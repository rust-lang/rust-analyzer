Desugar ast::WhileExpr from: [opt_ident]: while <cond> <body> into:

ⓘ
[opt_ident]: loop {
  if <cond> {
    <body>
  }
  else {
    break;
  }
}

FIXME: Rustc wraps the condition in a construct equivalent to { let _t = <cond>; _t } to preserve drop semantics. We should probably do the same in future.

Desugar ast::ForExpr from: [opt_ident]: for <pat> in <head> <body> into:

ⓘ
match IntoIterator::into_iter(<head>) {
    mut iter => {
        [opt_ident]: loop {
            match Iterator::next(&mut iter) {
                None => break,
                Some(<pat>) => <body>,
            };
        }
    }
}

fn collect_try_operator( &mut self, syntax_ptr: AstPtr<Expr>, e: TryExpr, ) -> ExprId

Desugar ast::TryExpr from: <expr>? into:

ⓘ
match Try::branch(<expr>) {
    ControlFlow::Continue(val) => val,
    ControlFlow::Break(residual) =>
        // If there is an enclosing `try {...}`:
        break 'catch_target Residual::into_try_type(residual),
        // If there is an enclosing `try bikeshed Ty {...}`:
        break 'catch_target Try::from_residual(residual),
        // Otherwise:
        return Try::from_residual(residual),
}
