35impl<'a> TtIter<'a> {
36    pub(crate) fn new(tt: TokenTreesView<'a>) -> TtIter<'a> {
37        TtIter { inner: tt }
38    }

40    pub fn expect_char(&mut self, char: char) -> Result<(), ()> {
41        match self.next() {
42            Some(TtElement::Leaf(Leaf::Punct(Punct { char: c, .. }))) if c == char => Ok(()),
43            _ => Err(()),
44        }
45    }
46
47    pub fn expect_any_char(&mut self, chars: &[char]) -> Result<(), ()> {
48        match self.next() {
49            Some(TtElement::Leaf(Leaf::Punct(Punct { char: c, .. }))) if chars.contains(&c) => {
50                Ok(())
51            }
52            _ => Err(()),
53        }
54    }
55
56    pub fn expect_subtree(&mut self) -> Result<(Subtree, TtIter<'a>), ()> {
57        match self.next() {
58            Some(TtElement::Subtree(subtree, iter)) => Ok((subtree, iter)),
59            _ => Err(()),
60        }
61    }
62
63    pub fn expect_leaf(&mut self) -> Result<Leaf, ()> {
64        match self.next() {
65            Some(TtElement::Leaf(it)) => Ok(it),
66            _ => Err(()),
67        }
68    }
69
70    pub fn expect_dollar(&mut self) -> Result<(), ()> {
71        match self.expect_leaf()? {
72            Leaf::Punct(Punct { char: '$', .. }) => Ok(()),
73            _ => Err(()),
74        }
75    }
76
77    pub fn expect_comma(&mut self) -> Result<(), ()> {
78        match self.expect_leaf()? {
79            Leaf::Punct(Punct { char: ',', .. }) => Ok(()),
80            _ => Err(()),
81        }
82    }
83
84    pub fn expect_ident(&mut self) -> Result<Ident, ()> {
85        match self.expect_leaf()? {
86            Leaf::Ident(it) if it.sym != sym::underscore => Ok(it),
87            _ => Err(()),
88        }
89    }
90
91    pub fn expect_ident_or_underscore(&mut self) -> Result<Ident, ()> {
92        match self.expect_leaf()? {
93            Leaf::Ident(it) => Ok(it),
94            _ => Err(()),
95        }
96    }
97
98    pub fn expect_literal(&mut self) -> Result<Leaf, ()> {
99        let it = self.expect_leaf()?;
100        match &it {
101            Leaf::Literal(_) => Ok(it),
102            Leaf::Ident(ident) if ident.sym == sym::true_ || ident.sym == sym::false_ => Ok(it),
103            _ => Err(()),
104        }
105    }
106
107    pub fn expect_single_punct(&mut self) -> Result<Punct, ()> {
108        match self.expect_leaf()? {
109            Leaf::Punct(it) => Ok(it),
110            _ => Err(()),
111        }
112    }

114    /// Returns consecutive `Punct`s that can be glued together.
115    ///
116    /// This method currently may return a single quotation, which is part of lifetime ident and
117    /// conceptually not a punct in the context of mbe. Callers should handle this.
118    pub fn expect_glued_punct(&mut self) -> Result<ArrayVec<Punct, MAX_GLUED_PUNCT_LEN>, ()> {
119        let TtElement::Leaf(Leaf::Punct(first)) = self.next().ok_or(())? else {
120            return Err(());
121        };
122
123        let mut res = ArrayVec::new();
124        if first.spacing == Spacing::Alone {
125            res.push(first);
126            return Ok(res);
127        }
128
129        let (second, third) = match (self.peek_n(0), self.peek_n(1)) {
130            (Some(TokenTree::Leaf(Leaf::Punct(p2))), Some(TokenTree::Leaf(Leaf::Punct(p3))))
131                if p2.spacing == Spacing::Joint =>
132            {
133                (p2, Some(p3))
134            }
135            (Some(TokenTree::Leaf(Leaf::Punct(p2))), _) => (p2, None),
136            _ => {
137                res.push(first);
138                return Ok(res);
139            }
140        };
141
142        match (first.char, second.char, third.map(|it| it.char)) {
143            ('.', '.', Some('.' | '=')) | ('<', '<', Some('=')) | ('>', '>', Some('=')) => {
144                let _ = self.next().unwrap();
145                let _ = self.next().unwrap();
146                res.push(first);
147                res.push(second);
148                res.push(third.unwrap());
149            }
150            ('-' | '!' | '*' | '/' | '&' | '%' | '^' | '+' | '<' | '=' | '>' | '|', '=', _)
151            | ('-' | '=' | '>', '>', _)
152            | ('<', '-', _)
153            | (':', ':', _)
154            | ('.', '.', _)
155            | ('&', '&', _)
156            | ('<', '<', _)
157            | ('|', '|', _) => {
158                let _ = self.next().unwrap();
159                res.push(first);
160                res.push(second);
161            }
162            _ => res.push(first),
163        }
164        Ok(res)
165    }

167    /// This method won't check for subtrees, so the nth token tree may not be the nth sibling of the current tree.
168    fn peek_n(&self, n: usize) -> Option<TokenTree> {
169        dispatch_ref! {
170            match self.inner.repr => tt => Some(tt.get(n)?.to_api(self.inner.span_parts))
171        }
172    }
173
174    pub fn peek(&self) -> Option<TtElement<'a>> {
175        match self.peek_n(0)? {
176            TokenTree::Leaf(leaf) => Some(TtElement::Leaf(leaf)),
177            TokenTree::Subtree(subtree) => {
178                let nested_repr = self.inner.repr.get(1..subtree.usize_len() + 1).unwrap();
179                let nested_iter = TtIter {
180                    inner: TokenTreesView { repr: nested_repr, span_parts: self.inner.span_parts },
181                };
182                Some(TtElement::Subtree(subtree, nested_iter))
183            }
184        }
185    }
186
187    /// Equivalent to `peek().is_none()`, but a bit faster.
188    pub fn is_empty(&self) -> bool {
189        self.inner.len() == 0
190    }
191
192    pub fn next_span(&self) -> Option<Span> {
193        Some(self.peek()?.first_span())
194    }
195
196    pub fn remaining(&self) -> TokenTreesView<'a> {
197        self.inner
198    }
199
200    /// **Warning**: This advances `skip` **flat** token trees, subtrees account for children+1!
201    pub fn flat_advance(&mut self, skip: usize) {
202        self.inner.repr = self.inner.repr.get(skip..).unwrap();
203    }

205    pub fn savepoint(&self) -> TtIterSavepoint<'a> {
206        TtIterSavepoint(self.inner)
207    }

209    pub fn from_savepoint(&self, savepoint: TtIterSavepoint<'a>) -> TokenTreesView<'a> {
210        let len = match (self.inner.repr, savepoint.0.repr) {
211            (
212                TokenTreesReprRef::SpanStorage32(this),
213                TokenTreesReprRef::SpanStorage32(savepoint),
214            ) => {
215                (this.as_ptr() as usize - savepoint.as_ptr() as usize)
216                    / size_of::<crate::storage::TokenTree<crate::storage::SpanStorage32>>()
217            }
218            (
219                TokenTreesReprRef::SpanStorage64(this),
220                TokenTreesReprRef::SpanStorage64(savepoint),
221            ) => {
222                (this.as_ptr() as usize - savepoint.as_ptr() as usize)
223                    / size_of::<crate::storage::TokenTree<crate::storage::SpanStorage64>>()
224            }
225            (
226                TokenTreesReprRef::SpanStorage96(this),
227                TokenTreesReprRef::SpanStorage96(savepoint),
228            ) => {
229                (this.as_ptr() as usize - savepoint.as_ptr() as usize)
230                    / size_of::<crate::storage::TokenTree<crate::storage::SpanStorage96>>()
231            }
232            _ => panic!("savepoint did not originate from this TtIter"),
233        };
234        TokenTreesView {
235            repr: savepoint.0.repr.get(..len).unwrap(),
236            span_parts: savepoint.0.span_parts,
237        }
238    }
239
240    pub fn next_as_view(&mut self) -> Option<TokenTreesView<'a>> {
241        let savepoint = self.savepoint();
242        self.next()?;
243        Some(self.from_savepoint(savepoint))
244    }

247#[derive(Clone)]
248pub enum TtElement<'a> {
249    Leaf(Leaf),
250    Subtree(Subtree, TtIter<'a>),
251}
252
253impl fmt::Debug for TtElement<'_> {
254    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
255        match self {
256            Self::Leaf(leaf) => f.debug_tuple("Leaf").field(leaf).finish(),
257            Self::Subtree(subtree, inner) => {
258                f.debug_tuple("Subtree").field(subtree).field(inner).finish()
259            }
260        }
261    }
262}
263
264impl TtElement<'_> {
265    #[inline]
266    pub fn first_span(&self) -> Span {
267        match self {
268            TtElement::Leaf(it) => *it.span(),
269            TtElement::Subtree(it, _) => it.delimiter.open,
270        }
271    }
272}

274impl<'a> Iterator for TtIter<'a> {
275    type Item = TtElement<'a>;
276    fn next(&mut self) -> Option<Self::Item> {
277        let result = self.peek()?;
278        let skip = match &result {
279            TtElement::Leaf(_) => 1,
280            TtElement::Subtree(subtree, _) => subtree.usize_len() + 1,
281        };
282        self.inner.repr = self.inner.repr.get(skip..).unwrap();
283        Some(result)
284    }
285}
