10pub(super) fn is_use_path_start(p: &Parser<'_>) -> bool {
11    match p.current() {
12        IDENT | T![self] | T![super] | T![crate] => true,
13        T![:] if p.at(T![::]) => true,
14        _ => false,
15    }

54fn path(p: &mut Parser<'_>, mode: Mode) -> Option<CompletedMarker> {
55    let path = p.start();
56    if path_segment(p, mode, true).is_none() {
57        path.abandon(p);
58        return None;
59    }
60    let qual = path.complete(p, PATH);
61    Some(path_for_qualifier(p, mode, qual))
62}
63
64fn path_for_qualifier(
65    p: &mut Parser<'_>,
66    mode: Mode,
67    mut qual: CompletedMarker,
68) -> CompletedMarker {
69    loop {
70        let use_tree = mode == Mode::Use && matches!(p.nth(2), T![*] | T!['{']);
71        if p.at(T![::]) && !use_tree {
72            let path = qual.precede(p);
73            p.bump(T![::]);
74            path_segment(p, mode, false);
75            let path = path.complete(p, PATH);
76            qual = path;
77        } else {
78            return qual;
79        }
80    }
81}

83const EXPR_PATH_SEGMENT_RECOVERY_SET: TokenSet =
84    expressions::EXPR_RECOVERY_SET.union(items::ITEM_RECOVERY_SET);
85const TYPE_PATH_SEGMENT_RECOVERY_SET: TokenSet = types::TYPE_RECOVERY_SET;

87fn path_segment(p: &mut Parser<'_>, mode: Mode, first: bool) -> Option<CompletedMarker> {
88    let m = p.start();
89    // test qual_paths
90    // type X = <A as B>::Output;
91    // fn foo() { <usize as Default>::default(); }
92    if first && p.at(T![<]) {
93        let m = p.start();
94        p.bump(T![<]);
95        // test_err angled_path_without_qual
96        // type X = <()>;
97        // type Y = <A as B>;
98        types::type_(p);
99        if p.eat(T![as]) {
100            if is_use_path_start(p) {
101                types::path_type_bounds(p, true);
102            } else {
103                p.error("expected a trait");
104            }
105        }
106        p.expect(T![>]);
107        m.complete(p, TYPE_ANCHOR);
108        if !p.at(T![::]) {
109            p.error("expected `::`");
110        }
111    } else {
112        let mut empty = if first { !p.eat(T![::]) } else { true };
113        if p.at_ts(PATH_NAME_REF_KINDS) {
114            // test crate_path
115            // use crate::foo;
116            name_ref_mod_path(p);
117            opt_path_args(p, mode);
118        } else {
119            let recover_set = match mode {
120                Mode::Use => items::ITEM_RECOVERY_SET,
121                Mode::Attr => {
122                    items::ITEM_RECOVERY_SET.union(TokenSet::new(&[T![']'], T![=], T![#]]))
123                }
124                Mode::Vis => items::ITEM_RECOVERY_SET.union(TokenSet::new(&[T![')']])),
125                Mode::Type => TYPE_PATH_SEGMENT_RECOVERY_SET,
126                Mode::Expr => EXPR_PATH_SEGMENT_RECOVERY_SET,
127            };
128            empty &= p.err_recover(
129                "expected identifier, `self`, `super`, `crate`, or `Self`",
130                recover_set,
131            );
132            if empty {
133                // test_err empty_segment
134                // use crate::;
135                m.abandon(p);
136                return None;
137            }
138        }
139    }
140    Some(m.complete(p, PATH_SEGMENT))
141}

143pub(crate) fn opt_path_type_args(p: &mut Parser<'_>) {
144    // test typepathfn_with_coloncolon
145    // type F = Start::(Middle) -> (Middle)::End;
146    // type GenericArg = S<Start(Middle)::End>;
147    let m;
148    if p.at(T![::]) && matches!(p.nth(2), T![<] | T!['(']) {
149        m = p.start();
150        p.bump(T![::]);
151    } else if (p.current() == T![<] && p.nth(1) != T![=]) || p.current() == T!['('] {
152        m = p.start();
153    } else {
154        return;
155    }
156    let current = p.current();
157    if current == T![<] {
158        // test_err generic_arg_list_recover
159        // type T = T<0, ,T>;
160        // type T = T::<0, ,T>;
161        delimited(
162            p,
163            T![<],
164            T![>],
165            T![,],
166            || "expected generic argument".into(),
167            generic_args::GENERIC_ARG_FIRST,
168            generic_args::generic_arg,
169        );
170        m.complete(p, GENERIC_ARG_LIST);
171    } else if p.nth_at(1, T![..]) {
172        // test return_type_syntax_in_path
173        // fn foo<T>()
174        // where
175        //     T::method(..): Send,
176        //     method(..): Send,
177        //     method::(..): Send,
178        // {}
179        p.bump(T!['(']);
180        p.bump(T![..]);
181        p.expect(T![')']);
182        m.complete(p, RETURN_TYPE_SYNTAX);
183    } else {
184        // test path_fn_trait_args
185        // type F = Box<Fn(i32) -> ()>;
186        // type F = Box<::Fn(i32) -> ()>;
187        // type F = Box<Fn::(i32) -> ()>;
188        // type F = Box<::Fn::(i32) -> ()>;
189        delimited(
190            p,
191            T!['('],
192            T![')'],
193            T![,],
194            || "expected type".into(),
195            types::TYPE_FIRST,
196            |p| {
197                let progress = types::TYPE_FIRST.contains(p.current());
198                generic_args::type_arg(p);
199                progress
200            },
201        );
202        m.complete(p, PARENTHESIZED_ARG_LIST);
203        opt_ret_type(p);
204    }
205}

207fn opt_path_args(p: &mut Parser<'_>, mode: Mode) {
208    match mode {
209        Mode::Use | Mode::Attr | Mode::Vis => {}
210        Mode::Type => opt_path_type_args(p),
211        Mode::Expr => generic_args::opt_generic_arg_list_expr(p),
212    }
