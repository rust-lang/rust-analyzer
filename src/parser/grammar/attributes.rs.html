17fn attr(p: &mut Parser<'_>, inner: bool) {
18    assert!(p.at(T![#]));
19
20    let attr = p.start();
21    p.bump(T![#]);
22
23    if inner {
24        p.bump(T![!]);
25    }
26
27    if p.expect(T!['[']) {
28        meta(p);
29        p.expect(T![']']);
30    }
31
32    attr.complete(p, ATTR);
33}

43fn cfg_attr_meta(p: &mut Parser<'_>, m: Marker) {
44    // test cfg_attr
45    // #![cfg_attr(not(foo), unsafe(bar()), cfg_attr(all(true, foo = "bar"), baz = "baz"))]
46    p.eat_contextual_kw(T![cfg_attr]);
47    p.bump(T!['(']);
48    cfg_predicate(p);
49    p.expect(T![,]);
50    while !p.at(T![')']) && !p.at(EOF) {
51        meta(p);
52        if !p.eat(T![,]) {
53            break;
54        }
55    }
56    p.expect(T![')']);
57    m.complete(p, CFG_ATTR_META);
58}

60const CFG_PREDICATE_FIRST_SET: TokenSet = TokenSet::new(&[T![true], T![false], T![ident]]);
61
62fn cfg_predicate(p: &mut Parser<'_>) {
63    let m = p.start();
64    if p.eat(T![true]) || p.eat(T![false]) {
65        // test cfg_true_false_pred
66        // #![cfg(true)]
67        // #![cfg(false)]
68        m.complete(p, CFG_ATOM);
69        return;
70    }
71    p.expect(T![ident]);
72    if p.eat(T![=]) {
73        if p.at(T![ident]) {
74            // This is required for completion, that inserts an identifier, to work in cases like
75            // `#[cfg(key = $0)]`, and also makes sense on itself.
76
77            // test_err key_ident_cfg_predicate
78            // #![cfg(key = value)]
79            p.err_and_bump("expected a string literal");
80        } else {
81            // test cfg_key_value_pred
82            // #![cfg(key = "value")]
83            p.expect(T![string]);
84        }
85        m.complete(p, CFG_ATOM);
86    } else if p.at(T!['(']) {
87        // test cfg_composite_pred
88        // #![cfg(any(a, all(b = "c", d)))]
89        delimited(
90            p,
91            T!['('],
92            T![')'],
93            T![,],
94            || "expected a cfg predicate".to_owned(),
95            CFG_PREDICATE_FIRST_SET,
96            |p| {
97                if p.at_ts(CFG_PREDICATE_FIRST_SET) {
98                    cfg_predicate(p);
99                    true
100                } else {
101                    false
102                }
103            },
104        );
105        m.complete(p, CFG_COMPOSITE);
106    } else {
107        m.complete(p, CFG_ATOM);
108    }
109}

111fn cfg_meta(p: &mut Parser<'_>, m: Marker) {
112    // test cfg_meta
113    // #![cfg(foo)]
114    // #![cfg(foo = "bar",)]
115    p.eat_contextual_kw(T![cfg]);
116    p.bump(T!['(']);
117    cfg_predicate(p);
118    p.eat(T![,]);
119    p.expect(T![')']);
120    m.complete(p, CFG_META);
121}

141// #![unsafe(simple::path::tt(a b c))]
142// #![unsafe(simple::path::tt[a b c])]

144pub(super) fn meta(p: &mut Parser<'_>) {
145    let m = p.start();
146    if p.eat(T![unsafe]) {
147        p.expect(T!['(']);
148        meta(p);
149        p.expect(T![')']);
150        m.complete(p, UNSAFE_META);
151        return;
152    }
153
154    if p.nth_at(1, T!['(']) {
155        if p.at_contextual_kw(T![cfg_attr]) {
156            return cfg_attr_meta(p, m);
157        } else if p.at_contextual_kw(T![cfg]) {
158            return cfg_meta(p, m);
159        }
160    }
161
162    paths::attr_path(p);
163
164    match p.current() {
165        T![=] if !p.at(T![=>]) && !p.at(T![==]) => {
166            p.bump(T![=]);
167            if expressions::expr(p).is_none() {
168                p.error("expected expression");
169            }
170            m.complete(p, KEY_VALUE_META);
171        }
172        T!['('] | T!['['] | T!['{'] => {
173            items::token_tree(p);
174            m.complete(p, TOKEN_TREE_META);
175        }
176        _ => {
177            m.complete(p, PATH_META);
178        }
179    }
180}
