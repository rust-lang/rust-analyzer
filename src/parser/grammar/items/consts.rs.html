3// test const_item
4// const C: u32 = 92;
5pub(super) fn konst(p: &mut Parser<'_>, m: Marker) {
6    p.eat(T![type]);
7    p.bump(T![const]);
8    const_or_static(p, m, true);
9}

11pub(super) fn static_(p: &mut Parser<'_>, m: Marker) {
12    p.bump(T![static]);
13    const_or_static(p, m, false);
14}

16fn const_or_static(p: &mut Parser<'_>, m: Marker, is_const: bool) {
17    p.eat(T![mut]);
18
19    if is_const && p.eat(T![_]) {
20        // test anonymous_const
21        // const _: u32 = 0;
22    } else {
23        // test_err anonymous_static
24        // static _: i32 = 5;
25        name(p);
26    }
27
28    // FIXME: Recover on statics with generic params/where clause.
29    if !is_const && p.at(T![<]) {
30        // test_err generic_static
31        // static C<i32>: u32 = 0;
32        p.error("`static` may not have generic parameters");
33    }
34    // test generic_const
35    // const C<i32>: u32 = 0;
36    // impl Foo {
37    //     const C<'a>: &'a () = &();
38    // }
39    generic_params::opt_generic_param_list(p);
40
41    if p.at(T![:]) {
42        types::ascription(p);
43    } else if is_const {
44        // test_err missing_const_type
45        // const C = 0;
46        p.error("missing type for `const`");
47    } else {
48        // test_err missing_static_type
49        // static C = 0;
50        p.error("missing type for `static`");
51    }
52    if p.eat(T![=]) {
53        expressions::expr(p);
54    }
55
56    if is_const {
57        // test const_where_clause
58        // const C<i32>: u32 = 0
59        // where i32: Copy;
60        // trait Foo {
61        //     const C: i32 where i32: Copy;
62        // }
63        generic_params::opt_where_clause(p);
64    }
65    // test_err static_where_clause
66    // static C: u32 = 0
67    // where i32: Copy;
68
69    p.expect(T![;]);
70    m.complete(p, if is_const { CONST } else { STATIC });
71}
