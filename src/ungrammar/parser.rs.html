1//! Simple hand-written ungrammar parser.
2#![allow(clippy::disallowed_types)]
3use std::collections::HashMap;
4
5use crate::{
6    Grammar, Node, NodeData, Rule, Token, TokenData,
7    error::{Result, format_err},
8    lexer::{self, TokenKind},
9};
10
11macro_rules! bail {
12    ($loc:expr, $($tt:tt)*) => {{
13        let err = $crate::error::format_err!($($tt)*)
14            .with_location($loc);
15        return Err(err);
16    }};
17}
18
19pub(crate) fn parse(tokens: Vec<lexer::Token>) -> Result<Grammar> {
20    let mut p = Parser::new(tokens);
21    while !p.is_eof() {
22        node(&mut p)?;
23    }
24    p.finish()
25}
26
27#[derive(Default)]
28struct Parser {
29    grammar: Grammar,
30    tokens: Vec<lexer::Token>,
31    node_table: HashMap<String, Node>,
32    token_table: HashMap<String, Token>,
33}
34
35const DUMMY_RULE: Rule = Rule::Node(Node(!0));
36
37impl Parser {
38    fn new(mut tokens: Vec<lexer::Token>) -> Parser {
39        tokens.reverse();
40        Parser { tokens, ..Parser::default() }
41    }
42
43    fn peek(&self) -> Option<&lexer::Token> {
44        self.peek_n(0)
45    }
46    fn peek_n(&self, n: usize) -> Option<&lexer::Token> {
47        self.tokens.iter().nth_back(n)
48    }
49    fn bump(&mut self) -> Result<lexer::Token> {
50        self.tokens.pop().ok_or_else(|| format_err!("unexpected EOF"))
51    }
52    fn expect(&mut self, kind: TokenKind, what: &str) -> Result<()> {
53        let token = self.bump()?;
54        if token.kind != kind {
55            bail!(token.loc, "unexpected token, expected `{}`", what);
56        }
57        Ok(())
58    }
59    fn is_eof(&self) -> bool {
60        self.tokens.is_empty()
61    }
62    fn finish(self) -> Result<Grammar> {
63        for node_data in &self.grammar.nodes {
64            if matches!(node_data.rule, DUMMY_RULE) {
65                crate::error::bail!("Undefined node: {}", node_data.name)
66            }
67        }
68        Ok(self.grammar)
69    }
70    fn intern_node(&mut self, name: String) -> Node {
71        let len = self.node_table.len();
72        let grammar = &mut self.grammar;
73        *self.node_table.entry(name.clone()).or_insert_with(|| {
74            grammar.nodes.push(NodeData { name, rule: DUMMY_RULE });
75            Node(len)
76        })
77    }
78    fn intern_token(&mut self, name: String) -> Token {
79        let len = self.token_table.len();
80        let grammar = &mut self.grammar;
81        *self.token_table.entry(name.clone()).or_insert_with(|| {
82            grammar.tokens.push(TokenData { name });
83            Token(len)
84        })
85    }
86}
87
88fn node(p: &mut Parser) -> Result<()> {
89    let token = p.bump()?;
90    let node = match token.kind {
91        TokenKind::Node(it) => p.intern_node(it),
92        _ => bail!(token.loc, "expected ident"),
93    };
94    p.expect(TokenKind::Eq, "=")?;
95    if !matches!(p.grammar[node].rule, DUMMY_RULE) {
96        bail!(token.loc, "duplicate rule: `{}`", p.grammar[node].name)
97    }
98
99    let rule = rule(p)?;
100    p.grammar.nodes[node.0].rule = rule;
101    Ok(())
102}
103
104fn rule(p: &mut Parser) -> Result<Rule> {
105    if let Some(lexer::Token { kind: TokenKind::Pipe, loc }) = p.peek() {
106        bail!(
107            *loc,
108            "The first element in a sequence of productions or alternatives \
109            must not have a leading pipe (`|`)"
110        );
111    }
112
113    let lhs = seq_rule(p)?;
114    let mut alt = vec![lhs];
115    while let Some(token) = p.peek() {
116        if token.kind != TokenKind::Pipe {
117            break;
118        }
119        p.bump()?;
120        let rule = seq_rule(p)?;
121        alt.push(rule)
122    }
123    let res = if alt.len() == 1 { alt.pop().unwrap() } else { Rule::Alt(alt) };
124    Ok(res)
125}
126
127fn seq_rule(p: &mut Parser) -> Result<Rule> {
128    let lhs = atom_rule(p)?;
129
130    let mut seq = vec![lhs];
131    while let Some(rule) = opt_atom_rule(p)? {
132        seq.push(rule)
133    }
134    let res = if seq.len() == 1 { seq.pop().unwrap() } else { Rule::Seq(seq) };
135    Ok(res)
136}
137
138fn atom_rule(p: &mut Parser) -> Result<Rule> {
139    match opt_atom_rule(p)? {
140        Some(it) => Ok(it),
141        None => {
142            let token = p.bump()?;
143            bail!(token.loc, "unexpected token")
144        }
145    }
146}
147
148fn opt_atom_rule(p: &mut Parser) -> Result<Option<Rule>> {
149    let token = match p.peek() {
150        Some(it) => it,
151        None => return Ok(None),
152    };
153    let mut res = match &token.kind {
154        TokenKind::Node(name) => {
155            if let Some(lookahead) = p.peek_n(1) {
156                match lookahead.kind {
157                    TokenKind::Eq => return Ok(None),
158                    TokenKind::Colon => {
159                        let label = name.clone();
160                        p.bump()?;
161                        p.bump()?;
162                        let rule = atom_rule(p)?;
163                        let res = Rule::Labeled { label, rule: Box::new(rule) };
164                        return Ok(Some(res));
165                    }
166                    _ => (),
167                }
168            }
169            match p.peek_n(1) {
170                Some(token) if token.kind == TokenKind::Eq => return Ok(None),
171                _ => (),
172            }
173            let name = name.clone();
174            p.bump()?;
175            let node = p.intern_node(name);
176            Rule::Node(node)
177        }
178        TokenKind::Token(name) => {
179            let name = name.clone();
180            p.bump()?;
181            let token = p.intern_token(name);
182            Rule::Token(token)
183        }
184        TokenKind::LParen => {
185            p.bump()?;
186            let rule = rule(p)?;
187            p.expect(TokenKind::RParen, ")")?;
188            rule
189        }
190        _ => return Ok(None),
191    };
192
193    if let Some(token) = p.peek() {
194        match &token.kind {
195            TokenKind::QMark => {
196                p.bump()?;
197                res = Rule::Opt(Box::new(res));
198            }
199            TokenKind::Star => {
200                p.bump()?;
201                res = Rule::Rep(Box::new(res));
202            }
203            _ => (),
204        }
205    }
206    Ok(Some(res))
207}
