1//! Boilerplate error definitions.
2use std::fmt;
3
4use crate::lexer::Location;
5
6/// A type alias for std's Result with the Error as our error type.
7pub type Result<T, E = Error> = std::result::Result<T, E>;
8
9/// An error encountered when parsing a Grammar.
10#[derive(Debug)]
11pub struct Error {
12    pub(crate) message: String,
13    pub(crate) location: Option<Location>,
14}
15
16impl fmt::Display for Error {
17    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
18        if let Some(loc) = self.location {
19            // Report 1-based indices, to match text editors
20            write!(f, "{}:{}: ", loc.line + 1, loc.column + 1)?
21        }
22        write!(f, "{}", self.message)
23    }
24}
25
26impl std::error::Error for Error {}
27
28impl Error {
29    pub(crate) fn with_location(self, location: Location) -> Error {
30        Error { location: Some(location), ..self }
31    }
32}
33
34macro_rules! _format_err {
35    ($($tt:tt)*) => {
36        $crate::error::Error {
37            message: format!($($tt)*),
38            location: None,
39        }
40    };
41}
42pub(crate) use _format_err as format_err;
43
44macro_rules! _bail {
45    ($($tt:tt)*) => { return Err($crate::error::format_err!($($tt)*)) };
46}
47pub(crate) use _bail as bail;
