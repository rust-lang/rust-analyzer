21// rustc does not have a cycle handling for the `specializes` query, meaning a cycle is a bug,
22// and indeed I was unable to cause cycles even with erroneous code. However, in r-a we can
23// create a cycle if there is an error in the impl's where clauses. I believe well formed code
24// cannot create a cycle, but a cycle handler is required nevertheless.
25fn specializes_query_cycle(
26    _db: &dyn HirDatabase,
27    _: salsa::Id,
28    _specializing_impl_def_id: ImplId,
29    _parent_impl_def_id: ImplId,
30) -> bool {
31    false

34/// Is `specializing_impl_def_id` a specialization of `parent_impl_def_id`?
35///
36/// For every type that could apply to `specializing_impl_def_id`, we prove that
37/// the `parent_impl_def_id` also applies (i.e. it has a valid impl header and
38/// its where-clauses hold).
39///
40/// For the purposes of const traits, we also check that the specializing
41/// impl is not more restrictive than the parent impl. That is, if the
42/// `parent_impl_def_id` is a const impl (conditionally based off of some `[const]`
43/// bounds), then `specializing_impl_def_id` must also be const for the same
44/// set of types.
45#[salsa::tracked(cycle_result = specializes_query_cycle)]
46fn specializes_query(
47    db: &dyn HirDatabase,
48    specializing_impl_def_id: ImplId,
49    parent_impl_def_id: ImplId,
50) -> bool {
51    let trait_env = db.trait_environment(ExpressionStoreOwnerId::from(GenericDefId::from(
52        specializing_impl_def_id,
53    )));
54    let interner = DbInterner::new_with(db, specializing_impl_def_id.krate(db));
55
56    let specializing_impl_signature = ImplSignature::of(db, specializing_impl_def_id);
57    let parent_impl_signature = ImplSignature::of(db, parent_impl_def_id);
58
59    // We determine whether there's a subset relationship by:
60    //
61    // - replacing bound vars with placeholders in impl1,
62    // - assuming the where clauses for impl1,
63    // - instantiating impl2 with fresh inference variables,
64    // - unifying,
65    // - attempting to prove the where clauses for impl2
66    //
67    // The last three steps are encapsulated in `fulfill_implication`.
68    //
69    // See RFC 1210 for more details and justification.
70
71    // Currently we do not allow e.g., a negative impl to specialize a positive one
72    if specializing_impl_signature.is_negative() != parent_impl_signature.is_negative() {
73        return false;
74    }
75
76    // create a parameter environment corresponding to an identity instantiation of the specializing impl,
77    // i.e. the most generic instantiation of the specializing impl.
78    let param_env = trait_env;
79
80    // Create an infcx, taking the predicates of the specializing impl as assumptions:
81    let infcx = interner.infer_ctxt().build(TypingMode::non_body_analysis());
82
83    let specializing_impl_trait_ref =
84        db.impl_trait(specializing_impl_def_id).unwrap().instantiate_identity();
85    let cause = &ObligationCause::dummy();
86    debug!(
87        "fulfill_implication({:?}, trait_ref={:?} |- {:?} applies)",
88        param_env, specializing_impl_trait_ref, parent_impl_def_id
89    );
90
91    // Attempt to prove that the parent impl applies, given all of the above.
92
93    let mut ocx = ObligationCtxt::new(&infcx);
94
95    let parent_args = infcx.fresh_args_for_item(Span::Dummy, parent_impl_def_id.into());
96    let parent_impl_trait_ref = db
97        .impl_trait(parent_impl_def_id)
98        .expect("expected source impl to be a trait impl")
99        .instantiate(interner, parent_args);
100
101    // do the impls unify? If not, no specialization.
102    let Ok(()) = ocx.eq(cause, param_env, specializing_impl_trait_ref, parent_impl_trait_ref)
103    else {
104        return false;
105    };
106
107    // Now check that the source trait ref satisfies all the where clauses of the target impl.
108    // This is not just for correctness; we also need this to constrain any params that may
109    // only be referenced via projection predicates.
110    ocx.register_obligations(clauses_as_obligations(
111        GenericPredicates::query_all(db, parent_impl_def_id.into())
112            .iter_instantiated(interner, parent_args.as_slice()),
113        *cause,
114        param_env,
115    ));
116
117    let errors = ocx.evaluate_obligations_error_on_ambiguity();
118    if !errors.is_empty() {
119        // no dice!
120        debug!(
121            "fulfill_implication: for impls on {:?} and {:?}, \
122                 could not fulfill: {:?} given {:?}",
123            specializing_impl_trait_ref, parent_impl_trait_ref, errors, param_env
124        );
125        return false;
126    }
127
128    // FIXME: Check impl constness (when we implement const impls).
129
130    debug!(
131        "fulfill_implication: an impl for {:?} specializes {:?}",
132        specializing_impl_trait_ref, parent_impl_trait_ref
133    );
134
135    true
136}

138// This function is used to avoid creating the query for crates that does not define `#![feature(specialization)]`,
139// as the solver is calling this a lot, and creating the query consumes a lot of memory.
140pub(crate) fn specializes(
141    db: &dyn HirDatabase,
142    specializing_impl_def_id: ImplId,
143    parent_impl_def_id: ImplId,
144) -> bool {
145    let module = specializing_impl_def_id.loc(db).container;
146
147    // We check that the specializing impl comes from a crate that has specialization enabled.
148    //
149    // We don't really care if the specialized impl (the parent) is in a crate that has
150    // specialization enabled, since it's not being specialized.
151    //
152    // rustc also checks whether the specializing impls comes from a macro marked
153    // `#[allow_internal_unstable(specialization)]`, but `#[allow_internal_unstable]`
154    // is an internal feature, std is not using it for specialization nor is likely to
155    // ever use it, and we don't have the span information necessary to replicate that.
156    let features = UnstableFeatures::query(db, module.krate(db));
157    if !features.specialization && !features.min_specialization {
158        return false;
159    }
160
161    specializes_query(db, specializing_impl_def_id, parent_impl_def_id)
162}
