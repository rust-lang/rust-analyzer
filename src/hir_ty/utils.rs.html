20/// SAFETY: `old_pointer` must be valid for unique writes
21pub(crate) unsafe fn unsafe_update_eq<T>(old_pointer: *mut T, new_value: T) -> bool
22where
23    T: PartialEq,
24{
25    // SAFETY: Caller obligation
26    let old_ref: &mut T = unsafe { &mut *old_pointer };
27
28    if *old_ref != new_value {
29        *old_ref = new_value;
30        true
31    } else {
32        // Subtle but important: Eq impls can be buggy or define equality
33        // in surprising ways. If it says that the value has not changed,
34        // we do not modify the existing value, and thus do not have to
35        // update the revision, as downstream code will not see the new value.
36        false
37    }
38}

54#[derive(Debug, Clone, Copy, PartialEq, Eq)]
55pub enum Unsafety {
56    Safe,
57    Unsafe,
58    /// A lint.
59    DeprecatedSafe2024,
60}

62#[derive(Debug, Clone, Copy, PartialEq, Eq)]
63pub enum TargetFeatureIsSafeInTarget {
64    No,
65    Yes,
66}

68pub fn target_feature_is_safe_in_target(target: &TargetData) -> TargetFeatureIsSafeInTarget {
69    match target.arch {
70        target::Arch::Wasm32 | target::Arch::Wasm64 => TargetFeatureIsSafeInTarget::Yes,
71        _ => TargetFeatureIsSafeInTarget::No,
72    }
73}

75pub fn is_fn_unsafe_to_call(
76    db: &dyn HirDatabase,
77    func: FunctionId,
78    caller_target_features: &TargetFeatures<'_>,
79    call_edition: Edition,
80    target_feature_is_safe: TargetFeatureIsSafeInTarget,
81) -> Unsafety {
82    let data = db.function_signature(func);
83    if data.is_unsafe() {
84        return Unsafety::Unsafe;
85    }
86
87    if data.has_target_feature() && target_feature_is_safe == TargetFeatureIsSafeInTarget::No {
88        // RFC 2396 <https://rust-lang.github.io/rfcs/2396-target-feature-1.1.html>.
89        let callee_target_features = TargetFeatures::from_fn_no_implications(db, func);
90        if !caller_target_features.enabled.is_superset(&callee_target_features.enabled) {
91            return Unsafety::Unsafe;
92        }
93    }
94
95    if data.is_deprecated_safe_2024() {
96        if call_edition.at_least_2024() {
97            return Unsafety::Unsafe;
98        } else {
99            return Unsafety::DeprecatedSafe2024;
100        }
101    }
102
103    let loc = func.lookup(db);
104    match loc.container {
105        hir_def::ItemContainerId::ExternBlockId(block) => {
106            let is_intrinsic_block = block.abi(db) == Some(sym::rust_dash_intrinsic);
107            if is_intrinsic_block {
108                // legacy intrinsics
109                // extern "rust-intrinsic" intrinsics are unsafe unless they have the rustc_safe_intrinsic attribute
110                if AttrFlags::query(db, func.into()).contains(AttrFlags::RUSTC_SAFE_INTRINSIC) {
111                    Unsafety::Safe
112                } else {
113                    Unsafety::Unsafe
114                }
115            } else {
116                // Function in an `extern` block are always unsafe to call, except when
117                // it is marked as `safe`.
118                if data.is_safe() { Unsafety::Safe } else { Unsafety::Unsafe }
119            }
120        }
121        _ => Unsafety::Safe,
122    }
123}

125pub(crate) fn detect_variant_from_bytes<'a>(
126    layout: &'a Layout,
127    db: &dyn HirDatabase,
128    target_data_layout: &TargetDataLayout,
129    b: &[u8],
130    e: EnumId,
131) -> Option<(EnumVariantId, &'a Layout)> {
132    let (var_id, var_layout) = match &layout.variants {
133        hir_def::layout::Variants::Empty => unreachable!(),
134        hir_def::layout::Variants::Single { index } => {
135            (e.enum_variants(db).variants[index.0].0, layout)
136        }
137        hir_def::layout::Variants::Multiple { tag, tag_encoding, variants, .. } => {
138            let size = tag.size(target_data_layout).bytes_usize();
139            let offset = layout.fields.offset(0).bytes_usize(); // The only field on enum variants is the tag field
140            let tag = i128::from_le_bytes(pad16(&b[offset..offset + size], false));
141            match tag_encoding {
142                TagEncoding::Direct => {
143                    let (var_idx, layout) =
144                        variants.iter_enumerated().find_map(|(var_idx, v)| {
145                            let def = e.enum_variants(db).variants[var_idx.0].0;
146                            (db.const_eval_discriminant(def) == Ok(tag)).then_some((def, v))
147                        })?;
148                    (var_idx, layout)
149                }
150                TagEncoding::Niche { untagged_variant, niche_start, .. } => {
151                    let candidate_tag = tag.wrapping_sub(*niche_start as i128) as usize;
152                    let variant = variants
153                        .iter_enumerated()
154                        .map(|(x, _)| x)
155                        .filter(|x| x != untagged_variant)
156                        .nth(candidate_tag)
157                        .unwrap_or(*untagged_variant);
158                    (e.enum_variants(db).variants[variant.0].0, &variants[variant])
159                }
160            }
161        }
162    };
163    Some((var_id, var_layout))
164}
