30pub(crate) fn fn_traits(db: &dyn DefDatabase, krate: Crate) -> impl Iterator<Item = TraitId> + '_ {
31    [LangItem::Fn, LangItem::FnMut, LangItem::FnOnce]
32        .into_iter()
33        .filter_map(move |lang| lang.resolve_trait(db, krate))

36/// Returns an iterator over the direct super traits (including the trait itself).
37pub fn direct_super_traits(db: &dyn DefDatabase, trait_: TraitId) -> SmallVec<[TraitId; 4]> {
38    let mut result = smallvec![trait_];
39    direct_super_traits_cb(db, trait_, |tt| {
40        if !result.contains(&tt) {
41            result.push(tt);
42        }
43    });
44    result
45}
46
47/// Returns an iterator over the whole super trait hierarchy (including the
48/// trait itself).
49pub fn all_super_traits(db: &dyn DefDatabase, trait_: TraitId) -> SmallVec<[TraitId; 4]> {
50    // we need to take care a bit here to avoid infinite loops in case of cycles
51    // (i.e. if we have `trait A: B; trait B: A;`)
52
53    let mut result = smallvec![trait_];
54    let mut i = 0;
55    while let Some(&t) = result.get(i) {
56        // yeah this is quadratic, but trait hierarchies should be flat
57        // enough that this doesn't matter
58        direct_super_traits_cb(db, t, |tt| {
59            if !result.contains(&tt) {
60                result.push(tt);
61            }
62        });
63        i += 1;
64    }
65    result
66}

68fn direct_super_traits_cb(db: &dyn DefDatabase, trait_: TraitId, cb: impl FnMut(TraitId)) {
69    let resolver = LazyCell::new(|| trait_.resolver(db));
70    let (generic_params, store) = db.generic_params_and_store(trait_.into());
71    let trait_self = generic_params.trait_self_param();
72    generic_params
73        .where_predicates()
74        .iter()
75        .filter_map(|pred| match pred {
76            WherePredicate::ForLifetime { target, bound, .. }
77            | WherePredicate::TypeBound { target, bound } => {
78                let is_trait = match &store[*target] {
79                    TypeRef::Path(p) => p.is_self_type(),
80                    TypeRef::TypeParam(p) => Some(p.local_id()) == trait_self,
81                    _ => false,
82                };
83                match is_trait {
84                    true => bound.as_path(&store),
85                    false => None,
86                }
87            }
88            WherePredicate::Lifetime { .. } => None,
89        })
90        .filter(|(_, bound_modifier)| matches!(bound_modifier, TraitBoundModifier::None))
91        .filter_map(|(path, _)| match resolver.resolve_path_in_type_ns_fully(db, path) {
92            Some(TypeNs::TraitId(t)) => Some(t),
93            _ => None,
94        })
95        .for_each(cb);
96}

99pub enum Unsafety {
100    Safe,
101    Unsafe,
102    /// A lint.
103    DeprecatedSafe2024,
104}

106#[derive(Debug, Clone, Copy, PartialEq, Eq)]
107pub enum TargetFeatureIsSafeInTarget {
108    No,
109    Yes,
110}

112pub fn target_feature_is_safe_in_target(target: &TargetData) -> TargetFeatureIsSafeInTarget {
113    match target.arch {
114        target::Arch::Wasm32 | target::Arch::Wasm64 => TargetFeatureIsSafeInTarget::Yes,
115        _ => TargetFeatureIsSafeInTarget::No,
116    }

119pub fn is_fn_unsafe_to_call(
120    db: &dyn HirDatabase,
121    func: FunctionId,
122    caller_target_features: &TargetFeatures,
123    call_edition: Edition,
124    target_feature_is_safe: TargetFeatureIsSafeInTarget,
125) -> Unsafety {
126    let data = db.function_signature(func);
127    if data.is_unsafe() {
128        return Unsafety::Unsafe;
129    }
130
131    if data.has_target_feature() && target_feature_is_safe == TargetFeatureIsSafeInTarget::No {
132        // RFC 2396 <https://rust-lang.github.io/rfcs/2396-target-feature-1.1.html>.
133        let callee_target_features =
134            TargetFeatures::from_attrs_no_implications(&db.attrs(func.into()));
135        if !caller_target_features.enabled.is_superset(&callee_target_features.enabled) {
136            return Unsafety::Unsafe;
137        }
138    }
139
140    if data.is_deprecated_safe_2024() {
141        if call_edition.at_least_2024() {
142            return Unsafety::Unsafe;
143        } else {
144            return Unsafety::DeprecatedSafe2024;
145        }
146    }
147
148    let loc = func.lookup(db);
149    match loc.container {
150        hir_def::ItemContainerId::ExternBlockId(block) => {
151            let is_intrinsic_block = block.abi(db) == Some(sym::rust_dash_intrinsic);
152            if is_intrinsic_block {
153                // legacy intrinsics
154                // extern "rust-intrinsic" intrinsics are unsafe unless they have the rustc_safe_intrinsic attribute
155                if db.attrs(func.into()).by_key(sym::rustc_safe_intrinsic).exists() {
156                    Unsafety::Safe
157                } else {
158                    Unsafety::Unsafe
159                }
160            } else {
161                // Function in an `extern` block are always unsafe to call, except when
162                // it is marked as `safe`.
163                if data.is_safe() { Unsafety::Safe } else { Unsafety::Unsafe }
164            }
165        }
166        _ => Unsafety::Safe,
167    }
168}

170pub(crate) fn detect_variant_from_bytes<'a>(
171    layout: &'a Layout,
172    db: &dyn HirDatabase,
173    target_data_layout: &TargetDataLayout,
174    b: &[u8],
175    e: EnumId,
176) -> Option<(EnumVariantId, &'a Layout)> {
177    let (var_id, var_layout) = match &layout.variants {
178        hir_def::layout::Variants::Empty => unreachable!(),
179        hir_def::layout::Variants::Single { index } => {
180            (e.enum_variants(db).variants[index.0].0, layout)
181        }
182        hir_def::layout::Variants::Multiple { tag, tag_encoding, variants, .. } => {
183            let size = tag.size(target_data_layout).bytes_usize();
184            let offset = layout.fields.offset(0).bytes_usize(); // The only field on enum variants is the tag field
185            let tag = i128::from_le_bytes(pad16(&b[offset..offset + size], false));
186            match tag_encoding {
187                TagEncoding::Direct => {
188                    let (var_idx, layout) =
189                        variants.iter_enumerated().find_map(|(var_idx, v)| {
190                            let def = e.enum_variants(db).variants[var_idx.0].0;
191                            (db.const_eval_discriminant(def) == Ok(tag)).then_some((def, v))
192                        })?;
193                    (var_idx, layout)
194                }
195                TagEncoding::Niche { untagged_variant, niche_start, .. } => {
196                    let candidate_tag = tag.wrapping_sub(*niche_start as i128) as usize;
197                    let variant = variants
198                        .iter_enumerated()
199                        .map(|(x, _)| x)
200                        .filter(|x| x != untagged_variant)
201                        .nth(candidate_tag)
202                        .unwrap_or(*untagged_variant);
203                    (e.enum_variants(db).variants[variant.0].0, &variants[variant])
204                }
205            }
206        }
207    };
208    Some((var_id, var_layout))
209}
