28#[derive(Debug, Clone, PartialEq, Eq, Hash)]
29pub enum DynCompatibilityViolation {
30    SizedSelf,
31    SelfReferential,
32    Method(FunctionId, MethodViolationCode),
33    AssocConst(ConstId),
34    GAT(TypeAliasId),
35    // This doesn't exist in rustc, but added for better visualization
36    HasNonCompatibleSuperTrait(TraitId),
37}
38
39#[derive(Debug, Clone, PartialEq, Eq, Hash)]
40pub enum MethodViolationCode {
41    StaticMethod,
42    ReferencesSelfInput,
43    ReferencesSelfOutput,
44    ReferencesImplTraitInTrait,
45    AsyncFn,
46    WhereClauseReferencesSelf,
47    Generic,
48    UndispatchableReceiver,
49}

51pub fn dyn_compatibility(
52    db: &dyn HirDatabase,
53    trait_: TraitId,
54) -> Option<DynCompatibilityViolation> {
55    let interner = DbInterner::new_no_crate(db);
56    for super_trait in elaborate::supertrait_def_ids(interner, trait_.into()) {
57        if let Some(v) = db.dyn_compatibility_of_trait(super_trait.0) {
58            return if super_trait.0 == trait_ {
59                Some(v)
60            } else {
61                Some(DynCompatibilityViolation::HasNonCompatibleSuperTrait(super_trait.0))
62            };
63        }
64    }
65
66    None
67}

69pub fn dyn_compatibility_with_callback<F>(
70    db: &dyn HirDatabase,
71    trait_: TraitId,
72    cb: &mut F,
73) -> ControlFlow<()>
74where
75    F: FnMut(DynCompatibilityViolation) -> ControlFlow<()>,
76{
77    let interner = DbInterner::new_no_crate(db);
78    for super_trait in elaborate::supertrait_def_ids(interner, trait_.into()).skip(1) {
79        if db.dyn_compatibility_of_trait(super_trait.0).is_some() {
80            cb(DynCompatibilityViolation::HasNonCompatibleSuperTrait(trait_))?;
81        }
82    }
83
84    dyn_compatibility_of_trait_with_callback(db, trait_, cb)
85}

87pub fn dyn_compatibility_of_trait_with_callback<F>(
88    db: &dyn HirDatabase,
89    trait_: TraitId,
90    cb: &mut F,
91) -> ControlFlow<()>
92where
93    F: FnMut(DynCompatibilityViolation) -> ControlFlow<()>,
94{
95    // Check whether this has a `Sized` bound
96    if generics_require_sized_self(db, trait_.into()) {
97        cb(DynCompatibilityViolation::SizedSelf)?;
98    }
99
100    // Check if there exist bounds that referencing self
101    if predicates_reference_self(db, trait_) {
102        cb(DynCompatibilityViolation::SelfReferential)?;
103    }
104    if bounds_reference_self(db, trait_) {
105        cb(DynCompatibilityViolation::SelfReferential)?;
106    }
107
108    // rustc checks for non-lifetime binders here, but we don't support HRTB yet
109
110    let trait_data = trait_.trait_items(db);
111    for (_, assoc_item) in &trait_data.items {
112        dyn_compatibility_violation_for_assoc_item(db, trait_, *assoc_item, cb)?;
113    }
114
115    ControlFlow::Continue(())
116}

118pub fn dyn_compatibility_of_trait_query(
119    db: &dyn HirDatabase,
120    trait_: TraitId,
121) -> Option<DynCompatibilityViolation> {
122    let mut res = None;
123    _ = dyn_compatibility_of_trait_with_callback(db, trait_, &mut |osv| {
124        res = Some(osv);
125        ControlFlow::Break(())
126    });
127
128    res
129}

131pub fn generics_require_sized_self(db: &dyn HirDatabase, def: GenericDefId) -> bool {
132    let krate = def.module(db).krate(db);
133    let interner = DbInterner::new_with(db, krate);
134    let Some(sized) = interner.lang_items().Sized else {
135        return false;
136    };
137
138    let predicates = GenericPredicates::query_explicit(db, def);
139    // FIXME: We should use `explicit_predicates_of` here, which hasn't been implemented to
140    // rust-analyzer yet
141    // https://github.com/rust-lang/rust/blob/ddaf12390d3ffb7d5ba74491a48f3cd528e5d777/compiler/rustc_hir_analysis/src/collect/predicates_of.rs#L490
142    elaborate::elaborate(interner, predicates.iter_identity_copied()).any(|pred| {
143        match pred.kind().skip_binder() {
144            ClauseKind::Trait(trait_pred) => {
145                if sized == trait_pred.def_id().0
146                    && let rustc_type_ir::TyKind::Param(param_ty) =
147                        trait_pred.trait_ref.self_ty().kind()
148                    && param_ty.index == 0
149                {
150                    true
151                } else {
152                    false
153                }
154            }
155            _ => false,
156        }
157    })
158}

160// rustc gathers all the spans that references `Self` for error rendering,
161// but we don't have good way to render such locations.
162// So, just return single boolean value for existence of such `Self` reference
163fn predicates_reference_self(db: &dyn HirDatabase, trait_: TraitId) -> bool {
164    GenericPredicates::query_explicit(db, trait_.into())
165        .iter_identity_copied()
166        .any(|pred| predicate_references_self(db, trait_, pred, AllowSelfProjection::No))
167}
168
169// Same as the above, `predicates_reference_self`
170fn bounds_reference_self(db: &dyn HirDatabase, trait_: TraitId) -> bool {
171    let trait_data = trait_.trait_items(db);
172    trait_data
173        .items
174        .iter()
175        .filter_map(|(_, it)| match *it {
176            AssocItemId::TypeAliasId(id) => Some(associated_ty_item_bounds(db, id)),
177            _ => None,
178        })
179        .any(|bounds| {
180            bounds.skip_binder().iter().any(|pred| match pred.skip_binder() {
181                rustc_type_ir::ExistentialPredicate::Trait(it) => it.args.iter().any(|arg| {
182                    contains_illegal_self_type_reference(db, trait_, &arg, AllowSelfProjection::Yes)
183                }),
184                rustc_type_ir::ExistentialPredicate::Projection(it) => it.args.iter().any(|arg| {
185                    contains_illegal_self_type_reference(db, trait_, &arg, AllowSelfProjection::Yes)
186                }),
187                rustc_type_ir::ExistentialPredicate::AutoTrait(_) => false,
188            })
189        })
190}

192#[derive(Clone, Copy)]
193enum AllowSelfProjection {
194    Yes,
195    No,
196}
197
198fn predicate_references_self<'db>(
199    db: &'db dyn HirDatabase,
200    trait_: TraitId,
201    predicate: Clause<'db>,
202    allow_self_projection: AllowSelfProjection,
203) -> bool {
204    match predicate.kind().skip_binder() {
205        ClauseKind::Trait(trait_pred) => trait_pred.trait_ref.args.iter().skip(1).any(|arg| {
206            contains_illegal_self_type_reference(db, trait_, &arg, allow_self_projection)
207        }),
208        ClauseKind::Projection(proj_pred) => {
209            proj_pred.projection_term.args.iter().skip(1).any(|arg| {
210                contains_illegal_self_type_reference(db, trait_, &arg, allow_self_projection)
211            })
212        }
213        _ => false,
214    }
215}

217fn contains_illegal_self_type_reference<'db, T: rustc_type_ir::TypeVisitable<DbInterner<'db>>>(
218    db: &'db dyn HirDatabase,
219    trait_: TraitId,
220    t: &T,
221    allow_self_projection: AllowSelfProjection,
222) -> bool {
223    struct IllegalSelfTypeVisitor<'db> {
224        db: &'db dyn HirDatabase,
225        trait_: TraitId,
226        super_traits: Option<SmallVec<[TraitId; 4]>>,
227        allow_self_projection: AllowSelfProjection,
228    }
229    impl<'db> rustc_type_ir::TypeVisitor<DbInterner<'db>> for IllegalSelfTypeVisitor<'db> {
230        type Result = ControlFlow<()>;
231
232        fn visit_ty(
233            &mut self,
234            ty: <DbInterner<'db> as rustc_type_ir::Interner>::Ty,
235        ) -> Self::Result {
236            let interner = DbInterner::new_no_crate(self.db);
237            match ty.kind() {
238                rustc_type_ir::TyKind::Param(param) if param.index == 0 => ControlFlow::Break(()),
239                rustc_type_ir::TyKind::Param(_) => ControlFlow::Continue(()),
240                rustc_type_ir::TyKind::Alias(AliasTyKind::Projection, proj) => {
241                    match self.allow_self_projection {
242                        AllowSelfProjection::Yes => {
243                            let trait_ = proj.trait_def_id(interner);
244                            let trait_ = match trait_ {
245                                SolverDefId::TraitId(id) => id,
246                                _ => unreachable!(),
247                            };
248                            if self.super_traits.is_none() {
249                                self.super_traits = Some(
250                                    elaborate::supertrait_def_ids(interner, self.trait_.into())
251                                        .map(|super_trait| super_trait.0)
252                                        .collect(),
253                                )
254                            }
255                            if self.super_traits.as_ref().is_some_and(|s| s.contains(&trait_)) {
256                                ControlFlow::Continue(())
257                            } else {
258                                ty.super_visit_with(self)
259                            }
260                        }
261                        AllowSelfProjection::No => ty.super_visit_with(self),
262                    }
263                }
264                _ => ty.super_visit_with(self),
265            }
266        }
267    }
268
269    let mut visitor =
270        IllegalSelfTypeVisitor { db, trait_, super_traits: None, allow_self_projection };
271    t.visit_with(&mut visitor).is_break()
272}

274fn dyn_compatibility_violation_for_assoc_item<F>(
275    db: &dyn HirDatabase,
276    trait_: TraitId,
277    item: AssocItemId,
278    cb: &mut F,
279) -> ControlFlow<()>
280where
281    F: FnMut(DynCompatibilityViolation) -> ControlFlow<()>,
282{
283    // Any item that has a `Self : Sized` requisite is otherwise
284    // exempt from the regulations.
285    if generics_require_sized_self(db, item.into()) {
286        return ControlFlow::Continue(());
287    }
288
289    match item {
290        AssocItemId::ConstId(it) => cb(DynCompatibilityViolation::AssocConst(it)),
291        AssocItemId::FunctionId(it) => {
292            virtual_call_violations_for_method(db, trait_, it, &mut |mvc| {
293                cb(DynCompatibilityViolation::Method(it, mvc))
294            })
295        }
296        AssocItemId::TypeAliasId(it) => {
297            let def_map = crate_def_map(db, trait_.krate(db));
298            if def_map.is_unstable_feature_enabled(&intern::sym::generic_associated_type_extended) {
299                ControlFlow::Continue(())
300            } else {
301                let generic_params = db.generic_params(item.into());
302                if !generic_params.is_empty() {
303                    cb(DynCompatibilityViolation::GAT(it))
304                } else {
305                    ControlFlow::Continue(())
306                }
307            }
308        }
309    }
310}

312fn virtual_call_violations_for_method<F>(
313    db: &dyn HirDatabase,
314    trait_: TraitId,
315    func: FunctionId,
316    cb: &mut F,
317) -> ControlFlow<()>
318where
319    F: FnMut(MethodViolationCode) -> ControlFlow<()>,
320{
321    let func_data = db.function_signature(func);
322    if !func_data.has_self_param() {
323        cb(MethodViolationCode::StaticMethod)?;
324    }
325
326    if func_data.is_async() {
327        cb(MethodViolationCode::AsyncFn)?;
328    }
329
330    let sig = db.callable_item_signature(func.into());
331    if sig.skip_binder().inputs().iter().skip(1).any(|ty| {
332        contains_illegal_self_type_reference(db, trait_, ty.skip_binder(), AllowSelfProjection::Yes)
333    }) {
334        cb(MethodViolationCode::ReferencesSelfInput)?;
335    }
336
337    if contains_illegal_self_type_reference(
338        db,
339        trait_,
340        &sig.skip_binder().output(),
341        AllowSelfProjection::Yes,
342    ) {
343        cb(MethodViolationCode::ReferencesSelfOutput)?;
344    }
345
346    if !func_data.is_async()
347        && let Some(mvc) = contains_illegal_impl_trait_in_trait(db, &sig)
348    {
349        cb(mvc)?;
350    }
351
352    let generic_params = db.generic_params(func.into());
353    if generic_params.len_type_or_consts() > 0 {
354        cb(MethodViolationCode::Generic)?;
355    }
356
357    if func_data.has_self_param() && !receiver_is_dispatchable(db, trait_, func, &sig) {
358        cb(MethodViolationCode::UndispatchableReceiver)?;
359    }
360
361    let predicates = GenericPredicates::query_own(db, func.into());
362    for pred in predicates.iter_identity_copied() {
363        let pred = pred.kind().skip_binder();
364
365        if matches!(pred, ClauseKind::TypeOutlives(_)) {
366            continue;
367        }
368
369        // Allow `impl AutoTrait` predicates
370        if let ClauseKind::Trait(TraitPredicate {
371            trait_ref: pred_trait_ref,
372            polarity: PredicatePolarity::Positive,
373        }) = pred
374            && let trait_data = db.trait_signature(pred_trait_ref.def_id.0)
375            && trait_data.flags.contains(TraitFlags::AUTO)
376            && let rustc_type_ir::TyKind::Param(ParamTy { index: 0, .. }) =
377                pred_trait_ref.self_ty().kind()
378        {
379            continue;
380        }
381
382        if contains_illegal_self_type_reference(db, trait_, &pred, AllowSelfProjection::Yes) {
383            cb(MethodViolationCode::WhereClauseReferencesSelf)?;
384            break;
385        }
386    }
387
388    ControlFlow::Continue(())
389}

391fn receiver_is_dispatchable<'db>(
392    db: &dyn HirDatabase,
393    trait_: TraitId,
394    func: FunctionId,
395    sig: &EarlyBinder<'db, Binder<'db, rustc_type_ir::FnSig<DbInterner<'db>>>>,
396) -> bool {
397    let sig = sig.instantiate_identity();
398
399    let module = trait_.module(db);
400    let interner = DbInterner::new_with(db, module.krate(db));
401    let self_param_id = TypeParamId::from_unchecked(TypeOrConstParamId {
402        parent: trait_.into(),
403        local_id: LocalTypeOrConstParamId::from_raw(la_arena::RawIdx::from_u32(0)),
404    });
405    let self_param_ty =
406        Ty::new(interner, rustc_type_ir::TyKind::Param(ParamTy { index: 0, id: self_param_id }));
407
408    // `self: Self` can't be dispatched on, but this is already considered dyn-compatible
409    // See rustc's comment on https://github.com/rust-lang/rust/blob/3f121b9461cce02a703a0e7e450568849dfaa074/compiler/rustc_trait_selection/src/traits/object_safety.rs#L433-L437
410    if sig.inputs().iter().next().is_some_and(|p| *p.skip_binder() == self_param_ty) {
411        return true;
412    }
413
414    let Some(&receiver_ty) = sig.inputs().skip_binder().first() else {
415        return false;
416    };
417
418    let lang_items = interner.lang_items();
419    let traits = (lang_items.Unsize, lang_items.DispatchFromDyn);
420    let (Some(unsize_did), Some(dispatch_from_dyn_did)) = traits else {
421        return false;
422    };
423
424    let meta_sized_did = lang_items.MetaSized;
425
426    // TODO: This is for supporting dyn compatibility for toolchains doesn't contain `MetaSized`
427    // trait. Uncomment and short circuit here once `MINIMUM_SUPPORTED_TOOLCHAIN_VERSION`
428    // become > 1.88.0
429    //
430    // let Some(meta_sized_did) = meta_sized_did else {
431    //     return false;
432    // };
433
434    // Type `U`
435    // FIXME: That seems problematic to fake a generic param like that?
436    let unsized_self_ty = Ty::new_param(interner, self_param_id, u32::MAX);
437    // `Receiver[Self => U]`
438    let unsized_receiver_ty = receiver_for_self_ty(interner, func, receiver_ty, unsized_self_ty);
439
440    let param_env = {
441        let generic_predicates = GenericPredicates::query_all(db, func.into());
442
443        // Self: Unsize<U>
444        let unsize_predicate =
445            TraitRef::new(interner, unsize_did.into(), [self_param_ty, unsized_self_ty]);
446
447        // U: Trait<Arg1, ..., ArgN>
448        let args = GenericArgs::for_item(interner, trait_.into(), |index, kind, _| {
449            if index == 0 { unsized_self_ty.into() } else { mk_param(interner, index, kind) }
450        });
451        let trait_predicate = TraitRef::new_from_args(interner, trait_.into(), args);
452
453        let meta_sized_predicate = meta_sized_did
454            .map(|did| TraitRef::new(interner, did.into(), [unsized_self_ty]).upcast(interner));
455
456        ParamEnv {
457            clauses: Clauses::new_from_iter(
458                interner,
459                generic_predicates
460                    .iter_identity_copied()
461                    .chain([unsize_predicate.upcast(interner), trait_predicate.upcast(interner)])
462                    .chain(meta_sized_predicate),
463            ),
464        }
465    };
466
467    // Receiver: DispatchFromDyn<Receiver[Self => U]>
468    let predicate =
469        TraitRef::new(interner, dispatch_from_dyn_did.into(), [receiver_ty, unsized_receiver_ty]);
470    let goal = Goal::new(interner, param_env, predicate);
471
472    let infcx = interner.infer_ctxt().build(TypingMode::non_body_analysis());
473    // the receiver is dispatchable iff the obligation holds
474    let res = next_trait_solve_in_ctxt(&infcx, goal);
475    res.map_or(false, |res| matches!(res.1, rustc_type_ir::solve::Certainty::Yes))
476}

478fn receiver_for_self_ty<'db>(
479    interner: DbInterner<'db>,
480    func: FunctionId,
481    receiver_ty: Ty<'db>,
482    self_ty: Ty<'db>,
483) -> Ty<'db> {
484    let args = GenericArgs::for_item(interner, SolverDefId::FunctionId(func), |index, kind, _| {
485        if index == 0 { self_ty.into() } else { mk_param(interner, index, kind) }
486    });
487
488    EarlyBinder::bind(receiver_ty).instantiate(interner, args)
489}

491fn contains_illegal_impl_trait_in_trait<'db>(
492    db: &'db dyn HirDatabase,
493    sig: &EarlyBinder<'db, Binder<'db, rustc_type_ir::FnSig<DbInterner<'db>>>>,
494) -> Option<MethodViolationCode> {
495    struct OpaqueTypeCollector(FxHashSet<InternedOpaqueTyId>);
496
497    impl<'db> rustc_type_ir::TypeVisitor<DbInterner<'db>> for OpaqueTypeCollector {
498        type Result = ControlFlow<()>;
499
500        fn visit_ty(
501            &mut self,
502            ty: <DbInterner<'db> as rustc_type_ir::Interner>::Ty,
503        ) -> Self::Result {
504            if let rustc_type_ir::TyKind::Alias(AliasTyKind::Opaque, op) = ty.kind() {
505                let id = match op.def_id {
506                    SolverDefId::InternedOpaqueTyId(id) => id,
507                    _ => unreachable!(),
508                };
509                self.0.insert(id);
510            }
511            ty.super_visit_with(self)
512        }
513    }
514
515    let ret = sig.skip_binder().output();
516    let mut visitor = OpaqueTypeCollector(FxHashSet::default());
517    _ = ret.visit_with(&mut visitor);
518
519    // Since we haven't implemented RPITIT in proper way like rustc yet,
520    // just check whether `ret` contains RPIT for now
521    for opaque_ty in visitor.0 {
522        let impl_trait_id = db.lookup_intern_impl_trait_id(opaque_ty);
523        if matches!(impl_trait_id, ImplTraitId::ReturnTypeImplTrait(..)) {
524            return Some(MethodViolationCode::ReferencesImplTraitInTrait);
525        }
526    }
527
528    None
529}
