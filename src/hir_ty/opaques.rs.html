23pub(crate) fn opaque_types_defined_by(
24    db: &dyn HirDatabase,
25    def_id: DefWithBodyId,
26    result: &mut Vec<SolverDefId>,
27) {
28    if let DefWithBodyId::FunctionId(func) = def_id {
29        // A function may define its own RPITs.
30        extend_with_opaques(
31            db,
32            db.return_type_impl_traits(func),
33            |opaque_idx| ImplTraitId::ReturnTypeImplTrait(func, opaque_idx),
34            result,
35        );
36    }
37
38    let extend_with_taits = |type_alias| {
39        extend_with_opaques(
40            db,
41            db.type_alias_impl_traits(type_alias),
42            |opaque_idx| ImplTraitId::TypeAliasImplTrait(type_alias, opaque_idx),
43            result,
44        );
45    };
46
47    // Collect opaques from assoc items.
48    let extend_with_atpit_from_assoc_items = |assoc_items: &[(Name, AssocItemId)]| {
49        assoc_items
50            .iter()
51            .filter_map(|&(_, assoc_id)| match assoc_id {
52                AssocItemId::TypeAliasId(it) => Some(it),
53                AssocItemId::FunctionId(_) | AssocItemId::ConstId(_) => None,
54            })
55            .for_each(extend_with_taits);
56    };
57    let extend_with_atpit_from_container = |container| match container {
58        ItemContainerId::ImplId(impl_id) => {
59            if db.impl_signature(impl_id).target_trait.is_some() {
60                extend_with_atpit_from_assoc_items(&impl_id.impl_items(db).items);
61            }
62        }
63        ItemContainerId::TraitId(trait_id) => {
64            extend_with_atpit_from_assoc_items(&trait_id.trait_items(db).items);
65        }
66        _ => {}
67    };
68    match def_id {
69        DefWithBodyId::ConstId(id) => extend_with_atpit_from_container(id.loc(db).container),
70        DefWithBodyId::FunctionId(id) => extend_with_atpit_from_container(id.loc(db).container),
71        DefWithBodyId::StaticId(_) | DefWithBodyId::VariantId(_) => {}
72    }
73
74    // FIXME: Collect opaques from `#[define_opaque]`.
75
76    fn extend_with_opaques<'db>(
77        db: &'db dyn HirDatabase,
78        opaques: Option<Arc<EarlyBinder<'db, ImplTraits<'db>>>>,
79        mut make_impl_trait: impl FnMut(ImplTraitIdx<'db>) -> ImplTraitId<'db>,
80        result: &mut Vec<SolverDefId>,
81    ) {
82        if let Some(opaques) = opaques {
83            for (opaque_idx, _) in (*opaques).as_ref().skip_binder().impl_traits.iter() {
84                let opaque_id = InternedOpaqueTyId::new(db, make_impl_trait(opaque_idx));
85                result.push(opaque_id.into());
86            }
87        }
88    }
89}

91// These are firewall queries to prevent drawing dependencies between infers:
92
93#[salsa::tracked(returns(ref), unsafe(non_update_return_type))]
94pub(crate) fn rpit_hidden_types<'db>(
95    db: &'db dyn HirDatabase,
96    function: FunctionId,
97) -> ArenaMap<ImplTraitIdx<'db>, EarlyBinder<'db, Ty<'db>>> {
98    let infer = db.infer(function.into());
99    let mut result = ArenaMap::new();
100    for (opaque, hidden_type) in infer.return_position_impl_trait_types(db) {
101        result.insert(opaque, EarlyBinder::bind(hidden_type));
102    }
103    result.shrink_to_fit();
104    result
105}

107#[salsa::tracked(returns(ref), unsafe(non_update_return_type))]
108pub(crate) fn tait_hidden_types<'db>(
109    db: &'db dyn HirDatabase,
110    type_alias: TypeAliasId,
111) -> ArenaMap<ImplTraitIdx<'db>, EarlyBinder<'db, Ty<'db>>> {
112    let loc = type_alias.loc(db);
113    let module = loc.module(db);
114    let interner = DbInterner::new_with(db, Some(module.krate()), module.containing_block());
115    let infcx = interner.infer_ctxt().build(TypingMode::non_body_analysis());
116    let mut ocx = ObligationCtxt::new(&infcx);
117    let cause = ObligationCause::dummy();
118    let param_env = db.trait_environment(type_alias.into()).env;
119
120    let defining_bodies = tait_defining_bodies(db, &loc);
121
122    let taits_count = db
123        .type_alias_impl_traits(type_alias)
124        .map_or(0, |taits| (*taits).as_ref().skip_binder().impl_traits.len());
125
126    let mut result = ArenaMap::with_capacity(taits_count);
127    for defining_body in defining_bodies {
128        let infer = db.infer(defining_body);
129        for (&opaque, &hidden_type) in &infer.type_of_opaque {
130            let ImplTraitId::TypeAliasImplTrait(opaque_owner, opaque_idx) = opaque.loc(db) else {
131                continue;
132            };
133            if opaque_owner != type_alias {
134                continue;
135            }
136            // In the presence of errors, we attempt to create a unified type from all
137            // types. rustc doesn't do that, but this should improve the experience.
138            let hidden_type = infcx.insert_type_vars(hidden_type);
139            match result.entry(opaque_idx) {
140                la_arena::Entry::Vacant(entry) => {
141                    entry.insert(EarlyBinder::bind(hidden_type));
142                }
143                la_arena::Entry::Occupied(entry) => {
144                    _ = ocx.eq(&cause, param_env, entry.get().instantiate_identity(), hidden_type);
145                }
146            }
147        }
148    }
149
150    _ = ocx.try_evaluate_obligations();
151
152    // Fill missing entries.
153    for idx in 0..taits_count {
154        let idx = la_arena::Idx::from_raw(la_arena::RawIdx::from_u32(idx as u32));
155        match result.entry(idx) {
156            la_arena::Entry::Vacant(entry) => {
157                entry.insert(EarlyBinder::bind(Ty::new_error(interner, ErrorGuaranteed)));
158            }
159            la_arena::Entry::Occupied(mut entry) => {
160                *entry.get_mut() = entry.get().map_bound(|hidden_type| {
161                    infcx.resolve_vars_if_possible(hidden_type).replace_infer_with_error(interner)
162                });
163            }
164        }
165    }
166
167    result
168}

170fn tait_defining_bodies(
171    db: &dyn HirDatabase,
172    loc: &AssocItemLoc<ast::TypeAlias>,
173) -> Vec<DefWithBodyId> {
174    let from_assoc_items = |assoc_items: &[(Name, AssocItemId)]| {
175        // Associated Type Position Impl Trait.
176        assoc_items
177            .iter()
178            .filter_map(|&(_, assoc_id)| match assoc_id {
179                AssocItemId::FunctionId(it) => Some(it.into()),
180                AssocItemId::ConstId(it) => Some(it.into()),
181                AssocItemId::TypeAliasId(_) => None,
182            })
183            .collect()
184    };
185    match loc.container {
186        ItemContainerId::ImplId(impl_id) => {
187            if db.impl_signature(impl_id).target_trait.is_some() {
188                return from_assoc_items(&impl_id.impl_items(db).items);
189            }
190        }
191        ItemContainerId::TraitId(trait_id) => {
192            return from_assoc_items(&trait_id.trait_items(db).items);
193        }
194        _ => {}
195    }
196
197    // FIXME: Support general TAITs, or decisively decide not to.
198    Vec::new()
