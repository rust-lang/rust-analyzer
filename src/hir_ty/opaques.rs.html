22pub(crate) fn opaque_types_defined_by(
23    db: &dyn HirDatabase,
24    def_id: DefWithBodyId,
25    result: &mut Vec<SolverDefId>,
26) {
27    if let DefWithBodyId::FunctionId(func) = def_id {
28        // A function may define its own RPITs.
29        extend_with_opaques(
30            db,
31            ImplTraits::return_type_impl_traits(db, func),
32            |opaque_idx| ImplTraitId::ReturnTypeImplTrait(func, opaque_idx),
33            result,
34        );
35    }
36
37    let extend_with_taits = |type_alias| {
38        extend_with_opaques(
39            db,
40            ImplTraits::type_alias_impl_traits(db, type_alias),
41            |opaque_idx| ImplTraitId::TypeAliasImplTrait(type_alias, opaque_idx),
42            result,
43        );
44    };
45
46    // Collect opaques from assoc items.
47    let extend_with_atpit_from_assoc_items = |assoc_items: &[(Name, AssocItemId)]| {
48        assoc_items
49            .iter()
50            .filter_map(|&(_, assoc_id)| match assoc_id {
51                AssocItemId::TypeAliasId(it) => Some(it),
52                AssocItemId::FunctionId(_) | AssocItemId::ConstId(_) => None,
53            })
54            .for_each(extend_with_taits);
55    };
56    let extend_with_atpit_from_container = |container| match container {
57        ItemContainerId::ImplId(impl_id) => {
58            if db.impl_signature(impl_id).target_trait.is_some() {
59                extend_with_atpit_from_assoc_items(&impl_id.impl_items(db).items);
60            }
61        }
62        ItemContainerId::TraitId(trait_id) => {
63            extend_with_atpit_from_assoc_items(&trait_id.trait_items(db).items);
64        }
65        _ => {}
66    };
67    match def_id {
68        DefWithBodyId::ConstId(id) => extend_with_atpit_from_container(id.loc(db).container),
69        DefWithBodyId::FunctionId(id) => extend_with_atpit_from_container(id.loc(db).container),
70        DefWithBodyId::StaticId(_) | DefWithBodyId::VariantId(_) => {}
71    }
72
73    // FIXME: Collect opaques from `#[define_opaque]`.
74
75    fn extend_with_opaques(
76        db: &dyn HirDatabase,
77        opaques: &Option<Box<StoredEarlyBinder<ImplTraits>>>,
78        mut make_impl_trait: impl FnMut(ImplTraitIdx) -> ImplTraitId,
79        result: &mut Vec<SolverDefId>,
80    ) {
81        if let Some(opaques) = opaques {
82            for (opaque_idx, _) in (**opaques).as_ref().skip_binder().impl_traits.iter() {
83                let opaque_id = InternedOpaqueTyId::new(db, make_impl_trait(opaque_idx));
84                result.push(opaque_id.into());
85            }
86        }
87    }
88}

90// These are firewall queries to prevent drawing dependencies between infers:
91
92#[salsa::tracked(returns(ref))]
93pub(crate) fn rpit_hidden_types<'db>(
94    db: &'db dyn HirDatabase,
95    function: FunctionId,
96) -> ArenaMap<ImplTraitIdx, StoredEarlyBinder<StoredTy>> {
97    let infer = InferenceResult::for_body(db, function.into());
98    let mut result = ArenaMap::new();
99    for (opaque, hidden_type) in infer.return_position_impl_trait_types(db) {
100        result.insert(opaque, StoredEarlyBinder::bind(hidden_type.store()));
101    }
102    result.shrink_to_fit();
103    result
104}

106#[salsa::tracked(returns(ref))]
107pub(crate) fn tait_hidden_types<'db>(
108    db: &'db dyn HirDatabase,
109    type_alias: TypeAliasId,
110) -> ArenaMap<ImplTraitIdx, StoredEarlyBinder<StoredTy>> {
111    // Call this first, to not perform redundant work if there are no TAITs.
112    let Some(taits_count) = ImplTraits::type_alias_impl_traits(db, type_alias)
113        .as_deref()
114        .map(|taits| taits.as_ref().skip_binder().impl_traits.len())
115    else {
116        return ArenaMap::new();
117    };
118
119    let loc = type_alias.loc(db);
120    let module = loc.module(db);
121    let interner = DbInterner::new_with(db, module.krate(db));
122    let infcx = interner.infer_ctxt().build(TypingMode::non_body_analysis());
123    let mut ocx = ObligationCtxt::new(&infcx);
124    let cause = ObligationCause::dummy();
125    let param_env = db.trait_environment(type_alias.into());
126
127    let defining_bodies = tait_defining_bodies(db, &loc);
128
129    let mut result = ArenaMap::with_capacity(taits_count);
130    for defining_body in defining_bodies {
131        let infer = InferenceResult::for_body(db, defining_body);
132        for (&opaque, hidden_type) in &infer.type_of_opaque {
133            let ImplTraitId::TypeAliasImplTrait(opaque_owner, opaque_idx) = opaque.loc(db) else {
134                continue;
135            };
136            if opaque_owner != type_alias {
137                continue;
138            }
139            // In the presence of errors, we attempt to create a unified type from all
140            // types. rustc doesn't do that, but this should improve the experience.
141            let hidden_type = infcx.insert_type_vars(hidden_type.as_ref());
142            match result.entry(opaque_idx) {
143                la_arena::Entry::Vacant(entry) => {
144                    entry.insert(StoredEarlyBinder::bind(hidden_type.store()));
145                }
146                la_arena::Entry::Occupied(entry) => {
147                    _ = ocx.eq(
148                        &cause,
149                        param_env,
150                        entry.get().get().instantiate_identity(),
151                        hidden_type,
152                    );
153                }
154            }
155        }
156    }
157
158    _ = ocx.try_evaluate_obligations();
159
160    // Fill missing entries.
161    for idx in 0..taits_count {
162        let idx = la_arena::Idx::from_raw(la_arena::RawIdx::from_u32(idx as u32));
163        match result.entry(idx) {
164            la_arena::Entry::Vacant(entry) => {
165                entry.insert(StoredEarlyBinder::bind(
166                    Ty::new_error(interner, ErrorGuaranteed).store(),
167                ));
168            }
169            la_arena::Entry::Occupied(mut entry) => {
170                let hidden_type = entry.get().get().skip_binder();
171                let hidden_type =
172                    infcx.resolve_vars_if_possible(hidden_type).replace_infer_with_error(interner);
173                *entry.get_mut() = StoredEarlyBinder::bind(hidden_type.store());
174            }
175        }
176    }
177
178    result
179}

181fn tait_defining_bodies(
182    db: &dyn HirDatabase,
183    loc: &AssocItemLoc<ast::TypeAlias>,
184) -> Vec<DefWithBodyId> {
185    let from_assoc_items = |assoc_items: &[(Name, AssocItemId)]| {
186        // Associated Type Position Impl Trait.
187        assoc_items
188            .iter()
189            .filter_map(|&(_, assoc_id)| match assoc_id {
190                AssocItemId::FunctionId(it) => Some(it.into()),
191                AssocItemId::ConstId(it) => Some(it.into()),
192                AssocItemId::TypeAliasId(_) => None,
193            })
194            .collect()
195    };
196    match loc.container {
197        ItemContainerId::ImplId(impl_id) => {
198            if db.impl_signature(impl_id).target_trait.is_some() {
199                return from_assoc_items(&impl_id.impl_items(db).items);
200            }
201        }
202        ItemContainerId::TraitId(trait_id) => {
203            return from_assoc_items(&trait_id.trait_items(db).items);
204        }
205        _ => {}
206    }
207
208    // FIXME: Support general TAITs, or decisively decide not to.
209    Vec::new()
