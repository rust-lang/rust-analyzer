96/// When adjusting a receiver we often want to do one of
97///
98/// - Add a `&` (or `&mut`), converting the receiver from `T` to `&T` (or `&mut T`)
99/// - If the receiver has type `*mut T`, convert it to `*const T`
100///
101/// This type tells us which one to do.
102///
103/// Note that in principle we could do both at the same time. For example, when the receiver has
104/// type `T`, we could autoref it to `&T`, then convert to `*const T`. Or, when it has type `*mut
105/// T`, we could convert it to `*const T`, then autoref to `&*const T`. However, currently we do
106/// (at most) one of these. Either the receiver has type `T` and we convert it to `&T` (or with
107/// `mut`), or it has type `*mut T` and we convert it to `*const T`.
108#[derive(Debug, PartialEq, Copy, Clone)]
109pub enum AutorefOrPtrAdjustment {
110    /// Receiver has type `T`, add `&` or `&mut` (if `T` is `mut`), and maybe also "unsize" it.
111    /// Unsizing is used to convert a `[T; N]` to `[T]`, which only makes sense when autorefing.
112    Autoref {
113        mutbl: Mutability,
114
115        /// Indicates that the source expression should be "unsized" to a target type.
116        /// This is special-cased for just arrays unsizing to slices.
117        unsize: bool,
118    },
119    /// Receiver has type `*mut T`, convert to `*const T`
120    ToConstPtr,
121}
122
123impl AutorefOrPtrAdjustment {
124    fn get_unsize(&self) -> bool {
125        match self {
126            AutorefOrPtrAdjustment::Autoref { mutbl: _, unsize } => *unsize,
127            AutorefOrPtrAdjustment::ToConstPtr => false,
128        }
129    }
130}

132/// Criteria to apply when searching for a given Pick. This is used during
133/// the search for potentially shadowed methods to ensure we don't search
134/// more candidates than strictly necessary.
135#[derive(Debug)]
136struct PickConstraintsForShadowed {
137    autoderefs: usize,
138    receiver_steps: Option<usize>,
139    def_id: CandidateId,
140}
141
142impl PickConstraintsForShadowed {
143    fn may_shadow_based_on_autoderefs(&self, autoderefs: usize) -> bool {
144        autoderefs == self.autoderefs
145    }
146
147    fn candidate_may_shadow(&self, candidate: &Candidate<'_>) -> bool {
148        // An item never shadows itself
149        candidate.item != self.def_id
150            // and we're only concerned about inherent impls doing the shadowing.
151            // Shadowing can only occur if the shadowed is further along
152            // the Receiver dereferencing chain than the shadowed.
153            && match candidate.kind {
154                CandidateKind::InherentImplCandidate { receiver_steps, .. } => match self.receiver_steps {
155                    Some(shadowed_receiver_steps) => receiver_steps > shadowed_receiver_steps,
156                    _ => false
157                },
158                _ => false
159            }
160    }
161}

213#[derive(Debug, Clone)]
214pub struct CandidateStep<'db> {
215    pub self_ty: Canonical<'db, QueryResponse<'db, Ty<'db>>>,
216    pub self_ty_is_opaque: bool,
217    pub autoderefs: usize,
218    /// `true` if the type results from a dereference of a raw pointer.
219    /// when assembling candidates, we include these steps, but not when
220    /// picking methods. This so that if we have `foo: *const Foo` and `Foo` has methods
221    /// `fn by_raw_ptr(self: *const Self)` and `fn by_ref(&self)`, then
222    /// `foo.by_raw_ptr()` will work and `foo.by_ref()` won't.
223    pub from_unsafe_deref: bool,
224    pub unsize: bool,
225    /// We will generate CandidateSteps which are reachable via a chain
226    /// of following `Receiver`. The first 'n' of those will be reachable
227    /// by following a chain of 'Deref' instead (since there's a blanket
228    /// implementation of Receiver for Deref).
229    /// We use the entire set of steps when identifying method candidates
230    /// (e.g. identifying relevant `impl` blocks) but only those that are
231    /// reachable via Deref when examining what the receiver type can
232    /// be converted into by autodereffing.
233    pub reachable_via_deref: bool,
234}

269    fn probe_op<Choice: ProbeChoice<'db>>(
270        &self,
271        mode: Mode,
272        self_ty: Ty<'db>,
273        choice: Choice,
274    ) -> Choice::FinalChoice {
275        let mut orig_values = OriginalQueryValues::default();
276        let query_input = self.infcx.canonicalize_query(self_ty, &mut orig_values);
277        let steps = match mode {
278            Mode::MethodCall => self.method_autoderef_steps(&query_input),
279            Mode::Path => self.infcx.probe(|_| {
280                // Mode::Path - the deref steps is "trivial". This turns
281                // our CanonicalQuery into a "trivial" QueryResponse. This
282                // is a bit inefficient, but I don't think that writing
283                // special handling for this "trivial case" is a good idea.
284
285                let infcx = self.infcx;
286                let (self_ty, var_values) = infcx.instantiate_canonical(&query_input);
287                debug!(?self_ty, ?query_input, "probe_op: Mode::Path");
288                let prev_opaque_entries =
289                    self.infcx.inner.borrow_mut().opaque_types().num_entries();
290                MethodAutoderefStepsResult {
291                    steps: smallvec![CandidateStep {
292                        self_ty: self.infcx.make_query_response_ignoring_pending_obligations(
293                            var_values,
294                            self_ty,
295                            prev_opaque_entries
296                        ),
297                        self_ty_is_opaque: false,
298                        autoderefs: 0,
299                        from_unsafe_deref: false,
300                        unsize: false,
301                        reachable_via_deref: true,
302                    }],
303                    opt_bad_ty: None,
304                    reached_recursion_limit: false,
305                }
306            }),
307        };
308
309        if steps.reached_recursion_limit {
310            // FIXME: Report an error.
311        }
312
313        // If we encountered an `_` type or an error type during autoderef, this is
314        // ambiguous.
315        if let Some(bad_ty) = &steps.opt_bad_ty {
316            if bad_ty.reached_raw_pointer
317                && !self.unstable_features.arbitrary_self_types_pointers
318                && self.edition.at_least_2018()
319            {
320                // this case used to be allowed by the compiler,
321                // so we do a future-compat lint here for the 2015 edition
322                // (see https://github.com/rust-lang/rust/issues/46906)
323                // FIXME: Emit the lint.
324                // self.tcx.node_span_lint(
325                //     lint::builtin::TYVAR_BEHIND_RAW_POINTER,
326                //     scope_expr_id,
327                //     span,
328                //     |lint| {
329                //         lint.primary_message("type annotations needed");
330                //     },
331                // );
332            } else {
333                // Ended up encountering a type variable when doing autoderef,
334                // but it may not be a type variable after processing obligations
335                // in our local `FnCtxt`, so don't call `structurally_resolve_type`.
336                let ty = &bad_ty.ty;
337                let ty = self
338                    .infcx
339                    .instantiate_query_response_and_region_obligations(
340                        &ObligationCause::new(),
341                        self.param_env,
342                        &orig_values,
343                        ty,
344                    )
345                    .unwrap_or_else(|_| panic!("instantiating {:?} failed?", ty));
346                let ty = self.infcx.resolve_vars_if_possible(ty.value);
347                match ty.kind() {
348                    TyKind::Infer(InferTy::TyVar(_)) => {
349                        // FIXME: Report "type annotations needed" error.
350                    }
351                    TyKind::Error(_) => {}
352                    _ => panic!("unexpected bad final type in method autoderef"),
353                };
354                return Choice::final_choice_from_err(MethodError::ErrorReported);
355            }
356        }
357
358        debug!("ProbeContext: steps for self_ty={:?} are {:?}", self_ty, steps);
359
360        // this creates one big transaction so that all type variables etc
361        // that we create during the probe process are removed later
362        self.infcx.probe(|_| {
363            let mut probe_cx = ProbeContext::new(self, mode, &orig_values, &steps.steps, choice);
364
365            probe_cx.assemble_inherent_candidates();
366            probe_cx.assemble_extension_candidates_for_traits_in_scope();
367            Choice::choose(probe_cx)
368        })
369    }

371    fn method_autoderef_steps(
372        &self,
373        self_ty: &Canonical<'db, Ty<'db>>,
374    ) -> MethodAutoderefStepsResult<'db> {
375        self.infcx.probe(|_| {
376            debug!("method_autoderef_steps({:?})", self_ty);
377
378            // We accept not-yet-defined opaque types in the autoderef
379            // chain to support recursive calls. We do error if the final
380            // infer var is not an opaque.
381            let infcx = self.infcx;
382            let (self_ty, inference_vars) = infcx.instantiate_canonical(self_ty);
383            let prev_opaque_entries = infcx.inner.borrow_mut().opaque_types().num_entries();
384
385            let self_ty_is_opaque = |ty: Ty<'_>| {
386                if let TyKind::Infer(InferTy::TyVar(vid)) = ty.kind() {
387                    infcx.has_opaques_with_sub_unified_hidden_type(vid)
388                } else {
389                    false
390                }
391            };
392
393            // If arbitrary self types is not enabled, we follow the chain of
394            // `Deref<Target=T>`. If arbitrary self types is enabled, we instead
395            // follow the chain of `Receiver<Target=T>`, but we also record whether
396            // such types are reachable by following the (potentially shorter)
397            // chain of `Deref<Target=T>`. We will use the first list when finding
398            // potentially relevant function implementations (e.g. relevant impl blocks)
399            // but the second list when determining types that the receiver may be
400            // converted to, in order to find out which of those methods might actually
401            // be callable.
402            let mut autoderef_via_deref =
403                Autoderef::new(infcx, self.param_env, self_ty).include_raw_pointers();
404
405            let mut reached_raw_pointer = false;
406            let arbitrary_self_types_enabled = self.unstable_features.arbitrary_self_types
407                || self.unstable_features.arbitrary_self_types_pointers;
408            let (mut steps, reached_recursion_limit) = if arbitrary_self_types_enabled {
409                let reachable_via_deref =
410                    autoderef_via_deref.by_ref().map(|_| true).chain(std::iter::repeat(false));
411
412                let mut autoderef_via_receiver = Autoderef::new(infcx, self.param_env, self_ty)
413                    .include_raw_pointers()
414                    .use_receiver_trait();
415                let steps = autoderef_via_receiver
416                    .by_ref()
417                    .zip(reachable_via_deref)
418                    .map(|((ty, d), reachable_via_deref)| {
419                        let step = CandidateStep {
420                            self_ty: infcx.make_query_response_ignoring_pending_obligations(
421                                inference_vars,
422                                ty,
423                                prev_opaque_entries,
424                            ),
425                            self_ty_is_opaque: self_ty_is_opaque(ty),
426                            autoderefs: d,
427                            from_unsafe_deref: reached_raw_pointer,
428                            unsize: false,
429                            reachable_via_deref,
430                        };
431                        if ty.is_raw_ptr() {
432                            // all the subsequent steps will be from_unsafe_deref
433                            reached_raw_pointer = true;
434                        }
435                        step
436                    })
437                    .collect::<SmallVec<[_; _]>>();
438                (steps, autoderef_via_receiver.reached_recursion_limit())
439            } else {
440                let steps = autoderef_via_deref
441                    .by_ref()
442                    .map(|(ty, d)| {
443                        let step = CandidateStep {
444                            self_ty: infcx.make_query_response_ignoring_pending_obligations(
445                                inference_vars,
446                                ty,
447                                prev_opaque_entries,
448                            ),
449                            self_ty_is_opaque: self_ty_is_opaque(ty),
450                            autoderefs: d,
451                            from_unsafe_deref: reached_raw_pointer,
452                            unsize: false,
453                            reachable_via_deref: true,
454                        };
455                        if ty.is_raw_ptr() {
456                            // all the subsequent steps will be from_unsafe_deref
457                            reached_raw_pointer = true;
458                        }
459                        step
460                    })
461                    .collect();
462                (steps, autoderef_via_deref.reached_recursion_limit())
463            };
464            let final_ty = autoderef_via_deref.final_ty();
465            let opt_bad_ty = match final_ty.kind() {
466                TyKind::Infer(InferTy::TyVar(_)) if !self_ty_is_opaque(final_ty) => {
467                    Some(MethodAutoderefBadTy {
468                        reached_raw_pointer,
469                        ty: infcx.make_query_response_ignoring_pending_obligations(
470                            inference_vars,
471                            final_ty,
472                            prev_opaque_entries,
473                        ),
474                    })
475                }
476                TyKind::Error(_) => Some(MethodAutoderefBadTy {
477                    reached_raw_pointer,
478                    ty: infcx.make_query_response_ignoring_pending_obligations(
479                        inference_vars,
480                        final_ty,
481                        prev_opaque_entries,
482                    ),
483                }),
484                TyKind::Array(elem_ty, _) => {
485                    let autoderefs = steps.iter().filter(|s| s.reachable_via_deref).count() - 1;
486                    steps.push(CandidateStep {
487                        self_ty: infcx.make_query_response_ignoring_pending_obligations(
488                            inference_vars,
489                            Ty::new_slice(infcx.interner, elem_ty),
490                            prev_opaque_entries,
491                        ),
492                        self_ty_is_opaque: false,
493                        autoderefs,
494                        // this could be from an unsafe deref if we had
495                        // a *mut/const [T; N]
496                        from_unsafe_deref: reached_raw_pointer,
497                        unsize: true,
498                        reachable_via_deref: true, // this is always the final type from
499                                                   // autoderef_via_deref
500                    });
501
502                    None
503                }
504                _ => None,
505            };
506
507            debug!("method_autoderef_steps: steps={:?} opt_bad_ty={:?}", steps, opt_bad_ty);
508            MethodAutoderefStepsResult { steps, opt_bad_ty, reached_recursion_limit }
509        })
510    }
511}

572impl<'db> ProbeChoice<'db> for ProbeForNameChoice<'db> {
573    type Choice = PickResult<'db>;
574    type FinalChoice = PickResult<'db>;
575
576    fn with_impl_or_trait_item<'a>(
577        this: &mut ProbeContext<'a, 'db, Self>,
578        items: &[(Name, AssocItemId)],
579        mut callback: impl FnMut(&mut ProbeContext<'a, 'db, Self>, CandidateId),
580    ) {
581        let item = items
582            .iter()
583            .filter_map(|(name, id)| {
584                let id = match *id {
585                    AssocItemId::FunctionId(id) => id.into(),
586                    AssocItemId::ConstId(id) => id.into(),
587                    AssocItemId::TypeAliasId(_) => return None,
588                };
589                Some((name, id))
590            })
591            .find(|(name, _)| **name == this.choice.item_name)
592            .map(|(_, id)| id)
593            .filter(|id| this.mode == Mode::Path || matches!(id, CandidateId::FunctionId(_)));
594        if let Some(item) = item {
595            callback(this, item);
596        }
597    }
598
599    fn consider_candidates(
600        this: &ProbeContext<'_, 'db, Self>,
601        self_ty: Ty<'db>,
602        mut applicable_candidates: Vec<&Candidate<'db>>,
603    ) -> ControlFlow<Self::Choice> {
604        if applicable_candidates.len() > 1
605            && let Some(pick) =
606                this.collapse_candidates_to_trait_pick(self_ty, &applicable_candidates)
607        {
608            return ControlFlow::Break(Ok(pick));
609        }
610
611        if applicable_candidates.len() > 1 {
612            // We collapse to a subtrait pick *after* filtering unstable candidates
613            // to make sure we don't prefer a unstable subtrait method over a stable
614            // supertrait method.
615            if this.ctx.unstable_features.supertrait_item_shadowing
616                && let Some(pick) =
617                    this.collapse_candidates_to_subtrait_pick(self_ty, &applicable_candidates)
618            {
619                return ControlFlow::Break(Ok(pick));
620            }
621
622            let sources =
623                applicable_candidates.iter().map(|p| this.candidate_source(p, self_ty)).collect();
624            return ControlFlow::Break(Err(MethodError::Ambiguity(sources)));
625        }
626
627        match applicable_candidates.pop() {
628            Some(probe) => ControlFlow::Break(Ok(probe.to_unadjusted_pick(self_ty))),
629            None => ControlFlow::Continue(()),
630        }
631    }
632
633    fn consider_private_candidates(
634        this: &mut ProbeContext<'_, 'db, Self>,
635        self_ty: Ty<'db>,
636        instantiate_self_ty_obligations: &[PredicateObligation<'db>],
637    ) {
638        if this.choice.private_candidate.is_none()
639            && let ControlFlow::Break(Ok(pick)) = this.consider_candidates(
640                self_ty,
641                instantiate_self_ty_obligations,
642                &this.private_candidates,
643                None,
644            )
645        {
646            this.choice.private_candidate = Some(pick);
647        }
648    }
649
650    fn map_choice_pick(
651        choice: Self::Choice,
652        f: impl FnOnce(Pick<'db>) -> Pick<'db>,
653    ) -> Self::Choice {
654        choice.map(f)
655    }
656
657    fn check_by_value_method_shadowing(
658        this: &mut ProbeContext<'_, 'db, Self>,
659        by_value_pick: &Self::Choice,
660        step: &CandidateStep<'db>,
661        self_ty: Ty<'db>,
662        instantiate_self_ty_obligations: &[PredicateObligation<'db>],
663    ) -> ControlFlow<Self::Choice> {
664        if let Ok(by_value_pick) = by_value_pick
665            && matches!(by_value_pick.kind, PickKind::InherentImplPick(_))
666        {
667            for mutbl in [Mutability::Not, Mutability::Mut] {
668                if let Err(e) = this.check_for_shadowed_autorefd_method(
669                    by_value_pick,
670                    step,
671                    self_ty,
672                    instantiate_self_ty_obligations,
673                    mutbl,
674                ) {
675                    return ControlFlow::Break(Err(e));
676                }
677            }
678        }
679        ControlFlow::Continue(())
680    }
681
682    fn check_autorefed_method_shadowing(
683        this: &mut ProbeContext<'_, 'db, Self>,
684        autoref_pick: &Self::Choice,
685        step: &CandidateStep<'db>,
686        self_ty: Ty<'db>,
687        instantiate_self_ty_obligations: &[PredicateObligation<'db>],
688    ) -> ControlFlow<Self::Choice> {
689        if let Ok(autoref_pick) = autoref_pick.as_ref() {
690            // Check we're not shadowing others
691            if matches!(autoref_pick.kind, PickKind::InherentImplPick(_))
692                && let Err(e) = this.check_for_shadowed_autorefd_method(
693                    autoref_pick,
694                    step,
695                    self_ty,
696                    instantiate_self_ty_obligations,
697                    Mutability::Mut,
698                )
699            {
700                return ControlFlow::Break(Err(e));
701            }
702        }
703        ControlFlow::Continue(())
704    }
705
706    fn final_choice_from_err(err: MethodError<'db>) -> Self::FinalChoice {
707        Err(err)
708    }
709
710    fn choose(this: ProbeContext<'_, 'db, Self>) -> Self::FinalChoice {
711        this.pick()
712    }

867    fn push_candidate(&mut self, candidate: Candidate<'db>, is_inherent: bool) {
868        let is_accessible = if is_inherent {
869            let candidate_id = match candidate.item {
870                CandidateId::FunctionId(id) => id.into(),
871                CandidateId::ConstId(id) => id.into(),
872            };
873            let visibility = self.db().assoc_visibility(candidate_id);
874            self.ctx.resolver.is_visible(self.db(), visibility)
875        } else {
876            true
877        };
878        if is_accessible {
879            if is_inherent {
880                self.inherent_candidates.push(candidate);
881            } else {
882                self.extension_candidates.push(candidate);
883            }
884        } else {
885            self.private_candidates.push(candidate);
886        }
887    }

890        for step in self.steps.iter() {

895    #[instrument(level = "debug", skip(self))]
896    fn assemble_probe(
897        &mut self,
898        self_ty: &Canonical<'db, QueryResponse<'db, Ty<'db>>>,
899        receiver_steps: usize,
900    ) {
901        let raw_self_ty = self_ty.value.value;
902        match raw_self_ty.kind() {
903            TyKind::Dynamic(data, ..) => {
904                if let Some(p) = data.principal() {
905                    // Subtle: we can't use `instantiate_query_response` here: using it will
906                    // commit to all of the type equalities assumed by inference going through
907                    // autoderef (see the `method-probe-no-guessing` test).
908                    //
909                    // However, in this code, it is OK if we end up with an object type that is
910                    // "more general" than the object type that we are evaluating. For *every*
911                    // object type `MY_OBJECT`, a function call that goes through a trait-ref
912                    // of the form `<MY_OBJECT as SuperTraitOf(MY_OBJECT)>::func` is a valid
913                    // `ObjectCandidate`, and it should be discoverable "exactly" through one
914                    // of the iterations in the autoderef loop, so there is no problem with it
915                    // being discoverable in another one of these iterations.
916                    //
917                    // Using `instantiate_canonical` on our
918                    // `Canonical<QueryResponse<Ty<'db>>>` and then *throwing away* the
919                    // `CanonicalVarValues` will exactly give us such a generalization - it
920                    // will still match the original object type, but it won't pollute our
921                    // type variables in any form, so just do that!
922                    let (QueryResponse { value: generalized_self_ty, .. }, _ignored_var_values) =
923                        self.infcx().instantiate_canonical(self_ty);
924
925                    self.assemble_inherent_candidates_from_object(generalized_self_ty);
926                    self.assemble_inherent_impl_candidates_for_type(
927                        &SimplifiedType::Trait(p.def_id().0.into()),
928                        receiver_steps,
929                    );
930                    self.assemble_inherent_candidates_for_incoherent_ty(
931                        raw_self_ty,
932                        receiver_steps,
933                    );
934                }
935            }
936            TyKind::Adt(def, _) => {
937                let def_id = def.def_id().0;
938                self.assemble_inherent_impl_candidates_for_type(
939                    &SimplifiedType::Adt(def_id.into()),
940                    receiver_steps,
941                );
942                self.assemble_inherent_candidates_for_incoherent_ty(raw_self_ty, receiver_steps);
943            }
944            TyKind::Foreign(did) => {
945                self.assemble_inherent_impl_candidates_for_type(
946                    &SimplifiedType::Foreign(did.0.into()),
947                    receiver_steps,
948                );
949                self.assemble_inherent_candidates_for_incoherent_ty(raw_self_ty, receiver_steps);
950            }
951            TyKind::Param(_) => {
952                self.assemble_inherent_candidates_from_param(raw_self_ty);
953            }
954            TyKind::Bool
955            | TyKind::Char
956            | TyKind::Int(_)
957            | TyKind::Uint(_)
958            | TyKind::Float(_)
959            | TyKind::Str
960            | TyKind::Array(..)
961            | TyKind::Slice(_)
962            | TyKind::RawPtr(_, _)
963            | TyKind::Ref(..)
964            | TyKind::Never
965            | TyKind::Tuple(..) => {
966                self.assemble_inherent_candidates_for_incoherent_ty(raw_self_ty, receiver_steps)
967            }
968            _ => {}
969        }
970    }

972    fn assemble_inherent_candidates_for_incoherent_ty(
973        &mut self,
974        self_ty: Ty<'db>,
975        receiver_steps: usize,
976    ) {
977        let Some(simp) = simplify_type(self.interner(), self_ty, TreatParams::InstantiateWithInfer)
978        else {
979            panic!("unexpected incoherent type: {:?}", self_ty)
980        };
981        with_incoherent_inherent_impls(self.db(), self.ctx.resolver.krate(), &simp, |impls| {
982            for &impl_def_id in impls {
983                self.assemble_inherent_impl_probe(impl_def_id, receiver_steps);
984            }
985        });
986    }

988    fn assemble_inherent_impl_candidates_for_type(
989        &mut self,
990        self_ty: &SimplifiedType,
991        receiver_steps: usize,
992    ) {
993        let Some(module) = simplified_type_module(self.db(), self_ty) else {
994            return;
995        };
996        InherentImpls::for_each_crate_and_block(
997            self.db(),
998            module.krate(self.db()),
999            module.block(self.db()),
1000            &mut |impls| {
1001                for &impl_def_id in impls.for_self_ty(self_ty) {
1002                    self.assemble_inherent_impl_probe(impl_def_id, receiver_steps);
1003                }
1004            },
1005        );
1006    }

1008    #[instrument(level = "debug", skip(self))]
1009    fn assemble_inherent_impl_probe(&mut self, impl_def_id: ImplId, receiver_steps: usize) {
1010        if !self.impl_dups.insert(impl_def_id) {
1011            return; // already visited
1012        }
1013
1014        self.with_impl_item(impl_def_id, |this, item| {
1015            if !this.has_applicable_self(item) {
1016                // No receiver declared. Not a candidate.
1017                this.record_static_candidate(CandidateSource::Impl(impl_def_id.into()));
1018                return;
1019            }
1020            this.push_candidate(
1021                Candidate { item, kind: InherentImplCandidate { impl_def_id, receiver_steps } },
1022                true,
1023            );
1024        });
1025    }

1027    #[instrument(level = "debug", skip(self))]
1028    fn assemble_inherent_candidates_from_object(&mut self, self_ty: Ty<'db>) {
1029        let principal = match self_ty.kind() {
1030            TyKind::Dynamic(data, ..) => Some(data),
1031            _ => None,
1032        }
1033        .and_then(|data| data.principal())
1034        .unwrap_or_else(|| {
1035            panic!("non-object {:?} in assemble_inherent_candidates_from_object", self_ty)
1036        });
1037
1038        // It is illegal to invoke a method on a trait instance that refers to
1039        // the `Self` type. An [`DynCompatibilityViolation::SupertraitSelf`] error
1040        // will be reported by `dyn_compatibility.rs` if the method refers to the
1041        // `Self` type anywhere other than the receiver. Here, we use a
1042        // instantiation that replaces `Self` with the object type itself. Hence,
1043        // a `&self` method will wind up with an argument type like `&dyn Trait`.
1044        let trait_ref = principal.with_self_ty(self.interner(), self_ty);
1045        self.assemble_candidates_for_bounds(
1046            elaborate::supertraits(self.interner(), trait_ref),
1047            |this, new_trait_ref, item| {
1048                this.push_candidate(Candidate { item, kind: ObjectCandidate(new_trait_ref) }, true);
1049            },
1050        );
1051    }

1053    #[instrument(level = "debug", skip(self))]
1054    fn assemble_inherent_candidates_from_param(&mut self, param_ty: Ty<'db>) {
1055        debug_assert!(matches!(param_ty.kind(), TyKind::Param(_)));
1056
1057        let interner = self.interner();
1058
1059        // We use `DeepRejectCtxt` here which may return false positive on where clauses
1060        // with alias self types. We need to later on reject these as inherent candidates
1061        // in `consider_probe`.
1062        let bounds = self.param_env().clauses.iter().filter_map(|predicate| {
1063            let bound_predicate = predicate.kind();
1064            match bound_predicate.skip_binder() {
1065                ClauseKind::Trait(trait_predicate) => DeepRejectCtxt::relate_rigid_rigid(interner)
1066                    .types_may_unify(param_ty, trait_predicate.trait_ref.self_ty())
1067                    .then(|| bound_predicate.rebind(trait_predicate.trait_ref)),
1068                ClauseKind::RegionOutlives(_)
1069                | ClauseKind::TypeOutlives(_)
1070                | ClauseKind::Projection(_)
1071                | ClauseKind::ConstArgHasType(_, _)
1072                | ClauseKind::WellFormed(_)
1073                | ClauseKind::ConstEvaluatable(_)
1074                | ClauseKind::UnstableFeature(_)
1075                | ClauseKind::HostEffect(..) => None,
1076            }
1077        });
1078
1079        self.assemble_candidates_for_bounds(bounds, |this, poly_trait_ref, item| {
1080            this.push_candidate(
1081                Candidate { item, kind: WhereClauseCandidate(poly_trait_ref) },
1082                true,
1083            );
1084        });
1085    }

1087    // Do a search through a list of bounds, using a callback to actually
1088    // create the candidates.
1089    fn assemble_candidates_for_bounds<F>(
1090        &mut self,
1091        bounds: impl Iterator<Item = PolyTraitRef<'db>>,
1092        mut mk_cand: F,
1093    ) where
1094        F: for<'b> FnMut(&mut ProbeContext<'b, 'db, Choice>, PolyTraitRef<'db>, CandidateId),
1095    {
1096        for bound_trait_ref in bounds {
1097            debug!("elaborate_bounds(bound_trait_ref={:?})", bound_trait_ref);
1098            self.with_trait_item(bound_trait_ref.def_id().0, |this, item| {
1099                if !this.has_applicable_self(item) {
1100                    this.record_static_candidate(CandidateSource::Trait(
1101                        bound_trait_ref.def_id().0,
1102                    ));
1103                } else {
1104                    mk_cand(this, bound_trait_ref, item);
1105                }
1106            });
1107        }
1108    }

1120        let trait_ref = TraitRef::new_from_args(self.interner(), trait_def_id.into(), trait_args);
1121
1122        self.with_trait_item(trait_def_id, |this, item| {
1123            // Check whether `trait_def_id` defines a method with suitable name.

1126                this.record_static_candidate(CandidateSource::Trait(trait_def_id));
1127                return;
1128            }

1132            );
1133        });

1139impl<'a, 'db> ProbeContext<'a, 'db, ProbeForNameChoice<'db>> {
1140    #[instrument(level = "debug", skip(self))]
1141    fn pick(mut self) -> PickResult<'db> {
1142        if let Some(r) = self.pick_core() {
1143            return r;
1144        }
1145
1146        debug!("pick: actual search failed, assemble diagnostics");
1147
1148        if let Some(candidate) = self.choice.private_candidate {
1149            return Err(MethodError::PrivateMatch(candidate));
1150        }
1151
1152        Err(MethodError::NoMatch)
1153    }
1154
1155    fn pick_core(&mut self) -> Option<PickResult<'db>> {
1156        self.pick_all_method().break_value()
1157    }
1158
1159    /// Check for cases where arbitrary self types allows shadowing
1160    /// of methods that might be a compatibility break. Specifically,
1161    /// we have something like:
1162    /// ```ignore (illustrative)
1163    /// struct A;
1164    /// impl A {
1165    ///   fn foo(self: &NonNull<A>) {}
1166    ///      // note this is by reference
1167    /// }
1168    /// ```
1169    /// then we've come along and added this method to `NonNull`:
1170    /// ```ignore (illustrative)
1171    ///   fn foo(self)  // note this is by value
1172    /// ```
1173    /// Report an error in this case.
1174    fn check_for_shadowed_autorefd_method(
1175        &mut self,
1176        possible_shadower: &Pick<'db>,
1177        step: &CandidateStep<'db>,
1178        self_ty: Ty<'db>,
1179        instantiate_self_ty_obligations: &[PredicateObligation<'db>],
1180        mutbl: Mutability,
1181    ) -> Result<(), MethodError<'db>> {
1182        // The errors emitted by this function are part of
1183        // the arbitrary self types work, and should not impact
1184        // other users.
1185        if !self.ctx.unstable_features.arbitrary_self_types
1186            && !self.ctx.unstable_features.arbitrary_self_types_pointers
1187        {
1188            return Ok(());
1189        }
1190
1191        // Set criteria for how we find methods possibly shadowed by 'possible_shadower'
1192        let pick_constraints = PickConstraintsForShadowed {
1193            // It's the same `self` type...
1194            autoderefs: possible_shadower.autoderefs,
1195            // ... but the method was found in an impl block determined
1196            // by searching further along the Receiver chain than the other,
1197            // showing that it's a smart pointer type causing the problem...
1198            receiver_steps: possible_shadower.receiver_steps,
1199            // ... and they don't end up pointing to the same item in the
1200            // first place (could happen with things like blanket impls for T)
1201            def_id: possible_shadower.item,
1202        };
1203        // A note on the autoderefs above. Within pick_by_value_method, an extra
1204        // autoderef may be applied in order to reborrow a reference with
1205        // a different lifetime. That seems as though it would break the
1206        // logic of these constraints, since the number of autoderefs could
1207        // no longer be used to identify the fundamental type of the receiver.
1208        // However, this extra autoderef is applied only to by-value calls
1209        // where the receiver is already a reference. So this situation would
1210        // only occur in cases where the shadowing looks like this:
1211        // ```
1212        // struct A;
1213        // impl A {
1214        //   fn foo(self: &&NonNull<A>) {}
1215        //      // note this is by DOUBLE reference
1216        // }
1217        // ```
1218        // then we've come along and added this method to `NonNull`:
1219        // ```
1220        //   fn foo(&self)  // note this is by single reference
1221        // ```
1222        // and the call is:
1223        // ```
1224        // let bar = NonNull<Foo>;
1225        // let bar = &foo;
1226        // bar.foo();
1227        // ```
1228        // In these circumstances, the logic is wrong, and we wouldn't spot
1229        // the shadowing, because the autoderef-based maths wouldn't line up.
1230        // This is a niche case and we can live without generating an error
1231        // in the case of such shadowing.
1232        let potentially_shadowed_pick = self.pick_autorefd_method(
1233            step,
1234            self_ty,
1235            instantiate_self_ty_obligations,
1236            mutbl,
1237            Some(&pick_constraints),
1238        );
1239        // Look for actual pairs of shadower/shadowed which are
1240        // the sort of shadowing case we want to avoid. Specifically...
1241        if let ControlFlow::Break(Ok(possible_shadowed)) = &potentially_shadowed_pick {
1242            let sources = [possible_shadower, possible_shadowed]
1243                .into_iter()
1244                .map(|p| self.candidate_source_from_pick(p))
1245                .collect();
1246            return Err(MethodError::Ambiguity(sources));
1247        }
1248        Ok(())
1249    }
1250}

1253    fn pick_all_method(&mut self) -> ControlFlow<Choice::Choice> {
1254        self.steps
1255            .iter()
1256            // At this point we're considering the types to which the receiver can be converted,
1257            // so we want to follow the `Deref` chain not the `Receiver` chain. Filter out
1258            // steps which can only be reached by following the (longer) `Receiver` chain.
1259            .filter(|step| step.reachable_via_deref)
1260            .filter(|step| {
1261                debug!("pick_all_method: step={:?}", step);
1262                // Skip types with type errors (but not const/lifetime errors, which are
1263                // often spurious due to incomplete const evaluation) and raw pointer derefs.
1264                !step.self_ty.value.value.references_only_ty_error() && !step.from_unsafe_deref
1265            })
1266            .try_for_each(|step| {
1267                let InferOk { value: self_ty, obligations: instantiate_self_ty_obligations } = self
1268                    .infcx()
1269                    .instantiate_query_response_and_region_obligations(
1270                        &ObligationCause::new(),
1271                        self.param_env(),
1272                        self.orig_steps_var_values,
1273                        &step.self_ty,
1274                    )
1275                    .unwrap_or_else(|_| panic!("{:?} was applicable but now isn't?", step.self_ty));
1276
1277                let by_value_pick =
1278                    self.pick_by_value_method(step, self_ty, &instantiate_self_ty_obligations);
1279
1280                // Check for shadowing of a by-reference method by a by-value method (see comments on check_for_shadowing)
1281                if let ControlFlow::Break(by_value_pick) = by_value_pick {
1282                    Choice::check_by_value_method_shadowing(
1283                        self,
1284                        &by_value_pick,
1285                        step,
1286                        self_ty,
1287                        &instantiate_self_ty_obligations,
1288                    )?;
1289                    return ControlFlow::Break(by_value_pick);
1290                }
1291
1292                let autoref_pick = self.pick_autorefd_method(
1293                    step,
1294                    self_ty,
1295                    &instantiate_self_ty_obligations,
1296                    Mutability::Not,
1297                    None,
1298                );
1299                // Check for shadowing of a by-mut-ref method by a by-reference method (see comments on check_for_shadowing)
1300                if let ControlFlow::Break(autoref_pick) = autoref_pick {
1301                    Choice::check_autorefed_method_shadowing(
1302                        self,
1303                        &autoref_pick,
1304                        step,
1305                        self_ty,
1306                        &instantiate_self_ty_obligations,
1307                    )?;
1308                    return ControlFlow::Break(autoref_pick);
1309                }
1310
1311                // Note that no shadowing errors are produced from here on,
1312                // as we consider const ptr methods.
1313                // We allow new methods that take *mut T to shadow
1314                // methods which took *const T, so there is no entry in
1315                // this list for the results of `pick_const_ptr_method`.
1316                // The reason is that the standard pointer cast method
1317                // (on a mutable pointer) always already shadows the
1318                // cast method (on a const pointer). So, if we added
1319                // `pick_const_ptr_method` to this method, the anti-
1320                // shadowing algorithm would always complain about
1321                // the conflict between *const::cast and *mut::cast.
1322                // In practice therefore this does constrain us:
1323                // we cannot add new
1324                //   self: *mut Self
1325                // methods to types such as NonNull or anything else
1326                // which implements Receiver, because this might in future
1327                // shadow existing methods taking
1328                //   self: *const NonNull<Self>
1329                // in the pointee. In practice, methods taking raw pointers
1330                // are rare, and it seems that it should be easily possible
1331                // to avoid such compatibility breaks.
1332                // We also don't check for reborrowed pin methods which
1333                // may be shadowed; these also seem unlikely to occur.
1334                self.pick_autorefd_method(
1335                    step,
1336                    self_ty,
1337                    &instantiate_self_ty_obligations,
1338                    Mutability::Mut,
1339                    None,
1340                )?;
1341                self.pick_const_ptr_method(step, self_ty, &instantiate_self_ty_obligations)
1342            })
1343    }

1345    /// For each type `T` in the step list, this attempts to find a method where
1346    /// the (transformed) self type is exactly `T`. We do however do one
1347    /// transformation on the adjustment: if we are passing a region pointer in,
1348    /// we will potentially *reborrow* it to a shorter lifetime. This allows us
1349    /// to transparently pass `&mut` pointers, in particular, without consuming
1350    /// them for their entire lifetime.
1351    fn pick_by_value_method(
1352        &mut self,
1353        step: &CandidateStep<'db>,
1354        self_ty: Ty<'db>,
1355        instantiate_self_ty_obligations: &[PredicateObligation<'db>],
1356    ) -> ControlFlow<Choice::Choice> {
1357        if step.unsize {
1358            return ControlFlow::Continue(());
1359        }
1360
1361        self.pick_method(self_ty, instantiate_self_ty_obligations, None).map_break(|r| {
1362            Choice::map_choice_pick(r, |mut pick| {
1363                pick.autoderefs = step.autoderefs;
1364
1365                match step.self_ty.value.value.kind() {
1366                    // Insert a `&*` or `&mut *` if this is a reference type:
1367                    TyKind::Ref(_, _, mutbl) => {
1368                        pick.autoderefs += 1;
1369                        pick.autoref_or_ptr_adjustment = Some(AutorefOrPtrAdjustment::Autoref {
1370                            mutbl,
1371                            unsize: pick.autoref_or_ptr_adjustment.is_some_and(|a| a.get_unsize()),
1372                        })
1373                    }
1374
1375                    _ => (),
1376                }
1377
1378                pick
1379            })
1380        })
1381    }

1383    fn pick_autorefd_method(
1384        &mut self,
1385        step: &CandidateStep<'db>,
1386        self_ty: Ty<'db>,
1387        instantiate_self_ty_obligations: &[PredicateObligation<'db>],
1388        mutbl: Mutability,
1389        pick_constraints: Option<&PickConstraintsForShadowed>,
1390    ) -> ControlFlow<Choice::Choice> {
1391        let interner = self.interner();
1392
1393        if let Some(pick_constraints) = pick_constraints
1394            && !pick_constraints.may_shadow_based_on_autoderefs(step.autoderefs)
1395        {
1396            return ControlFlow::Continue(());
1397        }
1398
1399        // In general, during probing we erase regions.
1400        let region = Region::new_erased(interner);
1401
1402        let autoref_ty = Ty::new_ref(interner, region, self_ty, mutbl);
1403        self.pick_method(autoref_ty, instantiate_self_ty_obligations, pick_constraints).map_break(
1404            |r| {
1405                Choice::map_choice_pick(r, |mut pick| {
1406                    pick.autoderefs = step.autoderefs;
1407                    pick.autoref_or_ptr_adjustment =
1408                        Some(AutorefOrPtrAdjustment::Autoref { mutbl, unsize: step.unsize });
1409                    pick
1410                })
1411            },
1412        )
1413    }

1415    /// If `self_ty` is `*mut T` then this picks `*const T` methods. The reason why we have a
1416    /// special case for this is because going from `*mut T` to `*const T` with autoderefs and
1417    /// autorefs would require dereferencing the pointer, which is not safe.
1418    fn pick_const_ptr_method(
1419        &mut self,
1420        step: &CandidateStep<'db>,
1421        self_ty: Ty<'db>,
1422        instantiate_self_ty_obligations: &[PredicateObligation<'db>],
1423    ) -> ControlFlow<Choice::Choice> {
1424        // Don't convert an unsized reference to ptr
1425        if step.unsize {
1426            return ControlFlow::Continue(());
1427        }
1428
1429        let TyKind::RawPtr(ty, Mutability::Mut) = self_ty.kind() else {
1430            return ControlFlow::Continue(());
1431        };
1432
1433        let const_ptr_ty = Ty::new_ptr(self.interner(), ty, Mutability::Not);
1434        self.pick_method(const_ptr_ty, instantiate_self_ty_obligations, None).map_break(|r| {
1435            Choice::map_choice_pick(r, |mut pick| {
1436                pick.autoderefs = step.autoderefs;
1437                pick.autoref_or_ptr_adjustment = Some(AutorefOrPtrAdjustment::ToConstPtr);
1438                pick
1439            })
1440        })
1441    }

1443    fn pick_method(
1444        &mut self,
1445        self_ty: Ty<'db>,
1446        instantiate_self_ty_obligations: &[PredicateObligation<'db>],
1447        pick_constraints: Option<&PickConstraintsForShadowed>,
1448    ) -> ControlFlow<Choice::Choice> {
1449        debug!("pick_method(self_ty={:?})", self_ty);
1450
1451        for (kind, candidates) in
1452            [("inherent", &self.inherent_candidates), ("extension", &self.extension_candidates)]
1453        {
1454            debug!("searching {} candidates", kind);
1455            self.consider_candidates(
1456                self_ty,
1457                instantiate_self_ty_obligations,
1458                candidates,
1459                pick_constraints,
1460            )?;
1461        }
1462
1463        Choice::consider_private_candidates(self, self_ty, instantiate_self_ty_obligations);
1464
1465        ControlFlow::Continue(())
1466    }

1468    fn consider_candidates(
1469        &self,
1470        self_ty: Ty<'db>,
1471        instantiate_self_ty_obligations: &[PredicateObligation<'db>],
1472        candidates: &[Candidate<'db>],
1473        pick_constraints: Option<&PickConstraintsForShadowed>,
1474    ) -> ControlFlow<Choice::Choice> {
1475        let applicable_candidates: Vec<_> = candidates
1476            .iter()
1477            .filter(|candidate| {
1478                pick_constraints
1479                    .map(|pick_constraints| pick_constraints.candidate_may_shadow(candidate))
1480                    .unwrap_or(true)
1481            })
1482            .filter(|probe| {
1483                self.consider_probe(self_ty, instantiate_self_ty_obligations, probe)
1484                    != ProbeResult::NoMatch
1485            })
1486            .collect();
1487
1488        debug!("applicable_candidates: {:?}", applicable_candidates);
1489
1490        Choice::consider_candidates(self, self_ty, applicable_candidates)
1491    }

1502    /// Used for ambiguous method call error reporting. Uses probing that throws away the result internally,
1503    /// so do not use to make a decision that may lead to a successful compilation.
1504    fn candidate_source(&self, candidate: &Candidate<'db>, self_ty: Ty<'db>) -> CandidateSource {
1505        match candidate.kind {
1506            InherentImplCandidate { impl_def_id, .. } => CandidateSource::Impl(impl_def_id.into()),
1507            ObjectCandidate(trait_ref) | WhereClauseCandidate(trait_ref) => {
1508                CandidateSource::Trait(trait_ref.def_id().0)
1509            }
1510            TraitCandidate(trait_ref) => self.infcx().probe(|_| {
1511                let trait_ref = self.infcx().instantiate_binder_with_fresh_vars(
1512                    BoundRegionConversionTime::FnCall,
1513                    trait_ref,
1514                );
1515                let (xform_self_ty, _) = self.xform_self_ty(
1516                    candidate.item,
1517                    trait_ref.self_ty(),
1518                    trait_ref.args.as_slice(),
1519                );
1520                // Guide the trait selection to show impls that have methods whose type matches
1521                // up with the `self` parameter of the method.
1522                let _ = self
1523                    .infcx()
1524                    .at(&ObligationCause::dummy(), self.param_env())
1525                    .sup(xform_self_ty, self_ty);
1526                match self.select_trait_candidate(trait_ref) {
1527                    Ok(Some(ImplSource::UserDefined(ref impl_data))) => {
1528                        // If only a single impl matches, make the error message point
1529                        // to that impl.
1530                        CandidateSource::Impl(impl_data.impl_def_id)
1531                    }
1532                    _ => CandidateSource::Trait(trait_ref.def_id.0),
1533                }
1534            }),
1535        }
1536    }

1542            WhereClausePick(trait_ref) => CandidateSource::Trait(trait_ref.skip_binder().def_id.0),

1546    #[instrument(level = "debug", skip(self), ret)]
1547    fn consider_probe(
1548        &self,
1549        self_ty: Ty<'db>,
1550        instantiate_self_ty_obligations: &[PredicateObligation<'db>],
1551        probe: &Candidate<'db>,
1552    ) -> ProbeResult {
1553        self.infcx().probe(|_| {
1554            let mut result = ProbeResult::Match;
1555            let cause = &ObligationCause::new();
1556            let mut ocx = ObligationCtxt::new(self.infcx());
1557
1558            // Subtle: we're not *really* instantiating the current self type while
1559            // probing, but instead fully recompute the autoderef steps once we've got
1560            // a final `Pick`. We can't nicely handle these obligations outside of a probe.
1561            //
1562            // We simply handle them for each candidate here for now. That's kinda scuffed
1563            // and ideally we just put them into the `FnCtxt` right away. We need to consider
1564            // them to deal with defining uses in `method_autoderef_steps`.
1565            ocx.register_obligations(instantiate_self_ty_obligations.iter().cloned());
1566            let errors = ocx.try_evaluate_obligations();
1567            if !errors.is_empty() {
1568                unreachable!("unexpected autoderef error {errors:?}");
1569            }
1570
1571            let mut trait_predicate = None;
1572            let (xform_self_ty, xform_ret_ty);
1573
1574            match probe.kind {
1575                InherentImplCandidate { impl_def_id, .. } => {
1576                    let impl_args = self.infcx().fresh_args_for_item(impl_def_id.into());
1577                    let impl_ty =
1578                        self.db().impl_self_ty(impl_def_id).instantiate(self.interner(), impl_args);
1579                    (xform_self_ty, xform_ret_ty) =
1580                        self.xform_self_ty(probe.item, impl_ty, impl_args.as_slice());
1581                    match ocx.relate(
1582                        cause,
1583                        self.param_env(),
1584                        self.variance(),
1585                        self_ty,
1586                        xform_self_ty,
1587                    ) {
1588                        Ok(()) => {}
1589                        Err(err) => {
1590                            debug!("--> cannot relate self-types {:?}", err);
1591                            return ProbeResult::NoMatch;
1592                        }
1593                    }
1594                    // Check whether the impl imposes obligations we have to worry about.
1595                    let impl_bounds = GenericPredicates::query_all(self.db(), impl_def_id.into());
1596                    let impl_bounds = clauses_as_obligations(
1597                        impl_bounds.iter_instantiated_copied(self.interner(), impl_args.as_slice()),
1598                        ObligationCause::new(),
1599                        self.param_env(),
1600                    );
1601                    // Convert the bounds into obligations.
1602                    ocx.register_obligations(impl_bounds);
1603                }
1604                TraitCandidate(poly_trait_ref) => {
1605                    // Some trait methods are excluded for arrays before 2021.
1606                    // (`array.into_iter()` wants a slice iterator for compatibility.)
1607                    if self_ty.is_array() && !self.ctx.edition.at_least_2021() {
1608                        let trait_signature = self.db().trait_signature(poly_trait_ref.def_id().0);
1609                        if trait_signature
1610                            .flags
1611                            .contains(TraitFlags::SKIP_ARRAY_DURING_METHOD_DISPATCH)
1612                        {
1613                            return ProbeResult::NoMatch;
1614                        }
1615                    }
1616
1617                    // Some trait methods are excluded for boxed slices before 2024.
1618                    // (`boxed_slice.into_iter()` wants a slice iterator for compatibility.)
1619                    if self_ty.boxed_ty().is_some_and(Ty::is_slice)
1620                        && !self.ctx.edition.at_least_2024()
1621                    {
1622                        let trait_signature = self.db().trait_signature(poly_trait_ref.def_id().0);
1623                        if trait_signature
1624                            .flags
1625                            .contains(TraitFlags::SKIP_BOXED_SLICE_DURING_METHOD_DISPATCH)
1626                        {
1627                            return ProbeResult::NoMatch;
1628                        }
1629                    }
1630
1631                    let trait_ref = self.infcx().instantiate_binder_with_fresh_vars(
1632                        BoundRegionConversionTime::FnCall,
1633                        poly_trait_ref,
1634                    );
1635                    (xform_self_ty, xform_ret_ty) = self.xform_self_ty(
1636                        probe.item,
1637                        trait_ref.self_ty(),
1638                        trait_ref.args.as_slice(),
1639                    );
1640                    match ocx.relate(
1641                        cause,
1642                        self.param_env(),
1643                        self.variance(),
1644                        self_ty,
1645                        xform_self_ty,
1646                    ) {
1647                        Ok(()) => {}
1648                        Err(err) => {
1649                            debug!("--> cannot relate self-types {:?}", err);
1650                            return ProbeResult::NoMatch;
1651                        }
1652                    }
1653                    let obligation = Obligation::new(
1654                        self.interner(),
1655                        cause.clone(),
1656                        self.param_env(),
1657                        Binder::dummy(trait_ref),
1658                    );
1659
1660                    // We only need this hack to deal with fatal overflow in the old solver.
1661                    ocx.register_obligation(obligation);
1662
1663                    trait_predicate = Some(trait_ref.upcast(self.interner()));
1664                }
1665                ObjectCandidate(poly_trait_ref) | WhereClauseCandidate(poly_trait_ref) => {
1666                    let trait_ref = self.infcx().instantiate_binder_with_fresh_vars(
1667                        BoundRegionConversionTime::FnCall,
1668                        poly_trait_ref,
1669                    );
1670                    (xform_self_ty, xform_ret_ty) = self.xform_self_ty(
1671                        probe.item,
1672                        trait_ref.self_ty(),
1673                        trait_ref.args.as_slice(),
1674                    );
1675
1676                    if matches!(probe.kind, WhereClauseCandidate(_)) {
1677                        // `WhereClauseCandidate` requires that the self type is a param,
1678                        // because it has special behavior with candidate preference as an
1679                        // inherent pick.
1680                        match ocx.structurally_normalize_ty(
1681                            cause,
1682                            self.param_env(),
1683                            trait_ref.self_ty(),
1684                        ) {
1685                            Ok(ty) => {
1686                                if !matches!(ty.kind(), TyKind::Param(_)) {
1687                                    debug!("--> not a param ty: {xform_self_ty:?}");
1688                                    return ProbeResult::NoMatch;
1689                                }
1690                            }
1691                            Err(errors) => {
1692                                debug!("--> cannot relate self-types {:?}", errors);
1693                                return ProbeResult::NoMatch;
1694                            }
1695                        }
1696                    }
1697
1698                    match ocx.relate(
1699                        cause,
1700                        self.param_env(),
1701                        self.variance(),
1702                        self_ty,
1703                        xform_self_ty,
1704                    ) {
1705                        Ok(()) => {}
1706                        Err(err) => {
1707                            debug!("--> cannot relate self-types {:?}", err);
1708                            return ProbeResult::NoMatch;
1709                        }
1710                    }
1711                }
1712            }
1713
1714            // See <https://github.com/rust-lang/trait-system-refactor-initiative/issues/134>.
1715            //
1716            // In the new solver, check the well-formedness of the return type.
1717            // This emulates, in a way, the predicates that fall out of
1718            // normalizing the return type in the old solver.
1719            //
1720            // FIXME(-Znext-solver): We alternatively could check the predicates of
1721            // the method itself hold, but we intentionally do not do this in the old
1722            // solver b/c of cycles, and doing it in the new solver would be stronger.
1723            // This should be fixed in the future, since it likely leads to much better
1724            // method winnowing.
1725            if let Some(xform_ret_ty) = xform_ret_ty {
1726                ocx.register_obligation(Obligation::new(
1727                    self.interner(),
1728                    cause.clone(),
1729                    self.param_env(),
1730                    ClauseKind::WellFormed(xform_ret_ty.into()),
1731                ));
1732            }
1733
1734            if !ocx.try_evaluate_obligations().is_empty() {
1735                result = ProbeResult::NoMatch;
1736            }
1737
1738            if self.should_reject_candidate_due_to_opaque_treated_as_rigid(trait_predicate) {
1739                result = ProbeResult::NoMatch;
1740            }
1741
1742            // FIXME: Need to leak-check here.
1743            // if let Err(_) = self.leak_check(outer_universe, Some(snapshot)) {
1744            //     result = ProbeResult::NoMatch;
1745            // }
1746
1747            result
1748        })
1749    }

1751    /// Trait candidates for not-yet-defined opaque types are a somewhat hacky.
1752    ///
1753    /// We want to only accept trait methods if they were hold even if the
1754    /// opaque types were rigid. To handle this, we both check that for trait
1755    /// candidates the goal were to hold even when treating opaques as rigid,
1756    /// see `rustc_trait_selection::solve::OpaqueTypesJank`.
1757    ///
1758    /// We also check that all opaque types encountered as self types in the
1759    /// autoderef chain don't get constrained when applying the candidate.
1760    /// Importantly, this also handles calling methods taking `&self` on
1761    /// `impl Trait` to reject the "by-self" candidate.
1762    ///
1763    /// This needs to happen at the end of `consider_probe` as we need to take
1764    /// all the constraints from that into account.
1765    #[instrument(level = "debug", skip(self), ret)]
1766    fn should_reject_candidate_due_to_opaque_treated_as_rigid(
1767        &self,
1768        trait_predicate: Option<Predicate<'db>>,
1769    ) -> bool {
1770        // This function is what hacky and doesn't perfectly do what we want it to.
1771        // It's not soundness critical and we should be able to freely improve this
1772        // in the future.
1773        //
1774        // Some concrete edge cases include the fact that `goal_may_hold_opaque_types_jank`
1775        // also fails if there are any constraints opaques which are never used as a self
1776        // type. We also allow where-bounds which are currently ambiguous but end up
1777        // constraining an opaque later on.
1778
1779        // Check whether the trait candidate would not be applicable if the
1780        // opaque type were rigid.
1781        if let Some(predicate) = trait_predicate {
1782            let goal = Goal { param_env: self.param_env(), predicate };
1783            if !self.infcx().goal_may_hold_opaque_types_jank(goal) {
1784                return true;
1785            }
1786        }
1787
1788        // Check whether any opaque types in the autoderef chain have been
1789        // constrained.
1790        for step in self.steps {
1791            if step.self_ty_is_opaque {
1792                debug!(?step.autoderefs, ?step.self_ty, "self_type_is_opaque");
1793                let constrained_opaque = self.infcx().probe(|_| {
1794                    // If we fail to instantiate the self type of this
1795                    // step, this part of the deref-chain is no longer
1796                    // reachable. In this case we don't care about opaque
1797                    // types there.
1798                    let Ok(ok) = self.infcx().instantiate_query_response_and_region_obligations(
1799                        &ObligationCause::new(),
1800                        self.param_env(),
1801                        self.orig_steps_var_values,
1802                        &step.self_ty,
1803                    ) else {
1804                        debug!("failed to instantiate self_ty");
1805                        return false;
1806                    };
1807                    let mut ocx = ObligationCtxt::new(self.infcx());
1808                    let self_ty = ocx.register_infer_ok_obligations(ok);
1809                    if !ocx.try_evaluate_obligations().is_empty() {
1810                        debug!("failed to prove instantiate self_ty obligations");
1811                        return false;
1812                    }
1813
1814                    !self.infcx().resolve_vars_if_possible(self_ty).is_ty_var()
1815                });
1816                if constrained_opaque {
1817                    debug!("opaque type has been constrained");
1818                    return true;
1819                }
1820            }
1821        }
1822
1823        false
1824    }

1826    /// Sometimes we get in a situation where we have multiple probes that are all impls of the
1827    /// same trait, but we don't know which impl to use. In this case, since in all cases the
1828    /// external interface of the method can be determined from the trait, it's ok not to decide.
1829    /// We can basically just collapse all of the probes for various impls into one where-clause
1830    /// probe. This will result in a pending obligation so when more type-info is available we can
1831    /// make the final decision.
1832    ///
1833    /// Example (`tests/ui/methods/method-two-trait-defer-resolution-1.rs`):
1834    ///
1835    /// ```ignore (illustrative)
1836    /// trait Foo { ... }
1837    /// impl Foo for Vec<i32> { ... }
1838    /// impl Foo for Vec<usize> { ... }
1839    /// ```
1840    ///
1841    /// Now imagine the receiver is `Vec<_>`. It doesn't really matter at this time which impl we
1842    /// use, so it's ok to just commit to "using the method from the trait Foo".
1843    fn collapse_candidates_to_trait_pick(
1844        &self,
1845        self_ty: Ty<'db>,
1846        probes: &[&Candidate<'db>],
1847    ) -> Option<Pick<'db>> {
1848        // Do all probes correspond to the same trait?
1849        let ItemContainerId::TraitId(container) = probes[0].item.container(self.db()) else {
1850            return None;
1851        };
1852        for p in &probes[1..] {
1853            let ItemContainerId::TraitId(p_container) = p.item.container(self.db()) else {
1854                return None;
1855            };
1856            if p_container != container {
1857                return None;
1858            }
1859        }
1860
1861        // FIXME: check the return type here somehow.
1862        // If so, just use this trait and call it a day.
1863        Some(Pick {
1864            item: probes[0].item,
1865            kind: TraitPick(container),
1866            autoderefs: 0,
1867            autoref_or_ptr_adjustment: None,
1868            self_ty,
1869            receiver_steps: None,
1870            shadowed_candidates: vec![],
1871        })
1872    }

1874    /// Much like `collapse_candidates_to_trait_pick`, this method allows us to collapse
1875    /// multiple conflicting picks if there is one pick whose trait container is a subtrait
1876    /// of the trait containers of all of the other picks.
1877    ///
1878    /// This implements RFC #3624.
1879    fn collapse_candidates_to_subtrait_pick(
1880        &self,
1881        self_ty: Ty<'db>,
1882        probes: &[&Candidate<'db>],
1883    ) -> Option<Pick<'db>> {
1884        let mut child_candidate = probes[0];
1885        let ItemContainerId::TraitId(mut child_trait) = child_candidate.item.container(self.db())
1886        else {
1887            return None;
1888        };
1889        let mut supertraits: FxHashSet<_> =
1890            supertrait_def_ids(self.interner(), child_trait.into()).collect();
1891
1892        let mut remaining_candidates: Vec<_> = probes[1..].to_vec();
1893        while !remaining_candidates.is_empty() {
1894            let mut made_progress = false;
1895            let mut next_round = vec![];
1896
1897            for remaining_candidate in remaining_candidates {
1898                let ItemContainerId::TraitId(remaining_trait) =
1899                    remaining_candidate.item.container(self.db())
1900                else {
1901                    return None;
1902                };
1903                if supertraits.contains(&remaining_trait.into()) {
1904                    made_progress = true;
1905                    continue;
1906                }
1907
1908                // This pick is not a supertrait of the `child_pick`.
1909                // Check if it's a subtrait of the `child_pick`, instead.
1910                // If it is, then it must have been a subtrait of every
1911                // other pick we've eliminated at this point. It will
1912                // take over at this point.
1913                let remaining_trait_supertraits: FxHashSet<_> =
1914                    supertrait_def_ids(self.interner(), remaining_trait.into()).collect();
1915                if remaining_trait_supertraits.contains(&child_trait.into()) {
1916                    child_candidate = remaining_candidate;
1917                    child_trait = remaining_trait;
1918                    supertraits = remaining_trait_supertraits;
1919                    made_progress = true;
1920                    continue;
1921                }
1922
1923                // `child_pick` is not a supertrait of this pick.
1924                // Don't bail here, since we may be comparing two supertraits
1925                // of a common subtrait. These two supertraits won't be related
1926                // at all, but we will pick them up next round when we find their
1927                // child as we continue iterating in this round.
1928                next_round.push(remaining_candidate);
1929            }
1930
1931            if made_progress {
1932                // If we've made progress, iterate again.
1933                remaining_candidates = next_round;
1934            } else {
1935                // Otherwise, we must have at least two candidates which
1936                // are not related to each other at all.
1937                return None;
1938            }
1939        }
1940
1941        Some(Pick {
1942            item: child_candidate.item,
1943            kind: TraitPick(child_trait),
1944            autoderefs: 0,
1945            autoref_or_ptr_adjustment: None,
1946            self_ty,
1947            shadowed_candidates: probes
1948                .iter()
1949                .map(|c| c.item)
1950                .filter(|item| *item != child_candidate.item)
1951                .collect(),
1952            receiver_steps: None,
1953        })
1954    }

1999    #[instrument(level = "debug", skip(self))]
2000    fn xform_method_sig(&self, method: FunctionId, args: &[GenericArg<'db>]) -> FnSig<'db> {
2001        let fn_sig = self.db().callable_item_signature(method.into());
2002        debug!(?fn_sig);
2003
2004        assert!(!args.has_escaping_bound_vars());
2005
2006        // It is possible for type parameters or early-bound lifetimes
2007        // to appear in the signature of `self`. The generic parameters
2008        // we are given do not include type/lifetime parameters for the
2009        // method yet. So create fresh variables here for those too,
2010        // if there are any.
2011        let generics = self.db().generic_params(method.into());
2012
2013        let xform_fn_sig = if generics.is_empty() {
2014            fn_sig.instantiate(self.interner(), args)
2015        } else {
2016            let args = GenericArgs::for_item(
2017                self.interner(),
2018                method.into(),
2019                |param_index, param_id, _| {
2020                    let i = param_index as usize;
2021                    if i < args.len() {
2022                        args[i]
2023                    } else {
2024                        match param_id {
2025                            GenericParamId::LifetimeParamId(_) => {
2026                                // In general, during probe we erase regions.
2027                                Region::new_erased(self.interner()).into()
2028                            }
2029                            GenericParamId::TypeParamId(_) => self.infcx().next_ty_var().into(),
2030                            GenericParamId::ConstParamId(_) => self.infcx().next_const_var().into(),
2031                        }
2032                    }
2033                },
2034            );
2035            fn_sig.instantiate(self.interner(), args)
2036        };
2037
2038        self.interner().instantiate_bound_regions_with_erased(xform_fn_sig)
2039    }
