44#[derive(Debug)]
45pub(crate) struct ConfirmResult<'db> {
46    pub(crate) callee: MethodCallee<'db>,
47    pub(crate) illegal_sized_bound: bool,
48    pub(crate) adjustments: Box<[Adjustment]>,

51impl<'a, 'db> InferenceContext<'a, 'db> {
52    pub(crate) fn confirm_method(
53        &mut self,
54        pick: &probe::Pick<'db>,
55        unadjusted_self_ty: Ty<'db>,
56        expr: ExprId,
57        generic_args: Option<&HirGenericArgs>,
58    ) -> ConfirmResult<'db> {
59        debug!(
60            "confirm(unadjusted_self_ty={:?}, pick={:?}, generic_args={:?})",
61            unadjusted_self_ty, pick, generic_args,
62        );
63
64        let CandidateId::FunctionId(candidate) = pick.item else {
65            panic!("confirmation is only done for method calls, not path lookups");
66        };
67        let mut confirm_cx = ConfirmContext::new(self, candidate, expr);
68        confirm_cx.confirm(unadjusted_self_ty, pick, generic_args)
69    }
70}

96    fn confirm(
97        &mut self,
98        unadjusted_self_ty: Ty<'db>,
99        pick: &probe::Pick<'db>,
100        generic_args: Option<&HirGenericArgs>,
101    ) -> ConfirmResult<'db> {
102        // Adjust the self expression the user provided and obtain the adjusted type.
103        let (self_ty, adjustments) = self.adjust_self_ty(unadjusted_self_ty, pick);
104
105        // Create generic args for the method's type parameters.
106        let rcvr_args = self.fresh_receiver_args(self_ty, pick);
107        let all_args = self.instantiate_method_args(generic_args, rcvr_args);
108
109        debug!("rcvr_args={rcvr_args:?}, all_args={all_args:?}");
110
111        // Create the final signature for the method, replacing late-bound regions.
112        let (method_sig, method_predicates) =
113            self.instantiate_method_sig(pick, all_args.as_slice());
114
115        // If there is a `Self: Sized` bound and `Self` is a trait object, it is possible that
116        // something which derefs to `Self` actually implements the trait and the caller
117        // wanted to make a static dispatch on it but forgot to import the trait.
118        // See test `tests/ui/issues/issue-35976.rs`.
119        //
120        // In that case, we'll error anyway, but we'll also re-run the search with all traits
121        // in scope, and if we find another method which can be used, we'll output an
122        // appropriate hint suggesting to import the trait.
123        let filler_args = GenericArgs::fill_rest(
124            self.interner(),
125            self.candidate.into(),
126            rcvr_args,
127            |index, id, _| match id {
128                GenericParamId::TypeParamId(id) => Ty::new_param(self.interner(), id, index).into(),
129                GenericParamId::ConstParamId(id) => {
130                    Const::new_param(self.interner(), ParamConst { id, index }).into()
131                }
132                GenericParamId::LifetimeParamId(id) => {
133                    Region::new_early_param(self.interner(), EarlyParamRegion { id, index }).into()
134                }
135            },
136        );
137        let illegal_sized_bound = self.predicates_require_illegal_sized_bound(
138            GenericPredicates::query_all(self.db(), self.candidate.into())
139                .iter_instantiated_copied(self.interner(), filler_args.as_slice()),
140        );
141
142        // Unify the (adjusted) self type with what the method expects.
143        //
144        // SUBTLE: if we want good error messages, because of "guessing" while matching
145        // traits, no trait system method can be called before this point because they
146        // could alter our Self-type, except for normalizing the receiver from the
147        // signature (which is also done during probing).
148        let method_sig_rcvr = method_sig.inputs()[0];
149        debug!(
150            "confirm: self_ty={:?} method_sig_rcvr={:?} method_sig={:?}",
151            self_ty, method_sig_rcvr, method_sig
152        );
153        self.unify_receivers(self_ty, method_sig_rcvr, pick);
154
155        // Make sure nobody calls `drop()` explicitly.
156        self.check_for_illegal_method_calls();
157
158        // Lint when an item is shadowing a supertrait item.
159        self.lint_shadowed_supertrait_items(pick);
160
161        // Add any trait/regions obligations specified on the method's type parameters.
162        // We won't add these if we encountered an illegal sized bound, so that we can use
163        // a custom error in that case.
164        if !illegal_sized_bound {
165            self.add_obligations(method_sig, all_args, method_predicates);
166        }
167
168        // Create the final `MethodCallee`.
169        let callee = MethodCallee { def_id: self.candidate, args: all_args, sig: method_sig };
170        ConfirmResult { callee, illegal_sized_bound, adjustments }
171    }

176    fn adjust_self_ty(
177        &mut self,
178        unadjusted_self_ty: Ty<'db>,
179        pick: &probe::Pick<'db>,
180    ) -> (Ty<'db>, Box<[Adjustment]>) {
181        // Commit the autoderefs by calling `autoderef` again, but this
182        // time writing the results into the various typeck results.
183        let mut autoderef = self.ctx.table.autoderef_with_tracking(unadjusted_self_ty);
184        let Some((mut target, n)) = autoderef.nth(pick.autoderefs) else {
185            return (Ty::new_error(self.interner(), ErrorGuaranteed), Box::new([]));
186        };
187        assert_eq!(n, pick.autoderefs);
188
189        let mut adjustments =
190            self.ctx.table.register_infer_ok(autoderef.adjust_steps_as_infer_ok());
191        match pick.autoref_or_ptr_adjustment {
192            Some(probe::AutorefOrPtrAdjustment::Autoref { mutbl, unsize }) => {
193                let region = self.infcx().next_region_var();
194                // Type we're wrapping in a reference, used later for unsizing
195                let base_ty = target;
196
197                target = Ty::new_ref(self.interner(), region, target, mutbl);
198
199                // Method call receivers are the primary use case
200                // for two-phase borrows.
201                let mutbl = AutoBorrowMutability::new(mutbl, AllowTwoPhase::Yes);
202
203                adjustments.push(Adjustment {
204                    kind: Adjust::Borrow(AutoBorrow::Ref(mutbl)),
205                    target: target.store(),
206                });
207
208                if unsize {
209                    let unsized_ty = if let TyKind::Array(elem_ty, _) = base_ty.kind() {
210                        Ty::new_slice(self.interner(), elem_ty)
211                    } else {
212                        panic!(
213                            "AutorefOrPtrAdjustment's unsize flag should only be set for array ty, found {:?}",
214                            base_ty
215                        )
216                    };
217                    target = Ty::new_ref(self.interner(), region, unsized_ty, mutbl.into());
218                    adjustments.push(Adjustment {
219                        kind: Adjust::Pointer(PointerCast::Unsize),
220                        target: target.store(),
221                    });
222                }
223            }
224            Some(probe::AutorefOrPtrAdjustment::ToConstPtr) => {
225                target = match target.kind() {
226                    TyKind::RawPtr(ty, mutbl) => {
227                        assert!(mutbl.is_mut());
228                        Ty::new_imm_ptr(self.interner(), ty)
229                    }
230                    other => panic!("Cannot adjust receiver type {other:?} to const ptr"),
231                };
232
233                adjustments.push(Adjustment {
234                    kind: Adjust::Pointer(PointerCast::MutToConstPointer),
235                    target: target.store(),
236                });
237            }
238            None => {}
239        }
240
241        (target, adjustments.into_boxed_slice())
242    }

244    /// Returns a set of generic parameters for the method *receiver* where all type and region
245    /// parameters are instantiated with fresh variables. This generic parameters does not include any
246    /// parameters declared on the method itself.
247    ///
248    /// Note that this generic parameters may include late-bound regions from the impl level. If so,
249    /// these are instantiated later in the `instantiate_method_sig` routine.
250    fn fresh_receiver_args(
251        &mut self,
252        self_ty: Ty<'db>,
253        pick: &probe::Pick<'db>,
254    ) -> GenericArgs<'db> {
255        match pick.kind {
256            probe::InherentImplPick(impl_def_id) => {
257                self.infcx().fresh_args_for_item(impl_def_id.into())
258            }
259
260            probe::ObjectPick(trait_def_id) => {
261                // If the trait is not object safe (specifically, we care about when
262                // the receiver is not valid), then there's a chance that we will not
263                // actually be able to recover the object by derefing the receiver like
264                // we should if it were valid.
265                if self.db().dyn_compatibility_of_trait(trait_def_id).is_some() {
266                    return GenericArgs::error_for_item(self.interner(), trait_def_id.into());
267                }
268
269                self.extract_existential_trait_ref(self_ty, |this, object_ty, principal| {
270                    // The object data has no entry for the Self
271                    // Type. For the purposes of this method call, we
272                    // instantiate the object type itself. This
273                    // wouldn't be a sound instantiation in all cases,
274                    // since each instance of the object type is a
275                    // different existential and hence could match
276                    // distinct types (e.g., if `Self` appeared as an
277                    // argument type), but those cases have already
278                    // been ruled out when we deemed the trait to be
279                    // "dyn-compatible".
280                    let original_poly_trait_ref =
281                        principal.with_self_ty(this.interner(), object_ty);
282                    let upcast_poly_trait_ref = this.upcast(original_poly_trait_ref, trait_def_id);
283                    let upcast_trait_ref =
284                        this.instantiate_binder_with_fresh_vars(upcast_poly_trait_ref);
285                    debug!(
286                        "original_poly_trait_ref={:?} upcast_trait_ref={:?} target_trait={:?}",
287                        original_poly_trait_ref, upcast_trait_ref, trait_def_id
288                    );
289                    upcast_trait_ref.args
290                })
291            }
292
293            probe::TraitPick(trait_def_id) => {
294                // Make a trait reference `$0 : Trait<$1...$n>`
295                // consisting entirely of type variables. Later on in
296                // the process we will unify the transformed-self-type
297                // of the method with the actual type in order to
298                // unify some of these variables.
299                self.infcx().fresh_args_for_item(trait_def_id.into())
300            }
301
302            probe::WhereClausePick(poly_trait_ref) => {
303                // Where clauses can have bound regions in them. We need to instantiate
304                // those to convert from a poly-trait-ref to a trait-ref.
305                self.instantiate_binder_with_fresh_vars(poly_trait_ref).args
306            }
307        }
308    }

310    fn extract_existential_trait_ref<R, F>(&self, self_ty: Ty<'db>, mut closure: F) -> R
311    where
312        F: FnMut(&ConfirmContext<'a, 'b, 'db>, Ty<'db>, PolyExistentialTraitRef<'db>) -> R,
313    {
314        // If we specified that this is an object method, then the
315        // self-type ought to be something that can be dereferenced to
316        // yield an object-type (e.g., `&Object` or `Box<Object>`
317        // etc).
318
319        let mut autoderef = self.ctx.table.autoderef(self_ty);
320
321        // We don't need to gate this behind arbitrary self types
322        // per se, but it does make things a bit more gated.
323        if self.ctx.unstable_features.arbitrary_self_types
324            || self.ctx.unstable_features.arbitrary_self_types_pointers
325        {
326            autoderef = autoderef.use_receiver_trait();
327        }
328
329        autoderef
330            .include_raw_pointers()
331            .find_map(|(ty, _)| match ty.kind() {
332                TyKind::Dynamic(data, ..) => Some(closure(
333                    self,
334                    ty,
335                    data.principal().expect("calling trait method on empty object?"),
336                )),
337                _ => None,
338            })
339            .unwrap_or_else(|| {
340                panic!("self-type `{:?}` for ObjectPick never dereferenced to an object", self_ty)
341            })
342    }

344    fn instantiate_method_args(
345        &mut self,
346        generic_args: Option<&HirGenericArgs>,
347        parent_args: GenericArgs<'db>,
348    ) -> GenericArgs<'db> {
349        struct LowererCtx<'a, 'b, 'db> {
350            ctx: &'a mut InferenceContext<'b, 'db>,
351            expr: ExprId,
352            parent_args: &'a [GenericArg<'db>],
353        }
354
355        impl<'db> GenericArgsLowerer<'db> for LowererCtx<'_, '_, 'db> {
356            fn report_len_mismatch(
357                &mut self,
358                def: GenericDefId,
359                provided_count: u32,
360                expected_count: u32,
361                kind: IncorrectGenericsLenKind,
362            ) {
363                self.ctx.push_diagnostic(InferenceDiagnostic::MethodCallIncorrectGenericsLen {
364                    expr: self.expr,
365                    provided_count,
366                    expected_count,
367                    kind,
368                    def,
369                });
370            }
371
372            fn report_arg_mismatch(
373                &mut self,
374                param_id: GenericParamId,
375                arg_idx: u32,
376                has_self_arg: bool,
377            ) {
378                self.ctx.push_diagnostic(InferenceDiagnostic::MethodCallIncorrectGenericsOrder {
379                    expr: self.expr,
380                    param_id,
381                    arg_idx,
382                    has_self_arg,
383                });
384            }
385
386            fn provided_kind(
387                &mut self,
388                param_id: GenericParamId,
389                param: GenericParamDataRef<'_>,
390                arg: &HirGenericArg,
391            ) -> GenericArg<'db> {
392                match (param, arg) {
393                    (
394                        GenericParamDataRef::LifetimeParamData(_),
395                        HirGenericArg::Lifetime(lifetime),
396                    ) => self.ctx.make_body_lifetime(*lifetime).into(),
397                    (GenericParamDataRef::TypeParamData(_), HirGenericArg::Type(type_ref)) => {
398                        self.ctx.make_body_ty(*type_ref).into()
399                    }
400                    (GenericParamDataRef::ConstParamData(_), HirGenericArg::Const(konst)) => {
401                        let GenericParamId::ConstParamId(const_id) = param_id else {
402                            unreachable!("non-const param ID for const param");
403                        };
404                        let const_ty = self.ctx.db.const_param_ty_ns(const_id);
405                        self.ctx.make_body_const(*konst, const_ty).into()
406                    }
407                    _ => unreachable!("unmatching param kinds were passed to `provided_kind()`"),
408                }
409            }
410
411            fn provided_type_like_const(
412                &mut self,
413                const_ty: Ty<'db>,
414                arg: TypeLikeConst<'_>,
415            ) -> Const<'db> {
416                match arg {
417                    TypeLikeConst::Path(path) => self.ctx.make_path_as_body_const(path, const_ty),
418                    TypeLikeConst::Infer => self.ctx.table.next_const_var(),
419                }
420            }
421
422            fn inferred_kind(
423                &mut self,
424                _def: GenericDefId,
425                param_id: GenericParamId,
426                _param: GenericParamDataRef<'_>,
427                _infer_args: bool,
428                _preceding_args: &[GenericArg<'db>],
429            ) -> GenericArg<'db> {
430                // Always create an inference var, even when `infer_args == false`. This helps with diagnostics,
431                // and I think it's also required in the presence of `impl Trait` (that must be inferred).
432                self.ctx.table.next_var_for_param(param_id)
433            }
434
435            fn parent_arg(&mut self, param_idx: u32, _param_id: GenericParamId) -> GenericArg<'db> {
436                self.parent_args[param_idx as usize]
437            }
438
439            fn report_elided_lifetimes_in_path(
440                &mut self,
441                _def: GenericDefId,
442                _expected_count: u32,
443                _hard_error: bool,
444            ) {
445                unreachable!("we set `LifetimeElisionKind::Infer`")
446            }
447
448            fn report_elision_failure(&mut self, _def: GenericDefId, _expected_count: u32) {
449                unreachable!("we set `LifetimeElisionKind::Infer`")
450            }
451
452            fn report_missing_lifetime(&mut self, _def: GenericDefId, _expected_count: u32) {
453                unreachable!("we set `LifetimeElisionKind::Infer`")
454            }
455        }
456
457        substs_from_args_and_bindings(
458            self.db(),
459            self.ctx.body,
460            generic_args,
461            self.candidate.into(),
462            true,
463            LifetimeElisionKind::Infer,
464            false,
465            None,
466            &mut LowererCtx { ctx: self.ctx, expr: self.expr, parent_args: parent_args.as_slice() },
467        )
468    }

470    fn unify_receivers(
471        &mut self,
472        self_ty: Ty<'db>,
473        method_self_ty: Ty<'db>,
474        pick: &probe::Pick<'db>,
475    ) {
476        debug!(
477            "unify_receivers: self_ty={:?} method_self_ty={:?} pick={:?}",
478            self_ty, method_self_ty, pick
479        );
480        let cause = ObligationCause::new();
481        match self.ctx.table.at(&cause).sup(method_self_ty, self_ty) {
482            Ok(infer_ok) => {
483                self.ctx.table.register_infer_ok(infer_ok);
484            }
485            Err(_) => {
486                if self.ctx.unstable_features.arbitrary_self_types {
487                    self.ctx.result.type_mismatches.get_or_insert_default().insert(
488                        self.expr.into(),
489                        TypeMismatch { expected: method_self_ty.store(), actual: self_ty.store() },
490                    );
491                }
492            }
493        }
494    }

496    // NOTE: this returns the *unnormalized* predicates and method sig. Because of
497    // inference guessing, the predicates and method signature can't be normalized
498    // until we unify the `Self` type.
499    fn instantiate_method_sig<'c>(
500        &mut self,
501        pick: &probe::Pick<'db>,
502        all_args: &'c [GenericArg<'db>],
503    ) -> (FnSig<'db>, impl Iterator<Item = PredicateObligation<'db>> + use<'c, 'db>) {
504        debug!("instantiate_method_sig(pick={:?}, all_args={:?})", pick, all_args);
505
506        // Instantiate the bounds on the method with the
507        // type/early-bound-regions instantiations performed. There can
508        // be no late-bound regions appearing here.
509        let def_id = self.candidate;
510        let method_predicates = clauses_as_obligations(
511            GenericPredicates::query_all(self.db(), def_id.into())
512                .iter_instantiated_copied(self.interner(), all_args),
513            ObligationCause::new(),
514            self.ctx.table.param_env,
515        );
516
517        let sig =
518            self.db().callable_item_signature(def_id.into()).instantiate(self.interner(), all_args);
519        debug!("type scheme instantiated, sig={:?}", sig);
520
521        let sig = self.instantiate_binder_with_fresh_vars(sig);
522        debug!("late-bound lifetimes from method instantiated, sig={:?}", sig);
523
524        (sig, method_predicates)
525    }
526
527    fn add_obligations(
528        &mut self,
529        sig: FnSig<'db>,
530        all_args: GenericArgs<'db>,
531        method_predicates: impl Iterator<Item = PredicateObligation<'db>>,
532    ) {
533        debug!("add_obligations: sig={:?} all_args={:?}", sig, all_args);
534
535        self.ctx.table.register_predicates(method_predicates);
536
537        // this is a projection from a trait reference, so we have to
538        // make sure that the trait reference inputs are well-formed.
539        self.ctx.table.add_wf_bounds(all_args);
540
541        // the function type must also be well-formed (this is not
542        // implied by the args being well-formed because of inherent
543        // impls and late-bound regions - see issue #28609).
544        for ty in sig.inputs_and_output {
545            self.ctx.table.register_wf_obligation(ty.into(), ObligationCause::new());
546        }
547    }

552    fn predicates_require_illegal_sized_bound(
553        &self,
554        predicates: impl Iterator<Item = Clause<'db>>,
555    ) -> bool {
556        let Some(sized_def_id) = self.ctx.lang_items.Sized else {
557            return false;
558        };
559
560        elaborate(self.interner(), predicates)
561            // We don't care about regions here.
562            .filter_map(|pred| match pred.kind().skip_binder() {
563                ClauseKind::Trait(trait_pred) if trait_pred.def_id().0 == sized_def_id => {
564                    Some(trait_pred)
565                }
566                _ => None,
567            })
568            .any(|trait_pred| matches!(trait_pred.self_ty().kind(), TyKind::Dynamic(..)))
569    }

589    fn upcast(
590        &self,
591        source_trait_ref: PolyTraitRef<'db>,
592        target_trait_def_id: TraitId,
593    ) -> PolyTraitRef<'db> {
594        let upcast_trait_refs =
595            upcast_choices(self.interner(), source_trait_ref, target_trait_def_id);
596
597        // must be exactly one trait ref or we'd get an ambig error etc
598        if let &[upcast_trait_ref] = upcast_trait_refs.as_slice() {
599            upcast_trait_ref
600        } else {
601            Binder::dummy(TraitRef::new_from_args(
602                self.interner(),
603                target_trait_def_id.into(),
604                GenericArgs::error_for_item(self.interner(), target_trait_def_id.into()),
605            ))
606        }
607    }

609    fn instantiate_binder_with_fresh_vars<T>(&self, value: Binder<'db, T>) -> T
610    where
611        T: TypeFoldable<DbInterner<'db>> + Copy,
612    {
613        self.infcx().instantiate_binder_with_fresh_vars(BoundRegionConversionTime::FnCall, value)
614    }
