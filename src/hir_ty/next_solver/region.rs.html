42impl<'db> Region<'db> {
43    pub fn new(_interner: DbInterner<'db>, kind: RegionKind<'db>) -> Self {
44        let kind = unsafe { std::mem::transmute::<RegionKind<'db>, RegionKind<'static>>(kind) };
45        Self { interned: Interned::new_gc(RegionInterned(kind)) }
46    }
47
48    pub fn inner(&self) -> &RegionKind<'db> {
49        let inner = &self.interned.0;
50        unsafe { std::mem::transmute::<&RegionKind<'static>, &RegionKind<'db>>(inner) }
51    }
52
53    pub fn new_early_param(
54        interner: DbInterner<'db>,
55        early_bound_region: EarlyParamRegion,
56    ) -> Self {
57        Region::new(interner, RegionKind::ReEarlyParam(early_bound_region))
58    }
59
60    pub fn new_placeholder(interner: DbInterner<'db>, placeholder: PlaceholderRegion) -> Self {
61        Region::new(interner, RegionKind::RePlaceholder(placeholder))
62    }
63
64    pub fn new_var(interner: DbInterner<'db>, v: RegionVid) -> Region<'db> {
65        Region::new(interner, RegionKind::ReVar(v))
66    }
67
68    pub fn new_erased(interner: DbInterner<'db>) -> Region<'db> {
69        interner.default_types().regions.erased
70    }
71
72    pub fn new_bound(
73        interner: DbInterner<'db>,
74        index: DebruijnIndex,
75        bound: BoundRegion,
76    ) -> Region<'db> {
77        Region::new(interner, RegionKind::ReBound(BoundVarIndexKind::Bound(index), bound))
78    }
79
80    pub fn is_placeholder(&self) -> bool {
81        matches!(self.inner(), RegionKind::RePlaceholder(..))
82    }
83
84    pub fn is_static(&self) -> bool {
85        matches!(self.inner(), RegionKind::ReStatic)
86    }
87
88    pub fn is_erased(&self) -> bool {
89        matches!(self.inner(), RegionKind::ReErased)
90    }
91
92    pub fn is_var(&self) -> bool {
93        matches!(self.inner(), RegionKind::ReVar(_))
94    }
95
96    pub fn is_error(&self) -> bool {
97        matches!(self.inner(), RegionKind::ReError(_))
98    }
99
100    pub fn error(interner: DbInterner<'db>) -> Self {
101        interner.default_types().regions.error
102    }
103
104    pub fn type_flags(&self) -> TypeFlags {
105        let mut flags = TypeFlags::empty();
106
107        match &self.inner() {
108            RegionKind::ReVar(..) => {
109                flags |= TypeFlags::HAS_FREE_REGIONS;
110                flags |= TypeFlags::HAS_FREE_LOCAL_REGIONS;
111                flags |= TypeFlags::HAS_RE_INFER;
112            }
113            RegionKind::RePlaceholder(..) => {
114                flags |= TypeFlags::HAS_FREE_REGIONS;
115                flags |= TypeFlags::HAS_FREE_LOCAL_REGIONS;
116                flags |= TypeFlags::HAS_RE_PLACEHOLDER;
117            }
118            RegionKind::ReEarlyParam(..) => {
119                flags |= TypeFlags::HAS_FREE_REGIONS;
120                flags |= TypeFlags::HAS_FREE_LOCAL_REGIONS;
121                flags |= TypeFlags::HAS_RE_PARAM;
122            }
123            RegionKind::ReLateParam(..) => {
124                flags |= TypeFlags::HAS_FREE_REGIONS;
125                flags |= TypeFlags::HAS_FREE_LOCAL_REGIONS;
126            }
127            RegionKind::ReStatic => {
128                flags |= TypeFlags::HAS_FREE_REGIONS;
129            }
130            RegionKind::ReBound(BoundVarIndexKind::Canonical, ..) => {
131                flags |= TypeFlags::HAS_RE_BOUND;
132                flags |= TypeFlags::HAS_CANONICAL_BOUND;
133            }
134            RegionKind::ReBound(BoundVarIndexKind::Bound(..), ..) => {
135                flags |= TypeFlags::HAS_RE_BOUND;
136            }
137            RegionKind::ReErased => {
138                flags |= TypeFlags::HAS_RE_ERASED;
139            }
140            RegionKind::ReError(..) => {
141                flags |= TypeFlags::HAS_FREE_REGIONS;
142                flags |= TypeFlags::HAS_ERROR;
143            }
144        }
145
146        flags
147    }
148}

159#[derive(Copy, Clone, PartialEq, Eq, Hash)]
160/// The parameter representation of late-bound function parameters, "some region
161/// at least as big as the scope `fr.scope`".
162///
163/// Similar to a placeholder region as we create `LateParam` regions when entering a binder
164/// except they are always in the root universe and instead of using a boundvar to distinguish
165/// between others we use the `DefId` of the parameter. For this reason the `bound_region` field
166/// should basically always be `BoundRegionKind::Named` as otherwise there is no way of telling
167/// different parameters apart.
168pub struct LateParamRegion {
169    pub scope: SolverDefId,
170    pub bound_region: BoundRegionKind,
171}
172
173impl std::fmt::Debug for LateParamRegion {
174    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
175        write!(f, "ReLateParam({:?}, {:?})", self.scope, self.bound_region)
176    }
177}

179#[derive(Copy, Clone, PartialEq, Eq, Hash)]
180pub enum BoundRegionKind {
181    /// An anonymous region parameter for a given fn (&T)
182    Anon,
183
184    /// Named region parameters for functions (a in &'a T)
185    ///
186    /// The `DefId` is needed to distinguish free regions in
187    /// the event of shadowing.
188    Named(SolverDefId),
189
190    /// Anonymous region for the implicit env pointer parameter
191    /// to a closure
192    ClosureEnv,
193}
194
195impl std::fmt::Debug for BoundRegionKind {
196    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
197        match *self {
198            BoundRegionKind::Anon => write!(f, "BrAnon"),
199            BoundRegionKind::Named(did) => {
200                write!(f, "BrNamed({did:?})")
201            }
202            BoundRegionKind::ClosureEnv => write!(f, "BrEnv"),
203        }
204    }
205}

236impl core::fmt::Debug for BoundRegion {
237    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
238        match &self.kind {
239            BoundRegionKind::Anon => write!(f, "{:?}", self.var),
240            BoundRegionKind::ClosureEnv => write!(f, "{:?}.Env", self.var),
241            BoundRegionKind::Named(def) => {
242                write!(f, "{:?}.Named({:?})", self.var, def)
243            }
244        }
245    }
