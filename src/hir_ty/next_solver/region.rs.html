33impl<'db> Region<'db> {
34    pub fn new(interner: DbInterner<'db>, kind: RegionKind<'db>) -> Self {
35        Region::new_(interner.db(), kind)
36    }
37
38    pub fn inner(&self) -> &RegionKind<'db> {
39        crate::with_attached_db(|db| {
40            let inner = self.kind_(db);
41            // SAFETY: The caller already has access to a `Region<'db>`, so borrowchecking will
42            // make sure that our returned value is valid for the lifetime `'db`.
43            unsafe { std::mem::transmute::<&RegionKind<'_>, &RegionKind<'db>>(inner) }
44        })
45    }
46
47    pub fn new_early_param(
48        interner: DbInterner<'db>,
49        early_bound_region: EarlyParamRegion,
50    ) -> Self {
51        Region::new(interner, RegionKind::ReEarlyParam(early_bound_region))
52    }
53
54    pub fn new_placeholder(interner: DbInterner<'db>, placeholder: PlaceholderRegion) -> Self {
55        Region::new(interner, RegionKind::RePlaceholder(placeholder))
56    }
57
58    pub fn new_var(interner: DbInterner<'db>, v: RegionVid) -> Region<'db> {
59        Region::new(interner, RegionKind::ReVar(v))
60    }
61
62    pub fn new_erased(interner: DbInterner<'db>) -> Region<'db> {
63        Region::new(interner, RegionKind::ReErased)
64    }
65
66    pub fn new_bound(
67        interner: DbInterner<'db>,
68        index: DebruijnIndex,
69        bound: BoundRegion,
70    ) -> Region<'db> {
71        Region::new(interner, RegionKind::ReBound(BoundVarIndexKind::Bound(index), bound))
72    }
73
74    pub fn is_placeholder(&self) -> bool {
75        matches!(self.inner(), RegionKind::RePlaceholder(..))
76    }
77
78    pub fn is_static(&self) -> bool {
79        matches!(self.inner(), RegionKind::ReStatic)
80    }
81
82    pub fn is_var(&self) -> bool {
83        matches!(self.inner(), RegionKind::ReVar(_))
84    }
85
86    pub fn is_error(&self) -> bool {
87        matches!(self.inner(), RegionKind::ReError(_))
88    }
89
90    pub fn error(interner: DbInterner<'db>) -> Self {
91        Region::new(interner, RegionKind::ReError(ErrorGuaranteed))
92    }
93
94    pub fn type_flags(&self) -> TypeFlags {
95        let mut flags = TypeFlags::empty();
96
97        match &self.inner() {
98            RegionKind::ReVar(..) => {
99                flags |= TypeFlags::HAS_FREE_REGIONS;
100                flags |= TypeFlags::HAS_FREE_LOCAL_REGIONS;
101                flags |= TypeFlags::HAS_RE_INFER;
102            }
103            RegionKind::RePlaceholder(..) => {
104                flags |= TypeFlags::HAS_FREE_REGIONS;
105                flags |= TypeFlags::HAS_FREE_LOCAL_REGIONS;
106                flags |= TypeFlags::HAS_RE_PLACEHOLDER;
107            }
108            RegionKind::ReEarlyParam(..) => {
109                flags |= TypeFlags::HAS_FREE_REGIONS;
110                flags |= TypeFlags::HAS_FREE_LOCAL_REGIONS;
111                flags |= TypeFlags::HAS_RE_PARAM;
112            }
113            RegionKind::ReLateParam(..) => {
114                flags |= TypeFlags::HAS_FREE_REGIONS;
115                flags |= TypeFlags::HAS_FREE_LOCAL_REGIONS;
116            }
117            RegionKind::ReStatic => {
118                flags |= TypeFlags::HAS_FREE_REGIONS;
119            }
120            RegionKind::ReBound(BoundVarIndexKind::Canonical, ..) => {
121                flags |= TypeFlags::HAS_RE_BOUND;
122                flags |= TypeFlags::HAS_CANONICAL_BOUND;
123            }
124            RegionKind::ReBound(BoundVarIndexKind::Bound(..), ..) => {
125                flags |= TypeFlags::HAS_RE_BOUND;
126            }
127            RegionKind::ReErased => {
128                flags |= TypeFlags::HAS_RE_ERASED;
129            }
130            RegionKind::ReError(..) => {
131                flags |= TypeFlags::HAS_FREE_REGIONS;
132                flags |= TypeFlags::HAS_ERROR;
133            }
134        }
135
136        flags
137    }
138}

149#[derive(Copy, Clone, PartialEq, Eq, Hash)]
150/// The parameter representation of late-bound function parameters, "some region
151/// at least as big as the scope `fr.scope`".
152///
153/// Similar to a placeholder region as we create `LateParam` regions when entering a binder
154/// except they are always in the root universe and instead of using a boundvar to distinguish
155/// between others we use the `DefId` of the parameter. For this reason the `bound_region` field
156/// should basically always be `BoundRegionKind::Named` as otherwise there is no way of telling
157/// different parameters apart.
158pub struct LateParamRegion {
159    pub scope: SolverDefId,
160    pub bound_region: BoundRegionKind,
161}
162
163impl std::fmt::Debug for LateParamRegion {
164    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
165        write!(f, "ReLateParam({:?}, {:?})", self.scope, self.bound_region)
166    }
167}

169#[derive(Copy, Clone, PartialEq, Eq, Hash)]
170pub enum BoundRegionKind {
171    /// An anonymous region parameter for a given fn (&T)
172    Anon,
173
174    /// Named region parameters for functions (a in &'a T)
175    ///
176    /// The `DefId` is needed to distinguish free regions in
177    /// the event of shadowing.
178    Named(SolverDefId),
179
180    /// Anonymous region for the implicit env pointer parameter
181    /// to a closure
182    ClosureEnv,
183}
184
185impl std::fmt::Debug for BoundRegionKind {
186    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
187        match *self {
188            BoundRegionKind::Anon => write!(f, "BrAnon"),
189            BoundRegionKind::Named(did) => {
190                write!(f, "BrNamed({did:?})")
191            }
192            BoundRegionKind::ClosureEnv => write!(f, "BrEnv"),
193        }
194    }
195}

226impl core::fmt::Debug for BoundRegion {
227    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
228        match &self.kind {
229            BoundRegionKind::Anon => write!(f, "{:?}", self.var),
230            BoundRegionKind::ClosureEnv => write!(f, "{:?}.Env", self.var),
231            BoundRegionKind::Named(def) => {
232                write!(f, "{:?}.Named({:?})", self.var, def)
233            }
234        }
235    }
236}
