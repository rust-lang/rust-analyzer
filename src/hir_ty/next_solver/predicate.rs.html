53/// Compares via an ordering that will not change if modules are reordered or other changes are
54/// made to the tree. In particular, this ordering is preserved across incremental compilations.
55fn stable_cmp_existential_predicate<'db>(
56    a: &ExistentialPredicate<'db>,
57    b: &ExistentialPredicate<'db>,
58) -> Ordering {
59    // FIXME: this is actual unstable - see impl in predicate.rs in `rustc_middle`
60    match (a, b) {
61        (ExistentialPredicate::Trait(_), ExistentialPredicate::Trait(_)) => Ordering::Equal,
62        (ExistentialPredicate::Projection(_a), ExistentialPredicate::Projection(_b)) => {
63            // Should sort by def path hash
64            Ordering::Equal
65        }
66        (ExistentialPredicate::AutoTrait(_a), ExistentialPredicate::AutoTrait(_b)) => {
67            // Should sort by def path hash
68            Ordering::Equal
69        }
70        (ExistentialPredicate::Trait(_), _) => Ordering::Less,
71        (ExistentialPredicate::Projection(_), ExistentialPredicate::Trait(_)) => Ordering::Greater,
72        (ExistentialPredicate::Projection(_), _) => Ordering::Less,
73        (ExistentialPredicate::AutoTrait(_), _) => Ordering::Greater,
74    }
75}

86impl<'db> rustc_type_ir::inherent::BoundExistentialPredicates<DbInterner<'db>>
87    for BoundExistentialPredicates<'db>
88{
89    fn principal_def_id(self) -> Option<TraitIdWrapper> {
90        self.principal().map(|trait_ref| trait_ref.skip_binder().def_id)
91    }
92
93    fn principal(
94        self,
95    ) -> Option<
96        rustc_type_ir::Binder<DbInterner<'db>, rustc_type_ir::ExistentialTraitRef<DbInterner<'db>>>,
97    > {
98        self[0]
99            .map_bound(|this| match this {
100                ExistentialPredicate::Trait(tr) => Some(tr),
101                _ => None,
102            })
103            .transpose()
104    }
105
106    fn auto_traits(self) -> impl IntoIterator<Item = TraitIdWrapper> {
107        self.iter().filter_map(|predicate| match predicate.skip_binder() {
108            ExistentialPredicate::AutoTrait(did) => Some(did),
109            _ => None,
110        })
111    }
112
113    fn projection_bounds(
114        self,
115    ) -> impl IntoIterator<
116        Item = rustc_type_ir::Binder<
117            DbInterner<'db>,
118            rustc_type_ir::ExistentialProjection<DbInterner<'db>>,
119        >,
120    > {
121        self.iter().filter_map(|predicate| {
122            predicate
123                .map_bound(|pred| match pred {
124                    ExistentialPredicate::Projection(projection) => Some(projection),
125                    _ => None,
126                })
127                .transpose()
128        })
129    }

132impl<'db> rustc_type_ir::relate::Relate<DbInterner<'db>> for BoundExistentialPredicates<'db> {
133    fn relate<R: rustc_type_ir::relate::TypeRelation<DbInterner<'db>>>(
134        relation: &mut R,
135        a: Self,
136        b: Self,
137    ) -> rustc_type_ir::relate::RelateResult<DbInterner<'db>, Self> {
138        let interner = relation.cx();
139
140        // We need to perform this deduplication as we sometimes generate duplicate projections in `a`.
141        let mut a_v: Vec<_> = a.into_iter().collect();
142        let mut b_v: Vec<_> = b.into_iter().collect();
143        // `skip_binder` here is okay because `stable_cmp` doesn't look at binders
144        a_v.sort_by(|a, b| {
145            stable_cmp_existential_predicate(a.as_ref().skip_binder(), b.as_ref().skip_binder())
146        });
147        a_v.dedup();
148        b_v.sort_by(|a, b| {
149            stable_cmp_existential_predicate(a.as_ref().skip_binder(), b.as_ref().skip_binder())
150        });
151        b_v.dedup();
152        if a_v.len() != b_v.len() {
153            return Err(TypeError::ExistentialMismatch(ExpectedFound::new(a, b)));
154        }
155
156        let v = std::iter::zip(a_v, b_v).map(
157            |(ep_a, ep_b): (
158                Binder<'_, ty::ExistentialPredicate<_>>,
159                Binder<'_, ty::ExistentialPredicate<_>>,
160            )| {
161                match (ep_a.skip_binder(), ep_b.skip_binder()) {
162                    (ty::ExistentialPredicate::Trait(a), ty::ExistentialPredicate::Trait(b)) => {
163                        Ok(ep_a.rebind(ty::ExistentialPredicate::Trait(
164                            relation.relate(ep_a.rebind(a), ep_b.rebind(b))?.skip_binder(),
165                        )))
166                    }
167                    (
168                        ty::ExistentialPredicate::Projection(a),
169                        ty::ExistentialPredicate::Projection(b),
170                    ) => Ok(ep_a.rebind(ty::ExistentialPredicate::Projection(
171                        relation.relate(ep_a.rebind(a), ep_b.rebind(b))?.skip_binder(),
172                    ))),
173                    (
174                        ty::ExistentialPredicate::AutoTrait(a),
175                        ty::ExistentialPredicate::AutoTrait(b),
176                    ) if a == b => Ok(ep_a.rebind(ty::ExistentialPredicate::AutoTrait(a))),
177                    _ => Err(TypeError::ExistentialMismatch(ExpectedFound::new(a, b))),
178                }
179            },
180        );
181
182        BoundExistentialPredicates::new_from_iter(interner, v)
183    }
184}

201impl<'db> Predicate<'db> {
202    pub fn new(_interner: DbInterner<'db>, kind: Binder<'db, PredicateKind<'db>>) -> Self {
203        let kind = unsafe {
204            std::mem::transmute::<
205                Binder<'db, PredicateKind<'db>>,
206                Binder<'static, PredicateKind<'static>>,
207            >(kind)
208        };
209        let flags = FlagComputation::for_predicate(kind);
210        let cached = WithCachedTypeInfo {
211            internee: kind,
212            flags: flags.flags,
213            outer_exclusive_binder: flags.outer_exclusive_binder,
214        };
215        Self { interned: Interned::new_gc(PredicateInterned(cached)) }
216    }
217
218    pub fn inner(&self) -> &WithCachedTypeInfo<Binder<'db, PredicateKind<'db>>> {
219        let inner = &self.interned.0;
220        unsafe {
221            std::mem::transmute::<
222                &WithCachedTypeInfo<Binder<'static, PredicateKind<'static>>>,
223                &WithCachedTypeInfo<Binder<'db, PredicateKind<'db>>>,
224            >(inner)
225        }
226    }
227
228    /// Flips the polarity of a Predicate.
229    ///
230    /// Given `T: Trait` predicate it returns `T: !Trait` and given `T: !Trait` returns `T: Trait`.
231    pub fn flip_polarity(self) -> Option<Predicate<'db>> {
232        let kind = self
233            .kind()
234            .map_bound(|kind| match kind {
235                PredicateKind::Clause(ClauseKind::Trait(TraitPredicate {
236                    trait_ref,
237                    polarity,
238                })) => Some(PredicateKind::Clause(ClauseKind::Trait(TraitPredicate {
239                    trait_ref,
240                    polarity: polarity.flip(),
241                }))),
242
243                _ => None,
244            })
245            .transpose()?;
246
247        Some(Predicate::new(DbInterner::conjure(), kind))
248    }
249}

701impl<'db> rustc_type_ir::inherent::Predicate<DbInterner<'db>> for Predicate<'db> {
702    fn as_clause(self) -> Option<<DbInterner<'db> as rustc_type_ir::Interner>::Clause> {
703        match self.kind().skip_binder() {
704            PredicateKind::Clause(..) => Some(self.expect_clause()),
705            _ => None,
706        }
707    }
708
709    /// Whether this projection can be soundly normalized.
710    ///
711    /// Wf predicates must not be normalized, as normalization
712    /// can remove required bounds which would cause us to
713    /// unsoundly accept some programs. See #91068.
714    fn allow_normalization(self) -> bool {
715        // TODO: this should probably live in rustc_type_ir
716        match self.inner().as_ref().skip_binder() {
717            PredicateKind::Clause(ClauseKind::WellFormed(_)) | PredicateKind::AliasRelate(..) => {
718                false
719            }
720            PredicateKind::Clause(ClauseKind::Trait(_))
721            | PredicateKind::Clause(ClauseKind::RegionOutlives(_))
722            | PredicateKind::Clause(ClauseKind::TypeOutlives(_))
723            | PredicateKind::Clause(ClauseKind::Projection(_))
724            | PredicateKind::Clause(ClauseKind::ConstArgHasType(..))
725            | PredicateKind::Clause(ClauseKind::HostEffect(..))
726            | PredicateKind::Clause(ClauseKind::UnstableFeature(_))
727            | PredicateKind::DynCompatible(_)
728            | PredicateKind::Subtype(_)
729            | PredicateKind::Coerce(_)
730            | PredicateKind::Clause(ClauseKind::ConstEvaluatable(_))
731            | PredicateKind::ConstEquate(_, _)
732            | PredicateKind::NormalizesTo(..)
733            | PredicateKind::Ambiguous => true,
734        }
735    }
736}

896impl<'db> rustc_type_ir::inherent::Clause<DbInterner<'db>> for Clause<'db> {
897    fn as_predicate(self) -> <DbInterner<'db> as rustc_type_ir::Interner>::Predicate {
898        self.0
899    }
900
901    fn instantiate_supertrait(
902        self,
903        cx: DbInterner<'db>,
904        trait_ref: rustc_type_ir::Binder<DbInterner<'db>, rustc_type_ir::TraitRef<DbInterner<'db>>>,
905    ) -> Self {
906        tracing::debug!(?self, ?trait_ref);
907        // See the rustc impl for a long comment
908        let bound_pred = self.kind();
909        let pred_bound_vars = bound_pred.bound_vars();
910        let trait_bound_vars = trait_ref.bound_vars();
911        // 1) Self: Bar1<'a, '^0.0> -> Self: Bar1<'a, '^0.1>
912        let shifted_pred =
913            cx.shift_bound_var_indices(trait_bound_vars.len(), bound_pred.skip_binder());
914        // 2) Self: Bar1<'a, '^0.1> -> T: Bar1<'^0.0, '^0.1>
915        let new = EarlyBinder::bind(shifted_pred).instantiate(cx, trait_ref.skip_binder().args);
916        // 3) ['x] + ['b] -> ['x, 'b]
917        let bound_vars =
918            BoundVarKinds::new_from_iter(cx, trait_bound_vars.iter().chain(pred_bound_vars.iter()));
919
920        let predicate: Predicate<'db> =
921            ty::Binder::bind_with_vars(PredicateKind::Clause(new), bound_vars).upcast(cx);
922        predicate.expect_clause()
923    }
