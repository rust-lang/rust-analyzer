47/// Compares via an ordering that will not change if modules are reordered or other changes are
48/// made to the tree. In particular, this ordering is preserved across incremental compilations.
49fn stable_cmp_existential_predicate<'db>(
50    a: &ExistentialPredicate<'db>,
51    b: &ExistentialPredicate<'db>,
52) -> Ordering {
53    // FIXME: this is actual unstable - see impl in predicate.rs in `rustc_middle`
54    match (a, b) {
55        (ExistentialPredicate::Trait(_), ExistentialPredicate::Trait(_)) => Ordering::Equal,
56        (ExistentialPredicate::Projection(_a), ExistentialPredicate::Projection(_b)) => {
57            // Should sort by def path hash
58            Ordering::Equal
59        }
60        (ExistentialPredicate::AutoTrait(_a), ExistentialPredicate::AutoTrait(_b)) => {
61            // Should sort by def path hash
62            Ordering::Equal
63        }
64        (ExistentialPredicate::Trait(_), _) => Ordering::Less,
65        (ExistentialPredicate::Projection(_), ExistentialPredicate::Trait(_)) => Ordering::Greater,
66        (ExistentialPredicate::Projection(_), _) => Ordering::Less,
67        (ExistentialPredicate::AutoTrait(_), _) => Ordering::Greater,
68    }
69}

108            predicate
109                .map_bound(|pred| match pred {

118impl<'db> rustc_type_ir::relate::Relate<DbInterner<'db>> for BoundExistentialPredicates<'db> {
119    fn relate<R: rustc_type_ir::relate::TypeRelation<DbInterner<'db>>>(
120        relation: &mut R,
121        a: Self,
122        b: Self,
123    ) -> rustc_type_ir::relate::RelateResult<DbInterner<'db>, Self> {
124        let interner = relation.cx();
125
126        // We need to perform this deduplication as we sometimes generate duplicate projections in `a`.
127        let mut a_v: Vec<_> = a.into_iter().collect();
128        let mut b_v: Vec<_> = b.into_iter().collect();
129        // `skip_binder` here is okay because `stable_cmp` doesn't look at binders
130        a_v.sort_by(|a, b| {
131            stable_cmp_existential_predicate(a.as_ref().skip_binder(), b.as_ref().skip_binder())
132        });
133        a_v.dedup();
134        b_v.sort_by(|a, b| {
135            stable_cmp_existential_predicate(a.as_ref().skip_binder(), b.as_ref().skip_binder())
136        });
137        b_v.dedup();
138        if a_v.len() != b_v.len() {
139            return Err(TypeError::ExistentialMismatch(ExpectedFound::new(a, b)));
140        }
141
142        let v = std::iter::zip(a_v, b_v).map(
143            |(ep_a, ep_b): (
144                Binder<'_, ty::ExistentialPredicate<_>>,
145                Binder<'_, ty::ExistentialPredicate<_>>,
146            )| {
147                match (ep_a.skip_binder(), ep_b.skip_binder()) {
148                    (ty::ExistentialPredicate::Trait(a), ty::ExistentialPredicate::Trait(b)) => {
149                        Ok(ep_a.rebind(ty::ExistentialPredicate::Trait(
150                            relation.relate(ep_a.rebind(a), ep_b.rebind(b))?.skip_binder(),
151                        )))
152                    }
153                    (
154                        ty::ExistentialPredicate::Projection(a),
155                        ty::ExistentialPredicate::Projection(b),
156                    ) => Ok(ep_a.rebind(ty::ExistentialPredicate::Projection(
157                        relation.relate(ep_a.rebind(a), ep_b.rebind(b))?.skip_binder(),
158                    ))),
159                    (
160                        ty::ExistentialPredicate::AutoTrait(a),
161                        ty::ExistentialPredicate::AutoTrait(b),
162                    ) if a == b => Ok(ep_a.rebind(ty::ExistentialPredicate::AutoTrait(a))),
163                    _ => Err(TypeError::ExistentialMismatch(ExpectedFound::new(a, b))),
164                }
165            },
166        );
167
168        CollectAndApply::collect_and_apply(v, |g| {
169            BoundExistentialPredicates::new_from_iter(interner, g.iter().cloned())
170        })
171    }
172}

186impl<'db> std::fmt::Debug
187    for InternedWrapperNoDebug<WithCachedTypeInfo<Binder<'db, PredicateKind<'db>>>>
188{
189    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
190        write!(f, "Binder<")?;
191        match self.0.internee.skip_binder() {
192            rustc_type_ir::PredicateKind::Clause(clause_kind) => {
193                write!(f, "{clause_kind:?}")
194            }
195            rustc_type_ir::PredicateKind::DynCompatible(trait_def_id) => {
196                write!(f, "the trait `{trait_def_id:?}` is dyn-compatible")
197            }
198            rustc_type_ir::PredicateKind::Subtype(subtype_predicate) => {
199                write!(f, "{subtype_predicate:?}")
200            }
201            rustc_type_ir::PredicateKind::Coerce(coerce_predicate) => {
202                write!(f, "{coerce_predicate:?}")
203            }
204            rustc_type_ir::PredicateKind::ConstEquate(c1, c2) => {
205                write!(f, "the constant `{c1:?}` equals `{c2:?}`")
206            }
207            rustc_type_ir::PredicateKind::Ambiguous => write!(f, "ambiguous"),
208            rustc_type_ir::PredicateKind::NormalizesTo(data) => write!(f, "{data:?}"),
209            rustc_type_ir::PredicateKind::AliasRelate(t1, t2, dir) => {
210                write!(f, "{t1:?} {dir:?} {t2:?}")
211            }
212        }?;
213        write!(f, ", [{:?}]>", self.0.internee.bound_vars())?;
214        Ok(())
215    }
216}

218impl<'db> Predicate<'db> {
219    pub fn new(interner: DbInterner<'db>, kind: Binder<'db, PredicateKind<'db>>) -> Self {
220        let flags = FlagComputation::for_predicate(kind);
221        let cached = WithCachedTypeInfo {
222            internee: kind,
223            flags: flags.flags,
224            outer_exclusive_binder: flags.outer_exclusive_binder,
225        };
226        Predicate::new_(interner.db(), InternedWrapperNoDebug(cached))
227    }
228
229    pub fn inner(&self) -> &WithCachedTypeInfo<Binder<'db, PredicateKind<'db>>> {
230        crate::with_attached_db(|db| {
231            let inner = &self.kind_(db).0;
232            // SAFETY: The caller already has access to a `Predicate<'db>`, so borrowchecking will
233            // make sure that our returned value is valid for the lifetime `'db`.
234            unsafe { std::mem::transmute(inner) }
235        })
236    }
237
238    /// Flips the polarity of a Predicate.
239    ///
240    /// Given `T: Trait` predicate it returns `T: !Trait` and given `T: !Trait` returns `T: Trait`.
241    pub fn flip_polarity(self) -> Option<Predicate<'db>> {
242        let kind = self
243            .kind()
244            .map_bound(|kind| match kind {
245                PredicateKind::Clause(ClauseKind::Trait(TraitPredicate {
246                    trait_ref,
247                    polarity,
248                })) => Some(PredicateKind::Clause(ClauseKind::Trait(TraitPredicate {
249                    trait_ref,
250                    polarity: polarity.flip(),
251                }))),
252
253                _ => None,
254            })
255            .transpose()?;
256
257        Some(Predicate::new(DbInterner::conjure(), kind))
258    }
259}

669impl<'db> rustc_type_ir::inherent::Predicate<DbInterner<'db>> for Predicate<'db> {
670    fn as_clause(self) -> Option<<DbInterner<'db> as rustc_type_ir::Interner>::Clause> {
671        match self.kind().skip_binder() {
672            PredicateKind::Clause(..) => Some(self.expect_clause()),
673            _ => None,
674        }
675    }
676
677    /// Whether this projection can be soundly normalized.
678    ///
679    /// Wf predicates must not be normalized, as normalization
680    /// can remove required bounds which would cause us to
681    /// unsoundly accept some programs. See #91068.
682    fn allow_normalization(self) -> bool {
683        // TODO: this should probably live in rustc_type_ir
684        match self.inner().as_ref().skip_binder() {
685            PredicateKind::Clause(ClauseKind::WellFormed(_))
686            | PredicateKind::AliasRelate(..)
687            | PredicateKind::NormalizesTo(..) => false,
688            PredicateKind::Clause(ClauseKind::Trait(_))
689            | PredicateKind::Clause(ClauseKind::RegionOutlives(_))
690            | PredicateKind::Clause(ClauseKind::TypeOutlives(_))
691            | PredicateKind::Clause(ClauseKind::Projection(_))
692            | PredicateKind::Clause(ClauseKind::ConstArgHasType(..))
693            | PredicateKind::Clause(ClauseKind::HostEffect(..))
694            | PredicateKind::Clause(ClauseKind::UnstableFeature(_))
695            | PredicateKind::DynCompatible(_)
696            | PredicateKind::Subtype(_)
697            | PredicateKind::Coerce(_)
698            | PredicateKind::Clause(ClauseKind::ConstEvaluatable(_))
699            | PredicateKind::ConstEquate(_, _)
700            | PredicateKind::Ambiguous => true,
701        }
702    }
703}

863impl<'db> rustc_type_ir::inherent::Clause<DbInterner<'db>> for Clause<'db> {
864    fn as_predicate(self) -> <DbInterner<'db> as rustc_type_ir::Interner>::Predicate {
865        self.0
866    }
867
868    fn instantiate_supertrait(
869        self,
870        cx: DbInterner<'db>,
871        trait_ref: rustc_type_ir::Binder<DbInterner<'db>, rustc_type_ir::TraitRef<DbInterner<'db>>>,
872    ) -> Self {
873        tracing::debug!(?self, ?trait_ref);
874        // See the rustc impl for a long comment
875        let bound_pred = self.kind();
876        let pred_bound_vars = bound_pred.bound_vars();
877        let trait_bound_vars = trait_ref.bound_vars();
878        // 1) Self: Bar1<'a, '^0.0> -> Self: Bar1<'a, '^0.1>
879        let shifted_pred =
880            cx.shift_bound_var_indices(trait_bound_vars.len(), bound_pred.skip_binder());
881        // 2) Self: Bar1<'a, '^0.1> -> T: Bar1<'^0.0, '^0.1>
882        let new = EarlyBinder::bind(shifted_pred).instantiate(cx, trait_ref.skip_binder().args);
883        // 3) ['x] + ['b] -> ['x, 'b]
884        let bound_vars =
885            BoundVarKinds::new_from_iter(cx, trait_bound_vars.iter().chain(pred_bound_vars.iter()));
886
887        let predicate: Predicate<'db> =
888            ty::Binder::bind_with_vars(PredicateKind::Clause(new), bound_vars).upcast(cx);
889        predicate.expect_clause()
890    }
