63macro_rules! interned_slice {
64    ($storage:ident, $name:ident, $stored_name:ident, $default_types_field:ident, $ty_db:ty, $ty_static:ty $(,)?) => {
65        const _: () = {
66            #[allow(unused_lifetimes)]
67            fn _ensure_correct_types<'db: 'static>(v: $ty_db) -> $ty_static { v }
68        };
69
70        ::intern::impl_slice_internable!(gc; $storage, (), $ty_static);
71
72        #[derive(Clone, Copy, PartialEq, Eq, Hash)]
73        pub struct $name<'db> {
74            interned: ::intern::InternedSliceRef<'db, $storage>,
75        }
76
77        impl<'db> std::fmt::Debug for $name<'db> {
78            fn fmt(&self, fmt: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
79                self.as_slice().fmt(fmt)
80            }
81        }
82
83        impl<'db> $name<'db> {
84            #[inline]
85            pub fn empty(interner: DbInterner<'db>) -> Self {
86                interner.default_types().empty.$default_types_field
87            }
88
89            #[inline]
90            pub fn new_from_slice(slice: &[$ty_db]) -> Self {
91                let slice = unsafe { ::std::mem::transmute::<&[$ty_db], &[$ty_static]>(slice) };
92                Self { interned: ::intern::InternedSlice::from_header_and_slice((), slice) }
93            }
94
95            #[inline]
96            pub fn new_from_iter<I, T>(_interner: DbInterner<'db>, args: I) -> T::Output
97            where
98                I: IntoIterator<Item = T>,
99                T: ::rustc_type_ir::CollectAndApply<$ty_db, Self>,
100            {
101                ::rustc_type_ir::CollectAndApply::collect_and_apply(args.into_iter(), |g| {
102                    Self::new_from_slice(g)
103                })
104            }
105
106            #[inline]
107            pub fn as_slice(self) -> &'db [$ty_db] {
108                let slice = &self.interned.get().slice;
109                unsafe { ::std::mem::transmute::<&[$ty_static], &[$ty_db]>(slice) }
110            }
111
112            #[inline]
113            pub fn iter(self) -> ::std::iter::Copied<::std::slice::Iter<'db, $ty_db>> {
114                self.as_slice().iter().copied()
115            }
116
117            #[inline]
118            pub fn len(self) -> usize {
119                self.as_slice().len()
120            }
121
122            #[inline]
123            pub fn is_empty(self) -> bool {
124                self.as_slice().is_empty()
125            }
126        }
127
128        impl<'db> IntoIterator for $name<'db> {
129            type IntoIter = ::std::iter::Copied<::std::slice::Iter<'db, $ty_db>>;
130            type Item = $ty_db;
131            #[inline]
132            fn into_iter(self) -> Self::IntoIter { self.iter() }
133        }
134
135        impl<'db> ::std::ops::Deref for $name<'db> {
136            type Target = [$ty_db];
137
138            #[inline]
139            fn deref(&self) -> &Self::Target {
140                (*self).as_slice()
141            }
142        }
143
144        impl<'db> rustc_type_ir::inherent::SliceLike for $name<'db> {
145            type Item = $ty_db;
146
147            type IntoIter = ::std::iter::Copied<::std::slice::Iter<'db, $ty_db>>;
148
149            #[inline]
150            fn iter(self) -> Self::IntoIter {
151                self.iter()
152            }
153
154            #[inline]
155            fn as_slice(&self) -> &[Self::Item] {
156                (*self).as_slice()
157            }
158        }
159
160        impl<'db> Default for $name<'db> {
161            #[inline]
162            fn default() -> Self {
163                $name::empty(DbInterner::conjure())
164            }
165        }

167
168        impl<'db, V: $crate::next_solver::interner::WorldExposer>
169            rustc_type_ir::GenericTypeVisitable<V> for $name<'db>
170        {
171            #[inline]
172            fn generic_visit_with(&self, visitor: &mut V) {
173                if visitor.on_interned_slice(self.interned).is_continue() {
174                    self.as_slice().iter().for_each(|it| it.generic_visit_with(visitor));
175                }
176            }
177        }

347            // As an approximation, when we call `new_with` we're trait solving, therefore we need the lang items.

349            lang_items: Some(hir_def::lang_item::lang_items(db, krate)),
350        }
351    }

355        self.db
356    }

361        self.lang_items.expect(
362            "Must have `DbInterner::lang_items`.\n\n\

370        crate::next_solver::default_types(self.db)
371    }

546impl AdtDef {
547    pub fn new<'db>(def_id: AdtId, interner: DbInterner<'db>) -> Self {
548        let db = interner.db();
549        let (flags, variants, repr) = match def_id {
550            AdtId::StructId(struct_id) => {
551                let data = db.struct_signature(struct_id);
552
553                let flags = AdtFlags {
554                    is_enum: false,
555                    is_union: false,
556                    is_struct: true,
557                    is_phantom_data: data.flags.contains(StructFlags::IS_PHANTOM_DATA),
558                    is_fundamental: data.flags.contains(StructFlags::FUNDAMENTAL),
559                    is_box: data.flags.contains(StructFlags::IS_BOX),
560                    is_manually_drop: data.flags.contains(StructFlags::IS_MANUALLY_DROP),
561                };
562
563                let variants = vec![(VariantIdx(0), VariantDef::Struct(struct_id))];
564
565                let data_repr = data.repr(db, struct_id);
566                let mut repr_flags = ReprFlags::empty();
567                if flags.is_box {
568                    repr_flags.insert(ReprFlags::IS_LINEAR);
569                }
570                if data_repr.is_some_and(|r| r.c()) {
571                    repr_flags.insert(ReprFlags::IS_C);
572                }
573                if data_repr.is_some_and(|r| r.simd()) {
574                    repr_flags.insert(ReprFlags::IS_SIMD);
575                }
576                let repr = ReprOptions {
577                    align: data_repr.and_then(|r| r.align),
578                    pack: data_repr.and_then(|r| r.pack),
579                    int: data_repr.and_then(|r| r.int),
580                    flags: repr_flags,
581                    ..ReprOptions::default()
582                };
583
584                (flags, variants, repr)
585            }
586            AdtId::UnionId(union_id) => {
587                let flags = AdtFlags {
588                    is_enum: false,
589                    is_union: true,
590                    is_struct: false,
591                    is_phantom_data: false,
592                    is_fundamental: false,
593                    is_box: false,
594                    is_manually_drop: false,
595                };
596
597                let variants = vec![(VariantIdx(0), VariantDef::Union(union_id))];
598
599                let data_repr = AttrFlags::repr(db, union_id.into());
600                let mut repr_flags = ReprFlags::empty();
601                if flags.is_box {
602                    repr_flags.insert(ReprFlags::IS_LINEAR);
603                }
604                if data_repr.is_some_and(|r| r.c()) {
605                    repr_flags.insert(ReprFlags::IS_C);
606                }
607                if data_repr.is_some_and(|r| r.simd()) {
608                    repr_flags.insert(ReprFlags::IS_SIMD);
609                }
610                let repr = ReprOptions {
611                    align: data_repr.and_then(|r| r.align),
612                    pack: data_repr.and_then(|r| r.pack),
613                    int: data_repr.and_then(|r| r.int),
614                    flags: repr_flags,
615                    ..ReprOptions::default()
616                };
617
618                (flags, variants, repr)
619            }
620            AdtId::EnumId(enum_id) => {
621                let flags = AdtFlags {
622                    is_enum: true,
623                    is_union: false,
624                    is_struct: false,
625                    is_phantom_data: false,
626                    is_fundamental: false,
627                    is_box: false,
628                    is_manually_drop: false,
629                };
630
631                let variants = enum_id
632                    .enum_variants(db)
633                    .variants
634                    .iter()
635                    .enumerate()
636                    .map(|(idx, v)| (VariantIdx(idx), v))
637                    .map(|(idx, v)| (idx, VariantDef::Enum(v.0)))
638                    .collect();
639
640                let data_repr = AttrFlags::repr(db, enum_id.into());
641
642                let mut repr_flags = ReprFlags::empty();
643                if flags.is_box {
644                    repr_flags.insert(ReprFlags::IS_LINEAR);
645                }
646                if data_repr.is_some_and(|r| r.c()) {
647                    repr_flags.insert(ReprFlags::IS_C);
648                }
649                if data_repr.is_some_and(|r| r.simd()) {
650                    repr_flags.insert(ReprFlags::IS_SIMD);
651                }
652
653                let repr = ReprOptions {
654                    align: data_repr.and_then(|r| r.align),
655                    pack: data_repr.and_then(|r| r.pack),
656                    int: data_repr.and_then(|r| r.int),
657                    flags: repr_flags,
658                    ..ReprOptions::default()
659                };
660
661                (flags, variants, repr)
662            }
663        };
664
665        AdtDef::new_(db, AdtDefInner { id: def_id, variants, flags, repr })
666    }

932impl<'db> rustc_type_ir::relate::Relate<DbInterner<'db>> for Pattern<'db> {
933    fn relate<R: rustc_type_ir::relate::TypeRelation<DbInterner<'db>>>(
934        relation: &mut R,
935        a: Self,
936        b: Self,
937    ) -> rustc_type_ir::relate::RelateResult<DbInterner<'db>, Self> {
938        let tcx = relation.cx();
939        match (a.kind(), b.kind()) {
940            (
941                PatternKind::Range { start: start_a, end: end_a },
942                PatternKind::Range { start: start_b, end: end_b },
943            ) => {
944                let start = relation.relate(start_a, start_b)?;
945                let end = relation.relate(end_a, end_b)?;
946                Ok(Pattern::new(tcx, PatternKind::Range { start, end }))
947            }
948            (PatternKind::Or(a), PatternKind::Or(b)) => {
949                if a.len() != b.len() {
950                    return Err(TypeError::Mismatch);
951                }
952                let pats = PatList::new_from_iter(
953                    relation.cx(),
954                    std::iter::zip(a.iter(), b.iter()).map(|(a, b)| relation.relate(a, b)),
955                )?;
956                Ok(Pattern::new(tcx, PatternKind::Or(pats)))
957            }
958            (PatternKind::NotNull, PatternKind::NotNull) => Ok(a),
959            (PatternKind::Range { .. } | PatternKind::Or(_) | PatternKind::NotNull, _) => {
960                Err(TypeError::Mismatch)
961            }
962        }
963    }

1304        use hir_def::Lookup;
1305
1306        let container = match def_id {
1307            SolverDefId::FunctionId(it) => it.lookup(self.db()).container,
1308            SolverDefId::TypeAliasId(it) => it.lookup(self.db()).container,
1309            SolverDefId::ConstId(it) => it.lookup(self.db()).container,
1310            SolverDefId::InternedClosureId(it) => {
1311                return self
1312                    .db()
1313                    .lookup_intern_closure(it)
1314                    .0
1315                    .as_generic_def_id(self.db())
1316                    .unwrap()
1317                    .into();

1323                    .0
1324                    .as_generic_def_id(self.db())
1325                    .unwrap()
1326                    .into();

1329            | SolverDefId::AdtId(_)
1330            | SolverDefId::TraitId(_)

1384    fn generics_require_sized_self(self, def_id: Self::DefId) -> bool {
1385        let sized_trait = self.lang_items().Sized;
1386        let Some(sized_id) = sized_trait else {
1387            return false; /* No Sized trait, can't require it! */
1388        };
1389        let sized_def_id = sized_id.into();
1390
1391        // Search for a predicate like `Self : Sized` amongst the trait bounds.
1392        let predicates = self.predicates_of(def_id);
1393        elaborate(self, predicates.iter_identity()).any(|pred| match pred.kind().skip_binder() {
1394            ClauseKind::Trait(ref trait_pred) => {
1395                trait_pred.def_id() == sized_def_id
1396                    && matches!(
1397                        trait_pred.self_ty().kind(),
1398                        TyKind::Param(ParamTy { index: 0, .. })
1399                    )
1400            }
1401            ClauseKind::RegionOutlives(_)
1402            | ClauseKind::TypeOutlives(_)
1403            | ClauseKind::Projection(_)
1404            | ClauseKind::ConstArgHasType(_, _)
1405            | ClauseKind::WellFormed(_)
1406            | ClauseKind::ConstEvaluatable(_)
1407            | ClauseKind::HostEffect(..)
1408            | ClauseKind::UnstableFeature(_) => false,
1409        })

1773    fn for_each_relevant_impl(

1779        let krate = self.krate.expect("trait solving requires setting `DbInterner::krate`");
1780        let trait_block = trait_def_id.0.loc(self.db).container.block(self.db);
1781        let mut consider_impls_for_simplified_type = |simp: SimplifiedType| {
1782            let type_block = simp.def().and_then(|def_id| {
1783                let module = match def_id {
1784                    SolverDefId::AdtId(AdtId::StructId(id)) => id.module(self.db),
1785                    SolverDefId::AdtId(AdtId::EnumId(id)) => id.module(self.db),
1786                    SolverDefId::AdtId(AdtId::UnionId(id)) => id.module(self.db),

1788                    SolverDefId::TypeAliasId(id) => id.module(self.db),
1789                    SolverDefId::ConstId(_)
1790                    | SolverDefId::FunctionId(_)
1791                    | SolverDefId::ImplId(_)
1792                    | SolverDefId::BuiltinDeriveImplId(_)
1793                    | SolverDefId::StaticId(_)
1794                    | SolverDefId::InternedClosureId(_)
1795                    | SolverDefId::InternedCoroutineId(_)
1796                    | SolverDefId::InternedOpaqueTyId(_)
1797                    | SolverDefId::EnumVariantId(_)
1798                    | SolverDefId::Ctor(_) => return None,
1799                };
1800                module.block(self.db)
1801            });
1802            TraitImpls::for_each_crate_and_block_trait_and_type(
1803                self.db,
1804                krate,
1805                type_block,
1806                trait_block,
1807                &mut |impls| {
1808                    let (regular_impls, builtin_derive_impls) =
1809                        impls.for_trait_and_self_ty(trait_def_id.0, &simp);
1810                    for &impl_ in regular_impls {
1811                        f(impl_.into());
1812                    }
1813                    for &impl_ in builtin_derive_impls {

1815                    }

1822            | TyKind::Char
1823            | TyKind::Int(_)
1824            | TyKind::Uint(_)
1825            | TyKind::Float(_)
1826            | TyKind::Adt(_, _)
1827            | TyKind::Foreign(_)
1828            | TyKind::Str
1829            | TyKind::Array(_, _)
1830            | TyKind::Pat(_, _)
1831            | TyKind::Slice(_)
1832            | TyKind::RawPtr(_, _)
1833            | TyKind::Ref(_, _, _)
1834            | TyKind::FnDef(_, _)

1836            | TyKind::Dynamic(_, _)
1837            | TyKind::Closure(..)
1838            | TyKind::CoroutineClosure(..)
1839            | TyKind::Coroutine(_, _)
1840            | TyKind::Never
1841            | TyKind::Tuple(_)
1842            | TyKind::UnsafeBinder(_) => {
1843                let simp =
1844                    fast_reject::simplify_type(self, self_ty, fast_reject::TreatParams::AsRigid)
1845                        .unwrap();
1846                consider_impls_for_simplified_type(simp);
1847            }
1848
1849            // HACK: For integer and float variables we have to manually look at all impls
1850            // which have some integer or float as a self type.
1851            TyKind::Infer(InferTy::IntVar(_)) => {
1852                use IntTy::*;
1853                use UintTy::*;
1854                // This causes a compiler error if any new integer kinds are added.
1855                let (I8 | I16 | I32 | I64 | I128 | Isize): IntTy;
1856                let (U8 | U16 | U32 | U64 | U128 | Usize): UintTy;
1857                let possible_integers = [

1860                    SimplifiedType::Int(I16),
1861                    SimplifiedType::Int(I32),
1862                    SimplifiedType::Int(I64),
1863                    SimplifiedType::Int(I128),
1864                    SimplifiedType::Int(Isize),

1868                    SimplifiedType::Uint(U32),
1869                    SimplifiedType::Uint(U64),
1870                    SimplifiedType::Uint(U128),
1871                    SimplifiedType::Uint(Usize),
1872                ];
1873                for simp in possible_integers {
1874                    consider_impls_for_simplified_type(simp);
1875                }
1876            }
1877
1878            TyKind::Infer(InferTy::FloatVar(_)) => {
1879                // This causes a compiler error if any new float kinds are added.
1880                let (FloatTy::F16 | FloatTy::F32 | FloatTy::F64 | FloatTy::F128);
1881                let possible_floats = [
1882                    SimplifiedType::Float(FloatTy::F16),
1883                    SimplifiedType::Float(FloatTy::F32),
1884                    SimplifiedType::Float(FloatTy::F64),
1885                    SimplifiedType::Float(FloatTy::F128),
1886                ];
1887
1888                for simp in possible_floats {
1889                    consider_impls_for_simplified_type(simp);
1890                }
1891            }

1893            // The only traits applying to aliases and placeholders are blanket impls.
1894            //
1895            // Impls which apply to an alias after normalization are handled by
1896            // `assemble_candidates_after_normalizing_self_ty`.

1907                | InferTy::FreshTy(_)
1908                | InferTy::FreshIntTy(_)
1909                | InferTy::FreshFloatTy(_),

1911            | TyKind::Param(_)
1912            | TyKind::Bound(_, _) => panic!("unexpected self type: {self_ty:?}"),

2041    fn unsizing_params_for_adt(self, id: Self::AdtId) -> Self::UnsizingParams {
2042        let def = AdtDef::new(id.0, self);
2043        let num_params = self.generics_of(id.into()).count();
2044
2045        let maybe_unsizing_param_idx = |arg: GenericArg<'db>| match arg.kind() {
2046            GenericArgKind::Type(ty) => match ty.kind() {
2047                rustc_type_ir::TyKind::Param(p) => Some(p.index),
2048                _ => None,
2049            },
2050            GenericArgKind::Lifetime(_) => None,
2051            GenericArgKind::Const(ct) => match ct.kind() {
2052                rustc_type_ir::ConstKind::Param(p) => Some(p.index),
2053                _ => None,
2054            },
2055        };
2056
2057        // The last field of the structure has to exist and contain type/const parameters.
2058        let variant = def.non_enum_variant();
2059        let fields = variant.fields(self.db());
2060        let Some((tail_field, prefix_fields)) = fields.split_last() else {
2061            return UnsizingParams(DenseBitSet::new_empty(num_params));
2062        };
2063
2064        let field_types = self.db().field_types(variant.id());
2065        let mut unsizing_params = DenseBitSet::new_empty(num_params);
2066        let ty = field_types[tail_field.0].get();
2067        for arg in ty.instantiate_identity().walk() {
2068            if let Some(i) = maybe_unsizing_param_idx(arg) {
2069                unsizing_params.insert(i);
2070            }
2071        }
2072
2073        // Ensure none of the other fields mention the parameters used
2074        // in unsizing.
2075        for field in prefix_fields {
2076            for arg in field_types[field.0].get().instantiate_identity().walk() {
2077                if let Some(i) = maybe_unsizing_param_idx(arg) {
2078                    unsizing_params.remove(i);
2079                }
2080            }
2081        }
2082
2083        UnsizingParams(unsizing_params)
2084    }

2447mod tls_db {
2448    use std::{cell::Cell, ptr::NonNull};
2449
2450    use crate::db::HirDatabase;
2451
2452    struct Attached {
2453        database: Cell<Option<NonNull<dyn HirDatabase>>>,
2454    }
2455
2456    impl Attached {
2457        #[inline]
2458        fn attach<R>(&self, db: &dyn HirDatabase, op: impl FnOnce() -> R) -> R {
2459            struct DbGuard<'s> {
2460                state: Option<&'s Attached>,
2461            }
2462
2463            impl<'s> DbGuard<'s> {
2464                #[inline]
2465                fn new(attached: &'s Attached, db: &dyn HirDatabase) -> Self {
2466                    match attached.database.get() {
2467                        Some(current_db) => {
2468                            let new_db = NonNull::from(db);
2469                            if !std::ptr::addr_eq(current_db.as_ptr(), new_db.as_ptr()) {
2470                                panic!(
2471                                    "Cannot change attached database. This is likely a bug.\n\
2472                                    If this is not a bug, you can use `attach_db_allow_change()`."
2473                                );
2474                            }
2475                            Self { state: None }
2476                        }
2477                        None => {
2478                            // Otherwise, set the database.
2479                            attached.database.set(Some(NonNull::from(db)));
2480                            Self { state: Some(attached) }
2481                        }
2482                    }
2483                }
2484            }
2485
2486            impl Drop for DbGuard<'_> {
2487                #[inline]
2488                fn drop(&mut self) {
2489                    // Reset database to null if we did anything in `DbGuard::new`.
2490                    if let Some(attached) = self.state {
2491                        attached.database.set(None);
2492                    }
2493                }
2494            }
2495
2496            let _guard = DbGuard::new(self, db);
2497            op()
2498        }
2499
2500        #[inline]
2501        fn attach_allow_change<R>(&self, db: &dyn HirDatabase, op: impl FnOnce() -> R) -> R {
2502            struct DbGuard<'s> {
2503                state: &'s Attached,
2504                prev: Option<NonNull<dyn HirDatabase>>,
2505            }
2506
2507            impl<'s> DbGuard<'s> {
2508                #[inline]
2509                fn new(attached: &'s Attached, db: &dyn HirDatabase) -> Self {
2510                    let prev = attached.database.replace(Some(NonNull::from(db)));
2511                    Self { state: attached, prev }
2512                }
2513            }
2514
2515            impl Drop for DbGuard<'_> {
2516                #[inline]
2517                fn drop(&mut self) {
2518                    self.state.database.set(self.prev);
2519                }
2520            }
2521
2522            let _guard = DbGuard::new(self, db);
2523            op()
2524        }
2525
2526        #[inline]
2527        fn with<R>(&self, op: impl FnOnce(&dyn HirDatabase) -> R) -> R {
2528            let db = self.database.get().expect("Try to use attached db, but not db is attached");
2529
2530            // SAFETY: The db is attached, so it must be valid.
2531            op(unsafe { db.as_ref() })
2532        }
2533    }
2534
2535    thread_local! {
2536        static GLOBAL_DB: Attached = const { Attached { database: Cell::new(None) } };
2537    }
2538
2539    #[inline]
2540    pub fn attach_db<R>(db: &dyn HirDatabase, op: impl FnOnce() -> R) -> R {
2541        GLOBAL_DB.with(|global_db| global_db.attach(db, op))
2542    }
2543
2544    #[inline]
2545    pub fn attach_db_allow_change<R>(db: &dyn HirDatabase, op: impl FnOnce() -> R) -> R {
2546        GLOBAL_DB.with(|global_db| global_db.attach_allow_change(db, op))
2547    }
2548
2549    #[inline]
2550    pub fn with_attached_db<R>(op: impl FnOnce(&dyn HirDatabase) -> R) -> R {
2551        GLOBAL_DB.with(
2552            #[inline]
2553            |a| a.with(op),
2554        )
2555    }
