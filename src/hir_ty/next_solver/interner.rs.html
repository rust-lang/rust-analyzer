456impl AdtDef {
457    pub fn new<'db>(def_id: AdtId, interner: DbInterner<'db>) -> Self {
458        let db = interner.db();
459        let (flags, variants, repr) = match def_id {
460            AdtId::StructId(struct_id) => {
461                let data = db.struct_signature(struct_id);
462
463                let flags = AdtFlags {
464                    is_enum: false,
465                    is_union: false,
466                    is_struct: true,
467                    is_phantom_data: data.flags.contains(StructFlags::IS_PHANTOM_DATA),
468                    is_fundamental: data.flags.contains(StructFlags::FUNDAMENTAL),
469                    is_box: data.flags.contains(StructFlags::IS_BOX),
470                    is_manually_drop: data.flags.contains(StructFlags::IS_MANUALLY_DROP),
471                };
472
473                let variants = vec![(VariantIdx(0), VariantDef::Struct(struct_id))];
474
475                let mut repr = ReprOptions::default();
476                repr.align = data.repr.and_then(|r| r.align);
477                repr.pack = data.repr.and_then(|r| r.pack);
478                repr.int = data.repr.and_then(|r| r.int);
479
480                let mut repr_flags = ReprFlags::empty();
481                if flags.is_box {
482                    repr_flags.insert(ReprFlags::IS_LINEAR);
483                }
484                if data.repr.is_some_and(|r| r.c()) {
485                    repr_flags.insert(ReprFlags::IS_C);
486                }
487                if data.repr.is_some_and(|r| r.simd()) {
488                    repr_flags.insert(ReprFlags::IS_SIMD);
489                }
490                repr.flags = repr_flags;
491
492                (flags, variants, repr)
493            }
494            AdtId::UnionId(union_id) => {
495                let data = db.union_signature(union_id);
496
497                let flags = AdtFlags {
498                    is_enum: false,
499                    is_union: true,
500                    is_struct: false,
501                    is_phantom_data: false,
502                    is_fundamental: false,
503                    is_box: false,
504                    is_manually_drop: false,
505                };
506
507                let variants = vec![(VariantIdx(0), VariantDef::Union(union_id))];
508
509                let mut repr = ReprOptions::default();
510                repr.align = data.repr.and_then(|r| r.align);
511                repr.pack = data.repr.and_then(|r| r.pack);
512                repr.int = data.repr.and_then(|r| r.int);
513
514                let mut repr_flags = ReprFlags::empty();
515                if flags.is_box {
516                    repr_flags.insert(ReprFlags::IS_LINEAR);
517                }
518                if data.repr.is_some_and(|r| r.c()) {
519                    repr_flags.insert(ReprFlags::IS_C);
520                }
521                if data.repr.is_some_and(|r| r.simd()) {
522                    repr_flags.insert(ReprFlags::IS_SIMD);
523                }
524                repr.flags = repr_flags;
525
526                (flags, variants, repr)
527            }
528            AdtId::EnumId(enum_id) => {
529                let flags = AdtFlags {
530                    is_enum: true,
531                    is_union: false,
532                    is_struct: false,
533                    is_phantom_data: false,
534                    is_fundamental: false,
535                    is_box: false,
536                    is_manually_drop: false,
537                };
538
539                let variants = enum_id
540                    .enum_variants(db)
541                    .variants
542                    .iter()
543                    .enumerate()
544                    .map(|(idx, v)| (VariantIdx(idx), v))
545                    .map(|(idx, v)| (idx, VariantDef::Enum(v.0)))
546                    .collect();
547
548                let data = db.enum_signature(enum_id);
549
550                let mut repr = ReprOptions::default();
551                repr.align = data.repr.and_then(|r| r.align);
552                repr.pack = data.repr.and_then(|r| r.pack);
553                repr.int = data.repr.and_then(|r| r.int);
554
555                let mut repr_flags = ReprFlags::empty();
556                if flags.is_box {
557                    repr_flags.insert(ReprFlags::IS_LINEAR);
558                }
559                if data.repr.is_some_and(|r| r.c()) {
560                    repr_flags.insert(ReprFlags::IS_C);
561                }
562                if data.repr.is_some_and(|r| r.simd()) {
563                    repr_flags.insert(ReprFlags::IS_SIMD);
564                }
565                repr.flags = repr_flags;
566
567                (flags, variants, repr)
568            }
569        };
570
571        AdtDef::new_(db, AdtDefInner { id: def_id, variants, flags, repr })
572    }

801impl<'db> rustc_type_ir::relate::Relate<DbInterner<'db>> for Pattern<'db> {
802    fn relate<R: rustc_type_ir::relate::TypeRelation<DbInterner<'db>>>(
803        relation: &mut R,
804        a: Self,
805        b: Self,
806    ) -> rustc_type_ir::relate::RelateResult<DbInterner<'db>, Self> {
807        let tcx = relation.cx();
808        match (a.kind(), b.kind()) {
809            (
810                PatternKind::Range { start: start_a, end: end_a },
811                PatternKind::Range { start: start_b, end: end_b },
812            ) => {
813                let start = relation.relate(start_a, start_b)?;
814                let end = relation.relate(end_a, end_b)?;
815                Ok(Pattern::new(tcx, PatternKind::Range { start, end }))
816            }
817            (PatternKind::Or(a), PatternKind::Or(b)) => {
818                if a.len() != b.len() {
819                    return Err(TypeError::Mismatch);
820                }
821                let pats = CollectAndApply::collect_and_apply(
822                    std::iter::zip(a.iter(), b.iter()).map(|(a, b)| relation.relate(a, b)),
823                    |g| PatList::new_from_iter(tcx, g.iter().cloned()),
824                )?;
825                Ok(Pattern::new(tcx, PatternKind::Or(pats)))
826            }
827            (PatternKind::Range { .. } | PatternKind::Or(_), _) => Err(TypeError::Mismatch),
828        }
829    }

1238    fn generics_require_sized_self(self, def_id: Self::DefId) -> bool {
1239        let sized_trait =
1240            LangItem::Sized.resolve_trait(self.db(), self.krate.expect("Must have self.krate"));
1241        let Some(sized_id) = sized_trait else {
1242            return false; /* No Sized trait, can't require it! */
1243        };
1244        let sized_def_id = sized_id.into();
1245
1246        // Search for a predicate like `Self : Sized` amongst the trait bounds.
1247        let predicates = self.predicates_of(def_id);
1248        elaborate(self, predicates.iter_identity()).any(|pred| match pred.kind().skip_binder() {
1249            ClauseKind::Trait(ref trait_pred) => {
1250                trait_pred.def_id() == sized_def_id
1251                    && matches!(
1252                        trait_pred.self_ty().kind(),
1253                        TyKind::Param(ParamTy { index: 0, .. })
1254                    )
1255            }
1256            ClauseKind::RegionOutlives(_)
1257            | ClauseKind::TypeOutlives(_)
1258            | ClauseKind::Projection(_)
1259            | ClauseKind::ConstArgHasType(_, _)
1260            | ClauseKind::WellFormed(_)
1261            | ClauseKind::ConstEvaluatable(_)
1262            | ClauseKind::HostEffect(..)
1263            | ClauseKind::UnstableFeature(_) => false,
1264        })
1265    }

1327    #[tracing::instrument(skip(self), ret)]
1328    fn explicit_super_predicates_of(
1329        self,
1330        def_id: Self::TraitId,
1331    ) -> EarlyBinder<Self, impl IntoIterator<Item = (Self::Clause, Self::Span)>> {
1332        let is_self = |ty: Ty<'db>| match ty.kind() {
1333            rustc_type_ir::TyKind::Param(param) => param.index == 0,
1334            _ => false,
1335        };

1337        let predicates: Vec<(Clause<'db>, Span)> = self
1338            .db()
1339            .generic_predicates(def_id.0.into())
1340            .iter()
1341            .filter(|p| match p.kind().skip_binder() {
1342                // rustc has the following assertion:
1343                // https://github.com/rust-lang/rust/blob/52618eb338609df44978b0ca4451ab7941fd1c7a/compiler/rustc_hir_analysis/src/hir_ty_lowering/bounds.rs#L525-L608
1344                rustc_type_ir::ClauseKind::Trait(it) => is_self(it.self_ty()),
1345                rustc_type_ir::ClauseKind::TypeOutlives(it) => is_self(it.0),
1346                rustc_type_ir::ClauseKind::Projection(it) => is_self(it.self_ty()),
1347                rustc_type_ir::ClauseKind::HostEffect(it) => is_self(it.self_ty()),
1348                _ => false,
1349            })
1350            .cloned()
1351            .map(|p| (p, Span::dummy()))
1352            .collect();
1353        EarlyBinder::bind(predicates)
1354    }

1610    fn for_each_relevant_impl(

1613        self_ty: Self::Ty,

1618        let fps: &[TyFingerprint] = match self_ty.kind() {
1619            TyKind::Infer(InferTy::IntVar(..)) => &ALL_INT_FPS,
1620            TyKind::Infer(InferTy::FloatVar(..)) => &ALL_FLOAT_FPS,
1621            _ => self_ty_fp.as_slice(),

1625            _ = for_trait_impls(
1626                self.db(),

1628                self.block,

1630                self_ty_fp,

1635                            b.skip_binder().visit_with(&mut ContainsTypeErrors).is_break()
1636                        });
1637                        if contains_errors {
1638                            continue;
1639                        }
1640
1641                        f(i.into());
1642                    }
1643                    ControlFlow::Continue(())
1644                },

1648                self.db(),

1786    fn unsizing_params_for_adt(self, id: Self::AdtId) -> Self::UnsizingParams {
1787        let def = AdtDef::new(id.0, self);
1788        let num_params = self.generics_of(id.into()).count();
1789
1790        let maybe_unsizing_param_idx = |arg: GenericArg<'db>| match arg.kind() {
1791            GenericArgKind::Type(ty) => match ty.kind() {
1792                rustc_type_ir::TyKind::Param(p) => Some(p.index),
1793                _ => None,
1794            },
1795            GenericArgKind::Lifetime(_) => None,
1796            GenericArgKind::Const(ct) => match ct.kind() {
1797                rustc_type_ir::ConstKind::Param(p) => Some(p.index),
1798                _ => None,
1799            },
1800        };
1801
1802        // The last field of the structure has to exist and contain type/const parameters.
1803        let variant = def.non_enum_variant();
1804        let fields = variant.fields(self.db());
1805        let Some((tail_field, prefix_fields)) = fields.split_last() else {
1806            return UnsizingParams(DenseBitSet::new_empty(num_params));
1807        };
1808
1809        let field_types = self.db().field_types(variant.id());
1810        let mut unsizing_params = DenseBitSet::new_empty(num_params);
1811        let ty = field_types[tail_field.0];
1812        for arg in ty.instantiate_identity().walk() {
1813            if let Some(i) = maybe_unsizing_param_idx(arg) {
1814                unsizing_params.insert(i);
1815            }
1816        }
1817
1818        // Ensure none of the other fields mention the parameters used
1819        // in unsizing.
1820        for field in prefix_fields {
1821            for arg in field_types[field.0].instantiate_identity().walk() {
1822                if let Some(i) = maybe_unsizing_param_idx(arg) {
1823                    unsizing_params.remove(i);
1824                }
1825            }
1826        }
1827
1828        UnsizingParams(unsizing_params)
1829    }

1832        self,
