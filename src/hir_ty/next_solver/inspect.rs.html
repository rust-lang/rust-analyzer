67/// The expected term of a `NormalizesTo` goal gets replaced
68/// with an unconstrained inference variable when computing
69/// `NormalizesTo` goals and we return the nested goals to the
70/// caller, who also equates the actual term with the expected.
71///
72/// This is an implementation detail of the trait solver and
73/// not something we want to leak to users. We therefore
74/// treat `NormalizesTo` goals as if they apply the expected
75/// type at the end of each candidate.
76#[derive(Debug, Copy, Clone)]
77pub(crate) struct NormalizesToTermHack<'db> {
78    term: Term<'db>,
79    unconstrained_term: Term<'db>,
80}
81
82impl<'db> NormalizesToTermHack<'db> {
83    /// Relate the `term` with the new `unconstrained_term` created
84    /// when computing the proof tree for this `NormalizesTo` goals.
85    /// This handles nested obligations.
86    fn constrain_and(
87        &self,
88        infcx: &InferCtxt<'db>,
89        param_env: ParamEnv<'db>,
90        f: impl FnOnce(&mut ObligationCtxt<'_, 'db>),
91    ) -> Result<Certainty, NoSolution> {
92        let mut ocx = ObligationCtxt::new(infcx);
93        ocx.eq(&ObligationCause::dummy(), param_env, self.term, self.unconstrained_term)?;
94        f(&mut ocx);
95        let errors = ocx.evaluate_obligations_error_on_ambiguity();
96        if errors.is_empty() {
97            Ok(Certainty::Yes)
98        } else if errors.iter().all(|e| !matches!(e, NextSolverError::TrueError(_))) {
99            Ok(Certainty::AMBIGUOUS)
100        } else {
101            Err(NoSolution)
102        }
103    }
104}

140    /// Visit all nested goals of this candidate without rolling
141    /// back their inference constraints. This function modifies
142    /// the state of the `infcx`.
143    pub(crate) fn visit_nested_no_probe<V: ProofTreeVisitor<'db>>(
144        &self,
145        visitor: &mut V,
146    ) -> V::Result {
147        for goal in self.instantiate_nested_goals() {
148            try_visit!(goal.visit_with(visitor));
149        }
150
151        V::Result::output()
152    }

154    /// Instantiate the nested goals for the candidate without rolling back their
155    /// inference constraints. This function modifies the state of the `infcx`.
156    ///
157    /// See [`Self::instantiate_impl_args`] if you need the impl args too.
158    pub(crate) fn instantiate_nested_goals(&self) -> Vec<InspectGoal<'a, 'db>> {
159        let infcx = self.goal.infcx;
160        let param_env = self.goal.goal.param_env;
161        let mut orig_values = self.goal.orig_values.to_vec();
162
163        let mut instantiated_goals = vec![];
164        for step in &self.steps {
165            match **step {
166                inspect::ProbeStep::AddGoal(source, goal) => instantiated_goals.push((
167                    source,
168                    instantiate_canonical_state(
169                        infcx,
170                        Span::dummy(),
171                        param_env,
172                        &mut orig_values,
173                        goal,
174                    ),
175                )),
176                inspect::ProbeStep::RecordImplArgs { .. } => {}
177                inspect::ProbeStep::MakeCanonicalResponse { .. }
178                | inspect::ProbeStep::NestedProbe(_) => unreachable!(),
179            }
180        }
181
182        let () = instantiate_canonical_state(
183            infcx,
184            Span::dummy(),
185            param_env,
186            &mut orig_values,
187            self.final_state,
188        );
189
190        if let Some(term_hack) = &self.goal.normalizes_to_term_hack {
191            // FIXME: We ignore the expected term of `NormalizesTo` goals
192            // when computing the result of its candidates. This is
193            // scuffed.
194            let _ = term_hack.constrain_and(infcx, param_env, |_| {});
195        }
196
197        instantiated_goals
198            .into_iter()
199            .map(|(source, goal)| self.instantiate_proof_tree_for_nested_goal(source, goal))
200            .collect()
201    }

203    /// Instantiate the args of an impl if this candidate came from a
204    /// `CandidateSource::Impl`. This function modifies the state of the
205    /// `infcx`.
206    pub(crate) fn instantiate_impl_args(&self) -> GenericArgs<'db> {
207        let infcx = self.goal.infcx;
208        let param_env = self.goal.goal.param_env;
209        let mut orig_values = self.goal.orig_values.to_vec();
210
211        for step in &self.steps {
212            match **step {
213                inspect::ProbeStep::RecordImplArgs { impl_args } => {
214                    let impl_args = instantiate_canonical_state(
215                        infcx,
216                        Span::dummy(),
217                        param_env,
218                        &mut orig_values,
219                        impl_args,
220                    );
221
222                    let () = instantiate_canonical_state(
223                        infcx,
224                        Span::dummy(),
225                        param_env,
226                        &mut orig_values,
227                        self.final_state,
228                    );
229
230                    // No reason we couldn't support this, but we don't need to for select.
231                    assert!(
232                        self.goal.normalizes_to_term_hack.is_none(),
233                        "cannot use `instantiate_impl_args` with a `NormalizesTo` goal"
234                    );
235
236                    return eager_resolve_vars(infcx, impl_args);
237                }
238                inspect::ProbeStep::AddGoal(..) => {}
239                inspect::ProbeStep::MakeCanonicalResponse { .. }
240                | inspect::ProbeStep::NestedProbe(_) => unreachable!(),
241            }
242        }
243
244        panic!("expected impl args probe step for `instantiate_impl_args`");
245    }

247    pub(crate) fn instantiate_proof_tree_for_nested_goal(
248        &self,
249        source: GoalSource,
250        goal: Goal<'db, Predicate<'db>>,
251    ) -> InspectGoal<'a, 'db> {
252        let infcx = self.goal.infcx;
253        match goal.predicate.kind().no_bound_vars() {
254            Some(PredicateKind::NormalizesTo(NormalizesTo { alias, term })) => {
255                let unconstrained_term = infcx.next_term_var_of_kind(term);
256                let goal =
257                    goal.with(infcx.interner, NormalizesTo { alias, term: unconstrained_term });
258                // We have to use a `probe` here as evaluating a `NormalizesTo` can constrain the
259                // expected term. This means that candidates which only fail due to nested goals
260                // and which normalize to a different term then the final result could ICE: when
261                // building their proof tree, the expected term was unconstrained, but when
262                // instantiating the candidate it is already constrained to the result of another
263                // candidate.
264                let normalizes_to_term_hack = NormalizesToTermHack { term, unconstrained_term };
265                let (proof_tree, nested_goals_result) = infcx.probe(|_| {
266                    // Here, if we have any nested goals, then we make sure to apply them
267                    // considering the constrained RHS, and pass the resulting certainty to
268                    // `InspectGoal::new` so that the goal has the right result (and maintains
269                    // the impression that we don't do this normalizes-to infer hack at all).
270                    let (nested, proof_tree) =
271                        infcx.evaluate_root_goal_for_proof_tree(goal, Span::dummy());
272                    let nested_goals_result = nested.and_then(|nested| {
273                        normalizes_to_term_hack.constrain_and(
274                            infcx,
275                            proof_tree.uncanonicalized_goal.param_env,
276                            |ocx| {
277                                ocx.register_obligations(nested.0.into_iter().map(|(_, goal)| {
278                                    Obligation::new(
279                                        infcx.interner,
280                                        ObligationCause::dummy(),
281                                        goal.param_env,
282                                        goal.predicate,
283                                    )
284                                }));
285                            },
286                        )
287                    });
288                    (proof_tree, nested_goals_result)
289                });
290                InspectGoal::new(
291                    infcx,
292                    self.goal.depth + 1,
293                    proof_tree,
294                    Some((normalizes_to_term_hack, nested_goals_result)),
295                    source,
296                )
297            }
298            _ => {
299                // We're using a probe here as evaluating a goal could constrain
300                // inference variables by choosing one candidate. If we then recurse
301                // into another candidate who ends up with different inference
302                // constraints, we get an ICE if we already applied the constraints
303                // from the chosen candidate.
304                let proof_tree =
305                    infcx.probe(|_| infcx.evaluate_root_goal_for_proof_tree(goal, Span::dummy()).1);
306                InspectGoal::new(infcx, self.goal.depth + 1, proof_tree, None, source)
307            }
308        }
309    }

340    fn candidates_recur(
341        &'a self,
342        candidates: &mut Vec<InspectCandidate<'a, 'db>>,
343        steps: &mut Vec<&'a inspect::ProbeStep<DbInterner<'db>>>,
344        probe: &'a inspect::Probe<DbInterner<'db>>,
345    ) {
346        let mut shallow_certainty = None;
347        for step in &probe.steps {
348            match *step {
349                inspect::ProbeStep::AddGoal(..) | inspect::ProbeStep::RecordImplArgs { .. } => {
350                    steps.push(step)
351                }
352                inspect::ProbeStep::MakeCanonicalResponse { shallow_certainty: c } => {
353                    assert!(matches!(
354                        shallow_certainty.replace(c),
355                        None | Some(Certainty::Maybe { cause: MaybeCause::Ambiguity, .. })
356                    ));
357                }
358                inspect::ProbeStep::NestedProbe(ref probe) => {
359                    match probe.kind {
360                        // These never assemble candidates for the goal we're trying to solve.
361                        inspect::ProbeKind::ProjectionCompatibility
362                        | inspect::ProbeKind::ShadowedEnvProbing => continue,
363
364                        inspect::ProbeKind::NormalizedSelfTyAssembly
365                        | inspect::ProbeKind::UnsizeAssembly
366                        | inspect::ProbeKind::Root { .. }
367                        | inspect::ProbeKind::TraitCandidate { .. }
368                        | inspect::ProbeKind::OpaqueTypeStorageLookup { .. }
369                        | inspect::ProbeKind::RigidAlias { .. } => {
370                            // Nested probes have to prove goals added in their parent
371                            // but do not leak them, so we truncate the added goals
372                            // afterwards.
373                            let num_steps = steps.len();
374                            self.candidates_recur(candidates, steps, probe);
375                            steps.truncate(num_steps);
376                        }
377                    }
378                }
379            }
380        }
381
382        match probe.kind {
383            inspect::ProbeKind::ProjectionCompatibility
384            | inspect::ProbeKind::ShadowedEnvProbing => {
385                panic!()
386            }
387
388            inspect::ProbeKind::NormalizedSelfTyAssembly | inspect::ProbeKind::UnsizeAssembly => {}
389
390            // We add a candidate even for the root evaluation if there
391            // is only one way to prove a given goal, e.g. for `WellFormed`.
392            inspect::ProbeKind::Root { result }
393            | inspect::ProbeKind::TraitCandidate { source: _, result }
394            | inspect::ProbeKind::OpaqueTypeStorageLookup { result }
395            | inspect::ProbeKind::RigidAlias { result } => {
396                // We only add a candidate if `shallow_certainty` was set, which means
397                // that we ended up calling `evaluate_added_goals_and_make_canonical_response`.
398                if let Some(shallow_certainty) = shallow_certainty {
399                    candidates.push(InspectCandidate {
400                        goal: self,
401                        kind: probe.kind,
402                        steps: steps.clone(),
403                        final_state: probe.final_state,
404                        shallow_certainty,
405                        result,
406                    });
407                }
408            }
409        }
410    }

412    pub(crate) fn candidates(&'a self) -> Vec<InspectCandidate<'a, 'db>> {
413        let mut candidates = vec![];
414        let mut nested_goals = vec![];
415        self.candidates_recur(&mut candidates, &mut nested_goals, &self.final_revision);
416        candidates
417    }

419    /// Returns the single candidate applicable for the current goal, if it exists.
420    ///
421    /// Returns `None` if there are either no or multiple applicable candidates.
422    pub(crate) fn unique_applicable_candidate(&'a self) -> Option<InspectCandidate<'a, 'db>> {
423        // FIXME(-Znext-solver): This does not handle impl candidates
424        // hidden by env candidates.
425        let mut candidates = self.candidates();
426        candidates.retain(|c| c.result().is_ok());
427        candidates.pop().filter(|_| candidates.is_empty())
428    }

430    pub(crate) fn new(
431        infcx: &'a InferCtxt<'db>,
432        depth: usize,
433        root: inspect::GoalEvaluation<DbInterner<'db>>,
434        term_hack_and_nested_certainty: Option<(
435            NormalizesToTermHack<'db>,
436            Result<Certainty, NoSolution>,
437        )>,
438        source: GoalSource,
439    ) -> Self {
440        let infcx = <&SolverContext<'db>>::from(infcx);
441
442        let inspect::GoalEvaluation { uncanonicalized_goal, orig_values, final_revision, result } =
443            root;
444        // If there's a normalizes-to goal, AND the evaluation result with the result of
445        // constraining the normalizes-to RHS and computing the nested goals.
446        let result = result.and_then(|ok| {
447            let nested_goals_certainty =
448                term_hack_and_nested_certainty.map_or(Ok(Certainty::Yes), |(_, c)| c)?;
449            Ok(ok.value.certainty.and(nested_goals_certainty))
450        });
451
452        InspectGoal {
453            infcx,
454            depth,
455            orig_values,
456            goal: eager_resolve_vars(infcx, uncanonicalized_goal),
457            result,
458            final_revision,
459            normalizes_to_term_hack: term_hack_and_nested_certainty.map(|(n, _)| n),
460            source,
461        }
462    }

464    pub(crate) fn visit_with<V: ProofTreeVisitor<'db>>(&self, visitor: &mut V) -> V::Result {
465        if self.depth < visitor.config().max_depth {
466            try_visit!(visitor.visit_goal(self));
467        }
468
469        V::Result::output()
470    }

473/// The public API to interact with proof trees.
474pub(crate) trait ProofTreeVisitor<'db> {
475    type Result: VisitorResult;
476
477    fn config(&self) -> InspectConfig {
478        InspectConfig { max_depth: 10 }
479    }
480
481    fn visit_goal(&mut self, goal: &InspectGoal<'_, 'db>) -> Self::Result;
482}

484impl<'db> InferCtxt<'db> {
485    pub(crate) fn visit_proof_tree<V: ProofTreeVisitor<'db>>(
486        &self,
487        goal: Goal<'db, Predicate<'db>>,
488        visitor: &mut V,
489    ) -> V::Result {
490        self.visit_proof_tree_at_depth(goal, 0, visitor)
491    }
492
493    pub(crate) fn visit_proof_tree_at_depth<V: ProofTreeVisitor<'db>>(
494        &self,
495        goal: Goal<'db, Predicate<'db>>,
496        depth: usize,
497        visitor: &mut V,
498    ) -> V::Result {
499        let (_, proof_tree) = <&SolverContext<'db>>::from(self)
500            .evaluate_root_goal_for_proof_tree(goal, Span::dummy());
501        visitor.visit_goal(&InspectGoal::new(self, depth, proof_tree, None, GoalSource::Misc))
502    }
