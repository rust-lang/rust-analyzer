41pub struct Discr<'db> {
42    /// Bit representation of the discriminant (e.g., `-128i8` is `0xFF_u128`).
43    pub val: u128,
44    pub ty: Ty<'db>,
45}
46
47impl<'db> Discr<'db> {
48    /// Adds `1` to the value and wraps around if the maximum for the type is reached.
49    pub fn wrap_incr(self, interner: DbInterner<'db>) -> Self {
50        self.checked_add(interner, 1).0
51    }
52    pub fn checked_add(self, interner: DbInterner<'db>, n: u128) -> (Self, bool) {
53        let (size, signed) = self.ty.int_size_and_signed(interner);
54        let (val, oflo) = if signed {
55            let min = size.signed_int_min();
56            let max = size.signed_int_max();
57            let val = size.sign_extend(self.val);
58            assert!(n < (i128::MAX as u128));
59            let n = n as i128;
60            let oflo = val > max - n;
61            let val = if oflo { min + (n - (max - val) - 1) } else { val + n };
62            // zero the upper bits
63            let val = val as u128;
64            let val = size.truncate(val);
65            (val, oflo)
66        } else {
67            let max = size.unsigned_int_max();
68            let val = self.val;
69            let oflo = val > max - n;
70            let val = if oflo { n - (max - val) - 1 } else { val + n };
71            (val, oflo)
72        };
73        (Self { val, ty: self.ty }, oflo)
74    }
75}

77pub trait IntegerTypeExt {
78    fn to_ty<'db>(&self, interner: DbInterner<'db>) -> Ty<'db>;
79    fn initial_discriminant<'db>(&self, interner: DbInterner<'db>) -> Discr<'db>;
80    fn disr_incr<'db>(
81        &self,
82        interner: DbInterner<'db>,
83        val: Option<Discr<'db>>,
84    ) -> Option<Discr<'db>>;
85}
86
87impl IntegerTypeExt for IntegerType {
88    fn to_ty<'db>(&self, interner: DbInterner<'db>) -> Ty<'db> {
89        match self {
90            IntegerType::Pointer(true) => Ty::new(interner, TyKind::Int(IntTy::Isize)),
91            IntegerType::Pointer(false) => Ty::new(interner, TyKind::Uint(UintTy::Usize)),
92            IntegerType::Fixed(i, s) => i.to_ty(interner, *s),
93        }
94    }
95
96    fn initial_discriminant<'db>(&self, interner: DbInterner<'db>) -> Discr<'db> {
97        Discr { val: 0, ty: self.to_ty(interner) }
98    }
99
100    fn disr_incr<'db>(
101        &self,
102        interner: DbInterner<'db>,
103        val: Option<Discr<'db>>,
104    ) -> Option<Discr<'db>> {
105        if let Some(val) = val {
106            assert_eq!(self.to_ty(interner), val.ty);
107            let (new, oflo) = val.checked_add(interner, 1);
108            if oflo { None } else { Some(new) }
109        } else {
110            Some(self.initial_discriminant(interner))
111        }
112    }
113}

120        interner: DbInterner<'db>,

123        min: i128,
124        max: i128,
125    ) -> (Integer, bool);
126}
127
128impl IntegerExt for Integer {
129    #[inline]
130    fn to_ty<'db>(&self, interner: DbInterner<'db>, signed: bool) -> Ty<'db> {
131        use Integer::*;
132        match (*self, signed) {
133            (I8, false) => Ty::new(interner, TyKind::Uint(UintTy::U8)),
134            (I16, false) => Ty::new(interner, TyKind::Uint(UintTy::U16)),
135            (I32, false) => Ty::new(interner, TyKind::Uint(UintTy::U32)),
136            (I64, false) => Ty::new(interner, TyKind::Uint(UintTy::U64)),
137            (I128, false) => Ty::new(interner, TyKind::Uint(UintTy::U128)),
138            (I8, true) => Ty::new(interner, TyKind::Int(IntTy::I8)),
139            (I16, true) => Ty::new(interner, TyKind::Int(IntTy::I16)),
140            (I32, true) => Ty::new(interner, TyKind::Int(IntTy::I32)),
141            (I64, true) => Ty::new(interner, TyKind::Int(IntTy::I64)),
142            (I128, true) => Ty::new(interner, TyKind::Int(IntTy::I128)),
143        }
144    }
145
146    fn from_int_ty<C: HasDataLayout>(cx: &C, ity: IntTy) -> Integer {
147        use Integer::*;
148        match ity {
149            IntTy::I8 => I8,
150            IntTy::I16 => I16,
151            IntTy::I32 => I32,
152            IntTy::I64 => I64,
153            IntTy::I128 => I128,
154            IntTy::Isize => cx.data_layout().ptr_sized_integer(),
155        }
156    }
157    fn from_uint_ty<C: HasDataLayout>(cx: &C, ity: UintTy) -> Integer {
158        use Integer::*;
159        match ity {
160            UintTy::U8 => I8,
161            UintTy::U16 => I16,
162            UintTy::U32 => I32,
163            UintTy::U64 => I64,
164            UintTy::U128 => I128,
165            UintTy::Usize => cx.data_layout().ptr_sized_integer(),
166        }
167    }
168
169    /// Finds the appropriate Integer type and signedness for the given
170    /// signed discriminant range and `#[repr]` attribute.
171    /// N.B.: `u128` values above `i128::MAX` will be treated as signed, but
172    /// that shouldn't affect anything, other than maybe debuginfo.
173    fn repr_discr<'db>(
174        interner: DbInterner<'db>,
175        ty: Ty<'db>,
176        repr: &ReprOptions,
177        min: i128,
178        max: i128,
179    ) -> (Integer, bool) {
180        // Theoretically, negative values could be larger in unsigned representation
181        // than the unsigned representation of the signed minimum. However, if there
182        // are any negative values, the only valid unsigned representation is u128
183        // which can fit all i128 values, so the result remains unaffected.
184        let unsigned_fit = Integer::fit_unsigned(std::cmp::max(min as u128, max as u128));
185        let signed_fit = std::cmp::max(Integer::fit_signed(min), Integer::fit_signed(max));
186
187        if let Some(ity) = repr.int {
188            let discr = Integer::from_attr(&interner, ity);
189            let fit = if ity.is_signed() { signed_fit } else { unsigned_fit };
190            if discr < fit {
191                panic!(
192                    "Integer::repr_discr: `#[repr]` hint too small for \
193                      discriminant range of enum `{ty:?}`"
194                )
195            }
196            return (discr, ity.is_signed());
197        }
198
199        let at_least = if repr.c() {
200            // This is usually I32, however it can be different on some platforms,
201            // notably hexagon and arm-none/thumb-none
202            interner.data_layout().c_enum_min_size
203        } else {
204            // repr(Rust) enums try to be as small as possible
205            Integer::I8
206        };
207
208        // If there are no negative values, we can use the unsigned fit.
209        if min >= 0 {
210            (std::cmp::max(unsigned_fit, at_least), false)
211        } else {
212            (std::cmp::max(signed_fit, at_least), true)
213        }
214    }
215}

276            Primitive::Float(_) => panic!("floats do not have an int type"),
277        }

326                self.max_universe.as_u32().max(placeholder.universe.as_u32()),

331    }

336                self.max_universe.as_u32().max(placeholder.universe.as_u32()),
337            );
338        }
339
340        c.super_visit_with(self)
341    }
342
343    fn visit_region(&mut self, r: Region<'db>) {
344        if let RegionKind::RePlaceholder(placeholder) = r.kind() {
345            self.max_universe = UniverseIndex::from_u32(
346                self.max_universe.as_u32().max(placeholder.universe.as_u32()),
347            );
348        }
349    }

352pub struct BottomUpFolder<'db, F, G, H>
353where
354    F: FnMut(Ty<'db>) -> Ty<'db>,
355    G: FnMut(Region<'db>) -> Region<'db>,
356    H: FnMut(Const<'db>) -> Const<'db>,
357{
358    pub interner: DbInterner<'db>,
359    pub ty_op: F,
360    pub lt_op: G,
361    pub ct_op: H,
362}
363
364impl<'db, F, G, H> TypeFolder<DbInterner<'db>> for BottomUpFolder<'db, F, G, H>
365where
366    F: FnMut(Ty<'db>) -> Ty<'db>,
367    G: FnMut(Region<'db>) -> Region<'db>,
368    H: FnMut(Const<'db>) -> Const<'db>,
369{
370    fn cx(&self) -> DbInterner<'db> {
371        self.interner
372    }
373
374    fn fold_ty(&mut self, ty: Ty<'db>) -> Ty<'db> {
375        let t = ty.super_fold_with(self);
376        (self.ty_op)(t)
377    }
378
379    fn fold_region(&mut self, r: Region<'db>) -> Region<'db> {
380        // This one is a little different, because `super_fold_with` is not
381        // implemented on non-recursive `Region`.
382        (self.lt_op)(r)
383    }
384
385    fn fold_const(&mut self, ct: Const<'db>) -> Const<'db> {
386        let ct = ct.super_fold_with(self);
387        (self.ct_op)(ct)
388    }
389}

391pub(crate) fn for_trait_impls(
392    db: &dyn HirDatabase,
393    krate: Crate,
394    block: Option<BlockId>,
395    trait_id: hir_def::TraitId,
396    self_ty_fp: Option<TyFingerprint>,
397    mut f: impl FnMut(&TraitImpls) -> ControlFlow<()>,
398) -> ControlFlow<()> {
399    // Note: Since we're using `impls_for_trait` and `impl_provided_for`,
400    // only impls where the trait can be resolved should ever reach Chalk.
401    // `impl_datum` relies on that and will panic if the trait can't be resolved.
402    let in_self_and_deps = db.trait_impls_in_deps(krate);
403    let trait_module = trait_id.module(db);
404    let type_module = match self_ty_fp {
405        Some(TyFingerprint::Adt(adt_id)) => Some(adt_id.module(db)),
406        Some(TyFingerprint::ForeignType(type_id)) => Some(type_id.module(db)),
407        Some(TyFingerprint::Dyn(trait_id)) => Some(trait_id.module(db)),
408        _ => None,
409    };
410
411    let mut def_blocks =
412        [trait_module.containing_block(), type_module.and_then(|it| it.containing_block())];
413
414    let block_impls = iter::successors(block, |&block_id| {
415        cov_mark::hit!(block_local_impls);
416        block_id.loc(db).module.containing_block()
417    })
418    .inspect(|&block_id| {
419        // make sure we don't search the same block twice
420        def_blocks.iter_mut().for_each(|block| {
421            if *block == Some(block_id) {
422                *block = None;
423            }
424        });
425    })
426    .filter_map(|block_id| db.trait_impls_in_block(block_id));
427    for it in in_self_and_deps.iter().map(ops::Deref::deref) {
428        f(it)?;
429    }
430    for it in block_impls {
431        f(&it)?;
432    }
433    for it in def_blocks.into_iter().flatten().filter_map(|it| db.trait_impls_in_block(it)) {
434        f(&it)?;
435    }
436    ControlFlow::Continue(())
437}

440pub fn sizedness_constraint_for_ty<'db>(
441    interner: DbInterner<'db>,
442    sizedness: SizedTraitKind,
443    ty: Ty<'db>,
444) -> Option<Ty<'db>> {
445    use rustc_type_ir::TyKind::*;
446
447    match ty.kind() {
448        // these are always sized
449        Bool | Char | Int(..) | Uint(..) | Float(..) | RawPtr(..) | Ref(..) | FnDef(..)
450        | FnPtr(..) | Array(..) | Closure(..) | CoroutineClosure(..) | Coroutine(..)
451        | CoroutineWitness(..) | Never => None,
452
453        // these are never sized
454        Str | Slice(..) | Dynamic(_, _) => match sizedness {
455            // Never `Sized`
456            SizedTraitKind::Sized => Some(ty),
457            // Always `MetaSized`
458            SizedTraitKind::MetaSized => None,
459        },
460
461        // Maybe `Sized` or `MetaSized`
462        Param(..) | Alias(..) | Error(_) => Some(ty),
463
464        // We cannot instantiate the binder, so just return the *original* type back,
465        // but only if the inner type has a sized constraint. Thus we skip the binder,
466        // but don't actually use the result from `sized_constraint_for_ty`.
467        UnsafeBinder(inner_ty) => {
468            sizedness_constraint_for_ty(interner, sizedness, inner_ty.skip_binder()).map(|_| ty)
469        }
470
471        // Never `MetaSized` or `Sized`
472        Foreign(..) => Some(ty),
473
474        // Recursive cases
475        Pat(ty, _) => sizedness_constraint_for_ty(interner, sizedness, ty),
476
477        Tuple(tys) => tys
478            .into_iter()
479            .next_back()
480            .and_then(|ty| sizedness_constraint_for_ty(interner, sizedness, ty)),
481
482        Adt(adt, args) => {
483            let tail_ty =
484                EarlyBinder::bind(adt.all_field_tys(interner).skip_binder().into_iter().last()?)
485                    .instantiate(interner, args);
486            sizedness_constraint_for_ty(interner, sizedness, tail_ty)
487        }
488
489        Placeholder(..) | Bound(..) | Infer(..) => {
490            panic!("unexpected type `{ty:?}` in sizedness_constraint_for_ty")
491        }
492    }
493}

495pub fn apply_args_to_binder<'db, T: TypeFoldable<DbInterner<'db>>>(
496    b: Binder<'db, T>,
497    args: GenericArgs<'db>,
498    interner: DbInterner<'db>,
499) -> T {
500    let types = &mut |ty: BoundTy| args.as_slice()[ty.var.index()].expect_ty();
501    let regions = &mut |region: BoundRegion| args.as_slice()[region.var.index()].expect_region();
502    let consts = &mut |const_: BoundConst| args.as_slice()[const_.var.index()].expect_const();
503    let mut instantiate = BoundVarReplacer::new(interner, FnMutDelegate { types, regions, consts });
504    b.skip_binder().fold_with(&mut instantiate)
505}

507pub fn explicit_item_bounds<'db>(
508    interner: DbInterner<'db>,
509    def_id: SolverDefId,
510) -> EarlyBinder<'db, Clauses<'db>> {
511    let db = interner.db();
512    match def_id {
513        SolverDefId::TypeAliasId(type_alias) => {
514            // Lower bounds -- we could/should maybe move this to a separate query in `lower`
515            let type_alias_data = db.type_alias_signature(type_alias);
516            let resolver = hir_def::resolver::HasResolver::resolver(type_alias, db);
517            let mut ctx = TyLoweringContext::new(
518                db,
519                &resolver,
520                &type_alias_data.store,
521                type_alias.into(),
522                LifetimeElisionKind::AnonymousReportError,
523            );
524
525            let item_args = GenericArgs::identity_for_item(interner, def_id);
526            let interner_ty = Ty::new_projection_from_args(interner, def_id, item_args);
527
528            let mut bounds = Vec::new();
529            for bound in &type_alias_data.bounds {
530                ctx.lower_type_bound(bound, interner_ty, false).for_each(|pred| {
531                    bounds.push(pred);
532                });
533            }
534
535            if !ctx.unsized_types.contains(&interner_ty) {
536                let sized_trait = LangItem::Sized
537                    .resolve_trait(ctx.db, interner.krate.expect("Must have interner.krate"));
538                let sized_bound = sized_trait.map(|trait_id| {
539                    let trait_ref = TraitRef::new_from_args(
540                        interner,
541                        trait_id.into(),
542                        GenericArgs::new_from_iter(interner, [interner_ty.into()]),
543                    );
544                    Clause(Predicate::new(
545                        interner,
546                        Binder::dummy(rustc_type_ir::PredicateKind::Clause(
547                            rustc_type_ir::ClauseKind::Trait(TraitPredicate {
548                                trait_ref,
549                                polarity: rustc_type_ir::PredicatePolarity::Positive,
550                            }),
551                        )),
552                    ))
553                });
554                bounds.extend(sized_bound);
555                bounds.shrink_to_fit();
556            }
557
558            rustc_type_ir::EarlyBinder::bind(Clauses::new_from_iter(interner, bounds))
559        }
560        SolverDefId::InternedOpaqueTyId(id) => {
561            let full_id = db.lookup_intern_impl_trait_id(id);
562            match full_id {
563                crate::ImplTraitId::ReturnTypeImplTrait(func, idx) => {
564                    let datas = db
565                        .return_type_impl_traits(func)
566                        .expect("impl trait id without impl traits");
567                    let datas = (*datas).as_ref().skip_binder();
568                    let data = &datas.impl_traits[Idx::from_raw(idx.into_raw())];
569                    EarlyBinder::bind(Clauses::new_from_iter(interner, data.predicates.clone()))
570                }
571                crate::ImplTraitId::TypeAliasImplTrait(alias, idx) => {
572                    let datas = db
573                        .type_alias_impl_traits(alias)
574                        .expect("impl trait id without impl traits");
575                    let datas = (*datas).as_ref().skip_binder();
576                    let data = &datas.impl_traits[Idx::from_raw(idx.into_raw())];
577                    EarlyBinder::bind(Clauses::new_from_iter(interner, data.predicates.clone()))
578                }
579            }
580        }
581        _ => panic!("Unexpected GenericDefId"),
582    }
583}

590    fn visit_ty(&mut self, t: Ty<'db>) -> Self::Result {
591        match t.kind() {
592            rustc_type_ir::TyKind::Error(_) => ControlFlow::Break(()),
593            _ => t.super_visit_with(self),
594        }
595    }
596}

598/// The inverse of [`BoundVarReplacer`]: replaces placeholders with the bound vars from which they came.
599pub struct PlaceholderReplacer<'a, 'db> {
600    infcx: &'a InferCtxt<'db>,
601    mapped_regions: FxIndexMap<PlaceholderRegion, BoundRegion>,
602    mapped_types: FxIndexMap<Placeholder<BoundTy>, BoundTy>,
603    mapped_consts: FxIndexMap<PlaceholderConst, BoundConst>,
604    universe_indices: &'a [Option<UniverseIndex>],
605    current_index: DebruijnIndex,
606}
607
608impl<'a, 'db> PlaceholderReplacer<'a, 'db> {
609    pub fn replace_placeholders<T: TypeFoldable<DbInterner<'db>>>(
610        infcx: &'a InferCtxt<'db>,
611        mapped_regions: FxIndexMap<PlaceholderRegion, BoundRegion>,
612        mapped_types: FxIndexMap<Placeholder<BoundTy>, BoundTy>,
613        mapped_consts: FxIndexMap<PlaceholderConst, BoundConst>,
614        universe_indices: &'a [Option<UniverseIndex>],
615        value: T,
616    ) -> T {
617        let mut replacer = PlaceholderReplacer {
618            infcx,
619            mapped_regions,
620            mapped_types,
621            mapped_consts,
622            universe_indices,
623            current_index: INNERMOST,
624        };
625        value.fold_with(&mut replacer)
626    }
627}
628
629impl<'db> TypeFolder<DbInterner<'db>> for PlaceholderReplacer<'_, 'db> {
630    fn cx(&self) -> DbInterner<'db> {
631        self.infcx.interner
632    }
633
634    fn fold_binder<T: TypeFoldable<DbInterner<'db>>>(
635        &mut self,
636        t: Binder<'db, T>,
637    ) -> Binder<'db, T> {
638        if !t.has_placeholders() && !t.has_infer() {
639            return t;
640        }
641        self.current_index.shift_in(1);
642        let t = t.super_fold_with(self);
643        self.current_index.shift_out(1);
644        t
645    }
646
647    fn fold_region(&mut self, r0: Region<'db>) -> Region<'db> {
648        let r1 = match r0.kind() {
649            RegionKind::ReVar(vid) => self
650                .infcx
651                .inner
652                .borrow_mut()
653                .unwrap_region_constraints()
654                .opportunistic_resolve_var(self.infcx.interner, vid),
655            _ => r0,
656        };
657
658        let r2 = match r1.kind() {
659            RegionKind::RePlaceholder(p) => {
660                let replace_var = self.mapped_regions.get(&p);
661                match replace_var {
662                    Some(replace_var) => {
663                        let index = self
664                            .universe_indices
665                            .iter()
666                            .position(|u| matches!(u, Some(pu) if *pu == p.universe))
667                            .unwrap_or_else(|| panic!("Unexpected placeholder universe."));
668                        let db = DebruijnIndex::from_usize(
669                            self.universe_indices.len() - index + self.current_index.as_usize() - 1,
670                        );
671                        Region::new_bound(self.cx(), db, *replace_var)
672                    }
673                    None => r1,
674                }
675            }
676            _ => r1,
677        };
678
679        tracing::debug!(?r0, ?r1, ?r2, "fold_region");
680
681        r2
682    }
683
684    fn fold_ty(&mut self, ty: Ty<'db>) -> Ty<'db> {
685        let ty = self.infcx.shallow_resolve(ty);
686        match ty.kind() {
687            TyKind::Placeholder(p) => {
688                let replace_var = self.mapped_types.get(&p);
689                match replace_var {
690                    Some(replace_var) => {
691                        let index = self
692                            .universe_indices
693                            .iter()
694                            .position(|u| matches!(u, Some(pu) if *pu == p.universe))
695                            .unwrap_or_else(|| panic!("Unexpected placeholder universe."));
696                        let db = DebruijnIndex::from_usize(
697                            self.universe_indices.len() - index + self.current_index.as_usize() - 1,
698                        );
699                        Ty::new_bound(self.infcx.interner, db, *replace_var)
700                    }
701                    None => {
702                        if ty.has_infer() {
703                            ty.super_fold_with(self)
704                        } else {
705                            ty
706                        }
707                    }
708                }
709            }
710
711            _ if ty.has_placeholders() || ty.has_infer() => ty.super_fold_with(self),
712            _ => ty,
713        }
714    }
715
716    fn fold_const(&mut self, ct: Const<'db>) -> Const<'db> {
717        let ct = self.infcx.shallow_resolve_const(ct);
718        if let ConstKind::Placeholder(p) = ct.kind() {
719            let replace_var = self.mapped_consts.get(&p);
720            match replace_var {
721                Some(replace_var) => {
722                    let index = self
723                        .universe_indices
724                        .iter()
725                        .position(|u| matches!(u, Some(pu) if *pu == p.universe))
726                        .unwrap_or_else(|| panic!("Unexpected placeholder universe."));
727                    let db = DebruijnIndex::from_usize(
728                        self.universe_indices.len() - index + self.current_index.as_usize() - 1,
729                    );
730                    Const::new_bound(self.infcx.interner, db, *replace_var)
731                }
732                None => {
733                    if ct.has_infer() {
734                        ct.super_fold_with(self)
735                    } else {
736                        ct
737                    }
738                }
739            }
740        } else {
741            ct.super_fold_with(self)
742        }
743    }
744}
745
746pub fn sizedness_fast_path<'db>(
747    tcx: DbInterner<'db>,
748    predicate: Predicate<'db>,
749    param_env: ParamEnv<'db>,
750) -> bool {
751    // Proving `Sized`/`MetaSized`, very often on "obviously sized" types like
752    // `&T`, accounts for about 60% percentage of the predicates we have to prove. No need to
753    // canonicalize and all that for such cases.
754    if let PredicateKind::Clause(ClauseKind::Trait(trait_pred)) = predicate.kind().skip_binder()
755        && trait_pred.polarity == PredicatePolarity::Positive
756    {
757        let sizedness = match tcx.as_trait_lang_item(trait_pred.def_id()) {
758            Some(SolverTraitLangItem::Sized) => SizedTraitKind::Sized,
759            Some(SolverTraitLangItem::MetaSized) => SizedTraitKind::MetaSized,
760            _ => return false,
761        };
762
763        // FIXME(sized_hierarchy): this temporarily reverts the `sized_hierarchy` feature
764        // while a proper fix for `tests/ui/sized-hierarchy/incomplete-inference-issue-143992.rs`
765        // is pending a proper fix
766        if matches!(sizedness, SizedTraitKind::MetaSized) {
767            return true;
768        }
769
770        if trait_pred.self_ty().has_trivial_sizedness(tcx, sizedness) {
771            tracing::debug!("fast path -- trivial sizedness");
772            return true;
773        }
774
775        if matches!(trait_pred.self_ty().kind(), TyKind::Param(_) | TyKind::Placeholder(_)) {
776            for clause in param_env.caller_bounds().iter() {
777                if let ClauseKind::Trait(clause_pred) = clause.kind().skip_binder()
778                    && clause_pred.polarity == PredicatePolarity::Positive
779                    && clause_pred.self_ty() == trait_pred.self_ty()
780                    && (clause_pred.def_id() == trait_pred.def_id()
781                        || (sizedness == SizedTraitKind::MetaSized
782                            && tcx.is_trait_lang_item(
783                                clause_pred.def_id(),
784                                SolverTraitLangItem::Sized,
785                            )))
786                {
787                    return true;
788                }
789            }
790        }
791    }
792
793    false
794}
