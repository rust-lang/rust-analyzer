15pub(crate) fn generics(interner: DbInterner<'_>, def: SolverDefId) -> Generics {
16    let mk_lt = |parent, index, local_id| {
17        let id = GenericParamId::LifetimeParamId(LifetimeParamId { parent, local_id });
18        GenericParamDef { index, id }
19    };
20    let mk_ty = |parent, index, local_id, p: &TypeOrConstParamData| {
21        let id = TypeOrConstParamId { parent, local_id };
22        let id = match p {
23            TypeOrConstParamData::TypeParamData(_) => {
24                GenericParamId::TypeParamId(TypeParamId::from_unchecked(id))
25            }
26            TypeOrConstParamData::ConstParamData(_) => {
27                GenericParamId::ConstParamId(ConstParamId::from_unchecked(id))
28            }
29        };
30        GenericParamDef { index, id }
31    };
32    let own_params_for_generic_params = |parent, params: &GenericParams| {
33        let mut result = Vec::with_capacity(params.len());
34        let mut type_and_consts = params.iter_type_or_consts();
35        let mut index = 0;
36        if let Some(self_param) = params.trait_self_param() {
37            result.push(mk_ty(parent, 0, self_param, &params[self_param]));
38            type_and_consts.next();
39            index += 1;
40        }
41        result.extend(params.iter_lt().map(|(local_id, _data)| {
42            let lt = mk_lt(parent, index, local_id);
43            index += 1;
44            lt
45        }));
46        result.extend(type_and_consts.map(|(local_id, data)| {
47            let ty = mk_ty(parent, index, local_id, data);
48            index += 1;
49            ty
50        }));
51        result
52    };
53
54    let db = interner.db;
55    let (parent, own_params) = match (def.try_into(), def) {
56        (Ok(def), _) => (
57            parent_generic_def(db, def),
58            own_params_for_generic_params(def, &db.generic_params(def)),
59        ),
60        (_, SolverDefId::InternedOpaqueTyId(id)) => {
61            match db.lookup_intern_impl_trait_id(id) {
62                crate::ImplTraitId::ReturnTypeImplTrait(function_id, _) => {
63                    // The opaque type itself does not have generics - only the parent function
64                    (Some(GenericDefId::FunctionId(function_id)), vec![])
65                }
66                crate::ImplTraitId::TypeAliasImplTrait(type_alias_id, _) => {
67                    (Some(type_alias_id.into()), Vec::new())
68                }
69            }
70        }
71        (_, SolverDefId::BuiltinDeriveImplId(id)) => {
72            return crate::builtin_derive::generics_of(interner, id);
73        }
74        _ => panic!("No generics for {def:?}"),
75    };
76    let parent_generics = parent.map(|def| Box::new(generics(interner, def.into())));
77
78    Generics {
79        parent,
80        parent_count: parent_generics.map_or(0, |g| g.parent_count + g.own_params.len()),
81        own_params,
82    }
83}
84
85#[derive(Debug)]
86pub struct Generics {
87    pub parent: Option<GenericDefId>,
88    pub parent_count: usize,
89    pub own_params: Vec<GenericParamDef>,
90}
91
92impl Generics {
93    pub(crate) fn push_param(&mut self, id: GenericParamId) {
94        let index = self.count() as u32;
95        self.own_params.push(GenericParamDef { index, id });
96    }
97}
98
99#[derive(Debug)]
100pub struct GenericParamDef {
101    index: u32,
102    pub(crate) id: GenericParamId,
103}
104
105impl GenericParamDef {
106    /// Returns the index of the param on the self generics only
107    /// (i.e. not including parent generics)
108    pub fn index(&self) -> u32 {
109        self.index
110    }
111}
112
113impl<'db> rustc_type_ir::inherent::GenericsOf<DbInterner<'db>> for Generics {
114    fn count(&self) -> usize {
115        self.parent_count + self.own_params.len()
116    }
117}
