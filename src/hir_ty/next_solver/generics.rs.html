11pub(crate) fn generics(interner: DbInterner<'_>, def: SolverDefId) -> Generics<'_> {
12    let db = interner.db;
13    let def = match (def.try_into(), def) {
14        (Ok(def), _) => def,
15        (_, SolverDefId::InternedOpaqueTyId(id)) => match id.loc(db) {
16            crate::ImplTraitId::ReturnTypeImplTrait(function_id, _) => function_id.into(),
17            crate::ImplTraitId::TypeAliasImplTrait(type_alias_id, _) => type_alias_id.into(),
18        },
19        (_, SolverDefId::BuiltinDeriveImplId(id)) => {
20            return crate::builtin_derive::generics_of(interner, id);
21        }
22        (_, SolverDefId::AnonConstId(id)) => {
23            let loc = id.loc(db);
24            let generic_def = loc.owner.generic_def(db);
25            return if loc.allow_using_generic_params {
26                Generics::from_generic_def(db, generic_def)
27            } else {
28                #[expect(
29                    deprecated,
30                    reason = "`Generics` only exposes an iterator over `GenericParamId`, \
31                        so you cannot exploit the erroneous `crate::generics::Generics`"
32                )]
33                Generics {
34                    generics: crate::generics::Generics::empty(generic_def),
35                    additional_param: None,
36                }
37            };
38        }
39        _ => panic!("No generics for {def:?}"),
40    };
41
42    Generics::from_generic_def(db, def)
43}

45#[derive(Debug)]
46pub struct Generics<'db> {
47    generics: crate::generics::Generics<'db>,
48    /// This is used for builtin derives, specifically `CoercePointee`.
49    additional_param: Option<GenericParamId>,
50}
51
52impl<'db> Generics<'db> {
53    pub(crate) fn from_generic_def(db: &'db dyn HirDatabase, def: GenericDefId) -> Generics<'db> {
54        Generics { generics: crate::generics::generics(db, def), additional_param: None }
55    }
56
57    pub(crate) fn from_generic_def_plus_one(
58        db: &'db dyn HirDatabase,
59        def: GenericDefId,
60        additional_param: GenericParamId,
61    ) -> Generics<'db> {
62        Generics {
63            generics: crate::generics::generics(db, def),
64            additional_param: Some(additional_param),
65        }
66    }
67
68    pub(super) fn iter_id(&self) -> impl Iterator<Item = GenericParamId> {
69        self.generics.iter_id().chain(self.additional_param)
70    }
71}
72
73impl<'db> rustc_type_ir::inherent::GenericsOf<DbInterner<'db>> for Generics<'db> {
74    fn count(&self) -> usize {
75        self.generics.len() + usize::from(self.additional_param.is_some())
76    }
