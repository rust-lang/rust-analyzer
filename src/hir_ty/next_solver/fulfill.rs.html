30/// A trait engine using the new trait solver.
31///
32/// This is mostly identical to how `evaluate_all` works inside of the
33/// solver, except that the requirements are slightly different.
34///
35/// Unlike `evaluate_all` it is possible to add new obligations later on
36/// and we also have to track diagnostics information by using `Obligation`
37/// instead of `Goal`.
38///
39/// It is also likely that we want to use slightly different datastructures
40/// here as this will have to deal with far more root goals than `evaluate_all`.
41#[derive(Debug, Clone)]
42pub struct FulfillmentCtxt<'db> {
43    obligations: ObligationStorage<'db>,
44
45    /// The snapshot in which this context was created. Using the context
46    /// outside of this snapshot leads to subtle bugs if the snapshot
47    /// gets rolled back. Because of this we explicitly check that we only
48    /// use the context in exactly this snapshot.
49    #[expect(unused)]
50    usable_in_snapshot: usize,
51    try_evaluate_obligations_scratch: PendingObligations<'db>,
52}
53
54#[derive(Default, Debug, Clone)]
55struct ObligationStorage<'db> {
56    /// Obligations which resulted in an overflow in fulfillment itself.
57    ///
58    /// We cannot eagerly return these as error so we instead store them here
59    /// to avoid recomputing them each time `try_evaluate_obligations` is called.
60    /// This also allows us to return the correct `FulfillmentError` for them.
61    overflowed: Vec<PredicateObligation<'db>>,
62    pending: PendingObligations<'db>,
63}
64
65impl<'db> ObligationStorage<'db> {
66    fn register(
67        &mut self,
68        obligation: PredicateObligation<'db>,
69        stalled_on: Option<GoalStalledOn<DbInterner<'db>>>,
70    ) {
71        self.pending.push((obligation, stalled_on));
72    }
73
74    fn clone_pending(&self) -> PredicateObligations<'db> {
75        let mut obligations: PredicateObligations<'db> =
76            self.pending.iter().map(|(o, _)| o.clone()).collect();
77        obligations.extend(self.overflowed.iter().cloned());
78        obligations
79    }
80
81    fn drain_pending<'this, 'cond>(
82        &'this mut self,
83        cond: impl 'cond + Fn(&PredicateObligation<'db>) -> bool,
84    ) -> impl Iterator<Item = (PredicateObligation<'db>, Option<GoalStalledOn<DbInterner<'db>>>)>
85    {
86        self.pending.extract_if(.., move |(o, _)| cond(o))
87    }
88
89    fn on_fulfillment_overflow(&mut self, infcx: &InferCtxt<'db>) {
90        infcx.probe(|_| {
91            // IMPORTANT: we must not use solve any inference variables in the obligations
92            // as this is all happening inside of a probe. We use a probe to make sure
93            // we get all obligations involved in the overflow. We pretty much check: if
94            // we were to do another step of `try_evaluate_obligations`, which goals would
95            // change.
96            // FIXME: <https://github.com/Gankra/thin-vec/pull/66> is merged, this can be removed.
97            self.overflowed.extend(
98                self.pending
99                    .extract_if(.., |(o, stalled_on)| {
100                        let goal = o.as_goal();
101                        let result = <&SolverContext<'db>>::from(infcx).evaluate_root_goal(
102                            goal,
103                            Span::dummy(),
104                            stalled_on.take(),
105                        );
106                        matches!(result, Ok(GoalEvaluation { has_changed: HasChanged::Yes, .. }))
107                    })
108                    .map(|(o, _)| o),
109            );
110        })
111    }
112}

114impl<'db> FulfillmentCtxt<'db> {
115    pub fn new(infcx: &InferCtxt<'db>) -> FulfillmentCtxt<'db> {
116        FulfillmentCtxt {

119            try_evaluate_obligations_scratch: Default::default(),
120        }
121    }

124impl<'db> FulfillmentCtxt<'db> {
125    #[tracing::instrument(level = "trace", skip(self, _infcx))]
126    pub(crate) fn register_predicate_obligation(
127        &mut self,
128        _infcx: &InferCtxt<'db>,
129        obligation: PredicateObligation<'db>,
130    ) {
131        // FIXME: See the comment in `try_evaluate_obligations()`.
132        // assert_eq!(self.usable_in_snapshot, infcx.num_open_snapshots());
133        self.obligations.register(obligation, None);
134    }
135
136    pub(crate) fn register_predicate_obligations(
137        &mut self,
138        _infcx: &InferCtxt<'db>,
139        obligations: impl IntoIterator<Item = PredicateObligation<'db>>,
140    ) {
141        // FIXME: See the comment in `try_evaluate_obligations()`.
142        // assert_eq!(self.usable_in_snapshot, infcx.num_open_snapshots());
143        obligations.into_iter().for_each(|obligation| self.obligations.register(obligation, None));
144    }
145
146    pub(crate) fn collect_remaining_errors(
147        &mut self,
148        _infcx: &InferCtxt<'db>,
149    ) -> Vec<NextSolverError<'db>> {
150        self.obligations
151            .pending
152            .drain(..)
153            .map(|(obligation, _)| NextSolverError::Ambiguity(obligation))
154            .chain(self.obligations.overflowed.drain(..).map(NextSolverError::Overflow))
155            .collect()
156    }
157
158    pub(crate) fn try_evaluate_obligations(
159        &mut self,
160        infcx: &InferCtxt<'db>,
161    ) -> Vec<NextSolverError<'db>> {
162        // FIXME(next-solver): We should bring this assertion back. Currently it panics because
163        // there are places which use `InferenceTable` and open a snapshot and register obligations
164        // and select. They should use a different `ObligationCtxt` instead. Then we'll be also able
165        // to not put the obligations queue in `InferenceTable`'s snapshots.
166        // assert_eq!(self.usable_in_snapshot, infcx.num_open_snapshots());
167        self.try_evaluate_obligations_scratch.clear();
168        let mut errors = Vec::new();
169        loop {
170            let mut any_changed = false;
171            self.try_evaluate_obligations_scratch.extend(self.obligations.drain_pending(|_| true));
172            for (mut obligation, stalled_on) in self.try_evaluate_obligations_scratch.drain(..) {
173                if obligation.recursion_depth >= infcx.interner.recursion_limit() {
174                    self.obligations.on_fulfillment_overflow(infcx);
175                    // Only return true errors that we have accumulated while processing.
176                    return errors;
177                }
178
179                let goal = obligation.as_goal();
180                let delegate = <&SolverContext<'db>>::from(infcx);
181                if let Some(certainty) = delegate.compute_goal_fast_path(goal, Span::dummy()) {
182                    match certainty {
183                        Certainty::Yes => {}
184                        Certainty::Maybe { .. } => {
185                            self.obligations.register(obligation, None);
186                        }
187                    }
188                    continue;
189                }
190
191                let result = delegate.evaluate_root_goal(goal, Span::dummy(), stalled_on);
192                infcx.inspect_evaluated_obligation(&obligation, &result, || {
193                    Some(delegate.evaluate_root_goal_for_proof_tree(goal, Span::dummy()).1)
194                });
195                let GoalEvaluation { goal: _, certainty, has_changed, stalled_on } = match result {
196                    Ok(result) => result,
197                    Err(NoSolution) => {
198                        errors.push(NextSolverError::TrueError(obligation));
199                        continue;
200                    }
201                };
202
203                if has_changed == HasChanged::Yes {
204                    // We increment the recursion depth here to track the number of times
205                    // this goal has resulted in inference progress. This doesn't precisely
206                    // model the way that we track recursion depth in the old solver due
207                    // to the fact that we only process root obligations, but it is a good
208                    // approximation and should only result in fulfillment overflow in
209                    // pathological cases.
210                    obligation.recursion_depth += 1;
211                    any_changed = true;
212                }
213
214                match certainty {
215                    Certainty::Yes => {}
216                    Certainty::Maybe { .. } => self.obligations.register(obligation, stalled_on),
217                }
218            }
219
220            if !any_changed {
221                break;
222            }
223        }
224
225        errors
226    }
227
228    pub(crate) fn evaluate_obligations_error_on_ambiguity(
229        &mut self,
230        infcx: &InferCtxt<'db>,
231    ) -> Vec<NextSolverError<'db>> {
232        let errors = self.try_evaluate_obligations(infcx);
233        if !errors.is_empty() {
234            return errors;
235        }
236
237        self.collect_remaining_errors(infcx)
238    }
239
240    pub(crate) fn pending_obligations(&self) -> PredicateObligations<'db> {
241        self.obligations.clone_pending()
242    }
243
244    pub(crate) fn drain_stalled_obligations_for_coroutines(
245        &mut self,
246        infcx: &InferCtxt<'db>,
247    ) -> PredicateObligations<'db> {
248        let stalled_coroutines = match infcx.typing_mode() {
249            TypingMode::Analysis { defining_opaque_types_and_generators } => {
250                defining_opaque_types_and_generators
251            }
252            TypingMode::Coherence
253            | TypingMode::Borrowck { defining_opaque_types: _ }
254            | TypingMode::PostBorrowckAnalysis { defined_opaque_types: _ }
255            | TypingMode::PostAnalysis => return Default::default(),
256        };
257        let stalled_coroutines = stalled_coroutines.as_slice();
258
259        if stalled_coroutines.is_empty() {
260            return Default::default();
261        }
262
263        self.obligations
264            .drain_pending(|obl| {
265                infcx.probe(|_| {
266                    infcx
267                        .visit_proof_tree(
268                            obl.as_goal(),
269                            &mut StalledOnCoroutines {
270                                stalled_coroutines,
271                                cache: Default::default(),
272                            },
273                        )
274                        .is_break()
275                })
276            })
277            .map(|(o, _)| o)
278            .collect()
279    }
280}

282/// Detect if a goal is stalled on a coroutine that is owned by the current typeck root.
283///
284/// This function can (erroneously) fail to detect a predicate, i.e. it doesn't need to
285/// be complete. However, this will lead to ambiguity errors, so we want to make it
286/// accurate.
287///
288/// This function can be also return false positives, which will lead to poor diagnostics
289/// so we want to keep this visitor *precise* too.
290pub struct StalledOnCoroutines<'a, 'db> {
291    pub stalled_coroutines: &'a [SolverDefId],
292    pub cache: FxHashSet<Ty<'db>>,
293}
294
295impl<'db> ProofTreeVisitor<'db> for StalledOnCoroutines<'_, 'db> {
296    type Result = ControlFlow<()>;
297
298    fn visit_goal(&mut self, inspect_goal: &super::inspect::InspectGoal<'_, 'db>) -> Self::Result {
299        inspect_goal.goal().predicate.visit_with(self)?;
300
301        if let Some(candidate) = inspect_goal.unique_applicable_candidate() {
302            candidate.visit_nested_no_probe(self)
303        } else {
304            ControlFlow::Continue(())
305        }
306    }
307}
308
309impl<'db> TypeVisitor<DbInterner<'db>> for StalledOnCoroutines<'_, 'db> {
310    type Result = ControlFlow<()>;
311
312    fn visit_ty(&mut self, ty: Ty<'db>) -> Self::Result {
313        if !self.cache.insert(ty) {
314            return ControlFlow::Continue(());
315        }
316
317        if let TyKind::Coroutine(def_id, _) = ty.kind()
318            && self.stalled_coroutines.contains(&def_id.into())
319        {
320            ControlFlow::Break(())
321        } else if ty.has_coroutines() {
322            ty.super_visit_with(self)
323        } else {
324            ControlFlow::Continue(())
325        }
326    }
327}

329#[derive(Debug)]
330pub enum NextSolverError<'db> {
331    TrueError(PredicateObligation<'db>),
332    Ambiguity(PredicateObligation<'db>),
333    Overflow(PredicateObligation<'db>),
334}
335
336impl NextSolverError<'_> {
337    #[inline]
338    pub fn is_true_error(&self) -> bool {
339        matches!(self, NextSolverError::TrueError(_))
340    }
