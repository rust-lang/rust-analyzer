9#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
10pub struct StoredEarlyBinder<T>(T);
11
12impl<T> StoredEarlyBinder<T> {
13    #[inline]
14    pub fn bind(value: T) -> Self {
15        Self(value)
16    }
17
18    #[inline]
19    pub fn skip_binder(self) -> T {
20        self.0
21    }
22
23    #[inline]
24    pub fn as_ref(&self) -> StoredEarlyBinder<&T> {
25        StoredEarlyBinder(&self.0)
26    }
27
28    #[inline]
29    pub fn get_with<'db, 'a, R>(&'a self, f: impl FnOnce(&'a T) -> R) -> EarlyBinder<'db, R> {
30        EarlyBinder::bind(f(&self.0))
31    }
32}
33
34impl StoredEarlyBinder<StoredTy> {
35    #[inline]
36    pub fn get<'db>(&self) -> EarlyBinder<'db, Ty<'db>> {
37        self.get_with(|it| it.as_ref())
38    }
39}
40
41impl StoredEarlyBinder<StoredClauses> {
42    #[inline]
43    pub fn get<'db>(&self) -> EarlyBinder<'db, Clauses<'db>> {
44        self.get_with(|it| it.as_ref())
45    }

48#[derive(Debug, Clone, PartialEq, Eq, Hash)]
49pub struct StoredPolyFnSig {
50    bound_vars: StoredBoundVarKinds,
51    inputs_and_output: StoredTys,
52    c_variadic: bool,
53    safety: Safety,
54    abi: FnAbi,
55}
56
57impl StoredPolyFnSig {
58    #[inline]
59    pub fn new(sig: PolyFnSig<'_>) -> Self {
60        let bound_vars = sig.bound_vars().store();
61        let sig = sig.skip_binder();
62        Self {
63            bound_vars,
64            inputs_and_output: sig.inputs_and_output.store(),
65            c_variadic: sig.c_variadic,
66            safety: sig.safety,
67            abi: sig.abi,
68        }
69    }
70
71    #[inline]
72    pub fn get(&self) -> PolyFnSig<'_> {
73        Binder::bind_with_vars(
74            FnSig {
75                inputs_and_output: self.inputs_and_output.as_ref(),
76                c_variadic: self.c_variadic,
77                safety: self.safety,
78                abi: self.abi,
79            },
80            self.bound_vars.as_ref(),
81        )
82    }
