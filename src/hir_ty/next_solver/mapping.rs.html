53pub fn convert_binder_to_early_binder<'db, T: rustc_type_ir::TypeFoldable<DbInterner<'db>>>(
54    interner: DbInterner<'db>,
55    def: GenericDefId,
56    binder: rustc_type_ir::Binder<DbInterner<'db>, T>,
57) -> rustc_type_ir::EarlyBinder<DbInterner<'db>, T> {
58    let mut folder = BinderToEarlyBinder {
59        interner,
60        debruijn: rustc_type_ir::DebruijnIndex::ZERO,
61        params: crate::generics::generics(interner.db, def).iter_id().collect(),
62    };
63    rustc_type_ir::EarlyBinder::bind(binder.skip_binder().fold_with(&mut folder))
64}

72impl<'db> rustc_type_ir::TypeFolder<DbInterner<'db>> for BinderToEarlyBinder<'db> {
73    fn cx(&self) -> DbInterner<'db> {
74        self.interner
75    }
76
77    fn fold_binder<T>(
78        &mut self,
79        t: rustc_type_ir::Binder<DbInterner<'db>, T>,
80    ) -> rustc_type_ir::Binder<DbInterner<'db>, T>
81    where
82        T: TypeFoldable<DbInterner<'db>>,
83    {
84        self.debruijn.shift_in(1);
85        let result = t.super_fold_with(self);
86        self.debruijn.shift_out(1);
87        result
88    }
89
90    fn fold_ty(&mut self, t: Ty<'db>) -> Ty<'db> {
91        match t.kind() {
92            rustc_type_ir::TyKind::Bound(debruijn, bound_ty) if self.debruijn == debruijn => {
93                let var: rustc_type_ir::BoundVar = bound_ty.var();
94                let GenericParamId::TypeParamId(id) = self.params[bound_ty.var.as_usize()] else {
95                    unreachable!()
96                };
97                Ty::new(
98                    self.cx(),
99                    rustc_type_ir::TyKind::Param(ParamTy { index: var.as_u32(), id }),
100                )
101            }
102            _ => t.super_fold_with(self),
103        }
104    }
105
106    fn fold_region(&mut self, r: Region<'db>) -> Region<'db> {
107        match r.kind() {
108            rustc_type_ir::ReBound(debruijn, bound_region) if self.debruijn == debruijn => {
109                let var: rustc_type_ir::BoundVar = bound_region.var();
110                let GenericParamId::LifetimeParamId(id) = self.params[bound_region.var.as_usize()]
111                else {
112                    unreachable!()
113                };
114                Region::new(
115                    self.cx(),
116                    rustc_type_ir::RegionKind::ReEarlyParam(EarlyParamRegion {
117                        index: var.as_u32(),
118                        id,
119                    }),
120                )
121            }
122            _ => r,
123        }
124    }
125
126    fn fold_const(&mut self, c: Const<'db>) -> Const<'db> {
127        match c.kind() {
128            rustc_type_ir::ConstKind::Bound(debruijn, var) if self.debruijn == debruijn => {
129                let GenericParamId::ConstParamId(id) = self.params[var.var.as_usize()] else {
130                    unreachable!()
131                };
132                Const::new(
133                    self.cx(),
134                    rustc_type_ir::ConstKind::Param(ParamConst { index: var.var.as_u32(), id }),
135                )
136            }
137            _ => c.super_fold_with(self),
138        }
139    }
140}

146impl<'db> ChalkToNextSolver<'db, Ty<'db>> for chalk_ir::Ty<Interner> {
147    fn to_nextsolver(&self, interner: DbInterner<'db>) -> Ty<'db> {
148        Ty::new(
149            interner,
150            match self.kind(Interner) {
151                chalk_ir::TyKind::Adt(adt_id, substitution) => {
152                    let def = AdtDef::new(adt_id.0, interner);
153                    let args = substitution.to_nextsolver(interner);
154                    rustc_type_ir::TyKind::Adt(def, args)
155                }
156                chalk_ir::TyKind::AssociatedType(assoc_type_id, substitution) => {
157                    let def_id = SolverDefId::TypeAliasId(from_assoc_type_id(*assoc_type_id));
158                    let args: GenericArgs<'db> = substitution.to_nextsolver(interner);
159                    let alias_ty = rustc_type_ir::AliasTy::new(interner, def_id, args.iter());
160                    rustc_type_ir::TyKind::Alias(rustc_type_ir::AliasTyKind::Projection, alias_ty)
161                }
162                chalk_ir::TyKind::Scalar(scalar) => match scalar {
163                    chalk_ir::Scalar::Bool => rustc_type_ir::TyKind::Bool,
164                    chalk_ir::Scalar::Char => rustc_type_ir::TyKind::Char,
165                    chalk_ir::Scalar::Int(chalk_ir::IntTy::Isize) => {
166                        rustc_type_ir::TyKind::Int(rustc_type_ir::IntTy::Isize)
167                    }
168                    chalk_ir::Scalar::Int(chalk_ir::IntTy::I8) => {
169                        rustc_type_ir::TyKind::Int(rustc_type_ir::IntTy::I8)
170                    }
171                    chalk_ir::Scalar::Int(chalk_ir::IntTy::I16) => {
172                        rustc_type_ir::TyKind::Int(rustc_type_ir::IntTy::I16)
173                    }
174                    chalk_ir::Scalar::Int(chalk_ir::IntTy::I32) => {
175                        rustc_type_ir::TyKind::Int(rustc_type_ir::IntTy::I32)
176                    }
177                    chalk_ir::Scalar::Int(chalk_ir::IntTy::I64) => {
178                        rustc_type_ir::TyKind::Int(rustc_type_ir::IntTy::I64)
179                    }
180                    chalk_ir::Scalar::Int(chalk_ir::IntTy::I128) => {
181                        rustc_type_ir::TyKind::Int(rustc_type_ir::IntTy::I128)
182                    }
183                    chalk_ir::Scalar::Uint(chalk_ir::UintTy::Usize) => {
184                        rustc_type_ir::TyKind::Uint(rustc_type_ir::UintTy::Usize)
185                    }
186                    chalk_ir::Scalar::Uint(chalk_ir::UintTy::U8) => {
187                        rustc_type_ir::TyKind::Uint(rustc_type_ir::UintTy::U8)
188                    }
189                    chalk_ir::Scalar::Uint(chalk_ir::UintTy::U16) => {
190                        rustc_type_ir::TyKind::Uint(rustc_type_ir::UintTy::U16)
191                    }
192                    chalk_ir::Scalar::Uint(chalk_ir::UintTy::U32) => {
193                        rustc_type_ir::TyKind::Uint(rustc_type_ir::UintTy::U32)
194                    }
195                    chalk_ir::Scalar::Uint(chalk_ir::UintTy::U64) => {
196                        rustc_type_ir::TyKind::Uint(rustc_type_ir::UintTy::U64)
197                    }
198                    chalk_ir::Scalar::Uint(chalk_ir::UintTy::U128) => {
199                        rustc_type_ir::TyKind::Uint(rustc_type_ir::UintTy::U128)
200                    }
201                    chalk_ir::Scalar::Float(chalk_ir::FloatTy::F16) => {
202                        rustc_type_ir::TyKind::Float(rustc_type_ir::FloatTy::F16)
203                    }
204                    chalk_ir::Scalar::Float(chalk_ir::FloatTy::F32) => {
205                        rustc_type_ir::TyKind::Float(rustc_type_ir::FloatTy::F32)
206                    }
207                    chalk_ir::Scalar::Float(chalk_ir::FloatTy::F64) => {
208                        rustc_type_ir::TyKind::Float(rustc_type_ir::FloatTy::F64)
209                    }
210                    chalk_ir::Scalar::Float(chalk_ir::FloatTy::F128) => {
211                        rustc_type_ir::TyKind::Float(rustc_type_ir::FloatTy::F128)
212                    }
213                },
214                chalk_ir::TyKind::Tuple(_, substitution) => {
215                    let args = substitution.to_nextsolver(interner);
216                    rustc_type_ir::TyKind::Tuple(args)
217                }
218                chalk_ir::TyKind::Array(ty, len) => rustc_type_ir::TyKind::Array(
219                    ty.to_nextsolver(interner),
220                    len.to_nextsolver(interner),
221                ),
222                chalk_ir::TyKind::Slice(ty) => {
223                    rustc_type_ir::TyKind::Slice(ty.to_nextsolver(interner))
224                }
225                chalk_ir::TyKind::Raw(mutability, ty) => rustc_type_ir::RawPtr(
226                    ty.to_nextsolver(interner),
227                    mutability.to_nextsolver(interner),
228                ),
229                chalk_ir::TyKind::Ref(mutability, lifetime, ty) => rustc_type_ir::TyKind::Ref(
230                    lifetime.to_nextsolver(interner),
231                    ty.to_nextsolver(interner),
232                    mutability.to_nextsolver(interner),
233                ),
234                chalk_ir::TyKind::OpaqueType(def_id, substitution) => {
235                    let id: InternedOpaqueTyId = (*def_id).into();
236                    let args: GenericArgs<'db> = substitution.to_nextsolver(interner);
237                    let alias_ty = rustc_type_ir::AliasTy::new(interner, id.into(), args);
238                    rustc_type_ir::TyKind::Alias(rustc_type_ir::AliasTyKind::Opaque, alias_ty)
239                }
240                chalk_ir::TyKind::FnDef(fn_def_id, substitution) => {
241                    let def_id = CallableDefId::from_chalk(interner.db(), *fn_def_id);
242                    let id: SolverDefId = match def_id {
243                        CallableDefId::FunctionId(id) => id.into(),
244                        CallableDefId::StructId(id) => SolverDefId::Ctor(Ctor::Struct(id)),
245                        CallableDefId::EnumVariantId(id) => SolverDefId::Ctor(Ctor::Enum(id)),
246                    };
247                    rustc_type_ir::TyKind::FnDef(id, substitution.to_nextsolver(interner))
248                }
249                chalk_ir::TyKind::Str => rustc_type_ir::TyKind::Str,
250                chalk_ir::TyKind::Never => rustc_type_ir::TyKind::Never,
251                chalk_ir::TyKind::Closure(closure_id, substitution) => {
252                    let id: InternedClosureId = (*closure_id).into();
253                    rustc_type_ir::TyKind::Closure(id.into(), substitution.to_nextsolver(interner))
254                }
255                chalk_ir::TyKind::Coroutine(coroutine_id, substitution) => {
256                    let id: InternedCoroutineId = (*coroutine_id).into();
257                    rustc_type_ir::TyKind::Coroutine(
258                        id.into(),
259                        substitution.to_nextsolver(interner),
260                    )
261                }
262                chalk_ir::TyKind::CoroutineWitness(coroutine_id, substitution) => {
263                    let id: InternedCoroutineId = (*coroutine_id).into();
264                    rustc_type_ir::TyKind::CoroutineWitness(
265                        id.into(),
266                        substitution.to_nextsolver(interner),
267                    )
268                }
269                chalk_ir::TyKind::Foreign(foreign_def_id) => rustc_type_ir::TyKind::Foreign(
270                    SolverDefId::TypeAliasId(crate::from_foreign_def_id(*foreign_def_id)),
271                ),
272                chalk_ir::TyKind::Error => rustc_type_ir::TyKind::Error(ErrorGuaranteed),
273                chalk_ir::TyKind::Dyn(dyn_ty) => {
274                    // exists<type> { for<...> ^1.0: ... }
275                    let bounds = BoundExistentialPredicates::new_from_iter(
276                        interner,
277                        dyn_ty.bounds.skip_binders().iter(Interner).filter_map(|pred| {
278                            // for<...> ^1.0: ...
279                            let (val, binders) = pred.clone().into_value_and_skipped_binders();
280                            let bound_vars = binders.to_nextsolver(interner);
281                            let clause = match val {
282                                chalk_ir::WhereClause::Implemented(trait_ref) => {
283                                    let trait_id = from_chalk_trait_id(trait_ref.trait_id);
284                                    if interner
285                                        .db()
286                                        .trait_signature(trait_id)
287                                        .flags
288                                        .contains(TraitFlags::AUTO)
289                                    {
290                                        ExistentialPredicate::AutoTrait(trait_id.into())
291                                    } else {
292                                        let args = GenericArgs::new_from_iter(
293                                            interner,
294                                            trait_ref
295                                                .substitution
296                                                .iter(Interner)
297                                                .skip(1)
298                                                .map(|a| a.clone().shifted_out(Interner).unwrap())
299                                                .map(|a| a.to_nextsolver(interner)),
300                                        );
301                                        let trait_ref = ExistentialTraitRef::new_from_args(
302                                            interner, trait_id.into(), args,
303                                        );
304                                        ExistentialPredicate::Trait(trait_ref)
305                                    }
306                                }
307                                chalk_ir::WhereClause::AliasEq(alias_eq) => {
308                                    let (def_id, args) = match &alias_eq.alias {
309                                        chalk_ir::AliasTy::Projection(projection) => {
310                                            let id =
311                                                from_assoc_type_id(projection.associated_ty_id);
312                                            let def_id = SolverDefId::TypeAliasId(id);
313                                            let generics = interner.generics_of(def_id);
314                                            let parent_len = generics.parent_count;
315                                            let substs = projection.substitution.iter(Interner).skip(1);
316
317                                            let args = GenericArgs::new_from_iter(
318                                                interner,
319                                                substs
320                                                    .map(|a| {
321                                                        a.clone().shifted_out(Interner).unwrap()
322                                                    })
323                                                    .map(|a| a.to_nextsolver(interner)),
324                                            );
325                                            (def_id, args)
326                                        }
327                                        chalk_ir::AliasTy::Opaque(opaque_ty) => {
328                                            panic!("Invalid ExistentialPredicate (opaques can't be named).");
329                                        }
330                                    };
331                                    let term = alias_eq
332                                        .ty
333                                        .clone()
334                                        .shifted_out(Interner)
335                                        .unwrap()
336                                        .to_nextsolver(interner)
337                                        .into();
338                                    let projection = ExistentialProjection::new_from_args(
339                                        interner, def_id, args, term,
340                                    );
341                                    ExistentialPredicate::Projection(projection)
342                                }
343                                chalk_ir::WhereClause::LifetimeOutlives(lifetime_outlives) => {
344                                    return None;
345                                }
346                                chalk_ir::WhereClause::TypeOutlives(type_outlives) => return None,
347                            };
348
349                            Some(Binder::bind_with_vars(clause, bound_vars))
350                        }),
351                    );
352                    let region = dyn_ty.lifetime.to_nextsolver(interner);
353                    let kind = rustc_type_ir::DynKind::Dyn;
354                    rustc_type_ir::TyKind::Dynamic(bounds, region, kind)
355                }
356                chalk_ir::TyKind::Alias(alias_ty) => match alias_ty {
357                    chalk_ir::AliasTy::Projection(projection_ty) => {
358                        let def_id = SolverDefId::TypeAliasId(from_assoc_type_id(
359                            projection_ty.associated_ty_id,
360                        ));
361                        let alias_ty = rustc_type_ir::AliasTy::new_from_args(
362                            interner,
363                            def_id,
364                            projection_ty.substitution.to_nextsolver(interner),
365                        );
366                        rustc_type_ir::TyKind::Alias(
367                            rustc_type_ir::AliasTyKind::Projection,
368                            alias_ty,
369                        )
370                    }
371                    chalk_ir::AliasTy::Opaque(opaque_ty) => {
372                        let id: InternedOpaqueTyId = opaque_ty.opaque_ty_id.into();
373                        let def_id = SolverDefId::InternedOpaqueTyId(id);
374                        let alias_ty = rustc_type_ir::AliasTy::new_from_args(
375                            interner,
376                            def_id,
377                            opaque_ty.substitution.to_nextsolver(interner),
378                        );
379                        rustc_type_ir::TyKind::Alias(rustc_type_ir::AliasTyKind::Opaque, alias_ty)
380                    }
381                },
382                chalk_ir::TyKind::Function(fn_pointer) => {
383                    let sig_tys = fn_pointer.clone().into_binders(Interner).to_nextsolver(interner);
384                    let header = rustc_type_ir::FnHeader {
385                        abi: fn_pointer.sig.abi,
386                        c_variadic: fn_pointer.sig.variadic,
387                        safety: match fn_pointer.sig.safety {
388                            chalk_ir::Safety::Safe => super::abi::Safety::Safe,
389                            chalk_ir::Safety::Unsafe => super::abi::Safety::Unsafe,
390                        },
391                    };
392
393                    rustc_type_ir::TyKind::FnPtr(sig_tys, header)
394                }
395                // The schema here is quite confusing.
396                // The new solver, like rustc, uses `Param` and `EarlyBinder` for generic params. It uses `BoundVar`
397                // and `Placeholder` together with `Binder` for HRTB, which we mostly don't handle.
398                // Chalk uses `Placeholder` for generic params and `BoundVar` quite liberally, and this is quite a
399                // problem. `chalk_ir::TyKind::BoundVar` can represent either HRTB or generic params, depending on the
400                // context. When returned from signature queries, the outer `Binders` represent the generic params.
401                // But there are also inner `Binders` for HRTB.
402                // AFAIK there is no way to tell which of the meanings is relevant, so we just use `rustc_type_ir::Bound`
403                // here, and hope for the best. If you are working with new solver types, therefore, use the new solver
404                // lower queries.
405                // Hopefully sooner than later Chalk will be ripped from the codebase and we can avoid that problem.
406                // For details about the rustc setup, read: https://rustc-dev-guide.rust-lang.org/generic_parameters_summary.html
407                // and the following chapters.
408                chalk_ir::TyKind::Placeholder(placeholder_index) => {
409                    let (id, index) = from_placeholder_idx(interner.db, *placeholder_index);
410                    rustc_type_ir::TyKind::Param(ParamTy {
411                        id: TypeParamId::from_unchecked(id),
412                        index,
413                    })
414                }
415                chalk_ir::TyKind::BoundVar(bound_var) => rustc_type_ir::TyKind::Bound(
416                    bound_var.debruijn.to_nextsolver(interner),
417                    BoundTy {
418                        var: rustc_type_ir::BoundVar::from_usize(bound_var.index),
419                        kind: BoundTyKind::Anon,
420                    },
421                ),
422                chalk_ir::TyKind::InferenceVar(inference_var, ty_variable_kind) => {
423                    rustc_type_ir::TyKind::Infer(
424                        (*inference_var, *ty_variable_kind).to_nextsolver(interner),
425                    )
426                }
427            },
428        )
429    }
430}

432impl<'db> ChalkToNextSolver<'db, Region<'db>> for chalk_ir::Lifetime<Interner> {
433    fn to_nextsolver(&self, interner: DbInterner<'db>) -> Region<'db> {
434        Region::new(
435            interner,
436            match self.data(Interner) {
437                chalk_ir::LifetimeData::BoundVar(bound_var) => rustc_type_ir::RegionKind::ReBound(
438                    bound_var.debruijn.to_nextsolver(interner),
439                    BoundRegion {
440                        var: rustc_type_ir::BoundVar::from_u32(bound_var.index as u32),
441                        kind: BoundRegionKind::Anon,
442                    },
443                ),
444                chalk_ir::LifetimeData::InferenceVar(inference_var) => {
445                    rustc_type_ir::RegionKind::ReVar(rustc_type_ir::RegionVid::from_u32(
446                        inference_var.index(),
447                    ))
448                }
449                chalk_ir::LifetimeData::Placeholder(placeholder_index) => {
450                    let (id, index) = lt_from_placeholder_idx(interner.db, *placeholder_index);
451                    rustc_type_ir::RegionKind::ReEarlyParam(EarlyParamRegion { id, index })
452                }
453                chalk_ir::LifetimeData::Static => rustc_type_ir::RegionKind::ReStatic,
454                chalk_ir::LifetimeData::Erased => rustc_type_ir::RegionKind::ReErased,
455                chalk_ir::LifetimeData::Phantom(_, _) => {
456                    unreachable!()
457                }
458                chalk_ir::LifetimeData::Error => {
459                    rustc_type_ir::RegionKind::ReError(ErrorGuaranteed)
460                }
461            },
462        )
463    }
464}

466impl<'db> ChalkToNextSolver<'db, Const<'db>> for chalk_ir::Const<Interner> {
467    fn to_nextsolver(&self, interner: DbInterner<'db>) -> Const<'db> {
468        let data = self.data(Interner);
469        Const::new(
470            interner,
471            match &data.value {
472                chalk_ir::ConstValue::BoundVar(bound_var) => rustc_type_ir::ConstKind::Bound(
473                    bound_var.debruijn.to_nextsolver(interner),
474                    BoundConst { var: rustc_type_ir::BoundVar::from_usize(bound_var.index) },
475                ),
476                chalk_ir::ConstValue::InferenceVar(inference_var) => {
477                    rustc_type_ir::ConstKind::Infer(rustc_type_ir::InferConst::Var(
478                        rustc_type_ir::ConstVid::from_u32(inference_var.index()),
479                    ))
480                }
481                chalk_ir::ConstValue::Placeholder(placeholder_index) => {
482                    let (id, index) = from_placeholder_idx(interner.db, *placeholder_index);
483                    rustc_type_ir::ConstKind::Param(ParamConst {
484                        id: ConstParamId::from_unchecked(id),
485                        index,
486                    })
487                }
488                chalk_ir::ConstValue::Concrete(concrete_const) => match &concrete_const.interned {
489                    ConstScalar::Bytes(bytes, memory) => {
490                        rustc_type_ir::ConstKind::Value(ValueConst::new(
491                            data.ty.to_nextsolver(interner),
492                            ConstBytes(bytes.clone(), memory.clone()),
493                        ))
494                    }
495                    ConstScalar::UnevaluatedConst(c, subst) => {
496                        let def = match *c {
497                            GeneralConstId::ConstId(id) => SolverDefId::ConstId(id),
498                            GeneralConstId::StaticId(id) => SolverDefId::StaticId(id),
499                        };
500                        let args = subst.to_nextsolver(interner);

681impl<'db> ChalkToNextSolver<'db, Predicate<'db>> for chalk_ir::Goal<Interner> {
682    fn to_nextsolver(&self, interner: DbInterner<'db>) -> Predicate<'db> {
683        match self.data(Interner) {
684            chalk_ir::GoalData::Quantified(quantifier_kind, binders) => {
685                if !binders.binders.is_empty(Interner) {
686                    panic!("Should not be constructed.");
687                }
688                let (val, _) = binders.clone().into_value_and_skipped_binders();
689                val.shifted_out(Interner).unwrap().to_nextsolver(interner)
690            }
691            chalk_ir::GoalData::Implies(program_clauses, goal) => {
692                panic!("Should not be constructed.")
693            }
694            chalk_ir::GoalData::All(goals) => panic!("Should not be constructed."),
695            chalk_ir::GoalData::Not(goal) => panic!("Should not be constructed."),
696            chalk_ir::GoalData::EqGoal(eq_goal) => panic!("Should not be constructed."),
697            chalk_ir::GoalData::SubtypeGoal(subtype_goal) => {
698                let subtype_predicate = SubtypePredicate {
699                    a: subtype_goal.a.to_nextsolver(interner),
700                    b: subtype_goal.b.to_nextsolver(interner),
701                    a_is_expected: true,
702                };
703                let pred_kind = PredicateKind::Subtype(subtype_predicate);
704                let pred_kind = Binder::bind_with_vars(
705                    shift_vars(interner, pred_kind, 1),
706                    BoundVarKinds::new_from_iter(interner, []),
707                );
708                Predicate::new(interner, pred_kind)
709            }
710            chalk_ir::GoalData::DomainGoal(domain_goal) => {
711                let pred_kind = domain_goal.to_nextsolver(interner);
712                let pred_kind = Binder::bind_with_vars(
713                    shift_vars(interner, pred_kind, 1),
714                    BoundVarKinds::new_from_iter(interner, []),
715                );
716                Predicate::new(interner, pred_kind)
717            }
718            chalk_ir::GoalData::CannotProve => panic!("Should not be constructed."),
719        }
720    }
721}

751impl<'db> ChalkToNextSolver<'db, PredicateKind<'db>> for chalk_ir::DomainGoal<Interner> {
752    fn to_nextsolver(&self, interner: DbInterner<'db>) -> PredicateKind<'db> {
753        match self {
754            chalk_ir::DomainGoal::Holds(where_clause) => match where_clause {
755                chalk_ir::WhereClause::Implemented(trait_ref) => {
756                    let predicate = TraitPredicate {
757                        trait_ref: trait_ref.to_nextsolver(interner),
758                        polarity: rustc_type_ir::PredicatePolarity::Positive,
759                    };
760                    PredicateKind::Clause(ClauseKind::Trait(predicate))
761                }
762                chalk_ir::WhereClause::AliasEq(alias_eq) => match &alias_eq.alias {
763                    chalk_ir::AliasTy::Projection(p) => {
764                        let def_id =
765                            SolverDefId::TypeAliasId(from_assoc_type_id(p.associated_ty_id));
766                        let args = p.substitution.to_nextsolver(interner);
767                        let term: Ty<'db> = alias_eq.ty.to_nextsolver(interner);
768                        let term: Term<'db> = term.into();
769                        let predicate = ProjectionPredicate {
770                            projection_term: AliasTerm::new_from_args(interner, def_id, args),
771                            term,
772                        };
773                        PredicateKind::Clause(ClauseKind::Projection(predicate))
774                    }
775                    chalk_ir::AliasTy::Opaque(opaque) => {
776                        let id: InternedOpaqueTyId = opaque.opaque_ty_id.into();
777                        let def_id = SolverDefId::InternedOpaqueTyId(id);
778                        let args = opaque.substitution.to_nextsolver(interner);
779                        let term: Ty<'db> = alias_eq.ty.to_nextsolver(interner);
780                        let term: Term<'db> = term.into();
781                        let opaque_ty = Ty::new(
782                            interner,
783                            rustc_type_ir::TyKind::Alias(
784                                rustc_type_ir::AliasTyKind::Opaque,
785                                rustc_type_ir::AliasTy::new_from_args(interner, def_id, args),
786                            ),
787                        )
788                        .into();
789                        PredicateKind::AliasRelate(
790                            opaque_ty,
791                            term,
792                            rustc_type_ir::AliasRelationDirection::Equate,
793                        )
794                    }
795                },
796                chalk_ir::WhereClause::LifetimeOutlives(lifetime_outlives) => {
797                    let predicate = OutlivesPredicate(
798                        lifetime_outlives.a.to_nextsolver(interner),
799                        lifetime_outlives.b.to_nextsolver(interner),
800                    );
801                    PredicateKind::Clause(ClauseKind::RegionOutlives(predicate))
802                }
803                chalk_ir::WhereClause::TypeOutlives(type_outlives) => {
804                    let predicate = OutlivesPredicate(
805                        type_outlives.ty.to_nextsolver(interner),
806                        type_outlives.lifetime.to_nextsolver(interner),
807                    );
808                    PredicateKind::Clause(ClauseKind::TypeOutlives(predicate))
809                }
810            },
811            chalk_ir::DomainGoal::Normalize(normalize) => {
812                let proj_ty = match &normalize.alias {
813                    chalk_ir::AliasTy::Projection(proj) => proj,
814                    _ => unimplemented!(),
815                };
816                let args: GenericArgs<'db> = proj_ty.substitution.to_nextsolver(interner);
817                let alias = Ty::new(
818                    interner,
819                    rustc_type_ir::TyKind::Alias(
820                        rustc_type_ir::AliasTyKind::Projection,
821                        rustc_type_ir::AliasTy::new(
822                            interner,
823                            from_assoc_type_id(proj_ty.associated_ty_id).into(),
824                            args,
825                        ),
826                    ),
827                )
828                .into();
829                let term = normalize.ty.to_nextsolver(interner).into();
830                PredicateKind::AliasRelate(
831                    alias,
832                    term,
833                    rustc_type_ir::AliasRelationDirection::Equate,
834                )
835            }
836            chalk_ir::DomainGoal::WellFormed(well_formed) => {
837                let term = match well_formed {
838                    WellFormed::Trait(_) => panic!("Should not be constructed."),
839                    WellFormed::Ty(ty) => Term::Ty(ty.to_nextsolver(interner)),
840                };
841                PredicateKind::Clause(rustc_type_ir::ClauseKind::WellFormed(term))
842            }
843            chalk_ir::DomainGoal::FromEnv(from_env) => match from_env {
844                chalk_ir::FromEnv::Trait(trait_ref) => {
845                    let predicate = TraitPredicate {
846                        trait_ref: trait_ref.to_nextsolver(interner),
847                        polarity: rustc_type_ir::PredicatePolarity::Positive,
848                    };
849                    PredicateKind::Clause(ClauseKind::Trait(predicate))
850                }
851                chalk_ir::FromEnv::Ty(ty) => PredicateKind::Clause(ClauseKind::WellFormed(
852                    Term::Ty(ty.to_nextsolver(interner)),
853                )),
854            },
855            chalk_ir::DomainGoal::IsLocal(ty) => panic!("Should not be constructed."),
856            chalk_ir::DomainGoal::IsUpstream(ty) => panic!("Should not be constructed."),
857            chalk_ir::DomainGoal::IsFullyVisible(ty) => panic!("Should not be constructed."),
858            chalk_ir::DomainGoal::LocalImplAllowed(trait_ref) => {
859                panic!("Should not be constructed.")
860            }
861            chalk_ir::DomainGoal::Compatible => panic!("Should not be constructed."),
862            chalk_ir::DomainGoal::DownstreamType(ty) => panic!("Should not be constructed."),
863            chalk_ir::DomainGoal::Reveal => panic!("Should not be constructed."),
864            chalk_ir::DomainGoal::ObjectSafe(trait_id) => panic!("Should not be constructed."),
865        }
866    }
867}

964pub fn convert_args_for_result<'db>(
965    interner: DbInterner<'db>,
966    args: &[GenericArg<'db>],
967) -> crate::Substitution {
968    let mut substs = Vec::with_capacity(args.len());
969    for arg in args {
970        match (*arg).kind() {
971            rustc_type_ir::GenericArgKind::Type(ty) => {
972                let ty = convert_ty_for_result(interner, ty);
973                substs.push(chalk_ir::GenericArgData::Ty(ty).intern(Interner));
974            }
975            rustc_type_ir::GenericArgKind::Lifetime(region) => {
976                let lifetime = convert_region_for_result(interner, region);
977                substs.push(chalk_ir::GenericArgData::Lifetime(lifetime).intern(Interner));
978            }
979            rustc_type_ir::GenericArgKind::Const(const_) => {
980                substs.push(
981                    chalk_ir::GenericArgData::Const(convert_const_for_result(interner, const_))
982                        .intern(Interner),
983                );
984            }
985        }
986    }
987    Substitution::from_iter(Interner, substs)
988}

990pub(crate) fn convert_ty_for_result<'db>(interner: DbInterner<'db>, ty: Ty<'db>) -> crate::Ty {
991    use crate::{Scalar, TyKind};
992    use chalk_ir::{FloatTy, IntTy, UintTy};
993    match ty.kind() {
994        rustc_type_ir::TyKind::Bool => TyKind::Scalar(Scalar::Bool),
995        rustc_type_ir::TyKind::Char => TyKind::Scalar(Scalar::Char),
996        rustc_type_ir::TyKind::Int(rustc_type_ir::IntTy::I8) => {
997            TyKind::Scalar(Scalar::Int(IntTy::I8))
998        }
999        rustc_type_ir::TyKind::Int(rustc_type_ir::IntTy::I16) => {
1000            TyKind::Scalar(Scalar::Int(IntTy::I16))
1001        }
1002        rustc_type_ir::TyKind::Int(rustc_type_ir::IntTy::I32) => {

1007        }

1009            TyKind::Scalar(Scalar::Int(IntTy::I128))
1010        }
1011        rustc_type_ir::TyKind::Int(rustc_type_ir::IntTy::Isize) => {
1012            TyKind::Scalar(Scalar::Int(IntTy::Isize))
1013        }

1015            TyKind::Scalar(Scalar::Uint(UintTy::U8))
1016        }
1017        rustc_type_ir::TyKind::Uint(rustc_type_ir::UintTy::U16) => {
1018            TyKind::Scalar(Scalar::Uint(UintTy::U16))
1019        }
1020        rustc_type_ir::TyKind::Uint(rustc_type_ir::UintTy::U32) => {

1022        }

1027            TyKind::Scalar(Scalar::Uint(UintTy::U128))
1028        }
1029        rustc_type_ir::TyKind::Uint(rustc_type_ir::UintTy::Usize) => {

1031        }

1033            TyKind::Scalar(Scalar::Float(FloatTy::F16))
1034        }
1035        rustc_type_ir::TyKind::Float(rustc_type_ir::FloatTy::F32) => {
1036            TyKind::Scalar(Scalar::Float(FloatTy::F32))
1037        }
1038        rustc_type_ir::TyKind::Float(rustc_type_ir::FloatTy::F64) => {
1039            TyKind::Scalar(Scalar::Float(FloatTy::F64))
1040        }
1041        rustc_type_ir::TyKind::Float(rustc_type_ir::FloatTy::F128) => {

1043        }
1044        rustc_type_ir::TyKind::Str => TyKind::Str,
1045        rustc_type_ir::TyKind::Error(_) => TyKind::Error,
1046        rustc_type_ir::TyKind::Never => TyKind::Never,
1047
1048        rustc_type_ir::TyKind::Adt(def, args) => {
1049            let adt_id = def.inner().id;
1050            let subst = convert_args_for_result(interner, args.as_slice());

1052        }

1054        rustc_type_ir::TyKind::Infer(infer_ty) => {
1055            let (var, kind) = match infer_ty {
1056                rustc_type_ir::InferTy::TyVar(var) => {
1057                    (InferenceVar::from(var.as_u32()), TyVariableKind::General)
1058                }
1059                rustc_type_ir::InferTy::IntVar(var) => {
1060                    (InferenceVar::from(var.as_u32()), TyVariableKind::Integer)
1061                }
1062                rustc_type_ir::InferTy::FloatVar(var) => {
1063                    (InferenceVar::from(var.as_u32()), TyVariableKind::Float)
1064                }
1065                rustc_type_ir::InferTy::FreshFloatTy(..)
1066                | rustc_type_ir::InferTy::FreshIntTy(..)
1067                | rustc_type_ir::InferTy::FreshTy(..) => {
1068                    panic!("Freshening shouldn't happen.")
1069                }
1070            };
1071            TyKind::InferenceVar(var, kind)
1072        }

1074        rustc_type_ir::TyKind::Ref(r, ty, mutability) => {
1075            let mutability = match mutability {
1076                rustc_ast_ir::Mutability::Mut => chalk_ir::Mutability::Mut,
1077                rustc_ast_ir::Mutability::Not => chalk_ir::Mutability::Not,
1078            };
1079            let r = convert_region_for_result(interner, r);
1080            let ty = convert_ty_for_result(interner, ty);
1081            TyKind::Ref(mutability, r, ty)
1082        }
1083
1084        rustc_type_ir::TyKind::Tuple(tys) => {
1085            let size = tys.len();
1086            let subst = Substitution::from_iter(
1087                Interner,
1088                tys.iter().map(|ty| {
1089                    chalk_ir::GenericArgData::Ty(convert_ty_for_result(interner, ty))
1090                        .intern(Interner)
1091                }),
1092            );
1093            TyKind::Tuple(size, subst)
1094        }
1095
1096        rustc_type_ir::TyKind::Array(ty, const_) => {
1097            let ty = convert_ty_for_result(interner, ty);
1098            let const_ = convert_const_for_result(interner, const_);
1099            TyKind::Array(ty, const_)
1100        }
1101
1102        rustc_type_ir::TyKind::Alias(alias_ty_kind, alias_ty) => match alias_ty_kind {
1103            rustc_type_ir::AliasTyKind::Projection => {
1104                let assoc_ty_id = match alias_ty.def_id {
1105                    SolverDefId::TypeAliasId(id) => id,
1106                    _ => unreachable!(),
1107                };
1108                let associated_ty_id = to_assoc_type_id(assoc_ty_id);
1109                let substitution = convert_args_for_result(interner, alias_ty.args.as_slice());
1110                TyKind::AssociatedType(associated_ty_id, substitution)
1111            }
1112            rustc_type_ir::AliasTyKind::Opaque => {
1113                let opaque_ty_id = match alias_ty.def_id {
1114                    SolverDefId::InternedOpaqueTyId(id) => id,
1115                    _ => unreachable!(),
1116                };
1117                let substitution = convert_args_for_result(interner, alias_ty.args.as_slice());
1118                TyKind::Alias(chalk_ir::AliasTy::Opaque(chalk_ir::OpaqueTy {
1119                    opaque_ty_id: opaque_ty_id.into(),
1120                    substitution,
1121                }))
1122            }
1123            rustc_type_ir::AliasTyKind::Inherent => unimplemented!(),
1124            rustc_type_ir::AliasTyKind::Free => unimplemented!(),
1125        },
1126
1127        // For `Placeholder`, `Bound` and `Param`, see the comment on the reverse conversion.
1128        rustc_type_ir::TyKind::Placeholder(placeholder) => {
1129            unimplemented!(
1130                "A `rustc_type_ir::TyKind::Placeholder` doesn't have a direct \
1131                correspondence in Chalk, as it represents a universally instantiated `Bound`.\n\
1132                It therefore feels safer to leave it panicking, but if you hit this panic \
1133                feel free to do the same as in `rustc_type_ir::TyKind::Bound` here."
1134            )
1135        }
1136        rustc_type_ir::TyKind::Bound(debruijn_index, ty) => TyKind::BoundVar(chalk_ir::BoundVar {
1137            debruijn: chalk_ir::DebruijnIndex::new(debruijn_index.as_u32()),
1138            index: ty.var.as_usize(),
1139        }),
1140        rustc_type_ir::TyKind::Param(param) => {
1141            let placeholder = to_placeholder_idx(interner.db, param.id.into(), param.index);
1142            TyKind::Placeholder(placeholder)
1143        }
1144
1145        rustc_type_ir::TyKind::FnPtr(bound_sig, fn_header) => {
1146            let num_binders = bound_sig.bound_vars().len();
1147            let sig = chalk_ir::FnSig {
1148                abi: fn_header.abi,
1149                safety: match fn_header.safety {
1150                    crate::next_solver::abi::Safety::Safe => chalk_ir::Safety::Safe,
1151                    crate::next_solver::abi::Safety::Unsafe => chalk_ir::Safety::Unsafe,
1152                },
1153                variadic: fn_header.c_variadic,
1154            };
1155            let args = GenericArgs::new_from_iter(
1156                interner,
1157                bound_sig.skip_binder().inputs_and_output.iter().map(|a| a.into()),
1158            );
1159            let substitution = convert_args_for_result(interner, args.as_slice());
1160            let substitution = chalk_ir::FnSubst(substitution);
1161            let fnptr = chalk_ir::FnPointer { num_binders, sig, substitution };
1162            TyKind::Function(fnptr)
1163        }
1164
1165        rustc_type_ir::TyKind::Dynamic(preds, region, dyn_kind) => {
1166            assert!(matches!(dyn_kind, rustc_type_ir::DynKind::Dyn));
1167            let self_ty = Ty::new_bound(
1168                interner,
1169                DebruijnIndex::from_u32(1),
1170                BoundTy { kind: BoundTyKind::Anon, var: BoundVar::from_u32(0) },
1171            );
1172            let bounds = chalk_ir::QuantifiedWhereClauses::from_iter(
1173                Interner,
1174                preds.iter().map(|p| {
1175                    let binders = chalk_ir::VariableKinds::from_iter(
1176                        Interner,
1177                        p.bound_vars().iter().map(|b| match b {
1178                            BoundVarKind::Ty(kind) => {
1179                                chalk_ir::VariableKind::Ty(TyVariableKind::General)
1180                            }
1181                            BoundVarKind::Region(kind) => chalk_ir::VariableKind::Lifetime,
1182                            BoundVarKind::Const => {
1183                                chalk_ir::VariableKind::Const(crate::TyKind::Error.intern(Interner))
1184                            }
1185                        }),
1186                    );
1187
1188                    // Rust and chalk have slightly different
1189                    // representation for trait objects.
1190                    //
1191                    // Chalk uses `for<T0> for<'a> T0: Trait<'a>` while rustc
1192                    // uses `ExistentialPredicate`s, which do not have a self ty.
1193                    // We need to shift escaping bound vars by 1 to accommodate
1194                    // the newly introduced `for<T0>` binder.
1195                    let p = shift_vars(interner, p, 1);
1196
1197                    let where_clause = match p.skip_binder() {
1198                        rustc_type_ir::ExistentialPredicate::Trait(trait_ref) => {
1199                            let trait_ref = TraitRef::new(
1200                                interner,
1201                                trait_ref.def_id,
1202                                [self_ty.into()].into_iter().chain(trait_ref.args.iter()),
1203                            );
1204                            let trait_id = to_chalk_trait_id(trait_ref.def_id.0);
1205                            let substitution =
1206                                convert_args_for_result(interner, trait_ref.args.as_slice());
1207                            let trait_ref = chalk_ir::TraitRef { trait_id, substitution };
1208                            chalk_ir::WhereClause::Implemented(trait_ref)
1209                        }
1210                        rustc_type_ir::ExistentialPredicate::AutoTrait(trait_) => {
1211                            let trait_id = to_chalk_trait_id(trait_.0);
1212                            let substitution = chalk_ir::Substitution::from1(
1213                                Interner,
1214                                convert_ty_for_result(interner, self_ty),
1215                            );
1216                            let trait_ref = chalk_ir::TraitRef { trait_id, substitution };
1217                            chalk_ir::WhereClause::Implemented(trait_ref)
1218                        }
1219                        rustc_type_ir::ExistentialPredicate::Projection(existential_projection) => {
1220                            let projection = ProjectionPredicate {
1221                                projection_term: AliasTerm::new(
1222                                    interner,
1223                                    existential_projection.def_id,
1224                                    [self_ty.into()]
1225                                        .iter()
1226                                        .chain(existential_projection.args.iter()),
1227                                ),
1228                                term: existential_projection.term,
1229                            };
1230                            let associated_ty_id = match projection.projection_term.def_id {
1231                                SolverDefId::TypeAliasId(id) => to_assoc_type_id(id),
1232                                _ => unreachable!(),
1233                            };
1234                            let substitution = convert_args_for_result(
1235                                interner,
1236                                projection.projection_term.args.as_slice(),
1237                            );
1238                            let alias = chalk_ir::AliasTy::Projection(chalk_ir::ProjectionTy {
1239                                associated_ty_id,
1240                                substitution,
1241                            });
1242                            let ty = match projection.term {
1243                                Term::Ty(ty) => ty,
1244                                _ => unreachable!(),
1245                            };
1246                            let ty = convert_ty_for_result(interner, ty);
1247                            let alias_eq = chalk_ir::AliasEq { alias, ty };
1248                            chalk_ir::WhereClause::AliasEq(alias_eq)
1249                        }
1250                    };
1251                    chalk_ir::Binders::new(binders, where_clause)
1252                }),
1253            );
1254            let binders = chalk_ir::VariableKinds::from1(
1255                Interner,
1256                chalk_ir::VariableKind::Ty(chalk_ir::TyVariableKind::General),
1257            );
1258            let bounds = chalk_ir::Binders::new(binders, bounds);
1259            let dyn_ty =
1260                chalk_ir::DynTy { bounds, lifetime: convert_region_for_result(interner, region) };
1261            TyKind::Dyn(dyn_ty)
1262        }
1263
1264        rustc_type_ir::TyKind::Slice(ty) => {
1265            let ty = convert_ty_for_result(interner, ty);
1266            TyKind::Slice(ty)
1267        }
1268
1269        rustc_type_ir::TyKind::Foreign(foreign) => {
1270            let def_id = match foreign {
1271                SolverDefId::TypeAliasId(id) => id,
1272                _ => unreachable!(),
1273            };
1274            TyKind::Foreign(to_foreign_def_id(def_id))
1275        }
1276        rustc_type_ir::TyKind::Pat(_, _) => unimplemented!(),
1277        rustc_type_ir::TyKind::RawPtr(ty, mutability) => {
1278            let mutability = match mutability {
1279                rustc_ast_ir::Mutability::Mut => chalk_ir::Mutability::Mut,
1280                rustc_ast_ir::Mutability::Not => chalk_ir::Mutability::Not,
1281            };
1282            let ty = convert_ty_for_result(interner, ty);
1283            TyKind::Raw(mutability, ty)
1284        }
1285        rustc_type_ir::TyKind::FnDef(def_id, args) => {
1286            let id = match def_id {
1287                SolverDefId::FunctionId(id) => CallableDefId::FunctionId(id),
1288                SolverDefId::Ctor(Ctor::Struct(id)) => CallableDefId::StructId(id),
1289                SolverDefId::Ctor(Ctor::Enum(id)) => CallableDefId::EnumVariantId(id),
1290                _ => unreachable!(),
1291            };
1292            let subst = convert_args_for_result(interner, args.as_slice());
1293            TyKind::FnDef(id.to_chalk(interner.db()), subst)
1294        }
1295
1296        rustc_type_ir::TyKind::Closure(def_id, args) => {
1297            let id = match def_id {
1298                SolverDefId::InternedClosureId(id) => id,
1299                _ => unreachable!(),
1300            };
1301            let subst = convert_args_for_result(interner, args.as_slice());
1302            TyKind::Closure(id.into(), subst)
1303        }
1304        rustc_type_ir::TyKind::CoroutineClosure(_, _) => unimplemented!(),
1305        rustc_type_ir::TyKind::Coroutine(def_id, args) => {
1306            let id = match def_id {
1307                SolverDefId::InternedCoroutineId(id) => id,
1308                _ => unreachable!(),
1309            };
1310            let subst = convert_args_for_result(interner, args.as_slice());
1311            TyKind::Coroutine(id.into(), subst)
1312        }
1313        rustc_type_ir::TyKind::CoroutineWitness(def_id, args) => {
1314            let id = match def_id {
1315                SolverDefId::InternedCoroutineId(id) => id,
1316                _ => unreachable!(),
1317            };
1318            let subst = convert_args_for_result(interner, args.as_slice());
1319            TyKind::CoroutineWitness(id.into(), subst)
1320        }
1321
1322        rustc_type_ir::TyKind::UnsafeBinder(_) => unimplemented!(),
1323    }
1324    .intern(Interner)
1325}

1327pub fn convert_const_for_result<'db>(
1328    interner: DbInterner<'db>,
1329    const_: Const<'db>,
1330) -> crate::Const {
1331    let value: chalk_ir::ConstValue<Interner> = match const_.kind() {
1332        rustc_type_ir::ConstKind::Infer(rustc_type_ir::InferConst::Var(var)) => {
1333            chalk_ir::ConstValue::InferenceVar(chalk_ir::InferenceVar::from(var.as_u32()))
1334        }
1335        rustc_type_ir::ConstKind::Infer(rustc_type_ir::InferConst::Fresh(fresh)) => {
1336            panic!("Vars should not be freshened.")
1337        }
1338        rustc_type_ir::ConstKind::Param(param) => {
1339            let placeholder = to_placeholder_idx(interner.db, param.id.into(), param.index);
1340            chalk_ir::ConstValue::Placeholder(placeholder)
1341        }
1342        rustc_type_ir::ConstKind::Bound(debruijn_index, var) => {
1343            chalk_ir::ConstValue::BoundVar(chalk_ir::BoundVar::new(
1344                chalk_ir::DebruijnIndex::new(debruijn_index.as_u32()),
1345                var.var.index(),
1346            ))
1347        }
1348        rustc_type_ir::ConstKind::Placeholder(placeholder_const) => {
1349            unimplemented!(
1350                "A `rustc_type_ir::ConstKind::Placeholder` doesn't have a direct \
1351                correspondence in Chalk, as it represents a universally instantiated `Bound`.\n\
1352                It therefore feels safer to leave it panicking, but if you hit this panic \
1353                feel free to do the same as in `rustc_type_ir::ConstKind::Bound` here."
1354            )
1355        }
1356        rustc_type_ir::ConstKind::Unevaluated(unevaluated_const) => {
1357            let id = match unevaluated_const.def {
1358                SolverDefId::ConstId(id) => GeneralConstId::ConstId(id),
1359                SolverDefId::StaticId(id) => GeneralConstId::StaticId(id),
1360                _ => unreachable!(),
1361            };
1362            let subst = convert_args_for_result(interner, unevaluated_const.args.as_slice());
1363            chalk_ir::ConstValue::Concrete(chalk_ir::ConcreteConst {
1364                interned: ConstScalar::UnevaluatedConst(id, subst),
1365            })
1366        }
1367        rustc_type_ir::ConstKind::Value(value_const) => {
1368            let bytes = value_const.value.inner();
1369            let value = chalk_ir::ConstValue::Concrete(chalk_ir::ConcreteConst {
1370                // SAFETY: we will never actually use this without a database
1371                interned: ConstScalar::Bytes(bytes.0.clone(), unsafe {
1372                    std::mem::transmute::<MemoryMap<'db>, MemoryMap<'static>>(bytes.1.clone())
1373                }),
1374            });
1375            return chalk_ir::ConstData {
1376                ty: convert_ty_for_result(interner, value_const.ty),
1377                value,
1378            }
1379            .intern(Interner);
1380        }
1381        rustc_type_ir::ConstKind::Error(_) => {
1382            chalk_ir::ConstValue::Concrete(chalk_ir::ConcreteConst {
1383                interned: ConstScalar::Unknown,
1384            })
1385        }
1386        rustc_type_ir::ConstKind::Expr(_) => unimplemented!(),
1387    };
1388    chalk_ir::ConstData { ty: crate::TyKind::Error.intern(Interner), value }.intern(Interner)
1389}

1391pub fn convert_region_for_result<'db>(
1392    interner: DbInterner<'db>,
1393    region: Region<'db>,
1394) -> crate::Lifetime {
1395    let lifetime = match region.kind() {
1396        rustc_type_ir::RegionKind::ReEarlyParam(early) => {
1397            let placeholder = lt_to_placeholder_idx(interner.db, early.id, early.index);
1398            chalk_ir::LifetimeData::Placeholder(placeholder)
1399        }
1400        rustc_type_ir::RegionKind::ReBound(db, bound) => {
1401            chalk_ir::LifetimeData::BoundVar(chalk_ir::BoundVar::new(
1402                chalk_ir::DebruijnIndex::new(db.as_u32()),
1403                bound.var.as_usize(),
1404            ))
1405        }
1406        rustc_type_ir::RegionKind::RePlaceholder(placeholder) => unimplemented!(
1407            "A `rustc_type_ir::RegionKind::RePlaceholder` doesn't have a direct \
1408            correspondence in Chalk, as it represents a universally instantiated `Bound`.\n\
1409            It therefore feels safer to leave it panicking, but if you hit this panic \
1410            feel free to do the same as in `rustc_type_ir::RegionKind::ReBound` here."
1411        ),
1412        rustc_type_ir::RegionKind::ReLateParam(_) => unimplemented!(),
1413        rustc_type_ir::RegionKind::ReStatic => chalk_ir::LifetimeData::Static,
1414        rustc_type_ir::RegionKind::ReVar(vid) => {
1415            chalk_ir::LifetimeData::InferenceVar(chalk_ir::InferenceVar::from(vid.as_u32()))
1416        }
1417        rustc_type_ir::RegionKind::ReErased => chalk_ir::LifetimeData::Erased,
1418        rustc_type_ir::RegionKind::ReError(_) => chalk_ir::LifetimeData::Error,
1419    };
1420    chalk_ir::Lifetime::new(Interner, lifetime)
1421}
