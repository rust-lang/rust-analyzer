150impl<'db> Relate<DbInterner<'db>> for GenericArg<'db> {
151    fn relate<R: rustc_type_ir::relate::TypeRelation<DbInterner<'db>>>(
152        relation: &mut R,
153        a: Self,
154        b: Self,
155    ) -> rustc_type_ir::relate::RelateResult<DbInterner<'db>, Self> {
156        match (a.kind(), b.kind()) {
157            (GenericArgKind::Lifetime(a_lt), GenericArgKind::Lifetime(b_lt)) => {
158                Ok(relation.relate(a_lt, b_lt)?.into())
159            }
160            (GenericArgKind::Type(a_ty), GenericArgKind::Type(b_ty)) => {
161                Ok(relation.relate(a_ty, b_ty)?.into())
162            }
163            (GenericArgKind::Const(a_ct), GenericArgKind::Const(b_ct)) => {
164                Ok(relation.relate(a_ct, b_ct)?.into())
165            }
166            (GenericArgKind::Lifetime(unpacked), x) => {
167                unreachable!("impossible case reached: can't relate: {:?} with {:?}", unpacked, x)
168            }
169            (GenericArgKind::Type(unpacked), x) => {
170                unreachable!("impossible case reached: can't relate: {:?} with {:?}", unpacked, x)
171            }
172            (GenericArgKind::Const(unpacked), x) => {
173                unreachable!("impossible case reached: can't relate: {:?} with {:?}", unpacked, x)
174            }
175        }
176    }

183impl<'db> GenericArgs<'db> {
184    /// Creates an `GenericArgs` for generic parameter definitions,
185    /// by calling closures to obtain each kind.
186    /// The closures get to observe the `GenericArgs` as they're
187    /// being built, which can be used to correctly
188    /// replace defaults of generic parameters.
189    pub fn for_item<F>(
190        interner: DbInterner<'db>,
191        def_id: SolverDefId,
192        mut mk_kind: F,
193    ) -> GenericArgs<'db>
194    where
195        F: FnMut(u32, GenericParamId, &[GenericArg<'db>]) -> GenericArg<'db>,
196    {
197        let defs = interner.generics_of(def_id);
198        let count = defs.count();
199        let mut args = SmallVec::with_capacity(count);
200        Self::fill_item(&mut args, interner, defs, &mut mk_kind);
201        interner.mk_args(&args)
202    }
203
204    /// Creates an all-error `GenericArgs`.
205    pub fn error_for_item(interner: DbInterner<'db>, def_id: SolverDefId) -> GenericArgs<'db> {
206        GenericArgs::for_item(interner, def_id, |_, id, _| GenericArg::error_from_id(interner, id))
207    }
208
209    /// Like `for_item`, but prefers the default of a parameter if it has any.
210    pub fn for_item_with_defaults<F>(
211        interner: DbInterner<'db>,
212        def_id: GenericDefId,
213        mut fallback: F,
214    ) -> GenericArgs<'db>
215    where
216        F: FnMut(u32, GenericParamId, &[GenericArg<'db>]) -> GenericArg<'db>,
217    {
218        let defaults = interner.db.generic_defaults(def_id);
219        Self::for_item(interner, def_id.into(), |idx, id, prev| match defaults.get(idx as usize) {
220            Some(default) => default.instantiate(interner, prev),
221            None => fallback(idx, id, prev),
222        })
223    }
224
225    /// Like `for_item()`, but calls first uses the args from `first`.
226    pub fn fill_rest<F>(
227        interner: DbInterner<'db>,
228        def_id: SolverDefId,
229        first: impl IntoIterator<Item = GenericArg<'db>>,
230        mut fallback: F,
231    ) -> GenericArgs<'db>
232    where
233        F: FnMut(u32, GenericParamId, &[GenericArg<'db>]) -> GenericArg<'db>,
234    {
235        let mut iter = first.into_iter();
236        Self::for_item(interner, def_id, |idx, id, prev| {
237            iter.next().unwrap_or_else(|| fallback(idx, id, prev))
238        })
239    }
240
241    /// Appends default param values to `first` if needed. Params without default will call `fallback()`.
242    pub fn fill_with_defaults<F>(
243        interner: DbInterner<'db>,
244        def_id: GenericDefId,
245        first: impl IntoIterator<Item = GenericArg<'db>>,
246        mut fallback: F,
247    ) -> GenericArgs<'db>
248    where
249        F: FnMut(u32, GenericParamId, &[GenericArg<'db>]) -> GenericArg<'db>,
250    {
251        let defaults = interner.db.generic_defaults(def_id);
252        Self::fill_rest(interner, def_id.into(), first, |idx, id, prev| {
253            defaults
254                .get(idx as usize)
255                .map(|default| default.instantiate(interner, prev))
256                .unwrap_or_else(|| fallback(idx, id, prev))
257        })
258    }
259
260    fn fill_item<F>(
261        args: &mut SmallVec<[GenericArg<'db>; 8]>,
262        interner: DbInterner<'_>,
263        defs: Generics,
264        mk_kind: &mut F,
265    ) where
266        F: FnMut(u32, GenericParamId, &[GenericArg<'db>]) -> GenericArg<'db>,
267    {
268        if let Some(def_id) = defs.parent {
269            let parent_defs = interner.generics_of(def_id.into());
270            Self::fill_item(args, interner, parent_defs, mk_kind);
271        }
272        Self::fill_single(args, &defs, mk_kind);
273    }
274
275    fn fill_single<F>(args: &mut SmallVec<[GenericArg<'db>; 8]>, defs: &Generics, mk_kind: &mut F)
276    where
277        F: FnMut(u32, GenericParamId, &[GenericArg<'db>]) -> GenericArg<'db>,
278    {
279        args.reserve(defs.own_params.len());
280        for param in &defs.own_params {
281            let kind = mk_kind(args.len() as u32, param.id, args);
282            args.push(kind);
283        }
284    }

286    pub fn closure_sig_untupled(self) -> PolyFnSig<'db> {
287        let TyKind::FnPtr(inputs_and_output, hdr) =
288            self.split_closure_args_untupled().closure_sig_as_fn_ptr_ty.kind()
289        else {
290            unreachable!("not a function pointer")
291        };
292        inputs_and_output.with(hdr)
293    }
294
295    /// A "sensible" `.split_closure_args()`, where the arguments are not in a tuple.
296    pub fn split_closure_args_untupled(self) -> rustc_type_ir::ClosureArgsParts<DbInterner<'db>> {
297        // FIXME: should use `ClosureSubst` when possible
298        match self.inner().as_slice() {
299            [parent_args @ .., closure_kind_ty, sig_ty, tupled_upvars_ty] => {
300                let interner = DbInterner::conjure();
301                rustc_type_ir::ClosureArgsParts {
302                    parent_args: GenericArgs::new_from_iter(interner, parent_args.iter().cloned()),
303                    closure_sig_as_fn_ptr_ty: sig_ty.expect_ty(),
304                    closure_kind_ty: closure_kind_ty.expect_ty(),
305                    tupled_upvars_ty: tupled_upvars_ty.expect_ty(),
306                }
307            }
308            _ => {
309                unreachable!("unexpected closure sig");
310            }
311        }
312    }

327impl<'db> rustc_type_ir::relate::Relate<DbInterner<'db>> for GenericArgs<'db> {
328    fn relate<R: rustc_type_ir::relate::TypeRelation<DbInterner<'db>>>(
329        relation: &mut R,
330        a: Self,
331        b: Self,
332    ) -> rustc_type_ir::relate::RelateResult<DbInterner<'db>, Self> {
333        let interner = relation.cx();
334        CollectAndApply::collect_and_apply(
335            std::iter::zip(a.iter(), b.iter()).map(|(a, b)| {
336                relation.relate_with_variance(
337                    Variance::Invariant,
338                    VarianceDiagInfo::default(),
339                    a,
340                    b,
341                )
342            }),
343            |g| GenericArgs::new_from_iter(interner, g.iter().cloned()),
344        )
345    }
346}

349    fn as_closure(self) -> ClosureArgs<DbInterner<'db>> {

352    fn as_coroutine(self) -> CoroutineArgs<DbInterner<'db>> {

355    fn as_coroutine_closure(self) -> CoroutineClosureArgs<DbInterner<'db>> {
356        CoroutineClosureArgs { args: self }
357    }
358    fn rebase_onto(
359        self,
360        interner: DbInterner<'db>,
361        source_def_id: <DbInterner<'db> as rustc_type_ir::Interner>::DefId,
362        target: <DbInterner<'db> as rustc_type_ir::Interner>::GenericArgs,
363    ) -> <DbInterner<'db> as rustc_type_ir::Interner>::GenericArgs {
364        let defs = interner.generics_of(source_def_id);
365        interner.mk_args_from_iter(target.iter().chain(self.iter().skip(defs.count())))
366    }

368    fn identity_for_item(
369        interner: DbInterner<'db>,
370        def_id: <DbInterner<'db> as rustc_type_ir::Interner>::DefId,
371    ) -> <DbInterner<'db> as rustc_type_ir::Interner>::GenericArgs {
372        Self::for_item(interner, def_id, |index, kind, _| mk_param(interner, index, kind))
373    }

375    fn extend_with_error(
376        interner: DbInterner<'db>,
377        def_id: <DbInterner<'db> as rustc_type_ir::Interner>::DefId,
378        original_args: &[<DbInterner<'db> as rustc_type_ir::Interner>::GenericArg],
379    ) -> <DbInterner<'db> as rustc_type_ir::Interner>::GenericArgs {
380        Self::for_item(interner, def_id, |index, kind, _| {
381            if let Some(arg) = original_args.get(index as usize) {
382                *arg
383            } else {
384                error_for_param_kind(kind, interner)
385            }
386        })
387    }

389        self.inner()

391            .and_then(|g| g.as_type())

409    fn split_closure_args(self) -> rustc_type_ir::ClosureArgsParts<DbInterner<'db>> {
410        // FIXME: should use `ClosureSubst` when possible
411        match self.inner().as_slice() {
412            [parent_args @ .., closure_kind_ty, sig_ty, tupled_upvars_ty] => {
413                let interner = DbInterner::conjure();
414                // This is stupid, but the next solver expects the first input to actually be a tuple
415                let sig_ty = match sig_ty.expect_ty().kind() {
416                    TyKind::FnPtr(sig_tys, header) => Ty::new(
417                        interner,
418                        TyKind::FnPtr(
419                            sig_tys.map_bound(|s| {
420                                let inputs = Ty::new_tup_from_iter(interner, s.inputs().iter());
421                                let output = s.output();
422                                FnSigTys {
423                                    inputs_and_output: Tys::new_from_iter(
424                                        interner,
425                                        [inputs, output],
426                                    ),
427                                }
428                            }),
429                            header,
430                        ),
431                    ),
432                    _ => unreachable!("sig_ty should be last"),
433                };
434                rustc_type_ir::ClosureArgsParts {
435                    parent_args: GenericArgs::new_from_iter(interner, parent_args.iter().cloned()),
436                    closure_sig_as_fn_ptr_ty: sig_ty,
437                    closure_kind_ty: closure_kind_ty.expect_ty(),
438                    tupled_upvars_ty: tupled_upvars_ty.expect_ty(),
439                }
440            }
441            _ => {
442                unreachable!("unexpected closure sig");
443            }
444        }
445    }
446
447    fn split_coroutine_closure_args(
448        self,
449    ) -> rustc_type_ir::CoroutineClosureArgsParts<DbInterner<'db>> {
450        match self.inner().as_slice() {
451            [
452                parent_args @ ..,
453                closure_kind_ty,
454                signature_parts_ty,
455                tupled_upvars_ty,
456                coroutine_captures_by_ref_ty,
457            ] => rustc_type_ir::CoroutineClosureArgsParts {
458                parent_args: GenericArgs::new_from_iter(
459                    DbInterner::conjure(),
460                    parent_args.iter().cloned(),
461                ),
462                closure_kind_ty: closure_kind_ty.expect_ty(),
463                signature_parts_ty: signature_parts_ty.expect_ty(),
464                tupled_upvars_ty: tupled_upvars_ty.expect_ty(),
465                coroutine_captures_by_ref_ty: coroutine_captures_by_ref_ty.expect_ty(),
466            },
467            _ => panic!("GenericArgs were likely not for a CoroutineClosure."),
468        }
469    }
470
471    fn split_coroutine_args(self) -> rustc_type_ir::CoroutineArgsParts<DbInterner<'db>> {
472        let interner = DbInterner::conjure();
473        match self.inner().as_slice() {
474            [parent_args @ .., kind_ty, resume_ty, yield_ty, return_ty, tupled_upvars_ty] => {
475                rustc_type_ir::CoroutineArgsParts {
476                    parent_args: GenericArgs::new_from_iter(interner, parent_args.iter().cloned()),
477                    kind_ty: kind_ty.expect_ty(),
478                    resume_ty: resume_ty.expect_ty(),
479                    yield_ty: yield_ty.expect_ty(),
480                    return_ty: return_ty.expect_ty(),
481                    tupled_upvars_ty: tupled_upvars_ty.expect_ty(),
482                }
483            }
484            _ => panic!("GenericArgs were likely not for a Coroutine."),
485        }
486    }

489pub fn mk_param<'db>(interner: DbInterner<'db>, index: u32, id: GenericParamId) -> GenericArg<'db> {
490    match id {
491        GenericParamId::LifetimeParamId(id) => {
492            Region::new_early_param(interner, EarlyParamRegion { index, id }).into()
493        }
494        GenericParamId::TypeParamId(id) => Ty::new_param(interner, id, index).into(),
495        GenericParamId::ConstParamId(id) => {
496            Const::new_param(interner, ParamConst { index, id }).into()
497        }
498    }
499}

501pub fn error_for_param_kind<'db>(id: GenericParamId, interner: DbInterner<'db>) -> GenericArg<'db> {
502    match id {
503        GenericParamId::LifetimeParamId(_) => Region::error(interner).into(),
504        GenericParamId::TypeParamId(_) => Ty::new_error(interner, ErrorGuaranteed).into(),
505        GenericParamId::ConstParamId(_) => Const::error(interner).into(),
506    }
