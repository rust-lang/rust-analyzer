38#[derive(Clone, Copy, PartialEq, Eq, Hash)]
39pub struct ExternalConstraints<'db> {
40    interned: InternedRef<'db, ExternalConstraintsInterned>,
41}
42
43#[derive(PartialEq, Eq, Hash, GenericTypeVisitable)]
44pub(super) struct ExternalConstraintsInterned(ExternalConstraintsData<'static>);
45
46impl_internable!(gc; ExternalConstraintsInterned);
47
48const _: () = {
49    const fn is_copy<T: Copy>() {}
50    is_copy::<ExternalConstraints<'static>>();
51};

53impl<'db> ExternalConstraints<'db> {
54    #[inline]
55    pub fn new(_interner: DbInterner<'db>, data: ExternalConstraintsData<'db>) -> Self {
56        let data = unsafe {
57            std::mem::transmute::<ExternalConstraintsData<'db>, ExternalConstraintsData<'static>>(
58                data,
59            )
60        };
61        Self { interned: Interned::new_gc(ExternalConstraintsInterned(data)) }
62    }
63
64    #[inline]
65    pub fn inner(&self) -> &ExternalConstraintsData<'db> {
66        let inner = &self.interned.0;
67        unsafe {
68            std::mem::transmute::<&ExternalConstraintsData<'static>, &ExternalConstraintsData<'db>>(
69                inner,
70            )
71        }
72    }
73}

89impl<'db> rustc_type_ir::TypeVisitable<DbInterner<'db>> for ExternalConstraints<'db> {
90    fn visit_with<V: rustc_type_ir::TypeVisitor<DbInterner<'db>>>(
91        &self,
92        visitor: &mut V,
93    ) -> V::Result {
94        try_visit!(self.region_constraints.visit_with(visitor));
95        try_visit!(self.opaque_types.visit_with(visitor));
96        self.normalization_nested_goals.visit_with(visitor)
97    }
98}

100impl<'db> rustc_type_ir::TypeFoldable<DbInterner<'db>> for ExternalConstraints<'db> {
101    fn try_fold_with<F: rustc_type_ir::FallibleTypeFolder<DbInterner<'db>>>(
102        self,
103        folder: &mut F,
104    ) -> Result<Self, F::Error> {
105        Ok(ExternalConstraints::new(
106            folder.cx(),
107            ExternalConstraintsData {
108                region_constraints: self.region_constraints.clone().try_fold_with(folder)?,
109                opaque_types: self
110                    .opaque_types
111                    .iter()
112                    .cloned()
113                    .map(|opaque| opaque.try_fold_with(folder))
114                    .collect::<Result<_, F::Error>>()?,
115                normalization_nested_goals: self
116                    .normalization_nested_goals
117                    .clone()
118                    .try_fold_with(folder)?,
119            },
120        ))
121    }
122    fn fold_with<F: rustc_type_ir::TypeFolder<DbInterner<'db>>>(self, folder: &mut F) -> Self {
123        ExternalConstraints::new(
124            folder.cx(),
125            ExternalConstraintsData {
126                region_constraints: self.region_constraints.clone().fold_with(folder),
127                opaque_types: self
128                    .opaque_types
129                    .iter()
130                    .cloned()
131                    .map(|opaque| opaque.fold_with(folder))
132                    .collect(),
133                normalization_nested_goals: self
134                    .normalization_nested_goals
135                    .clone()
136                    .fold_with(folder),
137            },
138        )
139    }
