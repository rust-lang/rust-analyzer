206    /// Fast path helper for testing if a type is `Sized` or `MetaSized`.
207    ///
208    /// Returning true means the type is known to implement the sizedness trait. Returning `false`
209    /// means nothing -- could be sized, might not be.
210    ///
211    /// Note that we could never rely on the fact that a type such as `[_]` is trivially `!Sized`
212    /// because we could be in a type environment with a bound such as `[_]: Copy`. A function with
213    /// such a bound obviously never can be called, but that doesn't mean it shouldn't typecheck.
214    /// This is why this method doesn't return `Option<bool>`.
215    #[tracing::instrument(skip(tcx), level = "debug")]
216    pub fn has_trivial_sizedness(self, tcx: DbInterner<'db>, sizedness: SizedTraitKind) -> bool {
217        match self.kind() {
218            TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_))
219            | TyKind::Uint(_)
220            | TyKind::Int(_)
221            | TyKind::Bool
222            | TyKind::Float(_)
223            | TyKind::FnDef(..)
224            | TyKind::FnPtr(..)
225            | TyKind::UnsafeBinder(_)
226            | TyKind::RawPtr(..)
227            | TyKind::Char
228            | TyKind::Ref(..)
229            | TyKind::Coroutine(..)
230            | TyKind::CoroutineWitness(..)
231            | TyKind::Array(..)
232            | TyKind::Pat(..)
233            | TyKind::Closure(..)
234            | TyKind::CoroutineClosure(..)
235            | TyKind::Never
236            | TyKind::Error(_) => true,
237
238            TyKind::Str | TyKind::Slice(_) | TyKind::Dynamic(_, _) => match sizedness {
239                SizedTraitKind::Sized => false,
240                SizedTraitKind::MetaSized => true,
241            },
242
243            TyKind::Foreign(..) => match sizedness {
244                SizedTraitKind::Sized | SizedTraitKind::MetaSized => false,
245            },
246
247            TyKind::Tuple(tys) => {
248                tys.last().is_none_or(|ty| ty.has_trivial_sizedness(tcx, sizedness))
249            }
250
251            TyKind::Adt(def, args) => def
252                .sizedness_constraint(tcx, sizedness)
253                .is_none_or(|ty| ty.instantiate(tcx, args).has_trivial_sizedness(tcx, sizedness)),
254
255            TyKind::Alias(..) | TyKind::Param(_) | TyKind::Placeholder(..) | TyKind::Bound(..) => {
256                false
257            }
258
259            TyKind::Infer(InferTy::TyVar(_)) => false,
260
261            TyKind::Infer(
262                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_),
263            ) => {
264                panic!("`has_trivial_sizedness` applied to unexpected type: {self:?}")
265            }
266        }
267    }

269    /// Fast path helper for primitives which are always `Copy` and which
270    /// have a side-effect-free `Clone` impl.
271    ///
272    /// Returning true means the type is known to be pure and `Copy+Clone`.
273    /// Returning `false` means nothing -- could be `Copy`, might not be.
274    ///
275    /// This is mostly useful for optimizations, as these are the types
276    /// on which we can replace cloning with dereferencing.
277    pub fn is_trivially_pure_clone_copy(self) -> bool {
278        match self.kind() {
279            TyKind::Bool | TyKind::Char | TyKind::Never => true,
280
281            // These aren't even `Clone`
282            TyKind::Str | TyKind::Slice(..) | TyKind::Foreign(..) | TyKind::Dynamic(..) => false,
283
284            TyKind::Infer(InferTy::FloatVar(_) | InferTy::IntVar(_))
285            | TyKind::Int(..)
286            | TyKind::Uint(..)
287            | TyKind::Float(..) => true,
288
289            // ZST which can't be named are fine.
290            TyKind::FnDef(..) => true,
291
292            TyKind::Array(element_ty, _len) => element_ty.is_trivially_pure_clone_copy(),
293
294            // A 100-tuple isn't "trivial", so doing this only for reasonable sizes.
295            TyKind::Tuple(field_tys) => {
296                field_tys.len() <= 3 && field_tys.iter().all(Self::is_trivially_pure_clone_copy)
297            }
298
299            TyKind::Pat(ty, _) => ty.is_trivially_pure_clone_copy(),
300
301            // Sometimes traits aren't implemented for every ABI or arity,
302            // because we can't be generic over everything yet.
303            TyKind::FnPtr(..) => false,
304
305            // Definitely absolutely not copy.
306            TyKind::Ref(_, _, Mutability::Mut) => false,
307
308            // The standard library has a blanket Copy impl for shared references and raw pointers,
309            // for all unsized types.
310            TyKind::Ref(_, _, Mutability::Not) | TyKind::RawPtr(..) => true,
311
312            TyKind::Coroutine(..) | TyKind::CoroutineWitness(..) => false,
313
314            // Might be, but not "trivial" so just giving the safe answer.
315            TyKind::Adt(..) | TyKind::Closure(..) | TyKind::CoroutineClosure(..) => false,
316
317            TyKind::UnsafeBinder(_) => false,
318
319            // Needs normalization or revealing to determine, so no is the safe answer.
320            TyKind::Alias(..) => false,
321
322            TyKind::Param(..)
323            | TyKind::Placeholder(..)
324            | TyKind::Bound(..)
325            | TyKind::Infer(..)
326            | TyKind::Error(..) => false,
327        }
328    }

330    pub fn is_trivially_wf(self, tcx: DbInterner<'db>) -> bool {
331        match self.kind() {
332            TyKind::Bool
333            | TyKind::Char
334            | TyKind::Int(_)
335            | TyKind::Uint(_)
336            | TyKind::Float(_)
337            | TyKind::Str
338            | TyKind::Never
339            | TyKind::Param(_)
340            | TyKind::Placeholder(_)
341            | TyKind::Bound(..) => true,
342
343            TyKind::Slice(ty) => {
344                ty.is_trivially_wf(tcx) && ty.has_trivial_sizedness(tcx, SizedTraitKind::Sized)
345            }
346            TyKind::RawPtr(ty, _) => ty.is_trivially_wf(tcx),
347
348            TyKind::FnPtr(sig_tys, _) => {
349                sig_tys.skip_binder().inputs_and_output.iter().all(|ty| ty.is_trivially_wf(tcx))
350            }
351            TyKind::Ref(_, ty, _) => ty.is_global() && ty.is_trivially_wf(tcx),
352
353            TyKind::Infer(infer) => match infer {
354                InferTy::TyVar(_) => false,
355                InferTy::IntVar(_) | InferTy::FloatVar(_) => true,
356                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_) => true,
357            },
358
359            TyKind::Adt(_, _)
360            | TyKind::Tuple(_)
361            | TyKind::Array(..)
362            | TyKind::Foreign(_)
363            | TyKind::Pat(_, _)
364            | TyKind::FnDef(..)
365            | TyKind::UnsafeBinder(..)
366            | TyKind::Dynamic(..)
367            | TyKind::Closure(..)
368            | TyKind::CoroutineClosure(..)
369            | TyKind::Coroutine(..)
370            | TyKind::CoroutineWitness(..)
371            | TyKind::Alias(..)
372            | TyKind::Error(_) => false,
373        }
374    }

516    pub fn callable_sig(self, interner: DbInterner<'db>) -> Option<Binder<'db, FnSig<'db>>> {
517        match self.kind() {
518            TyKind::FnDef(callable, args) => {
519                Some(interner.fn_sig(callable).instantiate(interner, args))
520            }
521            TyKind::FnPtr(sig, hdr) => Some(sig.with(hdr)),
522            TyKind::Closure(_, closure_args) => {
523                Some(interner.signature_unclosure(closure_args.as_closure().sig(), Safety::Safe))
524            }
525            TyKind::CoroutineClosure(coroutine_id, args) => {
526                Some(args.as_coroutine_closure().coroutine_closure_sig().map_bound(|sig| {
527                    let unit_ty = Ty::new_unit(interner);
528                    let return_ty = Ty::new_coroutine(
529                        interner,
530                        coroutine_id,
531                        CoroutineArgs::new(
532                            interner,
533                            CoroutineArgsParts {
534                                parent_args: args.as_coroutine_closure().parent_args(),
535                                kind_ty: unit_ty,
536                                resume_ty: unit_ty,
537                                yield_ty: unit_ty,
538                                return_ty: sig.return_ty,
539                                // FIXME: Deduce this from the coroutine closure's upvars.
540                                tupled_upvars_ty: unit_ty,
541                            },
542                        )
543                        .args,
544                    );
545                    FnSig {
546                        inputs_and_output: Tys::new_from_iter(
547                            interner,
548                            sig.tupled_inputs_ty
549                                .tuple_fields()
550                                .iter()
551                                .chain(std::iter::once(return_ty)),
552                        ),
553                        c_variadic: sig.c_variadic,
554                        safety: sig.safety,
555                        abi: sig.abi,
556                    }
557                }))
558            }
559            _ => None,
560        }
561    }

679    pub fn impl_trait_bounds(self, db: &'db dyn HirDatabase) -> Option<Vec<Clause<'db>>> {
680        let interner = DbInterner::new_no_crate(db);
681
682        match self.kind() {
683            TyKind::Alias(AliasTyKind::Opaque, opaque_ty) => Some(
684                opaque_ty
685                    .def_id
686                    .expect_opaque_ty()
687                    .predicates(db)
688                    .iter_instantiated_copied(interner, opaque_ty.args.as_slice())
689                    .collect(),
690            ),
691            TyKind::Param(param) => {
692                // FIXME: We shouldn't use `param.id` here.
693                let generic_params = db.generic_params(param.id.parent());
694                let param_data = &generic_params[param.id.local_id()];
695                match param_data {
696                    TypeOrConstParamData::TypeParamData(p) => match p.provenance {
697                        TypeParamProvenance::ArgumentImplTrait => {
698                            let predicates = GenericPredicates::query_all(db, param.id.parent())
699                                .iter_identity_copied()
700                                .filter(|wc| match wc.kind().skip_binder() {
701                                    ClauseKind::Trait(tr) => tr.self_ty() == self,
702                                    ClauseKind::Projection(pred) => pred.self_ty() == self,
703                                    ClauseKind::TypeOutlives(pred) => pred.0 == self,
704                                    _ => false,
705                                })
706                                .collect::<Vec<_>>();
707
708                            Some(predicates)
709                        }
710                        _ => None,
711                    },
712                    _ => None,
713                }
714            }
715            TyKind::Coroutine(coroutine_id, _args) => {
716                let InternedCoroutine(owner, _) = coroutine_id.0.loc(db);
717                let krate = owner.module(db).krate(db);
718                if let Some(future_trait) = hir_def::lang_item::lang_items(db, krate).Future {
719                    // This is only used by type walking.
720                    // Parameters will be walked outside, and projection predicate is not used.
721                    // So just provide the Future trait.
722                    let impl_bound = TraitRef::new_from_args(
723                        interner,
724                        future_trait.into(),
725                        GenericArgs::empty(interner),
726                    )
727                    .upcast(interner);
728                    Some(vec![impl_bound])
729                } else {
730                    None
731                }
732            }
733            _ => None,
734        }
735    }

738    pub fn equals_ctor(self, other: Ty<'db>) -> bool {
739        match (self.kind(), other.kind()) {
740            (TyKind::Adt(adt, ..), TyKind::Adt(adt2, ..)) => adt.def_id() == adt2.def_id(),
741            (TyKind::Slice(_), TyKind::Slice(_)) | (TyKind::Array(_, _), TyKind::Array(_, _)) => {
742                true
743            }
744            (TyKind::FnDef(def_id, ..), TyKind::FnDef(def_id2, ..)) => def_id == def_id2,
745            (TyKind::Alias(_, alias, ..), TyKind::Alias(_, alias2)) => {
746                alias.def_id == alias2.def_id
747            }
748            (TyKind::Foreign(ty_id, ..), TyKind::Foreign(ty_id2, ..)) => ty_id == ty_id2,
749            (TyKind::Closure(id1, _), TyKind::Closure(id2, _)) => id1 == id2,
750            (TyKind::Ref(.., mutability), TyKind::Ref(.., mutability2))
751            | (TyKind::RawPtr(.., mutability), TyKind::RawPtr(.., mutability2)) => {
752                mutability == mutability2
753            }
754            (TyKind::FnPtr(sig, hdr), TyKind::FnPtr(sig2, hdr2)) => sig == sig2 && hdr == hdr2,
755            (TyKind::Tuple(tys), TyKind::Tuple(tys2)) => tys.len() == tys2.len(),
756            (TyKind::Str, TyKind::Str)
757            | (TyKind::Never, TyKind::Never)
758            | (TyKind::Char, TyKind::Char)
759            | (TyKind::Bool, TyKind::Bool) => true,
760            (TyKind::Int(int), TyKind::Int(int2)) => int == int2,
761            (TyKind::Float(float), TyKind::Float(float2)) => float == float2,
762            _ => false,
763        }
764    }

1132    fn new_coroutine_witness_for_coroutine(
1133        interner: DbInterner<'db>,
1134        def_id: CoroutineIdWrapper,
1135        coroutine_args: <DbInterner<'db> as Interner>::GenericArgs,
1136    ) -> Self {
1137        // HACK: Coroutine witness types are lifetime erased, so they
1138        // never reference any lifetime args from the coroutine. We erase
1139        // the regions here since we may get into situations where a
1140        // coroutine is recursively contained within itself, leading to
1141        // witness types that differ by region args. This means that
1142        // cycle detection in fulfillment will not kick in, which leads
1143        // to unnecessary overflows in async code. See the issue:
1144        // <https://github.com/rust-lang/rust/issues/145151>.
1145        let coroutine_args = interner.mk_args_from_iter(coroutine_args.iter().map(|arg| {
1146            match arg.kind() {
1147                GenericArgKind::Type(_) | GenericArgKind::Const(_) => arg,
1148                GenericArgKind::Lifetime(_) => {
1149                    crate::next_solver::Region::new(interner, rustc_type_ir::RegionKind::ReErased)
1150                        .into()
1151                }
1152            }
1153        }));
1154        Ty::new_coroutine_witness(interner, def_id, coroutine_args)
1155    }

1467impl<'db> DbInterner<'db> {
1468    /// Given a closure signature, returns an equivalent fn signature. Detuples
1469    /// and so forth -- so e.g., if we have a sig with `Fn<(u32, i32)>` then
1470    /// you would get a `fn(u32, i32)`.
1471    /// `unsafety` determines the unsafety of the fn signature. If you pass
1472    /// `Safety::Unsafe` in the previous example, then you would get
1473    /// an `unsafe fn (u32, i32)`.
1474    /// It cannot convert a closure that requires unsafe.
1475    pub fn signature_unclosure(self, sig: PolyFnSig<'db>, safety: Safety) -> PolyFnSig<'db> {
1476        sig.map_bound(|s| {
1477            let params = match s.inputs()[0].kind() {
1478                TyKind::Tuple(params) => params,
1479                _ => panic!(),
1480            };
1481            self.mk_fn_sig(params, s.output(), s.c_variadic, safety, FnAbi::Rust)
1482        })
1483    }
