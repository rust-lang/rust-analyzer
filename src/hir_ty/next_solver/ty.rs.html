171    /// Fast path helper for testing if a type is `Sized` or `MetaSized`.
172    ///
173    /// Returning true means the type is known to implement the sizedness trait. Returning `false`
174    /// means nothing -- could be sized, might not be.
175    ///
176    /// Note that we could never rely on the fact that a type such as `[_]` is trivially `!Sized`
177    /// because we could be in a type environment with a bound such as `[_]: Copy`. A function with
178    /// such a bound obviously never can be called, but that doesn't mean it shouldn't typecheck.
179    /// This is why this method doesn't return `Option<bool>`.
180    #[tracing::instrument(skip(tcx), level = "debug")]
181    pub fn has_trivial_sizedness(self, tcx: DbInterner<'db>, sizedness: SizedTraitKind) -> bool {
182        match self.kind() {
183            TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_))
184            | TyKind::Uint(_)
185            | TyKind::Int(_)
186            | TyKind::Bool
187            | TyKind::Float(_)
188            | TyKind::FnDef(..)
189            | TyKind::FnPtr(..)
190            | TyKind::UnsafeBinder(_)
191            | TyKind::RawPtr(..)
192            | TyKind::Char
193            | TyKind::Ref(..)
194            | TyKind::Coroutine(..)
195            | TyKind::CoroutineWitness(..)
196            | TyKind::Array(..)
197            | TyKind::Pat(..)
198            | TyKind::Closure(..)
199            | TyKind::CoroutineClosure(..)
200            | TyKind::Never
201            | TyKind::Error(_) => true,
202
203            TyKind::Str | TyKind::Slice(_) | TyKind::Dynamic(_, _) => match sizedness {
204                SizedTraitKind::Sized => false,
205                SizedTraitKind::MetaSized => true,
206            },
207
208            TyKind::Foreign(..) => match sizedness {
209                SizedTraitKind::Sized | SizedTraitKind::MetaSized => false,
210            },
211
212            TyKind::Tuple(tys) => {
213                tys.last().is_none_or(|ty| ty.has_trivial_sizedness(tcx, sizedness))
214            }
215
216            TyKind::Adt(def, args) => def
217                .sizedness_constraint(tcx, sizedness)
218                .is_none_or(|ty| ty.instantiate(tcx, args).has_trivial_sizedness(tcx, sizedness)),
219
220            TyKind::Alias(..) | TyKind::Param(_) | TyKind::Placeholder(..) | TyKind::Bound(..) => {
221                false
222            }
223
224            TyKind::Infer(InferTy::TyVar(_)) => false,
225
226            TyKind::Infer(
227                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_),
228            ) => {
229                panic!("`has_trivial_sizedness` applied to unexpected type: {self:?}")
230            }
231        }
232    }

234    /// Fast path helper for primitives which are always `Copy` and which
235    /// have a side-effect-free `Clone` impl.
236    ///
237    /// Returning true means the type is known to be pure and `Copy+Clone`.
238    /// Returning `false` means nothing -- could be `Copy`, might not be.
239    ///
240    /// This is mostly useful for optimizations, as these are the types
241    /// on which we can replace cloning with dereferencing.
242    pub fn is_trivially_pure_clone_copy(self) -> bool {
243        match self.kind() {
244            TyKind::Bool | TyKind::Char | TyKind::Never => true,
245
246            // These aren't even `Clone`
247            TyKind::Str | TyKind::Slice(..) | TyKind::Foreign(..) | TyKind::Dynamic(..) => false,
248
249            TyKind::Infer(InferTy::FloatVar(_) | InferTy::IntVar(_))
250            | TyKind::Int(..)
251            | TyKind::Uint(..)
252            | TyKind::Float(..) => true,
253
254            // ZST which can't be named are fine.
255            TyKind::FnDef(..) => true,
256
257            TyKind::Array(element_ty, _len) => element_ty.is_trivially_pure_clone_copy(),
258
259            // A 100-tuple isn't "trivial", so doing this only for reasonable sizes.
260            TyKind::Tuple(field_tys) => {
261                field_tys.len() <= 3 && field_tys.iter().all(Self::is_trivially_pure_clone_copy)
262            }
263
264            TyKind::Pat(ty, _) => ty.is_trivially_pure_clone_copy(),
265
266            // Sometimes traits aren't implemented for every ABI or arity,
267            // because we can't be generic over everything yet.
268            TyKind::FnPtr(..) => false,
269
270            // Definitely absolutely not copy.
271            TyKind::Ref(_, _, Mutability::Mut) => false,
272
273            // The standard library has a blanket Copy impl for shared references and raw pointers,
274            // for all unsized types.
275            TyKind::Ref(_, _, Mutability::Not) | TyKind::RawPtr(..) => true,
276
277            TyKind::Coroutine(..) | TyKind::CoroutineWitness(..) => false,
278
279            // Might be, but not "trivial" so just giving the safe answer.
280            TyKind::Adt(..) | TyKind::Closure(..) | TyKind::CoroutineClosure(..) => false,
281
282            TyKind::UnsafeBinder(_) => false,
283
284            // Needs normalization or revealing to determine, so no is the safe answer.
285            TyKind::Alias(..) => false,
286
287            TyKind::Param(..)
288            | TyKind::Placeholder(..)
289            | TyKind::Bound(..)
290            | TyKind::Infer(..)
291            | TyKind::Error(..) => false,
292        }
293    }

295    pub fn is_trivially_wf(self, tcx: DbInterner<'db>) -> bool {
296        match self.kind() {
297            TyKind::Bool
298            | TyKind::Char
299            | TyKind::Int(_)
300            | TyKind::Uint(_)
301            | TyKind::Float(_)
302            | TyKind::Str
303            | TyKind::Never
304            | TyKind::Param(_)
305            | TyKind::Placeholder(_)
306            | TyKind::Bound(..) => true,
307
308            TyKind::Slice(ty) => {
309                ty.is_trivially_wf(tcx) && ty.has_trivial_sizedness(tcx, SizedTraitKind::Sized)
310            }
311            TyKind::RawPtr(ty, _) => ty.is_trivially_wf(tcx),
312
313            TyKind::FnPtr(sig_tys, _) => {
314                sig_tys.skip_binder().inputs_and_output.iter().all(|ty| ty.is_trivially_wf(tcx))
315            }
316            TyKind::Ref(_, ty, _) => ty.is_global() && ty.is_trivially_wf(tcx),
317
318            TyKind::Infer(infer) => match infer {
319                InferTy::TyVar(_) => false,
320                InferTy::IntVar(_) | InferTy::FloatVar(_) => true,
321                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_) => true,
322            },
323
324            TyKind::Adt(_, _)
325            | TyKind::Tuple(_)
326            | TyKind::Array(..)
327            | TyKind::Foreign(_)
328            | TyKind::Pat(_, _)
329            | TyKind::FnDef(..)
330            | TyKind::UnsafeBinder(..)
331            | TyKind::Dynamic(..)
332            | TyKind::Closure(..)
333            | TyKind::CoroutineClosure(..)
334            | TyKind::Coroutine(..)
335            | TyKind::CoroutineWitness(..)
336            | TyKind::Alias(..)
337            | TyKind::Error(_) => false,
338        }
339    }

458    pub fn callable_sig(self, interner: DbInterner<'db>) -> Option<Binder<'db, FnSig<'db>>> {
459        match self.kind() {
460            TyKind::FnDef(callable, args) => {
461                Some(interner.fn_sig(callable).instantiate(interner, args))
462            }
463            TyKind::FnPtr(sig, hdr) => Some(sig.with(hdr)),
464            TyKind::Closure(_, closure_args) => closure_args
465                .split_closure_args_untupled()
466                .closure_sig_as_fn_ptr_ty
467                .callable_sig(interner),
468            TyKind::CoroutineClosure(coroutine_id, args) => {
469                Some(args.as_coroutine_closure().coroutine_closure_sig().map_bound(|sig| {
470                    let unit_ty = Ty::new_unit(interner);
471                    let return_ty = Ty::new_coroutine(
472                        interner,
473                        coroutine_id,
474                        CoroutineArgs::new(
475                            interner,
476                            CoroutineArgsParts {
477                                parent_args: args.as_coroutine_closure().parent_args(),
478                                kind_ty: unit_ty,
479                                resume_ty: unit_ty,
480                                yield_ty: unit_ty,
481                                return_ty: sig.return_ty,
482                                // FIXME: Deduce this from the coroutine closure's upvars.
483                                tupled_upvars_ty: unit_ty,
484                            },
485                        )
486                        .args,
487                    );
488                    FnSig {
489                        inputs_and_output: Tys::new_from_iter(
490                            interner,
491                            sig.tupled_inputs_ty
492                                .tuple_fields()
493                                .iter()
494                                .chain(std::iter::once(return_ty)),
495                        ),
496                        c_variadic: sig.c_variadic,
497                        safety: sig.safety,
498                        abi: sig.abi,
499                    }
500                }))
501            }
502            _ => None,
503        }
504    }

622    pub fn impl_trait_bounds(self, db: &'db dyn HirDatabase) -> Option<Vec<Clause<'db>>> {
623        let interner = DbInterner::new_with(db, None, None);
624
625        match self.kind() {
626            TyKind::Alias(AliasTyKind::Opaque, opaque_ty) => Some(
627                opaque_ty
628                    .def_id
629                    .expect_opaque_ty()
630                    .predicates(db)
631                    .iter_instantiated_copied(interner, opaque_ty.args.as_slice())
632                    .collect(),
633            ),
634            TyKind::Param(param) => {
635                // FIXME: We shouldn't use `param.id` here.
636                let generic_params = db.generic_params(param.id.parent());
637                let param_data = &generic_params[param.id.local_id()];
638                match param_data {
639                    TypeOrConstParamData::TypeParamData(p) => match p.provenance {
640                        TypeParamProvenance::ArgumentImplTrait => {
641                            let predicates = GenericPredicates::query_all(db, param.id.parent())
642                                .iter_identity_copied()
643                                .filter(|wc| match wc.kind().skip_binder() {
644                                    ClauseKind::Trait(tr) => tr.self_ty() == self,
645                                    ClauseKind::Projection(pred) => pred.self_ty() == self,
646                                    ClauseKind::TypeOutlives(pred) => pred.0 == self,
647                                    _ => false,
648                                })
649                                .collect::<Vec<_>>();
650
651                            Some(predicates)
652                        }
653                        _ => None,
654                    },
655                    _ => None,
656                }
657            }
658            TyKind::Coroutine(coroutine_id, _args) => {
659                let InternedCoroutine(owner, _) = coroutine_id.0.loc(db);
660                let krate = owner.module(db).krate();
661                if let Some(future_trait) = LangItem::Future.resolve_trait(db, krate) {

663                    // Parameters will be walked outside, and projection predicate is not used.
664                    // So just provide the Future trait.

666                        interner,
667                        future_trait.into(),
668                        GenericArgs::new_from_iter(interner, []),

670                    .upcast(interner);
671                    Some(vec![impl_bound])
672                } else {
673                    None
674                }
675            }
676            _ => None,

690            }
691            (TyKind::Foreign(ty_id, ..), TyKind::Foreign(ty_id2, ..)) => ty_id == ty_id2,
692            (TyKind::Closure(id1, _), TyKind::Closure(id2, _)) => id1 == id2,
693            (TyKind::Ref(.., mutability), TyKind::Ref(.., mutability2))
694            | (TyKind::RawPtr(.., mutability), TyKind::RawPtr(.., mutability2)) => {
695                mutability == mutability2
696            }
697            (TyKind::FnPtr(sig, hdr), TyKind::FnPtr(sig2, hdr2)) => sig == sig2 && hdr == hdr2,
698            (TyKind::Tuple(tys), TyKind::Tuple(tys2)) => tys.len() == tys2.len(),
699            (TyKind::Str, TyKind::Str)
700            | (TyKind::Never, TyKind::Never)
701            | (TyKind::Char, TyKind::Char)
702            | (TyKind::Bool, TyKind::Bool) => true,
703            (TyKind::Int(int), TyKind::Int(int2)) => int == int2,
704            (TyKind::Float(float), TyKind::Float(float2)) => float == float2,
705            _ => false,
706        }

1219            TyKind::Bool
1220            | TyKind::Char
1221            | TyKind::Int(_)
1222            | TyKind::Uint(_)
1223            | TyKind::Float(_)
1224            | TyKind::Adt(..)

1237            | TyKind::Never
1238            | TyKind::Tuple(_)
1239            | TyKind::Error(_)
1240            | TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_)) => {
