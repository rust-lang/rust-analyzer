150    /// Fast path helper for testing if a type is `Sized` or `MetaSized`.
151    ///
152    /// Returning true means the type is known to implement the sizedness trait. Returning `false`
153    /// means nothing -- could be sized, might not be.
154    ///
155    /// Note that we could never rely on the fact that a type such as `[_]` is trivially `!Sized`
156    /// because we could be in a type environment with a bound such as `[_]: Copy`. A function with
157    /// such a bound obviously never can be called, but that doesn't mean it shouldn't typecheck.
158    /// This is why this method doesn't return `Option<bool>`.
159    #[tracing::instrument(skip(tcx), level = "debug")]
160    pub fn has_trivial_sizedness(self, tcx: DbInterner<'db>, sizedness: SizedTraitKind) -> bool {
161        match self.kind() {
162            TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_))
163            | TyKind::Uint(_)
164            | TyKind::Int(_)
165            | TyKind::Bool
166            | TyKind::Float(_)
167            | TyKind::FnDef(..)
168            | TyKind::FnPtr(..)
169            | TyKind::UnsafeBinder(_)
170            | TyKind::RawPtr(..)
171            | TyKind::Char
172            | TyKind::Ref(..)
173            | TyKind::Coroutine(..)
174            | TyKind::CoroutineWitness(..)
175            | TyKind::Array(..)
176            | TyKind::Pat(..)
177            | TyKind::Closure(..)
178            | TyKind::CoroutineClosure(..)
179            | TyKind::Never
180            | TyKind::Error(_) => true,
181
182            TyKind::Str | TyKind::Slice(_) | TyKind::Dynamic(_, _) => match sizedness {
183                SizedTraitKind::Sized => false,
184                SizedTraitKind::MetaSized => true,
185            },
186
187            TyKind::Foreign(..) => match sizedness {
188                SizedTraitKind::Sized | SizedTraitKind::MetaSized => false,
189            },
190
191            TyKind::Tuple(tys) => {
192                tys.last().is_none_or(|ty| ty.has_trivial_sizedness(tcx, sizedness))
193            }
194
195            TyKind::Adt(def, args) => def
196                .sizedness_constraint(tcx, sizedness)
197                .is_none_or(|ty| ty.instantiate(tcx, args).has_trivial_sizedness(tcx, sizedness)),
198
199            TyKind::Alias(..) | TyKind::Param(_) | TyKind::Placeholder(..) | TyKind::Bound(..) => {
200                false
201            }
202
203            TyKind::Infer(InferTy::TyVar(_)) => false,
204
205            TyKind::Infer(
206                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_),
207            ) => {
208                panic!("`has_trivial_sizedness` applied to unexpected type: {self:?}")
209            }
210        }
211    }

213    /// Fast path helper for primitives which are always `Copy` and which
214    /// have a side-effect-free `Clone` impl.
215    ///
216    /// Returning true means the type is known to be pure and `Copy+Clone`.
217    /// Returning `false` means nothing -- could be `Copy`, might not be.
218    ///
219    /// This is mostly useful for optimizations, as these are the types
220    /// on which we can replace cloning with dereferencing.
221    pub fn is_trivially_pure_clone_copy(self) -> bool {
222        match self.kind() {
223            TyKind::Bool | TyKind::Char | TyKind::Never => true,
224
225            // These aren't even `Clone`
226            TyKind::Str | TyKind::Slice(..) | TyKind::Foreign(..) | TyKind::Dynamic(..) => false,
227
228            TyKind::Infer(InferTy::FloatVar(_) | InferTy::IntVar(_))
229            | TyKind::Int(..)
230            | TyKind::Uint(..)
231            | TyKind::Float(..) => true,
232
233            // ZST which can't be named are fine.
234            TyKind::FnDef(..) => true,
235
236            TyKind::Array(element_ty, _len) => element_ty.is_trivially_pure_clone_copy(),
237
238            // A 100-tuple isn't "trivial", so doing this only for reasonable sizes.
239            TyKind::Tuple(field_tys) => {
240                field_tys.len() <= 3 && field_tys.iter().all(Self::is_trivially_pure_clone_copy)
241            }
242
243            TyKind::Pat(ty, _) => ty.is_trivially_pure_clone_copy(),
244
245            // Sometimes traits aren't implemented for every ABI or arity,
246            // because we can't be generic over everything yet.
247            TyKind::FnPtr(..) => false,
248
249            // Definitely absolutely not copy.
250            TyKind::Ref(_, _, Mutability::Mut) => false,
251
252            // The standard library has a blanket Copy impl for shared references and raw pointers,
253            // for all unsized types.
254            TyKind::Ref(_, _, Mutability::Not) | TyKind::RawPtr(..) => true,
255
256            TyKind::Coroutine(..) | TyKind::CoroutineWitness(..) => false,
257
258            // Might be, but not "trivial" so just giving the safe answer.
259            TyKind::Adt(..) | TyKind::Closure(..) | TyKind::CoroutineClosure(..) => false,
260
261            TyKind::UnsafeBinder(_) => false,
262
263            // Needs normalization or revealing to determine, so no is the safe answer.
264            TyKind::Alias(..) => false,
265
266            TyKind::Param(..)
267            | TyKind::Placeholder(..)
268            | TyKind::Bound(..)
269            | TyKind::Infer(..)
270            | TyKind::Error(..) => false,
271        }
272    }

274    pub fn is_trivially_wf(self, tcx: DbInterner<'db>) -> bool {
275        match self.kind() {
276            TyKind::Bool
277            | TyKind::Char
278            | TyKind::Int(_)
279            | TyKind::Uint(_)
280            | TyKind::Float(_)
281            | TyKind::Str
282            | TyKind::Never
283            | TyKind::Param(_)
284            | TyKind::Placeholder(_)
285            | TyKind::Bound(..) => true,
286
287            TyKind::Slice(ty) => {
288                ty.is_trivially_wf(tcx) && ty.has_trivial_sizedness(tcx, SizedTraitKind::Sized)
289            }
290            TyKind::RawPtr(ty, _) => ty.is_trivially_wf(tcx),
291
292            TyKind::FnPtr(sig_tys, _) => {
293                sig_tys.skip_binder().inputs_and_output.iter().all(|ty| ty.is_trivially_wf(tcx))
294            }
295            TyKind::Ref(_, ty, _) => ty.is_global() && ty.is_trivially_wf(tcx),
296
297            TyKind::Infer(infer) => match infer {
298                InferTy::TyVar(_) => false,
299                InferTy::IntVar(_) | InferTy::FloatVar(_) => true,
300                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_) => true,
301            },
302
303            TyKind::Adt(_, _)
304            | TyKind::Tuple(_)
305            | TyKind::Array(..)
306            | TyKind::Foreign(_)
307            | TyKind::Pat(_, _)
308            | TyKind::FnDef(..)
309            | TyKind::UnsafeBinder(..)
310            | TyKind::Dynamic(..)
311            | TyKind::Closure(..)
312            | TyKind::CoroutineClosure(..)
313            | TyKind::Coroutine(..)
314            | TyKind::CoroutineWitness(..)
315            | TyKind::Alias(..)
316            | TyKind::Error(_) => false,
317        }
318    }

356    /// Returns the type of `*ty`.
357    ///
358    /// The parameter `explicit` indicates if this is an *explicit* dereference.
359    /// Some types -- notably raw ptrs -- can only be dereferenced explicitly.
360    pub fn builtin_deref(self, db: &dyn HirDatabase, explicit: bool) -> Option<Ty<'db>> {
361        match self.kind() {
362            TyKind::Adt(adt, substs) if crate::lang_items::is_box(db, adt.def_id().0) => {
363                Some(substs.as_slice()[0].expect_ty())
364            }
365            TyKind::Ref(_, ty, _) => Some(ty),
366            TyKind::RawPtr(ty, _) if explicit => Some(ty),
367            _ => None,
368        }
369    }

379impl<'db> TypeVisitor<DbInterner<'db>> for ReferencesNonLifetimeError {

382    fn visit_ty(&mut self, ty: Ty<'db>) -> Self::Result {

384    }

387        if c.is_ct_error() { ControlFlow::Break(()) } else { c.super_visit_with(self) }
388    }

420impl<'db> TypeSuperVisitable<DbInterner<'db>> for Ty<'db> {
421    fn super_visit_with<V: rustc_type_ir::TypeVisitor<DbInterner<'db>>>(
422        &self,
423        visitor: &mut V,
424    ) -> V::Result {
425        match (*self).kind() {
426            TyKind::RawPtr(ty, _mutbl) => ty.visit_with(visitor),
427            TyKind::Array(typ, sz) => {
428                try_visit!(typ.visit_with(visitor));
429                sz.visit_with(visitor)
430            }
431            TyKind::Slice(typ) => typ.visit_with(visitor),
432            TyKind::Adt(_, args) => args.visit_with(visitor),
433            TyKind::Dynamic(ref trait_ty, ref reg) => {
434                try_visit!(trait_ty.visit_with(visitor));
435                reg.visit_with(visitor)
436            }
437            TyKind::Tuple(ts) => ts.visit_with(visitor),
438            TyKind::FnDef(_, args) => args.visit_with(visitor),
439            TyKind::FnPtr(ref sig_tys, _) => sig_tys.visit_with(visitor),
440            TyKind::UnsafeBinder(f) => f.visit_with(visitor),
441            TyKind::Ref(r, ty, _) => {
442                try_visit!(r.visit_with(visitor));
443                ty.visit_with(visitor)
444            }
445            TyKind::Coroutine(_did, ref args) => args.visit_with(visitor),
446            TyKind::CoroutineWitness(_did, ref args) => args.visit_with(visitor),
447            TyKind::Closure(_did, ref args) => args.visit_with(visitor),
448            TyKind::CoroutineClosure(_did, ref args) => args.visit_with(visitor),
449            TyKind::Alias(_, ref data) => data.visit_with(visitor),
450
451            TyKind::Pat(ty, pat) => {
452                try_visit!(ty.visit_with(visitor));
453                pat.visit_with(visitor)
454            }
455
456            TyKind::Error(guar) => guar.visit_with(visitor),
457
458            TyKind::Bool
459            | TyKind::Char
460            | TyKind::Str
461            | TyKind::Int(_)
462            | TyKind::Uint(_)
463            | TyKind::Float(_)
464            | TyKind::Infer(_)
465            | TyKind::Bound(..)
466            | TyKind::Placeholder(..)
467            | TyKind::Param(..)
468            | TyKind::Never
469            | TyKind::Foreign(..) => V::Result::output(),
470        }
471    }
472}

486impl<'db> TypeSuperFoldable<DbInterner<'db>> for Ty<'db> {
487    fn try_super_fold_with<F: rustc_type_ir::FallibleTypeFolder<DbInterner<'db>>>(
488        self,
489        folder: &mut F,
490    ) -> Result<Self, F::Error> {
491        let kind = match self.kind() {
492            TyKind::RawPtr(ty, mutbl) => TyKind::RawPtr(ty.try_fold_with(folder)?, mutbl),
493            TyKind::Array(typ, sz) => {
494                TyKind::Array(typ.try_fold_with(folder)?, sz.try_fold_with(folder)?)
495            }
496            TyKind::Slice(typ) => TyKind::Slice(typ.try_fold_with(folder)?),
497            TyKind::Adt(tid, args) => TyKind::Adt(tid, args.try_fold_with(folder)?),
498            TyKind::Dynamic(trait_ty, region) => {
499                TyKind::Dynamic(trait_ty.try_fold_with(folder)?, region.try_fold_with(folder)?)
500            }
501            TyKind::Tuple(ts) => TyKind::Tuple(ts.try_fold_with(folder)?),
502            TyKind::FnDef(def_id, args) => TyKind::FnDef(def_id, args.try_fold_with(folder)?),
503            TyKind::FnPtr(sig_tys, hdr) => TyKind::FnPtr(sig_tys.try_fold_with(folder)?, hdr),
504            TyKind::UnsafeBinder(f) => TyKind::UnsafeBinder(f.try_fold_with(folder)?),
505            TyKind::Ref(r, ty, mutbl) => {
506                TyKind::Ref(r.try_fold_with(folder)?, ty.try_fold_with(folder)?, mutbl)
507            }
508            TyKind::Coroutine(did, args) => TyKind::Coroutine(did, args.try_fold_with(folder)?),
509            TyKind::CoroutineWitness(did, args) => {
510                TyKind::CoroutineWitness(did, args.try_fold_with(folder)?)
511            }
512            TyKind::Closure(did, args) => TyKind::Closure(did, args.try_fold_with(folder)?),
513            TyKind::CoroutineClosure(did, args) => {
514                TyKind::CoroutineClosure(did, args.try_fold_with(folder)?)
515            }
516            TyKind::Alias(kind, data) => TyKind::Alias(kind, data.try_fold_with(folder)?),
517            TyKind::Pat(ty, pat) => {
518                TyKind::Pat(ty.try_fold_with(folder)?, pat.try_fold_with(folder)?)
519            }
520
521            TyKind::Bool
522            | TyKind::Char
523            | TyKind::Str
524            | TyKind::Int(_)
525            | TyKind::Uint(_)
526            | TyKind::Float(_)
527            | TyKind::Error(_)
528            | TyKind::Infer(_)
529            | TyKind::Param(..)
530            | TyKind::Bound(..)
531            | TyKind::Placeholder(..)
532            | TyKind::Never
533            | TyKind::Foreign(..) => return Ok(self),
534        };
535
536        Ok(if self.kind() == kind { self } else { Ty::new(folder.cx(), kind) })
537    }
538    fn super_fold_with<F: rustc_type_ir::TypeFolder<DbInterner<'db>>>(
539        self,
540        folder: &mut F,
541    ) -> Self {
542        let kind = match self.kind() {
543            TyKind::RawPtr(ty, mutbl) => TyKind::RawPtr(ty.fold_with(folder), mutbl),
544            TyKind::Array(typ, sz) => TyKind::Array(typ.fold_with(folder), sz.fold_with(folder)),
545            TyKind::Slice(typ) => TyKind::Slice(typ.fold_with(folder)),
546            TyKind::Adt(tid, args) => TyKind::Adt(tid, args.fold_with(folder)),
547            TyKind::Dynamic(trait_ty, region) => {
548                TyKind::Dynamic(trait_ty.fold_with(folder), region.fold_with(folder))
549            }
550            TyKind::Tuple(ts) => TyKind::Tuple(ts.fold_with(folder)),
551            TyKind::FnDef(def_id, args) => TyKind::FnDef(def_id, args.fold_with(folder)),
552            TyKind::FnPtr(sig_tys, hdr) => TyKind::FnPtr(sig_tys.fold_with(folder), hdr),
553            TyKind::UnsafeBinder(f) => TyKind::UnsafeBinder(f.fold_with(folder)),
554            TyKind::Ref(r, ty, mutbl) => {
555                TyKind::Ref(r.fold_with(folder), ty.fold_with(folder), mutbl)
556            }
557            TyKind::Coroutine(did, args) => TyKind::Coroutine(did, args.fold_with(folder)),
558            TyKind::CoroutineWitness(did, args) => {
559                TyKind::CoroutineWitness(did, args.fold_with(folder))
560            }
561            TyKind::Closure(did, args) => TyKind::Closure(did, args.fold_with(folder)),
562            TyKind::CoroutineClosure(did, args) => {
563                TyKind::CoroutineClosure(did, args.fold_with(folder))
564            }
565            TyKind::Alias(kind, data) => TyKind::Alias(kind, data.fold_with(folder)),
566            TyKind::Pat(ty, pat) => TyKind::Pat(ty.fold_with(folder), pat.fold_with(folder)),
567
568            TyKind::Bool
569            | TyKind::Char
570            | TyKind::Str
571            | TyKind::Int(_)
572            | TyKind::Uint(_)
573            | TyKind::Float(_)
574            | TyKind::Error(_)
575            | TyKind::Infer(_)
576            | TyKind::Param(..)
577            | TyKind::Bound(..)
578            | TyKind::Placeholder(..)
579            | TyKind::Never
580            | TyKind::Foreign(..) => return self,
581        };
582
583        if self.kind() == kind { self } else { Ty::new(folder.cx(), kind) }
584    }
585}

724    ) -> Self {
725        // HACK: Coroutine witness types are lifetime erased, so they
726        // never reference any lifetime args from the coroutine. We erase
727        // the regions here since we may get into situations where a
728        // coroutine is recursively contained within itself, leading to
729        // witness types that differ by region args. This means that
730        // cycle detection in fulfillment will not kick in, which leads
731        // to unnecessary overflows in async code. See the issue:
732        // <https://github.com/rust-lang/rust/issues/145151>.
733        let coroutine_args = interner.mk_args_from_iter(coroutine_args.iter().map(|arg| {
734            match arg {
735                GenericArg::Ty(_) | GenericArg::Const(_) => arg,
736                GenericArg::Lifetime(_) => {
737                    crate::next_solver::Region::new(interner, rustc_type_ir::RegionKind::ReErased)
738                        .into()
739                }
740            }
741        }));
742        Ty::new_coroutine_witness(interner, def_id, coroutine_args)
743    }
