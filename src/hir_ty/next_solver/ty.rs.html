154    /// Fast path helper for testing if a type is `Sized` or `MetaSized`.
155    ///
156    /// Returning true means the type is known to implement the sizedness trait. Returning `false`
157    /// means nothing -- could be sized, might not be.
158    ///
159    /// Note that we could never rely on the fact that a type such as `[_]` is trivially `!Sized`
160    /// because we could be in a type environment with a bound such as `[_]: Copy`. A function with
161    /// such a bound obviously never can be called, but that doesn't mean it shouldn't typecheck.
162    /// This is why this method doesn't return `Option<bool>`.
163    #[tracing::instrument(skip(tcx), level = "debug")]
164    pub fn has_trivial_sizedness(self, tcx: DbInterner<'db>, sizedness: SizedTraitKind) -> bool {
165        match self.kind() {
166            TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_))
167            | TyKind::Uint(_)
168            | TyKind::Int(_)
169            | TyKind::Bool
170            | TyKind::Float(_)
171            | TyKind::FnDef(..)
172            | TyKind::FnPtr(..)
173            | TyKind::UnsafeBinder(_)
174            | TyKind::RawPtr(..)
175            | TyKind::Char
176            | TyKind::Ref(..)
177            | TyKind::Coroutine(..)
178            | TyKind::CoroutineWitness(..)
179            | TyKind::Array(..)
180            | TyKind::Pat(..)
181            | TyKind::Closure(..)
182            | TyKind::CoroutineClosure(..)
183            | TyKind::Never
184            | TyKind::Error(_) => true,
185
186            TyKind::Str | TyKind::Slice(_) | TyKind::Dynamic(_, _) => match sizedness {
187                SizedTraitKind::Sized => false,
188                SizedTraitKind::MetaSized => true,
189            },
190
191            TyKind::Foreign(..) => match sizedness {
192                SizedTraitKind::Sized | SizedTraitKind::MetaSized => false,
193            },
194
195            TyKind::Tuple(tys) => {
196                tys.last().is_none_or(|ty| ty.has_trivial_sizedness(tcx, sizedness))
197            }
198
199            TyKind::Adt(def, args) => def
200                .sizedness_constraint(tcx, sizedness)
201                .is_none_or(|ty| ty.instantiate(tcx, args).has_trivial_sizedness(tcx, sizedness)),
202
203            TyKind::Alias(..) | TyKind::Param(_) | TyKind::Placeholder(..) | TyKind::Bound(..) => {
204                false
205            }
206
207            TyKind::Infer(InferTy::TyVar(_)) => false,
208
209            TyKind::Infer(
210                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_),
211            ) => {
212                panic!("`has_trivial_sizedness` applied to unexpected type: {self:?}")
213            }
214        }
215    }

217    /// Fast path helper for primitives which are always `Copy` and which
218    /// have a side-effect-free `Clone` impl.
219    ///
220    /// Returning true means the type is known to be pure and `Copy+Clone`.
221    /// Returning `false` means nothing -- could be `Copy`, might not be.
222    ///
223    /// This is mostly useful for optimizations, as these are the types
224    /// on which we can replace cloning with dereferencing.
225    pub fn is_trivially_pure_clone_copy(self) -> bool {
226        match self.kind() {
227            TyKind::Bool | TyKind::Char | TyKind::Never => true,
228
229            // These aren't even `Clone`
230            TyKind::Str | TyKind::Slice(..) | TyKind::Foreign(..) | TyKind::Dynamic(..) => false,
231
232            TyKind::Infer(InferTy::FloatVar(_) | InferTy::IntVar(_))
233            | TyKind::Int(..)
234            | TyKind::Uint(..)
235            | TyKind::Float(..) => true,
236
237            // ZST which can't be named are fine.
238            TyKind::FnDef(..) => true,
239
240            TyKind::Array(element_ty, _len) => element_ty.is_trivially_pure_clone_copy(),
241
242            // A 100-tuple isn't "trivial", so doing this only for reasonable sizes.
243            TyKind::Tuple(field_tys) => {
244                field_tys.len() <= 3 && field_tys.iter().all(Self::is_trivially_pure_clone_copy)
245            }
246
247            TyKind::Pat(ty, _) => ty.is_trivially_pure_clone_copy(),
248
249            // Sometimes traits aren't implemented for every ABI or arity,
250            // because we can't be generic over everything yet.
251            TyKind::FnPtr(..) => false,
252
253            // Definitely absolutely not copy.
254            TyKind::Ref(_, _, Mutability::Mut) => false,
255
256            // The standard library has a blanket Copy impl for shared references and raw pointers,
257            // for all unsized types.
258            TyKind::Ref(_, _, Mutability::Not) | TyKind::RawPtr(..) => true,
259
260            TyKind::Coroutine(..) | TyKind::CoroutineWitness(..) => false,
261
262            // Might be, but not "trivial" so just giving the safe answer.
263            TyKind::Adt(..) | TyKind::Closure(..) | TyKind::CoroutineClosure(..) => false,
264
265            TyKind::UnsafeBinder(_) => false,
266
267            // Needs normalization or revealing to determine, so no is the safe answer.
268            TyKind::Alias(..) => false,
269
270            TyKind::Param(..)
271            | TyKind::Placeholder(..)
272            | TyKind::Bound(..)
273            | TyKind::Infer(..)
274            | TyKind::Error(..) => false,
275        }
276    }

278    pub fn is_trivially_wf(self, tcx: DbInterner<'db>) -> bool {
279        match self.kind() {
280            TyKind::Bool
281            | TyKind::Char
282            | TyKind::Int(_)
283            | TyKind::Uint(_)
284            | TyKind::Float(_)
285            | TyKind::Str
286            | TyKind::Never
287            | TyKind::Param(_)
288            | TyKind::Placeholder(_)
289            | TyKind::Bound(..) => true,
290
291            TyKind::Slice(ty) => {
292                ty.is_trivially_wf(tcx) && ty.has_trivial_sizedness(tcx, SizedTraitKind::Sized)
293            }
294            TyKind::RawPtr(ty, _) => ty.is_trivially_wf(tcx),
295
296            TyKind::FnPtr(sig_tys, _) => {
297                sig_tys.skip_binder().inputs_and_output.iter().all(|ty| ty.is_trivially_wf(tcx))
298            }
299            TyKind::Ref(_, ty, _) => ty.is_global() && ty.is_trivially_wf(tcx),
300
301            TyKind::Infer(infer) => match infer {
302                InferTy::TyVar(_) => false,
303                InferTy::IntVar(_) | InferTy::FloatVar(_) => true,
304                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_) => true,
305            },
306
307            TyKind::Adt(_, _)
308            | TyKind::Tuple(_)
309            | TyKind::Array(..)
310            | TyKind::Foreign(_)
311            | TyKind::Pat(_, _)
312            | TyKind::FnDef(..)
313            | TyKind::UnsafeBinder(..)
314            | TyKind::Dynamic(..)
315            | TyKind::Closure(..)
316            | TyKind::CoroutineClosure(..)
317            | TyKind::Coroutine(..)
318            | TyKind::CoroutineWitness(..)
319            | TyKind::Alias(..)
320            | TyKind::Error(_) => false,
321        }
322    }

378    ///
379    /// The parameter `explicit` indicates if this is an *explicit* dereference.
380    /// Some types -- notably raw ptrs -- can only be dereferenced explicitly.

382        match self.kind() {

386            TyKind::Ref(_, ty, _) => Some(ty),
387            TyKind::RawPtr(ty, _) if explicit => Some(ty),
388            _ => None,
389        }

397    pub fn callable_sig(self, interner: DbInterner<'db>) -> Option<Binder<'db, FnSig<'db>>> {
398        match self.kind() {
399            TyKind::FnDef(callable, args) => {
400                Some(interner.fn_sig(callable).instantiate(interner, args))
401            }
402            TyKind::FnPtr(sig, hdr) => Some(sig.with(hdr)),
403            TyKind::Closure(_, closure_args) => closure_args
404                .split_closure_args_untupled()
405                .closure_sig_as_fn_ptr_ty
406                .callable_sig(interner),
407            TyKind::CoroutineClosure(coroutine_id, args) => {
408                Some(args.as_coroutine_closure().coroutine_closure_sig().map_bound(|sig| {
409                    let unit_ty = Ty::new_unit(interner);
410                    let return_ty = Ty::new_coroutine(
411                        interner,
412                        coroutine_id,
413                        CoroutineArgs::new(
414                            interner,
415                            CoroutineArgsParts {
416                                parent_args: args.as_coroutine_closure().parent_args(),
417                                kind_ty: unit_ty,
418                                resume_ty: unit_ty,
419                                yield_ty: unit_ty,
420                                return_ty: sig.return_ty,
421                                // FIXME: Deduce this from the coroutine closure's upvars.
422                                tupled_upvars_ty: unit_ty,
423                            },
424                        )
425                        .args,
426                    );
427                    FnSig {
428                        inputs_and_output: Tys::new_from_iter(
429                            interner,
430                            sig.tupled_inputs_ty
431                                .tuple_fields()
432                                .iter()
433                                .chain(std::iter::once(return_ty)),
434                        ),
435                        c_variadic: sig.c_variadic,
436                        safety: sig.safety,
437                        abi: sig.abi,
438                    }
439                }))
440            }
441            _ => None,
442        }
443    }

548            }),
549            TyKind::Float(float) => hir_def::builtin_type::BuiltinType::Float(match float {

560    // FIXME: Should this be here?
561    pub fn impl_trait_bounds(self, db: &'db dyn HirDatabase) -> Option<Vec<Clause<'db>>> {
562        let interner = DbInterner::new_with(db, None, None);
563
564        match self.kind() {
565            TyKind::Alias(AliasTyKind::Opaque, opaque_ty) => {
566                match db.lookup_intern_impl_trait_id(opaque_ty.def_id.expect_opaque_ty()) {
567                    ImplTraitId::ReturnTypeImplTrait(func, idx) => {
568                        db.return_type_impl_traits(func).map(|it| {
569                            let data =
570                                (*it).as_ref().map_bound(|rpit| &rpit.impl_traits[idx].predicates);
571                            data.iter_instantiated_copied(interner, opaque_ty.args.as_slice())
572                                .collect()
573                        })
574                    }
575                    ImplTraitId::TypeAliasImplTrait(alias, idx) => {
576                        db.type_alias_impl_traits(alias).map(|it| {
577                            let data =
578                                (*it).as_ref().map_bound(|rpit| &rpit.impl_traits[idx].predicates);
579                            data.iter_instantiated_copied(interner, opaque_ty.args.as_slice())
580                                .collect()
581                        })
582                    }
583                }
584            }
585            TyKind::Param(param) => {
586                // FIXME: We shouldn't use `param.id` here.
587                let generic_params = db.generic_params(param.id.parent());
588                let param_data = &generic_params[param.id.local_id()];
589                match param_data {
590                    TypeOrConstParamData::TypeParamData(p) => match p.provenance {
591                        TypeParamProvenance::ArgumentImplTrait => {
592                            let predicates = db
593                                .generic_predicates(param.id.parent())
594                                .instantiate_identity()
595                                .into_iter()
596                                .flatten()
597                                .filter(|wc| match wc.kind().skip_binder() {
598                                    ClauseKind::Trait(tr) => tr.self_ty() == self,
599                                    ClauseKind::Projection(pred) => pred.self_ty() == self,
600                                    ClauseKind::TypeOutlives(pred) => pred.0 == self,
601                                    _ => false,
602                                })
603                                .collect::<Vec<_>>();
604
605                            Some(predicates)
606                        }
607                        _ => None,
608                    },
609                    _ => None,
610                }
611            }
612            TyKind::Coroutine(coroutine_id, _args) => {
613                let InternedCoroutine(owner, _) = coroutine_id.0.loc(db);
614                let krate = owner.module(db).krate();
615                if let Some(future_trait) = LangItem::Future.resolve_trait(db, krate) {
616                    // This is only used by type walking.
617                    // Parameters will be walked outside, and projection predicate is not used.
618                    // So just provide the Future trait.
619                    let impl_bound = TraitRef::new(
620                        interner,
621                        future_trait.into(),
622                        GenericArgs::new_from_iter(interner, []),
623                    )
624                    .upcast(interner);
625                    Some(vec![impl_bound])
626                } else {
627                    None
628                }
629            }
630            _ => None,
631        }
632    }

634    /// FIXME: Get rid of this, it's not a good abstraction
635    pub fn equals_ctor(self, other: Ty<'db>) -> bool {
636        match (self.kind(), other.kind()) {
637            (TyKind::Adt(adt, ..), TyKind::Adt(adt2, ..)) => adt.def_id() == adt2.def_id(),
638            (TyKind::Slice(_), TyKind::Slice(_)) | (TyKind::Array(_, _), TyKind::Array(_, _)) => {
639                true
640            }
641            (TyKind::FnDef(def_id, ..), TyKind::FnDef(def_id2, ..)) => def_id == def_id2,
642            (TyKind::Alias(_, alias, ..), TyKind::Alias(_, alias2)) => {
643                alias.def_id == alias2.def_id
644            }
645            (TyKind::Foreign(ty_id, ..), TyKind::Foreign(ty_id2, ..)) => ty_id == ty_id2,
646            (TyKind::Closure(id1, _), TyKind::Closure(id2, _)) => id1 == id2,
647            (TyKind::Ref(.., mutability), TyKind::Ref(.., mutability2))
648            | (TyKind::RawPtr(.., mutability), TyKind::RawPtr(.., mutability2)) => {
649                mutability == mutability2
650            }
651            (TyKind::FnPtr(sig, hdr), TyKind::FnPtr(sig2, hdr2)) => sig == sig2 && hdr == hdr2,
652            (TyKind::Tuple(tys), TyKind::Tuple(tys2)) => tys.len() == tys2.len(),
653            (TyKind::Str, TyKind::Str)
654            | (TyKind::Never, TyKind::Never)
655            | (TyKind::Char, TyKind::Char)
656            | (TyKind::Bool, TyKind::Bool) => true,
657            (TyKind::Int(int), TyKind::Int(int2)) => int == int2,
658            (TyKind::Float(float), TyKind::Float(float2)) => float == float2,
659            _ => false,
660        }
661    }

747            TyKind::Error(guar) => guar.visit_with(visitor),

754            | TyKind::Float(_)
755            | TyKind::Infer(_)

1112    fn to_opt_closure_kind(self) -> Option<rustc_type_ir::ClosureKind> {
1113        match self.kind() {
1114            TyKind::Int(int_ty) => match int_ty {
1115                IntTy::I8 => Some(ClosureKind::Fn),
1116                IntTy::I16 => Some(ClosureKind::FnMut),
1117                IntTy::I32 => Some(ClosureKind::FnOnce),
1118                _ => unreachable!("cannot convert type `{:?}` to a closure kind", self),
1119            },
1120
1121            // "Bound" types appear in canonical queries when the
1122            // closure type is not yet known, and `Placeholder` and `Param`
1123            // may be encountered in generic `AsyncFnKindHelper` goals.
1124            TyKind::Bound(..) | TyKind::Placeholder(_) | TyKind::Param(_) | TyKind::Infer(_) => {
1125                None
1126            }
1127
1128            TyKind::Error(_) => Some(ClosureKind::Fn),
1129
1130            _ => unreachable!("cannot convert type `{:?}` to a closure kind", self),
1131        }
1132    }

1137            ClosureKind::FnMut => Ty::new(interner, TyKind::Int(IntTy::I16)),
1138            ClosureKind::FnOnce => Ty::new(interner, TyKind::Int(IntTy::I32)),
1139        }

1156    fn discriminant_ty(self, interner: DbInterner<'db>) -> <DbInterner<'db> as Interner>::Ty {
1157        match self.kind() {
1158            TyKind::Adt(adt, _) if adt.is_enum() => adt.repr().discr_type().to_ty(interner),
1159            TyKind::Coroutine(_, args) => args.as_coroutine().discr_ty(interner),
1160
1161            TyKind::Param(_) | TyKind::Alias(..) | TyKind::Infer(InferTy::TyVar(_)) => {
1162                /*
1163                let assoc_items = tcx.associated_item_def_ids(
1164                    tcx.require_lang_item(hir::LangItem::DiscriminantKind, None),
1165                );
1166                TyKind::new_projection_from_args(tcx, assoc_items[0], tcx.mk_args(&[self.into()]))
1167                */
1168                unimplemented!()
1169            }
1170
1171            TyKind::Pat(ty, _) => ty.discriminant_ty(interner),
1172
1173            TyKind::Bool
1174            | TyKind::Char
1175            | TyKind::Int(_)
1176            | TyKind::Uint(_)
1177            | TyKind::Float(_)
1178            | TyKind::Adt(..)
1179            | TyKind::Foreign(_)
1180            | TyKind::Str
1181            | TyKind::Array(..)
1182            | TyKind::Slice(_)
1183            | TyKind::RawPtr(_, _)
1184            | TyKind::Ref(..)
1185            | TyKind::FnDef(..)
1186            | TyKind::FnPtr(..)
1187            | TyKind::Dynamic(..)
1188            | TyKind::Closure(..)
1189            | TyKind::CoroutineClosure(..)
1190            | TyKind::CoroutineWitness(..)
1191            | TyKind::Never
1192            | TyKind::Tuple(_)
1193            | TyKind::Error(_)
1194            | TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_)) => {
1195                Ty::new(interner, TyKind::Uint(UintTy::U8))
1196            }
1197
1198            TyKind::Bound(..)
1199            | TyKind::Placeholder(_)
1200            | TyKind::Infer(
1201                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_),
1202            ) => {
1203                panic!(
1204                    "`dself.iter().map(|v| v.try_fold_with(folder)).collect::<Result<_, _>>()?iscriminant_ty` applied to unexpected type: {self:?}"
1205                )
1206            }
1207            TyKind::UnsafeBinder(..) => unimplemented!(),
1208        }
1209    }
1210}
