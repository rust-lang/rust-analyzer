1//! A nice interface for working with the infcx. The basic idea is to
2//! do `infcx.at(cause, param_env)`, which sets the "cause" of the
3//! operation as well as the surrounding parameter environment. Then
4//! you can do something like `.sub(a, b)` or `.eq(a, b)` to create a
5//! subtype or equality relationship respectively. The first argument
6//! is always the "expected" output from the POV of diagnostics.
7//!
8//! Examples:
9//! ```ignore (fragment)
10//!     infcx.at(cause, param_env).sub(a, b)
11//!     // requires that `a <: b`, with `a` considered the "expected" type
12//!
13//!     infcx.at(cause, param_env).sup(a, b)
14//!     // requires that `b <: a`, with `a` considered the "expected" type
15//!
16//!     infcx.at(cause, param_env).eq(a, b)
17//!     // requires that `a == b`, with `a` considered the "expected" type
18//! ```
19//! For finer-grained control, you can also do use `trace`:
20//! ```ignore (fragment)
21//!     infcx.at(...).trace(a, b).sub(&c, &d)
22//! ```
23//! This will set `a` and `b` as the "root" values for
24//! error-reporting, but actually operate on `c` and `d`. This is
25//! sometimes useful when the types of `c` and `d` are not traceable
26//! things. (That system should probably be refactored.)
27

48#[derive(Clone, Copy)]
49pub struct At<'a, 'db> {
50    pub infcx: &'a InferCtxt<'db>,
51    pub cause: &'a ObligationCause,
52    pub param_env: ParamEnv<'db>,
53}
54
55impl<'db> InferCtxt<'db> {
56    #[inline]
57    pub fn at<'a>(&'a self, cause: &'a ObligationCause, param_env: ParamEnv<'db>) -> At<'a, 'db> {
58        At { infcx: self, cause, param_env }
59    }
60
61    /// Forks the inference context, creating a new inference context with the same inference
62    /// variables in the same state. This can be used to "branch off" many tests from the same
63    /// common state.
64    pub fn fork(&self) -> Self {
65        Self {
66            interner: self.interner,
67            typing_mode: self.typing_mode,
68            inner: self.inner.clone(),
69            tainted_by_errors: self.tainted_by_errors.clone(),
70            universe: self.universe.clone(),
71            obligation_inspector: self.obligation_inspector.clone(),
72        }
73    }
74
75    /// Forks the inference context, creating a new inference context with the same inference
76    /// variables in the same state, except possibly changing the intercrate mode. This can be
77    /// used to "branch off" many tests from the same common state. Used in negative coherence.
78    pub fn fork_with_typing_mode(&self, typing_mode: TypingMode<DbInterner<'db>>) -> Self {
79        // Unlike `fork`, this invalidates all cache entries as they may depend on the
80        // typing mode.
81
82        Self {
83            interner: self.interner,
84            typing_mode,
85            inner: self.inner.clone(),
86            tainted_by_errors: self.tainted_by_errors.clone(),
87            universe: self.universe.clone(),
88            obligation_inspector: self.obligation_inspector.clone(),
89        }
90    }
91}

93pub trait ToTrace<'db>: Relate<DbInterner<'db>> {
94    fn to_trace(cause: &ObligationCause, a: Self, b: Self) -> TypeTrace<'db>;
95}
96
97impl<'a, 'db> At<'a, 'db> {
98    /// Makes `actual <: expected`. For example, if type-checking a
99    /// call like `foo(x)`, where `foo: fn(i32)`, you might have
100    /// `sup(i32, x)`, since the "expected" type is the type that
101    /// appears in the signature.
102    pub fn sup<T>(self, expected: T, actual: T) -> InferResult<'db, ()>
103    where
104        T: ToTrace<'db>,
105    {
106        RelateExt::relate(
107            self.infcx,
108            self.param_env,
109            expected,
110            Variance::Contravariant,
111            actual,
112            Span::dummy(),
113        )
114        .map(|goals| self.goals_to_obligations(goals))
115    }
116
117    /// Makes `expected <: actual`.
118    pub fn sub<T>(self, expected: T, actual: T) -> InferResult<'db, ()>
119    where
120        T: ToTrace<'db>,
121    {
122        RelateExt::relate(
123            self.infcx,
124            self.param_env,
125            expected,
126            Variance::Covariant,
127            actual,
128            Span::dummy(),
129        )
130        .map(|goals| self.goals_to_obligations(goals))
131    }
132
133    /// Makes `expected == actual`.
134    pub fn eq<T>(self, expected: T, actual: T) -> InferResult<'db, ()>
135    where
136        T: Relate<DbInterner<'db>>,
137    {
138        RelateExt::relate(
139            self.infcx,
140            self.param_env,
141            expected,
142            Variance::Invariant,
143            actual,
144            Span::dummy(),
145        )
146        .map(|goals| self.goals_to_obligations(goals))
147    }
148
149    pub fn relate<T>(self, expected: T, variance: Variance, actual: T) -> InferResult<'db, ()>
150    where
151        T: ToTrace<'db>,
152    {
153        match variance {
154            Variance::Covariant => self.sub(expected, actual),
155            Variance::Invariant => self.eq(expected, actual),
156            Variance::Contravariant => self.sup(expected, actual),
157
158            // We could make this make sense but it's not readily
159            // exposed and I don't feel like dealing with it. Note
160            // that bivariance in general does a bit more than just
161            // *nothing*, it checks that the types are the same
162            // "modulo variance" basically.
163            Variance::Bivariant => panic!("Bivariant given to `relate()`"),
164        }
165    }
166
167    /// Deeply normalizes `value`, replacing all aliases which can by normalized in
168    /// the current environment. This errors in case normalization fails or is ambiguous.
169    pub fn deeply_normalize<T>(self, value: T) -> Result<T, Vec<NextSolverError<'db>>>
170    where
171        T: TypeFoldable<DbInterner<'db>>,
172    {
173        crate::next_solver::normalize::deeply_normalize(self, value)
174    }
175
176    /// Computes the least-upper-bound, or mutual supertype, of two
177    /// values. The order of the arguments doesn't matter, but since
178    /// this can result in an error (e.g., if asked to compute LUB of
179    /// u32 and i32), it is meaningful to call one of them the
180    /// "expected type".
181    pub fn lub<T>(self, expected: T, actual: T) -> InferResult<'db, T>
182    where
183        T: ToTrace<'db>,
184    {
185        let mut op = LatticeOp::new(
186            self.infcx,
187            ToTrace::to_trace(self.cause, expected, actual),
188            self.param_env,
189            LatticeOpKind::Lub,
190        );
191        let value = op.relate(expected, actual)?;
192        Ok(InferOk { value, obligations: op.into_obligations() })
193    }
194
195    fn goals_to_obligations(&self, goals: Vec<Goal<'db, Predicate<'db>>>) -> InferOk<'db, ()> {
196        InferOk {
197            value: (),
198            obligations: goals
199                .into_iter()
200                .map(|goal| {
201                    Obligation::new(
202                        self.infcx.interner,
203                        self.cause.clone(),
204                        goal.param_env,
205                        goal.predicate,
206                    )
207                })
208                .collect(),
209        }
210    }
211}

237impl<'db> ToTrace<'db> for GenericArg<'db> {
238    fn to_trace(cause: &ObligationCause, a: Self, b: Self) -> TypeTrace<'db> {
239        TypeTrace {
240            cause: cause.clone(),
241            values: match (a.kind(), b.kind()) {
242                (GenericArgKind::Lifetime(a), GenericArgKind::Lifetime(b)) => {
243                    ValuePairs::Regions(ExpectedFound::new(a, b))
244                }
245                (GenericArgKind::Type(a), GenericArgKind::Type(b)) => {
246                    ValuePairs::Terms(ExpectedFound::new(a.into(), b.into()))
247                }
248                (GenericArgKind::Const(a), GenericArgKind::Const(b)) => {
249                    ValuePairs::Terms(ExpectedFound::new(a.into(), b.into()))
250                }
251                _ => panic!("relating different kinds: {a:?} {b:?}"),
252            },
253        }
254    }
255}
