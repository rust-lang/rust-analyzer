82/// This type contains all the things within `InferCtxt` that sit within a
83/// `RefCell` and are involved with taking/rolling back snapshots. Snapshot
84/// operations are hot enough that we want only one call to `borrow_mut` per
85/// call to `start_snapshot` and `rollback_to`.
86#[derive(Clone)]
87pub struct InferCtxtInner<'db> {
88    pub(crate) undo_log: InferCtxtUndoLogs<'db>,
89
90    /// We instantiate `UnificationTable` with `bounds<Ty>` because the types
91    /// that might instantiate a general type variable have an order,
92    /// represented by its upper and lower bounds.
93    pub(crate) type_variable_storage: type_variable::TypeVariableStorage<'db>,
94
95    /// Map from const parameter variable to the kind of const it represents.
96    pub(crate) const_unification_storage: ut::UnificationTableStorage<ConstVidKey<'db>>,
97
98    /// Map from integral variable to the kind of integer it represents.
99    pub(crate) int_unification_storage: ut::UnificationTableStorage<IntVid>,
100
101    /// Map from floating variable to the kind of float it represents.
102    pub(crate) float_unification_storage: ut::UnificationTableStorage<FloatVid>,
103
104    /// Tracks the set of region variables and the constraints between them.
105    ///
106    /// This is initially `Some(_)` but when
107    /// `resolve_regions_and_report_errors` is invoked, this gets set to `None`
108    /// -- further attempts to perform unification, etc., may fail if new
109    /// region constraints would've been added.
110    pub(crate) region_constraint_storage: Option<RegionConstraintStorage<'db>>,
111
112    /// A set of constraints that regionck must validate.
113    ///
114    /// Each constraint has the form `T:'a`, meaning "some type `T` must
115    /// outlive the lifetime 'a". These constraints derive from
116    /// instantiated type parameters. So if you had a struct defined
117    /// like the following:
118    /// ```ignore (illustrative)
119    /// struct Foo<T: 'static> { ... }
120    /// ```
121    /// In some expression `let x = Foo { ... }`, it will
122    /// instantiate the type parameter `T` with a fresh type `$0`. At
123    /// the same time, it will record a region obligation of
124    /// `$0: 'static`. This will get checked later by regionck. (We
125    /// can't generally check these things right away because we have
126    /// to wait until types are resolved.)
127    ///
128    /// These are stored in a map keyed to the id of the innermost
129    /// enclosing fn body / static initializer expression. This is
130    /// because the location where the obligation was incurred can be
131    /// relevant with respect to which sublifetime assumptions are in
132    /// place. The reason that we store under the fn-id, and not
133    /// something more fine-grained, is so that it is easier for
134    /// regionck to be sure that it has found *all* the region
135    /// obligations (otherwise, it's easy to fail to walk to a
136    /// particular node-id).
137    ///
138    /// Before running `resolve_regions_and_report_errors`, the creator
139    /// of the inference context is expected to invoke
140    /// [`InferCtxt::process_registered_region_obligations`]
141    /// for each body-id in this map, which will process the
142    /// obligations within. This is expected to be done 'late enough'
143    /// that all type inference variables have been bound and so forth.
144    pub(crate) region_obligations: Vec<RegionObligation<'db>>,
145
146    /// Caches for opaque type inference.
147    pub(crate) opaque_type_storage: OpaqueTypeStorage<'db>,
148}

342    /// Given a canonical value `C` as a starting point, create an
343    /// inference context that contains each of the bound values
344    /// within instantiated as a fresh variable. The `f` closure is
345    /// invoked with the new infcx, along with the instantiated value
346    /// `V` and a instantiation `S`. This instantiation `S` maps from
347    /// the bound values in `C` to their instantiated values in `V`
348    /// (in other words, `S(C) = V`).
349    pub fn build_with_canonical<T>(
350        mut self,
351        input: &CanonicalQueryInput<'db, T>,
352    ) -> (InferCtxt<'db>, T, CanonicalVarValues<'db>)
353    where

355    {
356        let infcx = self.build(input.typing_mode);
357        let (value, args) = infcx.instantiate_canonical(&input.canonical);
358        (infcx, value, args)
359    }

403    pub(crate) fn insert_type_vars<T>(&self, ty: T) -> T
404    where
405        T: TypeFoldable<DbInterner<'db>>,
406    {
407        struct Folder<'a, 'db> {
408            infcx: &'a InferCtxt<'db>,
409        }
410        impl<'db> TypeFolder<DbInterner<'db>> for Folder<'_, 'db> {
411            fn cx(&self) -> DbInterner<'db> {
412                self.infcx.interner
413            }
414
415            fn fold_ty(&mut self, ty: Ty<'db>) -> Ty<'db> {
416                if !ty.references_error() {
417                    return ty;
418                }
419
420                if ty.is_ty_error() { self.infcx.next_ty_var() } else { ty.super_fold_with(self) }
421            }
422
423            fn fold_const(&mut self, ct: Const<'db>) -> Const<'db> {
424                if !ct.references_error() {
425                    return ct;
426                }
427
428                if ct.is_ct_error() {
429                    self.infcx.next_const_var()
430                } else {
431                    ct.super_fold_with(self)
432                }
433            }
434
435            fn fold_region(&mut self, r: Region<'db>) -> Region<'db> {
436                if r.is_error() { self.infcx.next_region_var() } else { r }
437            }
438        }
439
440        ty.fold_with(&mut Folder { infcx: self })
441    }

443    /// Evaluates whether the predicate can be satisfied in the given
444    /// `ParamEnv`, and returns `false` if not certain. However, this is
445    /// not entirely accurate if inference variables are involved.
446    ///
447    /// This version may conservatively fail when outlives obligations
448    /// are required. Therefore, this version should only be used for
449    /// optimizations or diagnostics and be treated as if it can always
450    /// return `false`.
451    ///
452    /// # Example
453    ///
454    /// ```
455    /// # #![allow(dead_code)]
456    /// trait Trait {}
457    ///
458    /// fn check<T: Trait>() {}
459    ///
460    /// fn foo<T: 'static>()
461    /// where
462    ///     &'static T: Trait,
463    /// {
464    ///     // Evaluating `&'?0 T: Trait` adds a `'?0: 'static` outlives obligation,
465    ///     // which means that `predicate_must_hold_considering_regions` will return
466    ///     // `false`.
467    ///     check::<&'_ T>();
468    /// }
469    /// ```
470    #[expect(dead_code, reason = "this is used in rustc")]
471    fn predicate_must_hold_considering_regions(
472        &self,
473        obligation: &PredicateObligation<'db>,
474    ) -> bool {
475        self.evaluate_obligation(obligation).must_apply_considering_regions()
476    }

478    /// Evaluates whether the predicate can be satisfied in the given
479    /// `ParamEnv`, and returns `false` if not certain. However, this is
480    /// not entirely accurate if inference variables are involved.
481    ///
482    /// This version ignores all outlives constraints.
483    #[expect(dead_code, reason = "this is used in rustc")]
484    fn predicate_must_hold_modulo_regions(&self, obligation: &PredicateObligation<'db>) -> bool {
485        self.evaluate_obligation(obligation).must_apply_modulo_regions()
486    }

488    /// Evaluate a given predicate, capturing overflow and propagating it back.
489    fn evaluate_obligation(&self, obligation: &PredicateObligation<'db>) -> EvaluationResult {
490        self.probe(|snapshot| {
491            let mut ocx = ObligationCtxt::new(self);
492            ocx.register_obligation(obligation.clone());
493            let mut result = EvaluationResult::EvaluatedToOk;
494            for error in ocx.evaluate_obligations_error_on_ambiguity() {
495                if error.is_true_error() {
496                    return EvaluationResult::EvaluatedToErr;
497                } else {
498                    result = result.max(EvaluationResult::EvaluatedToAmbig);
499                }
500            }
501            if self.opaque_types_added_in_snapshot(snapshot) {
502                result = result.max(EvaluationResult::EvaluatedToOkModuloOpaqueTypes);
503            } else if self.region_constraints_added_in_snapshot(snapshot) {
504                result = result.max(EvaluationResult::EvaluatedToOkModuloRegions);
505            }
506            result
507        })
508    }

526    pub fn unresolved_variables(&self) -> Vec<Ty<'db>> {
527        let mut inner = self.inner.borrow_mut();
528        let mut vars: Vec<Ty<'db>> = inner
529            .type_variables()
530            .unresolved_variables()
531            .into_iter()
532            .map(|t| Ty::new_var(self.interner, t))
533            .collect();
534        vars.extend(
535            (0..inner.int_unification_table().len())
536                .map(IntVid::from_usize)
537                .filter(|&vid| inner.int_unification_table().probe_value(vid).is_unknown())
538                .map(|v| Ty::new_int_var(self.interner, v)),
539        );
540        vars.extend(
541            (0..inner.float_unification_table().len())
542                .map(FloatVid::from_usize)
543                .filter(|&vid| inner.float_unification_table().probe_value(vid).is_unknown())
544                .map(|v| Ty::new_float_var(self.interner, v)),
545        );
546        vars
547    }

554    /// Processes a `Coerce` predicate from the fulfillment context.
555    /// This is NOT the preferred way to handle coercion, which is to
556    /// invoke `FnCtxt::coerce` or a similar method (see `coercion.rs`).
557    ///
558    /// This method here is actually a fallback that winds up being
559    /// invoked when `FnCtxt::coerce` encounters unresolved type variables
560    /// and records a coercion predicate. Presently, this method is equivalent
561    /// to `subtype_predicate` -- that is, "coercing" `a` to `b` winds up
562    /// actually requiring `a <: b`. This is of course a valid coercion,
563    /// but it's not as flexible as `FnCtxt::coerce` would be.
564    ///
565    /// (We may refactor this in the future, but there are a number of
566    /// practical obstacles. Among other things, `FnCtxt::coerce` presently
567    /// records adjustments that are required on the HIR in order to perform
568    /// the coercion, and we don't currently have a way to manage that.)
569    pub fn coerce_predicate(
570        &self,
571        cause: &ObligationCause,
572        param_env: ParamEnv<'db>,
573        predicate: PolyCoercePredicate<'db>,
574    ) -> Result<InferResult<'db, ()>, (TyVid, TyVid)> {
575        let subtype_predicate = predicate.map_bound(|p| SubtypePredicate {
576            a_is_expected: false, // when coercing from `a` to `b`, `b` is expected
577            a: p.a,
578            b: p.b,
579        });
580        self.subtype_predicate(cause, param_env, subtype_predicate)
581    }

583    pub fn subtype_predicate(
584        &self,
585        cause: &ObligationCause,
586        param_env: ParamEnv<'db>,
587        predicate: PolySubtypePredicate<'db>,
588    ) -> Result<InferResult<'db, ()>, (TyVid, TyVid)> {
589        // Check for two unresolved inference variables, in which case we can
590        // make no progress. This is partly a micro-optimization, but it's
591        // also an opportunity to "sub-unify" the variables. This isn't
592        // *necessary* to prevent cycles, because they would eventually be sub-unified
593        // anyhow during generalization, but it helps with diagnostics (we can detect
594        // earlier that they are sub-unified).
595        //
596        // Note that we can just skip the binders here because
597        // type variables can't (at present, at
598        // least) capture any of the things bound by this binder.
599        //
600        // Note that this sub here is not just for diagnostics - it has semantic
601        // effects as well.
602        let r_a = self.shallow_resolve(predicate.skip_binder().a);
603        let r_b = self.shallow_resolve(predicate.skip_binder().b);
604        match (r_a.kind(), r_b.kind()) {
605            (TyKind::Infer(InferTy::TyVar(a_vid)), TyKind::Infer(InferTy::TyVar(b_vid))) => {
606                return Err((a_vid, b_vid));
607            }
608            _ => {}
609        }
610
611        self.enter_forall(predicate, |SubtypePredicate { a_is_expected, a, b }| {
612            if a_is_expected {
613                Ok(self.at(cause, param_env).sub(a, b))
614            } else {
615                Ok(self.at(cause, param_env).sup(b, a))
616            }
617        })
618    }

621        &self,
622        _cause: &traits::ObligationCause,
623        predicate: PolyRegionOutlivesPredicate<'db>,
624    ) {

769        match id {

872    pub fn shallow_resolve(&self, ty: Ty<'db>) -> Ty<'db> {
873        if let TyKind::Infer(v) = ty.kind() {
874            match v {
875                InferTy::TyVar(v) => {
876                    // Not entirely obvious: if `typ` is a type variable,
877                    // it can be resolved to an int/float variable, which
878                    // can then be recursively resolved, hence the
879                    // recursion. Note though that we prevent type
880                    // variables from unifying to other type variables
881                    // directly (though they may be embedded
882                    // structurally), and we prevent cycles in any case,
883                    // so this recursion should always be of very limited
884                    // depth.
885                    //
886                    // Note: if these two lines are combined into one we get
887                    // dynamic borrow errors on `self.inner`.
888                    let known = self.inner.borrow_mut().type_variables().probe(v).known();
889                    known.map_or(ty, |t| self.shallow_resolve(t))
890                }
891
892                InferTy::IntVar(v) => {
893                    match self.inner.borrow_mut().int_unification_table().probe_value(v) {
894                        IntVarValue::IntType(ty) => Ty::new_int(self.interner, ty),
895                        IntVarValue::UintType(ty) => Ty::new_uint(self.interner, ty),
896                        IntVarValue::Unknown => ty,
897                    }
898                }
899
900                InferTy::FloatVar(v) => {
901                    match self.inner.borrow_mut().float_unification_table().probe_value(v) {
902                        FloatVarValue::Known(ty) => Ty::new_float(self.interner, ty),
903                        FloatVarValue::Unknown => ty,
904                    }
905                }
906
907                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_) => ty,
908            }
909        } else {
910            ty
911        }
912    }

1017    // Instantiates the bound variables in a given binder with fresh inference
1018    // variables in the current universe.
1019    //
1020    // Use this method if you'd like to find some generic parameters of the binder's
1021    // variables (e.g. during a method call). If there isn't a [`BoundRegionConversionTime`]
1022    // that corresponds to your use case, consider whether or not you should
1023    // use [`InferCtxt::enter_forall`] instead.
1024    pub fn instantiate_binder_with_fresh_vars<T>(
1025        &self,
1026        _lbrct: BoundRegionConversionTime,
1027        value: Binder<'db, T>,
1028    ) -> T
1029    where
1030        T: TypeFoldable<DbInterner<'db>> + Clone,
1031    {
1032        if let Some(inner) = value.clone().no_bound_vars() {
1033            return inner;
1034        }
1035
1036        let bound_vars = value.clone().bound_vars();
1037        let mut args = Vec::with_capacity(bound_vars.len());
1038
1039        for bound_var_kind in bound_vars {
1040            let arg: GenericArg<'db> = match bound_var_kind {
1041                BoundVarKind::Ty(_) => self.next_ty_var().into(),
1042                BoundVarKind::Region(_) => self.next_region_var().into(),
1043                BoundVarKind::Const => self.next_const_var().into(),
1044            };
1045            args.push(arg);
1046        }
1047
1048        struct ToFreshVars<'db> {
1049            args: Vec<GenericArg<'db>>,
1050        }
1051
1052        impl<'db> BoundVarReplacerDelegate<'db> for ToFreshVars<'db> {
1053            fn replace_region(&mut self, br: BoundRegion) -> Region<'db> {
1054                self.args[br.var.index()].expect_region()
1055            }
1056            fn replace_ty(&mut self, bt: BoundTy) -> Ty<'db> {
1057                self.args[bt.var.index()].expect_ty()
1058            }
1059            fn replace_const(&mut self, bv: BoundConst) -> Const<'db> {
1060                self.args[bv.var.index()].expect_const()
1061            }
1062        }
1063        let delegate = ToFreshVars { args };
1064        self.interner.replace_bound_vars_uncached(value, delegate)
1065    }

1095    #[inline]
1096    pub fn is_ty_infer_var_definitely_unchanged<'a>(
1097        &'a self,
1098    ) -> impl Fn(TyOrConstInferVar) -> bool + Captures<'db> + 'a {
1099        // This hoists the borrow/release out of the loop body.
1100        let inner = self.inner.try_borrow();
1101
1102        move |infer_var: TyOrConstInferVar| match (infer_var, &inner) {
1103            (TyOrConstInferVar::Ty(ty_var), Ok(inner)) => {
1104                use self::type_variable::TypeVariableValue;
1105
1106                matches!(
1107                    inner.try_type_variables_probe_ref(ty_var),

1109                )

1115    /// `ty_or_const_infer_var_changed` is equivalent to one of these two:
1116    ///   * `shallow_resolve(ty) != ty` (where `ty.kind = Infer(_)`)
1117    ///   * `shallow_resolve(ct) != ct` (where `ct.kind = ConstKind::Infer(_)`)
1118    ///
1119    /// However, `ty_or_const_infer_var_changed` is more efficient. It's always
1120    /// inlined, despite being large, because it has only two call sites that
1121    /// are extremely hot (both in `traits::fulfill`'s checking of `stalled_on`
1122    /// inference variables), and it handles both `Ty` and `Const` without
1123    /// having to resort to storing full `GenericArg`s in `stalled_on`.
1124    #[inline(always)]
1125    pub fn ty_or_const_infer_var_changed(&self, infer_var: TyOrConstInferVar) -> bool {
1126        match infer_var {
1127            TyOrConstInferVar::Ty(v) => {
1128                use self::type_variable::TypeVariableValue;
1129
1130                // If `inlined_probe` returns a `Known` value, it never equals
1131                // `Infer(TyVar(v))`.
1132                match self.inner.borrow_mut().type_variables().inlined_probe(v) {
1133                    TypeVariableValue::Unknown { .. } => false,
1134                    TypeVariableValue::Known { .. } => true,
1135                }
1136            }
1137
1138            TyOrConstInferVar::TyInt(v) => {
1139                // If `inlined_probe_value` returns a value it's always a
1140                // `Int(_)` or `UInt(_)`, which never matches a
1141                // `Infer(_)`.
1142                self.inner.borrow_mut().int_unification_table().inlined_probe_value(v).is_known()
1143            }
1144
1145            TyOrConstInferVar::TyFloat(v) => {
1146                // If `probe_value` returns a value it's always a
1147                // `Float(_)`, which never matches a `Infer(_)`.
1148                //
1149                // Not `inlined_probe_value(v)` because this call site is colder.
1150                self.inner.borrow_mut().float_unification_table().probe_value(v).is_known()
1151            }
1152
1153            TyOrConstInferVar::Const(v) => {
1154                // If `probe_value` returns a `Known` value, it never equals
1155                // `ConstKind::Infer(InferConst::Var(v))`.
1156                //
1157                // Not `inlined_probe_value(v)` because this call site is colder.
1158                match self.inner.borrow_mut().const_unification_table().probe_value(v) {
1159                    ConstVariableValue::Unknown { .. } => false,
1160                    ConstVariableValue::Known { .. } => true,
1161                }
1162            }
1163        }
1164    }
1165
1166    fn sub_unification_table_root_var(&self, var: rustc_type_ir::TyVid) -> rustc_type_ir::TyVid {
1167        self.inner.borrow_mut().type_variables().sub_unification_table_root_var(var)
1168    }
1169
1170    fn sub_unify_ty_vids_raw(&self, a: rustc_type_ir::TyVid, b: rustc_type_ir::TyVid) {
1171        self.inner.borrow_mut().type_variables().sub_unify(a, b);
1172    }
1173}

1175/// Helper for [InferCtxt::ty_or_const_infer_var_changed] (see comment on that), currently
1176/// used only for `traits::fulfill`'s list of `stalled_on` inference variables.
1177#[derive(Copy, Clone, Debug)]
1178pub enum TyOrConstInferVar {
1179    /// Equivalent to `Infer(TyVar(_))`.
1180    Ty(TyVid),
1181    /// Equivalent to `Infer(IntVar(_))`.
1182    TyInt(IntVid),
1183    /// Equivalent to `Infer(FloatVar(_))`.
1184    TyFloat(FloatVid),
1185
1186    /// Equivalent to `ConstKind::Infer(InferConst::Var(_))`.
1187    Const(ConstVid),
1188}
1189
1190impl TyOrConstInferVar {
1191    /// Tries to extract an inference variable from a type or a constant, returns `None`
1192    /// for types other than `Infer(_)` (or `InferTy::Fresh*`) and
1193    /// for constants other than `ConstKind::Infer(_)` (or `InferConst::Fresh`).
1194    pub fn maybe_from_generic_arg<'db>(arg: GenericArg<'db>) -> Option<Self> {
1195        match arg.kind() {
1196            GenericArgKind::Type(ty) => Self::maybe_from_ty(ty),
1197            GenericArgKind::Const(ct) => Self::maybe_from_const(ct),
1198            GenericArgKind::Lifetime(_) => None,
1199        }
1200    }
1201
1202    /// Tries to extract an inference variable from a type, returns `None`
1203    /// for types other than `Infer(_)` (or `InferTy::Fresh*`).
1204    fn maybe_from_ty<'db>(ty: Ty<'db>) -> Option<Self> {
1205        match ty.kind() {
1206            TyKind::Infer(InferTy::TyVar(v)) => Some(TyOrConstInferVar::Ty(v)),
1207            TyKind::Infer(InferTy::IntVar(v)) => Some(TyOrConstInferVar::TyInt(v)),
1208            TyKind::Infer(InferTy::FloatVar(v)) => Some(TyOrConstInferVar::TyFloat(v)),
1209            _ => None,
1210        }
1211    }
1212
1213    /// Tries to extract an inference variable from a constant, returns `None`
1214    /// for constants other than `ConstKind::Infer(_)` (or `InferConst::Fresh`).
1215    fn maybe_from_const<'db>(ct: Const<'db>) -> Option<Self> {
1216        match ct.kind() {
1217            ConstKind::Infer(InferConst::Var(v)) => Some(TyOrConstInferVar::Const(v)),
1218            _ => None,
1219        }
1220    }
1221}
