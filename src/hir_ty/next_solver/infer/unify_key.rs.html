44pub(crate) struct RegionUnificationError;
45impl<'db> UnifyValue for RegionVariableValue<'db> {
46    type Error = RegionUnificationError;
47
48    fn unify_values(value1: &Self, value2: &Self) -> Result<Self, Self::Error> {
49        match (value1, value2) {
50            (RegionVariableValue::Known { .. }, RegionVariableValue::Known { .. }) => {
51                Err(RegionUnificationError)
52            }
53
54            (RegionVariableValue::Known { value }, RegionVariableValue::Unknown { universe })
55            | (RegionVariableValue::Unknown { universe }, RegionVariableValue::Known { value }) => {
56                let universe_of_value = match (*value).kind() {
57                    RegionKind::ReStatic
58                    | RegionKind::ReErased
59                    | RegionKind::ReLateParam(..)
60                    | RegionKind::ReEarlyParam(..)
61                    | RegionKind::ReError(_) => UniverseIndex::ROOT,
62                    RegionKind::RePlaceholder(placeholder) => placeholder.universe,
63                    RegionKind::ReVar(..) | RegionKind::ReBound(..) => {
64                        panic!("not a universal region")
65                    }
66                };
67
68                if universe.can_name(universe_of_value) {
69                    Ok(RegionVariableValue::Known { value: *value })
70                } else {
71                    Err(RegionUnificationError)
72                }
73            }
74
75            (
76                RegionVariableValue::Unknown { universe: a },
77                RegionVariableValue::Unknown { universe: b },
78            ) => {
79                // If we unify two unconstrained regions then whatever
80                // value they wind up taking (which must be the same value) must
81                // be nameable by both universes. Therefore, the resulting
82                // universe is the minimum of the two universes, because that is
83                // the one which contains the fewest names in scope.
84                Ok(RegionVariableValue::Unknown { universe: (*a).min(*b) })
85            }
86        }
87    }
88}

96pub(crate) enum ConstVariableValue<'db> {
97    Known { value: Const<'db> },
98    Unknown { origin: ConstVariableOrigin, universe: UniverseIndex },

101impl<'db> ConstVariableValue<'db> {
102    /// If this value is known, returns the const it is known to be.
103    /// Otherwise, `None`.
104    pub(crate) fn known(&self) -> Option<Const<'db>> {
105        match self {
106            ConstVariableValue::Unknown { .. } => None,
107            ConstVariableValue::Known { value } => Some(*value),
108        }
109    }
110}

135        "ConstVidKey"
136    }
137    fn order_roots(a: Self, _: &Self::Value, b: Self, _: &Self::Value) -> Option<(Self, Self)> {
138        if a.vid.as_u32() < b.vid.as_u32() { Some((a, b)) } else { Some((b, a)) }
139    }

142impl<'db> UnifyValue for ConstVariableValue<'db> {
143    type Error = NoError;
144
145    fn unify_values(value1: &Self, value2: &Self) -> Result<Self, Self::Error> {
146        match (value1, value2) {
147            (ConstVariableValue::Known { .. }, ConstVariableValue::Known { .. }) => {
148                panic!("equating two const variables, both of which have known values")
149            }
150
151            // If one side is known, prefer that one.
152            (ConstVariableValue::Known { .. }, ConstVariableValue::Unknown { .. }) => {
153                Ok(value1.clone())
154            }
155            (ConstVariableValue::Unknown { .. }, ConstVariableValue::Known { .. }) => {
156                Ok(value2.clone())
157            }
158
159            // If both sides are *unknown*, it hardly matters, does it?
160            (
161                ConstVariableValue::Unknown { origin, universe: universe1 },
162                ConstVariableValue::Unknown { origin: _, universe: universe2 },
163            ) => {
164                // If we unify two unbound variables, ?T and ?U, then whatever
165                // value they wind up taking (which must be the same value) must
166                // be nameable by both universes. Therefore, the resulting
167                // universe is the minimum of the two universes, because that is
168                // the one which contains the fewest names in scope.
169                let universe = cmp::min(*universe1, *universe2);
170                Ok(ConstVariableValue::Unknown { origin: *origin, universe })
171            }
172        }
173    }
174}
