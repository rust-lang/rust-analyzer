1//! # Lattice variables
2//!
3//! Generic code for operating on [lattices] of inference variables
4//! that are characterized by an upper- and lower-bound.
5//!
6//! The code is defined quite generically so that it can be
7//! applied both to type variables, which represent types being inferred,
8//! and fn variables, which represent function types being inferred.
9//! (It may eventually be applied to their types as well.)
10//! In some cases, the functions are also generic with respect to the
11//! operation on the lattice (GLB vs LUB).
12//!
13//! ## Note
14//!
15//! Although all the functions are generic, for simplicity, comments in the source code
16//! generally refer to type variables and the LUB operation.
17//!
18//! [lattices]: https://en.wikipedia.org/wiki/Lattice_(order)
19

41#[derive(Clone, Copy)]
42pub(crate) enum LatticeOpKind {
43    Glb,
44    Lub,
45}
46
47impl LatticeOpKind {
48    fn invert(self) -> Self {
49        match self {
50            LatticeOpKind::Glb => LatticeOpKind::Lub,
51            LatticeOpKind::Lub => LatticeOpKind::Glb,
52        }
53    }
54}

56/// A greatest lower bound" (common subtype) or least upper bound (common supertype).
57pub(crate) struct LatticeOp<'infcx, 'db> {
58    infcx: &'infcx InferCtxt<'db>,
59    // Immutable fields
60    trace: TypeTrace<'db>,
61    param_env: ParamEnv<'db>,
62    // Mutable fields
63    kind: LatticeOpKind,
64    obligations: PredicateObligations<'db>,
65}

82impl<'db> TypeRelation<DbInterner<'db>> for LatticeOp<'_, 'db> {
83    fn cx(&self) -> DbInterner<'db> {
84        self.infcx.interner
85    }
86
87    fn relate_ty_args(
88        &mut self,
89        a_ty: Ty<'db>,
90        b_ty: Ty<'db>,
91        def_id: SolverDefId,
92        a_args: GenericArgs<'db>,
93        b_args: GenericArgs<'db>,
94        mk: impl FnOnce(GenericArgs<'db>) -> Ty<'db>,
95    ) -> RelateResult<'db, Ty<'db>> {
96        let variances = self.cx().variances_of(def_id);
97        combine_ty_args(self.infcx, self, a_ty, b_ty, variances, a_args, b_args, mk)
98    }
99
100    fn relate_with_variance<T: Relate<DbInterner<'db>>>(
101        &mut self,
102        variance: Variance,
103        _info: VarianceDiagInfo<DbInterner<'db>>,
104        a: T,
105        b: T,
106    ) -> RelateResult<'db, T> {
107        match variance {
108            Variance::Invariant => {
109                self.obligations.extend(
110                    self.infcx.at(&self.trace.cause, self.param_env).eq(a, b)?.into_obligations(),
111                );
112                Ok(a)
113            }
114            Variance::Covariant => self.relate(a, b),
115            // FIXME(#41044) -- not correct, need test
116            Variance::Bivariant => Ok(a),
117            Variance::Contravariant => {
118                self.kind = self.kind.invert();
119                let res = self.relate(a, b);
120                self.kind = self.kind.invert();
121                res
122            }
123        }
124    }
125
126    /// Relates two types using a given lattice.
127    fn tys(&mut self, a: Ty<'db>, b: Ty<'db>) -> RelateResult<'db, Ty<'db>> {
128        if a == b {
129            return Ok(a);
130        }
131
132        let infcx = self.infcx;
133
134        let a = infcx.shallow_resolve(a);
135        let b = infcx.shallow_resolve(b);
136
137        match (a.kind(), b.kind()) {
138            // If one side is known to be a variable and one is not,
139            // create a variable (`v`) to represent the LUB. Make sure to
140            // relate `v` to the non-type-variable first (by passing it
141            // first to `relate_bound`). Otherwise, we would produce a
142            // subtype obligation that must then be processed.
143            //
144            // Example: if the LHS is a type variable, and RHS is
145            // `Box<i32>`, then we current compare `v` to the RHS first,
146            // which will instantiate `v` with `Box<i32>`. Then when `v`
147            // is compared to the LHS, we instantiate LHS with `Box<i32>`.
148            // But if we did in reverse order, we would create a `v <:
149            // LHS` (or vice versa) constraint and then instantiate
150            // `v`. This would require further processing to achieve same
151            // end-result; in particular, this screws up some of the logic
152            // in coercion, which expects LUB to figure out that the LHS
153            // is (e.g.) `Box<i32>`. A more obvious solution might be to
154            // iterate on the subtype obligations that are returned, but I
155            // think this suffices. -nmatsakis
156            (TyKind::Infer(rustc_type_ir::TyVar(..)), _) => {
157                let v = infcx.next_ty_var();
158                self.relate_bound(v, b, a)?;
159                Ok(v)
160            }
161            (_, TyKind::Infer(rustc_type_ir::TyVar(..))) => {
162                let v = infcx.next_ty_var();
163                self.relate_bound(v, a, b)?;
164                Ok(v)
165            }
166
167            (
168                TyKind::Alias(rustc_type_ir::Opaque, AliasTy { def_id: a_def_id, .. }),
169                TyKind::Alias(rustc_type_ir::Opaque, AliasTy { def_id: b_def_id, .. }),
170            ) if a_def_id == b_def_id => super_combine_tys(infcx, self, a, b),
171
172            _ => super_combine_tys(infcx, self, a, b),
173        }
174    }
175
176    fn regions(&mut self, a: Region<'db>, b: Region<'db>) -> RelateResult<'db, Region<'db>> {
177        let mut inner = self.infcx.inner.borrow_mut();
178        let mut constraints = inner.unwrap_region_constraints();
179        Ok(match self.kind {
180            // GLB(&'static u8, &'a u8) == &RegionLUB('static, 'a) u8 == &'static u8
181            LatticeOpKind::Glb => constraints.lub_regions(self.cx(), a, b),
182
183            // LUB(&'static u8, &'a u8) == &RegionGLB('static, 'a) u8 == &'a u8
184            LatticeOpKind::Lub => constraints.glb_regions(self.cx(), a, b),
185        })
186    }
187
188    fn consts(&mut self, a: Const<'db>, b: Const<'db>) -> RelateResult<'db, Const<'db>> {
189        super_combine_consts(self.infcx, self, a, b)
190    }
191
192    fn binders<T>(
193        &mut self,
194        a: Binder<'db, T>,
195        b: Binder<'db, T>,
196    ) -> RelateResult<'db, Binder<'db, T>>
197    where
198        T: Relate<DbInterner<'db>>,
199    {
200        // GLB/LUB of a binder and itself is just itself
201        if a == b {
202            return Ok(a);
203        }
204
205        if a.skip_binder().has_escaping_bound_vars() || b.skip_binder().has_escaping_bound_vars() {
206            // When higher-ranked types are involved, computing the GLB/LUB is
207            // very challenging, switch to invariance. This is obviously
208            // overly conservative but works ok in practice.
209            self.relate_with_variance(Variance::Invariant, VarianceDiagInfo::default(), a, b)?;
210            Ok(a)
211        } else {
212            Ok(Binder::dummy(self.relate(a.skip_binder(), b.skip_binder())?))
213        }
214    }
215}

217impl<'infcx, 'db> LatticeOp<'infcx, 'db> {
218    // Relates the type `v` to `a` and `b` such that `v` represents
219    // the LUB/GLB of `a` and `b` as appropriate.
220    //
221    // Subtle hack: ordering *may* be significant here. This method
222    // relates `v` to `a` first, which may help us to avoid unnecessary
223    // type variable obligations. See caller for details.
224    fn relate_bound(&mut self, v: Ty<'db>, a: Ty<'db>, b: Ty<'db>) -> RelateResult<'db, ()> {
225        let at = self.infcx.at(&self.trace.cause, self.param_env);
226        match self.kind {
227            LatticeOpKind::Glb => {
228                self.obligations.extend(at.sub(v, a)?.into_obligations());
229                self.obligations.extend(at.sub(v, b)?.into_obligations());
230            }
231            LatticeOpKind::Lub => {
232                self.obligations.extend(at.sub(a, v)?.into_obligations());
233                self.obligations.extend(at.sub(b, v)?.into_obligations());
234            }
235        }
236        Ok(())
237    }
238}

240impl<'db> PredicateEmittingRelation<InferCtxt<'db>> for LatticeOp<'_, 'db> {
241    fn span(&self) -> Span {
242        Span::dummy()
243    }
244
245    fn structurally_relate_aliases(&self) -> StructurallyRelateAliases {
246        StructurallyRelateAliases::No
247    }
248
249    fn param_env(&self) -> ParamEnv<'db> {
250        self.param_env
251    }
252
253    fn register_predicates(
254        &mut self,
255        preds: impl IntoIterator<Item: Upcast<DbInterner<'db>, Predicate<'db>>>,
256    ) {
257        self.obligations.extend(preds.into_iter().map(|pred| {
258            Obligation::new(self.infcx.interner, self.trace.cause.clone(), self.param_env, pred)
259        }))
260    }
261
262    fn register_goals(&mut self, goals: impl IntoIterator<Item = Goal<'db, Predicate<'db>>>) {
263        self.obligations.extend(goals.into_iter().map(|goal| {
264            Obligation::new(
265                self.infcx.interner,
266                self.trace.cause.clone(),
267                goal.param_env,
268                goal.predicate,
269            )
270        }))
271    }
272
273    fn register_alias_relate_predicate(&mut self, a: Ty<'db>, b: Ty<'db>) {
274        self.register_predicates([Binder::dummy(PredicateKind::AliasRelate(
275            a.into(),
276            b.into(),
277            // FIXME(deferred_projection_equality): This isn't right, I think?
278            AliasRelationDirection::Equate,
279        ))]);
280    }
