1//! # Lattice variables
2//!
3//! Generic code for operating on [lattices] of inference variables
4//! that are characterized by an upper- and lower-bound.
5//!
6//! The code is defined quite generically so that it can be
7//! applied both to type variables, which represent types being inferred,
8//! and fn variables, which represent function types being inferred.
9//! (It may eventually be applied to their types as well.)
10//! In some cases, the functions are also generic with respect to the
11//! operation on the lattice (GLB vs LUB).
12//!
13//! ## Note
14//!
15//! Although all the functions are generic, for simplicity, comments in the source code
16//! generally refer to type variables and the LUB operation.
17//!
18//! [lattices]: https://en.wikipedia.org/wiki/Lattice_(order)
19

39#[derive(Clone, Copy)]
40pub(crate) enum LatticeOpKind {
41    Glb,
42    Lub,
43}
44
45impl LatticeOpKind {
46    fn invert(self) -> Self {
47        match self {
48            LatticeOpKind::Glb => LatticeOpKind::Lub,
49            LatticeOpKind::Lub => LatticeOpKind::Glb,
50        }
51    }
52}

54/// A greatest lower bound" (common subtype) or least upper bound (common supertype).
55pub(crate) struct LatticeOp<'infcx, 'db> {
56    infcx: &'infcx InferCtxt<'db>,
57    // Immutable fields
58    trace: TypeTrace<'db>,
59    param_env: ParamEnv<'db>,
60    // Mutable fields
61    kind: LatticeOpKind,
62    obligations: PredicateObligations<'db>,
63}

65impl<'infcx, 'db> LatticeOp<'infcx, 'db> {
66    pub(crate) fn new(
67        infcx: &'infcx InferCtxt<'db>,
68        trace: TypeTrace<'db>,
69        param_env: ParamEnv<'db>,
70        kind: LatticeOpKind,
71    ) -> LatticeOp<'infcx, 'db> {

73    }

80impl<'db> TypeRelation<DbInterner<'db>> for LatticeOp<'_, 'db> {
81    fn cx(&self) -> DbInterner<'db> {
82        self.infcx.interner
83    }
84
85    fn relate_with_variance<T: Relate<DbInterner<'db>>>(
86        &mut self,
87        variance: Variance,
88        _info: VarianceDiagInfo<DbInterner<'db>>,
89        a: T,
90        b: T,
91    ) -> RelateResult<'db, T> {
92        match variance {
93            Variance::Invariant => {
94                self.obligations.extend(
95                    self.infcx.at(&self.trace.cause, self.param_env).eq(a, b)?.into_obligations(),
96                );
97                Ok(a)
98            }
99            Variance::Covariant => self.relate(a, b),
100            // FIXME(#41044) -- not correct, need test
101            Variance::Bivariant => Ok(a),
102            Variance::Contravariant => {
103                self.kind = self.kind.invert();
104                let res = self.relate(a, b);
105                self.kind = self.kind.invert();
106                res
107            }
108        }
109    }
110
111    /// Relates two types using a given lattice.
112    fn tys(&mut self, a: Ty<'db>, b: Ty<'db>) -> RelateResult<'db, Ty<'db>> {
113        if a == b {
114            return Ok(a);
115        }
116
117        let infcx = self.infcx;
118
119        let a = infcx.shallow_resolve(a);
120        let b = infcx.shallow_resolve(b);
121
122        match (a.kind(), b.kind()) {
123            // If one side is known to be a variable and one is not,
124            // create a variable (`v`) to represent the LUB. Make sure to
125            // relate `v` to the non-type-variable first (by passing it
126            // first to `relate_bound`). Otherwise, we would produce a
127            // subtype obligation that must then be processed.
128            //
129            // Example: if the LHS is a type variable, and RHS is
130            // `Box<i32>`, then we current compare `v` to the RHS first,
131            // which will instantiate `v` with `Box<i32>`. Then when `v`
132            // is compared to the LHS, we instantiate LHS with `Box<i32>`.
133            // But if we did in reverse order, we would create a `v <:
134            // LHS` (or vice versa) constraint and then instantiate
135            // `v`. This would require further processing to achieve same
136            // end-result; in particular, this screws up some of the logic
137            // in coercion, which expects LUB to figure out that the LHS
138            // is (e.g.) `Box<i32>`. A more obvious solution might be to
139            // iterate on the subtype obligations that are returned, but I
140            // think this suffices. -nmatsakis
141            (TyKind::Infer(rustc_type_ir::TyVar(..)), _) => {
142                let v = infcx.next_ty_var();
143                self.relate_bound(v, b, a)?;
144                Ok(v)
145            }
146            (_, TyKind::Infer(rustc_type_ir::TyVar(..))) => {
147                let v = infcx.next_ty_var();
148                self.relate_bound(v, a, b)?;
149                Ok(v)
150            }
151
152            (
153                TyKind::Alias(rustc_type_ir::Opaque, AliasTy { def_id: a_def_id, .. }),
154                TyKind::Alias(rustc_type_ir::Opaque, AliasTy { def_id: b_def_id, .. }),
155            ) if a_def_id == b_def_id => super_combine_tys(infcx, self, a, b),
156
157            _ => super_combine_tys(infcx, self, a, b),
158        }
159    }
160
161    fn regions(&mut self, a: Region<'db>, b: Region<'db>) -> RelateResult<'db, Region<'db>> {
162        let mut inner = self.infcx.inner.borrow_mut();
163        let mut constraints = inner.unwrap_region_constraints();
164        Ok(match self.kind {
165            // GLB(&'static u8, &'a u8) == &RegionLUB('static, 'a) u8 == &'static u8
166            LatticeOpKind::Glb => constraints.lub_regions(self.cx(), a, b),
167
168            // LUB(&'static u8, &'a u8) == &RegionGLB('static, 'a) u8 == &'a u8
169            LatticeOpKind::Lub => constraints.glb_regions(self.cx(), a, b),
170        })
171    }
172
173    fn consts(&mut self, a: Const<'db>, b: Const<'db>) -> RelateResult<'db, Const<'db>> {
174        super_combine_consts(self.infcx, self, a, b)
175    }
176
177    fn binders<T>(
178        &mut self,
179        a: Binder<'db, T>,
180        b: Binder<'db, T>,
181    ) -> RelateResult<'db, Binder<'db, T>>
182    where
183        T: Relate<DbInterner<'db>>,
184    {
185        // GLB/LUB of a binder and itself is just itself
186        if a == b {
187            return Ok(a);
188        }
189
190        if a.skip_binder().has_escaping_bound_vars() || b.skip_binder().has_escaping_bound_vars() {
191            // When higher-ranked types are involved, computing the GLB/LUB is
192            // very challenging, switch to invariance. This is obviously
193            // overly conservative but works ok in practice.
194            self.relate_with_variance(Variance::Invariant, VarianceDiagInfo::default(), a, b)?;
195            Ok(a)
196        } else {
197            Ok(Binder::dummy(self.relate(a.skip_binder(), b.skip_binder())?))
198        }
199    }
200}

202impl<'infcx, 'db> LatticeOp<'infcx, 'db> {
203    // Relates the type `v` to `a` and `b` such that `v` represents
204    // the LUB/GLB of `a` and `b` as appropriate.
205    //
206    // Subtle hack: ordering *may* be significant here. This method
207    // relates `v` to `a` first, which may help us to avoid unnecessary
208    // type variable obligations. See caller for details.
209    fn relate_bound(&mut self, v: Ty<'db>, a: Ty<'db>, b: Ty<'db>) -> RelateResult<'db, ()> {
210        let at = self.infcx.at(&self.trace.cause, self.param_env);
211        match self.kind {
212            LatticeOpKind::Glb => {
213                self.obligations.extend(at.sub(v, a)?.into_obligations());
214                self.obligations.extend(at.sub(v, b)?.into_obligations());
215            }
216            LatticeOpKind::Lub => {
217                self.obligations.extend(at.sub(a, v)?.into_obligations());
218                self.obligations.extend(at.sub(b, v)?.into_obligations());
219            }
220        }
221        Ok(())
222    }
223}

225impl<'db> PredicateEmittingRelation<InferCtxt<'db>> for LatticeOp<'_, 'db> {
226    fn span(&self) -> Span {
227        Span::dummy()
228    }
229
230    fn structurally_relate_aliases(&self) -> StructurallyRelateAliases {
231        StructurallyRelateAliases::No
232    }
233
234    fn param_env(&self) -> ParamEnv<'db> {
235        self.param_env
236    }
237
238    fn register_predicates(
239        &mut self,
240        preds: impl IntoIterator<Item: Upcast<DbInterner<'db>, Predicate<'db>>>,
241    ) {
242        self.obligations.extend(preds.into_iter().map(|pred| {
243            Obligation::new(self.infcx.interner, self.trace.cause.clone(), self.param_env, pred)
244        }))
245    }
246
247    fn register_goals(&mut self, goals: impl IntoIterator<Item = Goal<'db, Predicate<'db>>>) {
248        self.obligations.extend(goals.into_iter().map(|goal| {
249            Obligation::new(
250                self.infcx.interner,
251                self.trace.cause.clone(),
252                goal.param_env,
253                goal.predicate,
254            )
255        }))
256    }
257
258    fn register_alias_relate_predicate(&mut self, a: Ty<'db>, b: Ty<'db>) {
259        self.register_predicates([Binder::dummy(PredicateKind::AliasRelate(
260            a.into(),
261            b.into(),
262            // FIXME(deferred_projection_equality): This isn't right, I think?
263            AliasRelationDirection::Equate,
264        ))]);
265    }
