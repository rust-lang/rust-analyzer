12pub struct OpaqueHiddenType<'db> {

14}
15
16impl<'db> InferCtxt<'db> {
17    /// Insert a hidden type into the opaque type storage, making sure
18    /// it hasn't previously been defined. This does not emit any
19    /// constraints and it's the responsibility of the caller to make
20    /// sure that the item bounds of the opaque are checked.
21    pub fn register_hidden_type_in_storage(
22        &self,
23        opaque_type_key: OpaqueTypeKey<'db>,
24        hidden_ty: OpaqueHiddenType<'db>,
25    ) -> Option<Ty<'db>> {
26        self.inner.borrow_mut().opaque_types().register(opaque_type_key, hidden_ty)
27    }
