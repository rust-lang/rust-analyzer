25/// When we canonicalize a value to form a query, we wind up replacing
26/// various parts of it with canonical variables. This struct stores
27/// those replaced bits to remember for when we process the query
28/// result.
29#[derive(Clone, Debug)]
30pub struct OriginalQueryValues<'db> {
31    /// Map from the universes that appear in the query to the universes in the
32    /// caller context. For all queries except `evaluate_goal` (used by Chalk),
33    /// we only ever put ROOT values into the query, so this map is very
34    /// simple.
35    pub universe_map: SmallVec<[UniverseIndex; 4]>,
36
37    /// This is equivalent to `CanonicalVarValues`, but using a
38    /// `SmallVec` yields a significant performance win.
39    pub var_values: SmallVec<[GenericArg<'db>; 8]>,
40}
41
42impl<'db> Default for OriginalQueryValues<'db> {
43    fn default() -> Self {
44        let mut universe_map = SmallVec::default();
45        universe_map.push(UniverseIndex::ROOT);
46
47        Self { universe_map, var_values: SmallVec::default() }
48    }
49}

51impl<'db> InferCtxt<'db> {
52    /// Canonicalizes a query value `V`. When we canonicalize a query,
53    /// we not only canonicalize unbound inference variables, but we
54    /// *also* replace all free regions whatsoever. So for example a
55    /// query like `T: Trait<'static>` would be canonicalized to
56    ///
57    /// ```text
58    /// T: Trait<'?0>
59    /// ```
60    ///
61    /// with a mapping M that maps `'?0` to `'static`.
62    ///
63    /// To get a good understanding of what is happening here, check
64    /// out the [chapter in the rustc dev guide][c].
65    ///
66    /// [c]: https://rust-lang.github.io/chalk/book/canonical_queries/canonicalization.html#canonicalizing-the-query
67    pub fn canonicalize_query<V>(
68        &self,
69        value: ParamEnvAnd<'db, V>,
70        query_state: &mut OriginalQueryValues<'db>,
71    ) -> CanonicalQueryInput<DbInterner<'db>, ParamEnvAnd<'db, V>>
72    where
73        V: TypeFoldable<DbInterner<'db>>,
74    {
75        let (param_env, value) = value.into_parts();
76        // FIXME(#118965): We don't canonicalize the static lifetimes that appear in the
77        // `param_env` because they are treated differently by trait selection.
78        let canonical_param_env = Canonicalizer::canonicalize(
79            param_env,
80            self,
81            self.interner,
82            &CanonicalizeFreeRegionsOtherThanStatic,
83            query_state,
84        );
85
86        let canonical = Canonicalizer::canonicalize_with_base(
87            canonical_param_env,
88            value,
89            self,
90            self.interner,
91            &CanonicalizeAllFreeRegions,
92            query_state,
93        )
94        .unchecked_map(|(param_env, value)| ParamEnvAnd { param_env, value });
95        CanonicalQueryInput { canonical, typing_mode: self.typing_mode() }
96    }
97
98    /// Canonicalizes a query *response* `V`. When we canonicalize a
99    /// query response, we only canonicalize unbound inference
100    /// variables, and we leave other free regions alone. So,
101    /// continuing with the example from `canonicalize_query`, if
102    /// there was an input query `T: Trait<'static>`, it would have
103    /// been canonicalized to
104    ///
105    /// ```text
106    /// T: Trait<'?0>
107    /// ```
108    ///
109    /// with a mapping M that maps `'?0` to `'static`. But if we found that there
110    /// exists only one possible impl of `Trait`, and it looks like
111    /// ```ignore (illustrative)
112    /// impl<T> Trait<'static> for T { .. }
113    /// ```
114    /// then we would prepare a query result R that (among other
115    /// things) includes a mapping to `'?0 := 'static`. When
116    /// canonicalizing this query result R, we would leave this
117    /// reference to `'static` alone.
118    ///
119    /// To get a good understanding of what is happening here, check
120    /// out the [chapter in the rustc dev guide][c].
121    ///
122    /// [c]: https://rust-lang.github.io/chalk/book/canonical_queries/canonicalization.html#canonicalizing-the-query-result
123    pub fn canonicalize_response<V>(&self, value: V) -> Canonical<'db, V>
124    where
125        V: TypeFoldable<DbInterner<'db>>,
126    {
127        let mut query_state = OriginalQueryValues::default();
128        Canonicalizer::canonicalize(
129            value,
130            self,
131            self.interner,
132            &CanonicalizeQueryResponse,
133            &mut query_state,
134        )
135    }
136
137    pub fn canonicalize_user_type_annotation<V>(&self, value: V) -> Canonical<'db, V>
138    where
139        V: TypeFoldable<DbInterner<'db>>,
140    {
141        let mut query_state = OriginalQueryValues::default();
142        Canonicalizer::canonicalize(
143            value,
144            self,
145            self.interner,
146            &CanonicalizeUserTypeAnnotation,
147            &mut query_state,
148        )
149    }
150}

152/// Controls how we canonicalize "free regions" that are not inference
153/// variables. This depends on what we are canonicalizing *for* --
154/// e.g., if we are canonicalizing to create a query, we want to
155/// replace those with inference variables, since we want to make a
156/// maximally general query. But if we are canonicalizing a *query
157/// response*, then we don't typically replace free regions, as they
158/// must have been introduced from other parts of the system.
159trait CanonicalizeMode {
160    fn canonicalize_free_region<'db>(
161        &self,
162        canonicalizer: &mut Canonicalizer<'_, 'db>,
163        r: Region<'db>,
164    ) -> Region<'db>;
165
166    fn any(&self) -> bool;
167
168    // Do we preserve universe of variables.
169    fn preserve_universes(&self) -> bool;
170}
171
172struct CanonicalizeQueryResponse;
173
174impl CanonicalizeMode for CanonicalizeQueryResponse {
175    fn canonicalize_free_region<'db>(
176        &self,
177        canonicalizer: &mut Canonicalizer<'_, 'db>,
178        mut r: Region<'db>,
179    ) -> Region<'db> {
180        let infcx = canonicalizer.infcx;
181
182        if let RegionKind::ReVar(vid) = r.kind() {
183            r = infcx
184                .inner
185                .borrow_mut()
186                .unwrap_region_constraints()
187                .opportunistic_resolve_var(canonicalizer.tcx, vid);
188            debug!(
189                "canonical: region var found with vid {vid:?}, \
190                     opportunistically resolved to {r:?}",
191            );
192        };
193
194        match r.kind() {
195            RegionKind::ReLateParam(_)
196            | RegionKind::ReErased
197            | RegionKind::ReStatic
198            | RegionKind::ReEarlyParam(..)
199            | RegionKind::ReError(..) => r,
200
201            RegionKind::RePlaceholder(placeholder) => canonicalizer
202                .canonical_var_for_region(CanonicalVarKind::PlaceholderRegion(placeholder), r),
203
204            RegionKind::ReVar(vid) => {
205                let universe = infcx
206                    .inner
207                    .borrow_mut()
208                    .unwrap_region_constraints()
209                    .probe_value(vid)
210                    .unwrap_err();
211                canonicalizer.canonical_var_for_region(CanonicalVarKind::Region(universe), r)
212            }
213
214            _ => {
215                // Other than `'static` or `'empty`, the query
216                // response should be executing in a fully
217                // canonicalized environment, so there shouldn't be
218                // any other region names it can come up.
219                //
220                // rust-lang/rust#57464: `impl Trait` can leak local
221                // scopes (in manner violating typeck). Therefore, use
222                // `delayed_bug` to allow type error over an ICE.
223                panic!("unexpected region in query response: `{r:?}`");
224            }
225        }
226    }
227
228    fn any(&self) -> bool {
229        false
230    }
231
232    fn preserve_universes(&self) -> bool {
233        true
234    }
235}

239impl CanonicalizeMode for CanonicalizeUserTypeAnnotation {
240    fn canonicalize_free_region<'db>(
241        &self,
242        canonicalizer: &mut Canonicalizer<'_, 'db>,
243        r: Region<'db>,
244    ) -> Region<'db> {

246            RegionKind::ReEarlyParam(_)

248            | RegionKind::ReErased

254                panic!("unexpected region in query response: `{r:?}`")
255            }
256        }
257    }
258
259    fn any(&self) -> bool {
260        false
261    }
262
263    fn preserve_universes(&self) -> bool {
264        false
265    }

308struct Canonicalizer<'cx, 'db> {
309    /// Set to `None` to disable the resolution of inference variables.
310    infcx: &'cx InferCtxt<'db>,
311    tcx: DbInterner<'db>,
312    variables: SmallVec<[CanonicalVarKind<'db>; 8]>,
313    query_state: &'cx mut OriginalQueryValues<'db>,
314    // Note that indices is only used once `var_values` is big enough to be
315    // heap-allocated.
316    indices: FxHashMap<GenericArg<'db>, BoundVar>,
317    /// Maps each `sub_unification_table_root_var` to the index of the first
318    /// variable which used it.
319    ///
320    /// This means in case two type variables have the same sub relations root,
321    /// we set the `sub_root` of the second variable to the position of the first.
322    /// Otherwise the `sub_root` of each type variable is just its own position.
323    sub_root_lookup_table: FxHashMap<TyVid, usize>,
324    canonicalize_mode: &'cx dyn CanonicalizeMode,
325    needs_canonical_flags: TypeFlags,
326
327    binder_index: DebruijnIndex,
328}

330impl<'cx, 'db> TypeFolder<DbInterner<'db>> for Canonicalizer<'cx, 'db> {
331    fn cx(&self) -> DbInterner<'db> {
332        self.tcx
333    }
334
335    fn fold_binder<T>(&mut self, t: Binder<'db, T>) -> Binder<'db, T>
336    where
337        T: TypeFoldable<DbInterner<'db>>,
338    {
339        self.binder_index.shift_in(1);
340        let t = t.super_fold_with(self);
341        self.binder_index.shift_out(1);
342        t
343    }
344
345    fn fold_region(&mut self, r: Region<'db>) -> Region<'db> {
346        match r.kind() {
347            RegionKind::ReBound(BoundVarIndexKind::Bound(..), ..) => r,
348            RegionKind::ReBound(BoundVarIndexKind::Canonical, ..) => {
349                panic!("canonicalized bound var found during canonicalization");
350            }
351
352            RegionKind::ReStatic
353            | RegionKind::ReEarlyParam(..)
354            | RegionKind::ReError(_)
355            | RegionKind::ReLateParam(_)
356            | RegionKind::RePlaceholder(..)
357            | RegionKind::ReVar(_)
358            | RegionKind::ReErased => self.canonicalize_mode.canonicalize_free_region(self, r),
359        }
360    }
361
362    fn fold_ty(&mut self, mut t: Ty<'db>) -> Ty<'db> {
363        match t.kind() {
364            TyKind::Infer(TyVar(mut vid)) => {
365                // We need to canonicalize the *root* of our ty var.
366                // This is so that our canonical response correctly reflects
367                // any equated inference vars correctly!
368                let root_vid = self.infcx.root_var(vid);
369                if root_vid != vid {
370                    t = Ty::new_var(self.tcx, root_vid);
371                    vid = root_vid;
372                }
373
374                debug!("canonical: type var found with vid {:?}", vid);
375                match self.infcx.probe_ty_var(vid) {
376                    // `t` could be a float / int variable; canonicalize that instead.
377                    Ok(t) => {
378                        debug!("(resolved to {:?})", t);
379                        self.fold_ty(t)
380                    }
381
382                    // `TyVar(vid)` is unresolved, track its universe index in the canonicalized
383                    // result.
384                    Err(mut ui) => {
385                        if !self.canonicalize_mode.preserve_universes() {
386                            // FIXME: perf problem described in #55921.
387                            ui = UniverseIndex::ROOT;
388                        }
389
390                        let sub_root = self.get_or_insert_sub_root(vid);
391                        self.canonicalize_ty_var(CanonicalVarKind::Ty { ui, sub_root }, t)
392                    }
393                }
394            }
395
396            TyKind::Infer(IntVar(vid)) => {
397                let nt = self.infcx.opportunistic_resolve_int_var(vid);
398                if nt != t {
399                    self.fold_ty(nt)
400                } else {
401                    self.canonicalize_ty_var(CanonicalVarKind::Int, t)
402                }
403            }
404            TyKind::Infer(FloatVar(vid)) => {
405                let nt = self.infcx.opportunistic_resolve_float_var(vid);
406                if nt != t {
407                    self.fold_ty(nt)
408                } else {
409                    self.canonicalize_ty_var(CanonicalVarKind::Float, t)
410                }
411            }
412
413            TyKind::Infer(
414                InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_),
415            ) => {
416                panic!("encountered a fresh type during canonicalization")
417            }
418
419            TyKind::Placeholder(mut placeholder) => {
420                if !self.canonicalize_mode.preserve_universes() {
421                    placeholder.universe = UniverseIndex::ROOT;
422                }
423                self.canonicalize_ty_var(CanonicalVarKind::PlaceholderTy(placeholder), t)
424            }
425
426            TyKind::Bound(BoundVarIndexKind::Bound(..), _) => t,
427            TyKind::Bound(BoundVarIndexKind::Canonical, ..) => {
428                panic!("canonicalized bound var found during canonicalization");
429            }
430
431            TyKind::Closure(..)
432            | TyKind::CoroutineClosure(..)
433            | TyKind::Coroutine(..)
434            | TyKind::CoroutineWitness(..)
435            | TyKind::Bool
436            | TyKind::Char
437            | TyKind::Int(..)
438            | TyKind::Uint(..)
439            | TyKind::Float(..)
440            | TyKind::Adt(..)
441            | TyKind::Str
442            | TyKind::Error(_)
443            | TyKind::Array(..)
444            | TyKind::Slice(..)
445            | TyKind::RawPtr(..)
446            | TyKind::Ref(..)
447            | TyKind::FnDef(..)
448            | TyKind::FnPtr(..)
449            | TyKind::Dynamic(..)
450            | TyKind::UnsafeBinder(_)
451            | TyKind::Never
452            | TyKind::Tuple(..)
453            | TyKind::Alias(..)
454            | TyKind::Foreign(..)
455            | TyKind::Pat(..)
456            | TyKind::Param(..) => {
457                if t.flags().intersects(self.needs_canonical_flags) {
458                    t.super_fold_with(self)
459                } else {
460                    t
461                }
462            }
463        }
464    }
465
466    fn fold_const(&mut self, mut ct: Const<'db>) -> Const<'db> {
467        match ct.kind() {
468            ConstKind::Infer(InferConst::Var(mut vid)) => {
469                // We need to canonicalize the *root* of our const var.
470                // This is so that our canonical response correctly reflects
471                // any equated inference vars correctly!
472                let root_vid = self.infcx.root_const_var(vid);
473                if root_vid != vid {
474                    ct = Const::new_var(self.tcx, root_vid);
475                    vid = root_vid;
476                }
477
478                debug!("canonical: const var found with vid {:?}", vid);
479                match self.infcx.probe_const_var(vid) {
480                    Ok(c) => {
481                        debug!("(resolved to {:?})", c);
482                        return self.fold_const(c);
483                    }
484
485                    // `ConstVar(vid)` is unresolved, track its universe index in the
486                    // canonicalized result
487                    Err(mut ui) => {
488                        if !self.canonicalize_mode.preserve_universes() {
489                            // FIXME: perf problem described in #55921.
490                            ui = UniverseIndex::ROOT;
491                        }
492                        return self.canonicalize_const_var(CanonicalVarKind::Const(ui), ct);
493                    }
494                }
495            }
496            ConstKind::Infer(InferConst::Fresh(_)) => {
497                panic!("encountered a fresh const during canonicalization")
498            }
499            ConstKind::Bound(BoundVarIndexKind::Bound(..), _) => {
500                return ct;
501            }
502            ConstKind::Bound(BoundVarIndexKind::Canonical, ..) => {
503                panic!("canonicalized bound var found during canonicalization");
504            }
505            ConstKind::Placeholder(placeholder) => {
506                return self
507                    .canonicalize_const_var(CanonicalVarKind::PlaceholderConst(placeholder), ct);
508            }
509            _ => {}
510        }
511
512        if ct.flags().intersects(self.needs_canonical_flags) {
513            ct.super_fold_with(self)
514        } else {
515            ct
516        }
517    }
518}

520impl<'cx, 'db> Canonicalizer<'cx, 'db> {
521    /// The main `canonicalize` method, shared impl of
522    /// `canonicalize_query` and `canonicalize_response`.
523    fn canonicalize<V>(
524        value: V,
525        infcx: &InferCtxt<'db>,
526        tcx: DbInterner<'db>,
527        canonicalize_region_mode: &dyn CanonicalizeMode,
528        query_state: &mut OriginalQueryValues<'db>,
529    ) -> Canonical<'db, V>
530    where
531        V: TypeFoldable<DbInterner<'db>>,
532    {
533        let base = Canonical {
534            max_universe: UniverseIndex::ROOT,
535            variables: CanonicalVars::new_from_iter(tcx, []),
536            value: (),
537        };
538        Canonicalizer::canonicalize_with_base(
539            base,
540            value,
541            infcx,
542            tcx,
543            canonicalize_region_mode,
544            query_state,
545        )
546        .unchecked_map(|((), val)| val)
547    }
548
549    fn canonicalize_with_base<U, V>(
550        base: Canonical<'db, U>,
551        value: V,
552        infcx: &InferCtxt<'db>,
553        tcx: DbInterner<'db>,
554        canonicalize_region_mode: &dyn CanonicalizeMode,
555        query_state: &mut OriginalQueryValues<'db>,
556    ) -> Canonical<'db, (U, V)>
557    where
558        V: TypeFoldable<DbInterner<'db>>,
559    {
560        let needs_canonical_flags = if canonicalize_region_mode.any() {
561            TypeFlags::HAS_INFER | TypeFlags::HAS_PLACEHOLDER | TypeFlags::HAS_FREE_REGIONS
562        } else {
563            TypeFlags::HAS_INFER | TypeFlags::HAS_PLACEHOLDER
564        };
565
566        // Fast path: nothing that needs to be canonicalized.
567        if !value.has_type_flags(needs_canonical_flags) {
568            return base.unchecked_map(|b| (b, value));
569        }
570
571        let mut canonicalizer = Canonicalizer {
572            infcx,
573            tcx,
574            canonicalize_mode: canonicalize_region_mode,
575            needs_canonical_flags,
576            variables: SmallVec::from_slice(base.variables.as_slice()),
577            query_state,
578            indices: FxHashMap::default(),
579            sub_root_lookup_table: Default::default(),
580            binder_index: DebruijnIndex::ZERO,
581        };
582        if canonicalizer.query_state.var_values.spilled() {
583            canonicalizer.indices = canonicalizer
584                .query_state
585                .var_values
586                .iter()
587                .enumerate()
588                .map(|(i, &kind)| (kind, BoundVar::from(i)))
589                .collect();
590        }
591        let out_value = value.fold_with(&mut canonicalizer);
592
593        // Once we have canonicalized `out_value`, it should not
594        // contain anything that ties it to this inference context
595        // anymore.
596        debug_assert!(!out_value.has_infer() && !out_value.has_placeholders());
597
598        let canonical_variables =
599            CanonicalVars::new_from_iter(tcx, canonicalizer.universe_canonicalized_variables());
600
601        let max_universe = canonical_variables
602            .iter()
603            .map(|cvar| cvar.universe())
604            .max()
605            .unwrap_or(UniverseIndex::ROOT);
606
607        Canonical { max_universe, variables: canonical_variables, value: (base.value, out_value) }
608    }
609
610    /// Creates a canonical variable replacing `kind` from the input,
611    /// or returns an existing variable if `kind` has already been
612    /// seen. `kind` is expected to be an unbound variable (or
613    /// potentially a free region).
614    fn canonical_var(&mut self, info: CanonicalVarKind<'db>, kind: GenericArg<'db>) -> BoundVar {
615        let Canonicalizer { variables, query_state, indices, .. } = self;
616
617        let var_values = &mut query_state.var_values;
618
619        let universe = info.universe();
620        if universe != UniverseIndex::ROOT {
621            assert!(self.canonicalize_mode.preserve_universes());
622
623            // Insert universe into the universe map. To preserve the order of the
624            // universes in the value being canonicalized, we don't update the
625            // universe in `info` until we have finished canonicalizing.
626            match query_state.universe_map.binary_search(&universe) {
627                Err(idx) => query_state.universe_map.insert(idx, universe),
628                Ok(_) => {}
629            }
630        }
631
632        // This code is hot. `variables` and `var_values` are usually small
633        // (fewer than 8 elements ~95% of the time). They are SmallVec's to
634        // avoid allocations in those cases. We also don't use `indices` to
635        // determine if a kind has been seen before until the limit of 8 has
636        // been exceeded, to also avoid allocations for `indices`.
637        if !var_values.spilled() {
638            // `var_values` is stack-allocated. `indices` isn't used yet. Do a
639            // direct linear search of `var_values`.
640            if let Some(idx) = var_values.iter().position(|&k| k == kind) {
641                // `kind` is already present in `var_values`.
642                BoundVar::new(idx)
643            } else {
644                // `kind` isn't present in `var_values`. Append it. Likewise
645                // for `info` and `variables`.
646                variables.push(info);
647                var_values.push(kind);
648                assert_eq!(variables.len(), var_values.len());
649
650                // If `var_values` has become big enough to be heap-allocated,
651                // fill up `indices` to facilitate subsequent lookups.
652                if var_values.spilled() {
653                    assert!(indices.is_empty());
654                    *indices = var_values
655                        .iter()
656                        .enumerate()
657                        .map(|(i, &kind)| (kind, BoundVar::new(i)))
658                        .collect();
659                }
660                // The cv is the index of the appended element.
661                BoundVar::new(var_values.len() - 1)
662            }
663        } else {
664            // `var_values` is large. Do a hashmap search via `indices`.
665            *indices.entry(kind).or_insert_with(|| {
666                variables.push(info);
667                var_values.push(kind);
668                assert_eq!(variables.len(), var_values.len());
669                BoundVar::new(variables.len() - 1)
670            })
671        }
672    }
673
674    fn get_or_insert_sub_root(&mut self, vid: TyVid) -> BoundVar {
675        let root_vid = self.infcx.sub_unification_table_root_var(vid);
676        let idx =
677            *self.sub_root_lookup_table.entry(root_vid).or_insert_with(|| self.variables.len());
678        BoundVar::from(idx)
679    }

681    /// Replaces the universe indexes used in `var_values` with their index in
682    /// `query_state.universe_map`. This minimizes the maximum universe used in
683    /// the canonicalized value.
684    fn universe_canonicalized_variables(self) -> SmallVec<[CanonicalVarKind<'db>; 8]> {
685        if self.query_state.universe_map.len() == 1 {
686            return self.variables;
687        }
688
689        let reverse_universe_map: FxHashMap<UniverseIndex, UniverseIndex> = self
690            .query_state
691            .universe_map
692            .iter()
693            .enumerate()
694            .map(|(idx, universe)| (*universe, UniverseIndex::from_usize(idx)))
695            .collect();
696
697        self.variables
698            .iter()
699            .map(|v| match *v {
700                CanonicalVarKind::Int | CanonicalVarKind::Float => *v,
701                CanonicalVarKind::Ty { ui, sub_root } => {
702                    CanonicalVarKind::Ty { ui: reverse_universe_map[&ui], sub_root }
703                }
704                CanonicalVarKind::Region(u) => CanonicalVarKind::Region(reverse_universe_map[&u]),
705                CanonicalVarKind::Const(u) => CanonicalVarKind::Const(reverse_universe_map[&u]),
706                CanonicalVarKind::PlaceholderTy(placeholder) => {
707                    CanonicalVarKind::PlaceholderTy(Placeholder {
708                        universe: reverse_universe_map[&placeholder.universe],
709                        ..placeholder
710                    })
711                }
712                CanonicalVarKind::PlaceholderRegion(placeholder) => {
713                    CanonicalVarKind::PlaceholderRegion(Placeholder {
714                        universe: reverse_universe_map[&placeholder.universe],
715                        ..placeholder
716                    })
717                }
718                CanonicalVarKind::PlaceholderConst(placeholder) => {
719                    CanonicalVarKind::PlaceholderConst(Placeholder {
720                        universe: reverse_universe_map[&placeholder.universe],
721                        ..placeholder
722                    })
723                }
724            })
725            .collect()
726    }

728    /// Shorthand helper that creates a canonical region variable for
729    /// `r` (always in the root universe). The reason that we always
730    /// put these variables into the root universe is because this
731    /// method is used during **query construction:** in that case, we
732    /// are taking all the regions and just putting them into the most
733    /// generic context we can. This may generate solutions that don't
734    /// fit (e.g., that equate some region variable with a placeholder
735    /// it can't name) on the caller side, but that's ok, the caller
736    /// can figure that out. In the meantime, it maximizes our
737    /// caching.
738    ///
739    /// (This works because unification never fails -- and hence trait
740    /// selection is never affected -- due to a universe mismatch.)
741    fn canonical_var_for_region_in_root_universe(&mut self, r: Region<'db>) -> Region<'db> {
742        self.canonical_var_for_region(CanonicalVarKind::Region(UniverseIndex::ROOT), r)
743    }

745    /// Creates a canonical variable (with the given `info`)
746    /// representing the region `r`; return a region referencing it.
747    fn canonical_var_for_region(
748        &mut self,
749        info: CanonicalVarKind<'db>,
750        r: Region<'db>,
751    ) -> Region<'db> {
752        let var = self.canonical_var(info, r.into());
753        Region::new_canonical_bound(self.cx(), var)
754    }

756    /// Given a type variable `ty_var` of the given kind, first check
757    /// if `ty_var` is bound to anything; if so, canonicalize
758    /// *that*. Otherwise, create a new canonical variable for
759    /// `ty_var`.
760    fn canonicalize_ty_var(&mut self, info: CanonicalVarKind<'db>, ty_var: Ty<'db>) -> Ty<'db> {
