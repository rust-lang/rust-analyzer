1//! **Canonicalization** is the key to constructing a query in the
2//! middle of type inference. Ordinarily, it is not possible to store
3//! types from type inference in query keys, because they contain
4//! references to inference variables whose lifetimes are too short
5//! and so forth. Canonicalizing a value T1 using `canonicalize_query`
6//! produces two things:
7//!
8//! - a value T2 where each unbound inference variable has been
9//!   replaced with a **canonical variable**;
10//! - a map M (of type `CanonicalVarValues`) from those canonical
11//!   variables back to the original.
12//!
13//! We can then do queries using T2. These will give back constraints
14//! on the canonical variables which can be translated, using the map
15//! M, into constraints in our source context. This process of
16//! translating the results back is done by the
17//! `instantiate_query_result` method.
18//!
19//! For a more detailed look at what is happening here, check
20//! out the [chapter in the rustc dev guide][c].
21//!
22//! [c]: https://rust-lang.github.io/chalk/book/canonical_queries/canonicalization.html
23

36impl<'db> InferCtxt<'db> {
37    /// Creates an instantiation S for the canonical value with fresh inference
38    /// variables and placeholders then applies it to the canonical value.
39    /// Returns both the instantiated result *and* the instantiation S.
40    ///
41    /// This can be invoked as part of constructing an
42    /// inference context at the start of a query (see
43    /// `InferCtxtBuilder::build_with_canonical`). It basically
44    /// brings the canonical value "into scope" within your new infcx.
45    ///
46    /// At the end of processing, the instantiation S (once
47    /// canonicalized) then represents the values that you computed
48    /// for each of the canonical inputs to your query.
49    pub fn instantiate_canonical<T>(
50        &self,
51        canonical: &Canonical<'db, T>,
52    ) -> (T, CanonicalVarValues<'db>)
53    where
54        T: TypeFoldable<DbInterner<'db>>,
55    {
56        // For each universe that is referred to in the incoming
57        // query, create a universe in our local inference context. In
58        // practice, as of this writing, all queries have no universes
59        // in them, so this code has no effect, but it is looking
60        // forward to the day when we *do* want to carry universes
61        // through into queries.
62        //
63        // Instantiate the root-universe content into the current universe,
64        // and create fresh universes for the higher universes.
65        let universes: IndexVec<UniverseIndex, _> = std::iter::once(self.universe())
66            .chain((1..=canonical.max_universe.as_u32()).map(|_| self.create_next_universe()))
67            .collect();
68
69        let var_values = CanonicalVarValues::instantiate(
70            self.interner,
71            canonical.variables,
72            |var_values, info| self.instantiate_canonical_var(info, var_values, |ui| universes[ui]),
73        );
74        let result = canonical.instantiate(self.interner, &var_values);
75        (result, var_values)
76    }
77
78    /// Given the "info" about a canonical variable, creates a fresh
79    /// variable for it. If this is an existentially quantified
80    /// variable, then you'll get a new inference variable; if it is a
81    /// universally quantified variable, you get a placeholder.
82    ///
83    /// FIXME(-Znext-solver): This is public because it's used by the
84    /// new trait solver which has a different canonicalization routine.
85    /// We should somehow deduplicate all of this.
86    pub fn instantiate_canonical_var(
87        &self,
88        cv_info: CanonicalVarKind<DbInterner<'db>>,
89        previous_var_values: &[GenericArg<'db>],
90        universe_map: impl Fn(UniverseIndex) -> UniverseIndex,
91    ) -> GenericArg<'db> {
92        match cv_info {
93            CanonicalVarKind::Ty { ui, sub_root } => {
94                let vid = self.next_ty_var_id_in_universe(universe_map(ui));
95                // If this inference variable is related to an earlier variable
96                // via subtyping, we need to add that info to the inference context.
97                if let Some(prev) = previous_var_values.get(sub_root.as_usize()) {
98                    if let TyKind::Infer(InferTy::TyVar(sub_root)) = prev.expect_ty().kind() {
99                        self.sub_unify_ty_vids_raw(vid, sub_root);
100                    } else {
101                        unreachable!()
102                    }
103                }
104                Ty::new_var(self.interner, vid).into()
105            }
106
107            CanonicalVarKind::Int => self.next_int_var().into(),
108
109            CanonicalVarKind::Float => self.next_float_var().into(),
110
111            CanonicalVarKind::PlaceholderTy(PlaceholderTy { universe, bound }) => {
112                let universe_mapped = universe_map(universe);
113                let placeholder_mapped = PlaceholderTy { universe: universe_mapped, bound };
114                Ty::new_placeholder(self.interner, placeholder_mapped).into()
115            }
116
117            CanonicalVarKind::Region(ui) => {
118                self.next_region_var_in_universe(universe_map(ui)).into()
119            }
120
121            CanonicalVarKind::PlaceholderRegion(PlaceholderRegion { universe, bound }) => {
122                let universe_mapped = universe_map(universe);
123                let placeholder_mapped: crate::next_solver::Placeholder<
124                    crate::next_solver::BoundRegion,
125                > = PlaceholderRegion { universe: universe_mapped, bound };
126                Region::new_placeholder(self.interner, placeholder_mapped).into()
127            }
128
129            CanonicalVarKind::Const(ui) => self.next_const_var_in_universe(universe_map(ui)).into(),
130            CanonicalVarKind::PlaceholderConst(PlaceholderConst { universe, bound }) => {
131                let universe_mapped = universe_map(universe);
132                let placeholder_mapped = PlaceholderConst { universe: universe_mapped, bound };
133                Const::new_placeholder(self.interner, placeholder_mapped).into()
134            }
135        }
136    }
137}
