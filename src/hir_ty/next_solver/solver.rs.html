128    fn add_item_bounds_for_hidden_type(
129        &self,
130        def_id: SolverDefId,
131        args: GenericArgs<'db>,
132        param_env: ParamEnv<'db>,
133        hidden_ty: Ty<'db>,
134        goals: &mut Vec<Goal<'db, Predicate<'db>>>,
135    ) {
136        let interner = self.interner;
137        let opaque_id = def_id.expect_opaque_ty();
138        // Require that the hidden type is well-formed. We have to
139        // make sure we wf-check the hidden type to fix #114728.
140        //
141        // However, we don't check that all types are well-formed.
142        // We only do so for types provided by the user or if they are
143        // "used", e.g. for method selection.
144        //
145        // This means we never check the wf requirements of the hidden
146        // type during MIR borrowck, causing us to infer the wrong
147        // lifetime for its member constraints which then results in
148        // unexpected region errors.
149        goals.push(Goal::new(interner, param_env, ClauseKind::WellFormed(hidden_ty.into())));
150
151        let replace_opaques_in = |clause: Clause<'db>| {
152            fold_tys(interner, clause, |ty| match ty.kind() {
153                // Replace all other mentions of the same opaque type with the hidden type,
154                // as the bounds must hold on the hidden type after all.
155                TyKind::Alias(
156                    AliasTyKind::Opaque,
157                    AliasTy { def_id: def_id2, args: args2, .. },
158                ) if def_id == def_id2 && args == args2 => hidden_ty,
159                _ => ty,
160            })
161        };
162
163        let item_bounds = opaque_id.predicates(interner.db);
164        for predicate in item_bounds.iter_instantiated_copied(interner, args.as_slice()) {
165            let predicate = replace_opaques_in(predicate);
166
167            // Require that the predicate holds for the concrete type.
168            debug!(?predicate);
169            goals.push(Goal::new(interner, param_env, predicate));
170        }
171    }

173    fn fetch_eligible_assoc_item(
174        &self,
175        _goal_trait_ref: rustc_type_ir::TraitRef<Self::Interner>,
176        trait_assoc_def_id: SolverDefId,
177        impl_id: AnyImplId,
178    ) -> Result<Option<SolverDefId>, ErrorGuaranteed> {
179        let AnyImplId::ImplId(impl_id) = impl_id else {
180            // Builtin derive traits don't have type/consts assoc items.
181            return Ok(None);
182        };
183        let impl_items = impl_id.impl_items(self.0.interner.db());
184        let id =
185            match trait_assoc_def_id {
186                SolverDefId::TypeAliasId(trait_assoc_id) => {
187                    let trait_assoc_data = self.0.interner.db.type_alias_signature(trait_assoc_id);
188                    impl_items
189                        .items
190                        .iter()
191                        .find_map(|(impl_assoc_name, impl_assoc_id)| {
192                            if let AssocItemId::TypeAliasId(impl_assoc_id) = *impl_assoc_id
193                                && *impl_assoc_name == trait_assoc_data.name
194                            {
195                                Some(impl_assoc_id)
196                            } else {
197                                None
198                            }
199                        })
200                        .or_else(|| {
201                            if trait_assoc_data.ty.is_some() { Some(trait_assoc_id) } else { None }
202                        })
203                        .map(SolverDefId::TypeAliasId)
204                }
205                SolverDefId::ConstId(trait_assoc_id) => {
206                    let trait_assoc_data = self.0.interner.db.const_signature(trait_assoc_id);
207                    let trait_assoc_name = trait_assoc_data
208                        .name
209                        .as_ref()
210                        .expect("unnamed consts should not get passed to the solver");
211                    impl_items
212                        .items
213                        .iter()
214                        .find_map(|(impl_assoc_name, impl_assoc_id)| {
215                            if let AssocItemId::ConstId(impl_assoc_id) = *impl_assoc_id
216                                && impl_assoc_name == trait_assoc_name
217                            {
218                                Some(impl_assoc_id)
219                            } else {
220                                None
221                            }
222                        })
223                        .or_else(|| {
224                            if trait_assoc_data.has_body() { Some(trait_assoc_id) } else { None }
225                        })
226                        .map(SolverDefId::ConstId)
227                }
228                _ => panic!("Unexpected SolverDefId"),
229            };
230        Ok(id)
231    }

244    fn evaluate_const(
245        &self,
246        _param_env: ParamEnv<'db>,
247        uv: rustc_type_ir::UnevaluatedConst<Self::Interner>,
248    ) -> Option<<Self::Interner as rustc_type_ir::Interner>::Const> {
249        match uv.def.0 {
250            GeneralConstId::ConstId(c) => {
251                let subst = uv.args;
252                let ec = self.cx().db.const_eval(c, subst, None).ok()?;
253                Some(ec)
254            }
255            GeneralConstId::StaticId(c) => {
256                let ec = self.cx().db.const_eval_static(c).ok()?;
257                Some(ec)
258            }
259        }
260    }

262    fn compute_goal_fast_path(
263        &self,
264        goal: rustc_type_ir::solve::Goal<
265            Self::Interner,
266            <Self::Interner as rustc_type_ir::Interner>::Predicate,
267        >,
268        _span: <Self::Interner as rustc_type_ir::Interner>::Span,
269    ) -> Option<Certainty> {
270        if let Some(trait_pred) = goal.predicate.as_trait_clause() {
271            if self.shallow_resolve(trait_pred.self_ty().skip_binder()).is_ty_var()
272                // We don't do this fast path when opaques are defined since we may
273                // eventually use opaques to incompletely guide inference via ty var
274                // self types.
275                // FIXME: Properly consider opaques here.
276                && self.inner.borrow_mut().opaque_types().is_empty()
277            {
278                return Some(Certainty::AMBIGUOUS);
279            }
280
281            if trait_pred.polarity() == PredicatePolarity::Positive {
282                match self.0.cx().as_trait_lang_item(trait_pred.def_id()) {
283                    Some(SolverTraitLangItem::Sized) | Some(SolverTraitLangItem::MetaSized) => {
284                        let predicate = self.resolve_vars_if_possible(goal.predicate);
285                        if sizedness_fast_path(self.cx(), predicate, goal.param_env) {
286                            return Some(Certainty::Yes);
287                        }
288                    }
289                    Some(SolverTraitLangItem::Copy | SolverTraitLangItem::Clone) => {
290                        let self_ty =
291                            self.resolve_vars_if_possible(trait_pred.self_ty().skip_binder());
292                        // Unlike `Sized` traits, which always prefer the built-in impl,
293                        // `Copy`/`Clone` may be shadowed by a param-env candidate which
294                        // could force a lifetime error or guide inference. While that's
295                        // not generally desirable, it is observable, so for now let's
296                        // ignore this fast path for types that have regions or infer.
297                        if !self_ty
298                            .has_type_flags(TypeFlags::HAS_FREE_REGIONS | TypeFlags::HAS_INFER)
299                            && self_ty.is_trivially_pure_clone_copy()
300                        {
301                            return Some(Certainty::Yes);
302                        }
303                    }
304                    _ => {}
305                }
306            }
307        }
308
309        let pred = goal.predicate.kind();
310        match pred.no_bound_vars()? {
311            PredicateKind::Clause(ClauseKind::RegionOutlives(_outlives)) => Some(Certainty::Yes),
312            PredicateKind::Clause(ClauseKind::TypeOutlives(_outlives)) => Some(Certainty::Yes),
313            PredicateKind::Subtype(SubtypePredicate { a, b, .. })
314            | PredicateKind::Coerce(CoercePredicate { a, b }) => {
315                if self.shallow_resolve(a).is_ty_var() && self.shallow_resolve(b).is_ty_var() {
316                    // FIXME: We also need to register a subtype relation between these vars
317                    // when those are added, and if they aren't in the same sub root then
318                    // we should mark this goal as `has_changed`.
319                    Some(Certainty::AMBIGUOUS)
320                } else {
321                    None
322                }
323            }
324            PredicateKind::Clause(ClauseKind::ConstArgHasType(ct, _)) => {
325                if self.shallow_resolve_const(ct).is_ct_infer() {
326                    Some(Certainty::AMBIGUOUS)
327                } else {
328                    None
329                }
330            }
331            PredicateKind::Clause(ClauseKind::WellFormed(arg)) => {
332                if arg.is_trivially_wf(self.interner) {
333                    Some(Certainty::Yes)
334                } else if arg.is_infer() {
335                    Some(Certainty::AMBIGUOUS)
336                } else {
337                    None
338                }
339            }
340            _ => None,
341        }
342    }
