131    fn add_item_bounds_for_hidden_type(
132        &self,
133        def_id: SolverDefId,
134        args: GenericArgs<'db>,
135        param_env: ParamEnv<'db>,
136        hidden_ty: Ty<'db>,
137        goals: &mut Vec<Goal<'db, Predicate<'db>>>,
138    ) {
139        let interner = self.interner;
140        let opaque_id = def_id.expect_opaque_ty();
141        // Require that the hidden type is well-formed. We have to
142        // make sure we wf-check the hidden type to fix #114728.
143        //
144        // However, we don't check that all types are well-formed.
145        // We only do so for types provided by the user or if they are
146        // "used", e.g. for method selection.
147        //
148        // This means we never check the wf requirements of the hidden
149        // type during MIR borrowck, causing us to infer the wrong
150        // lifetime for its member constraints which then results in
151        // unexpected region errors.
152        goals.push(Goal::new(interner, param_env, ClauseKind::WellFormed(hidden_ty.into())));
153
154        let replace_opaques_in = |clause: Clause<'db>| {
155            fold_tys(interner, clause, |ty| match ty.kind() {
156                // Replace all other mentions of the same opaque type with the hidden type,
157                // as the bounds must hold on the hidden type after all.
158                TyKind::Alias(
159                    AliasTyKind::Opaque,
160                    AliasTy { def_id: def_id2, args: args2, .. },
161                ) if def_id == def_id2 && args == args2 => hidden_ty,
162                _ => ty,
163            })
164        };
165
166        let db = interner.db;
167        let (opaques_table, opaque_idx) = match opaque_id.loc(db) {
168            ImplTraitId::ReturnTypeImplTrait(func, opaque_idx) => {
169                (db.return_type_impl_traits(func), opaque_idx)
170            }
171            ImplTraitId::TypeAliasImplTrait(type_alias, opaque_idx) => {
172                (db.type_alias_impl_traits(type_alias), opaque_idx)
173            }
174        };
175        let item_bounds = opaques_table
176            .as_deref()
177            .unwrap()
178            .as_ref()
179            .map_bound(|table| &table.impl_traits[opaque_idx].predicates);
180        for predicate in item_bounds.iter_instantiated_copied(interner, args.as_slice()) {
181            let predicate = replace_opaques_in(predicate);
182
183            // Require that the predicate holds for the concrete type.
184            debug!(?predicate);
185            goals.push(Goal::new(interner, param_env, predicate));
186        }
187    }

189    fn fetch_eligible_assoc_item(
190        &self,
191        _goal_trait_ref: rustc_type_ir::TraitRef<Self::Interner>,
192        trait_assoc_def_id: SolverDefId,
193        impl_id: ImplIdWrapper,
194    ) -> Result<Option<SolverDefId>, ErrorGuaranteed> {
195        let impl_items = impl_id.0.impl_items(self.0.interner.db());
196        let id = match trait_assoc_def_id {
197            SolverDefId::TypeAliasId(trait_assoc_id) => {
198                let trait_assoc_data = self.0.interner.db.type_alias_signature(trait_assoc_id);
199                impl_items
200                    .items
201                    .iter()
202                    .find_map(|(impl_assoc_name, impl_assoc_id)| {
203                        if let AssocItemId::TypeAliasId(impl_assoc_id) = *impl_assoc_id
204                            && *impl_assoc_name == trait_assoc_data.name
205                        {
206                            Some(impl_assoc_id)
207                        } else {
208                            None
209                        }
210                    })
211                    .map(SolverDefId::TypeAliasId)
212            }
213            SolverDefId::ConstId(trait_assoc_id) => {
214                let trait_assoc_data = self.0.interner.db.const_signature(trait_assoc_id);
215                let trait_assoc_name = trait_assoc_data
216                    .name
217                    .as_ref()
218                    .expect("unnamed consts should not get passed to the solver");
219                impl_items
220                    .items
221                    .iter()
222                    .find_map(|(impl_assoc_name, impl_assoc_id)| {
223                        if let AssocItemId::ConstId(impl_assoc_id) = *impl_assoc_id
224                            && impl_assoc_name == trait_assoc_name
225                        {
226                            Some(impl_assoc_id)
227                        } else {
228                            None
229                        }
230                    })
231                    .map(SolverDefId::ConstId)
232            }
233            _ => panic!("Unexpected SolverDefId"),
234        };
235        Ok(id)
236    }

247    fn evaluate_const(
248        &self,
249        _param_env: ParamEnv<'db>,
250        uv: rustc_type_ir::UnevaluatedConst<Self::Interner>,
251    ) -> Option<<Self::Interner as rustc_type_ir::Interner>::Const> {
252        let c = match uv.def {
253            SolverDefId::ConstId(c) => GeneralConstId::ConstId(c),
254            SolverDefId::StaticId(c) => GeneralConstId::StaticId(c),
255            _ => unreachable!(),
256        };
257        let subst = uv.args;
258        let ec = self.cx().db.const_eval(c, subst, None).ok()?;
259        Some(ec)
260    }

262    fn compute_goal_fast_path(
263        &self,
264        goal: rustc_type_ir::solve::Goal<
265            Self::Interner,
266            <Self::Interner as rustc_type_ir::Interner>::Predicate,
267        >,
268        _span: <Self::Interner as rustc_type_ir::Interner>::Span,
269    ) -> Option<Certainty> {
270        if let Some(trait_pred) = goal.predicate.as_trait_clause() {
271            if self.shallow_resolve(trait_pred.self_ty().skip_binder()).is_ty_var()
272                // We don't do this fast path when opaques are defined since we may
273                // eventually use opaques to incompletely guide inference via ty var
274                // self types.
275                // FIXME: Properly consider opaques here.
276                && self.inner.borrow_mut().opaque_types().is_empty()
277            {
278                return Some(Certainty::AMBIGUOUS);
279            }
280
281            if trait_pred.polarity() == PredicatePolarity::Positive {
282                match self.0.cx().as_trait_lang_item(trait_pred.def_id()) {
283                    Some(SolverTraitLangItem::Sized) | Some(SolverTraitLangItem::MetaSized) => {
284                        let predicate = self.resolve_vars_if_possible(goal.predicate);
285                        if sizedness_fast_path(self.cx(), predicate, goal.param_env) {
286                            return Some(Certainty::Yes);
287                        }
288                    }
289                    Some(SolverTraitLangItem::Copy | SolverTraitLangItem::Clone) => {
290                        let self_ty =
291                            self.resolve_vars_if_possible(trait_pred.self_ty().skip_binder());
292                        // Unlike `Sized` traits, which always prefer the built-in impl,
293                        // `Copy`/`Clone` may be shadowed by a param-env candidate which
294                        // could force a lifetime error or guide inference. While that's
295                        // not generally desirable, it is observable, so for now let's
296                        // ignore this fast path for types that have regions or infer.
297                        if !self_ty
298                            .has_type_flags(TypeFlags::HAS_FREE_REGIONS | TypeFlags::HAS_INFER)
299                            && self_ty.is_trivially_pure_clone_copy()
300                        {
301                            return Some(Certainty::Yes);
302                        }
303                    }
304                    _ => {}
305                }
306            }
307        }
308
309        let pred = goal.predicate.kind();
310        match pred.no_bound_vars()? {
311            PredicateKind::Clause(ClauseKind::RegionOutlives(_outlives)) => Some(Certainty::Yes),
312            PredicateKind::Clause(ClauseKind::TypeOutlives(_outlives)) => Some(Certainty::Yes),
313            PredicateKind::Subtype(SubtypePredicate { a, b, .. })
314            | PredicateKind::Coerce(CoercePredicate { a, b }) => {
315                if self.shallow_resolve(a).is_ty_var() && self.shallow_resolve(b).is_ty_var() {
316                    // FIXME: We also need to register a subtype relation between these vars
317                    // when those are added, and if they aren't in the same sub root then
318                    // we should mark this goal as `has_changed`.
319                    Some(Certainty::AMBIGUOUS)
320                } else {
321                    None
322                }
323            }
324            PredicateKind::Clause(ClauseKind::ConstArgHasType(ct, _)) => {
325                if self.shallow_resolve_const(ct).is_ct_infer() {
326                    Some(Certainty::AMBIGUOUS)
327                } else {
328                    None
329                }
330            }
331            PredicateKind::Clause(ClauseKind::WellFormed(arg)) => {
332                if arg.is_trivially_wf(self.interner) {
333                    Some(Certainty::Yes)
334                } else if arg.is_infer() {
335                    Some(Certainty::AMBIGUOUS)
336                } else {
337                    None
338                }
339            }
340            _ => None,
341        }
342    }
