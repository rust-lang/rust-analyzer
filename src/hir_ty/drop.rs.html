17fn has_destructor(db: &dyn HirDatabase, adt: AdtId) -> bool {
18    let module = match adt {
19        AdtId::EnumId(id) => db.lookup_intern_enum(id).container,
20        AdtId::StructId(id) => db.lookup_intern_struct(id).container,
21        AdtId::UnionId(id) => db.lookup_intern_union(id).container,
22    };
23    let Some(drop_trait) =
24        db.lang_item(module.krate(), LangItem::Drop).and_then(|it| it.as_trait())
25    else {
26        return false;
27    };
28    let impls = match module.containing_block() {
29        Some(block) => match db.trait_impls_in_block(block) {
30            Some(it) => it,
31            None => return false,
32        },
33        None => db.trait_impls_in_crate(module.krate()),
34    };
35    impls.for_trait_and_self_ty(drop_trait, TyFingerprint::Adt(adt)).next().is_some()
36}

38#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
39pub enum DropGlue {
40    // Order of variants is important.
41    None,
42    /// May have a drop glue if some type parameter has it.
43    ///
44    /// For the compiler this is considered as a positive result, IDE distinguishes this from "yes".
45    DependOnParams,
46    HasDropGlue,
47}
48
49pub(crate) fn has_drop_glue(db: &dyn HirDatabase, ty: Ty, env: Arc<TraitEnvironment>) -> DropGlue {
50    match ty.kind(Interner) {
51        TyKind::Adt(adt, subst) => {
52            if has_destructor(db, adt.0) {
53                return DropGlue::HasDropGlue;
54            }
55            match adt.0 {
56                AdtId::StructId(id) => {
57                    if db.struct_signature(id).flags.contains(StructFlags::IS_MANUALLY_DROP) {
58                        return DropGlue::None;
59                    }
60                    db.field_types(id.into())
61                        .iter()
62                        .map(|(_, field_ty)| {
63                            db.has_drop_glue(
64                                field_ty.clone().substitute(Interner, subst),
65                                env.clone(),
66                            )
67                        })
68                        .max()
69                        .unwrap_or(DropGlue::None)
70                }
71                // Unions cannot have fields with destructors.
72                AdtId::UnionId(_) => DropGlue::None,
73                AdtId::EnumId(id) => db
74                    .enum_variants(id)
75                    .variants
76                    .iter()
77                    .map(|&(variant, _)| {
78                        db.field_types(variant.into())
79                            .iter()
80                            .map(|(_, field_ty)| {
81                                db.has_drop_glue(
82                                    field_ty.clone().substitute(Interner, subst),
83                                    env.clone(),
84                                )
85                            })
86                            .max()
87                            .unwrap_or(DropGlue::None)
88                    })
89                    .max()
90                    .unwrap_or(DropGlue::None),
91            }
92        }
93        TyKind::Tuple(_, subst) => subst
94            .iter(Interner)
95            .map(|ty| ty.assert_ty_ref(Interner))
96            .map(|ty| db.has_drop_glue(ty.clone(), env.clone()))
97            .max()
98            .unwrap_or(DropGlue::None),
99        TyKind::Array(ty, len) => {
100            if let ConstValue::Concrete(ConcreteConst { interned: ConstScalar::Bytes(len, _) }) =
101                &len.data(Interner).value
102            {
103                match (&**len).try_into() {
104                    Ok(len) => {
105                        let len = usize::from_le_bytes(len);
106                        if len == 0 {
107                            // Arrays of size 0 don't have drop glue.
108                            return DropGlue::None;
109                        }
110                    }
111                    Err(_) => {
112                        never!("const array size with non-usize len");
113                    }
114                }
115            }
116            db.has_drop_glue(ty.clone(), env)
117        }
118        TyKind::Slice(ty) => db.has_drop_glue(ty.clone(), env),
119        TyKind::Closure(closure_id, subst) => {
120            let owner = db.lookup_intern_closure((*closure_id).into()).0;
121            let infer = db.infer(owner);
122            let (captures, _) = infer.closure_info(closure_id);
123            let env = db.trait_environment_for_body(owner);
124            captures
125                .iter()
126                .map(|capture| db.has_drop_glue(capture.ty(subst), env.clone()))
127                .max()
128                .unwrap_or(DropGlue::None)
129        }
130        // FIXME: Handle coroutines.
131        TyKind::Coroutine(..) | TyKind::CoroutineWitness(..) => DropGlue::None,
132        TyKind::Ref(..)
133        | TyKind::Raw(..)
134        | TyKind::FnDef(..)
135        | TyKind::Str
136        | TyKind::Never
137        | TyKind::Scalar(_)
138        | TyKind::Function(_)
139        | TyKind::Foreign(_)
140        | TyKind::Error => DropGlue::None,
141        TyKind::Dyn(_) => DropGlue::HasDropGlue,
142        TyKind::AssociatedType(assoc_type_id, subst) => projection_has_drop_glue(
143            db,
144            env,
145            ProjectionTy { associated_ty_id: *assoc_type_id, substitution: subst.clone() },
146            ty,
147        ),
148        TyKind::Alias(AliasTy::Projection(projection)) => {
149            projection_has_drop_glue(db, env, projection.clone(), ty)
150        }
151        TyKind::OpaqueType(..) | TyKind::Alias(AliasTy::Opaque(_)) => {
152            if is_copy(db, ty, env) {
153                DropGlue::None
154            } else {
155                DropGlue::HasDropGlue
156            }
157        }
158        TyKind::Placeholder(_) | TyKind::BoundVar(_) => {
159            if is_copy(db, ty, env) {
160                DropGlue::None
161            } else {
162                DropGlue::DependOnParams
163            }
164        }
165        TyKind::InferenceVar(..) => unreachable!("inference vars shouldn't exist out of inference"),
166    }
167}

169fn projection_has_drop_glue(
170    db: &dyn HirDatabase,
171    env: Arc<TraitEnvironment>,
172    projection: ProjectionTy,
173    ty: Ty,
174) -> DropGlue {
175    let normalized = db.normalize_projection(projection, env.clone());
176    match normalized.kind(Interner) {
177        TyKind::Alias(AliasTy::Projection(_)) | TyKind::AssociatedType(..) => {
178            if is_copy(db, ty, env) { DropGlue::None } else { DropGlue::DependOnParams }
179        }
180        _ => db.has_drop_glue(normalized, env),
181    }
182}

184fn is_copy(db: &dyn HirDatabase, ty: Ty, env: Arc<TraitEnvironment>) -> bool {
185    let Some(copy_trait) = db.lang_item(env.krate, LangItem::Copy).and_then(|it| it.as_trait())
186    else {
187        return false;
188    };
189    let trait_ref = TyBuilder::trait_ref(db, copy_trait).push(ty).build();
190    let goal = Canonical {
191        value: InEnvironment::new(&env.env, trait_ref.cast(Interner)),
192        binders: CanonicalVarKinds::empty(Interner),
193    };
194    db.trait_solve(env.krate, env.block, goal).is_some()
195}
