67fn all_mir_bodies<'db>(
68    db: &'db dyn HirDatabase,
69    def: DefWithBodyId,
70    mut cb: impl FnMut(Arc<MirBody<'db>>),
71) -> Result<(), MirLowerError<'db>> {
72    fn for_closure<'db>(
73        db: &'db dyn HirDatabase,
74        c: InternedClosureId,
75        cb: &mut impl FnMut(Arc<MirBody<'db>>),
76    ) -> Result<(), MirLowerError<'db>> {
77        match db.mir_body_for_closure(c) {
78            Ok(body) => {
79                cb(body.clone());
80                body.closures.iter().try_for_each(|&it| for_closure(db, it, cb))
81            }
82            Err(e) => Err(e),
83        }
84    }
85    match db.mir_body(def) {
86        Ok(body) => {
87            cb(body.clone());
88            body.closures.iter().try_for_each(|&it| for_closure(db, it, &mut cb))
89        }
90        Err(e) => Err(e),
91    }
92}

94pub fn borrowck_query<'db>(
95    db: &'db dyn HirDatabase,
96    def: DefWithBodyId,
97) -> Result<Arc<[BorrowckResult<'db>]>, MirLowerError<'db>> {
98    let _p = tracing::info_span!("borrowck_query").entered();
99    let module = def.module(db);
100    let interner = DbInterner::new_with(db, Some(module.krate()), module.containing_block());
101    let env = db.trait_environment_for_body(def);
102    let mut res = vec![];
103    // This calculates opaques defining scope which is a bit costly therefore is put outside `all_mir_bodies()`.
104    let typing_mode = TypingMode::borrowck(interner, def.into());
105    all_mir_bodies(db, def, |body| {
106        // FIXME(next-solver): Opaques.
107        let infcx = interner.infer_ctxt().build(typing_mode);
108        res.push(BorrowckResult {
109            mutability_of_locals: mutability_of_locals(&infcx, &body),
110            moved_out_of_ref: moved_out_of_ref(&infcx, &env, &body),
111            partially_moved: partially_moved(&infcx, &env, &body),
112            borrow_regions: borrow_regions(db, &body),
113            mir_body: body,
114        });
115    })?;
116    Ok(res.into())
117}

119fn make_fetch_closure_field<'db>(
120    db: &'db dyn HirDatabase,
121) -> impl FnOnce(InternedClosureId, GenericArgs<'db>, usize) -> Ty<'db> + use<'db> {
122    |c: InternedClosureId, subst: GenericArgs<'db>, f: usize| {
123        let InternedClosure(def, _) = db.lookup_intern_closure(c);
124        let infer = db.infer(def);
125        let (captures, _) = infer.closure_info(c);
126        let parent_subst = subst.split_closure_args_untupled().parent_args;
127        let interner = DbInterner::new_with(db, None, None);
128        captures.get(f).expect("broken closure field").ty.instantiate(interner, parent_subst)
129    }
130}

132fn moved_out_of_ref<'db>(
133    infcx: &InferCtxt<'db>,
134    env: &TraitEnvironment<'db>,
135    body: &MirBody<'db>,
136) -> Vec<MovedOutOfRef<'db>> {
137    let db = infcx.interner.db;
138    let mut result = vec![];
139    let mut for_operand = |op: &Operand<'db>, span: MirSpan| match op.kind {
140        OperandKind::Copy(p) | OperandKind::Move(p) => {
141            let mut ty: Ty<'db> = body.locals[p.local].ty;
142            let mut is_dereference_of_ref = false;
143            for proj in p.projection.lookup(&body.projection_store) {
144                if *proj == ProjectionElem::Deref && ty.as_reference().is_some() {
145                    is_dereference_of_ref = true;
146                }
147                ty = proj.projected_ty(
148                    infcx,
149                    ty,
150                    make_fetch_closure_field(db),
151                    body.owner.module(db).krate(),
152                );
153            }
154            if is_dereference_of_ref
155                && !infcx.type_is_copy_modulo_regions(env.env, ty)
156                && !ty.references_non_lt_error()
157            {
158                result.push(MovedOutOfRef { span: op.span.unwrap_or(span), ty });
159            }
160        }
161        OperandKind::Constant { .. } | OperandKind::Static(_) => (),
162    };
163    for (_, block) in body.basic_blocks.iter() {
164        db.unwind_if_revision_cancelled();
165        for statement in &block.statements {
166            match &statement.kind {
167                StatementKind::Assign(_, r) => match r {
168                    Rvalue::ShallowInitBoxWithAlloc(_) => (),
169                    Rvalue::ShallowInitBox(o, _)
170                    | Rvalue::UnaryOp(_, o)
171                    | Rvalue::Cast(_, o, _)
172                    | Rvalue::Repeat(o, _)
173                    | Rvalue::Use(o) => for_operand(o, statement.span),
174                    Rvalue::CopyForDeref(_)
175                    | Rvalue::Discriminant(_)
176                    | Rvalue::Len(_)
177                    | Rvalue::Ref(_, _) => (),
178                    Rvalue::CheckedBinaryOp(_, o1, o2) => {
179                        for_operand(o1, statement.span);
180                        for_operand(o2, statement.span);
181                    }
182                    Rvalue::Aggregate(_, ops) => {
183                        for op in ops.iter() {
184                            for_operand(op, statement.span);
185                        }
186                    }
187                    Rvalue::ThreadLocalRef(n)
188                    | Rvalue::AddressOf(n)
189                    | Rvalue::BinaryOp(n)
190                    | Rvalue::NullaryOp(n) => match *n {},
191                },
192                StatementKind::FakeRead(_)
193                | StatementKind::Deinit(_)
194                | StatementKind::StorageLive(_)
195                | StatementKind::StorageDead(_)
196                | StatementKind::Nop => (),
197            }
198        }
199        match &block.terminator {
200            Some(terminator) => match &terminator.kind {
201                TerminatorKind::SwitchInt { discr, .. } => for_operand(discr, terminator.span),
202                TerminatorKind::FalseEdge { .. }
203                | TerminatorKind::FalseUnwind { .. }
204                | TerminatorKind::Goto { .. }
205                | TerminatorKind::UnwindResume
206                | TerminatorKind::CoroutineDrop
207                | TerminatorKind::Abort
208                | TerminatorKind::Return
209                | TerminatorKind::Unreachable
210                | TerminatorKind::Drop { .. } => (),
211                TerminatorKind::DropAndReplace { value, .. } => {
212                    for_operand(value, terminator.span);
213                }
214                TerminatorKind::Call { func, args, .. } => {
215                    for_operand(func, terminator.span);
216                    args.iter().for_each(|it| for_operand(it, terminator.span));
217                }
218                TerminatorKind::Assert { cond, .. } => {
219                    for_operand(cond, terminator.span);
220                }
221                TerminatorKind::Yield { value, .. } => {
222                    for_operand(value, terminator.span);
223                }
224            },
225            None => (),
226        }
227    }
228    result.shrink_to_fit();
229    result
230}

325fn borrow_regions<'db>(db: &'db dyn HirDatabase, body: &MirBody<'db>) -> Vec<BorrowRegion<'db>> {
326    let mut borrows = FxHashMap::default();
327    for (_, block) in body.basic_blocks.iter() {
328        db.unwind_if_revision_cancelled();
329        for statement in &block.statements {
330            if let StatementKind::Assign(_, Rvalue::Ref(kind, p)) = &statement.kind {
331                borrows
332                    .entry(p.local)
333                    .and_modify(|it: &mut BorrowRegion<'db>| {
334                        it.places.push(statement.span);
335                    })
336                    .or_insert_with(|| BorrowRegion {
337                        local: p.local,
338                        kind: *kind,
339                        places: vec![statement.span],
340                    });
341            }
342        }
343        match &block.terminator {
344            Some(terminator) => match &terminator.kind {
345                TerminatorKind::FalseEdge { .. }
346                | TerminatorKind::FalseUnwind { .. }
347                | TerminatorKind::Goto { .. }
348                | TerminatorKind::UnwindResume
349                | TerminatorKind::CoroutineDrop
350                | TerminatorKind::Abort
351                | TerminatorKind::Return
352                | TerminatorKind::Unreachable
353                | TerminatorKind::Drop { .. } => (),
354                TerminatorKind::DropAndReplace { .. } => {}
355                TerminatorKind::Call { .. } => {}
356                _ => (),
357            },
358            None => (),
359        }
360    }
361
362    borrows.into_values().collect()
363}

365#[derive(Debug, Clone, Copy, PartialEq, Eq)]
366enum ProjectionCase {
367    /// Projection is a local
368    Direct,
369    /// Projection is some field or slice of a local
370    DirectPart,
371    /// Projection is deref of something
372    Indirect,
373}

375fn place_case<'db>(
376    infcx: &InferCtxt<'db>,
377    body: &MirBody<'db>,
378    lvalue: &Place<'db>,
379) -> ProjectionCase {
380    let db = infcx.interner.db;
381    let mut is_part_of = false;
382    let mut ty = body.locals[lvalue.local].ty;
383    for proj in lvalue.projection.lookup(&body.projection_store).iter() {
384        match proj {
385            ProjectionElem::Deref if ty.as_adt().is_none() => return ProjectionCase::Indirect, // It's indirect in case of reference and raw
386            ProjectionElem::Deref // It's direct in case of `Box<T>`
387            | ProjectionElem::ConstantIndex { .. }
388            | ProjectionElem::Subslice { .. }
389            | ProjectionElem::Field(_)
390            | ProjectionElem::ClosureField(_)
391            | ProjectionElem::Index(_) => {
392                is_part_of = true;
393            }
394            ProjectionElem::OpaqueCast(_) => (),
395        }
396        ty = proj.projected_ty(
397            infcx,
398            ty,
399            make_fetch_closure_field(db),
400            body.owner.module(db).krate(),
401        );
402    }
403    if is_part_of { ProjectionCase::DirectPart } else { ProjectionCase::Direct }
404}

406/// Returns a map from basic blocks to the set of locals that might be ever initialized before
407/// the start of the block. Only `StorageDead` can remove something from this map, and we ignore
408/// `Uninit` and `drop` and similar after initialization.
409fn ever_initialized_map<'db>(
410    db: &'db dyn HirDatabase,
411    body: &MirBody<'db>,
412) -> ArenaMap<BasicBlockId<'db>, ArenaMap<LocalId<'db>, bool>> {
413    let mut result: ArenaMap<BasicBlockId<'db>, ArenaMap<LocalId<'db>, bool>> =
414        body.basic_blocks.iter().map(|it| (it.0, ArenaMap::default())).collect();
415    fn dfs<'db>(
416        db: &'db dyn HirDatabase,
417        body: &MirBody<'db>,
418        l: LocalId<'db>,
419        stack: &mut Vec<BasicBlockId<'db>>,
420        result: &mut ArenaMap<BasicBlockId<'db>, ArenaMap<LocalId<'db>, bool>>,
421    ) {
422        while let Some(b) = stack.pop() {
423            let mut is_ever_initialized = result[b][l]; // It must be filled, as we use it as mark for dfs
424            let block = &body.basic_blocks[b];
425            for statement in &block.statements {
426                match &statement.kind {
427                    StatementKind::Assign(p, _) => {
428                        if p.projection.lookup(&body.projection_store).is_empty() && p.local == l {
429                            is_ever_initialized = true;
430                        }
431                    }
432                    StatementKind::StorageDead(p) => {
433                        if *p == l {
434                            is_ever_initialized = false;
435                        }
436                    }
437                    StatementKind::Deinit(_)
438                    | StatementKind::FakeRead(_)
439                    | StatementKind::Nop
440                    | StatementKind::StorageLive(_) => (),
441                }
442            }
443            let Some(terminator) = &block.terminator else {
444                never!(
445                    "Terminator should be none only in construction.\nThe body:\n{}",
446                    body.pretty_print(db, DisplayTarget::from_crate(db, body.owner.krate(db)))
447                );
448                return;
449            };
450            let mut process = |target, is_ever_initialized| {
451                if !result[target].contains_idx(l) || !result[target][l] && is_ever_initialized {
452                    result[target].insert(l, is_ever_initialized);
453                    stack.push(target);
454                }
455            };
456            match &terminator.kind {
457                TerminatorKind::Goto { target } => process(*target, is_ever_initialized),
458                TerminatorKind::SwitchInt { targets, .. } => {
459                    targets.all_targets().iter().for_each(|&it| process(it, is_ever_initialized));
460                }
461                TerminatorKind::UnwindResume
462                | TerminatorKind::Abort
463                | TerminatorKind::Return
464                | TerminatorKind::Unreachable => (),
465                TerminatorKind::Call { target, cleanup, destination, .. } => {
466                    if destination.projection.lookup(&body.projection_store).is_empty()
467                        && destination.local == l
468                    {
469                        is_ever_initialized = true;
470                    }
471                    target.iter().chain(cleanup).for_each(|&it| process(it, is_ever_initialized));
472                }
473                TerminatorKind::Drop { target, unwind, place: _ } => {
474                    iter::once(target)
475                        .chain(unwind)
476                        .for_each(|&it| process(it, is_ever_initialized));
477                }
478                TerminatorKind::DropAndReplace { .. }
479                | TerminatorKind::Assert { .. }
480                | TerminatorKind::Yield { .. }
481                | TerminatorKind::CoroutineDrop
482                | TerminatorKind::FalseEdge { .. }
483                | TerminatorKind::FalseUnwind { .. } => {
484                    never!("We don't emit these MIR terminators yet");
485                }
486            }
487        }
488    }
489    let mut stack = Vec::new();
490    for &l in &body.param_locals {
491        result[body.start_block].insert(l, true);
492        stack.clear();
493        stack.push(body.start_block);
494        dfs(db, body, l, &mut stack, &mut result);
495    }
496    for l in body.locals.iter().map(|it| it.0) {
497        db.unwind_if_revision_cancelled();
498        if !result[body.start_block].contains_idx(l) {
499            result[body.start_block].insert(l, false);
500            stack.clear();
501            stack.push(body.start_block);
502            dfs(db, body, l, &mut stack, &mut result);
503        }
504    }
505    result
506}

508fn push_mut_span<'db>(
509    local: LocalId<'db>,
510    span: MirSpan,
511    result: &mut ArenaMap<LocalId<'db>, MutabilityReason>,
512) {
513    match &mut result[local] {
514        MutabilityReason::Mut { spans } => spans.push(span),
515        it @ (MutabilityReason::Not | MutabilityReason::Unused) => {
516            *it = MutabilityReason::Mut { spans: vec![span] }
517        }
518    };
519}

521fn record_usage<'db>(local: LocalId<'db>, result: &mut ArenaMap<LocalId<'db>, MutabilityReason>) {
522    if let it @ MutabilityReason::Unused = &mut result[local] {
523        *it = MutabilityReason::Not;
524    };
525}

527fn record_usage_for_operand<'db>(
528    arg: &Operand<'db>,
529    result: &mut ArenaMap<LocalId<'db>, MutabilityReason>,
530) {
531    if let OperandKind::Copy(p) | OperandKind::Move(p) = arg.kind {
532        record_usage(p.local, result);
533    }
534}

536fn mutability_of_locals<'db>(
537    infcx: &InferCtxt<'db>,
538    body: &MirBody<'db>,
539) -> ArenaMap<LocalId<'db>, MutabilityReason> {
540    let db = infcx.interner.db;
541    let mut result: ArenaMap<LocalId<'db>, MutabilityReason> =
542        body.locals.iter().map(|it| (it.0, MutabilityReason::Unused)).collect();
543
544    let ever_init_maps = ever_initialized_map(db, body);
545    for (block_id, mut ever_init_map) in ever_init_maps.into_iter() {
546        let block = &body.basic_blocks[block_id];
547        for statement in &block.statements {
548            match &statement.kind {
549                StatementKind::Assign(place, value) => {
550                    match place_case(infcx, body, place) {
551                        ProjectionCase::Direct => {
552                            if ever_init_map.get(place.local).copied().unwrap_or_default() {
553                                push_mut_span(place.local, statement.span, &mut result);
554                            } else {
555                                ever_init_map.insert(place.local, true);
556                            }
557                        }
558                        ProjectionCase::DirectPart => {
559                            // Partial initialization is not supported, so it is definitely `mut`
560                            push_mut_span(place.local, statement.span, &mut result);
561                        }
562                        ProjectionCase::Indirect => {
563                            record_usage(place.local, &mut result);
564                        }
565                    }
566                    match value {
567                        Rvalue::CopyForDeref(p)
568                        | Rvalue::Discriminant(p)
569                        | Rvalue::Len(p)
570                        | Rvalue::Ref(_, p) => {
571                            record_usage(p.local, &mut result);
572                        }
573                        Rvalue::Use(o)
574                        | Rvalue::Repeat(o, _)
575                        | Rvalue::Cast(_, o, _)
576                        | Rvalue::UnaryOp(_, o) => record_usage_for_operand(o, &mut result),
577                        Rvalue::CheckedBinaryOp(_, o1, o2) => {
578                            for o in [o1, o2] {
579                                record_usage_for_operand(o, &mut result);
580                            }
581                        }
582                        Rvalue::Aggregate(_, args) => {
583                            for arg in args.iter() {
584                                record_usage_for_operand(arg, &mut result);
585                            }
586                        }
587                        Rvalue::ShallowInitBox(_, _) | Rvalue::ShallowInitBoxWithAlloc(_) => (),
588                        Rvalue::ThreadLocalRef(n)
589                        | Rvalue::AddressOf(n)
590                        | Rvalue::BinaryOp(n)
591                        | Rvalue::NullaryOp(n) => match *n {},
592                    }
593                    if let Rvalue::Ref(
594                        BorrowKind::Mut {
595                            kind: MutBorrowKind::Default | MutBorrowKind::TwoPhasedBorrow,
596                        },
597                        p,
598                    ) = value
599                        && place_case(infcx, body, p) != ProjectionCase::Indirect
600                    {
601                        push_mut_span(p.local, statement.span, &mut result);
602                    }
603                }
604                StatementKind::FakeRead(p) => {
605                    record_usage(p.local, &mut result);
606                }
607                StatementKind::StorageDead(p) => {
608                    ever_init_map.insert(*p, false);
609                }
610                StatementKind::Deinit(_) | StatementKind::StorageLive(_) | StatementKind::Nop => (),
611            }
612        }
613        let Some(terminator) = &block.terminator else {
614            never!("Terminator should be none only in construction");
615            continue;
616        };
617        match &terminator.kind {
618            TerminatorKind::Goto { .. }
619            | TerminatorKind::UnwindResume
620            | TerminatorKind::Abort
621            | TerminatorKind::Return
622            | TerminatorKind::Unreachable
623            | TerminatorKind::FalseEdge { .. }
624            | TerminatorKind::FalseUnwind { .. }
625            | TerminatorKind::CoroutineDrop
626            | TerminatorKind::Drop { .. }
627            | TerminatorKind::DropAndReplace { .. }
628            | TerminatorKind::Assert { .. }
629            | TerminatorKind::Yield { .. } => (),
630            TerminatorKind::SwitchInt { discr, targets: _ } => {
631                record_usage_for_operand(discr, &mut result);
632            }
633            TerminatorKind::Call { destination, args, func, .. } => {
634                record_usage_for_operand(func, &mut result);
635                for arg in args.iter() {
636                    record_usage_for_operand(arg, &mut result);
637                }
638                if destination.projection.lookup(&body.projection_store).is_empty() {
639                    if ever_init_map.get(destination.local).copied().unwrap_or_default() {
640                        push_mut_span(destination.local, terminator.span, &mut result);
641                    } else {
642                        ever_init_map.insert(destination.local, true);
643                    }
644                }
645            }
646        }
647    }
648    result
649}
