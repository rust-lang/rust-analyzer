68fn all_mir_bodies(
69    db: &dyn HirDatabase,
70    def: DefWithBodyId,
71    mut cb: impl FnMut(Arc<MirBody>),
72) -> Result<(), MirLowerError> {
73    fn for_closure(
74        db: &dyn HirDatabase,
75        c: InternedClosureId,
76        cb: &mut impl FnMut(Arc<MirBody>),
77    ) -> Result<(), MirLowerError> {
78        match db.mir_body_for_closure(c) {
79            Ok(body) => {
80                cb(body.clone());
81                body.closures.iter().try_for_each(|&it| for_closure(db, it, cb))
82            }
83            Err(e) => Err(e),
84        }
85    }
86    match db.mir_body(def) {
87        Ok(body) => {
88            cb(body.clone());
89            body.closures.iter().try_for_each(|&it| for_closure(db, it, &mut cb))
90        }
91        Err(e) => Err(e),
92    }
93}

95pub fn borrowck_query(
96    db: &dyn HirDatabase,
97    def: DefWithBodyId,
98) -> Result<Arc<[BorrowckResult]>, MirLowerError> {
99    let _p = tracing::info_span!("borrowck_query").entered();
100    let module = def.module(db);
101    let interner = DbInterner::new_with(db, module.krate(db));
102    let env = db.trait_environment_for_body(def);
103    let mut res = vec![];
104    // This calculates opaques defining scope which is a bit costly therefore is put outside `all_mir_bodies()`.
105    let typing_mode = TypingMode::borrowck(interner, def.into());
106    all_mir_bodies(db, def, |body| {
107        // FIXME(next-solver): Opaques.
108        let infcx = interner.infer_ctxt().build(typing_mode);
109        res.push(BorrowckResult {
110            mutability_of_locals: mutability_of_locals(&infcx, env, &body),
111            moved_out_of_ref: moved_out_of_ref(&infcx, env, &body),
112            partially_moved: partially_moved(&infcx, env, &body),
113            borrow_regions: borrow_regions(db, &body),
114            mir_body: body,
115        });
116    })?;
117    Ok(res.into())
118}

120fn make_fetch_closure_field<'db>(
121    db: &'db dyn HirDatabase,
122) -> impl FnOnce(InternedClosureId, GenericArgs<'db>, usize) -> Ty<'db> + use<'db> {
123    |c: InternedClosureId, subst: GenericArgs<'db>, f: usize| {
124        let InternedClosure(def, _) = db.lookup_intern_closure(c);
125        let infer = InferenceResult::for_body(db, def);
126        let (captures, _) = infer.closure_info(c);
127        let parent_subst = subst.as_closure().parent_args();
128        let interner = DbInterner::new_no_crate(db);
129        captures.get(f).expect("broken closure field").ty.get().instantiate(interner, parent_subst)
130    }
131}

133fn moved_out_of_ref<'db>(
134    infcx: &InferCtxt<'db>,
135    env: ParamEnv<'db>,
136    body: &MirBody,
137) -> Vec<MovedOutOfRef> {
138    let db = infcx.interner.db;
139    let mut result = vec![];
140    let mut for_operand = |op: &Operand, span: MirSpan| match op.kind {
141        OperandKind::Copy(p) | OperandKind::Move(p) => {
142            let mut ty: Ty<'db> = body.locals[p.local].ty.as_ref();
143            let mut is_dereference_of_ref = false;
144            for proj in p.projection.lookup(&body.projection_store) {
145                if *proj == ProjectionElem::Deref && ty.as_reference().is_some() {
146                    is_dereference_of_ref = true;
147                }
148                ty = proj.projected_ty(
149                    infcx,
150                    env,
151                    ty,
152                    make_fetch_closure_field(db),
153                    body.owner.module(db).krate(db),
154                );
155            }
156            if is_dereference_of_ref
157                && !infcx.type_is_copy_modulo_regions(env, ty)
158                && !ty.references_non_lt_error()
159            {
160                result.push(MovedOutOfRef { span: op.span.unwrap_or(span), ty: ty.store() });
161            }
162        }
163        OperandKind::Constant { .. } | OperandKind::Static(_) => (),
164    };
165    for (_, block) in body.basic_blocks.iter() {
166        db.unwind_if_revision_cancelled();
167        for statement in &block.statements {
168            match &statement.kind {
169                StatementKind::Assign(_, r) => match r {
170                    Rvalue::ShallowInitBoxWithAlloc(_) => (),
171                    Rvalue::ShallowInitBox(o, _)
172                    | Rvalue::UnaryOp(_, o)
173                    | Rvalue::Cast(_, o, _)
174                    | Rvalue::Repeat(o, _)
175                    | Rvalue::Use(o) => for_operand(o, statement.span),
176                    Rvalue::CopyForDeref(_)
177                    | Rvalue::Discriminant(_)
178                    | Rvalue::Len(_)
179                    | Rvalue::Ref(_, _) => (),
180                    Rvalue::CheckedBinaryOp(_, o1, o2) => {
181                        for_operand(o1, statement.span);
182                        for_operand(o2, statement.span);
183                    }
184                    Rvalue::Aggregate(_, ops) => {
185                        for op in ops.iter() {
186                            for_operand(op, statement.span);
187                        }
188                    }
189                    Rvalue::ThreadLocalRef(n)
190                    | Rvalue::AddressOf(n)
191                    | Rvalue::BinaryOp(n)
192                    | Rvalue::NullaryOp(n) => match *n {},
193                },
194                StatementKind::FakeRead(_)
195                | StatementKind::Deinit(_)
196                | StatementKind::StorageLive(_)
197                | StatementKind::StorageDead(_)
198                | StatementKind::Nop => (),
199            }
200        }
201        match &block.terminator {
202            Some(terminator) => match &terminator.kind {
203                TerminatorKind::SwitchInt { discr, .. } => for_operand(discr, terminator.span),
204                TerminatorKind::FalseEdge { .. }
205                | TerminatorKind::FalseUnwind { .. }
206                | TerminatorKind::Goto { .. }
207                | TerminatorKind::UnwindResume
208                | TerminatorKind::CoroutineDrop
209                | TerminatorKind::Abort
210                | TerminatorKind::Return
211                | TerminatorKind::Unreachable
212                | TerminatorKind::Drop { .. } => (),
213                TerminatorKind::DropAndReplace { value, .. } => {
214                    for_operand(value, terminator.span);
215                }
216                TerminatorKind::Call { func, args, .. } => {
217                    for_operand(func, terminator.span);
218                    args.iter().for_each(|it| for_operand(it, terminator.span));
219                }
220                TerminatorKind::Assert { cond, .. } => {
221                    for_operand(cond, terminator.span);
222                }
223                TerminatorKind::Yield { value, .. } => {
224                    for_operand(value, terminator.span);
225                }
226            },
227            None => (),
228        }
229    }
230    result.shrink_to_fit();
231    result
232}

328fn borrow_regions(db: &dyn HirDatabase, body: &MirBody) -> Vec<BorrowRegion> {
329    let mut borrows = FxHashMap::default();
330    for (_, block) in body.basic_blocks.iter() {
331        db.unwind_if_revision_cancelled();
332        for statement in &block.statements {
333            if let StatementKind::Assign(_, Rvalue::Ref(kind, p)) = &statement.kind {
334                borrows
335                    .entry(p.local)
336                    .and_modify(|it: &mut BorrowRegion| {
337                        it.places.push(statement.span);
338                    })
339                    .or_insert_with(|| BorrowRegion {
340                        local: p.local,
341                        kind: *kind,
342                        places: vec![statement.span],
343                    });
344            }
345        }
346        match &block.terminator {
347            Some(terminator) => match &terminator.kind {
348                TerminatorKind::FalseEdge { .. }
349                | TerminatorKind::FalseUnwind { .. }
350                | TerminatorKind::Goto { .. }
351                | TerminatorKind::UnwindResume
352                | TerminatorKind::CoroutineDrop
353                | TerminatorKind::Abort
354                | TerminatorKind::Return
355                | TerminatorKind::Unreachable
356                | TerminatorKind::Drop { .. } => (),
357                TerminatorKind::DropAndReplace { .. } => {}
358                TerminatorKind::Call { .. } => {}
359                _ => (),
360            },
361            None => (),
362        }
363    }
364
365    borrows.into_values().collect()
366}

369enum ProjectionCase {
370    /// Projection is a local
371    Direct,
372    /// Projection is some field or slice of a local
373    DirectPart,
374    /// Projection is deref of something
375    Indirect,

378fn place_case<'db>(
379    infcx: &InferCtxt<'db>,
380    env: ParamEnv<'db>,
381    body: &MirBody,
382    lvalue: &Place,
383) -> ProjectionCase {
384    let db = infcx.interner.db;
385    let mut is_part_of = false;
386    let mut ty = body.locals[lvalue.local].ty.as_ref();
387    for proj in lvalue.projection.lookup(&body.projection_store).iter() {
388        match proj {
389            ProjectionElem::Deref if ty.as_adt().is_none() => return ProjectionCase::Indirect, // It's indirect in case of reference and raw
390            ProjectionElem::Deref // It's direct in case of `Box<T>`
391            | ProjectionElem::ConstantIndex { .. }
392            | ProjectionElem::Subslice { .. }
393            | ProjectionElem::Field(_)
394            | ProjectionElem::ClosureField(_)
395            | ProjectionElem::Index(_) => {
396                is_part_of = true;
397            }
398            ProjectionElem::OpaqueCast(_) => (),
399        }
400        ty = proj.projected_ty(
401            infcx,
402            env,
403            ty,
404            make_fetch_closure_field(db),
405            body.owner.module(db).krate(db),
406        );
407    }
408    if is_part_of { ProjectionCase::DirectPart } else { ProjectionCase::Direct }
409}

411/// Returns a map from basic blocks to the set of locals that might be ever initialized before
412/// the start of the block. Only `StorageDead` can remove something from this map, and we ignore
413/// `Uninit` and `drop` and similar after initialization.
414fn ever_initialized_map(
415    db: &dyn HirDatabase,
416    body: &MirBody,
417) -> ArenaMap<BasicBlockId, ArenaMap<LocalId, bool>> {
418    let mut result: ArenaMap<BasicBlockId, ArenaMap<LocalId, bool>> =
419        body.basic_blocks.iter().map(|it| (it.0, ArenaMap::default())).collect();
420    fn dfs(
421        db: &dyn HirDatabase,
422        body: &MirBody,
423        l: LocalId,
424        stack: &mut Vec<BasicBlockId>,
425        result: &mut ArenaMap<BasicBlockId, ArenaMap<LocalId, bool>>,
426    ) {
427        while let Some(b) = stack.pop() {
428            let mut is_ever_initialized = result[b][l]; // It must be filled, as we use it as mark for dfs
429            let block = &body.basic_blocks[b];
430            for statement in &block.statements {
431                match &statement.kind {
432                    StatementKind::Assign(p, _) => {
433                        if p.projection.lookup(&body.projection_store).is_empty() && p.local == l {
434                            is_ever_initialized = true;
435                        }
436                    }
437                    StatementKind::StorageDead(p) => {
438                        if *p == l {
439                            is_ever_initialized = false;
440                        }
441                    }
442                    StatementKind::Deinit(_)
443                    | StatementKind::FakeRead(_)
444                    | StatementKind::Nop
445                    | StatementKind::StorageLive(_) => (),
446                }
447            }
448            let Some(terminator) = &block.terminator else {
449                never!(
450                    "Terminator should be none only in construction.\nThe body:\n{}",
451                    body.pretty_print(db, DisplayTarget::from_crate(db, body.owner.krate(db)))
452                );
453                return;
454            };
455            let mut process = |target, is_ever_initialized| {
456                if !result[target].contains_idx(l) || !result[target][l] && is_ever_initialized {
457                    result[target].insert(l, is_ever_initialized);
458                    stack.push(target);
459                }
460            };
461            match &terminator.kind {
462                TerminatorKind::Goto { target } => process(*target, is_ever_initialized),
463                TerminatorKind::SwitchInt { targets, .. } => {
464                    targets.all_targets().iter().for_each(|&it| process(it, is_ever_initialized));
465                }
466                TerminatorKind::UnwindResume
467                | TerminatorKind::Abort
468                | TerminatorKind::Return
469                | TerminatorKind::Unreachable => (),
470                TerminatorKind::Call { target, cleanup, destination, .. } => {
471                    if destination.projection.lookup(&body.projection_store).is_empty()
472                        && destination.local == l
473                    {
474                        is_ever_initialized = true;
475                    }
476                    target.iter().chain(cleanup).for_each(|&it| process(it, is_ever_initialized));
477                }
478                TerminatorKind::Drop { target, unwind, place: _ } => {
479                    iter::once(target)
480                        .chain(unwind)
481                        .for_each(|&it| process(it, is_ever_initialized));
482                }
483                TerminatorKind::DropAndReplace { .. }
484                | TerminatorKind::Assert { .. }
485                | TerminatorKind::Yield { .. }
486                | TerminatorKind::CoroutineDrop
487                | TerminatorKind::FalseEdge { .. }
488                | TerminatorKind::FalseUnwind { .. } => {
489                    never!("We don't emit these MIR terminators yet");
490                }
491            }
492        }
493    }
494    let mut stack = Vec::new();
495    for &l in &body.param_locals {
496        result[body.start_block].insert(l, true);
497        stack.clear();
498        stack.push(body.start_block);
499        dfs(db, body, l, &mut stack, &mut result);
500    }
501    for l in body.locals.iter().map(|it| it.0) {
502        db.unwind_if_revision_cancelled();
503        if !result[body.start_block].contains_idx(l) {
504            result[body.start_block].insert(l, false);
505            stack.clear();
506            stack.push(body.start_block);
507            dfs(db, body, l, &mut stack, &mut result);
508        }
509    }
510    result
511}

513fn push_mut_span(local: LocalId, span: MirSpan, result: &mut ArenaMap<LocalId, MutabilityReason>) {
514    match &mut result[local] {
515        MutabilityReason::Mut { spans } => spans.push(span),
516        it @ (MutabilityReason::Not | MutabilityReason::Unused) => {
517            *it = MutabilityReason::Mut { spans: vec![span] }
518        }
519    };
520}
521
522fn record_usage(local: LocalId, result: &mut ArenaMap<LocalId, MutabilityReason>) {
523    if let it @ MutabilityReason::Unused = &mut result[local] {
524        *it = MutabilityReason::Not;
525    };
526}

528fn record_usage_for_operand(arg: &Operand, result: &mut ArenaMap<LocalId, MutabilityReason>) {
529    if let OperandKind::Copy(p) | OperandKind::Move(p) = arg.kind {
530        record_usage(p.local, result);
531    }
532}

534fn mutability_of_locals<'db>(
535    infcx: &InferCtxt<'db>,
536    env: ParamEnv<'db>,
537    body: &MirBody,
538) -> ArenaMap<LocalId, MutabilityReason> {
539    let db = infcx.interner.db;
540    let mut result: ArenaMap<LocalId, MutabilityReason> =
541        body.locals.iter().map(|it| (it.0, MutabilityReason::Unused)).collect();
542
543    let ever_init_maps = ever_initialized_map(db, body);
544    for (block_id, mut ever_init_map) in ever_init_maps.into_iter() {
545        let block = &body.basic_blocks[block_id];
546        for statement in &block.statements {
547            match &statement.kind {
548                StatementKind::Assign(place, value) => {
549                    match place_case(infcx, env, body, place) {
550                        ProjectionCase::Direct => {
551                            if ever_init_map.get(place.local).copied().unwrap_or_default() {
552                                push_mut_span(place.local, statement.span, &mut result);
553                            } else {
554                                ever_init_map.insert(place.local, true);
555                            }
556                        }
557                        ProjectionCase::DirectPart => {
558                            // Partial initialization is not supported, so it is definitely `mut`
559                            push_mut_span(place.local, statement.span, &mut result);
560                        }
561                        ProjectionCase::Indirect => {
562                            record_usage(place.local, &mut result);
563                        }
564                    }
565                    match value {
566                        Rvalue::CopyForDeref(p)
567                        | Rvalue::Discriminant(p)
568                        | Rvalue::Len(p)
569                        | Rvalue::Ref(_, p) => {
570                            record_usage(p.local, &mut result);
571                        }
572                        Rvalue::Use(o)
573                        | Rvalue::Repeat(o, _)
574                        | Rvalue::Cast(_, o, _)
575                        | Rvalue::UnaryOp(_, o) => record_usage_for_operand(o, &mut result),
576                        Rvalue::CheckedBinaryOp(_, o1, o2) => {
577                            for o in [o1, o2] {
578                                record_usage_for_operand(o, &mut result);
579                            }
580                        }
581                        Rvalue::Aggregate(_, args) => {
582                            for arg in args.iter() {
583                                record_usage_for_operand(arg, &mut result);
584                            }
585                        }
586                        Rvalue::ShallowInitBox(_, _) | Rvalue::ShallowInitBoxWithAlloc(_) => (),
587                        Rvalue::ThreadLocalRef(n)
588                        | Rvalue::AddressOf(n)
589                        | Rvalue::BinaryOp(n)
590                        | Rvalue::NullaryOp(n) => match *n {},
591                    }
592                    if let Rvalue::Ref(
593                        BorrowKind::Mut {
594                            kind: MutBorrowKind::Default | MutBorrowKind::TwoPhasedBorrow,
595                        },
596                        p,
597                    ) = value
598                        && place_case(infcx, env, body, p) != ProjectionCase::Indirect
599                    {
600                        push_mut_span(p.local, statement.span, &mut result);
601                    }
602                }
603                StatementKind::FakeRead(p) => {
604                    record_usage(p.local, &mut result);
605                }
606                StatementKind::StorageDead(p) => {
607                    ever_init_map.insert(*p, false);
608                }
609                StatementKind::Deinit(_) | StatementKind::StorageLive(_) | StatementKind::Nop => (),
610            }
611        }
612        let Some(terminator) = &block.terminator else {
613            never!("Terminator should be none only in construction");
614            continue;
615        };
616        match &terminator.kind {
617            TerminatorKind::Goto { .. }
618            | TerminatorKind::UnwindResume
619            | TerminatorKind::Abort
620            | TerminatorKind::Return
621            | TerminatorKind::Unreachable
622            | TerminatorKind::FalseEdge { .. }
623            | TerminatorKind::FalseUnwind { .. }
624            | TerminatorKind::CoroutineDrop
625            | TerminatorKind::Drop { .. }
626            | TerminatorKind::DropAndReplace { .. }
627            | TerminatorKind::Assert { .. }
628            | TerminatorKind::Yield { .. } => (),
629            TerminatorKind::SwitchInt { discr, targets: _ } => {
630                record_usage_for_operand(discr, &mut result);
631            }
632            TerminatorKind::Call { destination, args, func, .. } => {
633                record_usage_for_operand(func, &mut result);
634                for arg in args.iter() {
635                    record_usage_for_operand(arg, &mut result);
636                }
637                if destination.projection.lookup(&body.projection_store).is_empty() {
638                    if ever_init_map.get(destination.local).copied().unwrap_or_default() {
639                        push_mut_span(destination.local, terminator.span, &mut result);
640                    } else {
641                        ever_init_map.insert(destination.local, true);
642                    }
643                }
644            }
645        }
646    }
647    result
648}
