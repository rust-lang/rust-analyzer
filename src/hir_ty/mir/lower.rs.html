129/// A token to ensuring that each drop scope is popped at most once, thanks to the compiler that checks moves.
130struct DropScopeToken;
131impl DropScopeToken {
132    fn pop_and_drop<'db>(
133        self,
134        ctx: &mut MirLowerCtx<'_, 'db>,
135        current: BasicBlockId,
136        span: MirSpan,
137    ) -> BasicBlockId {
138        std::mem::forget(self);
139        ctx.pop_drop_scope_internal(current, span)
140    }
141
142    /// It is useful when we want a drop scope is syntactically closed, but we don't want to execute any drop
143    /// code. Either when the control flow is diverging (so drop code doesn't reached) or when drop is handled
144    /// for us (for example a block that ended with a return statement. Return will drop everything, so the block shouldn't
145    /// do anything)
146    fn pop_assume_dropped(self, ctx: &mut MirLowerCtx<'_, '_>) {
147        std::mem::forget(self);
148        ctx.pop_drop_scope_assume_dropped_internal();
149    }
150}

153    fn drop(&mut self) {}

167impl MirLowerError {
168    pub fn pretty_print(
169        &self,
170        f: &mut String,
171        db: &dyn HirDatabase,
172        span_formatter: impl Fn(FileId, TextRange) -> String,
173        display_target: DisplayTarget,
174    ) -> std::result::Result<(), std::fmt::Error> {
175        match self {
176            MirLowerError::ConstEvalError(name, e) => {
177                writeln!(f, "In evaluating constant {name}")?;
178                match &**e {
179                    ConstEvalError::MirLowerError(e) => {
180                        e.pretty_print(f, db, span_formatter, display_target)?
181                    }
182                    ConstEvalError::MirEvalError(e) => {
183                        e.pretty_print(f, db, span_formatter, display_target)?
184                    }
185                }
186            }
187            MirLowerError::MissingFunctionDefinition(owner, it) => {
188                let body = db.body(*owner);
189                writeln!(
190                    f,
191                    "Missing function definition for {}",
192                    body.pretty_print_expr(db, *owner, *it, display_target.edition)
193                )?;
194            }
195            MirLowerError::HasErrors => writeln!(f, "Type inference result contains errors")?,
196            MirLowerError::TypeMismatch(e) => writeln!(
197                f,
198                "Type mismatch: Expected {}, found {}",
199                e.expected.as_ref().display(db, display_target),
200                e.actual.as_ref().display(db, display_target),
201            )?,
202            MirLowerError::GenericArgNotProvided(id, subst) => {
203                let param_name = match *id {
204                    GenericParamId::TypeParamId(id) => {
205                        db.generic_params(id.parent())[id.local_id()].name().cloned()
206                    }
207                    GenericParamId::ConstParamId(id) => {
208                        db.generic_params(id.parent())[id.local_id()].name().cloned()
209                    }
210                    GenericParamId::LifetimeParamId(id) => {
211                        Some(db.generic_params(id.parent)[id.local_id].name.clone())
212                    }
213                };
214                writeln!(
215                    f,
216                    "Generic arg not provided for {}",
217                    param_name.unwrap_or(Name::missing()).display(db, display_target.edition)
218                )?;
219                writeln!(f, "Provided args: [")?;
220                for g in subst.as_ref() {
221                    write!(f, "    {},", g.display(db, display_target))?;
222                }
223                writeln!(f, "]")?;
224            }
225            MirLowerError::LayoutError(_)
226            | MirLowerError::UnsizedTemporary(_)
227            | MirLowerError::IncompleteExpr
228            | MirLowerError::IncompletePattern
229            | MirLowerError::InaccessibleLocal
230            | MirLowerError::TraitFunctionDefinition(_, _)
231            | MirLowerError::UnresolvedName(_)
232            | MirLowerError::RecordLiteralWithoutPath
233            | MirLowerError::UnresolvedMethod(_)
234            | MirLowerError::UnresolvedField
235            | MirLowerError::TypeError(_)
236            | MirLowerError::NotSupported(_)
237            | MirLowerError::ContinueWithoutLoop
238            | MirLowerError::BreakWithoutLoop
239            | MirLowerError::Loop
240            | MirLowerError::ImplementationError(_)
241            | MirLowerError::LangItemNotFound
242            | MirLowerError::MutatingRvalue
243            | MirLowerError::UnresolvedLabel
244            | MirLowerError::UnresolvedUpvar(_) => writeln!(f, "{self:?}")?,
245        }
246        Ok(())
247    }
248}

342    fn temp(&mut self, ty: Ty<'db>, current: BasicBlockId, span: MirSpan) -> Result<'db, LocalId> {
343        if matches!(ty.kind(), TyKind::Slice(_) | TyKind::Dynamic(..)) {
344            return Err(MirLowerError::UnsizedTemporary(ty.store()));
345        }
346        let l = self.result.locals.alloc(Local { ty: ty.store() });
347        self.push_storage_live_for_local(l, current, span)?;
348        Ok(l)
349    }

351    fn lower_expr_to_some_operand(
352        &mut self,
353        expr_id: ExprId,
354        current: BasicBlockId,
355    ) -> Result<'db, Option<(Operand, BasicBlockId)>> {
356        if !self.has_adjustments(expr_id)
357            && let Expr::Literal(l) = &self.body[expr_id]
358        {
359            let ty = self.expr_ty_without_adjust(expr_id);
360            return Ok(Some((self.lower_literal_to_operand(ty, l)?, current)));
361        }
362        let Some((p, current)) = self.lower_expr_as_place(current, expr_id, true)? else {
363            return Ok(None);
364        };
365        Ok(Some((Operand { kind: OperandKind::Copy(p), span: Some(expr_id.into()) }, current)))
366    }

368    fn lower_expr_to_place_with_adjust(
369        &mut self,
370        expr_id: ExprId,
371        place: Place,
372        current: BasicBlockId,
373        adjustments: &[Adjustment],
374    ) -> Result<'db, Option<BasicBlockId>> {
375        match adjustments.split_last() {
376            Some((last, rest)) => match &last.kind {
377                Adjust::NeverToAny => {
378                    let temp = self.temp(self.types.types.never, current, MirSpan::Unknown)?;
379                    self.lower_expr_to_place_with_adjust(expr_id, temp.into(), current, rest)
380                }
381                Adjust::Deref(_) => {
382                    let Some((p, current)) =
383                        self.lower_expr_as_place_with_adjust(current, expr_id, true, adjustments)?
384                    else {
385                        return Ok(None);
386                    };
387                    self.push_assignment(
388                        current,
389                        place,
390                        Operand { kind: OperandKind::Copy(p), span: None }.into(),
391                        expr_id.into(),
392                    );
393                    Ok(Some(current))
394                }
395                Adjust::Borrow(AutoBorrow::Ref(m)) => self.lower_expr_to_place_with_borrow_adjust(
396                    expr_id,
397                    place,
398                    current,
399                    rest,
400                    (*m).into(),
401                ),
402                Adjust::Borrow(AutoBorrow::RawPtr(m)) => {
403                    self.lower_expr_to_place_with_borrow_adjust(expr_id, place, current, rest, *m)
404                }
405                Adjust::Pointer(cast) => {
406                    let Some((p, current)) =
407                        self.lower_expr_as_place_with_adjust(current, expr_id, true, rest)?
408                    else {
409                        return Ok(None);
410                    };
411                    self.push_assignment(
412                        current,
413                        place,
414                        Rvalue::Cast(
415                            CastKind::PointerCoercion(*cast),
416                            Operand { kind: OperandKind::Copy(p), span: None },
417                            last.target.clone(),
418                        ),
419                        expr_id.into(),
420                    );
421                    Ok(Some(current))
422                }
423            },
424            None => self.lower_expr_to_place_without_adjust(expr_id, place, current),
425        }
426    }

428    fn lower_expr_to_place_with_borrow_adjust(
429        &mut self,
430        expr_id: ExprId,
431        place: Place,
432        current: BasicBlockId,
433        rest: &[Adjustment],
434        m: Mutability,
435    ) -> Result<'db, Option<BasicBlockId>> {
436        let Some((p, current)) =
437            self.lower_expr_as_place_with_adjust(current, expr_id, true, rest)?
438        else {
439            return Ok(None);
440        };
441        let bk = BorrowKind::from_rustc(m);
442        self.push_assignment(current, place, Rvalue::Ref(bk, p), expr_id.into());
443        Ok(Some(current))
444    }

446    fn lower_expr_to_place(
447        &mut self,
448        expr_id: ExprId,
449        place: Place,
450        prev_block: BasicBlockId,
451    ) -> Result<'db, Option<BasicBlockId>> {
452        if let Some(adjustments) = self.infer.expr_adjustments.get(&expr_id) {
453            return self.lower_expr_to_place_with_adjust(expr_id, place, prev_block, adjustments);
454        }
455        self.lower_expr_to_place_without_adjust(expr_id, place, prev_block)
456    }

458    fn lower_expr_to_place_without_adjust(
459        &mut self,
460        expr_id: ExprId,
461        place: Place,
462        mut current: BasicBlockId,
463    ) -> Result<'db, Option<BasicBlockId>> {
464        match &self.body[expr_id] {
465            Expr::OffsetOf(_) => {
466                not_supported!("builtin#offset_of")
467            }
468            Expr::InlineAsm(_) => {
469                not_supported!("builtin#asm")
470            }
471            Expr::Missing => {
472                if let DefWithBodyId::FunctionId(f) = self.owner {
473                    let assoc = f.lookup(self.db);
474                    if let ItemContainerId::TraitId(t) = assoc.container {
475                        let name = &self.db.function_signature(f).name;
476                        return Err(MirLowerError::TraitFunctionDefinition(t, name.clone()));
477                    }
478                }
479                Err(MirLowerError::IncompleteExpr)
480            }
481            Expr::Path(p) => {
482                let pr =
483                    if let Some((assoc, subst)) = self.infer.assoc_resolutions_for_expr(expr_id) {
484                        match assoc {
485                            CandidateId::ConstId(c) => {
486                                self.lower_const(c.into(), current, place, subst, expr_id.into())?;
487                                return Ok(Some(current));
488                            }
489                            CandidateId::FunctionId(_) => {
490                                // FnDefs are zero sized, no action is needed.
491                                return Ok(Some(current));
492                            }
493                        }
494                    } else if let Some(variant) = self.infer.variant_resolution_for_expr(expr_id) {
495                        match variant {
496                            VariantId::EnumVariantId(e) => ValueNs::EnumVariantId(e),
497                            VariantId::StructId(s) => ValueNs::StructId(s),
498                            VariantId::UnionId(_) => implementation_error!("Union variant as path"),
499                        }
500                    } else {
501                        let resolver_guard =
502                            self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
503                        let hygiene = self.body.expr_path_hygiene(expr_id);
504                        let result = self
505                            .resolver
506                            .resolve_path_in_value_ns_fully(self.db, p, hygiene)
507                            .ok_or_else(|| {
508                                MirLowerError::unresolved_path(
509                                    self.db,
510                                    p,
511                                    DisplayTarget::from_crate(self.db, self.krate()),
512                                    self.body,
513                                )
514                            })?;
515                        self.resolver.reset_to_guard(resolver_guard);
516                        result
517                    };
518                match pr {
519                    ValueNs::LocalBinding(_) | ValueNs::StaticId(_) => {
520                        let Some((temp, current)) =
521                            self.lower_expr_as_place_without_adjust(current, expr_id, false)?
522                        else {
523                            return Ok(None);
524                        };
525                        self.push_assignment(
526                            current,
527                            place,
528                            Operand { kind: OperandKind::Copy(temp), span: None }.into(),
529                            expr_id.into(),
530                        );
531                        Ok(Some(current))
532                    }
533                    ValueNs::ConstId(const_id) => {
534                        self.lower_const(
535                            const_id.into(),
536                            current,
537                            place,
538                            GenericArgs::empty(self.interner()),
539                            expr_id.into(),
540                        )?;
541                        Ok(Some(current))
542                    }
543                    ValueNs::EnumVariantId(variant_id) => {
544                        let variant_fields = variant_id.fields(self.db);
545                        if variant_fields.shape == FieldsShape::Unit {
546                            let ty = self.infer.expr_ty(expr_id);
547                            current = self.lower_enum_variant(
548                                variant_id,
549                                current,
550                                place,
551                                ty,
552                                Box::new([]),
553                                expr_id.into(),
554                            )?;
555                        }
556                        // Otherwise its a tuple like enum, treated like a zero sized function, so no action is needed
557                        Ok(Some(current))
558                    }
559                    ValueNs::GenericParam(p) => {
560                        let Some(def) = self.owner.as_generic_def_id(self.db) else {
561                            not_supported!("owner without generic def id");
562                        };
563                        let generics = generics(self.db, def);
564                        let index = generics
565                            .type_or_const_param_idx(p.into())
566                            .ok_or(MirLowerError::TypeError("fail to lower const generic param"))?
567                            as u32;
568                        self.push_assignment(
569                            current,
570                            place,
571                            Rvalue::from(Operand {
572                                kind: OperandKind::Constant {
573                                    konst: Const::new_param(
574                                        self.interner(),
575                                        ParamConst { id: p, index },
576                                    )
577                                    .store(),
578                                    ty: self.db.const_param_ty_ns(p).store(),
579                                },
580                                span: None,
581                            }),
582                            expr_id.into(),
583                        );
584                        Ok(Some(current))
585                    }
586                    ValueNs::FunctionId(_) | ValueNs::StructId(_) | ValueNs::ImplSelf(_) => {
587                        // It's probably a unit struct or a zero sized function, so no action is needed.
588                        Ok(Some(current))
589                    }
590                }
591            }
592            Expr::If { condition, then_branch, else_branch } => {
593                let Some((discr, current)) =
594                    self.lower_expr_to_some_operand(*condition, current)?
595                else {
596                    return Ok(None);
597                };
598                let start_of_then = self.new_basic_block();
599                let end_of_then = self.lower_expr_to_place(*then_branch, place, start_of_then)?;
600                let start_of_else = self.new_basic_block();
601                let end_of_else = if let Some(else_branch) = else_branch {
602                    self.lower_expr_to_place(*else_branch, place, start_of_else)?
603                } else {
604                    Some(start_of_else)
605                };
606                self.set_terminator(
607                    current,
608                    TerminatorKind::SwitchInt {
609                        discr,
610                        targets: SwitchTargets::static_if(1, start_of_then, start_of_else),
611                    },
612                    expr_id.into(),
613                );
614                Ok(self.merge_blocks(end_of_then, end_of_else, expr_id.into()))
615            }
616            Expr::Let { pat, expr } => {
617                let Some((cond_place, current)) = self.lower_expr_as_place(current, *expr, true)?
618                else {
619                    return Ok(None);
620                };
621                self.push_fake_read(current, cond_place, expr_id.into());
622                let resolver_guard =
623                    self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
624                let (then_target, else_target) =
625                    self.pattern_match(current, None, cond_place, *pat)?;
626                self.resolver.reset_to_guard(resolver_guard);
627                self.write_bytes_to_place(
628                    then_target,
629                    place,
630                    Box::new([1]),
631                    Ty::new_bool(self.interner()),
632                    MirSpan::Unknown,
633                )?;
634                if let Some(else_target) = else_target {
635                    self.write_bytes_to_place(
636                        else_target,
637                        place,
638                        Box::new([0]),
639                        Ty::new_bool(self.interner()),
640                        MirSpan::Unknown,
641                    )?;
642                }
643                Ok(self.merge_blocks(Some(then_target), else_target, expr_id.into()))
644            }
645            Expr::Unsafe { id: _, statements, tail } => {
646                self.lower_block_to_place(statements, current, *tail, place, expr_id.into())
647            }
648            Expr::Block { id: _, statements, tail, label } => {
649                if let Some(label) = label {
650                    self.lower_loop(current, place, Some(*label), expr_id.into(), |this, begin| {
651                        if let Some(current) = this.lower_block_to_place(
652                            statements,
653                            begin,
654                            *tail,
655                            place,
656                            expr_id.into(),
657                        )? {
658                            let end = this.current_loop_end()?;
659                            this.set_goto(current, end, expr_id.into());
660                        }
661                        Ok(())
662                    })
663                } else {
664                    self.lower_block_to_place(statements, current, *tail, place, expr_id.into())
665                }
666            }
667            Expr::Loop { body, label } => {
668                self.lower_loop(current, place, *label, expr_id.into(), |this, begin| {
669                    let scope = this.push_drop_scope();
670                    if let Some((_, mut current)) = this.lower_expr_as_place(begin, *body, true)? {
671                        current = scope.pop_and_drop(this, current, body.into());
672                        this.set_goto(current, begin, expr_id.into());
673                    } else {
674                        scope.pop_assume_dropped(this);
675                    }
676                    Ok(())
677                })
678            }
679            Expr::Call { callee, args, .. } => {
680                if let Some((func_id, generic_args)) = self.infer.method_resolution(expr_id) {
681                    let ty = Ty::new_fn_def(
682                        self.interner(),
683                        CallableDefId::FunctionId(func_id).into(),
684                        generic_args,
685                    );
686                    let func = Operand::from_bytes(Box::default(), ty);
687                    return self.lower_call_and_args(
688                        func,
689                        iter::once(*callee).chain(args.iter().copied()),
690                        place,
691                        current,
692                        self.is_uninhabited(expr_id),
693                        expr_id.into(),
694                    );
695                }
696                let callee_ty = self.expr_ty_after_adjustments(*callee);
697                match callee_ty.kind() {
698                    TyKind::FnDef(..) => {
699                        let func = Operand::from_bytes(Box::default(), callee_ty);
700                        self.lower_call_and_args(
701                            func,
702                            args.iter().copied(),
703                            place,
704                            current,
705                            self.is_uninhabited(expr_id),
706                            expr_id.into(),
707                        )
708                    }
709                    TyKind::FnPtr(..) => {
710                        let Some((func, current)) =
711                            self.lower_expr_to_some_operand(*callee, current)?
712                        else {
713                            return Ok(None);
714                        };
715                        self.lower_call_and_args(
716                            func,
717                            args.iter().copied(),
718                            place,
719                            current,
720                            self.is_uninhabited(expr_id),
721                            expr_id.into(),
722                        )
723                    }
724                    TyKind::Closure(_, _) => {
725                        not_supported!(
726                            "method resolution not emitted for closure (Are Fn traits available?)"
727                        );
728                    }
729                    TyKind::Error(_) => {
730                        Err(MirLowerError::MissingFunctionDefinition(self.owner, expr_id))
731                    }
732                    _ => Err(MirLowerError::TypeError("function call on bad type")),
733                }
734            }
735            Expr::MethodCall { receiver, args, method_name, .. } => {
736                let (func_id, generic_args) =
737                    self.infer.method_resolution(expr_id).ok_or_else(|| {
738                        MirLowerError::UnresolvedMethod(
739                            method_name.display(self.db, self.edition()).to_string(),
740                        )
741                    })?;
742                let func = Operand::from_fn(self.db, func_id, generic_args);
743                self.lower_call_and_args(
744                    func,
745                    iter::once(*receiver).chain(args.iter().copied()),
746                    place,
747                    current,
748                    self.is_uninhabited(expr_id),
749                    expr_id.into(),
750                )
751            }
752            Expr::Match { expr, arms } => {
753                let Some((cond_place, mut current)) =
754                    self.lower_expr_as_place(current, *expr, true)?
755                else {
756                    return Ok(None);
757                };
758                self.push_fake_read(current, cond_place, expr_id.into());
759                let mut end = None;
760                let resolver_guard =
761                    self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
762                for MatchArm { pat, guard, expr } in arms.iter() {
763                    let (then, mut otherwise) =
764                        self.pattern_match(current, None, cond_place, *pat)?;
765                    let then = if let &Some(guard) = guard {
766                        let next = self.new_basic_block();
767                        let o = otherwise.get_or_insert_with(|| self.new_basic_block());
768                        if let Some((discr, c)) = self.lower_expr_to_some_operand(guard, then)? {
769                            self.set_terminator(
770                                c,
771                                TerminatorKind::SwitchInt {
772                                    discr,
773                                    targets: SwitchTargets::static_if(1, next, *o),
774                                },
775                                expr_id.into(),
776                            );
777                        }
778                        next
779                    } else {
780                        then
781                    };
782                    if let Some(block) = self.lower_expr_to_place(*expr, place, then)? {
783                        let r = end.get_or_insert_with(|| self.new_basic_block());
784                        self.set_goto(block, *r, expr_id.into());
785                    }
786                    match otherwise {
787                        Some(o) => current = o,
788                        None => {
789                            // The current pattern was irrefutable, so there is no need to generate code
790                            // for the rest of patterns
791                            break;
792                        }
793                    }
794                }
795                self.resolver.reset_to_guard(resolver_guard);
796                if self.is_unterminated(current) {
797                    self.set_terminator(current, TerminatorKind::Unreachable, expr_id.into());
798                }
799                Ok(end)
800            }
801            Expr::Continue { label } => {
802                let loop_data = match label {
803                    Some(l) => {
804                        self.labeled_loop_blocks.get(l).ok_or(MirLowerError::UnresolvedLabel)?
805                    }
806                    None => self
807                        .current_loop_blocks
808                        .as_ref()
809                        .ok_or(MirLowerError::ContinueWithoutLoop)?,
810                };
811                let begin = loop_data.begin;
812                current =
813                    self.drop_until_scope(loop_data.drop_scope_index, current, expr_id.into());
814                self.set_goto(current, begin, expr_id.into());
815                Ok(None)
816            }
817            &Expr::Break { expr, label } => {
818                if let Some(expr) = expr {
819                    let loop_data = match label {
820                        Some(l) => self
821                            .labeled_loop_blocks
822                            .get(&l)
823                            .ok_or(MirLowerError::UnresolvedLabel)?,
824                        None => self
825                            .current_loop_blocks
826                            .as_ref()
827                            .ok_or(MirLowerError::BreakWithoutLoop)?,
828                    };
829                    let Some(c) = self.lower_expr_to_place(expr, loop_data.place, current)? else {
830                        return Ok(None);
831                    };
832                    current = c;
833                }
834                let (end, drop_scope) = match label {
835                    Some(l) => {
836                        let loop_blocks = self
837                            .labeled_loop_blocks
838                            .get(&l)
839                            .ok_or(MirLowerError::UnresolvedLabel)?;
840                        (
841                            loop_blocks.end.expect("We always generate end for labeled loops"),
842                            loop_blocks.drop_scope_index,
843                        )
844                    }
845                    None => (
846                        self.current_loop_end()?,
847                        self.current_loop_blocks.as_ref().unwrap().drop_scope_index,
848                    ),
849                };
850                current = self.drop_until_scope(drop_scope, current, expr_id.into());
851                self.set_goto(current, end, expr_id.into());
852                Ok(None)
853            }
854            Expr::Return { expr } => {
855                if let Some(expr) = expr {
856                    if let Some(c) =
857                        self.lower_expr_to_place(*expr, return_slot().into(), current)?
858                    {
859                        current = c;
860                    } else {
861                        return Ok(None);
862                    }
863                }
864                current = self.drop_until_scope(0, current, expr_id.into());
865                self.set_terminator(current, TerminatorKind::Return, expr_id.into());
866                Ok(None)
867            }
868            Expr::Become { .. } => not_supported!("tail-calls"),
869            Expr::Yield { .. } => not_supported!("yield"),
870            Expr::RecordLit { fields, path, spread, .. } => {
871                let spread_place = match *spread {
872                    RecordSpread::Expr(it) => {
873                        let Some((p, c)) = self.lower_expr_as_place(current, it, true)? else {
874                            return Ok(None);
875                        };

877                        Some(p)
878                    }
879                    RecordSpread::None => None,
880                    RecordSpread::FieldDefaults => not_supported!("empty record spread"),

889                        None => MirLowerError::RecordLiteralWithoutPath,

897                    VariantId::EnumVariantId(_) | VariantId::StructId(_) => {

901                                variant_fields.field(name).ok_or(MirLowerError::UnresolvedField)?;

903                            else {
904                                return Ok(None);
905                            };
906                            current = c;
907                            operands[u32::from(field_id.into_raw()) as usize] = Some(op);
908                        }
909                        let rvalue = Rvalue::Aggregate(
910                            AggregateKind::Adt(variant_id, subst.store()),
911                            match spread_place {
912                                Some(sp) => operands
913                                    .into_iter()
914                                    .enumerate()
915                                    .map(|(i, it)| match it {
916                                        Some(it) => it,
917                                        None => {
918                                            let p = sp.project(
919                                                ProjectionElem::Field(Either::Left(FieldId {
920                                                    parent: variant_id,
921                                                    local_id: LocalFieldId::from_raw(RawIdx::from(
922                                                        i as u32,
923                                                    )),
924                                                })),
925                                                &mut self.result.projection_store,
926                                            );
927                                            Operand { kind: OperandKind::Copy(p), span: None }
928                                        }
929                                    })
930                                    .collect(),
931                                None => operands.into_iter().collect::<Option<_>>().ok_or(
932                                    MirLowerError::TypeError("missing field in record literal"),
933                                )?,
934                            },
935                        );
936                        self.push_assignment(current, place, rvalue, expr_id.into());
937                        Ok(Some(current))
938                    }
939                    VariantId::UnionId(union_id) => {
940                        let [RecordLitField { name, expr }] = fields.as_ref() else {
941                            not_supported!("Union record literal with more than one field");
942                        };
943                        let local_id =
944                            variant_fields.field(name).ok_or(MirLowerError::UnresolvedField)?;
945                        let place = place.project(
946                            PlaceElem::Field(Either::Left(FieldId {
947                                parent: union_id.into(),
948                                local_id,
949                            })),
950                            &mut self.result.projection_store,
951                        );
952                        self.lower_expr_to_place(*expr, place, current)
953                    }
954                }
955            }
956            Expr::Await { .. } => not_supported!("await"),
957            Expr::Yeet { .. } => not_supported!("yeet"),
958            Expr::Async { .. } => not_supported!("async block"),
959            &Expr::Const(_) => {
960                // let subst = self.placeholder_subst();
961                // self.lower_const(
962                //     id.into(),
963                //     current,
964                //     place,
965                //     subst,
966                //     expr_id.into(),
967                //     self.expr_ty_without_adjust(expr_id),
968                // )?;
969                // Ok(Some(current))
970                not_supported!("const block")
971            }
972            Expr::Cast { expr, type_ref: _ } => {

974                    return Ok(None);
975                };

979                    Rvalue::Use(it)
980                } else {

982                    let target_ty = self.infer.expr_ty(expr_id);
983                    let cast_kind = if source_ty.as_reference().is_some() {
984                        CastKind::PointerCoercion(PointerCast::ArrayToPointer)
985                    } else {
986                        cast_kind(self.db, source_ty, target_ty)?
987                    };
988
989                    Rvalue::Cast(cast_kind, it, target_ty.store())
990                };
991                self.push_assignment(current, place, rvalue, expr_id.into());
992                Ok(Some(current))
993            }
994            Expr::Ref { expr, rawness: _, mutability } => {
995                let Some((p, current)) = self.lower_expr_as_place(current, *expr, true)? else {
996                    return Ok(None);
997                };
998                let bk = BorrowKind::from_hir(*mutability);
999                self.push_assignment(current, place, Rvalue::Ref(bk, p), expr_id.into());
1000                Ok(Some(current))
1001            }
1002            Expr::Box { expr } => {
1003                let ty = self.expr_ty_after_adjustments(*expr);
1004                self.push_assignment(
1005                    current,
1006                    place,
1007                    Rvalue::ShallowInitBoxWithAlloc(ty.store()),
1008                    expr_id.into(),
1009                );
1010                let Some((operand, current)) = self.lower_expr_to_some_operand(*expr, current)?
1011                else {
1012                    return Ok(None);
1013                };
1014                let p = place.project(ProjectionElem::Deref, &mut self.result.projection_store);
1015                self.push_assignment(current, p, operand.into(), expr_id.into());
1016                Ok(Some(current))
1017            }
1018            Expr::Field { .. }
1019            | Expr::Index { .. }
1020            | Expr::UnaryOp { op: hir_def::hir::UnaryOp::Deref, .. } => {
1021                let Some((p, current)) =
1022                    self.lower_expr_as_place_without_adjust(current, expr_id, true)?
1023                else {
1024                    return Ok(None);
1025                };
1026                self.push_assignment(
1027                    current,
1028                    place,
1029                    Operand { kind: OperandKind::Copy(p), span: None }.into(),
1030                    expr_id.into(),
1031                );
1032                Ok(Some(current))
1033            }
1034            Expr::UnaryOp {
1035                expr,
1036                op: op @ (hir_def::hir::UnaryOp::Not | hir_def::hir::UnaryOp::Neg),
1037            } => {
1038                let Some((operand, current)) = self.lower_expr_to_some_operand(*expr, current)?
1039                else {
1040                    return Ok(None);
1041                };
1042                let operation = match op {
1043                    hir_def::hir::UnaryOp::Not => UnOp::Not,
1044                    hir_def::hir::UnaryOp::Neg => UnOp::Neg,
1045                    _ => unreachable!(),
1046                };
1047                self.push_assignment(
1048                    current,
1049                    place,
1050                    Rvalue::UnaryOp(operation, operand),
1051                    expr_id.into(),
1052                );
1053                Ok(Some(current))
1054            }
1055            Expr::BinaryOp { lhs, rhs, op } => {
1056                let op: BinaryOp = op.ok_or(MirLowerError::IncompleteExpr)?;
1057                let is_builtin = 'b: {
1058                    // Without adjust here is a hack. We assume that we know every possible adjustment
1059                    // for binary operator, and use without adjust to simplify our conditions.
1060                    let lhs_ty = self.expr_ty_without_adjust(*lhs);
1061                    let rhs_ty = self.expr_ty_without_adjust(*rhs);
1062                    if matches!(op, BinaryOp::CmpOp(syntax::ast::CmpOp::Eq { .. }))
1063                        && matches!(lhs_ty.kind(), TyKind::RawPtr(..))
1064                        && matches!(rhs_ty.kind(), TyKind::RawPtr(..))
1065                    {
1066                        break 'b true;
1067                    }
1068                    let builtin_inequal_impls = matches!(
1069                        op,
1070                        BinaryOp::ArithOp(ArithOp::Shl | ArithOp::Shr)
1071                            | BinaryOp::Assignment { op: Some(ArithOp::Shl | ArithOp::Shr) }
1072                    );
1073                    matches!(
1074                        lhs_ty.kind(),
1075                        TyKind::Bool
1076                            | TyKind::Char
1077                            | TyKind::Int(_)
1078                            | TyKind::Uint(_)
1079                            | TyKind::Float(_)
1080                    ) && matches!(
1081                        rhs_ty.kind(),
1082                        TyKind::Bool
1083                            | TyKind::Char
1084                            | TyKind::Int(_)
1085                            | TyKind::Uint(_)
1086                            | TyKind::Float(_)
1087                    ) && (lhs_ty == rhs_ty || builtin_inequal_impls)
1088                };
1089                if !is_builtin
1090                    && let Some((func_id, generic_args)) = self.infer.method_resolution(expr_id)
1091                {
1092                    let func = Operand::from_fn(self.db, func_id, generic_args);
1093                    return self.lower_call_and_args(
1094                        func,
1095                        [*lhs, *rhs].into_iter(),
1096                        place,
1097                        current,
1098                        self.is_uninhabited(expr_id),
1099                        expr_id.into(),
1100                    );
1101                }
1102                if let hir_def::hir::BinaryOp::Assignment { op: Some(op) } = op {
1103                    // last adjustment is `&mut` which we don't want it.
1104                    let adjusts = self
1105                        .infer
1106                        .expr_adjustments
1107                        .get(lhs)
1108                        .and_then(|it| it.split_last())
1109                        .map(|it| it.1)
1110                        .ok_or(MirLowerError::TypeError("adjustment of binary op was missing"))?;
1111                    let Some((lhs_place, current)) =
1112                        self.lower_expr_as_place_with_adjust(current, *lhs, false, adjusts)?
1113                    else {
1114                        return Ok(None);
1115                    };
1116                    let Some((rhs_op, current)) = self.lower_expr_to_some_operand(*rhs, current)?
1117                    else {
1118                        return Ok(None);
1119                    };
1120                    let r_value = Rvalue::CheckedBinaryOp(
1121                        op.into(),
1122                        Operand { kind: OperandKind::Copy(lhs_place), span: None },
1123                        rhs_op,
1124                    );
1125                    self.push_assignment(current, lhs_place, r_value, expr_id.into());
1126                    return Ok(Some(current));
1127                }
1128                let Some((lhs_op, current)) = self.lower_expr_to_some_operand(*lhs, current)?
1129                else {
1130                    return Ok(None);
1131                };
1132                if let hir_def::hir::BinaryOp::LogicOp(op) = op {
1133                    let value_to_short = match op {
1134                        syntax::ast::LogicOp::And => 0,
1135                        syntax::ast::LogicOp::Or => 1,
1136                    };
1137                    let start_of_then = self.new_basic_block();
1138                    self.push_assignment(
1139                        start_of_then,
1140                        place,
1141                        lhs_op.clone().into(),
1142                        expr_id.into(),
1143                    );
1144                    let end_of_then = Some(start_of_then);
1145                    let start_of_else = self.new_basic_block();
1146                    let end_of_else = self.lower_expr_to_place(*rhs, place, start_of_else)?;
1147                    self.set_terminator(
1148                        current,
1149                        TerminatorKind::SwitchInt {
1150                            discr: lhs_op,
1151                            targets: SwitchTargets::static_if(
1152                                value_to_short,
1153                                start_of_then,
1154                                start_of_else,
1155                            ),
1156                        },
1157                        expr_id.into(),
1158                    );
1159                    return Ok(self.merge_blocks(end_of_then, end_of_else, expr_id.into()));
1160                }
1161                let Some((rhs_op, current)) = self.lower_expr_to_some_operand(*rhs, current)?
1162                else {
1163                    return Ok(None);
1164                };
1165                self.push_assignment(
1166                    current,
1167                    place,
1168                    Rvalue::CheckedBinaryOp(
1169                        match op {
1170                            hir_def::hir::BinaryOp::LogicOp(op) => match op {
1171                                hir_def::hir::LogicOp::And => BinOp::BitAnd, // FIXME: make these short circuit
1172                                hir_def::hir::LogicOp::Or => BinOp::BitOr,
1173                            },
1174                            hir_def::hir::BinaryOp::ArithOp(op) => BinOp::from(op),
1175                            hir_def::hir::BinaryOp::CmpOp(op) => BinOp::from(op),
1176                            hir_def::hir::BinaryOp::Assignment { .. } => unreachable!(), // handled above
1177                        },
1178                        lhs_op,
1179                        rhs_op,
1180                    ),
1181                    expr_id.into(),
1182                );
1183                Ok(Some(current))
1184            }
1185            &Expr::Assignment { target, value } => {

1187                else {

1196            }

1228                            .iter()
1229                            .map(|it| {

1234                                        Box::new([0]),
1235                                        Ty::new_bool(self.interner()),
1236                                    )),
1237                                    _ => None,
1238                                };
1239                                o.ok_or(MirLowerError::UnresolvedField)
1240                            })
1241                            .collect::<Result<'_, _>>()?,
1242                    ),
1243                    expr_id.into(),
1244                );
1245                Ok(Some(current))
1246            }
1247            Expr::Closure { .. } => {
1248                let ty = self.expr_ty_without_adjust(expr_id);
1249                let TyKind::Closure(id, _) = ty.kind() else {
1250                    not_supported!("closure with non closure type");
1251                };
1252                self.result.closures.push(id.0);
1253                let (captures, _) = self.infer.closure_info(id.0);
1254                let mut operands = vec![];
1255                for capture in captures.iter() {
1256                    let p = Place {
1257                        local: self.binding_local(capture.place.local)?,
1258                        projection: self.result.projection_store.intern(
1259                            capture
1260                                .place
1261                                .projections
1262                                .clone()
1263                                .into_iter()
1264                                .map(|it| match it {
1265                                    HirPlaceProjection::Deref => ProjectionElem::Deref,
1266                                    HirPlaceProjection::Field(field_id) => {
1267                                        ProjectionElem::Field(Either::Left(field_id))
1268                                    }
1269                                    HirPlaceProjection::TupleField(idx) => {
1270                                        ProjectionElem::Field(Either::Right(TupleFieldId {
1271                                            tuple: TupleId(!0), // Dummy as it's unused
1272                                            index: idx,
1273                                        }))
1274                                    }
1275                                })
1276                                .collect(),
1277                        ),
1278                    };
1279                    match &capture.kind {
1280                        CaptureKind::ByRef(bk) => {
1281                            let tmp_ty = capture.ty.get().instantiate_identity();
1282                            // FIXME: Handle more than one span.
1283                            let capture_spans = capture.spans();
1284                            let tmp: Place = self.temp(tmp_ty, current, capture_spans[0])?.into();
1285                            self.push_assignment(
1286                                current,
1287                                tmp,
1288                                Rvalue::Ref(*bk, p),
1289                                capture_spans[0],
1290                            );
1291                            operands.push(Operand { kind: OperandKind::Move(tmp), span: None });
1292                        }
1293                        CaptureKind::ByValue => {
1294                            operands.push(Operand { kind: OperandKind::Move(p), span: None })
1295                        }
1296                    }
1297                }
1298                self.push_assignment(
1299                    current,
1300                    place,
1301                    Rvalue::Aggregate(AggregateKind::Closure(ty.store()), operands.into()),
1302                    expr_id.into(),
1303                );
1304                Ok(Some(current))
1305            }
1306            Expr::Tuple { exprs } => {
1307                let Some(values) = exprs
1308                    .iter()
1309                    .map(|it| {
1310                        let Some((o, c)) = self.lower_expr_to_some_operand(*it, current)? else {
1311                            return Ok(None);
1312                        };
1313                        current = c;
1314                        Ok(Some(o))
1315                    })
1316                    .collect::<Result<'_, Option<_>>>()?
1317                else {
1318                    return Ok(None);
1319                };
1320                let r = Rvalue::Aggregate(
1321                    AggregateKind::Tuple(self.expr_ty_without_adjust(expr_id).store()),
1322                    values,
1323                );
1324                self.push_assignment(current, place, r, expr_id.into());
1325                Ok(Some(current))
1326            }
1327            Expr::Array(l) => match l {
1328                Array::ElementList { elements, .. } => {
1329                    let elem_ty = match self.expr_ty_without_adjust(expr_id).kind() {
1330                        TyKind::Array(ty, _) => ty,
1331                        _ => {
1332                            return Err(MirLowerError::TypeError(
1333                                "Array expression with non array type",
1334                            ));
1335                        }
1336                    };
1337                    let Some(values) = elements
1338                        .iter()
1339                        .map(|it| {
1340                            let Some((o, c)) = self.lower_expr_to_some_operand(*it, current)?
1341                            else {
1342                                return Ok(None);
1343                            };
1344                            current = c;
1345                            Ok(Some(o))
1346                        })
1347                        .collect::<Result<'_, Option<_>>>()?
1348                    else {
1349                        return Ok(None);
1350                    };
1351                    let r = Rvalue::Aggregate(AggregateKind::Array(elem_ty.store()), values);
1352                    self.push_assignment(current, place, r, expr_id.into());
1353                    Ok(Some(current))
1354                }
1355                Array::Repeat { initializer, .. } => {
1356                    let Some((init, current)) =
1357                        self.lower_expr_to_some_operand(*initializer, current)?
1358                    else {
1359                        return Ok(None);
1360                    };
1361                    let len = match self.expr_ty_without_adjust(expr_id).kind() {
1362                        TyKind::Array(_, len) => len,
1363                        _ => {
1364                            return Err(MirLowerError::TypeError(
1365                                "Array repeat expression with non array type",
1366                            ));
1367                        }
1368                    };
1369                    let r = Rvalue::Repeat(init, len.store());
1370                    self.push_assignment(current, place, r, expr_id.into());
1371                    Ok(Some(current))
1372                }
1373            },
1374            Expr::Literal(l) => {
1375                let ty = self.expr_ty_without_adjust(expr_id);
1376                let op = self.lower_literal_to_operand(ty, l)?;
1377                self.push_assignment(current, place, op.into(), expr_id.into());
1378                Ok(Some(current))
1379            }
1380            Expr::Underscore => Ok(Some(current)),
1381        }
1382    }

1410        &mut self,
1411        ty: Ty<'db>,
1412        loc: &ExprId,
1413    ) -> Result<'db, Operand> {
1414        match &self.body[*loc] {
1415            Expr::Literal(l) => self.lower_literal_to_operand(ty, l),
1416            Expr::Path(c) => {

1418                let db = self.db;

1421                        self.db,

1453    fn lower_literal_to_operand(&mut self, ty: Ty<'db>, l: &Literal) -> Result<'db, Operand> {
1454        let size = || {
1455            self.db
1456                .layout_of_ty(
1457                    ty.store(),
1458                    ParamEnvAndCrate { param_env: self.env, krate: self.krate() }.store(),
1459                )
1460                .map(|it| it.size.bytes_usize())
1461        };
1462        const USIZE_SIZE: usize = size_of::<usize>();
1463        let bytes: Box<[_]> = match l {
1464            hir_def::hir::Literal::String(b) => {
1465                let b = b.as_str();
1466                let mut data = [0; { 2 * USIZE_SIZE }];
1467                data[..USIZE_SIZE].copy_from_slice(&0usize.to_le_bytes());
1468                data[USIZE_SIZE..].copy_from_slice(&b.len().to_le_bytes());
1469                let mm = MemoryMap::simple(b.as_bytes().into());
1470                return Ok(Operand::from_concrete_const(Box::new(data), mm, ty));
1471            }
1472            hir_def::hir::Literal::CString(b) => {
1473                let bytes = b.iter().copied().chain(iter::once(0)).collect::<Box<_>>();
1474
1475                let mut data = [0; { 2 * USIZE_SIZE }];
1476                data[..USIZE_SIZE].copy_from_slice(&0usize.to_le_bytes());
1477                data[USIZE_SIZE..].copy_from_slice(&bytes.len().to_le_bytes());
1478                let mm = MemoryMap::simple(bytes);
1479                return Ok(Operand::from_concrete_const(Box::new(data), mm, ty));
1480            }
1481            hir_def::hir::Literal::ByteString(b) => {
1482                let mut data = [0; { 2 * USIZE_SIZE }];
1483                data[..USIZE_SIZE].copy_from_slice(&0usize.to_le_bytes());
1484                data[USIZE_SIZE..].copy_from_slice(&b.len().to_le_bytes());
1485                let mm = MemoryMap::simple(b.clone());
1486                return Ok(Operand::from_concrete_const(Box::new(data), mm, ty));
1487            }
1488            hir_def::hir::Literal::Char(c) => Box::new(u32::from(*c).to_le_bytes()),
1489            hir_def::hir::Literal::Bool(b) => Box::new([*b as u8]),
1490            hir_def::hir::Literal::Int(it, _) => Box::from(&it.to_le_bytes()[0..size()?]),
1491            hir_def::hir::Literal::Uint(it, _) => Box::from(&it.to_le_bytes()[0..size()?]),
1492            hir_def::hir::Literal::Float(f, _) => match size()? {
1493                16 => Box::new(f.to_f128().to_bits().to_le_bytes()),
1494                8 => Box::new(f.to_f64().to_le_bytes()),
1495                4 => Box::new(f.to_f32().to_le_bytes()),
1496                2 => Box::new(u16::try_from(f.to_f16().to_bits()).unwrap().to_le_bytes()),
1497                _ => {
1498                    return Err(MirLowerError::TypeError(
1499                        "float with size other than 2, 4, 8 or 16 bytes",
1500                    ));
1501                }
1502            },
1503        };
1504        Ok(Operand::from_concrete_const(bytes, MemoryMap::default(), ty))
1505    }

1524    fn lower_const_to_operand(
1525        &mut self,
1526        subst: GenericArgs<'db>,
1527        const_id: GeneralConstId,
1528    ) -> Result<'db, Operand> {
1529        let konst = if !subst.is_empty() {
1530            // We can't evaluate constant with substitution now, as generics are not monomorphized in lowering.
1531            Const::new_unevaluated(
1532                self.interner(),
1533                UnevaluatedConst { def: const_id.into(), args: subst },
1534            )
1535        } else {
1536            match const_id {
1537                id @ GeneralConstId::ConstId(const_id) => {
1538                    self.db.const_eval(const_id, subst, None).map_err(|e| {
1539                        let name = id.name(self.db);

1541                    })?
1542                }
1543                GeneralConstId::StaticId(static_id) => {
1544                    self.db.const_eval_static(static_id).map_err(|e| {
1545                        let name = const_id.name(self.db);
1546                        MirLowerError::ConstEvalError(name.into(), Box::new(e))
1547                    })?
1548                }
1549            }
1550        };
1551        let ty = self
1552            .db
1553            .value_ty(match const_id {
1554                GeneralConstId::ConstId(id) => id.into(),
1555                GeneralConstId::StaticId(id) => id.into(),
1556            })
1557            .unwrap()
1558            .instantiate(self.interner(), subst);
1559        Ok(Operand {
1560            kind: OperandKind::Constant { konst: konst.store(), ty: ty.store() },
1561            span: None,
1562        })
1563    }

1709    fn lower_loop(
1710        &mut self,
1711        prev_block: BasicBlockId,
1712        place: Place,
1713        label: Option<LabelId>,
1714        span: MirSpan,
1715        f: impl FnOnce(&mut MirLowerCtx<'_, 'db>, BasicBlockId) -> Result<'db, ()>,
1716    ) -> Result<'db, Option<BasicBlockId>> {
1717        let begin = self.new_basic_block();
1718        let prev = self.current_loop_blocks.replace(LoopBlocks {
1719            begin,
1720            end: None,
1721            place,
1722            drop_scope_index: self.drop_scopes.len(),
1723        });
1724        let prev_label = if let Some(label) = label {
1725            // We should generate the end now, to make sure that it wouldn't change later. It is
1726            // bad as we may emit end (unnecessary unreachable block) for unterminating loop, but
1727            // it should not affect correctness.
1728            self.current_loop_end()?;
1729            self.labeled_loop_blocks
1730                .insert(label, self.current_loop_blocks.as_ref().unwrap().clone())
1731        } else {
1732            None
1733        };
1734        self.set_goto(prev_block, begin, span);
1735        f(self, begin)?;
1736        let my = mem::replace(&mut self.current_loop_blocks, prev).ok_or(
1737            MirLowerError::ImplementationError("current_loop_blocks is corrupt".to_owned()),
1738        )?;
1739        if let Some(prev) = prev_label {
1740            self.labeled_loop_blocks.insert(label.unwrap(), prev);
1741        }
1742        Ok(my.end)
1743    }

1818    fn lower_block_to_place(
1819        &mut self,
1820        statements: &[hir_def::hir::Statement],
1821        mut current: BasicBlockId,
1822        tail: Option<ExprId>,
1823        place: Place,
1824        span: MirSpan,
1825    ) -> Result<'db, Option<Idx<BasicBlock>>> {
1826        let scope = self.push_drop_scope();
1827        for statement in statements.iter() {
1828            match statement {
1829                hir_def::hir::Statement::Let { pat, initializer, else_branch, type_ref: _ } => {
1830                    if let Some(expr_id) = initializer {
1831                        let else_block;
1832                        let Some((init_place, c)) =
1833                            self.lower_expr_as_place(current, *expr_id, true)?
1834                        else {
1835                            scope.pop_assume_dropped(self);
1836                            return Ok(None);
1837                        };
1838                        current = c;
1839                        self.push_fake_read(current, init_place, span);
1840                        // Using the initializer for the resolver scope is good enough for us, as it cannot create new declarations
1841                        // and has all declarations of the `let`.
1842                        let resolver_guard =
1843                            self.resolver.update_to_inner_scope(self.db, self.owner, *expr_id);
1844                        (current, else_block) =
1845                            self.pattern_match(current, None, init_place, *pat)?;
1846                        self.resolver.reset_to_guard(resolver_guard);
1847                        match (else_block, else_branch) {
1848                            (None, _) => (),
1849                            (Some(else_block), None) => {
1850                                self.set_terminator(else_block, TerminatorKind::Unreachable, span);
1851                            }
1852                            (Some(else_block), Some(else_branch)) => {
1853                                if let Some((_, b)) =
1854                                    self.lower_expr_as_place(else_block, *else_branch, true)?
1855                                {
1856                                    self.set_terminator(b, TerminatorKind::Unreachable, span);
1857                                }
1858                            }
1859                        }
1860                    } else {
1861                        let mut err = None;
1862                        self.body.walk_bindings_in_pat(*pat, |b| {
1863                            if let Err(e) = self.push_storage_live(b, current) {
1864                                err = Some(e);
1865                            }
1866                        });
1867                        if let Some(e) = err {
1868                            return Err(e);
1869                        }
1870                    }
1871                }
1872                &hir_def::hir::Statement::Expr { expr, has_semi: _ } => {
1873                    let scope2 = self.push_drop_scope();
1874                    let Some((p, c)) = self.lower_expr_as_place(current, expr, true)? else {
1875                        scope2.pop_assume_dropped(self);
1876                        scope.pop_assume_dropped(self);
1877                        return Ok(None);
1878                    };
1879                    self.push_fake_read(c, p, expr.into());
1880                    current = scope2.pop_and_drop(self, c, expr.into());
1881                }
1882                hir_def::hir::Statement::Item(_) => (),
1883            }
1884        }
1885        if let Some(tail) = tail {
1886            let Some(c) = self.lower_expr_to_place(tail, place, current)? else {
1887                scope.pop_assume_dropped(self);
1888                return Ok(None);
1889            };
1890            current = c;
1891        }
1892        current = scope.pop_and_drop(self, current, span);
1893        Ok(Some(current))
1894    }

1896    fn lower_params_and_bindings(
1897        &mut self,
1898        params: impl Iterator<Item = (PatId, Ty<'db>)> + Clone,
1899        self_binding: Option<(BindingId, Ty<'db>)>,
1900        pick_binding: impl Fn(BindingId) -> bool,
1901    ) -> Result<'db, BasicBlockId> {
1902        let base_param_count = self.result.param_locals.len();
1903        let self_binding = match self_binding {
1904            Some((self_binding, ty)) => {
1905                let local_id = self.result.locals.alloc(Local { ty: ty.store() });
1906                self.drop_scopes.last_mut().unwrap().locals.push(local_id);
1907                self.result.binding_locals.insert(self_binding, local_id);
1908                self.result.param_locals.push(local_id);
1909                Some(self_binding)
1910            }
1911            None => None,
1912        };
1913        self.result.param_locals.extend(params.clone().map(|(it, ty)| {
1914            let local_id = self.result.locals.alloc(Local { ty: ty.store() });
1915            self.drop_scopes.last_mut().unwrap().locals.push(local_id);
1916            if let Pat::Bind { id, subpat: None } = self.body[it]
1917                && matches!(
1918                    self.body[id].mode,
1919                    BindingAnnotation::Unannotated | BindingAnnotation::Mutable
1920                )
1921            {
1922                self.result.binding_locals.insert(id, local_id);
1923            }
1924            local_id
1925        }));
1926        // and then rest of bindings
1927        for (id, _) in self.body.bindings() {
1928            if !pick_binding(id) {
1929                continue;
1930            }
1931            if !self.result.binding_locals.contains_idx(id) {
1932                self.result.binding_locals.insert(
1933                    id,
1934                    self.result.locals.alloc(Local { ty: self.infer.binding_ty(id).store() }),
1935                );
1936            }
1937        }
1938        let mut current = self.result.start_block;
1939        if let Some(self_binding) = self_binding {
1940            let local = self.result.param_locals.clone()[base_param_count];
1941            if local != self.binding_local(self_binding)? {
1942                let r = self.match_self_param(self_binding, current, local)?;
1943                if let Some(b) = r.1 {
1944                    self.set_terminator(b, TerminatorKind::Unreachable, MirSpan::SelfParam);
1945                }
1946                current = r.0;
1947            }
1948        }
1949        let local_params = self
1950            .result
1951            .param_locals
1952            .clone()
1953            .into_iter()
1954            .skip(base_param_count + self_binding.is_some() as usize);
1955        for ((param, _), local) in params.zip(local_params) {
1956            if let Pat::Bind { id, .. } = self.body[param]
1957                && local == self.binding_local(id)?
1958            {
1959                continue;
1960            }
1961            let r = self.pattern_match(current, None, local.into(), param)?;
1962            if let Some(b) = r.1 {
1963                self.set_terminator(b, TerminatorKind::Unreachable, param.into());
1964            }
1965            current = r.0;
1966        }
1967        Ok(current)
1968    }

2082fn cast_kind<'db>(
2083    db: &'db dyn HirDatabase,
2084    source_ty: Ty<'db>,
2085    target_ty: Ty<'db>,

2154        let local = ctx.binding_local(capture.place.local)?;

2161        FnTrait::FnMut | FnTrait::Fn | FnTrait::AsyncFnMut | FnTrait::AsyncFn => {
2162            vec![ProjectionElem::Deref]
2163        }
2164    };

2166        if let Some(it) = upvar_map.get(&p.local) {

2169                    return false;
2170                }
2171                for (it, y) in p.projection.lookup(store).iter().zip(it.0.place.projections.iter())

2173                    match (it, y) {
2174                        (ProjectionElem::Deref, HirPlaceProjection::Deref) => (),
2175                        (ProjectionElem::Field(Either::Left(it)), HirPlaceProjection::Field(y))
2176                            if it == y => {}
2177                        (
2178                            ProjectionElem::Field(Either::Right(it)),
2179                            HirPlaceProjection::TupleField(y),
2180                        ) if it.index == *y => (),
2181                        _ => return false,
2182                    }
2183                }
2184                true
2185            });
2186            match r {
2187                Some(it) => {
2188                    p.local = closure_local;
2189                    let mut next_projs = closure_projection.clone();

2191                    let prev_projs = p.projection;
2192                    if it.0.kind != CaptureKind::ByValue {
2193                        next_projs.push(ProjectionElem::Deref);
2194                    }
2195                    next_projs.extend(
2196                        prev_projs.lookup(store).iter().skip(it.0.place.projections.len()).cloned(),

2198                    p.projection = store.intern(next_projs.into());

2200                None => err = Some(*p),

2202        }
2203    });

2206        .binding_locals

2261pub fn lower_to_mir<'db>(
2262    db: &'db dyn HirDatabase,
2263    owner: DefWithBodyId,
2264    body: &Body,
2265    infer: &InferenceResult,
2266    // FIXME: root_expr should always be the body.body_expr, but since `X` in `[(); X]` doesn't have its own specific body yet, we
2267    // need to take this input explicitly.
2268    root_expr: ExprId,
2269) -> Result<'db, MirBody> {
2270    if infer.type_mismatches().next().is_some() || infer.is_erroneous() {
2271        return Err(MirLowerError::HasErrors);
2272    }
2273    let mut ctx = MirLowerCtx::new(db, owner, body, infer);
2274    // 0 is return local
2275    ctx.result.locals.alloc(Local { ty: ctx.expr_ty_after_adjustments(root_expr).store() });
2276    let binding_picker = |b: BindingId| {
2277        let owner = ctx.body.binding_owner(b);
2278        if root_expr == body.body_expr { owner.is_none() } else { owner == Some(root_expr) }
2279    };
2280    // 1 to param_len is for params
2281    // FIXME: replace with let chain once it becomes stable
2282    let current = 'b: {
2283        if body.body_expr == root_expr {
2284            // otherwise it's an inline const, and has no parameter
2285            if let DefWithBodyId::FunctionId(fid) = owner {
2286                let callable_sig =
2287                    db.callable_item_signature(fid.into()).instantiate_identity().skip_binder();
2288                let mut params = callable_sig.inputs().iter().copied();
2289                let self_param = body.self_param.and_then(|id| Some((id, params.next()?)));
2290                break 'b ctx.lower_params_and_bindings(
2291                    body.params.iter().zip(params).map(|(it, y)| (*it, y)),
2292                    self_param,
2293                    binding_picker,
2294                )?;
2295            }
2296        }
2297        ctx.lower_params_and_bindings([].into_iter(), None, binding_picker)?
2298    };
2299    if let Some(current) = ctx.lower_expr_to_place(root_expr, return_slot().into(), current)? {
2300        let current = ctx.pop_drop_scope_assert_finished(current, root_expr.into())?;
2301        ctx.set_terminator(current, TerminatorKind::Return, root_expr.into());
2302    }
2303    Ok(ctx.result)
