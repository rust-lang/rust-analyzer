122/// A token to ensuring that each drop scope is popped at most once, thanks to the compiler that checks moves.
123struct DropScopeToken;
124impl DropScopeToken {
125    fn pop_and_drop<'db>(
126        self,
127        ctx: &mut MirLowerCtx<'_, 'db>,
128        current: BasicBlockId<'db>,
129        span: MirSpan,
130    ) -> BasicBlockId<'db> {
131        std::mem::forget(self);
132        ctx.pop_drop_scope_internal(current, span)
133    }
134
135    /// It is useful when we want a drop scope is syntactically closed, but we don't want to execute any drop
136    /// code. Either when the control flow is diverging (so drop code doesn't reached) or when drop is handled
137    /// for us (for example a block that ended with a return statement. Return will drop everything, so the block shouldn't
138    /// do anything)
139    fn pop_assume_dropped(self, ctx: &mut MirLowerCtx<'_, '_>) {
140        std::mem::forget(self);
141        ctx.pop_drop_scope_assume_dropped_internal();
142    }
143}

160impl MirLowerError<'_> {
161    pub fn pretty_print(
162        &self,
163        f: &mut String,
164        db: &dyn HirDatabase,
165        span_formatter: impl Fn(FileId, TextRange) -> String,
166        display_target: DisplayTarget,
167    ) -> std::result::Result<(), std::fmt::Error> {
168        match self {
169            MirLowerError::ConstEvalError(name, e) => {
170                writeln!(f, "In evaluating constant {name}")?;
171                match &**e {
172                    ConstEvalError::MirLowerError(e) => {
173                        e.pretty_print(f, db, span_formatter, display_target)?
174                    }
175                    ConstEvalError::MirEvalError(e) => {
176                        e.pretty_print(f, db, span_formatter, display_target)?
177                    }
178                }
179            }
180            MirLowerError::MissingFunctionDefinition(owner, it) => {
181                let body = db.body(*owner);
182                writeln!(
183                    f,
184                    "Missing function definition for {}",
185                    body.pretty_print_expr(db, *owner, *it, display_target.edition)
186                )?;
187            }
188            MirLowerError::HasErrors => writeln!(f, "Type inference result contains errors")?,
189            MirLowerError::TypeMismatch(e) => writeln!(
190                f,
191                "Type mismatch: Expected {}, found {}",
192                e.expected.display(db, display_target),
193                e.actual.display(db, display_target),
194            )?,
195            MirLowerError::GenericArgNotProvided(id, subst) => {
196                let param_name = match *id {
197                    GenericParamId::TypeParamId(id) => {
198                        db.generic_params(id.parent())[id.local_id()].name().cloned()
199                    }
200                    GenericParamId::ConstParamId(id) => {
201                        db.generic_params(id.parent())[id.local_id()].name().cloned()
202                    }
203                    GenericParamId::LifetimeParamId(id) => {
204                        Some(db.generic_params(id.parent)[id.local_id].name.clone())
205                    }
206                };
207                writeln!(
208                    f,
209                    "Generic arg not provided for {}",
210                    param_name.unwrap_or(Name::missing()).display(db, display_target.edition)
211                )?;
212                writeln!(f, "Provided args: [")?;
213                for g in subst.iter() {
214                    write!(f, "    {},", g.display(db, display_target))?;
215                }
216                writeln!(f, "]")?;
217            }
218            MirLowerError::LayoutError(_)
219            | MirLowerError::UnsizedTemporary(_)
220            | MirLowerError::IncompleteExpr
221            | MirLowerError::IncompletePattern
222            | MirLowerError::InaccessibleLocal
223            | MirLowerError::TraitFunctionDefinition(_, _)
224            | MirLowerError::UnresolvedName(_)
225            | MirLowerError::RecordLiteralWithoutPath
226            | MirLowerError::UnresolvedMethod(_)
227            | MirLowerError::UnresolvedField
228            | MirLowerError::TypeError(_)
229            | MirLowerError::NotSupported(_)
230            | MirLowerError::ContinueWithoutLoop
231            | MirLowerError::BreakWithoutLoop
232            | MirLowerError::Loop
233            | MirLowerError::ImplementationError(_)
234            | MirLowerError::LangItemNotFound(_)
235            | MirLowerError::MutatingRvalue
236            | MirLowerError::UnresolvedLabel
237            | MirLowerError::UnresolvedUpvar(_) => writeln!(f, "{self:?}")?,
238        }
239        Ok(())
240    }
241}

278    fn new(
279        db: &'db dyn HirDatabase,
280        owner: DefWithBodyId,
281        body: &'a Body,
282        infer: &'a InferenceResult<'db>,
283    ) -> Self {
284        let mut basic_blocks = Arena::new();
285        let start_block = basic_blocks.alloc(BasicBlock {
286            statements: vec![],
287            terminator: None,
288            is_cleanup: false,
289        });
290        let locals = Arena::new();
291        let binding_locals: ArenaMap<BindingId, LocalId<'db>> = ArenaMap::new();

293            projection_store: ProjectionStore::default(),
294            basic_blocks,
295            locals,
296            start_block,
297            binding_locals,
298            param_locals: vec![],
299            owner,
300            closures: vec![],
301        };
302        let resolver = owner.resolver(db);
303        let env = db.trait_environment_for_body(owner);
304        let interner = DbInterner::new_with(db, Some(env.krate), env.block);
305        // FIXME(next-solver): Is `non_body_analysis()` correct here? Don't we want to reveal opaque types defined by this body?
306        let infcx = interner.infer_ctxt().build(TypingMode::non_body_analysis());
307
308        MirLowerCtx {
309            result: mir,
310            db,
311            infer,
312            body,
313            owner,
314            resolver,
315            current_loop_blocks: None,
316            labeled_loop_blocks: Default::default(),
317            discr_temp: None,
318            drop_scopes: vec![DropScope::default()],
319            env,
320            infcx,
321        }
322    }

329    fn temp(
330        &mut self,
331        ty: Ty<'db>,
332        current: BasicBlockId<'db>,
333        span: MirSpan,
334    ) -> Result<'db, LocalId<'db>> {
335        if matches!(ty.kind(), TyKind::Slice(_) | TyKind::Dynamic(..)) {
336            return Err(MirLowerError::UnsizedTemporary(ty));
337        }
338        let l = self.result.locals.alloc(Local { ty });
339        self.push_storage_live_for_local(l, current, span)?;
340        Ok(l)
341    }

343    fn lower_expr_to_some_operand(
344        &mut self,
345        expr_id: ExprId,
346        current: BasicBlockId<'db>,
347    ) -> Result<'db, Option<(Operand<'db>, BasicBlockId<'db>)>> {
348        if !self.has_adjustments(expr_id)
349            && let Expr::Literal(l) = &self.body[expr_id]
350        {
351            let ty = self.expr_ty_without_adjust(expr_id);
352            return Ok(Some((self.lower_literal_to_operand(ty, l)?, current)));
353        }
354        let Some((p, current)) = self.lower_expr_as_place(current, expr_id, true)? else {
355            return Ok(None);
356        };
357        Ok(Some((Operand { kind: OperandKind::Copy(p), span: Some(expr_id.into()) }, current)))
358    }

360    fn lower_expr_to_place_with_adjust(
361        &mut self,
362        expr_id: ExprId,
363        place: Place<'db>,
364        current: BasicBlockId<'db>,
365        adjustments: &[Adjustment<'db>],
366    ) -> Result<'db, Option<BasicBlockId<'db>>> {
367        match adjustments.split_last() {
368            Some((last, rest)) => match &last.kind {
369                Adjust::NeverToAny => {
370                    let temp = self.temp(
371                        Ty::new(self.interner(), TyKind::Never),
372                        current,
373                        MirSpan::Unknown,
374                    )?;
375                    self.lower_expr_to_place_with_adjust(expr_id, temp.into(), current, rest)
376                }
377                Adjust::Deref(_) => {
378                    let Some((p, current)) =
379                        self.lower_expr_as_place_with_adjust(current, expr_id, true, adjustments)?
380                    else {
381                        return Ok(None);
382                    };
383                    self.push_assignment(
384                        current,
385                        place,
386                        Operand { kind: OperandKind::Copy(p), span: None }.into(),
387                        expr_id.into(),
388                    );
389                    Ok(Some(current))
390                }
391                Adjust::Borrow(AutoBorrow::Ref(_, m) | AutoBorrow::RawPtr(m)) => {
392                    let Some((p, current)) =
393                        self.lower_expr_as_place_with_adjust(current, expr_id, true, rest)?
394                    else {
395                        return Ok(None);
396                    };
397                    let bk = BorrowKind::from_rustc(*m);
398                    self.push_assignment(current, place, Rvalue::Ref(bk, p), expr_id.into());
399                    Ok(Some(current))
400                }
401                Adjust::Pointer(cast) => {
402                    let Some((p, current)) =
403                        self.lower_expr_as_place_with_adjust(current, expr_id, true, rest)?
404                    else {
405                        return Ok(None);
406                    };
407                    self.push_assignment(
408                        current,
409                        place,
410                        Rvalue::Cast(
411                            CastKind::PointerCoercion(*cast),
412                            Operand { kind: OperandKind::Copy(p), span: None },
413                            last.target,
414                        ),
415                        expr_id.into(),
416                    );
417                    Ok(Some(current))
418                }
419            },
420            None => self.lower_expr_to_place_without_adjust(expr_id, place, current),
421        }
422    }

424    fn lower_expr_to_place(
425        &mut self,
426        expr_id: ExprId,
427        place: Place<'db>,
428        prev_block: BasicBlockId<'db>,
429    ) -> Result<'db, Option<BasicBlockId<'db>>> {
430        if let Some(adjustments) = self.infer.expr_adjustments.get(&expr_id) {
431            return self.lower_expr_to_place_with_adjust(expr_id, place, prev_block, adjustments);
432        }
433        self.lower_expr_to_place_without_adjust(expr_id, place, prev_block)
434    }

436    fn lower_expr_to_place_without_adjust(
437        &mut self,
438        expr_id: ExprId,
439        place: Place<'db>,
440        mut current: BasicBlockId<'db>,
441    ) -> Result<'db, Option<BasicBlockId<'db>>> {
442        match &self.body[expr_id] {
443            Expr::OffsetOf(_) => {
444                not_supported!("builtin#offset_of")
445            }
446            Expr::InlineAsm(_) => {
447                not_supported!("builtin#asm")
448            }
449            Expr::Missing => {
450                if let DefWithBodyId::FunctionId(f) = self.owner {
451                    let assoc = f.lookup(self.db);
452                    if let ItemContainerId::TraitId(t) = assoc.container {
453                        let name = &self.db.function_signature(f).name;
454                        return Err(MirLowerError::TraitFunctionDefinition(t, name.clone()));
455                    }
456                }
457                Err(MirLowerError::IncompleteExpr)
458            }
459            Expr::Path(p) => {
460                let pr =
461                    if let Some((assoc, subst)) = self.infer.assoc_resolutions_for_expr(expr_id) {
462                        match assoc {
463                            hir_def::AssocItemId::ConstId(c) => {
464                                self.lower_const(c.into(), current, place, subst, expr_id.into())?;
465                                return Ok(Some(current));
466                            }
467                            hir_def::AssocItemId::FunctionId(_) => {
468                                // FnDefs are zero sized, no action is needed.
469                                return Ok(Some(current));
470                            }
471                            hir_def::AssocItemId::TypeAliasId(_) => {
472                                // FIXME: If it is unreachable, use proper error instead of `not_supported`.
473                                not_supported!("associated functions and types")
474                            }
475                        }
476                    } else if let Some(variant) = self.infer.variant_resolution_for_expr(expr_id) {
477                        match variant {
478                            VariantId::EnumVariantId(e) => ValueNs::EnumVariantId(e),
479                            VariantId::StructId(s) => ValueNs::StructId(s),
480                            VariantId::UnionId(_) => implementation_error!("Union variant as path"),
481                        }
482                    } else {
483                        let resolver_guard =
484                            self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
485                        let hygiene = self.body.expr_path_hygiene(expr_id);
486                        let result = self
487                            .resolver
488                            .resolve_path_in_value_ns_fully(self.db, p, hygiene)
489                            .ok_or_else(|| {
490                                MirLowerError::unresolved_path(
491                                    self.db,
492                                    p,
493                                    DisplayTarget::from_crate(self.db, self.krate()),
494                                    self.body,
495                                )
496                            })?;
497                        self.resolver.reset_to_guard(resolver_guard);
498                        result
499                    };
500                match pr {
501                    ValueNs::LocalBinding(_) | ValueNs::StaticId(_) => {
502                        let Some((temp, current)) =
503                            self.lower_expr_as_place_without_adjust(current, expr_id, false)?
504                        else {
505                            return Ok(None);
506                        };
507                        self.push_assignment(
508                            current,
509                            place,
510                            Operand { kind: OperandKind::Copy(temp), span: None }.into(),
511                            expr_id.into(),
512                        );
513                        Ok(Some(current))
514                    }
515                    ValueNs::ConstId(const_id) => {
516                        self.lower_const(
517                            const_id.into(),
518                            current,
519                            place,
520                            GenericArgs::new_from_iter(self.interner(), []),
521                            expr_id.into(),
522                        )?;
523                        Ok(Some(current))
524                    }
525                    ValueNs::EnumVariantId(variant_id) => {
526                        let variant_fields = variant_id.fields(self.db);
527                        if variant_fields.shape == FieldsShape::Unit {
528                            let ty = self.infer.type_of_expr[expr_id];
529                            current = self.lower_enum_variant(
530                                variant_id,
531                                current,
532                                place,
533                                ty,
534                                Box::new([]),
535                                expr_id.into(),
536                            )?;
537                        }
538                        // Otherwise its a tuple like enum, treated like a zero sized function, so no action is needed
539                        Ok(Some(current))
540                    }
541                    ValueNs::GenericParam(p) => {
542                        let Some(def) = self.owner.as_generic_def_id(self.db) else {
543                            not_supported!("owner without generic def id");
544                        };
545                        let generics = generics(self.db, def);
546                        let index = generics
547                            .type_or_const_param_idx(p.into())
548                            .ok_or(MirLowerError::TypeError("fail to lower const generic param"))?
549                            as u32;
550                        self.push_assignment(
551                            current,
552                            place,
553                            Rvalue::from(Operand {
554                                kind: OperandKind::Constant {
555                                    konst: Const::new_param(
556                                        self.interner(),
557                                        ParamConst { id: p, index },
558                                    ),
559                                    ty: self.db.const_param_ty_ns(p),
560                                },
561                                span: None,
562                            }),
563                            expr_id.into(),
564                        );
565                        Ok(Some(current))
566                    }
567                    ValueNs::FunctionId(_) | ValueNs::StructId(_) | ValueNs::ImplSelf(_) => {
568                        // It's probably a unit struct or a zero sized function, so no action is needed.
569                        Ok(Some(current))
570                    }
571                }
572            }
573            Expr::If { condition, then_branch, else_branch } => {
574                let Some((discr, current)) =
575                    self.lower_expr_to_some_operand(*condition, current)?
576                else {
577                    return Ok(None);
578                };
579                let start_of_then = self.new_basic_block();
580                let end_of_then = self.lower_expr_to_place(*then_branch, place, start_of_then)?;
581                let start_of_else = self.new_basic_block();
582                let end_of_else = if let Some(else_branch) = else_branch {
583                    self.lower_expr_to_place(*else_branch, place, start_of_else)?
584                } else {
585                    Some(start_of_else)
586                };
587                self.set_terminator(
588                    current,
589                    TerminatorKind::SwitchInt {
590                        discr,
591                        targets: SwitchTargets::static_if(1, start_of_then, start_of_else),
592                    },
593                    expr_id.into(),
594                );
595                Ok(self.merge_blocks(end_of_then, end_of_else, expr_id.into()))
596            }
597            Expr::Let { pat, expr } => {
598                let Some((cond_place, current)) = self.lower_expr_as_place(current, *expr, true)?
599                else {
600                    return Ok(None);
601                };
602                self.push_fake_read(current, cond_place, expr_id.into());
603                let resolver_guard =
604                    self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
605                let (then_target, else_target) =
606                    self.pattern_match(current, None, cond_place, *pat)?;
607                self.resolver.reset_to_guard(resolver_guard);
608                self.write_bytes_to_place(
609                    then_target,
610                    place,
611                    Box::new([1]),
612                    Ty::new_bool(self.interner()),
613                    MirSpan::Unknown,
614                )?;
615                if let Some(else_target) = else_target {
616                    self.write_bytes_to_place(
617                        else_target,
618                        place,
619                        Box::new([0]),
620                        Ty::new_bool(self.interner()),
621                        MirSpan::Unknown,
622                    )?;
623                }
624                Ok(self.merge_blocks(Some(then_target), else_target, expr_id.into()))
625            }
626            Expr::Unsafe { id: _, statements, tail } => {
627                self.lower_block_to_place(statements, current, *tail, place, expr_id.into())
628            }
629            Expr::Block { id: _, statements, tail, label } => {
630                if let Some(label) = label {
631                    self.lower_loop(current, place, Some(*label), expr_id.into(), |this, begin| {
632                        if let Some(current) = this.lower_block_to_place(
633                            statements,
634                            begin,
635                            *tail,
636                            place,
637                            expr_id.into(),
638                        )? {
639                            let end = this.current_loop_end()?;
640                            this.set_goto(current, end, expr_id.into());
641                        }
642                        Ok(())
643                    })
644                } else {
645                    self.lower_block_to_place(statements, current, *tail, place, expr_id.into())
646                }
647            }
648            Expr::Loop { body, label } => {
649                self.lower_loop(current, place, *label, expr_id.into(), |this, begin| {
650                    let scope = this.push_drop_scope();
651                    if let Some((_, mut current)) = this.lower_expr_as_place(begin, *body, true)? {
652                        current = scope.pop_and_drop(this, current, body.into());
653                        this.set_goto(current, begin, expr_id.into());
654                    } else {
655                        scope.pop_assume_dropped(this);
656                    }
657                    Ok(())
658                })
659            }
660            Expr::Call { callee, args, .. } => {
661                if let Some((func_id, generic_args)) = self.infer.method_resolution(expr_id) {
662                    let ty = Ty::new_fn_def(
663                        self.interner(),
664                        CallableDefId::FunctionId(func_id).into(),
665                        generic_args,
666                    );
667                    let func = Operand::from_bytes(Box::default(), ty);
668                    return self.lower_call_and_args(
669                        func,
670                        iter::once(*callee).chain(args.iter().copied()),
671                        place,
672                        current,
673                        self.is_uninhabited(expr_id),
674                        expr_id.into(),
675                    );
676                }
677                let callee_ty = self.expr_ty_after_adjustments(*callee);
678                match callee_ty.kind() {
679                    TyKind::FnDef(..) => {
680                        let func = Operand::from_bytes(Box::default(), callee_ty);
681                        self.lower_call_and_args(
682                            func,
683                            args.iter().copied(),
684                            place,
685                            current,
686                            self.is_uninhabited(expr_id),
687                            expr_id.into(),
688                        )
689                    }
690                    TyKind::FnPtr(..) => {
691                        let Some((func, current)) =
692                            self.lower_expr_to_some_operand(*callee, current)?
693                        else {
694                            return Ok(None);
695                        };
696                        self.lower_call_and_args(
697                            func,
698                            args.iter().copied(),
699                            place,
700                            current,
701                            self.is_uninhabited(expr_id),
702                            expr_id.into(),
703                        )
704                    }
705                    TyKind::Closure(_, _) => {
706                        not_supported!(
707                            "method resolution not emitted for closure (Are Fn traits available?)"
708                        );
709                    }
710                    TyKind::Error(_) => {
711                        Err(MirLowerError::MissingFunctionDefinition(self.owner, expr_id))
712                    }
713                    _ => Err(MirLowerError::TypeError("function call on bad type")),
714                }
715            }
716            Expr::MethodCall { receiver, args, method_name, .. } => {
717                let (func_id, generic_args) =
718                    self.infer.method_resolution(expr_id).ok_or_else(|| {
719                        MirLowerError::UnresolvedMethod(
720                            method_name.display(self.db, self.edition()).to_string(),
721                        )
722                    })?;
723                let func = Operand::from_fn(self.db, func_id, generic_args);
724                self.lower_call_and_args(
725                    func,
726                    iter::once(*receiver).chain(args.iter().copied()),
727                    place,
728                    current,
729                    self.is_uninhabited(expr_id),
730                    expr_id.into(),
731                )
732            }
733            Expr::Match { expr, arms } => {
734                let Some((cond_place, mut current)) =
735                    self.lower_expr_as_place(current, *expr, true)?
736                else {
737                    return Ok(None);
738                };
739                self.push_fake_read(current, cond_place, expr_id.into());
740                let mut end = None;
741                let resolver_guard =
742                    self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
743                for MatchArm { pat, guard, expr } in arms.iter() {
744                    let (then, mut otherwise) =
745                        self.pattern_match(current, None, cond_place, *pat)?;
746                    let then = if let &Some(guard) = guard {
747                        let next = self.new_basic_block();
748                        let o = otherwise.get_or_insert_with(|| self.new_basic_block());
749                        if let Some((discr, c)) = self.lower_expr_to_some_operand(guard, then)? {
750                            self.set_terminator(
751                                c,
752                                TerminatorKind::SwitchInt {
753                                    discr,
754                                    targets: SwitchTargets::static_if(1, next, *o),
755                                },
756                                expr_id.into(),
757                            );
758                        }
759                        next
760                    } else {
761                        then
762                    };
763                    if let Some(block) = self.lower_expr_to_place(*expr, place, then)? {
764                        let r = end.get_or_insert_with(|| self.new_basic_block());
765                        self.set_goto(block, *r, expr_id.into());
766                    }
767                    match otherwise {
768                        Some(o) => current = o,
769                        None => {
770                            // The current pattern was irrefutable, so there is no need to generate code
771                            // for the rest of patterns
772                            break;
773                        }
774                    }
775                }
776                self.resolver.reset_to_guard(resolver_guard);
777                if self.is_unterminated(current) {
778                    self.set_terminator(current, TerminatorKind::Unreachable, expr_id.into());
779                }
780                Ok(end)
781            }
782            Expr::Continue { label } => {
783                let loop_data = match label {
784                    Some(l) => {
785                        self.labeled_loop_blocks.get(l).ok_or(MirLowerError::UnresolvedLabel)?
786                    }
787                    None => self
788                        .current_loop_blocks
789                        .as_ref()
790                        .ok_or(MirLowerError::ContinueWithoutLoop)?,
791                };
792                let begin = loop_data.begin;
793                current =
794                    self.drop_until_scope(loop_data.drop_scope_index, current, expr_id.into());
795                self.set_goto(current, begin, expr_id.into());
796                Ok(None)
797            }
798            &Expr::Break { expr, label } => {
799                if let Some(expr) = expr {
800                    let loop_data = match label {
801                        Some(l) => self
802                            .labeled_loop_blocks
803                            .get(&l)
804                            .ok_or(MirLowerError::UnresolvedLabel)?,
805                        None => self
806                            .current_loop_blocks
807                            .as_ref()
808                            .ok_or(MirLowerError::BreakWithoutLoop)?,
809                    };
810                    let Some(c) = self.lower_expr_to_place(expr, loop_data.place, current)? else {
811                        return Ok(None);
812                    };
813                    current = c;
814                }
815                let (end, drop_scope) = match label {
816                    Some(l) => {
817                        let loop_blocks = self
818                            .labeled_loop_blocks
819                            .get(&l)
820                            .ok_or(MirLowerError::UnresolvedLabel)?;
821                        (
822                            loop_blocks.end.expect("We always generate end for labeled loops"),
823                            loop_blocks.drop_scope_index,
824                        )
825                    }
826                    None => (
827                        self.current_loop_end()?,
828                        self.current_loop_blocks.as_ref().unwrap().drop_scope_index,
829                    ),
830                };
831                current = self.drop_until_scope(drop_scope, current, expr_id.into());
832                self.set_goto(current, end, expr_id.into());
833                Ok(None)
834            }
835            Expr::Return { expr } => {
836                if let Some(expr) = expr {
837                    if let Some(c) =
838                        self.lower_expr_to_place(*expr, return_slot().into(), current)?
839                    {
840                        current = c;
841                    } else {
842                        return Ok(None);
843                    }
844                }
845                current = self.drop_until_scope(0, current, expr_id.into());
846                self.set_terminator(current, TerminatorKind::Return, expr_id.into());
847                Ok(None)
848            }
849            Expr::Become { .. } => not_supported!("tail-calls"),
850            Expr::Yield { .. } => not_supported!("yield"),
851            Expr::RecordLit { fields, path, spread } => {
852                let spread_place = match spread {
853                    &Some(it) => {
854                        let Some((p, c)) = self.lower_expr_as_place(current, it, true)? else {
855                            return Ok(None);
856                        };
857                        current = c;
858                        Some(p)
859                    }
860                    None => None,
861                };
862                let variant_id =
863                    self.infer.variant_resolution_for_expr(expr_id).ok_or_else(|| match path {
864                        Some(p) => MirLowerError::UnresolvedName(
865                            hir_display_with_store(&**p, self.body)
866                                .display(self.db, self.display_target())
867                                .to_string(),
868                        ),
869                        None => MirLowerError::RecordLiteralWithoutPath,
870                    })?;
871                let subst = match self.expr_ty_without_adjust(expr_id).kind() {
872                    TyKind::Adt(_, s) => s,
873                    _ => not_supported!("Non ADT record literal"),
874                };
875                let variant_fields = variant_id.fields(self.db);
876                match variant_id {
877                    VariantId::EnumVariantId(_) | VariantId::StructId(_) => {
878                        let mut operands = vec![None; variant_fields.fields().len()];
879                        for RecordLitField { name, expr } in fields.iter() {
880                            let field_id =
881                                variant_fields.field(name).ok_or(MirLowerError::UnresolvedField)?;
882                            let Some((op, c)) = self.lower_expr_to_some_operand(*expr, current)?
883                            else {
884                                return Ok(None);
885                            };
886                            current = c;
887                            operands[u32::from(field_id.into_raw()) as usize] = Some(op);
888                        }
889                        let rvalue = Rvalue::Aggregate(
890                            AggregateKind::Adt(variant_id, subst),
891                            match spread_place {
892                                Some(sp) => operands
893                                    .into_iter()
894                                    .enumerate()
895                                    .map(|(i, it)| match it {
896                                        Some(it) => it,
897                                        None => {
898                                            let p = sp.project(
899                                                ProjectionElem::Field(Either::Left(FieldId {
900                                                    parent: variant_id,
901                                                    local_id: LocalFieldId::from_raw(RawIdx::from(
902                                                        i as u32,
903                                                    )),
904                                                })),
905                                                &mut self.result.projection_store,
906                                            );
907                                            Operand { kind: OperandKind::Copy(p), span: None }
908                                        }
909                                    })
910                                    .collect(),
911                                None => operands.into_iter().collect::<Option<_>>().ok_or(
912                                    MirLowerError::TypeError("missing field in record literal"),
913                                )?,
914                            },
915                        );
916                        self.push_assignment(current, place, rvalue, expr_id.into());
917                        Ok(Some(current))
918                    }
919                    VariantId::UnionId(union_id) => {
920                        let [RecordLitField { name, expr }] = fields.as_ref() else {
921                            not_supported!("Union record literal with more than one field");
922                        };
923                        let local_id =
924                            variant_fields.field(name).ok_or(MirLowerError::UnresolvedField)?;
925                        let place = place.project(
926                            PlaceElem::Field(Either::Left(FieldId {
927                                parent: union_id.into(),
928                                local_id,
929                            })),
930                            &mut self.result.projection_store,
931                        );
932                        self.lower_expr_to_place(*expr, place, current)
933                    }
934                }
935            }
936            Expr::Await { .. } => not_supported!("await"),
937            Expr::Yeet { .. } => not_supported!("yeet"),
938            Expr::Async { .. } => not_supported!("async block"),
939            &Expr::Const(_) => {
940                // let subst = self.placeholder_subst();
941                // self.lower_const(
942                //     id.into(),
943                //     current,
944                //     place,
945                //     subst,
946                //     expr_id.into(),
947                //     self.expr_ty_without_adjust(expr_id),
948                // )?;
949                // Ok(Some(current))
950                not_supported!("const block")
951            }
952            Expr::Cast { expr, type_ref: _ } => {
953                let Some((it, current)) = self.lower_expr_to_some_operand(*expr, current)? else {
954                    return Ok(None);
955                };
956                // Since we don't have THIR, this is the "zipped" version of [rustc's HIR lowering](https://github.com/rust-lang/rust/blob/e71f9529121ca8f687e4b725e3c9adc3f1ebab4d/compiler/rustc_mir_build/src/thir/cx/expr.rs#L165-L178)
957                // and [THIR lowering as RValue](https://github.com/rust-lang/rust/blob/a4601859ae3875732797873612d424976d9e3dd0/compiler/rustc_mir_build/src/build/expr/as_rvalue.rs#L193-L313)
958                let rvalue = if self.infer.coercion_casts.contains(expr) {
959                    Rvalue::Use(it)
960                } else {
961                    let source_ty = self.infer[*expr];
962                    let target_ty = self.infer[expr_id];
963                    let cast_kind = if source_ty.as_reference().is_some() {
964                        CastKind::PointerCoercion(PointerCast::ArrayToPointer)
965                    } else {
966                        cast_kind(self.db, source_ty, target_ty)?
967                    };
968
969                    Rvalue::Cast(cast_kind, it, target_ty)
970                };
971                self.push_assignment(current, place, rvalue, expr_id.into());
972                Ok(Some(current))
973            }
974            Expr::Ref { expr, rawness: _, mutability } => {
975                let Some((p, current)) = self.lower_expr_as_place(current, *expr, true)? else {
976                    return Ok(None);
977                };
978                let bk = BorrowKind::from_hir(*mutability);
979                self.push_assignment(current, place, Rvalue::Ref(bk, p), expr_id.into());
980                Ok(Some(current))
981            }
982            Expr::Box { expr } => {
983                let ty = self.expr_ty_after_adjustments(*expr);
984                self.push_assignment(
985                    current,
986                    place,
987                    Rvalue::ShallowInitBoxWithAlloc(ty),
988                    expr_id.into(),
989                );
990                let Some((operand, current)) = self.lower_expr_to_some_operand(*expr, current)?
991                else {
992                    return Ok(None);
993                };
994                let p = place.project(ProjectionElem::Deref, &mut self.result.projection_store);
995                self.push_assignment(current, p, operand.into(), expr_id.into());
996                Ok(Some(current))
997            }
998            Expr::Field { .. }
999            | Expr::Index { .. }
1000            | Expr::UnaryOp { op: hir_def::hir::UnaryOp::Deref, .. } => {
1001                let Some((p, current)) =
1002                    self.lower_expr_as_place_without_adjust(current, expr_id, true)?
1003                else {
1004                    return Ok(None);
1005                };
1006                self.push_assignment(
1007                    current,
1008                    place,
1009                    Operand { kind: OperandKind::Copy(p), span: None }.into(),
1010                    expr_id.into(),
1011                );
1012                Ok(Some(current))
1013            }
1014            Expr::UnaryOp {
1015                expr,
1016                op: op @ (hir_def::hir::UnaryOp::Not | hir_def::hir::UnaryOp::Neg),
1017            } => {
1018                let Some((operand, current)) = self.lower_expr_to_some_operand(*expr, current)?
1019                else {
1020                    return Ok(None);
1021                };
1022                let operation = match op {
1023                    hir_def::hir::UnaryOp::Not => UnOp::Not,
1024                    hir_def::hir::UnaryOp::Neg => UnOp::Neg,
1025                    _ => unreachable!(),
1026                };
1027                self.push_assignment(
1028                    current,
1029                    place,
1030                    Rvalue::UnaryOp(operation, operand),
1031                    expr_id.into(),
1032                );
1033                Ok(Some(current))
1034            }
1035            Expr::BinaryOp { lhs, rhs, op } => {
1036                let op: BinaryOp = op.ok_or(MirLowerError::IncompleteExpr)?;
1037                let is_builtin = 'b: {
1038                    // Without adjust here is a hack. We assume that we know every possible adjustment
1039                    // for binary operator, and use without adjust to simplify our conditions.
1040                    let lhs_ty = self.expr_ty_without_adjust(*lhs);
1041                    let rhs_ty = self.expr_ty_without_adjust(*rhs);
1042                    if matches!(op, BinaryOp::CmpOp(syntax::ast::CmpOp::Eq { .. }))
1043                        && matches!(lhs_ty.kind(), TyKind::RawPtr(..))
1044                        && matches!(rhs_ty.kind(), TyKind::RawPtr(..))
1045                    {
1046                        break 'b true;
1047                    }
1048                    let builtin_inequal_impls = matches!(
1049                        op,
1050                        BinaryOp::ArithOp(ArithOp::Shl | ArithOp::Shr)
1051                            | BinaryOp::Assignment { op: Some(ArithOp::Shl | ArithOp::Shr) }
1052                    );
1053                    matches!(
1054                        lhs_ty.kind(),
1055                        TyKind::Bool
1056                            | TyKind::Char
1057                            | TyKind::Int(_)
1058                            | TyKind::Uint(_)
1059                            | TyKind::Float(_)
1060                    ) && matches!(
1061                        rhs_ty.kind(),
1062                        TyKind::Bool
1063                            | TyKind::Char
1064                            | TyKind::Int(_)
1065                            | TyKind::Uint(_)
1066                            | TyKind::Float(_)
1067                    ) && (lhs_ty == rhs_ty || builtin_inequal_impls)
1068                };
1069                if !is_builtin
1070                    && let Some((func_id, generic_args)) = self.infer.method_resolution(expr_id)
1071                {
1072                    let func = Operand::from_fn(self.db, func_id, generic_args);
1073                    return self.lower_call_and_args(
1074                        func,
1075                        [*lhs, *rhs].into_iter(),
1076                        place,
1077                        current,
1078                        self.is_uninhabited(expr_id),
1079                        expr_id.into(),
1080                    );
1081                }
1082                if let hir_def::hir::BinaryOp::Assignment { op: Some(op) } = op {
1083                    // last adjustment is `&mut` which we don't want it.
1084                    let adjusts = self
1085                        .infer
1086                        .expr_adjustments
1087                        .get(lhs)
1088                        .and_then(|it| it.split_last())
1089                        .map(|it| it.1)
1090                        .ok_or(MirLowerError::TypeError("adjustment of binary op was missing"))?;
1091                    let Some((lhs_place, current)) =
1092                        self.lower_expr_as_place_with_adjust(current, *lhs, false, adjusts)?
1093                    else {
1094                        return Ok(None);
1095                    };
1096                    let Some((rhs_op, current)) = self.lower_expr_to_some_operand(*rhs, current)?
1097                    else {
1098                        return Ok(None);
1099                    };
1100                    let r_value = Rvalue::CheckedBinaryOp(
1101                        op.into(),
1102                        Operand { kind: OperandKind::Copy(lhs_place), span: None },
1103                        rhs_op,
1104                    );
1105                    self.push_assignment(current, lhs_place, r_value, expr_id.into());
1106                    return Ok(Some(current));
1107                }
1108                let Some((lhs_op, current)) = self.lower_expr_to_some_operand(*lhs, current)?
1109                else {
1110                    return Ok(None);
1111                };
1112                if let hir_def::hir::BinaryOp::LogicOp(op) = op {
1113                    let value_to_short = match op {
1114                        syntax::ast::LogicOp::And => 0,
1115                        syntax::ast::LogicOp::Or => 1,
1116                    };
1117                    let start_of_then = self.new_basic_block();
1118                    self.push_assignment(
1119                        start_of_then,
1120                        place,
1121                        lhs_op.clone().into(),
1122                        expr_id.into(),
1123                    );
1124                    let end_of_then = Some(start_of_then);
1125                    let start_of_else = self.new_basic_block();
1126                    let end_of_else = self.lower_expr_to_place(*rhs, place, start_of_else)?;
1127                    self.set_terminator(
1128                        current,
1129                        TerminatorKind::SwitchInt {
1130                            discr: lhs_op,
1131                            targets: SwitchTargets::static_if(
1132                                value_to_short,
1133                                start_of_then,
1134                                start_of_else,
1135                            ),
1136                        },
1137                        expr_id.into(),
1138                    );
1139                    return Ok(self.merge_blocks(end_of_then, end_of_else, expr_id.into()));
1140                }
1141                let Some((rhs_op, current)) = self.lower_expr_to_some_operand(*rhs, current)?
1142                else {
1143                    return Ok(None);
1144                };
1145                self.push_assignment(
1146                    current,
1147                    place,
1148                    Rvalue::CheckedBinaryOp(
1149                        match op {
1150                            hir_def::hir::BinaryOp::LogicOp(op) => match op {
1151                                hir_def::hir::LogicOp::And => BinOp::BitAnd, // FIXME: make these short circuit
1152                                hir_def::hir::LogicOp::Or => BinOp::BitOr,
1153                            },
1154                            hir_def::hir::BinaryOp::ArithOp(op) => BinOp::from(op),
1155                            hir_def::hir::BinaryOp::CmpOp(op) => BinOp::from(op),
1156                            hir_def::hir::BinaryOp::Assignment { .. } => unreachable!(), // handled above
1157                        },
1158                        lhs_op,
1159                        rhs_op,
1160                    ),
1161                    expr_id.into(),
1162                );
1163                Ok(Some(current))
1164            }
1165            &Expr::Assignment { target, value } => {
1166                let Some((value, mut current)) = self.lower_expr_as_place(current, value, true)?
1167                else {
1168                    return Ok(None);
1169                };
1170                self.push_fake_read(current, value, expr_id.into());
1171                let resolver_guard =
1172                    self.resolver.update_to_inner_scope(self.db, self.owner, expr_id);
1173                current = self.pattern_match_assignment(current, value, target)?;
1174                self.resolver.reset_to_guard(resolver_guard);

1176            }
1177            &Expr::Range { lhs, rhs, range_type: _ } => {

1180                    return Err(MirLowerError::TypeError("Range type is not adt"));
1181                };

1183                    return Err(MirLowerError::TypeError("Range type is not struct"));
1184                };
1185                let mut lp = None;
1186                let mut rp = None;
1187                if let Some(it) = lhs {
1188                    let Some((o, c)) = self.lower_expr_to_some_operand(it, current)? else {
1189                        return Ok(None);
1190                    };
1191                    lp = Some(o);
1192                    current = c;
1193                }
1194                if let Some(it) = rhs {
1195                    let Some((o, c)) = self.lower_expr_to_some_operand(it, current)? else {
1196                        return Ok(None);
1197                    };
1198                    rp = Some(o);
1199                    current = c;
1200                }
1201                self.push_assignment(
1202                    current,
1203                    place,
1204                    Rvalue::Aggregate(
1205                        AggregateKind::Adt(st.into(), subst),
1206                        st.fields(self.db)
1207                            .fields()
1208                            .iter()
1209                            .map(|it| {
1210                                let o = match it.1.name.as_str() {
1211                                    "start" => lp.take(),
1212                                    "end" => rp.take(),
1213                                    "exhausted" => Some(Operand::from_bytes(
1214                                        Box::new([0]),
1215                                        Ty::new_bool(self.interner()),
1216                                    )),
1217                                    _ => None,
1218                                };
1219                                o.ok_or(MirLowerError::UnresolvedField)
1220                            })
1221                            .collect::<Result<'_, _>>()?,
1222                    ),
1223                    expr_id.into(),
1224                );
1225                Ok(Some(current))
1226            }
1227            Expr::Closure { .. } => {
1228                let ty = self.expr_ty_without_adjust(expr_id);
1229                let TyKind::Closure(id, _) = ty.kind() else {
1230                    not_supported!("closure with non closure type");
1231                };
1232                self.result.closures.push(id.0);
1233                let (captures, _) = self.infer.closure_info(id.0);
1234                let mut operands = vec![];
1235                for capture in captures.iter() {
1236                    let p = Place {
1237                        local: self.binding_local(capture.place.local)?,
1238                        projection: self.result.projection_store.intern(
1239                            capture
1240                                .place
1241                                .projections
1242                                .clone()
1243                                .into_iter()
1244                                .map(|it| match it {
1245                                    ProjectionElem::Deref => ProjectionElem::Deref,
1246                                    ProjectionElem::Field(it) => ProjectionElem::Field(it),
1247                                    ProjectionElem::ClosureField(it) => {
1248                                        ProjectionElem::ClosureField(it)
1249                                    }
1250                                    ProjectionElem::ConstantIndex { offset, from_end } => {
1251                                        ProjectionElem::ConstantIndex { offset, from_end }
1252                                    }
1253                                    ProjectionElem::Subslice { from, to } => {
1254                                        ProjectionElem::Subslice { from, to }
1255                                    }
1256                                    ProjectionElem::OpaqueCast(it) => {
1257                                        ProjectionElem::OpaqueCast(it)
1258                                    }
1259                                    #[allow(unreachable_patterns)]
1260                                    ProjectionElem::Index(it) => match it {},
1261                                })
1262                                .collect(),
1263                        ),
1264                    };
1265                    match &capture.kind {
1266                        CaptureKind::ByRef(bk) => {
1267                            let tmp_ty = capture.ty.instantiate_identity();
1268                            // FIXME: Handle more than one span.
1269                            let capture_spans = capture.spans();
1270                            let tmp: Place<'db> =
1271                                self.temp(tmp_ty, current, capture_spans[0])?.into();
1272                            self.push_assignment(
1273                                current,
1274                                tmp,
1275                                Rvalue::Ref(*bk, p),
1276                                capture_spans[0],
1277                            );
1278                            operands.push(Operand { kind: OperandKind::Move(tmp), span: None });
1279                        }
1280                        CaptureKind::ByValue => {
1281                            operands.push(Operand { kind: OperandKind::Move(p), span: None })
1282                        }
1283                    }
1284                }
1285                self.push_assignment(
1286                    current,
1287                    place,
1288                    Rvalue::Aggregate(AggregateKind::Closure(ty), operands.into()),
1289                    expr_id.into(),
1290                );
1291                Ok(Some(current))
1292            }
1293            Expr::Tuple { exprs } => {
1294                let Some(values) = exprs
1295                    .iter()
1296                    .map(|it| {
1297                        let Some((o, c)) = self.lower_expr_to_some_operand(*it, current)? else {
1298                            return Ok(None);
1299                        };
1300                        current = c;
1301                        Ok(Some(o))
1302                    })
1303                    .collect::<Result<'_, Option<_>>>()?
1304                else {
1305                    return Ok(None);
1306                };
1307                let r = Rvalue::Aggregate(
1308                    AggregateKind::Tuple(self.expr_ty_without_adjust(expr_id)),
1309                    values,
1310                );
1311                self.push_assignment(current, place, r, expr_id.into());
1312                Ok(Some(current))
1313            }
1314            Expr::Array(l) => match l {
1315                Array::ElementList { elements, .. } => {
1316                    let elem_ty = match self.expr_ty_without_adjust(expr_id).kind() {
1317                        TyKind::Array(ty, _) => ty,
1318                        _ => {
1319                            return Err(MirLowerError::TypeError(
1320                                "Array expression with non array type",
1321                            ));
1322                        }
1323                    };
1324                    let Some(values) = elements
1325                        .iter()
1326                        .map(|it| {
1327                            let Some((o, c)) = self.lower_expr_to_some_operand(*it, current)?
1328                            else {
1329                                return Ok(None);
1330                            };
1331                            current = c;
1332                            Ok(Some(o))
1333                        })
1334                        .collect::<Result<'_, Option<_>>>()?
1335                    else {
1336                        return Ok(None);
1337                    };
1338                    let r = Rvalue::Aggregate(AggregateKind::Array(elem_ty), values);
1339                    self.push_assignment(current, place, r, expr_id.into());
1340                    Ok(Some(current))
1341                }
1342                Array::Repeat { initializer, .. } => {
1343                    let Some((init, current)) =
1344                        self.lower_expr_to_some_operand(*initializer, current)?
1345                    else {
1346                        return Ok(None);
1347                    };
1348                    let len = match self.expr_ty_without_adjust(expr_id).kind() {
1349                        TyKind::Array(_, len) => len,
1350                        _ => {
1351                            return Err(MirLowerError::TypeError(
1352                                "Array repeat expression with non array type",
1353                            ));
1354                        }
1355                    };
1356                    let r = Rvalue::Repeat(init, len);
1357                    self.push_assignment(current, place, r, expr_id.into());
1358                    Ok(Some(current))
1359                }
1360            },
1361            Expr::Literal(l) => {
1362                let ty = self.expr_ty_without_adjust(expr_id);
1363                let op = self.lower_literal_to_operand(ty, l)?;
1364                self.push_assignment(current, place, op.into(), expr_id.into());
1365                Ok(Some(current))
1366            }
1367            Expr::Underscore => Ok(Some(current)),
1368        }
1369    }

1401        &mut self,
1402        ty: Ty<'db>,
1403        loc: &ExprId,
1404    ) -> Result<'db, Operand<'db>> {
1405        match &self.body[*loc] {
1406            Expr::Literal(l) => self.lower_literal_to_operand(ty, l),

1409                let db = self.db;
1410                let unresolved_name = || {

1444    fn lower_literal_to_operand(&mut self, ty: Ty<'db>, l: &Literal) -> Result<'db, Operand<'db>> {
1445        let size = || self.db.layout_of_ty(ty, self.env.clone()).map(|it| it.size.bytes_usize());
1446        const USIZE_SIZE: usize = size_of::<usize>();
1447        let bytes: Box<[_]> = match l {
1448            hir_def::hir::Literal::String(b) => {
1449                let b = b.as_str();
1450                let mut data = [0; { 2 * USIZE_SIZE }];
1451                data[..USIZE_SIZE].copy_from_slice(&0usize.to_le_bytes());
1452                data[USIZE_SIZE..].copy_from_slice(&b.len().to_le_bytes());
1453                let mm = MemoryMap::simple(b.as_bytes().into());
1454                return Ok(Operand::from_concrete_const(Box::new(data), mm, ty));
1455            }
1456            hir_def::hir::Literal::CString(b) => {
1457                let bytes = b.iter().copied().chain(iter::once(0)).collect::<Box<_>>();
1458
1459                let mut data = [0; { 2 * USIZE_SIZE }];
1460                data[..USIZE_SIZE].copy_from_slice(&0usize.to_le_bytes());
1461                data[USIZE_SIZE..].copy_from_slice(&bytes.len().to_le_bytes());
1462                let mm = MemoryMap::simple(bytes);
1463                return Ok(Operand::from_concrete_const(Box::new(data), mm, ty));
1464            }
1465            hir_def::hir::Literal::ByteString(b) => {
1466                let mut data = [0; { 2 * USIZE_SIZE }];
1467                data[..USIZE_SIZE].copy_from_slice(&0usize.to_le_bytes());
1468                data[USIZE_SIZE..].copy_from_slice(&b.len().to_le_bytes());
1469                let mm = MemoryMap::simple(b.clone());
1470                return Ok(Operand::from_concrete_const(Box::new(data), mm, ty));
1471            }
1472            hir_def::hir::Literal::Char(c) => Box::new(u32::from(*c).to_le_bytes()),
1473            hir_def::hir::Literal::Bool(b) => Box::new([*b as u8]),
1474            hir_def::hir::Literal::Int(it, _) => Box::from(&it.to_le_bytes()[0..size()?]),
1475            hir_def::hir::Literal::Uint(it, _) => Box::from(&it.to_le_bytes()[0..size()?]),
1476            hir_def::hir::Literal::Float(f, _) => match size()? {
1477                16 => Box::new(f.to_f128().to_bits().to_le_bytes()),
1478                8 => Box::new(f.to_f64().to_le_bytes()),
1479                4 => Box::new(f.to_f32().to_le_bytes()),
1480                2 => Box::new(u16::try_from(f.to_f16().to_bits()).unwrap().to_le_bytes()),
1481                _ => {
1482                    return Err(MirLowerError::TypeError(
1483                        "float with size other than 2, 4, 8 or 16 bytes",
1484                    ));
1485                }
1486            },
1487        };
1488        Ok(Operand::from_concrete_const(bytes, MemoryMap::default(), ty))
1489    }

1685    fn lower_loop(
1686        &mut self,
1687        prev_block: BasicBlockId<'db>,
1688        place: Place<'db>,
1689        label: Option<LabelId>,
1690        span: MirSpan,
1691        f: impl FnOnce(&mut MirLowerCtx<'_, 'db>, BasicBlockId<'db>) -> Result<'db, ()>,
1692    ) -> Result<'db, Option<BasicBlockId<'db>>> {
1693        let begin = self.new_basic_block();
1694        let prev = self.current_loop_blocks.replace(LoopBlocks {
1695            begin,
1696            end: None,
1697            place,
1698            drop_scope_index: self.drop_scopes.len(),
1699        });
1700        let prev_label = if let Some(label) = label {
1701            // We should generate the end now, to make sure that it wouldn't change later. It is
1702            // bad as we may emit end (unnecessary unreachable block) for unterminating loop, but
1703            // it should not affect correctness.
1704            self.current_loop_end()?;
1705            self.labeled_loop_blocks
1706                .insert(label, self.current_loop_blocks.as_ref().unwrap().clone())
1707        } else {
1708            None
1709        };
1710        self.set_goto(prev_block, begin, span);
1711        f(self, begin)?;
1712        let my = mem::replace(&mut self.current_loop_blocks, prev).ok_or(
1713            MirLowerError::ImplementationError("current_loop_blocks is corrupt".to_owned()),
1714        )?;
1715        if let Some(prev) = prev_label {
1716            self.labeled_loop_blocks.insert(label.unwrap(), prev);
1717        }
1718        Ok(my.end)
1719    }

1799    fn lower_block_to_place(
1800        &mut self,
1801        statements: &[hir_def::hir::Statement],
1802        mut current: BasicBlockId<'db>,
1803        tail: Option<ExprId>,
1804        place: Place<'db>,
1805        span: MirSpan,
1806    ) -> Result<'db, Option<Idx<BasicBlock<'db>>>> {
1807        let scope = self.push_drop_scope();
1808        for statement in statements.iter() {
1809            match statement {
1810                hir_def::hir::Statement::Let { pat, initializer, else_branch, type_ref: _ } => {
1811                    if let Some(expr_id) = initializer {
1812                        let else_block;
1813                        let Some((init_place, c)) =
1814                            self.lower_expr_as_place(current, *expr_id, true)?
1815                        else {
1816                            scope.pop_assume_dropped(self);
1817                            return Ok(None);
1818                        };
1819                        current = c;
1820                        self.push_fake_read(current, init_place, span);
1821                        // Using the initializer for the resolver scope is good enough for us, as it cannot create new declarations
1822                        // and has all declarations of the `let`.
1823                        let resolver_guard =
1824                            self.resolver.update_to_inner_scope(self.db, self.owner, *expr_id);
1825                        (current, else_block) =
1826                            self.pattern_match(current, None, init_place, *pat)?;
1827                        self.resolver.reset_to_guard(resolver_guard);
1828                        match (else_block, else_branch) {
1829                            (None, _) => (),
1830                            (Some(else_block), None) => {
1831                                self.set_terminator(else_block, TerminatorKind::Unreachable, span);
1832                            }
1833                            (Some(else_block), Some(else_branch)) => {
1834                                if let Some((_, b)) =
1835                                    self.lower_expr_as_place(else_block, *else_branch, true)?
1836                                {
1837                                    self.set_terminator(b, TerminatorKind::Unreachable, span);
1838                                }
1839                            }
1840                        }
1841                    } else {
1842                        let mut err = None;
1843                        self.body.walk_bindings_in_pat(*pat, |b| {
1844                            if let Err(e) = self.push_storage_live(b, current) {
1845                                err = Some(e);
1846                            }
1847                        });
1848                        if let Some(e) = err {
1849                            return Err(e);
1850                        }
1851                    }
1852                }
1853                &hir_def::hir::Statement::Expr { expr, has_semi: _ } => {
1854                    let scope2 = self.push_drop_scope();
1855                    let Some((p, c)) = self.lower_expr_as_place(current, expr, true)? else {
1856                        scope2.pop_assume_dropped(self);
1857                        scope.pop_assume_dropped(self);
1858                        return Ok(None);
1859                    };
1860                    self.push_fake_read(c, p, expr.into());
1861                    current = scope2.pop_and_drop(self, c, expr.into());
1862                }
1863                hir_def::hir::Statement::Item(_) => (),
1864            }
1865        }
1866        if let Some(tail) = tail {
1867            let Some(c) = self.lower_expr_to_place(tail, place, current)? else {
1868                scope.pop_assume_dropped(self);
1869                return Ok(None);
1870            };
1871            current = c;
1872        }
1873        current = scope.pop_and_drop(self, current, span);
1874        Ok(Some(current))
1875    }

1877    fn lower_params_and_bindings(
1878        &mut self,
1879        params: impl Iterator<Item = (PatId, Ty<'db>)> + Clone,
1880        self_binding: Option<(BindingId, Ty<'db>)>,
1881        pick_binding: impl Fn(BindingId) -> bool,
1882    ) -> Result<'db, BasicBlockId<'db>> {
1883        let base_param_count = self.result.param_locals.len();
1884        let self_binding = match self_binding {
1885            Some((self_binding, ty)) => {
1886                let local_id = self.result.locals.alloc(Local { ty });
1887                self.drop_scopes.last_mut().unwrap().locals.push(local_id);
1888                self.result.binding_locals.insert(self_binding, local_id);
1889                self.result.param_locals.push(local_id);
1890                Some(self_binding)
1891            }
1892            None => None,
1893        };
1894        self.result.param_locals.extend(params.clone().map(|(it, ty)| {
1895            let local_id = self.result.locals.alloc(Local { ty });
1896            self.drop_scopes.last_mut().unwrap().locals.push(local_id);
1897            if let Pat::Bind { id, subpat: None } = self.body[it]
1898                && matches!(
1899                    self.body[id].mode,
1900                    BindingAnnotation::Unannotated | BindingAnnotation::Mutable
1901                )
1902            {
1903                self.result.binding_locals.insert(id, local_id);
1904            }
1905            local_id
1906        }));
1907        // and then rest of bindings
1908        for (id, _) in self.body.bindings() {
1909            if !pick_binding(id) {
1910                continue;
1911            }
1912            if !self.result.binding_locals.contains_idx(id) {
1913                self.result
1914                    .binding_locals
1915                    .insert(id, self.result.locals.alloc(Local { ty: self.infer[id] }));
1916            }
1917        }
1918        let mut current = self.result.start_block;
1919        if let Some(self_binding) = self_binding {
1920            let local = self.result.param_locals.clone()[base_param_count];
1921            if local != self.binding_local(self_binding)? {
1922                let r = self.match_self_param(self_binding, current, local)?;
1923                if let Some(b) = r.1 {
1924                    self.set_terminator(b, TerminatorKind::Unreachable, MirSpan::SelfParam);
1925                }
1926                current = r.0;
1927            }
1928        }
1929        let local_params = self
1930            .result
1931            .param_locals
1932            .clone()
1933            .into_iter()
1934            .skip(base_param_count + self_binding.is_some() as usize);
1935        for ((param, _), local) in params.zip(local_params) {
1936            if let Pat::Bind { id, .. } = self.body[param]
1937                && local == self.binding_local(id)?
1938            {
1939                continue;
1940            }
1941            let r = self.pattern_match(current, None, local.into(), param)?;
1942            if let Some(b) = r.1 {
1943                self.set_terminator(b, TerminatorKind::Unreachable, param.into());
1944            }
1945            current = r.0;
1946        }
1947        Ok(current)
1948    }

2061fn cast_kind<'db>(
2062    db: &'db dyn HirDatabase,
2063    source_ty: Ty<'db>,
2064    target_ty: Ty<'db>,
2065) -> Result<'db, CastKind> {

2095    };

2104                ctx.interner(),
2105                Region::error(ctx.interner()),
2106                infer[expr],
2107                Mutability::Mut,

2111                Region::error(ctx.interner()),
2112                infer[expr],
2113                Mutability::Not,

2119        substs.split_closure_args_untupled().closure_sig_as_fn_ptr_ty.callable_sig(ctx.interner())

2122    };

2125        args.iter().zip(sig.skip_binder().inputs().iter()).map(|(it, y)| (*it, y)),

2132        ctx.set_terminator(current, TerminatorKind::Return, (*root).into());
2133    }
2134    let mut upvar_map: FxHashMap<LocalId<'db>, Vec<(&CapturedItem<'_>, usize)>> =
2135        FxHashMap::default();
2136    for (i, capture) in captures.iter().enumerate() {
2137        let local = ctx.binding_local(capture.place.local)?;
2138        upvar_map.entry(local).or_default().push((capture, i));
2139    }

2141    let closure_local = ctx.result.locals.iter().nth(1).unwrap().0;

2143        FnTrait::FnOnce | FnTrait::AsyncFnOnce => vec![],

2145            vec![ProjectionElem::Deref]
2146        }

2157                        (ProjectionElem::Deref, ProjectionElem::Deref) => (),

2159                        (ProjectionElem::ClosureField(it), ProjectionElem::ClosureField(y))
2160                            if it == y => {}
2161                        _ => return false,
2162                    }
2163                }
2164                true
2165            });
2166            match r {
2167                Some(it) => {
2168                    p.local = closure_local;
2169                    let mut next_projs = closure_projection.clone();
2170                    next_projs.push(PlaceElem::ClosureField(it.1));
2171                    let prev_projs = p.projection;
2172                    if it.0.kind != CaptureKind::ByValue {
2173                        next_projs.push(ProjectionElem::Deref);
2174                    }
2175                    next_projs.extend(
2176                        prev_projs.lookup(store).iter().skip(it.0.place.projections.len()).cloned(),
2177                    );
2178                    p.projection = store.intern(next_projs.into());
2179                }

2195}

2240pub fn lower_to_mir<'db>(
2241    db: &'db dyn HirDatabase,
2242    owner: DefWithBodyId,
2243    body: &Body,
2244    infer: &InferenceResult<'db>,
2245    // FIXME: root_expr should always be the body.body_expr, but since `X` in `[(); X]` doesn't have its own specific body yet, we
2246    // need to take this input explicitly.
2247    root_expr: ExprId,
2248) -> Result<'db, MirBody<'db>> {
2249    if infer.type_mismatches().next().is_some() || infer.is_erroneous() {
2250        return Err(MirLowerError::HasErrors);
2251    }
2252    let mut ctx = MirLowerCtx::new(db, owner, body, infer);
2253    // 0 is return local
2254    ctx.result.locals.alloc(Local { ty: ctx.expr_ty_after_adjustments(root_expr) });
2255    let binding_picker = |b: BindingId| {
2256        let owner = ctx.body.binding_owner(b);
2257        if root_expr == body.body_expr { owner.is_none() } else { owner == Some(root_expr) }
2258    };
2259    // 1 to param_len is for params
2260    // FIXME: replace with let chain once it becomes stable
2261    let current = 'b: {
2262        if body.body_expr == root_expr {
2263            // otherwise it's an inline const, and has no parameter

2265                let callable_sig =
2266                    db.callable_item_signature(fid.into()).instantiate_identity().skip_binder();
2267                let mut params = callable_sig.inputs().iter();
2268                let self_param = body.self_param.and_then(|id| Some((id, params.next()?)));
2269                break 'b ctx.lower_params_and_bindings(
2270                    body.params.iter().zip(params).map(|(it, y)| (*it, y)),
2271                    self_param,
2272                    binding_picker,
2273                )?;
2274            }
2275        }
2276        ctx.lower_params_and_bindings([].into_iter(), None, binding_picker)?
2277    };
2278    if let Some(current) = ctx.lower_expr_to_place(root_expr, return_slot().into(), current)? {
2279        let current = ctx.pop_drop_scope_assert_finished(current, root_expr.into())?;

2281    }
