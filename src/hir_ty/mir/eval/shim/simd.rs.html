9impl<'db> Evaluator<'db> {
10    fn detect_simd_ty(&self, ty: Ty<'db>) -> Result<'db, (usize, Ty<'db>)> {
11        match ty.kind() {
12            TyKind::Adt(adt_def, subst) => {
13                let len = match subst.as_slice().get(1).and_then(|it| it.konst()) {
14                    Some(len) => len,
15                    _ => {
16                        if let AdtId::StructId(id) = adt_def.def_id().0 {
17                            let struct_data = id.fields(self.db);
18                            let fields = struct_data.fields();
19                            let Some((first_field, _)) = fields.iter().next() else {
20                                not_supported!("simd type with no field");
21                            };
22                            let field_ty = self.db.field_types(id.into())[first_field]
23                                .get()
24                                .instantiate(self.interner(), subst);
25                            return Ok((fields.len(), field_ty));
26                        }
27                        return Err(MirEvalError::InternalError(
28                            "simd type with no len param".into(),
29                        ));
30                    }
31                };
32                match try_const_usize(self.db, len) {
33                    Some(len) => {
34                        let Some(ty) = subst.as_slice().first().and_then(|it| it.ty()) else {
35                            return Err(MirEvalError::InternalError(
36                                "simd type with no ty param".into(),
37                            ));
38                        };
39                        Ok((len as usize, ty))
40                    }
41                    None => Err(MirEvalError::InternalError(
42                        "simd type with unevaluatable len param".into(),
43                    )),
44                }
45            }
46            _ => Err(MirEvalError::InternalError("simd type which is not a struct".into())),
47        }
48    }

50    pub(super) fn exec_simd_intrinsic(
51        &mut self,
52        name: &str,
53        args: &[IntervalAndTy<'db>],
54        _generic_args: GenericArgs<'db>,
55        destination: Interval,
56        _locals: &Locals,
57        _span: MirSpan,
58    ) -> Result<'db, ()> {
59        match name {
60            "and" | "or" | "xor" => {
61                let [left, right] = args else {
62                    return Err(MirEvalError::InternalError(
63                        "simd bit op args are not provided".into(),
64                    ));
65                };
66                let result = left
67                    .get(self)?
68                    .iter()
69                    .zip(right.get(self)?)
70                    .map(|(&it, &y)| match name {
71                        "and" => it & y,
72                        "or" => it | y,
73                        "xor" => it ^ y,
74                        _ => unreachable!(),
75                    })
76                    .collect::<Vec<_>>();
77                destination.write_from_bytes(self, &result)
78            }
79            "eq" | "ne" | "lt" | "le" | "gt" | "ge" => {
80                let [left, right] = args else {
81                    return Err(MirEvalError::InternalError("simd args are not provided".into()));
82                };
83                let (len, ty) = self.detect_simd_ty(left.ty)?;
84                let is_signed = matches!(ty.kind(), TyKind::Int(_));
85                let size = left.interval.size / len;
86                let dest_size = destination.size / len;
87                let mut destination_bytes = vec![];
88                let vector = left.get(self)?.chunks(size).zip(right.get(self)?.chunks(size));
89                for (l, r) in vector {
90                    let mut result = Ordering::Equal;
91                    for (l, r) in l.iter().zip(r).rev() {
92                        let it = l.cmp(r);
93                        if it != Ordering::Equal {
94                            result = it;
95                            break;
96                        }
97                    }
98                    if is_signed
99                        && let Some((&l, &r)) = l.iter().zip(r).next_back()
100                        && l != r
101                    {
102                        result = (l as i8).cmp(&(r as i8));
103                    }
104                    let result = match result {
105                        Ordering::Less => ["lt", "le", "ne"].contains(&name),
106                        Ordering::Equal => ["ge", "le", "eq"].contains(&name),
107                        Ordering::Greater => ["ge", "gt", "ne"].contains(&name),
108                    };
109                    let result = if result { 255 } else { 0 };
110                    destination_bytes.extend(std::iter::repeat_n(result, dest_size));
111                }
112
113                destination.write_from_bytes(self, &destination_bytes)
114            }
115            "bitmask" => {
116                let [op] = args else {
117                    return Err(MirEvalError::InternalError(
118                        "simd_bitmask args are not provided".into(),
119                    ));
120                };
121                let (op_len, _) = self.detect_simd_ty(op.ty)?;
122                let op_count = op.interval.size / op_len;
123                let mut result: u64 = 0;
124                for (i, val) in op.get(self)?.chunks(op_count).enumerate() {
125                    if !val.iter().all(|&it| it == 0) {
126                        result |= 1 << i;
127                    }
128                }
129                destination.write_from_bytes(self, &result.to_le_bytes()[0..destination.size])
130            }
131            "shuffle" => {
132                let [left, right, index] = args else {
133                    return Err(MirEvalError::InternalError(
134                        "simd_shuffle args are not provided".into(),
135                    ));
136                };
137                let TyKind::Array(_, index_len) = index.ty.kind() else {
138                    return Err(MirEvalError::InternalError(
139                        "simd_shuffle index argument has non-array type".into(),
140                    ));
141                };
142                let index_len = match try_const_usize(self.db, index_len) {
143                    Some(it) => it as usize,
144                    None => {
145                        return Err(MirEvalError::InternalError(
146                            "simd type with unevaluatable len param".into(),
147                        ));
148                    }
149                };
150                let (left_len, _) = self.detect_simd_ty(left.ty)?;
151                let left_size = left.interval.size / left_len;
152                let vector =
153                    left.get(self)?.chunks(left_size).chain(right.get(self)?.chunks(left_size));
154                let mut result = vec![];
155                for index in index.get(self)?.chunks(index.interval.size / index_len) {
156                    let index = from_bytes!(u32, index) as usize;
157                    let val = match vector.clone().nth(index) {
158                        Some(it) => it,
159                        None => {
160                            return Err(MirEvalError::InternalError(
161                                "out of bound access in simd shuffle".into(),
162                            ));
163                        }
164                    };
165                    result.extend(val);
166                }
167                destination.write_from_bytes(self, &result)
168            }
169            _ => not_supported!("unknown simd intrinsic {name}"),
170        }
171    }
