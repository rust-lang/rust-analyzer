9macro_rules! from_bytes {
10    ($ty:tt, $value:expr) => {
11        ($ty::from_le_bytes(match ($value).try_into() {
12            Ok(it) => it,
13            Err(_) => return Err(MirEvalError::InternalError("mismatched size".into())),
14        }))
15    };
16}
17
18macro_rules! not_supported {
19    ($it: expr) => {
20        return Err(MirEvalError::NotSupported(format!($it)))
21    };
22}

24impl<'db> Evaluator<'db> {
25    fn detect_simd_ty(&self, ty: Ty<'db>) -> Result<'db, (usize, Ty<'db>)> {
26        match ty.kind() {
27            TyKind::Adt(adt_def, subst) => {
28                let len = match subst.as_slice().get(1).and_then(|it| it.konst()) {
29                    Some(len) => len,
30                    _ => {
31                        if let AdtId::StructId(id) = adt_def.def_id().0 {
32                            let struct_data = id.fields(self.db);
33                            let fields = struct_data.fields();
34                            let Some((first_field, _)) = fields.iter().next() else {
35                                not_supported!("simd type with no field");
36                            };
37                            let field_ty = self.db.field_types(id.into())[first_field]
38                                .instantiate(self.interner(), subst);
39                            return Ok((fields.len(), field_ty));
40                        }
41                        return Err(MirEvalError::InternalError(
42                            "simd type with no len param".into(),
43                        ));
44                    }
45                };
46                match try_const_usize(self.db, len) {
47                    Some(len) => {
48                        let Some(ty) = subst.as_slice().first().and_then(|it| it.ty()) else {
49                            return Err(MirEvalError::InternalError(
50                                "simd type with no ty param".into(),
51                            ));
52                        };
53                        Ok((len as usize, ty))
54                    }
55                    None => Err(MirEvalError::InternalError(
56                        "simd type with unevaluatable len param".into(),
57                    )),
58                }
59            }
60            _ => Err(MirEvalError::InternalError("simd type which is not a struct".into())),
61        }
62    }
63
64    pub(super) fn exec_simd_intrinsic(
65        &mut self,
66        name: &str,
67        args: &[IntervalAndTy<'db>],
68        _generic_args: GenericArgs<'db>,
69        destination: Interval,
70        _locals: &Locals<'db>,
71        _span: MirSpan,
72    ) -> Result<'db, ()> {
73        match name {
74            "and" | "or" | "xor" => {
75                let [left, right] = args else {
76                    return Err(MirEvalError::InternalError(
77                        "simd bit op args are not provided".into(),
78                    ));
79                };
80                let result = left
81                    .get(self)?
82                    .iter()
83                    .zip(right.get(self)?)
84                    .map(|(&it, &y)| match name {
85                        "and" => it & y,
86                        "or" => it | y,
87                        "xor" => it ^ y,
88                        _ => unreachable!(),
89                    })
90                    .collect::<Vec<_>>();
91                destination.write_from_bytes(self, &result)
92            }
93            "eq" | "ne" | "lt" | "le" | "gt" | "ge" => {
94                let [left, right] = args else {
95                    return Err(MirEvalError::InternalError("simd args are not provided".into()));
96                };
97                let (len, ty) = self.detect_simd_ty(left.ty)?;
98                let is_signed = matches!(ty.kind(), TyKind::Int(_));
99                let size = left.interval.size / len;
100                let dest_size = destination.size / len;
101                let mut destination_bytes = vec![];
102                let vector = left.get(self)?.chunks(size).zip(right.get(self)?.chunks(size));
103                for (l, r) in vector {
104                    let mut result = Ordering::Equal;
105                    for (l, r) in l.iter().zip(r).rev() {
106                        let it = l.cmp(r);
107                        if it != Ordering::Equal {
108                            result = it;
109                            break;
110                        }
111                    }
112                    if is_signed
113                        && let Some((&l, &r)) = l.iter().zip(r).next_back()
114                        && l != r
115                    {
116                        result = (l as i8).cmp(&(r as i8));
117                    }
118                    let result = match result {
119                        Ordering::Less => ["lt", "le", "ne"].contains(&name),
120                        Ordering::Equal => ["ge", "le", "eq"].contains(&name),
121                        Ordering::Greater => ["ge", "gt", "ne"].contains(&name),
122                    };
123                    let result = if result { 255 } else { 0 };
124                    destination_bytes.extend(std::iter::repeat_n(result, dest_size));
125                }
126
127                destination.write_from_bytes(self, &destination_bytes)
128            }
129            "bitmask" => {
130                let [op] = args else {
131                    return Err(MirEvalError::InternalError(
132                        "simd_bitmask args are not provided".into(),
133                    ));
134                };
135                let (op_len, _) = self.detect_simd_ty(op.ty)?;
136                let op_count = op.interval.size / op_len;
137                let mut result: u64 = 0;
138                for (i, val) in op.get(self)?.chunks(op_count).enumerate() {
139                    if !val.iter().all(|&it| it == 0) {
140                        result |= 1 << i;
141                    }
142                }
143                destination.write_from_bytes(self, &result.to_le_bytes()[0..destination.size])
144            }
145            "shuffle" => {
146                let [left, right, index] = args else {
147                    return Err(MirEvalError::InternalError(
148                        "simd_shuffle args are not provided".into(),
149                    ));
150                };
151                let TyKind::Array(_, index_len) = index.ty.kind() else {
152                    return Err(MirEvalError::InternalError(
153                        "simd_shuffle index argument has non-array type".into(),
154                    ));
155                };
156                let index_len = match try_const_usize(self.db, index_len) {
157                    Some(it) => it as usize,
158                    None => {
159                        return Err(MirEvalError::InternalError(
160                            "simd type with unevaluatable len param".into(),
161                        ));
162                    }
163                };
164                let (left_len, _) = self.detect_simd_ty(left.ty)?;
165                let left_size = left.interval.size / left_len;
166                let vector =
167                    left.get(self)?.chunks(left_size).chain(right.get(self)?.chunks(left_size));
168                let mut result = vec![];
169                for index in index.get(self)?.chunks(index.interval.size / index_len) {
170                    let index = from_bytes!(u32, index) as usize;
171                    let val = match vector.clone().nth(index) {
172                        Some(it) => it,
173                        None => {
174                            return Err(MirEvalError::InternalError(
175                                "out of bound access in simd shuffle".into(),
176                            ));
177                        }
178                    };
179                    result.extend(val);
180                }
181                destination.write_from_bytes(self, &result)
182            }
183            _ => not_supported!("unknown simd intrinsic {name}"),
184        }
185    }
