90pub struct VTableMap<'db> {
91    ty_to_id: FxHashMap<Ty<'db>, usize>,
92    id_to_ty: Vec<Ty<'db>>,
93}
94
95impl<'db> VTableMap<'db> {
96    const OFFSET: usize = 1000; // We should add some offset to ids to make 0 (null) an invalid id.
97
98    fn id(&mut self, ty: Ty<'db>) -> usize {
99        if let Some(it) = self.ty_to_id.get(&ty) {
100            return *it;
101        }
102        let id = self.id_to_ty.len() + VTableMap::OFFSET;
103        self.id_to_ty.push(ty);
104        self.ty_to_id.insert(ty, id);
105        id
106    }
107
108    pub(crate) fn ty(&self, id: usize) -> Result<'db, Ty<'db>> {
109        id.checked_sub(VTableMap::OFFSET)
110            .and_then(|id| self.id_to_ty.get(id).copied())
111            .ok_or(MirEvalError::InvalidVTableId(id))
112    }
113
114    fn ty_of_bytes(&self, bytes: &[u8]) -> Result<'db, Ty<'db>> {
115        let id = from_bytes!(usize, bytes);
116        self.ty(id)
117    }
118
119    pub fn shrink_to_fit(&mut self) {
120        self.id_to_ty.shrink_to_fit();
121        self.ty_to_id.shrink_to_fit();
122    }
123
124    fn is_empty(&self) -> bool {
125        self.id_to_ty.is_empty() && self.ty_to_id.is_empty()
126    }
127}

142            MirEvalError::UndefinedBehavior(format!("Getting invalid tls key {key}"))

144        Ok(*r)

150        })?;
151        *r = value;
152        Ok(())
153    }

294#[cfg(target_pointer_width = "64")]
295const STACK_OFFSET: usize = 1 << 60;
296#[cfg(target_pointer_width = "64")]
297const HEAP_OFFSET: usize = 1 << 59;
298
299#[cfg(target_pointer_width = "32")]
300const STACK_OFFSET: usize = 1 << 30;
301#[cfg(target_pointer_width = "32")]
302const HEAP_OFFSET: usize = 1 << 29;
303
304impl Address {
305    #[allow(clippy::double_parens)]
306    fn from_bytes<'db>(it: &[u8]) -> Result<'db, Self> {
307        Ok(Address::from_usize(from_bytes!(usize, it)))
308    }
309
310    fn from_usize(it: usize) -> Self {
311        if it > STACK_OFFSET {
312            Stack(it - STACK_OFFSET)
313        } else if it > HEAP_OFFSET {
314            Heap(it - HEAP_OFFSET)
315        } else {
316            Invalid(it)
317        }
318    }
319
320    fn to_bytes(&self) -> [u8; size_of::<usize>()] {
321        usize::to_le_bytes(self.to_usize())
322    }
323
324    fn to_usize(&self) -> usize {
325        match self {
326            Stack(it) => *it + STACK_OFFSET,
327            Heap(it) => *it + HEAP_OFFSET,
328            Invalid(it) => *it,
329        }
330    }
331
332    fn map(&self, f: impl FnOnce(usize) -> usize) -> Address {
333        match self {
334            Stack(it) => Stack(f(*it)),
335            Heap(it) => Heap(f(*it)),
336            Invalid(it) => Invalid(f(*it)),
337        }
338    }
339
340    fn offset(&self, offset: usize) -> Address {
341        self.map(|it| it + offset)
342    }
343}

366    /// FIXME: Fold this into InternalError

374impl MirEvalError {
375    pub fn pretty_print(
376        &self,
377        f: &mut String,
378        db: &dyn HirDatabase,
379        span_formatter: impl Fn(FileId, TextRange) -> String,
380        display_target: DisplayTarget,
381    ) -> std::result::Result<(), std::fmt::Error> {
382        writeln!(f, "Mir eval error:")?;
383        let mut err = self;
384        while let MirEvalError::InFunction(e, stack) = err {
385            err = e;
386            for (func, span, def) in stack.iter().take(30).rev() {
387                match func {
388                    Either::Left(func) => {
389                        let function_name = db.function_signature(*func);
390                        writeln!(
391                            f,
392                            "In function {} ({:?})",
393                            function_name.name.display(db, display_target.edition),
394                            func
395                        )?;
396                    }
397                    Either::Right(closure) => {
398                        writeln!(f, "In {closure:?}")?;
399                    }
400                }
401                let source_map = db.body_with_source_map(*def).1;
402                let span: InFile<SyntaxNodePtr> = match span {
403                    MirSpan::ExprId(e) => match source_map.expr_syntax(*e) {
404                        Ok(s) => s.map(|it| it.into()),
405                        Err(_) => continue,
406                    },
407                    MirSpan::PatId(p) => match source_map.pat_syntax(*p) {
408                        Ok(s) => s.map(|it| it.syntax_node_ptr()),
409                        Err(_) => continue,
410                    },
411                    MirSpan::BindingId(b) => {
412                        match source_map
413                            .patterns_for_binding(*b)
414                            .iter()
415                            .find_map(|p| source_map.pat_syntax(*p).ok())
416                        {
417                            Some(s) => s.map(|it| it.syntax_node_ptr()),
418                            None => continue,
419                        }
420                    }
421                    MirSpan::SelfParam => match source_map.self_param_syntax() {
422                        Some(s) => s.map(|it| it.syntax_node_ptr()),
423                        None => continue,
424                    },
425                    MirSpan::Unknown => continue,
426                };
427                let file_id = span.file_id.original_file(db);
428                let text_range = span.value.text_range();
429                writeln!(f, "{}", span_formatter(file_id.file_id(db), text_range))?;
430            }
431        }
432        match err {
433            MirEvalError::InFunction(..) => unreachable!(),
434            MirEvalError::LayoutError(err, ty) => {
435                write!(
436                    f,
437                    "Layout for type `{}` is not available due {err:?}",
438                    ty.as_ref()
439                        .display(db, display_target)
440                        .with_closure_style(ClosureStyle::ClosureWithId)
441                )?;
442            }
443            MirEvalError::MirLowerError(func, err) => {
444                let function_name = db.function_signature(*func);
445                let self_ = match func.lookup(db).container {
446                    ItemContainerId::ImplId(impl_id) => Some({
447                        db.impl_self_ty(impl_id)
448                            .instantiate_identity()
449                            .display(db, display_target)
450                            .to_string()
451                    }),
452                    ItemContainerId::TraitId(it) => Some(
453                        db.trait_signature(it).name.display(db, display_target.edition).to_string(),
454                    ),
455                    _ => None,
456                };
457                writeln!(
458                    f,
459                    "MIR lowering for function `{}{}{}` ({:?}) failed due:",
460                    self_.as_deref().unwrap_or_default(),
461                    if self_.is_some() { "::" } else { "" },
462                    function_name.name.display(db, display_target.edition),
463                    func
464                )?;
465                err.pretty_print(f, db, span_formatter, display_target)?;
466            }
467            MirEvalError::ConstEvalError(name, err) => {
468                MirLowerError::ConstEvalError((**name).into(), err.clone()).pretty_print(
469                    f,
470                    db,
471                    span_formatter,
472                    display_target,
473                )?;
474            }
475            MirEvalError::UndefinedBehavior(_)
476            | MirEvalError::TargetDataLayoutNotAvailable(_)
477            | MirEvalError::Panic(_)
478            | MirEvalError::MirLowerErrorForClosure(_, _)
479            | MirEvalError::TypeIsUnsized(_, _)
480            | MirEvalError::NotSupported(_)
481            | MirEvalError::InvalidConst(_)
482            | MirEvalError::ExecutionLimitExceeded
483            | MirEvalError::StackOverflow
484            | MirEvalError::CoerceUnsizedError(_)
485            | MirEvalError::InternalError(_)
486            | MirEvalError::InvalidVTableId(_) => writeln!(f, "{err:?}")?,
487        }
488        Ok(())
489    }
490
491    pub fn is_panic(&self) -> Option<&str> {
492        let mut err = self;
493        while let MirEvalError::InFunction(e, _) = err {
494            err = e;
495        }
496        match err {
497            MirEvalError::Panic(msg) => Some(msg),
498            _ => None,
499        }
500    }

593pub fn interpret_mir<'db>(
594    db: &'db dyn HirDatabase,
595    body: Arc<MirBody>,
596    // FIXME: This is workaround. Ideally, const generics should have a separate body (issue #7434), but now
597    // they share their body with their parent, so in MIR lowering we have locals of the parent body, which
598    // might have placeholders. With this argument, we (wrongly) assume that every placeholder type has
599    // a zero size, hoping that they are all outside of our current body. Even without a fix for #7434, we can
600    // (and probably should) do better here, for example by excluding bindings outside of the target expression.
601    assert_placeholder_ty_is_unused: bool,
602    trait_env: Option<ParamEnvAndCrate<'db>>,
603) -> Result<'db, (Result<'db, Const<'db>>, MirOutput)> {
604    let ty = body.locals[return_slot()].ty.as_ref();
605    let mut evaluator = Evaluator::new(db, body.owner, assert_placeholder_ty_is_unused, trait_env)?;
606    let it: Result<'db, Const<'db>> = (|| {
607        if evaluator.ptr_size() != size_of::<usize>() {
608            not_supported!("targets with different pointer size from host");
609        }
610        let interval = evaluator.interpret_mir(body.clone(), None.into_iter())?;
611        let bytes = interval.get(&evaluator)?;
612        let mut memory_map = evaluator.create_memory_map(
613            bytes,
614            ty,
615            &Locals { ptr: ArenaMap::new(), body, drop_flags: DropFlags::default() },
616        )?;
617        let bytes = bytes.into();
618        let memory_map = if memory_map.memory.is_empty() && evaluator.vtable_map.is_empty() {
619            MemoryMap::Empty
620        } else {
621            memory_map.vtable = mem::take(&mut evaluator.vtable_map);
622            memory_map.vtable.shrink_to_fit();
623            MemoryMap::Complex(Box::new(memory_map))
624        };
625        Ok(Const::new_valtree(evaluator.interner(), ty, bytes, memory_map))
626    })();
627    Ok((it, MirOutput { stdout: evaluator.stdout, stderr: evaluator.stderr }))
628}

723        let pair = (ty, proj);
724        if let Some(r) = self.projected_ty_cache.borrow().get(&pair) {

726        }

741                    .get()
742                    .instantiate(self.interner(), parent_subst)

750    fn place_addr_and_ty_and_metadata<'a>(
751        &'a self,
752        p: &Place,
753        locals: &'a Locals,
754    ) -> Result<'db, (Address, Ty<'db>, Option<IntervalOrOwned>)> {
755        let mut addr = locals.ptr[p.local].addr;
756        let mut ty: Ty<'db> = locals.body.locals[p.local].ty.as_ref();
757        let mut metadata: Option<IntervalOrOwned> = None; // locals are always sized
758        for proj in p.projection.lookup(&locals.body.projection_store) {
759            let prev_ty = ty;
760            ty = self.projected_ty(ty, proj.clone());
761            match proj {
762                ProjectionElem::Deref => {
763                    metadata = if self.size_align_of(ty, locals)?.is_none() {
764                        Some(
765                            Interval { addr: addr.offset(self.ptr_size()), size: self.ptr_size() }
766                                .into(),
767                        )
768                    } else {
769                        None
770                    };
771                    let it = from_bytes!(usize, self.read_memory(addr, self.ptr_size())?);
772                    addr = Address::from_usize(it);
773                }
774                ProjectionElem::Index(op) => {
775                    let offset = from_bytes!(
776                        usize,
777                        self.read_memory(locals.ptr[*op].addr, self.ptr_size())?
778                    );
779                    metadata = None; // Result of index is always sized
780                    let ty_size =
781                        self.size_of_sized(ty, locals, "array inner type should be sized")?;
782                    addr = addr.offset(ty_size * offset);
783                }
784                &ProjectionElem::ConstantIndex { from_end, offset } => {
785                    let offset = if from_end {
786                        let len = match prev_ty.kind() {
787                            TyKind::Array(_, c) => match try_const_usize(self.db, c) {
788                                Some(it) => it as u64,
789                                None => {
790                                    not_supported!("indexing array with unknown const from end")
791                                }
792                            },
793                            TyKind::Slice(_) => match metadata {
794                                Some(it) => from_bytes!(u64, it.get(self)?),
795                                None => not_supported!("slice place without metadata"),
796                            },
797                            _ => not_supported!("bad type for const index"),
798                        };
799                        (len - offset - 1) as usize
800                    } else {
801                        offset as usize
802                    };
803                    metadata = None; // Result of index is always sized
804                    let ty_size =
805                        self.size_of_sized(ty, locals, "array inner type should be sized")?;
806                    addr = addr.offset(ty_size * offset);
807                }
808                &ProjectionElem::Subslice { from, to } => {
809                    let inner_ty = match ty.kind() {
810                        TyKind::Array(inner, _) | TyKind::Slice(inner) => inner,
811                        _ => Ty::new_error(self.interner(), ErrorGuaranteed),
812                    };
813                    metadata = match metadata {
814                        Some(it) => {
815                            let prev_len = from_bytes!(u64, it.get(self)?);
816                            Some(IntervalOrOwned::Owned(
817                                (prev_len - from - to).to_le_bytes().to_vec(),
818                            ))
819                        }
820                        None => None,
821                    };
822                    let ty_size =
823                        self.size_of_sized(inner_ty, locals, "array inner type should be sized")?;
824                    addr = addr.offset(ty_size * (from as usize));
825                }
826                &ProjectionElem::ClosureField(f) => {
827                    let layout = self.layout(prev_ty)?;
828                    let offset = layout.fields.offset(f).bytes_usize();
829                    addr = addr.offset(offset);
830                    metadata = None;
831                }
832                ProjectionElem::Field(Either::Right(f)) => {
833                    let layout = self.layout(prev_ty)?;
834                    let offset = layout.fields.offset(f.index as usize).bytes_usize();
835                    addr = addr.offset(offset);
836                    metadata = None; // tuple field is always sized FIXME: This is wrong, the tail can be unsized
837                }
838                ProjectionElem::Field(Either::Left(f)) => {
839                    let layout = self.layout(prev_ty)?;
840                    let variant_layout = match &layout.variants {
841                        Variants::Single { .. } | Variants::Empty => &layout,
842                        Variants::Multiple { variants, .. } => {
843                            &variants[match f.parent {
844                                hir_def::VariantId::EnumVariantId(it) => {
845                                    RustcEnumVariantIdx(it.lookup(self.db).index as usize)
846                                }
847                                _ => {
848                                    return Err(MirEvalError::InternalError(
849                                        "mismatched layout".into(),
850                                    ));
851                                }
852                            }]
853                        }
854                    };
855                    let offset = variant_layout
856                        .fields
857                        .offset(u32::from(f.local_id.into_raw()) as usize)
858                        .bytes_usize();
859                    addr = addr.offset(offset);
860                    // Unsized field metadata is equal to the metadata of the struct
861                    if self.size_align_of(ty, locals)?.is_some() {
862                        metadata = None;
863                    }
864                }
865                ProjectionElem::OpaqueCast(_) => not_supported!("opaque cast"),
866            }
867        }
868        Ok((addr, ty, metadata))
869    }

919    fn interpret_mir(
920        &mut self,
921        body: Arc<MirBody>,
922        args: impl Iterator<Item = IntervalOrOwned>,
923    ) -> Result<'db, Interval> {
924        if let Some(it) = self.stack_depth_limit.checked_sub(1) {
925            self.stack_depth_limit = it;
926        } else {
927            return Err(MirEvalError::StackOverflow);
928        }
929        let mut current_block_idx = body.start_block;
930        let (mut locals, prev_stack_ptr) = self.create_locals_for_body(&body, None)?;
931        self.fill_locals_for_body(&body, &mut locals, args)?;
932        let prev_code_stack = mem::take(&mut self.code_stack);
933        let span = (MirSpan::Unknown, body.owner);
934        self.code_stack.push(StackFrame { locals, destination: None, prev_stack_ptr, span });
935        'stack: loop {
936            let Some(mut my_stack_frame) = self.code_stack.pop() else {
937                not_supported!("missing stack frame");
938            };
939            let e = (|| {
940                let locals = &mut my_stack_frame.locals;
941                let body = locals.body.clone();
942                loop {
943                    let current_block = &body.basic_blocks[current_block_idx];
944                    if let Some(it) = self.execution_limit.checked_sub(1) {
945                        self.execution_limit = it;
946                    } else {
947                        return Err(MirEvalError::ExecutionLimitExceeded);
948                    }
949                    for statement in &current_block.statements {
950                        match &statement.kind {
951                            StatementKind::Assign(l, r) => {
952                                let addr = self.place_addr(l, locals)?;
953                                let result = self.eval_rvalue(r, locals)?;
954                                self.copy_from_interval_or_owned(addr, result)?;
955                                locals.drop_flags.add_place(*l, &locals.body.projection_store);
956                            }
957                            StatementKind::Deinit(_) => not_supported!("de-init statement"),
958                            StatementKind::StorageLive(_)
959                            | StatementKind::FakeRead(_)
960                            | StatementKind::StorageDead(_)
961                            | StatementKind::Nop => (),
962                        }
963                    }
964                    let Some(terminator) = current_block.terminator.as_ref() else {
965                        not_supported!("block without terminator");
966                    };
967                    match &terminator.kind {
968                        TerminatorKind::Goto { target } => {
969                            current_block_idx = *target;
970                        }
971                        TerminatorKind::Call {
972                            func,
973                            args,
974                            destination,
975                            target,
976                            cleanup: _,
977                            from_hir_call: _,
978                        } => {
979                            let destination_interval = self.place_interval(destination, locals)?;
980                            let fn_ty = self.operand_ty(func, locals)?;
981                            let args = args
982                                .iter()
983                                .map(|it| self.operand_ty_and_eval(it, locals))
984                                .collect::<Result<'db, Vec<_>>>()?;
985                            let stack_frame = match fn_ty.kind() {
986                                TyKind::FnPtr(..) => {
987                                    let bytes = self.eval_operand(func, locals)?;
988                                    self.exec_fn_pointer(
989                                        bytes,
990                                        destination_interval,
991                                        &args,
992                                        locals,
993                                        *target,
994                                        terminator.span,
995                                    )?
996                                }
997                                TyKind::FnDef(def, generic_args) => self.exec_fn_def(
998                                    def.0,
999                                    generic_args,
1000                                    destination_interval,
1001                                    &args,
1002                                    locals,
1003                                    *target,
1004                                    terminator.span,
1005                                )?,
1006                                it => not_supported!("unknown function type {it:?}"),
1007                            };
1008                            locals
1009                                .drop_flags
1010                                .add_place(*destination, &locals.body.projection_store);
1011                            if let Some(stack_frame) = stack_frame {
1012                                self.code_stack.push(my_stack_frame);
1013                                current_block_idx = stack_frame.locals.body.start_block;
1014                                self.code_stack.push(stack_frame);
1015                                return Ok(None);
1016                            } else {
1017                                current_block_idx =
1018                                    target.ok_or(MirEvalError::UndefinedBehavior(
1019                                        "Diverging function returned".to_owned(),
1020                                    ))?;
1021                            }
1022                        }
1023                        TerminatorKind::SwitchInt { discr, targets } => {
1024                            let val = u128::from_le_bytes(pad16(
1025                                self.eval_operand(discr, locals)?.get(self)?,
1026                                false,
1027                            ));
1028                            current_block_idx = targets.target_for_value(val);
1029                        }
1030                        TerminatorKind::Return => {
1031                            break;
1032                        }
1033                        TerminatorKind::Unreachable => {
1034                            return Err(MirEvalError::UndefinedBehavior(
1035                                "unreachable executed".to_owned(),
1036                            ));
1037                        }
1038                        TerminatorKind::Drop { place, target, unwind: _ } => {
1039                            self.drop_place(place, locals, terminator.span)?;
1040                            current_block_idx = *target;
1041                        }
1042                        _ => not_supported!("unknown terminator"),
1043                    }
1044                }
1045                Ok(Some(my_stack_frame))
1046            })();
1047            let my_stack_frame = match e {
1048                Ok(None) => continue 'stack,
1049                Ok(Some(x)) => x,
1050                Err(e) => {
1051                    let my_code_stack = mem::replace(&mut self.code_stack, prev_code_stack);
1052                    let mut error_stack = vec![];
1053                    for frame in my_code_stack.into_iter().rev() {
1054                        if let DefWithBodyId::FunctionId(f) = frame.locals.body.owner {
1055                            error_stack.push((Either::Left(f), frame.span.0, frame.span.1));
1056                        }
1057                    }
1058                    return Err(MirEvalError::InFunction(Box::new(e), error_stack));
1059                }
1060            };
1061            let return_interval = my_stack_frame.locals.ptr[return_slot()];
1062            self.unused_locals_store
1063                .borrow_mut()
1064                .entry(my_stack_frame.locals.body.owner)
1065                .or_default()
1066                .push(my_stack_frame.locals);
1067            match my_stack_frame.destination {
1068                None => {
1069                    self.code_stack = prev_code_stack;
1070                    self.stack_depth_limit += 1;
1071                    return Ok(return_interval);
1072                }
1073                Some(bb) => {
1074                    // We don't support const promotion, so we can't truncate the stack yet.
1075                    let _ = my_stack_frame.prev_stack_ptr;
1076                    // self.stack.truncate(my_stack_frame.prev_stack_ptr);
1077                    current_block_idx = bb;
1078                }
1079            }
1080        }
1081    }

1083    fn fill_locals_for_body(
1084        &mut self,
1085        body: &MirBody,
1086        locals: &mut Locals,
1087        args: impl Iterator<Item = IntervalOrOwned>,
1088    ) -> Result<'db, ()> {
1089        let mut remain_args = body.param_locals.len();
1090        for ((l, interval), value) in locals.ptr.iter().skip(1).zip(args) {
1091            locals.drop_flags.add_place(l.into(), &locals.body.projection_store);
1092            match value {
1093                IntervalOrOwned::Owned(value) => interval.write_from_bytes(self, &value)?,
1094                IntervalOrOwned::Borrowed(value) => interval.write_from_interval(self, value)?,
1095            }
1096            if remain_args == 0 {
1097                return Err(MirEvalError::InternalError("too many arguments".into()));
1098            }
1099            remain_args -= 1;
1100        }
1101        if remain_args > 0 {
1102            return Err(MirEvalError::InternalError("too few arguments".into()));
1103        }
1104        Ok(())
1105    }

1107    fn create_locals_for_body(
1108        &mut self,
1109        body: &Arc<MirBody>,
1110        destination: Option<Interval>,
1111    ) -> Result<'db, (Locals, usize)> {
1112        let mut locals =
1113            match self.unused_locals_store.borrow_mut().entry(body.owner).or_default().pop() {
1114                None => Locals {
1115                    ptr: ArenaMap::new(),
1116                    body: body.clone(),
1117                    drop_flags: DropFlags::default(),
1118                },
1119                Some(mut l) => {
1120                    l.drop_flags.clear();
1121                    l.body = body.clone();
1122                    l
1123                }
1124            };
1125        let stack_size = {
1126            let mut stack_ptr = self.stack.len();
1127            for (id, it) in body.locals.iter() {
1128                if id == return_slot()
1129                    && let Some(destination) = destination
1130                {
1131                    locals.ptr.insert(id, destination);
1132                    continue;
1133                }
1134                let (size, align) = self.size_align_of_sized(
1135                    it.ty.as_ref(),
1136                    &locals,
1137                    "no unsized local in extending stack",
1138                )?;
1139                while !stack_ptr.is_multiple_of(align) {
1140                    stack_ptr += 1;
1141                }
1142                let my_ptr = stack_ptr;
1143                stack_ptr += size;
1144                locals.ptr.insert(id, Interval { addr: Stack(my_ptr), size });
1145            }
1146            stack_ptr - self.stack.len()
1147        };
1148        let prev_stack_pointer = self.stack.len();
1149        if stack_size > self.memory_limit {
1150            return Err(MirEvalError::Panic(format!(
1151                "Stack overflow. Tried to grow stack to {stack_size} bytes"
1152            )));
1153        }
1154        self.stack.extend(std::iter::repeat_n(0, stack_size));
1155        Ok((locals, prev_stack_pointer))
1156    }

1158    fn eval_rvalue(&mut self, r: &Rvalue, locals: &mut Locals) -> Result<'db, IntervalOrOwned> {
1159        use IntervalOrOwned::*;
1160        Ok(match r {
1161            Rvalue::Use(it) => Borrowed(self.eval_operand(it, locals)?),
1162            Rvalue::Ref(_, p) => {
1163                let (addr, _, metadata) = self.place_addr_and_ty_and_metadata(p, locals)?;
1164                let mut r = addr.to_bytes().to_vec();
1165                if let Some(metadata) = metadata {
1166                    r.extend(metadata.get(self)?);
1167                }
1168                Owned(r)
1169            }
1170            Rvalue::Len(p) => {
1171                let (_, _, metadata) = self.place_addr_and_ty_and_metadata(p, locals)?;
1172                match metadata {
1173                    Some(m) => m,
1174                    None => {
1175                        return Err(MirEvalError::InternalError(
1176                            "type without metadata is used for Rvalue::Len".into(),
1177                        ));
1178                    }
1179                }
1180            }
1181            Rvalue::UnaryOp(op, val) => {
1182                let mut c = self.eval_operand(val, locals)?.get(self)?;
1183                let mut ty = self.operand_ty(val, locals)?;
1184                while let TyKind::Ref(_, z, _) = ty.kind() {
1185                    ty = z;
1186                    let size = self.size_of_sized(ty, locals, "operand of unary op")?;
1187                    c = self.read_memory(Address::from_bytes(c)?, size)?;
1188                }
1189                if let TyKind::Float(f) = ty.kind() {
1190                    match f {
1191                        rustc_type_ir::FloatTy::F16 => {
1192                            let c = -from_bytes!(f16, u16, c);
1193                            Owned(u16::try_from(c.to_bits()).unwrap().to_le_bytes().into())
1194                        }
1195                        rustc_type_ir::FloatTy::F32 => {
1196                            let c = -from_bytes!(f32, c);
1197                            Owned(c.to_le_bytes().into())
1198                        }
1199                        rustc_type_ir::FloatTy::F64 => {
1200                            let c = -from_bytes!(f64, c);
1201                            Owned(c.to_le_bytes().into())
1202                        }

1205                            Owned(c.to_bits().to_le_bytes().into())
1206                        }

1211                        c[0] = 1 - c[0];
1212                    } else {

1214                            UnOp::Not => c.iter_mut().for_each(|it| *it = !*it),
1215                            UnOp::Neg => {
1216                                c.iter_mut().for_each(|it| *it = !*it);

1219                                    (*k, o) = k.overflowing_add(1);

1222                                    }

1224                            }

1277                                        BinOp::Mul => l * r,
1278                                        BinOp::Div => l / r,

1286                                    )
1287                                }

1289                                    "invalid binop {it:?} on floating point operators"

1292                        }

1659    fn compute_discriminant(&self, ty: Ty<'db>, bytes: &[u8]) -> Result<'db, i128> {
1660        let layout = self.layout(ty)?;
1661        let TyKind::Adt(adt_def, _) = ty.kind() else {
1662            return Ok(0);
1663        };
1664        let AdtId::EnumId(e) = adt_def.def_id().0 else {
1665            return Ok(0);
1666        };
1667        match &layout.variants {
1668            Variants::Empty => unreachable!(),
1669            Variants::Single { index } => {
1670                let r =
1671                    self.const_eval_discriminant(e.enum_variants(self.db).variants[index.0].0)?;
1672                Ok(r)
1673            }
1674            Variants::Multiple { tag, tag_encoding, variants, .. } => {
1675                let size = tag.size(&*self.target_data_layout).bytes_usize();
1676                let offset = layout.fields.offset(0).bytes_usize(); // The only field on enum variants is the tag field
1677                let is_signed = tag.is_signed();
1678                match tag_encoding {
1679                    TagEncoding::Direct => {
1680                        let tag = &bytes[offset..offset + size];
1681                        Ok(i128::from_le_bytes(pad16(tag, is_signed)))
1682                    }
1683                    TagEncoding::Niche { untagged_variant, niche_start, .. } => {
1684                        let tag = &bytes[offset..offset + size];
1685                        let candidate_tag = i128::from_le_bytes(pad16(tag, is_signed))
1686                            .wrapping_sub(*niche_start as i128)
1687                            as usize;
1688                        let idx = variants
1689                            .iter_enumerated()
1690                            .map(|(it, _)| it)
1691                            .filter(|it| it != untagged_variant)
1692                            .nth(candidate_tag)
1693                            .unwrap_or(*untagged_variant)
1694                            .0;
1695                        let result =
1696                            self.const_eval_discriminant(e.enum_variants(self.db).variants[idx].0)?;
1697                        Ok(result)
1698                    }
1699                }
1700            }
1701        }
1702    }

1704    fn coerce_unsized_look_through_fields<T>(
1705        &self,
1706        ty: Ty<'db>,
1707        goal: impl Fn(TyKind<'db>) -> Option<T>,
1708    ) -> Result<'db, T> {
1709        let kind = ty.kind();
1710        if let Some(it) = goal(kind) {
1711            return Ok(it);
1712        }
1713        if let TyKind::Adt(adt_ef, subst) = kind
1714            && let AdtId::StructId(struct_id) = adt_ef.def_id().0
1715        {
1716            let field_types = self.db.field_types(struct_id.into());
1717            if let Some(ty) =
1718                field_types.iter().last().map(|it| it.1.get().instantiate(self.interner(), subst))
1719            {
1720                return self.coerce_unsized_look_through_fields(ty, goal);
1721            }
1722        }
1723        Err(MirEvalError::CoerceUnsizedError(ty.store()))
1724    }

1726    fn coerce_unsized(
1727        &mut self,
1728        addr: Interval,
1729        current_ty: Ty<'db>,
1730        target_ty: Ty<'db>,
1731    ) -> Result<'db, IntervalOrOwned> {
1732        fn for_ptr<'db>(it: TyKind<'db>) -> Option<Ty<'db>> {
1733            match it {
1734                TyKind::RawPtr(ty, _) | TyKind::Ref(_, ty, _) => Some(ty),
1735                _ => None,
1736            }
1737        }
1738        let target_ty = self.coerce_unsized_look_through_fields(target_ty, for_ptr)?;
1739        let current_ty = self.coerce_unsized_look_through_fields(current_ty, for_ptr)?;
1740
1741        self.unsizing_ptr_from_addr(target_ty, current_ty, addr)
1742    }

1746        &mut self,
1747        target_ty: Ty<'db>,
1748        current_ty: Ty<'db>,
1749        addr: Interval,
1750    ) -> Result<'db, IntervalOrOwned> {
1751        use IntervalOrOwned::*;
1752        Ok(match &target_ty.kind() {
1753            TyKind::Slice(_) => match &current_ty.kind() {
1754                TyKind::Array(_, size) => {
1755                    let len = match try_const_usize(self.db, *size) {
1756                        None => {
1757                            not_supported!("unevaluatble len of array in coerce unsized")
1758                        }
1759                        Some(it) => it as usize,
1760                    };
1761                    let mut r = Vec::with_capacity(16);
1762                    let addr = addr.get(self)?;
1763                    r.extend(addr.iter().copied());
1764                    r.extend(len.to_le_bytes());
1765                    Owned(r)
1766                }
1767                t => {
1768                    not_supported!("slice unsizing from non array type {t:?}")
1769                }
1770            },
1771            TyKind::Dynamic(..) => {
1772                let vtable = self.vtable_map.id(current_ty);
1773                let mut r = Vec::with_capacity(16);
1774                let addr = addr.get(self)?;
1775                r.extend(addr.iter().copied());
1776                r.extend(vtable.to_le_bytes());
1777                Owned(r)
1778            }
1779            TyKind::Adt(adt_def, target_subst) => match &current_ty.kind() {
1780                TyKind::Adt(current_adt_def, current_subst) => {
1781                    let id = adt_def.def_id().0;
1782                    let current_id = current_adt_def.def_id().0;
1783                    if id != current_id {
1784                        not_supported!("unsizing struct with different type");
1785                    }
1786                    let id = match id {
1787                        AdtId::StructId(s) => s,
1788                        AdtId::UnionId(_) => not_supported!("unsizing unions"),
1789                        AdtId::EnumId(_) => not_supported!("unsizing enums"),
1790                    };
1791                    let Some((last_field, _)) = id.fields(self.db).fields().iter().next_back()
1792                    else {
1793                        not_supported!("unsizing struct without field");
1794                    };
1795                    let target_last_field = self.db.field_types(id.into())[last_field]
1796                        .get()
1797                        .instantiate(self.interner(), target_subst);
1798                    let current_last_field = self.db.field_types(id.into())[last_field]
1799                        .get()
1800                        .instantiate(self.interner(), current_subst);
1801                    return self.unsizing_ptr_from_addr(
1802                        target_last_field,
1803                        current_last_field,
1804                        addr,
1805                    );
1806                }
1807                _ => not_supported!("unsizing struct with non adt type"),
1808            },
1809            _ => not_supported!("unknown unsized cast"),
1810        })
1811    }

1813    fn layout_of_variant(
1814        &mut self,
1815        it: VariantId,
1816        subst: GenericArgs<'db>,
1817        locals: &Locals,
1818    ) -> Result<'db, (usize, Arc<Layout>, Option<(usize, usize, i128)>)> {
1819        let adt = it.adt_id(self.db);
1820        if let DefWithBodyId::VariantId(f) = locals.body.owner
1821            && let VariantId::EnumVariantId(it) = it
1822            && let AdtId::EnumId(e) = adt
1823            && f.lookup(self.db).parent == e
1824        {
1825            // Computing the exact size of enums require resolving the enum discriminants. In order to prevent loops (and
1826            // infinite sized type errors) we use a dummy layout
1827            let i = self.const_eval_discriminant(it)?;
1828            return Ok((16, self.layout(Ty::new_empty_tuple(self.interner()))?, Some((0, 16, i))));
1829        }
1830        let layout = self.layout_adt(adt, subst)?;
1831        Ok(match &layout.variants {
1832            Variants::Single { .. } | Variants::Empty => (layout.size.bytes_usize(), layout, None),
1833            Variants::Multiple { variants, tag, tag_encoding, .. } => {
1834                let enum_variant_id = match it {
1835                    VariantId::EnumVariantId(it) => it,
1836                    _ => not_supported!("multi variant layout for non-enums"),
1837                };
1838                let mut discriminant = self.const_eval_discriminant(enum_variant_id)?;
1839                let lookup = enum_variant_id.lookup(self.db);
1840                let rustc_enum_variant_idx = RustcEnumVariantIdx(lookup.index as usize);
1841                let variant_layout = variants[rustc_enum_variant_idx].clone();
1842                let have_tag = match tag_encoding {
1843                    TagEncoding::Direct => true,
1844                    TagEncoding::Niche { untagged_variant, niche_variants: _, niche_start } => {
1845                        if *untagged_variant == rustc_enum_variant_idx {
1846                            false
1847                        } else {
1848                            discriminant = (variants
1849                                .iter_enumerated()
1850                                .filter(|(it, _)| it != untagged_variant)
1851                                .position(|(it, _)| it == rustc_enum_variant_idx)
1852                                .unwrap() as i128)
1853                                .wrapping_add(*niche_start as i128);
1854                            true
1855                        }
1856                    }
1857                };
1858                (
1859                    layout.size.bytes_usize(),
1860                    Arc::new(variant_layout),
1861                    if have_tag {
1862                        Some((
1863                            layout.fields.offset(0).bytes_usize(),
1864                            tag.size(&*self.target_data_layout).bytes_usize(),
1865                            discriminant,
1866                        ))
1867                    } else {
1868                        None
1869                    },
1870                )
1871            }
1872        })
1873    }

1928    fn allocate_const_in_heap(
1929        &mut self,
1930        locals: &Locals,
1931        konst: Const<'db>,
1932    ) -> Result<'db, Interval> {
1933        let result_owner;
1934        let value = match konst.kind() {
1935            ConstKind::Value(value) => value,
1936            ConstKind::Unevaluated(UnevaluatedConst { def: const_id, args: subst }) => 'b: {
1937                let mut id = const_id.0;
1938                let mut subst = subst;
1939                if let hir_def::GeneralConstId::ConstId(c) = id {
1940                    let (c, s) = lookup_impl_const(&self.infcx, self.param_env.param_env, c, subst);
1941                    id = hir_def::GeneralConstId::ConstId(c);
1942                    subst = s;
1943                }
1944                result_owner = match id {
1945                    GeneralConstId::ConstId(const_id) => {
1946                        self.db.const_eval(const_id, subst, Some(self.param_env)).map_err(|e| {
1947                            let name = id.name(self.db);
1948                            MirEvalError::ConstEvalError(name, Box::new(e))
1949                        })?
1950                    }
1951                    GeneralConstId::StaticId(static_id) => {
1952                        self.db.const_eval_static(static_id).map_err(|e| {
1953                            let name = id.name(self.db);
1954                            MirEvalError::ConstEvalError(name, Box::new(e))
1955                        })?
1956                    }
1957                };
1958                if let ConstKind::Value(value) = result_owner.kind() {
1959                    break 'b value;
1960                }
1961                not_supported!("unevaluatable constant");
1962            }
1963            _ => not_supported!("evaluating unknown const"),
1964        };
1965        let ValueConst { ty, value } = value;
1966        let ConstBytes { memory: v, memory_map } = value.inner();
1967        let patch_map = memory_map.transform_addresses(|b, align| {
1968            let addr = self.heap_allocate(b.len(), align)?;
1969            self.write_memory(addr, b)?;
1970            Ok(addr.to_usize())
1971        })?;
1972        let (size, align) = self.size_align_of(ty, locals)?.unwrap_or((v.len(), 1));
1973        let v: Cow<'_, [u8]> = if size != v.len() {
1974            // Handle self enum
1975            if size == 16 && v.len() < 16 {
1976                Cow::Owned(pad16(v, false).to_vec())
1977            } else if size < 16 && v.len() == 16 {
1978                Cow::Borrowed(&v[0..size])
1979            } else {
1980                return Err(MirEvalError::InvalidConst(konst.store()));
1981            }
1982        } else {
1983            Cow::Borrowed(v)
1984        };
1985        let addr = self.heap_allocate(size, align)?;
1986        self.write_memory(addr, &v)?;
1987        self.patch_addresses(
1988            &patch_map,
1989            |bytes| match memory_map {
1990                MemoryMap::Empty | MemoryMap::Simple(_) => {
1991                    Err(MirEvalError::InvalidVTableId(from_bytes!(usize, bytes)))
1992                }
1993                MemoryMap::Complex(cm) => cm.vtable.ty_of_bytes(bytes),
1994            },
1995            addr,
1996            ty,
1997            locals,
1998        )?;
1999        Ok(Interval::new(addr, size))
2000    }

2060    fn copy_from_interval(&mut self, addr: Address, r: Interval) -> Result<'db, ()> {
2061        if r.size == 0 {
2062            return Ok(());
2063        }
2064
2065        let oob = || MirEvalError::UndefinedBehavior("out of bounds memory write".to_owned());
2066
2067        match (addr, r.addr) {
2068            (Stack(dst), Stack(src)) => {
2069                if self.stack.len() < src + r.size || self.stack.len() < dst + r.size {
2070                    return Err(oob());
2071                }
2072                self.stack.copy_within(src..src + r.size, dst)
2073            }
2074            (Heap(dst), Heap(src)) => {
2075                if self.stack.len() < src + r.size || self.stack.len() < dst + r.size {
2076                    return Err(oob());
2077                }
2078                self.heap.copy_within(src..src + r.size, dst)
2079            }
2080            (Stack(dst), Heap(src)) => {
2081                self.stack
2082                    .get_mut(dst..dst + r.size)
2083                    .ok_or_else(oob)?
2084                    .copy_from_slice(self.heap.get(src..src + r.size).ok_or_else(oob)?);
2085            }
2086            (Heap(dst), Stack(src)) => {
2087                self.heap
2088                    .get_mut(dst..dst + r.size)
2089                    .ok_or_else(oob)?
2090                    .copy_from_slice(self.stack.get(src..src + r.size).ok_or_else(oob)?);
2091            }
2092            _ => {
2093                return Err(MirEvalError::UndefinedBehavior(format!(
2094                    "invalid memory write at address {addr:?}"
2095                )));
2096            }
2097        }
2098
2099        Ok(())
2100    }

2155        if !align.is_power_of_two() || align > 10000 {
2156            return Err(MirEvalError::UndefinedBehavior(format!("Alignment {align} is invalid")));

2163        }

2165        self.heap.extend(std::iter::repeat_n(0, size));
2166        Ok(Address::Heap(pos))
2167    }

2189            this: &Evaluator<'db>,
2190            bytes: &[u8],

2206                                addr_usize,

2209                        }
2210                        None => {

2240                                        ty,
2241                                        locals,

2246                            }
2247                        }

2259                            this,
2260                            &bytes[offset..offset + size],

2266                    }

2274                            this,
2275                            &bytes[offset..offset + size],
2276                            ty,
2277                            locals,
2278                            mm,
2279                            stack_depth_limit - 1,
2280                        )?;
2281                    }
2282                }

2284                    AdtId::StructId(s) => {

2289                            let offset = layout
2290                                .fields
2291                                .offset(u32::from(f.into_raw()) as usize)

2296                                this,
2297                                &bytes[offset..offset + size],
2298                                ty,
2299                                locals,
2300                                mm,
2301                                stack_depth_limit - 1,
2302                            )?;
2303                        }
2304                    }
2305                    AdtId::EnumId(e) => {
2306                        let layout = this.layout(ty)?;
2307                        if let Some((v, l)) = detect_variant_from_bytes(

2324                                    ty,

2327                                    stack_depth_limit - 1,
2328                                )?;

2330                        }
2331                    }

2355    fn patch_addresses(
2356        &mut self,
2357        patch_map: &FxHashMap<usize, usize>,
2358        ty_of_bytes: impl Fn(&[u8]) -> Result<'db, Ty<'db>> + Copy,
2359        addr: Address,
2360        ty: Ty<'db>,
2361        locals: &Locals,
2362    ) -> Result<'db, ()> {
2363        // FIXME: support indirect references
2364        let layout = self.layout(ty)?;
2365        let my_size = self.size_of_sized(ty, locals, "value to patch address")?;
2366        use rustc_type_ir::TyKind;
2367        match ty.kind() {
2368            TyKind::Ref(_, t, _) => {
2369                let size = self.size_align_of(t, locals)?;
2370                match size {
2371                    Some(_) => {
2372                        let current = from_bytes!(usize, self.read_memory(addr, my_size)?);
2373                        if let Some(it) = patch_map.get(&current) {
2374                            self.write_memory(addr, &it.to_le_bytes())?;
2375                        }
2376                    }
2377                    None => {
2378                        let current = from_bytes!(usize, self.read_memory(addr, my_size / 2)?);
2379                        if let Some(it) = patch_map.get(&current) {
2380                            self.write_memory(addr, &it.to_le_bytes())?;
2381                        }
2382                    }
2383                }
2384            }
2385            TyKind::FnPtr(_, _) => {
2386                let ty = ty_of_bytes(self.read_memory(addr, my_size)?)?;
2387                let new_id = self.vtable_map.id(ty);
2388                self.write_memory(addr, &new_id.to_le_bytes())?;
2389            }
2390            TyKind::Adt(id, args) => match id.def_id().0 {
2391                AdtId::StructId(s) => {
2392                    for (i, (_, ty)) in self.db.field_types(s.into()).iter().enumerate() {
2393                        let offset = layout.fields.offset(i).bytes_usize();
2394                        let ty = ty.get().instantiate(self.interner(), args);
2395                        self.patch_addresses(
2396                            patch_map,
2397                            ty_of_bytes,
2398                            addr.offset(offset),
2399                            ty,
2400                            locals,
2401                        )?;
2402                    }
2403                }
2404                AdtId::UnionId(_) => (),
2405                AdtId::EnumId(e) => {
2406                    if let Some((ev, layout)) = detect_variant_from_bytes(
2407                        &layout,
2408                        self.db,
2409                        &self.target_data_layout,
2410                        self.read_memory(addr, layout.size.bytes_usize())?,
2411                        e,
2412                    ) {
2413                        for (i, (_, ty)) in self.db.field_types(ev.into()).iter().enumerate() {
2414                            let offset = layout.fields.offset(i).bytes_usize();
2415                            let ty = ty.get().instantiate(self.interner(), args);
2416                            self.patch_addresses(
2417                                patch_map,
2418                                ty_of_bytes,
2419                                addr.offset(offset),
2420                                ty,
2421                                locals,
2422                            )?;
2423                        }
2424                    }
2425                }
2426            },
2427            TyKind::Tuple(tys) => {
2428                for (id, ty) in tys.iter().enumerate() {
2429                    let offset = layout.fields.offset(id).bytes_usize();
2430                    self.patch_addresses(patch_map, ty_of_bytes, addr.offset(offset), ty, locals)?;
2431                }
2432            }
2433            TyKind::Array(inner, len) => {
2434                let len = match consteval::try_const_usize(self.db, len) {
2435                    Some(it) => it as usize,
2436                    None => not_supported!("non evaluatable array len in patching addresses"),
2437                };
2438                let size = self.size_of_sized(inner, locals, "inner of array")?;
2439                for i in 0..len {
2440                    self.patch_addresses(
2441                        patch_map,
2442                        ty_of_bytes,
2443                        addr.offset(i * size),
2444                        inner,
2445                        locals,
2446                    )?;
2447                }
2448            }
2449            TyKind::Bool
2450            | TyKind::Char
2451            | TyKind::Int(_)
2452            | TyKind::Uint(_)
2453            | TyKind::Float(_)
2454            | TyKind::Slice(_)
2455            | TyKind::RawPtr(_, _)
2456            | TyKind::FnDef(_, _)
2457            | TyKind::Str
2458            | TyKind::Never
2459            | TyKind::Closure(_, _)
2460            | TyKind::Coroutine(_, _)
2461            | TyKind::CoroutineWitness(_, _)
2462            | TyKind::Foreign(_)
2463            | TyKind::Error(_)
2464            | TyKind::Placeholder(_)
2465            | TyKind::Dynamic(_, _)
2466            | TyKind::Alias(_, _)
2467            | TyKind::Bound(_, _)
2468            | TyKind::Infer(_)
2469            | TyKind::Pat(_, _)
2470            | TyKind::Param(_)
2471            | TyKind::UnsafeBinder(_)
2472            | TyKind::CoroutineClosure(_, _) => (),
2473        }
2474        Ok(())
2475    }

2646    fn exec_fn_with_args(
2647        &mut self,
2648        mut def: FunctionId,
2649        args: &[IntervalAndTy<'db>],
2650        generic_args: GenericArgs<'db>,
2651        locals: &Locals,
2652        destination: Interval,
2653        target_bb: Option<BasicBlockId>,
2654        span: MirSpan,
2655    ) -> Result<'db, Option<StackFrame>> {
2656        if self.detect_and_exec_special_function(
2657            def,
2658            args,
2659            generic_args,
2660            locals,
2661            destination,
2662            span,
2663        )? {
2664            return Ok(None);
2665        }
2666        if let Some(redirect_def) = self.detect_and_redirect_special_function(def)? {
2667            def = redirect_def;
2668        }
2669        let arg_bytes = args.iter().map(|it| IntervalOrOwned::Borrowed(it.interval));
2670        match self.get_mir_or_dyn_index(def, generic_args, locals, span)? {
2671            MirOrDynIndex::Dyn(self_ty_idx) => {
2672                // In the layout of current possible receiver, which at the moment of writing this code is one of
2673                // `&T`, `&mut T`, `Box<T>`, `Rc<T>`, `Arc<T>`, and `Pin<P>` where `P` is one of possible receivers,
2674                // the vtable is exactly in the `[ptr_size..2*ptr_size]` bytes. So we can use it without branching on
2675                // the type.
2676                let first_arg = arg_bytes.clone().next().unwrap();
2677                let first_arg = first_arg.get(self)?;
2678                let ty = self
2679                    .vtable_map
2680                    .ty_of_bytes(&first_arg[self.ptr_size()..self.ptr_size() * 2])?;
2681                let mut args_for_target = args.to_vec();
2682                args_for_target[0] = IntervalAndTy {
2683                    interval: args_for_target[0].interval.slice(0..self.ptr_size()),
2684                    ty,
2685                };
2686                let generics_for_target = GenericArgs::new_from_iter(
2687                    self.interner(),
2688                    generic_args
2689                        .iter()
2690                        .enumerate()
2691                        .map(|(i, it)| if i == self_ty_idx { ty.into() } else { it }),
2692                );
2693                self.exec_fn_with_args(
2694                    def,
2695                    &args_for_target,
2696                    generics_for_target,
2697                    locals,
2698                    destination,
2699                    target_bb,
2700                    span,
2701                )
2702            }
2703            MirOrDynIndex::Mir(body) => self.exec_looked_up_function(
2704                body,
2705                locals,
2706                def,
2707                arg_bytes,
2708                span,
2709                destination,
2710                target_bb,
2711            ),
2712        }
2713    }

2743    fn exec_fn_trait(
2744        &mut self,
2745        def: FunctionId,
2746        args: &[IntervalAndTy<'db>],
2747        generic_args: GenericArgs<'db>,
2748        locals: &Locals,
2749        destination: Interval,
2750        target_bb: Option<BasicBlockId>,
2751        span: MirSpan,
2752    ) -> Result<'db, Option<StackFrame>> {
2753        let func = args
2754            .first()
2755            .ok_or_else(|| MirEvalError::InternalError("fn trait with no arg".into()))?;
2756        let mut func_ty = func.ty;
2757        let mut func_data = func.interval;
2758        while let TyKind::Ref(_, z, _) = func_ty.kind() {
2759            func_ty = z;
2760            if matches!(func_ty.kind(), TyKind::Dynamic(..)) {
2761                let id =
2762                    from_bytes!(usize, &func_data.get(self)?[self.ptr_size()..self.ptr_size() * 2]);
2763                func_data = func_data.slice(0..self.ptr_size());
2764                func_ty = self.vtable_map.ty(id)?;
2765            }
2766            let size = self.size_of_sized(func_ty, locals, "self type of fn trait")?;
2767            func_data = Interval { addr: Address::from_bytes(func_data.get(self)?)?, size };
2768        }
2769        match func_ty.kind() {
2770            TyKind::FnDef(def, subst) => {
2771                self.exec_fn_def(def.0, subst, destination, &args[1..], locals, target_bb, span)
2772            }
2773            TyKind::FnPtr(..) => {
2774                self.exec_fn_pointer(func_data, destination, &args[1..], locals, target_bb, span)
2775            }
2776            TyKind::Closure(closure, subst) => self.exec_closure(
2777                closure.0,
2778                func_data,
2779                GenericArgs::new_from_slice(subst.as_closure().parent_args()),
2780                destination,
2781                &args[1..],
2782                locals,
2783                span,
2784            ),
2785            _ => {
2786                // try to execute the manual impl of `FnTrait` for structs (nightly feature used in std)
2787                let arg0 = func;
2788                let args = &args[1..];
2789                let arg1 = {
2790                    let ty = Ty::new_tup_from_iter(self.interner(), args.iter().map(|it| it.ty));
2791                    let layout = self.layout(ty)?;
2792                    let result = self.construct_with_layout(
2793                        layout.size.bytes_usize(),
2794                        &layout,
2795                        None,
2796                        args.iter().map(|it| IntervalOrOwned::Borrowed(it.interval)),
2797                    )?;
2798                    // FIXME: there is some leak here
2799                    let size = layout.size.bytes_usize();
2800                    let addr = self.heap_allocate(size, layout.align.bytes() as usize)?;
2801                    self.write_memory(addr, &result)?;
2802                    IntervalAndTy { interval: Interval { addr, size }, ty }
2803                };
2804                self.exec_fn_with_args(
2805                    def,
2806                    &[arg0.clone(), arg1],
2807                    generic_args,
2808                    locals,
2809                    destination,
2810                    target_bb,
2811                    span,
2812                )
2813            }
2814        }
2815    }

2840    }

2876    fn run_drop_glue_deep(
2877        &mut self,
2878        ty: Ty<'db>,
2879        locals: &Locals,
2880        addr: Address,
2881        _metadata: &[u8],
2882        span: MirSpan,
2883    ) -> Result<'db, ()> {
2884        let Some(drop_fn) = (|| {
2885            let drop_trait = self.lang_items().Drop?;
2886            drop_trait.trait_items(self.db).method_by_name(&Name::new_symbol_root(sym::drop))
2887        })() else {
2888            // in some tests we don't have drop trait in minicore, and
2889            // we can ignore drop in them.
2890            return Ok(());
2891        };
2892
2893        let generic_args = GenericArgs::new_from_slice(&[ty.into()]);
2894        if let Ok(MirOrDynIndex::Mir(body)) =
2895            self.get_mir_or_dyn_index(drop_fn, generic_args, locals, span)
2896        {
2897            self.exec_looked_up_function(
2898                body,
2899                locals,
2900                drop_fn,
2901                iter::once(IntervalOrOwned::Owned(addr.to_bytes().to_vec())),
2902                span,
2903                Interval { addr: Address::Invalid(0), size: 0 },
2904                None,
2905            )?;
2906        }
2907        match ty.kind() {
2908            TyKind::Adt(adt_def, subst) => {

2932                            }
2933                            FieldsShape::Unit => (),
2934                        }
2935                    }
2936                    AdtId::UnionId(_) => (), // union fields don't need drop

2938                }

2940            TyKind::Bool
2941            | TyKind::Char
