64macro_rules! from_bytes {
65    ($ty:tt, $value:expr) => {
66        ($ty::from_le_bytes(match ($value).try_into() {
67            Ok(it) => it,
68            Err(_) => return Err(MirEvalError::InternalError(stringify!(mismatched size in constructing $ty).into())),

75            Err(_) => return Err(MirEvalError::InternalError(stringify!(mismatched size in constructing $apfloat).into())),
76        }).into())
77    };

87pub struct VTableMap<'db> {
88    ty_to_id: FxHashMap<Ty<'db>, usize>,
89    id_to_ty: Vec<Ty<'db>>,
90}
91
92impl<'db> VTableMap<'db> {
93    const OFFSET: usize = 1000; // We should add some offset to ids to make 0 (null) an invalid id.
94
95    fn id(&mut self, ty: Ty<'db>) -> usize {
96        if let Some(it) = self.ty_to_id.get(&ty) {
97            return *it;
98        }
99        let id = self.id_to_ty.len() + VTableMap::OFFSET;
100        self.id_to_ty.push(ty);
101        self.ty_to_id.insert(ty, id);
102        id
103    }
104
105    pub(crate) fn ty(&self, id: usize) -> Result<'db, Ty<'db>> {
106        id.checked_sub(VTableMap::OFFSET)
107            .and_then(|id| self.id_to_ty.get(id).copied())
108            .ok_or(MirEvalError::InvalidVTableId(id))
109    }
110
111    fn ty_of_bytes(&self, bytes: &[u8]) -> Result<'db, Ty<'db>> {
112        let id = from_bytes!(usize, bytes);
113        self.ty(id)
114    }
115
116    pub fn shrink_to_fit(&mut self) {
117        self.id_to_ty.shrink_to_fit();
118        self.ty_to_id.shrink_to_fit();
119    }
120
121    fn is_empty(&self) -> bool {
122        self.id_to_ty.is_empty() && self.ty_to_id.is_empty()
123    }
124}

141        Ok(*r)
142    }

147        })?;
148        *r = value;
149        Ok(())
150    }

291#[cfg(target_pointer_width = "64")]
292const STACK_OFFSET: usize = 1 << 60;
293#[cfg(target_pointer_width = "64")]
294const HEAP_OFFSET: usize = 1 << 59;
295
296#[cfg(target_pointer_width = "32")]
297const STACK_OFFSET: usize = 1 << 30;
298#[cfg(target_pointer_width = "32")]
299const HEAP_OFFSET: usize = 1 << 29;
300
301impl Address {
302    #[allow(clippy::double_parens)]
303    fn from_bytes<'db>(it: &[u8]) -> Result<'db, Self> {
304        Ok(Address::from_usize(from_bytes!(usize, it)))
305    }
306
307    fn from_usize(it: usize) -> Self {
308        if it > STACK_OFFSET {
309            Stack(it - STACK_OFFSET)
310        } else if it > HEAP_OFFSET {
311            Heap(it - HEAP_OFFSET)
312        } else {
313            Invalid(it)
314        }
315    }
316
317    fn to_bytes(&self) -> [u8; size_of::<usize>()] {
318        usize::to_le_bytes(self.to_usize())
319    }
320
321    fn to_usize(&self) -> usize {
322        match self {
323            Stack(it) => *it + STACK_OFFSET,
324            Heap(it) => *it + HEAP_OFFSET,
325            Invalid(it) => *it,
326        }
327    }
328
329    fn map(&self, f: impl FnOnce(usize) -> usize) -> Address {
330        match self {
331            Stack(it) => Stack(f(*it)),
332            Heap(it) => Heap(f(*it)),
333            Invalid(it) => Invalid(f(*it)),
334        }
335    }
336
337    fn offset(&self, offset: usize) -> Address {
338        self.map(|it| it + offset)
339    }
340}

344    ConstEvalError(String, Box<ConstEvalError<'db>>),
345    LayoutError(LayoutError, Ty<'db>),

347    /// Means that code had undefined behavior. We don't try to actively detect UB, but if it was detected

353    MirLowerErrorForClosure(InternedClosureId, MirLowerError<'db>),
354    TypeIsUnsized(Ty<'db>, &'static str),
355    NotSupported(String),
356    InvalidConst(Const<'db>),
357    InFunction(
358        Box<MirEvalError<'db>>,
359        Vec<(Either<FunctionId, InternedClosureId>, MirSpan, DefWithBodyId)>,

361    ExecutionLimitExceeded,
362    StackOverflow,
363    /// FIXME: Fold this into InternalError
364    InvalidVTableId(usize),

366    CoerceUnsizedError(Ty<'db>),
367    /// These should not occur, usually indicates a bug in mir lowering.

371impl MirEvalError<'_> {
372    pub fn pretty_print(
373        &self,
374        f: &mut String,
375        db: &dyn HirDatabase,
376        span_formatter: impl Fn(FileId, TextRange) -> String,
377        display_target: DisplayTarget,
378    ) -> std::result::Result<(), std::fmt::Error> {
379        writeln!(f, "Mir eval error:")?;
380        let mut err = self;
381        while let MirEvalError::InFunction(e, stack) = err {
382            err = e;
383            for (func, span, def) in stack.iter().take(30).rev() {
384                match func {
385                    Either::Left(func) => {
386                        let function_name = db.function_signature(*func);
387                        writeln!(
388                            f,
389                            "In function {} ({:?})",
390                            function_name.name.display(db, display_target.edition),
391                            func
392                        )?;
393                    }
394                    Either::Right(closure) => {
395                        writeln!(f, "In {closure:?}")?;
396                    }
397                }
398                let source_map = db.body_with_source_map(*def).1;
399                let span: InFile<SyntaxNodePtr> = match span {
400                    MirSpan::ExprId(e) => match source_map.expr_syntax(*e) {
401                        Ok(s) => s.map(|it| it.into()),
402                        Err(_) => continue,
403                    },
404                    MirSpan::PatId(p) => match source_map.pat_syntax(*p) {
405                        Ok(s) => s.map(|it| it.syntax_node_ptr()),
406                        Err(_) => continue,
407                    },
408                    MirSpan::BindingId(b) => {
409                        match source_map
410                            .patterns_for_binding(*b)
411                            .iter()
412                            .find_map(|p| source_map.pat_syntax(*p).ok())
413                        {
414                            Some(s) => s.map(|it| it.syntax_node_ptr()),
415                            None => continue,
416                        }
417                    }
418                    MirSpan::SelfParam => match source_map.self_param_syntax() {
419                        Some(s) => s.map(|it| it.syntax_node_ptr()),
420                        None => continue,
421                    },
422                    MirSpan::Unknown => continue,
423                };
424                let file_id = span.file_id.original_file(db);
425                let text_range = span.value.text_range();
426                writeln!(f, "{}", span_formatter(file_id.file_id(db), text_range))?;
427            }
428        }
429        match err {
430            MirEvalError::InFunction(..) => unreachable!(),
431            MirEvalError::LayoutError(err, ty) => {
432                write!(
433                    f,
434                    "Layout for type `{}` is not available due {err:?}",
435                    ty.display(db, display_target).with_closure_style(ClosureStyle::ClosureWithId)
436                )?;
437            }
438            MirEvalError::MirLowerError(func, err) => {
439                let function_name = db.function_signature(*func);
440                let self_ = match func.lookup(db).container {
441                    ItemContainerId::ImplId(impl_id) => Some({
442                        db.impl_self_ty(impl_id)
443                            .instantiate_identity()
444                            .display(db, display_target)
445                            .to_string()
446                    }),
447                    ItemContainerId::TraitId(it) => Some(
448                        db.trait_signature(it).name.display(db, display_target.edition).to_string(),
449                    ),
450                    _ => None,
451                };
452                writeln!(
453                    f,
454                    "MIR lowering for function `{}{}{}` ({:?}) failed due:",
455                    self_.as_deref().unwrap_or_default(),
456                    if self_.is_some() { "::" } else { "" },
457                    function_name.name.display(db, display_target.edition),
458                    func
459                )?;
460                err.pretty_print(f, db, span_formatter, display_target)?;
461            }
462            MirEvalError::ConstEvalError(name, err) => {
463                MirLowerError::ConstEvalError((**name).into(), err.clone()).pretty_print(
464                    f,
465                    db,
466                    span_formatter,
467                    display_target,
468                )?;
469            }
470            MirEvalError::UndefinedBehavior(_)
471            | MirEvalError::TargetDataLayoutNotAvailable(_)
472            | MirEvalError::Panic(_)
473            | MirEvalError::MirLowerErrorForClosure(_, _)
474            | MirEvalError::TypeIsUnsized(_, _)
475            | MirEvalError::NotSupported(_)
476            | MirEvalError::InvalidConst(_)
477            | MirEvalError::ExecutionLimitExceeded
478            | MirEvalError::StackOverflow
479            | MirEvalError::CoerceUnsizedError(_)
480            | MirEvalError::InternalError(_)
481            | MirEvalError::InvalidVTableId(_) => writeln!(f, "{err:?}")?,
482        }
483        Ok(())
484    }
485
486    pub fn is_panic(&self) -> Option<&str> {
487        let mut err = self;
488        while let MirEvalError::InFunction(e, _) = err {
489            err = e;
490        }
491        match err {
492            MirEvalError::Panic(msg) => Some(msg),
493            _ => None,
494        }
495    }
496}

544impl<'db> DropFlags<'db> {

546        if p.iterate_over_parents(store).any(|it| self.need_drop.contains(&it)) {
547            return;
548        }
549        self.need_drop.retain(|it| !p.is_parent(it, store));
550        self.need_drop.insert(p);
551    }
552
553    fn remove_place(&mut self, p: &Place<'db>, store: &ProjectionStore<'db>) -> bool {
554        // FIXME: replace parents with parts
555        if let Some(parent) = p.iterate_over_parents(store).find(|it| self.need_drop.contains(it)) {
556            self.need_drop.remove(&parent);
557            return true;
558        }
559        self.need_drop.remove(p)
560    }
561
562    fn clear(&mut self) {
563        self.need_drop.clear();
564    }

588pub fn interpret_mir<'db>(
589    db: &'db dyn HirDatabase,
590    body: Arc<MirBody<'db>>,
591    // FIXME: This is workaround. Ideally, const generics should have a separate body (issue #7434), but now
592    // they share their body with their parent, so in MIR lowering we have locals of the parent body, which
593    // might have placeholders. With this argument, we (wrongly) assume that every placeholder type has
594    // a zero size, hoping that they are all outside of our current body. Even without a fix for #7434, we can
595    // (and probably should) do better here, for example by excluding bindings outside of the target expression.
596    assert_placeholder_ty_is_unused: bool,
597    trait_env: Option<Arc<TraitEnvironment<'db>>>,
598) -> Result<'db, (Result<'db, Const<'db>>, MirOutput)> {
599    let ty = body.locals[return_slot()].ty;
600    let mut evaluator = Evaluator::new(db, body.owner, assert_placeholder_ty_is_unused, trait_env)?;
601    let it: Result<'db, Const<'db>> = (|| {
602        if evaluator.ptr_size() != size_of::<usize>() {
603            not_supported!("targets with different pointer size from host");
604        }
605        let interval = evaluator.interpret_mir(body.clone(), None.into_iter())?;
606        let bytes = interval.get(&evaluator)?;
607        let mut memory_map = evaluator.create_memory_map(
608            bytes,
609            ty,
610            &Locals { ptr: ArenaMap::new(), body, drop_flags: DropFlags::default() },
611        )?;
612        let bytes = bytes.into();
613        let memory_map = if memory_map.memory.is_empty() && evaluator.vtable_map.is_empty() {
614            MemoryMap::Empty
615        } else {
616            memory_map.vtable = mem::take(&mut evaluator.vtable_map);
617            memory_map.vtable.shrink_to_fit();
618            MemoryMap::Complex(Box::new(memory_map))
619        };
620        Ok(Const::new_valtree(evaluator.interner(), ty, bytes, memory_map))
621    })();
622    Ok((it, MirOutput { stdout: evaluator.stdout, stderr: evaluator.stderr }))
623}

708    fn projected_ty(&self, ty: Ty<'db>, proj: PlaceElem<'db>) -> Ty<'db> {
709        let pair = (ty, proj);
710        if let Some(r) = self.projected_ty_cache.borrow().get(&pair) {
711            return *r;
712        }
713        let (ty, proj) = pair;
714        let r = proj.projected_ty(
715            &self.infcx,
716            ty,
717            |c, subst, f| {
718                let InternedClosure(def, _) = self.db.lookup_intern_closure(c);
719                let infer = self.db.infer(def);
720                let (captures, _) = infer.closure_info(c);
721                let parent_subst = subst.split_closure_args_untupled().parent_args;
722                captures
723                    .get(f)
724                    .expect("broken closure field")
725                    .ty
726                    .instantiate(self.interner(), parent_subst)
727            },
728            self.crate_id,
729        );
730        self.projected_ty_cache.borrow_mut().insert((ty, proj), r);
731        r

734    fn place_addr_and_ty_and_metadata<'a>(
735        &'a self,
736        p: &Place<'db>,
737        locals: &'a Locals<'db>,
738    ) -> Result<'db, (Address, Ty<'db>, Option<IntervalOrOwned>)> {
739        let mut addr = locals.ptr[p.local].addr;
740        let mut ty: Ty<'db> = locals.body.locals[p.local].ty;
741        let mut metadata: Option<IntervalOrOwned> = None; // locals are always sized
742        for proj in p.projection.lookup(&locals.body.projection_store) {
743            let prev_ty = ty;
744            ty = self.projected_ty(ty, proj.clone());
745            match proj {
746                ProjectionElem::Deref => {
747                    metadata = if self.size_align_of(ty, locals)?.is_none() {
748                        Some(
749                            Interval { addr: addr.offset(self.ptr_size()), size: self.ptr_size() }
750                                .into(),
751                        )
752                    } else {
753                        None
754                    };
755                    let it = from_bytes!(usize, self.read_memory(addr, self.ptr_size())?);
756                    addr = Address::from_usize(it);
757                }
758                ProjectionElem::Index(op) => {
759                    let offset = from_bytes!(
760                        usize,
761                        self.read_memory(locals.ptr[*op].addr, self.ptr_size())?
762                    );
763                    metadata = None; // Result of index is always sized
764                    let ty_size =
765                        self.size_of_sized(ty, locals, "array inner type should be sized")?;
766                    addr = addr.offset(ty_size * offset);
767                }
768                &ProjectionElem::ConstantIndex { from_end, offset } => {
769                    let offset = if from_end {
770                        let len = match prev_ty.kind() {
771                            TyKind::Array(_, c) => match try_const_usize(self.db, c) {
772                                Some(it) => it as u64,
773                                None => {
774                                    not_supported!("indexing array with unknown const from end")
775                                }
776                            },
777                            TyKind::Slice(_) => match metadata {
778                                Some(it) => from_bytes!(u64, it.get(self)?),
779                                None => not_supported!("slice place without metadata"),
780                            },
781                            _ => not_supported!("bad type for const index"),
782                        };
783                        (len - offset - 1) as usize
784                    } else {
785                        offset as usize
786                    };
787                    metadata = None; // Result of index is always sized
788                    let ty_size =
789                        self.size_of_sized(ty, locals, "array inner type should be sized")?;
790                    addr = addr.offset(ty_size * offset);
791                }
792                &ProjectionElem::Subslice { from, to } => {
793                    let inner_ty = match ty.kind() {
794                        TyKind::Array(inner, _) | TyKind::Slice(inner) => inner,
795                        _ => Ty::new_error(self.interner(), ErrorGuaranteed),
796                    };
797                    metadata = match metadata {
798                        Some(it) => {
799                            let prev_len = from_bytes!(u64, it.get(self)?);
800                            Some(IntervalOrOwned::Owned(
801                                (prev_len - from - to).to_le_bytes().to_vec(),
802                            ))
803                        }
804                        None => None,
805                    };
806                    let ty_size =
807                        self.size_of_sized(inner_ty, locals, "array inner type should be sized")?;
808                    addr = addr.offset(ty_size * (from as usize));
809                }
810                &ProjectionElem::ClosureField(f) => {
811                    let layout = self.layout(prev_ty)?;
812                    let offset = layout.fields.offset(f).bytes_usize();
813                    addr = addr.offset(offset);
814                    metadata = None;
815                }
816                ProjectionElem::Field(Either::Right(f)) => {
817                    let layout = self.layout(prev_ty)?;
818                    let offset = layout.fields.offset(f.index as usize).bytes_usize();
819                    addr = addr.offset(offset);
820                    metadata = None; // tuple field is always sized FIXME: This is wrong, the tail can be unsized
821                }
822                ProjectionElem::Field(Either::Left(f)) => {
823                    let layout = self.layout(prev_ty)?;
824                    let variant_layout = match &layout.variants {
825                        Variants::Single { .. } | Variants::Empty => &layout,
826                        Variants::Multiple { variants, .. } => {
827                            &variants[match f.parent {
828                                hir_def::VariantId::EnumVariantId(it) => {
829                                    RustcEnumVariantIdx(it.lookup(self.db).index as usize)
830                                }
831                                _ => {
832                                    return Err(MirEvalError::InternalError(
833                                        "mismatched layout".into(),
834                                    ));
835                                }
836                            }]
837                        }
838                    };
839                    let offset = variant_layout
840                        .fields
841                        .offset(u32::from(f.local_id.into_raw()) as usize)
842                        .bytes_usize();
843                    addr = addr.offset(offset);
844                    // Unsized field metadata is equal to the metadata of the struct
845                    if self.size_align_of(ty, locals)?.is_some() {
846                        metadata = None;
847                    }
848                }
849                ProjectionElem::OpaqueCast(_) => not_supported!("opaque cast"),
850            }
851        }
852        Ok((addr, ty, metadata))
853    }

902    fn interpret_mir(
903        &mut self,
904        body: Arc<MirBody<'db>>,
905        args: impl Iterator<Item = IntervalOrOwned>,
906    ) -> Result<'db, Interval> {
907        if let Some(it) = self.stack_depth_limit.checked_sub(1) {
908            self.stack_depth_limit = it;
909        } else {
910            return Err(MirEvalError::StackOverflow);
911        }
912        let mut current_block_idx = body.start_block;
913        let (mut locals, prev_stack_ptr) = self.create_locals_for_body(&body, None)?;
914        self.fill_locals_for_body(&body, &mut locals, args)?;
915        let prev_code_stack = mem::take(&mut self.code_stack);
916        let span = (MirSpan::Unknown, body.owner);
917        self.code_stack.push(StackFrame { locals, destination: None, prev_stack_ptr, span });
918        'stack: loop {
919            let Some(mut my_stack_frame) = self.code_stack.pop() else {
920                not_supported!("missing stack frame");
921            };
922            let e = (|| {
923                let locals = &mut my_stack_frame.locals;
924                let body = locals.body.clone();
925                loop {
926                    let current_block = &body.basic_blocks[current_block_idx];
927                    if let Some(it) = self.execution_limit.checked_sub(1) {
928                        self.execution_limit = it;
929                    } else {
930                        return Err(MirEvalError::ExecutionLimitExceeded);
931                    }
932                    for statement in &current_block.statements {
933                        match &statement.kind {
934                            StatementKind::Assign(l, r) => {
935                                let addr = self.place_addr(l, locals)?;
936                                let result = self.eval_rvalue(r, locals)?;
937                                self.copy_from_interval_or_owned(addr, result)?;
938                                locals.drop_flags.add_place(*l, &locals.body.projection_store);
939                            }
940                            StatementKind::Deinit(_) => not_supported!("de-init statement"),
941                            StatementKind::StorageLive(_)
942                            | StatementKind::FakeRead(_)
943                            | StatementKind::StorageDead(_)
944                            | StatementKind::Nop => (),
945                        }
946                    }
947                    let Some(terminator) = current_block.terminator.as_ref() else {
948                        not_supported!("block without terminator");
949                    };
950                    match &terminator.kind {
951                        TerminatorKind::Goto { target } => {
952                            current_block_idx = *target;
953                        }
954                        TerminatorKind::Call {
955                            func,
956                            args,
957                            destination,
958                            target,
959                            cleanup: _,
960                            from_hir_call: _,
961                        } => {
962                            let destination_interval = self.place_interval(destination, locals)?;
963                            let fn_ty = self.operand_ty(func, locals)?;
964                            let args = args
965                                .iter()
966                                .map(|it| self.operand_ty_and_eval(it, locals))
967                                .collect::<Result<'db, Vec<_>>>()?;
968                            let stack_frame = match fn_ty.kind() {
969                                TyKind::FnPtr(..) => {
970                                    let bytes = self.eval_operand(func, locals)?;
971                                    self.exec_fn_pointer(
972                                        bytes,
973                                        destination_interval,
974                                        &args,
975                                        locals,
976                                        *target,
977                                        terminator.span,
978                                    )?
979                                }
980                                TyKind::FnDef(def, generic_args) => self.exec_fn_def(
981                                    def.0,
982                                    generic_args,
983                                    destination_interval,
984                                    &args,
985                                    locals,
986                                    *target,
987                                    terminator.span,
988                                )?,
989                                it => not_supported!("unknown function type {it:?}"),
990                            };
991                            locals
992                                .drop_flags
993                                .add_place(*destination, &locals.body.projection_store);
994                            if let Some(stack_frame) = stack_frame {
995                                self.code_stack.push(my_stack_frame);
996                                current_block_idx = stack_frame.locals.body.start_block;
997                                self.code_stack.push(stack_frame);
998                                return Ok(None);
999                            } else {
1000                                current_block_idx =
1001                                    target.ok_or(MirEvalError::UndefinedBehavior(
1002                                        "Diverging function returned".to_owned(),
1003                                    ))?;
1004                            }
1005                        }
1006                        TerminatorKind::SwitchInt { discr, targets } => {
1007                            let val = u128::from_le_bytes(pad16(
1008                                self.eval_operand(discr, locals)?.get(self)?,
1009                                false,
1010                            ));
1011                            current_block_idx = targets.target_for_value(val);
1012                        }
1013                        TerminatorKind::Return => {
1014                            break;
1015                        }
1016                        TerminatorKind::Unreachable => {
1017                            return Err(MirEvalError::UndefinedBehavior(
1018                                "unreachable executed".to_owned(),
1019                            ));
1020                        }
1021                        TerminatorKind::Drop { place, target, unwind: _ } => {
1022                            self.drop_place(place, locals, terminator.span)?;
1023                            current_block_idx = *target;
1024                        }
1025                        _ => not_supported!("unknown terminator"),
1026                    }
1027                }
1028                Ok(Some(my_stack_frame))
1029            })();
1030            let my_stack_frame = match e {
1031                Ok(None) => continue 'stack,
1032                Ok(Some(x)) => x,
1033                Err(e) => {
1034                    let my_code_stack = mem::replace(&mut self.code_stack, prev_code_stack);
1035                    let mut error_stack = vec![];
1036                    for frame in my_code_stack.into_iter().rev() {
1037                        if let DefWithBodyId::FunctionId(f) = frame.locals.body.owner {
1038                            error_stack.push((Either::Left(f), frame.span.0, frame.span.1));
1039                        }
1040                    }
1041                    return Err(MirEvalError::InFunction(Box::new(e), error_stack));
1042                }
1043            };
1044            let return_interval = my_stack_frame.locals.ptr[return_slot()];
1045            self.unused_locals_store
1046                .borrow_mut()
1047                .entry(my_stack_frame.locals.body.owner)
1048                .or_default()
1049                .push(my_stack_frame.locals);
1050            match my_stack_frame.destination {
1051                None => {
1052                    self.code_stack = prev_code_stack;
1053                    self.stack_depth_limit += 1;
1054                    return Ok(return_interval);
1055                }
1056                Some(bb) => {
1057                    // We don't support const promotion, so we can't truncate the stack yet.
1058                    let _ = my_stack_frame.prev_stack_ptr;
1059                    // self.stack.truncate(my_stack_frame.prev_stack_ptr);
1060                    current_block_idx = bb;
1061                }
1062            }
1063        }
1064    }

1066    fn fill_locals_for_body(
1067        &mut self,
1068        body: &MirBody<'db>,
1069        locals: &mut Locals<'db>,
1070        args: impl Iterator<Item = IntervalOrOwned>,
1071    ) -> Result<'db, ()> {
1072        let mut remain_args = body.param_locals.len();
1073        for ((l, interval), value) in locals.ptr.iter().skip(1).zip(args) {
1074            locals.drop_flags.add_place(l.into(), &locals.body.projection_store);
1075            match value {
1076                IntervalOrOwned::Owned(value) => interval.write_from_bytes(self, &value)?,
1077                IntervalOrOwned::Borrowed(value) => interval.write_from_interval(self, value)?,
1078            }
1079            if remain_args == 0 {
1080                return Err(MirEvalError::InternalError("too many arguments".into()));
1081            }
1082            remain_args -= 1;
1083        }
1084        if remain_args > 0 {
1085            return Err(MirEvalError::InternalError("too few arguments".into()));
1086        }
1087        Ok(())
1088    }

1090    fn create_locals_for_body(
1091        &mut self,
1092        body: &Arc<MirBody<'db>>,
1093        destination: Option<Interval>,
1094    ) -> Result<'db, (Locals<'db>, usize)> {
1095        let mut locals =
1096            match self.unused_locals_store.borrow_mut().entry(body.owner).or_default().pop() {
1097                None => Locals {
1098                    ptr: ArenaMap::new(),
1099                    body: body.clone(),
1100                    drop_flags: DropFlags::default(),
1101                },
1102                Some(mut l) => {
1103                    l.drop_flags.clear();
1104                    l.body = body.clone();
1105                    l
1106                }
1107            };
1108        let stack_size = {
1109            let mut stack_ptr = self.stack.len();
1110            for (id, it) in body.locals.iter() {
1111                if id == return_slot()
1112                    && let Some(destination) = destination
1113                {
1114                    locals.ptr.insert(id, destination);
1115                    continue;
1116                }
1117                let (size, align) = self.size_align_of_sized(
1118                    it.ty,
1119                    &locals,
1120                    "no unsized local in extending stack",
1121                )?;
1122                while !stack_ptr.is_multiple_of(align) {
1123                    stack_ptr += 1;
1124                }
1125                let my_ptr = stack_ptr;
1126                stack_ptr += size;
1127                locals.ptr.insert(id, Interval { addr: Stack(my_ptr), size });
1128            }
1129            stack_ptr - self.stack.len()
1130        };
1131        let prev_stack_pointer = self.stack.len();
1132        if stack_size > self.memory_limit {
1133            return Err(MirEvalError::Panic(format!(
1134                "Stack overflow. Tried to grow stack to {stack_size} bytes"
1135            )));
1136        }
1137        self.stack.extend(std::iter::repeat_n(0, stack_size));
1138        Ok((locals, prev_stack_pointer))
1139    }

1141    fn eval_rvalue(
1142        &mut self,
1143        r: &Rvalue<'db>,
1144        locals: &mut Locals<'db>,
1145    ) -> Result<'db, IntervalOrOwned> {
1146        use IntervalOrOwned::*;
1147        Ok(match r {
1148            Rvalue::Use(it) => Borrowed(self.eval_operand(it, locals)?),
1149            Rvalue::Ref(_, p) => {
1150                let (addr, _, metadata) = self.place_addr_and_ty_and_metadata(p, locals)?;
1151                let mut r = addr.to_bytes().to_vec();
1152                if let Some(metadata) = metadata {
1153                    r.extend(metadata.get(self)?);
1154                }
1155                Owned(r)
1156            }
1157            Rvalue::Len(p) => {
1158                let (_, _, metadata) = self.place_addr_and_ty_and_metadata(p, locals)?;
1159                match metadata {
1160                    Some(m) => m,
1161                    None => {
1162                        return Err(MirEvalError::InternalError(
1163                            "type without metadata is used for Rvalue::Len".into(),
1164                        ));
1165                    }
1166                }
1167            }
1168            Rvalue::UnaryOp(op, val) => {
1169                let mut c = self.eval_operand(val, locals)?.get(self)?;
1170                let mut ty = self.operand_ty(val, locals)?;
1171                while let TyKind::Ref(_, z, _) = ty.kind() {
1172                    ty = z;
1173                    let size = self.size_of_sized(ty, locals, "operand of unary op")?;
1174                    c = self.read_memory(Address::from_bytes(c)?, size)?;
1175                }
1176                if let TyKind::Float(f) = ty.kind() {
1177                    match f {
1178                        rustc_type_ir::FloatTy::F16 => {
1179                            let c = -from_bytes!(f16, u16, c);
1180                            Owned(u16::try_from(c.to_bits()).unwrap().to_le_bytes().into())
1181                        }
1182                        rustc_type_ir::FloatTy::F32 => {
1183                            let c = -from_bytes!(f32, c);
1184                            Owned(c.to_le_bytes().into())
1185                        }
1186                        rustc_type_ir::FloatTy::F64 => {
1187                            let c = -from_bytes!(f64, c);
1188                            Owned(c.to_le_bytes().into())
1189                        }
1190                        rustc_type_ir::FloatTy::F128 => {

1642    fn compute_discriminant(&self, ty: Ty<'db>, bytes: &[u8]) -> Result<'db, i128> {
1643        let layout = self.layout(ty)?;
1644        let TyKind::Adt(adt_def, _) = ty.kind() else {
1645            return Ok(0);
1646        };
1647        let AdtId::EnumId(e) = adt_def.def_id().0 else {
1648            return Ok(0);
1649        };
1650        match &layout.variants {
1651            Variants::Empty => unreachable!(),
1652            Variants::Single { index } => {
1653                let r =
1654                    self.const_eval_discriminant(e.enum_variants(self.db).variants[index.0].0)?;
1655                Ok(r)
1656            }
1657            Variants::Multiple { tag, tag_encoding, variants, .. } => {
1658                let size = tag.size(&*self.target_data_layout).bytes_usize();
1659                let offset = layout.fields.offset(0).bytes_usize(); // The only field on enum variants is the tag field
1660                let is_signed = tag.is_signed();
1661                match tag_encoding {
1662                    TagEncoding::Direct => {
1663                        let tag = &bytes[offset..offset + size];
1664                        Ok(i128::from_le_bytes(pad16(tag, is_signed)))
1665                    }
1666                    TagEncoding::Niche { untagged_variant, niche_start, .. } => {
1667                        let tag = &bytes[offset..offset + size];
1668                        let candidate_tag = i128::from_le_bytes(pad16(tag, is_signed))
1669                            .wrapping_sub(*niche_start as i128)
1670                            as usize;
1671                        let idx = variants
1672                            .iter_enumerated()
1673                            .map(|(it, _)| it)
1674                            .filter(|it| it != untagged_variant)
1675                            .nth(candidate_tag)
1676                            .unwrap_or(*untagged_variant)
1677                            .0;
1678                        let result =
1679                            self.const_eval_discriminant(e.enum_variants(self.db).variants[idx].0)?;
1680                        Ok(result)
1681                    }
1682                }
1683            }
1684        }
1685    }

1687    fn coerce_unsized_look_through_fields<T>(
1688        &self,
1689        ty: Ty<'db>,
1690        goal: impl Fn(TyKind<'db>) -> Option<T>,
1691    ) -> Result<'db, T> {
1692        let kind = ty.kind();
1693        if let Some(it) = goal(kind) {
1694            return Ok(it);
1695        }
1696        if let TyKind::Adt(adt_ef, subst) = kind
1697            && let AdtId::StructId(struct_id) = adt_ef.def_id().0
1698        {
1699            let field_types = self.db.field_types(struct_id.into());
1700            if let Some(ty) =
1701                field_types.iter().last().map(|it| it.1.instantiate(self.interner(), subst))
1702            {
1703                return self.coerce_unsized_look_through_fields(ty, goal);
1704            }
1705        }
1706        Err(MirEvalError::CoerceUnsizedError(ty))
1707    }

1709    fn coerce_unsized(
1710        &mut self,
1711        addr: Interval,
1712        current_ty: Ty<'db>,
1713        target_ty: Ty<'db>,
1714    ) -> Result<'db, IntervalOrOwned> {
1715        fn for_ptr<'db>(it: TyKind<'db>) -> Option<Ty<'db>> {
1716            match it {
1717                TyKind::RawPtr(ty, _) | TyKind::Ref(_, ty, _) => Some(ty),
1718                _ => None,
1719            }
1720        }
1721        let target_ty = self.coerce_unsized_look_through_fields(target_ty, for_ptr)?;
1722        let current_ty = self.coerce_unsized_look_through_fields(current_ty, for_ptr)?;
1723
1724        self.unsizing_ptr_from_addr(target_ty, current_ty, addr)
1725    }

1735        Ok(match &target_ty.kind() {
1736            TyKind::Slice(_) => match &current_ty.kind() {
1737                TyKind::Array(_, size) => {
1738                    let len = match try_const_usize(self.db, *size) {
1739                        None => {
1740                            not_supported!("unevaluatble len of array in coerce unsized")
1741                        }
1742                        Some(it) => it as usize,
1743                    };
1744                    let mut r = Vec::with_capacity(16);
1745                    let addr = addr.get(self)?;
1746                    r.extend(addr.iter().copied());
1747                    r.extend(len.to_le_bytes());
1748                    Owned(r)
1749                }
1750                t => {
1751                    not_supported!("slice unsizing from non array type {t:?}")
1752                }
1753            },
1754            TyKind::Dynamic(..) => {
1755                let vtable = self.vtable_map.id(current_ty);
1756                let mut r = Vec::with_capacity(16);
1757                let addr = addr.get(self)?;
1758                r.extend(addr.iter().copied());
1759                r.extend(vtable.to_le_bytes());
1760                Owned(r)
1761            }
1762            TyKind::Adt(adt_def, target_subst) => match &current_ty.kind() {
1763                TyKind::Adt(current_adt_def, current_subst) => {
1764                    let id = adt_def.def_id().0;
1765                    let current_id = current_adt_def.def_id().0;
1766                    if id != current_id {
1767                        not_supported!("unsizing struct with different type");
1768                    }
1769                    let id = match id {
1770                        AdtId::StructId(s) => s,
1771                        AdtId::UnionId(_) => not_supported!("unsizing unions"),
1772                        AdtId::EnumId(_) => not_supported!("unsizing enums"),
1773                    };
1774                    let Some((last_field, _)) = id.fields(self.db).fields().iter().next_back()
1775                    else {
1776                        not_supported!("unsizing struct without field");
1777                    };
1778                    let target_last_field = self.db.field_types(id.into())[last_field]
1779                        .instantiate(self.interner(), target_subst);
1780                    let current_last_field = self.db.field_types(id.into())[last_field]
1781                        .instantiate(self.interner(), current_subst);
1782                    return self.unsizing_ptr_from_addr(
1783                        target_last_field,
1784                        current_last_field,
1785                        addr,
1786                    );
1787                }
1788                _ => not_supported!("unsizing struct with non adt type"),
1789            },
1790            _ => not_supported!("unknown unsized cast"),
1791        })
1792    }

1794    fn layout_of_variant(
1795        &mut self,
1796        it: VariantId,
1797        subst: GenericArgs<'db>,
1798        locals: &Locals<'db>,
1799    ) -> Result<'db, (usize, Arc<Layout>, Option<(usize, usize, i128)>)> {
1800        let adt = it.adt_id(self.db);
1801        if let DefWithBodyId::VariantId(f) = locals.body.owner
1802            && let VariantId::EnumVariantId(it) = it
1803            && let AdtId::EnumId(e) = adt
1804            && f.lookup(self.db).parent == e
1805        {
1806            // Computing the exact size of enums require resolving the enum discriminants. In order to prevent loops (and
1807            // infinite sized type errors) we use a dummy layout
1808            let i = self.const_eval_discriminant(it)?;
1809            return Ok((16, self.layout(Ty::new_empty_tuple(self.interner()))?, Some((0, 16, i))));
1810        }
1811        let layout = self.layout_adt(adt, subst)?;
1812        Ok(match &layout.variants {
1813            Variants::Single { .. } | Variants::Empty => (layout.size.bytes_usize(), layout, None),
1814            Variants::Multiple { variants, tag, tag_encoding, .. } => {
1815                let enum_variant_id = match it {
1816                    VariantId::EnumVariantId(it) => it,
1817                    _ => not_supported!("multi variant layout for non-enums"),
1818                };
1819                let mut discriminant = self.const_eval_discriminant(enum_variant_id)?;
1820                let lookup = enum_variant_id.lookup(self.db);
1821                let rustc_enum_variant_idx = RustcEnumVariantIdx(lookup.index as usize);
1822                let variant_layout = variants[rustc_enum_variant_idx].clone();
1823                let have_tag = match tag_encoding {
1824                    TagEncoding::Direct => true,
1825                    TagEncoding::Niche { untagged_variant, niche_variants: _, niche_start } => {
1826                        if *untagged_variant == rustc_enum_variant_idx {
1827                            false
1828                        } else {
1829                            discriminant = (variants
1830                                .iter_enumerated()
1831                                .filter(|(it, _)| it != untagged_variant)
1832                                .position(|(it, _)| it == rustc_enum_variant_idx)
1833                                .unwrap() as i128)
1834                                .wrapping_add(*niche_start as i128);
1835                            true
1836                        }
1837                    }
1838                };
1839                (
1840                    layout.size.bytes_usize(),
1841                    Arc::new(variant_layout),
1842                    if have_tag {
1843                        Some((
1844                            layout.fields.offset(0).bytes_usize(),
1845                            tag.size(&*self.target_data_layout).bytes_usize(),
1846                            discriminant,
1847                        ))
1848                    } else {
1849                        None
1850                    },
1851                )
1852            }
1853        })
1854    }

1911    fn allocate_const_in_heap(
1912        &mut self,
1913        locals: &Locals<'db>,
1914        konst: Const<'db>,
1915    ) -> Result<'db, Interval> {
1916        let result_owner;
1917        let value = match konst.kind() {
1918            ConstKind::Value(value) => value,
1919            ConstKind::Unevaluated(UnevaluatedConst { def: const_id, args: subst }) => 'b: {
1920                let mut const_id = match const_id {
1921                    SolverDefId::ConstId(it) => GeneralConstId::from(it),
1922                    SolverDefId::StaticId(it) => it.into(),
1923                    _ => unreachable!("unevaluated consts should be consts or statics"),
1924                };
1925                let mut subst = subst;
1926                if let hir_def::GeneralConstId::ConstId(c) = const_id {
1927                    let (c, s) = lookup_impl_const(&self.infcx, self.trait_env.clone(), c, subst);
1928                    const_id = hir_def::GeneralConstId::ConstId(c);
1929                    subst = s;
1930                }
1931                result_owner = self
1932                    .db
1933                    .const_eval(const_id, subst, Some(self.trait_env.clone()))
1934                    .map_err(|e| {
1935                        let name = const_id.name(self.db);
1936                        MirEvalError::ConstEvalError(name, Box::new(e))
1937                    })?;
1938                if let ConstKind::Value(value) = result_owner.kind() {
1939                    break 'b value;
1940                }
1941                not_supported!("unevaluatable constant");
1942            }
1943            _ => not_supported!("evaluating unknown const"),
1944        };
1945        let ValueConst { ty, value } = value;
1946        let ConstBytes { memory: v, memory_map } = value.inner();
1947        let patch_map = memory_map.transform_addresses(|b, align| {
1948            let addr = self.heap_allocate(b.len(), align)?;
1949            self.write_memory(addr, b)?;
1950            Ok(addr.to_usize())
1951        })?;
1952        let (size, align) = self.size_align_of(ty, locals)?.unwrap_or((v.len(), 1));
1953        let v: Cow<'_, [u8]> = if size != v.len() {
1954            // Handle self enum
1955            if size == 16 && v.len() < 16 {
1956                Cow::Owned(pad16(v, false).to_vec())
1957            } else if size < 16 && v.len() == 16 {
1958                Cow::Borrowed(&v[0..size])
1959            } else {
1960                return Err(MirEvalError::InvalidConst(konst));
1961            }
1962        } else {
1963            Cow::Borrowed(v)
1964        };
1965        let addr = self.heap_allocate(size, align)?;
1966        self.write_memory(addr, &v)?;
1967        self.patch_addresses(
1968            &patch_map,
1969            |bytes| match memory_map {
1970                MemoryMap::Empty | MemoryMap::Simple(_) => {
1971                    Err(MirEvalError::InvalidVTableId(from_bytes!(usize, bytes)))
1972                }
1973                MemoryMap::Complex(cm) => cm.vtable.ty_of_bytes(bytes),
1974            },
1975            addr,
1976            ty,
1977            locals,
1978        )?;
1979        Ok(Interval::new(addr, size))
1980    }

2041        if r.size == 0 {
2042            return Ok(());
2043        }
2044

2046
2047        match (addr, r.addr) {

2050                    return Err(oob());
2051                }
2052                self.stack.copy_within(src..src + r.size, dst)
2053            }
2054            (Heap(dst), Heap(src)) => {
2055                if self.stack.len() < src + r.size || self.stack.len() < dst + r.size {
2056                    return Err(oob());
2057                }

2059            }
2060            (Stack(dst), Heap(src)) => {
2061                self.stack
2062                    .get_mut(dst..dst + r.size)
2063                    .ok_or_else(oob)?
2064                    .copy_from_slice(self.heap.get(src..src + r.size).ok_or_else(oob)?);

2066            (Heap(dst), Stack(src)) => {
2067                self.heap
2068                    .get_mut(dst..dst + r.size)
2069                    .ok_or_else(oob)?

2072            _ => {
2073                return Err(MirEvalError::UndefinedBehavior(format!(
2074                    "invalid memory write at address {addr:?}"
2075                )));
2076            }
2077        }

2139        if !align.is_power_of_two() || align > 10000 {
2140            return Err(MirEvalError::UndefinedBehavior(format!("Alignment {align} is invalid")));

2147        }

2149        self.heap.extend(std::iter::repeat_n(0, size));
2150        Ok(Address::Heap(pos))
2151    }

2311                                    stack_depth_limit - 1,
2312                                )?;
2313                            }
2314                        }
2315                    }
2316                    AdtId::UnionId(_) => (),
2317                },
2318                TyKind::Alias(AliasTyKind::Projection, _) => {
2319                    let mut ocx = ObligationCtxt::new(&this.infcx);
2320                    let ty = ocx
2321                        .structurally_normalize_ty(
2322                            &ObligationCause::dummy(),
2323                            this.trait_env.env,
2324                            ty,
2325                        )
2326                        .map_err(|_| MirEvalError::NotSupported("couldn't normalize".to_owned()))?;
2327

2329                }
2330                _ => (),
2331            }
2332            Ok(())
2333        }
2334        let mut mm = ComplexMemoryMap::default();

2339    fn patch_addresses(
2340        &mut self,
2341        patch_map: &FxHashMap<usize, usize>,
2342        ty_of_bytes: impl Fn(&[u8]) -> Result<'db, Ty<'db>> + Copy,
2343        addr: Address,
2344        ty: Ty<'db>,
2345        locals: &Locals<'db>,
2346    ) -> Result<'db, ()> {
2347        // FIXME: support indirect references
2348        let layout = self.layout(ty)?;
2349        let my_size = self.size_of_sized(ty, locals, "value to patch address")?;
2350        use rustc_type_ir::TyKind;
2351        match ty.kind() {
2352            TyKind::Ref(_, t, _) => {
2353                let size = self.size_align_of(t, locals)?;
2354                match size {
2355                    Some(_) => {
2356                        let current = from_bytes!(usize, self.read_memory(addr, my_size)?);
2357                        if let Some(it) = patch_map.get(&current) {
2358                            self.write_memory(addr, &it.to_le_bytes())?;
2359                        }
2360                    }
2361                    None => {
2362                        let current = from_bytes!(usize, self.read_memory(addr, my_size / 2)?);
2363                        if let Some(it) = patch_map.get(&current) {
2364                            self.write_memory(addr, &it.to_le_bytes())?;
2365                        }
2366                    }
2367                }
2368            }
2369            TyKind::FnPtr(_, _) => {
2370                let ty = ty_of_bytes(self.read_memory(addr, my_size)?)?;
2371                let new_id = self.vtable_map.id(ty);
2372                self.write_memory(addr, &new_id.to_le_bytes())?;
2373            }
2374            TyKind::Adt(id, args) => match id.def_id().0 {
2375                AdtId::StructId(s) => {
2376                    for (i, (_, ty)) in self.db.field_types(s.into()).iter().enumerate() {
2377                        let offset = layout.fields.offset(i).bytes_usize();
2378                        let ty = ty.instantiate(self.interner(), args);
2379                        self.patch_addresses(
2380                            patch_map,
2381                            ty_of_bytes,
2382                            addr.offset(offset),
2383                            ty,
2384                            locals,
2385                        )?;
2386                    }
2387                }
2388                AdtId::UnionId(_) => (),
2389                AdtId::EnumId(e) => {
2390                    if let Some((ev, layout)) = detect_variant_from_bytes(
2391                        &layout,
2392                        self.db,
2393                        &self.target_data_layout,
2394                        self.read_memory(addr, layout.size.bytes_usize())?,
2395                        e,
2396                    ) {
2397                        for (i, (_, ty)) in self.db.field_types(ev.into()).iter().enumerate() {
2398                            let offset = layout.fields.offset(i).bytes_usize();
2399                            let ty = ty.instantiate(self.interner(), args);
2400                            self.patch_addresses(
2401                                patch_map,
2402                                ty_of_bytes,
2403                                addr.offset(offset),
2404                                ty,
2405                                locals,
2406                            )?;
2407                        }
2408                    }
2409                }
2410            },
2411            TyKind::Tuple(tys) => {
2412                for (id, ty) in tys.iter().enumerate() {
2413                    let offset = layout.fields.offset(id).bytes_usize();
2414                    self.patch_addresses(patch_map, ty_of_bytes, addr.offset(offset), ty, locals)?;
2415                }
2416            }
2417            TyKind::Array(inner, len) => {
2418                let len = match consteval::try_const_usize(self.db, len) {
2419                    Some(it) => it as usize,
2420                    None => not_supported!("non evaluatable array len in patching addresses"),
2421                };
2422                let size = self.size_of_sized(inner, locals, "inner of array")?;
2423                for i in 0..len {
2424                    self.patch_addresses(
2425                        patch_map,
2426                        ty_of_bytes,
2427                        addr.offset(i * size),
2428                        inner,
2429                        locals,
2430                    )?;
2431                }
2432            }
2433            TyKind::Bool
2434            | TyKind::Char
2435            | TyKind::Int(_)
2436            | TyKind::Uint(_)
2437            | TyKind::Float(_)
2438            | TyKind::Slice(_)
2439            | TyKind::RawPtr(_, _)
2440            | TyKind::FnDef(_, _)
2441            | TyKind::Str
2442            | TyKind::Never
2443            | TyKind::Closure(_, _)
2444            | TyKind::Coroutine(_, _)
2445            | TyKind::CoroutineWitness(_, _)
2446            | TyKind::Foreign(_)
2447            | TyKind::Error(_)
2448            | TyKind::Placeholder(_)
2449            | TyKind::Dynamic(_, _)
2450            | TyKind::Alias(_, _)
2451            | TyKind::Bound(_, _)
2452            | TyKind::Infer(_)
2453            | TyKind::Pat(_, _)
2454            | TyKind::Param(_)
2455            | TyKind::UnsafeBinder(_)
2456            | TyKind::CoroutineClosure(_, _) => (),
2457        }
2458        Ok(())
2459    }

2491        &mut self,
2492        closure: InternedClosureId,
2493        closure_data: Interval,
2494        generic_args: GenericArgs<'db>,
2495        destination: Interval,
2496        args: &[IntervalAndTy<'db>],

2501            .db

2586        &self,
2587        def: FunctionId,

2590        span: MirSpan,
2591    ) -> Result<'db, MirOrDynIndex<'db>> {
2592        let pair = (def, generic_args);
2593        if let Some(r) = self.mir_or_dyn_index_cache.borrow().get(&pair) {
2594            return Ok(r.clone());
2595        }
2596        let (def, generic_args) = pair;

2598            is_dyn_method(self.interner(), self.trait_env.clone(), def, generic_args)
2599        {
2600            MirOrDynIndex::Dyn(self_ty_idx)
2601        } else {

2620    fn exec_fn_with_args(
2621        &mut self,
2622        mut def: FunctionId,
2623        args: &[IntervalAndTy<'db>],
2624        generic_args: GenericArgs<'db>,
2625        locals: &Locals<'db>,
2626        destination: Interval,
2627        target_bb: Option<BasicBlockId<'db>>,
2628        span: MirSpan,
2629    ) -> Result<'db, Option<StackFrame<'db>>> {
2630        if self.detect_and_exec_special_function(
2631            def,
2632            args,
2633            generic_args,
2634            locals,
2635            destination,
2636            span,
2637        )? {
2638            return Ok(None);
2639        }
2640        if let Some(redirect_def) = self.detect_and_redirect_special_function(def)? {
2641            def = redirect_def;
2642        }
2643        let arg_bytes = args.iter().map(|it| IntervalOrOwned::Borrowed(it.interval));
2644        match self.get_mir_or_dyn_index(def, generic_args, locals, span)? {
2645            MirOrDynIndex::Dyn(self_ty_idx) => {
2646                // In the layout of current possible receiver, which at the moment of writing this code is one of
2647                // `&T`, `&mut T`, `Box<T>`, `Rc<T>`, `Arc<T>`, and `Pin<P>` where `P` is one of possible receivers,
2648                // the vtable is exactly in the `[ptr_size..2*ptr_size]` bytes. So we can use it without branching on
2649                // the type.
2650                let first_arg = arg_bytes.clone().next().unwrap();
2651                let first_arg = first_arg.get(self)?;
2652                let ty = self
2653                    .vtable_map
2654                    .ty_of_bytes(&first_arg[self.ptr_size()..self.ptr_size() * 2])?;
2655                let mut args_for_target = args.to_vec();
2656                args_for_target[0] = IntervalAndTy {
2657                    interval: args_for_target[0].interval.slice(0..self.ptr_size()),
2658                    ty,
2659                };
2660                let generics_for_target = GenericArgs::new_from_iter(
2661                    self.interner(),
2662                    generic_args
2663                        .iter()
2664                        .enumerate()
2665                        .map(|(i, it)| if i == self_ty_idx { ty.into() } else { it }),
2666                );
2667                self.exec_fn_with_args(
2668                    def,
2669                    &args_for_target,
2670                    generics_for_target,
2671                    locals,
2672                    destination,
2673                    target_bb,
2674                    span,
2675                )
2676            }
2677            MirOrDynIndex::Mir(body) => self.exec_looked_up_function(
2678                body,
2679                locals,
2680                def,
2681                arg_bytes,
2682                span,
2683                destination,
2684                target_bb,
2685            ),
2686        }
2687    }

2690        &mut self,
2691        mir_body: Arc<MirBody<'db>>,
2692        locals: &Locals<'db>,
2693        def: FunctionId,
2694        arg_bytes: impl Iterator<Item = IntervalOrOwned>,
2695        span: MirSpan,
2696        destination: Interval,
2697        target_bb: Option<BasicBlockId<'db>>,

2700            let (mut locals, prev_stack_ptr) =
2701                self.create_locals_for_body(&mir_body, Some(destination))?;
2702            self.fill_locals_for_body(&mir_body, &mut locals, arg_bytes.into_iter())?;

2707                MirEvalError::InFunction(

2709                    vec![(Either::Left(def), span, locals.body.owner)],
2710                )
2711            })?;

2713            None
2714        })

2717    fn exec_fn_trait(
2718        &mut self,
2719        def: FunctionId,
2720        args: &[IntervalAndTy<'db>],
2721        generic_args: GenericArgs<'db>,
2722        locals: &Locals<'db>,
2723        destination: Interval,
2724        target_bb: Option<BasicBlockId<'db>>,
2725        span: MirSpan,
2726    ) -> Result<'db, Option<StackFrame<'db>>> {
2727        let func = args
2728            .first()
2729            .ok_or_else(|| MirEvalError::InternalError("fn trait with no arg".into()))?;
2730        let mut func_ty = func.ty;
2731        let mut func_data = func.interval;
2732        while let TyKind::Ref(_, z, _) = func_ty.kind() {
2733            func_ty = z;
2734            if matches!(func_ty.kind(), TyKind::Dynamic(..)) {
2735                let id =
2736                    from_bytes!(usize, &func_data.get(self)?[self.ptr_size()..self.ptr_size() * 2]);
2737                func_data = func_data.slice(0..self.ptr_size());
2738                func_ty = self.vtable_map.ty(id)?;
2739            }
2740            let size = self.size_of_sized(func_ty, locals, "self type of fn trait")?;
2741            func_data = Interval { addr: Address::from_bytes(func_data.get(self)?)?, size };
2742        }
2743        match func_ty.kind() {
2744            TyKind::FnDef(def, subst) => {
2745                self.exec_fn_def(def.0, subst, destination, &args[1..], locals, target_bb, span)
2746            }
2747            TyKind::FnPtr(..) => {
2748                self.exec_fn_pointer(func_data, destination, &args[1..], locals, target_bb, span)
2749            }
2750            TyKind::Closure(closure, subst) => self.exec_closure(
2751                closure.0,
2752                func_data,
2753                subst.split_closure_args_untupled().parent_args,
2754                destination,
2755                &args[1..],
2756                locals,
2757                span,
2758            ),
2759            _ => {
2760                // try to execute the manual impl of `FnTrait` for structs (nightly feature used in std)
2761                let arg0 = func;
2762                let args = &args[1..];
2763                let arg1 = {
2764                    let ty = Ty::new_tup_from_iter(self.interner(), args.iter().map(|it| it.ty));
2765                    let layout = self.layout(ty)?;
2766                    let result = self.construct_with_layout(
2767                        layout.size.bytes_usize(),
2768                        &layout,
2769                        None,
2770                        args.iter().map(|it| IntervalOrOwned::Borrowed(it.interval)),
2771                    )?;
2772                    // FIXME: there is some leak here
2773                    let size = layout.size.bytes_usize();
2774                    let addr = self.heap_allocate(size, layout.align.bytes() as usize)?;
2775                    self.write_memory(addr, &result)?;
2776                    IntervalAndTy { interval: Interval { addr, size }, ty }
2777                };
2778                self.exec_fn_with_args(
2779                    def,
2780                    &[arg0.clone(), arg1],
2781                    generic_args,
2782                    locals,
2783                    destination,
2784                    target_bb,
2785                    span,
2786                )
2787            }
2788        }
2789    }

2791    fn eval_static(&mut self, st: StaticId, locals: &Locals<'db>) -> Result<'db, Address> {
2792        if let Some(o) = self.static_locations.get(&st) {
2793            return Ok(*o);
2794        };
2795        let static_data = self.db.static_signature(st);
2796        let result = if !static_data.flags.contains(StaticFlags::EXTERN) {
2797            let konst = self.db.const_eval_static(st).map_err(|e| {
2798                MirEvalError::ConstEvalError(static_data.name.as_str().to_owned(), Box::new(e))
2799            })?;
2800            self.allocate_const_in_heap(locals, konst)?
2801        } else {
2802            let ty = self.db.infer(st.into())[self.db.body(st.into()).body_expr];
2803            let Some((size, align)) = self.size_align_of(ty, locals)? else {
2804                not_supported!("unsized extern static");
2805            };
2806            let addr = self.heap_allocate(size, align)?;
2807            Interval::new(addr, size)
2808        };
2809        let addr = self.heap_allocate(self.ptr_size(), self.ptr_size())?;
2810        self.write_memory(addr, &result.addr.to_bytes())?;
2811        self.static_locations.insert(st, addr);
2812        Ok(addr)
2813    }

2854    fn run_drop_glue_deep(
2855        &mut self,
2856        ty: Ty<'db>,
2857        locals: &Locals<'db>,
2858        addr: Address,
2859        _metadata: &[u8],
2860        span: MirSpan,
2861    ) -> Result<'db, ()> {
2862        let Some(drop_fn) = (|| {
2863            let drop_trait = LangItem::Drop.resolve_trait(self.db, self.crate_id)?;
2864            drop_trait.trait_items(self.db).method_by_name(&Name::new_symbol_root(sym::drop))
2865        })() else {
2866            // in some tests we don't have drop trait in minicore, and
2867            // we can ignore drop in them.
2868            return Ok(());
2869        };
2870
2871        let generic_args = GenericArgs::new_from_iter(self.interner(), [ty.into()]);
2872        if let Ok(MirOrDynIndex::Mir(body)) =
2873            self.get_mir_or_dyn_index(drop_fn, generic_args, locals, span)
2874        {
2875            self.exec_looked_up_function(
2876                body,
2877                locals,
2878                drop_fn,
2879                iter::once(IntervalOrOwned::Owned(addr.to_bytes().to_vec())),
2880                span,
2881                Interval { addr: Address::Invalid(0), size: 0 },
2882                None,
2883            )?;
2884        }
2885        match ty.kind() {
2886            TyKind::Adt(adt_def, subst) => {
2887                let id = adt_def.def_id().0;
2888                match id {
2889                    AdtId::StructId(s) => {
2890                        let data = self.db.struct_signature(s);
2891                        if data.flags.contains(StructFlags::IS_MANUALLY_DROP) {
2892                            return Ok(());
2893                        }
2894                        let layout = self.layout_adt(id, subst)?;
2895                        let variant_fields = s.fields(self.db);
