28fn coerce_pointee_new_type_param(trait_id: TraitId) -> TypeParamId {
29    // HACK: Fake the param.
30    // We cannot use a dummy param here, because it can leak into the IDE layer and that'll cause panics
31    // when e.g. trying to display it. So we use an existing param.
32    TypeParamId::from_unchecked(TypeOrConstParamId {
33        parent: trait_id.into(),
34        local_id: la_arena::Idx::from_raw(la_arena::RawIdx::from_u32(1)),
35    })
36}

38fn trait_args(trait_: BuiltinDeriveImplTrait, self_ty: Ty<'_>) -> GenericArgs<'_> {
39    match trait_ {
40        BuiltinDeriveImplTrait::Copy
41        | BuiltinDeriveImplTrait::Clone
42        | BuiltinDeriveImplTrait::Default
43        | BuiltinDeriveImplTrait::Debug
44        | BuiltinDeriveImplTrait::Hash
45        | BuiltinDeriveImplTrait::Eq
46        | BuiltinDeriveImplTrait::Ord => GenericArgs::new_from_slice(&[self_ty.into()]),
47        BuiltinDeriveImplTrait::PartialOrd | BuiltinDeriveImplTrait::PartialEq => {
48            GenericArgs::new_from_slice(&[self_ty.into(), self_ty.into()])
49        }
50        BuiltinDeriveImplTrait::CoerceUnsized | BuiltinDeriveImplTrait::DispatchFromDyn => {
51            panic!("`CoerceUnsized` and `DispatchFromDyn` have special generics")
52        }
53    }
54}

56pub(crate) fn generics_of<'db>(interner: DbInterner<'db>, id: BuiltinDeriveImplId) -> Generics {
57    let db = interner.db;
58    let loc = id.loc(db);
59    match loc.trait_ {
60        BuiltinDeriveImplTrait::Copy
61        | BuiltinDeriveImplTrait::Clone
62        | BuiltinDeriveImplTrait::Default
63        | BuiltinDeriveImplTrait::Debug
64        | BuiltinDeriveImplTrait::Hash
65        | BuiltinDeriveImplTrait::Ord
66        | BuiltinDeriveImplTrait::PartialOrd
67        | BuiltinDeriveImplTrait::Eq
68        | BuiltinDeriveImplTrait::PartialEq => interner.generics_of(loc.adt.into()),
69        BuiltinDeriveImplTrait::CoerceUnsized | BuiltinDeriveImplTrait::DispatchFromDyn => {
70            let mut generics = interner.generics_of(loc.adt.into());
71            let trait_id = loc
72                .trait_
73                .get_id(interner.lang_items())
74                .expect("we don't pass the impl to the solver if we can't resolve the trait");
75            generics.push_param(coerce_pointee_new_type_param(trait_id).into());
76            generics
77        }
78    }
79}

81pub fn generic_params_count(db: &dyn HirDatabase, id: BuiltinDeriveImplId) -> usize {
82    let loc = id.loc(db);
83    let adt_params = GenericParams::new(db, loc.adt.into());
84    let extra_params_count = match loc.trait_ {
85        BuiltinDeriveImplTrait::Copy
86        | BuiltinDeriveImplTrait::Clone
87        | BuiltinDeriveImplTrait::Default
88        | BuiltinDeriveImplTrait::Debug
89        | BuiltinDeriveImplTrait::Hash
90        | BuiltinDeriveImplTrait::Ord
91        | BuiltinDeriveImplTrait::PartialOrd
92        | BuiltinDeriveImplTrait::Eq
93        | BuiltinDeriveImplTrait::PartialEq => 0,
94        BuiltinDeriveImplTrait::CoerceUnsized | BuiltinDeriveImplTrait::DispatchFromDyn => 1,
95    };
96    adt_params.len() + extra_params_count
97}

99pub fn impl_trait<'db>(
100    interner: DbInterner<'db>,
101    id: BuiltinDeriveImplId,
102) -> EarlyBinder<'db, TraitRef<'db>> {
103    let db = interner.db;
104    let loc = id.loc(db);
105    let trait_id = loc
106        .trait_
107        .get_id(interner.lang_items())
108        .expect("we don't pass the impl to the solver if we can't resolve the trait");
109    match loc.trait_ {
110        BuiltinDeriveImplTrait::Copy
111        | BuiltinDeriveImplTrait::Clone
112        | BuiltinDeriveImplTrait::Default
113        | BuiltinDeriveImplTrait::Debug
114        | BuiltinDeriveImplTrait::Hash
115        | BuiltinDeriveImplTrait::Ord
116        | BuiltinDeriveImplTrait::Eq
117        | BuiltinDeriveImplTrait::PartialOrd
118        | BuiltinDeriveImplTrait::PartialEq => {
119            let self_ty = Ty::new_adt(
120                interner,
121                loc.adt,
122                GenericArgs::identity_for_item(interner, loc.adt.into()),
123            );
124            EarlyBinder::bind(TraitRef::new_from_args(
125                interner,
126                trait_id.into(),
127                trait_args(loc.trait_, self_ty),
128            ))
129        }
130        BuiltinDeriveImplTrait::CoerceUnsized | BuiltinDeriveImplTrait::DispatchFromDyn => {
131            let generic_params = GenericParams::new(db, loc.adt.into());
132            let interner = DbInterner::new_no_crate(db);
133            let args = GenericArgs::identity_for_item(interner, loc.adt.into());
134            let self_ty = Ty::new_adt(interner, loc.adt, args);
135            let Some((pointee_param_idx, _, new_param_ty)) =
136                coerce_pointee_params(interner, loc, &generic_params, trait_id)
137            else {
138                // Malformed derive.
139                return EarlyBinder::bind(TraitRef::new(
140                    interner,
141                    trait_id.into(),
142                    [self_ty, self_ty],
143                ));
144            };
145            let changed_args = replace_pointee(interner, pointee_param_idx, new_param_ty, args);
146            let changed_self_ty = Ty::new_adt(interner, loc.adt, changed_args);
147            EarlyBinder::bind(TraitRef::new(interner, trait_id.into(), [self_ty, changed_self_ty]))
148        }
149    }
150}

152#[salsa::tracked(returns(ref))]
153pub fn predicates<'db>(db: &'db dyn HirDatabase, impl_: BuiltinDeriveImplId) -> GenericPredicates {
154    let loc = impl_.loc(db);
155    let generic_params = GenericParams::new(db, loc.adt.into());
156    let interner = DbInterner::new_with(db, loc.module(db).krate(db));
157    let adt_predicates = GenericPredicates::query(db, loc.adt.into());
158    let trait_id = loc
159        .trait_
160        .get_id(interner.lang_items())
161        .expect("we don't pass the impl to the solver if we can't resolve the trait");
162    match loc.trait_ {
163        BuiltinDeriveImplTrait::Copy
164        | BuiltinDeriveImplTrait::Clone
165        | BuiltinDeriveImplTrait::Debug
166        | BuiltinDeriveImplTrait::Hash
167        | BuiltinDeriveImplTrait::Ord
168        | BuiltinDeriveImplTrait::PartialOrd
169        | BuiltinDeriveImplTrait::Eq
170        | BuiltinDeriveImplTrait::PartialEq => {
171            simple_trait_predicates(interner, loc, &generic_params, adt_predicates, trait_id)
172        }
173        BuiltinDeriveImplTrait::Default => {
174            if matches!(loc.adt, AdtId::EnumId(_)) {
175                // Enums don't have extra bounds.
176                GenericPredicates::from_explicit_own_predicates(StoredEarlyBinder::bind(
177                    Clauses::new_from_slice(adt_predicates.explicit_predicates().skip_binder())
178                        .store(),
179                ))
180            } else {
181                simple_trait_predicates(interner, loc, &generic_params, adt_predicates, trait_id)
182            }
183        }
184        BuiltinDeriveImplTrait::CoerceUnsized | BuiltinDeriveImplTrait::DispatchFromDyn => {
185            let Some((pointee_param_idx, pointee_param_id, new_param_ty)) =
186                coerce_pointee_params(interner, loc, &generic_params, trait_id)
187            else {
188                // Malformed derive.
189                return GenericPredicates::from_explicit_own_predicates(StoredEarlyBinder::bind(
190                    Clauses::default().store(),
191                ));
192            };
193            let duplicated_bounds =
194                adt_predicates.explicit_predicates().iter_identity_copied().filter_map(|pred| {
195                    let mentions_pointee =
196                        pred.visit_with(&mut MentionsPointee { pointee_param_idx }).is_break();
197                    if !mentions_pointee {
198                        return None;
199                    }
200                    let transformed =
201                        replace_pointee(interner, pointee_param_idx, new_param_ty, pred);
202                    Some(transformed)
203                });
204            let unsize_trait = interner.lang_items().Unsize;
205            let unsize_bound = unsize_trait.map(|unsize_trait| {
206                let pointee_param_ty = Ty::new_param(interner, pointee_param_id, pointee_param_idx);
207                TraitRef::new(interner, unsize_trait.into(), [pointee_param_ty, new_param_ty])
208                    .upcast(interner)
209            });
210            GenericPredicates::from_explicit_own_predicates(StoredEarlyBinder::bind(
211                Clauses::new_from_iter(
212                    interner,
213                    adt_predicates
214                        .explicit_predicates()
215                        .iter_identity_copied()
216                        .chain(duplicated_bounds)
217                        .chain(unsize_bound),
218                )
219                .store(),
220            ))
221        }
222    }
223}

225/// Not cached in a query, currently used in `hir` only. If you need this in `hir-ty` consider introducing a query.
226pub fn param_env<'db>(interner: DbInterner<'db>, id: BuiltinDeriveImplId) -> ParamEnv<'db> {
227    let predicates = predicates(interner.db, id);
228    crate::lower::param_env_from_predicates(interner, predicates)

231struct MentionsPointee {
232    pointee_param_idx: u32,
233}
234
235impl<'db> TypeVisitor<DbInterner<'db>> for MentionsPointee {
236    type Result = ControlFlow<()>;
237
238    fn visit_ty(&mut self, t: Ty<'db>) -> Self::Result {
239        if let TyKind::Param(param) = t.kind()
240            && param.index == self.pointee_param_idx
241        {
242            ControlFlow::Break(())
243        } else {
244            t.super_visit_with(self)
245        }
246    }
247}

249fn replace_pointee<'db, T: TypeFoldable<DbInterner<'db>>>(
250    interner: DbInterner<'db>,
251    pointee_param_idx: u32,
252    new_param_ty: Ty<'db>,
253    t: T,
254) -> T {
255    fold_tys(interner, t, |ty| match ty.kind() {
256        TyKind::Param(param) if param.index == pointee_param_idx => new_param_ty,
257        _ => ty,
258    })
259}

261fn simple_trait_predicates<'db>(
262    interner: DbInterner<'db>,
263    loc: &BuiltinDeriveImplLoc,
264    generic_params: &GenericParams,
265    adt_predicates: &GenericPredicates,
266    trait_id: TraitId,
267) -> GenericPredicates {
268    let extra_predicates = generic_params
269        .iter_type_or_consts()
270        .filter(|(_, data)| matches!(data, TypeOrConstParamData::TypeParamData(_)))
271        .map(|(param_idx, _)| {
272            let param_id = TypeParamId::from_unchecked(TypeOrConstParamId {
273                parent: loc.adt.into(),
274                local_id: param_idx,
275            });
276            let param_idx =
277                param_idx.into_raw().into_u32() + (generic_params.len_lifetimes() as u32);
278            let param_ty = Ty::new_param(interner, param_id, param_idx);
279            let trait_args = trait_args(loc.trait_, param_ty);
280            let trait_ref = TraitRef::new_from_args(interner, trait_id.into(), trait_args);
281            trait_ref.upcast(interner)
282        });
283    let mut assoc_type_bounds = Vec::new();
284    match loc.adt {
285        AdtId::StructId(id) => extend_assoc_type_bounds(
286            interner,
287            &mut assoc_type_bounds,
288            interner.db.field_types(id.into()),
289            trait_id,
290            loc.trait_,
291        ),
292        AdtId::UnionId(id) => extend_assoc_type_bounds(
293            interner,
294            &mut assoc_type_bounds,
295            interner.db.field_types(id.into()),
296            trait_id,
297            loc.trait_,
298        ),
299        AdtId::EnumId(id) => {
300            for &(variant_id, _, _) in &id.enum_variants(interner.db).variants {
301                extend_assoc_type_bounds(
302                    interner,
303                    &mut assoc_type_bounds,
304                    interner.db.field_types(variant_id.into()),
305                    trait_id,
306                    loc.trait_,
307                )
308            }
309        }
310    }
311    GenericPredicates::from_explicit_own_predicates(StoredEarlyBinder::bind(
312        Clauses::new_from_iter(
313            interner,
314            adt_predicates
315                .explicit_predicates()
316                .iter_identity_copied()
317                .chain(extra_predicates)
318                .chain(assoc_type_bounds),
319        )
320        .store(),
321    ))
322}

324fn extend_assoc_type_bounds<'db>(
325    interner: DbInterner<'db>,
326    assoc_type_bounds: &mut Vec<Clause<'db>>,
327    fields: &ArenaMap<LocalFieldId, StoredEarlyBinder<StoredTy>>,
328    trait_id: TraitId,
329    trait_: BuiltinDeriveImplTrait,
330) {
331    struct ProjectionFinder<'a, 'db> {
332        interner: DbInterner<'db>,
333        assoc_type_bounds: &'a mut Vec<Clause<'db>>,
334        trait_id: TraitId,
335        trait_: BuiltinDeriveImplTrait,
336    }
337
338    impl<'db> TypeVisitor<DbInterner<'db>> for ProjectionFinder<'_, 'db> {
339        type Result = ();
340
341        fn visit_ty(&mut self, t: Ty<'db>) -> Self::Result {
342            if let TyKind::Alias(AliasTyKind::Projection, _) = t.kind() {
343                self.assoc_type_bounds.push(
344                    TraitRef::new_from_args(
345                        self.interner,
346                        self.trait_id.into(),
347                        trait_args(self.trait_, t),
348                    )
349                    .upcast(self.interner),
350                );
351            }
352
353            t.super_visit_with(self)
354        }
355    }
356
357    let mut visitor = ProjectionFinder { interner, assoc_type_bounds, trait_id, trait_ };
358    for (_, field) in fields.iter() {
359        field.get().instantiate_identity().visit_with(&mut visitor);
360    }
361}

363fn coerce_pointee_params<'db>(
364    interner: DbInterner<'db>,
365    loc: &BuiltinDeriveImplLoc,
366    generic_params: &GenericParams,
367    trait_id: TraitId,
368) -> Option<(u32, TypeParamId, Ty<'db>)> {
369    let pointee_param = {
370        if let Ok((pointee_param, _)) = generic_params
371            .iter_type_or_consts()
372            .filter(|param| matches!(param.1, TypeOrConstParamData::TypeParamData(_)))
373            .exactly_one()
374        {
375            pointee_param
376        } else {
377            let (_, generic_param_attrs) =
378                AttrFlags::query_generic_params(interner.db, loc.adt.into());
379            generic_param_attrs
380                .iter()
381                .find(|param| param.1.contains(AttrFlags::IS_POINTEE))
382                .map(|(param, _)| param)
383                .or_else(|| {
384                    generic_params
385                        .iter_type_or_consts()
386                        .find(|param| matches!(param.1, TypeOrConstParamData::TypeParamData(_)))
387                        .map(|(idx, _)| idx)
388                })?
389        }
390    };
391    let pointee_param_id = TypeParamId::from_unchecked(TypeOrConstParamId {
392        parent: loc.adt.into(),
393        local_id: pointee_param,
394    });
395    let pointee_param_idx =
396        pointee_param.into_raw().into_u32() + (generic_params.len_lifetimes() as u32);
397    let new_param_idx = generic_params.len() as u32;
398    let new_param_id = coerce_pointee_new_type_param(trait_id);
399    let new_param_ty = Ty::new_param(interner, new_param_id, new_param_idx);
400    Some((pointee_param_idx, pointee_param_id, new_param_ty))
401}
