66/// This should be used in `hir` only.
67pub fn structurally_normalize_ty<'db>(
68    infcx: &InferCtxt<'db>,
69    ty: Ty<'db>,
70    env: ParamEnv<'db>,
71) -> Ty<'db> {
72    let TyKind::Alias(..) = ty.kind() else { return ty };
73    let mut ocx = ObligationCtxt::new(infcx);
74    let ty = ocx.structurally_normalize_ty(&ObligationCause::dummy(), env, ty).unwrap_or(ty);
75    ty.replace_infer_with_error(infcx.interner)
76}

99pub fn next_trait_solve_canonical_in_ctxt<'db>(
100    infer_ctxt: &InferCtxt<'db>,
101    goal: Canonical<'db, Goal<'db, Predicate<'db>>>,
102) -> NextTraitSolveResult {
103    infer_ctxt.probe(|_| {
104        let context = <&SolverContext<'db>>::from(infer_ctxt);
105
106        tracing::info!(?goal);
107
108        let (goal, var_values) = context.instantiate_canonical(&goal);
109        tracing::info!(?var_values);
110
111        let res = context.evaluate_root_goal(goal, Span::dummy(), None);
112
113        let obligation = Obligation {
114            cause: ObligationCause::dummy(),
115            param_env: goal.param_env,
116            recursion_depth: 0,
117            predicate: goal.predicate,
118        };
119        infer_ctxt.inspect_evaluated_obligation(&obligation, &res, || {
120            Some(context.evaluate_root_goal_for_proof_tree(goal, Span::dummy()).1)
121        });
122
123        let res = res.map(|r| (r.has_changed, r.certainty));
124
125        tracing::debug!("solve_nextsolver({:?}) => {:?}", goal, res);
126
127        match res {
128            Err(_) => NextTraitSolveResult::NoSolution,
129            Ok((_, Certainty::Yes)) => NextTraitSolveResult::Certain,
130            Ok((_, Certainty::Maybe { .. })) => NextTraitSolveResult::Uncertain,
131        }
132    })
133}

135/// Solve a trait goal using next trait solver.
136pub fn next_trait_solve_in_ctxt<'db, 'a>(
137    infer_ctxt: &'a InferCtxt<'db>,
138    goal: Goal<'db, Predicate<'db>>,
139) -> Result<(HasChanged, Certainty), rustc_type_ir::solve::NoSolution> {
140    tracing::info!(?goal);
141
142    let context = <&SolverContext<'db>>::from(infer_ctxt);
143
144    let res = context.evaluate_root_goal(goal, Span::dummy(), None);
145
146    let obligation = Obligation {
147        cause: ObligationCause::dummy(),
148        param_env: goal.param_env,
149        recursion_depth: 0,
150        predicate: goal.predicate,
151    };
152    infer_ctxt.inspect_evaluated_obligation(&obligation, &res, || {
153        Some(context.evaluate_root_goal_for_proof_tree(goal, Span::dummy()).1)
154    });
155
156    let res = res.map(|r| (r.has_changed, r.certainty));
157
158    tracing::debug!("solve_nextsolver({:?}) => {:?}", goal, res);
159
160    res
161}

163#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, PartialOrd, Ord, salsa::Update)]
164pub enum FnTrait {
165    // Warning: Order is important. If something implements `x` it should also implement
166    // `y` if `y <= x`.
167    FnOnce,
168    FnMut,
169    Fn,
170
171    AsyncFnOnce,
172    AsyncFnMut,
173    AsyncFn,
174}
175
176impl FnTrait {
177    pub fn method_name(self) -> Name {
178        match self {
179            FnTrait::FnOnce => Name::new_symbol_root(sym::call_once),
180            FnTrait::FnMut => Name::new_symbol_root(sym::call_mut),
181            FnTrait::Fn => Name::new_symbol_root(sym::call),
182            FnTrait::AsyncFnOnce => Name::new_symbol_root(sym::async_call_once),
183            FnTrait::AsyncFnMut => Name::new_symbol_root(sym::async_call_mut),
184            FnTrait::AsyncFn => Name::new_symbol_root(sym::async_call),
185        }
186    }
187
188    pub fn get_id(self, lang_items: &LangItems) -> Option<TraitId> {
189        match self {
190            FnTrait::FnOnce => lang_items.FnOnce,
191            FnTrait::FnMut => lang_items.FnMut,
192            FnTrait::Fn => lang_items.Fn,
193            FnTrait::AsyncFnOnce => lang_items.AsyncFnOnce,
194            FnTrait::AsyncFnMut => lang_items.AsyncFnMut,
195            FnTrait::AsyncFn => lang_items.AsyncFn,
196        }
197    }
198}

222fn implements_trait_unique_impl<'db>(
223    db: &'db dyn HirDatabase,
224    env: ParamEnvAndCrate<'db>,
225    trait_: TraitId,
226    create_args: &mut dyn FnMut(&InferCtxt<'db>) -> GenericArgs<'db>,
227) -> bool {
228    let interner = DbInterner::new_with(db, env.krate);
229    // FIXME(next-solver): I believe this should be `PostAnalysis`.
230    let infcx = interner.infer_ctxt().build(TypingMode::non_body_analysis());
231
232    let args = create_args(&infcx);
233    let trait_ref = rustc_type_ir::TraitRef::new_from_args(interner, trait_.into(), args);
234    let goal = Goal::new(interner, env.param_env, trait_ref);
235
236    let result = crate::traits::next_trait_solve_in_ctxt(&infcx, goal);
237    matches!(result, Ok((_, Certainty::Yes)))
238}

240pub fn is_inherent_impl_coherent(db: &dyn HirDatabase, def_map: &DefMap, impl_id: ImplId) -> bool {
241    let self_ty = db.impl_self_ty(impl_id).instantiate_identity();
242    let self_ty = self_ty.kind();
243    let impl_allowed = match self_ty {
244        TyKind::Tuple(_)
245        | TyKind::FnDef(_, _)
246        | TyKind::Array(_, _)
247        | TyKind::Never
248        | TyKind::RawPtr(_, _)
249        | TyKind::Ref(_, _, _)
250        | TyKind::Slice(_)
251        | TyKind::Str
252        | TyKind::Bool
253        | TyKind::Char
254        | TyKind::Int(_)
255        | TyKind::Uint(_)
256        | TyKind::Float(_) => def_map.is_rustc_coherence_is_core(),
257
258        TyKind::Adt(adt_def, _) => adt_def.def_id().0.module(db).krate(db) == def_map.krate(),
259        TyKind::Dynamic(it, _) => it
260            .principal_def_id()
261            .is_some_and(|trait_id| trait_id.0.module(db).krate(db) == def_map.krate()),
262
263        _ => true,
264    };
265    impl_allowed || {
266        let rustc_has_incoherent_inherent_impls = match self_ty {
267            TyKind::Tuple(_)
268            | TyKind::FnDef(_, _)
269            | TyKind::Array(_, _)
270            | TyKind::Never
271            | TyKind::RawPtr(_, _)
272            | TyKind::Ref(_, _, _)
273            | TyKind::Slice(_)
274            | TyKind::Str
275            | TyKind::Bool
276            | TyKind::Char
277            | TyKind::Int(_)
278            | TyKind::Uint(_)
279            | TyKind::Float(_) => true,
280
281            TyKind::Adt(adt_def, _) => match adt_def.def_id().0 {
282                hir_def::AdtId::StructId(id) => db
283                    .struct_signature(id)
284                    .flags
285                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
286                hir_def::AdtId::UnionId(id) => db
287                    .union_signature(id)
288                    .flags
289                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
290                hir_def::AdtId::EnumId(it) => db
291                    .enum_signature(it)
292                    .flags
293                    .contains(EnumFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
294            },
295            TyKind::Dynamic(it, _) => it.principal_def_id().is_some_and(|trait_id| {
296                db.trait_signature(trait_id.0)
297                    .flags
298                    .contains(TraitFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS)
299            }),
300
301            _ => false,
302        };
303        let items = impl_id.impl_items(db);
304        rustc_has_incoherent_inherent_impls
305            && !items.items.is_empty()
306            && items.items.iter().all(|&(_, assoc)| match assoc {
307                AssocItemId::FunctionId(it) => {
308                    db.function_signature(it).flags.contains(FnFlags::RUSTC_ALLOW_INCOHERENT_IMPL)
309                }
310                AssocItemId::ConstId(it) => {
311                    db.const_signature(it).flags.contains(ConstFlags::RUSTC_ALLOW_INCOHERENT_IMPL)
312                }
313                AssocItemId::TypeAliasId(it) => db
314                    .type_alias_signature(it)
315                    .flags
316                    .contains(TypeAliasFlags::RUSTC_ALLOW_INCOHERENT_IMPL),
317            })
318    }
319}

321/// Checks whether the impl satisfies the orphan rules.
322///
323/// Given `impl<P1..=Pn> Trait<T1..=Tn> for T0`, an `impl`` is valid only if at least one of the following is true:
324/// - Trait is a local trait
325/// - All of
326///   - At least one of the types `T0..=Tn`` must be a local type. Let `Ti`` be the first such type.
327///   - No uncovered type parameters `P1..=Pn` may appear in `T0..Ti`` (excluding `Ti`)
328pub fn check_orphan_rules<'db>(db: &'db dyn HirDatabase, impl_: ImplId) -> bool {
329    let Some(impl_trait) = db.impl_trait(impl_) else {
330        // not a trait impl
331        return true;
332    };
333
334    let local_crate = impl_.lookup(db).container.krate(db);
335    let is_local = |tgt_crate| tgt_crate == local_crate;
336
337    let trait_ref = impl_trait.instantiate_identity();
338    let trait_id = trait_ref.def_id.0;
339    if is_local(trait_id.module(db).krate(db)) {
340        // trait to be implemented is local
341        return true;
342    }
343
344    let unwrap_fundamental = |mut ty: Ty<'db>| {
345        // Unwrap all layers of fundamental types with a loop.
346        loop {
347            match ty.kind() {
348                TyKind::Ref(_, referenced, _) => ty = referenced,
349                TyKind::Adt(adt_def, subs) => {
350                    let AdtId::StructId(s) = adt_def.def_id().0 else {
351                        break ty;
352                    };
353                    let struct_signature = db.struct_signature(s);
354                    if struct_signature.flags.contains(StructFlags::FUNDAMENTAL) {
355                        let next = subs.types().next();
356                        match next {
357                            Some(it) => ty = it,
358                            None => break ty,
359                        }
360                    } else {
361                        break ty;
362                    }
363                }
364                _ => break ty,
365            }
366        }
367    };
368    //   - At least one of the types `T0..=Tn`` must be a local type. Let `Ti`` be the first such type.
369
370    // FIXME: param coverage
371    //   - No uncovered type parameters `P1..=Pn` may appear in `T0..Ti`` (excluding `Ti`)
372    let is_not_orphan = trait_ref.args.types().any(|ty| match unwrap_fundamental(ty).kind() {
373        TyKind::Adt(adt_def, _) => is_local(adt_def.def_id().0.module(db).krate(db)),
374        TyKind::Error(_) => true,
375        TyKind::Dynamic(it, _) => {
376            it.principal_def_id().is_some_and(|trait_id| is_local(trait_id.0.module(db).krate(db)))
377        }
378        _ => false,
379    });
380    #[allow(clippy::let_and_return)]
381    is_not_orphan
