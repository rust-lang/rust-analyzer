27const AUTODEREF_RECURSION_LIMIT: usize = 20;
28
29/// Returns types that `ty` transitively dereferences to. This function is only meant to be used
30/// outside `hir-ty`.
31///
32/// It is guaranteed that:
33/// - the yielded types don't contain inference variables (but may contain `TyKind::Error`).
34/// - a type won't be yielded more than once; in other words, the returned iterator will stop if it
35///   detects a cycle in the deref chain.
36pub fn autoderef<'db>(
37    db: &'db dyn HirDatabase,
38    env: Arc<TraitEnvironment<'db>>,
39    ty: Canonical<'db, Ty<'db>>,
40) -> impl Iterator<Item = Ty<'db>> + use<'db> {
41    let mut table = InferenceTable::new(db, env, None);
42    let ty = table.instantiate_canonical(ty);
43    let mut autoderef = Autoderef::new_no_tracking(&mut table, ty);
44    let mut v = Vec::new();
45    while let Some((ty, _steps)) = autoderef.next() {
46        // `ty` may contain unresolved inference variables. Since there's no chance they would be
47        // resolved, just replace with fallback type.
48        let resolved = autoderef.table.resolve_completely(ty);
49
50        // If the deref chain contains a cycle (e.g. `A` derefs to `B` and `B` derefs to `A`), we
51        // would revisit some already visited types. Stop here to avoid duplication.
52        //
53        // XXX: The recursion limit for `Autoderef` is currently 20, so `Vec::contains()` shouldn't
54        // be too expensive. Replace this duplicate check with `FxHashSet` if it proves to be more
55        // performant.
56        if v.contains(&resolved) {
57            break;
58        }
59        v.push(resolved);
60    }
61    v.into_iter()
62}

64pub(crate) trait TrackAutoderefSteps<'db>: Default + fmt::Debug {
65    fn len(&self) -> usize;
66    fn push(&mut self, ty: Ty<'db>, kind: AutoderefKind);
67}
68
69impl<'db> TrackAutoderefSteps<'db> for usize {
70    fn len(&self) -> usize {
71        *self
72    }
73    fn push(&mut self, _: Ty<'db>, _: AutoderefKind) {
74        *self += 1;
75    }
76}
77impl<'db> TrackAutoderefSteps<'db> for Vec<(Ty<'db>, AutoderefKind)> {
78    fn len(&self) -> usize {
79        self.len()
80    }
81    fn push(&mut self, ty: Ty<'db>, kind: AutoderefKind) {
82        self.push((ty, kind));
83    }
84}

87pub(crate) enum AutoderefKind {

89    Builtin,

108/// Recursively dereference a type, considering both built-in
109/// dereferences (`*`) and the `Deref` trait.
110/// Although called `Autoderef` it can be configured to use the
111/// `Receiver` trait instead of the `Deref` trait.
112pub(crate) struct Autoderef<'a, 'db, Steps = Vec<(Ty<'db>, AutoderefKind)>> {
113    // Meta infos:
114    pub(crate) table: &'a mut InferenceTable<'db>,
115    traits: Option<AutoderefTraits>,
116
117    // Current state:
118    state: AutoderefSnapshot<'db, Steps>,
119
120    // Configurations:
121    include_raw_pointers: bool,
122    use_receiver_trait: bool,
123}
124
125impl<'a, 'db, Steps: TrackAutoderefSteps<'db>> Iterator for Autoderef<'a, 'db, Steps> {
126    type Item = (Ty<'db>, usize);
127
128    fn next(&mut self) -> Option<Self::Item> {
129        debug!("autoderef: steps={:?}, cur_ty={:?}", self.state.steps, self.state.cur_ty);
130        if self.state.at_start {
131            self.state.at_start = false;
132            debug!("autoderef stage #0 is {:?}", self.state.cur_ty);
133            return Some((self.state.cur_ty, 0));
134        }
135
136        // If we have reached the recursion limit, error gracefully.
137        if self.state.steps.len() >= AUTODEREF_RECURSION_LIMIT {
138            self.state.reached_recursion_limit = true;
139            return None;
140        }
141
142        if self.state.cur_ty.is_ty_var() {
143            return None;
144        }
145
146        // Otherwise, deref if type is derefable:
147        // NOTE: in the case of self.use_receiver_trait = true, you might think it would
148        // be better to skip this clause and use the Overloaded case only, since &T
149        // and &mut T implement Receiver. But built-in derefs apply equally to Receiver
150        // and Deref, and this has benefits for const and the emitted MIR.
151        let (kind, new_ty) = if let Some(ty) =
152            self.state.cur_ty.builtin_deref(self.table.db, self.include_raw_pointers)
153        {
154            debug_assert_eq!(ty, self.table.infer_ctxt.resolve_vars_if_possible(ty));
155            // NOTE: we may still need to normalize the built-in deref in case
156            // we have some type like `&<Ty as Trait>::Assoc`, since users of
157            // autoderef expect this type to have been structurally normalized.
158            if let TyKind::Alias(..) = ty.kind() {
159                let (normalized_ty, obligations) = structurally_normalize_ty(self.table, ty)?;
160                self.state.obligations.extend(obligations);
161                (AutoderefKind::Builtin, normalized_ty)
162            } else {
163                (AutoderefKind::Builtin, ty)
164            }
165        } else if let Some(ty) = self.overloaded_deref_ty(self.state.cur_ty) {
166            // The overloaded deref check already normalizes the pointee type.
167            (AutoderefKind::Overloaded, ty)
168        } else {
169            return None;
170        };
171
172        self.state.steps.push(self.state.cur_ty, kind);
173        debug!(
174            "autoderef stage #{:?} is {:?} from {:?}",
175            self.step_count(),
176            new_ty,
177            (self.state.cur_ty, kind)
178        );
179        self.state.cur_ty = new_ty;
180
181        Some((self.state.cur_ty, self.step_count()))
182    }
183}

197impl<'a, 'db, Steps: TrackAutoderefSteps<'db>> Autoderef<'a, 'db, Steps> {
198    fn new_impl(table: &'a mut InferenceTable<'db>, base_ty: Ty<'db>) -> Self {
199        Autoderef {
200            state: AutoderefSnapshot {
201                steps: Steps::default(),
202                cur_ty: table.infer_ctxt.resolve_vars_if_possible(base_ty),
203                obligations: PredicateObligations::new(),
204                at_start: true,
205                reached_recursion_limit: false,
206            },
207            table,
208            traits: None,
209            include_raw_pointers: false,
210            use_receiver_trait: false,
211        }
212    }
213
214    fn autoderef_traits(&mut self) -> Option<AutoderefTraits> {
215        match &mut self.traits {
216            Some(it) => Some(*it),
217            None => {
218                let traits = if self.use_receiver_trait {
219                    (|| {
220                        Some(AutoderefTraits {
221                            trait_: LangItem::Receiver
222                                .resolve_trait(self.table.db, self.table.trait_env.krate)?,
223                            trait_target: LangItem::ReceiverTarget
224                                .resolve_type_alias(self.table.db, self.table.trait_env.krate)?,
225                        })
226                    })()
227                    .or_else(|| {
228                        Some(AutoderefTraits {
229                            trait_: LangItem::Deref
230                                .resolve_trait(self.table.db, self.table.trait_env.krate)?,
231                            trait_target: LangItem::DerefTarget
232                                .resolve_type_alias(self.table.db, self.table.trait_env.krate)?,
233                        })
234                    })?
235                } else {
236                    AutoderefTraits {
237                        trait_: LangItem::Deref
238                            .resolve_trait(self.table.db, self.table.trait_env.krate)?,
239                        trait_target: LangItem::DerefTarget
240                            .resolve_type_alias(self.table.db, self.table.trait_env.krate)?,
241                    }
242                };
243                Some(*self.traits.insert(traits))
244            }
245        }
246    }
247
248    fn overloaded_deref_ty(&mut self, ty: Ty<'db>) -> Option<Ty<'db>> {
249        debug!("overloaded_deref_ty({:?})", ty);
250        let interner = self.table.interner();
251
252        // <ty as Deref>, or whatever the equivalent trait is that we've been asked to walk.
253        let AutoderefTraits { trait_, trait_target } = self.autoderef_traits()?;
254
255        let trait_ref = TraitRef::new(interner, trait_.into(), [ty]);
256        let obligation =
257            Obligation::new(interner, ObligationCause::new(), self.table.trait_env.env, trait_ref);
258        // We detect whether the self type implements `Deref` before trying to
259        // structurally normalize. We use `predicate_may_hold_opaque_types_jank`
260        // to support not-yet-defined opaque types. It will succeed for `impl Deref`
261        // but fail for `impl OtherTrait`.
262        if !self.table.infer_ctxt.predicate_may_hold_opaque_types_jank(&obligation) {
263            debug!("overloaded_deref_ty: cannot match obligation");
264            return None;
265        }
266
267        let (normalized_ty, obligations) = structurally_normalize_ty(
268            self.table,
269            Ty::new_projection(interner, trait_target.into(), [ty]),
270        )?;
271        debug!("overloaded_deref_ty({:?}) = ({:?}, {:?})", ty, normalized_ty, obligations);
272        self.state.obligations.extend(obligations);
273
274        Some(self.table.infer_ctxt.resolve_vars_if_possible(normalized_ty))
275    }
276
277    /// Returns the final type we ended up with, which may be an unresolved
278    /// inference variable.
279    pub(crate) fn final_ty(&self) -> Ty<'db> {
280        self.state.cur_ty
281    }
282
283    pub(crate) fn step_count(&self) -> usize {
284        self.state.steps.len()
285    }
286
287    pub(crate) fn take_obligations(&mut self) -> PredicateObligations<'db> {
288        std::mem::take(&mut self.state.obligations)
289    }
290
291    pub(crate) fn steps(&self) -> &Steps {
292        &self.state.steps
293    }
294
295    #[expect(dead_code)]
296    pub(crate) fn reached_recursion_limit(&self) -> bool {
297        self.state.reached_recursion_limit
298    }
299
300    /// also dereference through raw pointer types
301    /// e.g., assuming ptr_to_Foo is the type `*const Foo`
302    /// fcx.autoderef(span, ptr_to_Foo)  => [*const Foo]
303    /// fcx.autoderef(span, ptr_to_Foo).include_raw_ptrs() => [*const Foo, Foo]
304    pub(crate) fn include_raw_pointers(mut self) -> Self {
305        self.include_raw_pointers = true;
306        self
307    }
308
309    /// Use `core::ops::Receiver` and `core::ops::Receiver::Target` as
310    /// the trait and associated type to iterate, instead of
311    /// `core::ops::Deref` and `core::ops::Deref::Target`
312    pub(crate) fn use_receiver_trait(mut self) -> Self {
313        self.use_receiver_trait = true;
314        self
315    }
316}

318fn structurally_normalize_ty<'db>(
319    table: &InferenceTable<'db>,
320    ty: Ty<'db>,
321) -> Option<(Ty<'db>, PredicateObligations<'db>)> {
322    let mut ocx = ObligationCtxt::new(&table.infer_ctxt);
323    let Ok(normalized_ty) =
324        ocx.structurally_normalize_ty(&ObligationCause::misc(), table.trait_env.env, ty)
325    else {
326        // We shouldn't have errors here in the old solver, except for
327        // evaluate/fulfill mismatches, but that's not a reason for an ICE.
328        return None;
329    };
330    let errors = ocx.try_evaluate_obligations();
331    if !errors.is_empty() {
332        unreachable!();
333    }
334
335    Some((normalized_ty, ocx.into_pending_obligations()))
336}

338pub(crate) fn overloaded_deref_ty<'db>(
339    table: &InferenceTable<'db>,
340    ty: Ty<'db>,
341) -> Option<InferOk<'db, Ty<'db>>> {
342    let interner = table.interner();
343
344    let trait_target = LangItem::DerefTarget.resolve_type_alias(table.db, table.trait_env.krate)?;
345
346    let (normalized_ty, obligations) =
347        structurally_normalize_ty(table, Ty::new_projection(interner, trait_target.into(), [ty]))?;
348
349    Some(InferOk { value: normalized_ty, obligations })
