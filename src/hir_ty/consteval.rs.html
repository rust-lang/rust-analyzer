39pub enum ConstEvalError {
40    MirLowerError(MirLowerError),
41    MirEvalError(MirEvalError),
42}
43
44impl ConstEvalError {
45    pub fn pretty_print(
46        &self,
47        f: &mut String,
48        db: &dyn HirDatabase,
49        span_formatter: impl Fn(span::FileId, span::TextRange) -> String,
50        display_target: DisplayTarget,
51    ) -> std::result::Result<(), std::fmt::Error> {
52        match self {
53            ConstEvalError::MirLowerError(e) => {
54                e.pretty_print(f, db, span_formatter, display_target)
55            }
56            ConstEvalError::MirEvalError(e) => {
57                e.pretty_print(f, db, span_formatter, display_target)
58            }
59        }
60    }
61}

63impl From<MirLowerError> for ConstEvalError {
64    fn from(value: MirLowerError) -> Self {
65        match value {
66            MirLowerError::ConstEvalError(_, e) => *e,
67            _ => ConstEvalError::MirLowerError(value),
68        }
69    }

78/// Interns a constant scalar with the given type
79fn intern_const_ref<'db>(
80    interner: DbInterner<'db>,
81    value: &Literal,
82    ty: Ty<'db>,
83) -> Result<Const<'db>, CreateConstError<'db>> {
84    let Ok(data_layout) = interner.db.target_data_layout(interner.expect_crate()) else {
85        return Ok(Const::error(interner));
86    };
87    let valtree = match (ty.kind(), value) {
88        (TyKind::Uint(uint), Literal::Uint(value, _)) => {
89            let size = uint.bit_width().map(Size::from_bits).unwrap_or(data_layout.pointer_size());
90            let Some(scalar) = ScalarInt::try_from_uint(*value, size) else {
91                return Ok(Const::error(interner));
92            };
93            ValTreeKind::Leaf(scalar)
94        }
95        (TyKind::Uint(uint), Literal::Int(value, _)) => {
96            // `Literal::Int` is the default, so we also need to account for the type being uint.
97            let size = uint.bit_width().map(Size::from_bits).unwrap_or(data_layout.pointer_size());
98            let Some(scalar) = ScalarInt::try_from_uint(*value as u128, size) else {
99                return Ok(Const::error(interner));
100            };
101            ValTreeKind::Leaf(scalar)
102        }
103        (TyKind::Int(int), Literal::Int(value, _)) => {
104            let size = int.bit_width().map(Size::from_bits).unwrap_or(data_layout.pointer_size());
105            let Some(scalar) = ScalarInt::try_from_int(*value, size) else {
106                return Ok(Const::error(interner));
107            };
108            ValTreeKind::Leaf(scalar)
109        }
110        (TyKind::Bool, Literal::Bool(value)) => ValTreeKind::Leaf(ScalarInt::from(*value)),
111        (TyKind::Char, Literal::Char(value)) => ValTreeKind::Leaf(ScalarInt::from(*value)),
112        (TyKind::Float(float), Literal::Float(value, _)) => {
113            let size = Size::from_bits(float.bit_width());
114            let value = match float {
115                rustc_ast_ir::FloatTy::F16 => value.to_f16().to_bits(),
116                rustc_ast_ir::FloatTy::F32 => value.to_f32().to_bits(),
117                rustc_ast_ir::FloatTy::F64 => value.to_f64().to_bits(),
118                rustc_ast_ir::FloatTy::F128 => value.to_f128().to_bits(),
119            };
120            let scalar = ScalarInt::try_from_uint(value, size).unwrap();
121            ValTreeKind::Leaf(scalar)
122        }
123        (_, Literal::String(value)) => {
124            let u8_values = &interner.default_types().consts.u8_values;
125            ValTreeKind::Branch(Consts::new_from_iter(
126                interner,
127                value.as_str().as_bytes().iter().map(|&byte| u8_values[usize::from(byte)]),
128            ))
129        }
130        (_, Literal::ByteString(value)) => {
131            let u8_values = &interner.default_types().consts.u8_values;
132            ValTreeKind::Branch(Consts::new_from_iter(
133                interner,
134                value.iter().map(|&byte| u8_values[usize::from(byte)]),
135            ))
136        }
137        (_, Literal::CString(_)) => {
138            // FIXME:
139            return Ok(Const::error(interner));
140        }
141        _ => {
142            never!("mismatching type for literal");
143            let actual = literal_ty(
144                interner,
145                value,
146                |types| types.types.i32,
147                |types| types.types.u32,
148                |types| types.types.f64,
149            );
150            return Err(CreateConstError::TypeMismatch { actual });
151        }
152    };
153    Ok(Const::new_valtree(interner, ty, valtree))
154}

156pub(crate) fn literal_ty<'db>(
157    interner: DbInterner<'db>,
158    value: &Literal,
159    default_int: impl FnOnce(&DefaultAny<'db>) -> Ty<'db>,
160    default_uint: impl FnOnce(&DefaultAny<'db>) -> Ty<'db>,
161    default_float: impl FnOnce(&DefaultAny<'db>) -> Ty<'db>,
162) -> Ty<'db> {
163    let types = interner.default_types();
164    match value {
165        Literal::Bool(..) => types.types.bool,
166        Literal::String(..) => types.types.static_str_ref,
167        Literal::ByteString(bs) => {
168            let byte_type = types.types.u8;
169            let array_type = Ty::new_array(interner, byte_type, bs.len() as u128);
170            Ty::new_ref(interner, types.regions.statik, array_type, Mutability::Not)
171        }
172        Literal::CString(..) => Ty::new_ref(
173            interner,
174            types.regions.statik,
175            interner.lang_items().CStr.map_or(types.types.error, |strukt| {
176                Ty::new_adt(interner, strukt.into(), types.empty.generic_args)
177            }),
178            Mutability::Not,
179        ),
180        Literal::Char(..) => types.types.char,
181        Literal::Int(_v, ty) => match ty {
182            Some(int_ty) => match int_ty {
183                hir_def::builtin_type::BuiltinInt::Isize => types.types.isize,
184                hir_def::builtin_type::BuiltinInt::I8 => types.types.i8,
185                hir_def::builtin_type::BuiltinInt::I16 => types.types.i16,
186                hir_def::builtin_type::BuiltinInt::I32 => types.types.i32,
187                hir_def::builtin_type::BuiltinInt::I64 => types.types.i64,
188                hir_def::builtin_type::BuiltinInt::I128 => types.types.i128,
189            },
190            None => default_int(types),
191        },
192        Literal::Uint(_v, ty) => match ty {
193            Some(int_ty) => match int_ty {
194                hir_def::builtin_type::BuiltinUint::Usize => types.types.usize,
195                hir_def::builtin_type::BuiltinUint::U8 => types.types.u8,
196                hir_def::builtin_type::BuiltinUint::U16 => types.types.u16,
197                hir_def::builtin_type::BuiltinUint::U32 => types.types.u32,
198                hir_def::builtin_type::BuiltinUint::U64 => types.types.u64,
199                hir_def::builtin_type::BuiltinUint::U128 => types.types.u128,
200            },
201            None => default_uint(types),
202        },
203        Literal::Float(_v, ty) => match ty {
204            Some(float_ty) => match float_ty {
205                hir_def::builtin_type::BuiltinFloat::F16 => types.types.f16,
206                hir_def::builtin_type::BuiltinFloat::F32 => types.types.f32,
207                hir_def::builtin_type::BuiltinFloat::F64 => types.types.f64,
208                hir_def::builtin_type::BuiltinFloat::F128 => types.types.f128,
209            },
210            None => default_float(types),
211        },
212    }
213}

215/// Interns a possibly-unknown target usize
216pub fn usize_const<'db>(db: &'db dyn HirDatabase, value: Option<u128>, krate: Crate) -> Const<'db> {
217    let interner = DbInterner::new_no_crate(db);
218    let value = match value {
219        Some(value) => value,
220        None => {
221            return Const::error(interner);
222        }
223    };
224    let Ok(data_layout) = db.target_data_layout(krate) else {
225        return Const::error(interner);
226    };
227    let usize_ty = interner.default_types().types.usize;
228    let Some(scalar) = ScalarInt::try_from_uint(value, data_layout.pointer_size()) else {
229        return Const::error(interner);
230    };
231    Const::new_valtree(interner, usize_ty, ValTreeKind::Leaf(scalar))
232}

238pub fn try_const_usize<'db>(db: &'db dyn HirDatabase, c: Const<'db>) -> Option<u128> {
239    match c.kind() {
240        ConstKind::Param(_) => None,
241        ConstKind::Infer(_) => None,
242        ConstKind::Bound(_, _) => None,
243        ConstKind::Placeholder(_) => None,
244        ConstKind::Unevaluated(unevaluated_const) => match unevaluated_const.def.0 {
245            GeneralConstId::ConstId(id) => {
246                let subst = unevaluated_const.args;
247                let ec = db.const_eval(id, subst, None).ok()?;
248                Some(allocation_as_usize(ec))
249            }
250            GeneralConstId::StaticId(id) => {
251                let ec = db.const_eval_static(id).ok()?;
252                Some(allocation_as_usize(ec))
253            }
254            GeneralConstId::AnonConstId(id) => {
255                let subst = unevaluated_const.args;
256                let ec = db.anon_const_eval(id, subst, None).ok()?;
257                Some(allocation_as_usize(ec))
258            }
259        },
260        ConstKind::Value(val) => {
261            if val.ty == default_types(db).types.usize {
262                Some(val.value.inner().to_leaf().to_uint_unchecked())
263            } else {
264                None
265            }
266        }
267        ConstKind::Error(_) => None,
268        ConstKind::Expr(_) => None,
269    }
270}

276pub fn try_const_isize<'db>(db: &'db dyn HirDatabase, c: Const<'db>) -> Option<i128> {
277    match c.kind() {
278        ConstKind::Param(_) => None,
279        ConstKind::Infer(_) => None,
280        ConstKind::Bound(_, _) => None,
281        ConstKind::Placeholder(_) => None,
282        ConstKind::Unevaluated(unevaluated_const) => match unevaluated_const.def.0 {
283            GeneralConstId::ConstId(id) => {
284                let subst = unevaluated_const.args;
285                let ec = db.const_eval(id, subst, None).ok()?;
286                Some(allocation_as_isize(ec))
287            }
288            GeneralConstId::StaticId(id) => {
289                let ec = db.const_eval_static(id).ok()?;
290                Some(allocation_as_isize(ec))
291            }
292            GeneralConstId::AnonConstId(id) => {
293                let subst = unevaluated_const.args;
294                let ec = db.anon_const_eval(id, subst, None).ok()?;
295                Some(allocation_as_isize(ec))
296            }
297        },
298        ConstKind::Value(val) => {
299            if val.ty == default_types(db).types.isize {
300                Some(val.value.inner().to_leaf().to_int_unchecked())
301            } else {
302                None
303            }
304        }
305        ConstKind::Error(_) => None,
306        ConstKind::Expr(_) => None,
307    }

311pub(crate) enum CreateConstError<'db> {
312    UsedForbiddenParam,
313    ResolveToNonConst,
314    DoesNotResolve,
315    ConstHasGenerics,
316    UnderscoreExpr,
317    AnonConstInterningDisabled,
318    TypeMismatch {

324pub(crate) fn path_to_const<'a, 'db>(
325    db: &'db dyn HirDatabase,
326    resolver: &Resolver<'db>,
327    generics: &dyn Fn() -> &'a Generics<'db>,
328    forbid_params_after: Option<u32>,
329    path: &Path,
330) -> Result<Const<'db>, CreateConstError<'db>> {
331    let interner = DbInterner::new_no_crate(db);
332    let resolution = resolver
333        .resolve_path_in_value_ns_fully(db, path, HygieneId::ROOT)
334        .ok_or(CreateConstError::DoesNotResolve)?;
335    let no_generics = |def| crate::generics::generics(db, def).has_no_params();
336    let konst = match resolution {
337        ValueNs::ConstId(id) if no_generics(id.into()) => GeneralConstId::ConstId(id),
338        ValueNs::StaticId(id) => GeneralConstId::StaticId(id),
339        ValueNs::ConstId(_) => return Err(CreateConstError::ConstHasGenerics),
340        ValueNs::GenericParam(param) => {
341            let index = generics().type_or_const_param_idx(param.into());
342            if forbid_params_after.is_some_and(|forbid_after| index >= forbid_after) {
343                return Err(CreateConstError::UsedForbiddenParam);
344            }
345            return Ok(Const::new_param(interner, ParamConst { id: param, index }));
346        }
347        // These are not valid as consts.
348        // FIXME: Report an error?
349        ValueNs::ImplSelf(_)
350        | ValueNs::LocalBinding(_)
351        | ValueNs::FunctionId(_)
352        | ValueNs::StructId(_)
353        | ValueNs::EnumVariantId(_) => return Err(CreateConstError::ResolveToNonConst),
354    };
355    let args = GenericArgs::empty(interner);
356    Ok(Const::new_unevaluated(interner, UnevaluatedConst { def: konst.into(), args }))
357}

359pub(crate) fn create_anon_const<'a, 'db>(
360    interner: DbInterner<'db>,
361    owner: ExpressionStoreOwnerId,
362    store: &ExpressionStore,
363    expr: ExprId,
364    resolver: &Resolver<'db>,
365    expected_ty: Ty<'db>,
366    generics: &dyn Fn() -> &'a Generics<'db>,
367    create_var: Option<&mut dyn FnMut(Span) -> Const<'db>>,
368    lowering_mode: LoweringMode,
369    forbid_params_after: Option<u32>,
370) -> Result<Const<'db>, CreateConstError<'db>> {
371    match &store[expr] {
372        Expr::Literal(literal) => intern_const_ref(interner, literal, expected_ty),
373        Expr::Underscore => match create_var {
374            Some(create_var) => Ok(create_var(expr.into())),
375            None => Err(CreateConstError::UnderscoreExpr),
376        },
377        Expr::Path(path)
378            if let konst =
379                path_to_const(interner.db, resolver, generics, forbid_params_after, path)
380                && !matches!(
381                    konst,
382                    Err(CreateConstError::DoesNotResolve | CreateConstError::ConstHasGenerics)
383                ) =>
384        {
385            konst
386        }
387        _ => {
388            let Some(token) = lowering_mode.allow_tracked_structs() else {
389                return Err(CreateConstError::AnonConstInterningDisabled);
390            };
391
392            let allow_using_generic_params = forbid_params_after.is_none();
393            let konst = AnonConstId::new(
394                interner.db,
395                AnonConstLoc {
396                    owner,
397                    expr,
398                    ty: StoredEarlyBinder::bind(expected_ty.store()),
399                    allow_using_generic_params,
400                },
401                token,
402            );
403            let args = if allow_using_generic_params {
404                GenericArgs::identity_for_item(interner, owner.generic_def(interner.db).into())
405            } else {
406                GenericArgs::empty(interner)
407            };
408            Ok(Const::new_unevaluated(
409                interner,
410                UnevaluatedConst { def: GeneralConstId::AnonConstId(konst).into(), args },
411            ))
412        }
413    }
414}

416pub(crate) fn const_eval_discriminant_variant(
417    db: &dyn HirDatabase,
418    variant_id: EnumVariantId,
419) -> Result<i128, ConstEvalError> {
420    let interner = DbInterner::new_no_crate(db);
421    let def = variant_id.into();
422    let body = Body::of(db, def);
423    let loc = variant_id.lookup(db);
424    if matches!(body[body.root_expr()], Expr::Missing) {
425        let prev_idx = loc.index(db).checked_sub(1);
426        let value = match prev_idx {
427            Some(prev_idx) => {
428                1 + db.const_eval_discriminant(loc.parent.enum_variants(db).variants[prev_idx].0)?
429            }
430            _ => 0,
431        };
432        return Ok(value);
433    }
434
435    let repr = AttrFlags::repr(db, loc.parent.into());
436    let is_signed = repr.and_then(|repr| repr.int).is_none_or(|int| int.is_signed());
437
438    let mir_body = db.monomorphized_mir_body(
439        def.into(),
440        GenericArgs::empty(interner).store(),
441        ParamEnvAndCrate {
442            param_env: db.trait_environment(def.generic_def(db)),
443            krate: def.krate(db),
444        }
445        .store(),
446    )?;
447    let c = interpret_mir(db, mir_body, false, None)?.0?;
448    let c = if is_signed { allocation_as_isize(c) } else { allocation_as_usize(c) as i128 };
449    Ok(c)
450}

460pub(crate) fn const_eval<'db>(
461    db: &'db dyn HirDatabase,
462    def: ConstId,
463    subst: GenericArgs<'db>,
464    trait_env: Option<ParamEnvAndCrate<'db>>,
465) -> Result<Allocation<'db>, ConstEvalError> {
466    return match const_eval_query(db, def, subst.store(), trait_env.map(|env| env.store())) {
467        Ok(konst) => Ok(konst.as_ref()),
468        Err(err) => Err(err.clone()),
469    };

471    #[salsa::tracked(returns(ref), cycle_result = const_eval_cycle_result)]
472    pub(crate) fn const_eval_query(
473        db: &dyn HirDatabase,
474        def: ConstId,
475        subst: StoredGenericArgs,
476        trait_env: Option<StoredParamEnvAndCrate>,
477    ) -> Result<StoredAllocation, ConstEvalError> {
478        let body = db.monomorphized_mir_body(
479            def.into(),
480            subst,
481            ParamEnvAndCrate { param_env: db.trait_environment(def.into()), krate: def.krate(db) }
482                .store(),
483        )?;
484        let c = interpret_mir(db, body, false, trait_env.as_ref().map(|env| env.as_ref()))?.0?;
485        Ok(c.store())
486    }
