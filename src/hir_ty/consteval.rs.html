39#[derive(Debug, Clone, PartialEq, Eq)]
40pub enum ConstEvalError<'db> {
41    MirLowerError(MirLowerError<'db>),
42    MirEvalError(MirEvalError<'db>),
43}
44
45impl ConstEvalError<'_> {
46    pub fn pretty_print(
47        &self,
48        f: &mut String,
49        db: &dyn HirDatabase,
50        span_formatter: impl Fn(span::FileId, span::TextRange) -> String,
51        display_target: DisplayTarget,
52    ) -> std::result::Result<(), std::fmt::Error> {
53        match self {
54            ConstEvalError::MirLowerError(e) => {
55                e.pretty_print(f, db, span_formatter, display_target)
56            }
57            ConstEvalError::MirEvalError(e) => {
58                e.pretty_print(f, db, span_formatter, display_target)
59            }
60        }
61    }
62}
63
64impl<'db> From<MirLowerError<'db>> for ConstEvalError<'db> {
65    fn from(value: MirLowerError<'db>) -> Self {
66        match value {
67            MirLowerError::ConstEvalError(_, e) => *e,
68            _ => ConstEvalError::MirLowerError(value),
69        }
70    }
71}

73impl<'db> From<MirEvalError<'db>> for ConstEvalError<'db> {
74    fn from(value: MirEvalError<'db>) -> Self {
75        ConstEvalError::MirEvalError(value)
76    }
77}

79/// Interns a constant scalar with the given type
80pub fn intern_const_ref<'a>(
81    db: &'a dyn HirDatabase,
82    value: &LiteralConstRef,
83    ty: Ty<'a>,
84    krate: Crate,
85) -> Const<'a> {
86    let interner = DbInterner::new_with(db, Some(krate), None);
87    let layout = db.layout_of_ty(ty, TraitEnvironment::empty(krate));
88    let kind = match value {
89        LiteralConstRef::Int(i) => {
90            // FIXME: We should handle failure of layout better.
91            let size = layout.map(|it| it.size.bytes_usize()).unwrap_or(16);
92            rustc_type_ir::ConstKind::Value(ValueConst::new(
93                ty,
94                ConstBytes {
95                    memory: i.to_le_bytes()[0..size].into(),
96                    memory_map: MemoryMap::default(),
97                },
98            ))
99        }
100        LiteralConstRef::UInt(i) => {
101            let size = layout.map(|it| it.size.bytes_usize()).unwrap_or(16);
102            rustc_type_ir::ConstKind::Value(ValueConst::new(
103                ty,
104                ConstBytes {
105                    memory: i.to_le_bytes()[0..size].into(),
106                    memory_map: MemoryMap::default(),
107                },
108            ))
109        }
110        LiteralConstRef::Bool(b) => rustc_type_ir::ConstKind::Value(ValueConst::new(
111            ty,
112            ConstBytes { memory: Box::new([*b as u8]), memory_map: MemoryMap::default() },
113        )),
114        LiteralConstRef::Char(c) => rustc_type_ir::ConstKind::Value(ValueConst::new(
115            ty,
116            ConstBytes {
117                memory: (*c as u32).to_le_bytes().into(),
118                memory_map: MemoryMap::default(),
119            },
120        )),
121        LiteralConstRef::Unknown => rustc_type_ir::ConstKind::Error(ErrorGuaranteed),
122    };
123    Const::new(interner, kind)
124}

126/// Interns a possibly-unknown target usize
127pub fn usize_const<'db>(db: &'db dyn HirDatabase, value: Option<u128>, krate: Crate) -> Const<'db> {
128    intern_const_ref(
129        db,
130        &value.map_or(LiteralConstRef::Unknown, LiteralConstRef::UInt),
131        Ty::new_uint(DbInterner::new_with(db, Some(krate), None), rustc_type_ir::UintTy::Usize),
132        krate,
133    )
134}

136pub fn try_const_usize<'db>(db: &'db dyn HirDatabase, c: Const<'db>) -> Option<u128> {
137    match c.kind() {
138        ConstKind::Param(_) => None,
139        ConstKind::Infer(_) => None,
140        ConstKind::Bound(_, _) => None,
141        ConstKind::Placeholder(_) => None,
142        ConstKind::Unevaluated(unevaluated_const) => {
143            let c = match unevaluated_const.def {
144                SolverDefId::ConstId(id) => GeneralConstId::ConstId(id),
145                SolverDefId::StaticId(id) => GeneralConstId::StaticId(id),
146                _ => unreachable!(),
147            };
148            let subst = unevaluated_const.args;
149            let ec = db.const_eval(c, subst, None).ok()?;
150            try_const_usize(db, ec)
151        }
152        ConstKind::Value(val) => Some(u128::from_le_bytes(pad16(&val.value.inner().memory, false))),
153        ConstKind::Error(_) => None,
154        ConstKind::Expr(_) => None,
155    }
156}

158pub fn try_const_isize<'db>(db: &'db dyn HirDatabase, c: &Const<'db>) -> Option<i128> {
159    match (*c).kind() {
160        ConstKind::Param(_) => None,
161        ConstKind::Infer(_) => None,
162        ConstKind::Bound(_, _) => None,
163        ConstKind::Placeholder(_) => None,
164        ConstKind::Unevaluated(unevaluated_const) => {
165            let c = match unevaluated_const.def {
166                SolverDefId::ConstId(id) => GeneralConstId::ConstId(id),
167                SolverDefId::StaticId(id) => GeneralConstId::StaticId(id),
168                _ => unreachable!(),
169            };
170            let subst = unevaluated_const.args;
171            let ec = db.const_eval(c, subst, None).ok()?;
172            try_const_isize(db, &ec)
173        }
174        ConstKind::Value(val) => Some(i128::from_le_bytes(pad16(&val.value.inner().memory, true))),
175        ConstKind::Error(_) => None,
176        ConstKind::Expr(_) => None,
177    }
178}

180pub(crate) fn const_eval_discriminant_variant<'db>(
181    db: &'db dyn HirDatabase,
182    variant_id: EnumVariantId,
183) -> Result<i128, ConstEvalError<'db>> {
184    let interner = DbInterner::new_with(db, None, None);
185    let def = variant_id.into();
186    let body = db.body(def);
187    let loc = variant_id.lookup(db);
188    if matches!(body[body.body_expr], Expr::Missing) {
189        let prev_idx = loc.index.checked_sub(1);
190        let value = match prev_idx {
191            Some(prev_idx) => {
192                1 + db.const_eval_discriminant(
193                    loc.parent.enum_variants(db).variants[prev_idx as usize].0,
194                )?
195            }
196            _ => 0,
197        };
198        return Ok(value);
199    }
200
201    let repr = db.enum_signature(loc.parent).repr;
202    let is_signed = repr.and_then(|repr| repr.int).is_none_or(|int| int.is_signed());
203
204    let mir_body = db.monomorphized_mir_body(
205        def,
206        GenericArgs::new_from_iter(interner, []),
207        db.trait_environment_for_body(def),
208    )?;
209    let c = interpret_mir(db, mir_body, false, None)?.0?;
210    let c = if is_signed {
211        try_const_isize(db, &c).unwrap()
212    } else {
213        try_const_usize(db, c).unwrap() as i128
214    };
215    Ok(c)
216}

218// FIXME: Ideally constants in const eval should have separate body (issue #7434), and this function should
219// get an `InferenceResult` instead of an `InferenceContext`. And we should remove `ctx.clone().resolve_all()` here
220// and make this function private. See the fixme comment on `InferenceContext::resolve_all`.
221pub(crate) fn eval_to_const<'db>(expr: ExprId, ctx: &mut InferenceContext<'_, 'db>) -> Const<'db> {
222    let infer = ctx.fixme_resolve_all_clone();
223    fn has_closure(body: &Body, expr: ExprId) -> bool {
224        if matches!(body[expr], Expr::Closure { .. }) {
225            return true;
226        }
227        let mut r = false;
228        body.walk_child_exprs(expr, |idx| r |= has_closure(body, idx));
229        r
230    }
231    if has_closure(ctx.body, expr) {
232        // Type checking clousres need an isolated body (See the above FIXME). Bail out early to prevent panic.
233        return unknown_const(infer[expr]);
234    }
235    if let Expr::Path(p) = &ctx.body[expr] {
236        let mut ctx = TyLoweringContext::new(
237            ctx.db,
238            &ctx.resolver,
239            ctx.body,
240            ctx.generic_def,
241            LifetimeElisionKind::Infer,
242        );
243        if let Some(c) = ctx.path_to_const(p) {
244            return c;
245        }
246    }
247    if let Ok(mir_body) = lower_to_mir(ctx.db, ctx.owner, ctx.body, &infer, expr)
248        && let Ok((Ok(result), _)) = interpret_mir(ctx.db, Arc::new(mir_body), true, None)
249    {
250        return result;
251    }
252    unknown_const(infer[expr])
253}

278pub(crate) fn const_eval_query<'db>(
279    db: &'db dyn HirDatabase,
280    def: GeneralConstId,
281    subst: GenericArgs<'db>,
282    trait_env: Option<Arc<TraitEnvironment<'db>>>,
283) -> Result<Const<'db>, ConstEvalError<'db>> {
284    let body = match def {
285        GeneralConstId::ConstId(c) => {
286            db.monomorphized_mir_body(c.into(), subst, db.trait_environment(c.into()))?
287        }
288        GeneralConstId::StaticId(s) => {
289            let krate = s.module(db).krate();
290            db.monomorphized_mir_body(s.into(), subst, TraitEnvironment::empty(krate))?
291        }
292    };
293    let c = interpret_mir(db, body, false, trait_env)?.0?;
294    Ok(c)
295}

305        db.trait_environment_for_body(def.into()),
306    )?;
307    let c = interpret_mir(db, body, false, None)?.0?;
