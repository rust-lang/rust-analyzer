41#[derive(Debug, Clone, PartialEq, Eq)]
42pub enum ConstEvalError {
43    MirLowerError(MirLowerError),
44    MirEvalError(MirEvalError),
45}
46
47impl ConstEvalError {
48    pub fn pretty_print(
49        &self,
50        f: &mut String,
51        db: &dyn HirDatabase,
52        span_formatter: impl Fn(span::FileId, span::TextRange) -> String,
53        display_target: DisplayTarget,
54    ) -> std::result::Result<(), std::fmt::Error> {
55        match self {
56            ConstEvalError::MirLowerError(e) => {
57                e.pretty_print(f, db, span_formatter, display_target)
58            }
59            ConstEvalError::MirEvalError(e) => {
60                e.pretty_print(f, db, span_formatter, display_target)
61            }
62        }
63    }
64}
65
66impl From<MirLowerError> for ConstEvalError {
67    fn from(value: MirLowerError) -> Self {
68        match value {
69            MirLowerError::ConstEvalError(_, e) => *e,
70            _ => ConstEvalError::MirLowerError(value),
71        }
72    }
73}
74
75impl From<MirEvalError> for ConstEvalError {
76    fn from(value: MirEvalError) -> Self {
77        ConstEvalError::MirEvalError(value)
78    }

81/// Interns a constant scalar with the given type
82pub fn intern_const_ref<'a>(
83    db: &'a dyn HirDatabase,
84    value: &Literal,
85    ty: Ty<'a>,
86    _krate: Crate,
87) -> Const<'a> {
88    let interner = DbInterner::new_no_crate(db);
89    let kind = match value {
90        &Literal::Uint(i, builtin_ty)
91            if builtin_ty.is_none() || ty.as_builtin() == builtin_ty.map(BuiltinType::Uint) =>
92        {
93            let memory = match ty.as_builtin() {
94                Some(BuiltinType::Uint(builtin_uint)) => match builtin_uint {
95                    BuiltinUint::U8 => Box::new([i as u8]) as Box<[u8]>,
96                    BuiltinUint::U16 => Box::new((i as u16).to_le_bytes()),
97                    BuiltinUint::U32 => Box::new((i as u32).to_le_bytes()),
98                    BuiltinUint::U64 => Box::new((i as u64).to_le_bytes()),
99                    BuiltinUint::U128 => Box::new((i).to_le_bytes()),
100                    BuiltinUint::Usize => Box::new((i as usize).to_le_bytes()),
101                },
102                _ => return Const::new(interner, rustc_type_ir::ConstKind::Error(ErrorGuaranteed)),
103            };
104            rustc_type_ir::ConstKind::Value(ValueConst::new(
105                ty,
106                ConstBytes { memory, memory_map: MemoryMap::default() },
107            ))
108        }
109        &Literal::Int(i, None)
110            if ty
111                .as_builtin()
112                .is_some_and(|builtin_ty| matches!(builtin_ty, BuiltinType::Uint(_))) =>
113        {
114            let memory = match ty.as_builtin() {
115                Some(BuiltinType::Uint(builtin_uint)) => match builtin_uint {
116                    BuiltinUint::U8 => Box::new([i as u8]) as Box<[u8]>,
117                    BuiltinUint::U16 => Box::new((i as u16).to_le_bytes()),
118                    BuiltinUint::U32 => Box::new((i as u32).to_le_bytes()),
119                    BuiltinUint::U64 => Box::new((i as u64).to_le_bytes()),
120                    BuiltinUint::U128 => Box::new((i as u128).to_le_bytes()),
121                    BuiltinUint::Usize => Box::new((i as usize).to_le_bytes()),
122                },
123                _ => return Const::new(interner, rustc_type_ir::ConstKind::Error(ErrorGuaranteed)),
124            };
125            rustc_type_ir::ConstKind::Value(ValueConst::new(
126                ty,
127                ConstBytes { memory, memory_map: MemoryMap::default() },
128            ))
129        }
130        &Literal::Int(i, builtin_ty)
131            if builtin_ty.is_none() || ty.as_builtin() == builtin_ty.map(BuiltinType::Int) =>
132        {
133            let memory = match ty.as_builtin() {
134                Some(BuiltinType::Int(builtin_int)) => match builtin_int {
135                    BuiltinInt::I8 => Box::new([i as u8]) as Box<[u8]>,
136                    BuiltinInt::I16 => Box::new((i as i16).to_le_bytes()),
137                    BuiltinInt::I32 => Box::new((i as i32).to_le_bytes()),
138                    BuiltinInt::I64 => Box::new((i as i64).to_le_bytes()),
139                    BuiltinInt::I128 => Box::new((i).to_le_bytes()),
140                    BuiltinInt::Isize => Box::new((i as isize).to_le_bytes()),
141                },
142                _ => return Const::new(interner, rustc_type_ir::ConstKind::Error(ErrorGuaranteed)),
143            };
144            rustc_type_ir::ConstKind::Value(ValueConst::new(
145                ty,
146                ConstBytes { memory, memory_map: MemoryMap::default() },
147            ))
148        }
149        Literal::Float(float_type_wrapper, builtin_float)
150            if builtin_float.is_none()
151                || ty.as_builtin() == builtin_float.map(BuiltinType::Float) =>
152        {
153            let memory = match ty.as_builtin().unwrap() {
154                BuiltinType::Float(builtin_float) => match builtin_float {
155                    // FIXME:
156                    hir_def::builtin_type::BuiltinFloat::F16 => Box::new([0u8; 2]) as Box<[u8]>,
157                    hir_def::builtin_type::BuiltinFloat::F32 => {
158                        Box::new(float_type_wrapper.to_f32().to_le_bytes())
159                    }
160                    hir_def::builtin_type::BuiltinFloat::F64 => {
161                        Box::new(float_type_wrapper.to_f64().to_le_bytes())
162                    }
163                    // FIXME:
164                    hir_def::builtin_type::BuiltinFloat::F128 => Box::new([0; 16]),
165                },
166                _ => unreachable!(),
167            };
168            rustc_type_ir::ConstKind::Value(ValueConst::new(
169                ty,
170                ConstBytes { memory, memory_map: MemoryMap::default() },
171            ))
172        }
173        Literal::Bool(b) if ty.is_bool() => rustc_type_ir::ConstKind::Value(ValueConst::new(
174            ty,
175            ConstBytes { memory: Box::new([*b as u8]), memory_map: MemoryMap::default() },
176        )),
177        Literal::Char(c) if ty.is_char() => rustc_type_ir::ConstKind::Value(ValueConst::new(
178            ty,
179            ConstBytes {
180                memory: (*c as u32).to_le_bytes().into(),
181                memory_map: MemoryMap::default(),
182            },
183        )),
184        Literal::String(symbol) if ty.is_str() => rustc_type_ir::ConstKind::Value(ValueConst::new(
185            ty,
186            ConstBytes {
187                memory: symbol.as_str().as_bytes().into(),
188                memory_map: MemoryMap::default(),
189            },
190        )),
191        Literal::ByteString(items) if ty.as_slice().is_some_and(|ty| ty.is_u8()) => {
192            rustc_type_ir::ConstKind::Value(ValueConst::new(
193                ty,
194                ConstBytes { memory: items.clone(), memory_map: MemoryMap::default() },
195            ))
196        }
197        // FIXME
198        Literal::CString(_items) => rustc_type_ir::ConstKind::Error(ErrorGuaranteed),
199        _ => rustc_type_ir::ConstKind::Error(ErrorGuaranteed),
200    };
201    Const::new(interner, kind)
202}

204/// Interns a possibly-unknown target usize
205pub fn usize_const<'db>(db: &'db dyn HirDatabase, value: Option<u128>, krate: Crate) -> Const<'db> {
206    intern_const_ref(
207        db,
208        &match value {
209            Some(value) => Literal::Uint(value, Some(BuiltinUint::Usize)),
210            None => {
211                return Const::new(
212                    DbInterner::new_no_crate(db),
213                    rustc_type_ir::ConstKind::Error(ErrorGuaranteed),
214                );
215            }
216        },
217        Ty::new_uint(DbInterner::new_no_crate(db), rustc_type_ir::UintTy::Usize),
218        krate,
219    )
220}

222pub fn try_const_usize<'db>(db: &'db dyn HirDatabase, c: Const<'db>) -> Option<u128> {
223    match c.kind() {
224        ConstKind::Param(_) => None,
225        ConstKind::Infer(_) => None,
226        ConstKind::Bound(_, _) => None,
227        ConstKind::Placeholder(_) => None,
228        ConstKind::Unevaluated(unevaluated_const) => match unevaluated_const.def.0 {
229            GeneralConstId::ConstId(id) => {
230                let subst = unevaluated_const.args;
231                let ec = db.const_eval(id, subst, None).ok()?;
232                try_const_usize(db, ec)
233            }
234            GeneralConstId::StaticId(id) => {
235                let ec = db.const_eval_static(id).ok()?;
236                try_const_usize(db, ec)
237            }
238        },
239        ConstKind::Value(val) => Some(u128::from_le_bytes(pad16(&val.value.inner().memory, false))),
240        ConstKind::Error(_) => None,
241        ConstKind::Expr(_) => None,
242    }
243}

245pub fn try_const_isize<'db>(db: &'db dyn HirDatabase, c: &Const<'db>) -> Option<i128> {
246    match (*c).kind() {
247        ConstKind::Param(_) => None,
248        ConstKind::Infer(_) => None,
249        ConstKind::Bound(_, _) => None,
250        ConstKind::Placeholder(_) => None,
251        ConstKind::Unevaluated(unevaluated_const) => match unevaluated_const.def.0 {
252            GeneralConstId::ConstId(id) => {
253                let subst = unevaluated_const.args;
254                let ec = db.const_eval(id, subst, None).ok()?;
255                try_const_isize(db, &ec)
256            }
257            GeneralConstId::StaticId(id) => {
258                let ec = db.const_eval_static(id).ok()?;
259                try_const_isize(db, &ec)
260            }
261        },
262        ConstKind::Value(val) => Some(i128::from_le_bytes(pad16(&val.value.inner().memory, true))),
263        ConstKind::Error(_) => None,
264        ConstKind::Expr(_) => None,
265    }
266}

268pub(crate) fn const_eval_discriminant_variant(
269    db: &dyn HirDatabase,
270    variant_id: EnumVariantId,
271) -> Result<i128, ConstEvalError> {
272    let interner = DbInterner::new_no_crate(db);
273    let def = variant_id.into();
274    let body = db.body(def);
275    let loc = variant_id.lookup(db);
276    if matches!(body[body.body_expr], Expr::Missing) {
277        let prev_idx = loc.index.checked_sub(1);
278        let value = match prev_idx {
279            Some(prev_idx) => {
280                1 + db.const_eval_discriminant(
281                    loc.parent.enum_variants(db).variants[prev_idx as usize].0,
282                )?
283            }
284            _ => 0,
285        };
286        return Ok(value);
287    }
288
289    let repr = AttrFlags::repr(db, loc.parent.into());
290    let is_signed = repr.and_then(|repr| repr.int).is_none_or(|int| int.is_signed());
291
292    let mir_body = db.monomorphized_mir_body(
293        def,
294        GenericArgs::empty(interner).store(),
295        ParamEnvAndCrate { param_env: db.trait_environment_for_body(def), krate: def.krate(db) }
296            .store(),
297    )?;
298    let c = interpret_mir(db, mir_body, false, None)?.0?;
299    let c = if is_signed {
300        try_const_isize(db, &c).unwrap()
301    } else {
302        try_const_usize(db, c).unwrap() as i128
303    };
304    Ok(c)
305}

307// FIXME: Ideally constants in const eval should have separate body (issue #7434), and this function should
308// get an `InferenceResult` instead of an `InferenceContext`. And we should remove `ctx.clone().resolve_all()` here
309// and make this function private. See the fixme comment on `InferenceContext::resolve_all`.
310pub(crate) fn eval_to_const<'db>(expr: ExprId, ctx: &mut InferenceContext<'_, 'db>) -> Const<'db> {
311    let infer = ctx.fixme_resolve_all_clone();
312    fn has_closure(body: &Body, expr: ExprId) -> bool {
313        if matches!(body[expr], Expr::Closure { .. }) {
314            return true;
315        }
316        let mut r = false;
317        body.walk_child_exprs(expr, |idx| r |= has_closure(body, idx));
318        r
319    }
320    if has_closure(ctx.body, expr) {
321        // Type checking clousres need an isolated body (See the above FIXME). Bail out early to prevent panic.
322        return Const::error(ctx.interner());
323    }
324    if let Expr::Path(p) = &ctx.body[expr] {
325        let mut ctx = TyLoweringContext::new(
326            ctx.db,
327            &ctx.resolver,
328            ctx.body,
329            ctx.generic_def,
330            LifetimeElisionKind::Infer,
331        );
332        if let Some(c) = ctx.path_to_const(p) {
333            return c;
334        }
335    }
336    if let Ok(mir_body) = lower_to_mir(ctx.db, ctx.owner, ctx.body, &infer, expr)
337        && let Ok((Ok(result), _)) = interpret_mir(ctx.db, Arc::new(mir_body), true, None)
338    {
339        return result;
340    }
341    Const::error(ctx.interner())
342}

352pub(crate) fn const_eval<'db>(
353    db: &'db dyn HirDatabase,
354    def: ConstId,
355    subst: GenericArgs<'db>,
356    trait_env: Option<ParamEnvAndCrate<'db>>,
357) -> Result<Const<'db>, ConstEvalError> {
358    return match const_eval_query(db, def, subst.store(), trait_env.map(|env| env.store())) {
359        Ok(konst) => Ok(konst.as_ref()),
360        Err(err) => Err(err.clone()),
361    };
362
363    #[salsa::tracked(returns(ref), cycle_result = const_eval_cycle_result)]
364    pub(crate) fn const_eval_query<'db>(
365        db: &'db dyn HirDatabase,
366        def: ConstId,
367        subst: StoredGenericArgs,
368        trait_env: Option<StoredParamEnvAndCrate>,
369    ) -> Result<StoredConst, ConstEvalError> {
370        let body = db.monomorphized_mir_body(
371            def.into(),
372            subst,
373            ParamEnvAndCrate { param_env: db.trait_environment(def.into()), krate: def.krate(db) }
374                .store(),
375        )?;
376        let c = interpret_mir(db, body, false, trait_env.as_ref().map(|env| env.as_ref()))?.0?;
377        Ok(c.store())
378    }

386    ) -> Result<StoredConst, ConstEvalError> {

391pub(crate) fn const_eval_static<'db>(
392    db: &'db dyn HirDatabase,
393    def: StaticId,
394) -> Result<Const<'db>, ConstEvalError> {
395    return match const_eval_static_query(db, def) {
396        Ok(konst) => Ok(konst.as_ref()),
397        Err(err) => Err(err.clone()),
398    };
399
400    #[salsa::tracked(returns(ref), cycle_result = const_eval_static_cycle_result)]
401    pub(crate) fn const_eval_static_query<'db>(
402        db: &'db dyn HirDatabase,
403        def: StaticId,
404    ) -> Result<StoredConst, ConstEvalError> {
405        let interner = DbInterner::new_no_crate(db);
406        let body = db.monomorphized_mir_body(
407            def.into(),
408            GenericArgs::empty(interner).store(),
409            ParamEnvAndCrate {
410                param_env: db.trait_environment_for_body(def.into()),
411                krate: def.krate(db),
412            }
413            .store(),
414        )?;
415        let c = interpret_mir(db, body, false, None)?.0?;
416        Ok(c.store())
417    }
