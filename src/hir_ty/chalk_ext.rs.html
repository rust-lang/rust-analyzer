13pub(crate) trait TyExt {
14    fn is_unit(&self) -> bool;
15    fn is_unknown(&self) -> bool;
16    fn contains_unknown(&self) -> bool;
17
18    fn as_adt(&self) -> Option<(hir_def::AdtId, &Substitution)>;
19    fn as_tuple(&self) -> Option<&Substitution>;
20    fn as_fn_def(&self, db: &dyn HirDatabase) -> Option<FunctionId>;
21    fn as_reference(&self) -> Option<(&Ty, Lifetime, Mutability)>;
22
23    fn callable_def(&self, db: &dyn HirDatabase) -> Option<CallableDefId>;
24    fn callable_sig(&self, db: &dyn HirDatabase) -> Option<CallableSig>;
25
26    fn strip_references(&self) -> &Ty;
27
28    /// If this is a `dyn Trait`, returns that trait.
29    fn dyn_trait(&self) -> Option<TraitId>;
30}
31
32impl TyExt for Ty {
33    fn is_unit(&self) -> bool {
34        matches!(self.kind(Interner), TyKind::Tuple(0, _))
35    }
36
37    fn is_unknown(&self) -> bool {
38        matches!(self.kind(Interner), TyKind::Error)
39    }
40
41    fn contains_unknown(&self) -> bool {
42        self.data(Interner).flags.contains(TypeFlags::HAS_ERROR)
43    }
44
45    fn as_adt(&self) -> Option<(hir_def::AdtId, &Substitution)> {
46        match self.kind(Interner) {
47            TyKind::Adt(AdtId(adt), parameters) => Some((*adt, parameters)),
48            _ => None,
49        }
50    }
51
52    fn as_tuple(&self) -> Option<&Substitution> {
53        match self.kind(Interner) {
54            TyKind::Tuple(_, substs) => Some(substs),
55            _ => None,
56        }
57    }
58
59    fn as_fn_def(&self, db: &dyn HirDatabase) -> Option<FunctionId> {
60        match self.callable_def(db) {
61            Some(CallableDefId::FunctionId(func)) => Some(func),
62            Some(CallableDefId::StructId(_) | CallableDefId::EnumVariantId(_)) | None => None,
63        }
64    }
65
66    fn as_reference(&self) -> Option<(&Ty, Lifetime, Mutability)> {
67        match self.kind(Interner) {
68            TyKind::Ref(mutability, lifetime, ty) => Some((ty, lifetime.clone(), *mutability)),
69            _ => None,
70        }
71    }
72
73    fn callable_def(&self, db: &dyn HirDatabase) -> Option<CallableDefId> {
74        match self.kind(Interner) {
75            &TyKind::FnDef(def, ..) => Some(ToChalk::from_chalk(db, def)),
76            _ => None,
77        }
78    }
79
80    fn callable_sig(&self, db: &dyn HirDatabase) -> Option<CallableSig> {
81        match self.kind(Interner) {
82            TyKind::Function(fn_ptr) => Some(CallableSig::from_fn_ptr(fn_ptr)),
83            TyKind::FnDef(def, parameters) => Some(CallableSig::from_def(db, *def, parameters)),
84            TyKind::Closure(.., substs) => ClosureSubst(substs).sig_ty(db).callable_sig(db),
85            _ => None,
86        }
87    }
88
89    fn dyn_trait(&self) -> Option<TraitId> {
90        let trait_ref = match self.kind(Interner) {
91            // The principal trait bound should be the first element of the bounds. This is an
92            // invariant ensured by `TyLoweringContext::lower_dyn_trait()`.
93            // FIXME: dyn types may not have principal trait and we don't want to return auto trait
94            // here.
95            TyKind::Dyn(dyn_ty) => dyn_ty.bounds.skip_binders().interned().first().and_then(|b| {
96                match b.skip_binders() {
97                    WhereClause::Implemented(trait_ref) => Some(trait_ref),
98                    _ => None,
99                }
100            }),
101            _ => None,
102        }?;
103        Some(from_chalk_trait_id(trait_ref.trait_id))
104    }
105
106    fn strip_references(&self) -> &Ty {
107        let mut t: &Ty = self;
108        while let TyKind::Ref(_mutability, _lifetime, ty) = t.kind(Interner) {
109            t = ty;
110        }
111        t
112    }
113}

115pub trait ProjectionTyExt {
116    fn trait_ref(&self, db: &dyn HirDatabase) -> TraitRef;
117    fn trait_(&self, db: &dyn HirDatabase) -> TraitId;
118    fn self_type_parameter(&self, db: &dyn HirDatabase) -> Ty;
119}
120
121impl ProjectionTyExt for ProjectionTy {
122    fn trait_ref(&self, db: &dyn HirDatabase) -> TraitRef {
123        // FIXME: something like `Split` trait from chalk-solve might be nice.
124        let generics = generics(db, from_assoc_type_id(self.associated_ty_id).into());
125        let parent_len = generics.parent_generics().map_or(0, |g| g.len_self());
126        let substitution =
127            Substitution::from_iter(Interner, self.substitution.iter(Interner).take(parent_len));
128        TraitRef { trait_id: to_chalk_trait_id(self.trait_(db)), substitution }
129    }
130
131    fn trait_(&self, db: &dyn HirDatabase) -> TraitId {
132        match from_assoc_type_id(self.associated_ty_id).lookup(db).container {
133            ItemContainerId::TraitId(it) => it,
134            _ => panic!("projection ty without parent trait"),
135        }
136    }
137
138    fn self_type_parameter(&self, db: &dyn HirDatabase) -> Ty {
139        self.trait_ref(db).self_type_parameter(Interner)
140    }
141}

143pub(crate) trait DynTyExt {
144    fn principal(&self) -> Option<Binders<Binders<&TraitRef>>>;
145}
146
147impl DynTyExt for DynTy {
148    fn principal(&self) -> Option<Binders<Binders<&TraitRef>>> {
149        self.bounds.as_ref().filter_map(|bounds| {
150            bounds.interned().first().and_then(|b| {
151                b.as_ref().filter_map(|b| match b {
152                    crate::WhereClause::Implemented(trait_ref) => Some(trait_ref),
153                    _ => None,
154                })
155            })
156        })
157    }
158}

160pub trait TraitRefExt {

162}

164impl TraitRefExt for TraitRef {
165    fn hir_trait_id(&self) -> TraitId {
