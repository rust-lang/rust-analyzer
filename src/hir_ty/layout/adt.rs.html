22pub fn layout_of_adt_query(
23    db: &dyn HirDatabase,
24    def: AdtId,
25    args: StoredGenericArgs,
26    trait_env: StoredParamEnvAndCrate,
27) -> Result<Arc<Layout>, LayoutError> {
28    let krate = trait_env.krate;
29    let Ok(target) = db.target_data_layout(krate) else {
30        return Err(LayoutError::TargetLayoutNotAvailable);
31    };
32    let dl = &*target;
33    let cx = LayoutCx::new(dl);
34    let handle_variant = |def: VariantId, var: &VariantFields| {
35        var.fields()
36            .iter()
37            .map(|(fd, _)| {
38                db.layout_of_ty(field_ty(db, def, fd, args.as_ref()).store(), trait_env.clone())
39            })
40            .collect::<Result<Vec<_>, _>>()
41    };
42    let (variants, repr, is_special_no_niche) = match def {
43        AdtId::StructId(s) => {
44            let sig = db.struct_signature(s);
45            let mut r = SmallVec::<[_; 1]>::new();
46            r.push(handle_variant(s.into(), s.fields(db))?);
47            (
48                r,
49                AttrFlags::repr(db, s.into()).unwrap_or_default(),
50                sig.flags.intersects(StructFlags::IS_UNSAFE_CELL | StructFlags::IS_UNSAFE_PINNED),
51            )
52        }
53        AdtId::UnionId(id) => {
54            let repr = AttrFlags::repr(db, id.into());
55            let mut r = SmallVec::new();
56            r.push(handle_variant(id.into(), id.fields(db))?);
57            (r, repr.unwrap_or_default(), false)
58        }
59        AdtId::EnumId(e) => {
60            let variants = e.enum_variants(db);
61            let r = variants
62                .variants
63                .iter()
64                .map(|&(v, _, _)| handle_variant(v.into(), v.fields(db)))
65                .collect::<Result<SmallVec<_>, _>>()?;
66            (r, AttrFlags::repr(db, e.into()).unwrap_or_default(), false)
67        }
68    };
69    let variants = variants
70        .iter()
71        .map(|it| it.iter().map(|it| &**it).collect::<Vec<_>>())
72        .collect::<SmallVec<[_; 1]>>();
73    let variants = variants.iter().map(|it| it.iter().collect()).collect::<IndexVec<_, _>>();
74    let result = if matches!(def, AdtId::UnionId(..)) {
75        cx.calc.layout_of_union(&repr, &variants)?
76    } else {
77        cx.calc.layout_of_struct_or_enum(
78            &repr,
79            &variants,
80            matches!(def, AdtId::EnumId(..)),
81            is_special_no_niche,
82            layout_scalar_valid_range(db, def),
83            |min, max| repr_discr(dl, &repr, min, max).unwrap_or((Integer::I8, false)),
84            variants.iter_enumerated().filter_map(|(id, _)| {
85                let AdtId::EnumId(e) = def else { return None };
86                let d = db.const_eval_discriminant(e.enum_variants(db).variants[id.0].0).ok()?;
87                Some((id, d))
88            }),
89            !matches!(def, AdtId::EnumId(..))
90                && variants
91                    .iter()
92                    .next()
93                    .and_then(|it| it.iter().last().map(|it| !it.is_unsized()))
94                    .unwrap_or(true),
95        )?
96    };
97    Ok(Arc::new(result))
98}

110fn layout_scalar_valid_range(db: &dyn HirDatabase, def: AdtId) -> (Bound<u128>, Bound<u128>) {
111    let range = AttrFlags::rustc_layout_scalar_valid_range(db, def);
112    let get = |value| match value {
113        Some(it) => Bound::Included(it),
114        None => Bound::Unbounded,
115    };
116    (get(range.start), get(range.end))
117}

119/// Finds the appropriate Integer type and signedness for the given
120/// signed discriminant range and `#[repr]` attribute.
121/// N.B.: `u128` values above `i128::MAX` will be treated as signed, but
122/// that shouldn't affect anything, other than maybe debuginfo.
123fn repr_discr(
124    dl: &TargetDataLayout,
125    repr: &ReprOptions,
126    min: i128,
127    max: i128,
128) -> Result<(Integer, bool), LayoutError> {
129    // Theoretically, negative values could be larger in unsigned representation
130    // than the unsigned representation of the signed minimum. However, if there
131    // are any negative values, the only valid unsigned representation is u128
132    // which can fit all i128 values, so the result remains unaffected.
133    let unsigned_fit = Integer::fit_unsigned(cmp::max(min as u128, max as u128));
134    let signed_fit = cmp::max(Integer::fit_signed(min), Integer::fit_signed(max));
135
136    if let Some(ity) = repr.int {
137        let discr = Integer::from_attr(dl, ity);
138        let fit = if ity.is_signed() { signed_fit } else { unsigned_fit };
139        if discr < fit {
140            return Err(LayoutError::UserReprTooSmall);
141        }
142        return Ok((discr, ity.is_signed()));
143    }
144
145    let at_least = if repr.c() {
146        // This is usually I32, however it can be different on some platforms,
147        // notably hexagon and arm-none/thumb-none
148        dl.c_enum_min_size
149    } else {
150        // repr(Rust) enums try to be as small as possible
151        Integer::I8
152    };
153
154    // If there are no negative values, we can use the unsigned fit.
155    Ok(if min >= 0 {
156        (cmp::max(unsigned_fit, at_least), false)
157    } else {
158        (cmp::max(signed_fit, at_least), true)
159    })
160}
