1//! Type inference, i.e. the process of walking through the code and determining
2//! the type of each expression and pattern.
3//!
4//! For type inference, compare the implementations in rustc (the various
5//! check_* methods in rustc_hir_analysis/check/mod.rs are a good entry point) and
6//! IntelliJ-Rust (org.rust.lang.core.types.infer). Our entry point for
7//! inference here is the `infer` function, which infers the types of all
8//! expressions in a given function.
9//!
10//! During inference, types (i.e. the `Ty` struct) can contain type 'variables'
11//! which represent currently unknown types; as we walk through the expressions,
12//! we might determine that certain variables need to be equal to each other, or
13//! to certain types. To record this, we use the union-find implementation from
14//! the `ena` crate, which is extracted from rustc.

88/// The entry point of type inference.
89pub(crate) fn infer_query(db: &dyn HirDatabase, def: DefWithBodyId) -> Arc<InferenceResult<'_>> {
90    let _p = tracing::info_span!("infer_query").entered();
91    let resolver = def.resolver(db);
92    let body = db.body(def);
93    let mut ctx = InferenceContext::new(db, def, &body, resolver);
94
95    match def {
96        DefWithBodyId::FunctionId(f) => {
97            ctx.collect_fn(f);
98        }
99        DefWithBodyId::ConstId(c) => ctx.collect_const(c, &db.const_signature(c)),
100        DefWithBodyId::StaticId(s) => ctx.collect_static(&db.static_signature(s)),
101        DefWithBodyId::VariantId(v) => {
102            ctx.return_ty = match db.enum_signature(v.lookup(db).parent).variant_body_type() {
103                hir_def::layout::IntegerType::Pointer(signed) => match signed {
104                    true => ctx.types.isize,
105                    false => ctx.types.usize,
106                },
107                hir_def::layout::IntegerType::Fixed(size, signed) => match signed {
108                    true => match size {
109                        Integer::I8 => ctx.types.i8,
110                        Integer::I16 => ctx.types.i16,
111                        Integer::I32 => ctx.types.i32,
112                        Integer::I64 => ctx.types.i64,
113                        Integer::I128 => ctx.types.i128,
114                    },
115                    false => match size {
116                        Integer::I8 => ctx.types.u8,
117                        Integer::I16 => ctx.types.u16,
118                        Integer::I32 => ctx.types.u32,
119                        Integer::I64 => ctx.types.u64,
120                        Integer::I128 => ctx.types.u128,
121                    },
122                },
123            };
124        }
125    }
126
127    ctx.infer_body();
128
129    ctx.infer_mut_body();
130
131    ctx.handle_opaque_type_uses();
132
133    ctx.type_inference_fallback();
134
135    // Comment from rustc:
136    // Even though coercion casts provide type hints, we check casts after fallback for
137    // backwards compatibility. This makes fallback a stronger type hint than a cast coercion.
138    let cast_checks = std::mem::take(&mut ctx.deferred_cast_checks);
139    for mut cast in cast_checks.into_iter() {
140        if let Err(diag) = cast.check(&mut ctx) {
141            ctx.diagnostics.push(diag);
142        }
143    }
144
145    ctx.table.select_obligations_where_possible();
146
147    ctx.infer_closures();
148
149    ctx.table.select_obligations_where_possible();
150
151    ctx.handle_opaque_type_uses();
152
153    Arc::new(ctx.resolve_all())
154}

293/// Represents coercing a value to a different type of value.
294///
295/// We transform values by following a number of `Adjust` steps in order.
296/// See the documentation on variants of `Adjust` for more details.
297///
298/// Here are some common scenarios:
299///
300/// 1. The simplest cases are where a pointer is not adjusted fat vs thin.
301///    Here the pointer will be dereferenced N times (where a dereference can
302///    happen to raw or borrowed pointers or any smart pointer which implements
303///    Deref, including Box<_>). The types of dereferences is given by
304///    `autoderefs`. It can then be auto-referenced zero or one times, indicated
305///    by `autoref`, to either a raw or borrowed pointer. In these cases unsize is
306///    `false`.
307///
308/// 2. A thin-to-fat coercion involves unsizing the underlying data. We start
309///    with a thin pointer, deref a number of times, unsize the underlying data,
310///    then autoref. The 'unsize' phase may change a fixed length array to a
311///    dynamically sized one, a concrete object to a trait object, or statically
312///    sized struct to a dynamically sized one. E.g., &[i32; 4] -> &[i32] is
313///    represented by:
314///
315///    ```ignore
316///    Deref(None) -> [i32; 4],
317///    Borrow(AutoBorrow::Ref) -> &[i32; 4],
318///    Unsize -> &[i32],
319///    ```
320///
321///    Note that for a struct, the 'deep' unsizing of the struct is not recorded.
322///    E.g., `struct Foo<T> { it: T }` we can coerce &Foo<[i32; 4]> to &Foo<[i32]>
323///    The autoderef and -ref are the same as in the above example, but the type
324///    stored in `unsize` is `Foo<[i32]>`, we don't store any further detail about
325///    the underlying conversions from `[i32; 4]` to `[i32]`.
326///
327/// 3. Coercing a `Box<T>` to `Box<dyn Trait>` is an interesting special case. In
328///    that case, we have the pointer we need coming in, so there are no
329///    autoderefs, and no autoref. Instead we just do the `Unsize` transformation.
330///    At some point, of course, `Box` should move out of the compiler, in which
331///    case this is analogous to transforming a struct. E.g., Box<[i32; 4]> ->
332///    Box<[i32]> is an `Adjust::Unsize` with the target `Box<[i32]>`.
333#[derive(Clone, Debug, PartialEq, Eq, Hash)]
334pub struct Adjustment<'db> {
335    pub kind: Adjust<'db>,
336    pub target: Ty<'db>,
337}
338
339impl<'db> Adjustment<'db> {
340    pub fn borrow(interner: DbInterner<'db>, m: Mutability, ty: Ty<'db>, lt: Region<'db>) -> Self {
341        let ty = Ty::new_ref(interner, lt, ty, m);
342        Adjustment { kind: Adjust::Borrow(AutoBorrow::Ref(lt, m)), target: ty }
343    }
344}

347/// only a few specific cases. Right now, those are mostly "things that desugar"
348/// into method calls:
349/// - using `x.some_method()` syntax, where some_method takes `&mut self`,

432///

435/// not appear in the final inference result.

818struct BreakableContext<'db> {
819    /// Whether this context contains at least one break expression.
820    may_break: bool,
821    /// The coercion target of the context.

828#[derive(Clone, Debug)]
829enum BreakableKind {
830    Block,
831    Loop,
832    /// A border is something like an async block, closure etc. Anything that prevents
833    /// breaking/continuing through
834    Border,
835}
836
837fn find_breakable<'a, 'db>(
838    ctxs: &'a mut [BreakableContext<'db>],
839    label: Option<LabelId>,
840) -> Option<&'a mut BreakableContext<'db>> {
841    let mut ctxs = ctxs
842        .iter_mut()
843        .rev()
844        .take_while(|it| matches!(it.kind, BreakableKind::Block | BreakableKind::Loop));
845    match label {
846        Some(_) => ctxs.find(|ctx| ctx.label == label),
847        None => ctxs.find(|ctx| matches!(ctx.kind, BreakableKind::Loop)),
848    }
849}
850
851fn find_continuable<'a, 'db>(
852    ctxs: &'a mut [BreakableContext<'db>],
853    label: Option<LabelId>,
854) -> Option<&'a mut BreakableContext<'db>> {
855    match label {
856        Some(_) => find_breakable(ctxs, label).filter(|it| matches!(it.kind, BreakableKind::Loop)),
857        None => find_breakable(ctxs, label),
858    }
859}

937        let mut ctx = self.clone();

956    // FIXME: This function should be private in module. It is currently only used in the consteval, since we need
957    // `InferenceResult` in the middle of inference. See the fixme comment in `consteval::eval_to_const`. If you
958    // used this function for another workaround, mention it here. If you really need this function and believe that
959    // there is no problem in it being `pub(crate)`, remove this comment.
960    fn resolve_all(self) -> InferenceResult<'db> {
961        let InferenceContext {
962            mut table, mut result, tuple_field_accesses_rev, diagnostics, ..
963        } = self;
964        let mut diagnostics = diagnostics.finish();
965        // Destructure every single field so whenever new fields are added to `InferenceResult` we
966        // don't forget to handle them here.
967        let InferenceResult {
968            method_resolutions,
969            field_resolutions: _,
970            variant_resolutions: _,
971            assoc_resolutions,
972            type_of_expr,
973            type_of_pat,
974            type_of_binding,
975            type_of_opaque,
976            type_mismatches,
977            has_errors,
978            error_ty: _,
979            pat_adjustments,
980            binding_modes: _,
981            expr_adjustments,
982            // Types in `closure_info` have already been `resolve_completely()`'d during
983            // `InferenceContext::infer_closures()` (in `HirPlace::ty()` specifically), so no need
984            // to resolve them here.
985            closure_info: _,
986            mutated_bindings_in_closure: _,
987            tuple_field_access_types: _,
988            coercion_casts: _,
989            diagnostics: _,
990        } = &mut result;
991
992        for ty in type_of_expr.values_mut() {
993            *ty = table.resolve_completely(*ty);
994            *has_errors = *has_errors || ty.references_non_lt_error();
995        }
996        type_of_expr.shrink_to_fit();
997        for ty in type_of_pat.values_mut() {
998            *ty = table.resolve_completely(*ty);
999            *has_errors = *has_errors || ty.references_non_lt_error();
1000        }
1001        type_of_pat.shrink_to_fit();
1002        for ty in type_of_binding.values_mut() {
1003            *ty = table.resolve_completely(*ty);
1004            *has_errors = *has_errors || ty.references_non_lt_error();
1005        }
1006        type_of_binding.shrink_to_fit();
1007        type_of_opaque.shrink_to_fit();
1008
1009        *has_errors |= !type_mismatches.is_empty();
1010
1011        for mismatch in (*type_mismatches).values_mut() {
1012            mismatch.expected = table.resolve_completely(mismatch.expected);
1013            mismatch.actual = table.resolve_completely(mismatch.actual);
1014        }
1015        type_mismatches.shrink_to_fit();
1016        diagnostics.retain_mut(|diagnostic| {
1017            use InferenceDiagnostic::*;
1018            match diagnostic {
1019                ExpectedFunction { found: ty, .. }
1020                | UnresolvedField { receiver: ty, .. }
1021                | UnresolvedMethodCall { receiver: ty, .. } => {
1022                    *ty = table.resolve_completely(*ty);
1023                    // FIXME: Remove this when we are on par with rustc in terms of inference
1024                    if ty.references_non_lt_error() {
1025                        return false;
1026                    }
1027
1028                    if let UnresolvedMethodCall { field_with_same_name, .. } = diagnostic
1029                        && let Some(ty) = field_with_same_name
1030                    {
1031                        *ty = table.resolve_completely(*ty);
1032                        if ty.references_non_lt_error() {
1033                            *field_with_same_name = None;
1034                        }
1035                    }
1036                }
1037                TypedHole { expected: ty, .. } => {
1038                    *ty = table.resolve_completely(*ty);
1039                }
1040                _ => (),
1041            }
1042            true
1043        });
1044        diagnostics.shrink_to_fit();
1045        for (_, subst) in method_resolutions.values_mut() {
1046            *subst = table.resolve_completely(*subst);
1047            *has_errors = *has_errors || subst.types().any(|ty| ty.references_non_lt_error());
1048        }
1049        method_resolutions.shrink_to_fit();
1050        for (_, subst) in assoc_resolutions.values_mut() {
1051            *subst = table.resolve_completely(*subst);
1052            *has_errors = *has_errors || subst.types().any(|ty| ty.references_non_lt_error());
1053        }
1054        assoc_resolutions.shrink_to_fit();
1055        for adjustment in expr_adjustments.values_mut().flatten() {
1056            adjustment.target = table.resolve_completely(adjustment.target);
1057            *has_errors = *has_errors || adjustment.target.references_non_lt_error();
1058        }
1059        expr_adjustments.shrink_to_fit();
1060        for adjustment in pat_adjustments.values_mut().flatten() {
1061            *adjustment = table.resolve_completely(*adjustment);
1062            *has_errors = *has_errors || adjustment.references_non_lt_error();
1063        }
1064        pat_adjustments.shrink_to_fit();
1065        result.tuple_field_access_types = tuple_field_accesses_rev
1066            .into_iter()
1067            .enumerate()
1068            .map(|(idx, subst)| (TupleId(idx as u32), table.resolve_completely(subst)))
1069            .inspect(|(_, subst)| {
1070                *has_errors = *has_errors || subst.iter().any(|ty| ty.references_non_lt_error());
1071            })
1072            .collect();
1073        result.tuple_field_access_types.shrink_to_fit();
1074
1075        result.diagnostics = diagnostics;
1076
1077        result
1078    }

1102    fn collect_fn(&mut self, func: FunctionId) {
1103        let data = self.db.function_signature(func);
1104        let mut param_tys = self.with_ty_lowering(
1105            &data.store,
1106            InferenceTyDiagnosticSource::Signature,
1107            LifetimeElisionKind::for_fn_params(&data),
1108            |ctx| data.params.iter().map(|&type_ref| ctx.lower_ty(type_ref)).collect::<Vec<_>>(),
1109        );
1110
1111        // Check if function contains a va_list, if it does then we append it to the parameter types
1112        // that are collected from the function data
1113        if data.is_varargs() {
1114            let va_list_ty = match self.resolve_va_list() {
1115                Some(va_list) => Ty::new_adt(
1116                    self.interner(),
1117                    va_list,
1118                    GenericArgs::for_item_with_defaults(
1119                        self.interner(),
1120                        va_list.into(),
1121                        |_, id, _| self.table.next_var_for_param(id),
1122                    ),
1123                ),
1124                None => self.err_ty(),
1125            };
1126
1127            param_tys.push(va_list_ty);
1128        }
1129        let mut param_tys = param_tys.into_iter().chain(iter::repeat(self.table.next_ty_var()));
1130        if let Some(self_param) = self.body.self_param
1131            && let Some(ty) = param_tys.next()
1132        {
1133            let ty = self.process_user_written_ty(ty);
1134            self.write_binding_ty(self_param, ty);
1135        }
1136        for (ty, pat) in param_tys.zip(&*self.body.params) {
1137            let ty = self.process_user_written_ty(ty);
1138
1139            self.infer_top_pat(*pat, ty, None);
1140        }
1141        self.return_ty = match data.ret_type {
1142            Some(return_ty) => {
1143                let return_ty = self.with_ty_lowering(
1144                    &data.store,
1145                    InferenceTyDiagnosticSource::Signature,

1353    /// Returns the deeply last field of nested structures, or the same type if
1354    /// not a structure at all. Corresponds to the only possible unsized field,
1355    /// and its type can be used to determine unsizing strategy.
1356    ///
1357    /// This is parameterized over the normalization strategy (i.e. how to
1358    /// handle `<T as Trait>::Assoc` and `impl Trait`); pass the identity
1359    /// function to indicate no normalization should take place.
1360    fn struct_tail_with_normalize(
1361        &mut self,
1362        mut ty: Ty<'db>,
1363        mut normalize: impl FnMut(Ty<'db>) -> Ty<'db>,
1364    ) -> Ty<'db> {
1365        // FIXME: fetch the limit properly
1366        let recursion_limit = 10;
1367        for iteration in 0.. {
1368            if iteration > recursion_limit {
1369                return self.err_ty();
1370            }
1371            match ty.kind() {
1372                TyKind::Adt(adt_def, substs) => match adt_def.def_id().0 {
1373                    AdtId::StructId(struct_id) => {
1374                        match self.db.field_types(struct_id.into()).values().next_back().copied() {
1375                            Some(field) => {
1376                                ty = field.instantiate(self.interner(), substs);
1377                            }
1378                            None => break,
1379                        }
1380                    }
1381                    _ => break,
1382                },
1383                TyKind::Tuple(substs) => match substs.as_slice().split_last() {
1384                    Some((last_ty, _)) => ty = *last_ty,
1385                    None => break,
1386                },
1387                TyKind::Alias(..) => {
1388                    let normalized = normalize(ty);
1389                    if ty == normalized {
1390                        return ty;
1391                    } else {
1392                        ty = normalized;
1393                    }
1394                }
1395                _ => break,
1396            }
1397        }
1398        ty
1399    }

1467    fn resolve_variant(
1468        &mut self,
1469        node: ExprOrPatId,
1470        path: Option<&Path>,
1471        value_ns: bool,
1472    ) -> (Ty<'db>, Option<VariantId>) {
1473        let path = match path {
1474            Some(path) => path,
1475            None => return (self.err_ty(), None),
1476        };
1477        let mut ctx = TyLoweringContext::new(
1478            self.db,
1479            &self.resolver,
1480            &self.body.store,
1481            &self.diagnostics,
1482            InferenceTyDiagnosticSource::Body,
1483            self.generic_def,
1484            LifetimeElisionKind::Infer,
1485        );
1486        let mut path_ctx = ctx.at_path(path, node);
1487        let interner = DbInterner::conjure();
1488        let (resolution, unresolved) = if value_ns {
1489            let Some(res) = path_ctx.resolve_path_in_value_ns(HygieneId::ROOT) else {
1490                return (self.err_ty(), None);
1491            };
1492            match res {
1493                ResolveValueResult::ValueNs(value, _) => match value {
1494                    ValueNs::EnumVariantId(var) => {
1495                        let args = path_ctx.substs_from_path(var.into(), true, false);
1496                        drop(ctx);
1497                        let ty = self
1498                            .db
1499                            .ty(var.lookup(self.db).parent.into())
1500                            .instantiate(interner, args);
1501                        let ty = self.insert_type_vars(ty);
1502                        return (ty, Some(var.into()));
1503                    }
1504                    ValueNs::StructId(strukt) => {
1505                        let args = path_ctx.substs_from_path(strukt.into(), true, false);
1506                        drop(ctx);
1507                        let ty = self.db.ty(strukt.into()).instantiate(interner, args);
1508                        let ty = self.insert_type_vars(ty);
1509                        return (ty, Some(strukt.into()));
1510                    }
1511                    ValueNs::ImplSelf(impl_id) => (TypeNs::SelfType(impl_id), None),
1512                    _ => {
1513                        drop(ctx);
1514                        return (self.err_ty(), None);
1515                    }
1516                },
1517                ResolveValueResult::Partial(typens, unresolved, _) => (typens, Some(unresolved)),
1518            }
1519        } else {
1520            match path_ctx.resolve_path_in_type_ns() {
1521                Some((it, idx)) => (it, idx),
1522                None => return (self.err_ty(), None),
1523            }
1524        };
1525        return match resolution {
1526            TypeNs::AdtId(AdtId::StructId(strukt)) => {
1527                let args = path_ctx.substs_from_path(strukt.into(), true, false);
1528                drop(ctx);
1529                let ty = self.db.ty(strukt.into()).instantiate(interner, args);
1530                let ty = self.insert_type_vars(ty);
1531                forbid_unresolved_segments(self, (ty, Some(strukt.into())), unresolved)
1532            }
1533            TypeNs::AdtId(AdtId::UnionId(u)) => {
1534                let args = path_ctx.substs_from_path(u.into(), true, false);
1535                drop(ctx);
1536                let ty = self.db.ty(u.into()).instantiate(interner, args);
1537                let ty = self.insert_type_vars(ty);
1538                forbid_unresolved_segments(self, (ty, Some(u.into())), unresolved)
1539            }
1540            TypeNs::EnumVariantId(var) => {
1541                let args = path_ctx.substs_from_path(var.into(), true, false);
1542                drop(ctx);
1543                let ty = self.db.ty(var.lookup(self.db).parent.into()).instantiate(interner, args);
1544                let ty = self.insert_type_vars(ty);
1545                forbid_unresolved_segments(self, (ty, Some(var.into())), unresolved)
1546            }
1547            TypeNs::SelfType(impl_id) => {
1548                let mut ty = self.db.impl_self_ty(impl_id).instantiate_identity();
1549
1550                let Some(remaining_idx) = unresolved else {
1551                    drop(ctx);
1552                    let Some(mod_path) = path.mod_path() else {
1553                        never!("resolver should always resolve lang item paths");
1554                        return (self.err_ty(), None);
1555                    };
1556                    return self.resolve_variant_on_alias(ty, None, mod_path);
1557                };
1558
1559                let mut remaining_segments = path.segments().skip(remaining_idx);
1560
1561                if remaining_segments.len() >= 2 {
1562                    path_ctx.ignore_last_segment();
1563                }
1564
1565                // We need to try resolving unresolved segments one by one because each may resolve
1566                // to a projection, which `TyLoweringContext` cannot handle on its own.
1567                let mut tried_resolving_once = false;
1568                while let Some(current_segment) = remaining_segments.first() {
1569                    // If we can resolve to an enum variant, it takes priority over associated type
1570                    // of the same name.
1571                    if let TyKind::Adt(adt_def, _) = ty.kind()
1572                        && let AdtId::EnumId(id) = adt_def.def_id().0
1573                    {
1574                        let enum_data = id.enum_variants(self.db);
1575                        if let Some(variant) = enum_data.variant(current_segment.name) {
1576                            return if remaining_segments.len() == 1 {
1577                                (ty, Some(variant.into()))
1578                            } else {
1579                                // We still have unresolved paths, but enum variants never have
1580                                // associated types!
1581                                // FIXME: Report an error.
1582                                (self.err_ty(), None)
1583                            };
1584                        }
1585                    }
1586
1587                    if tried_resolving_once {
1588                        // FIXME: with `inherent_associated_types` this is allowed, but our `lower_partly_resolved_path()`
1589                        // will need to be updated to err at the correct segment.
1590                        break;
1591                    }
1592
1593                    // `lower_partly_resolved_path()` returns `None` as type namespace unless
1594                    // `remaining_segments` is empty, which is never the case here. We don't know
1595                    // which namespace the new `ty` is in until normalized anyway.
1596                    (ty, _) = path_ctx.lower_partly_resolved_path(resolution, true);
1597                    tried_resolving_once = true;
1598
1599                    ty = self.table.insert_type_vars(ty);
1600                    ty = self.table.normalize_associated_types_in(ty);
1601                    ty = self.table.structurally_resolve_type(ty);
1602                    if ty.is_ty_error() {
1603                        return (self.err_ty(), None);
1604                    }
1605
1606                    remaining_segments = remaining_segments.skip(1);
1607                }
1608                drop(ctx);
1609
1610                let variant = ty.as_adt().and_then(|(id, _)| match id {
1611                    AdtId::StructId(s) => Some(VariantId::StructId(s)),
1612                    AdtId::UnionId(u) => Some(VariantId::UnionId(u)),
1613                    AdtId::EnumId(_) => {
1614                        // FIXME Error E0071, expected struct, variant or union type, found enum `Foo`
1615                        None
1616                    }
1617                });
1618                (ty, variant)
1619            }
1620            TypeNs::TypeAliasId(it) => {
1621                let Some(mod_path) = path.mod_path() else {
1622                    never!("resolver should always resolve lang item paths");
1623                    return (self.err_ty(), None);
1624                };
1625                let args = path_ctx.substs_from_path_segment(it.into(), true, None, false);
1626                drop(ctx);
1627                let interner = DbInterner::conjure();
1628                let ty = self.db.ty(it.into()).instantiate(interner, args);
1629                let ty = self.insert_type_vars(ty);
1630
1631                self.resolve_variant_on_alias(ty, unresolved, mod_path)
1632            }
1633            TypeNs::AdtSelfType(_) => {
1634                // FIXME this could happen in array size expressions, once we're checking them
1635                (self.err_ty(), None)
1636            }
1637            TypeNs::GenericParam(_) => {
1638                // FIXME potentially resolve assoc type
1639                (self.err_ty(), None)
1640            }
1641            TypeNs::AdtId(AdtId::EnumId(_))
1642            | TypeNs::BuiltinType(_)
1643            | TypeNs::TraitId(_)
1644            | TypeNs::ModuleId(_) => {
1645                // FIXME diagnostic
1646                (self.err_ty(), None)
1647            }
1648        };

1652            result: (Ty<'db>, Option<VariantId>),
1653            unresolved: Option<usize>,
1654        ) -> (Ty<'db>, Option<VariantId>) {

1665        &mut self,
1666        ty: Ty<'db>,
1667        unresolved: Option<usize>,
1668        path: &ModPath,
1669    ) -> (Ty<'db>, Option<VariantId>) {
1670        let remaining = unresolved.map(|it| path.segments()[it..].len()).filter(|it| it > &0);
1671        let ty = self.table.try_structurally_resolve_type(ty);
1672        match remaining {
1673            None => {
1674                let variant = ty.as_adt().and_then(|(adt_id, _)| match adt_id {
1675                    AdtId::StructId(s) => Some(VariantId::StructId(s)),
1676                    AdtId::UnionId(u) => Some(VariantId::UnionId(u)),
1677                    AdtId::EnumId(_) => {
1678                        // FIXME Error E0071, expected struct, variant or union type, found enum `Foo`
1679                        None
1680                    }
1681                });
1682                (ty, variant)
1683            }
1684            Some(1) => {
1685                let segment = path.segments().last().unwrap();
1686                // this could be an enum variant or associated type
1687                if let Some((AdtId::EnumId(enum_id), _)) = ty.as_adt() {
1688                    let enum_data = enum_id.enum_variants(self.db);

1694                (self.err_ty(), None)
1695            }
1696            Some(_) => {
1697                // FIXME diagnostic
1698                (self.err_ty(), None)
1699            }
1700        }

1793/// When inferring an expression, we propagate downward whatever type hint we
1794/// are able in the form of an `Expectation`.
1795#[derive(Clone, PartialEq, Eq, Debug)]
1796pub(crate) enum Expectation<'db> {
1797    None,
1798    HasType(Ty<'db>),
1799    Castable(Ty<'db>),
1800    RValueLikeUnsized(Ty<'db>),
1801}
1802
1803impl<'db> Expectation<'db> {
1804    /// The expectation that the type of the expression needs to equal the given
1805    /// type.
1806    fn has_type(ty: Ty<'db>) -> Self {
1807        if ty.is_ty_error() {
1808            // FIXME: get rid of this?
1809            Expectation::None
1810        } else {
1811            Expectation::HasType(ty)
1812        }
1813    }
1814
1815    /// The following explanation is copied straight from rustc:
1816    /// Provides an expectation for an rvalue expression given an *optional*
1817    /// hint, which is not required for type safety (the resulting type might
1818    /// be checked higher up, as is the case with `&expr` and `box expr`), but
1819    /// is useful in determining the concrete type.
1820    ///
1821    /// The primary use case is where the expected type is a fat pointer,
1822    /// like `&[isize]`. For example, consider the following statement:
1823    ///
1824    ///     let it: &[isize] = &[1, 2, 3];
1825    ///
1826    /// In this case, the expected type for the `&[1, 2, 3]` expression is
1827    /// `&[isize]`. If however we were to say that `[1, 2, 3]` has the
1828    /// expectation `ExpectHasType([isize])`, that would be too strong --
1829    /// `[1, 2, 3]` does not have the type `[isize]` but rather `[isize; 3]`.
1830    /// It is only the `&[1, 2, 3]` expression as a whole that can be coerced
1831    /// to the type `&[isize]`. Therefore, we propagate this more limited hint,
1832    /// which still is useful, because it informs integer literals and the like.
1833    /// See the test case `test/ui/coerce-expect-unsized.rs` and #20169
1834    /// for examples of where this comes up,.
1835    fn rvalue_hint(ctx: &mut InferenceContext<'_, 'db>, ty: Ty<'db>) -> Self {
1836        match ctx.struct_tail_without_normalization(ty).kind() {
1837            TyKind::Slice(_) | TyKind::Str | TyKind::Dynamic(..) => {
1838                Expectation::RValueLikeUnsized(ty)
1839            }
1840            _ => Expectation::has_type(ty),
1841        }
1842    }
1843
1844    /// This expresses no expectation on the type.
1845    fn none() -> Self {
1846        Expectation::None
1847    }
1848
1849    fn resolve(&self, table: &mut unify::InferenceTable<'db>) -> Expectation<'db> {
1850        match self {
1851            Expectation::None => Expectation::None,
1852            Expectation::HasType(t) => Expectation::HasType(table.shallow_resolve(*t)),
1853            Expectation::Castable(t) => Expectation::Castable(table.shallow_resolve(*t)),
1854            Expectation::RValueLikeUnsized(t) => {
1855                Expectation::RValueLikeUnsized(table.shallow_resolve(*t))
1856            }
1857        }
1858    }
1859
1860    fn to_option(&self, table: &mut unify::InferenceTable<'db>) -> Option<Ty<'db>> {
1861        match self.resolve(table) {
1862            Expectation::None => None,
1863            Expectation::HasType(t)
1864            | Expectation::Castable(t)
1865            | Expectation::RValueLikeUnsized(t) => Some(t),
1866        }
1867    }
1868
1869    fn only_has_type(&self, table: &mut unify::InferenceTable<'db>) -> Option<Ty<'db>> {
1870        match self {
1871            Expectation::HasType(t) => Some(table.shallow_resolve(*t)),
1872            Expectation::Castable(_) | Expectation::RValueLikeUnsized(_) | Expectation::None => {
1873                None
1874            }
1875        }
1876    }
1877
1878    fn coercion_target_type(&self, table: &mut unify::InferenceTable<'db>) -> Ty<'db> {
1879        self.only_has_type(table).unwrap_or_else(|| table.next_ty_var())
1880    }
1881
1882    /// Comment copied from rustc:
1883    /// Disregard "castable to" expectations because they
1884    /// can lead us astray. Consider for example `if cond
1885    /// {22} else {c} as u8` -- if we propagate the
1886    /// "castable to u8" constraint to 22, it will pick the
1887    /// type 22u8, which is overly constrained (c might not
1888    /// be a u8). In effect, the problem is that the
1889    /// "castable to" expectation is not the tightest thing
1890    /// we can say, so we want to drop it in this case.
1891    /// The tightest thing we can say is "must unify with
1892    /// else branch". Note that in the case of a "has type"
1893    /// constraint, this limitation does not hold.
1894    ///
1895    /// If the expected type is just a type variable, then don't use
1896    /// an expected type. Otherwise, we might write parts of the type
1897    /// when checking the 'then' block which are incompatible with the
1898    /// 'else' branch.
1899    fn adjust_for_branches(&self, table: &mut unify::InferenceTable<'db>) -> Expectation<'db> {
1900        match *self {
1901            Expectation::HasType(ety) => {
1902                let ety = table.structurally_resolve_type(ety);
1903                if ety.is_ty_var() { Expectation::None } else { Expectation::HasType(ety) }
1904            }
1905            Expectation::RValueLikeUnsized(ety) => Expectation::RValueLikeUnsized(ety),
1906            _ => Expectation::None,
1907        }
1908    }
1909}
