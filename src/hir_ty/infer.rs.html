1//! Type inference, i.e. the process of walking through the code and determining
2//! the type of each expression and pattern.
3//!
4//! For type inference, compare the implementations in rustc (the various
5//! check_* methods in rustc_hir_analysis/check/mod.rs are a good entry point) and
6//! IntelliJ-Rust (org.rust.lang.core.types.infer). Our entry point for
7//! inference here is the `infer` function, which infers the types of all
8//! expressions in a given function.
9//!
10//! During inference, types (i.e. the `Ty` struct) can contain type 'variables'
11//! which represent currently unknown types; as we walk through the expressions,
12//! we might determine that certain variables need to be equal to each other, or
13//! to certain types. To record this, we use the union-find implementation from
14//! the `ena` crate, which is extracted from rustc.

100pub fn infer_query_with_inspect<'db>(
101    db: &'db dyn HirDatabase,
102    def: DefWithBodyId,
103    inspect: Option<ObligationInspector<'db>>,
104) -> InferenceResult {
105    let _p = tracing::info_span!("infer_query").entered();
106    let resolver = def.resolver(db);
107    let body = db.body(def);
108    let mut ctx = InferenceContext::new(db, def, &body, resolver);
109
110    if let Some(inspect) = inspect {
111        ctx.table.infer_ctxt.attach_obligation_inspector(inspect);
112    }
113
114    match def {
115        DefWithBodyId::FunctionId(f) => {
116            ctx.collect_fn(f);
117        }
118        DefWithBodyId::ConstId(c) => ctx.collect_const(c, &db.const_signature(c)),
119        DefWithBodyId::StaticId(s) => ctx.collect_static(&db.static_signature(s)),
120        DefWithBodyId::VariantId(v) => {
121            ctx.return_ty = match EnumSignature::variant_body_type(db, v.lookup(db).parent) {
122                hir_def::layout::IntegerType::Pointer(signed) => match signed {
123                    true => ctx.types.types.isize,
124                    false => ctx.types.types.usize,
125                },
126                hir_def::layout::IntegerType::Fixed(size, signed) => match signed {
127                    true => match size {
128                        Integer::I8 => ctx.types.types.i8,
129                        Integer::I16 => ctx.types.types.i16,
130                        Integer::I32 => ctx.types.types.i32,
131                        Integer::I64 => ctx.types.types.i64,
132                        Integer::I128 => ctx.types.types.i128,
133                    },
134                    false => match size {
135                        Integer::I8 => ctx.types.types.u8,
136                        Integer::I16 => ctx.types.types.u16,
137                        Integer::I32 => ctx.types.types.u32,
138                        Integer::I64 => ctx.types.types.u64,
139                        Integer::I128 => ctx.types.types.u128,
140                    },
141                },
142            };
143        }
144    }
145
146    ctx.infer_body();
147
148    ctx.infer_mut_body();
149
150    ctx.handle_opaque_type_uses();
151
152    ctx.type_inference_fallback();
153
154    // Comment from rustc:
155    // Even though coercion casts provide type hints, we check casts after fallback for
156    // backwards compatibility. This makes fallback a stronger type hint than a cast coercion.
157    let cast_checks = std::mem::take(&mut ctx.deferred_cast_checks);
158    for mut cast in cast_checks.into_iter() {
159        if let Err(diag) = cast.check(&mut ctx) {
160            ctx.diagnostics.push(diag);
161        }
162    }
163
164    ctx.table.select_obligations_where_possible();
165
166    ctx.infer_closures();
167
168    ctx.table.select_obligations_where_possible();
169
170    ctx.handle_opaque_type_uses();
171
172    ctx.resolve_all()
173}

309/// Represents coercing a value to a different type of value.
310///
311/// We transform values by following a number of `Adjust` steps in order.
312/// See the documentation on variants of `Adjust` for more details.
313///
314/// Here are some common scenarios:
315///
316/// 1. The simplest cases are where a pointer is not adjusted fat vs thin.
317///    Here the pointer will be dereferenced N times (where a dereference can
318///    happen to raw or borrowed pointers or any smart pointer which implements
319///    Deref, including Box<_>). The types of dereferences is given by
320///    `autoderefs`. It can then be auto-referenced zero or one times, indicated
321///    by `autoref`, to either a raw or borrowed pointer. In these cases unsize is
322///    `false`.
323///
324/// 2. A thin-to-fat coercion involves unsizing the underlying data. We start
325///    with a thin pointer, deref a number of times, unsize the underlying data,
326///    then autoref. The 'unsize' phase may change a fixed length array to a
327///    dynamically sized one, a concrete object to a trait object, or statically
328///    sized struct to a dynamically sized one. E.g., &[i32; 4] -> &[i32] is
329///    represented by:
330///
331///    ```ignore
332///    Deref(None) -> [i32; 4],
333///    Borrow(AutoBorrow::Ref) -> &[i32; 4],
334///    Unsize -> &[i32],
335///    ```
336///
337///    Note that for a struct, the 'deep' unsizing of the struct is not recorded.
338///    E.g., `struct Foo<T> { it: T }` we can coerce &Foo<[i32; 4]> to &Foo<[i32]>
339///    The autoderef and -ref are the same as in the above example, but the type
340///    stored in `unsize` is `Foo<[i32]>`, we don't store any further detail about
341///    the underlying conversions from `[i32; 4]` to `[i32]`.
342///
343/// 3. Coercing a `Box<T>` to `Box<dyn Trait>` is an interesting special case. In
344///    that case, we have the pointer we need coming in, so there are no
345///    autoderefs, and no autoref. Instead we just do the `Unsize` transformation.
346///    At some point, of course, `Box` should move out of the compiler, in which
347///    case this is analogous to transforming a struct. E.g., Box<[i32; 4]> ->
348///    Box<[i32]> is an `Adjust::Unsize` with the target `Box<[i32]>`.
349#[derive(Clone, Debug, PartialEq, Eq, Hash)]
350pub struct Adjustment {
351    pub kind: Adjust,
352    pub target: StoredTy,
353}
354
355impl Adjustment {
356    pub fn borrow<'db>(
357        interner: DbInterner<'db>,
358        m: Mutability,
359        ty: Ty<'db>,
360        lt: Region<'db>,
361    ) -> Self {
362        let ty = Ty::new_ref(interner, lt, ty, m);
363        Adjustment {
364            kind: Adjust::Borrow(AutoBorrow::Ref(AutoBorrowMutability::new(m, AllowTwoPhase::No))),
365            target: ty.store(),
366        }
367    }
368}

370/// At least for initial deployment, we want to limit two-phase borrows to
371/// only a few specific cases. Right now, those are mostly "things that desugar"
372/// into method calls:
373/// - using `x.some_method()` syntax, where some_method takes `&mut self`,
374/// - using `Foo::some_method(&mut x, ...)` syntax,
375/// - binary assignment operators (`+=`, `-=`, `*=`, etc.).
376///
377/// Anything else should be rejected until generalized two-phase borrow support
378/// is implemented. Right now, dataflow can't handle the general case where there
379/// is more than one use of a mutable borrow, and we don't want to accept too much
380/// new code via two-phase borrows, so we try to limit where we create two-phase
381/// capable mutable borrows.
382/// See #49434 for tracking.
383#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
384pub enum AllowTwoPhase {
385    // FIXME: We should use this when appropriate.
386    Yes,
387    No,
388}

811#[derive(Clone, Debug)]
812struct BreakableContext<'db> {
813    /// Whether this context contains at least one break expression.
814    may_break: bool,
815    /// The coercion target of the context.
816    coerce: Option<DynamicCoerceMany<'db>>,
817    /// The optional label of the context.

822#[derive(Clone, Debug)]
823enum BreakableKind {
824    Block,
825    Loop,
826    /// A border is something like an async block, closure etc. Anything that prevents
827    /// breaking/continuing through
828    Border,
829}
830
831fn find_breakable<'a, 'db>(
832    ctxs: &'a mut [BreakableContext<'db>],
833    label: Option<LabelId>,
834) -> Option<&'a mut BreakableContext<'db>> {
835    let mut ctxs = ctxs
836        .iter_mut()
837        .rev()
838        .take_while(|it| matches!(it.kind, BreakableKind::Block | BreakableKind::Loop));
839    match label {
840        Some(_) => ctxs.find(|ctx| ctx.label == label),
841        None => ctxs.find(|ctx| matches!(ctx.kind, BreakableKind::Loop)),
842    }
843}
844
845fn find_continuable<'a, 'db>(
846    ctxs: &'a mut [BreakableContext<'db>],
847    label: Option<LabelId>,
848) -> Option<&'a mut BreakableContext<'db>> {
849    match label {
850        Some(_) => find_breakable(ctxs, label).filter(|it| matches!(it.kind, BreakableKind::Loop)),
851        None => find_breakable(ctxs, label),
852    }
853}

928    /// Clones `self` and calls `resolve_all()` on it.

930    pub(crate) fn fixme_resolve_all_clone(&self) -> InferenceResult {
931        let mut ctx = self.clone();

933        ctx.type_inference_fallback();

950    // FIXME: This function should be private in module. It is currently only used in the consteval, since we need
951    // `InferenceResult` in the middle of inference. See the fixme comment in `consteval::eval_to_const`. If you
952    // used this function for another workaround, mention it here. If you really need this function and believe that
953    // there is no problem in it being `pub(crate)`, remove this comment.
954    fn resolve_all(self) -> InferenceResult {
955        let InferenceContext {
956            mut table, mut result, tuple_field_accesses_rev, diagnostics, ..
957        } = self;
958        let mut diagnostics = diagnostics.finish();
959        // Destructure every single field so whenever new fields are added to `InferenceResult` we
960        // don't forget to handle them here.
961        let InferenceResult {
962            method_resolutions,
963            field_resolutions: _,
964            variant_resolutions: _,
965            assoc_resolutions,
966            type_of_expr,
967            type_of_pat,
968            type_of_binding,
969            type_of_type_placeholder,
970            type_of_opaque,
971            type_mismatches,
972            has_errors,
973            error_ty: _,
974            pat_adjustments,
975            binding_modes: _,
976            expr_adjustments,
977            // Types in `closure_info` have already been `resolve_completely()`'d during
978            // `InferenceContext::infer_closures()` (in `HirPlace::ty()` specifically), so no need
979            // to resolve them here.
980            closure_info: _,
981            mutated_bindings_in_closure: _,
982            tuple_field_access_types: _,
983            coercion_casts: _,
984            diagnostics: _,
985        } = &mut result;
986
987        for ty in type_of_expr.values_mut() {
988            *ty = table.resolve_completely(ty.as_ref()).store();
989            *has_errors = *has_errors || ty.as_ref().references_non_lt_error();
990        }
991        type_of_expr.shrink_to_fit();
992        for ty in type_of_pat.values_mut() {
993            *ty = table.resolve_completely(ty.as_ref()).store();
994            *has_errors = *has_errors || ty.as_ref().references_non_lt_error();
995        }
996        type_of_pat.shrink_to_fit();
997        for ty in type_of_binding.values_mut() {
998            *ty = table.resolve_completely(ty.as_ref()).store();
999            *has_errors = *has_errors || ty.as_ref().references_non_lt_error();
1000        }
1001        type_of_binding.shrink_to_fit();
1002        for ty in type_of_type_placeholder.values_mut() {
1003            *ty = table.resolve_completely(ty.as_ref()).store();
1004            *has_errors = *has_errors || ty.as_ref().references_non_lt_error();
1005        }
1006        type_of_type_placeholder.shrink_to_fit();
1007        type_of_opaque.shrink_to_fit();
1008
1009        if let Some(type_mismatches) = type_mismatches {
1010            *has_errors = true;
1011            for mismatch in type_mismatches.values_mut() {
1012                mismatch.expected = table.resolve_completely(mismatch.expected.as_ref()).store();
1013                mismatch.actual = table.resolve_completely(mismatch.actual.as_ref()).store();
1014            }
1015            type_mismatches.shrink_to_fit();
1016        }
1017        diagnostics.retain_mut(|diagnostic| {
1018            use InferenceDiagnostic::*;
1019            match diagnostic {
1020                ExpectedFunction { found: ty, .. }
1021                | UnresolvedField { receiver: ty, .. }
1022                | UnresolvedMethodCall { receiver: ty, .. } => {
1023                    *ty = table.resolve_completely(ty.as_ref()).store();
1024                    // FIXME: Remove this when we are on par with rustc in terms of inference
1025                    if ty.as_ref().references_non_lt_error() {
1026                        return false;
1027                    }
1028
1029                    if let UnresolvedMethodCall { field_with_same_name, .. } = diagnostic
1030                        && let Some(ty) = field_with_same_name
1031                    {
1032                        *ty = table.resolve_completely(ty.as_ref()).store();
1033                        if ty.as_ref().references_non_lt_error() {
1034                            *field_with_same_name = None;
1035                        }
1036                    }
1037                }
1038                TypedHole { expected: ty, .. } => {
1039                    *ty = table.resolve_completely(ty.as_ref()).store();
1040                }
1041                _ => (),
1042            }
1043            true
1044        });
1045        diagnostics.shrink_to_fit();
1046        for (_, subst) in method_resolutions.values_mut() {
1047            *subst = table.resolve_completely(subst.as_ref()).store();
1048            *has_errors =
1049                *has_errors || subst.as_ref().types().any(|ty| ty.references_non_lt_error());
1050        }
1051        method_resolutions.shrink_to_fit();
1052        for (_, subst) in assoc_resolutions.values_mut() {
1053            *subst = table.resolve_completely(subst.as_ref()).store();
1054            *has_errors =
1055                *has_errors || subst.as_ref().types().any(|ty| ty.references_non_lt_error());
1056        }
1057        assoc_resolutions.shrink_to_fit();
1058        for adjustment in expr_adjustments.values_mut().flatten() {
1059            adjustment.target = table.resolve_completely(adjustment.target.as_ref()).store();
1060            *has_errors = *has_errors || adjustment.target.as_ref().references_non_lt_error();
1061        }
1062        expr_adjustments.shrink_to_fit();
1063        for adjustment in pat_adjustments.values_mut().flatten() {
1064            *adjustment = table.resolve_completely(adjustment.as_ref()).store();
1065            *has_errors = *has_errors || adjustment.as_ref().references_non_lt_error();
1066        }
1067        pat_adjustments.shrink_to_fit();
1068        result.tuple_field_access_types = tuple_field_accesses_rev
1069            .into_iter()
1070            .map(|subst| table.resolve_completely(subst).store())
1071            .inspect(|subst| {
1072                *has_errors =
1073                    *has_errors || subst.as_ref().iter().any(|ty| ty.references_non_lt_error());
1074            })
1075            .collect();
1076        result.tuple_field_access_types.shrink_to_fit();
1077
1078        result.diagnostics = diagnostics;
1079
1080        result
1081    }

1105    fn collect_fn(&mut self, func: FunctionId) {
1106        let data = self.db.function_signature(func);
1107        let mut param_tys = self.with_ty_lowering(
1108            &data.store,
1109            InferenceTyDiagnosticSource::Signature,
1110            LifetimeElisionKind::for_fn_params(&data),
1111            |ctx| data.params.iter().map(|&type_ref| ctx.lower_ty(type_ref)).collect::<Vec<_>>(),
1112        );
1113
1114        // Check if function contains a va_list, if it does then we append it to the parameter types
1115        // that are collected from the function data
1116        if data.is_varargs() {
1117            let va_list_ty = match self.resolve_va_list() {
1118                Some(va_list) => Ty::new_adt(
1119                    self.interner(),
1120                    va_list,
1121                    GenericArgs::for_item_with_defaults(
1122                        self.interner(),
1123                        va_list.into(),
1124                        |_, id, _| self.table.next_var_for_param(id),
1125                    ),
1126                ),
1127                None => self.err_ty(),
1128            };
1129
1130            param_tys.push(va_list_ty);
1131        }
1132        let mut param_tys = param_tys.into_iter().chain(iter::repeat(self.table.next_ty_var()));
1133        if let Some(self_param) = self.body.self_param
1134            && let Some(ty) = param_tys.next()
1135        {
1136            let ty = self.process_user_written_ty(ty);
1137            self.write_binding_ty(self_param, ty);
1138        }
1139        for (ty, pat) in param_tys.zip(&*self.body.params) {
1140            let ty = self.process_user_written_ty(ty);
1141
1142            self.infer_top_pat(*pat, ty, None);
1143        }
1144        self.return_ty = match data.ret_type {

1292    fn make_ty(
1293        &mut self,
1294        type_ref: TypeRefId,
1295        store: &ExpressionStore,
1296        type_source: InferenceTyDiagnosticSource,
1297        lifetime_elision: LifetimeElisionKind<'db>,
1298    ) -> Ty<'db> {
1299        let ty = self
1300            .with_ty_lowering(store, type_source, lifetime_elision, |ctx| ctx.lower_ty(type_ref));
1301        let ty = self.process_user_written_ty(ty);
1302
1303        // Record the association from placeholders' TypeRefId to type variables.
1304        // We only record them if their number matches. This assumes TypeRef::walk and TypeVisitable process the items in the same order.
1305        let type_variables = collect_type_inference_vars(&ty);
1306        let mut placeholder_ids = vec![];
1307        TypeRef::walk(type_ref, store, &mut |type_ref_id, type_ref| {
1308            if matches!(type_ref, TypeRef::Placeholder) {
1309                placeholder_ids.push(type_ref_id);
1310            }
1311        });
1312
1313        if placeholder_ids.len() == type_variables.len() {
1314            for (placeholder_id, type_variable) in
1315                placeholder_ids.into_iter().zip(type_variables.into_iter())
1316            {
1317                self.write_type_placeholder_ty(placeholder_id, type_variable);
1318            }
1319        }
1320
1321        ty
1322    }

1390    /// Returns the deeply last field of nested structures, or the same type if
1391    /// not a structure at all. Corresponds to the only possible unsized field,
1392    /// and its type can be used to determine unsizing strategy.
1393    ///
1394    /// This is parameterized over the normalization strategy (i.e. how to
1395    /// handle `<T as Trait>::Assoc` and `impl Trait`); pass the identity
1396    /// function to indicate no normalization should take place.
1397    fn struct_tail_with_normalize(
1398        &mut self,
1399        mut ty: Ty<'db>,
1400        mut normalize: impl FnMut(Ty<'db>) -> Ty<'db>,
1401    ) -> Ty<'db> {
1402        // FIXME: fetch the limit properly
1403        let recursion_limit = 10;
1404        for iteration in 0.. {
1405            if iteration > recursion_limit {
1406                return self.err_ty();
1407            }
1408            match ty.kind() {
1409                TyKind::Adt(adt_def, substs) => match adt_def.def_id().0 {
1410                    AdtId::StructId(struct_id) => {
1411                        match self
1412                            .db
1413                            .field_types(struct_id.into())
1414                            .values()
1415                            .next_back()
1416                            .map(|it| it.get())
1417                        {
1418                            Some(field) => {
1419                                ty = field.instantiate(self.interner(), substs);
1420                            }
1421                            None => break,
1422                        }
1423                    }
1424                    _ => break,
1425                },
1426                TyKind::Tuple(substs) => match substs.as_slice().split_last() {
1427                    Some((last_ty, _)) => ty = *last_ty,
1428                    None => break,
1429                },
1430                TyKind::Alias(..) => {
1431                    let normalized = normalize(ty);
1432                    if ty == normalized {
1433                        return ty;
1434                    } else {
1435                        ty = normalized;
1436                    }
1437                }
1438                _ => break,
1439            }
1440        }
1441        ty
1442    }

1561    fn resolve_variant(
1562        &mut self,
1563        node: ExprOrPatId,
1564        path: Option<&Path>,
1565        value_ns: bool,
1566    ) -> (Ty<'db>, Option<VariantId>) {
1567        let path = match path {
1568            Some(path) => path,
1569            None => return (self.err_ty(), None),
1570        };
1571        let mut ctx = TyLoweringContext::new(
1572            self.db,
1573            &self.resolver,
1574            &self.body.store,
1575            &self.diagnostics,
1576            InferenceTyDiagnosticSource::Body,
1577            self.generic_def,
1578            LifetimeElisionKind::Infer,
1579        );
1580        let mut path_ctx = ctx.at_path(path, node);
1581        let interner = DbInterner::conjure();
1582        let (resolution, unresolved) = if value_ns {
1583            let Some(res) = path_ctx.resolve_path_in_value_ns(HygieneId::ROOT) else {
1584                return (self.err_ty(), None);
1585            };
1586            match res {
1587                ResolveValueResult::ValueNs(value) => match value {
1588                    ValueNs::EnumVariantId(var) => {
1589                        let args = path_ctx.substs_from_path(var.into(), true, false);
1590                        drop(ctx);
1591                        let ty = self
1592                            .db
1593                            .ty(var.lookup(self.db).parent.into())
1594                            .instantiate(interner, args);
1595                        let ty = self.insert_type_vars(ty);
1596                        return (ty, Some(var.into()));
1597                    }
1598                    ValueNs::StructId(strukt) => {
1599                        let args = path_ctx.substs_from_path(strukt.into(), true, false);
1600                        drop(ctx);
1601                        let ty = self.db.ty(strukt.into()).instantiate(interner, args);
1602                        let ty = self.insert_type_vars(ty);
1603                        return (ty, Some(strukt.into()));
1604                    }
1605                    ValueNs::ImplSelf(impl_id) => (TypeNs::SelfType(impl_id), None),
1606                    _ => {
1607                        drop(ctx);
1608                        return (self.err_ty(), None);
1609                    }
1610                },
1611                ResolveValueResult::Partial(typens, unresolved) => (typens, Some(unresolved)),
1612            }
1613        } else {
1614            match path_ctx.resolve_path_in_type_ns() {
1615                Some((it, idx)) => (it, idx),
1616                None => return (self.err_ty(), None),
1617            }
1618        };
1619        return match resolution {
1620            TypeNs::AdtId(AdtId::StructId(strukt)) => {
1621                let args = path_ctx.substs_from_path(strukt.into(), true, false);
1622                drop(ctx);
1623                let ty = self.db.ty(strukt.into()).instantiate(interner, args);
1624                let ty = self.insert_type_vars(ty);
1625                forbid_unresolved_segments(self, (ty, Some(strukt.into())), unresolved)
1626            }
1627            TypeNs::AdtId(AdtId::UnionId(u)) => {
1628                let args = path_ctx.substs_from_path(u.into(), true, false);
1629                drop(ctx);
1630                let ty = self.db.ty(u.into()).instantiate(interner, args);
1631                let ty = self.insert_type_vars(ty);
1632                forbid_unresolved_segments(self, (ty, Some(u.into())), unresolved)
1633            }
1634            TypeNs::EnumVariantId(var) => {
1635                let args = path_ctx.substs_from_path(var.into(), true, false);
1636                drop(ctx);
1637                let ty = self.db.ty(var.lookup(self.db).parent.into()).instantiate(interner, args);
1638                let ty = self.insert_type_vars(ty);
1639                forbid_unresolved_segments(self, (ty, Some(var.into())), unresolved)
1640            }
1641            TypeNs::SelfType(impl_id) => {
1642                let mut ty = self.db.impl_self_ty(impl_id).instantiate_identity();
1643
1644                let Some(remaining_idx) = unresolved else {
1645                    drop(ctx);
1646                    let Some(mod_path) = path.mod_path() else {
1647                        never!("resolver should always resolve lang item paths");
1648                        return (self.err_ty(), None);
1649                    };
1650                    return self.resolve_variant_on_alias(ty, None, mod_path);
1651                };
1652
1653                let mut remaining_segments = path.segments().skip(remaining_idx);
1654
1655                if remaining_segments.len() >= 2 {
1656                    path_ctx.ignore_last_segment();
1657                }
1658
1659                // We need to try resolving unresolved segments one by one because each may resolve
1660                // to a projection, which `TyLoweringContext` cannot handle on its own.
1661                let mut tried_resolving_once = false;
1662                while let Some(current_segment) = remaining_segments.first() {
1663                    // If we can resolve to an enum variant, it takes priority over associated type
1664                    // of the same name.
1665                    if let TyKind::Adt(adt_def, _) = ty.kind()
1666                        && let AdtId::EnumId(id) = adt_def.def_id().0
1667                    {
1668                        let enum_data = id.enum_variants(self.db);
1669                        if let Some(variant) = enum_data.variant(current_segment.name) {
1670                            return if remaining_segments.len() == 1 {
1671                                (ty, Some(variant.into()))
1672                            } else {
1673                                // We still have unresolved paths, but enum variants never have
1674                                // associated types!
1675                                // FIXME: Report an error.
1676                                (self.err_ty(), None)
1677                            };
1678                        }
1679                    }
1680
1681                    if tried_resolving_once {
1682                        // FIXME: with `inherent_associated_types` this is allowed, but our `lower_partly_resolved_path()`
1683                        // will need to be updated to err at the correct segment.
1684                        break;
1685                    }
1686
1687                    // `lower_partly_resolved_path()` returns `None` as type namespace unless
1688                    // `remaining_segments` is empty, which is never the case here. We don't know
1689                    // which namespace the new `ty` is in until normalized anyway.
1690                    (ty, _) = path_ctx.lower_partly_resolved_path(resolution, true);
1691                    tried_resolving_once = true;
1692
1693                    ty = self.table.process_user_written_ty(ty);
1694                    if ty.is_ty_error() {
1695                        return (self.err_ty(), None);
1696                    }
1697
1698                    remaining_segments = remaining_segments.skip(1);
1699                }
1700                drop(ctx);
1701
1702                let variant = ty.as_adt().and_then(|(id, _)| match id {
1703                    AdtId::StructId(s) => Some(VariantId::StructId(s)),
1704                    AdtId::UnionId(u) => Some(VariantId::UnionId(u)),
1705                    AdtId::EnumId(_) => {
1706                        // FIXME Error E0071, expected struct, variant or union type, found enum `Foo`
1707                        None
1708                    }
1709                });
1710                (ty, variant)
1711            }
1712            TypeNs::TypeAliasId(it) => {
1713                let Some(mod_path) = path.mod_path() else {
1714                    never!("resolver should always resolve lang item paths");
1715                    return (self.err_ty(), None);
1716                };
1717                let args = path_ctx.substs_from_path_segment(it.into(), true, None, false);
1718                drop(ctx);
1719                let interner = DbInterner::conjure();
1720                let ty = self.db.ty(it.into()).instantiate(interner, args);
1721                let ty = self.insert_type_vars(ty);
1722
1723                self.resolve_variant_on_alias(ty, unresolved, mod_path)
1724            }
1725            TypeNs::AdtSelfType(_) => {
1726                // FIXME this could happen in array size expressions, once we're checking them
1727                (self.err_ty(), None)
1728            }
1729            TypeNs::GenericParam(_) => {
1730                // FIXME potentially resolve assoc type
1731                (self.err_ty(), None)
1732            }
1733            TypeNs::AdtId(AdtId::EnumId(_))
1734            | TypeNs::BuiltinType(_)
1735            | TypeNs::TraitId(_)
1736            | TypeNs::ModuleId(_) => {
1737                // FIXME diagnostic
1738                (self.err_ty(), None)
1739            }
1740        };
1741
1742        fn forbid_unresolved_segments<'db>(
1743            ctx: &InferenceContext<'_, 'db>,
1744            result: (Ty<'db>, Option<VariantId>),
1745            unresolved: Option<usize>,

1878/// When inferring an expression, we propagate downward whatever type hint we
1879/// are able in the form of an `Expectation`.
1880#[derive(Clone, PartialEq, Eq, Debug)]
1881pub(crate) enum Expectation<'db> {
1882    None,
1883    HasType(Ty<'db>),
1884    Castable(Ty<'db>),
1885    RValueLikeUnsized(Ty<'db>),
1886}
1887
1888impl<'db> Expectation<'db> {
1889    /// The expectation that the type of the expression needs to equal the given
1890    /// type.
1891    fn has_type(ty: Ty<'db>) -> Self {
1892        if ty.is_ty_error() {
1893            // FIXME: get rid of this?
1894            Expectation::None
1895        } else {
1896            Expectation::HasType(ty)
1897        }
1898    }
1899
1900    /// The following explanation is copied straight from rustc:
1901    /// Provides an expectation for an rvalue expression given an *optional*
1902    /// hint, which is not required for type safety (the resulting type might
1903    /// be checked higher up, as is the case with `&expr` and `box expr`), but
1904    /// is useful in determining the concrete type.
1905    ///
1906    /// The primary use case is where the expected type is a fat pointer,
1907    /// like `&[isize]`. For example, consider the following statement:
1908    ///
1909    ///     let it: &[isize] = &[1, 2, 3];
1910    ///
1911    /// In this case, the expected type for the `&[1, 2, 3]` expression is
1912    /// `&[isize]`. If however we were to say that `[1, 2, 3]` has the
1913    /// expectation `ExpectHasType([isize])`, that would be too strong --
1914    /// `[1, 2, 3]` does not have the type `[isize]` but rather `[isize; 3]`.
1915    /// It is only the `&[1, 2, 3]` expression as a whole that can be coerced
1916    /// to the type `&[isize]`. Therefore, we propagate this more limited hint,
1917    /// which still is useful, because it informs integer literals and the like.
1918    /// See the test case `test/ui/coerce-expect-unsized.rs` and #20169
1919    /// for examples of where this comes up,.
1920    fn rvalue_hint(ctx: &mut InferenceContext<'_, 'db>, ty: Ty<'db>) -> Self {
1921        match ctx.struct_tail_without_normalization(ty).kind() {
1922            TyKind::Slice(_) | TyKind::Str | TyKind::Dynamic(..) => {
1923                Expectation::RValueLikeUnsized(ty)
1924            }
1925            _ => Expectation::has_type(ty),
1926        }
1927    }
1928
1929    /// This expresses no expectation on the type.
1930    fn none() -> Self {
1931        Expectation::None
1932    }
1933
1934    fn resolve(&self, table: &mut unify::InferenceTable<'db>) -> Expectation<'db> {
1935        match self {
1936            Expectation::None => Expectation::None,
1937            Expectation::HasType(t) => Expectation::HasType(table.shallow_resolve(*t)),
1938            Expectation::Castable(t) => Expectation::Castable(table.shallow_resolve(*t)),
1939            Expectation::RValueLikeUnsized(t) => {
1940                Expectation::RValueLikeUnsized(table.shallow_resolve(*t))
1941            }
1942        }
1943    }
1944
1945    fn to_option(&self, table: &mut unify::InferenceTable<'db>) -> Option<Ty<'db>> {
1946        match self.resolve(table) {
1947            Expectation::None => None,
1948            Expectation::HasType(t)
1949            | Expectation::Castable(t)
1950            | Expectation::RValueLikeUnsized(t) => Some(t),
1951        }
1952    }
1953
1954    fn only_has_type(&self, table: &mut unify::InferenceTable<'db>) -> Option<Ty<'db>> {
1955        match self {
1956            Expectation::HasType(t) => Some(table.shallow_resolve(*t)),
1957            Expectation::Castable(_) | Expectation::RValueLikeUnsized(_) | Expectation::None => {
1958                None
1959            }
1960        }
1961    }
1962
1963    fn coercion_target_type(&self, table: &mut unify::InferenceTable<'db>) -> Ty<'db> {
1964        self.only_has_type(table).unwrap_or_else(|| table.next_ty_var())
1965    }
1966
1967    /// Comment copied from rustc:
1968    /// Disregard "castable to" expectations because they
1969    /// can lead us astray. Consider for example `if cond
1970    /// {22} else {c} as u8` -- if we propagate the
1971    /// "castable to u8" constraint to 22, it will pick the
1972    /// type 22u8, which is overly constrained (c might not
1973    /// be a u8). In effect, the problem is that the
1974    /// "castable to" expectation is not the tightest thing
1975    /// we can say, so we want to drop it in this case.
1976    /// The tightest thing we can say is "must unify with
1977    /// else branch". Note that in the case of a "has type"
1978    /// constraint, this limitation does not hold.
1979    ///
1980    /// If the expected type is just a type variable, then don't use
1981    /// an expected type. Otherwise, we might write parts of the type
1982    /// when checking the 'then' block which are incompatible with the
1983    /// 'else' branch.
1984    fn adjust_for_branches(&self, table: &mut unify::InferenceTable<'db>) -> Expectation<'db> {
1985        match *self {
1986            Expectation::HasType(ety) => {
1987                let ety = table.structurally_resolve_type(ety);
1988                if ety.is_ty_var() { Expectation::None } else { Expectation::HasType(ety) }
1989            }
1990            Expectation::RValueLikeUnsized(ety) => Expectation::RValueLikeUnsized(ety),
1991            _ => Expectation::None,
1992        }
1993    }
1994}
