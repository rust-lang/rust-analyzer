1//! Type inference, i.e. the process of walking through the code and determining
2//! the type of each expression and pattern.
3//!
4//! For type inference, compare the implementations in rustc (the various
5//! check_* methods in rustc_hir_analysis/check/mod.rs are a good entry point) and
6//! IntelliJ-Rust (org.rust.lang.core.types.infer). Our entry point for
7//! inference here is the `infer` function, which infers the types of all
8//! expressions in a given function.
9//!
10//! During inference, types (i.e. the `Ty` struct) can contain type 'variables'
11//! which represent currently unknown types; as we walk through the expressions,
12//! we might determine that certain variables need to be equal to each other, or
13//! to certain types. To record this, we use the union-find implementation from
14//! the `ena` crate, which is extracted from rustc.

87/// The entry point of type inference.
88pub(crate) fn infer_query(db: &dyn HirDatabase, def: DefWithBodyId) -> Arc<InferenceResult> {
89    let _p = tracing::info_span!("infer_query").entered();
90    let resolver = def.resolver(db);
91    let body = db.body(def);
92    let mut ctx = InferenceContext::new(db, def, &body, resolver);
93
94    match def {
95        DefWithBodyId::FunctionId(f) => {
96            ctx.collect_fn(f);
97        }
98        DefWithBodyId::ConstId(c) => ctx.collect_const(&db.const_signature(c)),
99        DefWithBodyId::StaticId(s) => ctx.collect_static(&db.static_signature(s)),
100        DefWithBodyId::VariantId(v) => {
101            ctx.return_ty = TyBuilder::builtin(
102                match db.enum_signature(v.lookup(db).parent).variant_body_type() {
103                    hir_def::layout::IntegerType::Pointer(signed) => match signed {
104                        true => BuiltinType::Int(BuiltinInt::Isize),
105                        false => BuiltinType::Uint(BuiltinUint::Usize),
106                    },
107                    hir_def::layout::IntegerType::Fixed(size, signed) => match signed {
108                        true => BuiltinType::Int(match size {
109                            Integer::I8 => BuiltinInt::I8,
110                            Integer::I16 => BuiltinInt::I16,
111                            Integer::I32 => BuiltinInt::I32,
112                            Integer::I64 => BuiltinInt::I64,
113                            Integer::I128 => BuiltinInt::I128,
114                        }),
115                        false => BuiltinType::Uint(match size {
116                            Integer::I8 => BuiltinUint::U8,
117                            Integer::I16 => BuiltinUint::U16,
118                            Integer::I32 => BuiltinUint::U32,
119                            Integer::I64 => BuiltinUint::U64,
120                            Integer::I128 => BuiltinUint::U128,
121                        }),
122                    },
123                },
124            );
125        }
126    }
127
128    ctx.infer_body();
129
130    ctx.infer_mut_body();
131
132    ctx.infer_closures();
133
134    Arc::new(ctx.resolve_all())
135}

137/// Fully normalize all the types found within `ty` in context of `owner` body definition.
138///
139/// This is appropriate to use only after type-check: it assumes
140/// that normalization will succeed, for example.
141pub(crate) fn normalize(db: &dyn HirDatabase, trait_env: Arc<TraitEnvironment>, ty: Ty) -> Ty {
142    // FIXME: TypeFlags::HAS_CT_PROJECTION is not implemented in chalk, so TypeFlags::HAS_PROJECTION only
143    // works for the type case, so we check array unconditionally. Remove the array part
144    // when the bug in chalk becomes fixed.
145    if !ty.data(Interner).flags.intersects(TypeFlags::HAS_PROJECTION)
146        && !matches!(ty.kind(Interner), TyKind::Array(..))
147    {
148        return ty;
149    }
150    let mut table = unify::InferenceTable::new(db, trait_env);
151
152    let ty_with_vars = table.normalize_associated_types_in(ty);
153    table.resolve_obligations_as_possible();
154    table.propagate_diverging_flag();
155    table.resolve_completely(ty_with_vars)
156}

305/// Represents coercing a value to a different type of value.
306///
307/// We transform values by following a number of `Adjust` steps in order.
308/// See the documentation on variants of `Adjust` for more details.
309///
310/// Here are some common scenarios:
311///
312/// 1. The simplest cases are where a pointer is not adjusted fat vs thin.
313///    Here the pointer will be dereferenced N times (where a dereference can
314///    happen to raw or borrowed pointers or any smart pointer which implements
315///    Deref, including Box<_>). The types of dereferences is given by
316///    `autoderefs`. It can then be auto-referenced zero or one times, indicated
317///    by `autoref`, to either a raw or borrowed pointer. In these cases unsize is
318///    `false`.
319///
320/// 2. A thin-to-fat coercion involves unsizing the underlying data. We start
321///    with a thin pointer, deref a number of times, unsize the underlying data,
322///    then autoref. The 'unsize' phase may change a fixed length array to a
323///    dynamically sized one, a concrete object to a trait object, or statically
324///    sized struct to a dynamically sized one. E.g., &[i32; 4] -> &[i32] is
325///    represented by:
326///
327///    ```ignore
328///    Deref(None) -> [i32; 4],
329///    Borrow(AutoBorrow::Ref) -> &[i32; 4],
330///    Unsize -> &[i32],
331///    ```
332///
333///    Note that for a struct, the 'deep' unsizing of the struct is not recorded.
334///    E.g., `struct Foo<T> { it: T }` we can coerce &Foo<[i32; 4]> to &Foo<[i32]>
335///    The autoderef and -ref are the same as in the above example, but the type
336///    stored in `unsize` is `Foo<[i32]>`, we don't store any further detail about
337///    the underlying conversions from `[i32; 4]` to `[i32]`.
338///
339/// 3. Coercing a `Box<T>` to `Box<dyn Trait>` is an interesting special case. In
340///    that case, we have the pointer we need coming in, so there are no
341///    autoderefs, and no autoref. Instead we just do the `Unsize` transformation.
342///    At some point, of course, `Box` should move out of the compiler, in which
343///    case this is analogous to transforming a struct. E.g., Box<[i32; 4]> ->
344///    Box<[i32]> is an `Adjust::Unsize` with the target `Box<[i32]>`.
345#[derive(Clone, Debug, PartialEq, Eq, Hash)]
346pub struct Adjustment {
347    pub kind: Adjust,
348    pub target: Ty,
349}
350
351impl Adjustment {
352    pub fn borrow(m: Mutability, ty: Ty, lt: Lifetime) -> Self {
353        let ty = TyKind::Ref(m, lt.clone(), ty).intern(Interner);
354        Adjustment { kind: Adjust::Borrow(AutoBorrow::Ref(lt, m)), target: ty }
355    }
356}

425/// The result of type inference: A mapping from expressions and patterns to types.
426///
427/// When you add a field that stores types (including `Substitution` and the like), don't forget
428/// `resolve_completely()`'ing  them in `InferenceContext::resolve_all()`. Inference variables must
429/// not appear in the final inference result.
430#[derive(Clone, PartialEq, Eq, Debug, Default)]
431pub struct InferenceResult {
432    /// For each method call expr, records the function it resolves to.
433    method_resolutions: FxHashMap<ExprId, (FunctionId, Substitution)>,
434    /// For each field access expr, records the field it resolves to.
435    field_resolutions: FxHashMap<ExprId, Either<FieldId, TupleFieldId>>,
436    /// For each struct literal or pattern, records the variant it resolves to.
437    variant_resolutions: FxHashMap<ExprOrPatId, VariantId>,
438    /// For each associated item record what it resolves to
439    assoc_resolutions: FxHashMap<ExprOrPatId, (AssocItemId, Substitution)>,
440    /// Whenever a tuple field expression access a tuple field, we allocate a tuple id in
441    /// [`InferenceContext`] and store the tuples substitution there. This map is the reverse of
442    /// that which allows us to resolve a [`TupleFieldId`]s type.
443    pub tuple_field_access_types: FxHashMap<TupleId, Substitution>,
444    /// During inference this field is empty and [`InferenceContext::diagnostics`] is filled instead.
445    pub diagnostics: Vec<InferenceDiagnostic>,
446    pub type_of_expr: ArenaMap<ExprId, Ty>,
447    /// For each pattern record the type it resolves to.
448    ///
449    /// **Note**: When a pattern type is resolved it may still contain
450    /// unresolved or missing subpatterns or subpatterns of mismatched types.
451    pub type_of_pat: ArenaMap<PatId, Ty>,
452    pub type_of_binding: ArenaMap<BindingId, Ty>,
453    pub type_of_rpit: ArenaMap<ImplTraitIdx, Ty>,
454    /// Type of the result of `.into_iter()` on the for. `ExprId` is the one of the whole for loop.
455    pub type_of_for_iterator: FxHashMap<ExprId, Ty>,
456    type_mismatches: FxHashMap<ExprOrPatId, TypeMismatch>,
457    /// Whether there are any type-mismatching errors in the result.
458    // FIXME: This isn't as useful as initially thought due to us falling back placeholders to
459    // `TyKind::Error`.
460    // Which will then mark this field.
461    pub(crate) has_errors: bool,
462    /// Interned common types to return references to.
463    // FIXME: Move this into `InferenceContext`
464    standard_types: InternedStandardTypes,
465    /// Stores the types which were implicitly dereferenced in pattern binding modes.
466    pub pat_adjustments: FxHashMap<PatId, Vec<Ty>>,
467    /// Stores the binding mode (`ref` in `let ref x = 2`) of bindings.
468    ///
469    /// This one is tied to the `PatId` instead of `BindingId`, because in some rare cases, a binding in an
470    /// or pattern can have multiple binding modes. For example:
471    /// ```
472    /// fn foo(mut slice: &[u32]) -> usize {
473    ///     slice = match slice {
474    ///         [0, rest @ ..] | rest => rest,
475    ///     };
476    ///     0
477    /// }
478    /// ```
479    /// the first `rest` has implicit `ref` binding mode, but the second `rest` binding mode is `move`.
480    pub binding_modes: ArenaMap<PatId, BindingMode>,
481    pub expr_adjustments: FxHashMap<ExprId, Vec<Adjustment>>,
482    pub(crate) closure_info: FxHashMap<ClosureId, (Vec<CapturedItem>, FnTrait)>,
483    // FIXME: remove this field
484    pub mutated_bindings_in_closure: FxHashSet<BindingId>,
485    pub coercion_casts: FxHashSet<ExprId>,
486}

532        self.type_mismatches.iter().filter_map(|(expr_or_pat, mismatch)| match *expr_or_pat {

535        })
536    }
537    pub fn closure_info(&self, closure: &ClosureId) -> &(Vec<CapturedItem>, FnTrait) {
538        self.closure_info.get(closure).unwrap()
539    }
540    pub fn type_of_expr_or_pat(&self, id: ExprOrPatId) -> Option<&Ty> {
541        match id {
542            ExprOrPatId::ExprId(id) => self.type_of_expr.get(id),
543            ExprOrPatId::PatId(id) => self.type_of_pat.get(id),
544        }
545    }
546}

634#[derive(Clone, Debug)]
635struct BreakableContext {
636    /// Whether this context contains at least one break expression.
637    may_break: bool,
638    /// The coercion target of the context.
639    coerce: Option<CoerceMany>,
640    /// The optional label of the context.
641    label: Option<LabelId>,
642    kind: BreakableKind,
643}
644
645#[derive(Clone, Debug)]
646enum BreakableKind {
647    Block,
648    Loop,
649    /// A border is something like an async block, closure etc. Anything that prevents
650    /// breaking/continuing through
651    Border,
652}
653
654fn find_breakable(
655    ctxs: &mut [BreakableContext],
656    label: Option<LabelId>,
657) -> Option<&mut BreakableContext> {
658    let mut ctxs = ctxs
659        .iter_mut()
660        .rev()
661        .take_while(|it| matches!(it.kind, BreakableKind::Block | BreakableKind::Loop));
662    match label {
663        Some(_) => ctxs.find(|ctx| ctx.label == label),
664        None => ctxs.find(|ctx| matches!(ctx.kind, BreakableKind::Loop)),
665    }
666}
667
668fn find_continuable(
669    ctxs: &mut [BreakableContext],
670    label: Option<LabelId>,
671) -> Option<&mut BreakableContext> {
672    match label {
673        Some(_) => find_breakable(ctxs, label).filter(|it| matches!(it.kind, BreakableKind::Loop)),
674        None => find_breakable(ctxs, label),
675    }
676}

727    // FIXME: This function should be private in module. It is currently only used in the consteval, since we need
728    // `InferenceResult` in the middle of inference. See the fixme comment in `consteval::eval_to_const`. If you
729    // used this function for another workaround, mention it here. If you really need this function and believe that
730    // there is no problem in it being `pub(crate)`, remove this comment.
731    pub(crate) fn resolve_all(self) -> InferenceResult {
732        let InferenceContext {
733            mut table,
734            mut result,
735            mut deferred_cast_checks,
736            tuple_field_accesses_rev,
737            diagnostics,
738            ..
739        } = self;
740        let mut diagnostics = diagnostics.finish();
741        // Destructure every single field so whenever new fields are added to `InferenceResult` we
742        // don't forget to handle them here.
743        let InferenceResult {
744            method_resolutions,
745            field_resolutions: _,
746            variant_resolutions: _,
747            assoc_resolutions,
748            type_of_expr,
749            type_of_pat,
750            type_of_binding,
751            type_of_rpit,
752            type_of_for_iterator,
753            type_mismatches,
754            has_errors,
755            standard_types: _,
756            pat_adjustments,
757            binding_modes: _,
758            expr_adjustments,
759            // Types in `closure_info` have already been `resolve_completely()`'d during
760            // `InferenceContext::infer_closures()` (in `HirPlace::ty()` specifically), so no need
761            // to resolve them here.
762            closure_info: _,
763            mutated_bindings_in_closure: _,
764            tuple_field_access_types: _,
765            coercion_casts,
766            diagnostics: _,
767        } = &mut result;
768        table.fallback_if_possible();
769
770        // Comment from rustc:
771        // Even though coercion casts provide type hints, we check casts after fallback for
772        // backwards compatibility. This makes fallback a stronger type hint than a cast coercion.
773        let mut apply_adjustments = |expr, adj| {
774            expr_adjustments.insert(expr, adj);
775        };
776        let mut set_coercion_cast = |expr| {
777            coercion_casts.insert(expr);
778        };
779        for cast in deferred_cast_checks.iter_mut() {
780            if let Err(diag) =
781                cast.check(&mut table, &mut apply_adjustments, &mut set_coercion_cast)
782            {
783                diagnostics.push(diag);
784            }
785        }
786
787        // FIXME resolve obligations as well (use Guidance if necessary)
788        table.resolve_obligations_as_possible();
789
790        // make sure diverging type variables are marked as such
791        table.propagate_diverging_flag();
792        for ty in type_of_expr.values_mut() {
793            *ty = table.resolve_completely(ty.clone());
794            *has_errors = *has_errors || ty.contains_unknown();
795        }
796        for ty in type_of_pat.values_mut() {
797            *ty = table.resolve_completely(ty.clone());
798            *has_errors = *has_errors || ty.contains_unknown();
799        }
800        for ty in type_of_binding.values_mut() {
801            *ty = table.resolve_completely(ty.clone());
802            *has_errors = *has_errors || ty.contains_unknown();
803        }
804        for ty in type_of_rpit.values_mut() {
805            *ty = table.resolve_completely(ty.clone());
806            *has_errors = *has_errors || ty.contains_unknown();
807        }
808        for ty in type_of_for_iterator.values_mut() {
809            *ty = table.resolve_completely(ty.clone());
810            *has_errors = *has_errors || ty.contains_unknown();
811        }
812
813        *has_errors |= !type_mismatches.is_empty();
814
815        type_mismatches.retain(|_, mismatch| {
816            mismatch.expected = table.resolve_completely(mismatch.expected.clone());
817            mismatch.actual = table.resolve_completely(mismatch.actual.clone());
818            chalk_ir::zip::Zip::zip_with(
819                &mut UnknownMismatch(self.db),
820                Variance::Invariant,
821                &mismatch.expected,
822                &mismatch.actual,
823            )
824            .is_ok()
825        });
826        diagnostics.retain_mut(|diagnostic| {
827            use InferenceDiagnostic::*;
828            match diagnostic {
829                ExpectedFunction { found: ty, .. }
830                | UnresolvedField { receiver: ty, .. }
831                | UnresolvedMethodCall { receiver: ty, .. } => {
832                    *ty = table.resolve_completely(ty.clone());
833                    // FIXME: Remove this when we are on par with rustc in terms of inference
834                    if ty.contains_unknown() {
835                        return false;
836                    }
837
838                    if let UnresolvedMethodCall { field_with_same_name, .. } = diagnostic {
839                        if let Some(ty) = field_with_same_name {
840                            *ty = table.resolve_completely(ty.clone());
841                            if ty.contains_unknown() {
842                                *field_with_same_name = None;
843                            }
844                        }
845                    }
846                }
847                TypedHole { expected: ty, .. } => {
848                    *ty = table.resolve_completely(ty.clone());
849                }
850                _ => (),
851            }
852            true
853        });
854        for (_, subst) in method_resolutions.values_mut() {
855            *subst = table.resolve_completely(subst.clone());
856            *has_errors =
857                *has_errors || subst.type_parameters(Interner).any(|ty| ty.contains_unknown());
858        }
859        for (_, subst) in assoc_resolutions.values_mut() {
860            *subst = table.resolve_completely(subst.clone());
861            *has_errors =
862                *has_errors || subst.type_parameters(Interner).any(|ty| ty.contains_unknown());
863        }
864        for adjustment in expr_adjustments.values_mut().flatten() {
865            adjustment.target = table.resolve_completely(adjustment.target.clone());
866            *has_errors = *has_errors || adjustment.target.contains_unknown();
867        }
868        for adjustment in pat_adjustments.values_mut().flatten() {
869            *adjustment = table.resolve_completely(adjustment.clone());
870            *has_errors = *has_errors || adjustment.contains_unknown();
871        }
872        result.tuple_field_access_types = tuple_field_accesses_rev
873            .into_iter()
874            .enumerate()
875            .map(|(idx, subst)| (TupleId(idx as u32), table.resolve_completely(subst)))
876            .inspect(|(_, subst)| {
877                *has_errors =
878                    *has_errors || subst.type_parameters(Interner).any(|ty| ty.contains_unknown());
879            })
880            .collect();
881
882        result.diagnostics = diagnostics;
883
884        result
885    }

907    fn collect_fn(&mut self, func: FunctionId) {
908        let data = self.db.function_signature(func);
909        let mut param_tys =
910            self.with_ty_lowering(&data.store, InferenceTyDiagnosticSource::Signature, |ctx| {
911                ctx.type_param_mode(ParamLoweringMode::Placeholder);
912                data.params.iter().map(|&type_ref| ctx.lower_ty(type_ref)).collect::<Vec<_>>()
913            });
914
915        // Check if function contains a va_list, if it does then we append it to the parameter types
916        // that are collected from the function data
917        if data.is_varargs() {
918            let va_list_ty = match self.resolve_va_list() {
919                Some(va_list) => TyBuilder::adt(self.db, va_list)
920                    .fill_with_defaults(self.db, || self.table.new_type_var())
921                    .build(),
922                None => self.err_ty(),
923            };
924
925            param_tys.push(va_list_ty);
926        }
927        let mut param_tys = param_tys.into_iter().chain(iter::repeat(self.table.new_type_var()));
928        if let Some(self_param) = self.body.self_param {
929            if let Some(ty) = param_tys.next() {
930                let ty = self.insert_type_vars(ty);
931                let ty = self.normalize_associated_types_in(ty);
932                self.write_binding_ty(self_param, ty);
933            }
934        }
935        let mut tait_candidates = FxHashSet::default();
936        for (ty, pat) in param_tys.zip(&*self.body.params) {
937            let ty = self.insert_type_vars(ty);
938            let ty = self.normalize_associated_types_in(ty);
939
940            self.infer_top_pat(*pat, &ty, None);
941            if ty
942                .data(Interner)
943                .flags
944                .intersects(TypeFlags::HAS_TY_OPAQUE.union(TypeFlags::HAS_TY_INFER))
945            {
946                tait_candidates.insert(ty);
947            }
948        }
949        let return_ty = match data.ret_type {
950            Some(return_ty) => {
951                let return_ty = self.with_ty_lowering(
952                    &data.store,
953                    InferenceTyDiagnosticSource::Signature,
954                    |ctx| {
955                        ctx.type_param_mode(ParamLoweringMode::Placeholder)
956                            .impl_trait_mode(ImplTraitLoweringMode::Opaque)
957                            .lower_ty(return_ty)
958                    },
959                );
960                let return_ty = self.insert_type_vars(return_ty);
961                if let Some(rpits) = self.db.return_type_impl_traits(func) {
962                    // RPIT opaque types use substitution of their parent function.
963                    let fn_placeholders = TyBuilder::placeholder_subst(self.db, func);
964                    let mut mode = ImplTraitReplacingMode::ReturnPosition(FxHashSet::default());
965                    let result = self.insert_inference_vars_for_impl_trait(
966                        return_ty,
967                        fn_placeholders,
968                        &mut mode,
969                    );
970                    if let ImplTraitReplacingMode::ReturnPosition(taits) = mode {
971                        tait_candidates.extend(taits);
972                    }
973                    let rpits = rpits.skip_binders();
974                    for (id, _) in rpits.impl_traits.iter() {
975                        if let Entry::Vacant(e) = self.result.type_of_rpit.entry(id) {
976                            never!("Missed RPIT in `insert_inference_vars_for_rpit`");
977                            e.insert(TyKind::Error.intern(Interner));
978                        }
979                    }
980                    result
981                } else {
982                    return_ty
983                }
984            }
985            None => self.result.standard_types.unit.clone(),
986        };
987
988        self.return_ty = self.normalize_associated_types_in(return_ty);
989        self.return_coercion = Some(CoerceMany::new(self.return_ty.clone()));
990
991        // Functions might be defining usage sites of TAITs.
992        // To define an TAITs, that TAIT must appear in the function's signatures.
993        // So, it suffices to check for params and return types.
994        if self
995            .return_ty
996            .data(Interner)
997            .flags
998            .intersects(TypeFlags::HAS_TY_OPAQUE.union(TypeFlags::HAS_TY_INFER))
999        {
1000            tait_candidates.insert(self.return_ty.clone());
1001        }
1002        self.make_tait_coercion_table(tait_candidates.iter());
1003    }

1005    fn insert_inference_vars_for_impl_trait<T>(
1006        &mut self,
1007        t: T,
1008        placeholders: Substitution,
1009        mode: &mut ImplTraitReplacingMode,
1010    ) -> T
1011    where
1012        T: crate::HasInterner<Interner = Interner> + crate::TypeFoldable<Interner>,
1013    {
1014        fold_tys(
1015            t,
1016            |ty, _| {
1017                let opaque_ty_id = match ty.kind(Interner) {
1018                    TyKind::OpaqueType(opaque_ty_id, _) => *opaque_ty_id,
1019                    _ => return ty,
1020                };
1021                let (impl_traits, idx) =
1022                    match self.db.lookup_intern_impl_trait_id(opaque_ty_id.into()) {
1023                        // We don't replace opaque types from other kind with inference vars
1024                        // because `insert_inference_vars_for_impl_traits` for each kinds
1025                        // and unreplaced opaque types of other kind are resolved while
1026                        // inferencing because of `tait_coercion_table`.
1027                        // Moreover, calling `insert_inference_vars_for_impl_traits` with same
1028                        // `placeholders` for other kind may cause trouble because
1029                        // the substs for the bounds of each impl traits do not match
1030                        ImplTraitId::ReturnTypeImplTrait(def, idx) => {
1031                            if matches!(mode, ImplTraitReplacingMode::TypeAlias) {
1032                                // RPITs don't have `tait_coercion_table`, so use inserted inference
1033                                // vars for them.
1034                                if let Some(ty) = self.result.type_of_rpit.get(idx) {
1035                                    return ty.clone();
1036                                }
1037                                return ty;
1038                            }
1039                            (self.db.return_type_impl_traits(def), idx)
1040                        }
1041                        ImplTraitId::TypeAliasImplTrait(def, idx) => {
1042                            if let ImplTraitReplacingMode::ReturnPosition(taits) = mode {
1043                                // Gather TAITs while replacing RPITs because TAITs inside RPITs
1044                                // may not visited while replacing TAITs
1045                                taits.insert(ty.clone());
1046                                return ty;
1047                            }
1048                            (self.db.type_alias_impl_traits(def), idx)
1049                        }
1050                        _ => unreachable!(),
1051                    };
1052                let Some(impl_traits) = impl_traits else {
1053                    return ty;
1054                };
1055                let bounds = (*impl_traits)
1056                    .map_ref(|its| its.impl_traits[idx].bounds.map_ref(|it| it.iter()));
1057                let var = self.table.new_type_var();
1058                let var_subst = Substitution::from1(Interner, var.clone());
1059                for bound in bounds {
1060                    let predicate = bound.map(|it| it.cloned());
1061                    let predicate = predicate.substitute(Interner, &placeholders);
1062                    let (var_predicate, binders) =
1063                        predicate.substitute(Interner, &var_subst).into_value_and_skipped_binders();
1064                    always!(binders.is_empty(Interner)); // quantified where clauses not yet handled
1065                    let var_predicate = self.insert_inference_vars_for_impl_trait(
1066                        var_predicate,
1067                        placeholders.clone(),
1068                        mode,
1069                    );
1070                    self.push_obligation(var_predicate.cast(Interner));
1071                }
1072                self.result.type_of_rpit.insert(idx, var.clone());
1073                var
1074            },
1075            DebruijnIndex::INNERMOST,
1076        )
1077    }

1079    /// The coercion of a non-inference var into an opaque type should fail,
1080    /// but not in the defining sites of the TAITs.
1081    /// In such cases, we insert an proxy inference var for each TAIT,
1082    /// and coerce into it instead of TAIT itself.
1083    ///
1084    /// The inference var stretagy is effective because;
1085    ///
1086    /// - It can still unify types that coerced into TAITs
1087    /// - We are pushing `impl Trait` bounds into it
1088    ///
1089    /// This function inserts a map that maps the opaque type to that proxy inference var.
1090    fn make_tait_coercion_table<'b>(&mut self, tait_candidates: impl Iterator<Item = &'b Ty>) {
1091        struct TypeAliasImplTraitCollector<'a, 'b> {
1092            db: &'b dyn HirDatabase,
1093            table: &'b mut InferenceTable<'a>,
1094            assocs: FxHashMap<OpaqueTyId, (ImplId, Ty)>,
1095            non_assocs: FxHashMap<OpaqueTyId, Ty>,
1096        }
1097
1098        impl TypeVisitor<Interner> for TypeAliasImplTraitCollector<'_, '_> {
1099            type BreakTy = ();
1100
1101            fn as_dyn(&mut self) -> &mut dyn TypeVisitor<Interner, BreakTy = Self::BreakTy> {
1102                self
1103            }
1104
1105            fn interner(&self) -> Interner {
1106                Interner
1107            }
1108
1109            fn visit_ty(
1110                &mut self,
1111                ty: &chalk_ir::Ty<Interner>,
1112                outer_binder: DebruijnIndex,
1113            ) -> std::ops::ControlFlow<Self::BreakTy> {
1114                let ty = self.table.resolve_ty_shallow(ty);
1115
1116                if let TyKind::OpaqueType(id, _) = ty.kind(Interner) {
1117                    if let ImplTraitId::TypeAliasImplTrait(alias_id, _) =
1118                        self.db.lookup_intern_impl_trait_id((*id).into())
1119                    {
1120                        let loc = self.db.lookup_intern_type_alias(alias_id);
1121                        match loc.container {
1122                            ItemContainerId::ImplId(impl_id) => {
1123                                self.assocs.insert(*id, (impl_id, ty.clone()));
1124                            }
1125                            ItemContainerId::ModuleId(..) | ItemContainerId::ExternBlockId(..) => {
1126                                self.non_assocs.insert(*id, ty.clone());
1127                            }
1128                            _ => {}
1129                        }
1130                    }
1131                }
1132
1133                ty.super_visit_with(self, outer_binder)
1134            }
1135        }
1136
1137        let mut collector = TypeAliasImplTraitCollector {
1138            db: self.db,
1139            table: &mut self.table,
1140            assocs: FxHashMap::default(),
1141            non_assocs: FxHashMap::default(),
1142        };
1143        for ty in tait_candidates {
1144            _ = ty.visit_with(collector.as_dyn(), DebruijnIndex::INNERMOST);
1145        }
1146
1147        // Non-assoc TAITs can be define-used everywhere as long as they are
1148        // in function signatures or const types, etc
1149        let mut taits = collector.non_assocs;
1150
1151        // assoc TAITs(ATPITs) can be only define-used inside their impl block.
1152        // They cannot be define-used in inner items like in the following;
1153        //
1154        // ```
1155        // impl Trait for Struct {
1156        //     type Assoc = impl Default;
1157        //
1158        //     fn assoc_fn() -> Self::Assoc {
1159        //         let foo: Self::Assoc = true; // Allowed here
1160        //
1161        //         fn inner() -> Self::Assoc {
1162        //              false                   // Not allowed here
1163        //         }
1164        //
1165        //         foo
1166        //     }
1167        // }
1168        // ```
1169        let impl_id = match self.owner {
1170            DefWithBodyId::FunctionId(it) => {
1171                let loc = self.db.lookup_intern_function(it);
1172                if let ItemContainerId::ImplId(impl_id) = loc.container {
1173                    Some(impl_id)
1174                } else {
1175                    None
1176                }
1177            }
1178            DefWithBodyId::ConstId(it) => {
1179                let loc = self.db.lookup_intern_const(it);
1180                if let ItemContainerId::ImplId(impl_id) = loc.container {
1181                    Some(impl_id)
1182                } else {
1183                    None
1184                }
1185            }
1186            _ => None,
1187        };
1188
1189        if let Some(impl_id) = impl_id {

1196            .into_iter()

1199                    self.db.lookup_intern_impl_trait_id(id.into())
1200                {
1201                    let subst = TyBuilder::placeholder_subst(self.db, alias_id);
1202                    let ty = self.insert_inference_vars_for_impl_trait(
1203                        ty,
1204                        subst,
1205                        &mut ImplTraitReplacingMode::TypeAlias,
1206                    );
1207                    Some((id, ty))
1208                } else {
1209                    None
1210                }
1211            })
1212            .collect();

1390    /// Returns the deeply last field of nested structures, or the same type if
1391    /// not a structure at all. Corresponds to the only possible unsized field,
1392    /// and its type can be used to determine unsizing strategy.
1393    ///
1394    /// This is parameterized over the normalization strategy (i.e. how to
1395    /// handle `<T as Trait>::Assoc` and `impl Trait`); pass the identity
1396    /// function to indicate no normalization should take place.
1397    fn struct_tail_with_normalize(
1398        &mut self,
1399        mut ty: Ty,
1400        mut normalize: impl FnMut(Ty) -> Ty,
1401    ) -> Ty {
1402        // FIXME: fetch the limit properly
1403        let recursion_limit = 10;
1404        for iteration in 0.. {
1405            if iteration > recursion_limit {
1406                return self.err_ty();
1407            }
1408            match ty.kind(Interner) {
1409                TyKind::Adt(chalk_ir::AdtId(hir_def::AdtId::StructId(struct_id)), substs) => {
1410                    match self.db.field_types((*struct_id).into()).values().next_back().cloned() {
1411                        Some(field) => {
1412                            ty = field.substitute(Interner, substs);
1413                        }
1414                        None => break,
1415                    }
1416                }
1417                TyKind::Adt(..) => break,
1418                TyKind::Tuple(_, substs) => {
1419                    match substs
1420                        .as_slice(Interner)
1421                        .split_last()
1422                        .and_then(|(last_ty, _)| last_ty.ty(Interner))
1423                    {
1424                        Some(last_ty) => ty = last_ty.clone(),
1425                        None => break,
1426                    }
1427                }
1428                TyKind::Alias(..) => {
1429                    let normalized = normalize(ty.clone());
1430                    if ty == normalized {
1431                        return ty;
1432                    } else {
1433                        ty = normalized;
1434                    }
1435                }
1436                _ => break,
1437            }
1438        }
1439        ty
1440    }

1498    fn resolve_variant(
1499        &mut self,
1500        node: ExprOrPatId,
1501        path: Option<&Path>,
1502        value_ns: bool,
1503    ) -> (Ty, Option<VariantId>) {
1504        let path = match path {
1505            Some(path) => path,
1506            None => return (self.err_ty(), None),
1507        };
1508        let mut ctx = TyLoweringContext::new(
1509            self.db,
1510            &self.resolver,
1511            &self.body.store,
1512            &self.diagnostics,
1513            InferenceTyDiagnosticSource::Body,
1514            self.generic_def,
1515        );
1516        let mut path_ctx = ctx.at_path(path, node);
1517        let (resolution, unresolved) = if value_ns {
1518            let Some(res) = path_ctx.resolve_path_in_value_ns(HygieneId::ROOT) else {
1519                return (self.err_ty(), None);
1520            };
1521            match res {
1522                ResolveValueResult::ValueNs(value, _) => match value {
1523                    ValueNs::EnumVariantId(var) => {
1524                        let substs = path_ctx.substs_from_path(var.into(), true);
1525                        drop(ctx);
1526                        let ty = self.db.ty(var.lookup(self.db).parent.into());
1527                        let ty = self.insert_type_vars(ty.substitute(Interner, &substs));
1528                        return (ty, Some(var.into()));
1529                    }
1530                    ValueNs::StructId(strukt) => {
1531                        let substs = path_ctx.substs_from_path(strukt.into(), true);
1532                        drop(ctx);
1533                        let ty = self.db.ty(strukt.into());
1534                        let ty = self.insert_type_vars(ty.substitute(Interner, &substs));
1535                        return (ty, Some(strukt.into()));
1536                    }
1537                    ValueNs::ImplSelf(impl_id) => (TypeNs::SelfType(impl_id), None),
1538                    _ => {
1539                        drop(ctx);
1540                        return (self.err_ty(), None);
1541                    }
1542                },
1543                ResolveValueResult::Partial(typens, unresolved, _) => (typens, Some(unresolved)),
1544            }
1545        } else {
1546            match path_ctx.resolve_path_in_type_ns() {
1547                Some((it, idx)) => (it, idx),
1548                None => return (self.err_ty(), None),
1549            }
1550        };
1551        return match resolution {
1552            TypeNs::AdtId(AdtId::StructId(strukt)) => {
1553                let substs = path_ctx.substs_from_path(strukt.into(), true);
1554                drop(ctx);
1555                let ty = self.db.ty(strukt.into());
1556                let ty = self.insert_type_vars(ty.substitute(Interner, &substs));
1557                forbid_unresolved_segments((ty, Some(strukt.into())), unresolved)
1558            }
1559            TypeNs::AdtId(AdtId::UnionId(u)) => {
1560                let substs = path_ctx.substs_from_path(u.into(), true);
1561                drop(ctx);
1562                let ty = self.db.ty(u.into());
1563                let ty = self.insert_type_vars(ty.substitute(Interner, &substs));
1564                forbid_unresolved_segments((ty, Some(u.into())), unresolved)
1565            }
1566            TypeNs::EnumVariantId(var) => {
1567                let substs = path_ctx.substs_from_path(var.into(), true);
1568                drop(ctx);
1569                let ty = self.db.ty(var.lookup(self.db).parent.into());
1570                let ty = self.insert_type_vars(ty.substitute(Interner, &substs));
1571                forbid_unresolved_segments((ty, Some(var.into())), unresolved)
1572            }
1573            TypeNs::SelfType(impl_id) => {
1574                let generics = crate::generics::generics(self.db, impl_id.into());
1575                let substs = generics.placeholder_subst(self.db);
1576                let mut ty = self.db.impl_self_ty(impl_id).substitute(Interner, &substs);
1577
1578                let Some(remaining_idx) = unresolved else {
1579                    drop(ctx);
1580                    let Some(mod_path) = path.mod_path() else {
1581                        never!("resolver should always resolve lang item paths");
1582                        return (self.err_ty(), None);
1583                    };
1584                    return self.resolve_variant_on_alias(ty, None, mod_path);
1585                };
1586
1587                let mut remaining_segments = path.segments().skip(remaining_idx);
1588
1589                if remaining_segments.len() >= 2 {
1590                    path_ctx.ignore_last_segment();
1591                }
1592
1593                // We need to try resolving unresolved segments one by one because each may resolve
1594                // to a projection, which `TyLoweringContext` cannot handle on its own.
1595                let mut tried_resolving_once = false;
1596                while let Some(current_segment) = remaining_segments.first() {
1597                    // If we can resolve to an enum variant, it takes priority over associated type
1598                    // of the same name.
1599                    if let Some((AdtId::EnumId(id), _)) = ty.as_adt() {
1600                        let enum_data = self.db.enum_variants(id);
1601                        if let Some(variant) = enum_data.variant(current_segment.name) {
1602                            return if remaining_segments.len() == 1 {
1603                                (ty, Some(variant.into()))
1604                            } else {
1605                                // We still have unresolved paths, but enum variants never have
1606                                // associated types!
1607                                // FIXME: Report an error.
1608                                (self.err_ty(), None)
1609                            };
1610                        }
1611                    }
1612
1613                    if tried_resolving_once {
1614                        // FIXME: with `inherent_associated_types` this is allowed, but our `lower_partly_resolved_path()`
1615                        // will need to be updated to err at the correct segment.
1616                        break;
1617                    }
1618
1619                    // `lower_partly_resolved_path()` returns `None` as type namespace unless
1620                    // `remaining_segments` is empty, which is never the case here. We don't know
1621                    // which namespace the new `ty` is in until normalized anyway.
1622                    (ty, _) = path_ctx.lower_partly_resolved_path(resolution, false);
1623                    tried_resolving_once = true;
1624
1625                    ty = self.table.insert_type_vars(ty);
1626                    ty = self.table.normalize_associated_types_in(ty);
1627                    ty = self.table.resolve_ty_shallow(&ty);
1628                    if ty.is_unknown() {
1629                        return (self.err_ty(), None);
1630                    }
1631
1632                    remaining_segments = remaining_segments.skip(1);
1633                }
1634                drop(ctx);
1635
1636                let variant = ty.as_adt().and_then(|(id, _)| match id {
1637                    AdtId::StructId(s) => Some(VariantId::StructId(s)),
1638                    AdtId::UnionId(u) => Some(VariantId::UnionId(u)),
1639                    AdtId::EnumId(_) => {
1640                        // FIXME Error E0071, expected struct, variant or union type, found enum `Foo`
1641                        None
1642                    }
1643                });
1644                (ty, variant)
1645            }
1646            TypeNs::TypeAliasId(it) => {
1647                let Some(mod_path) = path.mod_path() else {
1648                    never!("resolver should always resolve lang item paths");
1649                    return (self.err_ty(), None);
1650                };
1651                let substs = path_ctx.substs_from_path_segment(it.into(), true, None);
1652                drop(ctx);
1653                let ty = self.db.ty(it.into());
1654                let ty = self.insert_type_vars(ty.substitute(Interner, &substs));
1655
1656                self.resolve_variant_on_alias(ty, unresolved, mod_path)
1657            }
1658            TypeNs::AdtSelfType(_) => {
1659                // FIXME this could happen in array size expressions, once we're checking them
1660                (self.err_ty(), None)
1661            }
1662            TypeNs::GenericParam(_) => {
1663                // FIXME potentially resolve assoc type
1664                (self.err_ty(), None)
1665            }
1666            TypeNs::AdtId(AdtId::EnumId(_))
1667            | TypeNs::BuiltinType(_)
1668            | TypeNs::TraitId(_)
1669            | TypeNs::TraitAliasId(_)
1670            | TypeNs::ModuleId(_) => {
1671                // FIXME diagnostic
1672                (self.err_ty(), None)
1673            }
1674        };
1675
1676        fn forbid_unresolved_segments(
1677            result: (Ty, Option<VariantId>),
1678            unresolved: Option<usize>,
1679        ) -> (Ty, Option<VariantId>) {
1680            if unresolved.is_none() {
1681                result
1682            } else {
1683                // FIXME diagnostic
1684                (TyKind::Error.intern(Interner), None)
1685            }
1686        }
1687    }

1689    fn resolve_variant_on_alias(
1690        &mut self,
1691        ty: Ty,
1692        unresolved: Option<usize>,
1693        path: &ModPath,
1694    ) -> (Ty, Option<VariantId>) {
1695        let remaining = unresolved.map(|it| path.segments()[it..].len()).filter(|it| it > &0);
1696        let ty = match ty.kind(Interner) {
1697            TyKind::Alias(AliasTy::Projection(proj_ty)) => {
1698                let ty = self.table.normalize_projection_ty(proj_ty.clone());
1699                self.table.resolve_ty_shallow(&ty)
1700            }
1701            _ => ty,
1702        };
1703        match remaining {
1704            None => {
1705                let variant = ty.as_adt().and_then(|(adt_id, _)| match adt_id {
1706                    AdtId::StructId(s) => Some(VariantId::StructId(s)),
1707                    AdtId::UnionId(u) => Some(VariantId::UnionId(u)),
1708                    AdtId::EnumId(_) => {
1709                        // FIXME Error E0071, expected struct, variant or union type, found enum `Foo`
1710                        None
1711                    }
1712                });
1713                (ty, variant)
1714            }
1715            Some(1) => {
1716                let segment = path.segments().last().unwrap();
1717                // this could be an enum variant or associated type
1718                if let Some((AdtId::EnumId(enum_id), _)) = ty.as_adt() {
1719                    let enum_data = self.db.enum_variants(enum_id);
1720                    if let Some(variant) = enum_data.variant(segment) {
1721                        return (ty, Some(variant.into()));
1722                    }
1723                }
1724                // FIXME potentially resolve assoc type
1725                (self.err_ty(), None)
1726            }
1727            Some(_) => {
1728                // FIXME diagnostic
1729                (self.err_ty(), None)
1730            }
1731        }
1732    }

1823/// When inferring an expression, we propagate downward whatever type hint we
1824/// are able in the form of an `Expectation`.
1825#[derive(Clone, PartialEq, Eq, Debug)]
1826pub(crate) enum Expectation {
1827    None,
1828    HasType(Ty),
1829    #[allow(dead_code)]
1830    Castable(Ty),
1831    RValueLikeUnsized(Ty),
1832}
1833
1834impl Expectation {
1835    /// The expectation that the type of the expression needs to equal the given
1836    /// type.
1837    fn has_type(ty: Ty) -> Self {
1838        if ty.is_unknown() {
1839            // FIXME: get rid of this?
1840            Expectation::None
1841        } else {
1842            Expectation::HasType(ty)
1843        }
1844    }
1845
1846    /// The following explanation is copied straight from rustc:
1847    /// Provides an expectation for an rvalue expression given an *optional*
1848    /// hint, which is not required for type safety (the resulting type might
1849    /// be checked higher up, as is the case with `&expr` and `box expr`), but
1850    /// is useful in determining the concrete type.
1851    ///
1852    /// The primary use case is where the expected type is a fat pointer,
1853    /// like `&[isize]`. For example, consider the following statement:
1854    ///
1855    ///     let it: &[isize] = &[1, 2, 3];
1856    ///
1857    /// In this case, the expected type for the `&[1, 2, 3]` expression is
1858    /// `&[isize]`. If however we were to say that `[1, 2, 3]` has the
1859    /// expectation `ExpectHasType([isize])`, that would be too strong --
1860    /// `[1, 2, 3]` does not have the type `[isize]` but rather `[isize; 3]`.
1861    /// It is only the `&[1, 2, 3]` expression as a whole that can be coerced
1862    /// to the type `&[isize]`. Therefore, we propagate this more limited hint,
1863    /// which still is useful, because it informs integer literals and the like.
1864    /// See the test case `test/ui/coerce-expect-unsized.rs` and #20169
1865    /// for examples of where this comes up,.
1866    fn rvalue_hint(ctx: &mut InferenceContext<'_>, ty: Ty) -> Self {
1867        match ctx.struct_tail_without_normalization(ty.clone()).kind(Interner) {
1868            TyKind::Slice(_) | TyKind::Str | TyKind::Dyn(_) => Expectation::RValueLikeUnsized(ty),
1869            _ => Expectation::has_type(ty),
1870        }
1871    }
1872
1873    /// This expresses no expectation on the type.
1874    fn none() -> Self {
1875        Expectation::None
1876    }
1877
1878    fn resolve(&self, table: &mut unify::InferenceTable<'_>) -> Expectation {
1879        match self {
1880            Expectation::None => Expectation::None,
1881            Expectation::HasType(t) => Expectation::HasType(table.resolve_ty_shallow(t)),
1882            Expectation::Castable(t) => Expectation::Castable(table.resolve_ty_shallow(t)),
1883            Expectation::RValueLikeUnsized(t) => {
1884                Expectation::RValueLikeUnsized(table.resolve_ty_shallow(t))
1885            }
1886        }
1887    }
1888
1889    fn to_option(&self, table: &mut unify::InferenceTable<'_>) -> Option<Ty> {
1890        match self.resolve(table) {
1891            Expectation::None => None,
1892            Expectation::HasType(t)
1893            | Expectation::Castable(t)
1894            | Expectation::RValueLikeUnsized(t) => Some(t),
1895        }
1896    }
1897
1898    fn only_has_type(&self, table: &mut unify::InferenceTable<'_>) -> Option<Ty> {
1899        match self {
1900            Expectation::HasType(t) => Some(table.resolve_ty_shallow(t)),
1901            Expectation::Castable(_) | Expectation::RValueLikeUnsized(_) | Expectation::None => {
1902                None
1903            }
1904        }
1905    }
1906
1907    fn coercion_target_type(&self, table: &mut unify::InferenceTable<'_>) -> Ty {
1908        self.only_has_type(table).unwrap_or_else(|| table.new_type_var())
1909    }
1910
1911    /// Comment copied from rustc:
1912    /// Disregard "castable to" expectations because they
1913    /// can lead us astray. Consider for example `if cond
1914    /// {22} else {c} as u8` -- if we propagate the
1915    /// "castable to u8" constraint to 22, it will pick the
1916    /// type 22u8, which is overly constrained (c might not
1917    /// be a u8). In effect, the problem is that the
1918    /// "castable to" expectation is not the tightest thing
1919    /// we can say, so we want to drop it in this case.
1920    /// The tightest thing we can say is "must unify with
1921    /// else branch". Note that in the case of a "has type"
1922    /// constraint, this limitation does not hold.
1923    ///
1924    /// If the expected type is just a type variable, then don't use
1925    /// an expected type. Otherwise, we might write parts of the type
1926    /// when checking the 'then' block which are incompatible with the
1927    /// 'else' branch.
1928    fn adjust_for_branches(&self, table: &mut unify::InferenceTable<'_>) -> Expectation {
1929        match self {
1930            Expectation::HasType(ety) => {
1931                let ety = table.resolve_ty_shallow(ety);
1932                if ety.is_ty_var() { Expectation::None } else { Expectation::HasType(ety) }
1933            }
1934            Expectation::RValueLikeUnsized(ety) => Expectation::RValueLikeUnsized(ety.clone()),
1935            _ => Expectation::None,
1936        }
1937    }
1938}

1978/// A zipper that checks for unequal occurrences of `{unknown}` and unresolved projections
1979/// in the two types. Used to filter out mismatch diagnostics that only differ in
1980/// `{unknown}` and unresolved projections. These mismatches are usually not helpful.
1981/// As the cause is usually an underlying name resolution problem
1982struct UnknownMismatch<'db>(&'db dyn HirDatabase);
1983impl chalk_ir::zip::Zipper<Interner> for UnknownMismatch<'_> {
1984    fn zip_tys(&mut self, variance: Variance, a: &Ty, b: &Ty) -> chalk_ir::Fallible<()> {
1985        let zip_substs = |this: &mut Self,
1986                          variances,
1987                          sub_a: &Substitution,
1988                          sub_b: &Substitution| {
1989            this.zip_substs(variance, variances, sub_a.as_slice(Interner), sub_b.as_slice(Interner))
1990        };
1991        match (a.kind(Interner), b.kind(Interner)) {
1992            (TyKind::Adt(id_a, sub_a), TyKind::Adt(id_b, sub_b)) if id_a == id_b => zip_substs(
1993                self,
1994                Some(self.unification_database().adt_variance(*id_a)),
1995                sub_a,
1996                sub_b,
1997            )?,
1998            (
1999                TyKind::AssociatedType(assoc_ty_a, sub_a),
2000                TyKind::AssociatedType(assoc_ty_b, sub_b),
2001            ) if assoc_ty_a == assoc_ty_b => zip_substs(self, None, sub_a, sub_b)?,
2002            (TyKind::Tuple(arity_a, sub_a), TyKind::Tuple(arity_b, sub_b))
2003                if arity_a == arity_b =>
2004            {
2005                zip_substs(self, None, sub_a, sub_b)?
2006            }
2007            (TyKind::OpaqueType(opaque_ty_a, sub_a), TyKind::OpaqueType(opaque_ty_b, sub_b))
2008                if opaque_ty_a == opaque_ty_b =>
2009            {
2010                zip_substs(self, None, sub_a, sub_b)?
2011            }
2012            (TyKind::Slice(ty_a), TyKind::Slice(ty_b)) => self.zip_tys(variance, ty_a, ty_b)?,
2013            (TyKind::FnDef(fn_def_a, sub_a), TyKind::FnDef(fn_def_b, sub_b))
2014                if fn_def_a == fn_def_b =>
2015            {
2016                zip_substs(
2017                    self,
2018                    Some(self.unification_database().fn_def_variance(*fn_def_a)),
2019                    sub_a,
2020                    sub_b,
2021                )?
2022            }
2023            (TyKind::Ref(mutability_a, _, ty_a), TyKind::Ref(mutability_b, _, ty_b))
2024                if mutability_a == mutability_b =>
2025            {
2026                self.zip_tys(variance, ty_a, ty_b)?
2027            }
2028            (TyKind::Raw(mutability_a, ty_a), TyKind::Raw(mutability_b, ty_b))
2029                if mutability_a == mutability_b =>
2030            {
2031                self.zip_tys(variance, ty_a, ty_b)?
2032            }
2033            (TyKind::Array(ty_a, const_a), TyKind::Array(ty_b, const_b)) if const_a == const_b => {
2034                self.zip_tys(variance, ty_a, ty_b)?
2035            }
2036            (TyKind::Closure(id_a, sub_a), TyKind::Closure(id_b, sub_b)) if id_a == id_b => {
2037                zip_substs(self, None, sub_a, sub_b)?
2038            }
2039            (TyKind::Coroutine(coroutine_a, sub_a), TyKind::Coroutine(coroutine_b, sub_b))
2040                if coroutine_a == coroutine_b =>
2041            {
2042                zip_substs(self, None, sub_a, sub_b)?
2043            }
2044            (
2045                TyKind::CoroutineWitness(coroutine_a, sub_a),
2046                TyKind::CoroutineWitness(coroutine_b, sub_b),
2047            ) if coroutine_a == coroutine_b => zip_substs(self, None, sub_a, sub_b)?,
2048            (TyKind::Function(fn_ptr_a), TyKind::Function(fn_ptr_b))
2049                if fn_ptr_a.sig == fn_ptr_b.sig && fn_ptr_a.num_binders == fn_ptr_b.num_binders =>
2050            {
2051                zip_substs(self, None, &fn_ptr_a.substitution.0, &fn_ptr_b.substitution.0)?
2052            }
2053            (TyKind::Error, TyKind::Error) => (),
2054            (TyKind::Error, _)
2055            | (_, TyKind::Error)
2056            | (TyKind::Alias(AliasTy::Projection(_)) | TyKind::AssociatedType(_, _), _)
2057            | (_, TyKind::Alias(AliasTy::Projection(_)) | TyKind::AssociatedType(_, _)) => {
2058                return Err(chalk_ir::NoSolution);
2059            }
2060            _ => (),
2061        }
2062
2063        Ok(())
2064    }
