50struct NestedObligationsForSelfTy<'a, 'db> {
51    ctx: &'a InferenceTable<'db>,
52    self_ty: TyVid,
53    root_cause: &'a ObligationCause,
54    obligations_for_self_ty: &'a mut SmallVec<[Obligation<'db, Predicate<'db>>; 4]>,
55}
56
57impl<'a, 'db> ProofTreeVisitor<'db> for NestedObligationsForSelfTy<'a, 'db> {
58    type Result = ();
59
60    fn config(&self) -> InspectConfig {
61        // Using an intentionally low depth to minimize the chance of future
62        // breaking changes in case we adapt the approach later on. This also
63        // avoids any hangs for exponentially growing proof trees.
64        InspectConfig { max_depth: 5 }
65    }
66
67    fn visit_goal(&mut self, inspect_goal: &InspectGoal<'_, 'db>) {
68        // No need to walk into goal subtrees that certainly hold, since they
69        // wouldn't then be stalled on an infer var.
70        if inspect_goal.result() == Ok(Certainty::Yes) {
71            return;
72        }
73
74        let db = self.ctx.interner();
75        let goal = inspect_goal.goal();
76        if self.ctx.predicate_has_self_ty(goal.predicate, self.self_ty)
77            // We do not push the instantiated forms of goals as it would cause any
78            // aliases referencing bound vars to go from having escaping bound vars to
79            // being able to be normalized to an inference variable.
80            //
81            // This is mostly just a hack as arbitrary nested goals could still contain
82            // such aliases while having a different `GoalSource`. Closure signature inference
83            // however can't really handle *every* higher ranked `Fn` goal also being present
84            // in the form of `?c: Fn<(<?x as Trait<'!a>>::Assoc)`.
85            //
86            // This also just better matches the behaviour of the old solver where we do not
87            // encounter instantiated forms of goals, only nested goals that referred to bound
88            // vars from instantiated goals.
89            && !matches!(inspect_goal.source(), GoalSource::InstantiateHigherRanked)
90        {
91            self.obligations_for_self_ty.push(Obligation::new(
92                db,
93                self.root_cause.clone(),
94                goal.param_env,
95                goal.predicate,
96            ));
97        }
98
99        // If there's a unique way to prove a given goal, recurse into
100        // that candidate. This means that for `impl<F: FnOnce(u32)> Trait<F> for () {}`
101        // and a `(): Trait<?0>` goal we recurse into the impl and look at
102        // the nested `?0: FnOnce(u32)` goal.
103        if let Some(candidate) = inspect_goal.unique_applicable_candidate() {
104            candidate.visit_nested_no_probe(self)
105        }
106    }
107}

109/// Check if types unify.
110///
111/// Note that we consider placeholder types to unify with everything.
112/// This means that there may be some unresolved goals that actually set bounds for the placeholder
113/// type for the types to unify. For example `Option<T>` and `Option<U>` unify although there is
114/// unresolved goal `T = U`.
115pub fn could_unify<'db>(
116    db: &'db dyn HirDatabase,
117    env: Arc<TraitEnvironment<'db>>,
118    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
119) -> bool {
120    could_unify_impl(db, env, tys, |ctxt| ctxt.try_evaluate_obligations())
121}
122
123/// Check if types unify eagerly making sure there are no unresolved goals.
124///
125/// This means that placeholder types are not considered to unify if there are any bounds set on
126/// them. For example `Option<T>` and `Option<U>` do not unify as we cannot show that `T = U`
127pub fn could_unify_deeply<'db>(
128    db: &'db dyn HirDatabase,
129    env: Arc<TraitEnvironment<'db>>,
130    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
131) -> bool {
132    could_unify_impl(db, env, tys, |ctxt| ctxt.evaluate_obligations_error_on_ambiguity())
133}
134
135fn could_unify_impl<'db>(
136    db: &'db dyn HirDatabase,
137    env: Arc<TraitEnvironment<'db>>,
138    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
139    select: for<'a> fn(&mut ObligationCtxt<'a, 'db>) -> Vec<NextSolverError<'db>>,
140) -> bool {
141    let interner = DbInterner::new_with(db, Some(env.krate), env.block);
142    let infcx = interner.infer_ctxt().build(TypingMode::PostAnalysis);
143    let cause = ObligationCause::dummy();
144    let at = infcx.at(&cause, env.env);
145    let ((ty1_with_vars, ty2_with_vars), _) = infcx.instantiate_canonical(tys);
146    let mut ctxt = ObligationCtxt::new(&infcx);
147    let can_unify = at
148        .eq(ty1_with_vars, ty2_with_vars)
149        .map(|infer_ok| ctxt.register_infer_ok_obligations(infer_ok))
150        .is_ok();
151    can_unify && select(&mut ctxt).is_empty()
152}

201    pub(crate) fn type_var_is_sized(&self, self_ty: TyVid) -> bool {
202        let Some(sized_did) = LangItem::Sized.resolve_trait(self.db, self.trait_env.krate) else {
203            return true;
204        };
205        self.obligations_for_self_ty(self_ty).into_iter().any(|obligation| {
206            match obligation.predicate.kind().skip_binder() {
207                PredicateKind::Clause(ClauseKind::Trait(data)) => data.def_id().0 == sized_did,
208                _ => false,
209            }
210        })

213    pub(super) fn obligations_for_self_ty(
214        &self,
215        self_ty: TyVid,
216    ) -> SmallVec<[Obligation<'db, Predicate<'db>>; 4]> {
217        let obligations = self.fulfillment_cx.pending_obligations();
218        let mut obligations_for_self_ty = SmallVec::new();
219        for obligation in obligations {
220            let mut visitor = NestedObligationsForSelfTy {
221                ctx: self,
222                self_ty,
223                obligations_for_self_ty: &mut obligations_for_self_ty,
224                root_cause: &obligation.cause,
225            };
226
227            let goal = obligation.as_goal();
228            self.infer_ctxt.visit_proof_tree(goal, &mut visitor);
229        }
230
231        obligations_for_self_ty.retain_mut(|obligation| {
232            obligation.predicate = self.infer_ctxt.resolve_vars_if_possible(obligation.predicate);
233            !obligation.predicate.has_placeholders()
234        });
235        obligations_for_self_ty
236    }

238    fn predicate_has_self_ty(&self, predicate: Predicate<'db>, expected_vid: TyVid) -> bool {
239        match predicate.kind().skip_binder() {
240            PredicateKind::Clause(ClauseKind::Trait(data)) => {
241                self.type_matches_expected_vid(expected_vid, data.self_ty())
242            }
243            PredicateKind::Clause(ClauseKind::Projection(data)) => {
244                self.type_matches_expected_vid(expected_vid, data.projection_term.self_ty())
245            }
246            PredicateKind::Clause(ClauseKind::ConstArgHasType(..))
247            | PredicateKind::Subtype(..)
248            | PredicateKind::Coerce(..)
249            | PredicateKind::Clause(ClauseKind::RegionOutlives(..))
250            | PredicateKind::Clause(ClauseKind::TypeOutlives(..))
251            | PredicateKind::Clause(ClauseKind::WellFormed(..))
252            | PredicateKind::DynCompatible(..)
253            | PredicateKind::NormalizesTo(..)
254            | PredicateKind::AliasRelate(..)
255            | PredicateKind::Clause(ClauseKind::ConstEvaluatable(..))
256            | PredicateKind::ConstEquate(..)
257            | PredicateKind::Clause(ClauseKind::HostEffect(..))
258            | PredicateKind::Clause(ClauseKind::UnstableFeature(_))
259            | PredicateKind::Ambiguous => false,
260        }
261    }

263    fn type_matches_expected_vid(&self, expected_vid: TyVid, ty: Ty<'db>) -> bool {
264        let ty = self.shallow_resolve(ty);
265
266        match ty.kind() {
267            TyKind::Infer(rustc_type_ir::TyVar(found_vid)) => {
268                self.infer_ctxt.root_var(expected_vid) == self.infer_ctxt.root_var(found_vid)
269            }
270            _ => false,
271        }
272    }

353    pub(crate) fn resolve_with_fallback<T>(
354        &mut self,
355        t: T,
356        fallback_ty: &mut dyn FnMut(DebruijnIndex, InferTy) -> Ty<'db>,
357        fallback_const: &mut dyn FnMut(DebruijnIndex, InferConst) -> Const<'db>,
358        fallback_region: &mut dyn FnMut(DebruijnIndex, RegionVid) -> Region<'db>,
359    ) -> T
360    where
361        T: TypeFoldable<DbInterner<'db>>,
362    {
363        struct Resolver<'a, 'db> {
364            table: &'a mut InferenceTable<'db>,
365            binder: DebruijnIndex,
366            fallback_ty: &'a mut dyn FnMut(DebruijnIndex, InferTy) -> Ty<'db>,
367            fallback_const: &'a mut dyn FnMut(DebruijnIndex, InferConst) -> Const<'db>,
368            fallback_region: &'a mut dyn FnMut(DebruijnIndex, RegionVid) -> Region<'db>,
369        }
370
371        impl<'db> TypeFolder<DbInterner<'db>> for Resolver<'_, 'db> {
372            fn cx(&self) -> DbInterner<'db> {
373                self.table.interner()
374            }
375
376            fn fold_binder<T>(&mut self, t: Binder<'db, T>) -> Binder<'db, T>
377            where
378                T: TypeFoldable<DbInterner<'db>>,
379            {
380                self.binder.shift_in(1);
381                let result = t.super_fold_with(self);
382                self.binder.shift_out(1);
383                result
384            }
385
386            fn fold_ty(&mut self, t: Ty<'db>) -> Ty<'db> {
387                if !t.has_infer() {
388                    return t;
389                }
390
391                if let TyKind::Infer(infer) = t.kind() {
392                    (self.fallback_ty)(self.binder, infer)
393                } else {
394                    t.super_fold_with(self)
395                }
396            }
397
398            fn fold_const(&mut self, c: Const<'db>) -> Const<'db> {
399                if !c.has_infer() {
400                    return c;
401                }
402
403                if let ConstKind::Infer(infer) = c.kind() {
404                    (self.fallback_const)(self.binder, infer)
405                } else {
406                    c.super_fold_with(self)
407                }
408            }
409
410            fn fold_region(&mut self, r: Region<'db>) -> Region<'db> {
411                if let RegionKind::ReVar(infer) = r.kind() {
412                    (self.fallback_region)(self.binder, infer)
413                } else {
414                    r
415                }
416            }
417        }
418
419        t.fold_with(&mut Resolver {
420            table: self,
421            binder: DebruijnIndex::ZERO,
422            fallback_ty,
423            fallback_const,
424            fallback_region,
425        })
426    }

498    /// Try to resolve `ty` to a structural type, normalizing aliases.
499    ///
500    /// In case there is still ambiguity, the returned type may be an inference
501    /// variable. This is different from `structurally_resolve_type` which errors
502    /// in this case.
503    pub(crate) fn try_structurally_resolve_type(&mut self, ty: Ty<'db>) -> Ty<'db> {
504        if let TyKind::Alias(..) = ty.kind() {
505            // We need to use a separate variable here as otherwise the temporary for
506            // `self.fulfillment_cx.borrow_mut()` is alive in the `Err` branch, resulting
507            // in a reentrant borrow, causing an ICE.
508            let result = self
509                .infer_ctxt
510                .at(&ObligationCause::misc(), self.trait_env.env)
511                .structurally_normalize_ty(ty, &mut self.fulfillment_cx);
512            match result {
513                Ok(normalized_ty) => normalized_ty,
514                Err(_errors) => Ty::new_error(self.interner(), ErrorGuaranteed),
515            }
516        } else {
517            self.resolve_vars_with_obligations(ty)
518        }
519    }

564    /// Checks an obligation without registering it. Useful mostly to check
565    /// whether a trait *might* be implemented before deciding to 'lock in' the
566    /// choice (during e.g. method resolution or deref).
567    #[tracing::instrument(level = "debug", skip(self))]
568    pub(crate) fn try_obligation(&mut self, predicate: Predicate<'db>) -> NextTraitSolveResult {
569        let goal = next_solver::Goal { param_env: self.trait_env.env, predicate };
570        let canonicalized = self.canonicalize(goal);
571
572        next_trait_solve_canonical_in_ctxt(&self.infer_ctxt, canonicalized)
573    }

577        self.register_obligation_in_env(goal)
578    }

580    #[tracing::instrument(level = "debug", skip(self))]
581    fn register_obligation_in_env(
582        &mut self,
583        goal: next_solver::Goal<'db, next_solver::Predicate<'db>>,
584    ) {
585        let result = next_trait_solve_in_ctxt(&self.infer_ctxt, goal);
586        tracing::debug!(?result);
587        match result {
588            Ok((_, Certainty::Yes)) => {}
589            Err(rustc_type_ir::solve::NoSolution) => {}
590            Ok((_, Certainty::Maybe { .. })) => {
591                self.fulfillment_cx.register_predicate_obligation(
592                    &self.infer_ctxt,
593                    Obligation::new(
594                        self.interner(),
595                        ObligationCause::new(),
596                        goal.param_env,
597                        goal.predicate,
598                    ),
599                );
600            }
601        }
602    }

631    pub(crate) fn callable_sig(
632        &mut self,
633        ty: Ty<'db>,
634        num_args: usize,
635    ) -> Option<(Option<FnTrait>, Vec<Ty<'db>>, Ty<'db>)> {
636        match ty.callable_sig(self.interner()) {
637            Some(sig) => {
638                let sig = sig.skip_binder();
639                Some((None, sig.inputs_and_output.inputs().to_vec(), sig.output()))
640            }
641            None => {
642                let (f, args_ty, return_ty) = self.callable_sig_from_fn_trait(ty, num_args)?;
643                Some((Some(f), args_ty, return_ty))
644            }
645        }
646    }
647
648    fn callable_sig_from_fn_trait(
649        &mut self,
650        ty: Ty<'db>,
651        num_args: usize,
652    ) -> Option<(FnTrait, Vec<Ty<'db>>, Ty<'db>)> {
653        for (fn_trait_name, output_assoc_name, subtraits) in [
654            (FnTrait::FnOnce, sym::Output, &[FnTrait::Fn, FnTrait::FnMut][..]),
655            (FnTrait::AsyncFnMut, sym::CallRefFuture, &[FnTrait::AsyncFn]),
656            (FnTrait::AsyncFnOnce, sym::CallOnceFuture, &[]),
657        ] {
658            let krate = self.trait_env.krate;
659            let fn_trait = fn_trait_name.get_id(self.db, krate)?;
660            let trait_data = fn_trait.trait_items(self.db);
661            let output_assoc_type =
662                trait_data.associated_type_by_name(&Name::new_symbol_root(output_assoc_name))?;
663
664            let mut arg_tys = Vec::with_capacity(num_args);
665            let arg_ty = Ty::new_tup_from_iter(
666                self.interner(),
667                std::iter::repeat_with(|| {
668                    let ty = self.next_ty_var();
669                    arg_tys.push(ty);
670                    ty
671                })
672                .take(num_args),
673            );
674            let args = [ty, arg_ty];
675            let trait_ref = TraitRef::new(self.interner(), fn_trait.into(), args);
676
677            let projection = Ty::new_alias(
678                self.interner(),
679                rustc_type_ir::AliasTyKind::Projection,
680                AliasTy::new(self.interner(), output_assoc_type.into(), args),
681            );
682
683            let pred = Predicate::upcast_from(trait_ref, self.interner());
684            if !self.try_obligation(pred).no_solution() {
685                self.register_obligation(pred);
686                let return_ty = self.normalize_alias_ty(projection);
687                for &fn_x in subtraits {
688                    let fn_x_trait = fn_x.get_id(self.db, krate)?;
689                    let trait_ref = TraitRef::new(self.interner(), fn_x_trait.into(), args);
690                    let pred = Predicate::upcast_from(trait_ref, self.interner());
691                    if !self.try_obligation(pred).no_solution() {
692                        return Some((fn_x, arg_tys, return_ty));
693                    }
694                }
695                return Some((fn_trait_name, arg_tys, return_ty));
696            }
697        }
698        None
699    }

741    /// Check if given type is `Sized` or not
742    pub(crate) fn is_sized(&mut self, ty: Ty<'db>) -> bool {
743        fn short_circuit_trivial_tys(ty: Ty<'_>) -> Option<bool> {
744            match ty.kind() {
745                TyKind::Bool
746                | TyKind::Char
747                | TyKind::Int(_)
748                | TyKind::Uint(_)
749                | TyKind::Float(_)
750                | TyKind::Ref(..)
751                | TyKind::RawPtr(..)
752                | TyKind::Never
753                | TyKind::FnDef(..)
754                | TyKind::Array(..)
755                | TyKind::FnPtr(..) => Some(true),
756                TyKind::Slice(..) | TyKind::Str | TyKind::Dynamic(..) => Some(false),
757                _ => None,
758            }
759        }
760
761        let mut ty = ty;
762        ty = self.eagerly_normalize_and_resolve_shallow_in(ty);
763        if let Some(sized) = short_circuit_trivial_tys(ty) {
764            return sized;
765        }
766
767        {
768            let mut structs = SmallVec::<[_; 8]>::new();
769            // Must use a loop here and not recursion because otherwise users will conduct completely
770            // artificial examples of structs that have themselves as the tail field and complain r-a crashes.
771            while let Some((AdtId::StructId(id), subst)) = ty.as_adt() {
772                let struct_data = id.fields(self.db);
773                if let Some((last_field, _)) = struct_data.fields().iter().next_back() {
774                    let last_field_ty = self.db.field_types(id.into())[last_field]
775                        .instantiate(self.interner(), subst);
776                    if structs.contains(&ty) {
777                        // A struct recursively contains itself as a tail field somewhere.
778                        return true; // Don't overload the users with too many errors.
779                    }
780                    structs.push(ty);
781                    // Structs can have DST as its last field and such cases are not handled
782                    // as unsized by the chalk, so we do this manually.
783                    ty = last_field_ty;
784                    ty = self.eagerly_normalize_and_resolve_shallow_in(ty);
785                    if let Some(sized) = short_circuit_trivial_tys(ty) {
786                        return sized;
787                    }
788                } else {
789                    break;
790                };
791            }
792        }
793
794        let Some(sized) = LangItem::Sized.resolve_trait(self.db, self.trait_env.krate) else {
795            return false;
796        };
797        let sized_pred = Predicate::upcast_from(
798            TraitRef::new(self.interner(), sized.into(), [ty]),
799            self.interner(),
800        );
801        self.try_obligation(sized_pred).certain()
802    }
803}

826    pub(super) struct Resolver<'a, 'db> {
827        ctx: &'a mut InferenceTable<'db>,
828        /// Whether we should normalize, disabled when resolving predicates.
829        should_normalize: bool,
830        nested_goals: &'a mut Vec<Goal<'db, Predicate<'db>>>,
831    }
832
833    impl<'a, 'db> Resolver<'a, 'db> {
834        pub(super) fn new(
835            ctx: &'a mut InferenceTable<'db>,
836            should_normalize: bool,
837            nested_goals: &'a mut Vec<Goal<'db, Predicate<'db>>>,
838        ) -> Resolver<'a, 'db> {
839            Resolver { ctx, nested_goals, should_normalize }
840        }
841
842        fn handle_term<T>(
843            &mut self,
844            value: T,
845            outer_exclusive_binder: impl FnOnce(T) -> DebruijnIndex,
846        ) -> T
847        where
848            T: Into<Term<'db>> + TypeSuperFoldable<DbInterner<'db>> + Copy,
849        {
850            let value = if self.should_normalize {
851                let cause = ObligationCause::new();
852                let at = self.ctx.infer_ctxt.at(&cause, self.ctx.trait_env.env);
853                let universes = vec![None; outer_exclusive_binder(value).as_usize()];
854                match deeply_normalize_with_skipped_universes_and_ambiguous_coroutine_goals(
855                    at, value, universes,
856                ) {
857                    Ok((value, goals)) => {
858                        self.nested_goals.extend(goals);
859                        value
860                    }
861                    Err(_errors) => {
862                        // FIXME: Report the error.
863                        value
864                    }
865                }
866            } else {
867                value
868            };
869
870            value.fold_with(&mut ReplaceInferWithError::new(self.ctx.interner()))
871        }
872    }
873
874    impl<'cx, 'db> TypeFolder<DbInterner<'db>> for Resolver<'cx, 'db> {
875        fn cx(&self) -> DbInterner<'db> {
876            self.ctx.interner()
877        }
878
879        fn fold_region(&mut self, r: Region<'db>) -> Region<'db> {
880            if r.is_var() { Region::error(self.ctx.interner()) } else { r }
881        }
882
883        fn fold_ty(&mut self, ty: Ty<'db>) -> Ty<'db> {
884            self.handle_term(ty, |it| it.outer_exclusive_binder())
885        }
886
887        fn fold_const(&mut self, ct: Const<'db>) -> Const<'db> {
888            self.handle_term(ct, |it| it.outer_exclusive_binder())
889        }
890
891        fn fold_predicate(&mut self, predicate: Predicate<'db>) -> Predicate<'db> {
892            assert!(
893                !self.should_normalize,
894                "normalizing predicates in writeback is not generally sound"
895            );
896            predicate.super_fold_with(self)
897        }
898    }
