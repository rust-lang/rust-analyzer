39struct NestedObligationsForSelfTy<'a, 'db> {
40    ctx: &'a InferenceTable<'db>,
41    self_ty: TyVid,
42    root_cause: &'a ObligationCause,
43    obligations_for_self_ty: &'a mut SmallVec<[Obligation<'db, Predicate<'db>>; 4]>,
44}
45
46impl<'a, 'db> ProofTreeVisitor<'db> for NestedObligationsForSelfTy<'a, 'db> {
47    type Result = ();
48
49    fn config(&self) -> InspectConfig {
50        // Using an intentionally low depth to minimize the chance of future
51        // breaking changes in case we adapt the approach later on. This also
52        // avoids any hangs for exponentially growing proof trees.
53        InspectConfig { max_depth: 5 }
54    }
55
56    fn visit_goal(&mut self, inspect_goal: &InspectGoal<'_, 'db>) {
57        // No need to walk into goal subtrees that certainly hold, since they
58        // wouldn't then be stalled on an infer var.
59        if inspect_goal.result() == Ok(Certainty::Yes) {
60            return;
61        }
62
63        let db = self.ctx.interner();
64        let goal = inspect_goal.goal();
65        if self.ctx.predicate_has_self_ty(goal.predicate, self.self_ty) {
66            self.obligations_for_self_ty.push(Obligation::new(
67                db,
68                self.root_cause.clone(),
69                goal.param_env,
70                goal.predicate,
71            ));
72        }
73
74        // If there's a unique way to prove a given goal, recurse into
75        // that candidate. This means that for `impl<F: FnOnce(u32)> Trait<F> for () {}`
76        // and a `(): Trait<?0>` goal we recurse into the impl and look at
77        // the nested `?0: FnOnce(u32)` goal.
78        if let Some(candidate) = inspect_goal.unique_applicable_candidate() {
79            candidate.visit_nested_no_probe(self)
80        }
81    }
82}

84/// Check if types unify.
85///
86/// Note that we consider placeholder types to unify with everything.
87/// This means that there may be some unresolved goals that actually set bounds for the placeholder
88/// type for the types to unify. For example `Option<T>` and `Option<U>` unify although there is
89/// unresolved goal `T = U`.
90pub fn could_unify<'db>(
91    db: &'db dyn HirDatabase,
92    env: ParamEnvAndCrate<'db>,
93    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
94) -> bool {
95    could_unify_impl(db, env, tys, |ctxt| ctxt.try_evaluate_obligations())
96}
97
98/// Check if types unify eagerly making sure there are no unresolved goals.
99///
100/// This means that placeholder types are not considered to unify if there are any bounds set on
101/// them. For example `Option<T>` and `Option<U>` do not unify as we cannot show that `T = U`
102pub fn could_unify_deeply<'db>(
103    db: &'db dyn HirDatabase,
104    env: ParamEnvAndCrate<'db>,
105    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
106) -> bool {
107    could_unify_impl(db, env, tys, |ctxt| ctxt.evaluate_obligations_error_on_ambiguity())
108}
109
110fn could_unify_impl<'db>(
111    db: &'db dyn HirDatabase,
112    env: ParamEnvAndCrate<'db>,
113    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
114    select: for<'a> fn(&mut ObligationCtxt<'a, 'db>) -> Vec<NextSolverError<'db>>,
115) -> bool {
116    let interner = DbInterner::new_with(db, env.krate);
117    let infcx = interner.infer_ctxt().build(TypingMode::PostAnalysis);
118    let cause = ObligationCause::dummy();
119    let at = infcx.at(&cause, env.param_env);
120    let ((ty1_with_vars, ty2_with_vars), _) = infcx.instantiate_canonical(tys);
121    let mut ctxt = ObligationCtxt::new(&infcx);
122    let can_unify = at
123        .eq(ty1_with_vars, ty2_with_vars)
124        .map(|infer_ok| ctxt.register_infer_ok_obligations(infer_ok))
125        .is_ok();
126    can_unify && select(&mut ctxt).is_empty()
127}

177    pub(crate) fn type_var_is_sized(&self, self_ty: TyVid) -> bool {
178        let Some(sized_did) = self.interner().lang_items().Sized else {
179            return true;
180        };
181        self.obligations_for_self_ty(self_ty).into_iter().any(|obligation| {
182            match obligation.predicate.kind().skip_binder() {
183                PredicateKind::Clause(ClauseKind::Trait(data)) => data.def_id().0 == sized_did,
184                _ => false,
185            }
186        })
187    }

189    pub(super) fn obligations_for_self_ty(
190        &self,
191        self_ty: TyVid,
192    ) -> SmallVec<[Obligation<'db, Predicate<'db>>; 4]> {
193        let obligations = self.fulfillment_cx.pending_obligations();
194        let mut obligations_for_self_ty = SmallVec::new();
195        for obligation in obligations {
196            let mut visitor = NestedObligationsForSelfTy {
197                ctx: self,
198                self_ty,
199                obligations_for_self_ty: &mut obligations_for_self_ty,
200                root_cause: &obligation.cause,
201            };
202
203            let goal = obligation.as_goal();
204            self.infer_ctxt.visit_proof_tree(goal, &mut visitor);
205        }
206
207        obligations_for_self_ty.retain_mut(|obligation| {
208            obligation.predicate = self.infer_ctxt.resolve_vars_if_possible(obligation.predicate);
209            !obligation.predicate.has_placeholders()
210        });
211        obligations_for_self_ty
212    }

214    fn predicate_has_self_ty(&self, predicate: Predicate<'db>, expected_vid: TyVid) -> bool {
215        match predicate.kind().skip_binder() {
216            PredicateKind::Clause(ClauseKind::Trait(data)) => {
217                self.type_matches_expected_vid(expected_vid, data.self_ty())
218            }
219            PredicateKind::Clause(ClauseKind::Projection(data)) => {
220                self.type_matches_expected_vid(expected_vid, data.projection_term.self_ty())
221            }
222            PredicateKind::Clause(ClauseKind::ConstArgHasType(..))
223            | PredicateKind::Subtype(..)
224            | PredicateKind::Coerce(..)
225            | PredicateKind::Clause(ClauseKind::RegionOutlives(..))
226            | PredicateKind::Clause(ClauseKind::TypeOutlives(..))
227            | PredicateKind::Clause(ClauseKind::WellFormed(..))
228            | PredicateKind::DynCompatible(..)
229            | PredicateKind::NormalizesTo(..)
230            | PredicateKind::AliasRelate(..)
231            | PredicateKind::Clause(ClauseKind::ConstEvaluatable(..))
232            | PredicateKind::ConstEquate(..)
233            | PredicateKind::Clause(ClauseKind::HostEffect(..))
234            | PredicateKind::Clause(ClauseKind::UnstableFeature(_))
235            | PredicateKind::Ambiguous => false,
236        }
237    }

239    fn type_matches_expected_vid(&self, expected_vid: TyVid, ty: Ty<'db>) -> bool {
240        let ty = self.shallow_resolve(ty);
241
242        match ty.kind() {
243            TyKind::Infer(rustc_type_ir::TyVar(found_vid)) => {
244                self.infer_ctxt.root_var(expected_vid) == self.infer_ctxt.root_var(found_vid)
245            }
246            _ => false,
247        }
248    }

334    where
335        T: rustc_type_ir::TypeFoldable<DbInterner<'db>>,
336    {
337        if !t.has_non_region_infer() {

356    /// Try to resolve `ty` to a structural type, normalizing aliases.
357    ///
358    /// In case there is still ambiguity, the returned type may be an inference
359    /// variable. This is different from `structurally_resolve_type` which errors
360    /// in this case.
361    pub(crate) fn try_structurally_resolve_type(&mut self, ty: Ty<'db>) -> Ty<'db> {
362        if let TyKind::Alias(..) = ty.kind() {
363            // We need to use a separate variable here as otherwise the temporary for
364            // `self.fulfillment_cx.borrow_mut()` is alive in the `Err` branch, resulting
365            // in a reentrant borrow, causing an ICE.
366            let result = self
367                .infer_ctxt
368                .at(&ObligationCause::misc(), self.param_env)
369                .structurally_normalize_ty(ty, &mut self.fulfillment_cx);
370            match result {
371                Ok(normalized_ty) => normalized_ty,
372                Err(_errors) => Ty::new_error(self.interner(), ErrorGuaranteed),
373            }
374        } else {
375            self.resolve_vars_with_obligations(ty)
376        }
377    }

427    #[tracing::instrument(level = "debug", skip(self))]
428    fn register_obligation_in_env(&mut self, goal: Goal<'db, Predicate<'db>>) {
429        let result = next_trait_solve_in_ctxt(&self.infer_ctxt, goal);
430        tracing::debug!(?result);
431        match result {
432            Ok((_, Certainty::Yes)) => {}
433            Err(rustc_type_ir::solve::NoSolution) => {}
434            Ok((_, Certainty::Maybe { .. })) => {
435                self.fulfillment_cx.register_predicate_obligation(
436                    &self.infer_ctxt,
437                    Obligation::new(
438                        self.interner(),
439                        ObligationCause::new(),
440                        goal.param_env,
441                        goal.predicate,
442                    ),
443                );
444            }
445        }
446    }

492    pub(crate) fn callable_sig(
493        &mut self,
494        ty: Ty<'db>,
495        num_args: usize,
496    ) -> Option<(Option<FnTrait>, Vec<Ty<'db>>, Ty<'db>)> {
497        match ty.callable_sig(self.interner()) {
498            Some(sig) => {
499                let sig = sig.skip_binder();
500                Some((None, sig.inputs_and_output.inputs().to_vec(), sig.output()))
501            }
502            None => {
503                let (f, args_ty, return_ty) = self.callable_sig_from_fn_trait(ty, num_args)?;
504                Some((Some(f), args_ty, return_ty))
505            }
506        }
507    }
508
509    fn callable_sig_from_fn_trait(
510        &mut self,
511        ty: Ty<'db>,
512        num_args: usize,
513    ) -> Option<(FnTrait, Vec<Ty<'db>>, Ty<'db>)> {
514        let lang_items = self.interner().lang_items();
515        for (fn_trait_name, output_assoc_name, subtraits) in [
516            (FnTrait::FnOnce, sym::Output, &[FnTrait::Fn, FnTrait::FnMut][..]),
517            (FnTrait::AsyncFnMut, sym::CallRefFuture, &[FnTrait::AsyncFn]),
518            (FnTrait::AsyncFnOnce, sym::CallOnceFuture, &[]),
519        ] {
520            let fn_trait = fn_trait_name.get_id(lang_items)?;
521            let trait_data = fn_trait.trait_items(self.db);
522            let output_assoc_type =
523                trait_data.associated_type_by_name(&Name::new_symbol_root(output_assoc_name))?;
524
525            let mut arg_tys = Vec::with_capacity(num_args);
526            let arg_ty = Ty::new_tup_from_iter(
527                self.interner(),
528                std::iter::repeat_with(|| {
529                    let ty = self.next_ty_var();
530                    arg_tys.push(ty);
531                    ty
532                })
533                .take(num_args),
534            );
535            let args = GenericArgs::new_from_slice(&[ty.into(), arg_ty.into()]);
536            let trait_ref = TraitRef::new_from_args(self.interner(), fn_trait.into(), args);
537
538            let proj_args = self.infer_ctxt.fill_rest_fresh_args(output_assoc_type.into(), args);
539            let projection = Ty::new_alias(
540                self.interner(),
541                rustc_type_ir::AliasTyKind::Projection,
542                AliasTy::new_from_args(self.interner(), output_assoc_type.into(), proj_args),
543            );
544
545            let pred = Predicate::upcast_from(trait_ref, self.interner());
546            if !self.try_obligation(pred).no_solution() {
547                self.register_obligation(pred);
548                let return_ty = self.normalize_alias_ty(projection);
549                for &fn_x in subtraits {
550                    let fn_x_trait = fn_x.get_id(lang_items)?;
551                    let trait_ref =
552                        TraitRef::new_from_args(self.interner(), fn_x_trait.into(), args);
553                    let pred = Predicate::upcast_from(trait_ref, self.interner());
554                    if !self.try_obligation(pred).no_solution() {
555                        return Some((fn_x, arg_tys, return_ty));
556                    }
557                }
558                return Some((fn_trait_name, arg_tys, return_ty));
559            }
560        }
561        None
562    }

564    pub(super) fn insert_type_vars<T>(&mut self, ty: T) -> T
565    where
566        T: TypeFoldable<DbInterner<'db>>,
567    {
568        self.infer_ctxt.insert_type_vars(ty)

584    pub(crate) fn process_remote_user_written_ty(&mut self, ty: Ty<'db>) -> Ty<'db> {
585        let ty = self.insert_type_vars(ty);

598    /// Check if given type is `Sized` or not
599    pub(crate) fn is_sized(&mut self, ty: Ty<'db>) -> bool {
600        fn short_circuit_trivial_tys(ty: Ty<'_>) -> Option<bool> {
601            match ty.kind() {
602                TyKind::Bool
603                | TyKind::Char
604                | TyKind::Int(_)
605                | TyKind::Uint(_)
606                | TyKind::Float(_)
607                | TyKind::Ref(..)
608                | TyKind::RawPtr(..)
609                | TyKind::Never
610                | TyKind::FnDef(..)
611                | TyKind::Array(..)
612                | TyKind::FnPtr(..) => Some(true),
613                TyKind::Slice(..) | TyKind::Str | TyKind::Dynamic(..) => Some(false),
614                _ => None,
615            }
616        }
617
618        let mut ty = ty;
619        ty = self.try_structurally_resolve_type(ty);
620        if let Some(sized) = short_circuit_trivial_tys(ty) {
621            return sized;
622        }
623
624        {
625            let mut structs = SmallVec::<[_; 8]>::new();
626            // Must use a loop here and not recursion because otherwise users will conduct completely
627            // artificial examples of structs that have themselves as the tail field and complain r-a crashes.
628            while let Some((AdtId::StructId(id), subst)) = ty.as_adt() {
629                let struct_data = id.fields(self.db);
630                if let Some((last_field, _)) = struct_data.fields().iter().next_back() {
631                    let last_field_ty = self.db.field_types(id.into())[last_field]
632                        .get()
633                        .instantiate(self.interner(), subst);
634                    if structs.contains(&ty) {
635                        // A struct recursively contains itself as a tail field somewhere.
636                        return true; // Don't overload the users with too many errors.
637                    }
638                    structs.push(ty);
639                    // Structs can have DST as its last field and such cases are not handled
640                    // as unsized by the chalk, so we do this manually.
641                    ty = last_field_ty;
642                    ty = self.try_structurally_resolve_type(ty);
643                    if let Some(sized) = short_circuit_trivial_tys(ty) {
644                        return sized;
645                    }
646                } else {
647                    break;
648                };
649            }
650        }
651
652        let Some(sized) = self.interner().lang_items().Sized else {
653            return false;
654        };
655        let sized_pred = Predicate::upcast_from(
656            TraitRef::new(self.interner(), sized.into(), [ty]),
657            self.interner(),
658        );
659        self.try_obligation(sized_pred).certain()
660    }
661}

672mod resolve_completely {
673    use rustc_type_ir::{DebruijnIndex, Flags, TypeFolder, TypeSuperFoldable};
674
675    use crate::{
676        infer::unify::InferenceTable,
677        next_solver::{
678            Const, DbInterner, Goal, Predicate, Region, Term, Ty,
679            infer::{resolve::ReplaceInferWithError, traits::ObligationCause},
680            normalize::deeply_normalize_with_skipped_universes_and_ambiguous_coroutine_goals,
681        },
682    };
683
684    pub(super) struct Resolver<'a, 'db> {
685        ctx: &'a mut InferenceTable<'db>,
686        /// Whether we should normalize, disabled when resolving predicates.
687        should_normalize: bool,
688        nested_goals: &'a mut Vec<Goal<'db, Predicate<'db>>>,
689    }
690
691    impl<'a, 'db> Resolver<'a, 'db> {
692        pub(super) fn new(
693            ctx: &'a mut InferenceTable<'db>,
694            should_normalize: bool,
695            nested_goals: &'a mut Vec<Goal<'db, Predicate<'db>>>,
696        ) -> Resolver<'a, 'db> {
697            Resolver { ctx, nested_goals, should_normalize }
698        }
699
700        fn handle_term<T>(
701            &mut self,
702            value: T,
703            outer_exclusive_binder: impl FnOnce(T) -> DebruijnIndex,
704        ) -> T
705        where
706            T: Into<Term<'db>> + TypeSuperFoldable<DbInterner<'db>> + Copy,
707        {
708            let value = if self.should_normalize {
709                let cause = ObligationCause::new();
710                let at = self.ctx.at(&cause);
711                let universes = vec![None; outer_exclusive_binder(value).as_usize()];
712                match deeply_normalize_with_skipped_universes_and_ambiguous_coroutine_goals(
713                    at, value, universes,
714                ) {
715                    Ok((value, goals)) => {
716                        self.nested_goals.extend(goals);
717                        value
718                    }
719                    Err(_errors) => {
720                        // FIXME: Report the error.
721                        value
722                    }
723                }
724            } else {
725                value
726            };
727
728            value.fold_with(&mut ReplaceInferWithError::new(self.ctx.interner()))
729        }
730    }
731
732    impl<'cx, 'db> TypeFolder<DbInterner<'db>> for Resolver<'cx, 'db> {
733        fn cx(&self) -> DbInterner<'db> {
734            self.ctx.interner()
735        }
736
737        fn fold_region(&mut self, r: Region<'db>) -> Region<'db> {
738            if r.is_var() { Region::error(self.ctx.interner()) } else { r }
739        }
740
741        fn fold_ty(&mut self, ty: Ty<'db>) -> Ty<'db> {
742            self.handle_term(ty, |it| it.outer_exclusive_binder())
743        }
744
745        fn fold_const(&mut self, ct: Const<'db>) -> Const<'db> {
746            self.handle_term(ct, |it| it.outer_exclusive_binder())
747        }
748
749        fn fold_predicate(&mut self, predicate: Predicate<'db>) -> Predicate<'db> {
750            assert!(
751                !self.should_normalize,
752                "normalizing predicates in writeback is not generally sound"
753            );
754            predicate.super_fold_with(self)
755        }
756    }
