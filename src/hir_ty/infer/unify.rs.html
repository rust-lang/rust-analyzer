37struct NestedObligationsForSelfTy<'a, 'db> {
38    ctx: &'a InferenceTable<'db>,
39    self_ty: TyVid,
40    root_cause: &'a ObligationCause,
41    obligations_for_self_ty: &'a mut SmallVec<[Obligation<'db, Predicate<'db>>; 4]>,
42}
43
44impl<'a, 'db> ProofTreeVisitor<'db> for NestedObligationsForSelfTy<'a, 'db> {
45    type Result = ();
46
47    fn span(&self) -> Span {
48        self.root_cause.span()
49    }
50
51    fn config(&self) -> InspectConfig {
52        // Using an intentionally low depth to minimize the chance of future
53        // breaking changes in case we adapt the approach later on. This also
54        // avoids any hangs for exponentially growing proof trees.
55        InspectConfig { max_depth: 5 }
56    }
57
58    fn visit_goal(&mut self, inspect_goal: &InspectGoal<'_, 'db>) {
59        // No need to walk into goal subtrees that certainly hold, since they
60        // wouldn't then be stalled on an infer var.
61        if inspect_goal.result() == Ok(Certainty::Yes) {
62            return;
63        }
64
65        let db = self.ctx.interner();
66        let goal = inspect_goal.goal();
67        if self.ctx.predicate_has_self_ty(goal.predicate, self.self_ty) {
68            self.obligations_for_self_ty.push(Obligation::new(
69                db,
70                *self.root_cause,
71                goal.param_env,
72                goal.predicate,
73            ));
74        }
75
76        // If there's a unique way to prove a given goal, recurse into
77        // that candidate. This means that for `impl<F: FnOnce(u32)> Trait<F> for () {}`
78        // and a `(): Trait<?0>` goal we recurse into the impl and look at
79        // the nested `?0: FnOnce(u32)` goal.
80        if let Some(candidate) = inspect_goal.unique_applicable_candidate() {
81            candidate.visit_nested_no_probe(self)
82        }
83    }
84}

86/// Check if types unify.
87///
88/// Note that we consider placeholder types to unify with everything.
89/// This means that there may be some unresolved goals that actually set bounds for the placeholder
90/// type for the types to unify. For example `Option<T>` and `Option<U>` unify although there is
91/// unresolved goal `T = U`.
92pub fn could_unify<'db>(
93    db: &'db dyn HirDatabase,
94    env: ParamEnvAndCrate<'db>,
95    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
96) -> bool {
97    could_unify_impl(db, env, tys, |ctxt| ctxt.try_evaluate_obligations())
98}
99
100/// Check if types unify eagerly making sure there are no unresolved goals.
101///
102/// This means that placeholder types are not considered to unify if there are any bounds set on
103/// them. For example `Option<T>` and `Option<U>` do not unify as we cannot show that `T = U`
104pub fn could_unify_deeply<'db>(
105    db: &'db dyn HirDatabase,
106    env: ParamEnvAndCrate<'db>,
107    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
108) -> bool {
109    could_unify_impl(db, env, tys, |ctxt| ctxt.evaluate_obligations_error_on_ambiguity())
110}
111
112fn could_unify_impl<'db>(
113    db: &'db dyn HirDatabase,
114    env: ParamEnvAndCrate<'db>,
115    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
116    select: for<'a> fn(&mut ObligationCtxt<'a, 'db>) -> Vec<NextSolverError<'db>>,
117) -> bool {
118    let interner = DbInterner::new_with(db, env.krate);
119    let infcx = interner.infer_ctxt().build(TypingMode::PostAnalysis);
120    let cause = ObligationCause::dummy();
121    let at = infcx.at(&cause, env.param_env);
122    let ((ty1_with_vars, ty2_with_vars), _) = infcx.instantiate_canonical(Span::Dummy, tys);
123    let mut ctxt = ObligationCtxt::new(&infcx);
124    let can_unify = at
125        .eq(ty1_with_vars, ty2_with_vars)
126        .map(|infer_ok| ctxt.register_infer_ok_obligations(infer_ok))
127        .is_ok();
128    can_unify && select(&mut ctxt).is_empty()
129}

180    pub(crate) fn type_var_is_sized(&self, self_ty: TyVid) -> bool {
181        let Some(sized_did) = self.interner().lang_items().Sized else {
182            return true;
183        };
184        self.obligations_for_self_ty(self_ty).into_iter().any(|obligation| {
185            match obligation.predicate.kind().skip_binder() {
186                PredicateKind::Clause(ClauseKind::Trait(data)) => data.def_id().0 == sized_did,
187                _ => false,
188            }
189        })
190    }

192    pub(super) fn obligations_for_self_ty(
193        &self,
194        self_ty: TyVid,
195    ) -> SmallVec<[Obligation<'db, Predicate<'db>>; 4]> {
196        let obligations = self.fulfillment_cx.pending_obligations();
197        let mut obligations_for_self_ty = SmallVec::new();
198        for obligation in obligations {
199            let mut visitor = NestedObligationsForSelfTy {
200                ctx: self,
201                self_ty,
202                obligations_for_self_ty: &mut obligations_for_self_ty,
203                root_cause: &obligation.cause,
204            };
205
206            let goal = obligation.as_goal();
207            self.infer_ctxt.visit_proof_tree(goal, &mut visitor);
208        }
209
210        obligations_for_self_ty.retain_mut(|obligation| {
211            obligation.predicate = self.infer_ctxt.resolve_vars_if_possible(obligation.predicate);
212            !obligation.predicate.has_placeholders()
213        });
214        obligations_for_self_ty
215    }

217    fn predicate_has_self_ty(&self, predicate: Predicate<'db>, expected_vid: TyVid) -> bool {
218        match predicate.kind().skip_binder() {
219            PredicateKind::Clause(ClauseKind::Trait(data)) => {
220                self.type_matches_expected_vid(expected_vid, data.self_ty())
221            }
222            PredicateKind::Clause(ClauseKind::Projection(data)) => {
223                self.type_matches_expected_vid(expected_vid, data.projection_term.self_ty())
224            }
225            PredicateKind::Clause(ClauseKind::ConstArgHasType(..))
226            | PredicateKind::Subtype(..)
227            | PredicateKind::Coerce(..)
228            | PredicateKind::Clause(ClauseKind::RegionOutlives(..))
229            | PredicateKind::Clause(ClauseKind::TypeOutlives(..))
230            | PredicateKind::Clause(ClauseKind::WellFormed(..))
231            | PredicateKind::DynCompatible(..)
232            | PredicateKind::NormalizesTo(..)
233            | PredicateKind::AliasRelate(..)
234            | PredicateKind::Clause(ClauseKind::ConstEvaluatable(..))
235            | PredicateKind::ConstEquate(..)
236            | PredicateKind::Clause(ClauseKind::HostEffect(..))
237            | PredicateKind::Clause(ClauseKind::UnstableFeature(_))
238            | PredicateKind::Ambiguous => false,
239        }
240    }

242    fn type_matches_expected_vid(&self, expected_vid: TyVid, ty: Ty<'db>) -> bool {
243        let ty = self.shallow_resolve(ty);
244
245        match ty.kind() {
246            TyKind::Infer(rustc_type_ir::TyVar(found_vid)) => {
247                self.infer_ctxt.root_var(expected_vid) == self.infer_ctxt.root_var(found_vid)
248            }
249            _ => false,
250        }
251    }

322    /// Try to resolve `ty` to a structural type, normalizing aliases.
323    ///
324    /// In case there is still ambiguity, the returned type may be an inference
325    /// variable. This is different from `structurally_resolve_type` which errors
326    /// in this case.
327    pub(crate) fn try_structurally_resolve_type(&mut self, span: Span, ty: Ty<'db>) -> Ty<'db> {
328        if let TyKind::Alias(..) = ty.kind() {
329            let result = self
330                .infer_ctxt
331                .at(&ObligationCause::new(span), self.param_env)
332                .structurally_normalize_ty(ty, &mut self.fulfillment_cx);
333            match result {
334                Ok(normalized_ty) => normalized_ty,
335                Err(errors) => {
336                    self.trait_errors.extend(errors);
337                    Ty::new_error(self.interner(), ErrorGuaranteed)
338                }
339            }
340        } else {
341            self.resolve_vars_with_obligations(ty)
342        }
343    }

486        table: InferenceTable<'db>,
487        diagnostics: ThinVec<InferenceDiagnostic>,

493    impl<'db> WriteBackCtxt<'db> {
494        pub(crate) fn new(
495            table: InferenceTable<'db>,
496            diagnostics: ThinVec<InferenceDiagnostic>,
497            vars_emitted_type_must_be_known_for: FxHashSet<Term<'db>>,
498        ) -> Self {
499            let spans_emitted_type_must_be_known_for = vars_emitted_type_must_be_known_for
500                .into_iter()
501                .filter_map(|term| match term.kind() {
502                    TermKind::Ty(ty) => match ty.kind() {
503                        TyKind::Infer(InferTy::TyVar(vid)) => {
504                            Some(table.infer_ctxt.type_var_span(vid))
505                        }
506                        _ => None,
507                    },
508                    TermKind::Const(ct) => match ct.kind() {
509                        ConstKind::Infer(InferConst::Var(vid)) => {
510                            table.infer_ctxt.const_var_span(vid)
511                        }
512                        _ => None,
513                    },
514                })
515                .collect();
516
517            Self {
518                types: table.interner().default_types(),
519                table,
520                diagnostics,
521                has_errors: false,
522                spans_emitted_type_must_be_known_for,
523            }
524        }
525
526        pub(crate) fn resolve_completely<T>(&mut self, value_ref: &mut T)
527        where
528            T: TypeFoldable<DbInterner<'db>>,
529        {
530            self.resolve_completely_with_default(value_ref, value_ref.clone());
531        }
532
533        pub(crate) fn resolve_completely_with_default<T>(&mut self, value_ref: &mut T, default: T)
534        where
535            T: TypeFoldable<DbInterner<'db>>,
536        {
537            let value = std::mem::replace(value_ref, default);
538
539            let value = self.table.resolve_vars_if_possible(value);
540
541            let mut goals = vec![];
542
543            // FIXME(next-solver): Handle `goals`.
544
545            *value_ref = value.fold_with(&mut Resolver::new(self, true, &mut goals));
546        }
547
548        pub(crate) fn resolve_diagnostics(mut self) -> (ThinVec<InferenceDiagnostic>, bool) {
549            let has_errors = self.has_errors;
550
551            self.table.emit_trait_errors(&mut self.diagnostics);
552
553            // Ignore diagnostics made from resolving diagnostics.
554            let mut diagnostics = std::mem::take(&mut self.diagnostics);
555            diagnostics.retain_mut(|diagnostic| {
556                self.resolve_completely(diagnostic);
557
558                if let InferenceDiagnostic::ExpectedFunction { found: ty, .. }
559                | InferenceDiagnostic::UnresolvedField { receiver: ty, .. }
560                | InferenceDiagnostic::UnresolvedMethodCall { receiver: ty, .. } = diagnostic
561                    && ty.as_ref().references_non_lt_error()
562                {
563                    false
564                } else {
565                    true
566                }
567            });
568            diagnostics.shrink_to_fit();
569
570            (diagnostics, has_errors)
571        }
572    }

579    impl<'db> DiagnoseInferVars<'_, 'db> {
580        const TYPE_FLAGS: TypeFlags = TypeFlags::HAS_INFER.union(TypeFlags::HAS_NON_REGION_ERROR);
581
582        fn err_on_span(&mut self, span: Span) {
583            if !self.ctx.spans_emitted_type_must_be_known_for.insert(span) {
584                // Suppress duplicate diagnostics.
585                return;
586            }
587
588            if span.is_dummy() {
589                return;
590            }
591
592            // We have to be careful not to insert infer vars here, as we won't resolve this new diagnostic.
593            let top_term = self.top_term.fold_with(&mut ReplaceInferWithError::new(self.cx()));
594            self.ctx.diagnostics.push(InferenceDiagnostic::TypeMustBeKnown {
595                at_point: span,
596                top_term: Some(GenericArg::from(top_term).store()),
597            });
598        }
599    }

601    impl<'db> TypeFolder<DbInterner<'db>> for DiagnoseInferVars<'_, 'db> {
602        fn cx(&self) -> DbInterner<'db> {
603            self.ctx.table.interner()
604        }
605
606        fn fold_ty(&mut self, t: Ty<'db>) -> Ty<'db> {
607            if !t.has_type_flags(Self::TYPE_FLAGS) {
608                return t;
609            }
610
611            match t.kind() {
612                TyKind::Error(_) => {
613                    self.ctx.has_errors = true;
614                    t
615                }
616                TyKind::Infer(infer_ty) => match infer_ty {
617                    InferTy::TyVar(vid) => {
618                        self.err_on_span(self.ctx.table.infer_ctxt.type_var_span(vid));
619                        self.ctx.has_errors = true;
620                        self.ctx.types.types.error
621                    }
622                    InferTy::IntVar(_) => {
623                        never!("fallback should have resolved all int vars");
624                        self.ctx.types.types.i32
625                    }
626                    InferTy::FloatVar(_) => {
627                        never!("fallback should have resolved all float vars");
628                        self.ctx.types.types.f64
629                    }
630                    InferTy::FreshTy(_) | InferTy::FreshIntTy(_) | InferTy::FreshFloatTy(_) => {
631                        never!("should not have fresh infer vars outside of caching");
632                        self.ctx.has_errors = true;
633                        self.ctx.types.types.error
634                    }
635                },
636                _ => t.super_fold_with(self),
637            }
638        }
639
640        fn fold_const(&mut self, c: Const<'db>) -> Const<'db> {
641            if !c.has_type_flags(Self::TYPE_FLAGS) {
642                return c;
643            }
644
645            match c.kind() {
646                ConstKind::Error(_) => {
647                    self.ctx.has_errors = true;
648                    c
649                }
650                ConstKind::Infer(infer_ct) => match infer_ct {
651                    InferConst::Var(vid) => {
652                        if let Some(span) = self.ctx.table.infer_ctxt.const_var_span(vid) {
653                            self.err_on_span(span);
654                        }
655                        self.ctx.has_errors = true;
656                        self.ctx.types.consts.error
657                    }
658                    InferConst::Fresh(_) => {
659                        never!("should not have fresh infer vars outside of caching");
660                        self.ctx.has_errors = true;
661                        self.ctx.types.consts.error
662                    }
663                },
664                _ => c.super_fold_with(self),
665            }
666        }
667
668        fn fold_predicate(&mut self, p: Predicate<'db>) -> Predicate<'db> {
669            if !p.has_type_flags(Self::TYPE_FLAGS) {
670                return p;
671            }
672            p.super_fold_with(self)
673        }
674
675        fn fold_region(&mut self, r: Region<'db>) -> Region<'db> {
676            if r.is_var() {
677                // For now, we don't error on regions.
678                self.ctx.types.regions.error
679            } else {
680                r
681            }
682        }
683    }

685    pub(super) struct Resolver<'a, 'db> {
686        ctx: &'a mut WriteBackCtxt<'db>,
687        /// Whether we should normalize, disabled when resolving predicates.
688        should_normalize: bool,
689        nested_goals: &'a mut Vec<Goal<'db, Predicate<'db>>>,
690    }
691
692    impl<'a, 'db> Resolver<'a, 'db> {
693        pub(super) fn new(
694            ctx: &'a mut WriteBackCtxt<'db>,
695            should_normalize: bool,
696            nested_goals: &'a mut Vec<Goal<'db, Predicate<'db>>>,
697        ) -> Resolver<'a, 'db> {
698            Resolver { ctx, nested_goals, should_normalize }
699        }
700
701        fn handle_term<T>(
702            &mut self,
703            value: T,
704            outer_exclusive_binder: impl FnOnce(T) -> DebruijnIndex,
705        ) -> T
706        where
707            T: Into<Term<'db>> + TypeSuperFoldable<DbInterner<'db>> + Copy,
708        {
709            let value = if self.should_normalize {
710                // FIXME: This should not use a dummy span.
711                let cause = ObligationCause::new(Span::Dummy);
712                let at = self.ctx.table.at(&cause);
713                let universes = vec![None; outer_exclusive_binder(value).as_usize()];
714                match deeply_normalize_with_skipped_universes_and_ambiguous_coroutine_goals(
715                    at, value, universes,
716                ) {
717                    Ok((value, goals)) => {
718                        self.nested_goals.extend(goals);
719                        value
720                    }
721                    Err(errors) => {
722                        self.ctx.table.trait_errors.extend(errors);
723                        value
724                    }
725                }
726            } else {
727                value
728            };
729
730            value.fold_with(&mut DiagnoseInferVars { ctx: self.ctx, top_term: value.into() })
731        }
732    }
733
734    impl<'db> TypeFolder<DbInterner<'db>> for Resolver<'_, 'db> {
735        fn cx(&self) -> DbInterner<'db> {
736            self.ctx.table.interner()
737        }
738
739        fn fold_region(&mut self, r: Region<'db>) -> Region<'db> {
740            if r.is_var() { self.ctx.types.regions.error } else { r }
741        }
742
743        fn fold_ty(&mut self, ty: Ty<'db>) -> Ty<'db> {
744            self.handle_term(ty, |it| it.outer_exclusive_binder())
745        }
746
747        fn fold_const(&mut self, ct: Const<'db>) -> Const<'db> {
748            self.handle_term(ct, |it| it.outer_exclusive_binder())
749        }
750
751        fn fold_predicate(&mut self, predicate: Predicate<'db>) -> Predicate<'db> {
752            assert!(
753                !self.should_normalize,
754                "normalizing predicates in writeback is not generally sound"
755            );
756            predicate.super_fold_with(self)
757        }
758    }
