81/// A `Place` represents how a value is located in memory. This does not
82/// always correspond to a syntactic place expression. For example, when
83/// processing a pattern, a `Place` can be used to refer to the sub-value
84/// currently being inspected.
85#[derive(Clone, Debug, PartialEq, Eq, Hash, TypeVisitable, TypeFoldable)]
86pub struct Place {
87    /// The type of the `PlaceBase`
88    pub base_ty: StoredTy,
89    /// The "outermost" place that holds this value.
90    #[type_visitable(ignore)]
91    pub base: PlaceBase,
92    /// How this place is derived from the base place.
93    pub projections: Vec<Projection>,
94}
95
96impl Place {
97    /// Returns an iterator of the types that have to be dereferenced to access
98    /// the `Place`.
99    ///
100    /// The types are in the reverse order that they are applied. So if
101    /// `x: &*const u32` and the `Place` is `**x`, then the types returned are
102    ///`*const u32` then `&*const u32`.
103    pub fn deref_tys<'db>(&self) -> impl Iterator<Item = Ty<'db>> {
104        self.projections.iter().enumerate().rev().filter_map(move |(index, proj)| {
105            if ProjectionKind::Deref == proj.kind {
106                Some(self.ty_before_projection(index))
107            } else {
108                None
109            }
110        })
111    }
112
113    /// Returns the type of this `Place` after all projections have been applied.
114    pub fn ty<'db>(&self) -> Ty<'db> {
115        self.projections.last().map_or(self.base_ty.as_ref(), |proj| proj.ty.as_ref())
116    }
117
118    /// Returns the type of this `Place` immediately before `projection_index`th projection
119    /// is applied.
120    pub fn ty_before_projection<'db>(&self, projection_index: usize) -> Ty<'db> {
121        assert!(projection_index < self.projections.len());
122        if projection_index == 0 {
123            self.base_ty.as_ref()
124        } else {
125            self.projections[projection_index - 1].ty.as_ref()
126        }
127    }
128}

130/// A `PlaceWithOrigin` represents how a value is located in memory. This does not
131/// always correspond to a syntactic place expression. For example, when
132/// processing a pattern, a `Place` can be used to refer to the sub-value
133/// currently being inspected.
134#[derive(Clone, Debug, PartialEq, Eq, Hash)]
135pub(crate) struct PlaceWithOrigin {
136    /// `ExprId`s or `PatId`s of the expressions or patterns producing this value.
137    pub origins: SmallVec<[CaptureSourceStack; 2]>,
138
139    /// Information about the `Place`.
140    pub place: Place,
141}
142
143impl PlaceWithOrigin {
144    fn new_no_projections<'db>(
145        origin: impl Into<ExprOrPatId>,
146        base_ty: Ty<'db>,
147        base: PlaceBase,
148    ) -> PlaceWithOrigin {
149        Self::new(
150            smallvec![CaptureSourceStack::from_single(origin.into())],
151            base_ty,
152            base,
153            Vec::new(),
154        )
155    }
156
157    fn new<'db>(
158        origins: SmallVec<[CaptureSourceStack; 2]>,
159        base_ty: Ty<'db>,
160        base: PlaceBase,
161        projections: Vec<Projection>,
162    ) -> PlaceWithOrigin {
163        debug_assert!(origins.iter().all(|origin| origin.len() == projections.len() + 1));
164        PlaceWithOrigin { origins, place: Place { base_ty: base_ty.store(), base, projections } }
165    }
166
167    fn push_projection(&mut self, projection: Projection, origin: ExprOrPatId) {
168        self.place.projections.push(projection);
169        for origin_stack in &mut self.origins {
170            origin_stack.push(origin);
171        }
172    }
173
174    pub(crate) fn span(&self) -> Span {
175        match self.origins.first() {
176            Some(origin) => origin.final_source().into(),
177            None => Span::Dummy,
178        }
179    }
180}

182/// The `FakeReadCause` describes the type of pattern why a FakeRead statement exists.
183#[derive(Copy, Clone, Debug, Hash, PartialEq, Eq)]
184pub enum FakeReadCause {
185    /// A fake read injected into a match guard to ensure that the discriminants
186    /// that are being matched on aren't modified while the match guard is being
187    /// evaluated.
188    ///
189    /// At the beginning of each match guard, a fake borrow is
190    /// inserted for each discriminant accessed in the entire `match` statement.
191    ///
192    /// Then, at the end of the match guard, a `FakeRead(ForMatchGuard)` is
193    /// inserted to keep the fake borrows alive until that point.
194    ///
195    /// This should ensure that you cannot change the variant for an enum while
196    /// you are in the midst of matching on it.
197    ForMatchGuard,
198
199    /// Fake read of the scrutinee of a `match` or destructuring `let`
200    /// (i.e. `let` with non-trivial pattern).
201    ///
202    /// In `match x { ... }`, we generate a `FakeRead(ForMatchedPlace, x)`
203    /// and insert it into the `otherwise_block` (which is supposed to be
204    /// unreachable for irrefutable pattern-matches like `match` or `let`).
205    ///
206    /// This is necessary because `let x: !; match x {}` doesn't generate any
207    /// actual read of x, so we need to generate a `FakeRead` to check that it
208    /// is initialized.
209    ///
210    /// If the `FakeRead(ForMatchedPlace)` is being performed with a closure
211    /// that doesn't capture the required upvars, the `FakeRead` within the
212    /// closure is omitted entirely.
213    ///
214    /// To make sure that this is still sound, if a closure matches against
215    /// a Place starting with an Upvar, we hoist the `FakeRead` to the
216    /// definition point of the closure.
217    ///
218    /// If the `FakeRead` comes from being hoisted out of a closure like this,
219    /// we record the `ExprId` of the closure. Otherwise, the `Option` will be `None`.
220    //
221    // We can use LocalDefId here since fake read statements are removed
222    // before codegen in the `CleanupNonCodegenStatements` pass.
223    ForMatchedPlace(Option<ExprId>),
224
225    /// A fake read injected into a match guard to ensure that the places
226    /// bound by the pattern are immutable for the duration of the match guard.
227    ///
228    /// Within a match guard, references are created for each place that the
229    /// pattern creates a binding for — this is known as the `RefWithinGuard`
230    /// version of the variables. To make sure that the references stay
231    /// alive until the end of the match guard, and properly prevent the
232    /// places in question from being modified, a `FakeRead(ForGuardBinding)`
233    /// is inserted at the end of the match guard.
234    ///
235    /// For details on how these references are created, see the extensive
236    /// documentation on `bind_matched_candidate_for_guard` in
237    /// `rustc_mir_build`.
238    ForGuardBinding,
239
240    /// Officially, the semantics of
241    ///
242    /// `let pattern = <expr>;`
243    ///
244    /// is that `<expr>` is evaluated into a temporary and then this temporary is
245    /// into the pattern.
246    ///
247    /// However, if we see the simple pattern `let var = <expr>`, we optimize this to
248    /// evaluate `<expr>` directly into the variable `var`. This is mostly unobservable,
249    /// but in some cases it can affect the borrow checker, as in #53695.
250    ///
251    /// Therefore, we insert a `FakeRead(ForLet)` immediately after each `let`
252    /// with a trivial pattern.
253    ///
254    /// FIXME: `ExprUseVisitor` has an entirely different opinion on what `FakeRead(ForLet)`
255    /// is supposed to mean. If it was accurate to what MIR lowering does,
256    /// would it even make sense to hoist these out of closures like
257    /// `ForMatchedPlace`?
258    ForLet(Option<ExprId>),
259
260    /// Currently, index expressions overloaded through the `Index` trait
261    /// get lowered differently than index expressions with builtin semantics
262    /// for arrays and slices — the latter will emit code to perform
263    /// bound checks, and then return a MIR place that will only perform the
264    /// indexing "for real" when it gets incorporated into an instruction.
265    ///
266    /// This is observable in the fact that the following compiles:
267    ///
268    /// ```
269    /// fn f(x: &mut [&mut [u32]], i: usize) {
270    ///     x[i][x[i].len() - 1] += 1;
271    /// }
272    /// ```
273    ///
274    /// However, we need to be careful to not let the user invalidate the
275    /// bound check with an expression like
276    ///
277    /// `(*x)[1][{ x = y; 4}]`
278    ///
279    /// Here, the first bounds check would be invalidated when we evaluate the
280    /// second index expression. To make sure that this doesn't happen, we
281    /// create a fake borrow of `x` and hold it while we evaluate the second
282    /// index.
283    ///
284    /// This borrow is kept alive by a `FakeRead(ForIndex)` at the end of its
285    /// scope.
286    ForIndex,
287}

289/// This trait defines the callbacks you can expect to receive when
290/// employing the ExprUseVisitor.
291pub(crate) trait Delegate<'db> {
292    /// The value found at `place` is moved, depending
293    /// on `mode`. Where `diag_expr_id` is the id used for diagnostics for `place`.
294    ///
295    /// If the value is `Copy`, [`copy`][Self::copy] is called instead, which
296    /// by default falls back to [`borrow`][Self::borrow].
297    ///
298    /// The parameter `diag_expr_id` indicates the HIR id that ought to be used for
299    /// diagnostics. Around pattern matching such as `let pat = expr`, the diagnostic
300    /// id will be the id of the expression `expr` but the place itself will have
301    /// the id of the binding in the pattern `pat`.
302    fn consume(&mut self, place_with_id: PlaceWithOrigin, ctx: &mut InferenceContext<'_, 'db>);
303
304    /// The value found at `place` is used, depending
305    /// on `mode`. Where `diag_expr_id` is the id used for diagnostics for `place`.
306    ///
307    /// Use of a `Copy` type in a ByUse context is considered a use
308    /// by `ImmBorrow` and `borrow` is called instead. This is because
309    /// a shared borrow is the "minimum access" that would be needed
310    /// to perform a copy.
311    ///
312    ///
313    /// The parameter `diag_expr_id` indicates the HIR id that ought to be used for
314    /// diagnostics. Around pattern matching such as `let pat = expr`, the diagnostic
315    /// id will be the id of the expression `expr` but the place itself will have
316    /// the id of the binding in the pattern `pat`.
317    fn use_cloned(&mut self, place_with_id: PlaceWithOrigin, ctx: &mut InferenceContext<'_, 'db>);
318
319    /// The value found at `place` is being borrowed with kind `bk`.
320    /// `diag_expr_id` is the id used for diagnostics (see `consume` for more details).
321    fn borrow(
322        &mut self,
323        place_with_id: PlaceWithOrigin,
324        bk: BorrowKind,
325        ctx: &mut InferenceContext<'_, 'db>,
326    );
327
328    /// The value found at `place` is being copied.
329    /// `diag_expr_id` is the id used for diagnostics (see `consume` for more details).
330    ///
331    /// If an implementation is not provided, use of a `Copy` type in a ByValue context is instead
332    /// considered a use by `ImmBorrow` and `borrow` is called instead. This is because a shared
333    /// borrow is the "minimum access" that would be needed to perform a copy.
334    fn copy(&mut self, place_with_id: PlaceWithOrigin, ctx: &mut InferenceContext<'_, 'db>) {
335        // In most cases, copying data from `x` is equivalent to doing `*&x`, so by default
336        // we treat a copy of `x` as a borrow of `x`.
337        self.borrow(place_with_id, BorrowKind::Immutable, ctx)
338    }
339
340    /// The path at `assignee_place` is being assigned to.
341    /// `diag_expr_id` is the id used for diagnostics (see `consume` for more details).
342    fn mutate(&mut self, assignee_place: PlaceWithOrigin, ctx: &mut InferenceContext<'_, 'db>);
343
344    /// The path at `binding_place` is a binding that is being initialized.
345    ///
346    /// This covers cases such as `let x = 42;`
347    fn bind(&mut self, binding_place: PlaceWithOrigin, ctx: &mut InferenceContext<'_, 'db>) {
348        // Bindings can normally be treated as a regular assignment, so by default we
349        // forward this to the mutate callback.
350        self.mutate(binding_place, ctx)
351    }
352
353    /// The `place` should be a fake read because of specified `cause`.
354    fn fake_read(
355        &mut self,
356        place_with_id: PlaceWithOrigin,
357        cause: FakeReadCause,
358        ctx: &mut InferenceContext<'_, 'db>,
359    );
360}

403///
404/// See [module-level docs][self] and [`Delegate`] for details.
405pub(crate) struct ExprUseVisitor<'a, 'b, 'db, D: Delegate<'db>> {
406    cx: &'a mut InferenceContext<'b, 'db>,
407    delegate: D,
408    closure_expr: ExprId,
409    upvars: UpvarsRef<'db>,
410}
411
412impl<'a, 'b, 'db, D: Delegate<'db>> ExprUseVisitor<'a, 'b, 'db, D> {
413    /// Creates the ExprUseVisitor, configuring it with the various options provided:
414    ///
415    /// - `delegate` -- who receives the callbacks
416    /// - `param_env` --- parameter environment for trait lookups (esp. pertaining to `Copy`)
417    /// - `typeck_results` --- typeck results for the code being analyzed
418    pub(crate) fn new(
419        cx: &'a mut InferenceContext<'b, 'db>,
420        closure_expr: ExprId,
421        upvars: UpvarsRef<'db>,
422        delegate: D,
423    ) -> Self {
424        ExprUseVisitor { delegate, closure_expr, upvars, cx }
425    }
426
427    pub(crate) fn consume_closure_body(&mut self, params: &[PatId], body: ExprId) -> Result {
428        for &param in params {
429            let param_ty = self.pat_ty_adjusted(param)?;
430            debug!("consume_body: param_ty = {:?}", param_ty);
431
432            let param_place = self.cat_rvalue(param.into(), param_ty);
433
434            self.fake_read_scrutinee(param_place.clone(), false);
435            self.walk_pat(param_place, param, false)?;
436        }
437
438        self.consume_expr(body)?;
439
440        Ok(())
441    }
442
443    #[instrument(skip(self), level = "debug")]
444    fn consume_or_copy(&mut self, place_with_id: PlaceWithOrigin) {
445        if self.cx.table.type_is_copy_modulo_regions(place_with_id.place.ty()) {
446            self.delegate.copy(place_with_id, self.cx);
447        } else {
448            self.delegate.consume(place_with_id, self.cx);
449        }
450    }
451
452    #[instrument(skip(self), level = "debug")]
453    pub(crate) fn consume_clone_or_copy(&mut self, place_with_id: PlaceWithOrigin) {
454        // `x.use` will do one of the following
455        // * if it implements `Copy`, it will be a copy
456        // * if it implements `UseCloned`, it will be a call to `clone`
457        // * otherwise, it is a move
458        //
459        // we do a conservative approximation of this, treating it as a move unless we know that it implements copy or `UseCloned`
460        if self.cx.table.type_is_copy_modulo_regions(place_with_id.place.ty()) {
461            self.delegate.copy(place_with_id, self.cx);
462        } else if self.cx.table.type_is_use_cloned_modulo_regions(place_with_id.place.ty()) {
463            self.delegate.use_cloned(place_with_id, self.cx);
464        } else {
465            self.delegate.consume(place_with_id, self.cx);
466        }
467    }
468
469    fn consume_exprs(&mut self, exprs: &[ExprId]) -> Result {
470        for &expr in exprs {
471            self.consume_expr(expr)?;
472        }
473        Ok(())
474    }
475
476    // FIXME: It's suspicious that this is public; clippy should probably use `walk_expr`.
477    #[instrument(skip(self), level = "debug")]
478    pub(crate) fn consume_expr(&mut self, expr: ExprId) -> Result {
479        let place_with_id = self.cat_expr(expr)?;
480        self.consume_or_copy(place_with_id);
481        self.walk_expr(expr)?;
482        Ok(())
483    }
484
485    fn mutate_expr(&mut self, expr: ExprId) -> Result {
486        let place_with_id = self.cat_expr(expr)?;
487        self.delegate.mutate(place_with_id, self.cx);
488        self.walk_expr(expr)?;
489        Ok(())
490    }
491
492    #[instrument(skip(self), level = "debug")]
493    fn borrow_expr(&mut self, expr: ExprId, bk: BorrowKind) -> Result {
494        let place_with_id = self.cat_expr(expr)?;
495        self.delegate.borrow(place_with_id, bk, self.cx);
496        self.walk_expr(expr)?;
497        Ok(())
498    }
499
500    #[instrument(skip(self), level = "debug")]
501    pub(crate) fn walk_expr(&mut self, expr: ExprId) -> Result {
502        self.walk_adjustment(expr)?;
503
504        match self.cx.store[expr] {
505            Expr::Path(_) => {}
506
507            Expr::UnaryOp { op: UnaryOp::Deref, expr: base } => {
508                // *base
509                self.walk_expr(base)?;
510            }
511
512            Expr::Field { expr: base, .. } => {
513                // base.f
514                self.walk_expr(base)?;
515            }
516
517            Expr::Index { base: lhs, index: rhs } => {
518                // lhs[rhs]
519                self.walk_expr(lhs)?;
520                self.consume_expr(rhs)?;
521            }
522
523            Expr::Call { callee, ref args } => {
524                // callee(args)
525                self.consume_expr(callee)?;
526                self.consume_exprs(args)?;
527            }
528
529            Expr::MethodCall { receiver, ref args, .. } => {
530                // callee.m(args)
531                self.consume_expr(receiver)?;
532                self.consume_exprs(args)?;
533            }
534
535            Expr::RecordLit { ref fields, spread, .. } => {
536                self.walk_struct_expr(fields, spread)?;
537            }
538
539            Expr::Tuple { ref exprs } => {
540                self.consume_exprs(exprs)?;
541            }
542
543            Expr::If {
544                condition: cond_expr,
545                then_branch: then_expr,
546                else_branch: opt_else_expr,
547            } => {
548                self.consume_expr(cond_expr)?;
549                self.consume_expr(then_expr)?;
550                if let Some(else_expr) = opt_else_expr {
551                    self.consume_expr(else_expr)?;
552                }
553            }
554
555            Expr::Let { pat, expr: init } => {
556                self.walk_local(init, pat, None, |this| {
557                    this.borrow_expr(init, BorrowKind::Immutable)
558                })?;
559            }
560
561            Expr::Match { expr: discr, ref arms } => {
562                let discr_place = self.cat_expr(discr)?;
563                self.fake_read_scrutinee(discr_place.clone(), true);
564                self.walk_expr(discr)?;
565
566                for arm in arms {
567                    self.walk_arm(discr_place.clone(), arm)?;
568                }
569            }
570
571            Expr::Array(Array::ElementList { elements: ref exprs }) => {
572                self.consume_exprs(exprs)?;
573            }
574
575            Expr::Ref { expr: base, mutability: m, .. } => {
576                // &base
577                // make sure that the thing we are pointing out stays valid
578                // for the lifetime `scope_r` of the resulting ptr:
579                let bk = BorrowKind::from_hir_mutbl(m);
580                self.borrow_expr(base, bk)?;
581            }
582
583            Expr::InlineAsm(ref asm) => {
584                for (_, op) in &asm.operands {
585                    match *op {
586                        AsmOperand::In { expr, .. } => {
587                            self.consume_expr(expr)?;
588                        }
589                        AsmOperand::Out { expr: Some(expr), .. }
590                        | AsmOperand::InOut { expr, .. } => {
591                            self.mutate_expr(expr)?;
592                        }
593                        AsmOperand::SplitInOut { in_expr, out_expr, .. } => {
594                            self.consume_expr(in_expr)?;
595                            if let Some(out_expr) = out_expr {
596                                self.mutate_expr(out_expr)?;
597                            }
598                        }
599                        AsmOperand::Out { expr: None, .. }
600                        | AsmOperand::Const { .. }
601                        | AsmOperand::Sym { .. } => {}
602                        AsmOperand::Label(block) => {
603                            self.walk_expr(block)?;
604                        }
605                    }
606                }
607            }
608
609            Expr::Continue { .. }
610            | Expr::Literal(..)
611            | Expr::Const(..)
612            | Expr::OffsetOf(..)
613            | Expr::Missing
614            | Expr::Underscore => {}
615
616            Expr::Loop { body: blk, .. } => {
617                self.walk_expr(blk)?;
618            }
619
620            Expr::UnaryOp { expr: lhs, .. } => {
621                self.consume_expr(lhs)?;
622            }
623
624            Expr::BinaryOp {
625                lhs,
626                rhs,
627                op: Some(BinaryOp::ArithOp(..) | BinaryOp::CmpOp(..) | BinaryOp::LogicOp(..)),
628            } => {
629                self.consume_expr(lhs)?;
630                self.consume_expr(rhs)?;
631            }
632
633            Expr::Block { ref statements, tail, .. }
634            | Expr::Unsafe { ref statements, tail, .. } => {
635                for stmt in statements {
636                    self.walk_stmt(stmt)?;
637                }
638
639                if let Some(tail_expr) = tail {
640                    self.consume_expr(tail_expr)?;
641                }
642            }
643
644            Expr::Break { expr: opt_expr, .. } | Expr::Return { expr: opt_expr } => {
645                if let Some(expr) = opt_expr {
646                    self.consume_expr(expr)?;
647                }
648            }
649
650            Expr::Become { expr } | Expr::Await { expr } | Expr::Box { expr } => {
651                self.consume_expr(expr)?;
652            }
653
654            Expr::Assignment { target, value } => {
655                self.walk_expr(value)?;
656                let expr_place = self.cat_expr(value)?;
657                let update_guard =
658                    self.cx.resolver.update_to_inner_scope(self.cx.db, self.cx.owner, expr);
659                self.walk_pat(expr_place, target, false)?;
660                self.cx.resolver.reset_to_guard(update_guard);
661            }
662
663            Expr::Cast { expr: base, .. } => {
664                self.consume_expr(base)?;
665            }
666
667            Expr::BinaryOp { lhs, rhs, op: None | Some(BinaryOp::Assignment { .. }) } => {
668                self.consume_expr(lhs)?;
669                self.consume_expr(rhs)?;
670            }
671
672            Expr::Array(Array::Repeat { initializer: base, .. }) => {
673                self.consume_expr(base)?;
674            }
675
676            Expr::Closure { .. } => {
677                self.walk_captures(expr);
678            }
679
680            Expr::Yield { expr: value } | Expr::Yeet { expr: value } => {
681                if let Some(value) = value {
682                    self.consume_expr(value)?;
683                }
684            }
685
686            Expr::Range { lhs, rhs, .. } => {
687                if let Some(lhs) = lhs {
688                    self.consume_expr(lhs)?;
689                }
690                if let Some(rhs) = rhs {
691                    self.consume_expr(rhs)?;
692                }
693            }
694        }
695        Ok(())
696    }

699        match *stmt {
700            Statement::Let { pat, initializer: Some(expr), else_branch: els, .. } => {

708                // only the fn body we were given.

711            Statement::Expr { expr, .. } => {
712                self.consume_expr(expr)?;
713            }
714        }

749    fn walk_struct_expr(&mut self, fields: &[RecordLitField], spread: RecordSpread) -> Result {
750        // Consume the expressions supplying values for each field.
751        for field in fields {
752            self.consume_expr(field.expr)?;
753        }
754
755        let RecordSpread::Expr(with_expr) = spread else { return Ok(()) };
756
757        let with_place = self.cat_expr(with_expr)?;
758
759        // Select just those fields of the `with`
760        // expression that will actually be used
761        match self.cx.structurally_resolve_type(with_expr.into(), with_place.place.ty()).kind() {
762            TyKind::Adt(adt, args) if adt.is_struct() => {
763                let AdtId::StructId(adt) = adt.def_id() else { unreachable!() };
764                let adt_fields = VariantId::from(adt).fields(self.cx.db).fields();
765                let adt_field_types = self.cx.db.field_types(adt.into());
766                // Consume those fields of the with expression that are needed.
767                for (f_index, with_field) in adt_fields.iter() {
768                    let is_mentioned = fields.iter().any(|f| f.name == with_field.name);
769                    if !is_mentioned {
770                        let field_place = self.cat_projection(
771                            with_expr.into(),
772                            with_place.clone(),
773                            adt_field_types[f_index].get().instantiate(self.cx.interner(), args),
774                            ProjectionKind::Field {
775                                field_idx: f_index.into_raw().into_u32(),
776                                variant_idx: 0,
777                            },
778                        );
779                        self.consume_or_copy(field_place);
780                    }
781                }
782            }
783            _ => {}
784        }
785
786        // walk the with expression so that complex expressions
787        // are properly handled.
788        self.walk_expr(with_expr)?;
789
790        Ok(())
791    }

798    /// Invoke the appropriate delegate calls for anything that gets
799    /// consumed or borrowed as part of the automatic adjustment
800    /// process.
801    fn walk_adjustment(&mut self, expr: ExprId) -> Result {
802        let adjustments = self.expr_adjustments(expr);
803        let mut place_with_id = self.cat_expr_unadjusted(expr)?;
804        for adjustment in &adjustments {
805            debug!("walk_adjustment expr={:?} adj={:?}", expr, adjustment);
806            match adjustment.kind {
807                Adjust::NeverToAny | Adjust::Pointer(_) => {
808                    // Creating a closure/fn-pointer or unsizing consumes
809                    // the input and stores it into the resulting rvalue.
810                    self.consume_or_copy(place_with_id.clone());
811                }
812
813                Adjust::Deref(None) => {}
814
815                // Autoderefs for overloaded Deref calls in fact reference
816                // their receiver. That is, if we have `(*x)` where `x`
817                // is of type `Rc<T>`, then this in fact is equivalent to
818                // `x.deref()`. Since `deref()` is declared with `&self`,
819                // this is an autoref of `x`.
820                Adjust::Deref(Some(ref deref)) => {
821                    let bk = BorrowKind::from_mutbl(deref.0);
822                    self.delegate.borrow(place_with_id.clone(), bk, self.cx);
823                }
824
825                Adjust::Borrow(ref autoref) => {
826                    self.walk_autoref(expr, place_with_id.clone(), autoref);
827                }
828            }
829            place_with_id = self.cat_expr_adjusted(expr, place_with_id, adjustment)?;
830        }
831        Ok(())
832    }

863    /// The core driver for walking a pattern
864    ///
865    /// This should mirror how pattern-matching gets lowered to MIR, as
866    /// otherwise lowering will ICE when trying to resolve the upvars.
867    ///
868    /// However, it is okay to approximate it here by doing *more* accesses than
869    /// the actual MIR builder will, which is useful when some checks are too
870    /// cumbersome to perform here. For example, if after typeck it becomes
871    /// clear that only one variant of an enum is inhabited, and therefore a
872    /// read of the discriminant is not necessary, `walk_pat` will have
873    /// over-approximated the necessary upvar capture granularity.
874    ///
875    /// Do note that discrepancies like these do still create obscure corners
876    /// in the semantics of the language, and should be avoided if possible.
877    #[instrument(skip(self), level = "debug")]
878    fn walk_pat(&mut self, discr_place: PlaceWithOrigin, pat: PatId, has_guard: bool) -> Result {
879        self.cat_pattern(discr_place.clone(), pat, &mut |this, place, pat| {
880            debug!("walk_pat: pat.kind={:?}", this.cx.store[pat]);
881            let read_discriminant = {
882                let place = place.clone();
883                |this: &mut Self| {
884                    this.delegate.borrow(place, BorrowKind::Immutable, this.cx);
885                }
886            };
887
888            match this.cx.store[pat] {
889                Pat::Bind { id, .. } => {
890                    debug!("walk_pat: binding place={:?} pat={:?}", place, pat);
891                    let bm = this.cx.result.binding_modes[pat];
892                    debug!("walk_pat: pat.hir_id={:?} bm={:?}", pat, bm);
893
894                    // pat_ty: the type of the binding being produced.
895                    let pat_ty = this.node_ty(pat.into())?;
896                    debug!("walk_pat: pat_ty={:?}", pat_ty);
897
898                    if let Ok(binding_place) = this.cat_local(pat.into(), pat_ty, id) {
899                        this.delegate.bind(binding_place, this.cx);
900                    }
901
902                    // Subtle: MIR desugaring introduces immutable borrows for each pattern
903                    // binding when lowering pattern guards to ensure that the guard does not
904                    // modify the scrutinee.
905                    if has_guard {
906                        read_discriminant(this);
907                    }
908
909                    // It is also a borrow or copy/move of the value being matched.
910                    // In a cases of pattern like `let pat = upvar`, don't use the span
911                    // of the pattern, as this just looks confusing, instead use the span
912                    // of the discriminant.
913                    match this.cx.result.binding_mode(pat).ok_or(ErrorGuaranteed)?.0 {
914                        ByRef::Yes(m) => {
915                            let bk = BorrowKind::from_mutbl(m);
916                            this.delegate.borrow(place, bk, this.cx);
917                        }
918                        ByRef::No => {
919                            debug!("walk_pat binding consuming pat");
920                            this.consume_or_copy(place);
921                        }
922                    }
923                }
924                Pat::Path(ref path) => {
925                    // A `Path` pattern is just a name like `Foo`. This is either a
926                    // named constant or else it refers to an ADT variant
927
928                    let is_assoc_const = this
929                        .cx
930                        .result
931                        .assoc_resolutions_for_pat(pat)
932                        .is_some_and(|it| matches!(it.0, CandidateId::ConstId(_)));
933                    let resolution = this.cx.resolver.resolve_path_in_value_ns_fully(
934                        this.cx.db,
935                        path,
936                        this.cx.store.pat_path_hygiene(pat),
937                    );
938                    let is_normal_const = matches!(resolution, Some(ValueNs::ConstId(_)));
939                    if is_assoc_const || is_normal_const {
940                        // Named constants have to be equated with the value
941                        // being matched, so that's a read of the value being matched.
942                        //
943                        // FIXME: Does the MIR code skip this read when matching on a ZST?
944                        // If so, we can also skip it here.
945                        read_discriminant(this);
946                    } else if this.is_multivariant_adt(pat.into(), place.place.ty()) {
947                        // Otherwise, this is a struct/enum variant, and so it's
948                        // only a read if we need to read the discriminant.
949                        read_discriminant(this);
950                    }
951                }
952                Pat::Lit(_) | Pat::ConstBlock(_) | Pat::Range { .. } => {
953                    // When matching against a literal or range, we need to
954                    // borrow the place to compare it against the pattern.
955                    //
956                    // Note that we do this read even if the range matches all
957                    // possible values, such as 0..=u8::MAX. This is because
958                    // we don't want to depend on consteval here.
959                    //
960                    // FIXME: What if the type being matched only has one
961                    // possible value?
962                    read_discriminant(this);
963                }
964                Pat::Record { .. } | Pat::TupleStruct { .. } => {
965                    if this.is_multivariant_adt(pat.into(), place.place.ty()) {
966                        read_discriminant(this);
967                    }
968                }
969                Pat::Slice { prefix: ref lhs, slice: wild, suffix: ref rhs } => {
970                    // We don't need to test the length if the pattern is `[..]`
971                    if matches!((&**lhs, wild, &**rhs), (&[], Some(_), &[]))
972                        // Arrays have a statically known size, so
973                        // there is no need to read their length
974                        || place.place.ty().strip_references().is_array()
975                    {
976                        // No read necessary
977                    } else {
978                        read_discriminant(this);
979                    }
980                }
981                Pat::Expr(expr) => {
982                    // Destructuring assignment.
983                    this.mutate_expr(expr)?;
984                }
985                Pat::Or(_)
986                | Pat::Box { .. }
987                | Pat::Ref { .. }
988                | Pat::Tuple { .. }
989                | Pat::Wild
990                | Pat::Missing
991                | Pat::Rest => {
992                    // If the PatKind is Or, Box, Ref, Guard, or Tuple, the relevant accesses
993                    // are made later as these patterns contains subpatterns.
994                    // If the PatKind is Missing, Wild or Err, any relevant accesses are made when processing
995                    // the other patterns that are part of the match
996                }
997            }
998
999            Ok(())
1000        })
1001    }

1003    /// Handle the case where the current body contains a closure.
1004    ///
1005    /// When the current body being handled is a closure, then we must make sure that
1006    /// - The parent closure only captures Places from the nested closure that are not local to it.
1007    ///
1008    /// In the following example the closures `c` only captures `p.x` even though `incr`
1009    /// is a capture of the nested closure
1010    ///
1011    /// ```
1012    /// struct P { x: i32 }
1013    /// let mut p = P { x: 4 };
1014    /// let c = || {
1015    ///    let incr = 10;
1016    ///    let nested = || p.x += incr;
1017    /// };
1018    /// ```
1019    ///
1020    /// - When reporting the Place back to the Delegate, ensure that the UpvarId uses the enclosing
1021    /// closure as the DefId.
1022    #[instrument(skip(self), level = "debug")]
1023    fn walk_captures(&mut self, closure_expr: ExprId) {
1024        fn upvar_is_local_variable(upvars: UpvarsRef<'_>, var_id: BindingId) -> bool {
1025            upvars.contains(var_id)
1026        }
1027
1028        // If we have a nested closure, we want to include the fake reads present in the nested
1029        // closure.
1030        // `remove()` then re-insert and not `get()` due to borrowck errors.
1031        if let Some(closure_data) = self.cx.result.closures_data.remove(&closure_expr) {
1032            for (fake_read, cause, origins) in closure_data.fake_reads.iter() {
1033                match fake_read.base {
1034                    PlaceBase::Upvar { var_id, closure: _ } => {
1035                        if upvar_is_local_variable(self.upvars, var_id) {
1036                            // The nested closure might be fake reading the current (enclosing) closure's local variables.
1037                            // The only places we want to fake read before creating the parent closure are the ones that
1038                            // are not local to it/ defined by it.
1039                            //
1040                            // ```rust,ignore(cannot-test-this-because-pseudo-code)
1041                            // let v1 = (0, 1);
1042                            // let c = || { // fake reads: v1
1043                            //    let v2 = (0, 1);
1044                            //    let e = || { // fake reads: v1, v2
1045                            //       let (_, t1) = v1;
1046                            //       let (_, t2) = v2;
1047                            //    }
1048                            // }
1049                            // ```
1050                            // This check is performed when visiting the body of the outermost closure (`c`) and ensures
1051                            // that we don't add a fake read of v2 in c.
1052                            continue;
1053                        }
1054                    }
1055                    _ => {
1056                        panic!(
1057                            "Do not know how to get ExprId out of Rvalue and StaticItem {:?}",
1058                            fake_read.base
1059                        );
1060                    }
1061                };
1062                self.delegate.fake_read(
1063                    PlaceWithOrigin { place: fake_read.clone(), origins: origins.clone() },
1064                    *cause,
1065                    self.cx,
1066                );
1067            }
1068
1069            for (var_id, min_list) in closure_data.min_captures.iter() {
1070                if !self.upvars.contains(*var_id) {
1071                    // The nested closure might be capturing the current (enclosing) closure's local variables.
1072                    // We check if the root variable is ever mentioned within the enclosing closure, if not
1073                    // then for the current body (if it's a closure) these aren't captures, we will ignore them.
1074                    continue;
1075                }
1076                for captured_place in min_list {
1077                    let place = &captured_place.place;
1078                    let capture_info = &captured_place.info;
1079
1080                    // Mark the place to be captured by the enclosing closure
1081                    let place_base =
1082                        PlaceBase::Upvar { var_id: *var_id, closure: self.closure_expr };
1083                    let place_with_id = PlaceWithOrigin::new(
1084                        capture_info.sources.clone(),
1085                        place.base_ty.as_ref(),
1086                        place_base,
1087                        place.projections.clone(),
1088                    );
1089
1090                    match capture_info.capture_kind {
1091                        UpvarCapture::ByValue => {
1092                            self.consume_or_copy(place_with_id);
1093                        }
1094                        UpvarCapture::ByUse => {
1095                            self.consume_clone_or_copy(place_with_id);
1096                        }
1097                        UpvarCapture::ByRef(upvar_borrow) => {
1098                            self.delegate.borrow(place_with_id, upvar_borrow, self.cx);
1099                        }
1100                    }
1101                }
1102            }
1103
1104            self.cx.result.closures_data.insert(closure_expr, closure_data);
1105        }
1106    }

1146    /// Returns the type of value that this pattern matches against.
1147    /// Some non-obvious cases:
1148    ///
1149    /// - a `ref x` binding matches against a value of type `T` and gives
1150    ///   `x` the type `&T`; we return `T`.
1151    /// - a pattern with implicit derefs (thanks to default binding
1152    ///   modes #42640) may look like `Some(x)` but in fact have
1153    ///   implicit deref patterns attached (e.g., it is really
1154    ///   `&Some(x)`). In that case, we return the "outermost" type
1155    ///   (e.g., `&Option<T>`).
1156    fn pat_ty_adjusted(&mut self, pat: PatId) -> Result<Ty<'db>> {
1157        // Check for implicit `&` types wrapping the pattern; note
1158        // that these are never attached to binding patterns, so
1159        // actually this is somewhat "disjoint" from the code below
1160        // that aims to account for `ref x`.
1161        if let Some(vec) = self.cx.result.pat_adjustment(pat) {
1162            if let Some(first_adjust) = vec.first() {
1163                debug!("pat_ty(pat={:?}) found adjustment `{:?}`", pat, first_adjust);
1164                return Ok(first_adjust.source.as_ref());
1165            }
1166        } else if let Pat::Ref { pat: subpat, .. } = self.cx.store[pat]
1167            && self.cx.result.is_skipped_ref_pat(pat)
1168        {
1169            return self.pat_ty_adjusted(subpat);
1170        }
1171
1172        self.pat_ty_unadjusted(pat)
1173    }
1174
1175    /// Like [`Self::pat_ty_adjusted`], but ignores implicit `&` patterns.
1176    fn pat_ty_unadjusted(&mut self, pat: PatId) -> Result<Ty<'db>> {
1177        let base_ty = self.node_ty(pat.into())?;
1178        trace!(?base_ty);
1179
1180        // This code detects whether we are looking at a `ref x`,
1181        // and if so, figures out what the type *being borrowed* is.
1182        match self.cx.store[pat] {
1183            Pat::Bind { .. } => {
1184                let bm = self.cx.result.binding_mode(pat).ok_or(ErrorGuaranteed)?;
1185
1186                if let ByRef::Yes(_) = bm.0 {
1187                    // a bind-by-ref means that the base_ty will be the type of the ident itself,
1188                    // but what we want here is the type of the underlying value being borrowed.
1189                    // So peel off one-level, turning the &T into T.
1190                    match self
1191                        .cx
1192                        .structurally_resolve_type(pat.into(), base_ty)
1193                        .builtin_deref(false)
1194                    {
1195                        Some(ty) => Ok(ty),
1196                        None => {
1197                            debug!("By-ref binding of non-derefable type: {base_ty:?}");
1198                            Err(ErrorGuaranteed)
1199                        }
1200                    }
1201                } else {
1202                    Ok(base_ty)
1203                }
1204            }
1205            _ => Ok(base_ty),
1206        }
1207    }

1213    /// This recursion helper avoids going through *too many*
1214    /// adjustments, since *only* non-overloaded deref recurses.
1215    fn cat_expr_(&mut self, expr: ExprId, adjustments: &[Adjustment]) -> Result<PlaceWithOrigin> {
1216        match adjustments.split_last() {
1217            None => self.cat_expr_unadjusted(expr),
1218            Some((adjustment, previous)) => {
1219                self.cat_expr_adjusted_with(expr, |this| this.cat_expr_(expr, previous), adjustment)
1220            }
1221        }
1222    }

1224    fn cat_expr_adjusted(
1225        &mut self,
1226        expr: ExprId,
1227        previous: PlaceWithOrigin,
1228        adjustment: &Adjustment,
1229    ) -> Result<PlaceWithOrigin> {
1230        self.cat_expr_adjusted_with(expr, |_this| Ok(previous), adjustment)
1231    }

1233    fn cat_expr_adjusted_with<F>(
1234        &mut self,
1235        expr: ExprId,
1236        previous: F,
1237        adjustment: &Adjustment,
1238    ) -> Result<PlaceWithOrigin>
1239    where
1240        F: FnOnce(&mut Self) -> Result<PlaceWithOrigin>,
1241    {
1242        let target = self.cx.infcx().resolve_vars_if_possible(adjustment.target.as_ref());
1243        match adjustment.kind {
1244            Adjust::Deref(overloaded) => {
1245                // Equivalent to *expr or something similar.
1246                let base = if let Some(deref) = overloaded {
1247                    let ref_ty = Ty::new_ref(
1248                        self.cx.interner(),
1249                        self.cx.types.regions.erased,
1250                        target,
1251                        deref.0,
1252                    );
1253                    self.cat_rvalue(expr.into(), ref_ty)
1254                } else {
1255                    previous(self)?
1256                };
1257                self.cat_deref(expr.into(), base)
1258            }
1259
1260            Adjust::NeverToAny | Adjust::Pointer(_) | Adjust::Borrow(_) => {
1261                // Result is an rvalue.
1262                Ok(self.cat_rvalue(expr.into(), target))
1263            }
1264        }
1265    }

1267    fn cat_expr_unadjusted(&mut self, expr: ExprId) -> Result<PlaceWithOrigin> {
1268        let expr_ty = self.expr_ty(expr)?;
1269        match self.cx.store[expr] {
1270            Expr::UnaryOp { expr: e_base, op: UnaryOp::Deref } => {
1271                if self.cx.result.method_resolutions.contains_key(&expr) {
1272                    self.cat_overloaded_place(expr, e_base)
1273                } else {
1274                    let base = self.cat_expr(e_base)?;
1275                    self.cat_deref(expr.into(), base)
1276                }
1277            }
1278
1279            Expr::Field { expr: base, .. } => {
1280                let base = self.cat_expr(base)?;
1281                debug!(?base);
1282
1283                let field_idx = self
1284                    .cx
1285                    .result
1286                    .field_resolutions
1287                    .get(&expr)
1288                    .map(|field| match *field {
1289                        Either::Left(field) => field.local_id.into_raw().into_u32(),
1290                        Either::Right(tuple_field) => tuple_field.index,
1291                    })
1292                    .ok_or(ErrorGuaranteed)?;
1293
1294                Ok(self.cat_projection(
1295                    expr.into(),
1296                    base,
1297                    expr_ty,
1298                    ProjectionKind::Field { field_idx, variant_idx: 0 },
1299                ))
1300            }
1301
1302            Expr::Index { base, index: _ } => {
1303                // rustc checks if this is an overloaded index, but the check is buggy and treats any indexing
1304                // as overloaded, see https://rust-lang.zulipchat.com/#narrow/channel/144729-t-types/topic/.E2.9C.94.20Is.20builtin.20indexing.20any.20special.20in.20typeck.3F/near/565881390.
1305                // So that's what we do here.
1306                self.cat_overloaded_place(expr, base)
1307            }
1308
1309            Expr::Path(ref path) => {
1310                let resolver_guard =
1311                    self.cx.resolver.update_to_inner_scope(self.cx.db, self.cx.owner, expr);
1312                let resolution = self.cx.resolver.resolve_path_in_value_ns_fully(
1313                    self.cx.db,
1314                    path,
1315                    self.cx.store.expr_path_hygiene(expr),
1316                );
1317                self.cx.resolver.reset_to_guard(resolver_guard);
1318                match (resolution, self.cx.result.assoc_resolutions_for_expr(expr)) {
1319                    (_, Some((CandidateId::FunctionId(_) | CandidateId::ConstId(_), _)))
1320                    | (
1321                        Some(
1322                            ValueNs::ConstId(_)
1323                            | ValueNs::GenericParam(_)
1324                            | ValueNs::FunctionId(_)
1325                            | ValueNs::ImplSelf(_)
1326                            | ValueNs::EnumVariantId(_)
1327                            | ValueNs::StructId(_),
1328                        ),
1329                        None,
1330                    ) => Ok(self.cat_rvalue(expr.into(), expr_ty)),
1331                    (Some(ValueNs::StaticId(_)), None) => Ok(PlaceWithOrigin::new_no_projections(
1332                        expr,
1333                        expr_ty,
1334                        PlaceBase::StaticItem,
1335                    )),
1336                    (Some(ValueNs::LocalBinding(var_id)), None) => {
1337                        self.cat_local(expr.into(), expr_ty, var_id)
1338                    }
1339                    (None, None) => Err(ErrorGuaranteed),
1340                }
1341            }
1342
1343            _ => Ok(self.cat_rvalue(expr.into(), expr_ty)),
1344        }
1345    }

1392    fn cat_overloaded_place(&mut self, expr: ExprId, base: ExprId) -> Result<PlaceWithOrigin> {
1393        // Reconstruct the output assuming it's a reference with the
1394        // same region and mutability as the receiver. This holds for
1395        // `Deref(Mut)::Deref(_mut)` and `Index(Mut)::index(_mut)`.
1396        let place_ty = self.expr_ty(expr)?;
1397        let base_ty = self.expr_ty_adjusted(base)?;
1398
1399        let TyKind::Ref(region, _, mutbl) =
1400            self.cx.structurally_resolve_type(base.into(), base_ty).kind()
1401        else {
1402            return Err(ErrorGuaranteed);
1403        };
1404        let ref_ty = Ty::new_ref(self.cx.interner(), region, place_ty, mutbl);
1405
1406        let base = self.cat_rvalue(expr.into(), ref_ty);
1407        self.cat_deref(expr.into(), base)
1408    }

1410    fn cat_deref(
1411        &mut self,
1412        node: ExprOrPatId,
1413        mut base_place: PlaceWithOrigin,
1414    ) -> Result<PlaceWithOrigin> {
1415        let base_curr_ty = base_place.place.ty();
1416        let Some(deref_ty) =
1417            self.cx.structurally_resolve_type(node, base_curr_ty).builtin_deref(true)
1418        else {
1419            debug!("explicit deref of non-derefable type: {:?}", base_curr_ty);
1420            return Err(ErrorGuaranteed);
1421        };
1422        base_place.push_projection(
1423            Projection { kind: ProjectionKind::Deref, ty: deref_ty.store() },
1424            node,
1425        );
1426        Ok(base_place)
1427    }

1444        match self.cx.structurally_resolve_type(pat_id.into(), ty).kind() {

1450    /// Here, `place` is the `PlaceWithId` being matched and pat is the pattern it
1451    /// is being matched against.
1452    ///
1453    /// In general, the way that this works is that we walk down the pattern,
1454    /// constructing a `PlaceWithId` that represents the path that will be taken
1455    /// to reach the value being matched.
1456    fn cat_pattern<F>(
1457        &mut self,
1458        mut place_with_id: PlaceWithOrigin,
1459        pat: PatId,
1460        op: &mut F,
1461    ) -> Result
1462    where
1463        F: FnMut(&mut Self, PlaceWithOrigin, PatId) -> Result,
1464    {
1465        // If (pattern) adjustments are active for this pattern, adjust the `PlaceWithId` correspondingly.
1466        // `PlaceWithId`s are constructed differently from patterns. For example, in
1467        //
1468        // ```
1469        // match foo {
1470        //     &&Some(x, ) => { ... },
1471        //     _ => { ... },
1472        // }
1473        // ```
1474        //
1475        // the pattern `&&Some(x,)` is represented as `Ref { Ref { TupleStruct }}`. To build the
1476        // corresponding `PlaceWithId` we start with the `PlaceWithId` for `foo`, and then, by traversing the
1477        // pattern, try to answer the question: given the address of `foo`, how is `x` reached?
1478        //
1479        // `&&Some(x,)` `place_foo`
1480        //  `&Some(x,)` `deref { place_foo}`
1481        //   `Some(x,)` `deref { deref { place_foo }}`
1482        //       `(x,)` `field0 { deref { deref { place_foo }}}` <- resulting place
1483        //
1484        // The above example has no adjustments. If the code were instead the (after adjustments,
1485        // equivalent) version
1486        //
1487        // ```
1488        // match foo {
1489        //     Some(x, ) => { ... },
1490        //     _ => { ... },
1491        // }
1492        // ```
1493        //
1494        // Then we see that to get the same result, we must start with
1495        // `deref { deref { place_foo }}` instead of `place_foo` since the pattern is now `Some(x,)`
1496        // and not `&&Some(x,)`, even though its assigned type is that of `&&Some(x,)`.
1497        let adjustments_len = self.cx.result.pat_adjustment(pat).map_or(0, |it| it.len());
1498        for _ in 0..adjustments_len {
1499            debug!("applying adjustment to place_with_id={:?}", place_with_id);
1500            // FIXME: We need to adjust this once we implement deref patterns (or pin ergonomics, for that matter).
1501            place_with_id = self.cat_deref(pat.into(), place_with_id)?;
1502        }
1503        let place_with_id = place_with_id; // lose mutability
1504        debug!("applied adjustment derefs to get place_with_id={:?}", place_with_id);
1505
1506        // Invoke the callback, but only now, after the `place_with_id` has adjusted.
1507        //
1508        // To see that this makes sense, consider `match &Some(3) { Some(x) => { ... }}`. In that
1509        // case, the initial `place_with_id` will be that for `&Some(3)` and the pattern is `Some(x)`. We
1510        // don't want to call `op` with these incompatible values. As written, what happens instead
1511        // is that `op` is called with the adjusted place (that for `*&Some(3)`) and the pattern
1512        // `Some(x)` (which matches). Recursing once more, `*&Some(3)` and the pattern `Some(x)`
1513        // result in the place `Downcast<Some>(*&Some(3)).0` associated to `x` and invoke `op` with
1514        // that (where the `ref` on `x` is implied).
1515        op(self, place_with_id.clone(), pat)?;
1516
1517        match self.cx.store[pat] {
1518            Pat::Tuple { args: ref subpats, ellipsis: dots_pos } => {
1519                // (p1, ..., pN)
1520                let total_fields = self.total_fields_in_tuple(pat);
1521
1522                for (i, &subpat) in subpats.iter().enumerate_and_adjust(total_fields, dots_pos) {
1523                    let subpat_ty = self.pat_ty_adjusted(subpat)?;
1524                    let projection_kind =
1525                        ProjectionKind::Field { field_idx: i as u32, variant_idx: 0 };
1526                    let sub_place = self.cat_projection(
1527                        pat.into(),
1528                        place_with_id.clone(),
1529                        subpat_ty,
1530                        projection_kind,
1531                    );
1532                    self.cat_pattern(sub_place, subpat, op)?;
1533                }
1534            }
1535
1536            Pat::TupleStruct { args: ref subpats, ellipsis: dots_pos, .. } => {
1537                // S(p1, ..., pN)
1538                let (variant_index, variant) = self.variant_index_for_adt(pat)?;
1539                let total_fields = variant.fields(self.cx.db).len();
1540
1541                for (i, &subpat) in subpats.iter().enumerate_and_adjust(total_fields, dots_pos) {
1542                    let subpat_ty = self.pat_ty_adjusted(subpat)?;
1543                    let projection_kind =
1544                        ProjectionKind::Field { variant_idx: variant_index, field_idx: i as u32 };
1545                    let sub_place = self.cat_projection(
1546                        pat.into(),
1547                        place_with_id.clone(),
1548                        subpat_ty,
1549                        projection_kind,
1550                    );
1551                    self.cat_pattern(sub_place, subpat, op)?;
1552                }
1553            }
1554
1555            Pat::Record { args: ref field_pats, .. } => {
1556                // S { f1: p1, ..., fN: pN }
1557
1558                let (variant_index, variant) = self.variant_index_for_adt(pat)?;
1559                let fields = variant.fields(self.cx.db);
1560
1561                for fp in field_pats {
1562                    let field_ty = self.pat_ty_adjusted(fp.pat)?;
1563                    let field_index = fields.field(&fp.name).ok_or(ErrorGuaranteed)?;
1564
1565                    let field_place = self.cat_projection(
1566                        pat.into(),
1567                        place_with_id.clone(),
1568                        field_ty,
1569                        ProjectionKind::Field {
1570                            variant_idx: variant_index,
1571                            field_idx: field_index.into_raw().into_u32(),
1572                        },
1573                    );
1574                    self.cat_pattern(field_place, fp.pat, op)?;
1575                }
1576            }
1577
1578            Pat::Or(ref pats) => {
1579                for &pat in pats {
1580                    self.cat_pattern(place_with_id.clone(), pat, op)?;
1581                }
1582            }
1583
1584            Pat::Bind { subpat: Some(subpat), .. } => {
1585                self.cat_pattern(place_with_id, subpat, op)?;
1586            }
1587
1588            Pat::Box { inner: subpat } | Pat::Ref { pat: subpat, .. } => {
1589                // box p1, &p1, &mut p1. we can ignore the mutability of
1590                // PatKind::Ref since that information is already contained
1591                // in the type.
1592                let subplace = self.cat_deref(pat.into(), place_with_id)?;
1593                self.cat_pattern(subplace, subpat, op)?;
1594            }
1595
1596            Pat::Slice { prefix: ref before, slice, suffix: ref after } => {
1597                let Some(element_ty) = self
1598                    .cx
1599                    .structurally_resolve_type(pat.into(), place_with_id.place.ty())
1600                    .builtin_index()
1601                else {
1602                    debug!("explicit index of non-indexable type {:?}", place_with_id);
1603                    return Err(ErrorGuaranteed);
1604                };
1605                let elt_place = self.cat_projection(
1606                    pat.into(),
1607                    place_with_id.clone(),
1608                    element_ty,
1609                    ProjectionKind::Index,
1610                );
1611                for &before_pat in before {
1612                    self.cat_pattern(elt_place.clone(), before_pat, op)?;
1613                }
1614                if let Some(slice_pat) = slice {
1615                    let slice_pat_ty = self.pat_ty_adjusted(slice_pat)?;
1616                    let slice_place = self.cat_projection(
1617                        pat.into(),
1618                        place_with_id,
1619                        slice_pat_ty,
1620                        ProjectionKind::Subslice,
1621                    );
1622                    self.cat_pattern(slice_place, slice_pat, op)?;
1623                }
1624                for &after_pat in after {
1625                    self.cat_pattern(elt_place.clone(), after_pat, op)?;
1626                }
1627            }
1628
1629            Pat::Bind { subpat: None, .. }
1630            | Pat::Expr(..)
1631            | Pat::Path(_)
1632            | Pat::Lit(..)
1633            | Pat::ConstBlock(..)
1634            | Pat::Range { .. }
1635            | Pat::Missing
1636            | Pat::Rest
1637            | Pat::Wild => {
1638                // always ok
1639            }
1640        }
1641
1642        Ok(())
1643    }

1645    /// Checks whether a type has multiple variants, and therefore, whether a
1646    /// read of the discriminant might be necessary. Note that the actual MIR
1647    /// builder code does a more specific check, filtering out variants that
1648    /// happen to be uninhabited.
1649    ///
1650    /// Here, it is not practical to perform such a check, because inhabitedness
1651    /// queries require typeck results, and typeck requires closure capture analysis.
1652    ///
1653    /// Moreover, the language is moving towards uninhabited variants still semantically
1654    /// causing a discriminant read, so we *shouldn't* perform any such check.
1655    ///
1656    /// FIXME(never_patterns): update this comment once the aforementioned MIR builder
1657    /// code is changed to be insensitive to inhhabitedness.
1658    #[instrument(skip(self), level = "debug")]
1659    fn is_multivariant_adt(&mut self, node: ExprOrPatId, ty: Ty<'db>) -> bool {
1660        if let TyKind::Adt(def, _) = self.cx.structurally_resolve_type(node, ty).kind() {
1661            // Note that if a non-exhaustive SingleVariant is defined in another crate, we need
1662            // to assume that more cases will be added to the variant in the future. This mean
1663            // that we should handle non-exhaustive SingleVariant the same way we would handle
1664            // a MultiVariant.
1665            match def.def_id() {
1666                AdtId::StructId(_) | AdtId::UnionId(_) => false,
1667                AdtId::EnumId(did) => {
1668                    let has_foreign_non_exhaustive = || {
1669                        AttrFlags::query(self.cx.db, did.into()).contains(AttrFlags::NON_EXHAUSTIVE)
1670                            && did.krate(self.cx.db) != self.cx.krate()
1671                    };
1672                    did.enum_variants(self.cx.db).variants.len() > 1 || has_foreign_non_exhaustive()
1673                }
1674            }
1675        } else {
1676            false
1677        }
1678    }
