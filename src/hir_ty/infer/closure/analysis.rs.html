37#[derive(Debug, Clone, PartialEq, Eq, Hash)]
38pub(crate) struct HirPlace<'db> {
39    pub(crate) local: BindingId,
40    pub(crate) projections: Vec<ProjectionElem<Infallible, Ty<'db>>>,
41}

43impl<'db> HirPlace<'db> {
44    fn ty(&self, ctx: &mut InferenceContext<'_, 'db>) -> Ty<'db> {
45        let mut ty = ctx.table.resolve_completely(ctx.result[self.local]);
46        for p in &self.projections {
47            ty = p.projected_ty(
48                &ctx.table.infer_ctxt,
49                ty,
50                |_, _, _| {
51                    unreachable!("Closure field only happens in MIR");
52                },
53                ctx.owner.module(ctx.db).krate(),
54            );
55        }
56        ty
57    }
58
59    fn capture_kind_of_truncated_place(
60        &self,
61        mut current_capture: CaptureKind,
62        len: usize,
63    ) -> CaptureKind {
64        if let CaptureKind::ByRef(BorrowKind::Mut {
65            kind: MutBorrowKind::Default | MutBorrowKind::TwoPhasedBorrow,
66        }) = current_capture
67            && self.projections[len..].contains(&ProjectionElem::Deref)
68        {
69            current_capture =
70                CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::ClosureCapture });
71        }
72        current_capture
73    }
74}

82#[derive(Debug, Clone, PartialEq, Eq)]
83pub struct CapturedItem<'db> {
84    pub(crate) place: HirPlace<'db>,
85    pub(crate) kind: CaptureKind,
86    /// The inner vec is the stacks; the outer vec is for each capture reference.
87    ///
88    /// Even though we always report only the last span (i.e. the most inclusive span),
89    /// we need to keep them all, since when a closure occurs inside a closure, we
90    /// copy all captures of the inner closure to the outer closure, and then we may
91    /// truncate them, and we want the correct span to be reported.
92    span_stacks: SmallVec<[SmallVec<[MirSpan; 3]>; 3]>,
93    pub(crate) ty: EarlyBinder<'db, Ty<'db>>,
94}
95
96impl<'db> CapturedItem<'db> {
97    pub fn local(&self) -> BindingId {
98        self.place.local
99    }
100
101    /// Returns whether this place has any field (aka. non-deref) projections.
102    pub fn has_field_projections(&self) -> bool {
103        self.place.projections.iter().any(|it| !matches!(it, ProjectionElem::Deref))
104    }
105
106    pub fn ty(&self, db: &'db dyn HirDatabase, subst: GenericArgs<'db>) -> Ty<'db> {
107        let interner = DbInterner::new_with(db, None, None);
108        self.ty.instantiate(interner, subst.split_closure_args_untupled().parent_args)
109    }
110
111    pub fn kind(&self) -> CaptureKind {
112        self.kind
113    }
114
115    pub fn spans(&self) -> SmallVec<[MirSpan; 3]> {
116        self.span_stacks.iter().map(|stack| *stack.last().expect("empty span stack")).collect()
117    }

119    /// Converts the place to a name that can be inserted into source code.
120    pub fn place_to_name(&self, owner: DefWithBodyId, db: &dyn HirDatabase) -> String {
121        let body = db.body(owner);
122        let mut result = body[self.place.local].name.as_str().to_owned();
123        for proj in &self.place.projections {
124            match proj {
125                ProjectionElem::Deref => {}
126                ProjectionElem::Field(Either::Left(f)) => {
127                    let variant_data = f.parent.fields(db);
128                    match variant_data.shape {
129                        FieldsShape::Record => {
130                            result.push('_');
131                            result.push_str(variant_data.fields()[f.local_id].name.as_str())
132                        }
133                        FieldsShape::Tuple => {
134                            let index =
135                                variant_data.fields().iter().position(|it| it.0 == f.local_id);
136                            if let Some(index) = index {
137                                format_to!(result, "_{index}");
138                            }
139                        }
140                        FieldsShape::Unit => {}
141                    }
142                }
143                ProjectionElem::Field(Either::Right(f)) => format_to!(result, "_{}", f.index),
144                &ProjectionElem::ClosureField(field) => format_to!(result, "_{field}"),
145                ProjectionElem::Index(_)
146                | ProjectionElem::ConstantIndex { .. }
147                | ProjectionElem::Subslice { .. }
148                | ProjectionElem::OpaqueCast(_) => {
149                    never!("Not happen in closure capture");
150                    continue;
151                }
152            }
153        }
154        if is_raw_identifier(&result, owner.module(db).krate().data(db).edition) {
155            result.insert_str(0, "r#");
156        }
157        result
158    }

160    pub fn display_place_source_code(&self, owner: DefWithBodyId, db: &dyn HirDatabase) -> String {
161        let body = db.body(owner);
162        let krate = owner.krate(db);
163        let edition = krate.data(db).edition;
164        let mut result = body[self.place.local].name.display(db, edition).to_string();
165        for proj in &self.place.projections {
166            match proj {
167                // In source code autoderef kicks in.
168                ProjectionElem::Deref => {}
169                ProjectionElem::Field(Either::Left(f)) => {
170                    let variant_data = f.parent.fields(db);
171                    match variant_data.shape {
172                        FieldsShape::Record => format_to!(
173                            result,
174                            ".{}",
175                            variant_data.fields()[f.local_id].name.display(db, edition)
176                        ),
177                        FieldsShape::Tuple => format_to!(
178                            result,
179                            ".{}",
180                            variant_data
181                                .fields()
182                                .iter()
183                                .position(|it| it.0 == f.local_id)
184                                .unwrap_or_default()
185                        ),
186                        FieldsShape::Unit => {}
187                    }
188                }
189                ProjectionElem::Field(Either::Right(f)) => {
190                    let field = f.index;
191                    format_to!(result, ".{field}");
192                }
193                &ProjectionElem::ClosureField(field) => {
194                    format_to!(result, ".{field}");
195                }
196                ProjectionElem::Index(_)
197                | ProjectionElem::ConstantIndex { .. }
198                | ProjectionElem::Subslice { .. }
199                | ProjectionElem::OpaqueCast(_) => {
200                    never!("Not happen in closure capture");
201                    continue;
202                }
203            }
204        }
205        let final_derefs_count = self
206            .place
207            .projections
208            .iter()
209            .rev()
210            .take_while(|proj| matches!(proj, ProjectionElem::Deref))
211            .count();
212        result.insert_str(0, &"*".repeat(final_derefs_count));
213        result
214    }
215
216    pub fn display_place(&self, owner: DefWithBodyId, db: &dyn HirDatabase) -> String {
217        let body = db.body(owner);
218        let krate = owner.krate(db);
219        let edition = krate.data(db).edition;
220        let mut result = body[self.place.local].name.display(db, edition).to_string();
221        let mut field_need_paren = false;
222        for proj in &self.place.projections {
223            match proj {
224                ProjectionElem::Deref => {
225                    result = format!("*{result}");
226                    field_need_paren = true;
227                }
228                ProjectionElem::Field(Either::Left(f)) => {
229                    if field_need_paren {
230                        result = format!("({result})");
231                    }
232                    let variant_data = f.parent.fields(db);
233                    let field = match variant_data.shape {
234                        FieldsShape::Record => {
235                            variant_data.fields()[f.local_id].name.as_str().to_owned()
236                        }
237                        FieldsShape::Tuple => variant_data
238                            .fields()
239                            .iter()
240                            .position(|it| it.0 == f.local_id)
241                            .unwrap_or_default()
242                            .to_string(),
243                        FieldsShape::Unit => "[missing field]".to_owned(),
244                    };
245                    result = format!("{result}.{field}");
246                    field_need_paren = false;
247                }
248                ProjectionElem::Field(Either::Right(f)) => {
249                    let field = f.index;
250                    if field_need_paren {
251                        result = format!("({result})");
252                    }
253                    result = format!("{result}.{field}");
254                    field_need_paren = false;
255                }
256                &ProjectionElem::ClosureField(field) => {
257                    if field_need_paren {
258                        result = format!("({result})");
259                    }
260                    result = format!("{result}.{field}");
261                    field_need_paren = false;
262                }
263                ProjectionElem::Index(_)
264                | ProjectionElem::ConstantIndex { .. }
265                | ProjectionElem::Subslice { .. }
266                | ProjectionElem::OpaqueCast(_) => {
267                    never!("Not happen in closure capture");
268                    continue;
269                }
270            }
271        }
272        result
273    }
274}

277pub(crate) struct CapturedItemWithoutTy<'db> {
278    pub(crate) place: HirPlace<'db>,
279    pub(crate) kind: CaptureKind,
280    /// The inner vec is the stacks; the outer vec is for each capture reference.
281    pub(crate) span_stacks: SmallVec<[SmallVec<[MirSpan; 3]>; 3]>,
282}
283
284impl<'db> CapturedItemWithoutTy<'db> {
285    fn with_ty(self, ctx: &mut InferenceContext<'_, 'db>) -> CapturedItem<'db> {
286        let ty = self.place.ty(ctx);
287        let ty = match &self.kind {
288            CaptureKind::ByValue => ty,
289            CaptureKind::ByRef(bk) => {
290                let m = match bk {
291                    BorrowKind::Mut { .. } => Mutability::Mut,
292                    _ => Mutability::Not,
293                };
294                Ty::new_ref(ctx.interner(), ctx.types.re_error, ty, m)
295            }
296        };
297        CapturedItem {
298            place: self.place,
299            kind: self.kind,
300            span_stacks: self.span_stacks,
301            ty: EarlyBinder::bind(ty),
302        }
303    }
304}
305
306impl<'db> InferenceContext<'_, 'db> {
307    fn place_of_expr(&mut self, tgt_expr: ExprId) -> Option<HirPlace<'db>> {
308        let r = self.place_of_expr_without_adjust(tgt_expr)?;
309        let adjustments =
310            self.result.expr_adjustments.get(&tgt_expr).map(|it| &**it).unwrap_or_default();
311        apply_adjusts_to_place(&mut self.current_capture_span_stack, r, adjustments)
312    }
313
314    /// Pushes the span into `current_capture_span_stack`, *without clearing it first*.
315    fn path_place(&mut self, path: &Path, id: ExprOrPatId) -> Option<HirPlace<'db>> {
316        if path.type_anchor().is_some() {
317            return None;
318        }
319        let hygiene = self.body.expr_or_pat_path_hygiene(id);
320        self.resolver.resolve_path_in_value_ns_fully(self.db, path, hygiene).and_then(|result| {
321            match result {
322                ValueNs::LocalBinding(binding) => {
323                    let mir_span = match id {
324                        ExprOrPatId::ExprId(id) => MirSpan::ExprId(id),
325                        ExprOrPatId::PatId(id) => MirSpan::PatId(id),
326                    };
327                    self.current_capture_span_stack.push(mir_span);
328                    Some(HirPlace { local: binding, projections: Vec::new() })
329                }
330                _ => None,
331            }
332        })
333    }
334
335    /// Changes `current_capture_span_stack` to contain the stack of spans for this expr.
336    fn place_of_expr_without_adjust(&mut self, tgt_expr: ExprId) -> Option<HirPlace<'db>> {
337        self.current_capture_span_stack.clear();
338        match &self.body[tgt_expr] {
339            Expr::Path(p) => {
340                let resolver_guard =
341                    self.resolver.update_to_inner_scope(self.db, self.owner, tgt_expr);
342                let result = self.path_place(p, tgt_expr.into());
343                self.resolver.reset_to_guard(resolver_guard);
344                return result;
345            }
346            Expr::Field { expr, name: _ } => {
347                let mut place = self.place_of_expr(*expr)?;
348                let field = self.result.field_resolution(tgt_expr)?;
349                self.current_capture_span_stack.push(MirSpan::ExprId(tgt_expr));
350                place.projections.push(ProjectionElem::Field(field));
351                return Some(place);
352            }
353            Expr::UnaryOp { expr, op: UnaryOp::Deref } => {
354                if matches!(
355                    self.expr_ty_after_adjustments(*expr).kind(),
356                    TyKind::Ref(..) | TyKind::RawPtr(..)
357                ) {
358                    let mut place = self.place_of_expr(*expr)?;
359                    self.current_capture_span_stack.push(MirSpan::ExprId(tgt_expr));
360                    place.projections.push(ProjectionElem::Deref);
361                    return Some(place);
362                }
363            }
364            _ => (),
365        }
366        None
367    }
368
369    fn push_capture(&mut self, place: HirPlace<'db>, kind: CaptureKind) {
370        self.current_captures.push(CapturedItemWithoutTy {
371            place,
372            kind,
373            span_stacks: smallvec![self.current_capture_span_stack.iter().copied().collect()],
374        });
375    }
376
377    fn truncate_capture_spans(
378        &self,
379        capture: &mut CapturedItemWithoutTy<'db>,
380        mut truncate_to: usize,
381    ) {
382        // The first span is the identifier, and it must always remain.
383        truncate_to += 1;
384        for span_stack in &mut capture.span_stacks {
385            let mut remained = truncate_to;
386            let mut actual_truncate_to = 0;
387            for &span in &*span_stack {
388                actual_truncate_to += 1;
389                if !span.is_ref_span(self.body) {
390                    remained -= 1;
391                    if remained == 0 {
392                        break;
393                    }
394                }
395            }
396            if actual_truncate_to < span_stack.len()
397                && span_stack[actual_truncate_to].is_ref_span(self.body)
398            {
399                // Include the ref operator if there is one, we will fix it later (in `strip_captures_ref_span()`) if it's incorrect.
400                actual_truncate_to += 1;
401            }
402            span_stack.truncate(actual_truncate_to);
403        }
404    }

406    fn ref_expr(&mut self, expr: ExprId, place: Option<HirPlace<'db>>) {
407        if let Some(place) = place {
408            self.add_capture(place, CaptureKind::ByRef(BorrowKind::Shared));
409        }
410        self.walk_expr(expr);
411    }
412
413    fn add_capture(&mut self, place: HirPlace<'db>, kind: CaptureKind) {
414        if self.is_upvar(&place) {
415            self.push_capture(place, kind);
416        }
417    }
418
419    fn mutate_path_pat(&mut self, path: &Path, id: PatId) {
420        if let Some(place) = self.path_place(path, id.into()) {
421            self.add_capture(
422                place,
423                CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::Default }),
424            );
425            self.current_capture_span_stack.pop(); // Remove the pattern span.
426        }
427    }
428
429    fn mutate_expr(&mut self, expr: ExprId, place: Option<HirPlace<'db>>) {
430        if let Some(place) = place {
431            self.add_capture(
432                place,
433                CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::Default }),
434            );
435        }
436        self.walk_expr(expr);
437    }
438
439    fn consume_expr(&mut self, expr: ExprId) {
440        if let Some(place) = self.place_of_expr(expr) {
441            self.consume_place(place);
442        }
443        self.walk_expr(expr);
444    }
445
446    fn consume_place(&mut self, place: HirPlace<'db>) {
447        if self.is_upvar(&place) {
448            let ty = place.ty(self);
449            let kind = if self.is_ty_copy(ty) {
450                CaptureKind::ByRef(BorrowKind::Shared)
451            } else {
452                CaptureKind::ByValue
453            };
454            self.push_capture(place, kind);
455        }
456    }
457
458    fn walk_expr_with_adjust(&mut self, tgt_expr: ExprId, adjustment: &[Adjustment<'db>]) {
459        if let Some((last, rest)) = adjustment.split_last() {
460            match &last.kind {
461                Adjust::NeverToAny | Adjust::Deref(None) | Adjust::Pointer(_) => {
462                    self.walk_expr_with_adjust(tgt_expr, rest)
463                }
464                Adjust::Deref(Some(m)) => match m.0 {
465                    Some(m) => {
466                        self.ref_capture_with_adjusts(m, tgt_expr, rest);
467                    }
468                    None => unreachable!(),
469                },
470                Adjust::Borrow(b) => {
471                    self.ref_capture_with_adjusts(b.mutability(), tgt_expr, rest);
472                }
473            }
474        } else {
475            self.walk_expr_without_adjust(tgt_expr);
476        }
477    }
478
479    fn ref_capture_with_adjusts(
480        &mut self,
481        m: Mutability,
482        tgt_expr: ExprId,
483        rest: &[Adjustment<'db>],
484    ) {
485        let capture_kind = match m {
486            Mutability::Mut => CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::Default }),
487            Mutability::Not => CaptureKind::ByRef(BorrowKind::Shared),
488        };
489        if let Some(place) = self.place_of_expr_without_adjust(tgt_expr)
490            && let Some(place) =
491                apply_adjusts_to_place(&mut self.current_capture_span_stack, place, rest)
492        {
493            self.add_capture(place, capture_kind);
494        }
495        self.walk_expr_with_adjust(tgt_expr, rest);
496    }
497
498    fn walk_expr(&mut self, tgt_expr: ExprId) {
499        if let Some(it) = self.result.expr_adjustments.get_mut(&tgt_expr) {
500            // FIXME: this take is completely unneeded, and just is here to make borrow checker
501            // happy. Remove it if you can.
502            let x_taken = mem::take(it);
503            self.walk_expr_with_adjust(tgt_expr, &x_taken);
504            *self.result.expr_adjustments.get_mut(&tgt_expr).unwrap() = x_taken;
505        } else {
506            self.walk_expr_without_adjust(tgt_expr);
507        }
508    }
509
510    fn walk_expr_without_adjust(&mut self, tgt_expr: ExprId) {
511        match &self.body[tgt_expr] {
512            Expr::OffsetOf(_) => (),
513            Expr::InlineAsm(e) => e.operands.iter().for_each(|(_, op)| match op {
514                AsmOperand::In { expr, .. }
515                | AsmOperand::Out { expr: Some(expr), .. }
516                | AsmOperand::InOut { expr, .. } => self.walk_expr_without_adjust(*expr),
517                AsmOperand::SplitInOut { in_expr, out_expr, .. } => {
518                    self.walk_expr_without_adjust(*in_expr);
519                    if let Some(out_expr) = out_expr {
520                        self.walk_expr_without_adjust(*out_expr);
521                    }
522                }
523                AsmOperand::Out { expr: None, .. }
524                | AsmOperand::Const(_)
525                | AsmOperand::Label(_)
526                | AsmOperand::Sym(_) => (),
527            }),
528            Expr::If { condition, then_branch, else_branch } => {
529                self.consume_expr(*condition);
530                self.consume_expr(*then_branch);
531                if let &Some(expr) = else_branch {
532                    self.consume_expr(expr);
533                }
534            }
535            Expr::Async { statements, tail, .. }
536            | Expr::Unsafe { statements, tail, .. }
537            | Expr::Block { statements, tail, .. } => {
538                for s in statements.iter() {
539                    match s {
540                        Statement::Let { pat, type_ref: _, initializer, else_branch } => {
541                            if let Some(else_branch) = else_branch {
542                                self.consume_expr(*else_branch);
543                            }
544                            if let Some(initializer) = initializer {
545                                if else_branch.is_some() {
546                                    self.consume_expr(*initializer);
547                                } else {
548                                    self.walk_expr(*initializer);
549                                }
550                                if let Some(place) = self.place_of_expr(*initializer) {
551                                    self.consume_with_pat(place, *pat);
552                                }
553                            }
554                        }
555                        Statement::Expr { expr, has_semi: _ } => {
556                            self.consume_expr(*expr);
557                        }
558                        Statement::Item(_) => (),
559                    }
560                }
561                if let Some(tail) = tail {
562                    self.consume_expr(*tail);
563                }
564            }
565            Expr::Call { callee, args } => {
566                self.consume_expr(*callee);
567                self.consume_exprs(args.iter().copied());
568            }
569            Expr::MethodCall { receiver, args, .. } => {
570                self.consume_expr(*receiver);
571                self.consume_exprs(args.iter().copied());
572            }
573            Expr::Match { expr, arms } => {
574                for arm in arms.iter() {
575                    self.consume_expr(arm.expr);
576                    if let Some(guard) = arm.guard {
577                        self.consume_expr(guard);
578                    }
579                }
580                self.walk_expr(*expr);
581                if let Some(discr_place) = self.place_of_expr(*expr)
582                    && self.is_upvar(&discr_place)
583                {
584                    let mut capture_mode = None;
585                    for arm in arms.iter() {
586                        self.walk_pat(&mut capture_mode, arm.pat);
587                    }
588                    if let Some(c) = capture_mode {
589                        self.push_capture(discr_place, c);
590                    }
591                }
592            }
593            Expr::Break { expr, label: _ }
594            | Expr::Return { expr }
595            | Expr::Yield { expr }
596            | Expr::Yeet { expr } => {
597                if let &Some(expr) = expr {
598                    self.consume_expr(expr);
599                }
600            }
601            &Expr::Become { expr } => {
602                self.consume_expr(expr);
603            }
604            Expr::RecordLit { fields, spread, .. } => {
605                if let &Some(expr) = spread {
606                    self.consume_expr(expr);
607                }
608                self.consume_exprs(fields.iter().map(|it| it.expr));
609            }
610            Expr::Field { expr, name: _ } => self.select_from_expr(*expr),
611            Expr::UnaryOp { expr, op: UnaryOp::Deref } => {
612                if matches!(
613                    self.expr_ty_after_adjustments(*expr).kind(),
614                    TyKind::Ref(..) | TyKind::RawPtr(..)
615                ) {
616                    self.select_from_expr(*expr);
617                } else if let Some((f, _)) = self.result.method_resolution(tgt_expr) {
618                    let mutability = 'b: {
619                        if let Some(deref_trait) =
620                            self.resolve_lang_item(LangItem::DerefMut).and_then(|it| it.as_trait())
621                            && let Some(deref_fn) = deref_trait
622                                .trait_items(self.db)
623                                .method_by_name(&Name::new_symbol_root(sym::deref_mut))
624                        {
625                            break 'b deref_fn == f;
626                        }
627                        false
628                    };
629                    let place = self.place_of_expr(*expr);
630                    if mutability {
631                        self.mutate_expr(*expr, place);
632                    } else {
633                        self.ref_expr(*expr, place);
634                    }
635                } else {
636                    self.select_from_expr(*expr);
637                }
638            }
639            Expr::Let { pat, expr } => {
640                self.walk_expr(*expr);
641                if let Some(place) = self.place_of_expr(*expr) {
642                    self.consume_with_pat(place, *pat);
643                }
644            }
645            Expr::UnaryOp { expr, op: _ }
646            | Expr::Array(Array::Repeat { initializer: expr, repeat: _ })
647            | Expr::Await { expr }
648            | Expr::Loop { body: expr, label: _ }
649            | Expr::Box { expr }
650            | Expr::Cast { expr, type_ref: _ } => {
651                self.consume_expr(*expr);
652            }
653            Expr::Ref { expr, rawness: _, mutability } => {
654                // We need to do this before we push the span so the order will be correct.
655                let place = self.place_of_expr(*expr);
656                self.current_capture_span_stack.push(MirSpan::ExprId(tgt_expr));
657                match mutability {
658                    hir_def::type_ref::Mutability::Shared => self.ref_expr(*expr, place),
659                    hir_def::type_ref::Mutability::Mut => self.mutate_expr(*expr, place),
660                }
661            }
662            Expr::BinaryOp { lhs, rhs, op } => {
663                let Some(op) = op else {
664                    return;
665                };
666                if matches!(op, BinaryOp::Assignment { .. }) {
667                    let place = self.place_of_expr(*lhs);
668                    self.mutate_expr(*lhs, place);
669                    self.consume_expr(*rhs);
670                    return;
671                }
672                self.consume_expr(*lhs);
673                self.consume_expr(*rhs);
674            }
675            Expr::Range { lhs, rhs, range_type: _ } => {
676                if let &Some(expr) = lhs {
677                    self.consume_expr(expr);
678                }
679                if let &Some(expr) = rhs {
680                    self.consume_expr(expr);
681                }
682            }
683            Expr::Index { base, index } => {
684                self.select_from_expr(*base);
685                self.consume_expr(*index);
686            }
687            Expr::Closure { .. } => {
688                let ty = self.expr_ty(tgt_expr);
689                let TyKind::Closure(id, _) = ty.kind() else {
690                    never!("closure type is always closure");
691                    return;
692                };
693                let (captures, _) =
694                    self.result.closure_info.get(&id.0).expect(
695                        "We sort closures, so we should always have data for inner closures",
696                    );
697                let mut cc = mem::take(&mut self.current_captures);
698                cc.extend(captures.iter().filter(|it| self.is_upvar(&it.place)).map(|it| {
699                    CapturedItemWithoutTy {
700                        place: it.place.clone(),
701                        kind: it.kind,
702                        span_stacks: it.span_stacks.clone(),
703                    }
704                }));
705                self.current_captures = cc;
706            }
707            Expr::Array(Array::ElementList { elements: exprs }) | Expr::Tuple { exprs } => {
708                self.consume_exprs(exprs.iter().copied())
709            }
710            &Expr::Assignment { target, value } => {
711                self.walk_expr(value);
712                let resolver_guard =
713                    self.resolver.update_to_inner_scope(self.db, self.owner, tgt_expr);
714                match self.place_of_expr(value) {
715                    Some(rhs_place) => {
716                        self.inside_assignment = true;
717                        self.consume_with_pat(rhs_place, target);
718                        self.inside_assignment = false;
719                    }
720                    None => self.body.walk_pats(target, &mut |pat| match &self.body[pat] {
721                        Pat::Path(path) => self.mutate_path_pat(path, pat),
722                        &Pat::Expr(expr) => {
723                            let place = self.place_of_expr(expr);
724                            self.mutate_expr(expr, place);
725                        }
726                        _ => {}
727                    }),
728                }
729                self.resolver.reset_to_guard(resolver_guard);
730            }
731
732            Expr::Missing
733            | Expr::Continue { .. }
734            | Expr::Path(_)
735            | Expr::Literal(_)
736            | Expr::Const(_)
737            | Expr::Underscore => (),
738        }
739    }

741    fn walk_pat(&mut self, result: &mut Option<CaptureKind>, pat: PatId) {
742        let mut update_result = |ck: CaptureKind| match result {
743            Some(r) => {
744                *r = cmp::max(*r, ck);
745            }
746            None => *result = Some(ck),
747        };
748
749        self.walk_pat_inner(
750            pat,
751            &mut update_result,
752            BorrowKind::Mut { kind: MutBorrowKind::Default },
753        );
754    }
755
756    fn walk_pat_inner(
757        &mut self,
758        p: PatId,
759        update_result: &mut impl FnMut(CaptureKind),
760        mut for_mut: BorrowKind,
761    ) {
762        match &self.body[p] {
763            Pat::Ref { .. }
764            | Pat::Box { .. }
765            | Pat::Missing
766            | Pat::Wild
767            | Pat::Tuple { .. }
768            | Pat::Expr(_)
769            | Pat::Or(_) => (),
770            Pat::TupleStruct { .. } | Pat::Record { .. } => {
771                if let Some(variant) = self.result.variant_resolution_for_pat(p) {
772                    let adt = variant.adt_id(self.db);
773                    let is_multivariant = match adt {
774                        hir_def::AdtId::EnumId(e) => e.enum_variants(self.db).variants.len() != 1,
775                        _ => false,
776                    };
777                    if is_multivariant {
778                        update_result(CaptureKind::ByRef(BorrowKind::Shared));
779                    }
780                }
781            }
782            Pat::Slice { .. }
783            | Pat::ConstBlock(_)
784            | Pat::Path(_)
785            | Pat::Lit(_)
786            | Pat::Range { .. } => {
787                update_result(CaptureKind::ByRef(BorrowKind::Shared));
788            }
789            Pat::Bind { id, .. } => match self.result.binding_modes[p] {
790                crate::BindingMode::Move => {
791                    if self.is_ty_copy(self.result.type_of_binding[*id]) {
792                        update_result(CaptureKind::ByRef(BorrowKind::Shared));
793                    } else {
794                        update_result(CaptureKind::ByValue);
795                    }
796                }
797                crate::BindingMode::Ref(r) => match r {
798                    Mutability::Mut => update_result(CaptureKind::ByRef(for_mut)),
799                    Mutability::Not => update_result(CaptureKind::ByRef(BorrowKind::Shared)),
800                },
801            },
802        }
803        if self.result.pat_adjustments.get(&p).is_some_and(|it| !it.is_empty()) {
804            for_mut = BorrowKind::Mut { kind: MutBorrowKind::ClosureCapture };
805        }
806        self.body.walk_pats_shallow(p, |p| self.walk_pat_inner(p, update_result, for_mut));
807    }

851    fn restrict_precision_for_unsafe(&mut self) {
852        // FIXME: Borrow checker problems without this.
853        let mut current_captures = std::mem::take(&mut self.current_captures);
854        for capture in &mut current_captures {
855            let mut ty = self.table.resolve_completely(self.result[capture.place.local]);
856            if ty.is_raw_ptr() || ty.is_union() {
857                capture.kind = CaptureKind::ByRef(BorrowKind::Shared);
858                self.truncate_capture_spans(capture, 0);
859                capture.place.projections.truncate(0);
860                continue;
861            }
862            for (i, p) in capture.place.projections.iter().enumerate() {
863                ty = p.projected_ty(
864                    &self.table.infer_ctxt,
865                    ty,
866                    |_, _, _| {
867                        unreachable!("Closure field only happens in MIR");
868                    },
869                    self.owner.module(self.db).krate(),
870                );
871                if ty.is_raw_ptr() || ty.is_union() {
872                    capture.kind = CaptureKind::ByRef(BorrowKind::Shared);
873                    self.truncate_capture_spans(capture, i + 1);
874                    capture.place.projections.truncate(i + 1);
875                    break;
876                }
877            }
878        }
879        self.current_captures = current_captures;
880    }

882    fn adjust_for_move_closure(&mut self) {
883        // FIXME: Borrow checker won't allow without this.
884        let mut current_captures = std::mem::take(&mut self.current_captures);
885        for capture in &mut current_captures {
886            if let Some(first_deref) =
887                capture.place.projections.iter().position(|proj| *proj == ProjectionElem::Deref)
888            {
889                self.truncate_capture_spans(capture, first_deref);
890                capture.place.projections.truncate(first_deref);
891            }
892            capture.kind = CaptureKind::ByValue;
893        }
894        self.current_captures = current_captures;
895    }

897    fn minimize_captures(&mut self) {
898        self.current_captures.sort_unstable_by_key(|it| it.place.projections.len());
899        let mut hash_map = FxHashMap::<HirPlace<'db>, usize>::default();
900        let result = mem::take(&mut self.current_captures);
901        for mut item in result {
902            let mut lookup_place = HirPlace { local: item.place.local, projections: vec![] };
903            let mut it = item.place.projections.iter();
904            let prev_index = loop {
905                if let Some(k) = hash_map.get(&lookup_place) {
906                    break Some(*k);
907                }
908                match it.next() {
909                    Some(it) => {
910                        lookup_place.projections.push(it.clone());
911                    }
912                    None => break None,
913                }
914            };
915            match prev_index {
916                Some(p) => {
917                    let prev_projections_len = self.current_captures[p].place.projections.len();
918                    self.truncate_capture_spans(&mut item, prev_projections_len);
919                    self.current_captures[p].span_stacks.extend(item.span_stacks);
920                    let len = self.current_captures[p].place.projections.len();
921                    let kind_after_truncate =
922                        item.place.capture_kind_of_truncated_place(item.kind, len);
923                    self.current_captures[p].kind =
924                        cmp::max(kind_after_truncate, self.current_captures[p].kind);
925                }
926                None => {
927                    hash_map.insert(item.place.clone(), self.current_captures.len());
928                    self.current_captures.push(item);
929                }
930            }
931        }
932    }

934    fn consume_with_pat(&mut self, mut place: HirPlace<'db>, tgt_pat: PatId) {
935        let adjustments_count =
936            self.result.pat_adjustments.get(&tgt_pat).map(|it| it.len()).unwrap_or_default();
937        place.projections.extend((0..adjustments_count).map(|_| ProjectionElem::Deref));
938        self.current_capture_span_stack
939            .extend((0..adjustments_count).map(|_| MirSpan::PatId(tgt_pat)));
940        'reset_span_stack: {
941            match &self.body[tgt_pat] {
942                Pat::Missing | Pat::Wild => (),
943                Pat::Tuple { args, ellipsis } => {
944                    let (al, ar) = args.split_at(ellipsis.map_or(args.len(), |it| it as usize));
945                    let field_count = match self.result[tgt_pat].kind() {
946                        TyKind::Tuple(s) => s.len(),
947                        _ => break 'reset_span_stack,
948                    };
949                    let fields = 0..field_count;
950                    let it = al.iter().zip(fields.clone()).chain(ar.iter().rev().zip(fields.rev()));
951                    for (&arg, i) in it {
952                        let mut p = place.clone();
953                        self.current_capture_span_stack.push(MirSpan::PatId(arg));
954                        p.projections.push(ProjectionElem::Field(Either::Right(TupleFieldId {
955                            tuple: TupleId(!0), // dummy this, as its unused anyways
956                            index: i as u32,
957                        })));
958                        self.consume_with_pat(p, arg);
959                        self.current_capture_span_stack.pop();
960                    }
961                }
962                Pat::Or(pats) => {
963                    for pat in pats.iter() {
964                        self.consume_with_pat(place.clone(), *pat);
965                    }
966                }
967                Pat::Record { args, .. } => {
968                    let Some(variant) = self.result.variant_resolution_for_pat(tgt_pat) else {
969                        break 'reset_span_stack;
970                    };
971                    match variant {
972                        VariantId::EnumVariantId(_) | VariantId::UnionId(_) => {
973                            self.consume_place(place)
974                        }
975                        VariantId::StructId(s) => {
976                            let vd = s.fields(self.db);
977                            for field_pat in args.iter() {
978                                let arg = field_pat.pat;
979                                let Some(local_id) = vd.field(&field_pat.name) else {
980                                    continue;
981                                };
982                                let mut p = place.clone();
983                                self.current_capture_span_stack.push(MirSpan::PatId(arg));
984                                p.projections.push(ProjectionElem::Field(Either::Left(FieldId {
985                                    parent: variant,
986                                    local_id,
987                                })));
988                                self.consume_with_pat(p, arg);
989                                self.current_capture_span_stack.pop();
990                            }
991                        }
992                    }
993                }
994                Pat::Range { .. } | Pat::Slice { .. } | Pat::ConstBlock(_) | Pat::Lit(_) => {
995                    self.consume_place(place)
996                }
997                Pat::Path(path) => {
998                    if self.inside_assignment {
999                        self.mutate_path_pat(path, tgt_pat);
1000                    }
1001                    self.consume_place(place);
1002                }
1003                &Pat::Bind { id, subpat: _ } => {
1004                    let mode = self.result.binding_modes[tgt_pat];
1005                    let capture_kind = match mode {
1006                        BindingMode::Move => {
1007                            self.consume_place(place);
1008                            break 'reset_span_stack;
1009                        }
1010                        BindingMode::Ref(Mutability::Not) => BorrowKind::Shared,
1011                        BindingMode::Ref(Mutability::Mut) => {
1012                            BorrowKind::Mut { kind: MutBorrowKind::Default }
1013                        }
1014                    };
1015                    self.current_capture_span_stack.push(MirSpan::BindingId(id));
1016                    self.add_capture(place, CaptureKind::ByRef(capture_kind));
1017                    self.current_capture_span_stack.pop();
1018                }
1019                Pat::TupleStruct { path: _, args, ellipsis } => {
1020                    let Some(variant) = self.result.variant_resolution_for_pat(tgt_pat) else {
1021                        break 'reset_span_stack;
1022                    };
1023                    match variant {
1024                        VariantId::EnumVariantId(_) | VariantId::UnionId(_) => {
1025                            self.consume_place(place)
1026                        }
1027                        VariantId::StructId(s) => {
1028                            let vd = s.fields(self.db);
1029                            let (al, ar) =
1030                                args.split_at(ellipsis.map_or(args.len(), |it| it as usize));
1031                            let fields = vd.fields().iter();
1032                            let it = al
1033                                .iter()
1034                                .zip(fields.clone())
1035                                .chain(ar.iter().rev().zip(fields.rev()));
1036                            for (&arg, (i, _)) in it {
1037                                let mut p = place.clone();
1038                                self.current_capture_span_stack.push(MirSpan::PatId(arg));
1039                                p.projections.push(ProjectionElem::Field(Either::Left(FieldId {
1040                                    parent: variant,
1041                                    local_id: i,
1042                                })));
1043                                self.consume_with_pat(p, arg);
1044                                self.current_capture_span_stack.pop();
1045                            }
1046                        }
1047                    }
1048                }
1049                Pat::Ref { pat, mutability: _ } => {
1050                    self.current_capture_span_stack.push(MirSpan::PatId(tgt_pat));
1051                    place.projections.push(ProjectionElem::Deref);
1052                    self.consume_with_pat(place, *pat);
1053                    self.current_capture_span_stack.pop();
1054                }
1055                Pat::Box { .. } => (), // not supported
1056                &Pat::Expr(expr) => {
1057                    self.consume_place(place);
1058                    let pat_capture_span_stack = mem::take(&mut self.current_capture_span_stack);
1059                    let old_inside_assignment = mem::replace(&mut self.inside_assignment, false);
1060                    let lhs_place = self.place_of_expr(expr);
1061                    self.mutate_expr(expr, lhs_place);
1062                    self.inside_assignment = old_inside_assignment;
1063                    self.current_capture_span_stack = pat_capture_span_stack;
1064                }
1065            }
1066        }
1067        self.current_capture_span_stack
1068            .truncate(self.current_capture_span_stack.len() - adjustments_count);
1069    }

1077    fn closure_kind(&self) -> FnTrait {
1078        let mut r = FnTrait::Fn;
1079        for it in &self.current_captures {
1080            r = cmp::min(
1081                r,
1082                match &it.kind {
1083                    CaptureKind::ByRef(BorrowKind::Mut { .. }) => FnTrait::FnMut,
1084                    CaptureKind::ByRef(BorrowKind::Shallow | BorrowKind::Shared) => FnTrait::Fn,
1085                    CaptureKind::ByValue => FnTrait::FnOnce,
1086                },
1087            )
1088        }
1089        r
1090    }

1092    fn analyze_closure(&mut self, closure: InternedClosureId) -> FnTrait {
1093        let InternedClosure(_, root) = self.db.lookup_intern_closure(closure);
1094        self.current_closure = Some(closure);
1095        let Expr::Closure { body, capture_by, .. } = &self.body[root] else {
1096            unreachable!("Closure expression id is always closure");
1097        };
1098        self.consume_expr(*body);
1099        for item in &self.current_captures {
1100            if matches!(
1101                item.kind,
1102                CaptureKind::ByRef(BorrowKind::Mut {
1103                    kind: MutBorrowKind::Default | MutBorrowKind::TwoPhasedBorrow
1104                })
1105            ) && !item.place.projections.contains(&ProjectionElem::Deref)
1106            {
1107                // FIXME: remove the `mutated_bindings_in_closure` completely and add proper fake reads in
1108                // MIR. I didn't do that due duplicate diagnostics.
1109                self.result.mutated_bindings_in_closure.insert(item.place.local);
1110            }
1111        }
1112        self.restrict_precision_for_unsafe();
1113        // `closure_kind` should be done before adjust_for_move_closure
1114        // If there exists pre-deduced kind of a closure, use it instead of one determined by capture, as rustc does.
1115        // rustc also does diagnostics here if the latter is not a subtype of the former.
1116        let closure_kind = self
1117            .result
1118            .closure_info
1119            .get(&closure)
1120            .map_or_else(|| self.closure_kind(), |info| info.1);
1121        match capture_by {
1122            CaptureBy::Value => self.adjust_for_move_closure(),
1123            CaptureBy::Ref => (),
1124        }
1125        self.minimize_captures();
1126        self.strip_captures_ref_span();
1127        let result = mem::take(&mut self.current_captures);
1128        let captures = result.into_iter().map(|it| it.with_ty(self)).collect::<Vec<_>>();
1129        self.result.closure_info.insert(closure, (captures, closure_kind));
1130        closure_kind
1131    }

1133    fn strip_captures_ref_span(&mut self) {
1134        // FIXME: Borrow checker won't allow without this.
1135        let mut captures = std::mem::take(&mut self.current_captures);
1136        for capture in &mut captures {
1137            if matches!(capture.kind, CaptureKind::ByValue) {
1138                for span_stack in &mut capture.span_stacks {
1139                    if span_stack[span_stack.len() - 1].is_ref_span(self.body) {
1140                        span_stack.truncate(span_stack.len() - 1);
1141                    }
1142                }
1143            }
1144        }
1145        self.current_captures = captures;
1146    }

1148    pub(crate) fn infer_closures(&mut self) {
1149        let deferred_closures = self.sort_closures();
1150        for (closure, exprs) in deferred_closures.into_iter().rev() {
1151            self.current_captures = vec![];
1152            let kind = self.analyze_closure(closure);
1153
1154            for (derefed_callee, callee_ty, params, expr) in exprs {
1155                if let &Expr::Call { callee, .. } = &self.body[expr] {
1156                    let mut adjustments =
1157                        self.result.expr_adjustments.remove(&callee).unwrap_or_default().into_vec();
1158                    self.write_fn_trait_method_resolution(
1159                        kind,
1160                        derefed_callee,
1161                        &mut adjustments,
1162                        callee_ty,
1163                        &params,
1164                        expr,
1165                    );
1166                    self.result.expr_adjustments.insert(callee, adjustments.into_boxed_slice());
1167                }
1168            }
1169        }
1170    }

1172    /// We want to analyze some closures before others, to have a correct analysis:
1173    /// * We should analyze nested closures before the parent, since the parent should capture some of
1174    ///   the things that its children captures.
1175    /// * If a closure calls another closure, we need to analyze the callee, to find out how we should
1176    ///   capture it (e.g. by move for FnOnce)
1177    ///
1178    /// These dependencies are collected in the main inference. We do a topological sort in this function. It
1179    /// will consume the `deferred_closures` field and return its content in a sorted vector.
1180    fn sort_closures(
1181        &mut self,
1182    ) -> Vec<(InternedClosureId, Vec<(Ty<'db>, Ty<'db>, Vec<Ty<'db>>, ExprId)>)> {
1183        let mut deferred_closures = mem::take(&mut self.deferred_closures);
1184        let mut dependents_count: FxHashMap<InternedClosureId, usize> =
1185            deferred_closures.keys().map(|it| (*it, 0)).collect();
1186        for deps in self.closure_dependencies.values() {
1187            for dep in deps {
1188                *dependents_count.entry(*dep).or_default() += 1;
1189            }
1190        }
1191        let mut queue: Vec<_> =
1192            deferred_closures.keys().copied().filter(|&it| dependents_count[&it] == 0).collect();
1193        let mut result = vec![];
1194        while let Some(it) = queue.pop() {
1195            if let Some(d) = deferred_closures.remove(&it) {
1196                result.push((it, d));
1197            }
1198            for &dep in self.closure_dependencies.get(&it).into_iter().flat_map(|it| it.iter()) {
1199                let cnt = dependents_count.get_mut(&dep).unwrap();
1200                *cnt -= 1;
1201                if *cnt == 0 {
1202                    queue.push(dep);
1203                }
1204            }
1205        }
1206        assert!(deferred_closures.is_empty(), "we should have analyzed all closures");
1207        result
1208    }

1210    pub(crate) fn add_current_closure_dependency(&mut self, dep: InternedClosureId) {
1211        if let Some(c) = self.current_closure
1212            && !dep_creates_cycle(&self.closure_dependencies, &mut FxHashSet::default(), c, dep)
1213        {
1214            self.closure_dependencies.entry(c).or_default().push(dep);
1215        }
1216
1217        fn dep_creates_cycle(
1218            closure_dependencies: &FxHashMap<InternedClosureId, Vec<InternedClosureId>>,
1219            visited: &mut FxHashSet<InternedClosureId>,
1220            from: InternedClosureId,
1221            to: InternedClosureId,
1222        ) -> bool {
1223            if !visited.insert(from) {
1224                return false;
1225            }
1226
1227            if from == to {
1228                return true;
1229            }
1230
1231            if let Some(deps) = closure_dependencies.get(&to) {
1232                for dep in deps {
1233                    if dep_creates_cycle(closure_dependencies, visited, from, *dep) {
1234                        return true;
1235                    }
1236                }
1237            }
1238
1239            false
1240        }
1241    }
1242}

1244/// Call this only when the last span in the stack isn't a split.
1245fn apply_adjusts_to_place<'db>(
1246    current_capture_span_stack: &mut Vec<MirSpan>,
1247    mut r: HirPlace<'db>,
1248    adjustments: &[Adjustment<'db>],
1249) -> Option<HirPlace<'db>> {
1250    let span = *current_capture_span_stack.last().expect("empty capture span stack");
1251    for adj in adjustments {
1252        match &adj.kind {
1253            Adjust::Deref(None) => {
1254                current_capture_span_stack.push(span);
1255                r.projections.push(ProjectionElem::Deref);
1256            }
1257            _ => return None,
1258        }
1259    }
1260    Some(r)
