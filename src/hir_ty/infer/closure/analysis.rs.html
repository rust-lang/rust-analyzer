40#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
41pub(crate) enum HirPlaceProjection {
42    Deref,
43    Field(FieldId),
44    TupleField(u32),
45}
46
47impl HirPlaceProjection {
48    fn projected_ty<'db>(
49        self,
50        infcx: &InferCtxt<'db>,
51        env: ParamEnv<'db>,
52        mut base: Ty<'db>,
53        krate: Crate,
54    ) -> Ty<'db> {
55        let interner = infcx.interner;
56        let db = interner.db;
57        if base.is_ty_error() {
58            return Ty::new_error(interner, ErrorGuaranteed);
59        }
60
61        if matches!(base.kind(), TyKind::Alias(..)) {
62            let mut ocx = ObligationCtxt::new(infcx);
63            match ocx.structurally_normalize_ty(&ObligationCause::dummy(), env, base) {
64                Ok(it) => base = it,
65                Err(_) => return Ty::new_error(interner, ErrorGuaranteed),
66            }
67        }
68        match self {
69            HirPlaceProjection::Deref => match base.kind() {
70                TyKind::RawPtr(inner, _) | TyKind::Ref(_, inner, _) => inner,
71                TyKind::Adt(adt_def, subst) if adt_def.is_box() => subst.type_at(0),
72                _ => {
73                    never!(
74                        "Overloaded deref on type {} is not a projection",
75                        base.display(db, DisplayTarget::from_crate(db, krate))
76                    );
77                    Ty::new_error(interner, ErrorGuaranteed)
78                }
79            },
80            HirPlaceProjection::Field(f) => match base.kind() {
81                TyKind::Adt(_, subst) => {
82                    db.field_types(f.parent)[f.local_id].get().instantiate(interner, subst)
83                }
84                ty => {
85                    never!("Only adt has field, found {:?}", ty);
86                    Ty::new_error(interner, ErrorGuaranteed)
87                }
88            },
89            HirPlaceProjection::TupleField(idx) => match base.kind() {
90                TyKind::Tuple(subst) => {
91                    subst.as_slice().get(idx as usize).copied().unwrap_or_else(|| {
92                        never!("Out of bound tuple field");
93                        Ty::new_error(interner, ErrorGuaranteed)
94                    })
95                }
96                ty => {
97                    never!("Only tuple has tuple field: {:?}", ty);
98                    Ty::new_error(interner, ErrorGuaranteed)
99                }
100            },
101        }
102    }
103}

111impl HirPlace {
112    fn ty<'db>(&self, ctx: &mut InferenceContext<'_, 'db>) -> Ty<'db> {
113        let krate = ctx.krate();
114        let mut ty = ctx.table.resolve_completely(ctx.result.binding_ty(self.local));
115        for p in &self.projections {
116            ty = p.projected_ty(ctx.infcx(), ctx.table.param_env, ty, krate);
117        }
118        ty
119    }
120
121    fn capture_kind_of_truncated_place(
122        &self,
123        mut current_capture: CaptureKind,
124        len: usize,
125    ) -> CaptureKind {
126        if let CaptureKind::ByRef(BorrowKind::Mut {
127            kind: MutBorrowKind::Default | MutBorrowKind::TwoPhasedBorrow,
128        }) = current_capture
129            && self.projections[len..].contains(&HirPlaceProjection::Deref)
130        {
131            current_capture =
132                CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::ClosureCapture });
133        }
134        current_capture
135    }
136}

144#[derive(Debug, Clone, PartialEq, Eq, salsa::Update)]
145pub struct CapturedItem {
146    pub(crate) place: HirPlace,
147    pub(crate) kind: CaptureKind,
148    /// The inner vec is the stacks; the outer vec is for each capture reference.
149    ///
150    /// Even though we always report only the last span (i.e. the most inclusive span),
151    /// we need to keep them all, since when a closure occurs inside a closure, we
152    /// copy all captures of the inner closure to the outer closure, and then we may
153    /// truncate them, and we want the correct span to be reported.
154    span_stacks: SmallVec<[SmallVec<[MirSpan; 3]>; 3]>,
155    pub(crate) ty: StoredEarlyBinder<StoredTy>,
156}
157
158impl CapturedItem {
159    pub fn local(&self) -> BindingId {
160        self.place.local
161    }
162
163    /// Returns whether this place has any field (aka. non-deref) projections.
164    pub fn has_field_projections(&self) -> bool {
165        self.place.projections.iter().any(|it| !matches!(it, HirPlaceProjection::Deref))
166    }

168    pub fn ty<'db>(&self, db: &'db dyn HirDatabase, subst: GenericArgs<'db>) -> Ty<'db> {
169        let interner = DbInterner::new_no_crate(db);
170        self.ty.get().instantiate(interner, subst.as_closure().parent_args())
171    }
172
173    pub fn kind(&self) -> CaptureKind {
174        self.kind
175    }
176
177    pub fn spans(&self) -> SmallVec<[MirSpan; 3]> {
178        self.span_stacks.iter().map(|stack| *stack.last().expect("empty span stack")).collect()
179    }

181    /// Converts the place to a name that can be inserted into source code.
182    pub fn place_to_name(&self, owner: DefWithBodyId, db: &dyn HirDatabase) -> String {
183        let body = db.body(owner);
184        let mut result = body[self.place.local].name.as_str().to_owned();
185        for proj in &self.place.projections {
186            match proj {
187                HirPlaceProjection::Deref => {}
188                HirPlaceProjection::Field(f) => {
189                    let variant_data = f.parent.fields(db);
190                    match variant_data.shape {
191                        FieldsShape::Record => {
192                            result.push('_');
193                            result.push_str(variant_data.fields()[f.local_id].name.as_str())
194                        }
195                        FieldsShape::Tuple => {
196                            let index =
197                                variant_data.fields().iter().position(|it| it.0 == f.local_id);
198                            if let Some(index) = index {
199                                format_to!(result, "_{index}");
200                            }
201                        }
202                        FieldsShape::Unit => {}
203                    }
204                }
205                HirPlaceProjection::TupleField(idx) => {
206                    format_to!(result, "_{idx}")
207                }
208            }
209        }
210        if is_raw_identifier(&result, owner.module(db).krate(db).data(db).edition) {
211            result.insert_str(0, "r#");
212        }
213        result
214    }

216    pub fn display_place_source_code(&self, owner: DefWithBodyId, db: &dyn HirDatabase) -> String {
217        let body = db.body(owner);
218        let krate = owner.krate(db);
219        let edition = krate.data(db).edition;
220        let mut result = body[self.place.local].name.display(db, edition).to_string();
221        for proj in &self.place.projections {
222            match proj {
223                // In source code autoderef kicks in.
224                HirPlaceProjection::Deref => {}
225                HirPlaceProjection::Field(f) => {
226                    let variant_data = f.parent.fields(db);
227                    match variant_data.shape {
228                        FieldsShape::Record => format_to!(
229                            result,
230                            ".{}",
231                            variant_data.fields()[f.local_id].name.display(db, edition)
232                        ),
233                        FieldsShape::Tuple => format_to!(
234                            result,
235                            ".{}",
236                            variant_data
237                                .fields()
238                                .iter()
239                                .position(|it| it.0 == f.local_id)
240                                .unwrap_or_default()
241                        ),
242                        FieldsShape::Unit => {}
243                    }
244                }
245                HirPlaceProjection::TupleField(idx) => {
246                    format_to!(result, ".{idx}")
247                }
248            }
249        }
250        let final_derefs_count = self
251            .place
252            .projections
253            .iter()
254            .rev()
255            .take_while(|proj| matches!(proj, HirPlaceProjection::Deref))
256            .count();
257        result.insert_str(0, &"*".repeat(final_derefs_count));
258        result
259    }
260
261    pub fn display_place(&self, owner: DefWithBodyId, db: &dyn HirDatabase) -> String {
262        let body = db.body(owner);
263        let krate = owner.krate(db);
264        let edition = krate.data(db).edition;
265        let mut result = body[self.place.local].name.display(db, edition).to_string();
266        let mut field_need_paren = false;
267        for proj in &self.place.projections {
268            match proj {
269                HirPlaceProjection::Deref => {
270                    result = format!("*{result}");
271                    field_need_paren = true;
272                }
273                HirPlaceProjection::Field(f) => {
274                    if field_need_paren {
275                        result = format!("({result})");
276                    }
277                    let variant_data = f.parent.fields(db);
278                    let field = match variant_data.shape {
279                        FieldsShape::Record => {
280                            variant_data.fields()[f.local_id].name.as_str().to_owned()
281                        }
282                        FieldsShape::Tuple => variant_data
283                            .fields()
284                            .iter()
285                            .position(|it| it.0 == f.local_id)
286                            .unwrap_or_default()
287                            .to_string(),
288                        FieldsShape::Unit => "[missing field]".to_owned(),
289                    };
290                    result = format!("{result}.{field}");
291                    field_need_paren = false;
292                }
293                HirPlaceProjection::TupleField(idx) => {
294                    if field_need_paren {
295                        result = format!("({result})");
296                    }
297                    result = format!("{result}.{idx}");
298                    field_need_paren = false;
299                }
300            }
301        }
302        result
303    }
304}

307pub(crate) struct CapturedItemWithoutTy {
308    pub(crate) place: HirPlace,
309    pub(crate) kind: CaptureKind,
310    /// The inner vec is the stacks; the outer vec is for each capture reference.
311    pub(crate) span_stacks: SmallVec<[SmallVec<[MirSpan; 3]>; 3]>,
312}
313
314impl CapturedItemWithoutTy {
315    fn with_ty(self, ctx: &mut InferenceContext<'_, '_>) -> CapturedItem {
316        let ty = self.place.ty(ctx);
317        let ty = match &self.kind {
318            CaptureKind::ByValue => ty,
319            CaptureKind::ByRef(bk) => {
320                let m = match bk {
321                    BorrowKind::Mut { .. } => Mutability::Mut,
322                    _ => Mutability::Not,
323                };
324                Ty::new_ref(ctx.interner(), ctx.types.regions.error, ty, m)
325            }
326        };
327        CapturedItem {
328            place: self.place,
329            kind: self.kind,
330            span_stacks: self.span_stacks,
331            ty: StoredEarlyBinder::bind(ty.store()),
332        }
333    }
334}

336impl<'db> InferenceContext<'_, 'db> {
337    fn place_of_expr(&mut self, tgt_expr: ExprId) -> Option<HirPlace> {
338        let r = self.place_of_expr_without_adjust(tgt_expr)?;
339        let adjustments =
340            self.result.expr_adjustments.get(&tgt_expr).map(|it| &**it).unwrap_or_default();
341        apply_adjusts_to_place(&mut self.current_capture_span_stack, r, adjustments)
342    }
343
344    /// Pushes the span into `current_capture_span_stack`, *without clearing it first*.
345    fn path_place(&mut self, path: &Path, id: ExprOrPatId) -> Option<HirPlace> {
346        if path.type_anchor().is_some() {
347            return None;
348        }
349        let hygiene = self.body.expr_or_pat_path_hygiene(id);
350        self.resolver.resolve_path_in_value_ns_fully(self.db, path, hygiene).and_then(|result| {
351            match result {
352                ValueNs::LocalBinding(binding) => {
353                    let mir_span = match id {
354                        ExprOrPatId::ExprId(id) => MirSpan::ExprId(id),
355                        ExprOrPatId::PatId(id) => MirSpan::PatId(id),
356                    };
357                    self.current_capture_span_stack.push(mir_span);
358                    Some(HirPlace { local: binding, projections: Vec::new() })
359                }
360                _ => None,
361            }
362        })
363    }
364
365    /// Changes `current_capture_span_stack` to contain the stack of spans for this expr.
366    fn place_of_expr_without_adjust(&mut self, tgt_expr: ExprId) -> Option<HirPlace> {
367        self.current_capture_span_stack.clear();
368        match &self.body[tgt_expr] {
369            Expr::Path(p) => {
370                let resolver_guard =
371                    self.resolver.update_to_inner_scope(self.db, self.owner, tgt_expr);
372                let result = self.path_place(p, tgt_expr.into());
373                self.resolver.reset_to_guard(resolver_guard);
374                return result;
375            }
376            Expr::Field { expr, name: _ } => {
377                let mut place = self.place_of_expr(*expr)?;
378                let field = self.result.field_resolution(tgt_expr)?;
379                self.current_capture_span_stack.push(MirSpan::ExprId(tgt_expr));
380                place.projections.push(field.either(HirPlaceProjection::Field, |f| {
381                    HirPlaceProjection::TupleField(f.index)
382                }));
383                return Some(place);
384            }
385            Expr::UnaryOp { expr, op: UnaryOp::Deref } => {
386                let is_builtin_deref = match self.expr_ty(*expr).kind() {
387                    TyKind::Ref(..) | TyKind::RawPtr(..) => true,
388                    TyKind::Adt(adt_def, _) if adt_def.is_box() => true,
389                    _ => false,
390                };
391                if is_builtin_deref {
392                    let mut place = self.place_of_expr(*expr)?;
393                    self.current_capture_span_stack.push(MirSpan::ExprId(tgt_expr));
394                    place.projections.push(HirPlaceProjection::Deref);
395                    return Some(place);
396                }
397            }
398            _ => (),
399        }
400        None
401    }
402
403    fn push_capture(&mut self, place: HirPlace, kind: CaptureKind) {
404        self.current_captures.push(CapturedItemWithoutTy {
405            place,
406            kind,
407            span_stacks: smallvec![self.current_capture_span_stack.iter().copied().collect()],
408        });
409    }
410
411    fn truncate_capture_spans(&self, capture: &mut CapturedItemWithoutTy, mut truncate_to: usize) {
412        // The first span is the identifier, and it must always remain.
413        truncate_to += 1;
414        for span_stack in &mut capture.span_stacks {
415            let mut remained = truncate_to;
416            let mut actual_truncate_to = 0;
417            for &span in &*span_stack {
418                actual_truncate_to += 1;
419                if !span.is_ref_span(self.body) {
420                    remained -= 1;
421                    if remained == 0 {
422                        break;
423                    }
424                }
425            }
426            if actual_truncate_to < span_stack.len()
427                && span_stack[actual_truncate_to].is_ref_span(self.body)
428            {
429                // Include the ref operator if there is one, we will fix it later (in `strip_captures_ref_span()`) if it's incorrect.
430                actual_truncate_to += 1;
431            }
432            span_stack.truncate(actual_truncate_to);
433        }
434    }
435
436    fn ref_expr(&mut self, expr: ExprId, place: Option<HirPlace>) {
437        if let Some(place) = place {
438            self.add_capture(place, CaptureKind::ByRef(BorrowKind::Shared));
439        }
440        self.walk_expr(expr);
441    }
442
443    fn add_capture(&mut self, place: HirPlace, kind: CaptureKind) {
444        if self.is_upvar(&place) {
445            self.push_capture(place, kind);
446        }
447    }
448
449    fn mutate_path_pat(&mut self, path: &Path, id: PatId) {
450        if let Some(place) = self.path_place(path, id.into()) {
451            self.add_capture(
452                place,
453                CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::Default }),
454            );
455            self.current_capture_span_stack.pop(); // Remove the pattern span.
456        }
457    }
458
459    fn mutate_expr(&mut self, expr: ExprId, place: Option<HirPlace>) {
460        if let Some(place) = place {
461            self.add_capture(
462                place,
463                CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::Default }),
464            );
465        }
466        self.walk_expr(expr);
467    }
468
469    fn consume_expr(&mut self, expr: ExprId) {
470        if let Some(place) = self.place_of_expr(expr) {
471            self.consume_place(place);
472        }
473        self.walk_expr(expr);
474    }
475
476    fn consume_place(&mut self, place: HirPlace) {
477        if self.is_upvar(&place) {
478            let ty = place.ty(self);
479            let kind = if self.is_ty_copy(ty) {
480                CaptureKind::ByRef(BorrowKind::Shared)
481            } else {
482                CaptureKind::ByValue
483            };
484            self.push_capture(place, kind);
485        }
486    }
487
488    fn walk_expr_with_adjust(&mut self, tgt_expr: ExprId, adjustment: &[Adjustment]) {
489        if let Some((last, rest)) = adjustment.split_last() {
490            match &last.kind {
491                Adjust::NeverToAny | Adjust::Deref(None) | Adjust::Pointer(_) => {
492                    self.walk_expr_with_adjust(tgt_expr, rest)
493                }
494                Adjust::Deref(Some(m)) => match m.0 {
495                    Some(m) => {
496                        self.ref_capture_with_adjusts(m, tgt_expr, rest);
497                    }
498                    None => unreachable!(),
499                },
500                Adjust::Borrow(b) => {
501                    self.ref_capture_with_adjusts(b.mutability(), tgt_expr, rest);
502                }
503            }
504        } else {
505            self.walk_expr_without_adjust(tgt_expr);
506        }
507    }
508
509    fn ref_capture_with_adjusts(&mut self, m: Mutability, tgt_expr: ExprId, rest: &[Adjustment]) {
510        let capture_kind = match m {
511            Mutability::Mut => CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::Default }),
512            Mutability::Not => CaptureKind::ByRef(BorrowKind::Shared),
513        };
514        if let Some(place) = self.place_of_expr_without_adjust(tgt_expr)
515            && let Some(place) =
516                apply_adjusts_to_place(&mut self.current_capture_span_stack, place, rest)
517        {
518            self.add_capture(place, capture_kind);
519        }
520        self.walk_expr_with_adjust(tgt_expr, rest);
521    }
522
523    fn walk_expr(&mut self, tgt_expr: ExprId) {
524        if let Some(it) = self.result.expr_adjustments.get_mut(&tgt_expr) {
525            // FIXME: this take is completely unneeded, and just is here to make borrow checker
526            // happy. Remove it if you can.
527            let x_taken = mem::take(it);
528            self.walk_expr_with_adjust(tgt_expr, &x_taken);
529            *self.result.expr_adjustments.get_mut(&tgt_expr).unwrap() = x_taken;
530        } else {
531            self.walk_expr_without_adjust(tgt_expr);
532        }
533    }
534
535    fn walk_expr_without_adjust(&mut self, tgt_expr: ExprId) {
536        match &self.body[tgt_expr] {
537            Expr::OffsetOf(_) => (),
538            Expr::InlineAsm(e) => e.operands.iter().for_each(|(_, op)| match op {
539                AsmOperand::In { expr, .. }
540                | AsmOperand::Out { expr: Some(expr), .. }
541                | AsmOperand::InOut { expr, .. } => self.walk_expr_without_adjust(*expr),
542                AsmOperand::SplitInOut { in_expr, out_expr, .. } => {
543                    self.walk_expr_without_adjust(*in_expr);
544                    if let Some(out_expr) = out_expr {
545                        self.walk_expr_without_adjust(*out_expr);
546                    }
547                }
548                AsmOperand::Out { expr: None, .. }
549                | AsmOperand::Const(_)
550                | AsmOperand::Label(_)
551                | AsmOperand::Sym(_) => (),
552            }),
553            Expr::If { condition, then_branch, else_branch } => {
554                self.consume_expr(*condition);
555                self.consume_expr(*then_branch);
556                if let &Some(expr) = else_branch {
557                    self.consume_expr(expr);
558                }
559            }
560            Expr::Async { statements, tail, .. }
561            | Expr::Unsafe { statements, tail, .. }
562            | Expr::Block { statements, tail, .. } => {
563                for s in statements.iter() {
564                    match s {
565                        Statement::Let { pat, type_ref: _, initializer, else_branch } => {
566                            if let Some(else_branch) = else_branch {
567                                self.consume_expr(*else_branch);
568                            }
569                            if let Some(initializer) = initializer {
570                                if else_branch.is_some() {
571                                    self.consume_expr(*initializer);
572                                } else {
573                                    self.walk_expr(*initializer);
574                                }
575                                if let Some(place) = self.place_of_expr(*initializer) {
576                                    self.consume_with_pat(place, *pat);
577                                }
578                            }
579                        }
580                        Statement::Expr { expr, has_semi: _ } => {
581                            self.consume_expr(*expr);
582                        }
583                        Statement::Item(_) => (),
584                    }
585                }
586                if let Some(tail) = tail {
587                    self.consume_expr(*tail);
588                }
589            }
590            Expr::Call { callee, args } => {
591                self.consume_expr(*callee);
592                self.consume_exprs(args.iter().copied());
593            }
594            Expr::MethodCall { receiver, args, .. } => {
595                self.consume_expr(*receiver);
596                self.consume_exprs(args.iter().copied());
597            }
598            Expr::Match { expr, arms } => {
599                for arm in arms.iter() {
600                    self.consume_expr(arm.expr);
601                    if let Some(guard) = arm.guard {
602                        self.consume_expr(guard);
603                    }
604                }
605                self.walk_expr(*expr);
606                if let Some(discr_place) = self.place_of_expr(*expr)
607                    && self.is_upvar(&discr_place)
608                {
609                    let mut capture_mode = None;
610                    for arm in arms.iter() {
611                        self.walk_pat(&mut capture_mode, arm.pat);
612                    }
613                    if let Some(c) = capture_mode {
614                        self.push_capture(discr_place, c);
615                    }
616                }
617            }
618            Expr::Break { expr, label: _ }
619            | Expr::Return { expr }
620            | Expr::Yield { expr }
621            | Expr::Yeet { expr } => {
622                if let &Some(expr) = expr {
623                    self.consume_expr(expr);
624                }
625            }
626            &Expr::Become { expr } => {
627                self.consume_expr(expr);
628            }
629            Expr::RecordLit { fields, spread, .. } => {
630                if let RecordSpread::Expr(expr) = *spread {
631                    self.consume_expr(expr);
632                }
633                self.consume_exprs(fields.iter().map(|it| it.expr));
634            }
635            Expr::Field { expr, name: _ } => self.select_from_expr(*expr),
636            Expr::UnaryOp { expr, op: UnaryOp::Deref } => {
637                if self.result.method_resolution(tgt_expr).is_some() {
638                    // Overloaded deref.
639                    match self.expr_ty_after_adjustments(*expr).kind() {
640                        TyKind::Ref(_, _, mutability) => {
641                            let place = self.place_of_expr(*expr);
642                            match mutability {
643                                Mutability::Mut => self.mutate_expr(*expr, place),
644                                Mutability::Not => self.ref_expr(*expr, place),
645                            }
646                        }
647                        // FIXME: Is this correct wrt. raw pointer derefs?
648                        TyKind::RawPtr(..) => self.select_from_expr(*expr),
649                        _ => never!("deref adjustments should include taking a mutable reference"),
650                    }
651                } else {
652                    self.select_from_expr(*expr);
653                }
654            }
655            Expr::Let { pat, expr } => {
656                self.walk_expr(*expr);
657                if let Some(place) = self.place_of_expr(*expr) {
658                    self.consume_with_pat(place, *pat);
659                }
660            }
661            Expr::UnaryOp { expr, op: _ }
662            | Expr::Array(Array::Repeat { initializer: expr, repeat: _ })
663            | Expr::Await { expr }
664            | Expr::Loop { body: expr, label: _ }
665            | Expr::Box { expr }
666            | Expr::Cast { expr, type_ref: _ } => {
667                self.consume_expr(*expr);
668            }
669            Expr::Ref { expr, rawness: _, mutability } => {
670                // We need to do this before we push the span so the order will be correct.
671                let place = self.place_of_expr(*expr);
672                self.current_capture_span_stack.push(MirSpan::ExprId(tgt_expr));
673                match mutability {
674                    hir_def::type_ref::Mutability::Shared => self.ref_expr(*expr, place),
675                    hir_def::type_ref::Mutability::Mut => self.mutate_expr(*expr, place),
676                }
677            }
678            Expr::BinaryOp { lhs, rhs, op } => {
679                let Some(op) = op else {
680                    return;
681                };
682                if matches!(op, BinaryOp::Assignment { .. }) {
683                    let place = self.place_of_expr(*lhs);
684                    self.mutate_expr(*lhs, place);
685                    self.consume_expr(*rhs);
686                    return;
687                }
688                self.consume_expr(*lhs);
689                self.consume_expr(*rhs);
690            }
691            Expr::Range { lhs, rhs, range_type: _ } => {
692                if let &Some(expr) = lhs {
693                    self.consume_expr(expr);
694                }
695                if let &Some(expr) = rhs {
696                    self.consume_expr(expr);
697                }
698            }
699            Expr::Index { base, index } => {
700                self.select_from_expr(*base);
701                self.consume_expr(*index);
702            }
703            Expr::Closure { .. } => {
704                let ty = self.expr_ty(tgt_expr);
705                let TyKind::Closure(id, _) = ty.kind() else {
706                    never!("closure type is always closure");
707                    return;
708                };
709                let (captures, _) =
710                    self.result.closure_info.get(&id.0).expect(
711                        "We sort closures, so we should always have data for inner closures",
712                    );
713                let mut cc = mem::take(&mut self.current_captures);
714                cc.extend(captures.iter().filter(|it| self.is_upvar(&it.place)).map(|it| {
715                    CapturedItemWithoutTy {
716                        place: it.place.clone(),
717                        kind: it.kind,
718                        span_stacks: it.span_stacks.clone(),
719                    }
720                }));
721                self.current_captures = cc;
722            }
723            Expr::Array(Array::ElementList { elements: exprs }) | Expr::Tuple { exprs } => {
724                self.consume_exprs(exprs.iter().copied())
725            }
726            &Expr::Assignment { target, value } => {
727                self.walk_expr(value);
728                let resolver_guard =
729                    self.resolver.update_to_inner_scope(self.db, self.owner, tgt_expr);
730                match self.place_of_expr(value) {
731                    Some(rhs_place) => {
732                        self.inside_assignment = true;
733                        self.consume_with_pat(rhs_place, target);
734                        self.inside_assignment = false;
735                    }
736                    None => self.body.walk_pats(target, &mut |pat| match &self.body[pat] {
737                        Pat::Path(path) => self.mutate_path_pat(path, pat),
738                        &Pat::Expr(expr) => {
739                            let place = self.place_of_expr(expr);
740                            self.mutate_expr(expr, place);
741                        }
742                        _ => {}
743                    }),
744                }
745                self.resolver.reset_to_guard(resolver_guard);
746            }
747
748            Expr::Missing
749            | Expr::Continue { .. }
750            | Expr::Path(_)
751            | Expr::Literal(_)
752            | Expr::Const(_)
753            | Expr::Underscore => (),
754        }
755    }

757    fn walk_pat(&mut self, result: &mut Option<CaptureKind>, pat: PatId) {
758        let mut update_result = |ck: CaptureKind| match result {
759            Some(r) => {
760                *r = cmp::max(*r, ck);
761            }
762            None => *result = Some(ck),
763        };
764
765        self.walk_pat_inner(
766            pat,
767            &mut update_result,
768            BorrowKind::Mut { kind: MutBorrowKind::Default },
769        );
770    }
771
772    fn walk_pat_inner(
773        &mut self,
774        p: PatId,
775        update_result: &mut impl FnMut(CaptureKind),
776        mut for_mut: BorrowKind,
777    ) {
778        match &self.body[p] {
779            Pat::Ref { .. }
780            | Pat::Box { .. }
781            | Pat::Missing
782            | Pat::Wild
783            | Pat::Tuple { .. }
784            | Pat::Expr(_)
785            | Pat::Or(_) => (),
786            Pat::TupleStruct { .. } | Pat::Record { .. } => {
787                if let Some(variant) = self.result.variant_resolution_for_pat(p) {
788                    let adt = variant.adt_id(self.db);
789                    let is_multivariant = match adt {
790                        hir_def::AdtId::EnumId(e) => e.enum_variants(self.db).variants.len() != 1,
791                        _ => false,
792                    };
793                    if is_multivariant {
794                        update_result(CaptureKind::ByRef(BorrowKind::Shared));
795                    }
796                }
797            }
798            Pat::Slice { .. }
799            | Pat::ConstBlock(_)
800            | Pat::Path(_)
801            | Pat::Lit(_)
802            | Pat::Range { .. } => {
803                update_result(CaptureKind::ByRef(BorrowKind::Shared));
804            }
805            Pat::Bind { id, .. } => match self.result.binding_modes[p] {
806                crate::BindingMode::Move => {
807                    if self.is_ty_copy(self.result.binding_ty(*id)) {
808                        update_result(CaptureKind::ByRef(BorrowKind::Shared));
809                    } else {
810                        update_result(CaptureKind::ByValue);
811                    }
812                }
813                crate::BindingMode::Ref(r) => match r {
814                    Mutability::Mut => update_result(CaptureKind::ByRef(for_mut)),
815                    Mutability::Not => update_result(CaptureKind::ByRef(BorrowKind::Shared)),
816                },
817            },
818        }
819        if self.result.pat_adjustments.get(&p).is_some_and(|it| !it.is_empty()) {
820            for_mut = BorrowKind::Mut { kind: MutBorrowKind::ClosureCapture };
821        }
822        self.body.walk_pats_shallow(p, |p| self.walk_pat_inner(p, update_result, for_mut));
823    }

833    fn is_ty_copy(&mut self, ty: Ty<'db>) -> bool {
834        if let TyKind::Closure(id, _) = ty.kind() {
835            // FIXME: We handle closure as a special case, since chalk consider every closure as copy. We
836            // should probably let chalk know which closures are copy, but I don't know how doing it
837            // without creating query cycles.
838            return self
839                .result
840                .closure_info
841                .get(&id.0)
842                .map(|it| it.1 == FnTrait::Fn)
843                .unwrap_or(true);
844        }
845        let ty = self.table.resolve_completely(ty);
846        self.table.type_is_copy_modulo_regions(ty)
847    }

853    fn restrict_precision_for_unsafe(&mut self) {
854        // FIXME: Borrow checker problems without this.
855        let mut current_captures = std::mem::take(&mut self.current_captures);
856        for capture in &mut current_captures {
857            let mut ty = self.table.resolve_completely(self.result.binding_ty(capture.place.local));
858            if ty.is_raw_ptr() || ty.is_union() {
859                capture.kind = CaptureKind::ByRef(BorrowKind::Shared);
860                self.truncate_capture_spans(capture, 0);
861                capture.place.projections.truncate(0);
862                continue;
863            }
864            for (i, p) in capture.place.projections.iter().enumerate() {
865                ty = p.projected_ty(
866                    &self.table.infer_ctxt,
867                    self.table.param_env,
868                    ty,
869                    self.owner.module(self.db).krate(self.db),
870                );
871                if ty.is_raw_ptr() || ty.is_union() {
872                    capture.kind = CaptureKind::ByRef(BorrowKind::Shared);
873                    self.truncate_capture_spans(capture, i + 1);
874                    capture.place.projections.truncate(i + 1);
875                    break;
876                }
877            }
878        }
879        self.current_captures = current_captures;
880    }

882    fn adjust_for_move_closure(&mut self) {
883        // FIXME: Borrow checker won't allow without this.
884        let mut current_captures = std::mem::take(&mut self.current_captures);
885        for capture in &mut current_captures {
886            if let Some(first_deref) =
887                capture.place.projections.iter().position(|proj| *proj == HirPlaceProjection::Deref)
888            {
889                self.truncate_capture_spans(capture, first_deref);
890                capture.place.projections.truncate(first_deref);
891            }
892            capture.kind = CaptureKind::ByValue;
893        }
894        self.current_captures = current_captures;
895    }

897    fn minimize_captures(&mut self) {
898        self.current_captures.sort_unstable_by_key(|it| it.place.projections.len());
899        let mut hash_map = FxHashMap::<HirPlace, usize>::default();
900        let result = mem::take(&mut self.current_captures);
901        for mut item in result {
902            let mut lookup_place = HirPlace { local: item.place.local, projections: vec![] };
903            let mut it = item.place.projections.iter();
904            let prev_index = loop {
905                if let Some(k) = hash_map.get(&lookup_place) {
906                    break Some(*k);
907                }
908                match it.next() {
909                    Some(it) => {
910                        lookup_place.projections.push(*it);
911                    }
912                    None => break None,
913                }
914            };
915            match prev_index {
916                Some(p) => {
917                    let prev_projections_len = self.current_captures[p].place.projections.len();
918                    self.truncate_capture_spans(&mut item, prev_projections_len);
919                    self.current_captures[p].span_stacks.extend(item.span_stacks);
920                    let len = self.current_captures[p].place.projections.len();
921                    let kind_after_truncate =
922                        item.place.capture_kind_of_truncated_place(item.kind, len);
923                    self.current_captures[p].kind =
924                        cmp::max(kind_after_truncate, self.current_captures[p].kind);
925                }
926                None => {
927                    hash_map.insert(item.place.clone(), self.current_captures.len());
928                    self.current_captures.push(item);
929                }
930            }
931        }
932    }

934    fn consume_with_pat(&mut self, mut place: HirPlace, tgt_pat: PatId) {
935        let adjustments_count =
936            self.result.pat_adjustments.get(&tgt_pat).map(|it| it.len()).unwrap_or_default();
937        place.projections.extend((0..adjustments_count).map(|_| HirPlaceProjection::Deref));
938        self.current_capture_span_stack
939            .extend((0..adjustments_count).map(|_| MirSpan::PatId(tgt_pat)));
940        'reset_span_stack: {
941            match &self.body[tgt_pat] {
942                Pat::Missing | Pat::Wild => (),
943                Pat::Tuple { args, ellipsis } => {
944                    let (al, ar) = args.split_at(ellipsis.map_or(args.len(), |it| it as usize));
945                    let field_count = match self.result.pat_ty(tgt_pat).kind() {
946                        TyKind::Tuple(s) => s.len(),
947                        _ => break 'reset_span_stack,
948                    };
949                    let fields = 0..field_count;
950                    let it = al.iter().zip(fields.clone()).chain(ar.iter().rev().zip(fields.rev()));
951                    for (&arg, i) in it {
952                        let mut p = place.clone();
953                        self.current_capture_span_stack.push(MirSpan::PatId(arg));
954                        p.projections.push(HirPlaceProjection::TupleField(i as u32));
955                        self.consume_with_pat(p, arg);
956                        self.current_capture_span_stack.pop();
957                    }
958                }
959                Pat::Or(pats) => {
960                    for pat in pats.iter() {
961                        self.consume_with_pat(place.clone(), *pat);
962                    }
963                }
964                Pat::Record { args, .. } => {
965                    let Some(variant) = self.result.variant_resolution_for_pat(tgt_pat) else {
966                        break 'reset_span_stack;
967                    };
968                    match variant {
969                        VariantId::EnumVariantId(_) | VariantId::UnionId(_) => {
970                            self.consume_place(place)
971                        }
972                        VariantId::StructId(s) => {
973                            let vd = s.fields(self.db);
974                            for field_pat in args.iter() {
975                                let arg = field_pat.pat;
976                                let Some(local_id) = vd.field(&field_pat.name) else {
977                                    continue;
978                                };
979                                let mut p = place.clone();
980                                self.current_capture_span_stack.push(MirSpan::PatId(arg));
981                                p.projections.push(HirPlaceProjection::Field(FieldId {
982                                    parent: variant,
983                                    local_id,
984                                }));
985                                self.consume_with_pat(p, arg);
986                                self.current_capture_span_stack.pop();
987                            }
988                        }
989                    }
990                }
991                Pat::Range { .. } | Pat::Slice { .. } | Pat::ConstBlock(_) | Pat::Lit(_) => {
992                    self.consume_place(place)
993                }
994                Pat::Path(path) => {
995                    if self.inside_assignment {
996                        self.mutate_path_pat(path, tgt_pat);
997                    }
998                    self.consume_place(place);
999                }
1000                &Pat::Bind { id, subpat: _ } => {
1001                    let mode = self.result.binding_modes[tgt_pat];
1002                    let capture_kind = match mode {
1003                        BindingMode::Move => {
1004                            self.consume_place(place);
1005                            break 'reset_span_stack;
1006                        }
1007                        BindingMode::Ref(Mutability::Not) => BorrowKind::Shared,
1008                        BindingMode::Ref(Mutability::Mut) => {
1009                            BorrowKind::Mut { kind: MutBorrowKind::Default }
1010                        }
1011                    };
1012                    self.current_capture_span_stack.push(MirSpan::BindingId(id));
1013                    self.add_capture(place, CaptureKind::ByRef(capture_kind));
1014                    self.current_capture_span_stack.pop();
1015                }
1016                Pat::TupleStruct { path: _, args, ellipsis } => {
1017                    let Some(variant) = self.result.variant_resolution_for_pat(tgt_pat) else {
1018                        break 'reset_span_stack;
1019                    };
1020                    match variant {
1021                        VariantId::EnumVariantId(_) | VariantId::UnionId(_) => {
1022                            self.consume_place(place)
1023                        }
1024                        VariantId::StructId(s) => {
1025                            let vd = s.fields(self.db);
1026                            let (al, ar) =
1027                                args.split_at(ellipsis.map_or(args.len(), |it| it as usize));
1028                            let fields = vd.fields().iter();
1029                            let it = al
1030                                .iter()
1031                                .zip(fields.clone())
1032                                .chain(ar.iter().rev().zip(fields.rev()));
1033                            for (&arg, (i, _)) in it {
1034                                let mut p = place.clone();
1035                                self.current_capture_span_stack.push(MirSpan::PatId(arg));
1036                                p.projections.push(HirPlaceProjection::Field(FieldId {
1037                                    parent: variant,
1038                                    local_id: i,
1039                                }));
1040                                self.consume_with_pat(p, arg);
1041                                self.current_capture_span_stack.pop();
1042                            }
1043                        }
1044                    }
1045                }
1046                Pat::Ref { pat, mutability: _ } => {
1047                    self.current_capture_span_stack.push(MirSpan::PatId(tgt_pat));
1048                    place.projections.push(HirPlaceProjection::Deref);
1049                    self.consume_with_pat(place, *pat);
1050                    self.current_capture_span_stack.pop();
1051                }
1052                Pat::Box { .. } => (), // not supported
1053                &Pat::Expr(expr) => {
1054                    self.consume_place(place);
1055                    let pat_capture_span_stack = mem::take(&mut self.current_capture_span_stack);
1056                    let old_inside_assignment = mem::replace(&mut self.inside_assignment, false);
1057                    let lhs_place = self.place_of_expr(expr);
1058                    self.mutate_expr(expr, lhs_place);
1059                    self.inside_assignment = old_inside_assignment;
1060                    self.current_capture_span_stack = pat_capture_span_stack;
1061                }
1062            }
1063        }
1064        self.current_capture_span_stack
1065            .truncate(self.current_capture_span_stack.len() - adjustments_count);
1066    }

1074    fn closure_kind(&self) -> FnTrait {
1075        let mut r = FnTrait::Fn;
1076        for it in &self.current_captures {
1077            r = cmp::min(
1078                r,
1079                match &it.kind {
1080                    CaptureKind::ByRef(BorrowKind::Mut { .. }) => FnTrait::FnMut,
1081                    CaptureKind::ByRef(BorrowKind::Shallow | BorrowKind::Shared) => FnTrait::Fn,
1082                    CaptureKind::ByValue => FnTrait::FnOnce,
1083                },
1084            )
1085        }
1086        r
1087    }

1089    fn analyze_closure(&mut self, closure: InternedClosureId) -> FnTrait {
1090        let InternedClosure(_, root) = self.db.lookup_intern_closure(closure);
1091        self.current_closure = Some(closure);
1092        let Expr::Closure { body, capture_by, .. } = &self.body[root] else {
1093            unreachable!("Closure expression id is always closure");
1094        };
1095        self.consume_expr(*body);
1096        for item in &self.current_captures {
1097            if matches!(
1098                item.kind,
1099                CaptureKind::ByRef(BorrowKind::Mut {
1100                    kind: MutBorrowKind::Default | MutBorrowKind::TwoPhasedBorrow
1101                })
1102            ) && !item.place.projections.contains(&HirPlaceProjection::Deref)
1103            {
1104                // FIXME: remove the `mutated_bindings_in_closure` completely and add proper fake reads in
1105                // MIR. I didn't do that due duplicate diagnostics.
1106                self.result.mutated_bindings_in_closure.insert(item.place.local);
1107            }
1108        }
1109        self.restrict_precision_for_unsafe();
1110        // `closure_kind` should be done before adjust_for_move_closure
1111        // If there exists pre-deduced kind of a closure, use it instead of one determined by capture, as rustc does.
1112        // rustc also does diagnostics here if the latter is not a subtype of the former.
1113        let closure_kind = self
1114            .result
1115            .closure_info
1116            .get(&closure)
1117            .map_or_else(|| self.closure_kind(), |info| info.1);
1118        match capture_by {
1119            CaptureBy::Value => self.adjust_for_move_closure(),
1120            CaptureBy::Ref => (),
1121        }
1122        self.minimize_captures();
1123        self.strip_captures_ref_span();
1124        let result = mem::take(&mut self.current_captures);
1125        let captures = result.into_iter().map(|it| it.with_ty(self)).collect::<Vec<_>>();
1126        self.result.closure_info.insert(closure, (captures, closure_kind));
1127        closure_kind
1128    }

1130    fn strip_captures_ref_span(&mut self) {
1131        // FIXME: Borrow checker won't allow without this.
1132        let mut captures = std::mem::take(&mut self.current_captures);
1133        for capture in &mut captures {
1134            if matches!(capture.kind, CaptureKind::ByValue) {
1135                for span_stack in &mut capture.span_stacks {
1136                    if span_stack[span_stack.len() - 1].is_ref_span(self.body) {
1137                        span_stack.truncate(span_stack.len() - 1);
1138                    }
1139                }
1140            }
1141        }
1142        self.current_captures = captures;
1143    }

1145    pub(crate) fn infer_closures(&mut self) {
1146        let deferred_closures = self.sort_closures();
1147        for (closure, exprs) in deferred_closures.into_iter().rev() {
1148            self.current_captures = vec![];
1149            let kind = self.analyze_closure(closure);
1150
1151            for (derefed_callee, callee_ty, params, expr) in exprs {
1152                if let &Expr::Call { callee, .. } = &self.body[expr] {
1153                    let mut adjustments =
1154                        self.result.expr_adjustments.remove(&callee).unwrap_or_default().into_vec();
1155                    self.write_fn_trait_method_resolution(
1156                        kind,
1157                        derefed_callee,
1158                        &mut adjustments,
1159                        callee_ty,
1160                        &params,
1161                        expr,
1162                    );
1163                    self.result.expr_adjustments.insert(callee, adjustments.into_boxed_slice());
1164                }
1165            }
1166        }
1167    }

1169    /// We want to analyze some closures before others, to have a correct analysis:
1170    /// * We should analyze nested closures before the parent, since the parent should capture some of
1171    ///   the things that its children captures.
1172    /// * If a closure calls another closure, we need to analyze the callee, to find out how we should
1173    ///   capture it (e.g. by move for FnOnce)
1174    ///
1175    /// These dependencies are collected in the main inference. We do a topological sort in this function. It
1176    /// will consume the `deferred_closures` field and return its content in a sorted vector.
1177    fn sort_closures(
1178        &mut self,
1179    ) -> Vec<(InternedClosureId, Vec<(Ty<'db>, Ty<'db>, Vec<Ty<'db>>, ExprId)>)> {
1180        let mut deferred_closures = mem::take(&mut self.deferred_closures);
1181        let mut dependents_count: FxHashMap<InternedClosureId, usize> =
1182            deferred_closures.keys().map(|it| (*it, 0)).collect();
1183        for deps in self.closure_dependencies.values() {
1184            for dep in deps {
1185                *dependents_count.entry(*dep).or_default() += 1;
1186            }
1187        }
1188        let mut queue: Vec<_> =
1189            deferred_closures.keys().copied().filter(|&it| dependents_count[&it] == 0).collect();
1190        let mut result = vec![];
1191        while let Some(it) = queue.pop() {
1192            if let Some(d) = deferred_closures.remove(&it) {
1193                result.push((it, d));
1194            }
1195            for &dep in self.closure_dependencies.get(&it).into_iter().flat_map(|it| it.iter()) {
1196                let cnt = dependents_count.get_mut(&dep).unwrap();
1197                *cnt -= 1;
1198                if *cnt == 0 {
1199                    queue.push(dep);
1200                }
1201            }
1202        }
1203        assert!(deferred_closures.is_empty(), "we should have analyzed all closures");
1204        result
1205    }
1206
1207    pub(crate) fn add_current_closure_dependency(&mut self, dep: InternedClosureId) {
1208        if let Some(c) = self.current_closure
1209            && !dep_creates_cycle(&self.closure_dependencies, &mut FxHashSet::default(), c, dep)
1210        {
1211            self.closure_dependencies.entry(c).or_default().push(dep);
1212        }
1213
1214        fn dep_creates_cycle(
1215            closure_dependencies: &FxHashMap<InternedClosureId, Vec<InternedClosureId>>,
1216            visited: &mut FxHashSet<InternedClosureId>,
1217            from: InternedClosureId,
1218            to: InternedClosureId,
1219        ) -> bool {
1220            if !visited.insert(from) {
1221                return false;
1222            }
1223
1224            if from == to {
1225                return true;
1226            }
1227
1228            if let Some(deps) = closure_dependencies.get(&to) {
1229                for dep in deps {
1230                    if dep_creates_cycle(closure_dependencies, visited, from, *dep) {
1231                        return true;
1232                    }
1233                }
1234            }
1235
1236            false
1237        }
1238    }
1239}

1241/// Call this only when the last span in the stack isn't a split.
1242fn apply_adjusts_to_place(
1243    current_capture_span_stack: &mut Vec<MirSpan>,
1244    mut r: HirPlace,
1245    adjustments: &[Adjustment],
1246) -> Option<HirPlace> {
1247    let span = *current_capture_span_stack.last().expect("empty capture span stack");
1248    for adj in adjustments {
1249        match &adj.kind {
1250            Adjust::Deref(None) => {
1251                current_capture_span_stack.push(span);
1252                r.projections.push(HirPlaceProjection::Deref);
1253            }
1254            _ => return None,
1255        }
1256    }
1257    Some(r)
