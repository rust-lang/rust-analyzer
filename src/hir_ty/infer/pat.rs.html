87/// Mode for adjusting the expected type and binding mode.
88#[derive(Clone, Copy, Debug, PartialEq, Eq)]
89enum AdjustMode {
90    /// Peel off all immediate reference types. If the `deref_patterns` feature is enabled, this
91    /// also peels smart pointer ADTs.
92    Peel { kind: PeelKind },
93    /// Pass on the input binding mode and expected type.
94    Pass,
95}

97/// Restrictions on what types to peel when adjusting the expected type and binding mode.
98#[derive(Clone, Copy, Debug, PartialEq, Eq)]
99enum PeelKind {
100    /// Only peel reference types. This is used for explicit `deref!(_)` patterns, which dereference
101    /// any number of `&`/`&mut` references, plus a single smart pointer.
102    ExplicitDerefPat,
103    /// Implicitly peel references, and if `deref_patterns` is enabled, smart pointer ADTs.
104    Implicit {
105        /// The ADT the pattern is a constructor for, if applicable, so that we don't peel it. See
106        /// [`ResolvedPat`] for more information.
107        until_adt: Option<AdtId>,
108        /// The number of references at the head of the pattern's type, so we can leave that many
109        /// untouched. This is `1` for string literals, and `0` for most patterns.
110        pat_ref_layers: usize,
111    },
112}
113
114impl AdjustMode {
115    const fn peel_until_adt(opt_adt_def: Option<AdtId>) -> AdjustMode {
116        AdjustMode::Peel { kind: PeelKind::Implicit { until_adt: opt_adt_def, pat_ref_layers: 0 } }
117    }
118    const fn peel_all() -> AdjustMode {
119        AdjustMode::peel_until_adt(None)
120    }

123/// `ref mut` bindings (explicit or match-ergonomics) are not allowed behind an `&` reference.
124/// Normally, the borrow checker enforces this, but for (currently experimental) match ergonomics,
125/// we track this when typing patterns for two purposes:
126///
127/// - For RFC 3627's Rule 3, when this would prevent us from binding with `ref mut`, we limit the
128///   default binding mode to be by shared `ref` when it would otherwise be `ref mut`.
129///
130/// - For RFC 3627's Rule 5, we allow `&` patterns to match against `&mut` references, treating them
131///   as if they were shared references. Since the scrutinee is mutable in this case, the borrow
132///   checker won't catch if we bind with `ref mut`, so we need to throw an error ourselves.
133#[derive(Clone, Copy, Debug, PartialEq, Eq)]
134enum MutblCap {
135    /// Mutability restricted to immutable.
136    Not,
137
138    /// Mutability restricted to immutable, but only because of the pattern
139    /// (not the scrutinee type).
140    ///
141    /// The contained span, if present, points to an `&` pattern
142    /// that is the reason for the restriction,
143    /// and which will be reported in a diagnostic.
144    WeaklyNot,
145
146    /// No restriction on mutability
147    Mut,
148}
149
150impl MutblCap {
151    #[must_use]
152    fn cap_to_weakly_not(self) -> Self {
153        match self {
154            MutblCap::Not => MutblCap::Not,
155            _ => MutblCap::WeaklyNot,
156        }
157    }
158
159    #[must_use]
160    fn as_mutbl(self) -> Mutability {
161        match self {
162            MutblCap::Not | MutblCap::WeaklyNot => Mutability::Not,
163            MutblCap::Mut => Mutability::Mut,
164        }
165    }
166}

168/// Variations on RFC 3627's Rule 4: when do reference patterns match against inherited references?
169///
170/// "Inherited reference" designates the `&`/`&mut` types that arise from using match ergonomics, i.e.
171/// from matching a reference type with a non-reference pattern. E.g. when `Some(x)` matches on
172/// `&mut Option<&T>`, `x` gets type `&mut &T` and the outer `&mut` is considered "inherited".
173#[derive(Clone, Copy, Debug, PartialEq, Eq)]
174enum InheritedRefMatchRule {
175    /// Reference patterns consume only the inherited reference if possible, regardless of whether
176    /// the underlying type being matched against is a reference type. If there is no inherited
177    /// reference, a reference will be consumed from the underlying type.
178    EatOuter,
179    /// Reference patterns consume only a reference from the underlying type if possible. If the
180    /// underlying type is not a reference type, the inherited reference will be consumed.
181    EatInner,
182    /// When the underlying type is a reference type, reference patterns consume both layers of
183    /// reference, i.e. they both reset the binding mode and consume the reference type.
184    EatBoth {
185        /// If `true`, an inherited reference will be considered when determining whether a reference
186        /// pattern matches a given type:
187        /// - If the underlying type is not a reference, a reference pattern may eat the inherited reference;
188        /// - If the underlying type is a reference, a reference pattern matches if it can eat either one
189        ///   of the underlying and inherited references. E.g. a `&mut` pattern is allowed if either the
190        ///   underlying type is `&mut` or the inherited reference is `&mut`.
191        ///
192        /// If `false`, a reference pattern is only matched against the underlying type.
193        /// This is `false` for stable Rust and `true` for both the `ref_pat_eat_one_layer_2024` and
194        /// `ref_pat_eat_one_layer_2024_structural` feature gates.
195        consider_inherited_ref: bool,
196    },
197}

199/// When checking patterns containing paths, we need to know the path's resolution to determine
200/// whether to apply match ergonomics and implicitly dereference the scrutinee. For instance, when
201/// the `deref_patterns` feature is enabled and we're matching against a scrutinee of type
202/// `Cow<'a, Option<u8>>`, we insert an implicit dereference to allow the pattern `Some(_)` to type,
203/// but we must not dereference it when checking the pattern `Cow::Borrowed(_)`.
204///
205/// `ResolvedPat` contains the information from resolution needed to determine match ergonomics
206/// adjustments, and to finish checking the pattern once we know its adjusted type.
207#[derive(Clone, Copy, Debug)]
208struct ResolvedPat<'db> {
209    /// The type of the pattern, to be checked against the type of the scrutinee after peeling. This
210    /// is also used to avoid peeling the scrutinee's constructors (see the `Cow` example above).
211    ty: Ty<'db>,
212    kind: ResolvedPatKind,
213}
214
215#[derive(Clone, Copy, Debug)]
216enum ResolvedPatKind {
217    Path { res: ValueNs },
218    Struct { variant: VariantId },
219    TupleStruct { variant: VariantId },
220}
221
222impl<'db> ResolvedPat<'db> {
223    fn adjust_mode(&self) -> AdjustMode {
224        if let ResolvedPatKind::Path { res, .. } = self.kind
225            && matches!(res, ValueNs::ConstId(_))
226        {
227            // These constants can be of a reference type, e.g. `const X: &u8 = &0;`.
228            // Peeling the reference types too early will cause type checking failures.
229            // Although it would be possible to *also* peel the types of the constants too.
230            AdjustMode::Pass
231        } else {
232            // The remaining possible resolutions for path, struct, and tuple struct patterns are
233            // ADT constructors. As such, we may peel references freely, but we must not peel the
234            // ADT itself from the scrutinee if it's a smart pointer.
235            AdjustMode::peel_until_adt(self.ty.as_adt().map(|(adt, _)| adt))
236        }
237    }
238}

241    /// Experimental pattern feature: after matching against a shared reference, do we limit the
242    /// default binding mode in subpatterns to be `ref` when it would otherwise be `ref mut`?
243    /// This corresponds to Rule 3 of RFC 3627.
244    fn downgrade_mut_inside_shared(&self) -> bool {
245        // NB: RFC 3627 proposes stabilizing Rule 3 in all editions. If we adopt the same behavior
246        // across all editions, this may be removed.
247        self.features.ref_pat_eat_one_layer_2024_structural
248    }

250    /// Experimental pattern feature: when do reference patterns match against inherited references?
251    /// This corresponds to variations on Rule 4 of RFC 3627.
252    fn ref_pat_matches_inherited_ref(&self, edition: Edition) -> InheritedRefMatchRule {
253        // NB: The particular rule used here is likely to differ across editions, so calls to this
254        // may need to become edition checks after match ergonomics stabilize.
255        if edition.at_least_2024() {
256            if self.features.ref_pat_eat_one_layer_2024 {
257                InheritedRefMatchRule::EatOuter
258            } else if self.features.ref_pat_eat_one_layer_2024_structural {
259                InheritedRefMatchRule::EatInner
260            } else {
261                // Currently, matching against an inherited ref on edition 2024 is an error.
262                // Use `EatBoth` as a fallback to be similar to stable Rust.
263                InheritedRefMatchRule::EatBoth { consider_inherited_ref: false }
264            }
265        } else {
266            InheritedRefMatchRule::EatBoth {
267                consider_inherited_ref: self.features.ref_pat_eat_one_layer_2024
268                    || self.features.ref_pat_eat_one_layer_2024_structural,
269            }
270        }
271    }

296    /// Type check the given `pat` against the `expected` type
297    /// with the provided `binding_mode` (default binding mode).
298    ///
299    /// Outside of this module, `check_pat_top` should always be used.
300    /// Conversely, inside this module, `check_pat_top` should never be used.
301    #[instrument(level = "debug", skip(self, pat_info))]
302    fn infer_pat(&mut self, pat_id: PatId, expected: Ty<'db>, pat_info: PatInfo) {
303        // For patterns containing paths, we need the path's resolution to determine whether to
304        // implicitly dereference the scrutinee before matching.
305        let pat = &self.store[pat_id];
306        let opt_path_res = match pat {
307            Pat::Path(path) => Some(self.resolve_pat_path(pat_id, path)),
308            Pat::Record { path, .. } => Some(self.resolve_record_pat(pat_id, path)),
309            Pat::TupleStruct { path, .. } => Some(self.resolve_tuple_struct_pat(pat_id, path)),
310            _ => None,
311        };
312        let adjust_mode = self.calc_adjust_mode(pat_id, pat, opt_path_res);
313        let ty = self.infer_pat_inner(pat_id, opt_path_res, adjust_mode, expected, pat_info);
314        let ty = self.insert_type_vars_shallow(ty);
315        self.write_pat_ty(pat_id, ty);
316
317        // If we implicitly inserted overloaded dereferences before matching check the pattern to
318        // see if the dereferenced types need `DerefMut` bounds.
319        if let Some(derefed_tys) = self.result.pat_adjustment(pat_id)
320            && derefed_tys.iter().any(|adjust| adjust.kind == PatAdjust::OverloadedDeref)
321        {
322            let infer_ok = self.register_deref_mut_bounds_if_needed(
323                pat_id,
324                pat_id,
325                derefed_tys.iter().filter_map(|adjust| match adjust.kind {
326                    PatAdjust::OverloadedDeref => Some(adjust.source.as_ref()),
327                    PatAdjust::BuiltinDeref => None,
328                }),
329            );
330            self.table.register_infer_ok(infer_ok);
331        }
332
333        // (note_1): In most of the cases where (note_1) is referenced
334        // (literals and constants being the exception), we relate types
335        // using strict equality, even though subtyping would be sufficient.
336        // There are a few reasons for this, some of which are fairly subtle
337        // and which cost me (nmatsakis) an hour or two debugging to remember,
338        // so I thought I'd write them down this time.
339        //
340        // 1. There is no loss of expressiveness here, though it does
341        // cause some inconvenience. What we are saying is that the type
342        // of `x` becomes *exactly* what is expected. This can cause unnecessary
343        // errors in some cases, such as this one:
344        //
345        // ```
346        // fn foo<'x>(x: &'x i32) {
347        //    let a = 1;
348        //    let mut z = x;
349        //    z = &a;
350        // }
351        // ```
352        //
353        // The reason we might get an error is that `z` might be
354        // assigned a type like `&'x i32`, and then we would have
355        // a problem when we try to assign `&a` to `z`, because
356        // the lifetime of `&a` (i.e., the enclosing block) is
357        // shorter than `'x`.
358        //
359        // HOWEVER, this code works fine. The reason is that the
360        // expected type here is whatever type the user wrote, not
361        // the initializer's type. In this case the user wrote
362        // nothing, so we are going to create a type variable `Z`.
363        // Then we will assign the type of the initializer (`&'x i32`)
364        // as a subtype of `Z`: `&'x i32 <: Z`. And hence we
365        // will instantiate `Z` as a type `&'0 i32` where `'0` is
366        // a fresh region variable, with the constraint that `'x : '0`.
367        // So basically we're all set.
368        //
369        // Note that there are two tests to check that this remains true
370        // (`regions-reassign-{match,let}-bound-pointer.rs`).
371        //
372        // 2. An outdated issue related to the old HIR borrowck. See the test
373        // `regions-relate-bound-regions-on-closures-to-inference-variables.rs`,
374    }

376    // Helper to avoid resolving the same path pattern several times.
377    fn infer_pat_inner(
378        &mut self,
379        pat: PatId,
380        opt_path_res: Option<Result<ResolvedPat<'db>, ()>>,
381        adjust_mode: AdjustMode,
382        expected: Ty<'db>,
383        pat_info: PatInfo,
384    ) -> Ty<'db> {
385        #[cfg(debug_assertions)]
386        if matches!(pat_info.binding_mode, ByRef::Yes(Mutability::Mut))
387            && pat_info.max_ref_mutbl != MutblCap::Mut
388            && self.downgrade_mut_inside_shared()
389        {
390            panic!("Pattern mutability cap violated!");
391        }
392
393        // Resolve type if needed.
394        let expected = if let AdjustMode::Peel { .. } = adjust_mode
395            && pat_info.pat_origin.default_binding_modes()
396        {
397            self.table.try_structurally_resolve_type(pat.into(), expected)
398        } else {
399            expected
400        };
401
402        match self.store[pat] {
403            // Peel off a `&` or `&mut`from the scrutinee type. See the examples in
404            // `tests/ui/rfcs/rfc-2005-default-binding-mode`.
405            _ if let AdjustMode::Peel { kind: peel_kind } = adjust_mode
406                && pat_info.pat_origin.default_binding_modes()
407                && let TyKind::Ref(_, inner_ty, inner_mutability) = expected.kind()
408                && self.should_peel_ref(peel_kind, expected) =>
409            {
410                debug!("inspecting {:?}", expected);
411
412                debug!("current discriminant is Ref, inserting implicit deref");
413                // Preserve the reference type. We'll need it later during THIR lowering.
414                self.result.pat_adjustments.entry(pat).or_default().push(PatAdjustment {
415                    kind: PatAdjust::BuiltinDeref,
416                    source: expected.store(),
417                });
418
419                // Use the old pat info to keep `current_depth` to its old value.
420                let new_pat_info = self.adjust_pat_info(inner_mutability, pat_info);
421
422                // Recurse with the new expected type.
423                self.infer_pat_inner(pat, opt_path_res, adjust_mode, inner_ty, new_pat_info)
424            }
425            // If `deref_patterns` is enabled, peel a smart pointer from the scrutinee type. See the
426            // examples in `tests/ui/pattern/deref_patterns/`.
427            _ if self.features.deref_patterns
428                && let AdjustMode::Peel { kind: peel_kind } = adjust_mode
429                && pat_info.pat_origin.default_binding_modes()
430                && self.should_peel_smart_pointer(peel_kind, expected) =>
431            {
432                debug!("scrutinee ty {expected:?} is a smart pointer, inserting pin deref");
433
434                // The scrutinee is a smart pointer; implicitly dereference it. This adds a
435                // requirement that `expected: DerefPure`.
436                let inner_ty = self.deref_pat_target(pat, expected);
437                // Once we've checked `pat`, we'll add a `DerefMut` bound if it contains any
438                // `ref mut` bindings. See `Self::register_deref_mut_bounds_if_needed`.
439
440                self.check_deref_pattern(
441                    pat,
442                    opt_path_res,
443                    adjust_mode,
444                    expected,
445                    inner_ty,
446                    PatAdjust::OverloadedDeref,
447                    pat_info,
448                )
449            }
450            Pat::Missing => self.types.types.error,
451            Pat::Wild | Pat::Rest => expected,
452            // We allow any type here; we ensure that the type is uninhabited during match checking.
453            // Pat::Never => expected,
454            Pat::Path(_) => {
455                let ty = match opt_path_res.unwrap() {
456                    Ok(ref pr) => self.infer_pat_path(pat, pr, expected),
457                    Err(()) => self.types.types.error,
458                };
459                self.write_pat_ty(pat, ty);
460                ty
461            }
462            Pat::Lit(expr) => self.infer_lit_pat(expr, expected),
463            Pat::Range { start: lhs, end: rhs, .. } => {
464                self.infer_range_pat(pat, lhs, rhs, expected)
465            }
466            Pat::Bind { id: var_id, subpat } => {
467                self.infer_bind_pat(pat, var_id, subpat, expected, pat_info)
468            }
469            Pat::TupleStruct { args: ref subpats, ellipsis: ddpos, .. } => match opt_path_res
470                .unwrap()
471            {
472                Ok(ResolvedPat { ty, kind: ResolvedPatKind::TupleStruct { variant } }) => self
473                    .infer_tuple_struct_pat(pat, subpats, ddpos, ty, variant, expected, pat_info),
474                Err(()) => {
475                    let ty_err = self.types.types.error;
476                    for &subpat in subpats {
477                        self.infer_pat(subpat, ty_err, pat_info);
478                    }
479                    ty_err
480                }
481                Ok(pr) => panic!("tuple struct pattern resolved to {pr:?}"),
482            },
483            Pat::Record { args: ref fields, ellipsis: has_rest_pat, .. } => {
484                match opt_path_res.unwrap() {
485                    Ok(ResolvedPat { ty, kind: ResolvedPatKind::Struct { variant } }) => self
486                        .infer_record_pat(
487                            pat,
488                            fields,
489                            has_rest_pat,
490                            ty,
491                            variant,
492                            expected,
493                            pat_info,
494                        ),
495                    Err(()) => {
496                        let ty_err = self.types.types.error;
497                        for field in fields {
498                            self.infer_pat(field.pat, ty_err, pat_info);
499                        }
500                        ty_err
501                    }
502                    Ok(pr) => panic!("struct pattern resolved to {pr:?}"),
503                }
504            }
505            // Pat::Guard(pat, cond) => {
506            //     self.infer_pat(pat, expected, pat_info);
507            //     self.check_expr_has_type_or_error(cond, self.tcx.types.bool, |_| {});
508            //     expected
509            // }
510            Pat::Or(ref pats) => {
511                for &pat in pats {
512                    self.infer_pat(pat, expected, pat_info);
513                }
514                expected
515            }
516            Pat::Tuple { args: ref elements, ellipsis: ddpos } => {
517                self.infer_tuple_pat(pat, elements, ddpos, expected, pat_info)
518            }
519            Pat::Box { inner } => self.infer_box_pat(pat, inner, expected, pat_info),
520            // Pat::Deref(inner) => self.infer_deref_pat(pat.span, inner, expected, pat_info),
521            Pat::Ref { pat: inner, mutability: mutbl } => self.infer_ref_pat(
522                pat,
523                inner,
524                if mutbl.is_mut() { Mutability::Mut } else { Mutability::Not },
525                expected,
526                pat_info,
527            ),
528            Pat::Slice { prefix: ref before, slice, suffix: ref after } => {
529                self.infer_slice_pat(pat, before, slice, after, expected, pat_info)
530            }
531            Pat::Expr(expr) => self.infer_destructuring_assignment_expr(expr, expected),
532            Pat::ConstBlock(expr) => {
533                self.infer_expr(expr, &Expectation::has_type(expected), ExprIsRead::Yes)
534            }
535        }
536    }

538    fn adjust_pat_info(&self, inner_mutability: Mutability, pat_info: PatInfo) -> PatInfo {
539        let mut binding_mode = match pat_info.binding_mode {
540            // If default binding mode is by value, make it `ref`, `ref mut`, `ref pin const`
541            // or `ref pin mut` (depending on whether we observe `&`, `&mut`, `&pin const` or
542            // `&pin mut`).
543            ByRef::No => ByRef::Yes(inner_mutability),
544            ByRef::Yes(mutability) => {
545                let mutability = match mutability {
546                    // When `ref mut`, stay a `ref mut` (on `&mut`) or downgrade to `ref` (on `&`).
547                    Mutability::Mut => inner_mutability,
548                    // Once a `ref`, always a `ref`.
549                    // This is because a `& &mut` cannot mutate the underlying value.
550                    Mutability::Not => Mutability::Not,
551                };
552                ByRef::Yes(mutability)
553            }
554        };
555
556        let PatInfo { mut max_ref_mutbl, .. } = pat_info;
557        if self.downgrade_mut_inside_shared() {
558            binding_mode = binding_mode.cap_ref_mutability(max_ref_mutbl.as_mutbl());
559        }
560        match binding_mode {
561            ByRef::Yes(Mutability::Not) => max_ref_mutbl = MutblCap::Not,
562            _ => {}
563        }
564        debug!("default binding mode is now {:?}", binding_mode);
565        PatInfo { binding_mode, max_ref_mutbl, ..pat_info }
566    }

569        &mut self,
570        pat: PatId,
571        opt_path_res: Option<Result<ResolvedPat<'db>, ()>>,
572        adjust_mode: AdjustMode,
573        expected: Ty<'db>,
574        mut inner_ty: Ty<'db>,
575        pat_adjust_kind: PatAdjust,
576        pat_info: PatInfo,
577    ) -> Ty<'db> {
578        debug_assert!(
579            !matches!(pat_adjust_kind, PatAdjust::BuiltinDeref),
580            "unexpected deref pattern for builtin reference type {expected:?}",
581        );
582
583        let pat_adjustments = self.result.pat_adjustments.entry(pat).or_default();
584        // We may reach the recursion limit if a user matches on a type `T` satisfying
585        // `T: Deref<Target = T>`; error gracefully in this case.
586        // FIXME(deref_patterns): If `deref_patterns` stabilizes, it may make sense to move
587        // this check out of this branch. Alternatively, this loop could be implemented with
588        // autoderef and this check removed. For now though, don't break code compiling on
589        // stable with lots of `&`s and a low recursion limit, if anyone's done that.
590        if pat_adjustments.len() < self.resolver.top_level_def_map().recursion_limit() as usize {
591            // Preserve the smart pointer type for THIR lowering and closure upvar analysis.
592            pat_adjustments.push(PatAdjustment { kind: pat_adjust_kind, source: expected.store() });
593        } else {
594            // FIXME: Emit an error.
595            inner_ty = self.types.types.error;
596        }
597
598        // Recurse, using the old pat info to keep `current_depth` to its old value.
599        // Peeling smart pointers does not update the default binding mode.
600        self.infer_pat_inner(pat, opt_path_res, adjust_mode, inner_ty, pat_info)
601    }

603    /// How should the binding mode and expected type be adjusted?
604    ///
605    /// When the pattern contains a path, `opt_path_res` must be `Some(path_res)`.
606    fn calc_adjust_mode(
607        &mut self,
608        pat_id: PatId,
609        pat: &Pat,
610        opt_path_res: Option<Result<ResolvedPat<'db>, ()>>,
611    ) -> AdjustMode {
612        match pat {
613            // Type checking these product-like types successfully always require
614            // that the expected type be of those types and not reference types.
615            Pat::Tuple { .. } | Pat::Range { .. } | Pat::Slice { .. } => AdjustMode::peel_all(),
616            // When checking an explicit deref pattern, only peel reference types.
617            // FIXME(deref_patterns): If box patterns and deref patterns need to coexist, box
618            // patterns may want `PeelKind::Implicit`, stopping on encountering a box.
619            Pat::Box { .. } /*  | Pat::Deref(_) */ => {
620                AdjustMode::Peel { kind: PeelKind::ExplicitDerefPat }
621            }
622            // A never pattern behaves somewhat like a literal or unit variant.
623            // Pat::Never => AdjustMode::peel_all(),
624            // For patterns with paths, how we peel the scrutinee depends on the path's resolution.
625            Pat::Record { .. }
626            | Pat::TupleStruct { .. }
627            | Pat::Path(_) => {
628                // If there was an error resolving the path, default to peeling everything.
629                opt_path_res.unwrap().map_or(AdjustMode::peel_all(), |pr| pr.adjust_mode())
630            }
631
632            // String and byte-string literals result in types `&str` and `&[u8]` respectively.
633            // All other literals result in non-reference types.
634            // As a result, we allow `if let 0 = &&0 {}` but not `if let "foo" = &&"foo" {}` unless
635            // `deref_patterns` is enabled.
636            &Pat::Lit(expr) | &Pat::ConstBlock(expr) => {
637                let lit_ty = self.infer_expr_pat_unadjusted(expr);
638                // Call `resolve_vars_if_possible` here for inline const blocks.
639                let lit_ty = self.infcx().resolve_vars_if_possible(lit_ty);
640                // If `deref_patterns` is enabled, allow `if let "foo" = &&"foo" {}`.
641                if self.features.deref_patterns {
642                    let mut peeled_ty = lit_ty;
643                    let mut pat_ref_layers = 0;
644                    while let TyKind::Ref(_, inner_ty, mutbl) =
645                        self.table.try_structurally_resolve_type(pat_id.into(), peeled_ty).kind()
646                    {
647                        // We rely on references at the head of constants being immutable.
648                        debug_assert!(mutbl.is_not());
649                        pat_ref_layers += 1;
650                        peeled_ty = inner_ty;
651                    }
652                    AdjustMode::Peel {
653                        kind: PeelKind::Implicit { until_adt: None, pat_ref_layers },
654                    }
655                } else {
656                    if lit_ty.is_ref() { AdjustMode::Pass } else { AdjustMode::peel_all() }
657                }
658            }
659
660            // Ref patterns are complicated, we handle them in `check_pat_ref`.
661            Pat::Ref { .. }
662            // No need to do anything on a missing pattern.
663            | Pat::Missing
664            // A `_`/`..` pattern works with any expected type, so there's no need to do anything.
665            | Pat::Wild | Pat::Rest
666            // Bindings also work with whatever the expected type is,
667            // and moreover if we peel references off, that will give us the wrong binding type.
668            // Also, we can have a subpattern `binding @ pat`.
669            // Each side of the `@` should be treated independently (like with OR-patterns).
670            | Pat::Bind { .. }
671            // `Pat::Expr(_)` inside assignments becomes a binding in rustc, therefore should be
672            // the same as `Pat::Bind`.
673            | Pat::Expr(_)
674            // An OR-pattern just propagates to each individual alternative.
675            // This is maximally flexible, allowing e.g., `Some(mut x) | &Some(mut x)`.
676            // In that example, `Some(mut x)` results in `Peel` whereas `&Some(mut x)` in `Reset`.
677            | Pat::Or(_)
678            // Like or-patterns, guard patterns just propagate to their subpatterns.
679            /* | Pat::Guard(..) */ => AdjustMode::Pass,
680        }
681    }

683    /// Assuming `expected` is a reference type, determine whether to peel it before matching.
684    fn should_peel_ref(&self, peel_kind: PeelKind, mut expected: Ty<'db>) -> bool {
685        debug_assert!(expected.is_ref());
686        let pat_ref_layers = match peel_kind {
687            PeelKind::ExplicitDerefPat => 0,
688            PeelKind::Implicit { pat_ref_layers, .. } => pat_ref_layers,
689        };
690
691        // Most patterns don't have reference types, so we'll want to peel all references from the
692        // scrutinee before matching. To optimize for the common case, return early.
693        if pat_ref_layers == 0 {
694            return true;
695        }
696        debug_assert!(
697            self.features.deref_patterns,
698            "Peeling for patterns with reference types is gated by `deref_patterns`."
699        );
700
701        // If the pattern has as many or more layers of reference as the expected type, we can match
702        // without peeling more, unless we find a smart pointer or `&mut` that we also need to peel.
703        // We don't treat `&` and `&mut` as interchangeable, but by peeling `&mut`s before matching,
704        // we can still, e.g., match on a `&mut str` with a string literal pattern. This is because
705        // string literal patterns may be used where `str` is expected.
706        let mut expected_ref_layers = 0;
707        while let TyKind::Ref(_, inner_ty, mutbl) = expected.kind() {
708            if mutbl.is_mut() {
709                // Mutable references can't be in the final value of constants, thus they can't be
710                // at the head of their types, thus we should always peel `&mut`.
711                return true;
712            }
713            expected_ref_layers += 1;
714            expected = inner_ty;
715        }
716        pat_ref_layers < expected_ref_layers || self.should_peel_smart_pointer(peel_kind, expected)
717    }

719    /// Determine whether `expected` is a smart pointer type that should be peeled before matching.
720    fn should_peel_smart_pointer(&self, peel_kind: PeelKind, expected: Ty<'db>) -> bool {
721        // Explicit `deref!(_)` patterns match against smart pointers; don't peel in that case.
722        if let PeelKind::Implicit { until_adt, .. } = peel_kind
723            // For simplicity, only apply overloaded derefs if `expected` is a known ADT.
724            // FIXME(deref_patterns): we'll get better diagnostics for users trying to
725            // implicitly deref generics if we allow them here, but primitives, tuples, and
726            // inference vars definitely should be stopped. Figure out what makes most sense.
727            && let TyKind::Adt(scrutinee_adt, _) = expected.kind()
728            // Don't peel if the pattern type already matches the scrutinee. E.g., stop here if
729            // matching on a `Cow<'a, T>` scrutinee with a `Cow::Owned(_)` pattern.
730            && until_adt != Some(scrutinee_adt.def_id())
731            // At this point, the pattern isn't able to match `expected` without peeling. Check
732            // that it implements `Deref` before assuming it's a smart pointer, to get a normal
733            // type error instead of a missing impl error if not. This only checks for `Deref`,
734            // not `DerefPure`: we require that too, but we want a trait error if it's missing.
735            && let Some(deref_trait) = self.lang_items.Deref
736            && self.infcx().type_implements_trait(deref_trait, [expected], self.table.param_env).may_apply()
737        {
738            true
739        } else {
740            false
741        }
742    }

748    fn infer_lit_pat(&mut self, expr: ExprId, expected: Ty<'db>) -> Ty<'db> {
749        let literal = match &self.store[expr] {
750            Expr::Literal(literal) => literal,
751            _ => panic!("expected a literal"),
752        };
753
754        // We've already computed the type above (when checking for a non-ref pat),
755        // so avoid computing it again.
756        let ty = self.expr_ty(expr);
757
758        // Byte string patterns behave the same way as array patterns
759        // They can denote both statically and dynamically-sized byte arrays.
760        // Additionally, when `deref_patterns` is enabled, byte string literal patterns may have
761        // types `[u8]` or `[u8; N]`, in order to type, e.g., `deref!(b"..."): Vec<u8>`.
762        let mut pat_ty = ty;
763        if matches!(literal, Literal::ByteString(_)) {
764            let expected = self.structurally_resolve_type(expr.into(), expected);
765            match expected.kind() {
766                // Allow `b"...": &[u8]`
767                TyKind::Ref(_, inner_ty, _)
768                    if self
769                        .table
770                        .try_structurally_resolve_type(expr.into(), inner_ty)
771                        .is_slice() =>
772                {
773                    trace!(?expr, "polymorphic byte string lit");
774                    pat_ty = self.types.types.static_u8_slice;
775                }
776                // Allow `b"...": [u8; 3]` for `deref_patterns`
777                TyKind::Array(..) if self.features.deref_patterns => {
778                    pat_ty = match ty.kind() {
779                        TyKind::Ref(_, inner_ty, _) => inner_ty,
780                        _ => panic!("found byte string literal with non-ref type {ty:?}"),
781                    }
782                }
783                // Allow `b"...": [u8]` for `deref_patterns`
784                TyKind::Slice(..) if self.features.deref_patterns => {
785                    pat_ty = self.types.types.u8_slice;
786                }
787                // Otherwise, `b"...": &[u8; 3]`
788                _ => {}
789            }
790        }
791
792        // When `deref_patterns` is enabled, in order to allow `deref!("..."): String`, we allow
793        // string literal patterns to have type `str`. This is accounted for when lowering to MIR.
794        if self.features.deref_patterns
795            && matches!(literal, Literal::String(_))
796            && self.table.try_structurally_resolve_type(expr.into(), expected).is_str()
797        {
798            pat_ty = self.types.types.str;
799        }
800
801        // Somewhat surprising: in this case, the subtyping relation goes the
802        // opposite way as the other cases. Actually what we really want is not
803        // a subtyping relation at all but rather that there exists a LUB
804        // (so that they can be compared). However, in practice, constants are
805        // always scalars or strings. For scalars subtyping is irrelevant,
806        // and for strings `ty` is type is `&'static str`, so if we say that
807        //
808        //     &'static str <: expected
809        //
810        // then that's equivalent to there existing a LUB.
811        _ = self.demand_suptype(expr.into(), expected, pat_ty);
812
813        pat_ty
814    }

816    fn infer_range_pat(
817        &mut self,
818        pat: PatId,
819        lhs_expr: Option<ExprId>,
820        rhs_expr: Option<ExprId>,
821        expected: Ty<'db>,
822    ) -> Ty<'db> {
823        let mut calc_side = |opt_expr: Option<ExprId>| match opt_expr {
824            None => None,
825            Some(expr) => {
826                let ty = self.infer_expr_pat_unadjusted(expr);
827                // Check that the end-point is possibly of numeric or char type.
828                // The early check here is not for correctness, but rather better
829                // diagnostics (e.g. when `&str` is being matched, `expected` will
830                // be peeled to `str` while ty here is still `&str`, if we don't
831                // err early here, a rather confusing unification error will be
832                // emitted instead).
833                let ty = self.table.try_structurally_resolve_type(expr.into(), ty);
834                let fail =
835                    !(ty.is_numeric() || ty.is_char() || ty.is_ty_var() || ty.references_error());
836                Some((fail, ty, expr))
837            }
838        };
839        let mut lhs = calc_side(lhs_expr);
840        let mut rhs = calc_side(rhs_expr);
841
842        if let (Some((true, ..)), _) | (_, Some((true, ..))) = (lhs, rhs) {
843            // There exists a side that didn't meet our criteria that the end-point
844            // be of a numeric or char type, as checked in `calc_side` above.
845            // FIXME: Emit an error.
846            return self.types.types.error;
847        }
848
849        // Unify each side with `expected`.
850        // Subtyping doesn't matter here, as the value is some kind of scalar.
851        let mut demand_eqtype = |x: &mut _| {
852            if let Some((_, x_ty, x_expr)) = *x {
853                _ = self.demand_eqtype(ExprOrPatId::from(x_expr), expected, x_ty);
854            }
855        };
856        demand_eqtype(&mut lhs);
857        demand_eqtype(&mut rhs);
858
859        if let (Some((true, ..)), _) | (_, Some((true, ..))) = (lhs, rhs) {
860            return self.types.types.error;
861        }
862
863        // Find the unified type and check if it's of numeric or char type again.
864        // This check is needed if both sides are inference variables.
865        // We require types to be resolved here so that we emit inference failure
866        // rather than "_ is not a char or numeric".
867        let ty = self.structurally_resolve_type(
868            lhs_expr.or(rhs_expr).map(ExprOrPatId::ExprId).unwrap_or(pat.into()),
869            expected,
870        );
871        if !(ty.is_numeric() || ty.is_char() || ty.references_error()) {
872            // FIXME: Emit an error.
873            return self.types.types.error;
874        }
875        ty
876    }

878    fn infer_bind_pat(
879        &mut self,
880        pat: PatId,
881        var_id: BindingId,
882        sub: Option<PatId>,
883        expected: Ty<'db>,
884        pat_info: PatInfo,
885    ) -> Ty<'db> {
886        let PatInfo { binding_mode: def_br, .. } = pat_info;
887        let binding_data = &self.store[var_id];
888
889        // Determine the binding mode...
890        let user_bind_annot = BindingMode::from_annotation(binding_data.mode);
891        let bm = match user_bind_annot {
892            BindingMode(ByRef::No, Mutability::Mut) if let ByRef::Yes(_) = def_br => {
893                // Only mention the experimental `mut_ref` feature if if we're in edition 2024 and
894                // using other experimental matching features compatible with it.
895                if self.edition.at_least_2024()
896                    && (self.features.ref_pat_eat_one_layer_2024
897                        || self.features.ref_pat_eat_one_layer_2024_structural)
898                {
899                    if !self.features.mut_ref {
900                        // FIXME: Emit an error: binding cannot be both mutable and by-reference.
901                    }
902
903                    BindingMode(def_br, Mutability::Mut)
904                } else {
905                    // `mut` resets the binding mode on edition <= 2021
906                    BindingMode(ByRef::No, Mutability::Mut)
907                }
908            }
909            BindingMode(ByRef::No, mutbl) => BindingMode(def_br, mutbl),
910            BindingMode(ByRef::Yes(_), _) => user_bind_annot,
911        };
912
913        if matches!(bm.0, ByRef::Yes(Mutability::Mut))
914            && let MutblCap::WeaklyNot = pat_info.max_ref_mutbl
915        {
916            // FIXME: Emit an error: cannot borrow as mutable inside an `&` pattern.
917        }
918
919        // ...and store it in a side table:
920        self.result.binding_modes.insert(pat, bm);
921
922        debug!("check_pat_ident: pat.hir_id={:?} bm={:?}", pat, bm);
923
924        let local_ty = match bm.0 {
925            ByRef::Yes(mutbl) => {
926                // If the binding is like `ref x | ref mut x`,
927                // then `x` is assigned a value of type `&M T` where M is the
928                // mutability and T is the expected type.
929                //
930                // Under pin ergonomics, if the binding is like `ref pin const|mut x`,
931                // then `x` is assigned a value of type `&pin M T` where M is the
932                // mutability and T is the expected type.
933                //
934                // `x` is assigned a value of type `&M T`, hence `&M T <: typeof(x)`
935                // is required. However, we use equality, which is stronger.
936                // See (note_1) for an explanation.
937                self.new_ref_ty(pat.into(), mutbl, expected)
938            }
939            // Otherwise, the type of x is the expected type `T`.
940            ByRef::No => expected, // As above, `T <: typeof(x)` is required, but we use equality, see (note_1).
941        };
942
943        // We have a concrete type for the local, so we do not need to taint it and hide follow up errors *using* the local.
944        if let Some(existing_local_ty) = self.result.type_of_binding.get(var_id) {
945            // If there are multiple arms, make sure they all agree on
946            // what the type of the binding `x` ought to be.
947            _ = self.demand_eqtype(pat.into(), existing_local_ty.as_ref(), local_ty);
948        } else {
949            self.write_binding_ty(var_id, local_ty);
950        }
951
952        if let Some(p) = sub {
953            self.infer_pat(p, expected, pat_info);
954        }
955
956        local_ty
957    }

1046    fn infer_tuple_struct_pat(
1047        &mut self,
1048        pat: PatId,
1049        subpats: &[PatId],
1050        ddpos: Option<u32>,
1051        pat_ty: Ty<'db>,
1052        variant: VariantId,
1053        expected: Ty<'db>,
1054        pat_info: PatInfo,
1055    ) -> Ty<'db> {
1056        let interner = self.interner();
1057
1058        // Type-check the tuple struct pattern against the expected type.
1059        let had_err = self.demand_eqtype(pat.into(), expected, pat_ty);
1060
1061        let variant_fields = variant.fields(self.db);
1062        let variant_field_tys = self.db.field_types(variant);
1063        let TyKind::Adt(_, args) = pat_ty.kind() else {
1064            panic!("unexpected pattern type {:?}", pat_ty);
1065        };
1066        // Type-check subpatterns.
1067        if subpats.len() == variant_fields.len()
1068            || subpats.len() < variant_fields.len() && ddpos.is_some()
1069        {
1070            for (i, &subpat) in subpats.iter().enumerate_and_adjust(variant_fields.len(), ddpos) {
1071                let field_id = LocalFieldId::from_raw(la_arena::RawIdx::from_u32(i as u32));
1072                let field_ty = variant_field_tys[field_id].get().instantiate(interner, args);
1073                self.infer_pat(subpat, field_ty, pat_info);
1074            }
1075            if let Err(()) = had_err {
1076                for &pat in subpats {
1077                    self.infer_pat(pat, self.types.types.error, pat_info);
1078                }
1079                return self.types.types.error;
1080            }
1081        } else {
1082            self.push_diagnostic(InferenceDiagnostic::MismatchedTupleStructPatArgCount {
1083                pat,
1084                expected: variant_fields.len(),
1085                found: subpats.len(),
1086            });
1087
1088            for (i, &pat) in subpats.iter().enumerate() {
1089                let field_id = LocalFieldId::from_raw(la_arena::RawIdx::from_u32(i as u32));
1090                let expected = match variant_field_tys.get(field_id) {
1091                    Some(field_ty) => field_ty.get().instantiate(interner, args),

1347    // Precondition: Pat is Ref(inner)
1348    fn infer_ref_pat(
1349        &mut self,
1350        pat: PatId,
1351        inner: PatId,
1352        pat_mutbl: Mutability,
1353        mut expected: Ty<'db>,
1354        mut pat_info: PatInfo,
1355    ) -> Ty<'db> {
1356        let ref_pat_matches_mut_ref = self.ref_pat_matches_mut_ref();
1357        if ref_pat_matches_mut_ref && pat_mutbl == Mutability::Not {
1358            // If `&` patterns can match against mutable reference types (RFC 3627, Rule 5), we need
1359            // to prevent subpatterns from binding with `ref mut`. Subpatterns of a shared reference
1360            // pattern should have read-only access to the scrutinee, and the borrow checker won't
1361            // catch it in this case.
1362            pat_info.max_ref_mutbl = pat_info.max_ref_mutbl.cap_to_weakly_not();
1363        }
1364
1365        expected = self.table.try_structurally_resolve_type(pat.into(), expected);
1366        // Determine whether we're consuming an inherited reference and resetting the default
1367        // binding mode, based on edition and enabled experimental features.
1368        if let ByRef::Yes(inh_mut) = pat_info.binding_mode {
1369            match self.ref_pat_matches_inherited_ref(self.edition) {
1370                InheritedRefMatchRule::EatOuter => {
1371                    // ref pattern attempts to consume inherited reference
1372                    if pat_mutbl > inh_mut {
1373                        // Tried to match inherited `ref` with `&mut`
1374                        // NB: This assumes that `&` patterns can match against mutable references
1375                        // (RFC 3627, Rule 5). If we implement a pattern typing ruleset with Rule 4E
1376                        // but not Rule 5, we'll need to check that here.
1377                        debug_assert!(ref_pat_matches_mut_ref);
1378                        // FIXME: Emit an error.
1379                    }
1380
1381                    pat_info.binding_mode = ByRef::No;
1382                    self.result.skipped_ref_pats.insert(pat);
1383                    self.infer_pat(inner, expected, pat_info);
1384                    return expected;
1385                }
1386                InheritedRefMatchRule::EatInner => {
1387                    if let TyKind::Ref(_, _, r_mutbl) = expected.kind()
1388                        && pat_mutbl <= r_mutbl
1389                    {
1390                        // Match against the reference type; don't consume the inherited ref.
1391                        // NB: The check for compatible pattern and ref type mutability assumes that
1392                        // `&` patterns can match against mutable references (RFC 3627, Rule 5). If
1393                        // we implement a pattern typing ruleset with Rule 4 (including the fallback
1394                        // to matching the inherited ref when the inner ref can't match) but not
1395                        // Rule 5, we'll need to check that here.
1396                        debug_assert!(ref_pat_matches_mut_ref);
1397                        // NB: For RFC 3627's Rule 3, we limit the default binding mode's ref
1398                        // mutability to `pat_info.max_ref_mutbl`. If we implement a pattern typing
1399                        // ruleset with Rule 4 but not Rule 3, we'll need to check that here.
1400                        debug_assert!(self.downgrade_mut_inside_shared());
1401                        let mutbl_cap = cmp::min(r_mutbl, pat_info.max_ref_mutbl.as_mutbl());
1402                        pat_info.binding_mode = pat_info.binding_mode.cap_ref_mutability(mutbl_cap);
1403                    } else {
1404                        // The reference pattern can't match against the expected type, so try
1405                        // matching against the inherited ref instead.
1406                        if pat_mutbl > inh_mut {
1407                            // We can't match an inherited shared reference with `&mut`.
1408                            // NB: This assumes that `&` patterns can match against mutable
1409                            // references (RFC 3627, Rule 5). If we implement a pattern typing
1410                            // ruleset with Rule 4 but not Rule 5, we'll need to check that here.
1411                            // FIXME(ref_pat_eat_one_layer_2024_structural): If we already tried
1412                            // matching the real reference, the error message should explain that
1413                            // falling back to the inherited reference didn't work. This should be
1414                            // the same error as the old-Edition version below.
1415                            debug_assert!(ref_pat_matches_mut_ref);
1416                            // FIXME: Emit an error.
1417                        }
1418
1419                        pat_info.binding_mode = ByRef::No;
1420                        self.result.skipped_ref_pats.insert(pat);
1421                        self.infer_pat(inner, expected, pat_info);
1422                        return expected;
1423                    }
1424                }
1425                InheritedRefMatchRule::EatBoth { consider_inherited_ref: true } => {
1426                    // Reset binding mode on old editions
1427                    pat_info.binding_mode = ByRef::No;
1428
1429                    if let TyKind::Ref(_, inner_ty, _) = expected.kind() {
1430                        // Consume both the inherited and inner references.
1431                        if pat_mutbl.is_mut() && inh_mut.is_mut() {
1432                            // As a special case, a `&mut` reference pattern will be able to match
1433                            // against a reference type of any mutability if the inherited ref is
1434                            // mutable. Since this allows us to match against a shared reference
1435                            // type, we refer to this as "falling back" to matching the inherited
1436                            // reference, though we consume the real reference as well. We handle
1437                            // this here to avoid adding this case to the common logic below.
1438                            self.infer_pat(inner, inner_ty, pat_info);
1439                            return expected;
1440                        } else {
1441                            // Otherwise, use the common logic below for matching the inner
1442                            // reference type.
1443                            // FIXME(ref_pat_eat_one_layer_2024_structural): If this results in a
1444                            // mutability mismatch, the error message should explain that falling
1445                            // back to the inherited reference didn't work. This should be the same
1446                            // error as the Edition 2024 version above.
1447                        }
1448                    } else {
1449                        // The expected type isn't a reference type, so only match against the
1450                        // inherited reference.
1451                        if pat_mutbl > inh_mut {
1452                            // We can't match a lone inherited shared reference with `&mut`.
1453                            // FIXME: Emit an error.
1454                        }
1455
1456                        self.result.skipped_ref_pats.insert(pat);
1457                        self.infer_pat(inner, expected, pat_info);
1458                        return expected;
1459                    }
1460                }
1461                InheritedRefMatchRule::EatBoth { consider_inherited_ref: false } => {
1462                    // Reset binding mode on stable Rust. This will be a type error below if
1463                    // `expected` is not a reference type.
1464                    pat_info.binding_mode = ByRef::No;
1465                }
1466            }
1467        }
1468
1469        let (ref_ty, inner_ty) = match self.check_dereferenceable(expected, inner) {
1470            Ok(()) => {
1471                // `demand::subtype` would be good enough, but using `eqtype` turns
1472                // out to be equally general. See (note_1) for details.
1473
1474                // Take region, inner-type from expected type if we can,
1475                // to avoid creating needless variables. This also helps with
1476                // the bad interactions of the given hack detailed in (note_1).
1477                debug!("check_pat_ref: expected={:?}", expected);
1478                match expected.as_reference() {
1479                    Some((r_ty, _, r_mutbl))
1480                        if ((ref_pat_matches_mut_ref && r_mutbl >= pat_mutbl)
1481                            || r_mutbl == pat_mutbl) =>
1482                    {
1483                        if r_mutbl == Mutability::Not {
1484                            pat_info.max_ref_mutbl = MutblCap::Not;
1485                        }
1486
1487                        (expected, r_ty)
1488                    }
1489                    _ => {
1490                        let inner_ty = self.table.next_ty_var(inner.into());
1491                        let ref_ty = self.new_ref_ty(inner.into(), pat_mutbl, inner_ty);
1492                        debug!("check_pat_ref: demanding {:?} = {:?}", expected, ref_ty);
1493                        _ = self.demand_eqtype(pat.into(), expected, ref_ty);
1494
1495                        (ref_ty, inner_ty)
1496                    }
1497                }
1498            }
1499            Err(()) => {
1500                let err = self.types.types.error;
1501                (err, err)
1502            }
1503        };
1504
1505        self.infer_pat(inner, inner_ty, pat_info);
1506        ref_ty
1507    }

1570    /// Type check a slice pattern.
1571    ///
1572    /// Syntactically, these look like `[pat_0, ..., pat_n]`.
1573    /// Semantically, we are type checking a pattern with structure:
1574    /// ```ignore (not-rust)
1575    /// [before_0, ..., before_n, (slice, after_0, ... after_n)?]
1576    /// ```
1577    /// The type of `slice`, if it is present, depends on the `expected` type.
1578    /// If `slice` is missing, then so is `after_i`.
1579    /// If `slice` is present, it can still represent 0 elements.
1580    fn infer_slice_pat(
1581        &mut self,
1582        pat: PatId,
1583        before: &[PatId],
1584        slice: Option<PatId>,
1585        after: &[PatId],
1586        expected: Ty<'db>,
1587        pat_info: PatInfo,
1588    ) -> Ty<'db> {
1589        let expected = self.table.try_structurally_resolve_type(pat.into(), expected);
1590
1591        // If the pattern is irrefutable and `expected` is an infer ty, we try to equate it
1592        // to an array if the given pattern allows it. See issue #76342
1593        if self.pat_is_irrefutable(pat_info.pat_origin)
1594            && expected.is_ty_var()
1595            && let Some(resolved_arr_ty) = self.try_resolve_slice_ty_to_array_ty(before, slice, pat)
1596        {
1597            debug!(?resolved_arr_ty);
1598            let _ = self.demand_eqtype(pat.into(), expected, resolved_arr_ty);
1599        }
1600
1601        let expected = self.structurally_resolve_type(pat.into(), expected);
1602        debug!(?expected);
1603
1604        let (element_ty, opt_slice_ty, inferred) = match expected.kind() {
1605            // An array, so we might have something like `let [a, b, c] = [0, 1, 2];`.
1606            TyKind::Array(element_ty, len) => {
1607                let min = before.len() as u64 + after.len() as u64;
1608                let (opt_slice_ty, expected) =
1609                    self.check_array_pat_len(pat, element_ty, expected, slice, len, min.into());
1610                // `opt_slice_ty.is_none()` => `slice.is_none()`.
1611                // Note, though, that opt_slice_ty could be `Some(error_ty)`.
1612                assert!(opt_slice_ty.is_some() || slice.is_none());
1613                (element_ty, opt_slice_ty, expected)
1614            }
1615            TyKind::Slice(element_ty) => (element_ty, Some(expected), expected),
1616            // The expected type must be an array or slice, but was neither, so error.
1617            _ => {
1618                // FIXME: Emit an error: expected an array or a slice.
1619                let err = self.types.types.error;
1620                (err, Some(err), err)
1621            }
1622        };
1623
1624        // Type check all the patterns before `slice`.
1625        for &elt in before {
1626            self.infer_pat(elt, element_ty, pat_info);
1627        }
1628        // Type check the `slice`, if present, against its expected type.
1629        if let Some(slice) = slice {
1630            self.infer_pat(slice, opt_slice_ty.unwrap(), pat_info);
1631        }
1632        // Type check the elements after `slice`, if present.
1633        for &elt in after {
1634            self.infer_pat(elt, element_ty, pat_info);
1635        }
1636        inferred
1637    }

1639    /// Type check the length of an array pattern.
1640    ///
1641    /// Returns both the type of the variable length pattern (or `None`), and the potentially
1642    /// inferred array type. We only return `None` for the slice type if `slice.is_none()`.
1643    fn check_array_pat_len(
1644        &mut self,
1645        pat: PatId,
1646        element_ty: Ty<'db>,
1647        arr_ty: Ty<'db>,
1648        slice: Option<PatId>,
1649        len: Const<'db>,
1650        min_len: u128,
1651    ) -> (Option<Ty<'db>>, Ty<'db>) {
1652        let len = crate::consteval::try_const_usize(self.db, len);
1653
1654        if let Some(len) = len {
1655            // Now we know the length...
1656            if slice.is_none() {
1657                // ...and since there is no variable-length pattern,
1658                // we require an exact match between the number of elements
1659                // in the array pattern and as provided by the matched type.
1660                if min_len == len {
1661                    return (None, arr_ty);
1662                }
1663
1664                self.push_diagnostic(InferenceDiagnostic::MismatchedArrayPatLen {
1665                    pat,
1666                    expected: len,
1667                    found: min_len,
1668                    has_rest: false,
1669                });
1670            } else if let Some(pat_len) = len.checked_sub(min_len) {
1671                // The variable-length pattern was there,
1672                // so it has an array type with the remaining elements left as its size...
1673                return (Some(Ty::new_array(self.interner(), element_ty, pat_len)), arr_ty);
1674            } else {
1675                // ...however, in this case, there were no remaining elements.
1676                // That is, the slice pattern requires more than the array type offers.
1677                self.push_diagnostic(InferenceDiagnostic::MismatchedArrayPatLen {
1678                    pat,
1679                    expected: len,
1680                    found: min_len,
1681                    has_rest: true,
1682                });
1683            }
1684        } else if slice.is_none() {
1685            // We have a pattern with a fixed length,
1686            // which we can use to infer the length of the array.
1687            let updated_arr_ty = Ty::new_array(self.interner(), element_ty, min_len);
1688            _ = self.demand_eqtype(pat.into(), updated_arr_ty, arr_ty);
1689            return (None, updated_arr_ty);
1690        } else {
1691            // We have a variable-length pattern and don't know the array length.
1692            // This happens if we have e.g.,
1693            // `let [a, b, ..] = arr` where `arr: [T; N]` where `const N: usize`.
1694            // FIXME: Emit an error: cannot pattern-match on an array without a fixed length.
1695        };
1696
1697        // If we get here, we must have emitted an error.
1698        (Some(self.types.types.error), arr_ty)
1699    }

1701    fn infer_destructuring_assignment_expr(&mut self, expr: ExprId, expected: Ty<'db>) -> Ty<'db> {
1702        // LHS of assignment doesn't constitute reads.
1703        let expr_is_read = ExprIsRead::No;
1704        let lhs_ty = self.infer_expr_inner(expr, &Expectation::has_type(expected), expr_is_read);
1705        match self.coerce(expr, expected, lhs_ty, AllowTwoPhase::No, expr_is_read) {
1706            Ok(ty) => ty,
1707            Err(_) => {
1708                self.emit_type_mismatch(expr.into(), expected, lhs_ty);
1709                // `rhs_ty` is returned so no further type mismatches are
1710                // reported because of this mismatch.
1711                expected
1712            }
1713        }
1714    }
