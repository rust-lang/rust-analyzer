27    /// Infers type for tuple struct pattern or its corresponding assignee expression.
28    ///
29    /// Ellipses found in the original pattern or expression must be filtered out.
30    pub(super) fn infer_tuple_struct_pat_like(
31        &mut self,
32        path: Option<&Path>,
33        expected: Ty<'db>,
34        default_bm: BindingMode,
35        id: PatId,
36        ellipsis: Option<u32>,
37        subs: &[PatId],
38        decl: Option<DeclContext>,
39    ) -> Ty<'db> {
40        let (ty, def) = self.resolve_variant(id.into(), path, true);
41        let var_data = def.map(|it| it.fields(self.db));
42        if let Some(variant) = def {
43            self.write_variant_resolution(id.into(), variant);
44        }
45        if let Some(var) = &var_data {
46            let cmp = if ellipsis.is_some() { usize::gt } else { usize::ne };
47
48            if cmp(&subs.len(), &var.fields().len()) {
49                self.push_diagnostic(InferenceDiagnostic::MismatchedTupleStructPatArgCount {
50                    pat: id.into(),
51                    expected: var.fields().len(),
52                    found: subs.len(),
53                });
54            }
55        }
56
57        self.unify(ty, expected);
58
59        match def {
60            _ if subs.is_empty() => {}
61            Some(def) => {
62                let field_types = self.db.field_types(def);
63                let variant_data = def.fields(self.db);
64                let visibilities = self.db.field_visibilities(def);
65
66                let (pre, post) = match ellipsis {
67                    Some(idx) => subs.split_at(idx as usize),
68                    None => (subs, &[][..]),
69                };
70                let post_idx_offset = field_types.iter().count().saturating_sub(post.len());
71
72                let pre_iter = pre.iter().enumerate();
73                let post_iter = (post_idx_offset..).zip(post.iter());
74
75                let substs = ty.as_adt().map(TupleExt::tail);
76
77                for (i, &subpat) in pre_iter.chain(post_iter) {
78                    let expected_ty = {
79                        match variant_data.field(&Name::new_tuple_field(i)) {
80                            Some(local_id) => {
81                                if !visibilities[local_id]
82                                    .is_visible_from(self.db, self.resolver.module())
83                                {
84                                    // FIXME(DIAGNOSE): private tuple field
85                                }
86                                let f = field_types[local_id];
87                                let expected_ty = match substs {
88                                    Some(substs) => f.instantiate(self.interner(), substs),
89                                    None => f.instantiate(self.interner(), &[]),
90                                };
91                                self.process_remote_user_written_ty(expected_ty)
92                            }
93                            None => self.err_ty(),
94                        }
95                    };
96
97                    self.infer_pat(subpat, expected_ty, default_bm, decl);
98                }
99            }
100            None => {
101                let err_ty = self.err_ty();
102                for &inner in subs {
103                    self.infer_pat(inner, err_ty, default_bm, decl);
104                }
105            }
106        }
107
108        ty
109    }

111    /// Infers type for record pattern or its corresponding assignee expression.
112    pub(super) fn infer_record_pat_like(
113        &mut self,
114        path: Option<&Path>,
115        expected: Ty<'db>,
116        default_bm: BindingMode,
117        id: PatId,
118        subs: impl ExactSizeIterator<Item = (Name, PatId)>,
119        decl: Option<DeclContext>,
120    ) -> Ty<'db> {
121        let (ty, def) = self.resolve_variant(id.into(), path, false);
122        if let Some(variant) = def {
123            self.write_variant_resolution(id.into(), variant);
124        }
125
126        self.unify(ty, expected);
127
128        match def {
129            _ if subs.len() == 0 => {}
130            Some(def) => {
131                let field_types = self.db.field_types(def);
132                let variant_data = def.fields(self.db);
133                let visibilities = self.db.field_visibilities(def);
134
135                let substs = ty.as_adt().map(TupleExt::tail);
136
137                for (name, inner) in subs {
138                    let expected_ty = {
139                        match variant_data.field(&name) {
140                            Some(local_id) => {
141                                if !visibilities[local_id]
142                                    .is_visible_from(self.db, self.resolver.module())
143                                {
144                                    self.push_diagnostic(InferenceDiagnostic::NoSuchField {
145                                        field: inner.into(),
146                                        private: Some(local_id),
147                                        variant: def,
148                                    });
149                                }
150                                let f = field_types[local_id];
151                                let expected_ty = match substs {
152                                    Some(substs) => f.instantiate(self.interner(), substs),
153                                    None => f.instantiate(self.interner(), &[]),
154                                };
155                                self.process_remote_user_written_ty(expected_ty)
156                            }
157                            None => {
158                                self.push_diagnostic(InferenceDiagnostic::NoSuchField {
159                                    field: inner.into(),
160                                    private: None,
161                                    variant: def,
162                                });
163                                self.err_ty()
164                            }
165                        }
166                    };
167
168                    self.infer_pat(inner, expected_ty, default_bm, decl);
169                }
170            }
171            None => {
172                let err_ty = self.err_ty();
173                for (_, inner) in subs {
174                    self.infer_pat(inner, err_ty, default_bm, decl);
175                }
176            }
177        }
178
179        ty
180    }

182    /// Infers type for tuple pattern or its corresponding assignee expression.
183    ///
184    /// Ellipses found in the original pattern or expression must be filtered out.
185    pub(super) fn infer_tuple_pat_like(
186        &mut self,
187        expected: Ty<'db>,
188        default_bm: BindingMode,
189        ellipsis: Option<u32>,
190        subs: &[PatId],
191        decl: Option<DeclContext>,
192    ) -> Ty<'db> {
193        let expected = self.table.structurally_resolve_type(expected);
194        let expectations = match expected.kind() {
195            TyKind::Tuple(parameters) => parameters,
196            _ => self.types.empty_tys,
197        };
198
199        let ((pre, post), n_uncovered_patterns) = match ellipsis {
200            Some(idx) => {
201                (subs.split_at(idx as usize), expectations.len().saturating_sub(subs.len()))
202            }
203            None => ((subs, &[][..]), 0),
204        };
205        let mut expectations_iter =
206            expectations.iter().chain(repeat_with(|| self.table.next_ty_var()));
207
208        let mut inner_tys = Vec::with_capacity(n_uncovered_patterns + subs.len());
209
210        inner_tys.extend(expectations_iter.by_ref().take(n_uncovered_patterns + subs.len()));
211
212        // Process pre
213        for (ty, pat) in inner_tys.iter_mut().zip(pre) {
214            *ty = self.infer_pat(*pat, *ty, default_bm, decl);
215        }
216
217        // Process post
218        for (ty, pat) in inner_tys.iter_mut().skip(pre.len() + n_uncovered_patterns).zip(post) {
219            *ty = self.infer_pat(*pat, *ty, default_bm, decl);
220        }
221
222        Ty::new_tup_from_iter(self.interner(), inner_tys.into_iter())
223    }

228        &mut self,

231        decl: Option<DeclContext>,
232    ) {
233        self.infer_pat(pat, expected, BindingMode::default(), decl);

236    fn infer_pat(
237        &mut self,
238        pat: PatId,
239        expected: Ty<'db>,
240        mut default_bm: BindingMode,
241        decl: Option<DeclContext>,
242    ) -> Ty<'db> {
243        let mut expected = self.table.structurally_resolve_type(expected);
244
245        if matches!(&self.body[pat], Pat::Ref { .. }) || self.inside_assignment {
246            cov_mark::hit!(match_ergonomics_ref);
247            // When you encounter a `&pat` pattern, reset to Move.
248            // This is so that `w` is by value: `let (_, &w) = &(1, &2);`
249            // Destructuring assignments also reset the binding mode and
250            // don't do match ergonomics.
251            default_bm = BindingMode::Move;
252        } else if self.is_non_ref_pat(self.body, pat) {
253            let mut pat_adjustments = Vec::new();
254            while let TyKind::Ref(_lifetime, inner, mutability) = expected.kind() {
255                pat_adjustments.push(expected);
256                expected = self.table.try_structurally_resolve_type(inner);
257                default_bm = match default_bm {
258                    BindingMode::Move => BindingMode::Ref(mutability),
259                    BindingMode::Ref(Mutability::Not) => BindingMode::Ref(Mutability::Not),
260                    BindingMode::Ref(Mutability::Mut) => BindingMode::Ref(mutability),
261                }
262            }
263
264            if !pat_adjustments.is_empty() {
265                pat_adjustments.shrink_to_fit();
266                self.result.pat_adjustments.insert(pat, pat_adjustments);
267            }
268        }
269
270        // Lose mutability.
271        let default_bm = default_bm;
272        let expected = expected;
273
274        let ty = match &self.body[pat] {
275            Pat::Tuple { args, ellipsis } => {
276                self.infer_tuple_pat_like(expected, default_bm, *ellipsis, args, decl)
277            }
278            Pat::Or(pats) => {
279                for pat in pats.iter() {
280                    self.infer_pat(*pat, expected, default_bm, decl);
281                }
282                expected
283            }
284            &Pat::Ref { pat, mutability } => {
285                self.infer_ref_pat(pat, lower_mutability(mutability), expected, default_bm, decl)
286            }
287            Pat::TupleStruct { path: p, args: subpats, ellipsis } => self
288                .infer_tuple_struct_pat_like(
289                    p.as_deref(),
290                    expected,
291                    default_bm,
292                    pat,
293                    *ellipsis,
294                    subpats,
295                    decl,
296                ),
297            Pat::Record { path: p, args: fields, ellipsis: _ } => {
298                let subs = fields.iter().map(|f| (f.name.clone(), f.pat));
299                self.infer_record_pat_like(p.as_deref(), expected, default_bm, pat, subs, decl)
300            }
301            Pat::Path(path) => {
302                let ty = self.infer_path(path, pat.into()).unwrap_or_else(|| self.err_ty());
303                let ty_inserted_vars = self.insert_type_vars_shallow(ty);
304                match self.coerce(
305                    pat.into(),
306                    expected,
307                    ty_inserted_vars,
308                    AllowTwoPhase::No,
309                    CoerceNever::Yes,
310                ) {
311                    Ok(coerced_ty) => {
312                        self.write_pat_ty(pat, coerced_ty);
313                        return self.pat_ty_after_adjustment(pat);
314                    }
315                    Err(_) => {
316                        self.result.type_mismatches.insert(
317                            pat.into(),
318                            TypeMismatch { expected, actual: ty_inserted_vars },
319                        );
320                        self.write_pat_ty(pat, ty);
321                        // We return `expected` to prevent cascading errors. I guess an alternative is to
322                        // not emit type mismatches for error types and emit an error type here.
323                        return expected;
324                    }
325                }
326            }
327            Pat::Bind { id, subpat } => {
328                return self.infer_bind_pat(pat, *id, default_bm, *subpat, expected, decl);
329            }
330            Pat::Slice { prefix, slice, suffix } => {
331                self.infer_slice_pat(expected, prefix, *slice, suffix, default_bm, decl)
332            }
333            Pat::Wild => expected,
334            Pat::Range { .. } => {
335                // FIXME: do some checks here.
336                expected
337            }
338            &Pat::Lit(expr) => {
339                // Don't emit type mismatches again, the expression lowering already did that.
340                let ty = self.infer_lit_pat(expr, expected);
341                self.write_pat_ty(pat, ty);
342                return self.pat_ty_after_adjustment(pat);
343            }
344            Pat::Box { inner } => match self.resolve_boxed_box() {
345                Some(box_adt) => {
346                    let (inner_ty, alloc_ty) = match expected.as_adt() {
347                        Some((adt, subst)) if adt == box_adt => {
348                            (subst.type_at(0), subst.as_slice().get(1).and_then(|a| a.as_type()))
349                        }
350                        _ => (self.types.error, None),
351                    };
352
353                    let inner_ty = self.infer_pat(*inner, inner_ty, default_bm, decl);
354                    Ty::new_adt(
355                        self.interner(),
356                        box_adt,
357                        GenericArgs::fill_with_defaults(
358                            self.interner(),
359                            box_adt.into(),
360                            std::iter::once(inner_ty.into()).chain(alloc_ty.map(Into::into)),
361                            |_, id, _| self.table.next_var_for_param(id),
362                        ),
363                    )
364                }
365                None => self.err_ty(),
366            },
367            Pat::ConstBlock(expr) => {
368                let old_inside_assign = std::mem::replace(&mut self.inside_assignment, false);
369                let result =
370                    self.infer_expr(*expr, &Expectation::has_type(expected), ExprIsRead::Yes);
371                self.inside_assignment = old_inside_assign;
372                result
373            }
374            Pat::Expr(expr) => {
375                let old_inside_assign = std::mem::replace(&mut self.inside_assignment, false);
376                // LHS of assignment doesn't constitute reads.
377                let result =
378                    self.infer_expr_coerce(*expr, &Expectation::has_type(expected), ExprIsRead::No);
379                // We are returning early to avoid the unifiability check below.
380                let lhs_ty = self.insert_type_vars_shallow(result);
381                let ty = match self.coerce(
382                    pat.into(),
383                    expected,
384                    lhs_ty,
385                    AllowTwoPhase::No,
386                    CoerceNever::Yes,
387                ) {
388                    Ok(ty) => ty,
389                    Err(_) => {
390                        self.result
391                            .type_mismatches
392                            .insert(pat.into(), TypeMismatch { expected, actual: lhs_ty });
393                        // `rhs_ty` is returned so no further type mismatches are
394                        // reported because of this mismatch.
395                        expected
396                    }
397                };
398                self.write_pat_ty(pat, ty);
399                self.inside_assignment = old_inside_assign;
400                return ty;
401            }
402            Pat::Missing => self.err_ty(),
403        };
404        // use a new type variable if we got error type here
405        let ty = self.insert_type_vars_shallow(ty);
406        // FIXME: This never check is odd, but required with out we do inference right now
407        if !expected.is_never() && !self.unify(ty, expected) {
408            self.result.type_mismatches.insert(pat.into(), TypeMismatch { expected, actual: ty });
409        }
410        self.write_pat_ty(pat, ty);
411        self.pat_ty_after_adjustment(pat)
412    }

414    fn pat_ty_after_adjustment(&self, pat: PatId) -> Ty<'db> {
415        *self
416            .result
417            .pat_adjustments
418            .get(&pat)
419            .and_then(|it| it.first())
420            .unwrap_or(&self.result.type_of_pat[pat])
421    }

423    fn infer_ref_pat(
424        &mut self,
425        inner_pat: PatId,
426        mutability: Mutability,
427        expected: Ty<'db>,
428        default_bm: BindingMode,
429        decl: Option<DeclContext>,
430    ) -> Ty<'db> {
431        let (expectation_type, expectation_lt) = match expected.kind() {
432            TyKind::Ref(lifetime, inner_ty, _exp_mut) => (inner_ty, lifetime),
433            _ => {
434                let inner_ty = self.table.next_ty_var();
435                let inner_lt = self.table.next_region_var();
436                let ref_ty = Ty::new_ref(self.interner(), inner_lt, inner_ty, mutability);
437                // Unification failure will be reported by the caller.
438                self.unify(ref_ty, expected);
439                (inner_ty, inner_lt)
440            }
441        };
442        let subty = self.infer_pat(inner_pat, expectation_type, default_bm, decl);
443        Ty::new_ref(self.interner(), expectation_lt, subty, mutability)
444    }

446    fn infer_bind_pat(
447        &mut self,
448        pat: PatId,
449        binding: BindingId,
450        default_bm: BindingMode,
451        subpat: Option<PatId>,
452        expected: Ty<'db>,
453        decl: Option<DeclContext>,
454    ) -> Ty<'db> {
455        let Binding { mode, .. } = self.body[binding];
456        let mode = if mode == BindingAnnotation::Unannotated {
457            default_bm
458        } else {
459            BindingMode::convert(mode)
460        };
461        self.result.binding_modes.insert(pat, mode);
462
463        let inner_ty = match subpat {
464            Some(subpat) => self.infer_pat(subpat, expected, default_bm, decl),
465            None => expected,
466        };
467        let inner_ty = self.insert_type_vars_shallow(inner_ty);
468
469        let bound_ty = match mode {
470            BindingMode::Ref(mutability) => {
471                let inner_lt = self.table.next_region_var();
472                Ty::new_ref(self.interner(), inner_lt, inner_ty, mutability)
473            }
474            BindingMode::Move => inner_ty,
475        };
476        self.write_pat_ty(pat, inner_ty);
477        self.write_binding_ty(binding, bound_ty);
478        inner_ty
479    }

481    fn infer_slice_pat(
482        &mut self,
483        expected: Ty<'db>,
484        prefix: &[PatId],
485        slice: Option<PatId>,
486        suffix: &[PatId],
487        default_bm: BindingMode,
488        decl: Option<DeclContext>,
489    ) -> Ty<'db> {
490        let expected = self.table.structurally_resolve_type(expected);
491
492        // If `expected` is an infer ty, we try to equate it to an array if the given pattern
493        // allows it. See issue #16609
494        if self.pat_is_irrefutable(decl)
495            && expected.is_ty_var()
496            && let Some(resolved_array_ty) =
497                self.try_resolve_slice_ty_to_array_ty(prefix, suffix, slice)
498        {
499            self.unify(expected, resolved_array_ty);
500        }
501
502        let expected = self.table.try_structurally_resolve_type(expected);
503        let elem_ty = match expected.kind() {
504            TyKind::Array(st, _) | TyKind::Slice(st) => st,
505            _ => self.err_ty(),
506        };
507
508        for &pat_id in prefix.iter().chain(suffix.iter()) {
509            self.infer_pat(pat_id, elem_ty, default_bm, decl);
510        }
511
512        if let Some(slice_pat_id) = slice {
513            let rest_pat_ty = match expected.kind() {
514                TyKind::Array(_, length) => {
515                    let len = try_const_usize(self.db, length);
516                    let len =
517                        len.and_then(|len| len.checked_sub((prefix.len() + suffix.len()) as u128));
518                    Ty::new_array_with_const_len(
519                        self.interner(),
520                        elem_ty,
521                        usize_const(self.db, len, self.resolver.krate()),
522                    )
523                }
524                _ => Ty::new_slice(self.interner(), elem_ty),
525            };
526            self.infer_pat(slice_pat_id, rest_pat_ty, default_bm, decl);
527        }
528
529        match expected.kind() {
530            TyKind::Array(_, const_) => {
531                Ty::new_array_with_const_len(self.interner(), elem_ty, const_)
532            }
533            _ => Ty::new_slice(self.interner(), elem_ty),
534        }
535    }

537    fn infer_lit_pat(&mut self, expr: ExprId, expected: Ty<'db>) -> Ty<'db> {
538        // Like slice patterns, byte string patterns can denote both `&[u8; N]` and `&[u8]`.
539        if let Expr::Literal(Literal::ByteString(_)) = self.body[expr]
540            && let TyKind::Ref(_, inner, _) = expected.kind()
541        {
542            let inner = self.table.try_structurally_resolve_type(inner);
543            if matches!(inner.kind(), TyKind::Slice(_)) {
544                let elem_ty = self.types.u8;
545                let slice_ty = Ty::new_slice(self.interner(), elem_ty);
546                let ty =
547                    Ty::new_ref(self.interner(), self.types.re_static, slice_ty, Mutability::Not);
548                self.write_expr_ty(expr, ty);
549                return ty;
550            }
551        }
552
553        self.infer_expr(expr, &Expectation::has_type(expected), ExprIsRead::Yes)
554    }

556    fn is_non_ref_pat(&mut self, body: &hir_def::expr_store::Body, pat: PatId) -> bool {
557        match &body[pat] {
558            Pat::Tuple { .. }
559            | Pat::TupleStruct { .. }
560            | Pat::Record { .. }
561            | Pat::Range { .. }
562            | Pat::Slice { .. } => true,
563            Pat::Or(pats) => pats.iter().all(|p| self.is_non_ref_pat(body, *p)),
564            Pat::Path(path) => {
565                // A const is a reference pattern, but other value ns things aren't (see #16131).
566                let resolved = self.resolve_value_path_inner(path, pat.into(), true);
567                resolved.is_some_and(|it| !matches!(it.0, hir_def::resolver::ValueNs::ConstId(_)))
568            }
569            Pat::ConstBlock(..) => false,
570            Pat::Lit(expr) => !matches!(
571                body[*expr],
572                Expr::Literal(Literal::String(..) | Literal::CString(..) | Literal::ByteString(..))
573            ),
574            Pat::Wild
575            | Pat::Bind { .. }
576            | Pat::Ref { .. }
577            | Pat::Box { .. }
578            | Pat::Missing
579            | Pat::Expr(_) => false,
580        }
581    }

583    fn try_resolve_slice_ty_to_array_ty(
584        &mut self,
585        before: &[PatId],
586        suffix: &[PatId],
587        slice: Option<PatId>,
588    ) -> Option<Ty<'db>> {
589        if slice.is_some() {
590            return None;
591        }
592
593        let len = before.len() + suffix.len();
594        let size = consteval::usize_const(self.db, Some(len as u128), self.owner.krate(self.db));
595
596        let elem_ty = self.table.next_ty_var();
597        let array_ty = Ty::new_array_with_const_len(self.interner(), elem_ty, size);
598        Some(array_ty)
599    }

601    /// Used to determine whether we can infer the expected type in the slice pattern to be of type array.
602    /// This is only possible if we're in an irrefutable pattern. If we were to allow this in refutable
603    /// patterns we wouldn't e.g. report ambiguity in the following situation:
604    ///
605    /// ```ignore(rust)
606    ///    struct Zeroes;
607    ///    const ARR: [usize; 2] = [0; 2];
608    ///    const ARR2: [usize; 2] = [2; 2];
609    ///
610    ///    impl Into<&'static [usize; 2]> for Zeroes {
611    ///        fn into(self) -> &'static [usize; 2] {
612    ///            &ARR
613    ///        }
614    ///    }
615    ///
616    ///    impl Into<&'static [usize]> for Zeroes {
617    ///        fn into(self) -> &'static [usize] {
618    ///            &ARR2
619    ///        }
620    ///    }
621    ///
622    ///    fn main() {
623    ///        let &[a, b]: &[usize] = Zeroes.into() else {
624    ///           ..
625    ///        };
626    ///    }
627    /// ```
628    ///
629    /// If we're in an irrefutable pattern we prefer the array impl candidate given that
630    /// the slice impl candidate would be rejected anyway (if no ambiguity existed).
631    fn pat_is_irrefutable(&self, decl_ctxt: Option<DeclContext>) -> bool {
632        matches!(decl_ctxt, Some(DeclContext { origin: DeclOrigin::LocalDecl { has_else: false } }))
633    }
634}

639        res |= matches!(body[pat], Pat::Bind { id, .. } if body[id].mode == BindingAnnotation::Ref);
640    });
