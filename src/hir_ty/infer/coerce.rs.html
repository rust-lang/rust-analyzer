1//! # Type Coercion
2//!
3//! Under certain circumstances we will coerce from one type to another,
4//! for example by auto-borrowing. This occurs in situations where the
5//! compiler has a firm 'expected type' that was supplied from the user,
6//! and where the actual type is similar to that expected type in purpose
7//! but not in representation (so actual subtyping is inappropriate).
8//!
9//! ## Reborrowing
10//!
11//! Note that if we are expecting a reference, we will *reborrow*
12//! even if the argument provided was already a reference. This is
13//! useful for freezing mut things (that is, when the expected type is &T
14//! but you have &mut T) and also for avoiding the linearity
15//! of mut things (when the expected is &mut T and you have &mut T). See
16//! the various `tests/ui/coerce/*.rs` tests for
17//! examples of where this is useful.
18//!
19//! ## Subtle note
20//!
21//! When inferring the generic arguments of functions, the argument
22//! order is relevant, which can lead to the following edge case:
23//!
24//! ```ignore (illustrative)
25//! fn foo<T>(a: T, b: T) {
26//!     // ...
27//! }
28//!
29//! foo(&7i32, &mut 7i32);
30//! // This compiles, as we first infer `T` to be `&i32`,
31//! // and then coerce `&mut 7i32` to `&7i32`.
32//!
33//! foo(&mut 7i32, &7i32);
34//! // This does not compile, as we first infer `T` to be `&mut i32`
35//! // and are then unable to coerce `&7i32` to `&mut i32`.
36//! ```

75struct Coerce<'a, 'b, 'db> {
76    table: &'a mut InferenceTable<'db>,
77    has_errors: &'a mut bool,
78    target_features: &'a mut dyn FnMut() -> (&'b TargetFeatures, TargetFeatureIsSafeInTarget),
79    use_lub: bool,
80    /// Determines whether or not allow_two_phase_borrow is set on any
81    /// autoref adjustments we create while coercing. We don't want to
82    /// allow deref coercions to create two-phase borrows, at least initially,
83    /// but we do need two-phase borrows for function argument reborrows.
84    /// See rust#47489 and rust#48598
85    /// See docs on the "AllowTwoPhase" type for a more detailed discussion
86    allow_two_phase: AllowTwoPhase,
87    /// Whether we allow `NeverToAny` coercions. This is unsound if we're
88    /// coercing a place expression without it counting as a read in the MIR.
89    /// This is a side-effect of HIR not really having a great distinction
90    /// between places and values.
91    coerce_never: bool,
92    cause: ObligationCause,
93}

97/// Coercing a mutable reference to an immutable works, while
98/// coercing `&T` to `&mut T` should be forbidden.
99fn coerce_mutbls<'db>(from_mutbl: Mutability, to_mutbl: Mutability) -> RelateResult<'db, ()> {
100    if from_mutbl >= to_mutbl { Ok(()) } else { Err(TypeError::Mutability) }
101}

104fn success<'db>(

106    target: Ty<'db>,
107    obligations: PredicateObligations<'db>,
108) -> CoerceResult<'db> {
109    Ok(InferOk { value: (adj, target), obligations })

128    pub(crate) fn commit_if_ok<T, E>(
129        &mut self,
130        f: impl FnOnce(&mut Self) -> Result<T, E>,
131    ) -> Result<T, E> {
132        let snapshot = self.table.snapshot();
133        let result = f(self);
134        match result {
135            Ok(_) => {}
136            Err(_) => {
137                self.table.rollback_to(snapshot);
138            }
139        }
140        result
141    }

143    fn unify_raw(&mut self, a: Ty<'db>, b: Ty<'db>) -> InferResult<'db, Ty<'db>> {
144        debug!("unify(a: {:?}, b: {:?}, use_lub: {})", a, b, self.use_lub);
145        self.commit_if_ok(|this| {
146            let at = this.infer_ctxt().at(&this.cause, this.table.trait_env.env);
147
148            let res = if this.use_lub {
149                at.lub(b, a)
150            } else {
151                at.sup(b, a)
152                    .map(|InferOk { value: (), obligations }| InferOk { value: b, obligations })
153            };
154
155            // In the new solver, lazy norm may allow us to shallowly equate
156            // more types, but we emit possibly impossible-to-satisfy obligations.
157            // Filter these cases out to make sure our coercion is more accurate.
158            match res {
159                Ok(InferOk { value, obligations }) => {
160                    let mut ocx = ObligationCtxt::new(this.infer_ctxt());
161                    ocx.register_obligations(obligations);
162                    if ocx.try_evaluate_obligations().is_empty() {
163                        Ok(InferOk { value, obligations: ocx.into_pending_obligations() })
164                    } else {
165                        Err(TypeError::Mismatch)
166                    }
167                }
168                res => res,
169            }
170        })
171    }

173    /// Unify two types (using sub or lub).
174    fn unify(&mut self, a: Ty<'db>, b: Ty<'db>) -> CoerceResult<'db> {
175        self.unify_raw(a, b)
176            .and_then(|InferOk { value: ty, obligations }| success(vec![], ty, obligations))
177    }

179    /// Unify two types (using sub or lub) and produce a specific coercion.
180    fn unify_and(
181        &mut self,
182        a: Ty<'db>,
183        b: Ty<'db>,
184        adjustments: impl IntoIterator<Item = Adjustment<'db>>,
185        final_adjustment: Adjust<'db>,
186    ) -> CoerceResult<'db> {
187        self.unify_raw(a, b).and_then(|InferOk { value: ty, obligations }| {
188            success(
189                adjustments
190                    .into_iter()
191                    .chain(std::iter::once(Adjustment { target: ty, kind: final_adjustment }))
192                    .collect(),
193                ty,
194                obligations,
195            )
196        })
197    }

199    #[instrument(skip(self))]
200    fn coerce(&mut self, a: Ty<'db>, b: Ty<'db>) -> CoerceResult<'db> {
201        // First, remove any resolved type variables (at the top level, at least):
202        let a = self.table.shallow_resolve(a);
203        let b = self.table.shallow_resolve(b);
204        debug!("Coerce.tys({:?} => {:?})", a, b);
205
206        // Coercing from `!` to any type is allowed:
207        if a.is_never() {
208            // If we're coercing into an inference var, mark it as possibly diverging.
209            if b.is_infer() {
210                self.table.set_diverging(b);
211            }
212
213            if self.coerce_never {
214                return success(
215                    vec![Adjustment { kind: Adjust::NeverToAny, target: b }],
216                    b,
217                    PredicateObligations::new(),
218                );
219            } else {
220                // Otherwise the only coercion we can do is unification.
221                return self.unify(a, b);
222            }
223        }
224
225        // Coercing *from* an unresolved inference variable means that
226        // we have no information about the source type. This will always
227        // ultimately fall back to some form of subtyping.
228        if a.is_infer() {
229            return self.coerce_from_inference_variable(a, b);
230        }
231
232        // Consider coercing the subtype to a DST
233        //
234        // NOTE: this is wrapped in a `commit_if_ok` because it creates
235        // a "spurious" type variable, and we don't want to have that
236        // type variable in memory if the coercion fails.
237        let unsize = self.commit_if_ok(|this| this.coerce_unsized(a, b));
238        match unsize {
239            Ok(_) => {
240                debug!("coerce: unsize successful");
241                return unsize;
242            }
243            Err(error) => {
244                debug!(?error, "coerce: unsize failed");
245            }
246        }
247
248        // Examine the supertype and consider type-specific coercions, such
249        // as auto-borrowing, coercing pointer mutability, a `dyn*` coercion,
250        // or pin-ergonomics.
251        match b.kind() {
252            TyKind::RawPtr(_, b_mutbl) => {
253                return self.coerce_raw_ptr(a, b, b_mutbl);
254            }
255            TyKind::Ref(r_b, _, mutbl_b) => {
256                return self.coerce_borrowed_pointer(a, b, r_b, mutbl_b);
257            }
258            _ => {}
259        }
260
261        match a.kind() {
262            TyKind::FnDef(..) => {
263                // Function items are coercible to any closure
264                // type; function pointers are not (that would
265                // require double indirection).
266                // Additionally, we permit coercion of function
267                // items to drop the unsafe qualifier.
268                self.coerce_from_fn_item(a, b)
269            }
270            TyKind::FnPtr(a_sig_tys, a_hdr) => {
271                // We permit coercion of fn pointers to drop the
272                // unsafe qualifier.
273                self.coerce_from_fn_pointer(a_sig_tys.with(a_hdr), b)
274            }
275            TyKind::Closure(closure_def_id_a, args_a) => {
276                // Non-capturing closures are coercible to
277                // function pointers or unsafe function pointers.
278                // It cannot convert closures that require unsafe.
279                self.coerce_closure_to_fn(a, closure_def_id_a.0, args_a, b)
280            }
281            _ => {
282                // Otherwise, just use unification rules.
283                self.unify(a, b)
284            }
285        }
286    }

288    /// Coercing *from* an inference variable. In this case, we have no information
289    /// about the source type, so we can't really do a true coercion and we always
290    /// fall back to subtyping (`unify_and`).
291    fn coerce_from_inference_variable(&mut self, a: Ty<'db>, b: Ty<'db>) -> CoerceResult<'db> {
292        debug!("coerce_from_inference_variable(a={:?}, b={:?})", a, b);
293        debug_assert!(a.is_infer() && self.table.shallow_resolve(a) == a);
294        debug_assert!(self.table.shallow_resolve(b) == b);
295
296        if b.is_infer() {
297            // Two unresolved type variables: create a `Coerce` predicate.
298            let target_ty = if self.use_lub { self.table.next_ty_var() } else { b };
299
300            let mut obligations = PredicateObligations::with_capacity(2);
301            for &source_ty in &[a, b] {
302                if source_ty != target_ty {
303                    obligations.push(Obligation::new(
304                        self.interner(),
305                        self.cause.clone(),
306                        self.table.trait_env.env,
307                        Binder::dummy(PredicateKind::Coerce(CoercePredicate {
308                            a: source_ty,
309                            b: target_ty,
310                        })),
311                    ));
312                }
313            }
314
315            debug!(
316                "coerce_from_inference_variable: two inference variables, target_ty={:?}, obligations={:?}",
317                target_ty, obligations
318            );
319            success(vec![], target_ty, obligations)
320        } else {
321            // One unresolved type variable: just apply subtyping, we may be able
322            // to do something useful.
323            self.unify(a, b)
324        }
325    }

327    /// Reborrows `&mut A` to `&mut B` and `&(mut) A` to `&B`.
328    /// To match `A` with `B`, autoderef will be performed,
329    /// calling `deref`/`deref_mut` where necessary.
330    fn coerce_borrowed_pointer(
331        &mut self,
332        a: Ty<'db>,
333        b: Ty<'db>,
334        r_b: Region<'db>,
335        mutbl_b: Mutability,
336    ) -> CoerceResult<'db> {
337        debug!("coerce_borrowed_pointer(a={:?}, b={:?})", a, b);
338        debug_assert!(self.table.shallow_resolve(a) == a);
339        debug_assert!(self.table.shallow_resolve(b) == b);
340
341        // If we have a parameter of type `&M T_a` and the value
342        // provided is `expr`, we will be adding an implicit borrow,
343        // meaning that we convert `f(expr)` to `f(&M *expr)`. Therefore,
344        // to type check, we will construct the type that `&M*expr` would
345        // yield.
346
347        let (r_a, mt_a) = match a.kind() {
348            TyKind::Ref(r_a, ty, mutbl) => {
349                let mt_a = TypeAndMut::<DbInterner<'db>> { ty, mutbl };
350                coerce_mutbls(mt_a.mutbl, mutbl_b)?;
351                (r_a, mt_a)
352            }
353            _ => return self.unify(a, b),
354        };
355
356        let mut first_error = None;
357        let mut r_borrow_var = None;
358        let mut autoderef = Autoderef::new(self.table, a);
359        let mut found = None;
360
361        while let Some((referent_ty, autoderefs)) = autoderef.next() {
362            if autoderefs == 0 {
363                // Don't let this pass, otherwise it would cause
364                // &T to autoref to &&T.
365                continue;
366            }
367
368            // At this point, we have deref'd `a` to `referent_ty`. So
369            // imagine we are coercing from `&'a mut Vec<T>` to `&'b mut [T]`.
370            // In the autoderef loop for `&'a mut Vec<T>`, we would get
371            // three callbacks:
372            //
373            // - `&'a mut Vec<T>` -- 0 derefs, just ignore it
374            // - `Vec<T>` -- 1 deref
375            // - `[T]` -- 2 deref
376            //
377            // At each point after the first callback, we want to
378            // check to see whether this would match out target type
379            // (`&'b mut [T]`) if we autoref'd it. We can't just
380            // compare the referent types, though, because we still
381            // have to consider the mutability. E.g., in the case
382            // we've been considering, we have an `&mut` reference, so
383            // the `T` in `[T]` needs to be unified with equality.
384            //
385            // Therefore, we construct reference types reflecting what
386            // the types will be after we do the final auto-ref and
387            // compare those. Note that this means we use the target
388            // mutability [1], since it may be that we are coercing
389            // from `&mut T` to `&U`.
390            //
391            // One fine point concerns the region that we use. We
392            // choose the region such that the region of the final
393            // type that results from `unify` will be the region we
394            // want for the autoref:
395            //
396            // - if in sub mode, that means we want to use `'b` (the
397            //   region from the target reference) for both
398            //   pointers [2]. This is because sub mode (somewhat
399            //   arbitrarily) returns the subtype region. In the case
400            //   where we are coercing to a target type, we know we
401            //   want to use that target type region (`'b`) because --
402            //   for the program to type-check -- it must be the
403            //   smaller of the two.
404            //   - One fine point. It may be surprising that we can
405            //     use `'b` without relating `'a` and `'b`. The reason
406            //     that this is ok is that what we produce is
407            //     effectively a `&'b *x` expression (if you could
408            //     annotate the region of a borrow), and regionck has
409            //     code that adds edges from the region of a borrow
410            //     (`'b`, here) into the regions in the borrowed
411            //     expression (`*x`, here). (Search for "link".)
412            // - if in lub mode, things can get fairly complicated. The
413            //   easiest thing is just to make a fresh
414            //   region variable [4], which effectively means we defer
415            //   the decision to region inference (and regionck, which will add
416            //   some more edges to this variable). However, this can wind up
417            //   creating a crippling number of variables in some cases --
418            //   e.g., #32278 -- so we optimize one particular case [3].
419            //   Let me try to explain with some examples:
420            //   - The "running example" above represents the simple case,
421            //     where we have one `&` reference at the outer level and
422            //     ownership all the rest of the way down. In this case,
423            //     we want `LUB('a, 'b)` as the resulting region.
424            //   - However, if there are nested borrows, that region is
425            //     too strong. Consider a coercion from `&'a &'x Rc<T>` to
426            //     `&'b T`. In this case, `'a` is actually irrelevant.
427            //     The pointer we want is `LUB('x, 'b`). If we choose `LUB('a,'b)`
428            //     we get spurious errors (`ui/regions-lub-ref-ref-rc.rs`).
429            //     (The errors actually show up in borrowck, typically, because
430            //     this extra edge causes the region `'a` to be inferred to something
431            //     too big, which then results in borrowck errors.)
432            //   - We could track the innermost shared reference, but there is already
433            //     code in regionck that has the job of creating links between
434            //     the region of a borrow and the regions in the thing being
435            //     borrowed (here, `'a` and `'x`), and it knows how to handle
436            //     all the various cases. So instead we just make a region variable
437            //     and let regionck figure it out.
438            let r = if !self.use_lub {
439                r_b // [2] above
440            } else if autoderefs == 1 {
441                r_a // [3] above
442            } else {
443                if r_borrow_var.is_none() {
444                    // create var lazily, at most once
445                    let r = autoderef.table.next_region_var();
446                    r_borrow_var = Some(r); // [4] above
447                }
448                r_borrow_var.unwrap()
449            };
450            let derefd_ty_a = Ty::new_ref(
451                autoderef.table.interner(),
452                r,
453                referent_ty,
454                mutbl_b, // [1] above
455            );
456            // We need to construct a new `Coerce` because of lifetimes.
457            let mut coerce = Coerce {
458                table: autoderef.table,
459                has_errors: self.has_errors,
460                target_features: self.target_features,
461                use_lub: self.use_lub,
462                allow_two_phase: self.allow_two_phase,
463                coerce_never: self.coerce_never,
464                cause: self.cause.clone(),
465            };
466            match coerce.unify_raw(derefd_ty_a, b) {
467                Ok(ok) => {
468                    found = Some(ok);
469                    break;
470                }
471                Err(err) => {
472                    if first_error.is_none() {
473                        first_error = Some(err);
474                    }
475                }
476            }
477        }
478
479        // Extract type or return an error. We return the first error
480        // we got, which should be from relating the "base" type
481        // (e.g., in example above, the failure from relating `Vec<T>`
482        // to the target type), since that should be the least
483        // confusing.
484        let Some(InferOk { value: ty, mut obligations }) = found else {
485            if let Some(first_error) = first_error {
486                debug!("coerce_borrowed_pointer: failed with err = {:?}", first_error);
487                return Err(first_error);
488            } else {
489                // This may happen in the new trait solver since autoderef requires
490                // the pointee to be structurally normalizable, or else it'll just bail.
491                // So when we have a type like `&<not well formed>`, then we get no
492                // autoderef steps (even though there should be at least one). That means
493                // we get no type mismatches, since the loop above just exits early.
494                return Err(TypeError::Mismatch);
495            }
496        };
497
498        if ty == a && mt_a.mutbl.is_not() && autoderef.step_count() == 1 {
499            // As a special case, if we would produce `&'a *x`, that's
500            // a total no-op. We end up with the type `&'a T` just as
501            // we started with. In that case, just skip it
502            // altogether. This is just an optimization.
503            //
504            // Note that for `&mut`, we DO want to reborrow --
505            // otherwise, this would be a move, which might be an
506            // error. For example `foo(self.x)` where `self` and
507            // `self.x` both have `&mut `type would be a move of
508            // `self.x`, but we auto-coerce it to `foo(&mut *self.x)`,
509            // which is a borrow.
510            assert!(mutbl_b.is_not()); // can only coerce &T -> &U
511            return success(vec![], ty, obligations);
512        }
513
514        let InferOk { value: mut adjustments, obligations: o } =
515            autoderef.adjust_steps_as_infer_ok();
516        obligations.extend(o);
517
518        // Now apply the autoref. We have to extract the region out of
519        // the final ref type we got.
520        let TyKind::Ref(region, _, _) = ty.kind() else {
521            panic!("expected a ref type, got {:?}", ty);
522        };
523        adjustments.push(Adjustment {
524            kind: Adjust::Borrow(AutoBorrow::Ref(region, mutbl_b)),
525            target: ty,
526        });
527
528        debug!("coerce_borrowed_pointer: succeeded ty={:?} adjustments={:?}", ty, adjustments);
529
530        success(adjustments, ty, obligations)
531    }

533    /// Performs [unsized coercion] by emulating a fulfillment loop on a
534    /// `CoerceUnsized` goal until all `CoerceUnsized` and `Unsize` goals
535    /// are successfully selected.
536    ///
537    /// [unsized coercion](https://doc.rust-lang.org/reference/type-coercions.html#unsized-coercions)
538    #[instrument(skip(self), level = "debug")]
539    fn coerce_unsized(&mut self, source: Ty<'db>, target: Ty<'db>) -> CoerceResult<'db> {
540        debug!(?source, ?target);
541        debug_assert!(self.table.shallow_resolve(source) == source);
542        debug_assert!(self.table.shallow_resolve(target) == target);
543
544        // We don't apply any coercions incase either the source or target
545        // aren't sufficiently well known but tend to instead just equate
546        // them both.
547        if source.is_infer() {
548            debug!("coerce_unsized: source is a TyVar, bailing out");
549            return Err(TypeError::Mismatch);
550        }
551        if target.is_infer() {
552            debug!("coerce_unsized: target is a TyVar, bailing out");
553            return Err(TypeError::Mismatch);
554        }
555
556        // This is an optimization because coercion is one of the most common
557        // operations that we do in typeck, since it happens at every assignment
558        // and call arg (among other positions).
559        //
560        // These targets are known to never be RHS in `LHS: CoerceUnsized<RHS>`.
561        // That's because these are built-in types for which a core-provided impl
562        // doesn't exist, and for which a user-written impl is invalid.
563        //
564        // This is technically incomplete when users write impossible bounds like
565        // `where T: CoerceUnsized<usize>`, for example, but that trait is unstable
566        // and coercion is allowed to be incomplete. The only case where this matters
567        // is impossible bounds.
568        //
569        // Note that some of these types implement `LHS: Unsize<RHS>`, but they
570        // do not implement *`CoerceUnsized`* which is the root obligation of the
571        // check below.
572        match target.kind() {
573            TyKind::Bool
574            | TyKind::Char
575            | TyKind::Int(_)
576            | TyKind::Uint(_)
577            | TyKind::Float(_)
578            | TyKind::Infer(rustc_type_ir::IntVar(_) | rustc_type_ir::FloatVar(_))
579            | TyKind::Str
580            | TyKind::Array(_, _)
581            | TyKind::Slice(_)
582            | TyKind::FnDef(_, _)
583            | TyKind::FnPtr(_, _)
584            | TyKind::Dynamic(_, _)
585            | TyKind::Closure(_, _)
586            | TyKind::CoroutineClosure(_, _)
587            | TyKind::Coroutine(_, _)
588            | TyKind::CoroutineWitness(_, _)
589            | TyKind::Never
590            | TyKind::Tuple(_) => return Err(TypeError::Mismatch),
591            _ => {}
592        }
593        // Additionally, we ignore `&str -> &str` coercions, which happen very
594        // commonly since strings are one of the most used argument types in Rust,
595        // we do coercions when type checking call expressions.
596        if let TyKind::Ref(_, source_pointee, Mutability::Not) = source.kind()
597            && source_pointee.is_str()
598            && let TyKind::Ref(_, target_pointee, Mutability::Not) = target.kind()
599            && target_pointee.is_str()
600        {
601            return Err(TypeError::Mismatch);
602        }
603
604        let traits = (
605            LangItem::Unsize.resolve_trait(self.table.db, self.table.trait_env.krate),
606            LangItem::CoerceUnsized.resolve_trait(self.table.db, self.table.trait_env.krate),
607        );
608        let (Some(unsize_did), Some(coerce_unsized_did)) = traits else {
609            debug!("missing Unsize or CoerceUnsized traits");
610            return Err(TypeError::Mismatch);
611        };
612
613        // Note, we want to avoid unnecessary unsizing. We don't want to coerce to
614        // a DST unless we have to. This currently comes out in the wash since
615        // we can't unify [T] with U. But to properly support DST, we need to allow
616        // that, at which point we will need extra checks on the target here.
617
618        // Handle reborrows before selecting `Source: CoerceUnsized<Target>`.
619        let reborrow = match (source.kind(), target.kind()) {
620            (TyKind::Ref(_, ty_a, mutbl_a), TyKind::Ref(_, _, mutbl_b)) => {
621                coerce_mutbls(mutbl_a, mutbl_b)?;
622
623                let r_borrow = self.table.next_region_var();
624
625                // We don't allow two-phase borrows here, at least for initial
626                // implementation. If it happens that this coercion is a function argument,
627                // the reborrow in coerce_borrowed_ptr will pick it up.
628                // let mutbl = AutoBorrowMutability::new(mutbl_b, AllowTwoPhase::No);
629                let mutbl = mutbl_b;
630
631                Some((
632                    Adjustment { kind: Adjust::Deref(None), target: ty_a },
633                    Adjustment {
634                        kind: Adjust::Borrow(AutoBorrow::Ref(r_borrow, mutbl)),
635                        target: Ty::new_ref(self.interner(), r_borrow, ty_a, mutbl_b),
636                    },
637                ))
638            }
639            (TyKind::Ref(_, ty_a, mt_a), TyKind::RawPtr(_, mt_b)) => {
640                coerce_mutbls(mt_a, mt_b)?;
641
642                Some((
643                    Adjustment { kind: Adjust::Deref(None), target: ty_a },
644                    Adjustment {
645                        kind: Adjust::Borrow(AutoBorrow::RawPtr(mt_b)),
646                        target: Ty::new_ptr(self.interner(), ty_a, mt_b),
647                    },
648                ))
649            }
650            _ => None,
651        };
652        let coerce_source = reborrow.as_ref().map_or(source, |(_, r)| r.target);
653
654        // Setup either a subtyping or a LUB relationship between
655        // the `CoerceUnsized` target type and the expected type.
656        // We only have the latter, so we use an inference variable
657        // for the former and let type inference do the rest.
658        let coerce_target = self.table.next_ty_var();
659
660        let mut coercion = self.unify_and(
661            coerce_target,
662            target,
663            reborrow.into_iter().flat_map(|(deref, autoref)| [deref, autoref]),
664            Adjust::Pointer(PointerCast::Unsize),
665        )?;
666
667        // Create an obligation for `Source: CoerceUnsized<Target>`.
668        let cause = self.cause.clone();
669
670        // Use a FIFO queue for this custom fulfillment procedure.
671        //
672        // A Vec (or SmallVec) is not a natural choice for a queue. However,
673        // this code path is hot, and this queue usually has a max length of 1
674        // and almost never more than 3. By using a SmallVec we avoid an
675        // allocation, at the (very small) cost of (occasionally) having to
676        // shift subsequent elements down when removing the front element.
677        let mut queue: SmallVec<[PredicateObligation<'db>; 4]> = smallvec![Obligation::new(
678            self.interner(),
679            cause,
680            self.table.trait_env.env,
681            TraitRef::new(
682                self.interner(),
683                coerce_unsized_did.into(),
684                [coerce_source, coerce_target]
685            )
686        )];
687        // Keep resolving `CoerceUnsized` and `Unsize` predicates to avoid
688        // emitting a coercion in cases like `Foo<$1>` -> `Foo<$2>`, where
689        // inference might unify those two inner type variables later.
690        let traits = [coerce_unsized_did, unsize_did];
691        while !queue.is_empty() {
692            let obligation = queue.remove(0);
693            let trait_pred = match obligation.predicate.kind().no_bound_vars() {
694                Some(PredicateKind::Clause(ClauseKind::Trait(trait_pred)))
695                    if traits.contains(&trait_pred.def_id().0) =>
696                {
697                    self.infer_ctxt().resolve_vars_if_possible(trait_pred)
698                }
699                // Eagerly process alias-relate obligations in new trait solver,
700                // since these can be emitted in the process of solving trait goals,
701                // but we need to constrain vars before processing goals mentioning
702                // them.
703                Some(PredicateKind::AliasRelate(..)) => {
704                    let mut ocx = ObligationCtxt::new(self.infer_ctxt());
705                    ocx.register_obligation(obligation);
706                    if !ocx.try_evaluate_obligations().is_empty() {
707                        return Err(TypeError::Mismatch);
708                    }
709                    coercion.obligations.extend(ocx.into_pending_obligations());
710                    continue;
711                }
712                _ => {
713                    coercion.obligations.push(obligation);
714                    continue;
715                }
716            };
717            debug!("coerce_unsized resolve step: {:?}", trait_pred);
718            match self.infer_ctxt().select(&obligation.with(self.interner(), trait_pred)) {
719                // Uncertain or unimplemented.
720                Ok(None) => {
721                    if trait_pred.def_id().0 == unsize_did {
722                        let self_ty = trait_pred.self_ty();
723                        let unsize_ty = trait_pred.trait_ref.args.inner()[1].expect_ty();
724                        debug!("coerce_unsized: ambiguous unsize case for {:?}", trait_pred);
725                        match (self_ty.kind(), unsize_ty.kind()) {
726                            (TyKind::Infer(rustc_type_ir::TyVar(v)), TyKind::Dynamic(..))
727                                if self.table.type_var_is_sized(v) =>
728                            {
729                                debug!("coerce_unsized: have sized infer {:?}", v);
730                                coercion.obligations.push(obligation);
731                                // `$0: Unsize<dyn Trait>` where we know that `$0: Sized`, try going
732                                // for unsizing.
733                            }
734                            _ => {
735                                // Some other case for `$0: Unsize<Something>`. Note that we
736                                // hit this case even if `Something` is a sized type, so just
737                                // don't do the coercion.
738                                debug!("coerce_unsized: ambiguous unsize");
739                                return Err(TypeError::Mismatch);
740                            }
741                        }
742                    } else {
743                        debug!("coerce_unsized: early return - ambiguous");
744                        if !coerce_source.references_non_lt_error()
745                            && !coerce_target.references_non_lt_error()
746                        {
747                            // rustc always early-returns here, even when the types contains errors. However not bailing
748                            // improves error recovery, and while we don't implement generic consts properly, it also helps
749                            // correct code.
750                            return Err(TypeError::Mismatch);
751                        }
752                    }
753                }
754                Err(SelectionError::Unimplemented) => {
755                    debug!("coerce_unsized: early return - can't prove obligation");
756                    return Err(TypeError::Mismatch);
757                }
758
759                Err(SelectionError::TraitDynIncompatible(_)) => {
760                    // Dyn compatibility errors in coercion will *always* be due to the
761                    // fact that the RHS of the coercion is a non-dyn compatible `dyn Trait`
762                    // written in source somewhere (otherwise we will never have lowered
763                    // the dyn trait from HIR to middle).
764                    //
765                    // There's no reason to emit yet another dyn compatibility error,
766                    // especially since the span will differ slightly and thus not be
767                    // deduplicated at all!
768                    self.set_tainted_by_errors();
769                }
770                Err(_err) => {
771                    // FIXME: Report an error:
772                    // let guar = self.err_ctxt().report_selection_error(
773                    //     obligation.clone(),
774                    //     &obligation,
775                    //     &err,
776                    // );
777                    self.set_tainted_by_errors();
778                    // Treat this like an obligation and follow through
779                    // with the unsizing - the lack of a coercion should
780                    // be silent, as it causes a type mismatch later.
781                }
782
783                Ok(Some(ImplSource::UserDefined(impl_source))) => {
784                    queue.extend(impl_source.nested);
785                }
786                Ok(Some(impl_source)) => queue.extend(impl_source.nested_obligations()),
787            }
788        }
789
790        Ok(coercion)
791    }

793    fn coerce_from_safe_fn(
794        &mut self,
795        fn_ty_a: PolyFnSig<'db>,
796        b: Ty<'db>,
797        adjustment: Option<Adjust<'db>>,
798    ) -> CoerceResult<'db> {
799        debug_assert!(self.table.shallow_resolve(b) == b);
800
801        self.commit_if_ok(|this| {
802            if let TyKind::FnPtr(_, hdr_b) = b.kind()
803                && fn_ty_a.safety().is_safe()
804                && !hdr_b.safety.is_safe()
805            {
806                let unsafe_a = Ty::safe_to_unsafe_fn_ty(this.interner(), fn_ty_a);
807                this.unify_and(
808                    unsafe_a,
809                    b,
810                    adjustment.map(|kind| Adjustment {
811                        kind,
812                        target: Ty::new_fn_ptr(this.interner(), fn_ty_a),
813                    }),
814                    Adjust::Pointer(PointerCast::UnsafeFnPointer),
815                )
816            } else {
817                let a = Ty::new_fn_ptr(this.interner(), fn_ty_a);
818                match adjustment {
819                    Some(adjust) => this.unify_and(a, b, [], adjust),
820                    None => this.unify(a, b),
821                }
822            }
823        })
824    }

833    fn coerce_from_fn_item(&mut self, a: Ty<'db>, b: Ty<'db>) -> CoerceResult<'db> {
834        debug!("coerce_from_fn_item(a={:?}, b={:?})", a, b);
835        debug_assert!(self.table.shallow_resolve(a) == a);
836        debug_assert!(self.table.shallow_resolve(b) == b);
837
838        match b.kind() {
839            TyKind::FnPtr(_, b_hdr) => {
840                let a_sig = a.fn_sig(self.interner());
841                if let TyKind::FnDef(def_id, _) = a.kind() {
842                    // Intrinsics are not coercible to function pointers
843                    if let CallableDefId::FunctionId(def_id) = def_id.0 {
844                        if FunctionSignature::is_intrinsic(self.table.db, def_id) {
845                            return Err(TypeError::IntrinsicCast);
846                        }
847
848                        let attrs = self.table.db.attrs(def_id.into());
849                        if attrs.by_key(sym::rustc_force_inline).exists() {
850                            return Err(TypeError::ForceInlineCast);
851                        }
852
853                        if b_hdr.safety.is_safe() && attrs.by_key(sym::target_feature).exists() {
854                            let fn_target_features =
855                                TargetFeatures::from_attrs_no_implications(&attrs);
856                            // Allow the coercion if the current function has all the features that would be
857                            // needed to call the coercee safely.
858                            let (target_features, target_feature_is_safe) =
859                                (self.target_features)();
860                            if target_feature_is_safe == TargetFeatureIsSafeInTarget::No
861                                && !target_features.enabled.is_superset(&fn_target_features.enabled)
862                            {
863                                return Err(TypeError::TargetFeatureCast(
864                                    CallableIdWrapper(def_id.into()).into(),
865                                ));
866                            }
867                        }
868                    }
869                }
870
871                self.coerce_from_safe_fn(
872                    a_sig,
873                    b,
874                    Some(Adjust::Pointer(PointerCast::ReifyFnPointer)),
875                )
876            }
877            _ => self.unify(a, b),
878        }

881    /// Attempts to coerce from the type of a non-capturing closure
882    /// into a function pointer.
883    fn coerce_closure_to_fn(
884        &mut self,
885        a: Ty<'db>,
886        _closure_def_id_a: InternedClosureId,
887        args_a: GenericArgs<'db>,
888        b: Ty<'db>,
889    ) -> CoerceResult<'db> {
890        debug_assert!(self.table.shallow_resolve(a) == a);
891        debug_assert!(self.table.shallow_resolve(b) == b);
892
893        match b.kind() {
894            // FIXME: We need to have an `upvars_mentioned()` query:
895            // At this point we haven't done capture analysis, which means
896            // that the ClosureArgs just contains an inference variable instead
897            // of tuple of captured types.
898            //
899            // All we care here is if any variable is being captured and not the exact paths,
900            // so we check `upvars_mentioned` for root variables being captured.
901            TyKind::FnPtr(_, hdr) =>
902            // if self
903            //     .db
904            //     .upvars_mentioned(closure_def_id_a.expect_local())
905            //     .is_none_or(|u| u.is_empty()) =>
906            {
907                // We coerce the closure, which has fn type
908                //     `extern "rust-call" fn((arg0,arg1,...)) -> _`
909                // to
910                //     `fn(arg0,arg1,...) -> _`
911                // or
912                //     `unsafe fn(arg0,arg1,...) -> _`
913                let safety = hdr.safety;
914                let closure_sig = args_a.closure_sig_untupled().map_bound(|mut sig| {
915                    sig.safety = hdr.safety;
916                    sig
917                });
918                let pointer_ty = Ty::new_fn_ptr(self.interner(), closure_sig);
919                debug!("coerce_closure_to_fn(a={:?}, b={:?}, pty={:?})", a, b, pointer_ty);
920                self.unify_and(
921                    pointer_ty,
922                    b,
923                    [],
924                    Adjust::Pointer(PointerCast::ClosureFnPointer(safety)),
925                )
926            }
927            _ => self.unify(a, b),
928        }

931    fn coerce_raw_ptr(&mut self, a: Ty<'db>, b: Ty<'db>, mutbl_b: Mutability) -> CoerceResult<'db> {
932        debug!("coerce_raw_ptr(a={:?}, b={:?})", a, b);
933        debug_assert!(self.table.shallow_resolve(a) == a);
934        debug_assert!(self.table.shallow_resolve(b) == b);
935
936        let (is_ref, mt_a) = match a.kind() {
937            TyKind::Ref(_, ty, mutbl) => (true, TypeAndMut::<DbInterner<'db>> { ty, mutbl }),
938            TyKind::RawPtr(ty, mutbl) => (false, TypeAndMut { ty, mutbl }),
939            _ => return self.unify(a, b),
940        };
941        coerce_mutbls(mt_a.mutbl, mutbl_b)?;
942
943        // Check that the types which they point at are compatible.
944        let a_raw = Ty::new_ptr(self.interner(), mt_a.ty, mutbl_b);
945        // Although references and raw ptrs have the same
946        // representation, we still register an Adjust::DerefRef so that
947        // regionck knows that the region for `a` must be valid here.
948        if is_ref {
949            self.unify_and(
950                a_raw,
951                b,
952                [Adjustment { kind: Adjust::Deref(None), target: mt_a.ty }],
953                Adjust::Borrow(AutoBorrow::RawPtr(mutbl_b)),
954            )
955        } else if mt_a.mutbl != mutbl_b {
956            self.unify_and(a_raw, b, [], Adjust::Pointer(PointerCast::MutToConstPointer))
957        } else {
958            self.unify(a_raw, b)
959        }
960    }

970    /// Attempt to coerce an expression to a type, and return the
971    /// adjusted type of the expression, if successful.
972    /// Adjustments are only recorded if the coercion succeeded.
973    /// The expressions *must not* have any preexisting adjustments.
974    pub(crate) fn coerce(
975        &mut self,
976        expr: ExprOrPatId,
977        expr_ty: Ty<'db>,
978        mut target: Ty<'db>,
979        allow_two_phase: AllowTwoPhase,
980        coerce_never: CoerceNever,
981    ) -> RelateResult<'db, Ty<'db>> {
982        let source = self.table.try_structurally_resolve_type(expr_ty);
983        target = self.table.try_structurally_resolve_type(target);
984        debug!("coercion::try({:?}: {:?} -> {:?})", expr, source, target);
985
986        let cause = ObligationCause::new();
987        let krate = self.krate();
988        let mut coerce = Coerce {
989            table: &mut self.table,
990            has_errors: &mut self.result.has_errors,
991            cause,
992            allow_two_phase,
993            coerce_never: matches!(coerce_never, CoerceNever::Yes),
994            use_lub: false,
995            target_features: &mut || {
996                Self::target_features(self.db, &self.target_features, self.owner, krate)
997            },
998        };
999        let ok = coerce.commit_if_ok(|coerce| coerce.coerce(source, target))?;
1000
1001        let (adjustments, _) = self.table.register_infer_ok(ok);
1002        match expr {
1003            ExprOrPatId::ExprId(expr) => self.write_expr_adj(expr, adjustments.into_boxed_slice()),
1004            ExprOrPatId::PatId(pat) => self
1005                .write_pat_adj(pat, adjustments.into_iter().map(|adjust| adjust.target).collect()),
1006        }
1007        Ok(target)
1008    }

1010    /// Given some expressions, their known unified type and another expression,
1011    /// tries to unify the types, potentially inserting coercions on any of the
1012    /// provided expressions and returns their LUB (aka "common supertype").
1013    ///
1014    /// This is really an internal helper. From outside the coercion
1015    /// module, you should instantiate a `CoerceMany` instance.
1016    fn try_find_coercion_lub(
1017        &mut self,
1018        exprs: &[ExprId],
1019        prev_ty: Ty<'db>,
1020        new: ExprId,
1021        new_ty: Ty<'db>,
1022    ) -> RelateResult<'db, Ty<'db>> {
1023        let prev_ty = self.table.try_structurally_resolve_type(prev_ty);
1024        let new_ty = self.table.try_structurally_resolve_type(new_ty);
1025        debug!(
1026            "coercion::try_find_coercion_lub({:?}, {:?}, exprs={:?} exprs)",
1027            prev_ty,
1028            new_ty,
1029            exprs.len()
1030        );
1031
1032        // The following check fixes #88097, where the compiler erroneously
1033        // attempted to coerce a closure type to itself via a function pointer.
1034        if prev_ty == new_ty {
1035            return Ok(prev_ty);
1036        }
1037
1038        let is_force_inline = |ty: Ty<'db>| {
1039            if let TyKind::FnDef(CallableIdWrapper(CallableDefId::FunctionId(did)), _) = ty.kind() {
1040                self.db.attrs(did.into()).by_key(sym::rustc_force_inline).exists()
1041            } else {
1042                false
1043            }
1044        };
1045        if is_force_inline(prev_ty) || is_force_inline(new_ty) {
1046            return Err(TypeError::ForceInlineCast);
1047        }
1048
1049        // Special-case that coercion alone cannot handle:
1050        // Function items or non-capturing closures of differing IDs or GenericArgs.
1051        let (a_sig, b_sig) = {
1052            let is_capturing_closure = |_ty: Ty<'db>| {
1053                // FIXME:
1054                // if let TyKind::Closure(closure_def_id, _args) = ty.kind() {
1055                //     self.db.upvars_mentioned(closure_def_id.expect_local()).is_some()
1056                // } else {
1057                //     false
1058                // }
1059                false
1060            };
1061            if is_capturing_closure(prev_ty) || is_capturing_closure(new_ty) {
1062                (None, None)
1063            } else {
1064                match (prev_ty.kind(), new_ty.kind()) {
1065                    (TyKind::FnDef(..), TyKind::FnDef(..)) => {
1066                        // Don't reify if the function types have a LUB, i.e., they
1067                        // are the same function and their parameters have a LUB.
1068                        match self.table.commit_if_ok(|table| {
1069                            // We need to eagerly handle nested obligations due to lazy norm.
1070                            let mut ocx = ObligationCtxt::new(&table.infer_ctxt);
1071                            let value = ocx.lub(
1072                                &ObligationCause::new(),
1073                                table.trait_env.env,
1074                                prev_ty,
1075                                new_ty,
1076                            )?;
1077                            if ocx.try_evaluate_obligations().is_empty() {
1078                                Ok(InferOk { value, obligations: ocx.into_pending_obligations() })
1079                            } else {
1080                                Err(TypeError::Mismatch)
1081                            }
1082                        }) {
1083                            // We have a LUB of prev_ty and new_ty, just return it.
1084                            Ok(ok) => return Ok(self.table.register_infer_ok(ok)),
1085                            Err(_) => (
1086                                Some(prev_ty.fn_sig(self.table.interner())),
1087                                Some(new_ty.fn_sig(self.table.interner())),
1088                            ),
1089                        }
1090                    }
1091                    (TyKind::Closure(_, args), TyKind::FnDef(..)) => {
1092                        let b_sig = new_ty.fn_sig(self.table.interner());
1093                        let a_sig = args.closure_sig_untupled().map_bound(|mut sig| {
1094                            sig.safety = b_sig.safety();
1095                            sig
1096                        });
1097                        (Some(a_sig), Some(b_sig))
1098                    }
1099                    (TyKind::FnDef(..), TyKind::Closure(_, args)) => {
1100                        let a_sig = prev_ty.fn_sig(self.table.interner());
1101                        let b_sig = args.closure_sig_untupled().map_bound(|mut sig| {
1102                            sig.safety = a_sig.safety();
1103                            sig
1104                        });
1105                        (Some(a_sig), Some(b_sig))
1106                    }
1107                    (TyKind::Closure(_, args_a), TyKind::Closure(_, args_b)) => {
1108                        (Some(args_a.closure_sig_untupled()), Some(args_b.closure_sig_untupled()))
1109                    }
1110                    _ => (None, None),
1111                }
1112            }
1113        };
1114        if let (Some(a_sig), Some(b_sig)) = (a_sig, b_sig) {
1115            // The signature must match.
1116            let sig = self
1117                .table
1118                .infer_ctxt
1119                .at(&ObligationCause::new(), self.table.trait_env.env)
1120                .lub(a_sig, b_sig)
1121                .map(|ok| self.table.register_infer_ok(ok))?;
1122
1123            // Reify both sides and return the reified fn pointer type.
1124            let fn_ptr = Ty::new_fn_ptr(self.table.interner(), sig);
1125            let prev_adjustment = match prev_ty.kind() {
1126                TyKind::Closure(..) => {
1127                    Adjust::Pointer(PointerCast::ClosureFnPointer(a_sig.safety()))
1128                }
1129                TyKind::FnDef(..) => Adjust::Pointer(PointerCast::ReifyFnPointer),
1130                _ => panic!("should not try to coerce a {prev_ty:?} to a fn pointer"),
1131            };
1132            let next_adjustment = match new_ty.kind() {
1133                TyKind::Closure(..) => {
1134                    Adjust::Pointer(PointerCast::ClosureFnPointer(b_sig.safety()))
1135                }
1136                TyKind::FnDef(..) => Adjust::Pointer(PointerCast::ReifyFnPointer),
1137                _ => panic!("should not try to coerce a {new_ty:?} to a fn pointer"),
1138            };
1139            for &expr in exprs {
1140                self.write_expr_adj(
1141                    expr,
1142                    Box::new([Adjustment { kind: prev_adjustment.clone(), target: fn_ptr }]),
1143                );
1144            }
1145            self.write_expr_adj(
1146                new,
1147                Box::new([Adjustment { kind: next_adjustment, target: fn_ptr }]),
1148            );
1149            return Ok(fn_ptr);
1150        }
1151
1152        // Configure a Coerce instance to compute the LUB.
1153        // We don't allow two-phase borrows on any autorefs this creates since we
1154        // probably aren't processing function arguments here and even if we were,
1155        // they're going to get autorefed again anyway and we can apply 2-phase borrows
1156        // at that time.
1157        //
1158        // NOTE: we set `coerce_never` to `true` here because coercion LUBs only
1159        // operate on values and not places, so a never coercion is valid.
1160        let krate = self.krate();
1161        let mut coerce = Coerce {
1162            table: &mut self.table,
1163            has_errors: &mut self.result.has_errors,
1164            cause: ObligationCause::new(),
1165            allow_two_phase: AllowTwoPhase::No,
1166            coerce_never: true,
1167            use_lub: true,
1168            target_features: &mut || {
1169                Self::target_features(self.db, &self.target_features, self.owner, krate)
1170            },
1171        };
1172
1173        // First try to coerce the new expression to the type of the previous ones,
1174        // but only if the new expression has no coercion already applied to it.
1175        let mut first_error = None;
1176        if !self.result.expr_adjustments.contains_key(&new) {
1177            let result = coerce.commit_if_ok(|coerce| coerce.coerce(new_ty, prev_ty));
1178            match result {
1179                Ok(ok) => {
1180                    let (adjustments, target) = self.table.register_infer_ok(ok);
1181                    self.write_expr_adj(new, adjustments.into_boxed_slice());
1182                    debug!(
1183                        "coercion::try_find_coercion_lub: was able to coerce from new type {:?} to previous type {:?} ({:?})",
1184                        new_ty, prev_ty, target
1185                    );
1186                    return Ok(target);
1187                }
1188                Err(e) => first_error = Some(e),
1189            }
1190        }
1191
1192        match coerce.commit_if_ok(|coerce| coerce.coerce(prev_ty, new_ty)) {
1193            Err(_) => {
1194                // Avoid giving strange errors on failed attempts.
1195                if let Some(e) = first_error {
1196                    Err(e)
1197                } else {
1198                    Err(self
1199                        .table
1200                        .commit_if_ok(|table| {
1201                            table
1202                                .infer_ctxt
1203                                .at(&ObligationCause::new(), table.trait_env.env)
1204                                .lub(prev_ty, new_ty)
1205                        })
1206                        .unwrap_err())
1207                }
1208            }
1209            Ok(ok) => {
1210                let (adjustments, target) = self.table.register_infer_ok(ok);
1211                for &expr in exprs {
1212                    self.write_expr_adj(expr, adjustments.as_slice().into());
1213                }
1214                debug!(
1215                    "coercion::try_find_coercion_lub: was able to coerce previous type {:?} to new type {:?} ({:?})",
1216                    prev_ty, new_ty, target
1217                );
1218                Ok(target)
1219            }
1220        }
1221    }
1222}

1224/// CoerceMany encapsulates the pattern you should use when you have
1225/// many expressions that are all getting coerced to a common
1226/// type. This arises, for example, when you have a match (the result
1227/// of each arm is coerced to a common type). It also arises in less
1228/// obvious places, such as when you have many `break foo` expressions
1229/// that target the same loop, or the various `return` expressions in
1230/// a function.
1231///
1232/// The basic protocol is as follows:
1233///
1234/// - Instantiate the `CoerceMany` with an initial `expected_ty`.
1235///   This will also serve as the "starting LUB". The expectation is
1236///   that this type is something which all of the expressions *must*
1237///   be coercible to. Use a fresh type variable if needed.
1238/// - For each expression whose result is to be coerced, invoke `coerce()` with.
1239///   - In some cases we wish to coerce "non-expressions" whose types are implicitly
1240///     unit. This happens for example if you have a `break` with no expression,
1241///     or an `if` with no `else`. In that case, invoke `coerce_forced_unit()`.
1242///   - `coerce()` and `coerce_forced_unit()` may report errors. They hide this
1243///     from you so that you don't have to worry your pretty head about it.
1244///     But if an error is reported, the final type will be `err`.
1245///   - Invoking `coerce()` may cause us to go and adjust the "adjustments" on
1246///     previously coerced expressions.
1247/// - When all done, invoke `complete()`. This will return the LUB of
1248///   all your expressions.
1249///   - WARNING: I don't believe this final type is guaranteed to be
1250///     related to your initial `expected_ty` in any particular way,
1251///     although it will typically be a subtype, so you should check it.
1252///   - Invoking `complete()` may cause us to go and adjust the "adjustments" on
1253///     previously coerced expressions.
1254///
1255/// Example:
1256///
1257/// ```ignore (illustrative)
1258/// let mut coerce = CoerceMany::new(expected_ty);
1259/// for expr in exprs {
1260///     let expr_ty = fcx.check_expr_with_expectation(expr, expected);
1261///     coerce.coerce(fcx, &cause, expr, expr_ty);
1262/// }
1263/// let final_ty = coerce.complete(fcx);
1264/// ```
1265#[derive(Debug, Clone)]
1266pub(crate) struct CoerceMany<'db, 'exprs> {
1267    expected_ty: Ty<'db>,
1268    final_ty: Option<Ty<'db>>,
1269    expressions: Expressions<'exprs>,
1270    pushed: usize,
1271}
1272
1273/// The type of a `CoerceMany` that is storing up the expressions into
1274/// a buffer. We use this for things like `break`.
1275pub(crate) type DynamicCoerceMany<'db> = CoerceMany<'db, 'db>;
1276
1277#[derive(Debug, Clone)]
1278enum Expressions<'exprs> {
1279    Dynamic(SmallVec<[ExprId; 4]>),
1280    UpFront(&'exprs [ExprId]),
1281}
1282
1283impl<'db, 'exprs> CoerceMany<'db, 'exprs> {
1284    /// The usual case; collect the set of expressions dynamically.
1285    /// If the full set of coercion sites is known before hand,
1286    /// consider `with_coercion_sites()` instead to avoid allocation.
1287    pub(crate) fn new(expected_ty: Ty<'db>) -> Self {
1288        Self::make(expected_ty, Expressions::Dynamic(SmallVec::new()))
1289    }
1290
1291    /// As an optimization, you can create a `CoerceMany` with a
1292    /// preexisting slice of expressions. In this case, you are
1293    /// expected to pass each element in the slice to `coerce(...)` in
1294    /// order. This is used with arrays in particular to avoid
1295    /// needlessly cloning the slice.
1296    pub(crate) fn with_coercion_sites(
1297        expected_ty: Ty<'db>,
1298        coercion_sites: &'exprs [ExprId],
1299    ) -> Self {
1300        Self::make(expected_ty, Expressions::UpFront(coercion_sites))
1301    }
1302
1303    fn make(expected_ty: Ty<'db>, expressions: Expressions<'exprs>) -> Self {
1304        CoerceMany { expected_ty, final_ty: None, expressions, pushed: 0 }
1305    }
1306
1307    /// Returns the "expected type" with which this coercion was
1308    /// constructed. This represents the "downward propagated" type
1309    /// that was given to us at the start of typing whatever construct
1310    /// we are typing (e.g., the match expression).
1311    ///
1312    /// Typically, this is used as the expected type when
1313    /// type-checking each of the alternative expressions whose types
1314    /// we are trying to merge.
1315    pub(crate) fn expected_ty(&self) -> Ty<'db> {
1316        self.expected_ty
1317    }
1318
1319    /// Returns the current "merged type", representing our best-guess
1320    /// at the LUB of the expressions we've seen so far (if any). This
1321    /// isn't *final* until you call `self.complete()`, which will return
1322    /// the merged type.
1323    pub(crate) fn merged_ty(&self) -> Ty<'db> {
1324        self.final_ty.unwrap_or(self.expected_ty)
1325    }
1326
1327    /// Indicates that the value generated by `expression`, which is
1328    /// of type `expression_ty`, is one of the possibilities that we
1329    /// could coerce from. This will record `expression`, and later
1330    /// calls to `coerce` may come back and add adjustments and things
1331    /// if necessary.
1332    pub(crate) fn coerce(
1333        &mut self,
1334        icx: &mut InferenceContext<'_, 'db>,
1335        cause: &ObligationCause,
1336        expression: ExprId,
1337        expression_ty: Ty<'db>,
1338    ) {
1339        self.coerce_inner(icx, cause, expression, expression_ty, false, false)
1340    }
1341
1342    /// Indicates that one of the inputs is a "forced unit". This
1343    /// occurs in a case like `if foo { ... };`, where the missing else
1344    /// generates a "forced unit". Another example is a `loop { break;
1345    /// }`, where the `break` has no argument expression. We treat
1346    /// these cases slightly differently for error-reporting
1347    /// purposes. Note that these tend to correspond to cases where
1348    /// the `()` expression is implicit in the source, and hence we do
1349    /// not take an expression argument.
1350    ///
1351    /// The `augment_error` gives you a chance to extend the error
1352    /// message, in case any results (e.g., we use this to suggest
1353    /// removing a `;`).
1354    pub(crate) fn coerce_forced_unit(
1355        &mut self,
1356        icx: &mut InferenceContext<'_, 'db>,
1357        expr: ExprId,
1358        cause: &ObligationCause,
1359        label_unit_as_expected: bool,
1360    ) {
1361        self.coerce_inner(icx, cause, expr, icx.types.unit, true, label_unit_as_expected)
1362    }
1363
1364    /// The inner coercion "engine". If `expression` is `None`, this
1365    /// is a forced-unit case, and hence `expression_ty` must be
1366    /// `Nil`.
1367    pub(crate) fn coerce_inner(
1368        &mut self,
1369        icx: &mut InferenceContext<'_, 'db>,
1370        cause: &ObligationCause,
1371        expression: ExprId,
1372        mut expression_ty: Ty<'db>,
1373        force_unit: bool,
1374        label_expression_as_expected: bool,
1375    ) {
1376        // Incorporate whatever type inference information we have
1377        // until now; in principle we might also want to process
1378        // pending obligations, but doing so should only improve
1379        // compatibility (hopefully that is true) by helping us
1380        // uncover never types better.
1381        if expression_ty.is_ty_var() {
1382            expression_ty = icx.shallow_resolve(expression_ty);
1383        }
1384
1385        let (expected, found) = if label_expression_as_expected {
1386            // In the case where this is a "forced unit", like
1387            // `break`, we want to call the `()` "expected"
1388            // since it is implied by the syntax.
1389            // (Note: not all force-units work this way.)"
1390            (expression_ty, self.merged_ty())
1391        } else {
1392            // Otherwise, the "expected" type for error
1393            // reporting is the current unification type,
1394            // which is basically the LUB of the expressions
1395            // we've seen so far (combined with the expected
1396            // type)
1397            (self.merged_ty(), expression_ty)
1398        };
1399
1400        // Handle the actual type unification etc.
1401        let result = if !force_unit {
1402            if self.pushed == 0 {
1403                // Special-case the first expression we are coercing.
1404                // To be honest, I'm not entirely sure why we do this.
1405                // We don't allow two-phase borrows, see comment in try_find_coercion_lub for why
1406                icx.coerce(
1407                    expression.into(),
1408                    expression_ty,
1409                    self.expected_ty,
1410                    AllowTwoPhase::No,
1411                    CoerceNever::Yes,
1412                )
1413            } else {
1414                match self.expressions {
1415                    Expressions::Dynamic(ref exprs) => icx.try_find_coercion_lub(
1416                        exprs,
1417                        self.merged_ty(),
1418                        expression,
1419                        expression_ty,
1420                    ),
1421                    Expressions::UpFront(coercion_sites) => icx.try_find_coercion_lub(
1422                        &coercion_sites[0..self.pushed],
1423                        self.merged_ty(),
1424                        expression,
1425                        expression_ty,
1426                    ),
1427                }
1428            }
1429        } else {
1430            // this is a hack for cases where we default to `()` because
1431            // the expression etc has been omitted from the source. An
1432            // example is an `if let` without an else:
1433            //
1434            //     if let Some(x) = ... { }
1435            //
1436            // we wind up with a second match arm that is like `_ =>
1437            // ()`. That is the case we are considering here. We take
1438            // a different path to get the right "expected, found"
1439            // message and so forth (and because we know that
1440            // `expression_ty` will be unit).
1441            //
1442            // Another example is `break` with no argument expression.
1443            assert!(expression_ty.is_unit(), "if let hack without unit type");
1444            icx.table.infer_ctxt.at(cause, icx.table.trait_env.env).eq(expected, found).map(
1445                |infer_ok| {
1446                    icx.table.register_infer_ok(infer_ok);
1447                    expression_ty
1448                },
1449            )
1450        };
1451
1452        debug!(?result);
1453        match result {
1454            Ok(v) => {
1455                self.final_ty = Some(v);
1456                match self.expressions {
1457                    Expressions::Dynamic(ref mut buffer) => buffer.push(expression),
1458                    Expressions::UpFront(coercion_sites) => {
1459                        // if the user gave us an array to validate, check that we got
1460                        // the next expression in the list, as expected
1461                        assert_eq!(coercion_sites[self.pushed], expression);
1462                    }
1463                }
1464            }
1465            Err(_coercion_error) => {
1466                // Mark that we've failed to coerce the types here to suppress
1467                // any superfluous errors we might encounter while trying to
1468                // emit or provide suggestions on how to fix the initial error.
1469                icx.set_tainted_by_errors();
1470
1471                self.final_ty = Some(icx.types.error);
1472
1473                icx.result.type_mismatches.insert(
1474                    expression.into(),
1475                    if label_expression_as_expected {
1476                        TypeMismatch { expected: found, actual: expected }
1477                    } else {
1478                        TypeMismatch { expected, actual: found }
1479                    },
1480                );
1481            }
1482        }
1483
1484        self.pushed += 1;
1485    }
1486
1487    pub(crate) fn complete(self, icx: &mut InferenceContext<'_, 'db>) -> Ty<'db> {
1488        if let Some(final_ty) = self.final_ty {
1489            final_ty
1490        } else {
1491            // If we only had inputs that were of type `!` (or no
1492            // inputs at all), then the final type is `!`.
1493            assert_eq!(self.pushed, 0);
1494            icx.types.never
1495        }
1496    }
1497}

1499pub fn could_coerce<'db>(
1500    db: &'db dyn HirDatabase,
1501    env: Arc<TraitEnvironment<'db>>,
1502    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
1503) -> bool {
1504    coerce(db, env, tys).is_ok()
1505}
1506
1507fn coerce<'db>(
1508    db: &'db dyn HirDatabase,
1509    env: Arc<TraitEnvironment<'db>>,
1510    tys: &Canonical<'db, (Ty<'db>, Ty<'db>)>,
1511) -> Result<(Vec<Adjustment<'db>>, Ty<'db>), TypeError<DbInterner<'db>>> {
1512    let mut table = InferenceTable::new(db, env, None);
1513    let interner = table.interner();
1514    let ((ty1_with_vars, ty2_with_vars), vars) = table.infer_ctxt.instantiate_canonical(tys);
1515
1516    let cause = ObligationCause::new();
1517    // FIXME: Target features.
1518    let target_features = TargetFeatures::default();
1519    let mut coerce = Coerce {
1520        table: &mut table,
1521        has_errors: &mut false,
1522        cause,
1523        allow_two_phase: AllowTwoPhase::No,
1524        coerce_never: true,
1525        use_lub: false,
1526        target_features: &mut || (&target_features, TargetFeatureIsSafeInTarget::No),
1527    };
1528    let InferOk { value: (adjustments, ty), obligations } =
1529        coerce.coerce(ty1_with_vars, ty2_with_vars)?;
1530    table.register_predicates(obligations);
1531
1532    // default any type vars that weren't unified back to their original bound vars
1533    // (kind of hacky)
1534    let mut fallback_ty = |debruijn, infer| {
1535        let var = vars.var_values.iter().position(|arg| {
1536            arg.as_type().is_some_and(|ty| match ty.kind() {
1537                TyKind::Infer(it) => infer == it,
1538                _ => false,
1539            })
1540        });
1541        var.map_or_else(
1542            || Ty::new_error(interner, ErrorGuaranteed),
1543            |i| {
1544                Ty::new_bound(
1545                    interner,
1546                    debruijn,
1547                    BoundTy { kind: BoundTyKind::Anon, var: BoundVar::from_usize(i) },
1548                )
1549            },
1550        )
1551    };
1552    let mut fallback_const = |debruijn, infer| {
1553        let var = vars.var_values.iter().position(|arg| {
1554            arg.as_const().is_some_and(|ty| match ty.kind() {
1555                ConstKind::Infer(it) => infer == it,
1556                _ => false,
1557            })
1558        });
1559        var.map_or_else(
1560            || Const::new_error(interner, ErrorGuaranteed),
1561            |i| Const::new_bound(interner, debruijn, BoundConst { var: BoundVar::from_usize(i) }),
1562        )
1563    };
1564    let mut fallback_region = |debruijn, infer| {
1565        let var = vars.var_values.iter().position(|arg| {
1566            arg.as_region().is_some_and(|ty| match ty.kind() {
1567                RegionKind::ReVar(it) => infer == it,
1568                _ => false,
1569            })
1570        });
1571        var.map_or_else(
1572            || Region::error(interner),
1573            |i| {
1574                Region::new_bound(
1575                    interner,
1576                    debruijn,
1577                    BoundRegion { kind: BoundRegionKind::Anon, var: BoundVar::from_usize(i) },
1578                )
1579            },
1580        )
1581    };
1582    // FIXME also map the types in the adjustments
1583    // FIXME: We don't fallback correctly since this is done on `InferenceContext` and we only have `InferenceTable`.
1584    let ty = table.resolve_with_fallback(
1585        ty,
1586        &mut fallback_ty,
1587        &mut fallback_const,
1588        &mut fallback_region,
1589    );
1590    Ok((adjustments, ty))
