70    pub(crate) fn infer_expr(
71        &mut self,
72        tgt_expr: ExprId,
73        expected: &Expectation<'db>,
74        is_read: ExprIsRead,
75    ) -> Ty<'db> {
76        let ty = self.infer_expr_inner(tgt_expr, expected, is_read);
77        if let Some(expected_ty) = expected.only_has_type(&mut self.table) {
78            let could_unify = self.unify(ty, expected_ty);
79            if !could_unify {
80                self.result
81                    .type_mismatches
82                    .insert(tgt_expr.into(), TypeMismatch { expected: expected_ty, actual: ty });
83            }
84        }
85        ty
86    }

96    /// Infer type of expression with possibly implicit coerce to the expected type.
97    /// Return the type after possible coercion.
98    pub(super) fn infer_expr_coerce(
99        &mut self,
100        expr: ExprId,
101        expected: &Expectation<'db>,
102        is_read: ExprIsRead,
103    ) -> Ty<'db> {
104        let ty = self.infer_expr_inner(expr, expected, is_read);
105        if let Some(target) = expected.only_has_type(&mut self.table) {
106            let coerce_never = if self.expr_guaranteed_to_constitute_read_for_never(expr, is_read) {
107                CoerceNever::Yes
108            } else {
109                CoerceNever::No
110            };
111            match self.coerce(expr.into(), ty, target, AllowTwoPhase::No, coerce_never) {
112                Ok(res) => res,
113                Err(_) => {
114                    self.result
115                        .type_mismatches
116                        .insert(expr.into(), TypeMismatch { expected: target, actual: ty });
117                    target
118                }
119            }
120        } else {
121            ty
122        }
123    }

125    /// Whether this expression constitutes a read of value of the type that
126    /// it evaluates to.
127    ///
128    /// This is used to determine if we should consider the block to diverge
129    /// if the expression evaluates to `!`, and if we should insert a `NeverToAny`
130    /// coercion for values of type `!`.
131    ///
132    /// This function generally returns `false` if the expression is a place
133    /// expression and the *parent* expression is the scrutinee of a match or
134    /// the pointee of an `&` addr-of expression, since both of those parent
135    /// expressions take a *place* and not a value.
136    pub(super) fn expr_guaranteed_to_constitute_read_for_never(
137        &mut self,
138        expr: ExprId,
139        is_read: ExprIsRead,
140    ) -> bool {
141        // rustc does the place expr check first, but since we are feeding
142        // readness of the `expr` as a given value, we just can short-circuit
143        // the place expr check if it's true(see codes and comments below)
144        if is_read == ExprIsRead::Yes {
145            return true;
146        }
147
148        // We only care about place exprs. Anything else returns an immediate
149        // which would constitute a read. We don't care about distinguishing
150        // "syntactic" place exprs since if the base of a field projection is
151        // not a place then it would've been UB to read from it anyways since
152        // that constitutes a read.
153        if !self.is_syntactic_place_expr(expr) {
154            return true;
155        }
156
157        // rustc queries parent hir node of `expr` here and determine whether
158        // the current `expr` is read of value per its parent.
159        // But since we don't have hir node, we cannot follow such "bottom-up"
160        // method.
161        // So, we pass down such readness from the parent expression through the
162        // recursive `infer_expr*` calls in a "top-down" manner.
163        is_read == ExprIsRead::Yes
164    }

166    /// Whether this pattern constitutes a read of value of the scrutinee that
167    /// it is matching against. This is used to determine whether we should
168    /// perform `NeverToAny` coercions.
169    fn pat_guaranteed_to_constitute_read_for_never(&self, pat: PatId) -> bool {
170        match &self.body[pat] {
171            // Does not constitute a read.
172            Pat::Wild => false,
173
174            // This is unnecessarily restrictive when the pattern that doesn't
175            // constitute a read is unreachable.
176            //
177            // For example `match *never_ptr { value => {}, _ => {} }` or
178            // `match *never_ptr { _ if false => {}, value => {} }`.
179            //
180            // It is however fine to be restrictive here; only returning `true`
181            // can lead to unsoundness.
182            Pat::Or(subpats) => {
183                subpats.iter().all(|pat| self.pat_guaranteed_to_constitute_read_for_never(*pat))
184            }
185
186            // All of these constitute a read, or match on something that isn't `!`,
187            // which would require a `NeverToAny` coercion.
188            Pat::Bind { .. }
189            | Pat::TupleStruct { .. }
190            | Pat::Path(_)
191            | Pat::Tuple { .. }
192            | Pat::Box { .. }
193            | Pat::Ref { .. }
194            | Pat::Lit(_)
195            | Pat::Range { .. }
196            | Pat::Slice { .. }
197            | Pat::ConstBlock(_)
198            | Pat::Record { .. }
199            | Pat::Missing => true,
200            Pat::Expr(_) => unreachable!(
201                "we don't call pat_guaranteed_to_constitute_read_for_never() with assignments"
202            ),
203        }
204    }

206    fn is_syntactic_place_expr(&self, expr: ExprId) -> bool {
207        match &self.body[expr] {
208            // Lang item paths cannot currently be local variables or statics.
209            Expr::Path(Path::LangItem(_, _)) => false,
210            Expr::Path(Path::Normal(path)) => path.type_anchor.is_none(),
211            Expr::Path(path) => self
212                .resolver
213                .resolve_path_in_value_ns_fully(self.db, path, self.body.expr_path_hygiene(expr))
214                .is_none_or(|res| matches!(res, ValueNs::LocalBinding(_) | ValueNs::StaticId(_))),
215            Expr::Underscore => true,
216            Expr::UnaryOp { op: UnaryOp::Deref, .. } => true,
217            Expr::Field { .. } | Expr::Index { .. } => true,
218            Expr::Call { .. }
219            | Expr::MethodCall { .. }
220            | Expr::Tuple { .. }
221            | Expr::If { .. }
222            | Expr::Match { .. }
223            | Expr::Closure { .. }
224            | Expr::Block { .. }
225            | Expr::Array(..)
226            | Expr::Break { .. }
227            | Expr::Continue { .. }
228            | Expr::Return { .. }
229            | Expr::Become { .. }
230            | Expr::Let { .. }
231            | Expr::Loop { .. }
232            | Expr::InlineAsm(..)
233            | Expr::OffsetOf(..)
234            | Expr::Literal(..)
235            | Expr::Const(..)
236            | Expr::UnaryOp { .. }
237            | Expr::BinaryOp { .. }
238            | Expr::Assignment { .. }
239            | Expr::Yield { .. }
240            | Expr::Cast { .. }
241            | Expr::Async { .. }
242            | Expr::Unsafe { .. }
243            | Expr::Await { .. }
244            | Expr::Ref { .. }
245            | Expr::Range { .. }
246            | Expr::Box { .. }
247            | Expr::RecordLit { .. }
248            | Expr::Yeet { .. }
249            | Expr::Missing => false,
250        }
251    }

253    fn infer_expr_coerce_never(
254        &mut self,
255        expr: ExprId,
256        expected: &Expectation<'db>,
257        is_read: ExprIsRead,
258    ) -> Ty<'db> {
259        let ty = self.infer_expr_inner(expr, expected, is_read);
260        // While we don't allow *arbitrary* coercions here, we *do* allow
261        // coercions from `!` to `expected`.
262        if ty.is_never() {
263            if let Some(adjustments) = self.result.expr_adjustments.get(&expr) {
264                return if let [Adjustment { kind: Adjust::NeverToAny, target }] = &**adjustments {
265                    *target
266                } else {
267                    self.err_ty()
268                };
269            }
270
271            if let Some(target) = expected.only_has_type(&mut self.table) {
272                self.coerce(expr.into(), ty, target, AllowTwoPhase::No, CoerceNever::Yes)
273                    .expect("never-to-any coercion should always succeed")
274            } else {
275                ty
276            }
277        } else {
278            if let Some(expected_ty) = expected.only_has_type(&mut self.table) {
279                let could_unify = self.unify(ty, expected_ty);
280                if !could_unify {
281                    self.result
282                        .type_mismatches
283                        .insert(expr.into(), TypeMismatch { expected: expected_ty, actual: ty });
284                }
285            }
286            ty
287        }
288    }

290    #[tracing::instrument(level = "debug", skip(self, is_read), ret)]
291    fn infer_expr_inner(
292        &mut self,
293        tgt_expr: ExprId,
294        expected: &Expectation<'db>,
295        is_read: ExprIsRead,
296    ) -> Ty<'db> {
297        self.db.unwind_if_revision_cancelled();
298
299        let expr = &self.body[tgt_expr];
300        tracing::trace!(?expr);
301        let ty = match expr {
302            Expr::Missing => self.err_ty(),
303            &Expr::If { condition, then_branch, else_branch } => {
304                let expected = &expected.adjust_for_branches(&mut self.table);
305                self.infer_expr_coerce_never(
306                    condition,
307                    &Expectation::HasType(self.types.bool),
308                    ExprIsRead::Yes,
309                );
310
311                let condition_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
312
313                let then_ty = self.infer_expr_inner(then_branch, expected, ExprIsRead::Yes);
314                let then_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
315                let mut coercion_sites = [then_branch, tgt_expr];
316                if let Some(else_branch) = else_branch {
317                    coercion_sites[1] = else_branch;
318                }
319                let mut coerce = CoerceMany::with_coercion_sites(
320                    expected.coercion_target_type(&mut self.table),
321                    &coercion_sites,
322                );
323                coerce.coerce(self, &ObligationCause::new(), then_branch, then_ty);
324                match else_branch {
325                    Some(else_branch) => {
326                        let else_ty = self.infer_expr_inner(else_branch, expected, ExprIsRead::Yes);
327                        let else_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
328                        coerce.coerce(self, &ObligationCause::new(), else_branch, else_ty);
329                        self.diverges = condition_diverges | then_diverges & else_diverges;
330                    }
331                    None => {
332                        coerce.coerce_forced_unit(self, tgt_expr, &ObligationCause::new(), true);
333                        self.diverges = condition_diverges;
334                    }
335                }
336
337                coerce.complete(self)
338            }
339            &Expr::Let { pat, expr } => {
340                let child_is_read = if self.pat_guaranteed_to_constitute_read_for_never(pat) {
341                    ExprIsRead::Yes
342                } else {
343                    ExprIsRead::No
344                };
345                let input_ty = self.infer_expr(expr, &Expectation::none(), child_is_read);
346                self.infer_top_pat(
347                    pat,
348                    input_ty,
349                    Some(DeclContext { origin: DeclOrigin::LetExpr }),
350                );
351                self.types.bool
352            }
353            Expr::Block { statements, tail, label, id } => {
354                self.infer_block(tgt_expr, *id, statements, *tail, *label, expected)
355            }
356            Expr::Unsafe { id, statements, tail } => {
357                self.infer_block(tgt_expr, *id, statements, *tail, None, expected)
358            }
359            Expr::Const(id) => {
360                self.with_breakable_ctx(BreakableKind::Border, None, None, |this| {
361                    this.infer_expr(*id, expected, ExprIsRead::Yes)
362                })
363                .1
364            }
365            Expr::Async { id, statements, tail } => {
366                self.infer_async_block(tgt_expr, id, statements, tail)
367            }
368            &Expr::Loop { body, label } => {
369                // FIXME: should be:
370                // let ty = expected.coercion_target_type(&mut self.table);
371                let ty = self.table.next_ty_var();
372                let (breaks, ()) =
373                    self.with_breakable_ctx(BreakableKind::Loop, Some(ty), label, |this| {
374                        this.infer_expr(
375                            body,
376                            &Expectation::HasType(this.types.unit),
377                            ExprIsRead::Yes,
378                        );
379                    });
380
381                match breaks {
382                    Some(breaks) => {
383                        self.diverges = Diverges::Maybe;
384                        breaks
385                    }
386                    None => self.types.never,
387                }
388            }
389            Expr::Closure { body, args, ret_type, arg_types, closure_kind, capture_by: _ } => self
390                .infer_closure(
391                    *body,
392                    args,
393                    *ret_type,
394                    arg_types,
395                    *closure_kind,
396                    tgt_expr,
397                    expected,
398                ),
399            Expr::Call { callee, args, .. } => self.infer_call(tgt_expr, *callee, args, expected),
400            Expr::MethodCall { receiver, args, method_name, generic_args } => self
401                .infer_method_call(
402                    tgt_expr,
403                    *receiver,
404                    args,
405                    method_name,
406                    generic_args.as_deref(),
407                    expected,
408                ),
409            Expr::Match { expr, arms } => {
410                let scrutinee_is_read = arms
411                    .iter()
412                    .all(|arm| self.pat_guaranteed_to_constitute_read_for_never(arm.pat));
413                let scrutinee_is_read =
414                    if scrutinee_is_read { ExprIsRead::Yes } else { ExprIsRead::No };
415                let input_ty = self.infer_expr(*expr, &Expectation::none(), scrutinee_is_read);
416
417                if arms.is_empty() {
418                    self.diverges = Diverges::Always;
419                    self.types.never
420                } else {
421                    let matchee_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
422                    let mut all_arms_diverge = Diverges::Always;
423                    for arm in arms.iter() {
424                        let input_ty = self.table.structurally_resolve_type(input_ty);
425                        self.infer_top_pat(arm.pat, input_ty, None);
426                    }
427
428                    let expected = expected.adjust_for_branches(&mut self.table);
429                    let result_ty = match &expected {
430                        // We don't coerce to `()` so that if the match expression is a
431                        // statement it's branches can have any consistent type.
432                        Expectation::HasType(ty) if *ty != self.types.unit => *ty,
433                        _ => self.table.next_ty_var(),
434                    };
435                    let mut coerce = CoerceMany::new(result_ty);
436
437                    for arm in arms.iter() {
438                        if let Some(guard_expr) = arm.guard {
439                            self.diverges = Diverges::Maybe;
440                            self.infer_expr_coerce_never(
441                                guard_expr,
442                                &Expectation::HasType(self.types.bool),
443                                ExprIsRead::Yes,
444                            );
445                        }
446                        self.diverges = Diverges::Maybe;
447
448                        let arm_ty = self.infer_expr_inner(arm.expr, &expected, ExprIsRead::Yes);
449                        all_arms_diverge &= self.diverges;
450                        coerce.coerce(self, &ObligationCause::new(), arm.expr, arm_ty);
451                    }
452
453                    self.diverges = matchee_diverges | all_arms_diverge;
454
455                    coerce.complete(self)
456                }
457            }
458            Expr::Path(p) => self.infer_expr_path(p, tgt_expr.into(), tgt_expr),
459            &Expr::Continue { label } => {
460                if find_continuable(&mut self.breakables, label).is_none() {
461                    self.push_diagnostic(InferenceDiagnostic::BreakOutsideOfLoop {
462                        expr: tgt_expr,
463                        is_break: false,
464                        bad_value_break: false,
465                    });
466                };
467                self.types.never
468            }
469            &Expr::Break { expr, label } => {
470                let val_ty = if let Some(expr) = expr {
471                    let opt_coerce_to = match find_breakable(&mut self.breakables, label) {
472                        Some(ctxt) => match &ctxt.coerce {
473                            Some(coerce) => coerce.expected_ty(),
474                            None => {
475                                self.push_diagnostic(InferenceDiagnostic::BreakOutsideOfLoop {
476                                    expr: tgt_expr,
477                                    is_break: true,
478                                    bad_value_break: true,
479                                });
480                                self.err_ty()
481                            }
482                        },
483                        None => self.err_ty(),
484                    };
485                    self.infer_expr_inner(
486                        expr,
487                        &Expectation::HasType(opt_coerce_to),
488                        ExprIsRead::Yes,
489                    )
490                } else {
491                    self.types.unit
492                };
493
494                match find_breakable(&mut self.breakables, label) {
495                    Some(ctxt) => match ctxt.coerce.take() {
496                        Some(mut coerce) => {
497                            coerce.coerce(
498                                self,
499                                &ObligationCause::new(),
500                                expr.unwrap_or(tgt_expr),
501                                val_ty,
502                            );
503
504                            // Avoiding borrowck
505                            let ctxt = find_breakable(&mut self.breakables, label)
506                                .expect("breakable stack changed during coercion");
507                            ctxt.may_break = true;
508                            ctxt.coerce = Some(coerce);
509                        }
510                        None => ctxt.may_break = true,
511                    },
512                    None => {
513                        self.push_diagnostic(InferenceDiagnostic::BreakOutsideOfLoop {
514                            expr: tgt_expr,
515                            is_break: true,
516                            bad_value_break: false,
517                        });
518                    }
519                }
520                self.types.never
521            }
522            &Expr::Return { expr } => self.infer_expr_return(tgt_expr, expr),
523            &Expr::Become { expr } => self.infer_expr_become(expr),
524            Expr::Yield { expr } => {
525                if let Some((resume_ty, yield_ty)) = self.resume_yield_tys {
526                    if let Some(expr) = expr {
527                        self.infer_expr_coerce(
528                            *expr,
529                            &Expectation::has_type(yield_ty),
530                            ExprIsRead::Yes,
531                        );
532                    } else {
533                        let unit = self.types.unit;
534                        let _ = self.coerce(
535                            tgt_expr.into(),
536                            unit,
537                            yield_ty,
538                            AllowTwoPhase::No,
539                            CoerceNever::Yes,
540                        );
541                    }
542                    resume_ty
543                } else {
544                    // FIXME: report error (yield expr in non-coroutine)
545                    self.types.error
546                }
547            }
548            Expr::Yeet { expr } => {
549                if let &Some(expr) = expr {
550                    self.infer_expr_no_expect(expr, ExprIsRead::Yes);
551                }
552                self.types.never
553            }
554            Expr::RecordLit { path, fields, spread, .. } => {
555                let (ty, def_id) = self.resolve_variant(tgt_expr.into(), path.as_deref(), false);
556
557                if let Some(t) = expected.only_has_type(&mut self.table) {
558                    self.unify(ty, t);
559                }
560
561                let substs = ty.as_adt().map(|(_, s)| s).unwrap_or(self.types.empty_args);
562                if let Some(variant) = def_id {
563                    self.write_variant_resolution(tgt_expr.into(), variant);
564                }
565                match def_id {
566                    _ if fields.is_empty() => {}
567                    Some(def) => {
568                        let field_types = self.db.field_types(def);
569                        let variant_data = def.fields(self.db);
570                        let visibilities = self.db.field_visibilities(def);
571                        for field in fields.iter() {
572                            let field_def = {
573                                match variant_data.field(&field.name) {
574                                    Some(local_id) => {
575                                        if !visibilities[local_id]
576                                            .is_visible_from(self.db, self.resolver.module())
577                                        {
578                                            self.push_diagnostic(
579                                                InferenceDiagnostic::NoSuchField {
580                                                    field: field.expr.into(),
581                                                    private: Some(local_id),
582                                                    variant: def,
583                                                },
584                                            );
585                                        }
586                                        Some(local_id)
587                                    }
588                                    None => {
589                                        self.push_diagnostic(InferenceDiagnostic::NoSuchField {
590                                            field: field.expr.into(),
591                                            private: None,
592                                            variant: def,
593                                        });
594                                        None
595                                    }
596                                }
597                            };
598                            let field_ty = field_def.map_or(self.err_ty(), |it| {
599                                field_types[it].instantiate(self.interner(), &substs)
600                            });
601
602                            // Field type might have some unknown types
603                            // FIXME: we may want to emit a single type variable for all instance of type fields?
604                            let field_ty = self.insert_type_vars(field_ty);
605                            self.infer_expr_coerce(
606                                field.expr,
607                                &Expectation::has_type(field_ty),
608                                ExprIsRead::Yes,
609                            );
610                        }
611                    }
612                    None => {
613                        for field in fields.iter() {
614                            // Field projections don't constitute reads.
615                            self.infer_expr_coerce(field.expr, &Expectation::None, ExprIsRead::No);
616                        }
617                    }
618                }
619                if let Some(expr) = spread {
620                    self.infer_expr(*expr, &Expectation::has_type(ty), ExprIsRead::Yes);
621                }
622                ty
623            }
624            Expr::Field { expr, name } => self.infer_field_access(tgt_expr, *expr, name, expected),
625            Expr::Await { expr } => {
626                let inner_ty = self.infer_expr_inner(*expr, &Expectation::none(), ExprIsRead::Yes);
627                self.resolve_associated_type(inner_ty, self.resolve_future_future_output())
628            }
629            Expr::Cast { expr, type_ref } => {
630                let cast_ty = self.make_body_ty(*type_ref);
631                let expr_ty =
632                    self.infer_expr(*expr, &Expectation::Castable(cast_ty), ExprIsRead::Yes);
633                self.deferred_cast_checks.push(CastCheck::new(tgt_expr, *expr, expr_ty, cast_ty));
634                cast_ty
635            }
636            Expr::Ref { expr, rawness, mutability } => {
637                let mutability = lower_mutability(*mutability);
638                let expectation = if let Some((exp_inner, exp_rawness, exp_mutability)) = expected
639                    .only_has_type(&mut self.table)
640                    .as_ref()
641                    .and_then(|t| t.as_reference_or_ptr())
642                {
643                    if exp_mutability == Mutability::Mut && mutability == Mutability::Not {
644                        // FIXME: record type error - expected mut reference but found shared ref,
645                        // which cannot be coerced
646                    }
647                    if exp_rawness == Rawness::Ref && *rawness == Rawness::RawPtr {
648                        // FIXME: record type error - expected reference but found ptr,
649                        // which cannot be coerced
650                    }
651                    Expectation::rvalue_hint(self, exp_inner)
652                } else {
653                    Expectation::none()
654                };
655                let inner_ty = self.infer_expr_inner(*expr, &expectation, ExprIsRead::Yes);
656                match rawness {
657                    Rawness::RawPtr => Ty::new_ptr(self.interner(), inner_ty, mutability),
658                    Rawness::Ref => {
659                        let lt = self.table.next_region_var();
660                        Ty::new_ref(self.interner(), lt, inner_ty, mutability)
661                    }
662                }
663            }
664            &Expr::Box { expr } => self.infer_expr_box(expr, expected),
665            Expr::UnaryOp { expr, op } => {
666                let inner_ty = self.infer_expr_inner(*expr, &Expectation::none(), ExprIsRead::Yes);
667                let inner_ty = self.table.try_structurally_resolve_type(inner_ty);
668                // FIXME: Note down method resolution her
669                match op {
670                    UnaryOp::Deref => {
671                        if let Some(deref_trait) = self.resolve_lang_trait(LangItem::Deref)
672                            && let Some(deref_fn) = deref_trait
673                                .trait_items(self.db)
674                                .method_by_name(&Name::new_symbol_root(sym::deref))
675                        {
676                            // FIXME: this is wrong in multiple ways, subst is empty, and we emit it even for builtin deref (note that
677                            // the mutability is not wrong, and will be fixed in `self.infer_mut`).
678                            self.write_method_resolution(tgt_expr, deref_fn, self.types.empty_args);
679                        }
680                        if let Some(derefed) = inner_ty.builtin_deref(self.db, true) {
681                            self.table.try_structurally_resolve_type(derefed)
682                        } else {
683                            let infer_ok = overloaded_deref_ty(&self.table, inner_ty);
684                            match infer_ok {
685                                Some(infer_ok) => self.table.register_infer_ok(infer_ok),
686                                None => self.err_ty(),
687                            }
688                        }
689                    }
690                    UnaryOp::Neg => {
691                        match inner_ty.kind() {
692                            // Fast path for builtins
693                            TyKind::Int(_)
694                            | TyKind::Uint(_)
695                            | TyKind::Float(_)
696                            | TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_)) => inner_ty,
697                            // Otherwise we resolve via the std::ops::Neg trait
698                            _ => self
699                                .resolve_associated_type(inner_ty, self.resolve_ops_neg_output()),
700                        }
701                    }
702                    UnaryOp::Not => {
703                        match inner_ty.kind() {
704                            // Fast path for builtins
705                            TyKind::Bool
706                            | TyKind::Int(_)
707                            | TyKind::Uint(_)
708                            | TyKind::Float(_)
709                            | TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_)) => inner_ty,
710                            // Otherwise we resolve via the std::ops::Not trait
711                            _ => self
712                                .resolve_associated_type(inner_ty, self.resolve_ops_not_output()),
713                        }
714                    }
715                }
716            }
717            Expr::BinaryOp { lhs, rhs, op } => match op {
718                Some(BinaryOp::LogicOp(_)) => {
719                    let bool_ty = self.types.bool;
720                    self.infer_expr_coerce(*lhs, &Expectation::HasType(bool_ty), ExprIsRead::Yes);
721                    let lhs_diverges = self.diverges;
722                    self.infer_expr_coerce(*rhs, &Expectation::HasType(bool_ty), ExprIsRead::Yes);
723                    // Depending on the LHS' value, the RHS can never execute.
724                    self.diverges = lhs_diverges;
725                    bool_ty
726                }
727                Some(op) => self.infer_overloadable_binop(*lhs, *op, *rhs, tgt_expr),
728                _ => self.err_ty(),
729            },
730            &Expr::Assignment { target, value } => {
731                // In ordinary (non-destructuring) assignments, the type of
732                // `lhs` must be inferred first so that the ADT fields
733                // instantiations in RHS can be coerced to it. Note that this
734                // cannot happen in destructuring assignments because of how
735                // they are desugared.
736                let lhs_ty = match &self.body[target] {
737                    // LHS of assignment doesn't constitute reads.
738                    &Pat::Expr(expr) => {
739                        Some(self.infer_expr(expr, &Expectation::none(), ExprIsRead::No))
740                    }
741                    Pat::Path(path) => {
742                        let resolver_guard =
743                            self.resolver.update_to_inner_scope(self.db, self.owner, tgt_expr);
744                        let resolution = self.resolver.resolve_path_in_value_ns_fully(
745                            self.db,
746                            path,
747                            self.body.pat_path_hygiene(target),
748                        );
749                        self.resolver.reset_to_guard(resolver_guard);
750
751                        if matches!(
752                            resolution,
753                            Some(
754                                ValueNs::ConstId(_)
755                                    | ValueNs::StructId(_)
756                                    | ValueNs::EnumVariantId(_)
757                            )
758                        ) {
759                            None
760                        } else {
761                            Some(self.infer_expr_path(path, target.into(), tgt_expr))
762                        }
763                    }
764                    _ => None,
765                };
766                let is_destructuring_assignment = lhs_ty.is_none();
767
768                if let Some(lhs_ty) = lhs_ty {
769                    self.write_pat_ty(target, lhs_ty);
770                    self.infer_expr_coerce(value, &Expectation::has_type(lhs_ty), ExprIsRead::No);
771                } else {
772                    let rhs_ty = self.infer_expr(value, &Expectation::none(), ExprIsRead::Yes);
773                    let resolver_guard =
774                        self.resolver.update_to_inner_scope(self.db, self.owner, tgt_expr);
775                    self.inside_assignment = true;
776                    self.infer_top_pat(target, rhs_ty, None);
777                    self.inside_assignment = false;
778                    self.resolver.reset_to_guard(resolver_guard);
779                }
780                if is_destructuring_assignment && self.diverges.is_always() {
781                    // Ordinary assignments always return `()`, even when they diverge.
782                    // However, rustc lowers destructuring assignments into blocks, and blocks return `!` if they have no tail
783                    // expression and they diverge. Therefore, we have to do the same here, even though we don't lower destructuring
784                    // assignments into blocks.
785                    self.table.new_maybe_never_var()
786                } else {
787                    self.types.unit
788                }
789            }
790            Expr::Range { lhs, rhs, range_type } => {
791                let lhs_ty =
792                    lhs.map(|e| self.infer_expr_inner(e, &Expectation::none(), ExprIsRead::Yes));
793                let rhs_expect =
794                    lhs_ty.as_ref().map_or_else(Expectation::none, |ty| Expectation::has_type(*ty));
795                let rhs_ty = rhs.map(|e| self.infer_expr(e, &rhs_expect, ExprIsRead::Yes));
796                let single_arg_adt = |adt, ty: Ty<'db>| {
797                    Ty::new_adt(
798                        self.interner(),
799                        adt,
800                        GenericArgs::new_from_iter(self.interner(), [ty.into()]),
801                    )
802                };
803                match (range_type, lhs_ty, rhs_ty) {
804                    (RangeOp::Exclusive, None, None) => match self.resolve_range_full() {
805                        Some(adt) => Ty::new_adt(self.interner(), adt, self.types.empty_args),
806                        None => self.err_ty(),
807                    },
808                    (RangeOp::Exclusive, None, Some(ty)) => match self.resolve_range_to() {
809                        Some(adt) => single_arg_adt(adt, ty),
810                        None => self.err_ty(),
811                    },
812                    (RangeOp::Inclusive, None, Some(ty)) => {
813                        match self.resolve_range_to_inclusive() {
814                            Some(adt) => single_arg_adt(adt, ty),
815                            None => self.err_ty(),
816                        }
817                    }
818                    (RangeOp::Exclusive, Some(_), Some(ty)) => match self.resolve_range() {
819                        Some(adt) => single_arg_adt(adt, ty),
820                        None => self.err_ty(),
821                    },
822                    (RangeOp::Inclusive, Some(_), Some(ty)) => {
823                        match self.resolve_range_inclusive() {
824                            Some(adt) => single_arg_adt(adt, ty),
825                            None => self.err_ty(),
826                        }
827                    }
828                    (RangeOp::Exclusive, Some(ty), None) => match self.resolve_range_from() {
829                        Some(adt) => single_arg_adt(adt, ty),
830                        None => self.err_ty(),
831                    },
832                    (RangeOp::Inclusive, _, None) => self.err_ty(),
833                }
834            }
835            Expr::Index { base, index } => {
836                let base_ty = self.infer_expr_inner(*base, &Expectation::none(), ExprIsRead::Yes);
837                let index_ty = self.infer_expr(*index, &Expectation::none(), ExprIsRead::Yes);
838
839                if let Some(index_trait) = self.resolve_lang_trait(LangItem::Index) {
840                    let canonicalized = self.canonicalize(base_ty);
841                    let receiver_adjustments = method_resolution::resolve_indexing_op(
842                        &mut self.table,
843                        canonicalized,
844                        index_trait,
845                    );
846                    let (self_ty, mut adj) = receiver_adjustments
847                        .map_or((self.err_ty(), Vec::new()), |adj| {
848                            adj.apply(&mut self.table, base_ty)
849                        });
850
851                    // mutability will be fixed up in `InferenceContext::infer_mut`;
852                    adj.push(Adjustment::borrow(
853                        self.interner(),
854                        Mutability::Not,
855                        self_ty,
856                        self.table.next_region_var(),
857                    ));
858                    self.write_expr_adj(*base, adj.into_boxed_slice());
859                    if let Some(func) = index_trait
860                        .trait_items(self.db)
861                        .method_by_name(&Name::new_symbol_root(sym::index))
862                    {
863                        let subst = GenericArgs::new_from_iter(
864                            self.interner(),
865                            [self_ty.into(), index_ty.into()],
866                        );
867                        self.write_method_resolution(tgt_expr, func, subst);
868                    }
869                    let assoc = self.resolve_ops_index_output();
870                    self.resolve_associated_type_with_params(self_ty, assoc, &[index_ty.into()])
871                } else {
872                    self.err_ty()
873                }
874            }
875            Expr::Tuple { exprs, .. } => {
876                let mut tys =
877                    match expected.only_has_type(&mut self.table).as_ref().map(|t| t.kind()) {
878                        Some(TyKind::Tuple(substs)) => substs
879                            .iter()
880                            .chain(repeat_with(|| self.table.next_ty_var()))
881                            .take(exprs.len())
882                            .collect::<Vec<_>>(),
883                        _ => (0..exprs.len()).map(|_| self.table.next_ty_var()).collect(),
884                    };
885
886                for (expr, ty) in exprs.iter().zip(tys.iter_mut()) {
887                    *ty =
888                        self.infer_expr_coerce(*expr, &Expectation::has_type(*ty), ExprIsRead::Yes);
889                }
890
891                Ty::new_tup(self.interner(), &tys)
892            }
893            Expr::Array(array) => self.infer_expr_array(array, expected),
894            Expr::Literal(lit) => match lit {
895                Literal::Bool(..) => self.types.bool,
896                Literal::String(..) => self.types.static_str_ref,
897                Literal::ByteString(bs) => {
898                    let byte_type = self.types.u8;
899
900                    let len = consteval::usize_const(
901                        self.db,
902                        Some(bs.len() as u128),
903                        self.resolver.krate(),
904                    );
905
906                    let array_type = Ty::new_array_with_const_len(self.interner(), byte_type, len);
907                    Ty::new_ref(self.interner(), self.types.re_static, array_type, Mutability::Not)
908                }
909                Literal::CString(..) => Ty::new_ref(
910                    self.interner(),
911                    self.types.re_static,
912                    self.resolve_lang_item(LangItem::CStr)
913                        .and_then(LangItemTarget::as_struct)
914                        .map_or_else(
915                            || self.err_ty(),
916                            |strukt| {
917                                Ty::new_adt(self.interner(), strukt.into(), self.types.empty_args)
918                            },
919                        ),
920                    Mutability::Not,
921                ),
922                Literal::Char(..) => self.types.char,
923                Literal::Int(_v, ty) => match ty {
924                    Some(int_ty) => match int_ty {
925                        hir_def::builtin_type::BuiltinInt::Isize => self.types.isize,
926                        hir_def::builtin_type::BuiltinInt::I8 => self.types.i8,
927                        hir_def::builtin_type::BuiltinInt::I16 => self.types.i16,
928                        hir_def::builtin_type::BuiltinInt::I32 => self.types.i32,
929                        hir_def::builtin_type::BuiltinInt::I64 => self.types.i64,
930                        hir_def::builtin_type::BuiltinInt::I128 => self.types.i128,
931                    },
932                    None => {
933                        let expected_ty = expected.to_option(&mut self.table);
934                        tracing::debug!(?expected_ty);
935                        let opt_ty = match expected_ty.as_ref().map(|it| it.kind()) {
936                            Some(TyKind::Int(_) | TyKind::Uint(_)) => expected_ty,
937                            Some(TyKind::Char) => Some(self.types.u8),
938                            Some(TyKind::RawPtr(..) | TyKind::FnDef(..) | TyKind::FnPtr(..)) => {
939                                Some(self.types.usize)
940                            }
941                            _ => None,
942                        };
943                        opt_ty.unwrap_or_else(|| self.table.next_int_var())
944                    }
945                },
946                Literal::Uint(_v, ty) => match ty {
947                    Some(int_ty) => match int_ty {
948                        hir_def::builtin_type::BuiltinUint::Usize => self.types.usize,
949                        hir_def::builtin_type::BuiltinUint::U8 => self.types.u8,
950                        hir_def::builtin_type::BuiltinUint::U16 => self.types.u16,
951                        hir_def::builtin_type::BuiltinUint::U32 => self.types.u32,
952                        hir_def::builtin_type::BuiltinUint::U64 => self.types.u64,
953                        hir_def::builtin_type::BuiltinUint::U128 => self.types.u128,
954                    },
955                    None => {
956                        let expected_ty = expected.to_option(&mut self.table);
957                        let opt_ty = match expected_ty.as_ref().map(|it| it.kind()) {
958                            Some(TyKind::Int(_) | TyKind::Uint(_)) => expected_ty,
959                            Some(TyKind::Char) => Some(self.types.u8),
960                            Some(TyKind::RawPtr(..) | TyKind::FnDef(..) | TyKind::FnPtr(..)) => {
961                                Some(self.types.usize)
962                            }
963                            _ => None,
964                        };
965                        opt_ty.unwrap_or_else(|| self.table.next_int_var())
966                    }
967                },
968                Literal::Float(_v, ty) => match ty {
969                    Some(float_ty) => match float_ty {
970                        hir_def::builtin_type::BuiltinFloat::F16 => self.types.f16,
971                        hir_def::builtin_type::BuiltinFloat::F32 => self.types.f32,
972                        hir_def::builtin_type::BuiltinFloat::F64 => self.types.f64,
973                        hir_def::builtin_type::BuiltinFloat::F128 => self.types.f128,
974                    },
975                    None => {
976                        let opt_ty = expected
977                            .to_option(&mut self.table)
978                            .filter(|ty| matches!(ty.kind(), TyKind::Float(_)));
979                        opt_ty.unwrap_or_else(|| self.table.next_float_var())
980                    }
981                },
982            },
983            Expr::Underscore => {
984                // Underscore expression is an error, we render a specialized diagnostic
985                // to let the user know what type is expected though.
986                let expected = expected.to_option(&mut self.table).unwrap_or_else(|| self.err_ty());
987                self.push_diagnostic(InferenceDiagnostic::TypedHole { expr: tgt_expr, expected });
988                expected
989            }
990            Expr::OffsetOf(_) => self.types.usize,
991            Expr::InlineAsm(asm) => {
992                let check_expr_asm_operand = |this: &mut Self, expr, is_input: bool| {
993                    let ty = this.infer_expr_no_expect(expr, ExprIsRead::Yes);
994
995                    // If this is an input value, we require its type to be fully resolved
996                    // at this point. This allows us to provide helpful coercions which help
997                    // pass the type candidate list in a later pass.
998                    //
999                    // We don't require output types to be resolved at this point, which
1000                    // allows them to be inferred based on how they are used later in the
1001                    // function.
1002                    if is_input {
1003                        let ty = this.table.structurally_resolve_type(ty);
1004                        match ty.kind() {
1005                            TyKind::FnDef(def, parameters) => {
1006                                let fnptr_ty = Ty::new_fn_ptr(
1007                                    this.interner(),
1008                                    this.interner()
1009                                        .fn_sig(def)
1010                                        .instantiate(this.interner(), parameters),
1011                                );
1012                                _ = this.coerce(
1013                                    expr.into(),
1014                                    ty,
1015                                    fnptr_ty,
1016                                    AllowTwoPhase::No,
1017                                    CoerceNever::Yes,
1018                                );
1019                            }
1020                            TyKind::Ref(_, base_ty, mutbl) => {
1021                                let ptr_ty = Ty::new_ptr(this.interner(), base_ty, mutbl);
1022                                _ = this.coerce(
1023                                    expr.into(),
1024                                    ty,
1025                                    ptr_ty,
1026                                    AllowTwoPhase::No,
1027                                    CoerceNever::Yes,
1028                                );
1029                            }
1030                            _ => {}
1031                        }
1032                    }
1033                };
1034
1035                let diverge = asm.options.contains(AsmOptions::NORETURN);
1036                asm.operands.iter().for_each(|(_, operand)| match *operand {
1037                    AsmOperand::In { expr, .. } => check_expr_asm_operand(self, expr, true),
1038                    AsmOperand::Out { expr: Some(expr), .. } | AsmOperand::InOut { expr, .. } => {
1039                        check_expr_asm_operand(self, expr, false)
1040                    }
1041                    AsmOperand::Out { expr: None, .. } => (),
1042                    AsmOperand::SplitInOut { in_expr, out_expr, .. } => {
1043                        check_expr_asm_operand(self, in_expr, true);
1044                        if let Some(out_expr) = out_expr {
1045                            check_expr_asm_operand(self, out_expr, false);
1046                        }
1047                    }
1048                    AsmOperand::Label(expr) => {
1049                        self.infer_expr(
1050                            expr,
1051                            &Expectation::HasType(self.types.unit),
1052                            ExprIsRead::No,
1053                        );
1054                    }
1055                    AsmOperand::Const(expr) => {
1056                        self.infer_expr(expr, &Expectation::None, ExprIsRead::No);
1057                    }
1058                    // FIXME: `sym` should report for things that are not functions or statics.
1059                    AsmOperand::Sym(_) => (),
1060                });
1061                if diverge { self.types.never } else { self.types.unit }
1062            }
1063        };
1064        // use a new type variable if we got unknown here
1065        let ty = self.insert_type_vars_shallow(ty);
1066        self.write_expr_ty(tgt_expr, ty);
1067        if self.shallow_resolve(ty).is_never()
1068            && self.expr_guaranteed_to_constitute_read_for_never(tgt_expr, is_read)
1069        {
1070            // Any expression that produces a value of type `!` must have diverged
1071            self.diverges = Diverges::Always;
1072        }
1073        ty
1074    }

1076    fn infer_expr_path(&mut self, path: &Path, id: ExprOrPatId, scope_id: ExprId) -> Ty<'db> {
1077        let g = self.resolver.update_to_inner_scope(self.db, self.owner, scope_id);
1078        let ty = match self.infer_path(path, id) {
1079            Some(ty) => ty,
1080            None => {
1081                if path.mod_path().is_some_and(|mod_path| mod_path.is_ident() || mod_path.is_self())
1082                {
1083                    self.push_diagnostic(InferenceDiagnostic::UnresolvedIdent { id });
1084                }
1085                self.err_ty()
1086            }
1087        };
1088        self.resolver.reset_to_guard(g);
1089        ty
1090    }

1092    fn infer_async_block(
1093        &mut self,
1094        tgt_expr: ExprId,
1095        id: &Option<BlockId>,
1096        statements: &[Statement],
1097        tail: &Option<ExprId>,
1098    ) -> Ty<'db> {
1099        let ret_ty = self.table.next_ty_var();
1100        let prev_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
1101        let prev_ret_ty = mem::replace(&mut self.return_ty, ret_ty);
1102        let prev_ret_coercion = self.return_coercion.replace(CoerceMany::new(ret_ty));
1103
1104        // FIXME: We should handle async blocks like we handle closures
1105        let expected = &Expectation::has_type(ret_ty);
1106        let (_, inner_ty) = self.with_breakable_ctx(BreakableKind::Border, None, None, |this| {
1107            let ty = this.infer_block(tgt_expr, *id, statements, *tail, None, expected);
1108            if let Some(target) = expected.only_has_type(&mut this.table) {
1109                match this.coerce(tgt_expr.into(), ty, target, AllowTwoPhase::No, CoerceNever::Yes)
1110                {
1111                    Ok(res) => res,
1112                    Err(_) => {
1113                        this.result
1114                            .type_mismatches
1115                            .insert(tgt_expr.into(), TypeMismatch { expected: target, actual: ty });
1116                        target
1117                    }
1118                }
1119            } else {
1120                ty
1121            }
1122        });
1123
1124        self.diverges = prev_diverges;
1125        self.return_ty = prev_ret_ty;
1126        self.return_coercion = prev_ret_coercion;
1127
1128        self.lower_async_block_type_impl_trait(inner_ty, tgt_expr)
1129    }

1131    pub(crate) fn lower_async_block_type_impl_trait(
1132        &mut self,
1133        inner_ty: Ty<'db>,
1134        tgt_expr: ExprId,
1135    ) -> Ty<'db> {
1136        let coroutine_id = InternedCoroutine(self.owner, tgt_expr);
1137        let coroutine_id = self.db.intern_coroutine(coroutine_id).into();
1138        let parent_args = GenericArgs::identity_for_item(self.interner(), self.generic_def.into());
1139        Ty::new_coroutine(
1140            self.interner(),
1141            coroutine_id,
1142            CoroutineArgs::new(
1143                self.interner(),
1144                CoroutineArgsParts {
1145                    parent_args,
1146                    kind_ty: self.types.unit,
1147                    // rustc uses a special lang item type for the resume ty. I don't believe this can cause us problems.
1148                    resume_ty: self.types.unit,
1149                    yield_ty: self.types.unit,
1150                    return_ty: inner_ty,
1151                    // FIXME: Infer upvars.
1152                    tupled_upvars_ty: self.types.unit,
1153                },
1154            )
1155            .args,
1156        )
1157    }

1163        adjustments: &mut Vec<Adjustment<'db>>,
1164        callee_ty: Ty<'db>,
1165        params: &[Ty<'db>],
1166        tgt_expr: ExprId,
1167    ) {
1168        match fn_x {
1169            FnTrait::FnOnce | FnTrait::AsyncFnOnce => (),
1170            FnTrait::FnMut | FnTrait::AsyncFnMut => {
1171                if let TyKind::Ref(lt, inner, Mutability::Mut) = derefed_callee.kind() {
1172                    if adjustments
1173                        .last()
1174                        .map(|it| matches!(it.kind, Adjust::Borrow(_)))
1175                        .unwrap_or(true)
1176                    {
1177                        // prefer reborrow to move
1178                        adjustments.push(Adjustment { kind: Adjust::Deref(None), target: inner });
1179                        adjustments.push(Adjustment::borrow(
1180                            self.interner(),
1181                            Mutability::Mut,
1182                            inner,
1183                            lt,
1184                        ))
1185                    }
1186                } else {
1187                    adjustments.push(Adjustment::borrow(
1188                        self.interner(),
1189                        Mutability::Mut,
1190                        derefed_callee,
1191                        self.table.next_region_var(),
1192                    ));
1193                }
1194            }
1195            FnTrait::Fn | FnTrait::AsyncFn => {
1196                if !matches!(derefed_callee.kind(), TyKind::Ref(_, _, Mutability::Not)) {
1197                    adjustments.push(Adjustment::borrow(
1198                        self.interner(),
1199                        Mutability::Not,
1200                        derefed_callee,
1201                        self.table.next_region_var(),
1202                    ));
1203                }
1204            }
1205        }
1206        let Some(trait_) = fn_x.get_id(self.db, self.table.trait_env.krate) else {
1207            return;
1208        };
1209        let trait_data = trait_.trait_items(self.db);
1210        if let Some(func) = trait_data.method_by_name(&fn_x.method_name()) {
1211            let subst = GenericArgs::new_from_iter(
1212                self.interner(),
1213                [
1214                    callee_ty.into(),
1215                    Ty::new_tup_from_iter(self.interner(), params.iter().copied()).into(),
1216                ],
1217            );
1218            self.write_method_resolution(tgt_expr, func, subst);
1219        }
1220    }

1222    fn infer_expr_array(&mut self, array: &Array, expected: &Expectation<'db>) -> Ty<'db> {
1223        let elem_ty = match expected.to_option(&mut self.table).as_ref().map(|t| t.kind()) {
1224            Some(TyKind::Array(st, _) | TyKind::Slice(st)) => st,
1225            _ => self.table.next_ty_var(),
1226        };
1227
1228        let krate = self.resolver.krate();
1229
1230        let expected = Expectation::has_type(elem_ty);
1231        let (elem_ty, len) = match array {
1232            Array::ElementList { elements, .. } if elements.is_empty() => {
1233                (elem_ty, consteval::usize_const(self.db, Some(0), krate))
1234            }
1235            Array::ElementList { elements, .. } => {
1236                let mut coerce = CoerceMany::with_coercion_sites(elem_ty, elements);
1237                for &expr in elements.iter() {
1238                    let cur_elem_ty = self.infer_expr_inner(expr, &expected, ExprIsRead::Yes);
1239                    coerce.coerce(self, &ObligationCause::new(), expr, cur_elem_ty);
1240                }
1241                (
1242                    coerce.complete(self),
1243                    consteval::usize_const(self.db, Some(elements.len() as u128), krate),
1244                )
1245            }
1246            &Array::Repeat { initializer, repeat } => {
1247                self.infer_expr_coerce(
1248                    initializer,
1249                    &Expectation::has_type(elem_ty),
1250                    ExprIsRead::Yes,
1251                );
1252                let usize = self.types.usize;
1253                match self.body[repeat] {
1254                    Expr::Underscore => {
1255                        self.write_expr_ty(repeat, usize);
1256                    }
1257                    _ => _ = self.infer_expr(repeat, &Expectation::HasType(usize), ExprIsRead::Yes),
1258                }
1259
1260                (elem_ty, consteval::eval_to_const(repeat, self))
1261            }
1262        };
1263        // Try to evaluate unevaluated constant, and insert variable if is not possible.
1264        let len = self.table.insert_const_vars_shallow(len);
1265        Ty::new_array_with_const_len(self.interner(), elem_ty, len)
1266    }

1353    fn infer_overloadable_binop(
1354        &mut self,
1355        lhs: ExprId,
1356        op: BinaryOp,
1357        rhs: ExprId,
1358        tgt_expr: ExprId,
1359    ) -> Ty<'db> {
1360        let lhs_expectation = Expectation::none();
1361        let is_read = if matches!(op, BinaryOp::Assignment { .. }) {
1362            ExprIsRead::Yes
1363        } else {
1364            ExprIsRead::No
1365        };
1366        let lhs_ty = self.infer_expr(lhs, &lhs_expectation, is_read);
1367        let rhs_ty = self.table.next_ty_var();
1368
1369        let trait_func = lang_items_for_bin_op(op).and_then(|(name, lang_item)| {
1370            let trait_id = self.resolve_lang_item(lang_item)?.as_trait()?;
1371            let func = trait_id.trait_items(self.db).method_by_name(&name)?;
1372            Some((trait_id, func))
1373        });
1374        let func = match trait_func {
1375            Some((_, it)) => it,
1376            None => {
1377                // HACK: `rhs_ty` is a general inference variable with no clue at all at this
1378                // point. Passing `lhs_ty` as both operands just to check if `lhs_ty` is a builtin
1379                // type applicable to `op`.
1380                let ret_ty = if self.is_builtin_binop(lhs_ty, lhs_ty, op) {
1381                    // Assume both operands are builtin so we can continue inference. No guarantee
1382                    // on the correctness, rustc would complain as necessary lang items don't seem
1383                    // to exist anyway.
1384                    self.enforce_builtin_binop_types(lhs_ty, rhs_ty, op)
1385                } else {
1386                    self.err_ty()
1387                };
1388
1389                self.infer_expr_coerce(rhs, &Expectation::has_type(rhs_ty), ExprIsRead::Yes);
1390
1391                return ret_ty;
1392            }
1393        };
1394
1395        // HACK: We can use this substitution for the function because the function itself doesn't
1396        // have its own generic parameters.
1397        let args = GenericArgs::new_from_iter(self.interner(), [lhs_ty.into(), rhs_ty.into()]);
1398
1399        self.write_method_resolution(tgt_expr, func, args);
1400
1401        let method_ty = self.db.value_ty(func.into()).unwrap().instantiate(self.interner(), args);
1402        self.register_obligations_for_call(method_ty);
1403
1404        self.infer_expr_coerce(rhs, &Expectation::has_type(rhs_ty), ExprIsRead::Yes);
1405
1406        let ret_ty = match method_ty.callable_sig(self.interner()) {
1407            Some(sig) => {
1408                let sig = sig.skip_binder();
1409                let p_left = sig.inputs_and_output.as_slice()[0];
1410                if matches!(op, BinaryOp::CmpOp(..) | BinaryOp::Assignment { .. })
1411                    && let TyKind::Ref(lt, _, mtbl) = p_left.kind()
1412                {
1413                    self.write_expr_adj(
1414                        lhs,
1415                        Box::new([Adjustment {
1416                            kind: Adjust::Borrow(AutoBorrow::Ref(lt, mtbl)),
1417                            target: p_left,
1418                        }]),
1419                    );
1420                }
1421                let p_right = sig.inputs_and_output.as_slice()[1];
1422                if matches!(op, BinaryOp::CmpOp(..))
1423                    && let TyKind::Ref(lt, _, mtbl) = p_right.kind()
1424                {
1425                    self.write_expr_adj(
1426                        rhs,
1427                        Box::new([Adjustment {
1428                            kind: Adjust::Borrow(AutoBorrow::Ref(lt, mtbl)),
1429                            target: p_right,
1430                        }]),
1431                    );
1432                }
1433                sig.output()
1434            }
1435            None => self.err_ty(),
1436        };
1437
1438        let ret_ty = self.process_remote_user_written_ty(ret_ty);
1439
1440        if self.is_builtin_binop(lhs_ty, rhs_ty, op) {
1441            // use knowledge of built-in binary ops, which can sometimes help inference
1442            let builtin_ret = self.enforce_builtin_binop_types(lhs_ty, rhs_ty, op);
1443            self.unify(builtin_ret, ret_ty);
1444            builtin_ret
1445        } else {
1446            ret_ty
1447        }
1448    }

1450    fn infer_block(
1451        &mut self,
1452        expr: ExprId,
1453        block_id: Option<BlockId>,
1454        statements: &[Statement],
1455        tail: Option<ExprId>,
1456        label: Option<LabelId>,
1457        expected: &Expectation<'db>,
1458    ) -> Ty<'db> {
1459        let coerce_ty = expected.coercion_target_type(&mut self.table);
1460        let g = self.resolver.update_to_inner_scope(self.db, self.owner, expr);
1461        let prev_state = block_id.map(|block_id| {
1462            let prev_env = self.table.trait_env.clone();
1463            TraitEnvironment::with_block(&mut self.table.trait_env, block_id);
1464            let prev_block = self.table.infer_ctxt.interner.block.replace(block_id);
1465            (prev_env, prev_block)
1466        });
1467
1468        let (break_ty, ty) =
1469            self.with_breakable_ctx(BreakableKind::Block, Some(coerce_ty), label, |this| {
1470                for stmt in statements {
1471                    match stmt {
1472                        Statement::Let { pat, type_ref, initializer, else_branch } => {
1473                            let decl_ty = type_ref
1474                                .as_ref()
1475                                .map(|&tr| this.make_body_ty(tr))
1476                                .unwrap_or_else(|| this.table.next_ty_var());
1477
1478                            let ty = if let Some(expr) = initializer {
1479                                // If we have a subpattern that performs a read, we want to consider this
1480                                // to diverge for compatibility to support something like `let x: () = *never_ptr;`.
1481                                let target_is_read =
1482                                    if this.pat_guaranteed_to_constitute_read_for_never(*pat) {
1483                                        ExprIsRead::Yes
1484                                    } else {
1485                                        ExprIsRead::No
1486                                    };
1487                                let ty = if contains_explicit_ref_binding(this.body, *pat) {
1488                                    this.infer_expr(
1489                                        *expr,
1490                                        &Expectation::has_type(decl_ty),
1491                                        target_is_read,
1492                                    )
1493                                } else {
1494                                    this.infer_expr_coerce(
1495                                        *expr,
1496                                        &Expectation::has_type(decl_ty),
1497                                        target_is_read,
1498                                    )
1499                                };
1500                                if type_ref.is_some() { decl_ty } else { ty }
1501                            } else {
1502                                decl_ty
1503                            };
1504
1505                            let decl = DeclContext {
1506                                origin: DeclOrigin::LocalDecl { has_else: else_branch.is_some() },
1507                            };
1508
1509                            this.infer_top_pat(*pat, ty, Some(decl));
1510                            if let Some(expr) = else_branch {
1511                                let previous_diverges =
1512                                    mem::replace(&mut this.diverges, Diverges::Maybe);
1513                                this.infer_expr_coerce(
1514                                    *expr,
1515                                    &Expectation::HasType(this.types.never),
1516                                    ExprIsRead::Yes,
1517                                );
1518                                this.diverges = previous_diverges;
1519                            }
1520                        }
1521                        &Statement::Expr { expr, has_semi } => {
1522                            if has_semi {
1523                                this.infer_expr(expr, &Expectation::none(), ExprIsRead::Yes);
1524                            } else {
1525                                this.infer_expr_coerce(
1526                                    expr,
1527                                    &Expectation::HasType(this.types.unit),
1528                                    ExprIsRead::Yes,
1529                                );
1530                            }
1531                        }
1532                        Statement::Item(_) => (),
1533                    }
1534                }
1535
1536                // FIXME: This should make use of the breakable CoerceMany
1537                if let Some(expr) = tail {
1538                    this.infer_expr_coerce(expr, expected, ExprIsRead::Yes)
1539                } else {
1540                    // Citing rustc: if there is no explicit tail expression,
1541                    // that is typically equivalent to a tail expression
1542                    // of `()` -- except if the block diverges. In that
1543                    // case, there is no value supplied from the tail
1544                    // expression (assuming there are no other breaks,
1545                    // this implies that the type of the block will be
1546                    // `!`).
1547                    if this.diverges.is_always() {
1548                        // we don't even make an attempt at coercion
1549                        this.table.new_maybe_never_var()
1550                    } else if let Some(t) = expected.only_has_type(&mut this.table) {
1551                        let coerce_never = if this
1552                            .expr_guaranteed_to_constitute_read_for_never(expr, ExprIsRead::Yes)
1553                        {
1554                            CoerceNever::Yes
1555                        } else {
1556                            CoerceNever::No
1557                        };
1558                        if this
1559                            .coerce(
1560                                expr.into(),
1561                                this.types.unit,
1562                                t,
1563                                AllowTwoPhase::No,
1564                                coerce_never,
1565                            )
1566                            .is_err()
1567                        {
1568                            this.result.type_mismatches.insert(
1569                                expr.into(),
1570                                TypeMismatch { expected: t, actual: this.types.unit },
1571                            );
1572                        }
1573                        t
1574                    } else {
1575                        this.types.unit
1576                    }
1577                }
1578            });
1579        self.resolver.reset_to_guard(g);
1580        if let Some((prev_env, prev_block)) = prev_state {
1581            self.table.trait_env = prev_env;
1582            self.table.infer_ctxt.interner.block = prev_block;
1583        }
1584
1585        break_ty.unwrap_or(ty)
1586    }

1588    fn lookup_field(
1589        &mut self,
1590        receiver_ty: Ty<'db>,
1591        name: &Name,
1592    ) -> Option<(Ty<'db>, Either<FieldId, TupleFieldId>, Vec<Adjustment<'db>>, bool)> {
1593        let interner = self.interner();
1594        let mut autoderef = self.table.autoderef(receiver_ty);
1595        let mut private_field = None;
1596        let res = autoderef.by_ref().find_map(|(derefed_ty, _)| {
1597            let (field_id, parameters) = match derefed_ty.kind() {
1598                TyKind::Tuple(substs) => {
1599                    return name.as_tuple_index().and_then(|idx| {
1600                        substs.as_slice().get(idx).copied().map(|ty| {
1601                            (
1602                                Either::Right(TupleFieldId {
1603                                    tuple: TupleId(
1604                                        self.tuple_field_accesses_rev.insert_full(substs).0 as u32,
1605                                    ),
1606                                    index: idx as u32,
1607                                }),
1608                                ty,
1609                            )
1610                        })
1611                    });
1612                }
1613                TyKind::Adt(adt, parameters) => match adt.def_id().0 {
1614                    hir_def::AdtId::StructId(s) => {
1615                        let local_id = s.fields(self.db).field(name)?;
1616                        let field = FieldId { parent: s.into(), local_id };
1617                        (field, parameters)
1618                    }
1619                    hir_def::AdtId::UnionId(u) => {
1620                        let local_id = u.fields(self.db).field(name)?;
1621                        let field = FieldId { parent: u.into(), local_id };
1622                        (field, parameters)
1623                    }
1624                    hir_def::AdtId::EnumId(_) => return None,
1625                },
1626                _ => return None,
1627            };
1628            let is_visible = self.db.field_visibilities(field_id.parent)[field_id.local_id]
1629                .is_visible_from(self.db, self.resolver.module());
1630            if !is_visible {
1631                if private_field.is_none() {
1632                    private_field = Some((field_id, parameters));
1633                }
1634                return None;
1635            }
1636            let ty = self.db.field_types(field_id.parent)[field_id.local_id]
1637                .instantiate(interner, parameters);
1638            Some((Either::Left(field_id), ty))
1639        });
1640
1641        Some(match res {
1642            Some((field_id, ty)) => {
1643                let adjustments = autoderef.adjust_steps();
1644                let ty = self.process_remote_user_written_ty(ty);
1645
1646                (ty, field_id, adjustments, true)
1647            }
1648            None => {
1649                let (field_id, subst) = private_field?;
1650                let adjustments = autoderef.adjust_steps();
1651                let ty = self.db.field_types(field_id.parent)[field_id.local_id]
1652                    .instantiate(self.interner(), subst);
1653                let ty = self.process_remote_user_written_ty(ty);
1654
1655                (ty, Either::Left(field_id), adjustments, false)
1656            }
1657        })
1658    }

1660    fn infer_field_access(
1661        &mut self,
1662        tgt_expr: ExprId,
1663        receiver: ExprId,
1664        name: &Name,
1665        expected: &Expectation<'db>,
1666    ) -> Ty<'db> {
1667        // Field projections don't constitute reads.
1668        let receiver_ty = self.infer_expr_inner(receiver, &Expectation::none(), ExprIsRead::No);
1669
1670        if name.is_missing() {
1671            // Bail out early, don't even try to look up field. Also, we don't issue an unresolved
1672            // field diagnostic because this is a syntax error rather than a semantic error.
1673            return self.err_ty();
1674        }
1675
1676        match self.lookup_field(receiver_ty, name) {
1677            Some((ty, field_id, adjustments, is_public)) => {
1678                self.write_expr_adj(receiver, adjustments.into_boxed_slice());
1679                self.result.field_resolutions.insert(tgt_expr, field_id);
1680                if !is_public && let Either::Left(field) = field_id {
1681                    // FIXME: Merge this diagnostic into UnresolvedField?
1682                    self.push_diagnostic(InferenceDiagnostic::PrivateField {
1683                        expr: tgt_expr,
1684                        field,
1685                    });
1686                }
1687                ty
1688            }
1689            None => {
1690                // no field found, lets attempt to resolve it like a function so that IDE things
1691                // work out while people are typing
1692                let canonicalized_receiver = self.canonicalize(receiver_ty);
1693                let resolved = method_resolution::lookup_method(
1694                    &canonicalized_receiver,
1695                    &mut self.table,
1696                    Self::get_traits_in_scope(&self.resolver, &self.traits_in_scope)
1697                        .as_ref()
1698                        .left_or_else(|&it| it),
1699                    VisibleFromModule::Filter(self.resolver.module()),
1700                    name,
1701                );
1702                self.push_diagnostic(InferenceDiagnostic::UnresolvedField {
1703                    expr: tgt_expr,
1704                    receiver: receiver_ty,
1705                    name: name.clone(),
1706                    method_with_same_name_exists: resolved.is_some(),
1707                });
1708                match resolved {
1709                    Some((adjust, func, _)) => {
1710                        let (ty, adjustments) = adjust.apply(&mut self.table, receiver_ty);
1711                        let args = self.substs_for_method_call(tgt_expr, func.into(), None);
1712                        self.write_expr_adj(receiver, adjustments.into_boxed_slice());
1713                        self.write_method_resolution(tgt_expr, func, args);
1714
1715                        self.check_method_call(
1716                            tgt_expr,
1717                            &[],
1718                            self.db
1719                                .value_ty(func.into())
1720                                .unwrap()
1721                                .instantiate(self.interner(), args),
1722                            ty,
1723                            expected,
1724                        )
1725                    }
1726                    None => self.err_ty(),
1727                }
1728            }
1729        }
1730    }

1732    fn infer_call(
1733        &mut self,
1734        tgt_expr: ExprId,
1735        callee: ExprId,
1736        args: &[ExprId],
1737        expected: &Expectation<'db>,
1738    ) -> Ty<'db> {
1739        let callee_ty = self.infer_expr(callee, &Expectation::none(), ExprIsRead::Yes);
1740        let interner = self.interner();
1741        let mut derefs = self.table.autoderef(callee_ty);
1742        let (res, derefed_callee) = loop {
1743            let Some((callee_deref_ty, _)) = derefs.next() else {
1744                break (None, callee_ty);
1745            };
1746            if let Some(res) = derefs.table.callable_sig(callee_deref_ty, args.len()) {
1747                break (Some(res), callee_deref_ty);
1748            }
1749        };
1750        // if the function is unresolved, we use is_varargs=true to
1751        // suppress the arg count diagnostic here
1752        let is_varargs = derefed_callee.callable_sig(interner).is_some_and(|sig| sig.c_variadic())
1753            || res.is_none();
1754        let (param_tys, ret_ty) = match res {
1755            Some((func, params, ret_ty)) => {
1756                let mut adjustments = derefs.adjust_steps();
1757                if let Some(fn_x) = func {
1758                    self.write_fn_trait_method_resolution(
1759                        fn_x,
1760                        derefed_callee,
1761                        &mut adjustments,
1762                        callee_ty,
1763                        &params,
1764                        tgt_expr,
1765                    );
1766                }
1767                if let TyKind::Closure(c, _) = self.table.resolve_completely(callee_ty).kind() {
1768                    self.add_current_closure_dependency(c.into());
1769                    self.deferred_closures.entry(c.into()).or_default().push((
1770                        derefed_callee,
1771                        callee_ty,
1772                        params.clone(),
1773                        tgt_expr,
1774                    ));
1775                }
1776                self.write_expr_adj(callee, adjustments.into_boxed_slice());
1777                (params, ret_ty)
1778            }
1779            None => {
1780                self.push_diagnostic(InferenceDiagnostic::ExpectedFunction {
1781                    call_expr: tgt_expr,
1782                    found: callee_ty,
1783                });
1784                (Vec::new(), Ty::new_error(interner, ErrorGuaranteed))
1785            }
1786        };
1787        let indices_to_skip = self.check_legacy_const_generics(derefed_callee, args);
1788        self.check_call(
1789            tgt_expr,
1790            args,
1791            callee_ty,
1792            &param_tys,
1793            ret_ty,
1794            &indices_to_skip,
1795            is_varargs,
1796            expected,
1797        )
1798    }

1825    fn infer_method_call(
1826        &mut self,
1827        tgt_expr: ExprId,
1828        receiver: ExprId,
1829        args: &[ExprId],
1830        method_name: &Name,
1831        generic_args: Option<&HirGenericArgs>,
1832        expected: &Expectation<'db>,
1833    ) -> Ty<'db> {
1834        let receiver_ty = self.infer_expr_inner(receiver, &Expectation::none(), ExprIsRead::Yes);
1835        let receiver_ty = self.table.try_structurally_resolve_type(receiver_ty);
1836
1837        if matches!(receiver_ty.kind(), TyKind::Error(_) | TyKind::Infer(InferTy::TyVar(_))) {
1838            // Don't probe on error type, or on a fully unresolved infer var.
1839            // FIXME: Emit an error if we're probing on an infer var (type annotations needed).
1840            for &arg in args {
1841                // Make sure we infer and record the arguments.
1842                self.infer_expr_no_expect(arg, ExprIsRead::Yes);
1843            }
1844            return receiver_ty;
1845        }
1846
1847        let canonicalized_receiver = self.canonicalize(receiver_ty);
1848
1849        let resolved = method_resolution::lookup_method(
1850            &canonicalized_receiver,
1851            &mut self.table,
1852            Self::get_traits_in_scope(&self.resolver, &self.traits_in_scope)
1853                .as_ref()
1854                .left_or_else(|&it| it),
1855            VisibleFromModule::Filter(self.resolver.module()),
1856            method_name,
1857        );
1858        match resolved {
1859            Some((adjust, func, visible)) => {
1860                if !visible {
1861                    self.push_diagnostic(InferenceDiagnostic::PrivateAssocItem {
1862                        id: tgt_expr.into(),
1863                        item: func.into(),
1864                    })
1865                }
1866
1867                let (ty, adjustments) = adjust.apply(&mut self.table, receiver_ty);
1868                self.write_expr_adj(receiver, adjustments.into_boxed_slice());
1869
1870                let gen_args = self.substs_for_method_call(tgt_expr, func.into(), generic_args);
1871                self.write_method_resolution(tgt_expr, func, gen_args);
1872                let interner = DbInterner::new_with(self.db, None, None);
1873                self.check_method_call(
1874                    tgt_expr,
1875                    args,
1876                    self.db
1877                        .value_ty(func.into())
1878                        .expect("we have a function def")
1879                        .instantiate(interner, gen_args),
1880                    ty,
1881                    expected,
1882                )
1883            }
1884            // Failed to resolve, report diagnostic and try to resolve as call to field access or
1885            // assoc function
1886            None => {
1887                let field_with_same_name_exists = match self.lookup_field(receiver_ty, method_name)
1888                {
1889                    Some((ty, field_id, adjustments, _public)) => {
1890                        self.write_expr_adj(receiver, adjustments.into_boxed_slice());
1891                        self.result.field_resolutions.insert(tgt_expr, field_id);
1892                        Some(ty)
1893                    }
1894                    None => None,
1895                };
1896
1897                let assoc_func_with_same_name = method_resolution::iterate_method_candidates(
1898                    &canonicalized_receiver,
1899                    &mut self.table,
1900                    Self::get_traits_in_scope(&self.resolver, &self.traits_in_scope)
1901                        .as_ref()
1902                        .left_or_else(|&it| it),
1903                    VisibleFromModule::Filter(self.resolver.module()),
1904                    Some(method_name),
1905                    method_resolution::LookupMode::Path,
1906                    |_ty, item, visible| match item {
1907                        hir_def::AssocItemId::FunctionId(function_id) if visible => {
1908                            Some(function_id)
1909                        }
1910                        _ => None,
1911                    },
1912                );
1913
1914                self.push_diagnostic(InferenceDiagnostic::UnresolvedMethodCall {
1915                    expr: tgt_expr,
1916                    receiver: receiver_ty,
1917                    name: method_name.clone(),
1918                    field_with_same_name: field_with_same_name_exists,
1919                    assoc_func_with_same_name,
1920                });
1921
1922                let recovered = match assoc_func_with_same_name {
1923                    Some(f) => {
1924                        let args = self.substs_for_method_call(tgt_expr, f.into(), generic_args);
1925                        let interner = DbInterner::new_with(self.db, None, None);
1926                        let f = self
1927                            .db
1928                            .value_ty(f.into())
1929                            .expect("we have a function def")
1930                            .instantiate(interner, args);
1931                        let sig = f.callable_sig(self.interner()).expect("we have a function def");
1932                        Some((f, sig, true))
1933                    }
1934                    None => field_with_same_name_exists.and_then(|field_ty| {
1935                        let callable_sig = field_ty.callable_sig(self.interner())?;
1936                        Some((field_ty, callable_sig, false))
1937                    }),
1938                };
1939                match recovered {
1940                    Some((callee_ty, sig, strip_first)) => {
1941                        let sig = sig.skip_binder();
1942                        self.check_call(
1943                            tgt_expr,
1944                            args,
1945                            callee_ty,
1946                            sig.inputs_and_output
1947                                .inputs()
1948                                .get(strip_first as usize..)
1949                                .unwrap_or(&[]),
1950                            sig.output(),
1951                            &[],
1952                            true,
1953                            expected,
1954                        )
1955                    }
1956                    None => {
1957                        for &arg in args.iter() {
1958                            self.infer_expr_no_expect(arg, ExprIsRead::Yes);
1959                        }
1960                        self.err_ty()
1961                    }
1962                }
1963            }
1964        }
1965    }

1968        &mut self,
1969        tgt_expr: ExprId,
1970        args: &[ExprId],
1971        method_ty: Ty<'db>,
1972        receiver_ty: Ty<'db>,

1986                        } else {
1987                            (self.types.error, Vec::new())

2005    /// Generic function that factors out common logic from function calls,
2006    /// method calls and overloaded operators.
2007    pub(in super::super) fn check_call_arguments(
2008        &mut self,
2009        call_expr: ExprId,
2010        // Types (as defined in the *signature* of the target function)
2011        formal_input_tys: &[Ty<'db>],
2012        formal_output: Ty<'db>,
2013        // Expected output from the parent expression or statement
2014        expectation: &Expectation<'db>,
2015        // The expressions for each provided argument
2016        provided_args: &[ExprId],
2017        skip_indices: &[u32],
2018        // Whether the function is variadic, for example when imported from C
2019        c_variadic: bool,
2020    ) {
2021        // First, let's unify the formal method signature with the expectation eagerly.
2022        // We use this to guide coercion inference; it's output is "fudged" which means
2023        // any remaining type variables are assigned to new, unrelated variables. This
2024        // is because the inference guidance here is only speculative.
2025        let formal_output = self.table.resolve_vars_with_obligations(formal_output);
2026        let expected_input_tys: Option<Vec<_>> = expectation
2027            .only_has_type(&mut self.table)
2028            .and_then(|expected_output| {
2029                self.table
2030                    .infer_ctxt
2031                    .fudge_inference_if_ok(|| {
2032                        let mut ocx = ObligationCtxt::new(&self.table.infer_ctxt);
2033
2034                        // Attempt to apply a subtyping relationship between the formal
2035                        // return type (likely containing type variables if the function
2036                        // is polymorphic) and the expected return type.
2037                        // No argument expectations are produced if unification fails.
2038                        let origin = ObligationCause::new();
2039                        ocx.sup(&origin, self.table.trait_env.env, expected_output, formal_output)?;
2040                        if !ocx.try_evaluate_obligations().is_empty() {
2041                            return Err(TypeError::Mismatch);
2042                        }
2043
2044                        // Record all the argument types, with the args
2045                        // produced from the above subtyping unification.
2046                        Ok(Some(
2047                            formal_input_tys
2048                                .iter()
2049                                .map(|&ty| self.table.infer_ctxt.resolve_vars_if_possible(ty))
2050                                .collect(),
2051                        ))
2052                    })
2053                    .ok()
2054            })
2055            .unwrap_or_default();
2056
2057        // If there are no external expectations at the call site, just use the types from the function defn
2058        let expected_input_tys = if let Some(expected_input_tys) = &expected_input_tys {
2059            assert_eq!(expected_input_tys.len(), formal_input_tys.len());
2060            expected_input_tys
2061        } else {
2062            formal_input_tys
2063        };
2064
2065        let minimum_input_count = expected_input_tys.len();
2066        let provided_arg_count = provided_args.len() - skip_indices.len();
2067
2068        // Keep track of whether we *could possibly* be satisfied, i.e. whether we're on the happy path
2069        // if the wrong number of arguments were supplied, we CAN'T be satisfied,
2070        // and if we're c_variadic, the supplied arguments must be >= the minimum count from the function
2071        // otherwise, they need to be identical, because rust doesn't currently support variadic functions
2072        let args_count_matches = if c_variadic {
2073            provided_arg_count >= minimum_input_count
2074        } else {
2075            provided_arg_count == minimum_input_count
2076        };
2077
2078        if !args_count_matches {
2079            self.push_diagnostic(InferenceDiagnostic::MismatchedArgCount {
2080                call_expr,
2081                expected: expected_input_tys.len() + skip_indices.len(),
2082                found: provided_args.len(),
2083            });
2084        }
2085
2086        // We introduce a helper function to demand that a given argument satisfy a given input
2087        // This is more complicated than just checking type equality, as arguments could be coerced
2088        // This version writes those types back so further type checking uses the narrowed types
2089        let demand_compatible = |this: &mut InferenceContext<'_, 'db>, idx| {
2090            let formal_input_ty: Ty<'db> = formal_input_tys[idx];
2091            let expected_input_ty: Ty<'db> = expected_input_tys[idx];
2092            let provided_arg = provided_args[idx];
2093
2094            debug!("checking argument {}: {:?} = {:?}", idx, provided_arg, formal_input_ty);
2095
2096            // We're on the happy path here, so we'll do a more involved check and write back types
2097            // To check compatibility, we'll do 3 things:
2098            // 1. Unify the provided argument with the expected type
2099            let expectation = Expectation::rvalue_hint(this, expected_input_ty);
2100
2101            let checked_ty = this.infer_expr_inner(provided_arg, &expectation, ExprIsRead::Yes);
2102
2103            // 2. Coerce to the most detailed type that could be coerced
2104            //    to, which is `expected_ty` if `rvalue_hint` returns an
2105            //    `ExpectHasType(expected_ty)`, or the `formal_ty` otherwise.
2106            let coerced_ty = expectation.only_has_type(&mut this.table).unwrap_or(formal_input_ty);
2107
2108            // Cause selection errors caused by resolving a single argument to point at the
2109            // argument and not the call. This lets us customize the span pointed to in the
2110            // fulfillment error to be more accurate.
2111            let coerced_ty = this.table.resolve_vars_with_obligations(coerced_ty);
2112
2113            let coerce_never = if this
2114                .expr_guaranteed_to_constitute_read_for_never(provided_arg, ExprIsRead::Yes)
2115            {
2116                CoerceNever::Yes
2117            } else {
2118                CoerceNever::No
2119            };
2120            let coerce_error = this
2121                .coerce(
2122                    provided_arg.into(),
2123                    checked_ty,
2124                    coerced_ty,
2125                    AllowTwoPhase::Yes,
2126                    coerce_never,
2127                )
2128                .err();
2129            if coerce_error.is_some() {
2130                return Err((coerce_error, coerced_ty, checked_ty));
2131            }
2132
2133            // 3. Check if the formal type is actually equal to the checked one
2134            //    and register any such obligations for future type checks.
2135            let formal_ty_error = this
2136                .table
2137                .infer_ctxt
2138                .at(&ObligationCause::new(), this.table.trait_env.env)
2139                .eq(formal_input_ty, coerced_ty);
2140
2141            // If neither check failed, the types are compatible
2142            match formal_ty_error {
2143                Ok(InferOk { obligations, value: () }) => {
2144                    this.table.register_predicates(obligations);
2145                    Ok(())
2146                }
2147                Err(err) => Err((Some(err), coerced_ty, checked_ty)),
2148            }
2149        };
2150
2151        // Check the arguments.
2152        // We do this in a pretty awful way: first we type-check any arguments
2153        // that are not closures, then we type-check the closures. This is so
2154        // that we have more information about the types of arguments when we
2155        // type-check the functions. This isn't really the right way to do this.
2156        for check_closures in [false, true] {
2157            // More awful hacks: before we check argument types, try to do
2158            // an "opportunistic" trait resolution of any trait bounds on
2159            // the call. This helps coercions.
2160            if check_closures {
2161                self.table.select_obligations_where_possible();
2162            }
2163
2164            let mut skip_indices = skip_indices.iter().copied();
2165            // Check each argument, to satisfy the input it was provided for
2166            // Visually, we're traveling down the diagonal of the compatibility matrix
2167            for (idx, arg) in provided_args.iter().enumerate() {
2168                if skip_indices.clone().next() == Some(idx as u32) {
2169                    skip_indices.next();
2170                    continue;
2171                }
2172
2173                // For this check, we do *not* want to treat async coroutine closures (async blocks)
2174                // as proper closures. Doing so would regress type inference when feeding
2175                // the return value of an argument-position async block to an argument-position
2176                // closure wrapped in a block.
2177                // See <https://github.com/rust-lang/rust/issues/112225>.
2178                let is_closure = if let Expr::Closure { closure_kind, .. } = self.body[*arg] {
2179                    !matches!(closure_kind, ClosureKind::Coroutine(_))
2180                } else {
2181                    false
2182                };
2183                if is_closure != check_closures {
2184                    continue;
2185                }
2186
2187                if idx >= minimum_input_count {
2188                    // Make sure we've checked this expr at least once.
2189                    self.infer_expr_no_expect(*arg, ExprIsRead::Yes);
2190                    continue;
2191                }
2192
2193                if let Err((_error, expected, found)) = demand_compatible(self, idx)
2194                    && args_count_matches
2195                {
2196                    // Don't report type mismatches if there is a mismatch in args count.
2197                    self.result
2198                        .type_mismatches
2199                        .insert((*arg).into(), TypeMismatch { expected, actual: found });
2200                }
2201            }
2202        }
2203
2204        if !args_count_matches {}
2205    }

2336            let generic_predicates =
2337                self.db.generic_predicates(GenericDefId::from_callable(self.db, fn_def.0));
2338            if let Some(predicates) = generic_predicates.instantiate(self.interner(), parameters) {
2339                let interner = self.interner();
2340                let param_env = self.table.trait_env.env;
2341                self.table.register_predicates(predicates.map(|predicate| {
2342                    Obligation::new(interner, ObligationCause::new(), param_env, predicate)
2343                }));
2344            }
2345            // add obligation for trait implementation, if this is a trait method
2346            match fn_def.0 {
2347                CallableDefId::FunctionId(f) => {
2348                    if let ItemContainerId::TraitId(trait_) = f.lookup(self.db).container {
2349                        // construct a TraitRef
2350                        let trait_params_len = generics(self.db, trait_.into()).len();
2351                        let substs = GenericArgs::new_from_iter(
2352                            self.interner(),
2353                            parameters.as_slice()[..trait_params_len].iter().copied(),
2354                        );
2355                        self.table.register_predicate(Obligation::new(
2356                            self.interner(),
2357                            ObligationCause::new(),
2358                            self.table.trait_env.env,
2359                            TraitRef::new(self.interner(), trait_.into(), substs),
2360                        ));
2361                    }
2362                }
2363                CallableDefId::StructId(_) | CallableDefId::EnumVariantId(_) => {}
2364            }
2365        }
2366    }

2368    /// Returns the argument indices to skip.
2369    fn check_legacy_const_generics(&mut self, callee: Ty<'db>, args: &[ExprId]) -> Box<[u32]> {
2370        let (func, _subst) = match callee.kind() {
2371            TyKind::FnDef(callable, subst) => {
2372                let func = match callable.0 {
2373                    CallableDefId::FunctionId(f) => f,
2374                    _ => return Default::default(),
2375                };
2376                (func, subst)
2377            }
2378            _ => return Default::default(),
2379        };
2380
2381        let data = self.db.function_signature(func);
2382        let Some(legacy_const_generics_indices) = &data.legacy_const_generics_indices else {
2383            return Default::default();
2384        };
2385
2386        // only use legacy const generics if the param count matches with them
2387        if data.params.len() + legacy_const_generics_indices.len() != args.len() {
2388            if args.len() <= data.params.len() {
2389                return Default::default();
2390            } else {
2391                // there are more parameters than there should be without legacy
2392                // const params; use them
2393                let mut indices = legacy_const_generics_indices.as_ref().clone();
2394                indices.sort();
2395                return indices;
2396            }
2397        }
2398
2399        // check legacy const parameters
2400        for arg_idx in legacy_const_generics_indices.iter().copied() {
2401            if arg_idx >= args.len() as u32 {
2402                continue;
2403            }
2404            let expected = Expectation::none(); // FIXME use actual const ty, when that is lowered correctly
2405            self.infer_expr(args[arg_idx as usize], &expected, ExprIsRead::Yes);
2406            // FIXME: evaluate and unify with the const
2407        }
2408        let mut indices = legacy_const_generics_indices.as_ref().clone();
2409        indices.sort();
2410        indices
2411    }

2424    /// Enforces expectations on lhs type and rhs type depending on the operator and returns the
2425    /// output type of the binary op.
2426    fn enforce_builtin_binop_types(&mut self, lhs: Ty<'db>, rhs: Ty<'db>, op: BinaryOp) -> Ty<'db> {
2427        // Special-case a single layer of referencing, so that things like `5.0 + &6.0f32` work (See rust-lang/rust#57447).
2428        let lhs = self.deref_ty_if_possible(lhs);
2429        let rhs = self.deref_ty_if_possible(rhs);
2430
2431        let (op, is_assign) = match op {
2432            BinaryOp::Assignment { op: Some(inner) } => (BinaryOp::ArithOp(inner), true),
2433            _ => (op, false),
2434        };
2435
2436        let output_ty = match op {
2437            BinaryOp::LogicOp(_) => {
2438                let bool_ = self.types.bool;
2439                self.unify(lhs, bool_);
2440                self.unify(rhs, bool_);
2441                bool_
2442            }
2443
2444            BinaryOp::ArithOp(ArithOp::Shl | ArithOp::Shr) => {
2445                // result type is same as LHS always
2446                lhs
2447            }
2448
2449            BinaryOp::ArithOp(_) => {
2450                // LHS, RHS, and result will have the same type
2451                self.unify(lhs, rhs);
2452                lhs
2453            }
2454
2455            BinaryOp::CmpOp(_) => {
2456                // LHS and RHS will have the same type
2457                self.unify(lhs, rhs);
2458                self.types.bool
2459            }
2460
2461            BinaryOp::Assignment { op: None } => {
2462                stdx::never!("Simple assignment operator is not binary op.");
2463                lhs
2464            }
2465
2466            BinaryOp::Assignment { .. } => unreachable!("handled above"),
2467        };
2468
2469        if is_assign { self.types.unit } else { output_ty }
2470    }

2472    fn is_builtin_binop(&mut self, lhs: Ty<'db>, rhs: Ty<'db>, op: BinaryOp) -> bool {
2473        // Special-case a single layer of referencing, so that things like `5.0 + &6.0f32` work (See rust-lang/rust#57447).
2474        let lhs = self.deref_ty_if_possible(lhs);
2475        let rhs = self.deref_ty_if_possible(rhs);
2476
2477        let op = match op {
2478            BinaryOp::Assignment { op: Some(inner) } => BinaryOp::ArithOp(inner),
2479            _ => op,
2480        };
2481
2482        match op {
2483            BinaryOp::LogicOp(_) => true,
2484
2485            BinaryOp::ArithOp(ArithOp::Shl | ArithOp::Shr) => {
2486                lhs.is_integral() && rhs.is_integral()
2487            }
2488
2489            BinaryOp::ArithOp(
2490                ArithOp::Add | ArithOp::Sub | ArithOp::Mul | ArithOp::Div | ArithOp::Rem,
2491            ) => {
2492                lhs.is_integral() && rhs.is_integral()
2493                    || lhs.is_floating_point() && rhs.is_floating_point()
2494            }
2495
2496            BinaryOp::ArithOp(ArithOp::BitAnd | ArithOp::BitOr | ArithOp::BitXor) => {
2497                lhs.is_integral() && rhs.is_integral()
2498                    || lhs.is_floating_point() && rhs.is_floating_point()
2499                    || matches!((lhs.kind(), rhs.kind()), (TyKind::Bool, TyKind::Bool))
2500            }
2501
2502            BinaryOp::CmpOp(_) => {
2503                let is_scalar = |kind| {
2504                    matches!(
2505                        kind,
2506                        TyKind::Bool
2507                            | TyKind::Char
2508                            | TyKind::Int(_)
2509                            | TyKind::Uint(_)
2510                            | TyKind::Float(_)
2511                            | TyKind::FnDef(..)
2512                            | TyKind::FnPtr(..)
2513                            | TyKind::RawPtr(..)
2514                            | TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_))
2515                    )
2516                };
2517                is_scalar(lhs.kind()) && is_scalar(rhs.kind())
2518            }
2519
2520            BinaryOp::Assignment { op: None } => {
2521                stdx::never!("Simple assignment operator is not binary op.");
2522                false
2523            }
2524
2525            BinaryOp::Assignment { .. } => unreachable!("handled above"),
2526        }
2527    }

2529    pub(super) fn with_breakable_ctx<T>(
2530        &mut self,
2531        kind: BreakableKind,
2532        ty: Option<Ty<'db>>,
2533        label: Option<LabelId>,
2534        cb: impl FnOnce(&mut Self) -> T,
2535    ) -> (Option<Ty<'db>>, T) {
2536        self.breakables.push({
2537            BreakableContext { kind, may_break: false, coerce: ty.map(CoerceMany::new), label }
2538        });
2539        let res = cb(self);
2540        let ctx = self.breakables.pop().expect("breakable stack broken");
2541        (if ctx.may_break { ctx.coerce.map(|ctx| ctx.complete(self)) } else { None }, res)
2542    }
