61    pub(crate) fn infer_expr(
62        &mut self,
63        tgt_expr: ExprId,
64        expected: &Expectation<'db>,
65        is_read: ExprIsRead,
66    ) -> Ty<'db> {
67        let ty = self.infer_expr_inner(tgt_expr, expected, is_read);
68        if let Some(expected_ty) = expected.only_has_type(&mut self.table) {
69            let could_unify = self.unify(ty, expected_ty);
70            if !could_unify {
71                self.result.type_mismatches.get_or_insert_default().insert(
72                    tgt_expr.into(),
73                    TypeMismatch { expected: expected_ty.store(), actual: ty.store() },
74                );
75            }
76        }
77        ty
78    }

90    pub(super) fn infer_expr_coerce(
91        &mut self,
92        expr: ExprId,
93        expected: &Expectation<'db>,
94        is_read: ExprIsRead,
95    ) -> Ty<'db> {
96        let ty = self.infer_expr_inner(expr, expected, is_read);
97        if let Some(target) = expected.only_has_type(&mut self.table) {
98            match self.coerce(expr.into(), ty, target, AllowTwoPhase::No, is_read) {
99                Ok(res) => res,
100                Err(_) => {
101                    self.result.type_mismatches.get_or_insert_default().insert(
102                        expr.into(),
103                        TypeMismatch { expected: target.store(), actual: ty.store() },
104                    );
105                    target
106                }
107            }
108        } else {
109            ty
110        }
111    }

113    /// Whether this expression constitutes a read of value of the type that
114    /// it evaluates to.
115    ///
116    /// This is used to determine if we should consider the block to diverge
117    /// if the expression evaluates to `!`, and if we should insert a `NeverToAny`
118    /// coercion for values of type `!`.
119    ///
120    /// This function generally returns `false` if the expression is a place
121    /// expression and the *parent* expression is the scrutinee of a match or
122    /// the pointee of an `&` addr-of expression, since both of those parent
123    /// expressions take a *place* and not a value.
124    pub(super) fn expr_guaranteed_to_constitute_read_for_never(
125        &mut self,
126        expr: ExprId,
127        is_read: ExprIsRead,
128    ) -> bool {
129        // rustc does the place expr check first, but since we are feeding
130        // readness of the `expr` as a given value, we just can short-circuit
131        // the place expr check if it's true(see codes and comments below)
132        if is_read == ExprIsRead::Yes {
133            return true;
134        }
135
136        // We only care about place exprs. Anything else returns an immediate
137        // which would constitute a read. We don't care about distinguishing
138        // "syntactic" place exprs since if the base of a field projection is
139        // not a place then it would've been UB to read from it anyways since
140        // that constitutes a read.
141        if !self.is_syntactic_place_expr(expr) {
142            return true;
143        }
144
145        // rustc queries parent hir node of `expr` here and determine whether
146        // the current `expr` is read of value per its parent.
147        // But since we don't have hir node, we cannot follow such "bottom-up"
148        // method.
149        // So, we pass down such readness from the parent expression through the
150        // recursive `infer_expr*` calls in a "top-down" manner.
151        is_read == ExprIsRead::Yes
152    }

154    /// Whether this pattern constitutes a read of value of the scrutinee that
155    /// it is matching against. This is used to determine whether we should
156    /// perform `NeverToAny` coercions.
157    fn pat_guaranteed_to_constitute_read_for_never(&self, pat: PatId) -> bool {
158        match &self.body[pat] {
159            // Does not constitute a read.
160            Pat::Wild => false,
161
162            // This is unnecessarily restrictive when the pattern that doesn't
163            // constitute a read is unreachable.
164            //
165            // For example `match *never_ptr { value => {}, _ => {} }` or
166            // `match *never_ptr { _ if false => {}, value => {} }`.
167            //
168            // It is however fine to be restrictive here; only returning `true`
169            // can lead to unsoundness.
170            Pat::Or(subpats) => {
171                subpats.iter().all(|pat| self.pat_guaranteed_to_constitute_read_for_never(*pat))
172            }
173
174            // All of these constitute a read, or match on something that isn't `!`,
175            // which would require a `NeverToAny` coercion.
176            Pat::Bind { .. }
177            | Pat::TupleStruct { .. }
178            | Pat::Path(_)
179            | Pat::Tuple { .. }
180            | Pat::Box { .. }
181            | Pat::Ref { .. }
182            | Pat::Lit(_)
183            | Pat::Range { .. }
184            | Pat::Slice { .. }
185            | Pat::ConstBlock(_)
186            | Pat::Record { .. }
187            | Pat::Missing => true,
188            Pat::Expr(_) => unreachable!(
189                "we don't call pat_guaranteed_to_constitute_read_for_never() with assignments"
190            ),
191        }
192    }

194    /// Checks if the pattern contains any `ref` or `ref mut` bindings, and if
195    /// yes whether it contains mutable or just immutables ones.
196    //
197    // FIXME(tschottdorf): this is problematic as the HIR is being scraped, but
198    // ref bindings are be implicit after #42640 (default match binding modes). See issue #44848.
199    fn contains_explicit_ref_binding(&self, pat: PatId) -> bool {
200        if let Pat::Bind { id, .. } = self.body[pat]
201            && matches!(self.body[id].mode, BindingAnnotation::Ref | BindingAnnotation::RefMut)
202        {
203            return true;
204        }
205
206        let mut result = false;
207        self.body.walk_pats_shallow(pat, |pat| result |= self.contains_explicit_ref_binding(pat));
208        result
209    }

211    fn is_syntactic_place_expr(&self, expr: ExprId) -> bool {
212        match &self.body[expr] {
213            // Lang item paths cannot currently be local variables or statics.
214            Expr::Path(Path::LangItem(_, _)) => false,
215            Expr::Path(Path::Normal(path)) => path.type_anchor.is_none(),
216            Expr::Path(path) => self
217                .resolver
218                .resolve_path_in_value_ns_fully(self.db, path, self.body.expr_path_hygiene(expr))
219                .is_none_or(|res| matches!(res, ValueNs::LocalBinding(_) | ValueNs::StaticId(_))),
220            Expr::Underscore => true,
221            Expr::UnaryOp { op: UnaryOp::Deref, .. } => true,
222            Expr::Field { .. } | Expr::Index { .. } => true,
223            Expr::Call { .. }
224            | Expr::MethodCall { .. }
225            | Expr::Tuple { .. }
226            | Expr::If { .. }
227            | Expr::Match { .. }
228            | Expr::Closure { .. }
229            | Expr::Block { .. }
230            | Expr::Array(..)
231            | Expr::Break { .. }
232            | Expr::Continue { .. }
233            | Expr::Return { .. }
234            | Expr::Become { .. }
235            | Expr::Let { .. }
236            | Expr::Loop { .. }
237            | Expr::InlineAsm(..)
238            | Expr::OffsetOf(..)
239            | Expr::Literal(..)
240            | Expr::Const(..)
241            | Expr::UnaryOp { .. }
242            | Expr::BinaryOp { .. }
243            | Expr::Assignment { .. }
244            | Expr::Yield { .. }
245            | Expr::Cast { .. }
246            | Expr::Async { .. }
247            | Expr::Unsafe { .. }
248            | Expr::Await { .. }
249            | Expr::Ref { .. }
250            | Expr::Range { .. }
251            | Expr::Box { .. }
252            | Expr::RecordLit { .. }
253            | Expr::Yeet { .. }
254            | Expr::Missing => false,
255        }
256    }

267    fn infer_expr_coerce_never(
268        &mut self,
269        expr: ExprId,
270        expected: &Expectation<'db>,
271        is_read: ExprIsRead,
272    ) -> Ty<'db> {
273        let ty = self.infer_expr_inner(expr, expected, is_read);
274        // While we don't allow *arbitrary* coercions here, we *do* allow
275        // coercions from `!` to `expected`.
276        if ty.is_never() {
277            if let Some(adjustments) = self.result.expr_adjustments.get(&expr) {
278                return if let [Adjustment { kind: Adjust::NeverToAny, target }] = &**adjustments {
279                    target.as_ref()
280                } else {
281                    self.err_ty()
282                };
283            }
284
285            if let Some(target) = expected.only_has_type(&mut self.table) {
286                self.coerce(expr.into(), ty, target, AllowTwoPhase::No, ExprIsRead::Yes)
287                    .expect("never-to-any coercion should always succeed")
288            } else {
289                ty
290            }
291        } else {
292            if let Some(expected_ty) = expected.only_has_type(&mut self.table) {
293                let could_unify = self.unify(ty, expected_ty);
294                if !could_unify {
295                    self.result.type_mismatches.get_or_insert_default().insert(
296                        expr.into(),
297                        TypeMismatch { expected: expected_ty.store(), actual: ty.store() },
298                    );
299                }
300            }
301            ty
302        }
303    }

305    #[tracing::instrument(level = "debug", skip(self, is_read), ret)]
306    fn infer_expr_inner(
307        &mut self,
308        tgt_expr: ExprId,
309        expected: &Expectation<'db>,
310        is_read: ExprIsRead,
311    ) -> Ty<'db> {
312        self.db.unwind_if_revision_cancelled();
313
314        let expr = &self.body[tgt_expr];
315        tracing::trace!(?expr);
316        let ty = match expr {
317            Expr::Missing => self.err_ty(),
318            &Expr::If { condition, then_branch, else_branch } => {
319                let expected = &expected.adjust_for_branches(&mut self.table);
320                self.infer_expr_coerce_never(
321                    condition,
322                    &Expectation::HasType(self.types.types.bool),
323                    ExprIsRead::Yes,
324                );
325
326                let condition_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
327
328                let then_ty = self.infer_expr_inner(then_branch, expected, ExprIsRead::Yes);
329                let then_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
330                let mut coercion_sites = [then_branch, tgt_expr];
331                if let Some(else_branch) = else_branch {
332                    coercion_sites[1] = else_branch;
333                }
334                let mut coerce = CoerceMany::with_coercion_sites(
335                    expected.coercion_target_type(&mut self.table),
336                    &coercion_sites,
337                );
338                coerce.coerce(self, &ObligationCause::new(), then_branch, then_ty, ExprIsRead::Yes);
339                match else_branch {
340                    Some(else_branch) => {
341                        let else_ty = self.infer_expr_inner(else_branch, expected, ExprIsRead::Yes);
342                        let else_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
343                        coerce.coerce(
344                            self,
345                            &ObligationCause::new(),
346                            else_branch,
347                            else_ty,
348                            ExprIsRead::Yes,
349                        );
350                        self.diverges = condition_diverges | then_diverges & else_diverges;
351                    }
352                    None => {
353                        coerce.coerce_forced_unit(
354                            self,
355                            tgt_expr,
356                            &ObligationCause::new(),
357                            true,
358                            ExprIsRead::Yes,
359                        );
360                        self.diverges = condition_diverges;
361                    }
362                }
363
364                coerce.complete(self)
365            }
366            &Expr::Let { pat, expr } => {
367                let child_is_read = if self.pat_guaranteed_to_constitute_read_for_never(pat) {
368                    ExprIsRead::Yes
369                } else {
370                    ExprIsRead::No
371                };
372                let input_ty = self.infer_expr(expr, &Expectation::none(), child_is_read);
373                self.infer_top_pat(
374                    pat,
375                    input_ty,
376                    Some(DeclContext { origin: DeclOrigin::LetExpr }),
377                );
378                self.types.types.bool
379            }
380            Expr::Block { statements, tail, label, id: _ } => {
381                self.infer_block(tgt_expr, statements, *tail, *label, expected)
382            }
383            Expr::Unsafe { id: _, statements, tail } => {
384                self.infer_block(tgt_expr, statements, *tail, None, expected)
385            }
386            Expr::Const(id) => {
387                self.with_breakable_ctx(BreakableKind::Border, None, None, |this| {
388                    this.infer_expr(*id, expected, ExprIsRead::Yes)
389                })
390                .1
391            }
392            Expr::Async { id: _, statements, tail } => {
393                self.infer_async_block(tgt_expr, statements, tail)
394            }
395            &Expr::Loop { body, label } => {
396                // FIXME: should be:
397                // let ty = expected.coercion_target_type(&mut self.table);
398                let ty = self.table.next_ty_var();
399                let (breaks, ()) =
400                    self.with_breakable_ctx(BreakableKind::Loop, Some(ty), label, |this| {
401                        this.infer_expr(
402                            body,
403                            &Expectation::HasType(this.types.types.unit),
404                            ExprIsRead::Yes,
405                        );
406                    });
407
408                match breaks {
409                    Some(breaks) => {
410                        self.diverges = Diverges::Maybe;
411                        breaks
412                    }
413                    None => self.types.types.never,
414                }
415            }
416            Expr::Closure { body, args, ret_type, arg_types, closure_kind, capture_by: _ } => self
417                .infer_closure(
418                    *body,
419                    args,
420                    *ret_type,
421                    arg_types,
422                    *closure_kind,
423                    tgt_expr,
424                    expected,
425                ),
426            Expr::Call { callee, args, .. } => self.infer_call(tgt_expr, *callee, args, expected),
427            Expr::MethodCall { receiver, args, method_name, generic_args } => self
428                .infer_method_call(
429                    tgt_expr,
430                    *receiver,
431                    args,
432                    method_name,
433                    generic_args.as_deref(),
434                    expected,
435                ),
436            Expr::Match { expr, arms } => {
437                let mut scrutinee_is_read = true;
438                let mut contains_ref_bindings = false;
439                for arm in arms {
440                    scrutinee_is_read &= self.pat_guaranteed_to_constitute_read_for_never(arm.pat);
441                    contains_ref_bindings |= self.contains_explicit_ref_binding(arm.pat);
442                }
443                let scrutinee_is_read =
444                    if scrutinee_is_read { ExprIsRead::Yes } else { ExprIsRead::No };
445                let input_ty = self.demand_scrutinee_type(
446                    *expr,
447                    contains_ref_bindings,
448                    arms.is_empty(),
449                    scrutinee_is_read,
450                );
451
452                if arms.is_empty() {
453                    self.diverges = Diverges::Always;
454                    self.types.types.never
455                } else {
456                    let matchee_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
457                    let mut all_arms_diverge = Diverges::Always;
458                    for arm in arms.iter() {
459                        self.infer_top_pat(arm.pat, input_ty, None);
460                    }
461
462                    let expected = expected.adjust_for_branches(&mut self.table);
463                    let result_ty = match &expected {
464                        // We don't coerce to `()` so that if the match expression is a
465                        // statement it's branches can have any consistent type.
466                        Expectation::HasType(ty) if *ty != self.types.types.unit => *ty,
467                        _ => self.table.next_ty_var(),
468                    };
469                    let mut coerce = CoerceMany::new(result_ty);
470
471                    for arm in arms.iter() {
472                        if let Some(guard_expr) = arm.guard {
473                            self.diverges = Diverges::Maybe;
474                            self.infer_expr_coerce_never(
475                                guard_expr,
476                                &Expectation::HasType(self.types.types.bool),
477                                ExprIsRead::Yes,
478                            );
479                        }
480                        self.diverges = Diverges::Maybe;
481
482                        let arm_ty = self.infer_expr_inner(arm.expr, &expected, ExprIsRead::Yes);
483                        all_arms_diverge &= self.diverges;
484                        coerce.coerce(
485                            self,
486                            &ObligationCause::new(),
487                            arm.expr,
488                            arm_ty,
489                            ExprIsRead::Yes,
490                        );
491                    }
492
493                    self.diverges = matchee_diverges | all_arms_diverge;
494
495                    coerce.complete(self)
496                }
497            }
498            Expr::Path(p) => self.infer_expr_path(p, tgt_expr.into(), tgt_expr),
499            &Expr::Continue { label } => {
500                if find_continuable(&mut self.breakables, label).is_none() {
501                    self.push_diagnostic(InferenceDiagnostic::BreakOutsideOfLoop {
502                        expr: tgt_expr,
503                        is_break: false,
504                        bad_value_break: false,
505                    });
506                };
507                self.types.types.never
508            }
509            &Expr::Break { expr, label } => {
510                let val_ty = if let Some(expr) = expr {
511                    let opt_coerce_to = match find_breakable(&mut self.breakables, label) {
512                        Some(ctxt) => match &ctxt.coerce {
513                            Some(coerce) => coerce.expected_ty(),
514                            None => {
515                                self.push_diagnostic(InferenceDiagnostic::BreakOutsideOfLoop {
516                                    expr: tgt_expr,
517                                    is_break: true,
518                                    bad_value_break: true,
519                                });
520                                self.err_ty()
521                            }
522                        },
523                        None => self.err_ty(),
524                    };
525                    self.infer_expr_inner(
526                        expr,
527                        &Expectation::HasType(opt_coerce_to),
528                        ExprIsRead::Yes,
529                    )
530                } else {
531                    self.types.types.unit
532                };
533
534                match find_breakable(&mut self.breakables, label) {
535                    Some(ctxt) => match ctxt.coerce.take() {
536                        Some(mut coerce) => {
537                            coerce.coerce(
538                                self,
539                                &ObligationCause::new(),
540                                expr.unwrap_or(tgt_expr),
541                                val_ty,
542                                ExprIsRead::Yes,
543                            );
544
545                            // Avoiding borrowck
546                            let ctxt = find_breakable(&mut self.breakables, label)
547                                .expect("breakable stack changed during coercion");
548                            ctxt.may_break = true;
549                            ctxt.coerce = Some(coerce);
550                        }
551                        None => ctxt.may_break = true,
552                    },
553                    None => {
554                        self.push_diagnostic(InferenceDiagnostic::BreakOutsideOfLoop {
555                            expr: tgt_expr,
556                            is_break: true,
557                            bad_value_break: false,
558                        });
559                    }
560                }
561                self.types.types.never
562            }
563            &Expr::Return { expr } => self.infer_expr_return(tgt_expr, expr),
564            &Expr::Become { expr } => self.infer_expr_become(expr),
565            Expr::Yield { expr } => {
566                if let Some((resume_ty, yield_ty)) = self.resume_yield_tys {
567                    if let Some(expr) = expr {
568                        self.infer_expr_coerce(
569                            *expr,
570                            &Expectation::has_type(yield_ty),
571                            ExprIsRead::Yes,
572                        );
573                    } else {
574                        let unit = self.types.types.unit;
575                        let _ = self.coerce(
576                            tgt_expr.into(),
577                            unit,
578                            yield_ty,
579                            AllowTwoPhase::No,
580                            ExprIsRead::Yes,
581                        );
582                    }
583                    resume_ty
584                } else {
585                    // FIXME: report error (yield expr in non-coroutine)
586                    self.types.types.error
587                }
588            }
589            Expr::Yeet { expr } => {
590                if let &Some(expr) = expr {
591                    self.infer_expr_no_expect(expr, ExprIsRead::Yes);
592                }
593                self.types.types.never
594            }
595            Expr::RecordLit { path, fields, spread, .. } => {
596                let (ty, def_id) = self.resolve_variant(tgt_expr.into(), path.as_deref(), false);
597
598                if let Some(t) = expected.only_has_type(&mut self.table) {
599                    self.unify(ty, t);
600                }
601
602                let substs = ty.as_adt().map(|(_, s)| s).unwrap_or(self.types.empty.generic_args);
603                if let Some(variant) = def_id {
604                    self.write_variant_resolution(tgt_expr.into(), variant);
605                }
606                match def_id {
607                    _ if fields.is_empty() => {}
608                    Some(def) => {
609                        let field_types = self.db.field_types(def);
610                        let variant_data = def.fields(self.db);
611                        let visibilities = self.db.field_visibilities(def);
612                        for field in fields.iter() {
613                            let field_def = {
614                                match variant_data.field(&field.name) {
615                                    Some(local_id) => {
616                                        if !visibilities[local_id]
617                                            .is_visible_from(self.db, self.resolver.module())
618                                        {
619                                            self.push_diagnostic(
620                                                InferenceDiagnostic::NoSuchField {
621                                                    field: field.expr.into(),
622                                                    private: Some(local_id),
623                                                    variant: def,
624                                                },
625                                            );
626                                        }
627                                        Some(local_id)
628                                    }
629                                    None => {
630                                        self.push_diagnostic(InferenceDiagnostic::NoSuchField {
631                                            field: field.expr.into(),
632                                            private: None,
633                                            variant: def,
634                                        });
635                                        None
636                                    }
637                                }
638                            };
639                            let field_ty = field_def.map_or(self.err_ty(), |it| {
640                                field_types[it].get().instantiate(self.interner(), &substs)
641                            });
642
643                            // Field type might have some unknown types
644                            // FIXME: we may want to emit a single type variable for all instance of type fields?
645                            let field_ty = self.insert_type_vars(field_ty);
646                            self.infer_expr_coerce(
647                                field.expr,
648                                &Expectation::has_type(field_ty),
649                                ExprIsRead::Yes,
650                            );
651                        }
652                    }
653                    None => {
654                        for field in fields.iter() {
655                            // Field projections don't constitute reads.
656                            self.infer_expr_coerce(field.expr, &Expectation::None, ExprIsRead::No);
657                        }
658                    }
659                }
660                if let RecordSpread::Expr(expr) = *spread {
661                    self.infer_expr(expr, &Expectation::has_type(ty), ExprIsRead::Yes);
662                }
663                ty
664            }
665            Expr::Field { expr, name } => self.infer_field_access(tgt_expr, *expr, name, expected),
666            Expr::Await { expr } => {
667                let inner_ty = self.infer_expr_inner(*expr, &Expectation::none(), ExprIsRead::Yes);
668                self.resolve_associated_type(inner_ty, self.resolve_future_future_output())
669            }
670            Expr::Cast { expr, type_ref } => {
671                let cast_ty = self.make_body_ty(*type_ref);
672                let expr_ty =
673                    self.infer_expr(*expr, &Expectation::Castable(cast_ty), ExprIsRead::Yes);
674                self.deferred_cast_checks.push(CastCheck::new(tgt_expr, *expr, expr_ty, cast_ty));
675                cast_ty
676            }
677            Expr::Ref { expr, rawness, mutability } => {
678                let mutability = lower_mutability(*mutability);
679                let expectation = if let Some((exp_inner, exp_rawness, exp_mutability)) = expected
680                    .only_has_type(&mut self.table)
681                    .as_ref()
682                    .and_then(|t| t.as_reference_or_ptr())
683                {
684                    if exp_mutability == Mutability::Mut && mutability == Mutability::Not {
685                        // FIXME: record type error - expected mut reference but found shared ref,
686                        // which cannot be coerced
687                    }
688                    if exp_rawness == Rawness::Ref && *rawness == Rawness::RawPtr {
689                        // FIXME: record type error - expected reference but found ptr,
690                        // which cannot be coerced
691                    }
692                    Expectation::rvalue_hint(self, exp_inner)
693                } else {
694                    Expectation::none()
695                };
696                let inner_ty = self.infer_expr_inner(*expr, &expectation, ExprIsRead::Yes);
697                match rawness {
698                    Rawness::RawPtr => Ty::new_ptr(self.interner(), inner_ty, mutability),
699                    Rawness::Ref => {
700                        let lt = self.table.next_region_var();
701                        Ty::new_ref(self.interner(), lt, inner_ty, mutability)
702                    }
703                }
704            }
705            &Expr::Box { expr } => self.infer_expr_box(expr, expected),
706            Expr::UnaryOp { expr, op } => self.infer_unop_expr(*op, *expr, expected, tgt_expr),
707            Expr::BinaryOp { lhs, rhs, op } => match op {
708                Some(BinaryOp::Assignment { op: Some(op) }) => {
709                    self.infer_assign_op_expr(tgt_expr, *op, *lhs, *rhs)
710                }
711                Some(op) => self.infer_binop_expr(tgt_expr, *op, *lhs, *rhs),
712                None => self.err_ty(),
713            },
714            &Expr::Assignment { target, value } => {
715                // In ordinary (non-destructuring) assignments, the type of
716                // `lhs` must be inferred first so that the ADT fields
717                // instantiations in RHS can be coerced to it. Note that this
718                // cannot happen in destructuring assignments because of how
719                // they are desugared.
720                let lhs_ty = match &self.body[target] {
721                    // LHS of assignment doesn't constitute reads.
722                    &Pat::Expr(expr) => {
723                        Some(self.infer_expr(expr, &Expectation::none(), ExprIsRead::No))
724                    }
725                    Pat::Path(path) => {
726                        let resolver_guard =
727                            self.resolver.update_to_inner_scope(self.db, self.owner, tgt_expr);
728                        let resolution = self.resolver.resolve_path_in_value_ns_fully(
729                            self.db,
730                            path,
731                            self.body.pat_path_hygiene(target),
732                        );
733                        self.resolver.reset_to_guard(resolver_guard);
734
735                        if matches!(
736                            resolution,
737                            Some(
738                                ValueNs::ConstId(_)
739                                    | ValueNs::StructId(_)
740                                    | ValueNs::EnumVariantId(_)
741                            )
742                        ) {
743                            None
744                        } else {
745                            Some(self.infer_expr_path(path, target.into(), tgt_expr))
746                        }
747                    }
748                    _ => None,
749                };
750                let is_destructuring_assignment = lhs_ty.is_none();
751
752                if let Some(lhs_ty) = lhs_ty {
753                    self.write_pat_ty(target, lhs_ty);
754                    self.infer_expr_coerce(value, &Expectation::has_type(lhs_ty), ExprIsRead::Yes);
755                } else {
756                    let rhs_ty = self.infer_expr(value, &Expectation::none(), ExprIsRead::Yes);
757                    let resolver_guard =
758                        self.resolver.update_to_inner_scope(self.db, self.owner, tgt_expr);
759                    self.inside_assignment = true;
760                    self.infer_top_pat(target, rhs_ty, None);
761                    self.inside_assignment = false;
762                    self.resolver.reset_to_guard(resolver_guard);
763                }
764                if is_destructuring_assignment && self.diverges.is_always() {
765                    // Ordinary assignments always return `()`, even when they diverge.
766                    // However, rustc lowers destructuring assignments into blocks, and blocks return `!` if they have no tail
767                    // expression and they diverge. Therefore, we have to do the same here, even though we don't lower destructuring
768                    // assignments into blocks.
769                    self.table.new_maybe_never_var()
770                } else {
771                    self.types.types.unit
772                }
773            }
774            Expr::Range { lhs, rhs, range_type } => {
775                let lhs_ty =
776                    lhs.map(|e| self.infer_expr_inner(e, &Expectation::none(), ExprIsRead::Yes));
777                let rhs_expect = lhs_ty.map_or_else(Expectation::none, Expectation::has_type);
778                let rhs_ty = rhs.map(|e| self.infer_expr(e, &rhs_expect, ExprIsRead::Yes));
779                let single_arg_adt = |adt, ty: Ty<'db>| {
780                    Ty::new_adt(
781                        self.interner(),
782                        adt,
783                        GenericArgs::new_from_slice(&[GenericArg::from(ty)]),
784                    )
785                };
786                match (range_type, lhs_ty, rhs_ty) {
787                    (RangeOp::Exclusive, None, None) => match self.resolve_range_full() {
788                        Some(adt) => {
789                            Ty::new_adt(self.interner(), adt, self.types.empty.generic_args)
790                        }
791                        None => self.err_ty(),
792                    },
793                    (RangeOp::Exclusive, None, Some(ty)) => match self.resolve_range_to() {
794                        Some(adt) => single_arg_adt(adt, ty),
795                        None => self.err_ty(),
796                    },
797                    (RangeOp::Inclusive, None, Some(ty)) => {
798                        match self.resolve_range_to_inclusive() {
799                            Some(adt) => single_arg_adt(adt, ty),
800                            None => self.err_ty(),
801                        }
802                    }
803                    (RangeOp::Exclusive, Some(_), Some(ty)) => match self.resolve_range() {
804                        Some(adt) => single_arg_adt(adt, ty),
805                        None => self.err_ty(),
806                    },
807                    (RangeOp::Inclusive, Some(_), Some(ty)) => {
808                        match self.resolve_range_inclusive() {
809                            Some(adt) => single_arg_adt(adt, ty),
810                            None => self.err_ty(),
811                        }
812                    }
813                    (RangeOp::Exclusive, Some(ty), None) => match self.resolve_range_from() {
814                        Some(adt) => single_arg_adt(adt, ty),
815                        None => self.err_ty(),
816                    },
817                    (RangeOp::Inclusive, _, None) => self.err_ty(),
818                }
819            }
820            Expr::Index { base, index } => {
821                let base_t = self.infer_expr_no_expect(*base, ExprIsRead::Yes);
822                let idx_t = self.infer_expr_no_expect(*index, ExprIsRead::Yes);
823
824                let base_t = self.table.structurally_resolve_type(base_t);
825                match self.lookup_indexing(tgt_expr, *base, base_t, idx_t) {
826                    Some((trait_index_ty, trait_element_ty)) => {
827                        // two-phase not needed because index_ty is never mutable
828                        self.demand_coerce(
829                            *index,
830                            idx_t,
831                            trait_index_ty,
832                            AllowTwoPhase::No,
833                            ExprIsRead::Yes,
834                        );
835                        self.table.select_obligations_where_possible();
836                        trait_element_ty
837                    }
838                    // FIXME: Report an error.
839                    None => self.types.types.error,
840                }
841            }
842            Expr::Tuple { exprs, .. } => {
843                let mut tys = match expected
844                    .only_has_type(&mut self.table)
845                    .map(|t| self.table.try_structurally_resolve_type(t).kind())
846                {
847                    Some(TyKind::Tuple(substs)) => substs
848                        .iter()
849                        .chain(repeat_with(|| self.table.next_ty_var()))
850                        .take(exprs.len())
851                        .collect::<Vec<_>>(),
852                    _ => (0..exprs.len()).map(|_| self.table.next_ty_var()).collect(),
853                };
854
855                for (expr, ty) in exprs.iter().zip(tys.iter_mut()) {
856                    *ty =
857                        self.infer_expr_coerce(*expr, &Expectation::has_type(*ty), ExprIsRead::Yes);
858                }
859
860                Ty::new_tup(self.interner(), &tys)
861            }
862            Expr::Array(array) => self.infer_expr_array(array, expected),
863            Expr::Literal(lit) => match lit {
864                Literal::Bool(..) => self.types.types.bool,
865                Literal::String(..) => self.types.types.static_str_ref,
866                Literal::ByteString(bs) => {
867                    let byte_type = self.types.types.u8;
868
869                    let len = consteval::usize_const(
870                        self.db,
871                        Some(bs.len() as u128),
872                        self.resolver.krate(),
873                    );
874
875                    let array_type = Ty::new_array_with_const_len(self.interner(), byte_type, len);
876                    Ty::new_ref(
877                        self.interner(),
878                        self.types.regions.statik,
879                        array_type,
880                        Mutability::Not,
881                    )
882                }
883                Literal::CString(..) => Ty::new_ref(
884                    self.interner(),
885                    self.types.regions.statik,
886                    self.lang_items.CStr.map_or_else(
887                        || self.err_ty(),
888                        |strukt| {
889                            Ty::new_adt(
890                                self.interner(),
891                                strukt.into(),
892                                self.types.empty.generic_args,
893                            )
894                        },
895                    ),
896                    Mutability::Not,
897                ),
898                Literal::Char(..) => self.types.types.char,
899                Literal::Int(_v, ty) => match ty {
900                    Some(int_ty) => match int_ty {
901                        hir_def::builtin_type::BuiltinInt::Isize => self.types.types.isize,
902                        hir_def::builtin_type::BuiltinInt::I8 => self.types.types.i8,
903                        hir_def::builtin_type::BuiltinInt::I16 => self.types.types.i16,
904                        hir_def::builtin_type::BuiltinInt::I32 => self.types.types.i32,
905                        hir_def::builtin_type::BuiltinInt::I64 => self.types.types.i64,
906                        hir_def::builtin_type::BuiltinInt::I128 => self.types.types.i128,
907                    },
908                    None => {
909                        let expected_ty = expected.to_option(&mut self.table);
910                        tracing::debug!(?expected_ty);
911                        let opt_ty = match expected_ty.as_ref().map(|it| it.kind()) {
912                            Some(TyKind::Int(_) | TyKind::Uint(_)) => expected_ty,
913                            Some(TyKind::Char) => Some(self.types.types.u8),
914                            Some(TyKind::RawPtr(..) | TyKind::FnDef(..) | TyKind::FnPtr(..)) => {
915                                Some(self.types.types.usize)
916                            }
917                            _ => None,
918                        };
919                        opt_ty.unwrap_or_else(|| self.table.next_int_var())
920                    }
921                },
922                Literal::Uint(_v, ty) => match ty {
923                    Some(int_ty) => match int_ty {
924                        hir_def::builtin_type::BuiltinUint::Usize => self.types.types.usize,
925                        hir_def::builtin_type::BuiltinUint::U8 => self.types.types.u8,
926                        hir_def::builtin_type::BuiltinUint::U16 => self.types.types.u16,
927                        hir_def::builtin_type::BuiltinUint::U32 => self.types.types.u32,
928                        hir_def::builtin_type::BuiltinUint::U64 => self.types.types.u64,
929                        hir_def::builtin_type::BuiltinUint::U128 => self.types.types.u128,
930                    },
931                    None => {
932                        let expected_ty = expected.to_option(&mut self.table);
933                        let opt_ty = match expected_ty.as_ref().map(|it| it.kind()) {
934                            Some(TyKind::Int(_) | TyKind::Uint(_)) => expected_ty,
935                            Some(TyKind::Char) => Some(self.types.types.u8),
936                            Some(TyKind::RawPtr(..) | TyKind::FnDef(..) | TyKind::FnPtr(..)) => {
937                                Some(self.types.types.usize)
938                            }
939                            _ => None,
940                        };
941                        opt_ty.unwrap_or_else(|| self.table.next_int_var())
942                    }
943                },
944                Literal::Float(_v, ty) => match ty {
945                    Some(float_ty) => match float_ty {
946                        hir_def::builtin_type::BuiltinFloat::F16 => self.types.types.f16,
947                        hir_def::builtin_type::BuiltinFloat::F32 => self.types.types.f32,
948                        hir_def::builtin_type::BuiltinFloat::F64 => self.types.types.f64,
949                        hir_def::builtin_type::BuiltinFloat::F128 => self.types.types.f128,
950                    },
951                    None => {
952                        let opt_ty = expected
953                            .to_option(&mut self.table)
954                            .filter(|ty| matches!(ty.kind(), TyKind::Float(_)));
955                        opt_ty.unwrap_or_else(|| self.table.next_float_var())
956                    }
957                },
958            },
959            Expr::Underscore => {
960                // Underscore expression is an error, we render a specialized diagnostic
961                // to let the user know what type is expected though.
962                let expected = expected.to_option(&mut self.table).unwrap_or_else(|| self.err_ty());
963                self.push_diagnostic(InferenceDiagnostic::TypedHole {
964                    expr: tgt_expr,
965                    expected: expected.store(),
966                });
967                expected
968            }
969            Expr::OffsetOf(_) => self.types.types.usize,
970            Expr::InlineAsm(asm) => {
971                let check_expr_asm_operand = |this: &mut Self, expr, is_input: bool| {
972                    let ty = this.infer_expr_no_expect(expr, ExprIsRead::Yes);
973
974                    // If this is an input value, we require its type to be fully resolved
975                    // at this point. This allows us to provide helpful coercions which help
976                    // pass the type candidate list in a later pass.
977                    //
978                    // We don't require output types to be resolved at this point, which
979                    // allows them to be inferred based on how they are used later in the
980                    // function.
981                    if is_input {
982                        let ty = this.table.structurally_resolve_type(ty);
983                        match ty.kind() {
984                            TyKind::FnDef(def, parameters) => {
985                                let fnptr_ty = Ty::new_fn_ptr(
986                                    this.interner(),
987                                    this.interner()
988                                        .fn_sig(def)
989                                        .instantiate(this.interner(), parameters),
990                                );
991                                _ = this.coerce(
992                                    expr.into(),
993                                    ty,
994                                    fnptr_ty,
995                                    AllowTwoPhase::No,
996                                    ExprIsRead::Yes,
997                                );
998                            }
999                            TyKind::Ref(_, base_ty, mutbl) => {
1000                                let ptr_ty = Ty::new_ptr(this.interner(), base_ty, mutbl);
1001                                _ = this.coerce(
1002                                    expr.into(),
1003                                    ty,
1004                                    ptr_ty,
1005                                    AllowTwoPhase::No,
1006                                    ExprIsRead::Yes,
1007                                );
1008                            }
1009                            _ => {}
1010                        }
1011                    }
1012                };
1013
1014                let diverge = asm.options.contains(AsmOptions::NORETURN);
1015                asm.operands.iter().for_each(|(_, operand)| match *operand {
1016                    AsmOperand::In { expr, .. } => check_expr_asm_operand(self, expr, true),
1017                    AsmOperand::Out { expr: Some(expr), .. } | AsmOperand::InOut { expr, .. } => {
1018                        check_expr_asm_operand(self, expr, false)
1019                    }
1020                    AsmOperand::Out { expr: None, .. } => (),
1021                    AsmOperand::SplitInOut { in_expr, out_expr, .. } => {
1022                        check_expr_asm_operand(self, in_expr, true);
1023                        if let Some(out_expr) = out_expr {
1024                            check_expr_asm_operand(self, out_expr, false);
1025                        }
1026                    }
1027                    AsmOperand::Label(expr) => {
1028                        self.infer_expr(
1029                            expr,
1030                            &Expectation::HasType(self.types.types.unit),
1031                            ExprIsRead::No,
1032                        );
1033                    }
1034                    AsmOperand::Const(expr) => {
1035                        self.infer_expr(expr, &Expectation::None, ExprIsRead::No);
1036                    }
1037                    // FIXME: `sym` should report for things that are not functions or statics.
1038                    AsmOperand::Sym(_) => (),
1039                });
1040                if diverge || asm.kind == InlineAsmKind::NakedAsm {
1041                    self.types.types.never
1042                } else {
1043                    self.types.types.unit
1044                }
1045            }
1046        };
1047        // use a new type variable if we got unknown here
1048        let ty = self.insert_type_vars_shallow(ty);
1049        self.write_expr_ty(tgt_expr, ty);
1050        if self.shallow_resolve(ty).is_never()
1051            && self.expr_guaranteed_to_constitute_read_for_never(tgt_expr, is_read)
1052        {
1053            // Any expression that produces a value of type `!` must have diverged
1054            self.diverges = Diverges::Always;
1055        }
1056        ty
1057    }

1059    fn demand_scrutinee_type(
1060        &mut self,
1061        scrut: ExprId,
1062        contains_ref_bindings: bool,
1063        no_arms: bool,
1064        scrutinee_is_read: ExprIsRead,
1065    ) -> Ty<'db> {
1066        // Not entirely obvious: if matches may create ref bindings, we want to
1067        // use the *precise* type of the scrutinee, *not* some supertype, as
1068        // the "scrutinee type" (issue #23116).
1069        //
1070        // arielb1 [writes here in this comment thread][c] that there
1071        // is certainly *some* potential danger, e.g., for an example
1072        // like:
1073        //
1074        // [c]: https://github.com/rust-lang/rust/pull/43399#discussion_r130223956
1075        //
1076        // ```
1077        // let Foo(x) = f()[0];
1078        // ```
1079        //
1080        // Then if the pattern matches by reference, we want to match
1081        // `f()[0]` as a lexpr, so we can't allow it to be
1082        // coerced. But if the pattern matches by value, `f()[0]` is
1083        // still syntactically a lexpr, but we *do* want to allow
1084        // coercions.
1085        //
1086        // However, *likely* we are ok with allowing coercions to
1087        // happen if there are no explicit ref mut patterns - all
1088        // implicit ref mut patterns must occur behind a reference, so
1089        // they will have the "correct" variance and lifetime.
1090        //
1091        // This does mean that the following pattern would be legal:
1092        //
1093        // ```
1094        // struct Foo(Bar);
1095        // struct Bar(u32);
1096        // impl Deref for Foo {
1097        //     type Target = Bar;
1098        //     fn deref(&self) -> &Bar { &self.0 }
1099        // }
1100        // impl DerefMut for Foo {
1101        //     fn deref_mut(&mut self) -> &mut Bar { &mut self.0 }
1102        // }
1103        // fn foo(x: &mut Foo) {
1104        //     {
1105        //         let Bar(z): &mut Bar = x;
1106        //         *z = 42;
1107        //     }
1108        //     assert_eq!(foo.0.0, 42);
1109        // }
1110        // ```
1111        //
1112        // FIXME(tschottdorf): don't call contains_explicit_ref_binding, which
1113        // is problematic as the HIR is being scraped, but ref bindings may be
1114        // implicit after #42640. We need to make sure that pat_adjustments
1115        // (once introduced) is populated by the time we get here.
1116        //
1117        // See #44848.
1118        if contains_ref_bindings || no_arms {
1119            self.infer_expr_no_expect(scrut, scrutinee_is_read)
1120        } else {
1121            // ...but otherwise we want to use any supertype of the
1122            // scrutinee. This is sort of a workaround, see note (*) in
1123            // `check_pat` for some details.
1124            let scrut_ty = self.table.next_ty_var();
1125            self.infer_expr_coerce_never(scrut, &Expectation::HasType(scrut_ty), scrutinee_is_read);
1126            scrut_ty
1127        }
1128    }

1130    fn infer_expr_path(&mut self, path: &Path, id: ExprOrPatId, scope_id: ExprId) -> Ty<'db> {
1131        let g = self.resolver.update_to_inner_scope(self.db, self.owner, scope_id);
1132        let ty = match self.infer_path(path, id) {
1133            Some(ty) => ty,
1134            None => {
1135                if path.mod_path().is_some_and(|mod_path| mod_path.is_ident() || mod_path.is_self())
1136                {
1137                    self.push_diagnostic(InferenceDiagnostic::UnresolvedIdent { id });
1138                }
1139                self.err_ty()
1140            }
1141        };
1142        self.resolver.reset_to_guard(g);
1143        ty
1144    }

1146    fn infer_unop_expr(
1147        &mut self,
1148        unop: UnaryOp,
1149        oprnd: ExprId,
1150        expected: &Expectation<'db>,
1151        expr: ExprId,
1152    ) -> Ty<'db> {
1153        let expected_inner = match unop {
1154            UnaryOp::Not | UnaryOp::Neg => expected,
1155            UnaryOp::Deref => &Expectation::None,
1156        };
1157        let mut oprnd_t = self.infer_expr_inner(oprnd, expected_inner, ExprIsRead::Yes);
1158
1159        oprnd_t = self.table.structurally_resolve_type(oprnd_t);
1160        match unop {
1161            UnaryOp::Deref => {
1162                if let Some(ty) = self.lookup_derefing(expr, oprnd, oprnd_t) {
1163                    oprnd_t = ty;
1164                } else {
1165                    // FIXME: Report an error.
1166                    oprnd_t = self.types.types.error;
1167                }
1168            }
1169            UnaryOp::Not => {
1170                let result = self.infer_user_unop(expr, oprnd_t, unop);
1171                // If it's builtin, we can reuse the type, this helps inference.
1172                if !(oprnd_t.is_integral() || oprnd_t.kind() == TyKind::Bool) {
1173                    oprnd_t = result;
1174                }
1175            }
1176            UnaryOp::Neg => {
1177                let result = self.infer_user_unop(expr, oprnd_t, unop);
1178                // If it's builtin, we can reuse the type, this helps inference.
1179                if !oprnd_t.is_numeric() {
1180                    oprnd_t = result;
1181                }
1182            }
1183        }
1184        oprnd_t
1185    }

1187    fn infer_async_block(
1188        &mut self,
1189        tgt_expr: ExprId,
1190        statements: &[Statement],
1191        tail: &Option<ExprId>,
1192    ) -> Ty<'db> {
1193        let ret_ty = self.table.next_ty_var();
1194        let prev_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
1195        let prev_ret_ty = mem::replace(&mut self.return_ty, ret_ty);

1197

1199        let expected = &Expectation::has_type(ret_ty);
1200        let (_, inner_ty) = self.with_breakable_ctx(BreakableKind::Border, None, None, |this| {
1201            let ty = this.infer_block(tgt_expr, statements, *tail, None, expected);
1202            if let Some(target) = expected.only_has_type(&mut this.table) {
1203                match this.coerce(tgt_expr.into(), ty, target, AllowTwoPhase::No, ExprIsRead::Yes) {
1204                    Ok(res) => res,
1205                    Err(_) => {
1206                        this.result.type_mismatches.get_or_insert_default().insert(
1207                            tgt_expr.into(),

1209                        );
1210                        target
1211                    }
1212                }
1213            } else {
1214                ty
1215            }
1216        });
1217
1218        self.diverges = prev_diverges;
1219        self.return_ty = prev_ret_ty;
1220        self.return_coercion = prev_ret_coercion;

1253    pub(crate) fn write_fn_trait_method_resolution(
1254        &mut self,
1255        fn_x: FnTrait,
1256        derefed_callee: Ty<'db>,
1257        adjustments: &mut Vec<Adjustment>,
1258        callee_ty: Ty<'db>,
1259        params: &[Ty<'db>],
1260        tgt_expr: ExprId,
1261    ) {
1262        match fn_x {
1263            FnTrait::FnOnce | FnTrait::AsyncFnOnce => (),
1264            FnTrait::FnMut | FnTrait::AsyncFnMut => {
1265                if let TyKind::Ref(lt, inner, Mutability::Mut) = derefed_callee.kind() {
1266                    if adjustments
1267                        .last()
1268                        .map(|it| matches!(it.kind, Adjust::Borrow(_)))
1269                        .unwrap_or(true)
1270                    {
1271                        // prefer reborrow to move
1272                        adjustments
1273                            .push(Adjustment { kind: Adjust::Deref(None), target: inner.store() });
1274                        adjustments.push(Adjustment::borrow(
1275                            self.interner(),
1276                            Mutability::Mut,
1277                            inner,
1278                            lt,
1279                        ))
1280                    }
1281                } else {
1282                    adjustments.push(Adjustment::borrow(
1283                        self.interner(),
1284                        Mutability::Mut,
1285                        derefed_callee,
1286                        self.table.next_region_var(),
1287                    ));
1288                }
1289            }
1290            FnTrait::Fn | FnTrait::AsyncFn => {
1291                if !matches!(derefed_callee.kind(), TyKind::Ref(_, _, Mutability::Not)) {
1292                    adjustments.push(Adjustment::borrow(
1293                        self.interner(),
1294                        Mutability::Not,
1295                        derefed_callee,
1296                        self.table.next_region_var(),
1297                    ));
1298                }
1299            }
1300        }
1301        let Some(trait_) = fn_x.get_id(self.lang_items) else {
1302            return;
1303        };
1304        let trait_data = trait_.trait_items(self.db);
1305        if let Some(func) = trait_data.method_by_name(&fn_x.method_name()) {
1306            let subst = GenericArgs::new_from_slice(&[
1307                callee_ty.into(),
1308                Ty::new_tup(self.interner(), params).into(),
1309            ]);
1310            self.write_method_resolution(tgt_expr, func, subst);
1311        }
1312    }

1314    fn infer_expr_array(&mut self, array: &Array, expected: &Expectation<'db>) -> Ty<'db> {
1315        let elem_ty = match expected
1316            .to_option(&mut self.table)
1317            .map(|t| self.table.try_structurally_resolve_type(t).kind())
1318        {
1319            Some(TyKind::Array(st, _) | TyKind::Slice(st)) => st,
1320            _ => self.table.next_ty_var(),
1321        };
1322
1323        let krate = self.resolver.krate();
1324
1325        let expected = Expectation::has_type(elem_ty);
1326        let (elem_ty, len) = match array {
1327            Array::ElementList { elements, .. } if elements.is_empty() => {
1328                (elem_ty, consteval::usize_const(self.db, Some(0), krate))
1329            }
1330            Array::ElementList { elements, .. } => {
1331                let mut coerce = CoerceMany::with_coercion_sites(elem_ty, elements);
1332                for &expr in elements.iter() {
1333                    let cur_elem_ty = self.infer_expr_inner(expr, &expected, ExprIsRead::Yes);
1334                    coerce.coerce(
1335                        self,
1336                        &ObligationCause::new(),
1337                        expr,
1338                        cur_elem_ty,
1339                        ExprIsRead::Yes,
1340                    );
1341                }
1342                (
1343                    coerce.complete(self),
1344                    consteval::usize_const(self.db, Some(elements.len() as u128), krate),
1345                )
1346            }
1347            &Array::Repeat { initializer, repeat } => {
1348                self.infer_expr_coerce(
1349                    initializer,
1350                    &Expectation::has_type(elem_ty),
1351                    ExprIsRead::Yes,
1352                );
1353                let usize = self.types.types.usize;
1354                let len = match self.body[repeat] {
1355                    Expr::Underscore => {
1356                        self.write_expr_ty(repeat, usize);
1357                        self.table.next_const_var()
1358                    }
1359                    _ => {
1360                        self.infer_expr(repeat, &Expectation::HasType(usize), ExprIsRead::Yes);
1361                        consteval::eval_to_const(repeat, self)
1362                    }
1363                };
1364
1365                (elem_ty, len)
1366            }
1367        };
1368        // Try to evaluate unevaluated constant, and insert variable if is not possible.
1369        let len = self.table.insert_const_vars_shallow(len);
1370        Ty::new_array_with_const_len(self.interner(), elem_ty, len)
1371    }

1464    fn infer_block(
1465        &mut self,
1466        expr: ExprId,
1467        statements: &[Statement],
1468        tail: Option<ExprId>,
1469        label: Option<LabelId>,
1470        expected: &Expectation<'db>,
1471    ) -> Ty<'db> {
1472        let coerce_ty = expected.coercion_target_type(&mut self.table);
1473        let g = self.resolver.update_to_inner_scope(self.db, self.owner, expr);
1474
1475        let (break_ty, ty) =
1476            self.with_breakable_ctx(BreakableKind::Block, Some(coerce_ty), label, |this| {
1477                for stmt in statements {
1478                    match stmt {
1479                        Statement::Let { pat, type_ref, initializer, else_branch } => {
1480                            let decl_ty = type_ref
1481                                .as_ref()
1482                                .map(|&tr| this.make_body_ty(tr))
1483                                .unwrap_or_else(|| this.table.next_ty_var());
1484
1485                            let ty = if let Some(expr) = initializer {
1486                                // If we have a subpattern that performs a read, we want to consider this
1487                                // to diverge for compatibility to support something like `let x: () = *never_ptr;`.
1488                                let target_is_read =
1489                                    if this.pat_guaranteed_to_constitute_read_for_never(*pat) {
1490                                        ExprIsRead::Yes
1491                                    } else {
1492                                        ExprIsRead::No
1493                                    };
1494                                let ty = if contains_explicit_ref_binding(this.body, *pat) {
1495                                    this.infer_expr(
1496                                        *expr,
1497                                        &Expectation::has_type(decl_ty),
1498                                        target_is_read,
1499                                    )
1500                                } else {
1501                                    this.infer_expr_coerce(
1502                                        *expr,
1503                                        &Expectation::has_type(decl_ty),
1504                                        target_is_read,
1505                                    )
1506                                };
1507                                if type_ref.is_some() { decl_ty } else { ty }
1508                            } else {
1509                                decl_ty
1510                            };
1511
1512                            let decl = DeclContext {
1513                                origin: DeclOrigin::LocalDecl { has_else: else_branch.is_some() },
1514                            };
1515
1516                            this.infer_top_pat(*pat, ty, Some(decl));
1517                            if let Some(expr) = else_branch {
1518                                let previous_diverges =
1519                                    mem::replace(&mut this.diverges, Diverges::Maybe);
1520                                this.infer_expr_coerce(
1521                                    *expr,
1522                                    &Expectation::HasType(this.types.types.never),
1523                                    ExprIsRead::Yes,
1524                                );
1525                                this.diverges = previous_diverges;
1526                            }
1527                        }
1528                        &Statement::Expr { expr, has_semi } => {
1529                            if has_semi {
1530                                this.infer_expr(expr, &Expectation::none(), ExprIsRead::Yes);
1531                            } else {
1532                                this.infer_expr_coerce(
1533                                    expr,
1534                                    &Expectation::HasType(this.types.types.unit),
1535                                    ExprIsRead::Yes,
1536                                );
1537                            }
1538                        }
1539                        Statement::Item(_) => (),
1540                    }
1541                }
1542
1543                // FIXME: This should make use of the breakable CoerceMany
1544                if let Some(expr) = tail {
1545                    this.infer_expr_coerce(expr, expected, ExprIsRead::Yes)
1546                } else {
1547                    // Citing rustc: if there is no explicit tail expression,
1548                    // that is typically equivalent to a tail expression
1549                    // of `()` -- except if the block diverges. In that
1550                    // case, there is no value supplied from the tail
1551                    // expression (assuming there are no other breaks,
1552                    // this implies that the type of the block will be
1553                    // `!`).
1554                    if this.diverges.is_always() {
1555                        // we don't even make an attempt at coercion
1556                        this.table.new_maybe_never_var()
1557                    } else if let Some(t) = expected.only_has_type(&mut this.table) {
1558                        if this
1559                            .coerce(
1560                                expr.into(),
1561                                this.types.types.unit,
1562                                t,
1563                                AllowTwoPhase::No,
1564                                ExprIsRead::Yes,
1565                            )
1566                            .is_err()
1567                        {
1568                            this.result.type_mismatches.get_or_insert_default().insert(
1569                                expr.into(),
1570                                TypeMismatch {
1571                                    expected: t.store(),
1572                                    actual: this.types.types.unit.store(),
1573                                },
1574                            );
1575                        }
1576                        t
1577                    } else {
1578                        this.types.types.unit
1579                    }
1580                }
1581            });
1582        self.resolver.reset_to_guard(g);
1583
1584        break_ty.unwrap_or(ty)
1585    }

1587    fn lookup_field(
1588        &mut self,
1589        receiver_ty: Ty<'db>,
1590        name: &Name,
1591    ) -> Option<(Ty<'db>, Either<FieldId, TupleFieldId>, Vec<Adjustment>, bool)> {
1592        let interner = self.interner();
1593        let mut autoderef = self.table.autoderef_with_tracking(receiver_ty);
1594        let mut private_field = None;
1595        let res = autoderef.by_ref().find_map(|(derefed_ty, _)| {
1596            let (field_id, parameters) = match derefed_ty.kind() {
1597                TyKind::Tuple(substs) => {
1598                    return name.as_tuple_index().and_then(|idx| {
1599                        substs.as_slice().get(idx).copied().map(|ty| {
1600                            (
1601                                Either::Right(TupleFieldId {
1602                                    tuple: TupleId(
1603                                        self.tuple_field_accesses_rev.insert_full(substs).0 as u32,
1604                                    ),
1605                                    index: idx as u32,
1606                                }),
1607                                ty,
1608                            )
1609                        })
1610                    });
1611                }
1612                TyKind::Adt(adt, parameters) => match adt.def_id().0 {
1613                    hir_def::AdtId::StructId(s) => {
1614                        let local_id = s.fields(self.db).field(name)?;
1615                        let field = FieldId { parent: s.into(), local_id };
1616                        (field, parameters)
1617                    }
1618                    hir_def::AdtId::UnionId(u) => {
1619                        let local_id = u.fields(self.db).field(name)?;
1620                        let field = FieldId { parent: u.into(), local_id };
1621                        (field, parameters)
1622                    }
1623                    hir_def::AdtId::EnumId(_) => return None,
1624                },
1625                _ => return None,
1626            };
1627            let is_visible = self.db.field_visibilities(field_id.parent)[field_id.local_id]
1628                .is_visible_from(self.db, self.resolver.module());
1629            if !is_visible {
1630                if private_field.is_none() {
1631                    private_field = Some((field_id, parameters));
1632                }
1633                return None;
1634            }
1635            let ty = self.db.field_types(field_id.parent)[field_id.local_id]
1636                .get()
1637                .instantiate(interner, parameters);
1638            Some((Either::Left(field_id), ty))
1639        });
1640
1641        Some(match res {
1642            Some((field_id, ty)) => {
1643                let adjustments =
1644                    self.table.register_infer_ok(autoderef.adjust_steps_as_infer_ok());
1645                let ty = self.process_remote_user_written_ty(ty);
1646
1647                (ty, field_id, adjustments, true)
1648            }
1649            None => {
1650                let (field_id, subst) = private_field?;
1651                let adjustments =
1652                    self.table.register_infer_ok(autoderef.adjust_steps_as_infer_ok());
1653                let ty = self.db.field_types(field_id.parent)[field_id.local_id]
1654                    .get()
1655                    .instantiate(self.interner(), subst);
1656                let ty = self.process_remote_user_written_ty(ty);
1657
1658                (ty, Either::Left(field_id), adjustments, false)
1659            }
1660        })
1661    }

1663    fn infer_field_access(
1664        &mut self,
1665        tgt_expr: ExprId,
1666        receiver: ExprId,
1667        name: &Name,
1668        expected: &Expectation<'db>,
1669    ) -> Ty<'db> {
1670        // Field projections don't constitute reads.
1671        let receiver_ty = self.infer_expr_inner(receiver, &Expectation::none(), ExprIsRead::No);
1672        let receiver_ty = self.table.structurally_resolve_type(receiver_ty);
1673
1674        if name.is_missing() {
1675            // Bail out early, don't even try to look up field. Also, we don't issue an unresolved
1676            // field diagnostic because this is a syntax error rather than a semantic error.
1677            return self.err_ty();
1678        }
1679
1680        match self.lookup_field(receiver_ty, name) {
1681            Some((ty, field_id, adjustments, is_public)) => {

1687                        expr: tgt_expr,
1688                        field,
1689                    });
1690                }
1691                ty
1692            }
1693            None => {
1694                // no field found, lets attempt to resolve it like a function so that IDE things
1695                // work out while people are typing
1696                let resolved = self.lookup_method_including_private(
1697                    receiver_ty,
1698                    name.clone(),
1699                    None,
1700                    receiver,
1701                    tgt_expr,
1702                );
1703                self.push_diagnostic(InferenceDiagnostic::UnresolvedField {
1704                    expr: tgt_expr,
1705                    receiver: receiver_ty.store(),
1706                    name: name.clone(),
1707                    method_with_same_name_exists: resolved.is_ok(),
1708                });
1709                match resolved {
1710                    Ok((func, _is_visible)) => {
1711                        self.check_method_call(tgt_expr, &[], func.sig, expected)
1712                    }
1713                    Err(_) => self.err_ty(),
1714                }
1715            }
1716        }
1717    }

1729    fn infer_call(
1730        &mut self,
1731        tgt_expr: ExprId,
1732        callee: ExprId,
1733        args: &[ExprId],
1734        expected: &Expectation<'db>,
1735    ) -> Ty<'db> {
1736        let callee_ty = self.infer_expr(callee, &Expectation::none(), ExprIsRead::Yes);
1737        let callee_ty = self.table.try_structurally_resolve_type(callee_ty);
1738        let interner = self.interner();
1739        let mut derefs = InferenceContextAutoderef::new_from_inference_context(self, callee_ty);
1740        let (res, derefed_callee) = loop {
1741            let Some((callee_deref_ty, _)) = derefs.next() else {
1742                break (None, callee_ty);
1743            };
1744            if let Some(res) = derefs.ctx().table.callable_sig(callee_deref_ty, args.len()) {
1745                break (Some(res), callee_deref_ty);
1746            }
1747        };
1748        // if the function is unresolved, we use is_varargs=true to
1749        // suppress the arg count diagnostic here
1750        let is_varargs = derefed_callee.callable_sig(interner).is_some_and(|sig| sig.c_variadic())
1751            || res.is_none();
1752        let (param_tys, ret_ty) = match res {
1753            Some((func, params, ret_ty)) => {
1754                let infer_ok = derefs.adjust_steps_as_infer_ok();
1755                let mut adjustments = self.table.register_infer_ok(infer_ok);
1756                if let Some(fn_x) = func {
1757                    self.write_fn_trait_method_resolution(
1758                        fn_x,
1759                        derefed_callee,
1760                        &mut adjustments,
1761                        callee_ty,
1762                        &params,
1763                        tgt_expr,
1764                    );
1765                }
1766                if let TyKind::Closure(c, _) = self.table.resolve_completely(callee_ty).kind() {
1767                    self.add_current_closure_dependency(c.into());
1768                    self.deferred_closures.entry(c.into()).or_default().push((
1769                        derefed_callee,
1770                        callee_ty,
1771                        params.clone(),
1772                        tgt_expr,
1773                    ));
1774                }
1775                self.write_expr_adj(callee, adjustments.into_boxed_slice());
1776                (params, ret_ty)
1777            }
1778            None => {
1779                self.push_diagnostic(InferenceDiagnostic::ExpectedFunction {
1780                    call_expr: tgt_expr,
1781                    found: callee_ty.store(),
1782                });
1783                (Vec::new(), Ty::new_error(interner, ErrorGuaranteed))
1784            }
1785        };
1786        let indices_to_skip = self.check_legacy_const_generics(derefed_callee, args);
1787        self.check_call(
1788            tgt_expr,
1789            args,
1790            callee_ty,
1791            &param_tys,
1792            ret_ty,
1793            &indices_to_skip,
1794            is_varargs,
1795            expected,
1796        )
1797    }

1824    fn infer_method_call(
1825        &mut self,
1826        tgt_expr: ExprId,
1827        receiver: ExprId,
1828        args: &[ExprId],
1829        method_name: &Name,
1830        generic_args: Option<&HirGenericArgs>,
1831        expected: &Expectation<'db>,
1832    ) -> Ty<'db> {
1833        let receiver_ty = self.infer_expr_inner(receiver, &Expectation::none(), ExprIsRead::Yes);
1834        let receiver_ty = self.table.try_structurally_resolve_type(receiver_ty);
1835
1836        let resolved = self.lookup_method_including_private(
1837            receiver_ty,
1838            method_name.clone(),
1839            generic_args,
1840            receiver,
1841            tgt_expr,
1842        );
1843        match resolved {
1844            Ok((func, visible)) => {
1845                if !visible {
1846                    self.push_diagnostic(InferenceDiagnostic::PrivateAssocItem {
1847                        id: tgt_expr.into(),
1848                        item: func.def_id.into(),
1849                    })
1850                }
1851                self.check_method_call(tgt_expr, args, func.sig, expected)
1852            }
1853            // Failed to resolve, report diagnostic and try to resolve as call to field access or
1854            // assoc function
1855            Err(_) => {
1856                let field_with_same_name_exists = match self.lookup_field(receiver_ty, method_name)
1857                {
1858                    Some((ty, field_id, adjustments, _public)) => {
1859                        self.write_expr_adj(receiver, adjustments.into_boxed_slice());
1860                        self.result.field_resolutions.insert(tgt_expr, field_id);
1861                        Some(ty)
1862                    }
1863                    None => None,
1864                };
1865
1866                let assoc_func_with_same_name = self.with_method_resolution(|ctx| {
1867                    if !matches!(
1868                        receiver_ty.kind(),
1869                        TyKind::Infer(InferTy::TyVar(_)) | TyKind::Error(_)
1870                    ) {
1871                        ctx.probe_for_name(
1872                            method_resolution::Mode::Path,
1873                            method_name.clone(),
1874                            receiver_ty,
1875                        )
1876                    } else {
1877                        Err(MethodError::ErrorReported)
1878                    }
1879                });
1880                let assoc_func_with_same_name = match assoc_func_with_same_name {
1881                    Ok(method_resolution::Pick {
1882                        item: CandidateId::FunctionId(def_id), ..
1883                    })
1884                    | Err(MethodError::PrivateMatch(method_resolution::Pick {
1885                        item: CandidateId::FunctionId(def_id),
1886                        ..
1887                    })) => Some(self.instantiate_erroneous_method(def_id)),
1888                    _ => None,
1889                };
1890
1891                self.push_diagnostic(InferenceDiagnostic::UnresolvedMethodCall {
1892                    expr: tgt_expr,
1893                    receiver: receiver_ty.store(),
1894                    name: method_name.clone(),
1895                    field_with_same_name: field_with_same_name_exists.map(|it| it.store()),
1896                    assoc_func_with_same_name: assoc_func_with_same_name.map(|it| it.def_id),
1897                });
1898
1899                let recovered = match assoc_func_with_same_name {
1900                    Some(it) => Some((
1901                        Ty::new_fn_def(
1902                            self.interner(),
1903                            CallableDefId::FunctionId(it.def_id).into(),
1904                            it.args,
1905                        ),
1906                        it.sig,
1907                        true,
1908                    )),
1909                    None => field_with_same_name_exists.and_then(|field_ty| {
1910                        let callable_sig = field_ty.callable_sig(self.interner())?;
1911                        Some((field_ty, callable_sig.skip_binder(), false))
1912                    }),
1913                };
1914                match recovered {
1915                    Some((callee_ty, sig, strip_first)) => self.check_call(
1916                        tgt_expr,
1917                        args,
1918                        callee_ty,
1919                        sig.inputs_and_output.inputs().get(strip_first as usize..).unwrap_or(&[]),
1920                        sig.output(),
1921                        &[],
1922                        true,
1923                        expected,
1924                    ),
1925                    None => {
1926                        for &arg in args.iter() {
1927                            self.infer_expr_no_expect(arg, ExprIsRead::Yes);
1928                        }
1929                        self.err_ty()
1930                    }
1931                }
1932            }
1933        }
1934    }

1954    /// Generic function that factors out common logic from function calls,
1955    /// method calls and overloaded operators.
1956    pub(in super::super) fn check_call_arguments(
1957        &mut self,
1958        call_expr: ExprId,
1959        // Types (as defined in the *signature* of the target function)
1960        formal_input_tys: &[Ty<'db>],
1961        formal_output: Ty<'db>,
1962        // Expected output from the parent expression or statement
1963        expectation: &Expectation<'db>,
1964        // The expressions for each provided argument
1965        provided_args: &[ExprId],
1966        skip_indices: &[u32],
1967        // Whether the function is variadic, for example when imported from C
1968        c_variadic: bool,
1969    ) {
1970        // First, let's unify the formal method signature with the expectation eagerly.
1971        // We use this to guide coercion inference; it's output is "fudged" which means
1972        // any remaining type variables are assigned to new, unrelated variables. This
1973        // is because the inference guidance here is only speculative.
1974        let formal_output = self.table.resolve_vars_with_obligations(formal_output);
1975        let expected_input_tys: Option<Vec<_>> = expectation
1976            .only_has_type(&mut self.table)
1977            .and_then(|expected_output| {
1978                self.table
1979                    .infer_ctxt
1980                    .fudge_inference_if_ok(|| {
1981                        let mut ocx = ObligationCtxt::new(&self.table.infer_ctxt);
1982
1983                        // Attempt to apply a subtyping relationship between the formal
1984                        // return type (likely containing type variables if the function
1985                        // is polymorphic) and the expected return type.
1986                        // No argument expectations are produced if unification fails.
1987                        let origin = ObligationCause::new();
1988                        ocx.sup(&origin, self.table.param_env, expected_output, formal_output)?;
1989                        if !ocx.try_evaluate_obligations().is_empty() {
1990                            return Err(TypeError::Mismatch);
1991                        }
1992
1993                        // Record all the argument types, with the args
1994                        // produced from the above subtyping unification.
1995                        Ok(Some(
1996                            formal_input_tys
1997                                .iter()
1998                                .map(|&ty| self.table.infer_ctxt.resolve_vars_if_possible(ty))
1999                                .collect(),
2000                        ))
2001                    })
2002                    .ok()
2003            })
2004            .unwrap_or_default();
2005
2006        // If there are no external expectations at the call site, just use the types from the function defn
2007        let expected_input_tys = if let Some(expected_input_tys) = &expected_input_tys {
2008            assert_eq!(expected_input_tys.len(), formal_input_tys.len());
2009            expected_input_tys
2010        } else {
2011            formal_input_tys
2012        };
2013
2014        let minimum_input_count = expected_input_tys.len();
2015        let provided_arg_count = provided_args.len() - skip_indices.len();
2016
2017        // Keep track of whether we *could possibly* be satisfied, i.e. whether we're on the happy path
2018        // if the wrong number of arguments were supplied, we CAN'T be satisfied,
2019        // and if we're c_variadic, the supplied arguments must be >= the minimum count from the function
2020        // otherwise, they need to be identical, because rust doesn't currently support variadic functions
2021        let args_count_matches = if c_variadic {
2022            provided_arg_count >= minimum_input_count
2023        } else {
2024            provided_arg_count == minimum_input_count
2025        };
2026
2027        if !args_count_matches {
2028            self.push_diagnostic(InferenceDiagnostic::MismatchedArgCount {
2029                call_expr,
2030                expected: expected_input_tys.len() + skip_indices.len(),
2031                found: provided_args.len(),
2032            });
2033        }
2034
2035        // We introduce a helper function to demand that a given argument satisfy a given input
2036        // This is more complicated than just checking type equality, as arguments could be coerced
2037        // This version writes those types back so further type checking uses the narrowed types
2038        let demand_compatible = |this: &mut InferenceContext<'_, 'db>, idx| {
2039            let formal_input_ty: Ty<'db> = formal_input_tys[idx];
2040            let expected_input_ty: Ty<'db> = expected_input_tys[idx];
2041            let provided_arg = provided_args[idx];
2042
2043            debug!("checking argument {}: {:?} = {:?}", idx, provided_arg, formal_input_ty);
2044
2045            // We're on the happy path here, so we'll do a more involved check and write back types
2046            // To check compatibility, we'll do 3 things:
2047            // 1. Unify the provided argument with the expected type
2048            let expectation = Expectation::rvalue_hint(this, expected_input_ty);
2049
2050            let checked_ty = this.infer_expr_inner(provided_arg, &expectation, ExprIsRead::Yes);
2051
2052            // 2. Coerce to the most detailed type that could be coerced
2053            //    to, which is `expected_ty` if `rvalue_hint` returns an
2054            //    `ExpectHasType(expected_ty)`, or the `formal_ty` otherwise.
2055            let coerced_ty = expectation.only_has_type(&mut this.table).unwrap_or(formal_input_ty);
2056
2057            // Cause selection errors caused by resolving a single argument to point at the
2058            // argument and not the call. This lets us customize the span pointed to in the
2059            // fulfillment error to be more accurate.
2060            let coerced_ty = this.table.resolve_vars_with_obligations(coerced_ty);
2061
2062            let coerce_error = this
2063                .coerce(
2064                    provided_arg.into(),
2065                    checked_ty,
2066                    coerced_ty,
2067                    AllowTwoPhase::Yes,
2068                    ExprIsRead::Yes,
2069                )
2070                .err();
2071            if coerce_error.is_some() {
2072                return Err((coerce_error, coerced_ty, checked_ty));
2073            }
2074
2075            // 3. Check if the formal type is actually equal to the checked one
2076            //    and register any such obligations for future type checks.
2077            let formal_ty_error = this
2078                .table
2079                .infer_ctxt
2080                .at(&ObligationCause::new(), this.table.param_env)
2081                .eq(formal_input_ty, coerced_ty);
2082
2083            // If neither check failed, the types are compatible
2084            match formal_ty_error {
2085                Ok(InferOk { obligations, value: () }) => {
2086                    this.table.register_predicates(obligations);
2087                    Ok(())
2088                }
2089                Err(err) => Err((Some(err), coerced_ty, checked_ty)),
2090            }
2091        };
2092
2093        // Check the arguments.
2094        // We do this in a pretty awful way: first we type-check any arguments
2095        // that are not closures, then we type-check the closures. This is so
2096        // that we have more information about the types of arguments when we
2097        // type-check the functions. This isn't really the right way to do this.
2098        for check_closures in [false, true] {
2099            // More awful hacks: before we check argument types, try to do
2100            // an "opportunistic" trait resolution of any trait bounds on
2101            // the call. This helps coercions.
2102            if check_closures {
2103                self.table.select_obligations_where_possible();
2104            }
2105
2106            let mut skip_indices = skip_indices.iter().copied();
2107            // Check each argument, to satisfy the input it was provided for
2108            // Visually, we're traveling down the diagonal of the compatibility matrix
2109            for (idx, arg) in provided_args.iter().enumerate() {
2110                if skip_indices.clone().next() == Some(idx as u32) {
2111                    skip_indices.next();
2112                    continue;
2113                }
2114
2115                // For this check, we do *not* want to treat async coroutine closures (async blocks)
2116                // as proper closures. Doing so would regress type inference when feeding
2117                // the return value of an argument-position async block to an argument-position
2118                // closure wrapped in a block.
2119                // See <https://github.com/rust-lang/rust/issues/112225>.
2120                let is_closure = if let Expr::Closure { closure_kind, .. } = self.body[*arg] {
2121                    !matches!(closure_kind, ClosureKind::Coroutine(_))
2122                } else {
2123                    false
2124                };
2125                if is_closure != check_closures {
2126                    continue;
2127                }
2128
2129                if idx >= minimum_input_count {
2130                    // Make sure we've checked this expr at least once.
2131                    self.infer_expr_no_expect(*arg, ExprIsRead::Yes);
2132                    continue;
2133                }
2134
2135                if let Err((_error, expected, found)) = demand_compatible(self, idx)
2136                    && args_count_matches
2137                {
2138                    // Don't report type mismatches if there is a mismatch in args count.
2139                    self.result.type_mismatches.get_or_insert_default().insert(
2140                        (*arg).into(),
2141                        TypeMismatch { expected: expected.store(), actual: found.store() },
2142                    );
2143                }
2144            }
2145        }
2146
2147        if !args_count_matches {}
2148    }

2150    fn register_obligations_for_call(&mut self, callable_ty: Ty<'db>) {
2151        let callable_ty = self.table.try_structurally_resolve_type(callable_ty);
2152        if let TyKind::FnDef(fn_def, parameters) = callable_ty.kind() {
2153            let generic_predicates = GenericPredicates::query_all(
2154                self.db,
2155                GenericDefId::from_callable(self.db, fn_def.0),
2156            );
2157            let param_env = self.table.param_env;
2158            self.table.register_predicates(clauses_as_obligations(
2159                generic_predicates.iter_instantiated_copied(self.interner(), parameters.as_slice()),
2160                ObligationCause::new(),
2161                param_env,
2162            ));
2163            // add obligation for trait implementation, if this is a trait method
2164            match fn_def.0 {
2165                CallableDefId::FunctionId(f) => {
2166                    if let ItemContainerId::TraitId(trait_) = f.lookup(self.db).container {
2167                        // construct a TraitRef
2168                        let trait_params_len = generics(self.db, trait_.into()).len();
2169                        let substs =
2170                            GenericArgs::new_from_slice(&parameters.as_slice()[..trait_params_len]);
2171                        self.table.register_predicate(Obligation::new(
2172                            self.interner(),
2173                            ObligationCause::new(),
2174                            self.table.param_env,
2175                            TraitRef::new_from_args(self.interner(), trait_.into(), substs),
2176                        ));
2177                    }
2178                }
2179                CallableDefId::StructId(_) | CallableDefId::EnumVariantId(_) => {}
2180            }
2181        }
2182    }

2184    /// Returns the argument indices to skip.
2185    fn check_legacy_const_generics(&mut self, callee: Ty<'db>, args: &[ExprId]) -> Box<[u32]> {
2186        let (func, _subst) = match callee.kind() {
2187            TyKind::FnDef(callable, subst) => {
2188                let func = match callable.0 {
2189                    CallableDefId::FunctionId(f) => f,
2190                    _ => return Default::default(),
2191                };
2192                (func, subst)
2193            }
2194            _ => return Default::default(),
2195        };
2196
2197        let data = self.db.function_signature(func);
2198        let Some(legacy_const_generics_indices) = data.legacy_const_generics_indices(self.db, func)
2199        else {
2200            return Default::default();
2201        };
2202        let mut legacy_const_generics_indices = Box::<[u32]>::from(legacy_const_generics_indices);
2203
2204        // only use legacy const generics if the param count matches with them
2205        if data.params.len() + legacy_const_generics_indices.len() != args.len() {
2206            if args.len() <= data.params.len() {
2207                return Default::default();
2208            } else {
2209                // there are more parameters than there should be without legacy
2210                // const params; use them
2211                legacy_const_generics_indices.sort_unstable();
2212                return legacy_const_generics_indices;
2213            }
2214        }
2215
2216        // check legacy const parameters
2217        for arg_idx in legacy_const_generics_indices.iter().copied() {
2218            if arg_idx >= args.len() as u32 {
2219                continue;
2220            }
2221            let expected = Expectation::none(); // FIXME use actual const ty, when that is lowered correctly
2222            self.infer_expr(args[arg_idx as usize], &expected, ExprIsRead::Yes);
2223            // FIXME: evaluate and unify with the const
2224        }
2225        legacy_const_generics_indices.sort_unstable();
2226        legacy_const_generics_indices
2227    }
