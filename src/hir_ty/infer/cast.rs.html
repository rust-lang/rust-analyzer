33#[derive(Debug)]
34pub(crate) enum CastTy<'db> {
35    Int(Int),
36    Float,
37    FnPtr,
38    Ptr(Ty<'db>, Mutability),
39    // `DynStar` is Not supported yet in r-a
40}
41
42impl<'db> CastTy<'db> {
43    pub(crate) fn from_ty(db: &dyn HirDatabase, t: Ty<'db>) -> Option<Self> {
44        match t.kind() {
45            TyKind::Bool => Some(Self::Int(Int::Bool)),
46            TyKind::Char => Some(Self::Int(Int::Char)),
47            TyKind::Int(_) => Some(Self::Int(Int::I)),
48            TyKind::Uint(it) => Some(Self::Int(Int::U(it))),
49            TyKind::Infer(InferTy::IntVar(_)) => Some(Self::Int(Int::InferenceVar)),
50            TyKind::Infer(InferTy::FloatVar(_)) => Some(Self::Float),
51            TyKind::Float(_) => Some(Self::Float),
52            TyKind::Adt(..) => {
53                let (AdtId::EnumId(id), _) = t.as_adt()? else {
54                    return None;
55                };
56                let enum_data = id.enum_variants(db);
57                if enum_data.is_payload_free(db) { Some(Self::Int(Int::CEnum)) } else { None }
58            }
59            TyKind::RawPtr(ty, m) => Some(Self::Ptr(ty, m)),
60            TyKind::FnPtr(..) => Some(Self::FnPtr),
61            _ => None,
62        }
63    }
64}

102#[derive(Clone, Debug)]
103pub(super) struct CastCheck<'db> {
104    expr: ExprId,
105    source_expr: ExprId,
106    expr_ty: Ty<'db>,
107    cast_ty: Ty<'db>,
108}
109
110impl<'db> CastCheck<'db> {
111    pub(super) fn new(
112        expr: ExprId,
113        source_expr: ExprId,
114        expr_ty: Ty<'db>,
115        cast_ty: Ty<'db>,
116    ) -> Self {
117        Self { expr, source_expr, expr_ty, cast_ty }
118    }
119
120    pub(super) fn check(
121        &mut self,
122        ctx: &mut InferenceContext<'_, 'db>,
123    ) -> Result<(), InferenceDiagnostic> {
124        self.expr_ty = ctx.table.try_structurally_resolve_type(self.expr_ty);
125        self.cast_ty = ctx.table.try_structurally_resolve_type(self.cast_ty);
126
127        // This should always come first so that we apply the coercion, which impacts infer vars.
128        if ctx
129            .coerce(
130                self.source_expr.into(),
131                self.expr_ty,
132                self.cast_ty,
133                AllowTwoPhase::No,
134                ExprIsRead::Yes,
135            )
136            .is_ok()
137        {
138            ctx.result.coercion_casts.insert(self.source_expr);
139            return Ok(());
140        }
141
142        if self.expr_ty.references_non_lt_error() || self.cast_ty.references_non_lt_error() {
143            return Ok(());
144        }
145
146        if !self.cast_ty.has_infer_types() && !ctx.table.is_sized(self.cast_ty) {
147            return Err(InferenceDiagnostic::CastToUnsized {
148                expr: self.expr,
149                cast_ty: self.cast_ty.store(),
150            });
151        }
152
153        self.do_check(ctx).map_err(|e| e.into_diagnostic(self.expr, self.expr_ty, self.cast_ty))
154    }
155
156    fn do_check(&self, ctx: &mut InferenceContext<'_, 'db>) -> Result<(), CastError> {
157        let (t_from, t_cast) =
158            match (CastTy::from_ty(ctx.db, self.expr_ty), CastTy::from_ty(ctx.db, self.cast_ty)) {
159                (Some(t_from), Some(t_cast)) => (t_from, t_cast),
160                (None, Some(t_cast)) => match self.expr_ty.kind() {
161                    TyKind::FnDef(..) => {
162                        // rustc calls `FnCtxt::normalize` on this but it's a no-op in next-solver
163                        let sig = self.expr_ty.fn_sig(ctx.interner());
164                        let fn_ptr = Ty::new_fn_ptr(ctx.interner(), sig);
165                        match ctx.coerce(
166                            self.source_expr.into(),
167                            self.expr_ty,
168                            fn_ptr,
169                            AllowTwoPhase::No,
170                            ExprIsRead::Yes,
171                        ) {
172                            Ok(_) => {}
173                            Err(TypeError::IntrinsicCast) => {
174                                return Err(CastError::IllegalCast);
175                            }
176                            Err(_) => {
177                                return Err(CastError::NonScalar);
178                            }
179                        }
180
181                        (CastTy::FnPtr, t_cast)
182                    }
183                    TyKind::Ref(_, inner_ty, mutbl) => {
184                        return match t_cast {
185                            CastTy::Int(_) | CastTy::Float => match inner_ty.kind() {
186                                TyKind::Int(_)
187                                | TyKind::Uint(_)
188                                | TyKind::Float(_)
189                                | TyKind::Infer(InferTy::IntVar(_) | InferTy::FloatVar(_)) => {
190                                    Err(CastError::NeedDeref)
191                                }
192
193                                _ => Err(CastError::NeedViaPtr),
194                            },
195                            // array-ptr-cast
196                            CastTy::Ptr(t, m) => {
197                                let t = ctx.table.try_structurally_resolve_type(t);
198                                if !ctx.table.is_sized(t) {
199                                    return Err(CastError::IllegalCast);
200                                }
201                                self.check_ref_cast(ctx, inner_ty, mutbl, t, m)
202                            }
203                            _ => Err(CastError::NonScalar),
204                        };
205                    }
206                    _ => return Err(CastError::NonScalar),
207                },
208                _ => return Err(CastError::NonScalar),
209            };
210
211        // rustc checks whether the `expr_ty` is foreign adt with `non_exhaustive` sym
212
213        match (t_from, t_cast) {
214            // These types have invariants! can't cast into them.
215            (_, CastTy::Int(Int::CEnum) | CastTy::FnPtr) => Err(CastError::NonScalar),
216
217            // * -> Bool
218            (_, CastTy::Int(Int::Bool)) => Err(CastError::CastToBool),
219
220            // * -> Char
221            (CastTy::Int(Int::U(UintTy::U8)), CastTy::Int(Int::Char)) => Ok(()), // u8-char-cast
222            (_, CastTy::Int(Int::Char)) => Err(CastError::CastToChar),
223
224            // prim -> float,ptr
225            (CastTy::Int(Int::Bool | Int::CEnum | Int::Char), CastTy::Float) => {
226                Err(CastError::NeedViaInt)
227            }
228
229            (CastTy::Int(Int::Bool | Int::CEnum | Int::Char) | CastTy::Float, CastTy::Ptr(..))
230            | (CastTy::Ptr(..) | CastTy::FnPtr, CastTy::Float) => Err(CastError::IllegalCast),
231
232            // ptr -> ptr
233            (CastTy::Ptr(src, _), CastTy::Ptr(dst, _)) => self.check_ptr_ptr_cast(ctx, src, dst), // ptr-ptr-cast
234
235            // // ptr-addr-cast
236            (CastTy::Ptr(src, _), CastTy::Int(_)) => self.check_ptr_addr_cast(ctx, src),
237            (CastTy::FnPtr, CastTy::Int(_)) => Ok(()),
238
239            // addr-ptr-cast
240            (CastTy::Int(_), CastTy::Ptr(dst, _)) => self.check_addr_ptr_cast(ctx, dst),
241
242            // fn-ptr-cast
243            (CastTy::FnPtr, CastTy::Ptr(dst, _)) => self.check_fptr_ptr_cast(ctx, dst),
244
245            // prim -> prim
246            (CastTy::Int(Int::CEnum), CastTy::Int(_)) => Ok(()),
247            (CastTy::Int(Int::Char | Int::Bool), CastTy::Int(_)) => Ok(()),
248            (CastTy::Int(_) | CastTy::Float, CastTy::Int(_) | CastTy::Float) => Ok(()),
249        }
250    }

252    fn check_ref_cast(
253        &self,
254        ctx: &mut InferenceContext<'_, 'db>,
255        t_expr: Ty<'db>,
256        m_expr: Mutability,
257        t_cast: Ty<'db>,
258        m_cast: Mutability,
259    ) -> Result<(), CastError> {
260        let t_expr = ctx.table.try_structurally_resolve_type(t_expr);
261        let t_cast = ctx.table.try_structurally_resolve_type(t_cast);
262
263        if m_expr >= m_cast
264            && let TyKind::Array(ety, _) = t_expr.kind()
265            && ctx.infcx().can_eq(ctx.table.param_env, ety, t_cast)
266        {
267            // Due to historical reasons we allow directly casting references of
268            // arrays into raw pointers of their element type.
269
270            // Coerce to a raw pointer so that we generate RawPtr in MIR.
271            let array_ptr_type = Ty::new_ptr(ctx.interner(), t_expr, m_expr);
272            if ctx
273                .coerce(
274                    self.source_expr.into(),
275                    self.expr_ty,
276                    array_ptr_type,
277                    AllowTwoPhase::No,
278                    ExprIsRead::Yes,
279                )
280                .is_ok()
281            {
282            } else {
283                never!(
284                    "could not cast from reference to array to pointer to array ({:?} to {:?})",
285                    self.expr_ty,
286                    array_ptr_type
287                );
288            }
289
290            // this will report a type mismatch if needed
291            let _ = ctx.demand_eqtype(self.expr.into(), ety, t_cast);
292            return Ok(());
293        }
294
295        Err(CastError::IllegalCast)
296    }

298    fn check_ptr_ptr_cast(
299        &self,
300        ctx: &mut InferenceContext<'_, 'db>,
301        src: Ty<'db>,
302        dst: Ty<'db>,
303    ) -> Result<(), CastError> {
304        let src_kind = pointer_kind(src, ctx).map_err(|_| CastError::Unknown)?;
305        let dst_kind = pointer_kind(dst, ctx).map_err(|_| CastError::Unknown)?;
306
307        match (src_kind, dst_kind) {
308            (Some(PointerKind::Error), _) | (_, Some(PointerKind::Error)) => Ok(()),
309
310            // (_, None) => Err(CastError::UnknownCastPtrKind),
311            // (None, _) => Err(CastError::UnknownExprPtrKind),
312            (_, None) | (None, _) => Ok(()),
313
314            // Cast to thin pointer is OK
315            (_, Some(PointerKind::Thin)) => Ok(()),
316
317            // thin -> fat? report invalid cast (don't complain about vtable kinds)
318            (Some(PointerKind::Thin), _) => Err(CastError::SizedUnsizedCast),
319
320            // trait object -> trait object? need to do additional checks
321            (Some(PointerKind::VTable(src_tty)), Some(PointerKind::VTable(dst_tty))) => {
322                match (src_tty.principal_def_id(), dst_tty.principal_def_id()) {
323                    // A<dyn Src<...> + SrcAuto> -> B<dyn Dst<...> + DstAuto>. need to make sure
324                    // - `Src` and `Dst` traits are the same
325                    // - traits have the same generic arguments
326                    // - projections are the same
327                    // - `SrcAuto` (+auto traits implied by `Src`) is a superset of `DstAuto`
328                    //
329                    // Note that trait upcasting goes through a different mechanism (`coerce_unsized`)
330                    // and is unaffected by this check.
331                    (Some(src_principal), Some(_)) => {
332                        // We need to reconstruct trait object types.
333                        // `m_src` and `m_dst` won't work for us here because they will potentially
334                        // contain wrappers, which we do not care about.
335                        //
336                        // e.g. we want to allow `dyn T -> (dyn T,)`, etc.
337                        //
338                        // We also need to skip auto traits to emit an FCW and not an error.
339                        let src_obj = Ty::new_dynamic(
340                            ctx.interner(),
341                            BoundExistentialPredicates::new_from_iter(
342                                ctx.interner(),
343                                src_tty.iter().filter(|pred| {
344                                    !matches!(
345                                        pred.skip_binder(),
346                                        ExistentialPredicate::AutoTrait(_)
347                                    )
348                                }),
349                            ),
350                            Region::new_erased(ctx.interner()),
351                        );
352                        let dst_obj = Ty::new_dynamic(
353                            ctx.interner(),
354                            BoundExistentialPredicates::new_from_iter(
355                                ctx.interner(),
356                                dst_tty.iter().filter(|pred| {
357                                    !matches!(
358                                        pred.skip_binder(),
359                                        ExistentialPredicate::AutoTrait(_)
360                                    )
361                                }),
362                            ),
363                            Region::new_erased(ctx.interner()),
364                        );
365
366                        // `dyn Src = dyn Dst`, this checks for matching traits/generics/projections
367                        // This is `fcx.demand_eqtype`, but inlined to give a better error.
368                        if ctx
369                            .table
370                            .at(&ObligationCause::dummy())
371                            .eq(src_obj, dst_obj)
372                            .map(|infer_ok| ctx.table.register_infer_ok(infer_ok))
373                            .is_err()
374                        {
375                            return Err(CastError::DifferingKinds);
376                        }
377
378                        // Check that `SrcAuto` (+auto traits implied by `Src`) is a superset of `DstAuto`.
379                        // Emit an FCW otherwise.
380                        let src_auto: FxHashSet<_> = src_tty
381                            .auto_traits()
382                            .into_iter()
383                            .chain(
384                                elaborate::supertrait_def_ids(ctx.interner(), src_principal)
385                                    .filter(|trait_| {
386                                        ctx.db
387                                            .trait_signature(trait_.0)
388                                            .flags
389                                            .contains(TraitFlags::AUTO)
390                                    }),
391                            )
392                            .collect();
393
394                        let added = dst_tty
395                            .auto_traits()
396                            .into_iter()
397                            .any(|trait_| !src_auto.contains(&trait_));
398
399                        if added {
400                            return Err(CastError::PtrPtrAddingAutoTraits);
401                        }
402
403                        Ok(())
404                    }
405
406                    // dyn Auto -> dyn Auto'? ok.
407                    (None, None) => Ok(()),
408
409                    // dyn Trait -> dyn Auto? not ok (for now).
410                    //
411                    // Although dropping the principal is already allowed for unsizing coercions
412                    // (e.g. `*const (dyn Trait + Auto)` to `*const dyn Auto`), dropping it is
413                    // currently **NOT** allowed for (non-coercion) ptr-to-ptr casts (e.g
414                    // `*const Foo` to `*const Bar` where `Foo` has a `dyn Trait + Auto` tail
415                    // and `Bar` has a `dyn Auto` tail), because the underlying MIR operations
416                    // currently work very differently:
417                    //
418                    // * A MIR unsizing coercion on raw pointers to trait objects (`*const dyn Src`
419                    //   to `*const dyn Dst`) is currently equivalent to downcasting the source to
420                    //   the concrete sized type that it was originally unsized from first (via a
421                    //   ptr-to-ptr cast from `*const Src` to `*const T` with `T: Sized`) and then
422                    //   unsizing this thin pointer to the target type (unsizing `*const T` to
423                    //   `*const Dst`). In particular, this means that the pointer's metadata
424                    //   (vtable) will semantically change, e.g. for const eval and miri, even
425                    //   though the vtables will always be merged for codegen.
426                    //
427                    // * A MIR ptr-to-ptr cast is currently equivalent to a transmute and does not
428                    //   change the pointer metadata (vtable) at all.
429                    //
430                    // In addition to this potentially surprising difference between coercion and
431                    // non-coercion casts, casting away the principal with a MIR ptr-to-ptr cast
432                    // is currently considered undefined behavior:
433                    //
434                    // As a validity invariant of pointers to trait objects, we currently require
435                    // that the principal of the vtable in the pointer metadata exactly matches
436                    // the principal of the pointee type, where "no principal" is also considered
437                    // a kind of principal.
438                    (Some(_), None) => Err(CastError::DifferingKinds),
439
440                    // dyn Auto -> dyn Trait? not ok.
441                    (None, Some(_)) => Err(CastError::DifferingKinds),
442                }
443            }
444
445            // fat -> fat? metadata kinds must match
446            (Some(src_kind), Some(dst_kind)) if src_kind == dst_kind => Ok(()),
447            (_, _) => Err(CastError::DifferingKinds),
448        }
449    }

451    fn check_ptr_addr_cast(
452        &self,
453        ctx: &mut InferenceContext<'_, 'db>,
454        expr_ty: Ty<'db>,
455    ) -> Result<(), CastError> {
456        match pointer_kind(expr_ty, ctx).map_err(|_| CastError::Unknown)? {
457            // None => Err(CastError::UnknownExprPtrKind),
458            None => Ok(()),
459            Some(PointerKind::Error) => Ok(()),
460            Some(PointerKind::Thin) => Ok(()),
461            _ => Err(CastError::NeedViaThinPtr),
462        }
463    }

465    fn check_addr_ptr_cast(
466        &self,
467        ctx: &mut InferenceContext<'_, 'db>,
468        cast_ty: Ty<'db>,
469    ) -> Result<(), CastError> {
470        match pointer_kind(cast_ty, ctx).map_err(|_| CastError::Unknown)? {
471            // None => Err(CastError::UnknownCastPtrKind),
472            None => Ok(()),
473            Some(PointerKind::Error) => Ok(()),
474            Some(PointerKind::Thin) => Ok(()),
475            Some(PointerKind::VTable(_)) => Err(CastError::IntToWideCast),
476            Some(PointerKind::Length) => Err(CastError::IntToWideCast),
477            Some(PointerKind::OfAlias | PointerKind::OfParam(_)) => Err(CastError::IntToWideCast),
478        }
479    }

481    fn check_fptr_ptr_cast(
482        &self,
483        ctx: &mut InferenceContext<'_, 'db>,
484        cast_ty: Ty<'db>,
485    ) -> Result<(), CastError> {
486        match pointer_kind(cast_ty, ctx).map_err(|_| CastError::Unknown)? {
487            // None => Err(CastError::UnknownCastPtrKind),
488            None => Ok(()),
489            Some(PointerKind::Error) => Ok(()),
490            Some(PointerKind::Thin) => Ok(()),
491            _ => Err(CastError::IllegalCast),
492        }

496/// The kind of pointer and associated metadata (thin, length or vtable) - we
497/// only allow casts between wide pointers if their metadata have the same
498/// kind.
499#[derive(Debug, PartialEq, Eq)]
500enum PointerKind<'db> {
501    /// No metadata attached, ie pointer to sized type or foreign type
502    Thin,
503    /// A trait object
504    VTable(BoundExistentialPredicates<'db>),
505    /// Slice
506    Length,
507    /// The unsize info of this projection or opaque type
508    OfAlias,
509    /// The unsize info of this parameter
510    OfParam(ParamTy),
511    Error,
512}
513
514fn pointer_kind<'db>(
515    ty: Ty<'db>,
516    ctx: &mut InferenceContext<'_, 'db>,
517) -> Result<Option<PointerKind<'db>>, ()> {
518    let ty = ctx.table.try_structurally_resolve_type(ty);
519
520    if ctx.table.is_sized(ty) {
521        return Ok(Some(PointerKind::Thin));
522    }
523
524    match ty.kind() {
525        TyKind::Slice(_) | TyKind::Str => Ok(Some(PointerKind::Length)),
526        TyKind::Dynamic(bounds, _) => Ok(Some(PointerKind::VTable(bounds))),
527        TyKind::Adt(adt_def, subst) => {
528            let id = adt_def.def_id().0;
529            let AdtId::StructId(id) = id else {
530                never!("`{:?}` should be sized but is not?", ty);
531                return Err(());
532            };
533
534            let struct_data = id.fields(ctx.db);
535            if let Some((last_field, _)) = struct_data.fields().iter().last() {
536                let last_field_ty = ctx.db.field_types(id.into())[last_field]
537                    .get()
538                    .instantiate(ctx.interner(), subst);
539                pointer_kind(last_field_ty, ctx)
540            } else {
541                Ok(Some(PointerKind::Thin))
542            }
543        }
544        TyKind::Tuple(subst) => match subst.iter().next_back() {
545            None => Ok(Some(PointerKind::Thin)),
546            Some(ty) => pointer_kind(ty, ctx),
547        },
548        TyKind::Foreign(_) => Ok(Some(PointerKind::Thin)),
549        TyKind::Alias(..) => Ok(Some(PointerKind::OfAlias)),
550        TyKind::Error(_) => Ok(Some(PointerKind::Error)),
551        TyKind::Param(idx) => Ok(Some(PointerKind::OfParam(idx))),
552        TyKind::Bound(..) | TyKind::Placeholder(..) | TyKind::Infer(..) => Ok(None),
553        TyKind::Int(_)
554        | TyKind::Uint(_)
555        | TyKind::Float(_)
556        | TyKind::Bool
557        | TyKind::Char
558        | TyKind::Array(..)
559        | TyKind::CoroutineWitness(..)
560        | TyKind::RawPtr(..)
561        | TyKind::Ref(..)
562        | TyKind::FnDef(..)
563        | TyKind::FnPtr(..)
564        | TyKind::Closure(..)
565        | TyKind::Coroutine(..)
566        | TyKind::CoroutineClosure(..)
567        | TyKind::Never => {
568            never!("`{:?}` should be sized but is not?", ty);
569            Err(())
570        }
571        TyKind::UnsafeBinder(..) | TyKind::Pat(..) => {
572            never!("we don't produce these types: {ty:?}");
573            Err(())
574        }
575    }
