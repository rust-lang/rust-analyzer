55    pub(super) fn infer_closure(
56        &mut self,
57        body: &ExprId,
58        args: &[PatId],
59        ret_type: &Option<TypeRefId>,
60        arg_types: &[Option<TypeRefId>],
61        closure_kind: ClosureKind,
62        tgt_expr: ExprId,
63        expected: &Expectation,
64    ) -> Ty {
65        assert_eq!(args.len(), arg_types.len());
66
67        let (expected_sig, expected_kind) = match expected.to_option(&mut self.table) {
68            Some(expected_ty) => self.deduce_closure_signature(&expected_ty, closure_kind),
69            None => (None, None),
70        };
71
72        let ClosureSignature { expected_sig: bound_sig, ret_ty: body_ret_ty } =
73            self.sig_of_closure(body, ret_type, arg_types, closure_kind, expected_sig);
74        let bound_sig = self.normalize_associated_types_in(bound_sig);
75        let sig_ty = TyKind::Function(bound_sig.clone()).intern(Interner);
76
77        let (id, ty, resume_yield_tys) = match closure_kind {
78            ClosureKind::Coroutine(_) => {
79                let sig_tys = bound_sig.substitution.0.as_slice(Interner);
80                // FIXME: report error when there are more than 1 parameter.
81                let resume_ty = match sig_tys.first() {
82                    // When `sig_tys.len() == 1` the first type is the return type, not the
83                    // first parameter type.
84                    Some(ty) if sig_tys.len() > 1 => ty.assert_ty_ref(Interner).clone(),
85                    _ => self.result.standard_types.unit.clone(),
86                };
87                let yield_ty = self.table.new_type_var();
88
89                let subst = TyBuilder::subst_for_coroutine(self.db, self.owner)
90                    .push(resume_ty.clone())
91                    .push(yield_ty.clone())
92                    .push(body_ret_ty.clone())
93                    .build();
94
95                let coroutine_id =
96                    self.db.intern_coroutine(InternedCoroutine(self.owner, tgt_expr)).into();
97                let coroutine_ty = TyKind::Coroutine(coroutine_id, subst).intern(Interner);
98
99                (None, coroutine_ty, Some((resume_ty, yield_ty)))
100            }
101            ClosureKind::Closure | ClosureKind::Async => {
102                let closure_id =
103                    self.db.intern_closure(InternedClosure(self.owner, tgt_expr)).into();
104                let closure_ty = TyKind::Closure(
105                    closure_id,
106                    TyBuilder::subst_for_closure(self.db, self.owner, sig_ty.clone()),
107                )
108                .intern(Interner);
109                self.deferred_closures.entry(closure_id).or_default();
110                self.add_current_closure_dependency(closure_id);
111                (Some(closure_id), closure_ty, None)
112            }
113        };
114
115        // Eagerly try to relate the closure type with the expected
116        // type, otherwise we often won't have enough information to
117        // infer the body.
118        self.deduce_closure_type_from_expectations(tgt_expr, &ty, &sig_ty, expected, expected_kind);
119
120        // Now go through the argument patterns
121        for (arg_pat, arg_ty) in args.iter().zip(bound_sig.substitution.0.as_slice(Interner).iter())
122        {
123            self.infer_top_pat(*arg_pat, arg_ty.assert_ty_ref(Interner), None);
124        }
125
126        // FIXME: lift these out into a struct
127        let prev_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
128        let prev_closure = mem::replace(&mut self.current_closure, id);
129        let prev_ret_ty = mem::replace(&mut self.return_ty, body_ret_ty.clone());
130        let prev_ret_coercion = self.return_coercion.replace(CoerceMany::new(body_ret_ty.clone()));
131        let prev_resume_yield_tys = mem::replace(&mut self.resume_yield_tys, resume_yield_tys);
132
133        self.with_breakable_ctx(BreakableKind::Border, None, None, |this| {
134            this.infer_return(*body);
135        });
136
137        self.diverges = prev_diverges;
138        self.return_ty = prev_ret_ty;
139        self.return_coercion = prev_ret_coercion;
140        self.current_closure = prev_closure;
141        self.resume_yield_tys = prev_resume_yield_tys;
142
143        self.table.normalize_associated_types_in(ty)
144    }

146    // This function handles both closures and coroutines.
147    pub(super) fn deduce_closure_type_from_expectations(
148        &mut self,
149        closure_expr: ExprId,
150        closure_ty: &Ty,
151        sig_ty: &Ty,
152        expectation: &Expectation,
153        expected_kind: Option<FnTrait>,
154    ) {
155        let expected_ty = match expectation.to_option(&mut self.table) {
156            Some(ty) => ty,
157            None => return,
158        };
159
160        match (closure_ty.kind(Interner), expected_kind) {
161            (TyKind::Closure(closure_id, _), Some(closure_kind)) => {
162                self.result
163                    .closure_info
164                    .entry(*closure_id)
165                    .or_insert_with(|| (Vec::new(), closure_kind));
166            }
167            _ => {}
168        }
169
170        // Deduction from where-clauses in scope, as well as fn-pointer coercion are handled here.
171        let _ = self.coerce(Some(closure_expr), closure_ty, &expected_ty, CoerceNever::Yes);
172
173        // Coroutines are not Fn* so return early.
174        if matches!(closure_ty.kind(Interner), TyKind::Coroutine(..)) {
175            return;
176        }
177
178        // Deduction based on the expected `dyn Fn` is done separately.
179        if let TyKind::Dyn(dyn_ty) = expected_ty.kind(Interner) {
180            if let Some(sig) = self.deduce_sig_from_dyn_ty(dyn_ty) {
181                let expected_sig_ty = TyKind::Function(sig).intern(Interner);
182
183                self.unify(sig_ty, &expected_sig_ty);
184            }
185        }
186    }

190    pub(super) fn deduce_closure_signature(
191        &mut self,
192        expected_ty: &Ty,
193        closure_kind: ClosureKind,
194    ) -> (Option<FnSubst<Interner>>, Option<FnTrait>) {
195        match expected_ty.kind(Interner) {
196            TyKind::Alias(AliasTy::Opaque(OpaqueTy { .. })) | TyKind::OpaqueType(..) => {
197                let clauses = expected_ty.impl_trait_bounds(self.db).into_iter().flatten().map(
198                    |b: chalk_ir::Binders<chalk_ir::WhereClause<Interner>>| {
199                        b.into_value_and_skipped_binders().0
200                    },
201                );
202                self.deduce_closure_kind_from_predicate_clauses(expected_ty, clauses, closure_kind)
203            }
204            TyKind::Dyn(dyn_ty) => {
205                let sig =
206                    dyn_ty.bounds.skip_binders().as_slice(Interner).iter().find_map(|bound| {
207                        if let WhereClause::AliasEq(AliasEq {
208                            alias: AliasTy::Projection(projection_ty),
209                            ty: projected_ty,
210                        }) = bound.skip_binders()
211                        {
212                            if let Some(sig) = self.deduce_sig_from_projection(
213                                closure_kind,
214                                projection_ty,
215                                projected_ty,
216                            ) {
217                                return Some(sig);
218                            }
219                        }
220                        None
221                    });
222
223                let kind = dyn_ty.principal().and_then(|principal_trait_ref| {
224                    self.fn_trait_kind_from_trait_id(from_chalk_trait_id(
225                        principal_trait_ref.skip_binders().skip_binders().trait_id,
226                    ))
227                });
228
229                (sig, kind)
230            }
231            TyKind::InferenceVar(ty, chalk_ir::TyVariableKind::General) => {
232                let clauses = self.clauses_for_self_ty(*ty);
233                self.deduce_closure_kind_from_predicate_clauses(
234                    expected_ty,
235                    clauses.into_iter(),
236                    closure_kind,
237                )
238            }
239            TyKind::Function(fn_ptr) => match closure_kind {
240                ClosureKind::Closure => (Some(fn_ptr.substitution.clone()), Some(FnTrait::Fn)),
241                ClosureKind::Async | ClosureKind::Coroutine(_) => (None, None),
242            },
243            _ => (None, None),
244        }
245    }

247    fn deduce_closure_kind_from_predicate_clauses(
248        &self,
249        expected_ty: &Ty,
250        clauses: impl DoubleEndedIterator<Item = WhereClause>,
251        closure_kind: ClosureKind,
252    ) -> (Option<FnSubst<Interner>>, Option<FnTrait>) {
253        let mut expected_sig = None;
254        let mut expected_kind = None;
255
256        for clause in elaborate_clause_supertraits(self.db, clauses.rev()) {
257            if expected_sig.is_none() {
258                if let WhereClause::AliasEq(AliasEq {
259                    alias: AliasTy::Projection(projection),
260                    ty,
261                }) = &clause
262                {
263                    let inferred_sig =
264                        self.deduce_sig_from_projection(closure_kind, projection, ty);
265                    // Make sure that we didn't infer a signature that mentions itself.
266                    // This can happen when we elaborate certain supertrait bounds that
267                    // mention projections containing the `Self` type. See rust-lang/rust#105401.
268                    struct MentionsTy<'a> {
269                        expected_ty: &'a Ty,
270                    }
271                    impl TypeVisitor<Interner> for MentionsTy<'_> {
272                        type BreakTy = ();
273
274                        fn interner(&self) -> Interner {
275                            Interner
276                        }
277
278                        fn as_dyn(
279                            &mut self,
280                        ) -> &mut dyn TypeVisitor<Interner, BreakTy = Self::BreakTy>
281                        {
282                            self
283                        }
284
285                        fn visit_ty(
286                            &mut self,
287                            t: &Ty,
288                            db: chalk_ir::DebruijnIndex,
289                        ) -> ControlFlow<()> {
290                            if t == self.expected_ty {
291                                ControlFlow::Break(())
292                            } else {
293                                t.super_visit_with(self, db)
294                            }
295                        }
296                    }
297                    if inferred_sig
298                        .visit_with(
299                            &mut MentionsTy { expected_ty },
300                            chalk_ir::DebruijnIndex::INNERMOST,
301                        )
302                        .is_continue()
303                    {
304                        expected_sig = inferred_sig;
305                    }
306                }
307            }
308
309            let trait_id = match clause {
310                WhereClause::AliasEq(AliasEq {
311                    alias: AliasTy::Projection(projection), ..
312                }) => projection.trait_(self.db),
313                WhereClause::Implemented(trait_ref) => from_chalk_trait_id(trait_ref.trait_id),
314                _ => continue,
315            };
316            if let Some(closure_kind) = self.fn_trait_kind_from_trait_id(trait_id) {
317                // always use the closure kind that is more permissive.
318                match (expected_kind, closure_kind) {
319                    (None, _) => expected_kind = Some(closure_kind),
320                    (Some(FnTrait::FnMut), FnTrait::Fn) => expected_kind = Some(FnTrait::Fn),
321                    (Some(FnTrait::FnOnce), FnTrait::Fn | FnTrait::FnMut) => {
322                        expected_kind = Some(closure_kind)
323                    }
324                    _ => {}
325                }
326            }
327        }
328
329        (expected_sig, expected_kind)
330    }

332    fn deduce_sig_from_dyn_ty(&self, dyn_ty: &DynTy) -> Option<FnPointer> {
333        // Search for a predicate like `<$self as FnX<Args>>::Output == Ret`
334
335        let fn_traits: SmallVec<[ChalkTraitId; 3]> =
336            utils::fn_traits(self.db, self.owner.module(self.db).krate())
337                .map(to_chalk_trait_id)
338                .collect();
339
340        let self_ty = self.result.standard_types.unknown.clone();
341        let bounds = dyn_ty.bounds.clone().substitute(Interner, &[self_ty.cast(Interner)]);
342        for bound in bounds.iter(Interner) {
343            // NOTE(skip_binders): the extracted types are rebound by the returned `FnPointer`
344            if let WhereClause::AliasEq(AliasEq { alias: AliasTy::Projection(projection), ty }) =
345                bound.skip_binders()
346            {
347                let assoc_data =
348                    self.db.associated_ty_data(from_assoc_type_id(projection.associated_ty_id));
349                if !fn_traits.contains(&assoc_data.trait_id) {
350                    return None;
351                }
352
353                // Skip `Self`, get the type argument.
354                let arg = projection.substitution.as_slice(Interner).get(1)?;
355                if let Some(subst) = arg.ty(Interner)?.as_tuple() {
356                    let generic_args = subst.as_slice(Interner);
357                    let mut sig_tys = Vec::with_capacity(generic_args.len() + 1);
358                    for arg in generic_args {
359                        sig_tys.push(arg.ty(Interner)?.clone());
360                    }
361                    sig_tys.push(ty.clone());
362
363                    cov_mark::hit!(dyn_fn_param_informs_call_site_closure_signature);
364                    return Some(FnPointer {
365                        num_binders: bound.len(Interner),
366                        sig: FnSig {
367                            abi: FnAbi::RustCall,
368                            safety: chalk_ir::Safety::Safe,
369                            variadic: false,
370                        },
371                        substitution: FnSubst(Substitution::from_iter(Interner, sig_tys)),
372                    });
373                }
374            }
375        }
376
377        None
378    }

380    fn deduce_sig_from_projection(
381        &self,
382        closure_kind: ClosureKind,
383        projection_ty: &ProjectionTy,
384        projected_ty: &Ty,
385    ) -> Option<FnSubst<Interner>> {
386        let container =
387            from_assoc_type_id(projection_ty.associated_ty_id).lookup(self.db).container;
388        let trait_ = match container {
389            hir_def::ItemContainerId::TraitId(trait_) => trait_,
390            _ => return None,
391        };
392
393        // For now, we only do signature deduction based off of the `Fn` and `AsyncFn` traits,
394        // for closures and async closures, respectively.
395        match closure_kind {
396            ClosureKind::Closure | ClosureKind::Async
397                if self.fn_trait_kind_from_trait_id(trait_).is_some() =>
398            {
399                self.extract_sig_from_projection(projection_ty, projected_ty)
400            }
401            _ => None,
402        }
403    }

405    fn extract_sig_from_projection(
406        &self,
407        projection_ty: &ProjectionTy,
408        projected_ty: &Ty,
409    ) -> Option<FnSubst<Interner>> {
410        let arg_param_ty = projection_ty.substitution.as_slice(Interner)[1].assert_ty_ref(Interner);
411
412        let TyKind::Tuple(_, input_tys) = arg_param_ty.kind(Interner) else {
413            return None;
414        };
415
416        let ret_param_ty = projected_ty;
417
418        Some(FnSubst(Substitution::from_iter(
419            Interner,
420            input_tys.iter(Interner).map(|t| t.cast(Interner)).chain(Some(GenericArg::new(
421                Interner,
422                chalk_ir::GenericArgData::Ty(ret_param_ty.clone()),
423            ))),
424        )))
425    }

431    fn supplied_sig_of_closure(
432        &mut self,
433        body: &ExprId,
434        ret_type: &Option<TypeRefId>,
435        arg_types: &[Option<TypeRefId>],
436        closure_kind: ClosureKind,
437    ) -> ClosureSignature {
438        let mut sig_tys = Vec::with_capacity(arg_types.len() + 1);
439
440        // collect explicitly written argument types
441        for arg_type in arg_types.iter() {
442            let arg_ty = match arg_type {
443                Some(type_ref) => self.make_body_ty(*type_ref),
444                None => self.table.new_type_var(),
445            };
446            sig_tys.push(arg_ty);
447        }
448
449        // add return type
450        let ret_ty = match ret_type {
451            Some(type_ref) => self.make_body_ty(*type_ref),
452            None => self.table.new_type_var(),
453        };
454        if let ClosureKind::Async = closure_kind {
455            sig_tys.push(self.lower_async_block_type_impl_trait(ret_ty.clone(), *body));
456        } else {
457            sig_tys.push(ret_ty.clone());
458        }
459
460        let expected_sig = FnPointer {
461            num_binders: 0,
462            sig: FnSig { abi: FnAbi::RustCall, safety: chalk_ir::Safety::Safe, variadic: false },
463            substitution: FnSubst(
464                Substitution::from_iter(Interner, sig_tys.iter().cloned()).shifted_in(Interner),
465            ),
466        };
467
468        ClosureSignature { ret_ty, expected_sig }
469    }

471    /// The return type is the signature of the closure, and the return type
472    /// *as represented inside the body* (so, for async closures, the `Output` ty)
473    pub(super) fn sig_of_closure(
474        &mut self,
475        body: &ExprId,
476        ret_type: &Option<TypeRefId>,
477        arg_types: &[Option<TypeRefId>],
478        closure_kind: ClosureKind,
479        expected_sig: Option<FnSubst<Interner>>,
480    ) -> ClosureSignature {
481        if let Some(e) = expected_sig {
482            self.sig_of_closure_with_expectation(body, ret_type, arg_types, closure_kind, e)
483        } else {
484            self.sig_of_closure_no_expectation(body, ret_type, arg_types, closure_kind)
485        }
486    }

488    fn sig_of_closure_no_expectation(
489        &mut self,
490        body: &ExprId,
491        ret_type: &Option<TypeRefId>,
492        arg_types: &[Option<TypeRefId>],
493        closure_kind: ClosureKind,
494    ) -> ClosureSignature {
495        self.supplied_sig_of_closure(body, ret_type, arg_types, closure_kind)

498    fn sig_of_closure_with_expectation(
499        &mut self,
500        body: &ExprId,
501        ret_type: &Option<TypeRefId>,
502        arg_types: &[Option<TypeRefId>],
503        closure_kind: ClosureKind,
504        expected_sig: FnSubst<Interner>,
505    ) -> ClosureSignature {
506        let expected_sig = FnPointer {
507            num_binders: 0,
508            sig: FnSig { abi: FnAbi::RustCall, safety: chalk_ir::Safety::Safe, variadic: false },
509            substitution: expected_sig,
510        };
511
512        // If the expected signature does not match the actual arg types,
513        // then just return the expected signature
514        if expected_sig.substitution.0.len(Interner) != arg_types.len() + 1 {
515            let ret_ty = match ret_type {
516                Some(type_ref) => self.make_body_ty(*type_ref),
517                None => self.table.new_type_var(),
518            };
519            return ClosureSignature { expected_sig, ret_ty };
520        }
521
522        self.merge_supplied_sig_with_expectation(
523            body,
524            ret_type,
525            arg_types,
526            closure_kind,
527            expected_sig,
528        )
529    }
530
531    fn merge_supplied_sig_with_expectation(
532        &mut self,
533        body: &ExprId,
534        ret_type: &Option<TypeRefId>,
535        arg_types: &[Option<TypeRefId>],
536        closure_kind: ClosureKind,
537        expected_sig: FnPointer,
538    ) -> ClosureSignature {
539        let supplied_sig = self.supplied_sig_of_closure(body, ret_type, arg_types, closure_kind);
540
541        let snapshot = self.table.snapshot();
542        if !self.table.unify(&expected_sig.substitution, &supplied_sig.expected_sig.substitution) {
543            self.table.rollback_to(snapshot);
544        }
545
546        supplied_sig
547    }

552#[derive(Debug, Clone, PartialEq, Eq, Hash)]
553pub(crate) struct HirPlace {
554    pub(crate) local: BindingId,
555    pub(crate) projections: Vec<ProjectionElem<Infallible, Ty>>,
556}
557
558impl HirPlace {
559    fn ty(&self, ctx: &mut InferenceContext<'_>) -> Ty {
560        let mut ty = ctx.table.resolve_completely(ctx.result[self.local].clone());
561        for p in &self.projections {
562            ty = p.projected_ty(
563                ty,
564                ctx.db,
565                |_, _, _| {
566                    unreachable!("Closure field only happens in MIR");
567                },
568                ctx.owner.module(ctx.db).krate(),
569            );
570        }
571        ty
572    }
573
574    fn capture_kind_of_truncated_place(
575        &self,
576        mut current_capture: CaptureKind,
577        len: usize,
578    ) -> CaptureKind {
579        if let CaptureKind::ByRef(BorrowKind::Mut {
580            kind: MutBorrowKind::Default | MutBorrowKind::TwoPhasedBorrow,
581        }) = current_capture
582        {
583            if self.projections[len..].contains(&ProjectionElem::Deref) {
584                current_capture =
585                    CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::ClosureCapture });
586            }
587        }
588        current_capture
589    }
590}

634    /// Converts the place to a name that can be inserted into source code.
635    pub fn place_to_name(&self, owner: DefWithBodyId, db: &dyn HirDatabase) -> String {
636        let body = db.body(owner);
637        let mut result = body[self.place.local].name.as_str().to_owned();
638        for proj in &self.place.projections {
639            match proj {
640                ProjectionElem::Deref => {}
641                ProjectionElem::Field(Either::Left(f)) => {
642                    let variant_data = f.parent.variant_data(db);
643                    match variant_data.shape {
644                        FieldsShape::Record => {
645                            result.push('_');
646                            result.push_str(variant_data.fields()[f.local_id].name.as_str())
647                        }
648                        FieldsShape::Tuple => {
649                            let index =
650                                variant_data.fields().iter().position(|it| it.0 == f.local_id);
651                            if let Some(index) = index {
652                                format_to!(result, "_{index}");
653                            }
654                        }
655                        FieldsShape::Unit => {}
656                    }
657                }
658                ProjectionElem::Field(Either::Right(f)) => format_to!(result, "_{}", f.index),
659                &ProjectionElem::ClosureField(field) => format_to!(result, "_{field}"),
660                ProjectionElem::Index(_)
661                | ProjectionElem::ConstantIndex { .. }
662                | ProjectionElem::Subslice { .. }
663                | ProjectionElem::OpaqueCast(_) => {
664                    never!("Not happen in closure capture");
665                    continue;
666                }
667            }
668        }
669        if is_raw_identifier(&result, owner.module(db).krate().data(db).edition) {
670            result.insert_str(0, "r#");
671        }
672        result
673    }

675    pub fn display_place_source_code(&self, owner: DefWithBodyId, db: &dyn HirDatabase) -> String {
676        let body = db.body(owner);
677        let krate = owner.krate(db);
678        let edition = krate.data(db).edition;
679        let mut result = body[self.place.local].name.display(db, edition).to_string();
680        for proj in &self.place.projections {
681            match proj {
682                // In source code autoderef kicks in.
683                ProjectionElem::Deref => {}
684                ProjectionElem::Field(Either::Left(f)) => {
685                    let variant_data = f.parent.variant_data(db);
686                    match variant_data.shape {
687                        FieldsShape::Record => format_to!(
688                            result,
689                            ".{}",
690                            variant_data.fields()[f.local_id].name.display(db, edition)
691                        ),
692                        FieldsShape::Tuple => format_to!(
693                            result,
694                            ".{}",
695                            variant_data
696                                .fields()
697                                .iter()
698                                .position(|it| it.0 == f.local_id)
699                                .unwrap_or_default()
700                        ),
701                        FieldsShape::Unit => {}
702                    }
703                }
704                ProjectionElem::Field(Either::Right(f)) => {
705                    let field = f.index;
706                    format_to!(result, ".{field}");
707                }
708                &ProjectionElem::ClosureField(field) => {
709                    format_to!(result, ".{field}");
710                }
711                ProjectionElem::Index(_)
712                | ProjectionElem::ConstantIndex { .. }
713                | ProjectionElem::Subslice { .. }
714                | ProjectionElem::OpaqueCast(_) => {
715                    never!("Not happen in closure capture");
716                    continue;
717                }
718            }
719        }
720        let final_derefs_count = self
721            .place
722            .projections
723            .iter()
724            .rev()
725            .take_while(|proj| matches!(proj, ProjectionElem::Deref))
726            .count();
727        result.insert_str(0, &"*".repeat(final_derefs_count));
728        result
729    }

731    pub fn display_place(&self, owner: DefWithBodyId, db: &dyn HirDatabase) -> String {
732        let body = db.body(owner);
733        let krate = owner.krate(db);
734        let edition = krate.data(db).edition;
735        let mut result = body[self.place.local].name.display(db, edition).to_string();
736        let mut field_need_paren = false;
737        for proj in &self.place.projections {
738            match proj {
739                ProjectionElem::Deref => {
740                    result = format!("*{result}");
741                    field_need_paren = true;
742                }
743                ProjectionElem::Field(Either::Left(f)) => {
744                    if field_need_paren {
745                        result = format!("({result})");
746                    }
747                    let variant_data = f.parent.variant_data(db);
748                    let field = match variant_data.shape {
749                        FieldsShape::Record => {
750                            variant_data.fields()[f.local_id].name.as_str().to_owned()
751                        }
752                        FieldsShape::Tuple => variant_data
753                            .fields()
754                            .iter()
755                            .position(|it| it.0 == f.local_id)
756                            .unwrap_or_default()
757                            .to_string(),
758                        FieldsShape::Unit => "[missing field]".to_owned(),
759                    };
760                    result = format!("{result}.{field}");
761                    field_need_paren = false;
762                }
763                ProjectionElem::Field(Either::Right(f)) => {
764                    let field = f.index;
765                    if field_need_paren {
766                        result = format!("({result})");
767                    }
768                    result = format!("{result}.{field}");
769                    field_need_paren = false;
770                }
771                &ProjectionElem::ClosureField(field) => {
772                    if field_need_paren {
773                        result = format!("({result})");
774                    }
775                    result = format!("{result}.{field}");
776                    field_need_paren = false;
777                }
778                ProjectionElem::Index(_)
779                | ProjectionElem::ConstantIndex { .. }
780                | ProjectionElem::Subslice { .. }
781                | ProjectionElem::OpaqueCast(_) => {
782                    never!("Not happen in closure capture");
783                    continue;
784                }
785            }
786        }
787        result
788    }

799impl CapturedItemWithoutTy {
800    fn with_ty(self, ctx: &mut InferenceContext<'_>) -> CapturedItem {
801        let ty = self.place.ty(ctx);
802        let ty = match &self.kind {
803            CaptureKind::ByValue => ty,
804            CaptureKind::ByRef(bk) => {
805                let m = match bk {
806                    BorrowKind::Mut { .. } => Mutability::Mut,
807                    _ => Mutability::Not,
808                };
809                TyKind::Ref(m, error_lifetime(), ty).intern(Interner)
810            }
811        };
812        return CapturedItem {
813            place: self.place,
814            kind: self.kind,
815            span_stacks: self.span_stacks,
816            ty: replace_placeholder_with_binder(ctx, ty),
817        };
818
819        fn replace_placeholder_with_binder(ctx: &mut InferenceContext<'_>, ty: Ty) -> Binders<Ty> {
820            struct Filler<'a> {
821                db: &'a dyn HirDatabase,
822                generics: &'a Generics,
823            }
824            impl FallibleTypeFolder<Interner> for Filler<'_> {
825                type Error = ();
826
827                fn as_dyn(&mut self) -> &mut dyn FallibleTypeFolder<Interner, Error = Self::Error> {
828                    self
829                }
830
831                fn interner(&self) -> Interner {
832                    Interner
833                }
834
835                fn try_fold_free_placeholder_const(
836                    &mut self,
837                    ty: chalk_ir::Ty<Interner>,
838                    idx: chalk_ir::PlaceholderIndex,
839                    outer_binder: DebruijnIndex,
840                ) -> Result<chalk_ir::Const<Interner>, Self::Error> {
841                    let x = from_placeholder_idx(self.db, idx);
842                    let Some(idx) = self.generics.type_or_const_param_idx(x) else {
843                        return Err(());
844                    };
845                    Ok(BoundVar::new(outer_binder, idx).to_const(Interner, ty))
846                }
847
848                fn try_fold_free_placeholder_ty(
849                    &mut self,
850                    idx: chalk_ir::PlaceholderIndex,
851                    outer_binder: DebruijnIndex,
852                ) -> std::result::Result<Ty, Self::Error> {
853                    let x = from_placeholder_idx(self.db, idx);
854                    let Some(idx) = self.generics.type_or_const_param_idx(x) else {
855                        return Err(());
856                    };
857                    Ok(BoundVar::new(outer_binder, idx).to_ty(Interner))
858                }
859            }
860            let filler = &mut Filler { db: ctx.db, generics: ctx.generics() };
861            let result = ty.clone().try_fold_with(filler, DebruijnIndex::INNERMOST).unwrap_or(ty);
862            make_binders(ctx.db, filler.generics, result)
863        }
864    }
865}

867impl InferenceContext<'_> {
868    fn place_of_expr(&mut self, tgt_expr: ExprId) -> Option<HirPlace> {
869        let r = self.place_of_expr_without_adjust(tgt_expr)?;
870        let default = vec![];
871        let adjustments = self.result.expr_adjustments.get(&tgt_expr).unwrap_or(&default);
872        apply_adjusts_to_place(&mut self.current_capture_span_stack, r, adjustments)
873    }

875    /// Pushes the span into `current_capture_span_stack`, *without clearing it first*.
876    fn path_place(&mut self, path: &Path, id: ExprOrPatId) -> Option<HirPlace> {
877        if path.type_anchor().is_some() {
878            return None;
879        }
880        let hygiene = self.body.expr_or_pat_path_hygiene(id);
881        self.resolver.resolve_path_in_value_ns_fully(self.db, path, hygiene).and_then(|result| {
882            match result {
883                ValueNs::LocalBinding(binding) => {
884                    let mir_span = match id {
885                        ExprOrPatId::ExprId(id) => MirSpan::ExprId(id),
886                        ExprOrPatId::PatId(id) => MirSpan::PatId(id),
887                    };
888                    self.current_capture_span_stack.push(mir_span);
889                    Some(HirPlace { local: binding, projections: Vec::new() })
890                }
891                _ => None,
892            }
893        })
894    }
895
896    /// Changes `current_capture_span_stack` to contain the stack of spans for this expr.
897    fn place_of_expr_without_adjust(&mut self, tgt_expr: ExprId) -> Option<HirPlace> {
898        self.current_capture_span_stack.clear();
899        match &self.body[tgt_expr] {
900            Expr::Path(p) => {
901                let resolver_guard =
902                    self.resolver.update_to_inner_scope(self.db, self.owner, tgt_expr);
903                let result = self.path_place(p, tgt_expr.into());
904                self.resolver.reset_to_guard(resolver_guard);
905                return result;
906            }
907            Expr::Field { expr, name: _ } => {
908                let mut place = self.place_of_expr(*expr)?;
909                let field = self.result.field_resolution(tgt_expr)?;
910                self.current_capture_span_stack.push(MirSpan::ExprId(tgt_expr));
911                place.projections.push(ProjectionElem::Field(field));
912                return Some(place);
913            }
914            Expr::UnaryOp { expr, op: UnaryOp::Deref } => {
915                if matches!(
916                    self.expr_ty_after_adjustments(*expr).kind(Interner),
917                    TyKind::Ref(..) | TyKind::Raw(..)
918                ) {
919                    let mut place = self.place_of_expr(*expr)?;
920                    self.current_capture_span_stack.push(MirSpan::ExprId(tgt_expr));
921                    place.projections.push(ProjectionElem::Deref);
922                    return Some(place);
923                }
924            }
925            _ => (),
926        }
927        None
928    }

930    fn push_capture(&mut self, place: HirPlace, kind: CaptureKind) {
931        self.current_captures.push(CapturedItemWithoutTy {
932            place,
933            kind,
934            span_stacks: smallvec![self.current_capture_span_stack.iter().copied().collect()],
935        });
936    }
937
938    fn truncate_capture_spans(&self, capture: &mut CapturedItemWithoutTy, mut truncate_to: usize) {
939        // The first span is the identifier, and it must always remain.
940        truncate_to += 1;
941        for span_stack in &mut capture.span_stacks {
942            let mut remained = truncate_to;
943            let mut actual_truncate_to = 0;
944            for &span in &*span_stack {
945                actual_truncate_to += 1;
946                if !span.is_ref_span(self.body) {
947                    remained -= 1;
948                    if remained == 0 {
949                        break;
950                    }
951                }
952            }
953            if actual_truncate_to < span_stack.len()
954                && span_stack[actual_truncate_to].is_ref_span(self.body)
955            {
956                // Include the ref operator if there is one, we will fix it later (in `strip_captures_ref_span()`) if it's incorrect.
957                actual_truncate_to += 1;
958            }
959            span_stack.truncate(actual_truncate_to);
960        }
961    }

963    fn ref_expr(&mut self, expr: ExprId, place: Option<HirPlace>) {
964        if let Some(place) = place {
965            self.add_capture(place, CaptureKind::ByRef(BorrowKind::Shared));
966        }
967        self.walk_expr(expr);
968    }
969
970    fn add_capture(&mut self, place: HirPlace, kind: CaptureKind) {
971        if self.is_upvar(&place) {
972            self.push_capture(place, kind);
973        }
974    }

976    fn mutate_path_pat(&mut self, path: &Path, id: PatId) {
977        if let Some(place) = self.path_place(path, id.into()) {
978            self.add_capture(
979                place,
980                CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::Default }),
981            );
982            self.current_capture_span_stack.pop(); // Remove the pattern span.
983        }
984    }
985
986    fn mutate_expr(&mut self, expr: ExprId, place: Option<HirPlace>) {
987        if let Some(place) = place {
988            self.add_capture(
989                place,
990                CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::Default }),
991            );
992        }
993        self.walk_expr(expr);
994    }
995
996    fn consume_expr(&mut self, expr: ExprId) {
997        if let Some(place) = self.place_of_expr(expr) {
998            self.consume_place(place);
999        }
1000        self.walk_expr(expr);
1001    }
1002
1003    fn consume_place(&mut self, place: HirPlace) {
1004        if self.is_upvar(&place) {
1005            let ty = place.ty(self);
1006            let kind = if self.is_ty_copy(ty) {
1007                CaptureKind::ByRef(BorrowKind::Shared)
1008            } else {
1009                CaptureKind::ByValue
1010            };
1011            self.push_capture(place, kind);
1012        }
1013    }
1014
1015    fn walk_expr_with_adjust(&mut self, tgt_expr: ExprId, adjustment: &[Adjustment]) {
1016        if let Some((last, rest)) = adjustment.split_last() {
1017            match &last.kind {
1018                Adjust::NeverToAny | Adjust::Deref(None) | Adjust::Pointer(_) => {
1019                    self.walk_expr_with_adjust(tgt_expr, rest)
1020                }
1021                Adjust::Deref(Some(m)) => match m.0 {
1022                    Some(m) => {
1023                        self.ref_capture_with_adjusts(m, tgt_expr, rest);
1024                    }
1025                    None => unreachable!(),
1026                },
1027                Adjust::Borrow(b) => {
1028                    self.ref_capture_with_adjusts(b.mutability(), tgt_expr, rest);
1029                }
1030            }
1031        } else {
1032            self.walk_expr_without_adjust(tgt_expr);
1033        }
1034    }
1035
1036    fn ref_capture_with_adjusts(&mut self, m: Mutability, tgt_expr: ExprId, rest: &[Adjustment]) {
1037        let capture_kind = match m {
1038            Mutability::Mut => CaptureKind::ByRef(BorrowKind::Mut { kind: MutBorrowKind::Default }),
1039            Mutability::Not => CaptureKind::ByRef(BorrowKind::Shared),
1040        };
1041        if let Some(place) = self.place_of_expr_without_adjust(tgt_expr) {
1042            if let Some(place) =
1043                apply_adjusts_to_place(&mut self.current_capture_span_stack, place, rest)
1044            {
1045                self.add_capture(place, capture_kind);
1046            }
1047        }
1048        self.walk_expr_with_adjust(tgt_expr, rest);
1049    }

1051    fn walk_expr(&mut self, tgt_expr: ExprId) {
1052        if let Some(it) = self.result.expr_adjustments.get_mut(&tgt_expr) {
1053            // FIXME: this take is completely unneeded, and just is here to make borrow checker
1054            // happy. Remove it if you can.
1055            let x_taken = mem::take(it);
1056            self.walk_expr_with_adjust(tgt_expr, &x_taken);
1057            *self.result.expr_adjustments.get_mut(&tgt_expr).unwrap() = x_taken;
1058        } else {
1059            self.walk_expr_without_adjust(tgt_expr);
1060        }
1061    }
1062
1063    fn walk_expr_without_adjust(&mut self, tgt_expr: ExprId) {
1064        match &self.body[tgt_expr] {
1065            Expr::OffsetOf(_) => (),
1066            Expr::InlineAsm(e) => e.operands.iter().for_each(|(_, op)| match op {
1067                AsmOperand::In { expr, .. }
1068                | AsmOperand::Out { expr: Some(expr), .. }
1069                | AsmOperand::InOut { expr, .. } => self.walk_expr_without_adjust(*expr),
1070                AsmOperand::SplitInOut { in_expr, out_expr, .. } => {
1071                    self.walk_expr_without_adjust(*in_expr);
1072                    if let Some(out_expr) = out_expr {
1073                        self.walk_expr_without_adjust(*out_expr);
1074                    }
1075                }
1076                AsmOperand::Out { expr: None, .. }
1077                | AsmOperand::Const(_)
1078                | AsmOperand::Label(_)
1079                | AsmOperand::Sym(_) => (),
1080            }),
1081            Expr::If { condition, then_branch, else_branch } => {
1082                self.consume_expr(*condition);
1083                self.consume_expr(*then_branch);
1084                if let &Some(expr) = else_branch {
1085                    self.consume_expr(expr);
1086                }
1087            }
1088            Expr::Async { statements, tail, .. }
1089            | Expr::Unsafe { statements, tail, .. }
1090            | Expr::Block { statements, tail, .. } => {
1091                for s in statements.iter() {
1092                    match s {
1093                        Statement::Let { pat, type_ref: _, initializer, else_branch } => {
1094                            if let Some(else_branch) = else_branch {
1095                                self.consume_expr(*else_branch);
1096                            }
1097                            if let Some(initializer) = initializer {
1098                                if else_branch.is_some() {
1099                                    self.consume_expr(*initializer);
1100                                } else {
1101                                    self.walk_expr(*initializer);
1102                                }
1103                                if let Some(place) = self.place_of_expr(*initializer) {
1104                                    self.consume_with_pat(place, *pat);
1105                                }
1106                            }
1107                        }
1108                        Statement::Expr { expr, has_semi: _ } => {
1109                            self.consume_expr(*expr);
1110                        }
1111                        Statement::Item(_) => (),
1112                    }
1113                }
1114                if let Some(tail) = tail {
1115                    self.consume_expr(*tail);
1116                }
1117            }
1118            Expr::Call { callee, args } => {
1119                self.consume_expr(*callee);
1120                self.consume_exprs(args.iter().copied());
1121            }
1122            Expr::MethodCall { receiver, args, .. } => {
1123                self.consume_expr(*receiver);
1124                self.consume_exprs(args.iter().copied());
1125            }
1126            Expr::Match { expr, arms } => {
1127                for arm in arms.iter() {
1128                    self.consume_expr(arm.expr);
1129                    if let Some(guard) = arm.guard {
1130                        self.consume_expr(guard);
1131                    }
1132                }
1133                self.walk_expr(*expr);
1134                if let Some(discr_place) = self.place_of_expr(*expr) {
1135                    if self.is_upvar(&discr_place) {
1136                        let mut capture_mode = None;
1137                        for arm in arms.iter() {
1138                            self.walk_pat(&mut capture_mode, arm.pat);
1139                        }
1140                        if let Some(c) = capture_mode {
1141                            self.push_capture(discr_place, c);
1142                        }
1143                    }
1144                }
1145            }
1146            Expr::Break { expr, label: _ }
1147            | Expr::Return { expr }
1148            | Expr::Yield { expr }
1149            | Expr::Yeet { expr } => {
1150                if let &Some(expr) = expr {
1151                    self.consume_expr(expr);
1152                }
1153            }
1154            &Expr::Become { expr } => {
1155                self.consume_expr(expr);
1156            }
1157            Expr::RecordLit { fields, spread, .. } => {
1158                if let &Some(expr) = spread {
1159                    self.consume_expr(expr);
1160                }
1161                self.consume_exprs(fields.iter().map(|it| it.expr));
1162            }
1163            Expr::Field { expr, name: _ } => self.select_from_expr(*expr),
1164            Expr::UnaryOp { expr, op: UnaryOp::Deref } => {
1165                if matches!(
1166                    self.expr_ty_after_adjustments(*expr).kind(Interner),
1167                    TyKind::Ref(..) | TyKind::Raw(..)
1168                ) {
1169                    self.select_from_expr(*expr);
1170                } else if let Some((f, _)) = self.result.method_resolution(tgt_expr) {
1171                    let mutability = 'b: {
1172                        if let Some(deref_trait) =
1173                            self.resolve_lang_item(LangItem::DerefMut).and_then(|it| it.as_trait())
1174                        {
1175                            if let Some(deref_fn) = self
1176                                .db
1177                                .trait_items(deref_trait)
1178                                .method_by_name(&Name::new_symbol_root(sym::deref_mut.clone()))
1179                            {
1180                                break 'b deref_fn == f;
1181                            }
1182                        }
1183                        false
1184                    };
1185                    let place = self.place_of_expr(*expr);
1186                    if mutability {
1187                        self.mutate_expr(*expr, place);
1188                    } else {
1189                        self.ref_expr(*expr, place);
1190                    }
1191                } else {
1192                    self.select_from_expr(*expr);
1193                }
1194            }
1195            Expr::UnaryOp { expr, op: _ }
1196            | Expr::Array(Array::Repeat { initializer: expr, repeat: _ })
1197            | Expr::Await { expr }
1198            | Expr::Loop { body: expr, label: _ }
1199            | Expr::Let { pat: _, expr }
1200            | Expr::Box { expr }
1201            | Expr::Cast { expr, type_ref: _ } => {
1202                self.consume_expr(*expr);
1203            }
1204            Expr::Ref { expr, rawness: _, mutability } => {
1205                // We need to do this before we push the span so the order will be correct.
1206                let place = self.place_of_expr(*expr);
1207                self.current_capture_span_stack.push(MirSpan::ExprId(tgt_expr));
1208                match mutability {
1209                    hir_def::type_ref::Mutability::Shared => self.ref_expr(*expr, place),
1210                    hir_def::type_ref::Mutability::Mut => self.mutate_expr(*expr, place),
1211                }
1212            }
1213            Expr::BinaryOp { lhs, rhs, op } => {
1214                let Some(op) = op else {
1215                    return;
1216                };
1217                if matches!(op, BinaryOp::Assignment { .. }) {
1218                    let place = self.place_of_expr(*lhs);
1219                    self.mutate_expr(*lhs, place);
1220                    self.consume_expr(*rhs);
1221                    return;
1222                }
1223                self.consume_expr(*lhs);
1224                self.consume_expr(*rhs);
1225            }
1226            Expr::Range { lhs, rhs, range_type: _ } => {
1227                if let &Some(expr) = lhs {
1228                    self.consume_expr(expr);
1229                }
1230                if let &Some(expr) = rhs {
1231                    self.consume_expr(expr);
1232                }
1233            }
1234            Expr::Index { base, index } => {
1235                self.select_from_expr(*base);
1236                self.consume_expr(*index);
1237            }
1238            Expr::Closure { .. } => {
1239                let ty = self.expr_ty(tgt_expr);
1240                let TyKind::Closure(id, _) = ty.kind(Interner) else {
1241                    never!("closure type is always closure");
1242                    return;
1243                };
1244                let (captures, _) =
1245                    self.result.closure_info.get(id).expect(
1246                        "We sort closures, so we should always have data for inner closures",
1247                    );
1248                let mut cc = mem::take(&mut self.current_captures);
1249                cc.extend(captures.iter().filter(|it| self.is_upvar(&it.place)).map(|it| {
1250                    CapturedItemWithoutTy {
1251                        place: it.place.clone(),
1252                        kind: it.kind,
1253                        span_stacks: it.span_stacks.clone(),
1254                    }
1255                }));
1256                self.current_captures = cc;
1257            }
1258            Expr::Array(Array::ElementList { elements: exprs }) | Expr::Tuple { exprs } => {
1259                self.consume_exprs(exprs.iter().copied())
1260            }
1261            &Expr::Assignment { target, value } => {
1262                self.walk_expr(value);
1263                let resolver_guard =
1264                    self.resolver.update_to_inner_scope(self.db, self.owner, tgt_expr);
1265                match self.place_of_expr(value) {
1266                    Some(rhs_place) => {
1267                        self.inside_assignment = true;
1268                        self.consume_with_pat(rhs_place, target);
1269                        self.inside_assignment = false;
1270                    }
1271                    None => self.body.walk_pats(target, &mut |pat| match &self.body[pat] {
1272                        Pat::Path(path) => self.mutate_path_pat(path, pat),
1273                        &Pat::Expr(expr) => {
1274                            let place = self.place_of_expr(expr);
1275                            self.mutate_expr(expr, place);
1276                        }
1277                        _ => {}
1278                    }),
1279                }
1280                self.resolver.reset_to_guard(resolver_guard);
1281            }
1282
1283            Expr::Missing
1284            | Expr::Continue { .. }
1285            | Expr::Path(_)
1286            | Expr::Literal(_)
1287            | Expr::Const(_)
1288            | Expr::Underscore => (),
1289        }
1290    }

1292    fn walk_pat(&mut self, result: &mut Option<CaptureKind>, pat: PatId) {
1293        let mut update_result = |ck: CaptureKind| match result {
1294            Some(r) => {
1295                *r = cmp::max(*r, ck);
1296            }
1297            None => *result = Some(ck),
1298        };
1299
1300        self.walk_pat_inner(
1301            pat,
1302            &mut update_result,
1303            BorrowKind::Mut { kind: MutBorrowKind::Default },
1304        );
1305    }

1307    fn walk_pat_inner(
1308        &mut self,
1309        p: PatId,
1310        update_result: &mut impl FnMut(CaptureKind),
1311        mut for_mut: BorrowKind,
1312    ) {
1313        match &self.body[p] {
1314            Pat::Ref { .. }
1315            | Pat::Box { .. }
1316            | Pat::Missing
1317            | Pat::Wild
1318            | Pat::Tuple { .. }
1319            | Pat::Expr(_)
1320            | Pat::Or(_) => (),
1321            Pat::TupleStruct { .. } | Pat::Record { .. } => {
1322                if let Some(variant) = self.result.variant_resolution_for_pat(p) {
1323                    let adt = variant.adt_id(self.db);
1324                    let is_multivariant = match adt {
1325                        hir_def::AdtId::EnumId(e) => self.db.enum_variants(e).variants.len() != 1,
1326                        _ => false,
1327                    };
1328                    if is_multivariant {
1329                        update_result(CaptureKind::ByRef(BorrowKind::Shared));
1330                    }
1331                }
1332            }
1333            Pat::Slice { .. }
1334            | Pat::ConstBlock(_)
1335            | Pat::Path(_)
1336            | Pat::Lit(_)
1337            | Pat::Range { .. } => {
1338                update_result(CaptureKind::ByRef(BorrowKind::Shared));
1339            }
1340            Pat::Bind { id, .. } => match self.result.binding_modes[p] {
1341                crate::BindingMode::Move => {
1342                    if self.is_ty_copy(self.result.type_of_binding[*id].clone()) {
1343                        update_result(CaptureKind::ByRef(BorrowKind::Shared));
1344                    } else {
1345                        update_result(CaptureKind::ByValue);
1346                    }
1347                }
1348                crate::BindingMode::Ref(r) => match r {
1349                    Mutability::Mut => update_result(CaptureKind::ByRef(for_mut)),
1350                    Mutability::Not => update_result(CaptureKind::ByRef(BorrowKind::Shared)),
1351                },
1352            },
1353        }
1354        if self.result.pat_adjustments.get(&p).is_some_and(|it| !it.is_empty()) {
1355            for_mut = BorrowKind::Mut { kind: MutBorrowKind::ClosureCapture };
1356        }
1357        self.body.walk_pats_shallow(p, |p| self.walk_pat_inner(p, update_result, for_mut));
1358    }

1396    fn restrict_precision_for_unsafe(&mut self) {
1397        // FIXME: Borrow checker problems without this.
1398        let mut current_captures = std::mem::take(&mut self.current_captures);
1399        for capture in &mut current_captures {
1400            let mut ty = self.table.resolve_completely(self.result[capture.place.local].clone());
1401            if ty.as_raw_ptr().is_some() || ty.is_union() {
1402                capture.kind = CaptureKind::ByRef(BorrowKind::Shared);
1403                self.truncate_capture_spans(capture, 0);
1404                capture.place.projections.truncate(0);
1405                continue;
1406            }
1407            for (i, p) in capture.place.projections.iter().enumerate() {
1408                ty = p.projected_ty(
1409                    ty,
1410                    self.db,
1411                    |_, _, _| {
1412                        unreachable!("Closure field only happens in MIR");
1413                    },
1414                    self.owner.module(self.db).krate(),
1415                );
1416                if ty.as_raw_ptr().is_some() || ty.is_union() {
1417                    capture.kind = CaptureKind::ByRef(BorrowKind::Shared);
1418                    self.truncate_capture_spans(capture, i + 1);
1419                    capture.place.projections.truncate(i + 1);
1420                    break;
1421                }
1422            }
1423        }
1424        self.current_captures = current_captures;
1425    }

1428        // FIXME: Borrow checker won't allow without this.
1429        let mut current_captures = std::mem::take(&mut self.current_captures);

1433            {

1442    fn minimize_captures(&mut self) {
1443        self.current_captures.sort_unstable_by_key(|it| it.place.projections.len());
1444        let mut hash_map = FxHashMap::<HirPlace, usize>::default();
1445        let result = mem::take(&mut self.current_captures);
1446        for mut item in result {
1447            let mut lookup_place = HirPlace { local: item.place.local, projections: vec![] };
1448            let mut it = item.place.projections.iter();
1449            let prev_index = loop {
1450                if let Some(k) = hash_map.get(&lookup_place) {
1451                    break Some(*k);
1452                }
1453                match it.next() {
1454                    Some(it) => {
1455                        lookup_place.projections.push(it.clone());
1456                    }
1457                    None => break None,
1458                }
1459            };
1460            match prev_index {
1461                Some(p) => {
1462                    let prev_projections_len = self.current_captures[p].place.projections.len();
1463                    self.truncate_capture_spans(&mut item, prev_projections_len);
1464                    self.current_captures[p].span_stacks.extend(item.span_stacks);
1465                    let len = self.current_captures[p].place.projections.len();
1466                    let kind_after_truncate =
1467                        item.place.capture_kind_of_truncated_place(item.kind, len);
1468                    self.current_captures[p].kind =
1469                        cmp::max(kind_after_truncate, self.current_captures[p].kind);
1470                }
1471                None => {
1472                    hash_map.insert(item.place.clone(), self.current_captures.len());
1473                    self.current_captures.push(item);
1474                }
1475            }
1476        }
1477    }

1479    fn consume_with_pat(&mut self, mut place: HirPlace, tgt_pat: PatId) {
1480        let adjustments_count =
1481            self.result.pat_adjustments.get(&tgt_pat).map(|it| it.len()).unwrap_or_default();
1482        place.projections.extend((0..adjustments_count).map(|_| ProjectionElem::Deref));
1483        self.current_capture_span_stack
1484            .extend((0..adjustments_count).map(|_| MirSpan::PatId(tgt_pat)));
1485        'reset_span_stack: {
1486            match &self.body[tgt_pat] {
1487                Pat::Missing | Pat::Wild => (),
1488                Pat::Tuple { args, ellipsis } => {
1489                    let (al, ar) = args.split_at(ellipsis.map_or(args.len(), |it| it as usize));
1490                    let field_count = match self.result[tgt_pat].kind(Interner) {
1491                        TyKind::Tuple(_, s) => s.len(Interner),
1492                        _ => break 'reset_span_stack,
1493                    };
1494                    let fields = 0..field_count;
1495                    let it = al.iter().zip(fields.clone()).chain(ar.iter().rev().zip(fields.rev()));
1496                    for (&arg, i) in it {
1497                        let mut p = place.clone();
1498                        self.current_capture_span_stack.push(MirSpan::PatId(arg));
1499                        p.projections.push(ProjectionElem::Field(Either::Right(TupleFieldId {
1500                            tuple: TupleId(!0), // dummy this, as its unused anyways
1501                            index: i as u32,
1502                        })));
1503                        self.consume_with_pat(p, arg);
1504                        self.current_capture_span_stack.pop();
1505                    }
1506                }
1507                Pat::Or(pats) => {
1508                    for pat in pats.iter() {
1509                        self.consume_with_pat(place.clone(), *pat);
1510                    }
1511                }
1512                Pat::Record { args, .. } => {
1513                    let Some(variant) = self.result.variant_resolution_for_pat(tgt_pat) else {
1514                        break 'reset_span_stack;
1515                    };
1516                    match variant {
1517                        VariantId::EnumVariantId(_) | VariantId::UnionId(_) => {
1518                            self.consume_place(place)
1519                        }
1520                        VariantId::StructId(s) => {
1521                            let vd = &*self.db.variant_fields(s.into());
1522                            for field_pat in args.iter() {
1523                                let arg = field_pat.pat;
1524                                let Some(local_id) = vd.field(&field_pat.name) else {
1525                                    continue;
1526                                };
1527                                let mut p = place.clone();
1528                                self.current_capture_span_stack.push(MirSpan::PatId(arg));
1529                                p.projections.push(ProjectionElem::Field(Either::Left(FieldId {
1530                                    parent: variant,
1531                                    local_id,
1532                                })));
1533                                self.consume_with_pat(p, arg);
1534                                self.current_capture_span_stack.pop();
1535                            }
1536                        }
1537                    }
1538                }
1539                Pat::Range { .. } | Pat::Slice { .. } | Pat::ConstBlock(_) | Pat::Lit(_) => {
1540                    self.consume_place(place)
1541                }
1542                Pat::Path(path) => {
1543                    if self.inside_assignment {
1544                        self.mutate_path_pat(path, tgt_pat);
1545                    }
1546                    self.consume_place(place);
1547                }
1548                &Pat::Bind { id, subpat: _ } => {
1549                    let mode = self.result.binding_modes[tgt_pat];
1550                    let capture_kind = match mode {
1551                        BindingMode::Move => {
1552                            self.consume_place(place);
1553                            break 'reset_span_stack;
1554                        }
1555                        BindingMode::Ref(Mutability::Not) => BorrowKind::Shared,
1556                        BindingMode::Ref(Mutability::Mut) => {
1557                            BorrowKind::Mut { kind: MutBorrowKind::Default }
1558                        }
1559                    };
1560                    self.current_capture_span_stack.push(MirSpan::BindingId(id));
1561                    self.add_capture(place, CaptureKind::ByRef(capture_kind));
1562                    self.current_capture_span_stack.pop();
1563                }
1564                Pat::TupleStruct { path: _, args, ellipsis } => {
1565                    let Some(variant) = self.result.variant_resolution_for_pat(tgt_pat) else {
1566                        break 'reset_span_stack;
1567                    };
1568                    match variant {
1569                        VariantId::EnumVariantId(_) | VariantId::UnionId(_) => {
1570                            self.consume_place(place)
1571                        }
1572                        VariantId::StructId(s) => {
1573                            let vd = &*self.db.variant_fields(s.into());
1574                            let (al, ar) =
1575                                args.split_at(ellipsis.map_or(args.len(), |it| it as usize));
1576                            let fields = vd.fields().iter();
1577                            let it = al
1578                                .iter()
1579                                .zip(fields.clone())
1580                                .chain(ar.iter().rev().zip(fields.rev()));
1581                            for (&arg, (i, _)) in it {
1582                                let mut p = place.clone();
1583                                self.current_capture_span_stack.push(MirSpan::PatId(arg));
1584                                p.projections.push(ProjectionElem::Field(Either::Left(FieldId {
1585                                    parent: variant,
1586                                    local_id: i,
1587                                })));
1588                                self.consume_with_pat(p, arg);
1589                                self.current_capture_span_stack.pop();
1590                            }
1591                        }
1592                    }
1593                }
1594                Pat::Ref { pat, mutability: _ } => {
1595                    self.current_capture_span_stack.push(MirSpan::PatId(tgt_pat));
1596                    place.projections.push(ProjectionElem::Deref);
1597                    self.consume_with_pat(place, *pat);
1598                    self.current_capture_span_stack.pop();
1599                }
1600                Pat::Box { .. } => (), // not supported
1601                &Pat::Expr(expr) => {
1602                    self.consume_place(place);
1603                    let pat_capture_span_stack = mem::take(&mut self.current_capture_span_stack);
1604                    let old_inside_assignment = mem::replace(&mut self.inside_assignment, false);
1605                    let lhs_place = self.place_of_expr(expr);
1606                    self.mutate_expr(expr, lhs_place);
1607                    self.inside_assignment = old_inside_assignment;
1608                    self.current_capture_span_stack = pat_capture_span_stack;
1609                }
1610            }
1611        }
1612        self.current_capture_span_stack
1613            .truncate(self.current_capture_span_stack.len() - adjustments_count);
1614    }

1637    fn analyze_closure(&mut self, closure: ClosureId) -> FnTrait {
1638        let InternedClosure(_, root) = self.db.lookup_intern_closure(closure.into());
1639        self.current_closure = Some(closure);
1640        let Expr::Closure { body, capture_by, .. } = &self.body[root] else {
1641            unreachable!("Closure expression id is always closure");
1642        };
1643        self.consume_expr(*body);
1644        for item in &self.current_captures {
1645            if matches!(
1646                item.kind,
1647                CaptureKind::ByRef(BorrowKind::Mut {
1648                    kind: MutBorrowKind::Default | MutBorrowKind::TwoPhasedBorrow
1649                })
1650            ) && !item.place.projections.contains(&ProjectionElem::Deref)
1651            {
1652                // FIXME: remove the `mutated_bindings_in_closure` completely and add proper fake reads in
1653                // MIR. I didn't do that due duplicate diagnostics.
1654                self.result.mutated_bindings_in_closure.insert(item.place.local);
1655            }
1656        }
1657        self.restrict_precision_for_unsafe();
1658        // `closure_kind` should be done before adjust_for_move_closure
1659        // If there exists pre-deduced kind of a closure, use it instead of one determined by capture, as rustc does.
1660        // rustc also does diagnostics here if the latter is not a subtype of the former.
1661        let closure_kind = self
1662            .result
1663            .closure_info
1664            .get(&closure)
1665            .map_or_else(|| self.closure_kind(), |info| info.1);
1666        match capture_by {
1667            CaptureBy::Value => self.adjust_for_move_closure(),
1668            CaptureBy::Ref => (),
1669        }
1670        self.minimize_captures();
1671        self.strip_captures_ref_span();
1672        let result = mem::take(&mut self.current_captures);
1673        let captures = result.into_iter().map(|it| it.with_ty(self)).collect::<Vec<_>>();
1674        self.result.closure_info.insert(closure, (captures, closure_kind));
1675        closure_kind
1676    }

1678    fn strip_captures_ref_span(&mut self) {
1679        // FIXME: Borrow checker won't allow without this.
1680        let mut captures = std::mem::take(&mut self.current_captures);
1681        for capture in &mut captures {
1682            if matches!(capture.kind, CaptureKind::ByValue) {

1686                    }
1687                }
1688            }
1689        }
1690        self.current_captures = captures;
1691    }

1693    pub(crate) fn infer_closures(&mut self) {
1694        let deferred_closures = self.sort_closures();
1695        for (closure, exprs) in deferred_closures.into_iter().rev() {
1696            self.current_captures = vec![];
1697            let kind = self.analyze_closure(closure);
1698
1699            for (derefed_callee, callee_ty, params, expr) in exprs {
1700                if let &Expr::Call { callee, .. } = &self.body[expr] {
1701                    let mut adjustments =
1702                        self.result.expr_adjustments.remove(&callee).unwrap_or_default();
1703                    self.write_fn_trait_method_resolution(
1704                        kind,
1705                        &derefed_callee,
1706                        &mut adjustments,
1707                        &callee_ty,
1708                        &params,
1709                        expr,
1710                    );
1711                    self.result.expr_adjustments.insert(callee, adjustments);
1712                }
1713            }
1714        }
1715    }

1717    /// We want to analyze some closures before others, to have a correct analysis:
1718    /// * We should analyze nested closures before the parent, since the parent should capture some of
1719    ///   the things that its children captures.
1720    /// * If a closure calls another closure, we need to analyze the callee, to find out how we should
1721    ///   capture it (e.g. by move for FnOnce)
1722    ///
1723    /// These dependencies are collected in the main inference. We do a topological sort in this function. It
1724    /// will consume the `deferred_closures` field and return its content in a sorted vector.
1725    fn sort_closures(&mut self) -> Vec<(ClosureId, Vec<(Ty, Ty, Vec<Ty>, ExprId)>)> {
1726        let mut deferred_closures = mem::take(&mut self.deferred_closures);
1727        let mut dependents_count: FxHashMap<ClosureId, usize> =
1728            deferred_closures.keys().map(|it| (*it, 0)).collect();
1729        for deps in self.closure_dependencies.values() {
1730            for dep in deps {
1731                *dependents_count.entry(*dep).or_default() += 1;
1732            }
1733        }
1734        let mut queue: Vec<_> =
1735            deferred_closures.keys().copied().filter(|it| dependents_count[it] == 0).collect();
1736        let mut result = vec![];
1737        while let Some(it) = queue.pop() {
1738            if let Some(d) = deferred_closures.remove(&it) {
1739                result.push((it, d));
1740            }
1741            for dep in self.closure_dependencies.get(&it).into_iter().flat_map(|it| it.iter()) {
1742                let cnt = dependents_count.get_mut(dep).unwrap();
1743                *cnt -= 1;
1744                if *cnt == 0 {
1745                    queue.push(*dep);
1746                }
1747            }
1748        }
1749        assert!(deferred_closures.is_empty(), "we should have analyzed all closures");
1750        result
1751    }

1753    pub(super) fn add_current_closure_dependency(&mut self, dep: ClosureId) {
1754        if let Some(c) = self.current_closure {
1755            if !dep_creates_cycle(&self.closure_dependencies, &mut FxHashSet::default(), c, dep) {
1756                self.closure_dependencies.entry(c).or_default().push(dep);
1757            }
1758        }
1759
1760        fn dep_creates_cycle(
1761            closure_dependencies: &FxHashMap<ClosureId, Vec<ClosureId>>,
1762            visited: &mut FxHashSet<ClosureId>,
1763            from: ClosureId,
1764            to: ClosureId,
1765        ) -> bool {
1766            if !visited.insert(from) {
1767                return false;
1768            }
1769
1770            if from == to {
1771                return true;
1772            }
1773
1774            if let Some(deps) = closure_dependencies.get(&to) {
1775                for dep in deps {
1776                    if dep_creates_cycle(closure_dependencies, visited, from, *dep) {
1777                        return true;
1778                    }
1779                }
1780            }
1781
1782            false
1783        }
1784    }
1785}
