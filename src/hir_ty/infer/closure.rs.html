42struct ClosureSignatures<'db> {
43    /// The signature users of the closure see.
44    bound_sig: PolyFnSig<'db>,
45    /// The signature within the function body.
46    /// This mostly differs in the sense that lifetimes are now early bound and any
47    /// opaque types from the signature expectation are overridden in case there are

53    pub(super) fn infer_closure(
54        &mut self,
55        body: ExprId,
56        args: &[PatId],
57        ret_type: Option<TypeRefId>,
58        arg_types: &[Option<TypeRefId>],
59        closure_kind: ClosureKind,
60        tgt_expr: ExprId,
61        expected: &Expectation<'db>,
62    ) -> Ty<'db> {
63        assert_eq!(args.len(), arg_types.len());
64
65        let interner = self.interner();
66        let (expected_sig, expected_kind) = match expected.to_option(&mut self.table) {
67            Some(expected_ty) => self.deduce_closure_signature(expected_ty, closure_kind),
68            None => (None, None),
69        };
70
71        let ClosureSignatures { bound_sig, liberated_sig } =
72            self.sig_of_closure(arg_types, ret_type, expected_sig);
73        let body_ret_ty = bound_sig.output().skip_binder();
74        let sig_ty = Ty::new_fn_ptr(interner, bound_sig);
75
76        let parent_args = GenericArgs::identity_for_item(interner, self.generic_def.into());
77        // FIXME: Make this an infer var and infer it later.
78        let tupled_upvars_ty = self.types.unit;
79        let (id, ty, resume_yield_tys) = match closure_kind {
80            ClosureKind::Coroutine(_) => {
81                let yield_ty = self.table.next_ty_var();
82                let resume_ty = liberated_sig.inputs().get(0).unwrap_or(self.types.unit);
83
84                // FIXME: Infer the upvars later.
85                let parts = CoroutineArgsParts {
86                    parent_args,
87                    kind_ty: self.types.unit,
88                    resume_ty,
89                    yield_ty,
90                    return_ty: body_ret_ty,
91                    tupled_upvars_ty,
92                };
93
94                let coroutine_id =
95                    self.db.intern_coroutine(InternedCoroutine(self.owner, tgt_expr)).into();
96                let coroutine_ty = Ty::new_coroutine(
97                    interner,
98                    coroutine_id,
99                    CoroutineArgs::new(interner, parts).args,
100                );
101
102                (None, coroutine_ty, Some((resume_ty, yield_ty)))
103            }
104            ClosureKind::Closure => {
105                let closure_id = self.db.intern_closure(InternedClosure(self.owner, tgt_expr));
106                match expected_kind {
107                    Some(kind) => {
108                        self.result.closure_info.insert(
109                            closure_id,
110                            (
111                                Vec::new(),
112                                match kind {
113                                    rustc_type_ir::ClosureKind::Fn => FnTrait::Fn,
114                                    rustc_type_ir::ClosureKind::FnMut => FnTrait::FnMut,
115                                    rustc_type_ir::ClosureKind::FnOnce => FnTrait::FnOnce,
116                                },
117                            ),
118                        );
119                    }
120                    None => {}
121                };
122                // FIXME: Infer the kind later if needed.
123                let parts = ClosureArgsParts {
124                    parent_args,
125                    closure_kind_ty: Ty::from_closure_kind(
126                        interner,
127                        expected_kind.unwrap_or(rustc_type_ir::ClosureKind::Fn),
128                    ),
129                    closure_sig_as_fn_ptr_ty: sig_ty,
130                    tupled_upvars_ty,
131                };
132                let closure_ty = Ty::new_closure(
133                    interner,
134                    closure_id.into(),
135                    ClosureArgs::new(interner, parts).args,
136                );
137                self.deferred_closures.entry(closure_id).or_default();
138                self.add_current_closure_dependency(closure_id);
139                (Some(closure_id), closure_ty, None)
140            }
141            ClosureKind::Async => {
142                // async closures always return the type ascribed after the `->` (if present),
143                // and yield `()`.
144                let bound_return_ty = bound_sig.skip_binder().output();
145                let bound_yield_ty = self.types.unit;
146                // rustc uses a special lang item type for the resume ty. I don't believe this can cause us problems.
147                let resume_ty = self.types.unit;
148
149                // FIXME: Infer the kind later if needed.
150                let closure_kind_ty = Ty::from_closure_kind(
151                    interner,
152                    expected_kind.unwrap_or(rustc_type_ir::ClosureKind::Fn),
153                );
154
155                // FIXME: Infer captures later.
156                // `for<'env> fn() -> ()`, for no captures.
157                let coroutine_captures_by_ref_ty = Ty::new_fn_ptr(
158                    interner,
159                    Binder::bind_with_vars(
160                        interner.mk_fn_sig([], self.types.unit, false, Safety::Safe, FnAbi::Rust),
161                        BoundVarKinds::new_from_iter(
162                            interner,
163                            [BoundVarKind::Region(BoundRegionKind::ClosureEnv)],
164                        ),
165                    ),
166                );
167                let closure_args = CoroutineClosureArgs::new(
168                    interner,
169                    CoroutineClosureArgsParts {
170                        parent_args,
171                        closure_kind_ty,
172                        signature_parts_ty: Ty::new_fn_ptr(
173                            interner,
174                            bound_sig.map_bound(|sig| {
175                                interner.mk_fn_sig(
176                                    [
177                                        resume_ty,
178                                        Ty::new_tup_from_iter(interner, sig.inputs().iter()),
179                                    ],
180                                    Ty::new_tup(interner, &[bound_yield_ty, bound_return_ty]),
181                                    sig.c_variadic,
182                                    sig.safety,
183                                    sig.abi,
184                                )
185                            }),
186                        ),
187                        tupled_upvars_ty,
188                        coroutine_captures_by_ref_ty,
189                    },
190                );
191
192                let coroutine_id =
193                    self.db.intern_coroutine(InternedCoroutine(self.owner, tgt_expr)).into();
194                (None, Ty::new_coroutine_closure(interner, coroutine_id, closure_args.args), None)
195            }
196        };
197
198        // Now go through the argument patterns
199        for (arg_pat, arg_ty) in args.iter().zip(bound_sig.skip_binder().inputs()) {
200            self.infer_top_pat(*arg_pat, arg_ty, None);
201        }
202
203        // FIXME: lift these out into a struct
204        let prev_diverges = mem::replace(&mut self.diverges, Diverges::Maybe);
205        let prev_closure = mem::replace(&mut self.current_closure, id);
206        let prev_ret_ty = mem::replace(&mut self.return_ty, body_ret_ty);
207        let prev_ret_coercion = self.return_coercion.replace(CoerceMany::new(body_ret_ty));
208        let prev_resume_yield_tys = mem::replace(&mut self.resume_yield_tys, resume_yield_tys);
209
210        self.with_breakable_ctx(BreakableKind::Border, None, None, |this| {
211            this.infer_return(body);
212        });
213
214        self.diverges = prev_diverges;
215        self.return_ty = prev_ret_ty;
216        self.return_coercion = prev_ret_coercion;
217        self.current_closure = prev_closure;
218        self.resume_yield_tys = prev_resume_yield_tys;
219
220        ty
221    }

223    fn fn_trait_kind_from_def_id(&self, trait_id: TraitId) -> Option<rustc_type_ir::ClosureKind> {
224        let lang_item = self.db.lang_attr(trait_id.into())?;
225        match lang_item {
226            LangItem::Fn => Some(rustc_type_ir::ClosureKind::Fn),
227            LangItem::FnMut => Some(rustc_type_ir::ClosureKind::FnMut),
228            LangItem::FnOnce => Some(rustc_type_ir::ClosureKind::FnOnce),
229            _ => None,
230        }
231    }

246    /// Given the expected type, figures out what it can about this closure we
247    /// are about to type check:
248    fn deduce_closure_signature(
249        &mut self,
250        expected_ty: Ty<'db>,
251        closure_kind: ClosureKind,
252    ) -> (Option<PolyFnSig<'db>>, Option<rustc_type_ir::ClosureKind>) {
253        match expected_ty.kind() {
254            TyKind::Alias(rustc_type_ir::Opaque, AliasTy { def_id, args, .. }) => self
255                .deduce_closure_signature_from_predicates(
256                    expected_ty,
257                    closure_kind,
258                    explicit_item_bounds(self.interner(), def_id)
259                        .iter_instantiated(self.interner(), args)
260                        .map(|clause| clause.as_predicate()),
261                ),
262            TyKind::Dynamic(object_type, ..) => {
263                let sig = object_type.projection_bounds().into_iter().find_map(|pb| {
264                    let pb = pb.with_self_ty(self.interner(), Ty::new_unit(self.interner()));
265                    self.deduce_sig_from_projection(closure_kind, pb)
266                });
267                let kind = object_type
268                    .principal_def_id()
269                    .and_then(|did| self.fn_trait_kind_from_def_id(did.0));
270                (sig, kind)
271            }
272            TyKind::Infer(rustc_type_ir::TyVar(vid)) => self
273                .deduce_closure_signature_from_predicates(
274                    Ty::new_var(self.interner(), self.table.infer_ctxt.root_var(vid)),
275                    closure_kind,
276                    self.table.obligations_for_self_ty(vid).into_iter().map(|obl| obl.predicate),
277                ),
278            TyKind::FnPtr(sig_tys, hdr) => match closure_kind {
279                ClosureKind::Closure => {
280                    let expected_sig = sig_tys.with(hdr);
281                    (Some(expected_sig), Some(rustc_type_ir::ClosureKind::Fn))
282                }
283                ClosureKind::Coroutine(_) | ClosureKind::Async => (None, None),
284            },
285            _ => (None, None),
286        }
287    }

289    fn deduce_closure_signature_from_predicates(
290        &mut self,
291        expected_ty: Ty<'db>,
292        closure_kind: ClosureKind,
293        predicates: impl DoubleEndedIterator<Item = Predicate<'db>>,
294    ) -> (Option<PolyFnSig<'db>>, Option<rustc_type_ir::ClosureKind>) {
295        let mut expected_sig = None;
296        let mut expected_kind = None;
297
298        for pred in rustc_type_ir::elaborate::elaborate(
299            self.interner(),
300            // Reverse the obligations here, since `elaborate_*` uses a stack,
301            // and we want to keep inference generally in the same order of
302            // the registered obligations.
303            predicates.rev(),
304        )
305        // We only care about self bounds
306        .filter_only_self()
307        {
308            debug!(?pred);
309            let bound_predicate = pred.kind();
310
311            // Given a Projection predicate, we can potentially infer
312            // the complete signature.
313            if expected_sig.is_none()
314                && let PredicateKind::Clause(ClauseKind::Projection(proj_predicate)) =
315                    bound_predicate.skip_binder()
316            {
317                let inferred_sig = self.deduce_sig_from_projection(
318                    closure_kind,
319                    bound_predicate.rebind(proj_predicate),
320                );
321
322                // Make sure that we didn't infer a signature that mentions itself.
323                // This can happen when we elaborate certain supertrait bounds that
324                // mention projections containing the `Self` type. See #105401.
325                struct MentionsTy<'db> {
326                    expected_ty: Ty<'db>,
327                }
328                impl<'db> TypeVisitor<DbInterner<'db>> for MentionsTy<'db> {
329                    type Result = ControlFlow<()>;
330
331                    fn visit_ty(&mut self, t: Ty<'db>) -> Self::Result {
332                        if t == self.expected_ty {
333                            ControlFlow::Break(())
334                        } else {
335                            t.super_visit_with(self)
336                        }
337                    }
338                }
339
340                // Don't infer a closure signature from a goal that names the closure type as this will
341                // (almost always) lead to occurs check errors later in type checking.
342                if let Some(inferred_sig) = inferred_sig {
343                    // In the new solver it is difficult to explicitly normalize the inferred signature as we
344                    // would have to manually handle universes and rewriting bound vars and placeholders back
345                    // and forth.
346                    //
347                    // Instead we take advantage of the fact that we relating an inference variable with an alias
348                    // will only instantiate the variable if the alias is rigid(*not quite). Concretely we:
349                    // - Create some new variable `?sig`
350                    // - Equate `?sig` with the unnormalized signature, e.g. `fn(<Foo<?x> as Trait>::Assoc)`
351                    // - Depending on whether `<Foo<?x> as Trait>::Assoc` is rigid, ambiguous or normalizeable,
352                    //   we will either wind up with `?sig=<Foo<?x> as Trait>::Assoc/?y/ConcreteTy` respectively.
353                    //
354                    // *: In cases where there are ambiguous aliases in the signature that make use of bound vars
355                    //    they will wind up present in `?sig` even though they are non-rigid.
356                    //
357                    //    This is a bit weird and means we may wind up discarding the goal due to it naming `expected_ty`
358                    //    even though the normalized form may not name `expected_ty`. However, this matches the existing
359                    //    behaviour of the old solver and would be technically a breaking change to fix.
360                    let generalized_fnptr_sig = self.table.next_ty_var();
361                    let inferred_fnptr_sig = Ty::new_fn_ptr(self.interner(), inferred_sig);
362                    // FIXME: Report diagnostics.
363                    _ = self
364                        .table
365                        .infer_ctxt
366                        .at(&ObligationCause::new(), self.table.trait_env.env)
367                        .eq(inferred_fnptr_sig, generalized_fnptr_sig)
368                        .map(|infer_ok| self.table.register_infer_ok(infer_ok));
369
370                    let resolved_sig =
371                        self.table.infer_ctxt.resolve_vars_if_possible(generalized_fnptr_sig);
372
373                    if resolved_sig.visit_with(&mut MentionsTy { expected_ty }).is_continue() {
374                        expected_sig = Some(resolved_sig.fn_sig(self.interner()));
375                    }
376                } else if inferred_sig.visit_with(&mut MentionsTy { expected_ty }).is_continue() {
377                    expected_sig = inferred_sig;
378                }
379            }
380
381            // Even if we can't infer the full signature, we may be able to
382            // infer the kind. This can occur when we elaborate a predicate
383            // like `F : Fn<A>`. Note that due to subtyping we could encounter
384            // many viable options, so pick the most restrictive.
385            let trait_def_id = match bound_predicate.skip_binder() {
386                PredicateKind::Clause(ClauseKind::Projection(data)) => {
387                    Some(data.projection_term.trait_def_id(self.interner()).0)
388                }
389                PredicateKind::Clause(ClauseKind::Trait(data)) => Some(data.def_id().0),
390                _ => None,
391            };
392
393            if let Some(trait_def_id) = trait_def_id {
394                let found_kind = match closure_kind {
395                    ClosureKind::Closure => self.fn_trait_kind_from_def_id(trait_def_id),
396                    ClosureKind::Async => self
397                        .async_fn_trait_kind_from_def_id(trait_def_id)
398                        .or_else(|| self.fn_trait_kind_from_def_id(trait_def_id)),
399                    _ => None,
400                };
401
402                if let Some(found_kind) = found_kind {
403                    // always use the closure kind that is more permissive.
404                    match (expected_kind, found_kind) {
405                        (None, _) => expected_kind = Some(found_kind),
406                        (
407                            Some(rustc_type_ir::ClosureKind::FnMut),
408                            rustc_type_ir::ClosureKind::Fn,
409                        ) => expected_kind = Some(rustc_type_ir::ClosureKind::Fn),
410                        (
411                            Some(rustc_type_ir::ClosureKind::FnOnce),
412                            rustc_type_ir::ClosureKind::Fn | rustc_type_ir::ClosureKind::FnMut,
413                        ) => expected_kind = Some(found_kind),
414                        _ => {}
415                    }
416                }
417            }
418        }
419
420        (expected_sig, expected_kind)
421    }

423    /// Given a projection like "<F as Fn(X)>::Result == Y", we can deduce
424    /// everything we need to know about a closure or coroutine.
425    ///
426    /// The `cause_span` should be the span that caused us to
427    /// have this expected signature, or `None` if we can't readily
428    /// know that.
429    fn deduce_sig_from_projection(
430        &mut self,
431        closure_kind: ClosureKind,
432        projection: PolyProjectionPredicate<'db>,
433    ) -> Option<PolyFnSig<'db>> {
434        let SolverDefId::TypeAliasId(def_id) = projection.item_def_id() else { unreachable!() };
435        let lang_item = self.db.lang_attr(def_id.into());
436
437        // For now, we only do signature deduction based off of the `Fn` and `AsyncFn` traits,
438        // for closures and async closures, respectively.
439        match closure_kind {
440            ClosureKind::Closure if lang_item == Some(LangItem::FnOnceOutput) => {
441                self.extract_sig_from_projection(projection)
442            }
443            ClosureKind::Async if lang_item == Some(LangItem::AsyncFnOnceOutput) => {
444                self.extract_sig_from_projection(projection)
445            }
446            // It's possible we've passed the closure to a (somewhat out-of-fashion)
447            // `F: FnOnce() -> Fut, Fut: Future<Output = T>` style bound. Let's still
448            // guide inference here, since it's beneficial for the user.
449            ClosureKind::Async if lang_item == Some(LangItem::FnOnceOutput) => {
450                self.extract_sig_from_projection_and_future_bound(projection)
451            }
452            _ => None,
453        }
454    }

456    /// Given an `FnOnce::Output` or `AsyncFn::Output` projection, extract the args
457    /// and return type to infer a [`ty::PolyFnSig`] for the closure.
458    fn extract_sig_from_projection(
459        &self,
460        projection: PolyProjectionPredicate<'db>,
461    ) -> Option<PolyFnSig<'db>> {
462        let projection = self.table.infer_ctxt.resolve_vars_if_possible(projection);
463
464        let arg_param_ty = projection.skip_binder().projection_term.args.type_at(1);
465        debug!(?arg_param_ty);
466
467        let TyKind::Tuple(input_tys) = arg_param_ty.kind() else {
468            return None;
469        };
470
471        // Since this is a return parameter type it is safe to unwrap.
472        let ret_param_ty = projection.skip_binder().term.expect_type();
473        debug!(?ret_param_ty);
474
475        let sig = projection.rebind(self.interner().mk_fn_sig(
476            input_tys,
477            ret_param_ty,
478            false,
479            Safety::Safe,
480            FnAbi::Rust,
481        ));
482
483        Some(sig)
484    }

486    /// When an async closure is passed to a function that has a "two-part" `Fn`
487    /// and `Future` trait bound, like:
488    ///
489    /// ```rust
490    /// use std::future::Future;
491    ///
492    /// fn not_exactly_an_async_closure<F, Fut>(_f: F)
493    /// where
494    ///     F: FnOnce(String, u32) -> Fut,
495    ///     Fut: Future<Output = i32>,
496    /// {}
497    /// ```
498    ///
499    /// The we want to be able to extract the signature to guide inference in the async
500    /// closure. We will have two projection predicates registered in this case. First,
501    /// we identify the `FnOnce<Args, Output = ?Fut>` bound, and if the output type is
502    /// an inference variable `?Fut`, we check if that is bounded by a `Future<Output = Ty>`
503    /// projection.
504    ///
505    /// This function is actually best-effort with the return type; if we don't find a
506    /// `Future` projection, we still will return arguments that we extracted from the `FnOnce`
507    /// projection, and the output will be an unconstrained type variable instead.
508    fn extract_sig_from_projection_and_future_bound(
509        &mut self,
510        projection: PolyProjectionPredicate<'db>,
511    ) -> Option<PolyFnSig<'db>> {
512        let projection = self.table.infer_ctxt.resolve_vars_if_possible(projection);
513
514        let arg_param_ty = projection.skip_binder().projection_term.args.type_at(1);
515        debug!(?arg_param_ty);
516
517        let TyKind::Tuple(input_tys) = arg_param_ty.kind() else {
518            return None;
519        };
520
521        // If the return type is a type variable, look for bounds on it.
522        // We could theoretically support other kinds of return types here,
523        // but none of them would be useful, since async closures return
524        // concrete anonymous future types, and their futures are not coerced
525        // into any other type within the body of the async closure.
526        let TyKind::Infer(rustc_type_ir::TyVar(return_vid)) =
527            projection.skip_binder().term.expect_type().kind()
528        else {
529            return None;
530        };
531
532        // FIXME: We may want to elaborate here, though I assume this will be exceedingly rare.
533        let mut return_ty = None;
534        for bound in self.table.obligations_for_self_ty(return_vid) {
535            if let PredicateKind::Clause(ClauseKind::Projection(ret_projection)) =
536                bound.predicate.kind().skip_binder()
537                && let ret_projection = bound.predicate.kind().rebind(ret_projection)
538                && let Some(ret_projection) = ret_projection.no_bound_vars()
539                && let SolverDefId::TypeAliasId(assoc_type) = ret_projection.def_id()
540                && self.db.lang_attr(assoc_type.into()) == Some(LangItem::FutureOutput)
541            {
542                return_ty = Some(ret_projection.term.expect_type());
543                break;
544            }
545        }
546
547        // SUBTLE: If we didn't find a `Future<Output = ...>` bound for the return
548        // vid, we still want to attempt to provide inference guidance for the async
549        // closure's arguments. Instantiate a new vid to plug into the output type.
550        //
551        // You may be wondering, what if it's higher-ranked? Well, given that we
552        // found a type variable for the `FnOnce::Output` projection above, we know
553        // that the output can't mention any of the vars.
554        //
555        // Also note that we use a fresh var here for the signature since the signature
556        // records the output of the *future*, and `return_vid` above is the type
557        // variable of the future, not its output.
558        //
559        // FIXME: We probably should store this signature inference output in a way
560        // that does not misuse a `FnSig` type, but that can be done separately.
561        let return_ty = return_ty.unwrap_or_else(|| self.table.next_ty_var());
562
563        let sig = projection.rebind(self.interner().mk_fn_sig(
564            input_tys,
565            return_ty,
566            false,
567            Safety::Safe,
568            FnAbi::Rust,
569        ));
570
571        Some(sig)
572    }

574    fn sig_of_closure(
575        &mut self,
576        decl_inputs: &[Option<TypeRefId>],
577        decl_output: Option<TypeRefId>,
578        expected_sig: Option<PolyFnSig<'db>>,
579    ) -> ClosureSignatures<'db> {
580        if let Some(e) = expected_sig {
581            self.sig_of_closure_with_expectation(decl_inputs, decl_output, e)
582        } else {
583            self.sig_of_closure_no_expectation(decl_inputs, decl_output)
584        }
585    }

587    /// If there is no expected signature, then we will convert the
588    /// types that the user gave into a signature.
589    fn sig_of_closure_no_expectation(
590        &mut self,
591        decl_inputs: &[Option<TypeRefId>],
592        decl_output: Option<TypeRefId>,
593    ) -> ClosureSignatures<'db> {
594        let bound_sig = self.supplied_sig_of_closure(decl_inputs, decl_output);
595
596        self.closure_sigs(bound_sig)
597    }

599    /// Invoked to compute the signature of a closure expression. This
600    /// combines any user-provided type annotations (e.g., `|x: u32|
601    /// -> u32 { .. }`) with the expected signature.
602    ///
603    /// The approach is as follows:
604    ///
605    /// - Let `S` be the (higher-ranked) signature that we derive from the user's annotations.
606    /// - Let `E` be the (higher-ranked) signature that we derive from the expectations, if any.
607    ///   - If we have no expectation `E`, then the signature of the closure is `S`.
608    ///   - Otherwise, the signature of the closure is E. Moreover:
609    ///     - Skolemize the late-bound regions in `E`, yielding `E'`.
610    ///     - Instantiate all the late-bound regions bound in the closure within `S`
611    ///       with fresh (existential) variables, yielding `S'`
612    ///     - Require that `E' = S'`
613    ///       - We could use some kind of subtyping relationship here,
614    ///         I imagine, but equality is easier and works fine for
615    ///         our purposes.
616    ///
617    /// The key intuition here is that the user's types must be valid
618    /// from "the inside" of the closure, but the expectation
619    /// ultimately drives the overall signature.
620    ///
621    /// # Examples
622    ///
623    /// ```ignore (illustrative)
624    /// fn with_closure<F>(_: F)
625    ///   where F: Fn(&u32) -> &u32 { .. }
626    ///
627    /// with_closure(|x: &u32| { ... })
628    /// ```
629    ///
630    /// Here:
631    /// - E would be `fn(&u32) -> &u32`.
632    /// - S would be `fn(&u32) -> ?T`
633    /// - E' is `&'!0 u32 -> &'!0 u32`
634    /// - S' is `&'?0 u32 -> ?T`
635    ///
636    /// S' can be unified with E' with `['?0 = '!0, ?T = &'!10 u32]`.
637    ///
638    /// # Arguments
639    ///
640    /// - `expr_def_id`: the `LocalDefId` of the closure expression
641    /// - `decl`: the HIR declaration of the closure
642    /// - `body`: the body of the closure
643    /// - `expected_sig`: the expected signature (if any). Note that
644    ///   this is missing a binder: that is, there may be late-bound
645    ///   regions with depth 1, which are bound then by the closure.
646    fn sig_of_closure_with_expectation(
647        &mut self,
648        decl_inputs: &[Option<TypeRefId>],
649        decl_output: Option<TypeRefId>,
650        expected_sig: PolyFnSig<'db>,
651    ) -> ClosureSignatures<'db> {
652        // Watch out for some surprises and just ignore the
653        // expectation if things don't see to match up with what we
654        // expect.
655        if expected_sig.c_variadic() {
656            return self.sig_of_closure_no_expectation(decl_inputs, decl_output);
657        } else if expected_sig.skip_binder().inputs_and_output.len() != decl_inputs.len() + 1 {
658            return self
659                .sig_of_closure_with_mismatched_number_of_arguments(decl_inputs, decl_output);
660        }
661
662        // Create a `PolyFnSig`. Note the oddity that late bound
663        // regions appearing free in `expected_sig` are now bound up
664        // in this binder we are creating.
665        assert!(!expected_sig.skip_binder().has_vars_bound_above(rustc_type_ir::INNERMOST));
666        let bound_sig = expected_sig.map_bound(|sig| {
667            self.interner().mk_fn_sig(
668                sig.inputs(),
669                sig.output(),
670                sig.c_variadic,
671                Safety::Safe,
672                FnAbi::RustCall,
673            )
674        });
675
676        // `deduce_expectations_from_expected_type` introduces
677        // late-bound lifetimes defined elsewhere, which we now
678        // anonymize away, so as not to confuse the user.
679        let bound_sig = self.interner().anonymize_bound_vars(bound_sig);
680
681        let closure_sigs = self.closure_sigs(bound_sig);
682
683        // Up till this point, we have ignored the annotations that the user
684        // gave. This function will check that they unify successfully.
685        // Along the way, it also writes out entries for types that the user
686        // wrote into our typeck results, which are then later used by the privacy
687        // check.
688        match self.merge_supplied_sig_with_expectation(decl_inputs, decl_output, closure_sigs) {
689            Ok(infer_ok) => self.table.register_infer_ok(infer_ok),
690            Err(_) => self.sig_of_closure_no_expectation(decl_inputs, decl_output),
691        }
692    }

694    fn sig_of_closure_with_mismatched_number_of_arguments(
695        &mut self,
696        decl_inputs: &[Option<TypeRefId>],
697        decl_output: Option<TypeRefId>,
698    ) -> ClosureSignatures<'db> {
699        let error_sig = self.error_sig_of_closure(decl_inputs, decl_output);
700
701        self.closure_sigs(error_sig)
702    }

704    /// Enforce the user's types against the expectation. See
705    /// `sig_of_closure_with_expectation` for details on the overall
706    /// strategy.
707    fn merge_supplied_sig_with_expectation(
708        &mut self,
709        decl_inputs: &[Option<TypeRefId>],
710        decl_output: Option<TypeRefId>,
711        mut expected_sigs: ClosureSignatures<'db>,
712    ) -> InferResult<'db, ClosureSignatures<'db>> {
713        // Get the signature S that the user gave.
714        //
715        // (See comment on `sig_of_closure_with_expectation` for the
716        // meaning of these letters.)
717        let supplied_sig = self.supplied_sig_of_closure(decl_inputs, decl_output);
718
719        debug!(?supplied_sig);
720
721        // FIXME(#45727): As discussed in [this comment][c1], naively
722        // forcing equality here actually results in suboptimal error
723        // messages in some cases. For now, if there would have been
724        // an obvious error, we fallback to declaring the type of the
725        // closure to be the one the user gave, which allows other
726        // error message code to trigger.
727        //
728        // However, I think [there is potential to do even better
729        // here][c2], since in *this* code we have the precise span of
730        // the type parameter in question in hand when we report the
731        // error.
732        //
733        // [c1]: https://github.com/rust-lang/rust/pull/45072#issuecomment-341089706
734        // [c2]: https://github.com/rust-lang/rust/pull/45072#issuecomment-341096796
735        self.table.commit_if_ok(|table| {
736            let mut all_obligations = PredicateObligations::new();
737            let supplied_sig = table.infer_ctxt.instantiate_binder_with_fresh_vars(
738                BoundRegionConversionTime::FnCall,
739                supplied_sig,
740            );
741
742            // The liberated version of this signature should be a subtype
743            // of the liberated form of the expectation.
744            for (supplied_ty, expected_ty) in
745                iter::zip(supplied_sig.inputs(), expected_sigs.liberated_sig.inputs())
746            {
747                // Check that E' = S'.
748                let cause = ObligationCause::new();
749                let InferOk { value: (), obligations } = table
750                    .infer_ctxt
751                    .at(&cause, table.trait_env.env)
752                    .eq(expected_ty, supplied_ty)?;
753                all_obligations.extend(obligations);
754            }
755
756            let supplied_output_ty = supplied_sig.output();
757            let cause = ObligationCause::new();
758            let InferOk { value: (), obligations } = table
759                .infer_ctxt
760                .at(&cause, table.trait_env.env)
761                .eq(expected_sigs.liberated_sig.output(), supplied_output_ty)?;
762            all_obligations.extend(obligations);
763
764            let inputs = supplied_sig
765                .inputs()
766                .into_iter()
767                .map(|ty| table.infer_ctxt.resolve_vars_if_possible(ty));
768
769            expected_sigs.liberated_sig = table.interner().mk_fn_sig(
770                inputs,
771                supplied_output_ty,
772                expected_sigs.liberated_sig.c_variadic,
773                Safety::Safe,
774                FnAbi::RustCall,
775            );
776
777            Ok(InferOk { value: expected_sigs, obligations: all_obligations })
778        })
779    }

781    /// If there is no expected signature, then we will convert the
782    /// types that the user gave into a signature.
783    ///
784    /// Also, record this closure signature for later.
785    fn supplied_sig_of_closure(
786        &mut self,
787        decl_inputs: &[Option<TypeRefId>],
788        decl_output: Option<TypeRefId>,
789    ) -> PolyFnSig<'db> {
790        let interner = self.interner();
791
792        let supplied_return = match decl_output {
793            Some(output) => {
794                let output = self.make_body_ty(output);
795                self.process_user_written_ty(output)
796            }
797            None => self.table.next_ty_var(),
798        };
799        // First, convert the types that the user supplied (if any).
800        let supplied_arguments = decl_inputs.iter().map(|&input| match input {
801            Some(input) => {
802                let input = self.make_body_ty(input);
803                self.process_user_written_ty(input)
804            }
805            None => self.table.next_ty_var(),
806        });
807
808        Binder::dummy(interner.mk_fn_sig(
809            supplied_arguments,
810            supplied_return,
811            false,
812            Safety::Safe,
813            FnAbi::RustCall,
814        ))
815    }

817    /// Converts the types that the user supplied, in case that doing
818    /// so should yield an error, but returns back a signature where
819    /// all parameters are of type `ty::Error`.
820    fn error_sig_of_closure(
821        &mut self,
822        decl_inputs: &[Option<TypeRefId>],
823        decl_output: Option<TypeRefId>,
824    ) -> PolyFnSig<'db> {
825        let interner = self.interner();
826        let err_ty = Ty::new_error(interner, ErrorGuaranteed);
827
828        if let Some(output) = decl_output {
829            self.make_body_ty(output);
830        }
831        let supplied_arguments = decl_inputs.iter().map(|&input| match input {
832            Some(input) => {
833                self.make_body_ty(input);
834                err_ty
835            }
836            None => err_ty,
837        });
838
839        let result = Binder::dummy(interner.mk_fn_sig(
840            supplied_arguments,
841            err_ty,
842            false,
843            Safety::Safe,
844            FnAbi::RustCall,
845        ));
846
847        debug!("supplied_sig_of_closure: result={:?}", result);
848
849        result
850    }

852    fn closure_sigs(&self, bound_sig: PolyFnSig<'db>) -> ClosureSignatures<'db> {
853        let liberated_sig = bound_sig.skip_binder();
854        // FIXME: When we lower HRTB we'll need to actually liberate regions here.
855        ClosureSignatures { bound_sig, liberated_sig }
856    }
