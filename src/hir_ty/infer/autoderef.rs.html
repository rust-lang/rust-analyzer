17impl<'db> InferenceTable<'db> {
18    pub(crate) fn autoderef(&self, base_ty: Ty<'db>) -> Autoderef<'_, 'db, usize> {
19        Autoderef::new(&self.infer_ctxt, self.param_env, base_ty)
20    }
21
22    pub(crate) fn autoderef_with_tracking(&self, base_ty: Ty<'db>) -> Autoderef<'_, 'db> {
23        Autoderef::new_with_tracking(&self.infer_ctxt, self.param_env, base_ty)
24    }
25}

27impl<'db, Ctx: AutoderefCtx<'db>> GeneralAutoderef<'db, Ctx> {
28    pub(crate) fn adjust_steps_as_infer_ok(&mut self) -> InferOk<'db, Vec<Adjustment>> {
29        let steps = self.steps();
30        if steps.is_empty() {
31            return InferOk { obligations: PredicateObligations::new(), value: vec![] };
32        }
33
34        let targets = steps.iter().skip(1).map(|&(ty, _)| ty).chain(iter::once(self.final_ty()));
35        let steps: Vec<_> = steps
36            .iter()
37            .map(|&(_source, kind)| {
38                if let AutoderefKind::Overloaded = kind {
39                    Some(OverloadedDeref(Some(Mutability::Not)))
40                } else {
41                    None
42                }
43            })
44            .zip(targets)
45            .map(|(autoderef, target)| Adjustment {
46                kind: Adjust::Deref(autoderef),
47                target: target.store(),
48            })
49            .collect();
50
51        InferOk { obligations: self.take_obligations(), value: steps }
52    }
53}
