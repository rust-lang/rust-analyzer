1//! Autoderef helpers for inference.
2
3use std::iter;
4
5use rustc_ast_ir::Mutability;
6
7use crate::{
8    Adjust, Adjustment, OverloadedDeref,
9    autoderef::{Autoderef, AutoderefKind},
10    infer::unify::InferenceTable,
11    next_solver::{
12        Ty,
13        infer::{InferOk, traits::PredicateObligations},
14    },
15};
16
17impl<'db> InferenceTable<'db> {
18    pub(crate) fn autoderef(&mut self, base_ty: Ty<'db>) -> Autoderef<'_, 'db> {
19        Autoderef::new(self, base_ty)
20    }
21}
22
23impl<'db> Autoderef<'_, 'db> {
24    /// Returns the adjustment steps.
25    pub(crate) fn adjust_steps(mut self) -> Vec<Adjustment<'db>> {
26        let infer_ok = self.adjust_steps_as_infer_ok();
27        self.table.register_infer_ok(infer_ok)
28    }
29
30    pub(crate) fn adjust_steps_as_infer_ok(&mut self) -> InferOk<'db, Vec<Adjustment<'db>>> {
31        let steps = self.steps();
32        if steps.is_empty() {
33            return InferOk { obligations: PredicateObligations::new(), value: vec![] };
34        }
35
36        let targets = steps.iter().skip(1).map(|&(ty, _)| ty).chain(iter::once(self.final_ty()));
37        let steps: Vec<_> = steps
38            .iter()
39            .map(|&(_source, kind)| {
40                if let AutoderefKind::Overloaded = kind {
41                    Some(OverloadedDeref(Some(Mutability::Not)))
42                } else {
43                    None
44                }
45            })
46            .zip(targets)
47            .map(|(autoderef, target)| Adjustment { kind: Adjust::Deref(autoderef), target })
48            .collect();
49
50        InferOk { obligations: self.take_obligations(), value: steps }
51    }
52}
