1//! Finds if an expression is an immutable context or a mutable context, which is used in selecting
2//! between `Deref` and `DerefMut` or `Index` and `IndexMut` or similar.
3
4use hir_def::{
5    hir::{
6        Array, AsmOperand, BinaryOp, BindingAnnotation, Expr, ExprId, Pat, PatId, Statement,
7        UnaryOp,
8    },
9    lang_item::LangItem,
10};
11use hir_expand::name::Name;
12use intern::sym;
13use rustc_ast_ir::Mutability;
14use rustc_type_ir::inherent::IntoKind;
15
16use crate::next_solver::infer::traits::{Obligation, ObligationCause};
17use crate::next_solver::{GenericArgs, TraitRef};
18use crate::{

20    infer::{Expectation, InferenceContext, expr::ExprIsRead},

23};

25impl<'db> InferenceContext<'_, 'db> {
26    pub(crate) fn infer_mut_body(&mut self) {
27        self.infer_mut_expr(self.body.body_expr, Mutability::Not);
28    }
29
30    fn infer_mut_expr(&mut self, tgt_expr: ExprId, mut mutability: Mutability) {
31        if let Some(adjustments) = self.result.expr_adjustments.get_mut(&tgt_expr) {
32            for adj in adjustments.iter_mut().rev() {
33                match &mut adj.kind {
34                    Adjust::NeverToAny | Adjust::Deref(None) | Adjust::Pointer(_) => (),
35                    Adjust::Deref(Some(d)) => *d = OverloadedDeref(Some(mutability)),
36                    Adjust::Borrow(b) => match b {
37                        AutoBorrow::Ref(_, m) | AutoBorrow::RawPtr(m) => mutability = *m,
38                    },
39                }
40            }
41        }
42        self.infer_mut_expr_without_adjust(tgt_expr, mutability);
43    }
44
45    fn infer_mut_expr_without_adjust(&mut self, tgt_expr: ExprId, mutability: Mutability) {
46        match &self.body[tgt_expr] {
47            Expr::Missing => (),
48            Expr::InlineAsm(e) => {
49                e.operands.iter().for_each(|(_, op)| match op {
50                    AsmOperand::In { expr, .. }
51                    | AsmOperand::Out { expr: Some(expr), .. }
52                    | AsmOperand::InOut { expr, .. } => {
53                        self.infer_mut_expr_without_adjust(*expr, Mutability::Not)
54                    }
55                    AsmOperand::SplitInOut { in_expr, out_expr, .. } => {
56                        self.infer_mut_expr_without_adjust(*in_expr, Mutability::Not);
57                        if let Some(out_expr) = out_expr {
58                            self.infer_mut_expr_without_adjust(*out_expr, Mutability::Not);
59                        }
60                    }
61                    AsmOperand::Out { expr: None, .. }
62                    | AsmOperand::Label(_)
63                    | AsmOperand::Sym(_)
64                    | AsmOperand::Const(_) => (),
65                });
66            }
67            Expr::OffsetOf(_) => (),
68            &Expr::If { condition, then_branch, else_branch } => {
69                self.infer_mut_expr(condition, Mutability::Not);
70                self.infer_mut_expr(then_branch, Mutability::Not);
71                if let Some(else_branch) = else_branch {
72                    self.infer_mut_expr(else_branch, Mutability::Not);
73                }
74            }
75            Expr::Const(id) => {
76                self.infer_mut_expr(*id, Mutability::Not);
77            }
78            Expr::Let { pat, expr } => self.infer_mut_expr(*expr, self.pat_bound_mutability(*pat)),
79            Expr::Block { id: _, statements, tail, label: _ }
80            | Expr::Async { id: _, statements, tail }
81            | Expr::Unsafe { id: _, statements, tail } => {
82                for st in statements.iter() {
83                    match st {
84                        Statement::Let { pat, type_ref: _, initializer, else_branch } => {
85                            if let Some(i) = initializer {
86                                self.infer_mut_expr(*i, self.pat_bound_mutability(*pat));
87                            }
88                            if let Some(e) = else_branch {
89                                self.infer_mut_expr(*e, Mutability::Not);
90                            }
91                        }
92                        Statement::Expr { expr, has_semi: _ } => {
93                            self.infer_mut_expr(*expr, Mutability::Not);
94                        }
95                        Statement::Item(_) => (),
96                    }
97                }
98                if let Some(tail) = tail {
99                    self.infer_mut_expr(*tail, Mutability::Not);
100                }
101            }
102            Expr::MethodCall { receiver: it, method_name: _, args, generic_args: _ }
103            | Expr::Call { callee: it, args } => {
104                self.infer_mut_not_expr_iter(args.iter().copied().chain(Some(*it)));
105            }
106            Expr::Match { expr, arms } => {
107                let m = self.pat_iter_bound_mutability(arms.iter().map(|it| it.pat));
108                self.infer_mut_expr(*expr, m);
109                for arm in arms.iter() {
110                    self.infer_mut_expr(arm.expr, Mutability::Not);
111                    if let Some(g) = arm.guard {
112                        self.infer_mut_expr(g, Mutability::Not);
113                    }
114                }
115            }
116            Expr::Yield { expr }
117            | Expr::Yeet { expr }
118            | Expr::Return { expr }
119            | Expr::Break { expr, label: _ } => {
120                if let &Some(expr) = expr {
121                    self.infer_mut_expr(expr, Mutability::Not);
122                }
123            }
124            Expr::Become { expr } => {
125                self.infer_mut_expr(*expr, Mutability::Not);
126            }
127            Expr::RecordLit { path: _, fields, spread } => {
128                self.infer_mut_not_expr_iter(fields.iter().map(|it| it.expr).chain(*spread))
129            }
130            &Expr::Index { base, index } => {
131                if mutability == Mutability::Mut
132                    && let Some((f, _)) = self.result.method_resolutions.get_mut(&tgt_expr)
133                    && let Some(index_trait) =
134                        LangItem::IndexMut.resolve_trait(self.db, self.table.trait_env.krate)
135                    && let Some(index_fn) = index_trait
136                        .trait_items(self.db)
137                        .method_by_name(&Name::new_symbol_root(sym::index_mut))
138                {
139                    *f = index_fn;
140                    let mut base_ty = None;
141                    let base_adjustments =
142                        self.result.expr_adjustments.get_mut(&base).and_then(|it| it.last_mut());
143                    if let Some(Adjustment {
144                        kind: Adjust::Borrow(AutoBorrow::Ref(_, mutability)),
145                        target,
146                    }) = base_adjustments
147                    {
148                        if let TyKind::Ref(_, ty, _) = target.kind() {
149                            base_ty = Some(ty);
150                        }
151                        *mutability = Mutability::Mut;
152                    }
153
154                    // Apply `IndexMut` obligation for non-assignee expr
155                    if let Some(base_ty) = base_ty {
156                        let index_ty = if let Some(ty) = self.result.type_of_expr.get(index) {
157                            *ty
158                        } else {
159                            self.infer_expr(index, &Expectation::none(), ExprIsRead::Yes)
160                        };
161                        let trait_ref = TraitRef::new(
162                            self.interner(),
163                            index_trait.into(),
164                            GenericArgs::new_from_iter(
165                                self.interner(),
166                                [base_ty.into(), index_ty.into()],
167                            ),
168                        );
169                        self.table.register_predicate(Obligation::new(
170                            self.interner(),
171                            ObligationCause::new(),
172                            self.table.trait_env.env,
173                            trait_ref,
174                        ));
175                    }
176                }
177                self.infer_mut_expr(base, mutability);
178                self.infer_mut_expr(index, Mutability::Not);
179            }
180            Expr::UnaryOp { expr, op: UnaryOp::Deref } => {
181                let mut mutability = mutability;
182                if let Some((f, _)) = self.result.method_resolutions.get_mut(&tgt_expr)
183                    && mutability == Mutability::Mut
184                    && let Some(deref_trait) =
185                        LangItem::DerefMut.resolve_trait(self.db, self.table.trait_env.krate)
186                {
187                    let ty = self.result.type_of_expr.get(*expr);
188                    let is_mut_ptr = ty.is_some_and(|ty| {
189                        let ty = self.table.shallow_resolve(*ty);
190                        matches!(ty.kind(), TyKind::RawPtr(_, Mutability::Mut))
191                    });
192                    if is_mut_ptr {
193                        mutability = Mutability::Not;
194                    } else if let Some(deref_fn) = deref_trait
195                        .trait_items(self.db)
196                        .method_by_name(&Name::new_symbol_root(sym::deref_mut))
197                    {
198                        *f = deref_fn;
199                    }
200                }
201                self.infer_mut_expr(*expr, mutability);
202            }
203            Expr::Field { expr, name: _ } => {
204                self.infer_mut_expr(*expr, mutability);
205            }
206            Expr::UnaryOp { expr, op: _ }
207            | Expr::Range { lhs: Some(expr), rhs: None, range_type: _ }
208            | Expr::Range { rhs: Some(expr), lhs: None, range_type: _ }
209            | Expr::Await { expr }
210            | Expr::Box { expr }
211            | Expr::Loop { body: expr, label: _ }
212            | Expr::Cast { expr, type_ref: _ } => {
213                self.infer_mut_expr(*expr, Mutability::Not);
214            }
215            Expr::Ref { expr, rawness: _, mutability } => {
216                let mutability = lower_mutability(*mutability);
217                self.infer_mut_expr(*expr, mutability);
218            }
219            Expr::BinaryOp { lhs, rhs, op: Some(BinaryOp::Assignment { .. }) } => {
220                self.infer_mut_expr(*lhs, Mutability::Mut);
221                self.infer_mut_expr(*rhs, Mutability::Not);
222            }
223            &Expr::Assignment { target, value } => {
224                self.body.walk_pats(target, &mut |pat| match self.body[pat] {
225                    Pat::Expr(expr) => self.infer_mut_expr(expr, Mutability::Mut),
226                    Pat::ConstBlock(block) => self.infer_mut_expr(block, Mutability::Not),
227                    _ => {}
228                });
229                self.infer_mut_expr(value, Mutability::Not);
230            }
231            Expr::Array(Array::Repeat { initializer: lhs, repeat: rhs })
232            | Expr::BinaryOp { lhs, rhs, op: _ }
233            | Expr::Range { lhs: Some(lhs), rhs: Some(rhs), range_type: _ } => {
234                self.infer_mut_expr(*lhs, Mutability::Not);
235                self.infer_mut_expr(*rhs, Mutability::Not);
236            }
237            Expr::Closure { body, .. } => {
238                self.infer_mut_expr(*body, Mutability::Not);
239            }
240            Expr::Tuple { exprs } | Expr::Array(Array::ElementList { elements: exprs }) => {
241                self.infer_mut_not_expr_iter(exprs.iter().copied());
242            }
243            // These don't need any action, as they don't have sub expressions
244            Expr::Range { lhs: None, rhs: None, range_type: _ }
245            | Expr::Literal(_)
246            | Expr::Path(_)
247            | Expr::Continue { .. }
248            | Expr::Underscore => (),
249        }
250    }
251
252    fn infer_mut_not_expr_iter(&mut self, exprs: impl Iterator<Item = ExprId>) {
253        for expr in exprs {
254            self.infer_mut_expr(expr, Mutability::Not);
255        }
256    }
257
258    fn pat_iter_bound_mutability(&self, mut pat: impl Iterator<Item = PatId>) -> Mutability {
259        if pat.any(|p| self.pat_bound_mutability(p) == Mutability::Mut) {
260            Mutability::Mut
261        } else {
262            Mutability::Not
263        }
264    }
265
266    /// Checks if the pat contains a `ref mut` binding. Such paths makes the context of bounded expressions
267    /// mutable. For example in `let (ref mut x0, ref x1) = *it;` we need to use `DerefMut` for `*it` but in
268    /// `let (ref x0, ref x1) = *it;` we should use `Deref`.
269    fn pat_bound_mutability(&self, pat: PatId) -> Mutability {
270        let mut r = Mutability::Not;
271        self.body.walk_bindings_in_pat(pat, |b| {
272            if self.body[b].mode == BindingAnnotation::RefMut {
273                r = Mutability::Mut;
274            }
275        });
276        r
277    }
