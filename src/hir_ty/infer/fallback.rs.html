20#[derive(Copy, Clone)]
21pub(crate) enum DivergingFallbackBehavior {
22    /// Always fallback to `()` (aka "always spontaneous decay")
23    ToUnit,
24    /// Sometimes fallback to `!`, but mainly fallback to `()` so that most of the crates are not broken.
25    ContextDependent,
26    /// Always fallback to `!` (which should be equivalent to never falling back + not making
27    /// never-to-any coercions unless necessary)
28    ToNever,
29}

31impl<'db> InferenceContext<'_, 'db> {
32    pub(super) fn type_inference_fallback(&mut self) {
33        debug!(
34            "type-inference-fallback start obligations: {:#?}",
35            self.table.fulfillment_cx.pending_obligations()
36        );
37
38        // All type checking constraints were added, try to fallback unsolved variables.
39        self.table.select_obligations_where_possible();
40
41        debug!(
42            "type-inference-fallback post selection obligations: {:#?}",
43            self.table.fulfillment_cx.pending_obligations()
44        );
45
46        let fallback_occurred = self.fallback_types();
47
48        if !fallback_occurred {
49            return;
50        }
51
52        // We now see if we can make progress. This might cause us to
53        // unify inference variables for opaque types, since we may
54        // have unified some other type variables during the first
55        // phase of fallback. This means that we only replace
56        // inference variables with their underlying opaque types as a
57        // last resort.
58        //
59        // In code like this:
60        //
61        // ```rust
62        // type MyType = impl Copy;
63        // fn produce() -> MyType { true }
64        // fn bad_produce() -> MyType { panic!() }
65        // ```
66        //
67        // we want to unify the opaque inference variable in `bad_produce`
68        // with the diverging fallback for `panic!` (e.g. `()` or `!`).
69        // This will produce a nice error message about conflicting concrete
70        // types for `MyType`.
71        //
72        // If we had tried to fallback the opaque inference variable to `MyType`,
73        // we will generate a confusing type-check error that does not explicitly
74        // refer to opaque types.
75        self.table.select_obligations_where_possible();
76    }
77
78    fn diverging_fallback_behavior(&self) -> DivergingFallbackBehavior {
79        if self.krate().data(self.db).edition.at_least_2024() {
80            return DivergingFallbackBehavior::ToNever;
81        }
82
83        if self.resolver.def_map().is_unstable_feature_enabled(&sym::never_type_fallback) {
84            return DivergingFallbackBehavior::ContextDependent;
85        }
86
87        DivergingFallbackBehavior::ToUnit
88    }

90    fn fallback_types(&mut self) -> bool {
91        // Check if we have any unresolved variables. If not, no need for fallback.
92        let unresolved_variables = self.table.infer_ctxt.unresolved_variables();
93
94        if unresolved_variables.is_empty() {
95            return false;
96        }
97
98        let diverging_fallback_behavior = self.diverging_fallback_behavior();
99
100        let diverging_fallback =
101            self.calculate_diverging_fallback(&unresolved_variables, diverging_fallback_behavior);
102
103        // We do fallback in two passes, to try to generate
104        // better error messages.
105        // The first time, we do *not* replace opaque types.
106        let mut fallback_occurred = false;
107        for ty in unresolved_variables {
108            debug!("unsolved_variable = {:?}", ty);
109            fallback_occurred |= self.fallback_if_possible(ty, &diverging_fallback);
110        }
111
112        fallback_occurred
113    }

115    // Tries to apply a fallback to `ty` if it is an unsolved variable.
116    //
117    // - Unconstrained ints are replaced with `i32`.
118    //
119    // - Unconstrained floats are replaced with `f64`.
120    //
121    // - Non-numerics may get replaced with `()` or `!`, depending on
122    //   how they were categorized by `calculate_diverging_fallback`
123    //   (and the setting of `#![feature(never_type_fallback)]`).
124    //
125    // Fallback becomes very dubious if we have encountered
126    // type-checking errors. In that case, fallback to Error.
127    //
128    // Sets `FnCtxt::fallback_has_occurred` if fallback is performed
129    // during this call.
130    fn fallback_if_possible(
131        &mut self,
132        ty: Ty<'db>,
133        diverging_fallback: &FxHashMap<Ty<'db>, Ty<'db>>,
134    ) -> bool {
135        // Careful: we do NOT shallow-resolve `ty`. We know that `ty`
136        // is an unsolved variable, and we determine its fallback
137        // based solely on how it was created, not what other type
138        // variables it may have been unified with since then.
139        //
140        // The reason this matters is that other attempts at fallback
141        // may (in principle) conflict with this fallback, and we wish
142        // to generate a type error in that case. (However, this
143        // actually isn't true right now, because we're only using the
144        // builtin fallback rules. This would be true if we were using
145        // user-supplied fallbacks. But it's still useful to write the
146        // code to detect bugs.)
147        //
148        // (Note though that if we have a general type variable `?T`
149        // that is then unified with an integer type variable `?I`
150        // that ultimately never gets resolved to a special integral
151        // type, `?T` is not considered unsolved, but `?I` is. The
152        // same is true for float variables.)
153        let fallback = match ty.kind() {
154            TyKind::Infer(rustc_type_ir::IntVar(_)) => self.types.types.i32,
155            TyKind::Infer(rustc_type_ir::FloatVar(_)) => self.types.types.f64,
156            _ => match diverging_fallback.get(&ty) {
157                Some(&fallback_ty) => fallback_ty,
158                None => return false,
159            },
160        };
161        debug!("fallback_if_possible(ty={:?}): defaulting to `{:?}`", ty, fallback);
162
163        _ = self.demand_eqtype_fixme_no_diag(ty, fallback);
164        true
165    }

167    /// The "diverging fallback" system is rather complicated. This is
168    /// a result of our need to balance 'do the right thing' with
169    /// backwards compatibility.
170    ///
171    /// "Diverging" type variables are variables created when we
172    /// coerce a `!` type into an unbound type variable `?X`. If they
173    /// never wind up being constrained, the "right and natural" thing
174    /// is that `?X` should "fallback" to `!`. This means that e.g. an
175    /// expression like `Some(return)` will ultimately wind up with a
176    /// type like `Option<!>` (presuming it is not assigned or
177    /// constrained to have some other type).
178    ///
179    /// However, the fallback used to be `()` (before the `!` type was
180    /// added). Moreover, there are cases where the `!` type 'leaks
181    /// out' from dead code into type variables that affect live
182    /// code. The most common case is something like this:
183    ///
184    /// ```rust
185    /// # fn foo() -> i32 { 4 }
186    /// match foo() {
187    ///     22 => Default::default(), // call this type `?D`
188    ///     _ => return, // return has type `!`
189    /// } // call the type of this match `?M`
190    /// ```
191    ///
192    /// Here, coercing the type `!` into `?M` will create a diverging
193    /// type variable `?X` where `?X <: ?M`. We also have that `?D <:
194    /// ?M`. If `?M` winds up unconstrained, then `?X` will
195    /// fallback. If it falls back to `!`, then all the type variables
196    /// will wind up equal to `!` -- this includes the type `?D`
197    /// (since `!` doesn't implement `Default`, we wind up a "trait
198    /// not implemented" error in code like this). But since the
199    /// original fallback was `()`, this code used to compile with `?D
200    /// = ()`. This is somewhat surprising, since `Default::default()`
201    /// on its own would give an error because the types are
202    /// insufficiently constrained.
203    ///
204    /// Our solution to this dilemma is to modify diverging variables
205    /// so that they can *either* fallback to `!` (the default) or to
206    /// `()` (the backwards compatibility case). We decide which
207    /// fallback to use based on whether there is a coercion pattern
208    /// like this:
209    ///
210    /// ```ignore (not-rust)
211    /// ?Diverging -> ?V
212    /// ?NonDiverging -> ?V
213    /// ?V != ?NonDiverging
214    /// ```
215    ///
216    /// Here `?Diverging` represents some diverging type variable and
217    /// `?NonDiverging` represents some non-diverging type
218    /// variable. `?V` can be any type variable (diverging or not), so
219    /// long as it is not equal to `?NonDiverging`.
220    ///
221    /// Intuitively, what we are looking for is a case where a
222    /// "non-diverging" type variable (like `?M` in our example above)
223    /// is coerced *into* some variable `?V` that would otherwise
224    /// fallback to `!`. In that case, we make `?V` fallback to `!`,
225    /// along with anything that would flow into `?V`.
226    ///
227    /// The algorithm we use:
228    /// * Identify all variables that are coerced *into* by a
229    ///   diverging variable. Do this by iterating over each
230    ///   diverging, unsolved variable and finding all variables
231    ///   reachable from there. Call that set `D`.
232    /// * Walk over all unsolved, non-diverging variables, and find
233    ///   any variable that has an edge into `D`.
234    fn calculate_diverging_fallback(
235        &self,
236        unresolved_variables: &[Ty<'db>],
237        behavior: DivergingFallbackBehavior,
238    ) -> FxHashMap<Ty<'db>, Ty<'db>> {
239        debug!("calculate_diverging_fallback({:?})", unresolved_variables);
240
241        // Construct a coercion graph where an edge `A -> B` indicates
242        // a type variable is that is coerced
243        let coercion_graph = self.create_coercion_graph();
244
245        // Extract the unsolved type inference variable vids; note that some
246        // unsolved variables are integer/float variables and are excluded.
247        let unsolved_vids = unresolved_variables.iter().filter_map(|ty| ty.ty_vid());
248
249        // Compute the diverging root vids D -- that is, the root vid of
250        // those type variables that (a) are the target of a coercion from
251        // a `!` type and (b) have not yet been solved.
252        //
253        // These variables are the ones that are targets for fallback to
254        // either `!` or `()`.
255        let diverging_roots: FxHashSet<TyVid> = self
256            .table
257            .diverging_type_vars
258            .iter()
259            .map(|&ty| self.shallow_resolve(ty))
260            .filter_map(|ty| ty.ty_vid())
261            .map(|vid| self.table.infer_ctxt.root_var(vid))
262            .collect();
263        debug!(
264            "calculate_diverging_fallback: diverging_type_vars={:?}",
265            self.table.diverging_type_vars
266        );
267        debug!("calculate_diverging_fallback: diverging_roots={:?}", diverging_roots);
268
269        // Find all type variables that are reachable from a diverging
270        // type variable. These will typically default to `!`, unless
271        // we find later that they are *also* reachable from some
272        // other type variable outside this set.
273        let mut roots_reachable_from_diverging = Dfs::empty(&coercion_graph);
274        let mut diverging_vids = vec![];
275        let mut non_diverging_vids = vec![];
276        for unsolved_vid in unsolved_vids {
277            let root_vid = self.table.infer_ctxt.root_var(unsolved_vid);
278            debug!(
279                "calculate_diverging_fallback: unsolved_vid={:?} root_vid={:?} diverges={:?}",
280                unsolved_vid,
281                root_vid,
282                diverging_roots.contains(&root_vid),
283            );
284            if diverging_roots.contains(&root_vid) {
285                diverging_vids.push(unsolved_vid);
286                roots_reachable_from_diverging.move_to(root_vid.as_u32().into());
287
288                // drain the iterator to visit all nodes reachable from this node
289                while roots_reachable_from_diverging.next(&coercion_graph).is_some() {}
290            } else {
291                non_diverging_vids.push(unsolved_vid);
292            }
293        }
294
295        debug!(
296            "calculate_diverging_fallback: roots_reachable_from_diverging={:?}",
297            roots_reachable_from_diverging,
298        );
299
300        // Find all type variables N0 that are not reachable from a
301        // diverging variable, and then compute the set reachable from
302        // N0, which we call N. These are the *non-diverging* type
303        // variables. (Note that this set consists of "root variables".)
304        let mut roots_reachable_from_non_diverging = Dfs::empty(&coercion_graph);
305        for &non_diverging_vid in &non_diverging_vids {
306            let root_vid = self.table.infer_ctxt.root_var(non_diverging_vid);
307            if roots_reachable_from_diverging.discovered.contains(root_vid.as_usize()) {
308                continue;
309            }
310            roots_reachable_from_non_diverging.move_to(root_vid.as_u32().into());
311            while roots_reachable_from_non_diverging.next(&coercion_graph).is_some() {}
312        }
313        debug!(
314            "calculate_diverging_fallback: roots_reachable_from_non_diverging={:?}",
315            roots_reachable_from_non_diverging,
316        );
317
318        debug!("obligations: {:#?}", self.table.fulfillment_cx.pending_obligations());
319
320        // For each diverging variable, figure out whether it can
321        // reach a member of N. If so, it falls back to `()`. Else
322        // `!`.
323        let mut diverging_fallback =
324            FxHashMap::with_capacity_and_hasher(diverging_vids.len(), FxBuildHasher);
325
326        for &diverging_vid in &diverging_vids {
327            let diverging_ty = Ty::new_var(self.interner(), diverging_vid);
328            let root_vid = self.table.infer_ctxt.root_var(diverging_vid);
329            let can_reach_non_diverging = Dfs::new(&coercion_graph, root_vid.as_u32().into())
330                .iter(&coercion_graph)
331                .any(|n| roots_reachable_from_non_diverging.discovered.contains(n.index()));
332
333            let mut fallback_to = |ty| {
334                diverging_fallback.insert(diverging_ty, ty);
335            };
336
337            match behavior {
338                DivergingFallbackBehavior::ToUnit => {
339                    debug!("fallback to () - legacy: {:?}", diverging_vid);
340                    fallback_to(self.types.types.unit);
341                }
342                DivergingFallbackBehavior::ContextDependent => {
343                    // FIXME: rustc does the following, but given this is only relevant when the unstable
344                    // `never_type_fallback` feature is active, I chose to not port this.
345                    // if found_infer_var_info.self_in_trait && found_infer_var_info.output {
346                    //     // This case falls back to () to ensure that the code pattern in
347                    //     // tests/ui/never_type/fallback-closure-ret.rs continues to
348                    //     // compile when never_type_fallback is enabled.
349                    //     //
350                    //     // This rule is not readily explainable from first principles,
351                    //     // but is rather intended as a patchwork fix to ensure code
352                    //     // which compiles before the stabilization of never type
353                    //     // fallback continues to work.
354                    //     //
355                    //     // Typically this pattern is encountered in a function taking a
356                    //     // closure as a parameter, where the return type of that closure
357                    //     // (checked by `relationship.output`) is expected to implement
358                    //     // some trait (checked by `relationship.self_in_trait`). This
359                    //     // can come up in non-closure cases too, so we do not limit this
360                    //     // rule to specifically `FnOnce`.
361                    //     //
362                    //     // When the closure's body is something like `panic!()`, the
363                    //     // return type would normally be inferred to `!`. However, it
364                    //     // needs to fall back to `()` in order to still compile, as the
365                    //     // trait is specifically implemented for `()` but not `!`.
366                    //     //
367                    //     // For details on the requirements for these relationships to be
368                    //     // set, see the relationship finding module in
369                    //     // compiler/rustc_trait_selection/src/traits/relationships.rs.
370                    //     debug!("fallback to () - found trait and projection: {:?}", diverging_vid);
371                    //     fallback_to(self.types.types.unit);
372                    // }
373                    if can_reach_non_diverging {
374                        debug!("fallback to () - reached non-diverging: {:?}", diverging_vid);
375                        fallback_to(self.types.types.unit);
376                    } else {
377                        debug!("fallback to ! - all diverging: {:?}", diverging_vid);
378                        fallback_to(self.types.types.never);
379                    }
380                }
381                DivergingFallbackBehavior::ToNever => {
382                    debug!(
383                        "fallback to ! - `rustc_never_type_mode = \"fallback_to_never\")`: {:?}",
384                        diverging_vid
385                    );
386                    fallback_to(self.types.types.never);
387                }
388            }
389        }
390
391        diverging_fallback
392    }

394    /// Returns a graph whose nodes are (unresolved) inference variables and where
395    /// an edge `?A -> ?B` indicates that the variable `?A` is coerced to `?B`.
396    fn create_coercion_graph(&self) -> Graph<(), ()> {
397        let pending_obligations = self.table.fulfillment_cx.pending_obligations();
398        let pending_obligations_len = pending_obligations.len();
399        debug!("create_coercion_graph: pending_obligations={:?}", pending_obligations);
400        let coercion_edges = pending_obligations
401            .into_iter()
402            .filter_map(|obligation| {
403                // The predicates we are looking for look like `Coerce(?A -> ?B)`.
404                // They will have no bound variables.
405                obligation.predicate.kind().no_bound_vars()
406            })
407            .filter_map(|atom| {
408                // We consider both subtyping and coercion to imply 'flow' from
409                // some position in the code `a` to a different position `b`.
410                // This is then used to determine which variables interact with
411                // live code, and as such must fall back to `()` to preserve
412                // soundness.
413                //
414                // In practice currently the two ways that this happens is
415                // coercion and subtyping.
416                let (a, b) = match atom {
417                    PredicateKind::Coerce(CoercePredicate { a, b }) => (a, b),
418                    PredicateKind::Subtype(SubtypePredicate { a_is_expected: _, a, b }) => (a, b),
419                    _ => return None,
420                };
421
422                let a_vid = self.root_vid(a)?;
423                let b_vid = self.root_vid(b)?;
424                Some((a_vid.as_u32(), b_vid.as_u32()))
425            });
426        let num_ty_vars = self.table.infer_ctxt.num_ty_vars();
427        let mut graph = Graph::with_capacity(num_ty_vars, pending_obligations_len);
428        for _ in 0..num_ty_vars {
429            graph.add_node(());
430        }
431        graph.extend_with_edges(coercion_edges);
432        graph
433    }

435    /// If `ty` is an unresolved type variable, returns its root vid.
436    fn root_vid(&self, ty: Ty<'db>) -> Option<TyVid> {
437        Some(self.table.infer_ctxt.root_var(self.shallow_resolve(ty).ty_vid()?))
438    }
