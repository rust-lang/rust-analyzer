27impl<'db> InferenceContext<'_, 'db> {
28    pub(super) fn infer_path(&mut self, path: &Path, id: ExprOrPatId) -> Option<Ty<'db>> {
29        let (value_def, generic_def, substs) = match self.resolve_value_path(path, id)? {
30            ValuePathResolution::GenericDef(value_def, generic_def, substs) => {
31                (value_def, generic_def, substs)
32            }
33            ValuePathResolution::NonGeneric(ty) => return Some(ty),
34        };
35        let args = self.insert_type_vars(substs);
36
37        self.add_required_obligations_for_value_path(generic_def, args);
38
39        let ty = self.db.value_ty(value_def)?.instantiate(self.interner(), args);
40        let ty = self.process_remote_user_written_ty(ty);
41        Some(ty)
42    }

44    fn resolve_value_path(
45        &mut self,
46        path: &Path,
47        id: ExprOrPatId,
48    ) -> Option<ValuePathResolution<'db>> {
49        let (value, self_subst) = self.resolve_value_path_inner(path, id, false)?;
50
51        let value_def: ValueTyDefId = match value {
52            ValueNs::FunctionId(it) => it.into(),
53            ValueNs::ConstId(it) => it.into(),
54            ValueNs::StaticId(it) => it.into(),
55            ValueNs::StructId(it) => {
56                self.write_variant_resolution(id, it.into());
57
58                it.into()
59            }
60            ValueNs::EnumVariantId(it) => {
61                self.write_variant_resolution(id, it.into());
62
63                it.into()
64            }
65            ValueNs::LocalBinding(pat) => {
66                return match self.result.type_of_binding.get(pat) {
67                    Some(ty) => Some(ValuePathResolution::NonGeneric(ty.as_ref())),
68                    None => {
69                        never!("uninferred pattern?");
70                        None
71                    }
72                };
73            }
74            ValueNs::ImplSelf(impl_id) => {
75                let ty = self.db.impl_self_ty(impl_id).instantiate_identity();
76                return if let Some((AdtId::StructId(struct_id), substs)) = ty.as_adt() {
77                    Some(ValuePathResolution::GenericDef(
78                        struct_id.into(),
79                        struct_id.into(),
80                        substs,
81                    ))
82                } else {
83                    // FIXME: report error, invalid Self reference
84                    None
85                };
86            }
87            ValueNs::GenericParam(it) => {
88                return Some(ValuePathResolution::NonGeneric(self.db.const_param_ty_ns(it)));
89            }
90        };
91
92        let generic_def = value_def.to_generic_def_id(self.db);
93        if let GenericDefId::StaticId(_) = generic_def {
94            // `Static` is the kind of item that can never be generic currently. We can just skip the binders to get its type.
95            let ty = self.db.value_ty(value_def)?.skip_binder();
96            let ty = self.process_remote_user_written_ty(ty);
97            return Some(ValuePathResolution::NonGeneric(ty));
98        };
99
100        let substs = if self_subst.is_some_and(|it| !it.is_empty())
101            && matches!(value_def, ValueTyDefId::EnumVariantId(_))
102        {
103            // This is something like `TypeAlias::<Args>::EnumVariant`. Do not call `substs_from_path()`,
104            // as it'll try to re-lower the previous segment assuming it refers to the enum, but it refers
105            // to the type alias and they may have different generics.
106            self.types.empty.generic_args
107        } else {
108            self.with_body_ty_lowering(|ctx| {
109                let mut path_ctx = ctx.at_path(path, id);
110                let last_segment = path.segments().len().checked_sub(1);
111                if let Some(last_segment) = last_segment {
112                    path_ctx.set_current_segment(last_segment)
113                }
114                path_ctx.substs_from_path(value_def, true, false)
115            })
116        };
117
118        let parent_substs_len = self_subst.map_or(0, |it| it.len());
119        let substs = GenericArgs::fill_rest(
120            self.interner(),
121            generic_def.into(),
122            self_subst.iter().flat_map(|it| it.iter()).chain(substs.iter().skip(parent_substs_len)),
123            |_, id, _| GenericArg::error_from_id(self.interner(), id),
124        );
125
126        Some(ValuePathResolution::GenericDef(value_def, generic_def, substs))
127    }

129    pub(super) fn resolve_value_path_inner(
130        &mut self,
131        path: &Path,
132        id: ExprOrPatId,
133        no_diagnostics: bool,
134    ) -> Option<(ValueNs, Option<GenericArgs<'db>>)> {
135        // Don't use `self.make_ty()` here as we need `orig_ns`.
136        let mut ctx = TyLoweringContext::new(
137            self.db,
138            &self.resolver,
139            self.body,
140            &self.diagnostics,
141            InferenceTyDiagnosticSource::Body,
142            self.generic_def,
143            LifetimeElisionKind::Infer,
144        );
145        let mut path_ctx = if no_diagnostics {
146            ctx.at_path_forget_diagnostics(path)
147        } else {
148            ctx.at_path(path, id)
149        };
150        let (value, self_subst) = if let Some(type_ref) = path.type_anchor() {
151            let last = path.segments().last()?;
152
153            let (ty, orig_ns) = path_ctx.ty_ctx().lower_ty_ext(type_ref);
154            let ty = self.table.process_user_written_ty(ty);
155
156            path_ctx.ignore_last_segment();
157            let (ty, _) = path_ctx.lower_ty_relative_path(ty, orig_ns, true);
158            drop_ctx(ctx, no_diagnostics);
159            let ty = self.table.process_user_written_ty(ty);
160            self.resolve_ty_assoc_item(ty, last.name, id).map(|(it, substs)| (it, Some(substs)))?
161        } else {
162            let hygiene = self.body.expr_or_pat_path_hygiene(id);
163            // FIXME: report error, unresolved first path segment
164            let value_or_partial = path_ctx.resolve_path_in_value_ns(hygiene)?;
165
166            match value_or_partial {
167                ResolveValueResult::ValueNs(it) => {
168                    drop_ctx(ctx, no_diagnostics);
169                    (it, None)
170                }
171                ResolveValueResult::Partial(def, remaining_index) => {
172                    // there may be more intermediate segments between the resolved one and
173                    // the end. Only the last segment needs to be resolved to a value; from
174                    // the segments before that, we need to get either a type or a trait ref.
175
176                    let remaining_segments = path.segments().skip(remaining_index);
177                    let is_before_last = remaining_segments.len() == 1;
178                    let last_segment = remaining_segments
179                        .last()
180                        .expect("there should be at least one segment here");
181
182                    let (resolution, substs) = match (def, is_before_last) {
183                        (TypeNs::TraitId(trait_), true) => {
184                            let self_ty = self.table.next_ty_var();
185                            let trait_ref =
186                                path_ctx.lower_trait_ref_from_resolved_path(trait_, self_ty, true);
187                            drop_ctx(ctx, no_diagnostics);
188                            self.resolve_trait_assoc_item(trait_ref, last_segment, id)
189                        }
190                        (def, _) => {
191                            // Either we already have a type (e.g. `Vec::new`), or we have a
192                            // trait but it's not the last segment, so the next segment
193                            // should resolve to an associated type of that trait (e.g. `<T
194                            // as Iterator>::Item::default`)
195                            path_ctx.ignore_last_segment();
196                            let (ty, _) = path_ctx.lower_partly_resolved_path(def, true);
197                            drop_ctx(ctx, no_diagnostics);
198                            if ty.is_ty_error() {
199                                return None;
200                            }
201
202                            let ty = self.process_user_written_ty(ty);
203
204                            self.resolve_ty_assoc_item(ty, last_segment.name, id)
205                        }
206                    }?;
207                    (resolution, Some(substs))
208                }
209            }
210        };
211        return Some((value, self_subst));
212
213        #[inline]
214        fn drop_ctx(mut ctx: TyLoweringContext<'_, '_>, no_diagnostics: bool) {
215            if no_diagnostics {
216                ctx.forget_diagnostics();
217            }
218        }
219    }

221    fn add_required_obligations_for_value_path(
222        &mut self,
223        def: GenericDefId,
224        subst: GenericArgs<'db>,
225    ) {
226        let interner = self.interner();
227        let predicates = GenericPredicates::query_all(self.db, def);
228        let param_env = self.table.param_env;
229        self.table.register_predicates(clauses_as_obligations(
230            predicates.iter_instantiated_copied(interner, subst.as_slice()),
231            ObligationCause::new(),
232            param_env,
233        ));
234
235        // We need to add `Self: Trait` obligation when `def` is a trait assoc item.
236        let container = match def {
237            GenericDefId::FunctionId(id) => id.lookup(self.db).container,
238            GenericDefId::ConstId(id) => id.lookup(self.db).container,
239            _ => return,
240        };
241
242        if let ItemContainerId::TraitId(trait_) = container {
243            let parent_len = generics(self.db, def).parent_generics().map_or(0, |g| g.len_self());
244            let parent_subst = GenericArgs::new_from_slice(&subst.as_slice()[..parent_len]);
245            let trait_ref = TraitRef::new_from_args(interner, trait_.into(), parent_subst);
246            self.table.register_predicate(Obligation::new(
247                interner,
248                ObligationCause::new(),
249                param_env,
250                trait_ref,
251            ));
252        }
253    }

255    fn resolve_trait_assoc_item(
256        &mut self,
257        trait_ref: TraitRef<'db>,
258        segment: PathSegment<'_>,
259        id: ExprOrPatId,
260    ) -> Option<(ValueNs, GenericArgs<'db>)> {
261        let trait_ = trait_ref.def_id.0;
262        let item =
263            trait_.trait_items(self.db).items.iter().map(|(_name, id)| *id).find_map(|item| {
264                match item {
265                    AssocItemId::FunctionId(func) => {
266                        if segment.name == &self.db.function_signature(func).name {
267                            Some(CandidateId::FunctionId(func))
268                        } else {
269                            None
270                        }
271                    }
272
273                    AssocItemId::ConstId(konst) => {
274                        if self.db.const_signature(konst).name.as_ref() == Some(segment.name) {
275                            Some(CandidateId::ConstId(konst))
276                        } else {
277                            None
278                        }
279                    }
280                    AssocItemId::TypeAliasId(_) => None,
281                }
282            })?;
283        let def = match item {
284            CandidateId::FunctionId(f) => ValueNs::FunctionId(f),
285            CandidateId::ConstId(c) => ValueNs::ConstId(c),
286        };
287
288        self.write_assoc_resolution(id, item, trait_ref.args);
289        Some((def, trait_ref.args))
290    }

292    fn resolve_ty_assoc_item(
293        &mut self,
294        ty: Ty<'db>,
295        name: &Name,
296        id: ExprOrPatId,
297    ) -> Option<(ValueNs, GenericArgs<'db>)> {
298        if ty.is_ty_error() {
299            return None;
300        }
301
302        if let Some(result) = self.resolve_enum_variant_on_ty(ty, name, id) {
303            return Some(result);
304        }
305
306        let res = self.with_method_resolution(|ctx| {
307            ctx.probe_for_name(method_resolution::Mode::Path, name.clone(), ty)
308        });
309        let (item, visible) = match res {
310            Ok(res) => (res.item, true),
311            Err(error) => match error {
312                MethodError::PrivateMatch(candidate_id) => (candidate_id.item, false),
313                _ => {
314                    self.push_diagnostic(InferenceDiagnostic::UnresolvedAssocItem { id });
315                    return None;
316                }
317            },
318        };
319
320        let (def, container) = match item {
321            CandidateId::FunctionId(f) => (ValueNs::FunctionId(f), f.lookup(self.db).container),
322            CandidateId::ConstId(c) => (ValueNs::ConstId(c), c.lookup(self.db).container),
323        };
324        let substs = match container {
325            ItemContainerId::ImplId(impl_id) => {
326                let impl_substs = self.table.fresh_args_for_item(impl_id.into());
327                let impl_self_ty =
328                    self.db.impl_self_ty(impl_id).instantiate(self.interner(), impl_substs);
329                self.unify(impl_self_ty, ty);
330                impl_substs
331            }
332            ItemContainerId::TraitId(trait_) => {
333                // we're picking this method
334                let args = GenericArgs::fill_rest(
335                    self.interner(),
336                    trait_.into(),
337                    [ty.into()],
338                    |_, id, _| self.table.next_var_for_param(id),
339                );
340                let trait_ref = TraitRef::new_from_args(self.interner(), trait_.into(), args);
341                self.table.register_predicate(Obligation::new(
342                    self.interner(),
343                    ObligationCause::new(),
344                    self.table.param_env,
345                    trait_ref,
346                ));
347                args
348            }
349            ItemContainerId::ModuleId(_) | ItemContainerId::ExternBlockId(_) => {
350                never!("assoc item contained in module/extern block");
351                return None;
352            }
353        };
354
355        self.write_assoc_resolution(id, item, substs);
356        if !visible {
357            let item = match item {
358                CandidateId::FunctionId(it) => it.into(),
359                CandidateId::ConstId(it) => it.into(),
360            };
361            self.push_diagnostic(InferenceDiagnostic::PrivateAssocItem { id, item });
362        }
363        Some((def, substs))
364    }

366    fn resolve_enum_variant_on_ty(
367        &mut self,
368        ty: Ty<'db>,
369        name: &Name,
370        id: ExprOrPatId,
371    ) -> Option<(ValueNs, GenericArgs<'db>)> {
372        let ty = self.table.try_structurally_resolve_type(ty);
373        let (enum_id, subst) = match ty.as_adt() {
374            Some((AdtId::EnumId(e), subst)) => (e, subst),
375            _ => return None,
376        };
377        let enum_data = enum_id.enum_variants(self.db);
378        let variant = enum_data.variant(name)?;
379        self.write_variant_resolution(id, variant.into());
380        Some((ValueNs::EnumVariantId(variant), subst))
381    }

384#[derive(Debug)]
385enum ValuePathResolution<'db> {
386    // It's awkward to wrap a single ID in two enums, but we need both and this saves fallible
387    // conversion between them + `unwrap()`.
388    GenericDef(ValueTyDefId, GenericDefId, GenericArgs<'db>),
389    NonGeneric(Ty<'db>),
