26impl<'db> InferenceContext<'_, 'db> {
27    pub(super) fn infer_path(&mut self, path: &Path, id: ExprOrPatId) -> Option<Ty<'db>> {
28        let (value_def, generic_def, substs) = match self.resolve_value_path(path, id)? {
29            ValuePathResolution::GenericDef(value_def, generic_def, substs) => {
30                (value_def, generic_def, substs)
31            }
32            ValuePathResolution::NonGeneric(ty) => return Some(ty),
33        };
34        let args = self.process_remote_user_written_ty(substs);
35
36        self.add_required_obligations_for_value_path(generic_def, args);
37
38        let ty = self.db.value_ty(value_def)?.instantiate(self.interner(), args);
39        let ty = self.process_remote_user_written_ty(ty);
40        Some(ty)
41    }

43    fn resolve_value_path(
44        &mut self,
45        path: &Path,
46        id: ExprOrPatId,
47    ) -> Option<ValuePathResolution<'db>> {
48        let (value, self_subst) = self.resolve_value_path_inner(path, id, false)?;
49
50        let value_def: ValueTyDefId = match value {
51            ValueNs::FunctionId(it) => it.into(),
52            ValueNs::ConstId(it) => it.into(),
53            ValueNs::StaticId(it) => it.into(),
54            ValueNs::StructId(it) => {
55                self.write_variant_resolution(id, it.into());
56
57                it.into()
58            }
59            ValueNs::EnumVariantId(it) => {
60                self.write_variant_resolution(id, it.into());
61
62                it.into()
63            }
64            ValueNs::LocalBinding(pat) => {
65                return match self.result.type_of_binding.get(pat) {
66                    Some(ty) => Some(ValuePathResolution::NonGeneric(*ty)),
67                    None => {
68                        never!("uninferred pattern?");
69                        None
70                    }
71                };
72            }
73            ValueNs::ImplSelf(impl_id) => {
74                let ty = self.db.impl_self_ty(impl_id).instantiate_identity();
75                return if let Some((AdtId::StructId(struct_id), substs)) = ty.as_adt() {
76                    Some(ValuePathResolution::GenericDef(
77                        struct_id.into(),
78                        struct_id.into(),
79                        substs,
80                    ))
81                } else {
82                    // FIXME: report error, invalid Self reference
83                    None
84                };
85            }
86            ValueNs::GenericParam(it) => {
87                return Some(ValuePathResolution::NonGeneric(self.db.const_param_ty_ns(it)));
88            }
89        };
90
91        let generic_def = value_def.to_generic_def_id(self.db);
92        if let GenericDefId::StaticId(_) = generic_def {
93            // `Static` is the kind of item that can never be generic currently. We can just skip the binders to get its type.
94            let ty = self.db.value_ty(value_def)?.skip_binder();
95            return Some(ValuePathResolution::NonGeneric(ty));
96        };
97
98        let substs = if self_subst.is_some_and(|it| !it.is_empty())
99            && matches!(value_def, ValueTyDefId::EnumVariantId(_))
100        {
101            // This is something like `TypeAlias::<Args>::EnumVariant`. Do not call `substs_from_path()`,
102            // as it'll try to re-lower the previous segment assuming it refers to the enum, but it refers
103            // to the type alias and they may have different generics.
104            self.types.empty_args
105        } else {
106            self.with_body_ty_lowering(|ctx| {
107                let mut path_ctx = ctx.at_path(path, id);
108                let last_segment = path.segments().len().checked_sub(1);
109                if let Some(last_segment) = last_segment {
110                    path_ctx.set_current_segment(last_segment)
111                }
112                path_ctx.substs_from_path(value_def, true, false)
113            })
114        };
115
116        let parent_substs_len = self_subst.map_or(0, |it| it.len());
117        let substs = GenericArgs::fill_rest(
118            self.interner(),
119            generic_def.into(),
120            self_subst.iter().flat_map(|it| it.iter()).chain(substs.iter().skip(parent_substs_len)),
121            |_, id, _| GenericArg::error_from_id(self.interner(), id),
122        );
123
124        Some(ValuePathResolution::GenericDef(value_def, generic_def, substs))
125    }

127    pub(super) fn resolve_value_path_inner(
128        &mut self,
129        path: &Path,
130        id: ExprOrPatId,
131        no_diagnostics: bool,
132    ) -> Option<(ValueNs, Option<GenericArgs<'db>>)> {
133        // Don't use `self.make_ty()` here as we need `orig_ns`.
134        let mut ctx = TyLoweringContext::new(
135            self.db,
136            &self.resolver,
137            self.body,
138            &self.diagnostics,
139            InferenceTyDiagnosticSource::Body,
140            self.generic_def,
141            LifetimeElisionKind::Infer,
142        );
143        let mut path_ctx = if no_diagnostics {
144            ctx.at_path_forget_diagnostics(path)
145        } else {
146            ctx.at_path(path, id)
147        };
148        let (value, self_subst) = if let Some(type_ref) = path.type_anchor() {
149            let last = path.segments().last()?;
150
151            let (ty, orig_ns) = path_ctx.ty_ctx().lower_ty_ext(type_ref);
152            let ty = self.table.process_user_written_ty(ty);
153
154            path_ctx.ignore_last_segment();
155            let (ty, _) = path_ctx.lower_ty_relative_path(ty, orig_ns, true);
156            drop_ctx(ctx, no_diagnostics);
157            let ty = self.table.process_user_written_ty(ty);
158            self.resolve_ty_assoc_item(ty, last.name, id).map(|(it, substs)| (it, Some(substs)))?
159        } else {
160            let hygiene = self.body.expr_or_pat_path_hygiene(id);
161            // FIXME: report error, unresolved first path segment
162            let value_or_partial = path_ctx.resolve_path_in_value_ns(hygiene)?;
163
164            match value_or_partial {
165                ResolveValueResult::ValueNs(it, _) => {
166                    drop_ctx(ctx, no_diagnostics);
167                    (it, None)
168                }
169                ResolveValueResult::Partial(def, remaining_index, _) => {
170                    // there may be more intermediate segments between the resolved one and
171                    // the end. Only the last segment needs to be resolved to a value; from
172                    // the segments before that, we need to get either a type or a trait ref.
173
174                    let remaining_segments = path.segments().skip(remaining_index);
175                    let is_before_last = remaining_segments.len() == 1;
176                    let last_segment = remaining_segments
177                        .last()
178                        .expect("there should be at least one segment here");
179
180                    let (resolution, substs) = match (def, is_before_last) {
181                        (TypeNs::TraitId(trait_), true) => {
182                            let self_ty = self.table.next_ty_var();
183                            let trait_ref =
184                                path_ctx.lower_trait_ref_from_resolved_path(trait_, self_ty, true);
185                            drop_ctx(ctx, no_diagnostics);
186                            self.resolve_trait_assoc_item(trait_ref, last_segment, id)
187                        }
188                        (def, _) => {
189                            // Either we already have a type (e.g. `Vec::new`), or we have a
190                            // trait but it's not the last segment, so the next segment
191                            // should resolve to an associated type of that trait (e.g. `<T
192                            // as Iterator>::Item::default`)
193                            path_ctx.ignore_last_segment();
194                            let (ty, _) = path_ctx.lower_partly_resolved_path(def, true);
195                            drop_ctx(ctx, no_diagnostics);
196                            if ty.is_ty_error() {
197                                return None;
198                            }
199
200                            let ty = self.process_user_written_ty(ty);
201
202                            self.resolve_ty_assoc_item(ty, last_segment.name, id)
203                        }
204                    }?;
205                    (resolution, Some(substs))
206                }
207            }
208        };
209        return Some((value, self_subst));
210
211        #[inline]
212        fn drop_ctx(mut ctx: TyLoweringContext<'_, '_>, no_diagnostics: bool) {
213            if no_diagnostics {
214                ctx.forget_diagnostics();
215            }
216        }
217    }

219    fn add_required_obligations_for_value_path(
220        &mut self,
221        def: GenericDefId,
222        subst: GenericArgs<'db>,
223    ) {
224        let predicates = self.db.generic_predicates(def);
225        let interner = self.interner();
226        let param_env = self.table.trait_env.env;
227        if let Some(predicates) = predicates.instantiate(self.interner(), subst) {
228            self.table.register_predicates(predicates.map(|predicate| {
229                Obligation::new(interner, ObligationCause::new(), param_env, predicate)
230            }));
231        }
232
233        // We need to add `Self: Trait` obligation when `def` is a trait assoc item.
234        let container = match def {
235            GenericDefId::FunctionId(id) => id.lookup(self.db).container,
236            GenericDefId::ConstId(id) => id.lookup(self.db).container,
237            _ => return,
238        };
239
240        if let ItemContainerId::TraitId(trait_) = container {
241            let parent_len = generics(self.db, def).parent_generics().map_or(0, |g| g.len_self());
242            let parent_subst = GenericArgs::new_from_iter(
243                interner,
244                subst.as_slice()[..parent_len].iter().copied(),
245            );
246            let trait_ref = TraitRef::new(interner, trait_.into(), parent_subst);
247            self.table.register_predicate(Obligation::new(
248                interner,
249                ObligationCause::new(),
250                param_env,
251                trait_ref,
252            ));
253        }
254    }

256    fn resolve_trait_assoc_item(
257        &mut self,
258        trait_ref: TraitRef<'db>,
259        segment: PathSegment<'_>,
260        id: ExprOrPatId,
261    ) -> Option<(ValueNs, GenericArgs<'db>)> {
262        let trait_ = trait_ref.def_id.0;
263        let item =
264            trait_.trait_items(self.db).items.iter().map(|(_name, id)| *id).find_map(|item| {
265                match item {
266                    AssocItemId::FunctionId(func) => {
267                        if segment.name == &self.db.function_signature(func).name {
268                            Some(AssocItemId::FunctionId(func))
269                        } else {
270                            None
271                        }
272                    }
273
274                    AssocItemId::ConstId(konst) => {
275                        if self.db.const_signature(konst).name.as_ref() == Some(segment.name) {
276                            Some(AssocItemId::ConstId(konst))
277                        } else {
278                            None
279                        }
280                    }
281                    AssocItemId::TypeAliasId(_) => None,
282                }
283            })?;
284        let def = match item {
285            AssocItemId::FunctionId(f) => ValueNs::FunctionId(f),
286            AssocItemId::ConstId(c) => ValueNs::ConstId(c),
287            AssocItemId::TypeAliasId(_) => unreachable!(),
288        };
289
290        self.write_assoc_resolution(id, item, trait_ref.args);
291        Some((def, trait_ref.args))
292    }

294    fn resolve_ty_assoc_item(
295        &mut self,
296        ty: Ty<'db>,
297        name: &Name,
298        id: ExprOrPatId,
299    ) -> Option<(ValueNs, GenericArgs<'db>)> {
300        if ty.is_ty_error() {
301            return None;
302        }
303
304        if let Some(result) = self.resolve_enum_variant_on_ty(ty, name, id) {
305            return Some(result);
306        }
307
308        let canonical_ty = self.canonicalize(ty);
309
310        let mut not_visible = None;
311        let res = method_resolution::iterate_method_candidates(
312            &canonical_ty,
313            &mut self.table,
314            Self::get_traits_in_scope(&self.resolver, &self.traits_in_scope)
315                .as_ref()
316                .left_or_else(|&it| it),
317            VisibleFromModule::Filter(self.resolver.module()),
318            Some(name),
319            method_resolution::LookupMode::Path,
320            |_ty, item, visible| {
321                if visible {
322                    Some((item, true))
323                } else {
324                    if not_visible.is_none() {
325                        not_visible = Some((item, false));
326                    }
327                    None
328                }
329            },
330        );
331        let res = res.or(not_visible);
332        if res.is_none() {
333            self.push_diagnostic(InferenceDiagnostic::UnresolvedAssocItem { id });
334        }
335        let (item, visible) = res?;
336
337        let (def, container) = match item {
338            AssocItemId::FunctionId(f) => (ValueNs::FunctionId(f), f.lookup(self.db).container),
339            AssocItemId::ConstId(c) => (ValueNs::ConstId(c), c.lookup(self.db).container),
340            AssocItemId::TypeAliasId(_) => unreachable!(),
341        };
342        let substs = match container {
343            ItemContainerId::ImplId(impl_id) => {
344                let impl_substs = self.table.fresh_args_for_item(impl_id.into());
345                let impl_self_ty =
346                    self.db.impl_self_ty(impl_id).instantiate(self.interner(), impl_substs);
347                self.unify(impl_self_ty, ty);
348                impl_substs
349            }
350            ItemContainerId::TraitId(trait_) => {
351                // we're picking this method
352                let args = GenericArgs::fill_rest(
353                    self.interner(),
354                    trait_.into(),
355                    [ty.into()],
356                    |_, id, _| self.table.next_var_for_param(id),
357                );
358                let trait_ref = TraitRef::new(self.interner(), trait_.into(), args);
359                self.table.register_predicate(Obligation::new(
360                    self.interner(),
361                    ObligationCause::new(),
362                    self.table.trait_env.env,
363                    trait_ref,
364                ));
365                args
366            }
367            ItemContainerId::ModuleId(_) | ItemContainerId::ExternBlockId(_) => {
368                never!("assoc item contained in module/extern block");
369                return None;
370            }
371        };
372
373        self.write_assoc_resolution(id, item, substs);
374        if !visible {
375            self.push_diagnostic(InferenceDiagnostic::PrivateAssocItem { id, item });
376        }
377        Some((def, substs))
378    }

380    fn resolve_enum_variant_on_ty(
381        &mut self,
382        ty: Ty<'db>,
383        name: &Name,
384        id: ExprOrPatId,
385    ) -> Option<(ValueNs, GenericArgs<'db>)> {
386        let ty = self.table.try_structurally_resolve_type(ty);
387        let (enum_id, subst) = match ty.as_adt() {
388            Some((AdtId::EnumId(e), subst)) => (e, subst),
389            _ => return None,
390        };
391        let enum_data = enum_id.enum_variants(self.db);
392        let variant = enum_data.variant(name)?;
393        self.write_variant_resolution(id, variant.into());
394        Some((ValueNs::EnumVariantId(variant), subst))
395    }

399enum ValuePathResolution<'db> {
400    // It's awkward to wrap a single ID in two enums, but we need both and this saves fallible
401    // conversion between them + `unwrap()`.
402    GenericDef(ValueTyDefId, GenericDefId, GenericArgs<'db>),
403    NonGeneric(Ty<'db>),
