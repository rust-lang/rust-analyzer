25    /// Checks a `a <op>= b`
26    pub(crate) fn infer_assign_op_expr(
27        &mut self,
28        expr: ExprId,
29        op: ArithOp,
30        lhs: ExprId,
31        rhs: ExprId,
32    ) -> Ty<'db> {
33        let (lhs_ty, rhs_ty, return_ty) =
34            self.infer_overloaded_binop(expr, lhs, rhs, BinaryOp::Assignment { op: Some(op) });
35
36        let category = BinOpCategory::from(op);
37        let ty = if !lhs_ty.is_ty_var()
38            && !rhs_ty.is_ty_var()
39            && is_builtin_binop(lhs_ty, rhs_ty, category)
40        {
41            self.enforce_builtin_binop_types(lhs_ty, rhs_ty, category);
42            self.types.types.unit
43        } else {
44            return_ty
45        };
46
47        self.check_lhs_assignable(lhs);
48
49        ty
50    }

52    /// Checks a potentially overloaded binary operator.
53    pub(crate) fn infer_binop_expr(
54        &mut self,
55        expr: ExprId,
56        op: BinaryOp,
57        lhs_expr: ExprId,
58        rhs_expr: ExprId,
59    ) -> Ty<'db> {
60        debug!(
61            "check_binop(expr.hir_id={:?}, expr={:?}, op={:?}, lhs_expr={:?}, rhs_expr={:?})",
62            expr, expr, op, lhs_expr, rhs_expr
63        );
64
65        match op {
66            BinaryOp::LogicOp(_) => {
67                // && and || are a simple case.
68                self.infer_expr_coerce(
69                    lhs_expr,
70                    &Expectation::HasType(self.types.types.bool),
71                    ExprIsRead::Yes,
72                );
73                let lhs_diverges = self.diverges;
74                self.infer_expr_coerce(
75                    rhs_expr,
76                    &Expectation::HasType(self.types.types.bool),
77                    ExprIsRead::Yes,
78                );
79
80                // Depending on the LHS' value, the RHS can never execute.
81                self.diverges = lhs_diverges;
82
83                self.types.types.bool
84            }
85            _ => {
86                // Otherwise, we always treat operators as if they are
87                // overloaded. This is the way to be most flexible w/r/t
88                // types that get inferred.
89                let (lhs_ty, rhs_ty, return_ty) =
90                    self.infer_overloaded_binop(expr, lhs_expr, rhs_expr, op);
91
92                // Supply type inference hints if relevant. Probably these
93                // hints should be enforced during select as part of the
94                // `consider_unification_despite_ambiguity` routine, but this
95                // more convenient for now.
96                //
97                // The basic idea is to help type inference by taking
98                // advantage of things we know about how the impls for
99                // scalar types are arranged. This is important in a
100                // scenario like `1_u32 << 2`, because it lets us quickly
101                // deduce that the result type should be `u32`, even
102                // though we don't know yet what type 2 has and hence
103                // can't pin this down to a specific impl.
104                let category = BinOpCategory::from(op);
105                if !lhs_ty.is_ty_var()
106                    && !rhs_ty.is_ty_var()
107                    && is_builtin_binop(lhs_ty, rhs_ty, category)
108                {
109                    let builtin_return_ty =
110                        self.enforce_builtin_binop_types(lhs_ty, rhs_ty, category);
111                    _ = self.demand_eqtype(expr.into(), builtin_return_ty, return_ty);
112                    builtin_return_ty
113                } else {
114                    return_ty
115                }
116            }
117        }
118    }

120    fn enforce_builtin_binop_types(
121        &mut self,
122        lhs_ty: Ty<'db>,
123        rhs_ty: Ty<'db>,
124        category: BinOpCategory,
125    ) -> Ty<'db> {
126        debug_assert!(is_builtin_binop(lhs_ty, rhs_ty, category));
127
128        // Special-case a single layer of referencing, so that things like `5.0 + &6.0f32` work.
129        // (See https://github.com/rust-lang/rust/issues/57447.)
130        let (lhs_ty, rhs_ty) = (deref_ty_if_possible(lhs_ty), deref_ty_if_possible(rhs_ty));
131
132        match category {
133            BinOpCategory::Shortcircuit => {
134                self.demand_suptype(self.types.types.bool, lhs_ty);
135                self.demand_suptype(self.types.types.bool, rhs_ty);
136                self.types.types.bool
137            }
138
139            BinOpCategory::Shift => {
140                // result type is same as LHS always
141                lhs_ty
142            }
143
144            BinOpCategory::Math | BinOpCategory::Bitwise => {
145                // both LHS and RHS and result will have the same type
146                self.demand_suptype(lhs_ty, rhs_ty);
147                lhs_ty
148            }
149
150            BinOpCategory::Comparison => {
151                // both LHS and RHS and result will have the same type
152                self.demand_suptype(lhs_ty, rhs_ty);
153                self.types.types.bool
154            }
155        }
156    }

158    fn infer_overloaded_binop(
159        &mut self,
160        expr: ExprId,
161        lhs_expr: ExprId,
162        rhs_expr: ExprId,
163        op: BinaryOp,
164    ) -> (Ty<'db>, Ty<'db>, Ty<'db>) {
165        debug!("infer_overloaded_binop(expr.hir_id={:?}, op={:?})", expr, op);
166
167        let lhs_ty = match op {
168            BinaryOp::Assignment { .. } => {
169                // rust-lang/rust#52126: We have to use strict
170                // equivalence on the LHS of an assign-op like `+=`;
171                // overwritten or mutably-borrowed places cannot be
172                // coerced to a supertype.
173                self.infer_expr_no_expect(lhs_expr, ExprIsRead::Yes)
174            }
175            _ => {
176                // Find a suitable supertype of the LHS expression's type, by coercing to
177                // a type variable, to pass as the `Self` to the trait, avoiding invariant
178                // trait matching creating lifetime constraints that are too strict.
179                // e.g., adding `&'a T` and `&'b T`, given `&'x T: Add<&'x T>`, will result
180                // in `&'a T <: &'x T` and `&'b T <: &'x T`, instead of `'a = 'b = 'x`.
181                let lhs_ty = self.infer_expr_no_expect(lhs_expr, ExprIsRead::Yes);
182                let fresh_var = self.table.next_ty_var();
183                self.demand_coerce(lhs_expr, lhs_ty, fresh_var, AllowTwoPhase::No, ExprIsRead::Yes)
184            }
185        };
186        let lhs_ty = self.table.resolve_vars_with_obligations(lhs_ty);
187
188        // N.B., as we have not yet type-checked the RHS, we don't have the
189        // type at hand. Make a variable to represent it. The whole reason
190        // for this indirection is so that, below, we can check the expr
191        // using this variable as the expected type, which sometimes lets
192        // us do better coercions than we would be able to do otherwise,
193        // particularly for things like `String + &String`.
194        let rhs_ty_var = self.table.next_ty_var();
195        let result = self.lookup_op_method(
196            lhs_ty,
197            Some((rhs_expr, rhs_ty_var)),
198            self.lang_item_for_bin_op(op),
199        );
200
201        // see `NB` above
202        let rhs_ty =
203            self.infer_expr_coerce(rhs_expr, &Expectation::HasType(rhs_ty_var), ExprIsRead::Yes);
204        let rhs_ty = self.table.resolve_vars_with_obligations(rhs_ty);
205
206        let return_ty = match result {
207            Ok(method) => {
208                let by_ref_binop = !is_op_by_value(op);
209                if (matches!(op, BinaryOp::Assignment { .. }) || by_ref_binop)
210                    && let TyKind::Ref(_, _, mutbl) =
211                        method.sig.inputs_and_output.inputs()[0].kind()
212                {
213                    let mutbl = AutoBorrowMutability::new(mutbl, AllowTwoPhase::Yes);
214                    let autoref = Adjustment {
215                        kind: Adjust::Borrow(AutoBorrow::Ref(mutbl)),
216                        target: method.sig.inputs_and_output.inputs()[0].store(),
217                    };
218                    self.write_expr_adj(lhs_expr, Box::new([autoref]));
219                }
220                if by_ref_binop
221                    && let TyKind::Ref(_, _, mutbl) =
222                        method.sig.inputs_and_output.inputs()[1].kind()
223                {
224                    // Allow two-phase borrows for binops in initial deployment
225                    // since they desugar to methods
226                    let mutbl = AutoBorrowMutability::new(mutbl, AllowTwoPhase::Yes);
227
228                    let autoref = Adjustment {
229                        kind: Adjust::Borrow(AutoBorrow::Ref(mutbl)),
230                        target: method.sig.inputs_and_output.inputs()[1].store(),
231                    };
232                    // HACK(eddyb) Bypass checks due to reborrows being in
233                    // some cases applied on the RHS, on top of which we need
234                    // to autoref, which is not allowed by write_expr_adj.
235                    // self.write_expr_adj(rhs_expr, Box::new([autoref]));
236                    match self.result.expr_adjustments.entry(rhs_expr) {
237                        hash_map::Entry::Occupied(mut entry) => {
238                            let mut adjustments = Vec::from(std::mem::take(entry.get_mut()));
239                            adjustments.reserve_exact(1);
240                            adjustments.push(autoref);
241                            entry.insert(adjustments.into_boxed_slice());
242                        }
243                        hash_map::Entry::Vacant(entry) => {
244                            entry.insert(Box::new([autoref]));
245                        }
246                    };
247                }
248                self.write_method_resolution(expr, method.def_id, method.args);
249
250                method.sig.output()
251            }
252            Err(_errors) => {
253                // FIXME: Report diagnostic.
254                self.types.types.error
255            }
256        };
257
258        (lhs_ty, rhs_ty, return_ty)
259    }

261    pub(crate) fn infer_user_unop(
262        &mut self,
263        ex: ExprId,
264        operand_ty: Ty<'db>,
265        op: UnaryOp,
266    ) -> Ty<'db> {
267        match self.lookup_op_method(operand_ty, None, self.lang_item_for_unop(op)) {
268            Ok(method) => {
269                self.write_method_resolution(ex, method.def_id, method.args);
270                method.sig.output()
271            }
272            Err(_errors) => {
273                // FIXME: Report diagnostic.
274                self.types.types.error
275            }
276        }
277    }

279    fn lookup_op_method(
280        &mut self,
281        lhs_ty: Ty<'db>,
282        opt_rhs: Option<(ExprId, Ty<'db>)>,
283        (opname, trait_did): (Symbol, Option<TraitId>),
284    ) -> Result<MethodCallee<'db>, Vec<NextSolverError<'db>>> {
285        let Some(trait_did) = trait_did else {
286            // Bail if the operator trait is not defined.
287            return Err(vec![]);
288        };
289
290        debug!(
291            "lookup_op_method(lhs_ty={:?}, opname={:?}, trait_did={:?})",
292            lhs_ty, opname, trait_did
293        );
294
295        let opt_rhs_ty = opt_rhs.map(|it| it.1);
296        let cause = ObligationCause::new();
297
298        // We don't consider any other candidates if this lookup fails
299        // so we can freely treat opaque types as inference variables here
300        // to allow more code to compile.
301        let treat_opaques = TreatNotYetDefinedOpaques::AsInfer;
302        let method = self.table.lookup_method_for_operator(
303            cause.clone(),
304            opname,
305            trait_did,
306            lhs_ty,
307            opt_rhs_ty,
308            treat_opaques,
309        );
310        match method {
311            Some(ok) => {
312                let method = self.table.register_infer_ok(ok);
313                self.table.select_obligations_where_possible();
314                Ok(method)
315            }
316            None => {
317                // Guide inference for the RHS expression if it's provided --
318                // this will allow us to better error reporting, at the expense
319                // of making some error messages a bit more specific.
320                if let Some((rhs_expr, rhs_ty)) = opt_rhs
321                    && rhs_ty.is_ty_var()
322                {
323                    self.infer_expr_coerce(
324                        rhs_expr,
325                        &Expectation::HasType(rhs_ty),
326                        ExprIsRead::Yes,
327                    );
328                }
329
330                // Construct an obligation `self_ty : Trait<input_tys>`
331                let args = GenericArgs::for_item(
332                    self.interner(),
333                    trait_did.into(),
334                    |param_idx, param_id, _| match param_id {
335                        GenericParamId::LifetimeParamId(_) | GenericParamId::ConstParamId(_) => {
336                            unreachable!("did not expect operand trait to have lifetime/const args")
337                        }
338                        GenericParamId::TypeParamId(_) => {
339                            if param_idx == 0 {
340                                lhs_ty.into()
341                            } else {
342                                opt_rhs_ty.expect("expected RHS for binop").into()
343                            }
344                        }
345                    },
346                );
347                let obligation = Obligation::new(
348                    self.interner(),
349                    cause,
350                    self.table.param_env,
351                    TraitRef::new_from_args(self.interner(), trait_did.into(), args),
352                );
353                let mut ocx = ObligationCtxt::new(self.infcx());
354                ocx.register_obligation(obligation);
355                Err(ocx.evaluate_obligations_error_on_ambiguity())
356            }
357        }
358    }

360    fn lang_item_for_bin_op(&self, op: BinaryOp) -> (Symbol, Option<TraitId>) {
361        let (method_name, trait_lang_item) =
362            crate::lang_items::lang_items_for_bin_op(self.lang_items, op)
363                .expect("invalid operator provided");
364        (method_name, trait_lang_item)
365    }
366
367    fn lang_item_for_unop(&self, op: UnaryOp) -> (Symbol, Option<TraitId>) {
368        let (method_name, trait_lang_item) = match op {
369            UnaryOp::Not => (sym::not, self.lang_items.Not),
370            UnaryOp::Neg => (sym::neg, self.lang_items.Neg),
371            UnaryOp::Deref => panic!("Deref is not overloadable"),
372        };
373        (method_name, trait_lang_item)
374    }

377// Binary operator categories. These categories summarize the behavior
378// with respect to the builtin operations supported.
379#[derive(Clone, Copy)]
380enum BinOpCategory {
381    /// &&, || -- cannot be overridden
382    Shortcircuit,
383
384    /// <<, >> -- when shifting a single integer, rhs can be any
385    /// integer type. For simd, types must match.
386    Shift,
387
388    /// +, -, etc -- takes equal types, produces same type as input,
389    /// applicable to ints/floats/simd
390    Math,
391
392    /// &, |, ^ -- takes equal types, produces same type as input,
393    /// applicable to ints/floats/simd/bool
394    Bitwise,
395
396    /// ==, !=, etc -- takes equal types, produces bools, except for simd,
397    /// which produce the input type
398    Comparison,
399}
400
401impl From<BinaryOp> for BinOpCategory {
402    fn from(op: BinaryOp) -> BinOpCategory {
403        match op {
404            BinaryOp::LogicOp(_) => BinOpCategory::Shortcircuit,
405            BinaryOp::ArithOp(op) | BinaryOp::Assignment { op: Some(op) } => op.into(),
406            BinaryOp::CmpOp(_) => BinOpCategory::Comparison,
407            BinaryOp::Assignment { op: None } => unreachable!(
408                "assignment is lowered into `Expr::Assignment`, not into `Expr::BinaryOp`"
409            ),
410        }
411    }
412}
413
414impl From<ArithOp> for BinOpCategory {
415    fn from(op: ArithOp) -> BinOpCategory {
416        use ArithOp::*;
417        match op {
418            Shl | Shr => BinOpCategory::Shift,
419            Add | Sub | Mul | Div | Rem => BinOpCategory::Math,
420            BitXor | BitAnd | BitOr => BinOpCategory::Bitwise,
421        }
422    }
423}

425/// Returns `true` if the binary operator takes its arguments by value.
426fn is_op_by_value(op: BinaryOp) -> bool {
427    !matches!(op, BinaryOp::CmpOp(_))
428}

430/// Dereferences a single level of immutable referencing.
431fn deref_ty_if_possible(ty: Ty<'_>) -> Ty<'_> {
432    match ty.kind() {
433        TyKind::Ref(_, ty, Mutability::Not) => ty,
434        _ => ty,
435    }

438/// Returns `true` if this is a built-in arithmetic operation (e.g.,
439/// u32 + u32, i16x4 == i16x4) and false if these types would have to be
440/// overloaded to be legal. There are two reasons that we distinguish
441/// builtin operations from overloaded ones (vs trying to drive
442/// everything uniformly through the trait system and intrinsics or
443/// something like that):
444///
445/// 1. Builtin operations can trivially be evaluated in constants.
446/// 2. For comparison operators applied to SIMD types the result is
447///    not of type `bool`. For example, `i16x4 == i16x4` yields a
448///    type like `i16x4`. This means that the overloaded trait
449///    `PartialEq` is not applicable.
450///
451/// Reason #2 is the killer. I tried for a while to always use
452/// overloaded logic and just check the types in constants/codegen after
453/// the fact, and it worked fine, except for SIMD types. -nmatsakis
454fn is_builtin_binop<'db>(lhs: Ty<'db>, rhs: Ty<'db>, category: BinOpCategory) -> bool {
455    // Special-case a single layer of referencing, so that things like `5.0 + &6.0f32` work.
456    // (See https://github.com/rust-lang/rust/issues/57447.)
457    let (lhs, rhs) = (deref_ty_if_possible(lhs), deref_ty_if_possible(rhs));
458
459    match category {
460        BinOpCategory::Shortcircuit => true,
461        BinOpCategory::Shift => lhs.is_integral() && rhs.is_integral(),
462        BinOpCategory::Math => {
463            lhs.is_integral() && rhs.is_integral()
464                || lhs.is_floating_point() && rhs.is_floating_point()
465        }
466        BinOpCategory::Bitwise => {
467            lhs.is_integral() && rhs.is_integral()
468                || lhs.is_floating_point() && rhs.is_floating_point()
469                || lhs.is_bool() && rhs.is_bool()
470        }
471        BinOpCategory::Comparison => lhs.is_scalar() && rhs.is_scalar(),
472    }
