6pub fn int_ty_to_string(ty: IntTy) -> &'static str {
7    match ty {
8        IntTy::Isize => "isize",
9        IntTy::I8 => "i8",
10        IntTy::I16 => "i16",
11        IntTy::I32 => "i32",

14    }
15}
