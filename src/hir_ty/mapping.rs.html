1//! This module contains the implementations of the `ToChalk` trait, which
2//! handles conversion between our data types and their corresponding types in
3//! Chalk (in both directions); plus some helper functions for more specialized
4//! conversions.
5

19pub(crate) trait ToChalk {
20    type Chalk;
21    fn to_chalk(self, db: &dyn HirDatabase) -> Self::Chalk;
22    fn from_chalk(db: &dyn HirDatabase, chalk: Self::Chalk) -> Self;
23}
24
25pub(crate) fn from_chalk<T, ChalkT>(db: &dyn HirDatabase, chalk: ChalkT) -> T
26where
27    T: ToChalk<Chalk = ChalkT>,
28{
29    T::from_chalk(db, chalk)
30}
31
32impl ToChalk for hir_def::ImplId {
33    type Chalk = chalk_db::ImplId;
34
35    fn to_chalk(self, _db: &dyn HirDatabase) -> chalk_db::ImplId {
36        chalk_ir::ImplId(self.as_id())
37    }
38
39    fn from_chalk(_db: &dyn HirDatabase, impl_id: chalk_db::ImplId) -> hir_def::ImplId {
40        FromId::from_id(impl_id.0.as_id())
41    }
42}
43
44impl ToChalk for CallableDefId {
45    type Chalk = FnDefId;
46
47    fn to_chalk(self, db: &dyn HirDatabase) -> FnDefId {
48        db.intern_callable_def(self).into()
49    }
50
51    fn from_chalk(db: &dyn HirDatabase, fn_def_id: FnDefId) -> CallableDefId {
52        db.lookup_intern_callable_def(fn_def_id.into())
53    }
54}

137pub fn from_placeholder_idx(db: &dyn HirDatabase, idx: PlaceholderIndex) -> TypeOrConstParamId {
138    assert_eq!(idx.ui, chalk_ir::UniverseIndex::ROOT);
139    let interned_id = FromId::from_id(Id::from_u32(idx.idx.try_into().unwrap()));
140    db.lookup_intern_type_or_const_param_id(interned_id)
141}
142
143pub fn to_placeholder_idx(db: &dyn HirDatabase, id: TypeOrConstParamId) -> PlaceholderIndex {
144    let interned_id = db.intern_type_or_const_param_id(id);
145    PlaceholderIndex {
146        ui: chalk_ir::UniverseIndex::ROOT,
147        idx: interned_id.as_id().as_u32() as usize,
148    }
149}

151pub fn lt_from_placeholder_idx(db: &dyn HirDatabase, idx: PlaceholderIndex) -> LifetimeParamId {
