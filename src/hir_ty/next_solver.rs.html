133#[inline]
134pub fn default_types<'a, 'db>(db: &'db dyn HirDatabase) -> &'a DefaultAny<'db> {
135    static TYPES: OnceLock<DefaultAny<'static>> = OnceLock::new();
136
137    let interner = DbInterner::new_no_crate(db);
138    TYPES.get_or_init(|| {
139        let create_ty = |kind| {
140            let ty = Ty::new(interner, kind);
141            // We need to increase the refcount (forever), so that the types won't be freed.
142            let ty = ManuallyDrop::new(ty.store());
143            ty.as_ref()
144        };
145        let create_const = |kind| {
146            let ty = Const::new(interner, kind);
147            // We need to increase the refcount (forever), so that the types won't be freed.
148            let ty = ManuallyDrop::new(ty.store());
149            ty.as_ref()
150        };
151        let create_region = |kind| {
152            let ty = Region::new(interner, kind);
153            // We need to increase the refcount (forever), so that the types won't be freed.
154            let ty = ManuallyDrop::new(ty.store());
155            ty.as_ref()
156        };
157        let create_generic_args = |slice| {
158            let ty = GenericArgs::new_from_slice(slice);
159            // We need to increase the refcount (forever), so that the types won't be freed.
160            let ty = ManuallyDrop::new(ty.store());
161            ty.as_ref()
162        };
163        let create_bound_var_kinds = |slice| {
164            let ty = BoundVarKinds::new_from_slice(slice);
165            // We need to increase the refcount (forever), so that the types won't be freed.
166            let ty = ManuallyDrop::new(ty.store());
167            ty.as_ref()
168        };
169        let create_canonical_vars = |slice| {
170            let ty = CanonicalVars::new_from_slice(slice);
171            // We need to increase the refcount (forever), so that the types won't be freed.
172            let ty = ManuallyDrop::new(ty.store());
173            ty.as_ref()
174        };
175        let create_variances_of = |slice| {
176            let ty = VariancesOf::new_from_slice(slice);
177            // We need to increase the refcount (forever), so that the types won't be freed.
178            let ty = ManuallyDrop::new(ty.store());
179            ty.as_ref()
180        };
181        let create_pat_list = |slice| {
182            let ty = PatList::new_from_slice(slice);
183            // We need to increase the refcount (forever), so that the types won't be freed.
184            let ty = ManuallyDrop::new(ty.store());
185            ty.as_ref()
186        };
187        let create_predefined_opaques = |slice| {
188            let ty = PredefinedOpaques::new_from_slice(slice);
189            // We need to increase the refcount (forever), so that the types won't be freed.
190            let ty = ManuallyDrop::new(ty.store());
191            ty.as_ref()
192        };
193        let create_solver_def_ids = |slice| {
194            let ty = SolverDefIds::new_from_slice(slice);
195            // We need to increase the refcount (forever), so that the types won't be freed.
196            let ty = ManuallyDrop::new(ty.store());
197            ty.as_ref()
198        };
199        let create_bound_existential_predicates = |slice| {
200            let ty = BoundExistentialPredicates::new_from_slice(slice);
201            // We need to increase the refcount (forever), so that the types won't be freed.
202            let ty = ManuallyDrop::new(ty.store());
203            ty.as_ref()
204        };
205        let create_clauses = |slice| {
206            let ty = Clauses::new_from_slice(slice);
207            // We need to increase the refcount (forever), so that the types won't be freed.
208            let ty = ManuallyDrop::new(ty.store());
209            ty.as_ref()
210        };
211        let create_region_assumptions = |slice| {
212            let ty = RegionAssumptions::new_from_slice(slice);
213            // We need to increase the refcount (forever), so that the types won't be freed.
214            let ty = ManuallyDrop::new(ty.store());
215            ty.as_ref()
216        };
217        let create_tys = |slice| {
218            let ty = Tys::new_from_slice(slice);
219            // We need to increase the refcount (forever), so that the types won't be freed.
220            let ty = ManuallyDrop::new(ty.store());
221            ty.as_ref()
222        };
223
224        let str = create_ty(TyKind::Str);
225        let statik = create_region(RegionKind::ReStatic);
226        let empty_tys = create_tys(&[]);
227        let unit = create_ty(TyKind::Tuple(empty_tys));
228        DefaultAny {
229            types: DefaultTypes {
230                usize: create_ty(TyKind::Uint(rustc_ast_ir::UintTy::Usize)),
231                u8: create_ty(TyKind::Uint(rustc_ast_ir::UintTy::U8)),
232                u16: create_ty(TyKind::Uint(rustc_ast_ir::UintTy::U16)),
233                u32: create_ty(TyKind::Uint(rustc_ast_ir::UintTy::U32)),
234                u64: create_ty(TyKind::Uint(rustc_ast_ir::UintTy::U64)),
235                u128: create_ty(TyKind::Uint(rustc_ast_ir::UintTy::U128)),
236                isize: create_ty(TyKind::Int(rustc_ast_ir::IntTy::Isize)),
237                i8: create_ty(TyKind::Int(rustc_ast_ir::IntTy::I8)),
238                i16: create_ty(TyKind::Int(rustc_ast_ir::IntTy::I16)),
239                i32: create_ty(TyKind::Int(rustc_ast_ir::IntTy::I32)),
240                i64: create_ty(TyKind::Int(rustc_ast_ir::IntTy::I64)),
241                i128: create_ty(TyKind::Int(rustc_ast_ir::IntTy::I128)),
242                f16: create_ty(TyKind::Float(rustc_ast_ir::FloatTy::F16)),
243                f32: create_ty(TyKind::Float(rustc_ast_ir::FloatTy::F32)),
244                f64: create_ty(TyKind::Float(rustc_ast_ir::FloatTy::F64)),
245                f128: create_ty(TyKind::Float(rustc_ast_ir::FloatTy::F128)),
246                unit,
247                bool: create_ty(TyKind::Bool),
248                char: create_ty(TyKind::Char),
249                str,
250                never: create_ty(TyKind::Never),
251                error: create_ty(TyKind::Error(ErrorGuaranteed)),
252                static_str_ref: create_ty(TyKind::Ref(statik, str, rustc_ast_ir::Mutability::Not)),
253                mut_unit_ptr: create_ty(TyKind::RawPtr(unit, rustc_ast_ir::Mutability::Mut)),
254            },
255            consts: DefaultConsts { error: create_const(ConstKind::Error(ErrorGuaranteed)) },
256            regions: DefaultRegions {
257                error: create_region(RegionKind::ReError(ErrorGuaranteed)),
258                statik,
259                erased: create_region(RegionKind::ReErased),
260            },
261            empty: DefaultEmpty {
262                tys: empty_tys,
263                generic_args: create_generic_args(&[]),
264                bound_var_kinds: create_bound_var_kinds(&[]),
265                canonical_vars: create_canonical_vars(&[]),
266                variances: create_variances_of(&[]),
267                pat_list: create_pat_list(&[]),
268                predefined_opaques: create_predefined_opaques(&[]),
269                def_ids: create_solver_def_ids(&[]),
270                bound_existential_predicates: create_bound_existential_predicates(&[]),
271                clauses: create_clauses(&[]),
272                region_assumptions: create_region_assumptions(&[]),
273            },
274            one_invariant: create_variances_of(&[rustc_type_ir::Variance::Invariant]),
275            one_covariant: create_variances_of(&[rustc_type_ir::Variance::Covariant]),
276            coroutine_captures_by_ref_bound_var_kinds: create_bound_var_kinds(&[
277                BoundVarKind::Region(BoundRegionKind::ClosureEnv),
278            ]),
279        }
280    })
