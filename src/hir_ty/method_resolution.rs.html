48/// This is used as a key for indexing impls.
49#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, PartialOrd, Ord)]
50pub enum TyFingerprint {
51    // These are lang item impls:
52    Str,
53    Slice,
54    Array,
55    Never,
56    Ref(Mutability),
57    RawPtr(Mutability),
58    Bool,
59    Char,
60    Int(IntTy),
61    Uint(UintTy),
62    Float(FloatTy),
63    // These can have user-defined impls:
64    Adt(hir_def::AdtId),
65    Dyn(TraitId),
66    ForeignType(TypeAliasId),
67    // These only exist for trait impls
68    Unit,
69    Unnameable,
70    Function(u32),
71}

73impl TyFingerprint {
74    /// Creates a TyFingerprint for looking up an inherent impl. Only certain
75    /// types can have inherent impls: if we have some `struct S`, we can have
76    /// an `impl S`, but not `impl &S`. Hence, this will return `None` for
77    /// reference types and such.
78    pub fn for_inherent_impl<'db>(ty: Ty<'db>) -> Option<TyFingerprint> {
79        let fp = match ty.kind() {
80            TyKind::Str => TyFingerprint::Str,
81            TyKind::Never => TyFingerprint::Never,
82            TyKind::Slice(..) => TyFingerprint::Slice,
83            TyKind::Array(..) => TyFingerprint::Array,
84            TyKind::Bool => TyFingerprint::Bool,
85            TyKind::Char => TyFingerprint::Char,
86            TyKind::Int(int) => TyFingerprint::Int(int),
87            TyKind::Uint(int) => TyFingerprint::Uint(int),
88            TyKind::Float(float) => TyFingerprint::Float(float),
89            TyKind::Adt(adt_def, _) => TyFingerprint::Adt(adt_def.def_id().0),
90            TyKind::RawPtr(_, mutability) => TyFingerprint::RawPtr(mutability),
91            TyKind::Foreign(alias_id, ..) => TyFingerprint::ForeignType(alias_id.0),
92            TyKind::Dynamic(bounds, _) => {
93                bounds.principal_def_id().map(|trait_| TyFingerprint::Dyn(trait_.0))?
94            }
95            _ => return None,
96        };
97        Some(fp)
98    }
99
100    /// Creates a TyFingerprint for looking up a trait impl.
101    pub fn for_trait_impl<'db>(ty: Ty<'db>) -> Option<TyFingerprint> {
102        let fp = match ty.kind() {
103            TyKind::Str => TyFingerprint::Str,
104            TyKind::Never => TyFingerprint::Never,
105            TyKind::Slice(..) => TyFingerprint::Slice,
106            TyKind::Array(..) => TyFingerprint::Array,
107            TyKind::Bool => TyFingerprint::Bool,
108            TyKind::Char => TyFingerprint::Char,
109            TyKind::Int(int) => TyFingerprint::Int(int),
110            TyKind::Uint(int) => TyFingerprint::Uint(int),
111            TyKind::Float(float) => TyFingerprint::Float(float),
112            TyKind::Adt(adt_def, _) => TyFingerprint::Adt(adt_def.def_id().0),
113            TyKind::RawPtr(_, mutability) => TyFingerprint::RawPtr(mutability),
114            TyKind::Foreign(alias_id, ..) => TyFingerprint::ForeignType(alias_id.0),
115            TyKind::Dynamic(bounds, _) => {
116                bounds.principal_def_id().map(|trait_| TyFingerprint::Dyn(trait_.0))?
117            }
118            TyKind::Ref(_, _, mutability) => TyFingerprint::Ref(mutability),
119            TyKind::Tuple(subst) => {
120                let first_ty = subst.as_slice().first();
121                match first_ty {
122                    Some(ty) => return TyFingerprint::for_trait_impl(*ty),
123                    None => TyFingerprint::Unit,
124                }
125            }
126            // FIXME(next-solver): Putting `Alias` here is *probably* incorrect, AFAIK it should return `None`. But this breaks
127            // flyimport, which uses an incorrect but fast method resolution algorithm. Therefore we put it here,
128            // because this function is only called by flyimport, and anyway we should get rid of `TyFingerprint`
129            // and switch to `rustc_type_ir`'s `SimplifiedType`.
130            TyKind::Alias(..)
131            | TyKind::FnDef(_, _)
132            | TyKind::Closure(_, _)
133            | TyKind::Coroutine(..)
134            | TyKind::CoroutineClosure(..)
135            | TyKind::CoroutineWitness(..) => TyFingerprint::Unnameable,
136            TyKind::FnPtr(sig, _) => {
137                TyFingerprint::Function(sig.skip_binder().inputs_and_output.inner().len() as u32)
138            }
139            TyKind::Param(_)
140            | TyKind::Bound(..)
141            | TyKind::Placeholder(..)
142            | TyKind::Infer(_)
143            | TyKind::Error(_)
144            | TyKind::Pat(..)
145            | TyKind::UnsafeBinder(..) => return None,
146        };
147        Some(fp)
148    }
149}

183impl TraitImpls {
184    pub(crate) fn trait_impls_in_crate_query(db: &dyn HirDatabase, krate: Crate) -> Arc<Self> {
185        let _p = tracing::info_span!("trait_impls_in_crate_query", ?krate).entered();
186        let mut impls = FxHashMap::default();

193    pub(crate) fn trait_impls_in_block_query(
194        db: &dyn HirDatabase,
195        block: BlockId,

197        let _p = tracing::info_span!("trait_impls_in_block_query").entered();
198        let mut impls = FxHashMap::default();
199
200        Self::collect_def_map(db, &mut impls, block_def_map(db, block));
201
202        if impls.is_empty() { None } else { Some(Arc::new(Self::finish(impls))) }
203    }

205    pub(crate) fn trait_impls_in_deps_query(
206        db: &dyn HirDatabase,
207        krate: Crate,

209        let _p = tracing::info_span!("trait_impls_in_deps_query", ?krate).entered();
210        Arc::from_iter(
211            db.transitive_deps(krate).into_iter().map(|krate| db.trait_impls_in_crate(krate)),

215    fn finish(map: TraitFpMapCollector) -> TraitImpls {
216        TraitImpls {
217            map: map
218                .into_iter()
219                .map(|(k, v)| (k, v.into_iter().map(|(k, v)| (k, v.into_boxed_slice())).collect()))
220                .collect(),
221        }
222    }
223
224    fn collect_def_map(db: &dyn HirDatabase, map: &mut TraitFpMapCollector, def_map: &DefMap) {
225        for (_module_id, module_data) in def_map.modules() {
226            for impl_id in module_data.scope.impls() {
227                // Reservation impls should be ignored during trait resolution, so we never need
228                // them during type analysis. See rust-lang/rust#64631 for details.
229                //
230                // FIXME: Reservation impls should be considered during coherence checks. If we are
231                // (ever) to implement coherence checks, this filtering should be done by the trait
232                // solver.
233                if db.attrs(impl_id.into()).by_key(sym::rustc_reservation_impl).exists() {
234                    continue;
235                }
236                let target_trait = match db.impl_trait(impl_id) {
237                    Some(tr) => tr.skip_binder().def_id.0,
238                    None => continue,
239                };
240                let self_ty = db.impl_self_ty(impl_id);
241                let self_ty_fp = TyFingerprint::for_trait_impl(self_ty.instantiate_identity());
242                map.entry(target_trait).or_default().entry(self_ty_fp).or_default().push(impl_id);
243            }
244
245            // To better support custom derives, collect impls in all unnamed const items.
246            // const _: () = { ... };
247            for konst in module_data.scope.unnamed_consts() {
248                let body = db.body(konst.into());
249                for (_, block_def_map) in body.blocks(db) {
250                    Self::collect_def_map(db, map, block_def_map);
251                }
252            }
253        }
254    }

298/// Inherent impls defined in some crate.
299///
300/// Inherent impls can only be defined in the crate that also defines the self type of the impl
301/// (note that some primitives are considered to be defined by both libcore and liballoc).
302///
303/// This makes inherent impl lookup easier than trait impl lookup since we only have to consider a
304/// single crate.
305#[derive(Debug, Eq, PartialEq)]
306pub struct InherentImpls {
307    map: FxHashMap<TyFingerprint, Vec<ImplId>>,
308    invalid_impls: Vec<ImplId>,
309}
310
311impl InherentImpls {
312    pub(crate) fn inherent_impls_in_crate_query(db: &dyn HirDatabase, krate: Crate) -> Arc<Self> {
313        let _p = tracing::info_span!("inherent_impls_in_crate_query", ?krate).entered();
314        let mut impls = Self { map: FxHashMap::default(), invalid_impls: Vec::default() };
315
316        let crate_def_map = crate_def_map(db, krate);
317        impls.collect_def_map(db, crate_def_map);
318        impls.shrink_to_fit();
319
320        Arc::new(impls)
321    }
322
323    pub(crate) fn inherent_impls_in_block_query(
324        db: &dyn HirDatabase,
325        block: BlockId,
326    ) -> Option<Arc<Self>> {
327        let _p = tracing::info_span!("inherent_impls_in_block_query").entered();
328        let mut impls = Self { map: FxHashMap::default(), invalid_impls: Vec::default() };
329
330        let block_def_map = block_def_map(db, block);
331        impls.collect_def_map(db, block_def_map);
332        impls.shrink_to_fit();
333
334        if impls.map.is_empty() && impls.invalid_impls.is_empty() {
335            None
336        } else {
337            Some(Arc::new(impls))
338        }
339    }
340
341    fn shrink_to_fit(&mut self) {
342        self.map.values_mut().for_each(Vec::shrink_to_fit);
343        self.map.shrink_to_fit();
344    }
345
346    fn collect_def_map(&mut self, db: &dyn HirDatabase, def_map: &DefMap) {
347        for (_module_id, module_data) in def_map.modules() {
348            for impl_id in module_data.scope.impls() {
349                let data = db.impl_signature(impl_id);
350                if data.target_trait.is_some() {
351                    continue;
352                }
353
354                let self_ty = db.impl_self_ty(impl_id);
355                let self_ty = self_ty.instantiate_identity();
356
357                match is_inherent_impl_coherent(db, def_map, impl_id, self_ty) {
358                    true => {
359                        // `fp` should only be `None` in error cases (either erroneous code or incomplete name resolution)
360                        if let Some(fp) = TyFingerprint::for_inherent_impl(self_ty) {
361                            self.map.entry(fp).or_default().push(impl_id);
362                        }
363                    }
364                    false => self.invalid_impls.push(impl_id),
365                }
366            }
367
368            // To better support custom derives, collect impls in all unnamed const items.
369            // const _: () = { ... };
370            for konst in module_data.scope.unnamed_consts() {
371                let body = db.body(konst.into());
372                for (_, block_def_map) in body.blocks(db) {
373                    self.collect_def_map(db, block_def_map);
374                }
375            }
376        }
377    }
378
379    pub fn for_self_ty<'db>(&self, self_ty: Ty<'db>) -> &[ImplId] {
380        match TyFingerprint::for_inherent_impl(self_ty) {
381            Some(fp) => self.map.get(&fp).map(|vec| vec.as_ref()).unwrap_or(&[]),
382            None => &[],
383        }
384    }
385
386    pub fn all_impls(&self) -> impl Iterator<Item = ImplId> + '_ {
387        self.map.values().flat_map(|v| v.iter().copied())
388    }
389
390    pub fn invalid_impls(&self) -> &[ImplId] {
391        &self.invalid_impls
392    }
393}

395pub(crate) fn incoherent_inherent_impl_crates(
396    db: &dyn HirDatabase,
397    krate: Crate,
398    fp: TyFingerprint,
399) -> SmallVec<[Crate; 2]> {
400    let _p = tracing::info_span!("incoherent_inherent_impl_crates").entered();
401    let mut res = SmallVec::new();
402
403    // should pass crate for finger print and do reverse deps
404
405    for krate in db.transitive_deps(krate) {
406        let impls = db.inherent_impls_in_crate(krate);
407        if impls.map.get(&fp).is_some_and(|v| !v.is_empty()) {
408            res.push(krate);
409        }
410    }
411
412    res

415pub fn def_crates<'db>(
416    db: &'db dyn HirDatabase,
417    ty: Ty<'db>,
418    cur_crate: Crate,
419) -> Option<SmallVec<[Crate; 2]>> {
420    match ty.kind() {
421        TyKind::Adt(adt_def, _) => {
422            let def_id = adt_def.def_id().0;
423            let rustc_has_incoherent_inherent_impls = match def_id {
424                hir_def::AdtId::StructId(id) => db
425                    .struct_signature(id)
426                    .flags
427                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
428                hir_def::AdtId::UnionId(id) => db
429                    .union_signature(id)
430                    .flags
431                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
432                hir_def::AdtId::EnumId(id) => db
433                    .enum_signature(id)
434                    .flags
435                    .contains(EnumFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
436            };
437            Some(if rustc_has_incoherent_inherent_impls {
438                db.incoherent_inherent_impl_crates(cur_crate, TyFingerprint::Adt(def_id))
439            } else {
440                smallvec![def_id.module(db).krate()]
441            })
442        }
443        TyKind::Foreign(alias) => {
444            let alias = alias.0;
445            Some(
446                if db
447                    .type_alias_signature(alias)
448                    .flags
449                    .contains(TypeAliasFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPL)
450                {
451                    db.incoherent_inherent_impl_crates(cur_crate, TyFingerprint::ForeignType(alias))
452                } else {
453                    smallvec![alias.module(db).krate()]
454                },
455            )
456        }
457        TyKind::Dynamic(bounds, _) => {
458            let trait_id = bounds.principal_def_id()?.0;
459            Some(
460                if db
461                    .trait_signature(trait_id)
462                    .flags
463                    .contains(TraitFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS)
464                {
465                    db.incoherent_inherent_impl_crates(cur_crate, TyFingerprint::Dyn(trait_id))
466                } else {
467                    smallvec![trait_id.module(db).krate()]
468                },
469            )
470        }
471        // for primitives, there may be impls in various places (core and alloc
472        // mostly). We just check the whole crate graph for crates with impls
473        // (cached behind a query).
474        TyKind::Bool
475        | TyKind::Char
476        | TyKind::Int(_)
477        | TyKind::Uint(_)
478        | TyKind::Float(_)
479        | TyKind::Str
480        | TyKind::Slice(_)
481        | TyKind::Array(..)
482        | TyKind::RawPtr(..) => Some(db.incoherent_inherent_impl_crates(
483            cur_crate,
484            TyFingerprint::for_inherent_impl(ty).expect("fingerprint for primitive"),
485        )),
486        _ => None,
487    }
488}

490/// Look up the method with the given name.
491pub(crate) fn lookup_method<'db>(
492    ty: &Canonical<'db, Ty<'db>>,
493    table: &mut InferenceTable<'db>,
494    traits_in_scope: &FxHashSet<TraitId>,
495    visible_from_module: VisibleFromModule,
496    name: &Name,
497) -> Option<(ReceiverAdjustments, FunctionId, bool)> {
498    let mut not_visible = None;
499    let res = iterate_method_candidates(
500        ty,
501        table,
502        traits_in_scope,
503        visible_from_module,
504        Some(name),
505        LookupMode::MethodCall,
506        |adjustments, f, visible| match f {
507            AssocItemId::FunctionId(f) if visible => Some((adjustments, f, true)),
508            AssocItemId::FunctionId(f) if not_visible.is_none() => {
509                not_visible = Some((adjustments, f, false));
510                None
511            }
512            _ => None,
513        },
514    );
515    res.or(not_visible)
516}

518/// Whether we're looking up a dotted method call (like `v.len()`) or a path
519/// (like `Vec::new`).
520#[derive(Copy, Clone, Debug, PartialEq, Eq)]
521pub enum LookupMode {
522    /// Looking up a method call like `v.len()`: We only consider candidates
523    /// that have a `self` parameter, and do autoderef.
524    MethodCall,
525    /// Looking up a path like `Vec::new` or `Vec::default`: We consider all
526    /// candidates including associated constants, but don't do autoderef.
527    Path,
528}

571impl ReceiverAdjustments {
572    pub(crate) fn apply<'db>(
573        &self,
574        table: &mut InferenceTable<'db>,
575        mut ty: Ty<'db>,
576    ) -> (Ty<'db>, Vec<Adjustment<'db>>) {
577        let mut adjust = Vec::new();
578        let mut autoderef = table.autoderef(ty);
579        autoderef.next();
580        for _ in 0..self.autoderefs {
581            match autoderef.next() {
582                None => {
583                    never!("autoderef not possible for {:?}", ty);
584                    ty = Ty::new_error(table.interner(), ErrorGuaranteed);
585                    break;
586                }
587                Some((new_ty, _)) => {
588                    ty = new_ty;
589                    let mutbl = match self.autoref {
590                        Some(AutorefOrPtrAdjustment::Autoref(m)) => Some(m),
591                        Some(AutorefOrPtrAdjustment::ToConstPtr) => Some(Mutability::Not),
592                        // FIXME should we know the mutability here, when autoref is `None`?
593                        None => None,
594                    };
595                    adjust.push(Adjustment {
596                        kind: Adjust::Deref(match autoderef.steps().last().unwrap().1 {
597                            AutoderefKind::Overloaded => Some(OverloadedDeref(mutbl)),
598                            AutoderefKind::Builtin => None,
599                        }),
600                        target: ty,
601                    });
602                }
603            }
604        }
605        if let Some(autoref) = &self.autoref {
606            let lt = table.next_region_var();
607            match autoref {
608                AutorefOrPtrAdjustment::Autoref(m) => {
609                    let a = Adjustment::borrow(table.interner(), *m, ty, lt);
610                    ty = a.target;
611                    adjust.push(a);
612                }
613                AutorefOrPtrAdjustment::ToConstPtr => {
614                    if let TyKind::RawPtr(pointee, Mutability::Mut) = ty.kind() {
615                        let a = Adjustment {
616                            kind: Adjust::Pointer(PointerCast::MutToConstPointer),
617                            target: Ty::new_ptr(table.interner(), pointee, Mutability::Not),
618                        };
619                        ty = a.target;
620                        adjust.push(a);
621                    } else {
622                        never!("`ToConstPtr` target is not a raw mutable pointer");
623                    }
624                }
625            };
626        }
627        if self.unsize_array {
628            ty = 'it: {
629                if let TyKind::Ref(l, inner, m) = ty.kind()
630                    && let TyKind::Array(inner, _) = inner.kind()
631                {
632                    break 'it Ty::new_ref(
633                        table.interner(),
634                        l,
635                        Ty::new_slice(table.interner(), inner),
636                        m,
637                    );
638                }
639                // FIXME: report diagnostic if array unsizing happens without indirection.
640                ty
641            };
642            adjust.push(Adjustment { kind: Adjust::Pointer(PointerCast::Unsize), target: ty });
643        }
644        (ty, adjust)
645    }
646
647    fn with_autoref(&self, a: AutorefOrPtrAdjustment) -> ReceiverAdjustments {
648        Self { autoref: Some(a), ..*self }
649    }
650}

712/// Checks if the self parameter of `Trait` method is the `dyn Trait` and we should
713/// call the method using the vtable.
714pub fn is_dyn_method<'db>(
715    interner: DbInterner<'db>,
716    _env: Arc<TraitEnvironment<'db>>,
717    func: FunctionId,
718    fn_subst: GenericArgs<'db>,
719) -> Option<usize> {
720    let db = interner.db;
721
722    let ItemContainerId::TraitId(trait_id) = func.lookup(db).container else {
723        return None;
724    };
725    let trait_params = db.generic_params(trait_id.into()).len();
726    let fn_params = fn_subst.len() - trait_params;
727    let trait_ref = TraitRef::new(
728        interner,
729        trait_id.into(),
730        GenericArgs::new_from_iter(interner, fn_subst.iter().take(trait_params)),
731    );
732    let self_ty = trait_ref.self_ty();
733    if let TyKind::Dynamic(d, _) = self_ty.kind() {
734        // rustc doesn't accept `impl Foo<2> for dyn Foo<5>`, so if the trait id is equal, no matter
735        // what the generics are, we are sure that the method is come from the vtable.
736        let is_my_trait_in_bounds = d
737            .principal_def_id()
738            .is_some_and(|trait_| all_super_traits(db, trait_.0).contains(&trait_id));
739        if is_my_trait_in_bounds {
740            return Some(fn_params);
741        }
742    }
743    None
744}

746/// Looks up the impl method that actually runs for the trait method `func`.
747///
748/// Returns `func` if it's not a method defined in a trait or the lookup failed.
749pub(crate) fn lookup_impl_method_query<'db>(
750    db: &'db dyn HirDatabase,
751    env: Arc<TraitEnvironment<'db>>,
752    func: FunctionId,
753    fn_subst: GenericArgs<'db>,
754) -> (FunctionId, GenericArgs<'db>) {
755    let interner = DbInterner::new_with(db, Some(env.krate), env.block);
756    let infcx = interner.infer_ctxt().build(TypingMode::PostAnalysis);
757
758    let ItemContainerId::TraitId(trait_id) = func.lookup(db).container else {
759        return (func, fn_subst);
760    };
761    let trait_params = db.generic_params(trait_id.into()).len();
762    let trait_ref = TraitRef::new(
763        interner,
764        trait_id.into(),
765        GenericArgs::new_from_iter(interner, fn_subst.iter().take(trait_params)),
766    );
767
768    let name = &db.function_signature(func).name;
769    let Some((impl_fn, impl_subst)) =
770        lookup_impl_assoc_item_for_trait_ref(&infcx, trait_ref, env, name).and_then(|assoc| {
771            if let (AssocItemId::FunctionId(id), subst) = assoc { Some((id, subst)) } else { None }
772        })
773    else {
774        return (func, fn_subst);
775    };
776
777    (
778        impl_fn,
779        GenericArgs::new_from_iter(
780            interner,
781            impl_subst.iter().chain(fn_subst.iter().skip(trait_params)),
782        ),
783    )
784}

786fn lookup_impl_assoc_item_for_trait_ref<'db>(
787    infcx: &InferCtxt<'db>,
788    trait_ref: TraitRef<'db>,
789    env: Arc<TraitEnvironment<'db>>,
790    name: &Name,
791) -> Option<(AssocItemId, GenericArgs<'db>)> {
792    let (impl_id, impl_subst) = find_matching_impl(infcx, &env, trait_ref)?;
793    let item =
794        impl_id.impl_items(infcx.interner.db).items.iter().find_map(|(n, it)| match *it {
795            AssocItemId::FunctionId(f) => (n == name).then_some(AssocItemId::FunctionId(f)),
796            AssocItemId::ConstId(c) => (n == name).then_some(AssocItemId::ConstId(c)),
797            AssocItemId::TypeAliasId(_) => None,
798        })?;
799    Some((item, impl_subst))
800}

802pub(crate) fn find_matching_impl<'db>(
803    infcx: &InferCtxt<'db>,
804    env: &TraitEnvironment<'db>,
805    trait_ref: TraitRef<'db>,
806) -> Option<(ImplId, GenericArgs<'db>)> {
807    let trait_ref =
808        infcx.at(&ObligationCause::dummy(), env.env).deeply_normalize(trait_ref).ok()?;
809
810    let obligation = Obligation::new(infcx.interner, ObligationCause::dummy(), env.env, trait_ref);
811
812    let selection = infcx.select(&obligation).ok()??;
813
814    // Currently, we use a fulfillment context to completely resolve
815    // all nested obligations. This is because they can inform the
816    // inference of the impl's type parameters.
817    let mut ocx = ObligationCtxt::new(infcx);
818    let impl_source = selection.map(|obligation| ocx.register_obligation(obligation));
819
820    let errors = ocx.evaluate_obligations_error_on_ambiguity();
821    if !errors.is_empty() {
822        return None;
823    }
824
825    let impl_source = infcx.resolve_vars_if_possible(impl_source);
826    if impl_source.has_non_region_infer() {
827        return None;
828    }
829
830    match impl_source {
831        ImplSource::UserDefined(impl_source) => Some((impl_source.impl_def_id, impl_source.args)),
832        ImplSource::Param(_) | ImplSource::Builtin(..) => None,
833    }
834}

836fn is_inherent_impl_coherent<'db>(
837    db: &'db dyn HirDatabase,
838    def_map: &DefMap,
839    impl_id: ImplId,
840    self_ty: Ty<'db>,
841) -> bool {
842    let self_ty = self_ty.kind();
843    let impl_allowed = match self_ty {
844        TyKind::Tuple(_)
845        | TyKind::FnDef(_, _)
846        | TyKind::Array(_, _)
847        | TyKind::Never
848        | TyKind::RawPtr(_, _)
849        | TyKind::Ref(_, _, _)
850        | TyKind::Slice(_)
851        | TyKind::Str
852        | TyKind::Bool
853        | TyKind::Char
854        | TyKind::Int(_)
855        | TyKind::Uint(_)
856        | TyKind::Float(_) => def_map.is_rustc_coherence_is_core(),
857
858        TyKind::Adt(adt_def, _) => adt_def.def_id().0.module(db).krate() == def_map.krate(),
859        TyKind::Dynamic(it, _) => it
860            .principal_def_id()
861            .is_some_and(|trait_id| trait_id.0.module(db).krate() == def_map.krate()),
862
863        _ => true,
864    };
865    impl_allowed || {
866        let rustc_has_incoherent_inherent_impls = match self_ty {
867            TyKind::Tuple(_)
868            | TyKind::FnDef(_, _)
869            | TyKind::Array(_, _)
870            | TyKind::Never
871            | TyKind::RawPtr(_, _)
872            | TyKind::Ref(_, _, _)
873            | TyKind::Slice(_)
874            | TyKind::Str
875            | TyKind::Bool
876            | TyKind::Char
877            | TyKind::Int(_)
878            | TyKind::Uint(_)
879            | TyKind::Float(_) => true,
880
881            TyKind::Adt(adt_def, _) => match adt_def.def_id().0 {
882                hir_def::AdtId::StructId(id) => db
883                    .struct_signature(id)
884                    .flags
885                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
886                hir_def::AdtId::UnionId(id) => db
887                    .union_signature(id)
888                    .flags
889                    .contains(StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
890                hir_def::AdtId::EnumId(it) => db
891                    .enum_signature(it)
892                    .flags
893                    .contains(EnumFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
894            },
895            TyKind::Dynamic(it, _) => it.principal_def_id().is_some_and(|trait_id| {
896                db.trait_signature(trait_id.0)
897                    .flags
898                    .contains(TraitFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS)
899            }),
900
901            _ => false,
902        };
903        let items = impl_id.impl_items(db);
904        rustc_has_incoherent_inherent_impls
905            && !items.items.is_empty()
906            && items.items.iter().all(|&(_, assoc)| match assoc {
907                AssocItemId::FunctionId(it) => {
908                    db.function_signature(it).flags.contains(FnFlags::RUSTC_ALLOW_INCOHERENT_IMPL)
909                }
910                AssocItemId::ConstId(it) => {
911                    db.const_signature(it).flags.contains(ConstFlags::RUSTC_ALLOW_INCOHERENT_IMPL)
912                }
913                AssocItemId::TypeAliasId(it) => db
914                    .type_alias_signature(it)
915                    .flags
916                    .contains(TypeAliasFlags::RUSTC_ALLOW_INCOHERENT_IMPL),
917            })
918    }
919}

921/// Checks whether the impl satisfies the orphan rules.
922///
923/// Given `impl<P1..=Pn> Trait<T1..=Tn> for T0`, an `impl`` is valid only if at least one of the following is true:
924/// - Trait is a local trait
925/// - All of
926///   - At least one of the types `T0..=Tn`` must be a local type. Let `Ti`` be the first such type.
927///   - No uncovered type parameters `P1..=Pn` may appear in `T0..Ti`` (excluding `Ti`)
928pub fn check_orphan_rules<'db>(db: &'db dyn HirDatabase, impl_: ImplId) -> bool {
929    let Some(impl_trait) = db.impl_trait(impl_) else {
930        // not a trait impl
931        return true;
932    };
933
934    let local_crate = impl_.lookup(db).container.krate();
935    let is_local = |tgt_crate| tgt_crate == local_crate;
936
937    let trait_ref = impl_trait.instantiate_identity();
938    let trait_id = trait_ref.def_id.0;
939    if is_local(trait_id.module(db).krate()) {
940        // trait to be implemented is local
941        return true;
942    }
943
944    let unwrap_fundamental = |mut ty: Ty<'db>| {
945        // Unwrap all layers of fundamental types with a loop.
946        loop {
947            match ty.kind() {
948                TyKind::Ref(_, referenced, _) => ty = referenced,
949                TyKind::Adt(adt_def, subs) => {
950                    let AdtId::StructId(s) = adt_def.def_id().0 else {
951                        break ty;
952                    };
953                    let struct_signature = db.struct_signature(s);
954                    if struct_signature.flags.contains(StructFlags::FUNDAMENTAL) {
955                        let next = subs.types().next();
956                        match next {
957                            Some(it) => ty = it,
958                            None => break ty,
959                        }
960                    } else {
961                        break ty;
962                    }
963                }
964                _ => break ty,
965            }
966        }
967    };
968    //   - At least one of the types `T0..=Tn`` must be a local type. Let `Ti`` be the first such type.
969
970    // FIXME: param coverage
971    //   - No uncovered type parameters `P1..=Pn` may appear in `T0..Ti`` (excluding `Ti`)
972    let is_not_orphan = trait_ref.args.types().any(|ty| match unwrap_fundamental(ty).kind() {
973        TyKind::Adt(adt_def, _) => is_local(adt_def.def_id().0.module(db).krate()),
974        TyKind::Error(_) => true,
975        TyKind::Dynamic(it, _) => {
976            it.principal_def_id().is_some_and(|trait_id| is_local(trait_id.0.module(db).krate()))
977        }
978        _ => false,
979    });
980    #[allow(clippy::let_and_return)]
981    is_not_orphan
982}

1029fn iterate_method_candidates_dyn_impl<'db>(
1030    ty: &Canonical<'db, Ty<'db>>,
1031    table: &mut InferenceTable<'db>,
1032    traits_in_scope: &FxHashSet<TraitId>,
1033    visible_from_module: VisibleFromModule,
1034    name: Option<&Name>,
1035    mode: LookupMode,
1036    callback: &mut dyn MethodCandidateCallback,
1037) -> ControlFlow<()> {
1038    let _p = tracing::info_span!(
1039        "iterate_method_candidates_dyn",
1040        ?mode,
1041        ?name,
1042        traits_in_scope_len = traits_in_scope.len()
1043    )
1044    .entered();
1045
1046    match mode {
1047        LookupMode::MethodCall => {
1048            // For method calls, rust first does any number of autoderef, and
1049            // then one autoref (i.e. when the method takes &self or &mut self).
1050            // Note that when we've got a receiver like &S, even if the method
1051            // we find in the end takes &self, we still do the autoderef step
1052            // (just as rustc does an autoderef and then autoref again).
1053
1054            // We have to be careful about the order we're looking at candidates
1055            // in here. Consider the case where we're resolving `it.clone()`
1056            // where `it: &Vec<_>`. This resolves to the clone method with self
1057            // type `Vec<_>`, *not* `&_`. I.e. we need to consider methods where
1058            // the receiver type exactly matches before cases where we have to
1059            // do autoref. But in the autoderef steps, the `&_` self type comes
1060            // up *before* the `Vec<_>` self type.
1061            //
1062            // On the other hand, we don't want to just pick any by-value method
1063            // before any by-autoref method; it's just that we need to consider
1064            // the methods by autoderef order of *receiver types*, not *self
1065            // types*.
1066
1067            table.run_in_snapshot(|table| {
1068                let ty = table.instantiate_canonical(*ty);
1069                let deref_chain = autoderef_method_receiver(table, ty);
1070
1071                deref_chain.into_iter().try_for_each(|(receiver_ty, adj)| {
1072                    iterate_method_candidates_with_autoref(
1073                        table,
1074                        receiver_ty,
1075                        adj,
1076                        traits_in_scope,
1077                        visible_from_module,
1078                        name,
1079                        callback,
1080                    )
1081                })
1082            })
1083        }
1084        LookupMode::Path => {
1085            // No autoderef for path lookups
1086            iterate_method_candidates_for_self_ty(
1087                ty,
1088                table,
1089                traits_in_scope,
1090                visible_from_module,
1091                name,
1092                callback,
1093            )
1094        }
1095    }
1096}

1098#[tracing::instrument(skip_all, fields(name = ?name))]
1099fn iterate_method_candidates_with_autoref<'db>(
1100    table: &mut InferenceTable<'db>,
1101    receiver_ty: Canonical<'db, Ty<'db>>,
1102    first_adjustment: ReceiverAdjustments,
1103    traits_in_scope: &FxHashSet<TraitId>,
1104    visible_from_module: VisibleFromModule,
1105    name: Option<&Name>,
1106    callback: &mut dyn MethodCandidateCallback,
1107) -> ControlFlow<()> {
1108    let interner = table.interner();
1109
1110    let mut iterate_method_candidates_by_receiver = move |receiver_ty, first_adjustment| {
1111        iterate_method_candidates_by_receiver(
1112            table,
1113            receiver_ty,
1114            first_adjustment,
1115            traits_in_scope,
1116            visible_from_module,
1117            name,
1118            callback,
1119        )
1120    };
1121
1122    let mut maybe_reborrowed = first_adjustment.clone();
1123    if let TyKind::Ref(_, _, m) = receiver_ty.value.kind() {
1124        // Prefer reborrow of references to move
1125        maybe_reborrowed.autoref = Some(AutorefOrPtrAdjustment::Autoref(m));
1126        maybe_reborrowed.autoderefs += 1;
1127    }
1128
1129    iterate_method_candidates_by_receiver(receiver_ty, maybe_reborrowed)?;
1130
1131    let refed = Canonical {
1132        max_universe: receiver_ty.max_universe,
1133        variables: receiver_ty.variables,
1134        value: Ty::new_ref(interner, Region::error(interner), receiver_ty.value, Mutability::Not),
1135    };
1136
1137    iterate_method_candidates_by_receiver(
1138        refed,
1139        first_adjustment.with_autoref(AutorefOrPtrAdjustment::Autoref(Mutability::Not)),
1140    )?;
1141
1142    let ref_muted = Canonical {
1143        max_universe: receiver_ty.max_universe,
1144        variables: receiver_ty.variables,
1145        value: Ty::new_ref(interner, Region::error(interner), receiver_ty.value, Mutability::Mut),
1146    };
1147
1148    iterate_method_candidates_by_receiver(
1149        ref_muted,
1150        first_adjustment.with_autoref(AutorefOrPtrAdjustment::Autoref(Mutability::Mut)),
1151    )?;
1152
1153    if let TyKind::RawPtr(ty, Mutability::Mut) = receiver_ty.value.kind() {
1154        let const_ptr_ty = rustc_type_ir::Canonical {
1155            max_universe: rustc_type_ir::UniverseIndex::ZERO,
1156            value: Ty::new_ptr(interner, ty, Mutability::Not),
1157            variables: receiver_ty.variables,
1158        };
1159        iterate_method_candidates_by_receiver(
1160            const_ptr_ty,
1161            first_adjustment.with_autoref(AutorefOrPtrAdjustment::ToConstPtr),
1162        )?;
1163    }
1164
1165    ControlFlow::Continue(())
1166}

1207#[tracing::instrument(skip_all, fields(name = ?name))]
1208fn iterate_method_candidates_by_receiver<'db>(
1209    table: &mut InferenceTable<'db>,
1210    receiver_ty: Canonical<'db, Ty<'db>>,
1211    receiver_adjustments: ReceiverAdjustments,
1212    traits_in_scope: &FxHashSet<TraitId>,
1213    visible_from_module: VisibleFromModule,
1214    name: Option<&Name>,
1215    callback: &mut dyn MethodCandidateCallback,
1216) -> ControlFlow<()> {
1217    let receiver_ty = table.instantiate_canonical(receiver_ty);
1218    // We're looking for methods with *receiver* type receiver_ty. These could
1219    // be found in any of the derefs of receiver_ty, so we have to go through
1220    // that, including raw derefs.
1221    table.run_in_snapshot(|table| {
1222        let mut autoderef = autoderef::Autoderef::new_no_tracking(table, receiver_ty)
1223            .include_raw_pointers()
1224            .use_receiver_trait();
1225        while let Some((self_ty, _)) = autoderef.next() {
1226            iterate_inherent_methods(
1227                self_ty,
1228                autoderef.table,
1229                name,
1230                Some(receiver_ty),
1231                Some(receiver_adjustments.clone()),
1232                visible_from_module,
1233                LookupMode::MethodCall,
1234                &mut |adjustments, item, is_visible| {
1235                    callback.on_inherent_method(adjustments, item, is_visible)
1236                },
1237            )?
1238        }
1239        ControlFlow::Continue(())
1240    })?;
1241    table.run_in_snapshot(|table| {
1242        let mut autoderef = autoderef::Autoderef::new_no_tracking(table, receiver_ty)
1243            .include_raw_pointers()
1244            .use_receiver_trait();
1245        while let Some((self_ty, _)) = autoderef.next() {
1246            if matches!(self_ty.kind(), TyKind::Infer(rustc_type_ir::TyVar(_))) {
1247                // don't try to resolve methods on unknown types
1248                return ControlFlow::Continue(());
1249            }
1250
1251            iterate_trait_method_candidates(
1252                self_ty,
1253                autoderef.table,
1254                traits_in_scope,
1255                name,
1256                Some(receiver_ty),
1257                Some(receiver_adjustments.clone()),
1258                LookupMode::MethodCall,
1259                &mut |adjustments, item, is_visible| {
1260                    callback.on_trait_method(adjustments, item, is_visible)
1261                },
1262            )?
1263        }
1264        ControlFlow::Continue(())
1265    })
1266}

1306#[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1307fn iterate_trait_method_candidates<'db>(
1308    self_ty: Ty<'db>,
1309    table: &mut InferenceTable<'db>,
1310    traits_in_scope: &FxHashSet<TraitId>,
1311    name: Option<&Name>,
1312    receiver_ty: Option<Ty<'db>>,
1313    receiver_adjustments: Option<ReceiverAdjustments>,
1314    mode: LookupMode,
1315    callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1316) -> ControlFlow<()> {
1317    let db = table.db;
1318
1319    let canonical_self_ty = table.canonicalize(self_ty);
1320    let krate = table.trait_env.krate;
1321
1322    'traits: for &t in traits_in_scope {
1323        let data = db.trait_signature(t);
1324
1325        // Traits annotated with `#[rustc_skip_during_method_dispatch]` are skipped during
1326        // method resolution, if the receiver is an array, and we're compiling for editions before
1327        // 2021.
1328        // This is to make `[a].into_iter()` not break code with the new `IntoIterator` impl for
1329        // arrays.
1330        if data.flags.contains(TraitFlags::SKIP_ARRAY_DURING_METHOD_DISPATCH)
1331            && matches!(self_ty.kind(), TyKind::Array(..))
1332        {
1333            // FIXME: this should really be using the edition of the method name's span, in case it
1334            // comes from a macro
1335            if !krate.data(db).edition.at_least_2021() {
1336                continue;
1337            }
1338        }
1339        if data.flags.contains(TraitFlags::SKIP_BOXED_SLICE_DURING_METHOD_DISPATCH)
1340            && matches!(
1341                self_ty.kind(), TyKind::Adt(adt_def, subst)
1342                if is_box(table.db, adt_def.def_id().0)
1343                    && matches!(subst.type_at(0).kind(), TyKind::Slice(..))
1344            )
1345        {
1346            // FIXME: this should really be using the edition of the method name's span, in case it
1347            // comes from a macro
1348            if !krate.data(db).edition.at_least_2024() {
1349                continue;
1350            }
1351        }
1352
1353        // we'll be lazy about checking whether the type implements the
1354        // trait, but if we find out it doesn't, we'll skip the rest of the
1355        // iteration
1356        let mut known_implemented = false;
1357        for &(_, item) in t.trait_items(db).items.iter() {
1358            // Don't pass a `visible_from_module` down to `is_valid_candidate`,
1359            // since only inherent methods should be included into visibility checking.
1360            let visible = match is_valid_trait_method_candidate(
1361                table,
1362                t,
1363                name,
1364                receiver_ty,
1365                item,
1366                self_ty,
1367                mode,
1368            ) {
1369                IsValidCandidate::Yes => true,
1370                IsValidCandidate::NotVisible => false,
1371                IsValidCandidate::No => continue,
1372            };
1373            if !known_implemented {
1374                let goal = generic_implements_goal_ns(table, t, canonical_self_ty);
1375                if next_trait_solve_canonical_in_ctxt(&table.infer_ctxt, goal).no_solution() {
1376                    continue 'traits;
1377                }
1378            }
1379            known_implemented = true;
1380            callback(receiver_adjustments.clone().unwrap_or_default(), item, visible)?;
1381        }
1382    }
1383    ControlFlow::Continue(())
1384}

1386#[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1387fn iterate_inherent_methods<'db>(
1388    self_ty: Ty<'db>,
1389    table: &mut InferenceTable<'db>,
1390    name: Option<&Name>,
1391    receiver_ty: Option<Ty<'db>>,
1392    receiver_adjustments: Option<ReceiverAdjustments>,
1393    visible_from_module: VisibleFromModule,
1394    mode: LookupMode,
1395    callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1396) -> ControlFlow<()> {
1397    let db = table.db;
1398    let env = table.trait_env.clone();
1399
1400    // For trait object types and placeholder types with trait bounds, the methods of the trait and
1401    // its super traits are considered inherent methods. This matters because these methods have
1402    // higher priority than the other traits' methods, which would be considered in
1403    // `iterate_trait_method_candidates()` only after this function.
1404    match self_ty.kind() {
1405        TyKind::Param(_) => {
1406            let env = table.trait_env.clone();
1407            let traits =
1408                env.traits_in_scope_from_clauses(self_ty).flat_map(|t| all_super_traits(db, t));
1409            iterate_inherent_trait_methods(
1410                self_ty,
1411                table,
1412                name,
1413                receiver_ty,
1414                receiver_adjustments.clone(),
1415                callback,
1416                traits,
1417                mode,
1418            )?;
1419        }
1420        TyKind::Dynamic(bounds, _) => {
1421            if let Some(principal_trait) = bounds.principal_def_id() {
1422                let traits = all_super_traits(db, principal_trait.0);
1423                iterate_inherent_trait_methods(
1424                    self_ty,
1425                    table,
1426                    name,
1427                    receiver_ty,
1428                    receiver_adjustments.clone(),
1429                    callback,
1430                    traits.into_iter(),
1431                    mode,
1432                )?;
1433            }
1434        }
1435        _ => {}
1436    }
1437
1438    let def_crates = match def_crates(db, self_ty, env.krate) {
1439        Some(k) => k,
1440        None => return ControlFlow::Continue(()),
1441    };
1442
1443    let (module, mut block) = match visible_from_module {
1444        VisibleFromModule::Filter(module) => (Some(module), module.containing_block()),
1445        VisibleFromModule::IncludeBlock(block) => (None, Some(block)),
1446        VisibleFromModule::None => (None, None),
1447    };
1448
1449    while let Some(block_id) = block {
1450        if let Some(impls) = db.inherent_impls_in_block(block_id) {
1451            impls_for_self_ty(
1452                &impls,
1453                self_ty,
1454                table,
1455                name,
1456                receiver_ty,
1457                receiver_adjustments.clone(),
1458                module,
1459                callback,
1460            )?;
1461        }
1462
1463        block = block_def_map(db, block_id).parent().and_then(|module| module.containing_block());
1464    }
1465
1466    for krate in def_crates {
1467        let impls = db.inherent_impls_in_crate(krate);
1468        impls_for_self_ty(
1469            &impls,
1470            self_ty,
1471            table,
1472            name,
1473            receiver_ty,
1474            receiver_adjustments.clone(),
1475            module,
1476            callback,
1477        )?;
1478    }
1479    return ControlFlow::Continue(());
1480
1481    #[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1482    fn iterate_inherent_trait_methods<'db>(
1483        self_ty: Ty<'db>,
1484        table: &mut InferenceTable<'db>,
1485        name: Option<&Name>,
1486        receiver_ty: Option<Ty<'db>>,
1487        receiver_adjustments: Option<ReceiverAdjustments>,
1488        callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1489        traits: impl Iterator<Item = TraitId>,
1490        mode: LookupMode,
1491    ) -> ControlFlow<()> {
1492        let db = table.db;
1493        for t in traits {
1494            let data = t.trait_items(db);
1495            for &(_, item) in data.items.iter() {
1496                // We don't pass `visible_from_module` as all trait items should be visible.
1497                let visible = match is_valid_trait_method_candidate(
1498                    table,
1499                    t,
1500                    name,
1501                    receiver_ty,
1502                    item,
1503                    self_ty,
1504                    mode,
1505                ) {
1506                    IsValidCandidate::Yes => true,
1507                    IsValidCandidate::NotVisible => false,
1508                    IsValidCandidate::No => continue,
1509                };
1510                callback(receiver_adjustments.clone().unwrap_or_default(), item, visible)?;
1511            }
1512        }
1513        ControlFlow::Continue(())
1514    }
1515
1516    #[tracing::instrument(skip_all, fields(name = ?name, visible_from_module, receiver_ty))]
1517    fn impls_for_self_ty<'db>(
1518        impls: &InherentImpls,
1519        self_ty: Ty<'db>,
1520        table: &mut InferenceTable<'db>,
1521        name: Option<&Name>,
1522        receiver_ty: Option<Ty<'db>>,
1523        receiver_adjustments: Option<ReceiverAdjustments>,
1524        visible_from_module: Option<ModuleId>,
1525        callback: &mut dyn FnMut(ReceiverAdjustments, AssocItemId, bool) -> ControlFlow<()>,
1526    ) -> ControlFlow<()> {
1527        for &impl_id in impls.for_self_ty(self_ty) {
1528            for &(ref item_name, item) in impl_id.impl_items(table.db).items.iter() {
1529                let visible = match is_valid_impl_method_candidate(
1530                    table,
1531                    self_ty,
1532                    receiver_ty,
1533                    visible_from_module,
1534                    name,
1535                    impl_id,
1536                    item,
1537                    item_name,
1538                ) {
1539                    IsValidCandidate::Yes => true,
1540                    IsValidCandidate::NotVisible => false,
1541                    IsValidCandidate::No => continue,
1542                };
1543                callback(receiver_adjustments.clone().unwrap_or_default(), item, visible)?;
1544            }
1545        }
1546        ControlFlow::Continue(())
1547    }
1548}

1583#[tracing::instrument(skip_all, fields(name))]
1584fn is_valid_impl_method_candidate<'db>(
1585    table: &mut InferenceTable<'db>,
1586    self_ty: Ty<'db>,
1587    receiver_ty: Option<Ty<'db>>,
1588    visible_from_module: Option<ModuleId>,
1589    name: Option<&Name>,
1590    impl_id: ImplId,
1591    item: AssocItemId,
1592    item_name: &Name,
1593) -> IsValidCandidate {
1594    match item {
1595        AssocItemId::FunctionId(f) => is_valid_impl_fn_candidate(
1596            table,
1597            impl_id,
1598            f,
1599            name,
1600            receiver_ty,
1601            self_ty,
1602            visible_from_module,
1603            item_name,
1604        ),
1605        AssocItemId::ConstId(c) => {
1606            let db = table.db;
1607            check_that!(receiver_ty.is_none());
1608            check_that!(name.is_none_or(|n| n == item_name));
1609
1610            if let Some(from_module) = visible_from_module
1611                && !db.assoc_visibility(c.into()).is_visible_from(db, from_module)
1612            {
1613                cov_mark::hit!(const_candidate_not_visible);
1614                return IsValidCandidate::NotVisible;
1615            }
1616            let self_ty_matches = table.run_in_snapshot(|table| {
1617                let impl_args = table.fresh_args_for_item(impl_id.into());
1618                let expected_self_ty =
1619                    db.impl_self_ty(impl_id).instantiate(table.interner(), impl_args);
1620                table.unify(expected_self_ty, self_ty)
1621            });
1622            if !self_ty_matches {
1623                cov_mark::hit!(const_candidate_self_type_mismatch);
1624                return IsValidCandidate::No;
1625            }
1626            IsValidCandidate::Yes
1627        }
1628        _ => IsValidCandidate::No,
1629    }
1630}

1632/// Checks whether a given `AssocItemId` is applicable for `receiver_ty`.
1633#[tracing::instrument(skip_all, fields(name))]
1634fn is_valid_trait_method_candidate<'db>(
1635    table: &mut InferenceTable<'db>,
1636    trait_id: TraitId,
1637    name: Option<&Name>,
1638    receiver_ty: Option<Ty<'db>>,
1639    item: AssocItemId,
1640    self_ty: Ty<'db>,
1641    mode: LookupMode,
1642) -> IsValidCandidate {
1643    let db = table.db;
1644    match item {
1645        AssocItemId::FunctionId(fn_id) => {
1646            let data = db.function_signature(fn_id);
1647
1648            check_that!(name.is_none_or(|n| n == &data.name));
1649
1650            table.run_in_snapshot(|table| {
1651                let impl_subst = table.fresh_args_for_item(trait_id.into());
1652                let expect_self_ty = impl_subst.type_at(0);
1653
1654                check_that!(table.unify(expect_self_ty, self_ty));
1655
1656                if let Some(receiver_ty) = receiver_ty {
1657                    check_that!(data.has_self_param());
1658
1659                    let args = table.fill_rest_fresh_args(fn_id.into(), impl_subst);
1660
1661                    let sig = db.callable_item_signature(fn_id.into());
1662                    let expected_receiver = sig
1663                        .map_bound(|s| s.skip_binder().inputs_and_output.as_slice()[0])
1664                        .instantiate(table.interner(), args);
1665
1666                    // FIXME: Clean up this mess with some context struct like rustc's `ProbeContext`
1667                    let variance = match mode {
1668                        LookupMode::MethodCall => rustc_type_ir::Variance::Covariant,
1669                        LookupMode::Path => rustc_type_ir::Variance::Invariant,
1670                    };
1671                    let res = table
1672                        .infer_ctxt
1673                        .at(&ObligationCause::dummy(), table.trait_env.env)
1674                        .relate(expected_receiver, variance, receiver_ty);
1675                    let Ok(infer_ok) = res else {
1676                        return IsValidCandidate::No;
1677                    };
1678
1679                    if !infer_ok.obligations.is_empty() {
1680                        let mut ctxt = ObligationCtxt::new(&table.infer_ctxt);
1681                        ctxt.register_obligations(infer_ok.into_obligations());
1682                        // FIXME: Are we doing this correctly? Probably better to follow rustc more closely.
1683                        check_that!(ctxt.try_evaluate_obligations().is_empty());
1684                    }
1685
1686                    check_that!(table.unify(receiver_ty, expected_receiver));
1687                }
1688
1689                IsValidCandidate::Yes
1690            })
1691        }
1692        AssocItemId::ConstId(c) => {
1693            check_that!(receiver_ty.is_none());
1694            check_that!(name.is_none_or(|n| db.const_signature(c).name.as_ref() == Some(n)));
1695
1696            IsValidCandidate::Yes
1697        }
1698        _ => IsValidCandidate::No,
1699    }
1700}

1702#[tracing::instrument(skip_all, fields(name))]
1703fn is_valid_impl_fn_candidate<'db>(
1704    table: &mut InferenceTable<'db>,
1705    impl_id: ImplId,
1706    fn_id: FunctionId,
1707    name: Option<&Name>,
1708    receiver_ty: Option<Ty<'db>>,
1709    self_ty: Ty<'db>,
1710    visible_from_module: Option<ModuleId>,
1711    item_name: &Name,
1712) -> IsValidCandidate {
1713    check_that!(name.is_none_or(|n| n == item_name));
1714
1715    let db = table.db;
1716    let data = db.function_signature(fn_id);
1717
1718    if let Some(from_module) = visible_from_module
1719        && !db.assoc_visibility(fn_id.into()).is_visible_from(db, from_module)
1720    {
1721        cov_mark::hit!(autoderef_candidate_not_visible);
1722        return IsValidCandidate::NotVisible;
1723    }
1724    table.run_in_snapshot(|table| {
1725        let _p = tracing::info_span!("subst_for_def").entered();
1726        let impl_subst = table.infer_ctxt.fresh_args_for_item(impl_id.into());
1727        let expect_self_ty = db.impl_self_ty(impl_id).instantiate(table.interner(), &impl_subst);
1728
1729        check_that!(table.unify(expect_self_ty, self_ty));
1730
1731        if let Some(receiver_ty) = receiver_ty {
1732            let _p = tracing::info_span!("check_receiver_ty").entered();
1733            check_that!(data.has_self_param());
1734
1735            let args = table.infer_ctxt.fresh_args_for_item(fn_id.into());
1736
1737            let sig = db.callable_item_signature(fn_id.into());
1738            let expected_receiver = sig
1739                .map_bound(|s| s.skip_binder().inputs_and_output.as_slice()[0])
1740                .instantiate(table.interner(), args);
1741
1742            check_that!(table.unify(receiver_ty, expected_receiver));
1743        }
1744
1745        // We need to consider the bounds on the impl to distinguish functions of the same name
1746        // for a type.
1747        let predicates = db.generic_predicates(impl_id.into());
1748        let Some(predicates) = predicates.instantiate(table.interner(), impl_subst) else {
1749            return IsValidCandidate::Yes;
1750        };
1751
1752        let mut ctxt = ObligationCtxt::new(&table.infer_ctxt);
1753
1754        ctxt.register_obligations(predicates.into_iter().map(|p| {
1755            PredicateObligation::new(
1756                table.interner(),
1757                ObligationCause::new(),
1758                table.trait_env.env,
1759                p.0,
1760            )
1761        }));
1762
1763        if ctxt.try_evaluate_obligations().is_empty() {
1764            IsValidCandidate::Yes
1765        } else {
1766            IsValidCandidate::No
1767        }
1768    })
1769}

1771/// This creates Substs for a trait with the given Self type and type variables
1772/// for all other parameters, to query the trait solver with it.
1773#[tracing::instrument(skip_all)]
1774fn generic_implements_goal_ns<'db>(
1775    table: &mut InferenceTable<'db>,
1776    trait_: TraitId,
1777    self_ty: Canonical<'db, Ty<'db>>,
1778) -> Canonical<'db, Goal<'db, Predicate<'db>>> {
1779    let args = table.infer_ctxt.fresh_args_for_item(SolverDefId::TraitId(trait_));
1780    let self_ty = table.instantiate_canonical(self_ty);
1781    let trait_ref =
1782        rustc_type_ir::TraitRef::new_from_args(table.infer_ctxt.interner, trait_.into(), args)
1783            .with_replaced_self_ty(table.infer_ctxt.interner, self_ty);
1784    let goal = Goal::new(table.infer_ctxt.interner, table.trait_env.env, trait_ref);
1785
1786    table.canonicalize(goal)
1787}

1789fn autoderef_method_receiver<'db>(
1790    table: &mut InferenceTable<'db>,
1791    ty: Ty<'db>,
1792) -> Vec<(Canonical<'db, Ty<'db>>, ReceiverAdjustments)> {
1793    let interner = table.interner();
1794    let mut deref_chain = Vec::new();
1795    let mut autoderef = autoderef::Autoderef::new_no_tracking(table, ty).use_receiver_trait();
1796    while let Some((ty, derefs)) = autoderef.next() {
1797        deref_chain.push((
1798            autoderef.table.canonicalize(ty),
1799            ReceiverAdjustments { autoref: None, autoderefs: derefs, unsize_array: false },
1800        ));
1801    }
1802    // As a last step, we can do array unsizing (that's the only unsizing that rustc does for method receivers!)
1803    if let Some((rustc_type_ir::Array(parameters, _), variables, max_universe, adj)) =
1804        deref_chain.last().map(|d| (d.0.value.kind(), d.0.variables, d.0.max_universe, d.1.clone()))
1805    {
1806        let unsized_ty = Ty::new_slice(interner, parameters);
1807        deref_chain.push((
1808            Canonical { max_universe, value: unsized_ty, variables },
1809            ReceiverAdjustments { unsize_array: true, ..adj.clone() },
1810        ));
1811    }
1812    deref_chain
1813}
