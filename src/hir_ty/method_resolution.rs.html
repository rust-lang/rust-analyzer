115#[derive(Debug)]
116pub enum MethodError<'db> {
117    /// Did not find an applicable method.
118    NoMatch,
119
120    /// Multiple methods might apply.
121    Ambiguity(Vec<CandidateSource>),
122
123    /// Found an applicable method, but it is not visible.
124    PrivateMatch(Pick<'db>),
125
126    /// Found a `Self: Sized` bound where `Self` is a trait object.
127    IllegalSizedBound { candidates: Vec<FunctionId>, needs_mut: bool },
128
129    /// Error has already been emitted, no need to emit another one.
130    ErrorReported,
131}

141impl<'a, 'db> InferenceContext<'a, 'db> {
142    /// Performs method lookup. If lookup is successful, it will return the callee
143    /// and store an appropriate adjustment for the self-expr. In some cases it may
144    /// report an error (e.g., invoking the `drop` method).
145    #[instrument(level = "debug", skip(self))]
146    pub(crate) fn lookup_method_including_private(
147        &mut self,
148        self_ty: Ty<'db>,
149        name: Name,
150        generic_args: Option<&HirGenericArgs>,
151        receiver: ExprId,
152        call_expr: ExprId,
153    ) -> Result<(MethodCallee<'db>, bool), MethodError<'db>> {
154        let (pick, is_visible) = match self.lookup_probe(name, self_ty) {
155            Ok(it) => (it, true),
156            Err(MethodError::PrivateMatch(it)) => {
157                // FIXME: Report error.
158                (it, false)
159            }
160            Err(err) => return Err(err),
161        };
162
163        let result = self.confirm_method(&pick, self_ty, call_expr, generic_args);
164        debug!("result = {:?}", result);
165
166        if result.illegal_sized_bound {
167            // FIXME: Report an error.
168        }
169
170        self.write_expr_adj(receiver, result.adjustments);
171        self.write_method_resolution(call_expr, result.callee.def_id, result.callee.args);
172
173        Ok((result.callee, is_visible))
174    }
175
176    #[instrument(level = "debug", skip(self))]
177    pub(crate) fn lookup_probe(
178        &self,
179        method_name: Name,
180        self_ty: Ty<'db>,
181    ) -> probe::PickResult<'db> {
182        self.with_method_resolution(|ctx| {
183            let pick = ctx.probe_for_name(probe::Mode::MethodCall, method_name, self_ty)?;
184            Ok(pick)
185        })
186    }
187
188    pub(crate) fn with_method_resolution<R>(
189        &self,
190        f: impl FnOnce(&MethodResolutionContext<'_, 'db>) -> R,
191    ) -> R {
192        let traits_in_scope = self.get_traits_in_scope();
193        let traits_in_scope = match &traits_in_scope {
194            Either::Left(it) => it,
195            Either::Right(it) => *it,
196        };
197        let ctx = MethodResolutionContext {
198            infcx: &self.table.infer_ctxt,
199            resolver: &self.resolver,
200            param_env: self.table.param_env,
201            traits_in_scope,
202            edition: self.edition,
203            unstable_features: &self.unstable_features,
204        };
205        f(&ctx)
206    }
207}

209/// Used by `FnCtxt::lookup_method_for_operator` with `-Znext-solver`.
210///
211/// With `AsRigid` we error on `impl Opaque: NotInItemBounds` while
212/// `AsInfer` just treats it as ambiguous and succeeds. This is necessary
213/// as we want `FnCtxt::check_expr_call` to treat not-yet-defined opaque
214/// types as rigid to support `impl Deref<Target = impl FnOnce()>` and
215/// `Box<impl FnOnce()>`.
216///
217/// We only want to treat opaque types as rigid if we need to eagerly choose
218/// between multiple candidates. We otherwise treat them as ordinary inference
219/// variable to avoid rejecting otherwise correct code.
220#[derive(Debug)]
221#[expect(dead_code)]
222pub(super) enum TreatNotYetDefinedOpaques {
223    AsInfer,
224    AsRigid,
225}

227impl<'db> InferenceTable<'db> {
228    /// `lookup_method_in_trait` is used for overloaded operators.
229    /// It does a very narrow slice of what the normal probe/confirm path does.
230    /// In particular, it doesn't really do any probing: it simply constructs
231    /// an obligation for a particular trait with the given self type and checks
232    /// whether that trait is implemented.
233    #[instrument(level = "debug", skip(self))]
234    pub(super) fn lookup_method_for_operator(
235        &self,
236        cause: ObligationCause,
237        method_name: Symbol,
238        trait_def_id: TraitId,
239        self_ty: Ty<'db>,
240        opt_rhs_ty: Option<Ty<'db>>,
241        treat_opaques: TreatNotYetDefinedOpaques,
242    ) -> Option<InferOk<'db, MethodCallee<'db>>> {
243        // Construct a trait-reference `self_ty : Trait<input_tys>`
244        let args = GenericArgs::for_item(
245            self.interner(),
246            trait_def_id.into(),
247            |param_idx, param_id, _| match param_id {
248                GenericParamId::LifetimeParamId(_) | GenericParamId::ConstParamId(_) => {
249                    unreachable!("did not expect operator trait to have lifetime/const")
250                }
251                GenericParamId::TypeParamId(_) => {
252                    if param_idx == 0 {
253                        self_ty.into()
254                    } else if let Some(rhs_ty) = opt_rhs_ty {
255                        assert_eq!(param_idx, 1, "did not expect >1 param on operator trait");
256                        rhs_ty.into()
257                    } else {
258                        // FIXME: We should stop passing `None` for the failure case
259                        // when probing for call exprs. I.e. `opt_rhs_ty` should always
260                        // be set when it needs to be.
261                        self.next_var_for_param(param_id)
262                    }
263                }
264            },
265        );
266
267        let obligation = Obligation::new(
268            self.interner(),
269            cause,
270            self.param_env,
271            TraitRef::new_from_args(self.interner(), trait_def_id.into(), args),
272        );
273
274        // Now we want to know if this can be matched
275        let matches_trait = match treat_opaques {
276            TreatNotYetDefinedOpaques::AsInfer => self.infer_ctxt.predicate_may_hold(&obligation),
277            TreatNotYetDefinedOpaques::AsRigid => {
278                self.infer_ctxt.predicate_may_hold_opaque_types_jank(&obligation)
279            }
280        };
281
282        if !matches_trait {
283            debug!("--> Cannot match obligation");
284            // Cannot be matched, no such method resolution is possible.
285            return None;
286        }
287
288        // Trait must have a method named `m_name` and it should not have
289        // type parameters or early-bound regions.
290        let interner = self.interner();
291        // We use `Ident::with_dummy_span` since no built-in operator methods have
292        // any macro-specific hygiene, so the span's context doesn't really matter.
293        let Some(method_item) =
294            trait_def_id.trait_items(self.db).method_by_name(&Name::new_symbol_root(method_name))
295        else {
296            panic!("expected associated item for operator trait")
297        };
298
299        let def_id = method_item;
300
301        debug!("lookup_in_trait_adjusted: method_item={:?}", method_item);
302        let mut obligations = PredicateObligations::new();
303
304        // Instantiate late-bound regions and instantiate the trait
305        // parameters into the method type to get the actual method type.
306        //
307        // N.B., instantiate late-bound regions before normalizing the
308        // function signature so that normalization does not need to deal
309        // with bound regions.
310        let fn_sig =
311            self.db.callable_item_signature(method_item.into()).instantiate(interner, args);
312        let fn_sig = self
313            .infer_ctxt
314            .instantiate_binder_with_fresh_vars(BoundRegionConversionTime::FnCall, fn_sig);
315
316        // Register obligations for the parameters. This will include the
317        // `Self` parameter, which in turn has a bound of the main trait,
318        // so this also effectively registers `obligation` as well. (We
319        // used to register `obligation` explicitly, but that resulted in
320        // double error messages being reported.)
321        //
322        // Note that as the method comes from a trait, it should not have
323        // any late-bound regions appearing in its bounds.
324        let bounds = GenericPredicates::query_all(self.db, method_item.into());
325        let bounds = clauses_as_obligations(
326            bounds.iter_instantiated_copied(interner, args.as_slice()),
327            ObligationCause::new(),
328            self.param_env,
329        );
330
331        obligations.extend(bounds);
332
333        // Also add an obligation for the method type being well-formed.
334        debug!(
335            "lookup_method_in_trait: matched method fn_sig={:?} obligation={:?}",
336            fn_sig, obligation
337        );
338        for ty in fn_sig.inputs_and_output {
339            obligations.push(Obligation::new(
340                interner,
341                obligation.cause.clone(),
342                self.param_env,
343                Binder::dummy(PredicateKind::Clause(ClauseKind::WellFormed(ty.into()))),
344            ));
345        }
346
347        let callee = MethodCallee { def_id, args, sig: fn_sig };
348        debug!("callee = {:?}", callee);
349
350        Some(InferOk { obligations, value: callee })
351    }
352}

354pub fn lookup_impl_const<'db>(
355    infcx: &InferCtxt<'db>,
356    env: ParamEnv<'db>,
357    const_id: ConstId,
358    subs: GenericArgs<'db>,
359) -> (ConstId, GenericArgs<'db>) {
360    let interner = infcx.interner;
361    let db = interner.db;
362
363    let trait_id = match const_id.loc(db).container {
364        ItemContainerId::TraitId(id) => id,
365        _ => return (const_id, subs),
366    };
367    let trait_ref = TraitRef::new_from_args(interner, trait_id.into(), subs);
368
369    let const_signature = db.const_signature(const_id);
370    let name = match const_signature.name.as_ref() {
371        Some(name) => name,
372        None => return (const_id, subs),
373    };
374
375    lookup_impl_assoc_item_for_trait_ref(infcx, trait_ref, env, name)
376        .and_then(|assoc| {
377            if let (Either::Left(AssocItemId::ConstId(id)), s) = assoc {
378                Some((id, s))
379            } else {
380                None
381            }
382        })
383        .unwrap_or((const_id, subs))
384}

386/// Checks if the self parameter of `Trait` method is the `dyn Trait` and we should
387/// call the method using the vtable.
388pub fn is_dyn_method<'db>(
389    interner: DbInterner<'db>,
390    _env: ParamEnv<'db>,
391    func: FunctionId,
392    fn_subst: GenericArgs<'db>,
393) -> Option<usize> {
394    let db = interner.db;
395
396    let ItemContainerId::TraitId(trait_id) = func.loc(db).container else {
397        return None;
398    };
399    let trait_params = db.generic_params(trait_id.into()).len();
400    let fn_params = fn_subst.len() - trait_params;
401    let trait_ref = TraitRef::new_from_args(
402        interner,
403        trait_id.into(),
404        GenericArgs::new_from_slice(&fn_subst[..trait_params]),
405    );
406    let self_ty = trait_ref.self_ty();
407    if let TyKind::Dynamic(d, _) = self_ty.kind() {
408        // rustc doesn't accept `impl Foo<2> for dyn Foo<5>`, so if the trait id is equal, no matter
409        // what the generics are, we are sure that the method is come from the vtable.
410        let is_my_trait_in_bounds = d
411            .principal_def_id()
412            .is_some_and(|trait_| all_super_traits(db, trait_.0).contains(&trait_id));
413        if is_my_trait_in_bounds {
414            return Some(fn_params);
415        }
416    }
417    None
418}

420/// Looks up the impl method that actually runs for the trait method `func`.
421///
422/// Returns `func` if it's not a method defined in a trait or the lookup failed.
423pub(crate) fn lookup_impl_method_query<'db>(
424    db: &'db dyn HirDatabase,
425    env: ParamEnvAndCrate<'db>,
426    func: FunctionId,
427    fn_subst: GenericArgs<'db>,
428) -> (Either<FunctionId, (BuiltinDeriveImplId, BuiltinDeriveImplMethod)>, GenericArgs<'db>) {
429    let interner = DbInterner::new_with(db, env.krate);
430    let infcx = interner.infer_ctxt().build(TypingMode::PostAnalysis);
431
432    let ItemContainerId::TraitId(trait_id) = func.loc(db).container else {
433        return (Either::Left(func), fn_subst);
434    };
435    let trait_params = db.generic_params(trait_id.into()).len();
436    let trait_ref = TraitRef::new_from_args(
437        interner,
438        trait_id.into(),
439        GenericArgs::new_from_slice(&fn_subst[..trait_params]),
440    );
441
442    let name = &db.function_signature(func).name;
443    let Some((impl_fn, impl_subst)) =
444        lookup_impl_assoc_item_for_trait_ref(&infcx, trait_ref, env.param_env, name).and_then(
445            |(assoc, impl_args)| {
446                let assoc = match assoc {
447                    Either::Left(AssocItemId::FunctionId(id)) => Either::Left(id),
448                    Either::Right(it) => Either::Right(it),
449                    _ => return None,
450                };
451                Some((assoc, impl_args))
452            },
453        )
454    else {
455        return (Either::Left(func), fn_subst);
456    };
457
458    (
459        impl_fn,
460        GenericArgs::new_from_iter(
461            interner,
462            impl_subst.iter().chain(fn_subst.iter().skip(trait_params)),
463        ),
464    )
465}

467fn lookup_impl_assoc_item_for_trait_ref<'db>(
468    infcx: &InferCtxt<'db>,
469    trait_ref: TraitRef<'db>,
470    env: ParamEnv<'db>,
471    name: &Name,
472) -> Option<(Either<AssocItemId, (BuiltinDeriveImplId, BuiltinDeriveImplMethod)>, GenericArgs<'db>)>
473{
474    let (impl_id, impl_subst) = find_matching_impl(infcx, env, trait_ref)?;
475    let impl_id = match impl_id {
476        AnyImplId::ImplId(it) => it,
477        AnyImplId::BuiltinDeriveImplId(impl_) => {
478            return impl_
479                .loc(infcx.interner.db)
480                .trait_
481                .get_method(name.symbol())
482                .map(|method| (Either::Right((impl_, method)), impl_subst));
483        }
484    };
485    let item =
486        impl_id.impl_items(infcx.interner.db).items.iter().find_map(|(n, it)| match *it {
487            AssocItemId::FunctionId(f) => (n == name).then_some(AssocItemId::FunctionId(f)),
488            AssocItemId::ConstId(c) => (n == name).then_some(AssocItemId::ConstId(c)),
489            AssocItemId::TypeAliasId(_) => None,
490        })?;
491    Some((Either::Left(item), impl_subst))
492}

494pub(crate) fn find_matching_impl<'db>(
495    infcx: &InferCtxt<'db>,
496    env: ParamEnv<'db>,
497    trait_ref: TraitRef<'db>,
498) -> Option<(AnyImplId, GenericArgs<'db>)> {
499    let trait_ref = infcx.at(&ObligationCause::dummy(), env).deeply_normalize(trait_ref).ok()?;
500
501    let obligation = Obligation::new(infcx.interner, ObligationCause::dummy(), env, trait_ref);
502
503    let selection = infcx.select(&obligation).ok()??;
504
505    // Currently, we use a fulfillment context to completely resolve
506    // all nested obligations. This is because they can inform the
507    // inference of the impl's type parameters.
508    let mut ocx = ObligationCtxt::new(infcx);
509    let impl_source = selection.map(|obligation| ocx.register_obligation(obligation));
510
511    let errors = ocx.evaluate_obligations_error_on_ambiguity();
512    if !errors.is_empty() {
513        return None;
514    }
515
516    let impl_source = infcx.resolve_vars_if_possible(impl_source);
517    if impl_source.has_non_region_infer() {
518        return None;
519    }
520
521    match impl_source {
522        ImplSource::UserDefined(impl_source) => Some((impl_source.impl_def_id, impl_source.args)),
523        ImplSource::Param(_) | ImplSource::Builtin(..) => None,
524    }
525}

535pub fn with_incoherent_inherent_impls(
536    db: &dyn HirDatabase,
537    krate: Crate,
538    self_ty: &SimplifiedType,
539    mut callback: impl FnMut(&[ImplId]),
540) {
541    let has_incoherent_impls = match self_ty.def() {
542        Some(def_id) => match def_id.try_into() {
543            Ok(def_id) => AttrFlags::query(db, def_id)
544                .contains(AttrFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS),
545            Err(()) => true,
546        },
547        _ => true,
548    };
549    if !has_incoherent_impls {
550        return;
551    }
552    let _p = tracing::info_span!("incoherent_inherent_impls").entered();
553    let crates = crates_containing_incoherent_inherent_impls(db, krate);
554    for &krate in crates {
555        let impls = InherentImpls::for_crate(db, krate);
556        callback(impls.for_self_ty(self_ty));
557    }
558}

595impl InherentImpls {
596    fn collect_def_map(db: &dyn HirDatabase, def_map: &DefMap) -> Self {
597        let mut map = FxHashMap::default();
598        collect(db, def_map, &mut map);
599        let mut map = map
600            .into_iter()
601            .map(|(self_ty, impls)| (self_ty, impls.into_boxed_slice()))
602            .collect::<FxHashMap<_, _>>();
603        map.shrink_to_fit();
604        return Self { map };
605
606        fn collect(
607            db: &dyn HirDatabase,
608            def_map: &DefMap,
609            map: &mut FxHashMap<SimplifiedType, Vec<ImplId>>,
610        ) {
611            for (_module_id, module_data) in def_map.modules() {
612                for impl_id in module_data.scope.inherent_impls() {
613                    let interner = DbInterner::new_no_crate(db);
614                    let self_ty = db.impl_self_ty(impl_id);
615                    let self_ty = self_ty.instantiate_identity();
616                    if let Some(self_ty) =
617                        simplify_type(interner, self_ty, TreatParams::InstantiateWithInfer)
618                    {
619                        map.entry(self_ty).or_default().push(impl_id);
620                    }
621                }
622
623                // To better support custom derives, collect impls in all unnamed const items.
624                // const _: () = { ... };
625                for konst in module_data.scope.unnamed_consts() {
626                    let body = db.body(konst.into());
627                    for (_, block_def_map) in body.blocks(db) {
628                        collect(db, block_def_map, map);
629                    }
630                }
631            }
632        }
633    }
634
635    pub fn for_self_ty(&self, self_ty: &SimplifiedType) -> &[ImplId] {
636        self.map.get(self_ty).map(|it| &**it).unwrap_or_default()
637    }
638
639    pub fn for_each_crate_and_block(
640        db: &dyn HirDatabase,
641        krate: Crate,
642        block: Option<BlockId>,
643        for_each: &mut dyn FnMut(&InherentImpls),
644    ) {
645        let blocks = std::iter::successors(block, |block| block.loc(db).module.block(db));
646        blocks.filter_map(|block| Self::for_block(db, block).as_deref()).for_each(&mut *for_each);
647        for_each(Self::for_crate(db, krate));
648    }
649}

651#[derive(Debug, PartialEq)]
652struct OneTraitImpls {
653    non_blanket_impls: FxHashMap<SimplifiedType, (Box<[ImplId]>, Box<[BuiltinDeriveImplId]>)>,
654    blanket_impls: Box<[ImplId]>,
655}
656
657#[derive(Default)]
658struct OneTraitImplsBuilder {
659    non_blanket_impls: FxHashMap<SimplifiedType, (Vec<ImplId>, Vec<BuiltinDeriveImplId>)>,
660    blanket_impls: Vec<ImplId>,
661}
662
663impl OneTraitImplsBuilder {
664    fn finish(self) -> OneTraitImpls {
665        let mut non_blanket_impls = self
666            .non_blanket_impls
667            .into_iter()
668            .map(|(self_ty, (impls, builtin_derive_impls))| {
669                (self_ty, (impls.into_boxed_slice(), builtin_derive_impls.into_boxed_slice()))
670            })
671            .collect::<FxHashMap<_, _>>();
672        non_blanket_impls.shrink_to_fit();
673        let blanket_impls = self.blanket_impls.into_boxed_slice();
674        OneTraitImpls { non_blanket_impls, blanket_impls }
675    }
676}
677
678#[derive(Debug, PartialEq)]
679pub struct TraitImpls {
680    map: FxHashMap<TraitId, OneTraitImpls>,
681}
682
683#[salsa::tracked]
684impl TraitImpls {
685    #[salsa::tracked(returns(ref))]
686    pub fn for_crate(db: &dyn HirDatabase, krate: Crate) -> Arc<Self> {
687        let _p = tracing::info_span!("inherent_impls_in_crate_query", ?krate).entered();
688
689        let crate_def_map = crate_def_map(db, krate);
690        let result = Self::collect_def_map(db, crate_def_map);
691        Arc::new(result)
692    }
693
694    #[salsa::tracked(returns(ref))]
695    pub fn for_block(db: &dyn HirDatabase, block: BlockId) -> Option<Box<Self>> {
696        let _p = tracing::info_span!("inherent_impls_in_block_query").entered();
697
698        let block_def_map = block_def_map(db, block);
699        let result = Self::collect_def_map(db, block_def_map);
700        if result.map.is_empty() { None } else { Some(Box::new(result)) }
701    }
702
703    #[salsa::tracked(returns(ref))]
704    pub fn for_crate_and_deps(db: &dyn HirDatabase, krate: Crate) -> Box<[Arc<Self>]> {
705        krate.transitive_deps(db).iter().map(|&dep| Self::for_crate(db, dep).clone()).collect()
706    }
707}
708
709impl TraitImpls {
710    fn collect_def_map(db: &dyn HirDatabase, def_map: &DefMap) -> Self {
711        let lang_items = hir_def::lang_item::lang_items(db, def_map.krate());
712        let mut map = FxHashMap::default();
713        collect(db, def_map, lang_items, &mut map);
714        let mut map = map
715            .into_iter()
716            .map(|(trait_id, trait_map)| (trait_id, trait_map.finish()))
717            .collect::<FxHashMap<_, _>>();
718        map.shrink_to_fit();
719        return Self { map };
720
721        fn collect(
722            db: &dyn HirDatabase,
723            def_map: &DefMap,
724            lang_items: &LangItems,
725            map: &mut FxHashMap<TraitId, OneTraitImplsBuilder>,
726        ) {
727            for (_module_id, module_data) in def_map.modules() {
728                for impl_id in module_data.scope.trait_impls() {
729                    let trait_ref = match db.impl_trait(impl_id) {
730                        Some(tr) => tr.instantiate_identity(),
731                        None => continue,
732                    };
733                    // Reservation impls should be ignored during trait resolution, so we never need
734                    // them during type analysis. See rust-lang/rust#64631 for details.
735                    //
736                    // FIXME: Reservation impls should be considered during coherence checks. If we are
737                    // (ever) to implement coherence checks, this filtering should be done by the trait
738                    // solver.
739                    if AttrFlags::query(db, impl_id.into())
740                        .contains(AttrFlags::RUSTC_RESERVATION_IMPL)
741                    {
742                        continue;
743                    }
744                    let self_ty = trait_ref.self_ty();
745                    let interner = DbInterner::new_no_crate(db);
746                    let entry = map.entry(trait_ref.def_id.0).or_default();
747                    match simplify_type(interner, self_ty, TreatParams::InstantiateWithInfer) {
748                        Some(self_ty) => {
749                            entry.non_blanket_impls.entry(self_ty).or_default().0.push(impl_id)
750                        }
751                        None => entry.blanket_impls.push(impl_id),
752                    }
753                }
754
755                for impl_id in module_data.scope.builtin_derive_impls() {
756                    let loc = impl_id.loc(db);
757                    let Some(trait_id) = loc.trait_.get_id(lang_items) else { continue };
758                    let entry = map.entry(trait_id).or_default();
759                    let entry = entry
760                        .non_blanket_impls
761                        .entry(SimplifiedType::Adt(loc.adt.into()))
762                        .or_default();
763                    entry.1.push(impl_id);
764                }
765
766                // To better support custom derives, collect impls in all unnamed const items.
767                // const _: () = { ... };
768                for konst in module_data.scope.unnamed_consts() {
769                    let body = db.body(konst.into());
770                    for (_, block_def_map) in body.blocks(db) {
771                        collect(db, block_def_map, lang_items, map);
772                    }
773                }
774            }
775        }
776    }

778    pub fn blanket_impls(&self, for_trait: TraitId) -> &[ImplId] {
779        self.map.get(&for_trait).map(|it| &*it.blanket_impls).unwrap_or_default()
780    }

782    /// Queries whether `self_ty` has potentially applicable implementations of `trait_`.
783    pub fn has_impls_for_trait_and_self_ty(
784        &self,
785        trait_: TraitId,
786        self_ty: &SimplifiedType,
787    ) -> bool {
788        self.map.get(&trait_).is_some_and(|trait_impls| {
789            trait_impls.non_blanket_impls.contains_key(self_ty)
790                || !trait_impls.blanket_impls.is_empty()
791        })
792    }
793
794    pub fn for_trait_and_self_ty(
795        &self,
796        trait_: TraitId,
797        self_ty: &SimplifiedType,
798    ) -> (&[ImplId], &[BuiltinDeriveImplId]) {
799        self.map
800            .get(&trait_)
801            .and_then(|map| map.non_blanket_impls.get(self_ty))
802            .map(|it| (&*it.0, &*it.1))
803            .unwrap_or_default()
804    }
805
806    pub fn for_trait(
807        &self,
808        trait_: TraitId,
809        mut callback: impl FnMut(Either<&[ImplId], &[BuiltinDeriveImplId]>),
810    ) {
811        if let Some(impls) = self.map.get(&trait_) {
812            callback(Either::Left(&impls.blanket_impls));
813            for impls in impls.non_blanket_impls.values() {
814                callback(Either::Left(&impls.0));
815                callback(Either::Right(&impls.1));
816            }
817        }
818    }
819
820    pub fn for_self_ty(
821        &self,
822        self_ty: &SimplifiedType,
823        mut callback: impl FnMut(Either<&[ImplId], &[BuiltinDeriveImplId]>),
824    ) {
825        for for_trait in self.map.values() {
826            if let Some(for_ty) = for_trait.non_blanket_impls.get(self_ty) {
827                callback(Either::Left(&for_ty.0));
828                callback(Either::Right(&for_ty.1));
829            }
830        }
831    }

833    pub fn for_each_crate_and_block(
834        db: &dyn HirDatabase,
835        krate: Crate,
836        block: Option<BlockId>,
837        for_each: &mut dyn FnMut(&TraitImpls),
838    ) {
839        let blocks = std::iter::successors(block, |block| block.loc(db).module.block(db));
840        blocks.filter_map(|block| Self::for_block(db, block).as_deref()).for_each(&mut *for_each);
841        Self::for_crate_and_deps(db, krate).iter().map(|it| &**it).for_each(for_each);
842    }
843
844    /// Like [`Self::for_each_crate_and_block()`], but takes in account two blocks, one for a trait and one for a self type.
845    pub fn for_each_crate_and_block_trait_and_type(
846        db: &dyn HirDatabase,
847        krate: Crate,
848        type_block: Option<BlockId>,
849        trait_block: Option<BlockId>,
850        for_each: &mut dyn FnMut(&TraitImpls),
851    ) {
852        let in_self_and_deps = TraitImpls::for_crate_and_deps(db, krate);
853        in_self_and_deps.iter().for_each(|impls| for_each(impls));
854
855        // We must not provide duplicate impls to the solver. Therefore we work with the following strategy:
856        // start from each block, and walk ancestors until you meet the other block. If they never meet,
857        // that means there can't be duplicate impls; if they meet, we stop the search of the deeper block.
858        // This breaks when they are equal (both will stop immediately), therefore we handle this case
859        // specifically.
860        let blocks_iter = |block: Option<BlockId>| {
861            std::iter::successors(block, |block| block.loc(db).module.block(db))
862        };
863        let for_each_block = |current_block: Option<BlockId>, other_block: Option<BlockId>| {
864            blocks_iter(current_block)
865                .take_while(move |&block| {
866                    other_block.is_none_or(|other_block| other_block != block)
867                })
868                .filter_map(move |block| TraitImpls::for_block(db, block).as_deref())
869        };
870        if trait_block == type_block {
871            blocks_iter(trait_block)
872                .filter_map(|block| TraitImpls::for_block(db, block).as_deref())
873                .for_each(for_each);
874        } else {
875            for_each_block(trait_block, type_block).for_each(&mut *for_each);
876            for_each_block(type_block, trait_block).for_each(for_each);
877        }
878    }
