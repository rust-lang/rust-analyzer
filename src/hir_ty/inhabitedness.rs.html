21// FIXME: Turn this into a query, it can be quite slow
22/// Checks whether a type is visibly uninhabited from a particular module.
23pub(crate) fn is_ty_uninhabited_from<'db>(
24    infcx: &InferCtxt<'db>,
25    ty: Ty<'db>,
26    target_mod: ModuleId,
27    env: ParamEnv<'db>,
28) -> bool {
29    let _p = tracing::info_span!("is_ty_uninhabited_from", ?ty).entered();
30    let mut uninhabited_from = UninhabitedFrom::new(infcx, target_mod, env);
31    let inhabitedness = ty.visit_with(&mut uninhabited_from);
32    inhabitedness == BREAK_VISIBLY_UNINHABITED
33}

35// FIXME: Turn this into a query, it can be quite slow
36/// Checks whether a variant is visibly uninhabited from a particular module.
37pub(crate) fn is_enum_variant_uninhabited_from<'db>(
38    infcx: &InferCtxt<'db>,
39    variant: EnumVariantId,
40    subst: GenericArgs<'db>,
41    target_mod: ModuleId,
42    env: ParamEnv<'db>,
43) -> bool {
44    let _p = tracing::info_span!("is_enum_variant_uninhabited_from").entered();
45
46    let mut uninhabited_from = UninhabitedFrom::new(infcx, target_mod, env);
47    let inhabitedness = uninhabited_from.visit_variant(variant.into(), subst);
48    inhabitedness == BREAK_VISIBLY_UNINHABITED
49}

51struct UninhabitedFrom<'a, 'db> {
52    target_mod: ModuleId,
53    recursive_ty: FxHashSet<Ty<'db>>,
54    // guard for preventing stack overflow in non trivial non terminating types
55    max_depth: usize,
56    infcx: &'a InferCtxt<'db>,
57    env: ParamEnv<'db>,
58}
59
60const CONTINUE_OPAQUELY_INHABITED: ControlFlow<VisiblyUninhabited> = Continue(());
61const BREAK_VISIBLY_UNINHABITED: ControlFlow<VisiblyUninhabited> = Break(VisiblyUninhabited);
62#[derive(PartialEq, Eq)]
63struct VisiblyUninhabited;
64
65impl<'db> TypeVisitor<DbInterner<'db>> for UninhabitedFrom<'_, 'db> {
66    type Result = ControlFlow<VisiblyUninhabited>;
67
68    fn visit_ty(&mut self, mut ty: Ty<'db>) -> ControlFlow<VisiblyUninhabited> {
69        if self.recursive_ty.contains(&ty) || self.max_depth == 0 {
70            // rustc considers recursive types always inhabited. I think it is valid to consider
71            // recursive types as always uninhabited, but we should do what rustc is doing.
72            return CONTINUE_OPAQUELY_INHABITED;
73        }
74        self.recursive_ty.insert(ty);
75        self.max_depth -= 1;
76
77        if matches!(ty.kind(), TyKind::Alias(..)) {
78            let mut ocx = ObligationCtxt::new(self.infcx);
79            match ocx.structurally_normalize_ty(&ObligationCause::dummy(), self.env, ty) {
80                Ok(it) => ty = it,
81                Err(_) => return CONTINUE_OPAQUELY_INHABITED,
82            }
83        }
84
85        let r = match ty.kind() {
86            TyKind::Adt(adt, subst) => self.visit_adt(adt.def_id().0, subst),
87            TyKind::Never => BREAK_VISIBLY_UNINHABITED,
88            TyKind::Tuple(..) => ty.super_visit_with(self),
89            TyKind::Array(item_ty, len) => match try_const_usize(self.infcx.interner.db, len) {
90                Some(0) | None => CONTINUE_OPAQUELY_INHABITED,
91                Some(1..) => item_ty.visit_with(self),
92            },
93            _ => CONTINUE_OPAQUELY_INHABITED,
94        };
95        self.recursive_ty.remove(&ty);
96        self.max_depth += 1;
97        r
98    }
99}

101impl<'a, 'db> UninhabitedFrom<'a, 'db> {
102    fn new(infcx: &'a InferCtxt<'db>, target_mod: ModuleId, env: ParamEnv<'db>) -> Self {
103        Self { target_mod, recursive_ty: FxHashSet::default(), max_depth: 500, infcx, env }
104    }
105
106    #[inline]
107    fn interner(&self) -> DbInterner<'db> {
108        self.infcx.interner
109    }
110
111    #[inline]
112    fn db(&self) -> &'db dyn HirDatabase {
113        self.interner().db
114    }
115
116    fn visit_adt(
117        &mut self,
118        adt: AdtId,
119        subst: GenericArgs<'db>,
120    ) -> ControlFlow<VisiblyUninhabited> {
121        // An ADT is uninhabited iff all its variants uninhabited.
122        match adt {
123            // rustc: For now, `union`s are never considered uninhabited.
124            AdtId::UnionId(_) => CONTINUE_OPAQUELY_INHABITED,
125            AdtId::StructId(s) => self.visit_variant(s.into(), subst),
126            AdtId::EnumId(e) => {
127                let enum_data = e.enum_variants(self.db());
128
129                for &(variant, _, _) in enum_data.variants.iter() {
130                    let variant_inhabitedness = self.visit_variant(variant.into(), subst);
131                    match variant_inhabitedness {
132                        Break(VisiblyUninhabited) => (),
133                        Continue(()) => return CONTINUE_OPAQUELY_INHABITED,
134                    }
135                }
136                BREAK_VISIBLY_UNINHABITED
137            }
138        }
139    }
140
141    fn visit_variant(
142        &mut self,
143        variant: VariantId,
144        subst: GenericArgs<'db>,
145    ) -> ControlFlow<VisiblyUninhabited> {
146        let variant_data = variant.fields(self.db());
147        let fields = variant_data.fields();
148        if fields.is_empty() {
149            return CONTINUE_OPAQUELY_INHABITED;
150        }
151
152        let is_enum = matches!(variant, VariantId::EnumVariantId(..));
153        let field_tys = self.db().field_types(variant);
154        let field_vis = if is_enum { None } else { Some(self.db().field_visibilities(variant)) };
155
156        for (fid, _) in fields.iter() {
157            self.visit_field(field_vis.as_ref().map(|it| it[fid]), &field_tys[fid].get(), subst)?;
158        }
159        CONTINUE_OPAQUELY_INHABITED
160    }
161
162    fn visit_field(
163        &mut self,
164        vis: Option<Visibility>,
165        ty: &EarlyBinder<'db, Ty<'db>>,
166        subst: GenericArgs<'db>,
167    ) -> ControlFlow<VisiblyUninhabited> {
168        if vis.is_none_or(|it| it.is_visible_from(self.db(), self.target_mod)) {
169            let ty = ty.instantiate(self.interner(), subst);
170            ty.visit_with(self)
171        } else {
172            CONTINUE_OPAQUELY_INHABITED
173        }
174    }
175}
