23// FIXME: Turn this into a query, it can be quite slow
24/// Checks whether a type is visibly uninhabited from a particular module.
25pub(crate) fn is_ty_uninhabited_from<'db>(
26    infcx: &InferCtxt<'db>,
27    ty: Ty<'db>,
28    target_mod: ModuleId,
29    env: Arc<TraitEnvironment<'db>>,
30) -> bool {
31    let _p = tracing::info_span!("is_ty_uninhabited_from", ?ty).entered();
32    let mut uninhabited_from = UninhabitedFrom::new(infcx, target_mod, env);
33    let inhabitedness = ty.visit_with(&mut uninhabited_from);
34    inhabitedness == BREAK_VISIBLY_UNINHABITED
35}

37// FIXME: Turn this into a query, it can be quite slow
38/// Checks whether a variant is visibly uninhabited from a particular module.
39pub(crate) fn is_enum_variant_uninhabited_from<'db>(
40    infcx: &InferCtxt<'db>,
41    variant: EnumVariantId,
42    subst: GenericArgs<'db>,
43    target_mod: ModuleId,
44    env: Arc<TraitEnvironment<'db>>,
45) -> bool {
46    let _p = tracing::info_span!("is_enum_variant_uninhabited_from").entered();
47
48    let mut uninhabited_from = UninhabitedFrom::new(infcx, target_mod, env);
49    let inhabitedness = uninhabited_from.visit_variant(variant.into(), subst);
50    inhabitedness == BREAK_VISIBLY_UNINHABITED
51}

53struct UninhabitedFrom<'a, 'db> {
54    target_mod: ModuleId,
55    recursive_ty: FxHashSet<Ty<'db>>,
56    // guard for preventing stack overflow in non trivial non terminating types
57    max_depth: usize,
58    infcx: &'a InferCtxt<'db>,
59    env: Arc<TraitEnvironment<'db>>,
60}
61
62const CONTINUE_OPAQUELY_INHABITED: ControlFlow<VisiblyUninhabited> = Continue(());
63const BREAK_VISIBLY_UNINHABITED: ControlFlow<VisiblyUninhabited> = Break(VisiblyUninhabited);
64#[derive(PartialEq, Eq)]
65struct VisiblyUninhabited;
66
67impl<'db> TypeVisitor<DbInterner<'db>> for UninhabitedFrom<'_, 'db> {
68    type Result = ControlFlow<VisiblyUninhabited>;
69
70    fn visit_ty(&mut self, mut ty: Ty<'db>) -> ControlFlow<VisiblyUninhabited> {
71        if self.recursive_ty.contains(&ty) || self.max_depth == 0 {
72            // rustc considers recursive types always inhabited. I think it is valid to consider
73            // recursive types as always uninhabited, but we should do what rustc is doing.
74            return CONTINUE_OPAQUELY_INHABITED;
75        }
76        self.recursive_ty.insert(ty);
77        self.max_depth -= 1;
78
79        if matches!(ty.kind(), TyKind::Alias(..)) {
80            let mut ocx = ObligationCtxt::new(self.infcx);
81            match ocx.structurally_normalize_ty(&ObligationCause::dummy(), self.env.env, ty) {
82                Ok(it) => ty = it,
83                Err(_) => return CONTINUE_OPAQUELY_INHABITED,
84            }
85        }
86
87        let r = match ty.kind() {
88            TyKind::Adt(adt, subst) => self.visit_adt(adt.def_id().0, subst),
89            TyKind::Never => BREAK_VISIBLY_UNINHABITED,
90            TyKind::Tuple(..) => ty.super_visit_with(self),
91            TyKind::Array(item_ty, len) => match try_const_usize(self.infcx.interner.db, len) {
92                Some(0) | None => CONTINUE_OPAQUELY_INHABITED,
93                Some(1..) => item_ty.visit_with(self),
94            },
95            _ => CONTINUE_OPAQUELY_INHABITED,
96        };
97        self.recursive_ty.remove(&ty);
98        self.max_depth += 1;
99        r
100    }
101}

103impl<'a, 'db> UninhabitedFrom<'a, 'db> {
104    fn new(
105        infcx: &'a InferCtxt<'db>,
106        target_mod: ModuleId,
107        env: Arc<TraitEnvironment<'db>>,
108    ) -> Self {
109        Self { target_mod, recursive_ty: FxHashSet::default(), max_depth: 500, infcx, env }
110    }
111
112    #[inline]
113    fn interner(&self) -> DbInterner<'db> {
114        self.infcx.interner
115    }
116
117    #[inline]
118    fn db(&self) -> &'db dyn HirDatabase {
119        self.interner().db
120    }
121
122    fn visit_adt(
123        &mut self,
124        adt: AdtId,
125        subst: GenericArgs<'db>,
126    ) -> ControlFlow<VisiblyUninhabited> {
127        // An ADT is uninhabited iff all its variants uninhabited.
128        match adt {
129            // rustc: For now, `union`s are never considered uninhabited.
130            AdtId::UnionId(_) => CONTINUE_OPAQUELY_INHABITED,
131            AdtId::StructId(s) => self.visit_variant(s.into(), subst),
132            AdtId::EnumId(e) => {
133                let enum_data = e.enum_variants(self.db());
134
135                for &(variant, _, _) in enum_data.variants.iter() {
136                    let variant_inhabitedness = self.visit_variant(variant.into(), subst);
137                    match variant_inhabitedness {
138                        Break(VisiblyUninhabited) => (),
139                        Continue(()) => return CONTINUE_OPAQUELY_INHABITED,
140                    }
141                }
142                BREAK_VISIBLY_UNINHABITED
143            }
144        }
145    }
146
147    fn visit_variant(
148        &mut self,
149        variant: VariantId,
150        subst: GenericArgs<'db>,
151    ) -> ControlFlow<VisiblyUninhabited> {
152        let variant_data = variant.fields(self.db());
153        let fields = variant_data.fields();
154        if fields.is_empty() {
155            return CONTINUE_OPAQUELY_INHABITED;
156        }
157
158        let is_enum = matches!(variant, VariantId::EnumVariantId(..));
159        let field_tys = self.db().field_types(variant);
160        let field_vis = if is_enum { None } else { Some(self.db().field_visibilities(variant)) };
161
162        for (fid, _) in fields.iter() {
163            self.visit_field(field_vis.as_ref().map(|it| it[fid]), &field_tys[fid], subst)?;
164        }
165        CONTINUE_OPAQUELY_INHABITED
166    }
167
168    fn visit_field(
169        &mut self,
170        vis: Option<Visibility>,
171        ty: &EarlyBinder<'db, Ty<'db>>,
172        subst: GenericArgs<'db>,
173    ) -> ControlFlow<VisiblyUninhabited> {
174        if vis.is_none_or(|it| it.is_visible_from(self.db(), self.target_mod)) {
175            let ty = ty.instantiate(self.interner(), subst);
176            ty.visit_with(self)
177        } else {
178            CONTINUE_OPAQUELY_INHABITED
179        }
180    }
181}
