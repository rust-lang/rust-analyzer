48// We cannot use `&mut dyn FnMut()` because of lifetime issues, and we don't want to use `Box<dyn FnMut()>`
49// because of the allocation, so we create a lifetime-less callback, tailored for our needs.
50pub(crate) struct PathDiagnosticCallback<'a, 'db> {
51    pub(crate) data: CallbackData<'a>,
52    pub(crate) callback:
53        fn(&CallbackData<'_>, &mut TyLoweringContext<'db, '_>, PathLoweringDiagnostic),
54}

56pub(crate) struct PathLoweringContext<'a, 'b, 'db> {
57    ctx: &'a mut TyLoweringContext<'db, 'b>,
58    on_diagnostic: PathDiagnosticCallback<'a, 'db>,
59    path: &'a Path,
60    segments: PathSegments<'a>,
61    current_segment_idx: usize,
62    /// Contains the previous segment if `current_segment_idx == segments.len()`
63    current_or_prev_segment: PathSegment<'a>,
64}

66impl<'a, 'b, 'db> PathLoweringContext<'a, 'b, 'db> {
67    #[inline]
68    pub(crate) fn new(
69        ctx: &'a mut TyLoweringContext<'db, 'b>,
70        on_diagnostic: PathDiagnosticCallback<'a, 'db>,
71        path: &'a Path,
72    ) -> Self {
73        let segments = path.segments();
74        let first_segment = segments.first().unwrap_or(PathSegment::MISSING);
75        Self {
76            ctx,
77            on_diagnostic,
78            path,
79            segments,
80            current_segment_idx: 0,
81            current_or_prev_segment: first_segment,
82        }
83    }

88        (self.on_diagnostic.callback)(&self.on_diagnostic.data, self.ctx, diag);
89    }

101    #[inline]
102    fn skip_resolved_segment(&mut self) {
103        if !matches!(self.path, Path::LangItem(..)) {
104            // In lang items, the resolved "segment" is not one of the segments. Perhaps we should've put it
105            // point at -1, but I don't feel this is clearer.
106            self.current_segment_idx += 1;
107        }
108        self.update_current_segment();
109    }
110
111    #[inline]
112    fn update_current_segment(&mut self) {
113        self.current_or_prev_segment =
114            self.segments.get(self.current_segment_idx).unwrap_or(self.current_or_prev_segment);
115    }

117    #[inline]
118    pub(crate) fn ignore_last_segment(&mut self) {
119        self.segments = self.segments.strip_last();
120    }

131    #[inline]
132    fn with_lifetime_elision<T>(
133        &mut self,
134        lifetime_elision: LifetimeElisionKind<'db>,
135        f: impl FnOnce(&mut PathLoweringContext<'_, '_, 'db>) -> T,
136    ) -> T {
137        let old_lifetime_elision =
138            std::mem::replace(&mut self.ctx.lifetime_elision, lifetime_elision);

140        self.ctx.lifetime_elision = old_lifetime_elision;

144    pub(crate) fn lower_ty_relative_path(
145        &mut self,
146        ty: Ty<'db>,
147        // We need the original resolution to lower `Self::AssocTy` correctly
148        res: Option<TypeNs>,
149        infer_args: bool,
150    ) -> (Ty<'db>, Option<TypeNs>) {
151        let remaining_segments = self.segments.len() - self.current_segment_idx;
152        match remaining_segments {
153            0 => (ty, res),
154            1 => {
155                // resolve unselected assoc types
156                (self.select_associated_type(res, infer_args), None)
157            }
158            _ => {
159                // FIXME report error (ambiguous associated type)
160                (Ty::new_error(self.ctx.interner, ErrorGuaranteed), None)
161            }
162        }
163    }

165    // When calling this, the current segment is the resolved segment (we don't advance it yet).
166    pub(crate) fn lower_partly_resolved_path(
167        &mut self,
168        resolution: TypeNs,
169        infer_args: bool,
170    ) -> (Ty<'db>, Option<TypeNs>) {
171        let remaining_segments = self.segments.skip(self.current_segment_idx + 1);
172        tracing::debug!(?remaining_segments);
173        let rem_seg_len = remaining_segments.len();
174        tracing::debug!(?rem_seg_len);
175
176        let ty = match resolution {
177            TypeNs::TraitId(trait_) => {
178                let ty = match remaining_segments.len() {
179                    1 => {
180                        let trait_ref = self.lower_trait_ref_from_resolved_path(
181                            trait_,
182                            Ty::new_error(self.ctx.interner, ErrorGuaranteed),
183                            false,
184                        );
185                        tracing::debug!(?trait_ref);
186                        self.skip_resolved_segment();
187                        let segment = self.current_or_prev_segment;
188                        let trait_id = trait_ref.def_id.0;
189                        let found =
190                            trait_id.trait_items(self.ctx.db).associated_type_by_name(segment.name);
191
192                        tracing::debug!(?found);
193                        match found {
194                            Some(associated_ty) => {
195                                // FIXME: `substs_from_path_segment()` pushes `TyKind::Error` for every parent
196                                // generic params. It's inefficient to splice the `Substitution`s, so we may want
197                                // that method to optionally take parent `Substitution` as we already know them at
198                                // this point (`trait_ref.substitution`).
199                                let substitution = self.substs_from_path_segment(
200                                    associated_ty.into(),
201                                    false,
202                                    None,
203                                    true,
204                                );
205                                let args = GenericArgs::new_from_iter(
206                                    self.ctx.interner,
207                                    trait_ref
208                                        .args
209                                        .iter()
210                                        .chain(substitution.iter().skip(trait_ref.args.len())),
211                                );
212                                Ty::new_alias(
213                                    self.ctx.interner,
214                                    AliasTyKind::Projection,
215                                    AliasTy::new_from_args(
216                                        self.ctx.interner,
217                                        associated_ty.into(),
218                                        args,
219                                    ),
220                                )
221                            }
222                            None => {
223                                // FIXME: report error (associated type not found)
224                                Ty::new_error(self.ctx.interner, ErrorGuaranteed)
225                            }
226                        }
227                    }
228                    0 => {
229                        // Trait object type without dyn; this should be handled in upstream. See
230                        // `lower_path()`.
231                        stdx::never!("unexpected fully resolved trait path");
232                        Ty::new_error(self.ctx.interner, ErrorGuaranteed)
233                    }
234                    _ => {
235                        // FIXME report error (ambiguous associated type)
236                        Ty::new_error(self.ctx.interner, ErrorGuaranteed)
237                    }
238                };
239                return (ty, None);
240            }
241            TypeNs::GenericParam(param_id) => {
242                let generics = self.ctx.generics();
243                let idx = generics.type_or_const_param_idx(param_id.into());
244                match idx {
245                    None => {
246                        never!("no matching generics");
247                        Ty::new_error(self.ctx.interner, ErrorGuaranteed)
248                    }
249                    Some(idx) => {
250                        let (pidx, _param) = generics.iter().nth(idx).unwrap();
251                        assert_eq!(pidx, param_id.into());
252                        self.ctx.type_param(param_id, idx as u32)
253                    }
254                }
255            }
256            TypeNs::SelfType(impl_id) => self.ctx.db.impl_self_ty(impl_id).skip_binder(),
257            TypeNs::AdtSelfType(adt) => {
258                let args = GenericArgs::identity_for_item(self.ctx.interner, adt.into());
259                Ty::new_adt(self.ctx.interner, adt, args)
260            }
261
262            TypeNs::AdtId(it) => self.lower_path_inner(it.into(), infer_args),
263            TypeNs::BuiltinType(it) => self.lower_path_inner(it.into(), infer_args),
264            TypeNs::TypeAliasId(it) => self.lower_path_inner(it.into(), infer_args),
265            // FIXME: report error
266            TypeNs::EnumVariantId(_) | TypeNs::ModuleId(_) => {
267                return (Ty::new_error(self.ctx.interner, ErrorGuaranteed), None);
268            }
269        };
270
271        tracing::debug!(?ty);
272
273        self.skip_resolved_segment();
274        self.lower_ty_relative_path(ty, Some(resolution), infer_args)
275    }

277    /// This returns whether to keep the resolution (`true`) of throw it (`false`).
278    #[must_use]
279    fn handle_type_ns_resolution(&mut self, resolution: &TypeNs) -> bool {
280        let mut prohibit_generics_on_resolved = |reason| {
281            if self.current_or_prev_segment.args_and_bindings.is_some() {
282                let segment = self.current_segment_u32();
283                self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
284                    segment,
285                    reason,
286                });
287            }
288        };
289
290        match resolution {
291            TypeNs::SelfType(_) => {
292                prohibit_generics_on_resolved(GenericArgsProhibitedReason::SelfTy)
293            }
294            TypeNs::GenericParam(_) => {
295                prohibit_generics_on_resolved(GenericArgsProhibitedReason::TyParam)
296            }
297            TypeNs::AdtSelfType(_) => {
298                prohibit_generics_on_resolved(GenericArgsProhibitedReason::SelfTy);
299
300                if self.ctx.lowering_param_default.is_some() {
301                    // Generic defaults are not allowed to refer to `Self`.
302                    // FIXME: Emit an error.
303                    return false;
304                }
305            }
306            TypeNs::BuiltinType(_) => {
307                prohibit_generics_on_resolved(GenericArgsProhibitedReason::PrimitiveTy)
308            }
309            TypeNs::ModuleId(_) => {
310                prohibit_generics_on_resolved(GenericArgsProhibitedReason::Module)
311            }
312            TypeNs::AdtId(_)
313            | TypeNs::EnumVariantId(_)
314            | TypeNs::TypeAliasId(_)
315            | TypeNs::TraitId(_) => {}
316        }
317
318        true
319    }

329    #[tracing::instrument(skip(self), ret)]
330    pub(crate) fn resolve_path_in_type_ns(&mut self) -> Option<(TypeNs, Option<usize>)> {
331        let (resolution, remaining_index, _, prefix_info) =
332            self.ctx.resolver.resolve_path_in_type_ns_with_prefix_info(self.ctx.db, self.path)?;
333
334        let segments = self.segments;
335        if segments.is_empty() || matches!(self.path, Path::LangItem(..)) {
336            // `segments.is_empty()` can occur with `self`.
337            return Some((resolution, remaining_index));
338        }
339
340        let (module_segments, resolved_segment_idx, enum_segment) = match remaining_index {
341            None if prefix_info.enum_variant => {
342                (segments.strip_last_two(), segments.len() - 1, Some(segments.len() - 2))
343            }
344            None => (segments.strip_last(), segments.len() - 1, None),
345            Some(i) => (segments.take(i - 1), i - 1, None),
346        };
347
348        self.current_segment_idx = resolved_segment_idx;
349        self.current_or_prev_segment =
350            segments.get(resolved_segment_idx).expect("should have resolved segment");
351
352        for (i, mod_segment) in module_segments.iter().enumerate() {
353            if mod_segment.args_and_bindings.is_some() {
354                self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
355                    segment: i as u32,
356                    reason: GenericArgsProhibitedReason::Module,
357                });
358            }
359        }
360
361        if let Some(enum_segment) = enum_segment
362            && segments.get(enum_segment).is_some_and(|it| it.args_and_bindings.is_some())
363            && segments.get(enum_segment + 1).is_some_and(|it| it.args_and_bindings.is_some())
364        {
365            self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
366                segment: (enum_segment + 1) as u32,
367                reason: GenericArgsProhibitedReason::EnumVariant,
368            });
369        }
370
371        if !self.handle_type_ns_resolution(&resolution) {
372            return None;
373        }
374
375        Some((resolution, remaining_index))
376    }

378    pub(crate) fn resolve_path_in_value_ns(
379        &mut self,
380        hygiene_id: HygieneId,
381    ) -> Option<ResolveValueResult> {
382        let (res, prefix_info) = self.ctx.resolver.resolve_path_in_value_ns_with_prefix_info(
383            self.ctx.db,
384            self.path,
385            hygiene_id,
386        )?;
387
388        let segments = self.segments;
389        if segments.is_empty() || matches!(self.path, Path::LangItem(..)) {
390            // `segments.is_empty()` can occur with `self`.
391            return Some(res);
392        }
393
394        let (mod_segments, enum_segment, resolved_segment_idx) = match res {
395            ResolveValueResult::Partial(_, unresolved_segment) => {
396                (segments.take(unresolved_segment - 1), None, unresolved_segment - 1)
397            }
398            ResolveValueResult::ValueNs(ValueNs::EnumVariantId(_)) if prefix_info.enum_variant => {
399                (segments.strip_last_two(), segments.len().checked_sub(2), segments.len() - 1)
400            }
401            ResolveValueResult::ValueNs(..) => (segments.strip_last(), None, segments.len() - 1),
402        };
403
404        self.current_segment_idx = resolved_segment_idx;
405        self.current_or_prev_segment =
406            segments.get(resolved_segment_idx).expect("should have resolved segment");
407
408        for (i, mod_segment) in mod_segments.iter().enumerate() {
409            if mod_segment.args_and_bindings.is_some() {
410                self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
411                    segment: i as u32,
412                    reason: GenericArgsProhibitedReason::Module,
413                });
414            }
415        }
416
417        if let Some(enum_segment) = enum_segment
418            && segments.get(enum_segment).is_some_and(|it| it.args_and_bindings.is_some())
419            && segments.get(enum_segment + 1).is_some_and(|it| it.args_and_bindings.is_some())
420        {
421            self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
422                segment: (enum_segment + 1) as u32,
423                reason: GenericArgsProhibitedReason::EnumVariant,
424            });
425        }
426
427        match &res {
428            ResolveValueResult::ValueNs(resolution) => {
429                let resolved_segment_idx = self.current_segment_u32();
430                let resolved_segment = self.current_or_prev_segment;
431
432                let mut prohibit_generics_on_resolved = |reason| {
433                    if resolved_segment.args_and_bindings.is_some() {
434                        self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
435                            segment: resolved_segment_idx,
436                            reason,
437                        });
438                    }
439                };
440
441                match resolution {
442                    ValueNs::ImplSelf(_) => {
443                        prohibit_generics_on_resolved(GenericArgsProhibitedReason::SelfTy);
444                    }
445                    // FIXME: rustc generates E0107 (incorrect number of generic arguments) and not
446                    // E0109 (generic arguments provided for a type that doesn't accept them) for
447                    // consts and statics, presumably as a defense against future in which consts
448                    // and statics can be generic, or just because it was easier for rustc implementors.
449                    // That means we'll show the wrong error code. Because of us it's easier to do it
450                    // this way :)
451                    ValueNs::GenericParam(_) => {
452                        prohibit_generics_on_resolved(GenericArgsProhibitedReason::Const)
453                    }
454                    ValueNs::StaticId(_) => {
455                        prohibit_generics_on_resolved(GenericArgsProhibitedReason::Static)
456                    }
457                    ValueNs::LocalBinding(_) => {
458                        prohibit_generics_on_resolved(GenericArgsProhibitedReason::LocalVariable)
459                    }
460                    ValueNs::FunctionId(_)
461                    | ValueNs::StructId(_)
462                    | ValueNs::EnumVariantId(_)
463                    | ValueNs::ConstId(_) => {}
464                }
465            }
466            ResolveValueResult::Partial(resolution, _) => {
467                if !self.handle_type_ns_resolution(resolution) {
468                    return None;
469                }
470            }
471        };
472        Some(res)
473    }

475    #[tracing::instrument(skip(self), ret)]
476    fn select_associated_type(&mut self, res: Option<TypeNs>, infer_args: bool) -> Ty<'db> {
477        let interner = self.ctx.interner;
478        let db = self.ctx.db;
479        let def = self.ctx.def;
480        let segment = self.current_or_prev_segment;
481        let assoc_name = segment.name;
482        let error_ty = || Ty::new_error(self.ctx.interner, ErrorGuaranteed);
483        let (assoc_type, trait_args) = match res {
484            Some(TypeNs::GenericParam(param)) => {
485                let Ok(assoc_type) = super::resolve_type_param_assoc_type_shorthand(
486                    db,
487                    def,
488                    param,
489                    assoc_name.clone(),
490                ) else {
491                    return error_ty();
492                };
493                assoc_type
494                    .get_with(|(assoc_type, trait_args)| (*assoc_type, trait_args.as_ref()))
495                    .skip_binder()
496            }
497            Some(TypeNs::SelfType(impl_)) => {
498                let Some(impl_trait) = db.impl_trait(impl_) else {
499                    return error_ty();
500                };
501                let impl_trait = impl_trait.instantiate_identity();
502                // Searching for `Self::Assoc` in `impl Trait for Type` is like searching for `Self::Assoc` in `Trait`.
503                let Ok(assoc_type) = super::resolve_type_param_assoc_type_shorthand(
504                    db,
505                    impl_trait.def_id.0.into(),
506                    TypeParamId::trait_self(impl_trait.def_id.0),
507                    assoc_name.clone(),
508                ) else {
509                    return error_ty();
510                };
511                let (assoc_type, trait_args) = assoc_type
512                    .get_with(|(assoc_type, trait_args)| (*assoc_type, trait_args.as_ref()))
513                    .skip_binder();
514                (assoc_type, EarlyBinder::bind(trait_args).instantiate(interner, impl_trait.args))
515            }
516            _ => return error_ty(),
517        };
518
519        // FIXME: `substs_from_path_segment()` pushes `TyKind::Error` for every parent
520        // generic params. It's inefficient to splice the `Substitution`s, so we may want
521        // that method to optionally take parent `Substitution` as we already know them at
522        // this point (`t.substitution`).
523        let substs = self.substs_from_path_segment(assoc_type.into(), infer_args, None, true);
524
525        let substs = GenericArgs::new_from_iter(
526            interner,
527            trait_args.iter().chain(substs.iter().skip(trait_args.len())),
528        );
529
530        Ty::new_projection_from_args(interner, assoc_type.into(), substs)
531    }

533    fn lower_path_inner(&mut self, typeable: TyDefId, infer_args: bool) -> Ty<'db> {
534        let generic_def = match typeable {
535            TyDefId::BuiltinType(builtinty) => {
536                return Ty::from_builtin_type(self.ctx.interner, builtinty);
537            }
538            TyDefId::AdtId(it) => it.into(),
539            TyDefId::TypeAliasId(it) => it.into(),
540        };
541        let args = self.substs_from_path_segment(generic_def, infer_args, None, false);
542        let ty = ty_query(self.ctx.db, typeable);
543        ty.instantiate(self.ctx.interner, args)
544    }

546    /// Collect generic arguments from a path into a `Substs`. See also
547    /// `create_substs_for_ast_path` and `def_to_ty` in rustc.
548    pub(crate) fn substs_from_path(
549        &mut self,
550        // Note that we don't call `db.value_type(resolved)` here,
551        // `ValueTyDefId` is just a convenient way to pass generics and
552        // special-case enum variants
553        resolved: ValueTyDefId,
554        infer_args: bool,
555        lowering_assoc_type_generics: bool,
556    ) -> GenericArgs<'db> {
557        let interner = self.ctx.interner;
558        let prev_current_segment_idx = self.current_segment_idx;
559        let prev_current_segment = self.current_or_prev_segment;
560
561        let generic_def = match resolved {
562            ValueTyDefId::FunctionId(it) => it.into(),
563            ValueTyDefId::StructId(it) => it.into(),
564            ValueTyDefId::UnionId(it) => it.into(),
565            ValueTyDefId::ConstId(it) => it.into(),
566            ValueTyDefId::StaticId(_) => {
567                return GenericArgs::empty(interner);
568            }
569            ValueTyDefId::EnumVariantId(var) => {
570                // the generic args for an enum variant may be either specified
571                // on the segment referring to the enum, or on the segment
572                // referring to the variant. So `Option::<T>::None` and
573                // `Option::None::<T>` are both allowed (though the former is
574                // FIXME: This isn't strictly correct, enum variants may be used not through the enum
575                // (via `use Enum::Variant`). The resolver returns whether they were, but we don't have its result
576                // available here. The worst that can happen is that we will show some confusing diagnostics to the user,
577                // if generics exist on the module and they don't match with the variant.
578                // preferred). See also `def_ids_for_path_segments` in rustc.
579                //
580                // `wrapping_sub(1)` will return a number which `get` will return None for if current_segment_idx<2.
581                // This simplifies the code a bit.
582                let penultimate_idx = self.current_segment_idx.wrapping_sub(1);
583                let penultimate = self.segments.get(penultimate_idx);
584                if let Some(penultimate) = penultimate
585                    && self.current_or_prev_segment.args_and_bindings.is_none()
586                    && penultimate.args_and_bindings.is_some()
587                {
588                    self.current_segment_idx = penultimate_idx;
589                    self.current_or_prev_segment = penultimate;
590                }
591                var.lookup(self.ctx.db).parent.into()
592            }
593        };
594        let result = self.substs_from_path_segment(
595            generic_def,
596            infer_args,
597            None,
598            lowering_assoc_type_generics,
599        );
600        self.current_segment_idx = prev_current_segment_idx;
601        self.current_or_prev_segment = prev_current_segment;
602        result
603    }

605    pub(crate) fn substs_from_path_segment(
606        &mut self,
607        def: GenericDefId,
608        infer_args: bool,
609        explicit_self_ty: Option<Ty<'db>>,
610        lowering_assoc_type_generics: bool,
611    ) -> GenericArgs<'db> {
612        let old_lifetime_elision = self.ctx.lifetime_elision;
613
614        if let Some(args) = self.current_or_prev_segment.args_and_bindings
615            && args.parenthesized != GenericArgsParentheses::No
616        {
617            let prohibit_parens = match def {
618                GenericDefId::TraitId(trait_) => {
619                    // RTN is prohibited anyways if we got here.
620                    let is_rtn = args.parenthesized == GenericArgsParentheses::ReturnTypeNotation;
621                    let is_fn_trait = self
622                        .ctx
623                        .db
624                        .trait_signature(trait_)
625                        .flags
626                        .contains(TraitFlags::RUSTC_PAREN_SUGAR);
627                    is_rtn || !is_fn_trait
628                }
629                _ => true,
630            };
631
632            if prohibit_parens {
633                let segment = self.current_segment_u32();
634                self.on_diagnostic(
635                    PathLoweringDiagnostic::ParenthesizedGenericArgsWithoutFnTrait { segment },
636                );
637
638                return unknown_subst(self.ctx.interner, def);
639            }
640
641            // `Fn()`-style generics are treated like functions for the purpose of lifetime elision.
642            self.ctx.lifetime_elision =
643                LifetimeElisionKind::AnonymousCreateParameter { report_in_path: false };
644        }
645
646        let result = self.substs_from_args_and_bindings(
647            self.current_or_prev_segment.args_and_bindings,
648            def,
649            infer_args,
650            explicit_self_ty,
651            PathGenericsSource::Segment(self.current_segment_u32()),
652            lowering_assoc_type_generics,
653            self.ctx.lifetime_elision,
654        );
655        self.ctx.lifetime_elision = old_lifetime_elision;
656        result
657    }

659    pub(super) fn substs_from_args_and_bindings(
660        &mut self,
661        args_and_bindings: Option<&HirGenericArgs>,
662        def: GenericDefId,
663        infer_args: bool,
664        explicit_self_ty: Option<Ty<'db>>,
665        generics_source: PathGenericsSource,
666        lowering_assoc_type_generics: bool,
667        lifetime_elision: LifetimeElisionKind<'db>,
668    ) -> GenericArgs<'db> {
669        struct LowererCtx<'a, 'b, 'c, 'db> {
670            ctx: &'a mut PathLoweringContext<'b, 'c, 'db>,
671            generics_source: PathGenericsSource,
672        }
673
674        impl<'db> GenericArgsLowerer<'db> for LowererCtx<'_, '_, '_, 'db> {
675            fn report_len_mismatch(
676                &mut self,
677                def: GenericDefId,
678                provided_count: u32,
679                expected_count: u32,
680                kind: IncorrectGenericsLenKind,
681            ) {
682                self.ctx.on_diagnostic(PathLoweringDiagnostic::IncorrectGenericsLen {
683                    generics_source: self.generics_source,
684                    provided_count,
685                    expected_count,
686                    kind,
687                    def,
688                });
689            }
690
691            fn report_arg_mismatch(
692                &mut self,
693                param_id: GenericParamId,
694                arg_idx: u32,
695                has_self_arg: bool,
696            ) {
697                self.ctx.on_diagnostic(PathLoweringDiagnostic::IncorrectGenericsOrder {
698                    generics_source: self.generics_source,
699                    param_id,
700                    arg_idx,
701                    has_self_arg,
702                });
703            }
704
705            fn provided_kind(
706                &mut self,
707                param_id: GenericParamId,
708                param: GenericParamDataRef<'_>,
709                arg: &HirGenericArg,
710            ) -> GenericArg<'db> {
711                match (param, *arg) {
712                    (
713                        GenericParamDataRef::LifetimeParamData(_),
714                        HirGenericArg::Lifetime(lifetime),
715                    ) => self.ctx.ctx.lower_lifetime(lifetime).into(),
716                    (GenericParamDataRef::TypeParamData(_), HirGenericArg::Type(type_ref)) => {
717                        self.ctx.ctx.lower_ty(type_ref).into()
718                    }
719                    (GenericParamDataRef::ConstParamData(_), HirGenericArg::Const(konst)) => {
720                        let GenericParamId::ConstParamId(const_id) = param_id else {
721                            unreachable!("non-const param ID for const param");
722                        };
723                        self.ctx
724                            .ctx
725                            .lower_const(konst, const_param_ty_query(self.ctx.ctx.db, const_id))
726                            .into()
727                    }
728                    _ => unreachable!("unmatching param kinds were passed to `provided_kind()`"),
729                }
730            }
731
732            fn provided_type_like_const(
733                &mut self,
734                const_ty: Ty<'db>,
735                arg: TypeLikeConst<'_>,
736            ) -> Const<'db> {
737                match arg {
738                    TypeLikeConst::Path(path) => self.ctx.ctx.lower_path_as_const(path, const_ty),
739                    TypeLikeConst::Infer => unknown_const(const_ty),
740                }
741            }
742
743            fn inferred_kind(
744                &mut self,
745                def: GenericDefId,
746                param_id: GenericParamId,
747                param: GenericParamDataRef<'_>,
748                infer_args: bool,
749                preceding_args: &[GenericArg<'db>],
750            ) -> GenericArg<'db> {
751                let default =
752                    || {
753                        self.ctx.ctx.db.generic_defaults(def).get(preceding_args.len()).map(
754                            |default| default.instantiate(self.ctx.ctx.interner, preceding_args),
755                        )
756                    };
757                match param {
758                    GenericParamDataRef::LifetimeParamData(_) => {
759                        Region::new(self.ctx.ctx.interner, rustc_type_ir::ReError(ErrorGuaranteed))
760                            .into()
761                    }
762                    GenericParamDataRef::TypeParamData(param) => {
763                        if !infer_args
764                            && param.default.is_some()
765                            && let Some(default) = default()
766                        {
767                            return default;
768                        }
769                        Ty::new_error(self.ctx.ctx.interner, ErrorGuaranteed).into()
770                    }
771                    GenericParamDataRef::ConstParamData(param) => {
772                        if !infer_args
773                            && param.default.is_some()
774                            && let Some(default) = default()
775                        {
776                            return default;
777                        }
778                        let GenericParamId::ConstParamId(const_id) = param_id else {
779                            unreachable!("non-const param ID for const param");
780                        };
781                        unknown_const_as_generic(const_param_ty_query(self.ctx.ctx.db, const_id))
782                    }
783                }
784            }
785
786            fn parent_arg(&mut self, _param_idx: u32, param_id: GenericParamId) -> GenericArg<'db> {
787                match param_id {
788                    GenericParamId::TypeParamId(_) => {
789                        Ty::new_error(self.ctx.ctx.interner, ErrorGuaranteed).into()
790                    }
791                    GenericParamId::ConstParamId(const_id) => {
792                        unknown_const_as_generic(const_param_ty_query(self.ctx.ctx.db, const_id))
793                    }
794                    GenericParamId::LifetimeParamId(_) => {
795                        Region::new(self.ctx.ctx.interner, rustc_type_ir::ReError(ErrorGuaranteed))
796                            .into()
797                    }
798                }
799            }
800
801            fn report_elided_lifetimes_in_path(
802                &mut self,
803                def: GenericDefId,
804                expected_count: u32,
805                hard_error: bool,
806            ) {
807                self.ctx.on_diagnostic(PathLoweringDiagnostic::ElidedLifetimesInPath {
808                    generics_source: self.generics_source,
809                    def,
810                    expected_count,
811                    hard_error,
812                });
813            }
814
815            fn report_elision_failure(&mut self, def: GenericDefId, expected_count: u32) {
816                self.ctx.on_diagnostic(PathLoweringDiagnostic::ElisionFailure {
817                    generics_source: self.generics_source,
818                    def,
819                    expected_count,
820                });
821            }
822
823            fn report_missing_lifetime(&mut self, def: GenericDefId, expected_count: u32) {
824                self.ctx.on_diagnostic(PathLoweringDiagnostic::MissingLifetime {
825                    generics_source: self.generics_source,
826                    def,
827                    expected_count,
828                });
829            }
830        }
831
832        substs_from_args_and_bindings(
833            self.ctx.db,
834            self.ctx.store,
835            args_and_bindings,
836            def,
837            infer_args,
838            lifetime_elision,
839            lowering_assoc_type_generics,
840            explicit_self_ty,
841            &mut LowererCtx { ctx: self, generics_source },
842        )
843    }

846        &mut self,
847        resolved: TraitId,
848        explicit_self_ty: Ty<'db>,
849        infer_args: bool,
850    ) -> TraitRef<'db> {
851        let args = self.trait_ref_substs_from_path(resolved, explicit_self_ty, infer_args);
852        TraitRef::new_from_args(self.ctx.interner, resolved.into(), args)

856        &mut self,

864    pub(super) fn assoc_type_bindings_from_type_bound<'c>(
865        mut self,
866        trait_ref: TraitRef<'db>,
867    ) -> Option<impl Iterator<Item = (Clause<'db>, GenericPredicateSource)> + use<'a, 'b, 'c, 'db>>
868    {
869        let interner = self.ctx.interner;
870        self.current_or_prev_segment.args_and_bindings.map(|args_and_bindings| {
871            args_and_bindings.bindings.iter().enumerate().flat_map(move |(binding_idx, binding)| {
872                let found = associated_type_by_name_including_super_traits(
873                    self.ctx.db,
874                    trait_ref,
875                    binding.name.clone(),
876                );
877                let (associated_ty, super_trait_args) = match found {
878                    None => return SmallVec::new(),
879                    Some(t) => t,
880                };
881                let args =
882                    self.with_lifetime_elision(LifetimeElisionKind::AnonymousReportError, |this| {
883                        // FIXME: `substs_from_path_segment()` pushes `TyKind::Error` for every parent
884                        // generic params. It's inefficient to splice the `Substitution`s, so we may want
885                        // that method to optionally take parent `Substitution` as we already know them at
886                        // this point (`super_trait_ref.substitution`).
887                        this.substs_from_args_and_bindings(
888                            binding.args.as_ref(),
889                            associated_ty.into(),
890                            false, // this is not relevant
891                            Some(super_trait_args.type_at(0)),
892                            PathGenericsSource::AssocType {
893                                segment: this.current_segment_u32(),
894                                assoc_type: binding_idx as u32,
895                            },
896                            false,
897                            this.ctx.lifetime_elision,
898                        )
899                    });
900                let args = GenericArgs::new_from_iter(
901                    interner,
902                    super_trait_args.iter().chain(args.iter().skip(super_trait_args.len())),
903                );
904                let projection_term =
905                    AliasTerm::new_from_args(interner, associated_ty.into(), args);
906                let mut predicates: SmallVec<[_; 1]> = SmallVec::with_capacity(
907                    binding.type_ref.as_ref().map_or(0, |_| 1) + binding.bounds.len(),
908                );
909                if let Some(type_ref) = binding.type_ref {
910                    let lifetime_elision =
911                        if args_and_bindings.parenthesized == GenericArgsParentheses::ParenSugar {
912                            // `Fn()`-style generics are elided like functions. This is `Output` (we lower to it in hir-def).
913                            LifetimeElisionKind::for_fn_ret(self.ctx.interner)
914                        } else {
915                            self.ctx.lifetime_elision
916                        };
917                    self.with_lifetime_elision(lifetime_elision, |this| {
918                        match (&this.ctx.store[type_ref], this.ctx.impl_trait_mode.mode) {
919                            (TypeRef::ImplTrait(_), ImplTraitLoweringMode::Disallowed) => (),
920                            (
921                                _,
922                                ImplTraitLoweringMode::Disallowed | ImplTraitLoweringMode::Opaque,
923                            ) => {
924                                let ty = this.ctx.lower_ty(type_ref);
925                                let pred = Clause(Predicate::new(
926                                    interner,
927                                    Binder::dummy(rustc_type_ir::PredicateKind::Clause(
928                                        rustc_type_ir::ClauseKind::Projection(
929                                            ProjectionPredicate {
930                                                projection_term,
931                                                term: ty.into(),
932                                            },
933                                        ),
934                                    )),
935                                ));
936                                predicates.push((pred, GenericPredicateSource::SelfOnly));
937                            }
938                        }
939                    })
940                }
941                for bound in binding.bounds.iter() {
942                    predicates.extend(
943                        self.ctx
944                            .lower_type_bound(
945                                bound,
946                                Ty::new_alias(
947                                    self.ctx.interner,
948                                    AliasTyKind::Projection,
949                                    AliasTy::new_from_args(
950                                        self.ctx.interner,
951                                        associated_ty.into(),
952                                        args,
953                                    ),
954                                ),
955                                false,
956                            )
957                            .map(|(pred, _)| (pred, GenericPredicateSource::AssocTyBound)),
958                    );
959                }
960                predicates
961            })
962        })
963    }

1016/// Returns true if there was an error.
1017fn check_generic_args_len<'db>(
1018    args_and_bindings: Option<&HirGenericArgs>,
1019    def: GenericDefId,
1020    def_generics: &Generics,
1021    infer_args: bool,
1022    lifetime_elision: &LifetimeElisionKind<'db>,
1023    lowering_assoc_type_generics: bool,
1024    ctx: &mut impl GenericArgsLowerer<'db>,
1025) -> bool {
1026    let mut had_error = false;
1027
1028    let (mut provided_lifetimes_count, mut provided_types_and_consts_count) = (0usize, 0usize);
1029    if let Some(args_and_bindings) = args_and_bindings {
1030        let args_no_self = &args_and_bindings.args[usize::from(args_and_bindings.has_self_type)..];
1031        for arg in args_no_self {
1032            match arg {
1033                HirGenericArg::Lifetime(_) => provided_lifetimes_count += 1,
1034                HirGenericArg::Type(_) | HirGenericArg::Const(_) => {
1035                    provided_types_and_consts_count += 1
1036                }
1037            }
1038        }
1039    }
1040
1041    let lifetime_args_len = def_generics.len_lifetimes_self();
1042    if provided_lifetimes_count == 0
1043        && lifetime_args_len > 0
1044        && (!lowering_assoc_type_generics || infer_args)
1045    {
1046        // In generic associated types, we never allow inferring the lifetimes, but only in type context, that is
1047        // when `infer_args == false`. In expression/pattern context we always allow inferring them, even for GATs.
1048        match lifetime_elision {
1049            &LifetimeElisionKind::AnonymousCreateParameter { report_in_path } => {
1050                ctx.report_elided_lifetimes_in_path(def, lifetime_args_len as u32, report_in_path);
1051                had_error |= report_in_path;
1052            }
1053            LifetimeElisionKind::AnonymousReportError => {
1054                ctx.report_missing_lifetime(def, lifetime_args_len as u32);
1055                had_error = true
1056            }
1057            LifetimeElisionKind::ElisionFailure => {
1058                ctx.report_elision_failure(def, lifetime_args_len as u32);
1059                had_error = true;
1060            }
1061            LifetimeElisionKind::StaticIfNoLifetimeInScope { only_lint: _ } => {
1062                // FIXME: Check there are other lifetimes in scope, and error/lint.
1063            }
1064            LifetimeElisionKind::Elided(_) => {
1065                ctx.report_elided_lifetimes_in_path(def, lifetime_args_len as u32, false);
1066            }
1067            LifetimeElisionKind::Infer => {
1068                // Allow eliding lifetimes.
1069            }
1070        }
1071    } else if lifetime_args_len != provided_lifetimes_count {
1072        ctx.report_len_mismatch(
1073            def,
1074            provided_lifetimes_count as u32,
1075            lifetime_args_len as u32,
1076            IncorrectGenericsLenKind::Lifetimes,
1077        );
1078        had_error = true;
1079    }
1080
1081    let defaults_count =
1082        def_generics.iter_self_type_or_consts().filter(|(_, param)| param.has_default()).count();
1083    let named_type_and_const_params_count = def_generics
1084        .iter_self_type_or_consts()
1085        .filter(|(_, param)| match param {
1086            TypeOrConstParamData::TypeParamData(param) => {
1087                param.provenance == TypeParamProvenance::TypeParamList
1088            }
1089            TypeOrConstParamData::ConstParamData(_) => true,
1090        })
1091        .count();
1092    let expected_max = named_type_and_const_params_count;
1093    let expected_min =
1094        if infer_args { 0 } else { named_type_and_const_params_count - defaults_count };
1095    if provided_types_and_consts_count < expected_min
1096        || expected_max < provided_types_and_consts_count
1097    {
1098        ctx.report_len_mismatch(
1099            def,
1100            provided_types_and_consts_count as u32,
1101            named_type_and_const_params_count as u32,
1102            IncorrectGenericsLenKind::TypesAndConsts,
1103        );
1104        had_error = true;
1105    }
1106
1107    had_error
1108}

1110pub(crate) fn substs_from_args_and_bindings<'db>(
1111    db: &'db dyn HirDatabase,
1112    store: &ExpressionStore,
1113    args_and_bindings: Option<&HirGenericArgs>,
1114    def: GenericDefId,
1115    mut infer_args: bool,
1116    lifetime_elision: LifetimeElisionKind<'db>,
1117    lowering_assoc_type_generics: bool,
1118    explicit_self_ty: Option<Ty<'db>>,
1119    ctx: &mut impl GenericArgsLowerer<'db>,
1120) -> GenericArgs<'db> {
1121    let interner = DbInterner::new_no_crate(db);
1122
1123    tracing::debug!(?args_and_bindings);
1124
1125    // Order is
1126    // - Parent parameters
1127    // - Optional Self parameter
1128    // - Lifetime parameters
1129    // - Type or Const parameters
1130    let def_generics = generics(db, def);
1131    let args_slice = args_and_bindings.map(|it| &*it.args).unwrap_or_default();
1132
1133    // We do not allow inference if there are specified args, i.e. we do not allow partial inference.
1134    let has_non_lifetime_args =
1135        args_slice.iter().any(|arg| !matches!(arg, HirGenericArg::Lifetime(_)));
1136    infer_args &= !has_non_lifetime_args;
1137
1138    let had_count_error = check_generic_args_len(
1139        args_and_bindings,
1140        def,
1141        &def_generics,
1142        infer_args,
1143        &lifetime_elision,
1144        lowering_assoc_type_generics,
1145        ctx,
1146    );
1147
1148    let mut substs = Vec::with_capacity(def_generics.len());
1149
1150    substs.extend(
1151        def_generics.iter_parent_id().enumerate().map(|(idx, id)| ctx.parent_arg(idx as u32, id)),
1152    );
1153
1154    let mut args = args_slice.iter().enumerate().peekable();
1155    let mut params = def_generics.iter_self().peekable();
1156
1157    // If we encounter a type or const when we expect a lifetime, we infer the lifetimes.
1158    // If we later encounter a lifetime, we know that the arguments were provided in the
1159    // wrong order. `force_infer_lt` records the type or const that forced lifetimes to be
1160    // inferred, so we can use it for diagnostics later.
1161    let mut force_infer_lt = None;
1162
1163    let has_self_arg = args_and_bindings.is_some_and(|it| it.has_self_type);
1164    // First, handle `Self` parameter. Consume it from the args if provided, otherwise from `explicit_self_ty`,
1165    // and lastly infer it.
1166    if let Some(&(
1167        self_param_id,
1168        self_param @ GenericParamDataRef::TypeParamData(TypeParamData {
1169            provenance: TypeParamProvenance::TraitSelf,
1170            ..
1171        }),
1172    )) = params.peek()
1173    {
1174        let self_ty = if has_self_arg {
1175            let (_, self_ty) = args.next().expect("has_self_type=true, should have Self type");
1176            ctx.provided_kind(self_param_id, self_param, self_ty)
1177        } else {
1178            explicit_self_ty.map(|it| it.into()).unwrap_or_else(|| {
1179                ctx.inferred_kind(def, self_param_id, self_param, infer_args, &substs)
1180            })
1181        };
1182        params.next();
1183        substs.push(self_ty);
1184    }
1185
1186    loop {
1187        // We're going to iterate through the generic arguments that the user
1188        // provided, matching them with the generic parameters we expect.
1189        // Mismatches can occur as a result of elided lifetimes, or for malformed
1190        // input. We try to handle both sensibly.
1191        match (args.peek(), params.peek()) {
1192            (Some(&(arg_idx, arg)), Some(&(param_id, param))) => match (arg, param) {
1193                (HirGenericArg::Type(_), GenericParamDataRef::TypeParamData(type_param))
1194                    if type_param.provenance == TypeParamProvenance::ArgumentImplTrait =>
1195                {
1196                    // Do not allow specifying `impl Trait` explicitly. We already err at that, but if we won't handle it here
1197                    // we will handle it as if it was specified, instead of inferring it.
1198                    substs.push(ctx.inferred_kind(def, param_id, param, infer_args, &substs));
1199                    params.next();
1200                }
1201                (HirGenericArg::Lifetime(_), GenericParamDataRef::LifetimeParamData(_))
1202                | (HirGenericArg::Type(_), GenericParamDataRef::TypeParamData(_))
1203                | (HirGenericArg::Const(_), GenericParamDataRef::ConstParamData(_)) => {
1204                    substs.push(ctx.provided_kind(param_id, param, arg));
1205                    args.next();
1206                    params.next();
1207                }
1208                (
1209                    HirGenericArg::Type(_) | HirGenericArg::Const(_),
1210                    GenericParamDataRef::LifetimeParamData(_),
1211                ) => {
1212                    // We expected a lifetime argument, but got a type or const
1213                    // argument. That means we're inferring the lifetime.
1214                    substs.push(ctx.inferred_kind(def, param_id, param, infer_args, &substs));
1215                    params.next();
1216                    force_infer_lt = Some((arg_idx as u32, param_id));
1217                }
1218                (HirGenericArg::Type(type_ref), GenericParamDataRef::ConstParamData(_)) => {
1219                    if let Some(konst) = type_looks_like_const(store, *type_ref) {
1220                        let GenericParamId::ConstParamId(param_id) = param_id else {
1221                            panic!("unmatching param kinds");
1222                        };
1223                        let const_ty = const_param_ty_query(db, param_id);
1224                        substs.push(ctx.provided_type_like_const(const_ty, konst).into());
1225                        args.next();
1226                        params.next();
1227                    } else {
1228                        // See the `_ => { ... }` branch.
1229                        if !had_count_error {
1230                            ctx.report_arg_mismatch(param_id, arg_idx as u32, has_self_arg);
1231                        }
1232                        while args.next().is_some() {}
1233                    }
1234                }
1235                _ => {
1236                    // We expected one kind of parameter, but the user provided
1237                    // another. This is an error. However, if we already know that
1238                    // the arguments don't match up with the parameters, we won't issue
1239                    // an additional error, as the user already knows what's wrong.
1240                    if !had_count_error {
1241                        ctx.report_arg_mismatch(param_id, arg_idx as u32, has_self_arg);
1242                    }
1243
1244                    // We've reported the error, but we want to make sure that this
1245                    // problem doesn't bubble down and create additional, irrelevant
1246                    // errors. In this case, we're simply going to ignore the argument
1247                    // and any following arguments. The rest of the parameters will be
1248                    // inferred.
1249                    while args.next().is_some() {}
1250                }
1251            },
1252
1253            (Some(&(_, arg)), None) => {
1254                // We should never be able to reach this point with well-formed input.
1255                // There are two situations in which we can encounter this issue.
1256                //
1257                //  1. The number of arguments is incorrect. In this case, an error
1258                //     will already have been emitted, and we can ignore it.
1259                //  2. We've inferred some lifetimes, which have been provided later (i.e.
1260                //     after a type or const). We want to throw an error in this case.
1261                if !had_count_error {
1262                    assert!(
1263                        matches!(arg, HirGenericArg::Lifetime(_)),
1264                        "the only possible situation here is incorrect lifetime order"
1265                    );
1266                    let (provided_arg_idx, param_id) =
1267                        force_infer_lt.expect("lifetimes ought to have been inferred");
1268                    ctx.report_arg_mismatch(param_id, provided_arg_idx, has_self_arg);
1269                }
1270
1271                break;
1272            }
1273
1274            (None, Some(&(param_id, param))) => {
1275                // If there are fewer arguments than parameters, it means we're inferring the remaining arguments.
1276                let param = if let GenericParamId::LifetimeParamId(_) = param_id {
1277                    match &lifetime_elision {
1278                        LifetimeElisionKind::ElisionFailure
1279                        | LifetimeElisionKind::AnonymousCreateParameter { report_in_path: true }
1280                        | LifetimeElisionKind::AnonymousReportError => {
1281                            assert!(had_count_error);
1282                            ctx.inferred_kind(def, param_id, param, infer_args, &substs)
1283                        }
1284                        LifetimeElisionKind::StaticIfNoLifetimeInScope { only_lint: _ } => {
1285                            Region::new_static(interner).into()
1286                        }
1287                        LifetimeElisionKind::Elided(lifetime) => (*lifetime).into(),
1288                        LifetimeElisionKind::AnonymousCreateParameter { report_in_path: false }
1289                        | LifetimeElisionKind::Infer => {
1290                            // FIXME: With `AnonymousCreateParameter`, we need to create a new lifetime parameter here
1291                            // (but this will probably be done in hir-def lowering instead).
1292                            ctx.inferred_kind(def, param_id, param, infer_args, &substs)
1293                        }
1294                    }
1295                } else {
1296                    ctx.inferred_kind(def, param_id, param, infer_args, &substs)
1297                };
1298                substs.push(param);
1299                params.next();
1300            }
1301
1302            (None, None) => break,
1303        }
1304    }
1305
1306    GenericArgs::new_from_slice(&substs)
1307}
1308
1309fn type_looks_like_const(
1310    store: &ExpressionStore,
1311    type_ref: TypeRefId,
1312) -> Option<TypeLikeConst<'_>> {
1313    // A path/`_` const will be parsed as a type, instead of a const, because when parsing/lowering
1314    // in hir-def we don't yet know the expected argument kind. rustc does this a bit differently,
1315    // when lowering to HIR it resolves the path, and if it doesn't resolve to the type namespace
1316    // it is lowered as a const. Our behavior could deviate from rustc when the value is resolvable
1317    // in both the type and value namespaces, but I believe we only allow more code.
1318    let type_ref = &store[type_ref];
1319    match type_ref {
1320        TypeRef::Path(path) => Some(TypeLikeConst::Path(path)),
1321        TypeRef::Placeholder => Some(TypeLikeConst::Infer),
1322        _ => None,
1323    }
1324}

1326fn unknown_subst<'db>(interner: DbInterner<'db>, def: impl Into<GenericDefId>) -> GenericArgs<'db> {
1327    let params = generics(interner.db(), def.into());
1328    GenericArgs::new_from_iter(
1329        interner,
1330        params.iter_id().map(|id| match id {
1331            GenericParamId::TypeParamId(_) => Ty::new_error(interner, ErrorGuaranteed).into(),
1332            GenericParamId::ConstParamId(id) => {
1333                unknown_const_as_generic(const_param_ty_query(interner.db(), id))
1334            }
1335            GenericParamId::LifetimeParamId(_) => Region::error(interner).into(),
1336        }),
1337    )
1338}
