38// We cannot use `&mut dyn FnMut()` because of lifetime issues, and we don't want to use `Box<dyn FnMut()>`
39// because of the allocation, so we create a lifetime-less callback, tailored for our needs.
40pub(crate) struct PathDiagnosticCallback<'a> {
41    pub(crate) data: CallbackData<'a>,
42    pub(crate) callback: fn(&CallbackData<'_>, &mut TyLoweringContext<'_>, PathLoweringDiagnostic),
43}

45pub(crate) struct PathLoweringContext<'a, 'b> {
46    ctx: &'a mut TyLoweringContext<'b>,
47    on_diagnostic: PathDiagnosticCallback<'a>,
48    path: &'a Path,
49    segments: PathSegments<'a>,
50    current_segment_idx: usize,
51    /// Contains the previous segment if `current_segment_idx == segments.len()`
52    current_or_prev_segment: PathSegment<'a>,
53}

56    #[inline]
57    pub(crate) fn new(
58        ctx: &'a mut TyLoweringContext<'b>,
59        on_diagnostic: PathDiagnosticCallback<'a>,
60        path: &'a Path,
61    ) -> Self {
62        let segments = path.segments();
63        let first_segment = segments.first().unwrap_or(PathSegment::MISSING);
64        Self {
65            ctx,
66            on_diagnostic,
67            path,
68            segments,
69            current_segment_idx: 0,
70            current_or_prev_segment: first_segment,
71        }
72    }

90    #[inline]
91    fn skip_resolved_segment(&mut self) {
92        if !matches!(self.path, Path::LangItem(..)) {
93            // In lang items, the resolved "segment" is not one of the segments. Perhaps we should've put it
94            // point at -1, but I don't feel this is clearer.
95            self.current_segment_idx += 1;
96        }
97        self.update_current_segment();
98    }

102        self.current_or_prev_segment =
103            self.segments.get(self.current_segment_idx).unwrap_or(self.current_or_prev_segment);
104    }

120    pub(crate) fn lower_ty_relative_path(
121        &mut self,
122        ty: Ty,
123        // We need the original resolution to lower `Self::AssocTy` correctly
124        res: Option<TypeNs>,
125    ) -> (Ty, Option<TypeNs>) {
126        match self.segments.len() - self.current_segment_idx {
127            0 => (ty, res),
128            1 => {
129                // resolve unselected assoc types
130                (self.select_associated_type(res), None)
131            }
132            _ => {
133                // FIXME report error (ambiguous associated type)
134                (TyKind::Error.intern(Interner), None)
135            }
136        }
137    }

139    fn prohibit_parenthesized_generic_args(&mut self) -> bool {
140        if let Some(generic_args) = self.current_or_prev_segment.args_and_bindings {
141            match generic_args.parenthesized {
142                GenericArgsParentheses::No => {}
143                GenericArgsParentheses::ReturnTypeNotation | GenericArgsParentheses::ParenSugar => {
144                    let segment = self.current_segment_u32();
145                    self.on_diagnostic(
146                        PathLoweringDiagnostic::ParenthesizedGenericArgsWithoutFnTrait { segment },
147                    );
148                    return true;
149                }
150            }
151        }
152        false
153    }

155    // When calling this, the current segment is the resolved segment (we don't advance it yet).
156    pub(crate) fn lower_partly_resolved_path(
157        &mut self,
158        resolution: TypeNs,
159        infer_args: bool,
160    ) -> (Ty, Option<TypeNs>) {
161        let remaining_segments = self.segments.skip(self.current_segment_idx + 1);
162
163        let ty = match resolution {
164            TypeNs::TraitId(trait_) => {
165                let ty = match remaining_segments.len() {
166                    1 => {
167                        let trait_ref = self.lower_trait_ref_from_resolved_path(
168                            trait_,
169                            TyKind::Error.intern(Interner),
170                        );
171
172                        self.skip_resolved_segment();
173                        let segment = self.current_or_prev_segment;
174                        let found =
175                            self.ctx.db.trait_items(trait_).associated_type_by_name(segment.name);
176
177                        match found {
178                            Some(associated_ty) => {
179                                // FIXME: `substs_from_path_segment()` pushes `TyKind::Error` for every parent
180                                // generic params. It's inefficient to splice the `Substitution`s, so we may want
181                                // that method to optionally take parent `Substitution` as we already know them at
182                                // this point (`trait_ref.substitution`).
183                                let substitution = self.substs_from_path_segment(
184                                    associated_ty.into(),
185                                    false,
186                                    None,
187                                );
188                                let len_self =
189                                    generics(self.ctx.db.upcast(), associated_ty.into()).len_self();
190                                let substitution = Substitution::from_iter(
191                                    Interner,
192                                    substitution
193                                        .iter(Interner)
194                                        .take(len_self)
195                                        .chain(trait_ref.substitution.iter(Interner)),
196                                );
197                                TyKind::Alias(AliasTy::Projection(ProjectionTy {
198                                    associated_ty_id: to_assoc_type_id(associated_ty),
199                                    substitution,
200                                }))
201                                .intern(Interner)
202                            }
203                            None => {
204                                // FIXME: report error (associated type not found)
205                                TyKind::Error.intern(Interner)
206                            }
207                        }
208                    }
209                    0 => {
210                        // Trait object type without dyn; this should be handled in upstream. See
211                        // `lower_path()`.
212                        stdx::never!("unexpected fully resolved trait path");
213                        TyKind::Error.intern(Interner)
214                    }
215                    _ => {
216                        // FIXME report error (ambiguous associated type)
217                        TyKind::Error.intern(Interner)
218                    }
219                };
220                return (ty, None);
221            }
222            TypeNs::TraitAliasId(_) => {
223                // FIXME(trait_alias): Implement trait alias.
224                return (TyKind::Error.intern(Interner), None);
225            }
226            TypeNs::GenericParam(param_id) => match self.ctx.type_param_mode {
227                ParamLoweringMode::Placeholder => {
228                    TyKind::Placeholder(to_placeholder_idx(self.ctx.db, param_id.into()))
229                }
230                ParamLoweringMode::Variable => {
231                    let idx = match self
232                        .ctx
233                        .generics()
234                        .expect("generics in scope")
235                        .type_or_const_param_idx(param_id.into())
236                    {
237                        None => {
238                            never!("no matching generics");
239                            return (TyKind::Error.intern(Interner), None);
240                        }
241                        Some(idx) => idx,
242                    };
243
244                    TyKind::BoundVar(BoundVar::new(self.ctx.in_binders, idx))
245                }
246            }
247            .intern(Interner),
248            TypeNs::SelfType(impl_id) => {
249                let generics = self.ctx.generics().expect("impl should have generic param scope");
250
251                match self.ctx.type_param_mode {
252                    ParamLoweringMode::Placeholder => {
253                        // `def` can be either impl itself or item within, and we need impl itself
254                        // now.
255                        let generics = generics.parent_or_self();
256                        let subst = generics.placeholder_subst(self.ctx.db);
257                        self.ctx.db.impl_self_ty(impl_id).substitute(Interner, &subst)
258                    }
259                    ParamLoweringMode::Variable => {
260                        let starting_from = match generics.def() {
261                            GenericDefId::ImplId(_) => 0,
262                            // `def` is an item within impl. We need to substitute `BoundVar`s but
263                            // remember that they are for parent (i.e. impl) generic params so they
264                            // come after our own params.
265                            _ => generics.len_self(),
266                        };
267                        TyBuilder::impl_self_ty(self.ctx.db, impl_id)
268                            .fill_with_bound_vars(self.ctx.in_binders, starting_from)
269                            .build()
270                    }
271                }
272            }
273            TypeNs::AdtSelfType(adt) => {
274                let generics = generics(self.ctx.db.upcast(), adt.into());
275                let substs = match self.ctx.type_param_mode {
276                    ParamLoweringMode::Placeholder => generics.placeholder_subst(self.ctx.db),
277                    ParamLoweringMode::Variable => {
278                        generics.bound_vars_subst(self.ctx.db, self.ctx.in_binders)
279                    }
280                };
281                self.ctx.db.ty(adt.into()).substitute(Interner, &substs)
282            }
283
284            TypeNs::AdtId(it) => self.lower_path_inner(it.into(), infer_args),
285            TypeNs::BuiltinType(it) => self.lower_path_inner(it.into(), infer_args),
286            TypeNs::TypeAliasId(it) => self.lower_path_inner(it.into(), infer_args),
287            // FIXME: report error
288            TypeNs::EnumVariantId(_) => return (TyKind::Error.intern(Interner), None),
289        };
290
291        self.skip_resolved_segment();
292        self.lower_ty_relative_path(ty, Some(resolution))
293    }

295    fn handle_type_ns_resolution(&mut self, resolution: &TypeNs) {
296        let mut prohibit_generics_on_resolved = |reason| {
297            if self.current_or_prev_segment.args_and_bindings.is_some() {
298                let segment = self.current_segment_u32();
299                self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
300                    segment,
301                    reason,
302                });
303            }
304        };
305
306        match resolution {
307            TypeNs::SelfType(_) => {
308                prohibit_generics_on_resolved(GenericArgsProhibitedReason::SelfTy)
309            }
310            TypeNs::GenericParam(_) => {
311                prohibit_generics_on_resolved(GenericArgsProhibitedReason::TyParam)
312            }
313            TypeNs::AdtSelfType(_) => {
314                prohibit_generics_on_resolved(GenericArgsProhibitedReason::SelfTy)
315            }
316            TypeNs::BuiltinType(_) => {
317                prohibit_generics_on_resolved(GenericArgsProhibitedReason::PrimitiveTy)
318            }
319            TypeNs::AdtId(_)
320            | TypeNs::EnumVariantId(_)
321            | TypeNs::TypeAliasId(_)
322            | TypeNs::TraitId(_)
323            | TypeNs::TraitAliasId(_) => {}
324        }
325    }

328        let (res, unresolved) = self.resolve_path_in_type_ns()?;
329        if unresolved.is_some() {
330            return None;
331        }
332        Some(res)
333    }

335    pub(crate) fn resolve_path_in_type_ns(&mut self) -> Option<(TypeNs, Option<usize>)> {
336        let (resolution, remaining_index, _, prefix_info) = self
337            .ctx
338            .resolver
339            .resolve_path_in_type_ns_with_prefix_info(self.ctx.db.upcast(), self.path)?;
340
341        let segments = self.segments;
342        if segments.is_empty() || matches!(self.path, Path::LangItem(..)) {
343            // `segments.is_empty()` can occur with `self`.
344            return Some((resolution, remaining_index));
345        }
346
347        let (module_segments, resolved_segment_idx, enum_segment) = match remaining_index {
348            None if prefix_info.enum_variant => {
349                (segments.strip_last_two(), segments.len() - 1, Some(segments.len() - 2))
350            }
351            None => (segments.strip_last(), segments.len() - 1, None),
352            Some(i) => (segments.take(i - 1), i - 1, None),
353        };
354
355        self.current_segment_idx = resolved_segment_idx;
356        self.current_or_prev_segment =
357            segments.get(resolved_segment_idx).expect("should have resolved segment");
358
359        if matches!(self.path, Path::BarePath(..)) {
360            // Bare paths cannot have generics, so skip them as an optimization.
361            return Some((resolution, remaining_index));
362        }
363
364        for (i, mod_segment) in module_segments.iter().enumerate() {
365            if mod_segment.args_and_bindings.is_some() {
366                self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
367                    segment: i as u32,
368                    reason: GenericArgsProhibitedReason::Module,
369                });
370            }
371        }
372
373        if let Some(enum_segment) = enum_segment {
374            if segments.get(enum_segment).is_some_and(|it| it.args_and_bindings.is_some())
375                && segments.get(enum_segment + 1).is_some_and(|it| it.args_and_bindings.is_some())
376            {
377                self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
378                    segment: (enum_segment + 1) as u32,
379                    reason: GenericArgsProhibitedReason::EnumVariant,
380                });
381            }
382        }
383
384        self.handle_type_ns_resolution(&resolution);
385
386        Some((resolution, remaining_index))
387    }

389    pub(crate) fn resolve_path_in_value_ns(
390        &mut self,
391        hygiene_id: HygieneId,
392    ) -> Option<ResolveValueResult> {
393        let (res, prefix_info) = self.ctx.resolver.resolve_path_in_value_ns_with_prefix_info(
394            self.ctx.db.upcast(),
395            self.path,
396            hygiene_id,
397        )?;
398
399        let segments = self.segments;
400        if segments.is_empty() || matches!(self.path, Path::LangItem(..)) {
401            // `segments.is_empty()` can occur with `self`.
402            return Some(res);
403        }
404
405        let (mod_segments, enum_segment, resolved_segment_idx) = match res {
406            ResolveValueResult::Partial(_, unresolved_segment, _) => {
407                (segments.take(unresolved_segment - 1), None, unresolved_segment - 1)
408            }
409            ResolveValueResult::ValueNs(ValueNs::EnumVariantId(_), _)
410                if prefix_info.enum_variant =>
411            {
412                (segments.strip_last_two(), segments.len().checked_sub(2), segments.len() - 1)
413            }
414            ResolveValueResult::ValueNs(..) => (segments.strip_last(), None, segments.len() - 1),
415        };
416
417        self.current_segment_idx = resolved_segment_idx;
418        self.current_or_prev_segment =
419            segments.get(resolved_segment_idx).expect("should have resolved segment");
420
421        for (i, mod_segment) in mod_segments.iter().enumerate() {
422            if mod_segment.args_and_bindings.is_some() {
423                self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
424                    segment: i as u32,
425                    reason: GenericArgsProhibitedReason::Module,
426                });
427            }
428        }
429
430        if let Some(enum_segment) = enum_segment {
431            if segments.get(enum_segment).is_some_and(|it| it.args_and_bindings.is_some())
432                && segments.get(enum_segment + 1).is_some_and(|it| it.args_and_bindings.is_some())
433            {
434                self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
435                    segment: (enum_segment + 1) as u32,
436                    reason: GenericArgsProhibitedReason::EnumVariant,
437                });
438            }
439        }
440
441        match &res {
442            ResolveValueResult::ValueNs(resolution, _) => {
443                let resolved_segment_idx = self.current_segment_u32();
444                let resolved_segment = self.current_or_prev_segment;
445
446                let mut prohibit_generics_on_resolved = |reason| {
447                    if resolved_segment.args_and_bindings.is_some() {
448                        self.on_diagnostic(PathLoweringDiagnostic::GenericArgsProhibited {
449                            segment: resolved_segment_idx,
450                            reason,
451                        });
452                    }
453                };
454
455                match resolution {
456                    ValueNs::ImplSelf(_) => {
457                        prohibit_generics_on_resolved(GenericArgsProhibitedReason::SelfTy)
458                    }
459                    // FIXME: rustc generates E0107 (incorrect number of generic arguments) and not
460                    // E0109 (generic arguments provided for a type that doesn't accept them) for
461                    // consts and statics, presumably as a defense against future in which consts
462                    // and statics can be generic, or just because it was easier for rustc implementors.
463                    // That means we'll show the wrong error code. Because of us it's easier to do it
464                    // this way :)
465                    ValueNs::GenericParam(_) | ValueNs::ConstId(_) => {
466                        prohibit_generics_on_resolved(GenericArgsProhibitedReason::Const)
467                    }
468                    ValueNs::StaticId(_) => {
469                        prohibit_generics_on_resolved(GenericArgsProhibitedReason::Static)
470                    }
471                    ValueNs::FunctionId(_) | ValueNs::StructId(_) | ValueNs::EnumVariantId(_) => {}
472                    ValueNs::LocalBinding(_) => {}
473                }
474            }
475            ResolveValueResult::Partial(resolution, _, _) => {
476                self.handle_type_ns_resolution(resolution);
477            }
478        };
479        Some(res)
480    }

482    fn select_associated_type(&mut self, res: Option<TypeNs>) -> Ty {
483        let Some((generics, res)) = self.ctx.generics().zip(res) else {
484            return TyKind::Error.intern(Interner);
485        };
486        let segment = self.current_or_prev_segment;
487        let ty = named_associated_type_shorthand_candidates(
488            self.ctx.db,
489            generics.def(),
490            res,
491            Some(segment.name.clone()),
492            move |name, t, associated_ty| {
493                let generics = self.ctx.generics().unwrap();
494
495                if name != segment.name {
496                    return None;
497                }
498
499                let parent_subst = t.substitution.clone();
500                let parent_subst = match self.ctx.type_param_mode {
501                    ParamLoweringMode::Placeholder => {
502                        // if we're lowering to placeholders, we have to put them in now.
503                        let s = generics.placeholder_subst(self.ctx.db);
504                        s.apply(parent_subst, Interner)
505                    }
506                    ParamLoweringMode::Variable => {
507                        // We need to shift in the bound vars, since
508                        // `named_associated_type_shorthand_candidates` does not do that.
509                        parent_subst.shifted_in_from(Interner, self.ctx.in_binders)
510                    }
511                };
512
513                // FIXME: `substs_from_path_segment()` pushes `TyKind::Error` for every parent
514                // generic params. It's inefficient to splice the `Substitution`s, so we may want
515                // that method to optionally take parent `Substitution` as we already know them at
516                // this point (`t.substitution`).
517                let substs = self.substs_from_path_segment(associated_ty.into(), false, None);
518
519                let len_self =
520                    crate::generics::generics(self.ctx.db.upcast(), associated_ty.into())
521                        .len_self();
522
523                let substs = Substitution::from_iter(
524                    Interner,
525                    substs.iter(Interner).take(len_self).chain(parent_subst.iter(Interner)),
526                );
527
528                Some(
529                    TyKind::Alias(AliasTy::Projection(ProjectionTy {
530                        associated_ty_id: to_assoc_type_id(associated_ty),
531                        substitution: substs,
532                    }))
533                    .intern(Interner),
534                )
535            },
536        );
537
538        ty.unwrap_or_else(|| TyKind::Error.intern(Interner))
539    }

541    fn lower_path_inner(&mut self, typeable: TyDefId, infer_args: bool) -> Ty {
542        let generic_def = match typeable {
543            TyDefId::BuiltinType(builtin) => return TyBuilder::builtin(builtin),
544            TyDefId::AdtId(it) => it.into(),
545            TyDefId::TypeAliasId(it) => it.into(),
546        };
547        let substs = self.substs_from_path_segment(generic_def, infer_args, None);
548        self.ctx.db.ty(typeable).substitute(Interner, &substs)
549    }

551    /// Collect generic arguments from a path into a `Substs`. See also
552    /// `create_substs_for_ast_path` and `def_to_ty` in rustc.
553    pub(crate) fn substs_from_path(
554        &mut self,
555        // Note that we don't call `db.value_type(resolved)` here,
556        // `ValueTyDefId` is just a convenient way to pass generics and
557        // special-case enum variants
558        resolved: ValueTyDefId,
559        infer_args: bool,
560    ) -> Substitution {
561        let prev_current_segment_idx = self.current_segment_idx;
562        let prev_current_segment = self.current_or_prev_segment;
563
564        let generic_def = match resolved {
565            ValueTyDefId::FunctionId(it) => it.into(),
566            ValueTyDefId::StructId(it) => it.into(),
567            ValueTyDefId::UnionId(it) => it.into(),
568            ValueTyDefId::ConstId(it) => it.into(),
569            ValueTyDefId::StaticId(_) => return Substitution::empty(Interner),
570            ValueTyDefId::EnumVariantId(var) => {
571                // the generic args for an enum variant may be either specified
572                // on the segment referring to the enum, or on the segment
573                // referring to the variant. So `Option::<T>::None` and
574                // `Option::None::<T>` are both allowed (though the former is
575                // FIXME: This isn't strictly correct, enum variants may be used not through the enum
576                // (via `use Enum::Variant`). The resolver returns whether they were, but we don't have its result
577                // available here. The worst that can happen is that we will show some confusing diagnostics to the user,
578                // if generics exist on the module and they don't match with the variant.
579                // preferred). See also `def_ids_for_path_segments` in rustc.
580                //
581                // `wrapping_sub(1)` will return a number which `get` will return None for if current_segment_idx<2.
582                // This simplifies the code a bit.
583                let penultimate_idx = self.current_segment_idx.wrapping_sub(1);
584                let penultimate = self.segments.get(penultimate_idx);
585                if let Some(penultimate) = penultimate {
586                    if self.current_or_prev_segment.args_and_bindings.is_none()
587                        && penultimate.args_and_bindings.is_some()
588                    {
589                        self.current_segment_idx = penultimate_idx;
590                        self.current_or_prev_segment = penultimate;
591                    }
592                }
593                var.lookup(self.ctx.db.upcast()).parent.into()
594            }
595        };
596        let result = self.substs_from_path_segment(generic_def, infer_args, None);
597        self.current_segment_idx = prev_current_segment_idx;
598        self.current_or_prev_segment = prev_current_segment;
599        result
600    }

602    pub(crate) fn substs_from_path_segment(
603        &mut self,
604        def: GenericDefId,
605        infer_args: bool,
606        explicit_self_ty: Option<Ty>,
607    ) -> Substitution {
608        let prohibit_parens = match def {
609            GenericDefId::TraitId(trait_) => {
610                // RTN is prohibited anyways if we got here.
611                let is_rtn =
612                    self.current_or_prev_segment.args_and_bindings.is_some_and(|generics| {
613                        generics.parenthesized == GenericArgsParentheses::ReturnTypeNotation
614                    });
615                let is_fn_trait =
616                    !self.ctx.db.trait_data(trait_).flags.contains(TraitFlags::RUSTC_PAREN_SUGAR);
617                is_rtn || is_fn_trait
618            }
619            _ => true,
620        };
621        if prohibit_parens && self.prohibit_parenthesized_generic_args() {
622            return TyBuilder::unknown_subst(self.ctx.db, def);
623        }
624
625        self.substs_from_args_and_bindings(
626            self.current_or_prev_segment.args_and_bindings,
627            def,
628            infer_args,
629            explicit_self_ty,
630        )
631    }

633    pub(super) fn substs_from_args_and_bindings(
634        &mut self,
635        args_and_bindings: Option<&GenericArgs>,
636        def: GenericDefId,
637        infer_args: bool,
638        explicit_self_ty: Option<Ty>,
639    ) -> Substitution {
640        // Order is
641        // - Optional Self parameter
642        // - Lifetime parameters
643        // - Type or Const parameters
644        // - Parent parameters
645        let def_generics = generics(self.ctx.db.upcast(), def);
646        let (
647            parent_params,
648            self_param,
649            type_params,
650            const_params,
651            impl_trait_params,
652            lifetime_params,
653        ) = def_generics.provenance_split();
654        let item_len =
655            self_param as usize + type_params + const_params + impl_trait_params + lifetime_params;
656        let total_len = parent_params + item_len;
657
658        let mut substs = Vec::new();
659
660        // we need to iterate the lifetime and type/const params separately as our order of them
661        // differs from the supplied syntax
662
663        let ty_error = || TyKind::Error.intern(Interner).cast(Interner);
664        let mut def_toc_iter = def_generics.iter_self_type_or_consts_id();
665        let fill_self_param = || {
666            if self_param {
667                let self_ty = explicit_self_ty.map(|x| x.cast(Interner)).unwrap_or_else(ty_error);
668
669                if let Some(id) = def_toc_iter.next() {
670                    assert!(matches!(id, GenericParamId::TypeParamId(_)));
671                    substs.push(self_ty);
672                }
673            }
674        };
675        let mut had_explicit_args = false;
676
677        if let Some(&GenericArgs { ref args, has_self_type, .. }) = args_and_bindings {
678            // Fill in the self param first
679            if has_self_type && self_param {
680                had_explicit_args = true;
681                if let Some(id) = def_toc_iter.next() {
682                    assert!(matches!(id, GenericParamId::TypeParamId(_)));
683                    had_explicit_args = true;
684                    if let GenericArg::Type(ty) = &args[0] {
685                        substs.push(self.ctx.lower_ty(*ty).cast(Interner));
686                    }
687                }
688            } else {
689                fill_self_param()
690            };
691
692            // Then fill in the supplied lifetime args, or error lifetimes if there are too few
693            // (default lifetimes aren't a thing)
694            for arg in args
695                .iter()
696                .filter_map(|arg| match arg {
697                    GenericArg::Lifetime(arg) => Some(self.ctx.lower_lifetime(arg)),
698                    _ => None,
699                })
700                .chain(iter::repeat(error_lifetime()))
701                .take(lifetime_params)
702            {
703                substs.push(arg.cast(Interner));
704            }
705
706            let skip = if has_self_type { 1 } else { 0 };
707            // Fill in supplied type and const args
708            // Note if non-lifetime args are provided, it should be all of them, but we can't rely on that
709            for (arg, id) in args
710                .iter()
711                .filter(|arg| !matches!(arg, GenericArg::Lifetime(_)))
712                .skip(skip)
713                .take(type_params + const_params)
714                .zip(def_toc_iter)
715            {
716                had_explicit_args = true;
717                let arg = generic_arg_to_chalk(
718                    self.ctx.db,
719                    id,
720                    arg,
721                    self.ctx,
722                    self.ctx.types_map,
723                    |ctx, type_ref| ctx.lower_ty(type_ref),
724                    |ctx, const_ref, ty| ctx.lower_const(const_ref, ty),
725                    |ctx, lifetime_ref| ctx.lower_lifetime(lifetime_ref),
726                );
727                substs.push(arg);
728            }
729        } else {
730            fill_self_param();
731        }
732
733        let param_to_err = |id| match id {
734            GenericParamId::ConstParamId(x) => {
735                unknown_const_as_generic(self.ctx.db.const_param_ty(x))
736            }
737            GenericParamId::TypeParamId(_) => ty_error(),
738            GenericParamId::LifetimeParamId(_) => error_lifetime().cast(Interner),
739        };
740        // handle defaults. In expression or pattern path segments without
741        // explicitly specified type arguments, missing type arguments are inferred
742        // (i.e. defaults aren't used).
743        // Generic parameters for associated types are not supposed to have defaults, so we just
744        // ignore them.
745        let is_assoc_ty = || match def {
746            GenericDefId::TypeAliasId(id) => {
747                matches!(id.lookup(self.ctx.db.upcast()).container, ItemContainerId::TraitId(_))
748            }
749            _ => false,
750        };
751        let fill_defaults = (!infer_args || had_explicit_args) && !is_assoc_ty();
752        if fill_defaults {
753            let defaults = &*self.ctx.db.generic_defaults(def);
754            let (item, _parent) = defaults.split_at(item_len);
755            let parent_from = item_len - substs.len();
756
757            let mut rem =
758                def_generics.iter_id().skip(substs.len()).map(param_to_err).collect::<Vec<_>>();
759            // Fill in defaults for type/const params
760            for (idx, default_ty) in item[substs.len()..].iter().enumerate() {
761                // each default can depend on the previous parameters
762                let substs_so_far = Substitution::from_iter(
763                    Interner,
764                    substs.iter().cloned().chain(rem[idx..].iter().cloned()),
765                );
766                substs.push(default_ty.clone().substitute(Interner, &substs_so_far));
767            }
768            // Fill in remaining parent params
769            substs.extend(rem.drain(parent_from..));
770        } else {
771            // Fill in remaining def params and parent params
772            substs.extend(def_generics.iter_id().skip(substs.len()).map(param_to_err));
773        }
774
775        assert_eq!(substs.len(), total_len, "expected {} substs, got {}", total_len, substs.len());
776        Substitution::from_iter(Interner, substs)
777    }

779    pub(crate) fn lower_trait_ref_from_resolved_path(
780        &mut self,
781        resolved: TraitId,
782        explicit_self_ty: Ty,
783    ) -> TraitRef {
784        let substs = self.trait_ref_substs_from_path(resolved, explicit_self_ty);
785        TraitRef { trait_id: to_chalk_trait_id(resolved), substitution: substs }
786    }
787
788    fn trait_ref_substs_from_path(
789        &mut self,
790        resolved: TraitId,
791        explicit_self_ty: Ty,
792    ) -> Substitution {
793        self.substs_from_path_segment(resolved.into(), false, Some(explicit_self_ty))
794    }

796    pub(super) fn assoc_type_bindings_from_type_bound<'c>(
797        mut self,
798        bound: &'c TypeBound,
799        trait_ref: TraitRef,
800    ) -> Option<impl Iterator<Item = QuantifiedWhereClause> + use<'a, 'b, 'c>> {
801        self.current_or_prev_segment.args_and_bindings.map(|args_and_bindings| {
802            args_and_bindings.bindings.iter().flat_map(move |binding| {
803                let found = associated_type_by_name_including_super_traits(
804                    self.ctx.db,
805                    trait_ref.clone(),
806                    &binding.name,
807                );
808                let (super_trait_ref, associated_ty) = match found {
809                    None => return SmallVec::new(),
810                    Some(t) => t,
811                };
812                // FIXME: `substs_from_path_segment()` pushes `TyKind::Error` for every parent
813                // generic params. It's inefficient to splice the `Substitution`s, so we may want
814                // that method to optionally take parent `Substitution` as we already know them at
815                // this point (`super_trait_ref.substitution`).
816                let substitution = self.substs_from_args_and_bindings(
817                    binding.args.as_ref(),
818                    associated_ty.into(),
819                    false, // this is not relevant
820                    Some(super_trait_ref.self_type_parameter(Interner)),
821                );
822                let self_params = generics(self.ctx.db.upcast(), associated_ty.into()).len_self();
823                let substitution = Substitution::from_iter(
824                    Interner,
825                    substitution
826                        .iter(Interner)
827                        .take(self_params)
828                        .chain(super_trait_ref.substitution.iter(Interner)),
829                );
830                let projection_ty = ProjectionTy {
831                    associated_ty_id: to_assoc_type_id(associated_ty),
832                    substitution,
833                };
834                let mut predicates: SmallVec<[_; 1]> = SmallVec::with_capacity(
835                    binding.type_ref.as_ref().map_or(0, |_| 1) + binding.bounds.len(),
836                );
837                if let Some(type_ref) = binding.type_ref {
838                    match (&self.ctx.types_map[type_ref], self.ctx.impl_trait_mode.mode) {
839                        (TypeRef::ImplTrait(_), ImplTraitLoweringMode::Disallowed) => (),
840                        (_, ImplTraitLoweringMode::Disallowed | ImplTraitLoweringMode::Opaque) => {
841                            let ty = self.ctx.lower_ty(type_ref);
842                            let alias_eq =
843                                AliasEq { alias: AliasTy::Projection(projection_ty.clone()), ty };
844                            predicates
845                                .push(crate::wrap_empty_binders(WhereClause::AliasEq(alias_eq)));
846                        }
847                        (_, ImplTraitLoweringMode::Param | ImplTraitLoweringMode::Variable) => {
848                            // Find the generic index for the target of our `bound`
849                            let target_param_idx =
850                                self.ctx.resolver.where_predicates_in_scope().find_map(
851                                    |(p, (_, types_map))| match p {
852                                        WherePredicate::TypeBound {
853                                            target: WherePredicateTypeTarget::TypeOrConstParam(idx),
854                                            bound: b,
855                                        } if std::ptr::eq::<TypesMap>(
856                                            self.ctx.types_map,
857                                            types_map,
858                                        ) && bound == b =>
859                                        {
860                                            Some(idx)
861                                        }
862                                        _ => None,
863                                    },
864                                );
865                            let ty = if let Some(target_param_idx) = target_param_idx {
866                                let mut counter = 0;
867                                let generics = self.ctx.generics().expect("generics in scope");
868                                for (idx, data) in generics.iter_self_type_or_consts() {
869                                    // Count the number of `impl Trait` things that appear before
870                                    // the target of our `bound`.
871                                    // Our counter within `impl_trait_mode` should be that number
872                                    // to properly lower each types within `type_ref`
873                                    if data.type_param().is_some_and(|p| {
874                                        p.provenance == TypeParamProvenance::ArgumentImplTrait
875                                    }) {
876                                        counter += 1;
877                                    }
878                                    if idx == *target_param_idx {
879                                        break;
880                                    }
881                                }
882                                let mut ext = TyLoweringContext::new_maybe_unowned(
883                                    self.ctx.db,
884                                    self.ctx.resolver,
885                                    self.ctx.types_map,
886                                    self.ctx.types_source_map,
887                                    self.ctx.owner,
888                                )
889                                .with_type_param_mode(self.ctx.type_param_mode);
890                                match self.ctx.impl_trait_mode.mode {
891                                    ImplTraitLoweringMode::Param => {
892                                        ext.impl_trait_mode =
893                                            ImplTraitLoweringState::param(counter);
894                                    }
895                                    ImplTraitLoweringMode::Variable => {
896                                        ext.impl_trait_mode =
897                                            ImplTraitLoweringState::variable(counter);
898                                    }
899                                    _ => unreachable!(),
900                                }
901                                let ty = ext.lower_ty(type_ref);
902                                self.ctx.diagnostics.extend(ext.diagnostics);
903                                ty
904                            } else {
905                                self.ctx.lower_ty(type_ref)
906                            };
907
908                            let alias_eq =
909                                AliasEq { alias: AliasTy::Projection(projection_ty.clone()), ty };
910                            predicates
911                                .push(crate::wrap_empty_binders(WhereClause::AliasEq(alias_eq)));
912                        }
913                    }
914                }
915                for bound in binding.bounds.iter() {
916                    predicates.extend(self.ctx.lower_type_bound(
917                        bound,
918                        TyKind::Alias(AliasTy::Projection(projection_ty.clone())).intern(Interner),
919                        false,
920                    ));
921                }
922                predicates
923            })
924        })
925    }
