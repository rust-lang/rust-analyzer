117#[derive(Default)]
118enum BoundsFormattingCtx<'db> {
119    Entered {
120        /// We can have recursive bounds like the following case:
121        /// ```ignore
122        /// where
123        ///     T: Foo,
124        ///     T::FooAssoc: Baz<<T::FooAssoc as Bar>::BarAssoc> + Bar
125        /// ```
126        /// So, record the projection types met while formatting bounds and
127        //. prevent recursing into their bounds to avoid infinite loops.
128        projection_tys_met: FxHashSet<AliasTy<'db>>,
129    },
130    #[default]
131    Exited,
132}
133
134impl<'db> BoundsFormattingCtx<'db> {
135    fn contains(&self, proj: &AliasTy<'db>) -> bool {
136        match self {
137            BoundsFormattingCtx::Entered { projection_tys_met } => {
138                projection_tys_met.contains(proj)
139            }
140            BoundsFormattingCtx::Exited => false,
141        }
142    }
143}

145impl<'db> HirFormatter<'_, 'db> {
146    fn start_location_link(&mut self, location: ModuleDefId) {
147        self.fmt.start_location_link(location);
148    }
149
150    fn end_location_link(&mut self) {
151        self.fmt.end_location_link();
152    }
153
154    fn format_bounds_with<T, F: FnOnce(&mut Self) -> T>(
155        &mut self,
156        target: AliasTy<'db>,
157        format_bounds: F,
158    ) -> T {
159        match self.bounds_formatting_ctx {
160            BoundsFormattingCtx::Entered { ref mut projection_tys_met } => {
161                projection_tys_met.insert(target);
162                format_bounds(self)
163            }
164            BoundsFormattingCtx::Exited => {
165                let mut projection_tys_met = FxHashSet::default();
166                projection_tys_met.insert(target);
167                self.bounds_formatting_ctx = BoundsFormattingCtx::Entered { projection_tys_met };
168                let res = format_bounds(self);
169                // Since we want to prevent only the infinite recursions in bounds formatting
170                // and do not want to skip formatting of other separate bounds, clear context
171                // when exiting the formatting of outermost bounds
172                self.bounds_formatting_ctx = BoundsFormattingCtx::Exited;
173                res
174            }
175        }
176    }
177
178    fn render_region(&self, lifetime: Region<'db>) -> bool {
179        match self.display_lifetimes {
180            DisplayLifetime::Always => true,
181            DisplayLifetime::OnlyStatic => matches!(lifetime.kind(), RegionKind::ReStatic),
182            DisplayLifetime::OnlyNamed => {
183                matches!(lifetime.kind(), RegionKind::ReEarlyParam(_))
184            }
185            DisplayLifetime::OnlyNamedOrStatic => {
186                matches!(lifetime.kind(), RegionKind::ReStatic | RegionKind::ReEarlyParam(_))
187            }
188            DisplayLifetime::Never => false,
189        }
190    }
191}

395    pub fn write_joined<T: HirDisplay<'db>>(
396        &mut self,
397        iter: impl IntoIterator<Item = T>,
398        sep: &str,
399    ) -> Result<(), HirDisplayError> {
400        let mut first = true;
401        for e in iter {
402            if !first {
403                write!(self, "{sep}")?;
404            }
405            first = false;
406
407            // Abbreviate multiple omitted types with a single ellipsis.
408            if self.should_truncate() {
409                return write!(self, "{TYPE_HINT_TRUNCATION}");
410            }
411
412            e.hir_fmt(self)?;
413        }
414        Ok(())
415    }
416
417    /// This allows using the `write!` macro directly with a `HirFormatter`.
418    pub fn write_fmt(&mut self, args: fmt::Arguments<'_>) -> Result<(), HirDisplayError> {
419        // We write to a buffer first to track output size
420        self.buf.clear();
421        fmt::write(&mut self.buf, args)?;
422        self.curr_size += self.buf.len();
423
424        // Then we write to the internal formatter from the buffer
425        self.fmt.write_str(&self.buf).map_err(HirDisplayError::from)
426    }

429        self.fmt.write_str(s)?;
430        Ok(())
431    }

434        self.fmt.write_char(c)?;
435        Ok(())
436    }

609fn write_projection<'db>(
610    f: &mut HirFormatter<'_, 'db>,
611    alias: &AliasTy<'db>,
612) -> Result<(), HirDisplayError> {
613    if f.should_truncate() {
614        return write!(f, "{TYPE_HINT_TRUNCATION}");
615    }
616    let trait_ref = alias.trait_ref(f.interner);
617    let self_ty = trait_ref.self_ty();
618
619    // if we are projection on a type parameter, check if the projection target has bounds
620    // itself, if so, we render them directly as `impl Bound` instead of the less useful
621    // `<Param as Trait>::Assoc`
622    if !f.display_kind.is_source_code()
623        && let TyKind::Param(param) = self_ty.kind()
624        && !f.bounds_formatting_ctx.contains(alias)
625    {
626        // FIXME: We shouldn't use `param.id`, it should be removed. We should know the
627        // `GenericDefId` from the formatted type (store it inside the `HirFormatter`).
628        let bounds =
629            f.db.generic_predicates(param.id.parent())
630                .instantiate_identity()
631                .into_iter()
632                .flatten()
633                .filter(|wc| {
634                    let ty = match wc.kind().skip_binder() {
635                        ClauseKind::Trait(tr) => tr.self_ty(),
636                        ClauseKind::TypeOutlives(t) => t.0,
637                        _ => return false,
638                    };
639                    let TyKind::Alias(AliasTyKind::Projection, a) = ty.kind() else {
640                        return false;
641                    };
642                    a == *alias
643                })
644                .collect::<Vec<_>>();
645        if !bounds.is_empty() {
646            return f.format_bounds_with(*alias, |f| {
647                write_bounds_like_dyn_trait_with_prefix(
648                    f,
649                    "impl",
650                    Either::Left(Ty::new_alias(f.interner, AliasTyKind::Projection, *alias)),
651                    &bounds,
652                    SizedByDefault::NotSized,
653                )
654            });
655        }
656    }
657
658    write!(f, "<")?;
659    self_ty.hir_fmt(f)?;
660    write!(f, " as ")?;
661    trait_ref.hir_fmt(f)?;
662    write!(
663        f,
664        ">::{}",
665        f.db.type_alias_signature(alias.def_id.expect_type_alias()).name.display(f.db, f.edition())
666    )?;
667    let proj_params = &alias.args.as_slice()[trait_ref.args.len()..];
668    hir_fmt_generics(f, proj_params, None, None)
669}

681impl<'db> HirDisplay<'db> for Const<'db> {
682    fn hir_fmt(&self, f: &mut HirFormatter<'_, 'db>) -> Result<(), HirDisplayError> {
683        match self.kind() {
684            ConstKind::Placeholder(_) => write!(f, "<placeholder>"),
685            ConstKind::Bound(BoundVarIndexKind::Bound(db), bound_const) => {
686                write!(f, "?{}.{}", db.as_u32(), bound_const.var.as_u32())
687            }
688            ConstKind::Bound(BoundVarIndexKind::Canonical, bound_const) => {
689                write!(f, "?c.{}", bound_const.var.as_u32())
690            }
691            ConstKind::Infer(..) => write!(f, "#c#"),
692            ConstKind::Param(param) => {
693                let generics = generics(f.db, param.id.parent());
694                let param_data = &generics[param.id.local_id()];
695                write!(f, "{}", param_data.name().unwrap().display(f.db, f.edition()))?;
696                Ok(())
697            }
698            ConstKind::Value(const_bytes) => render_const_scalar(
699                f,
700                &const_bytes.value.inner().memory,
701                &const_bytes.value.inner().memory_map,
702                const_bytes.ty,
703            ),
704            ConstKind::Unevaluated(unev) => {
705                let c = match unev.def {
706                    SolverDefId::ConstId(id) => GeneralConstId::ConstId(id),
707                    SolverDefId::StaticId(id) => GeneralConstId::StaticId(id),
708                    _ => unreachable!(),
709                };
710                write!(f, "{}", c.name(f.db))?;
711                hir_fmt_generics(f, unev.args.as_slice(), c.generic_def(f.db), None)?;
712                Ok(())
713            }
714            ConstKind::Error(..) => f.write_char('_'),
715            ConstKind::Expr(..) => write!(f, "<const-expr>"),
716        }
717    }
718}

720fn render_const_scalar<'db>(
721    f: &mut HirFormatter<'_, 'db>,
722    b: &[u8],
723    memory_map: &MemoryMap<'db>,
724    ty: Ty<'db>,
725) -> Result<(), HirDisplayError> {
726    let trait_env = TraitEnvironment::empty(f.krate());
727    let infcx = f.interner.infer_ctxt().build(TypingMode::PostAnalysis);
728    let ty = infcx.at(&ObligationCause::new(), trait_env.env).deeply_normalize(ty).unwrap_or(ty);
729    render_const_scalar_inner(f, b, memory_map, ty, trait_env)
730}

732fn render_const_scalar_inner<'db>(
733    f: &mut HirFormatter<'_, 'db>,
734    b: &[u8],
735    memory_map: &MemoryMap<'db>,
736    ty: Ty<'db>,
737    trait_env: Arc<TraitEnvironment<'db>>,
738) -> Result<(), HirDisplayError> {
739    use TyKind;
740    match ty.kind() {
741        TyKind::Bool => write!(f, "{}", b[0] != 0),
742        TyKind::Char => {
743            let it = u128::from_le_bytes(pad16(b, false)) as u32;
744            let Ok(c) = char::try_from(it) else {
745                return f.write_str("<unicode-error>");
746            };
747            write!(f, "{c:?}")
748        }
749        TyKind::Int(_) => {
750            let it = i128::from_le_bytes(pad16(b, true));
751            write!(f, "{it}")
752        }
753        TyKind::Uint(_) => {
754            let it = u128::from_le_bytes(pad16(b, false));
755            write!(f, "{it}")
756        }
757        TyKind::Float(fl) => match fl {
758            FloatTy::F16 => {
759                // FIXME(#17451): Replace with builtins once they are stabilised.
760                let it = f16::from_bits(u16::from_le_bytes(b.try_into().unwrap()).into());
761                let s = it.to_string();
762                if s.strip_prefix('-').unwrap_or(&s).chars().all(|c| c.is_ascii_digit()) {
763                    // Match Rust debug formatting
764                    write!(f, "{s}.0")
765                } else {
766                    write!(f, "{s}")
767                }
768            }
769            FloatTy::F32 => {
770                let it = f32::from_le_bytes(b.try_into().unwrap());
771                write!(f, "{it:?}")
772            }
773            FloatTy::F64 => {
774                let it = f64::from_le_bytes(b.try_into().unwrap());
775                write!(f, "{it:?}")
776            }
777            FloatTy::F128 => {
778                // FIXME(#17451): Replace with builtins once they are stabilised.
779                let it = f128::from_bits(u128::from_le_bytes(b.try_into().unwrap()));
780                let s = it.to_string();
781                if s.strip_prefix('-').unwrap_or(&s).chars().all(|c| c.is_ascii_digit()) {
782                    // Match Rust debug formatting
783                    write!(f, "{s}.0")
784                } else {
785                    write!(f, "{s}")
786                }
787            }
788        },
789        TyKind::Ref(_, t, _) => match t.kind() {
790            TyKind::Str => {
791                let addr = usize::from_le_bytes(b[0..b.len() / 2].try_into().unwrap());
792                let size = usize::from_le_bytes(b[b.len() / 2..].try_into().unwrap());
793                let Some(bytes) = memory_map.get(addr, size) else {
794                    return f.write_str("<ref-data-not-available>");
795                };
796                let s = std::str::from_utf8(bytes).unwrap_or("<utf8-error>");
797                write!(f, "{s:?}")
798            }
799            TyKind::Slice(ty) => {
800                let addr = usize::from_le_bytes(b[0..b.len() / 2].try_into().unwrap());
801                let count = usize::from_le_bytes(b[b.len() / 2..].try_into().unwrap());
802                let Ok(layout) = f.db.layout_of_ty(ty, trait_env) else {
803                    return f.write_str("<layout-error>");
804                };
805                let size_one = layout.size.bytes_usize();
806                let Some(bytes) = memory_map.get(addr, size_one * count) else {
807                    return f.write_str("<ref-data-not-available>");
808                };
809                let expected_len = count * size_one;
810                if bytes.len() < expected_len {
811                    never!(
812                        "Memory map size is too small. Expected {expected_len}, got {}",
813                        bytes.len(),
814                    );
815                    return f.write_str("<layout-error>");
816                }
817                f.write_str("&[")?;
818                let mut first = true;
819                for i in 0..count {
820                    if first {
821                        first = false;
822                    } else {
823                        f.write_str(", ")?;
824                    }
825                    let offset = size_one * i;
826                    render_const_scalar(f, &bytes[offset..offset + size_one], memory_map, ty)?;
827                }
828                f.write_str("]")
829            }
830            TyKind::Dynamic(_, _) => {
831                let addr = usize::from_le_bytes(b[0..b.len() / 2].try_into().unwrap());
832                let ty_id = usize::from_le_bytes(b[b.len() / 2..].try_into().unwrap());
833                let Ok(t) = memory_map.vtable_ty(ty_id) else {
834                    return f.write_str("<ty-missing-in-vtable-map>");
835                };
836                let Ok(layout) = f.db.layout_of_ty(t, trait_env) else {
837                    return f.write_str("<layout-error>");
838                };
839                let size = layout.size.bytes_usize();
840                let Some(bytes) = memory_map.get(addr, size) else {
841                    return f.write_str("<ref-data-not-available>");
842                };
843                f.write_str("&")?;
844                render_const_scalar(f, bytes, memory_map, t)
845            }
846            TyKind::Adt(adt, _) if b.len() == 2 * size_of::<usize>() => match adt.def_id().0 {
847                hir_def::AdtId::StructId(s) => {
848                    let data = f.db.struct_signature(s);
849                    write!(f, "&{}", data.name.display(f.db, f.edition()))?;
850                    Ok(())
851                }
852                _ => f.write_str("<unsized-enum-or-union>"),
853            },
854            _ => {
855                let addr = usize::from_le_bytes(match b.try_into() {
856                    Ok(b) => b,
857                    Err(_) => {
858                        never!(
859                            "tried rendering ty {:?} in const ref with incorrect byte count {}",
860                            t,
861                            b.len()
862                        );
863                        return f.write_str("<layout-error>");
864                    }
865                });
866                let Ok(layout) = f.db.layout_of_ty(t, trait_env) else {
867                    return f.write_str("<layout-error>");
868                };
869                let size = layout.size.bytes_usize();
870                let Some(bytes) = memory_map.get(addr, size) else {
871                    return f.write_str("<ref-data-not-available>");
872                };
873                f.write_str("&")?;
874                render_const_scalar(f, bytes, memory_map, t)
875            }
876        },
877        TyKind::Tuple(tys) => {
878            let Ok(layout) = f.db.layout_of_ty(ty, trait_env.clone()) else {
879                return f.write_str("<layout-error>");
880            };
881            f.write_str("(")?;
882            let mut first = true;
883            for (id, ty) in tys.iter().enumerate() {
884                if first {
885                    first = false;
886                } else {
887                    f.write_str(", ")?;
888                }
889                let offset = layout.fields.offset(id).bytes_usize();
890                let Ok(layout) = f.db.layout_of_ty(ty, trait_env.clone()) else {
891                    f.write_str("<layout-error>")?;
892                    continue;
893                };
894                let size = layout.size.bytes_usize();
895                render_const_scalar(f, &b[offset..offset + size], memory_map, ty)?;
896            }
897            f.write_str(")")
898        }
899        TyKind::Adt(def, args) => {
900            let def = def.def_id().0;
901            let Ok(layout) = f.db.layout_of_adt(def, args, trait_env.clone()) else {
902                return f.write_str("<layout-error>");
903            };
904            match def {
905                hir_def::AdtId::StructId(s) => {
906                    let data = f.db.struct_signature(s);
907                    write!(f, "{}", data.name.display(f.db, f.edition()))?;
908                    let field_types = f.db.field_types(s.into());
909                    render_variant_after_name(
910                        s.fields(f.db),
911                        f,
912                        &field_types,
913                        f.db.trait_environment(def.into()),
914                        &layout,
915                        args,
916                        b,
917                        memory_map,
918                    )
919                }
920                hir_def::AdtId::UnionId(u) => {
921                    write!(f, "{}", f.db.union_signature(u).name.display(f.db, f.edition()))
922                }
923                hir_def::AdtId::EnumId(e) => {
924                    let Ok(target_data_layout) = f.db.target_data_layout(trait_env.krate) else {
925                        return f.write_str("<target-layout-not-available>");
926                    };
927                    let Some((var_id, var_layout)) =
928                        detect_variant_from_bytes(&layout, f.db, &target_data_layout, b, e)
929                    else {
930                        return f.write_str("<failed-to-detect-variant>");
931                    };
932                    let loc = var_id.lookup(f.db);
933                    write!(
934                        f,
935                        "{}",
936                        loc.parent.enum_variants(f.db).variants[loc.index as usize]
937                            .1
938                            .display(f.db, f.edition())
939                    )?;
940                    let field_types = f.db.field_types(var_id.into());
941                    render_variant_after_name(
942                        var_id.fields(f.db),
943                        f,
944                        &field_types,
945                        f.db.trait_environment(def.into()),
946                        var_layout,
947                        args,
948                        b,
949                        memory_map,
950                    )
951                }
952            }
953        }
954        TyKind::FnDef(..) => ty.hir_fmt(f),
955        TyKind::FnPtr(_, _) | TyKind::RawPtr(_, _) => {
956            let it = u128::from_le_bytes(pad16(b, false));
957            write!(f, "{it:#X} as ")?;
958            ty.hir_fmt(f)
959        }
960        TyKind::Array(ty, len) => {
961            let Some(len) = consteval::try_const_usize(f.db, len) else {
962                return f.write_str("<unknown-array-len>");
963            };
964            let Ok(layout) = f.db.layout_of_ty(ty, trait_env) else {
965                return f.write_str("<layout-error>");
966            };
967            let size_one = layout.size.bytes_usize();
968            f.write_str("[")?;
969            let mut first = true;
970            for i in 0..len as usize {
971                if first {
972                    first = false;
973                } else {
974                    f.write_str(", ")?;
975                }
976                let offset = size_one * i;
977                render_const_scalar(f, &b[offset..offset + size_one], memory_map, ty)?;
978            }
979            f.write_str("]")
980        }
981        TyKind::Never => f.write_str("!"),
982        TyKind::Closure(_, _) => f.write_str("<closure>"),
983        TyKind::Coroutine(_, _) => f.write_str("<coroutine>"),
984        TyKind::CoroutineWitness(_, _) => f.write_str("<coroutine-witness>"),
985        TyKind::CoroutineClosure(_, _) => f.write_str("<coroutine-closure>"),
986        TyKind::UnsafeBinder(_) => f.write_str("<unsafe-binder>"),
987        // The below arms are unreachable, since const eval will bail out before here.
988        TyKind::Foreign(_) => f.write_str("<extern-type>"),
989        TyKind::Pat(_, _) => f.write_str("<pat>"),
990        TyKind::Error(..)
991        | TyKind::Placeholder(_)
992        | TyKind::Alias(_, _)
993        | TyKind::Param(_)
994        | TyKind::Bound(_, _)
995        | TyKind::Infer(_) => f.write_str("<placeholder-or-unknown-type>"),
996        // The below arms are unreachable, since we handled them in ref case.
997        TyKind::Slice(_) | TyKind::Str | TyKind::Dynamic(_, _) => f.write_str("<unsized-value>"),
998    }
999}

1001fn render_variant_after_name<'db>(
1002    data: &VariantFields,
1003    f: &mut HirFormatter<'_, 'db>,
1004    field_types: &ArenaMap<LocalFieldId, EarlyBinder<'db, Ty<'db>>>,
1005    trait_env: Arc<TraitEnvironment<'db>>,
1006    layout: &Layout,
1007    args: GenericArgs<'db>,
1008    b: &[u8],
1009    memory_map: &MemoryMap<'db>,
1010) -> Result<(), HirDisplayError> {
1011    match data.shape {
1012        FieldsShape::Record | FieldsShape::Tuple => {
1013            let render_field = |f: &mut HirFormatter<'_, 'db>, id: LocalFieldId| {
1014                let offset = layout.fields.offset(u32::from(id.into_raw()) as usize).bytes_usize();
1015                let ty = field_types[id].instantiate(f.interner, args);
1016                let Ok(layout) = f.db.layout_of_ty(ty, trait_env.clone()) else {
1017                    return f.write_str("<layout-error>");
1018                };
1019                let size = layout.size.bytes_usize();
1020                render_const_scalar(f, &b[offset..offset + size], memory_map, ty)
1021            };
1022            let mut it = data.fields().iter();
1023            if matches!(data.shape, FieldsShape::Record) {
1024                write!(f, " {{")?;
1025                if let Some((id, data)) = it.next() {
1026                    write!(f, " {}: ", data.name.display(f.db, f.edition()))?;
1027                    render_field(f, id)?;
1028                }
1029                for (id, data) in it {
1030                    write!(f, ", {}: ", data.name.display(f.db, f.edition()))?;
1031                    render_field(f, id)?;
1032                }
1033                write!(f, " }}")?;
1034            } else {
1035                let mut it = it.map(|it| it.0);
1036                write!(f, "(")?;
1037                if let Some(id) = it.next() {
1038                    render_field(f, id)?;
1039                }
1040                for id in it {
1041                    write!(f, ", ")?;
1042                    render_field(f, id)?;
1043                }
1044                write!(f, ")")?;
1045            }
1046            Ok(())
1047        }
1048        FieldsShape::Unit => Ok(()),
1049    }
1050}

1052impl<'db> HirDisplay<'db> for Ty<'db> {
1053    fn hir_fmt(
1054        &self,
1055        f @ &mut HirFormatter { db, .. }: &mut HirFormatter<'_, 'db>,
1056    ) -> Result<(), HirDisplayError> {
1057        let interner = f.interner;
1058        if f.should_truncate() {
1059            return write!(f, "{TYPE_HINT_TRUNCATION}");
1060        }
1061
1062        use TyKind;
1063        match self.kind() {
1064            TyKind::Never => write!(f, "!")?,
1065            TyKind::Str => write!(f, "str")?,
1066            TyKind::Bool => write!(f, "bool")?,
1067            TyKind::Char => write!(f, "char")?,
1068            TyKind::Float(t) => write!(f, "{}", primitive::float_ty_to_string(t))?,
1069            TyKind::Int(t) => write!(f, "{}", primitive::int_ty_to_string(t))?,
1070            TyKind::Uint(t) => write!(f, "{}", primitive::uint_ty_to_string(t))?,
1071            TyKind::Slice(t) => {
1072                write!(f, "[")?;
1073                t.hir_fmt(f)?;
1074                write!(f, "]")?;
1075            }
1076            TyKind::Array(t, c) => {
1077                write!(f, "[")?;
1078                t.hir_fmt(f)?;
1079                write!(f, "; ")?;
1080                c.hir_fmt(f)?;
1081                write!(f, "]")?;
1082            }
1083            kind @ (TyKind::RawPtr(t, m) | TyKind::Ref(_, t, m)) => {
1084                if let TyKind::Ref(l, _, _) = kind {
1085                    f.write_char('&')?;
1086                    if f.render_region(l) {
1087                        l.hir_fmt(f)?;
1088                        f.write_char(' ')?;
1089                    }
1090                    match m {
1091                        rustc_ast_ir::Mutability::Not => (),
1092                        rustc_ast_ir::Mutability::Mut => f.write_str("mut ")?,
1093                    }
1094                } else {
1095                    write!(
1096                        f,
1097                        "*{}",
1098                        match m {
1099                            rustc_ast_ir::Mutability::Not => "const ",
1100                            rustc_ast_ir::Mutability::Mut => "mut ",
1101                        }
1102                    )?;
1103                }
1104
1105                // FIXME: all this just to decide whether to use parentheses...
1106                let (preds_to_print, has_impl_fn_pred) = match t.kind() {
1107                    TyKind::Dynamic(bounds, region) => {
1108                        let contains_impl_fn =
1109                            bounds.iter().any(|bound| match bound.skip_binder() {
1110                                ExistentialPredicate::Trait(trait_ref) => {
1111                                    let trait_ = trait_ref.def_id.0;
1112                                    fn_traits(db, trait_).any(|it| it == trait_)
1113                                }
1114                                _ => false,
1115                            });
1116                        let render_lifetime = f.render_region(region);
1117                        (bounds.len() + render_lifetime as usize, contains_impl_fn)
1118                    }
1119                    TyKind::Alias(AliasTyKind::Opaque, ty) => {
1120                        let opaque_ty_id = match ty.def_id {
1121                            SolverDefId::InternedOpaqueTyId(id) => id,
1122                            _ => unreachable!(),
1123                        };
1124                        let impl_trait_id = db.lookup_intern_impl_trait_id(opaque_ty_id);
1125                        if let ImplTraitId::ReturnTypeImplTrait(func, idx) = impl_trait_id {
1126                            let datas = db
1127                                .return_type_impl_traits(func)
1128                                .expect("impl trait id without data");
1129                            let data = (*datas)
1130                                .as_ref()
1131                                .map_bound(|rpit| &rpit.impl_traits[idx].predicates);
1132                            let bounds =
1133                                || data.iter_instantiated_copied(f.interner, ty.args.as_slice());
1134                            let mut len = bounds().count();
1135
1136                            // Don't count Sized but count when it absent
1137                            // (i.e. when explicit ?Sized bound is set).
1138                            let default_sized = SizedByDefault::Sized { anchor: func.krate(db) };
1139                            let sized_bounds = bounds()
1140                                .filter(|b| {
1141                                    matches!(
1142                                        b.kind().skip_binder(),
1143                                        ClauseKind::Trait(trait_ref)
1144                                            if default_sized.is_sized_trait(
1145                                                trait_ref.def_id().0,
1146                                                db,
1147                                            ),
1148                                    )
1149                                })
1150                                .count();
1151                            match sized_bounds {
1152                                0 => len += 1,
1153                                _ => {
1154                                    len = len.saturating_sub(sized_bounds);
1155                                }
1156                            }
1157
1158                            let contains_impl_fn = bounds().any(|bound| {
1159                                if let ClauseKind::Trait(trait_ref) = bound.kind().skip_binder() {
1160                                    let trait_ = trait_ref.def_id().0;
1161                                    fn_traits(db, trait_).any(|it| it == trait_)
1162                                } else {
1163                                    false
1164                                }
1165                            });
1166                            (len, contains_impl_fn)
1167                        } else {
1168                            (0, false)
1169                        }
1170                    }
1171                    _ => (0, false),
1172                };
1173
1174                if has_impl_fn_pred && preds_to_print <= 2 {
1175                    return t.hir_fmt(f);
1176                }
1177
1178                if preds_to_print > 1 {
1179                    write!(f, "(")?;
1180                    t.hir_fmt(f)?;
1181                    write!(f, ")")?;
1182                } else {
1183                    t.hir_fmt(f)?;
1184                }
1185            }
1186            TyKind::Tuple(tys) => {
1187                if tys.len() == 1 {
1188                    write!(f, "(")?;
1189                    tys.as_slice()[0].hir_fmt(f)?;
1190                    write!(f, ",)")?;
1191                } else {
1192                    write!(f, "(")?;
1193                    f.write_joined(tys.as_slice(), ", ")?;
1194                    write!(f, ")")?;
1195                }
1196            }
1197            TyKind::FnPtr(sig, header) => {
1198                let sig = sig.with(header);
1199                sig.hir_fmt(f)?;
1200            }
1201            TyKind::FnDef(def, args) => {
1202                let def = def.0;
1203                let sig = db.callable_item_signature(def).instantiate(interner, args);
1204
1205                if f.display_kind.is_source_code() {
1206                    // `FnDef` is anonymous and there's no surface syntax for it. Show it as a
1207                    // function pointer type.
1208                    return sig.hir_fmt(f);
1209                }
1210                if let Safety::Unsafe = sig.safety() {
1211                    write!(f, "unsafe ")?;
1212                }
1213                if !matches!(sig.abi(), FnAbi::Rust | FnAbi::RustCall) {
1214                    f.write_str("extern \"")?;
1215                    f.write_str(sig.abi().as_str())?;
1216                    f.write_str("\" ")?;
1217                }
1218
1219                let sig = sig.skip_binder();
1220                write!(f, "fn ")?;
1221                f.start_location_link(def.into());
1222                match def {
1223                    CallableDefId::FunctionId(ff) => {
1224                        write!(f, "{}", db.function_signature(ff).name.display(f.db, f.edition()))?
1225                    }
1226                    CallableDefId::StructId(s) => {
1227                        write!(f, "{}", db.struct_signature(s).name.display(f.db, f.edition()))?
1228                    }
1229                    CallableDefId::EnumVariantId(e) => {
1230                        let loc = e.lookup(db);
1231                        write!(
1232                            f,
1233                            "{}",
1234                            loc.parent.enum_variants(db).variants[loc.index as usize]
1235                                .1
1236                                .display(db, f.edition())
1237                        )?
1238                    }
1239                };
1240                f.end_location_link();
1241
1242                if args.len() > 0 {
1243                    let generic_def_id = GenericDefId::from_callable(db, def);
1244                    let generics = generics(db, generic_def_id);
1245                    let (parent_len, self_param, type_, const_, impl_, lifetime) =
1246                        generics.provenance_split();
1247                    let parameters = args.as_slice();
1248                    debug_assert_eq!(
1249                        parameters.len(),
1250                        parent_len + self_param as usize + type_ + const_ + impl_ + lifetime
1251                    );
1252                    // We print all params except implicit impl Trait params. Still a bit weird; should we leave out parent and self?
1253                    if parameters.len() - impl_ > 0 {
1254                        let params_len = parameters.len();
1255                        // `parameters` are in the order of fn's params (including impl traits), fn's lifetimes
1256                        let parameters =
1257                            generic_args_sans_defaults(f, Some(generic_def_id), parameters);
1258                        assert!(params_len >= parameters.len());
1259                        let defaults = params_len - parameters.len();
1260
1261                        // Normally, functions cannot have default parameters, but they can,
1262                        // for function-like things such as struct names or enum variants.
1263                        // The former cannot have defaults but does have parents,
1264                        // but the latter cannot have parents but can have defaults.
1265                        //
1266                        // However, it's also true that *traits* can have defaults too.
1267                        // In this case, there can be no function params.
1268                        let parent_end = if parent_len > 0 {
1269                            // If `parent_len` > 0, then there cannot be defaults on the function
1270                            // and all defaults must come from the parent.
1271                            parent_len - defaults
1272                        } else {
1273                            parent_len
1274                        };
1275                        let fn_params_no_impl_or_defaults = parameters.len() - parent_end - impl_;
1276                        let (parent_params, fn_params) = parameters.split_at(parent_end);
1277
1278                        write!(f, "<")?;
1279                        hir_fmt_generic_arguments(f, parent_params, None)?;
1280                        if !parent_params.is_empty() && !fn_params.is_empty() {
1281                            write!(f, ", ")?;
1282                        }
1283                        hir_fmt_generic_arguments(
1284                            f,
1285                            &fn_params[..fn_params_no_impl_or_defaults],
1286                            None,
1287                        )?;
1288                        write!(f, ">")?;
1289                    }
1290                }
1291                write!(f, "(")?;
1292                f.write_joined(sig.inputs(), ", ")?;
1293                write!(f, ")")?;
1294                let ret = sig.output();
1295                if !ret.is_unit() {
1296                    write!(f, " -> ")?;
1297                    ret.hir_fmt(f)?;
1298                }
1299            }
1300            TyKind::Adt(def, parameters) => {
1301                let def_id = def.def_id().0;
1302                f.start_location_link(def_id.into());
1303                match f.display_kind {
1304                    DisplayKind::Diagnostics | DisplayKind::Test => {
1305                        let name = match def_id {
1306                            hir_def::AdtId::StructId(it) => db.struct_signature(it).name.clone(),
1307                            hir_def::AdtId::UnionId(it) => db.union_signature(it).name.clone(),
1308                            hir_def::AdtId::EnumId(it) => db.enum_signature(it).name.clone(),
1309                        };
1310                        write!(f, "{}", name.display(f.db, f.edition()))?;
1311                    }
1312                    DisplayKind::SourceCode { target_module_id: module_id, allow_opaque: _ } => {
1313                        if let Some(path) = find_path::find_path(
1314                            db,
1315                            ItemInNs::Types(def_id.into()),
1316                            module_id,
1317                            PrefixKind::Plain,
1318                            false,
1319                            // FIXME: no_std Cfg?
1320                            FindPathConfig {
1321                                prefer_no_std: false,
1322                                prefer_prelude: true,
1323                                prefer_absolute: false,
1324                                allow_unstable: true,
1325                            },
1326                        ) {
1327                            write!(f, "{}", path.display(f.db, f.edition()))?;
1328                        } else {
1329                            return Err(HirDisplayError::DisplaySourceCodeError(
1330                                DisplaySourceCodeError::PathNotFound,
1331                            ));
1332                        }
1333                    }
1334                }
1335                f.end_location_link();
1336
1337                hir_fmt_generics(f, parameters.as_slice(), Some(def.def_id().0.into()), None)?;
1338            }
1339            TyKind::Alias(AliasTyKind::Projection, alias_ty) => write_projection(f, &alias_ty)?,
1340            TyKind::Foreign(alias) => {
1341                let type_alias = db.type_alias_signature(alias.0);
1342                f.start_location_link(alias.0.into());
1343                write!(f, "{}", type_alias.name.display(f.db, f.edition()))?;
1344                f.end_location_link();
1345            }
1346            TyKind::Alias(AliasTyKind::Opaque, alias_ty) => {
1347                let opaque_ty_id = match alias_ty.def_id {
1348                    SolverDefId::InternedOpaqueTyId(id) => id,
1349                    _ => unreachable!(),
1350                };
1351                if !f.display_kind.allows_opaque() {
1352                    return Err(HirDisplayError::DisplaySourceCodeError(
1353                        DisplaySourceCodeError::OpaqueType,
1354                    ));
1355                }
1356                let impl_trait_id = db.lookup_intern_impl_trait_id(opaque_ty_id);
1357                match impl_trait_id {
1358                    ImplTraitId::ReturnTypeImplTrait(func, idx) => {
1359                        let datas =
1360                            db.return_type_impl_traits(func).expect("impl trait id without data");
1361                        let data =
1362                            (*datas).as_ref().map_bound(|rpit| &rpit.impl_traits[idx].predicates);
1363                        let bounds = data
1364                            .iter_instantiated_copied(interner, alias_ty.args.as_slice())
1365                            .collect::<Vec<_>>();
1366                        let krate = func.krate(db);
1367                        write_bounds_like_dyn_trait_with_prefix(
1368                            f,
1369                            "impl",
1370                            Either::Left(*self),
1371                            &bounds,
1372                            SizedByDefault::Sized { anchor: krate },
1373                        )?;
1374                        // FIXME: it would maybe be good to distinguish this from the alias type (when debug printing), and to show the substitution
1375                    }
1376                    ImplTraitId::TypeAliasImplTrait(alias, idx) => {
1377                        let datas =
1378                            db.type_alias_impl_traits(alias).expect("impl trait id without data");
1379                        let data =
1380                            (*datas).as_ref().map_bound(|rpit| &rpit.impl_traits[idx].predicates);
1381                        let bounds = data
1382                            .iter_instantiated_copied(interner, alias_ty.args.as_slice())
1383                            .collect::<Vec<_>>();
1384                        let krate = alias.krate(db);
1385                        write_bounds_like_dyn_trait_with_prefix(
1386                            f,
1387                            "impl",
1388                            Either::Left(*self),
1389                            &bounds,
1390                            SizedByDefault::Sized { anchor: krate },
1391                        )?;
1392                    }
1393                }
1394            }
1395            TyKind::Closure(id, substs) => {
1396                let id = id.0;
1397                if f.display_kind.is_source_code() {
1398                    if !f.display_kind.allows_opaque() {
1399                        return Err(HirDisplayError::DisplaySourceCodeError(
1400                            DisplaySourceCodeError::OpaqueType,
1401                        ));
1402                    } else if f.closure_style != ClosureStyle::ImplFn {
1403                        never!("Only `impl Fn` is valid for displaying closures in source code");
1404                    }
1405                }
1406                match f.closure_style {
1407                    ClosureStyle::Hide => return write!(f, "{TYPE_HINT_TRUNCATION}"),
1408                    ClosureStyle::ClosureWithId => {
1409                        return write!(
1410                            f,
1411                            "{{closure#{:?}}}",
1412                            salsa::plumbing::AsId::as_id(&id).index()
1413                        );
1414                    }
1415                    ClosureStyle::ClosureWithSubst => {
1416                        write!(f, "{{closure#{:?}}}", salsa::plumbing::AsId::as_id(&id).index())?;
1417                        return hir_fmt_generics(f, substs.as_slice(), None, None);
1418                    }
1419                    _ => (),
1420                }
1421                let sig = substs
1422                    .split_closure_args_untupled()
1423                    .closure_sig_as_fn_ptr_ty
1424                    .callable_sig(interner);
1425                if let Some(sig) = sig {
1426                    let sig = sig.skip_binder();
1427                    let InternedClosure(def, _) = db.lookup_intern_closure(id);
1428                    let infer = db.infer(def);
1429                    let (_, kind) = infer.closure_info(id);
1430                    match f.closure_style {
1431                        ClosureStyle::ImplFn => write!(f, "impl {kind:?}(")?,
1432                        ClosureStyle::RANotation => write!(f, "|")?,
1433                        _ => unreachable!(),
1434                    }
1435                    if sig.inputs().is_empty() {
1436                    } else if f.should_truncate() {
1437                        write!(f, "{TYPE_HINT_TRUNCATION}")?;
1438                    } else {
1439                        f.write_joined(sig.inputs(), ", ")?;
1440                    };
1441                    match f.closure_style {
1442                        ClosureStyle::ImplFn => write!(f, ")")?,
1443                        ClosureStyle::RANotation => write!(f, "|")?,
1444                        _ => unreachable!(),
1445                    }
1446                    if f.closure_style == ClosureStyle::RANotation || !sig.output().is_unit() {
1447                        write!(f, " -> ")?;
1448                        sig.output().hir_fmt(f)?;
1449                    }
1450                } else {
1451                    write!(f, "{{closure}}")?;
1452                }
1453            }
1454            TyKind::CoroutineClosure(id, args) => {
1455                let id = id.0;
1456                if f.display_kind.is_source_code() {
1457                    if !f.display_kind.allows_opaque() {
1458                        return Err(HirDisplayError::DisplaySourceCodeError(
1459                            DisplaySourceCodeError::OpaqueType,
1460                        ));
1461                    } else if f.closure_style != ClosureStyle::ImplFn {
1462                        never!("Only `impl Fn` is valid for displaying closures in source code");
1463                    }
1464                }
1465                match f.closure_style {
1466                    ClosureStyle::Hide => return write!(f, "{TYPE_HINT_TRUNCATION}"),
1467                    ClosureStyle::ClosureWithId => {
1468                        return write!(
1469                            f,
1470                            "{{async closure#{:?}}}",
1471                            salsa::plumbing::AsId::as_id(&id).index()
1472                        );
1473                    }
1474                    ClosureStyle::ClosureWithSubst => {
1475                        write!(
1476                            f,
1477                            "{{async closure#{:?}}}",
1478                            salsa::plumbing::AsId::as_id(&id).index()
1479                        )?;
1480                        return hir_fmt_generics(f, args.as_slice(), None, None);
1481                    }
1482                    _ => (),
1483                }
1484                let CoroutineClosureArgsParts { closure_kind_ty, signature_parts_ty, .. } =
1485                    args.split_coroutine_closure_args();
1486                let kind = closure_kind_ty.to_opt_closure_kind().unwrap();
1487                let kind = match kind {
1488                    rustc_type_ir::ClosureKind::Fn => "AsyncFn",
1489                    rustc_type_ir::ClosureKind::FnMut => "AsyncFnMut",
1490                    rustc_type_ir::ClosureKind::FnOnce => "AsyncFnOnce",
1491                };
1492                let TyKind::FnPtr(coroutine_sig, _) = signature_parts_ty.kind() else {
1493                    unreachable!("invalid coroutine closure signature");
1494                };
1495                let coroutine_sig = coroutine_sig.skip_binder();
1496                let coroutine_inputs = coroutine_sig.inputs();
1497                let TyKind::Tuple(coroutine_inputs) = coroutine_inputs.as_slice()[1].kind() else {
1498                    unreachable!("invalid coroutine closure signature");
1499                };
1500                let TyKind::Tuple(coroutine_output) = coroutine_sig.output().kind() else {
1501                    unreachable!("invalid coroutine closure signature");
1502                };
1503                let coroutine_output = coroutine_output.as_slice()[1];
1504                match f.closure_style {
1505                    ClosureStyle::ImplFn => write!(f, "impl {kind}(")?,
1506                    ClosureStyle::RANotation => write!(f, "async |")?,
1507                    _ => unreachable!(),
1508                }
1509                if coroutine_inputs.is_empty() {
1510                } else if f.should_truncate() {
1511                    write!(f, "{TYPE_HINT_TRUNCATION}")?;
1512                } else {
1513                    f.write_joined(coroutine_inputs, ", ")?;
1514                };
1515                match f.closure_style {
1516                    ClosureStyle::ImplFn => write!(f, ")")?,
1517                    ClosureStyle::RANotation => write!(f, "|")?,
1518                    _ => unreachable!(),
1519                }
1520                if f.closure_style == ClosureStyle::RANotation || !coroutine_output.is_unit() {
1521                    write!(f, " -> ")?;
1522                    coroutine_output.hir_fmt(f)?;
1523                }
1524            }
1525            TyKind::Placeholder(_) => write!(f, "{{placeholder}}")?,
1526            TyKind::Param(param) => {
1527                // FIXME: We should not access `param.id`, it should be removed, and we should know the
1528                // parent from the formatted type.
1529                let generics = generics(db, param.id.parent());
1530                let param_data = &generics[param.id.local_id()];
1531                match param_data {
1532                    TypeOrConstParamData::TypeParamData(p) => match p.provenance {
1533                        TypeParamProvenance::TypeParamList | TypeParamProvenance::TraitSelf => {
1534                            write!(
1535                                f,
1536                                "{}",
1537                                p.name
1538                                    .clone()
1539                                    .unwrap_or_else(Name::missing)
1540                                    .display(f.db, f.edition())
1541                            )?
1542                        }
1543                        TypeParamProvenance::ArgumentImplTrait => {
1544                            let bounds = db
1545                                .generic_predicates(param.id.parent())
1546                                .instantiate_identity()
1547                                .into_iter()
1548                                .flatten()
1549                                .filter(|wc| match wc.kind().skip_binder() {
1550                                    ClauseKind::Trait(tr) => tr.self_ty() == *self,
1551                                    ClauseKind::Projection(proj) => proj.self_ty() == *self,
1552                                    ClauseKind::TypeOutlives(to) => to.0 == *self,
1553                                    _ => false,
1554                                })
1555                                .collect::<Vec<_>>();
1556                            let krate = param.id.parent().module(db).krate();
1557                            write_bounds_like_dyn_trait_with_prefix(
1558                                f,
1559                                "impl",
1560                                Either::Left(*self),
1561                                &bounds,
1562                                SizedByDefault::Sized { anchor: krate },
1563                            )?;
1564                        }
1565                    },
1566                    TypeOrConstParamData::ConstParamData(p) => {
1567                        write!(f, "{}", p.name.display(f.db, f.edition()))?;
1568                    }
1569                }
1570            }
1571            TyKind::Bound(BoundVarIndexKind::Bound(debruijn), ty) => {
1572                write!(f, "?{}.{}", debruijn.as_usize(), ty.var.as_usize())?
1573            }
1574            TyKind::Bound(BoundVarIndexKind::Canonical, ty) => {
1575                write!(f, "?c.{}", ty.var.as_usize())?
1576            }
1577            TyKind::Dynamic(bounds, region) => {
1578                // We want to put auto traits after principal traits, regardless of their written order.
1579                let mut bounds_to_display = SmallVec::<[_; 4]>::new();
1580                let mut auto_trait_bounds = SmallVec::<[_; 4]>::new();
1581                for bound in bounds.iter() {
1582                    let clause = bound.with_self_ty(interner, *self);
1583                    match bound.skip_binder() {
1584                        ExistentialPredicate::Trait(_) | ExistentialPredicate::Projection(_) => {
1585                            bounds_to_display.push(clause);
1586                        }
1587                        ExistentialPredicate::AutoTrait(_) => auto_trait_bounds.push(clause),
1588                    }
1589                }
1590                bounds_to_display.append(&mut auto_trait_bounds);
1591
1592                if f.render_region(region) {
1593                    bounds_to_display
1594                        .push(rustc_type_ir::OutlivesPredicate(*self, region).upcast(interner));
1595                }
1596
1597                write_bounds_like_dyn_trait_with_prefix(
1598                    f,
1599                    "dyn",
1600                    Either::Left(*self),
1601                    &bounds_to_display,
1602                    SizedByDefault::NotSized,
1603                )?;
1604            }
1605            TyKind::Error(_) => {
1606                if f.display_kind.is_source_code() {
1607                    f.write_char('_')?;
1608                } else {
1609                    write!(f, "{{unknown}}")?;
1610                }
1611            }
1612            TyKind::Infer(..) => write!(f, "_")?,
1613            TyKind::Coroutine(coroutine_id, subst) => {
1614                let InternedCoroutine(owner, expr_id) = coroutine_id.0.loc(db);
1615                let CoroutineArgsParts { resume_ty, yield_ty, return_ty, .. } =
1616                    subst.split_coroutine_args();
1617                let body = db.body(owner);
1618                let expr = &body[expr_id];
1619                match expr {
1620                    hir_def::hir::Expr::Closure {
1621                        closure_kind: hir_def::hir::ClosureKind::Async,
1622                        ..
1623                    }
1624                    | hir_def::hir::Expr::Async { .. } => {
1625                        let future_trait =
1626                            LangItem::Future.resolve_trait(db, owner.module(db).krate());
1627                        let output = future_trait.and_then(|t| {
1628                            t.trait_items(db)
1629                                .associated_type_by_name(&Name::new_symbol_root(sym::Output))
1630                        });
1631                        write!(f, "impl ")?;
1632                        if let Some(t) = future_trait {
1633                            f.start_location_link(t.into());
1634                        }
1635                        write!(f, "Future")?;
1636                        if future_trait.is_some() {
1637                            f.end_location_link();
1638                        }
1639                        write!(f, "<")?;
1640                        if let Some(t) = output {
1641                            f.start_location_link(t.into());
1642                        }
1643                        write!(f, "Output")?;
1644                        if output.is_some() {
1645                            f.end_location_link();
1646                        }
1647                        write!(f, " = ")?;
1648                        return_ty.hir_fmt(f)?;
1649                        write!(f, ">")?;
1650                    }
1651                    hir_def::hir::Expr::Closure {
1652                        closure_kind: hir_def::hir::ClosureKind::Coroutine(..),
1653                        ..
1654                    } => {
1655                        if f.display_kind.is_source_code() {
1656                            return Err(HirDisplayError::DisplaySourceCodeError(
1657                                DisplaySourceCodeError::Coroutine,
1658                            ));
1659                        }
1660                        write!(f, "|")?;
1661                        resume_ty.hir_fmt(f)?;
1662                        write!(f, "|")?;
1663
1664                        write!(f, " yields ")?;
1665                        yield_ty.hir_fmt(f)?;
1666
1667                        write!(f, " -> ")?;
1668                        return_ty.hir_fmt(f)?;
1669                    }
1670                    _ => panic!("invalid expr for coroutine: {expr:?}"),
1671                }
1672            }
1673            TyKind::CoroutineWitness(..) => write!(f, "{{coroutine witness}}")?,
1674            TyKind::Pat(_, _) => write!(f, "{{pat}}")?,
1675            TyKind::UnsafeBinder(_) => write!(f, "{{unsafe binder}}")?,
1676            TyKind::Alias(_, _) => write!(f, "{{alias}}")?,
1677        }
1678        Ok(())
1679    }
1680}

1683    f: &mut HirFormatter<'_, 'db>,
1684    parameters: &[GenericArg<'db>],
1685    generic_def: Option<hir_def::GenericDefId>,

1688    if parameters.is_empty() {
1689        return Ok(());
1690    }

1693
1694    if !parameters_to_write.is_empty() {
1695        write!(f, "<")?;

1697        write!(f, ">")?;
1698    }

1700    Ok(())

1703fn generic_args_sans_defaults<'ga, 'db>(
1704    f: &mut HirFormatter<'_, 'db>,
1705    generic_def: Option<hir_def::GenericDefId>,
1706    parameters: &'ga [GenericArg<'db>],
1707) -> &'ga [GenericArg<'db>] {
1708    if f.display_kind.is_source_code() || f.omit_verbose_types() {
1709        match generic_def.map(|generic_def_id| f.db.generic_defaults(generic_def_id)) {
1710            None => parameters,
1711            Some(default_parameters) => {
1712                let should_show = |arg: GenericArg<'db>, i: usize| match default_parameters.get(i) {
1713                    None => true,
1714                    Some(default_parameter) => {
1715                        arg != default_parameter.instantiate(f.interner, &parameters[..i])
1716                    }
1717                };
1718                let mut default_from = 0;
1719                for (i, &parameter) in parameters.iter().enumerate() {
1720                    if should_show(parameter, i) {
1721                        default_from = i + 1;
1722                    }
1723                }
1724                &parameters[0..default_from]
1725            }
1726        }
1727    } else {
1728        parameters
1729    }
1730}

1753fn hir_fmt_generic_arguments<'db>(
1754    f: &mut HirFormatter<'_, 'db>,
1755    parameters: &[GenericArg<'db>],
1756    self_: Option<Ty<'db>>,
1757) -> Result<(), HirDisplayError> {
1758    let mut first = true;
1759    let lifetime_offset = parameters.iter().position(|arg| arg.region().is_some());
1760
1761    let (ty_or_const, lifetimes) = match lifetime_offset {
1762        Some(offset) => parameters.split_at(offset),
1763        None => (parameters, &[][..]),
1764    };
1765    for generic_arg in lifetimes.iter().chain(ty_or_const) {
1766        if !mem::take(&mut first) {
1767            write!(f, ", ")?;
1768        }
1769        match self_ {
1770            self_ @ Some(_) if generic_arg.ty() == self_ => write!(f, "Self")?,
1771            _ => generic_arg.hir_fmt(f)?,
1772        }
1773    }
1774    Ok(())
1775}

1790            _ => ty.hir_fmt(f)?,
1791        }

1877fn write_bounds_like_dyn_trait<'db>(
1878    f: &mut HirFormatter<'_, 'db>,
1879    this: Either<Ty<'db>, Region<'db>>,
1880    predicates: &[Clause<'db>],
1881    default_sized: SizedByDefault,
1882) -> Result<(), HirDisplayError> {
1883    // Note: This code is written to produce nice results (i.e.
1884    // corresponding to surface Rust) for types that can occur in
1885    // actual Rust. It will have weird results if the predicates
1886    // aren't as expected (i.e. self types = $0, projection
1887    // predicates for a certain trait come after the Implemented
1888    // predicate for that trait).
1889    let mut first = true;
1890    let mut angle_open = false;
1891    let mut is_fn_trait = false;
1892    let mut is_sized = false;
1893    for p in predicates {
1894        match p.kind().skip_binder() {
1895            ClauseKind::Trait(trait_ref) => {
1896                let trait_ = trait_ref.def_id().0;
1897                if default_sized.is_sized_trait(trait_, f.db) {
1898                    is_sized = true;
1899                    if matches!(default_sized, SizedByDefault::Sized { .. }) {
1900                        // Don't print +Sized, but rather +?Sized if absent.
1901                        continue;
1902                    }
1903                }
1904                if !is_fn_trait {
1905                    is_fn_trait = fn_traits(f.db, trait_).any(|it| it == trait_);
1906                }
1907                if !is_fn_trait && angle_open {
1908                    write!(f, ">")?;
1909                    angle_open = false;
1910                }
1911                if !first {
1912                    write!(f, " + ")?;
1913                }
1914                // We assume that the self type is ^0.0 (i.e. the
1915                // existential) here, which is the only thing that's
1916                // possible in actual Rust, and hence don't print it
1917                f.start_location_link(trait_.into());
1918                write!(f, "{}", f.db.trait_signature(trait_).name.display(f.db, f.edition()))?;
1919                f.end_location_link();
1920                if is_fn_trait {
1921                    if let [_self, params @ ..] = trait_ref.trait_ref.args.as_slice()
1922                        && let Some(args) = params.first().and_then(|it| it.ty()?.as_tuple())
1923                    {
1924                        write!(f, "(")?;
1925                        hir_fmt_tys(f, args.as_slice(), Some(trait_ref.trait_ref.self_ty()))?;
1926                        write!(f, ")")?;
1927                    }
1928                } else {
1929                    let params = generic_args_sans_defaults(
1930                        f,
1931                        Some(trait_.into()),
1932                        trait_ref.trait_ref.args.as_slice(),
1933                    );
1934                    if let [_self, params @ ..] = params
1935                        && !params.is_empty()
1936                    {
1937                        write!(f, "<")?;
1938                        hir_fmt_generic_arguments(f, params, Some(trait_ref.trait_ref.self_ty()))?;
1939                        // there might be assoc type bindings, so we leave the angle brackets open
1940                        angle_open = true;
1941                    }
1942                }
1943            }
1944            ClauseKind::TypeOutlives(to) if Either::Left(to.0) == this => {
1945                if !is_fn_trait && angle_open {
1946                    write!(f, ">")?;
1947                    angle_open = false;
1948                }
1949                if !first {
1950                    write!(f, " + ")?;
1951                }
1952                to.1.hir_fmt(f)?;
1953            }
1954            ClauseKind::RegionOutlives(lo) if Either::Right(lo.0) == this => {
1955                if !is_fn_trait && angle_open {
1956                    write!(f, ">")?;
1957                    angle_open = false;
1958                }
1959                if !first {
1960                    write!(f, " + ")?;
1961                }
1962                lo.1.hir_fmt(f)?;
1963            }
1964            ClauseKind::Projection(projection) if is_fn_trait => {
1965                is_fn_trait = false;
1966                if !projection.term.as_type().is_some_and(|it| it.is_unit()) {
1967                    write!(f, " -> ")?;
1968                    projection.term.hir_fmt(f)?;
1969                }
1970            }
1971            ClauseKind::Projection(projection) => {
1972                // in types in actual Rust, these will always come
1973                // after the corresponding Implemented predicate
1974                if angle_open {
1975                    write!(f, ", ")?;
1976                } else {
1977                    write!(f, "<")?;
1978                    angle_open = true;
1979                }
1980                let assoc_ty_id = projection.def_id().expect_type_alias();
1981                let type_alias = f.db.type_alias_signature(assoc_ty_id);
1982                f.start_location_link(assoc_ty_id.into());
1983                write!(f, "{}", type_alias.name.display(f.db, f.edition()))?;
1984                f.end_location_link();
1985
1986                let own_args = projection.projection_term.own_args(f.interner);
1987                if !own_args.is_empty() {
1988                    write!(f, "<")?;
1989                    hir_fmt_generic_arguments(f, own_args.as_slice(), None)?;
1990                    write!(f, ">")?;
1991                }
1992                write!(f, " = ")?;
1993                projection.term.hir_fmt(f)?;
1994            }
1995            _ => {}
1996        }
1997        first = false;
1998    }
1999    if angle_open {
2000        write!(f, ">")?;
2001    }
2002    if let SizedByDefault::Sized { anchor } = default_sized {
2003        let sized_trait = LangItem::Sized.resolve_trait(f.db, anchor);
2004        if !is_sized {
2005            if !first {
2006                write!(f, " + ")?;
2007            }
2008            if let Some(sized_trait) = sized_trait {
2009                f.start_location_link(sized_trait.into());
2010            }
2011            write!(f, "?Sized")?;
2012        } else if first {
2013            if let Some(sized_trait) = sized_trait {
2014                f.start_location_link(sized_trait.into());
2015            }
2016            write!(f, "Sized")?;
2017        }
2018        if sized_trait.is_some() {
2019            f.end_location_link();
2020        }
2021    }
2022    Ok(())
2023}

2067pub fn write_visibility<'db>(
2068    module_id: ModuleId,
2069    vis: Visibility,
2070    f: &mut HirFormatter<'_, 'db>,
2071) -> Result<(), HirDisplayError> {
2072    match vis {
2073        Visibility::Public => write!(f, "pub "),
2074        Visibility::PubCrate(_) => write!(f, "pub(crate) "),
2075        Visibility::Module(vis_id, _) => {
2076            let def_map = module_id.def_map(f.db);
2077            let root_module_id = def_map.module_id(DefMap::ROOT);
2078            if vis_id == module_id {
2079                // pub(self) or omitted
2080                Ok(())
2081            } else if root_module_id == vis_id && !root_module_id.is_within_block() {
2082                write!(f, "pub(crate) ")
2083            } else if module_id.containing_module(f.db) == Some(vis_id) && !vis_id.is_block_module()
2084            {
2085                write!(f, "pub(super) ")
2086            } else {
2087                write!(f, "pub(in ...) ")
2088            }
2089        }
2090    }
2091}

2342impl<'db> HirDisplayWithExpressionStore<'db> for Path {
2343    fn hir_fmt(
2344        &self,
2345        f: &mut HirFormatter<'_, 'db>,
2346        store: &ExpressionStore,
2347    ) -> Result<(), HirDisplayError> {
2348        match (self.type_anchor(), self.kind()) {
2349            (Some(anchor), _) => {
2350                write!(f, "<")?;
2351                anchor.hir_fmt(f, store)?;
2352                write!(f, ">")?;
2353            }
2354            (_, PathKind::Plain) => {}
2355            (_, PathKind::Abs) => {}
2356            (_, PathKind::Crate) => write!(f, "crate")?,
2357            (_, &PathKind::SELF) => write!(f, "self")?,
2358            (_, PathKind::Super(n)) => {
2359                for i in 0..*n {
2360                    if i > 0 {
2361                        write!(f, "::")?;
2362                    }
2363                    write!(f, "super")?;
2364                }
2365            }
2366            (_, PathKind::DollarCrate(id)) => {
2367                // Resolve `$crate` to the crate's display name.
2368                // FIXME: should use the dependency name instead if available, but that depends on
2369                // the crate invoking `HirDisplay`
2370                let crate_data = id.extra_data(f.db);
2371                let name = crate_data
2372                    .display_name
2373                    .as_ref()
2374                    .map(|name| (*name.canonical_name()).clone())
2375                    .unwrap_or(sym::dollar_crate);
2376                write!(f, "{name}")?
2377            }
2378        }
2379
2380        // Convert trait's `Self` bound back to the surface syntax. Note there is no associated
2381        // trait, so there can only be one path segment that `has_self_type`. The `Self` type
2382        // itself can contain further qualified path through, which will be handled by recursive
2383        // `hir_fmt`s.
2384        //
2385        // `trait_mod::Trait<Self = type_mod::Type, Args>::Assoc`
2386        // =>
2387        // `<type_mod::Type as trait_mod::Trait<Args>>::Assoc`
2388        let trait_self_ty = self.segments().iter().find_map(|seg| {
2389            let generic_args = seg.args_and_bindings?;
2390            generic_args.has_self_type.then(|| &generic_args.args[0])
2391        });
2392        if let Some(ty) = trait_self_ty {
2393            write!(f, "<")?;
2394            ty.hir_fmt(f, store)?;
2395            write!(f, " as ")?;
2396            // Now format the path of the trait...
2397        }
2398
2399        for (seg_idx, segment) in self.segments().iter().enumerate() {
2400            if !matches!(self.kind(), PathKind::Plain) || seg_idx > 0 {
2401                write!(f, "::")?;
2402            }
2403            write!(f, "{}", segment.name.display(f.db, f.edition()))?;
2404            if let Some(generic_args) = segment.args_and_bindings {
2405                // We should be in type context, so format as `Foo<Bar>` instead of `Foo::<Bar>`.
2406                // Do we actually format expressions?
2407                match generic_args.parenthesized {
2408                    hir_def::expr_store::path::GenericArgsParentheses::ReturnTypeNotation => {
2409                        write!(f, "(..)")?;
2410                    }
2411                    hir_def::expr_store::path::GenericArgsParentheses::ParenSugar => {
2412                        // First argument will be a tuple, which already includes the parentheses.
2413                        // If the tuple only contains 1 item, write it manually to avoid the trailing `,`.
2414                        let tuple = match generic_args.args[0] {
2415                            hir_def::expr_store::path::GenericArg::Type(ty) => match &store[ty] {
2416                                TypeRef::Tuple(it) => Some(it),
2417                                _ => None,
2418                            },
2419                            _ => None,
2420                        };
2421                        if let Some(v) = tuple {
2422                            if v.len() == 1 {
2423                                write!(f, "(")?;
2424                                v[0].hir_fmt(f, store)?;
2425                                write!(f, ")")?;
2426                            } else {
2427                                generic_args.args[0].hir_fmt(f, store)?;
2428                            }
2429                        }
2430                        if let Some(ret) = generic_args.bindings[0].type_ref
2431                            && !matches!(&store[ret], TypeRef::Tuple(v) if v.is_empty())
2432                        {
2433                            write!(f, " -> ")?;
2434                            ret.hir_fmt(f, store)?;
2435                        }
2436                    }
2437                    hir_def::expr_store::path::GenericArgsParentheses::No => {
2438                        let mut first = true;
2439                        // Skip the `Self` bound if exists. It's handled outside the loop.
2440                        for arg in &generic_args.args[generic_args.has_self_type as usize..] {
2441                            if first {
2442                                first = false;
2443                                write!(f, "<")?;
2444                            } else {
2445                                write!(f, ", ")?;
2446                            }
2447                            arg.hir_fmt(f, store)?;
2448                        }
2449                        for binding in generic_args.bindings.iter() {
2450                            if first {
2451                                first = false;
2452                                write!(f, "<")?;
2453                            } else {
2454                                write!(f, ", ")?;
2455                            }
2456                            write!(f, "{}", binding.name.display(f.db, f.edition()))?;
2457                            match &binding.type_ref {
2458                                Some(ty) => {
2459                                    write!(f, " = ")?;
2460                                    ty.hir_fmt(f, store)?
2461                                }
2462                                None => {
2463                                    write!(f, ": ")?;
2464                                    f.write_joined(
2465                                        binding
2466                                            .bounds
2467                                            .iter()
2468                                            .map(ExpressionStoreAdapter::wrap(store)),
2469                                        " + ",
2470                                    )?;
2471                                }
2472                            }
2473                        }
2474
2475                        // There may be no generic arguments to print, in case of a trait having only a
2476                        // single `Self` bound which is converted to `<Ty as Trait>::Assoc`.
2477                        if !first {
2478                            write!(f, ">")?;
2479                        }
2480
2481                        // Current position: `<Ty as Trait<Args>|`
2482                        if generic_args.has_self_type {
2483                            write!(f, ">")?;
2484                        }
2485                    }
2486                }
2487            }
2488        }
2489
2490        Ok(())
2491    }
2492}

2498        store: &ExpressionStore,

2507        }
