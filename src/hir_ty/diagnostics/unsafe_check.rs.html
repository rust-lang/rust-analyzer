22#[derive(Debug, Default)]
23pub struct MissingUnsafeResult {
24    pub unsafe_exprs: Vec<(ExprOrPatId, UnsafetyReason)>,
25    /// If `fn_is_unsafe` is false, `unsafe_exprs` are hard errors. If true, they're `unsafe_op_in_unsafe_fn`.
26    pub fn_is_unsafe: bool,
27    pub deprecated_safe_calls: Vec<ExprId>,
28}
29
30pub fn missing_unsafe(db: &dyn HirDatabase, def: DefWithBodyId) -> MissingUnsafeResult {
31    let _p = tracing::info_span!("missing_unsafe").entered();
32
33    let is_unsafe = match def {
34        DefWithBodyId::FunctionId(it) => db.function_signature(it).is_unsafe(),
35        DefWithBodyId::StaticId(_) | DefWithBodyId::ConstId(_) | DefWithBodyId::VariantId(_) => {
36            false
37        }
38    };
39
40    let mut res = MissingUnsafeResult { fn_is_unsafe: is_unsafe, ..MissingUnsafeResult::default() };
41    let body = db.body(def);
42    let infer = db.infer(def);
43    let mut callback = |diag| match diag {
44        UnsafeDiagnostic::UnsafeOperation { node, inside_unsafe_block, reason } => {
45            if inside_unsafe_block == InsideUnsafeBlock::No {
46                res.unsafe_exprs.push((node, reason));
47            }
48        }
49        UnsafeDiagnostic::DeprecatedSafe2024 { node, inside_unsafe_block } => {
50            if inside_unsafe_block == InsideUnsafeBlock::No {
51                res.deprecated_safe_calls.push(node)
52            }
53        }
54    };
55    let mut visitor = UnsafeVisitor::new(db, &infer, &body, def, &mut callback);
56    visitor.walk_expr(body.body_expr);
57
58    if !is_unsafe {
59        // Unsafety in function parameter patterns (that can only be union destructuring)
60        // cannot be inserted into an unsafe block, so even with `unsafe_op_in_unsafe_fn`
61        // it is turned off for unsafe functions.
62        for &param in &body.params {
63            visitor.walk_pat(param);
64        }
65    }
66
67    res
68}

70#[derive(Debug, Clone, Copy)]
71pub enum UnsafetyReason {
72    UnionField,
73    UnsafeFnCall,
74    InlineAsm,
75    RawPtrDeref,
76    MutableStatic,
77    ExternStatic,
78}
79
80#[derive(Debug, Clone, Copy, PartialEq, Eq)]
81pub enum InsideUnsafeBlock {
82    No,
83    Yes,
84}
85
86#[derive(Debug)]
87enum UnsafeDiagnostic {
88    UnsafeOperation {
89        node: ExprOrPatId,
90        inside_unsafe_block: InsideUnsafeBlock,
91        reason: UnsafetyReason,
92    },
93    /// A lint.
94    DeprecatedSafe2024 { node: ExprId, inside_unsafe_block: InsideUnsafeBlock },
95}

97pub fn unsafe_operations_for_body(
98    db: &dyn HirDatabase,
99    infer: &InferenceResult,
100    def: DefWithBodyId,
101    body: &Body,
102    callback: &mut dyn FnMut(ExprOrPatId),
103) {
104    let mut visitor_callback = |diag| {
105        if let UnsafeDiagnostic::UnsafeOperation { node, .. } = diag {
106            callback(node);
107        }
108    };
109    let mut visitor = UnsafeVisitor::new(db, infer, body, def, &mut visitor_callback);
110    visitor.walk_expr(body.body_expr);
111    for &param in &body.params {
112        visitor.walk_pat(param);
113    }
114}

116pub fn unsafe_operations(
117    db: &dyn HirDatabase,
118    infer: &InferenceResult,
119    def: DefWithBodyId,
120    body: &Body,
121    current: ExprId,
122    callback: &mut dyn FnMut(InsideUnsafeBlock),
123) {
124    let mut visitor_callback = |diag| {
125        if let UnsafeDiagnostic::UnsafeOperation { inside_unsafe_block, .. } = diag {
126            callback(inside_unsafe_block);
127        }
128    };
129    let mut visitor = UnsafeVisitor::new(db, infer, body, def, &mut visitor_callback);
130    _ = visitor.resolver.update_to_inner_scope(db, def, current);
131    visitor.walk_expr(current);
132}

134struct UnsafeVisitor<'a> {
135    db: &'a dyn HirDatabase,
136    infer: &'a InferenceResult,
137    body: &'a Body,
138    resolver: Resolver,
139    def: DefWithBodyId,
140    inside_unsafe_block: InsideUnsafeBlock,
141    inside_assignment: bool,
142    inside_union_destructure: bool,
143    callback: &'a mut dyn FnMut(UnsafeDiagnostic),
144    def_target_features: TargetFeatures,
145    // FIXME: This needs to be the edition of the span of each call.
146    edition: Edition,
147}
148
149impl<'a> UnsafeVisitor<'a> {
150    fn new(
151        db: &'a dyn HirDatabase,
152        infer: &'a InferenceResult,
153        body: &'a Body,
154        def: DefWithBodyId,
155        unsafe_expr_cb: &'a mut dyn FnMut(UnsafeDiagnostic),
156    ) -> Self {
157        let resolver = def.resolver(db);
158        let def_target_features = match def {
159            DefWithBodyId::FunctionId(func) => TargetFeatures::from_attrs(&db.attrs(func.into())),
160            _ => TargetFeatures::default(),
161        };
162        let edition = resolver.module().krate().data(db).edition;
163        Self {
164            db,
165            infer,
166            body,
167            resolver,
168            def,
169            inside_unsafe_block: InsideUnsafeBlock::No,
170            inside_assignment: false,
171            inside_union_destructure: false,
172            callback: unsafe_expr_cb,
173            def_target_features,
174            edition,
175        }
176    }
177
178    fn on_unsafe_op(&mut self, node: ExprOrPatId, reason: UnsafetyReason) {
179        (self.callback)(UnsafeDiagnostic::UnsafeOperation {
180            node,
181            inside_unsafe_block: self.inside_unsafe_block,
182            reason,
183        });
184    }
185
186    fn check_call(&mut self, node: ExprId, func: FunctionId) {
187        let unsafety = is_fn_unsafe_to_call(self.db, func, &self.def_target_features, self.edition);
188        match unsafety {
189            crate::utils::Unsafety::Safe => {}
190            crate::utils::Unsafety::Unsafe => {
191                self.on_unsafe_op(node.into(), UnsafetyReason::UnsafeFnCall)
192            }
193            crate::utils::Unsafety::DeprecatedSafe2024 => {
194                (self.callback)(UnsafeDiagnostic::DeprecatedSafe2024 {
195                    node,
196                    inside_unsafe_block: self.inside_unsafe_block,
197                })
198            }
199        }
200    }
201
202    fn walk_pats_top(&mut self, pats: impl Iterator<Item = PatId>, parent_expr: ExprId) {
203        let guard = self.resolver.update_to_inner_scope(self.db, self.def, parent_expr);
204        pats.for_each(|pat| self.walk_pat(pat));
205        self.resolver.reset_to_guard(guard);
206    }
207
208    fn walk_pat(&mut self, current: PatId) {
209        let pat = &self.body.pats[current];
210
211        if self.inside_union_destructure {
212            match pat {
213                Pat::Tuple { .. }
214                | Pat::Record { .. }
215                | Pat::Range { .. }
216                | Pat::Slice { .. }
217                | Pat::Path(..)
218                | Pat::Lit(..)
219                | Pat::Bind { .. }
220                | Pat::TupleStruct { .. }
221                | Pat::Ref { .. }
222                | Pat::Box { .. }
223                | Pat::Expr(..)
224                | Pat::ConstBlock(..) => {
225                    self.on_unsafe_op(current.into(), UnsafetyReason::UnionField)
226                }
227                // `Or` only wraps other patterns, and `Missing`/`Wild` do not constitute a read.
228                Pat::Missing | Pat::Wild | Pat::Or(_) => {}
229            }
230        }
231
232        match pat {
233            Pat::Record { .. } => {
234                if let Some((AdtId::UnionId(_), _)) = self.infer[current].as_adt() {
235                    let old_inside_union_destructure =
236                        mem::replace(&mut self.inside_union_destructure, true);
237                    self.body.walk_pats_shallow(current, |pat| self.walk_pat(pat));
238                    self.inside_union_destructure = old_inside_union_destructure;
239                    return;
240                }
241            }
242            Pat::Path(path) => self.mark_unsafe_path(current.into(), path),
243            &Pat::ConstBlock(expr) => {
244                let old_inside_assignment = mem::replace(&mut self.inside_assignment, false);
245                self.walk_expr(expr);
246                self.inside_assignment = old_inside_assignment;
247            }
248            &Pat::Expr(expr) => self.walk_expr(expr),
249            _ => {}
250        }
251
252        self.body.walk_pats_shallow(current, |pat| self.walk_pat(pat));
253    }
254
255    fn walk_expr(&mut self, current: ExprId) {
256        let expr = &self.body.exprs[current];
257        let inside_assignment = mem::replace(&mut self.inside_assignment, false);
258        match expr {
259            &Expr::Call { callee, .. } => {
260                let callee = &self.infer[callee];
261                if let Some(func) = callee.as_fn_def(self.db) {
262                    self.check_call(current, func);
263                }
264                if let TyKind::Function(fn_ptr) = callee.kind(Interner) {
265                    if fn_ptr.sig.safety == chalk_ir::Safety::Unsafe {
266                        self.on_unsafe_op(current.into(), UnsafetyReason::UnsafeFnCall);
267                    }
268                }
269            }
270            Expr::Path(path) => {
271                let guard = self.resolver.update_to_inner_scope(self.db, self.def, current);
272                self.mark_unsafe_path(current.into(), path);
273                self.resolver.reset_to_guard(guard);
274            }
275            Expr::Ref { expr, rawness: Rawness::RawPtr, mutability: _ } => {
276                match self.body.exprs[*expr] {
277                    // Do not report unsafe for `addr_of[_mut]!(EXTERN_OR_MUT_STATIC)`,
278                    // see https://github.com/rust-lang/rust/pull/125834.
279                    Expr::Path(_) => return,
280                    // https://github.com/rust-lang/rust/pull/129248
281                    // Taking a raw ref to a deref place expr is always safe.
282                    Expr::UnaryOp { expr, op: UnaryOp::Deref } => {
283                        self.body
284                            .walk_child_exprs_without_pats(expr, |child| self.walk_expr(child));
285
286                        return;
287                    }
288                    _ => (),
289                }
290            }
291            Expr::MethodCall { .. } => {
292                if let Some((func, _)) = self.infer.method_resolution(current) {
293                    self.check_call(current, func);
294                }
295            }
296            Expr::UnaryOp { expr, op: UnaryOp::Deref } => {
297                if let TyKind::Raw(..) = &self.infer[*expr].kind(Interner) {
298                    self.on_unsafe_op(current.into(), UnsafetyReason::RawPtrDeref);
299                }
300            }
301            &Expr::Assignment { target, value: _ } => {
302                let old_inside_assignment = mem::replace(&mut self.inside_assignment, true);
303                self.walk_pats_top(std::iter::once(target), current);
304                self.inside_assignment = old_inside_assignment;
305            }
306            Expr::InlineAsm(_) => self.on_unsafe_op(current.into(), UnsafetyReason::InlineAsm),
307            // rustc allows union assignment to propagate through field accesses and casts.
308            Expr::Cast { .. } => self.inside_assignment = inside_assignment,
309            Expr::Field { .. } => {
310                self.inside_assignment = inside_assignment;
311                if !inside_assignment {
312                    if let Some(Either::Left(FieldId { parent: VariantId::UnionId(_), .. })) =
313                        self.infer.field_resolution(current)
314                    {
315                        self.on_unsafe_op(current.into(), UnsafetyReason::UnionField);
316                    }
317                }
318            }
319            Expr::Unsafe { statements, .. } => {
320                let old_inside_unsafe_block =
321                    mem::replace(&mut self.inside_unsafe_block, InsideUnsafeBlock::Yes);
322                self.walk_pats_top(
323                    statements.iter().filter_map(|statement| match statement {
324                        &Statement::Let { pat, .. } => Some(pat),
325                        _ => None,
326                    }),
327                    current,
328                );
329                self.body.walk_child_exprs_without_pats(current, |child| self.walk_expr(child));
330                self.inside_unsafe_block = old_inside_unsafe_block;
331                return;
332            }
333            Expr::Block { statements, .. } | Expr::Async { statements, .. } => {
334                self.walk_pats_top(
335                    statements.iter().filter_map(|statement| match statement {
336                        &Statement::Let { pat, .. } => Some(pat),
337                        _ => None,
338                    }),
339                    current,
340                );
341            }
342            Expr::Match { arms, .. } => {
343                self.walk_pats_top(arms.iter().map(|arm| arm.pat), current);
344            }
345            &Expr::Let { pat, .. } => {
346                self.walk_pats_top(std::iter::once(pat), current);
347            }
348            Expr::Closure { args, .. } => {
349                self.walk_pats_top(args.iter().copied(), current);
350            }
351            Expr::Const(e) => self.walk_expr(*e),
352            _ => {}
353        }
354
355        self.body.walk_child_exprs_without_pats(current, |child| self.walk_expr(child));
356    }
357
358    fn mark_unsafe_path(&mut self, node: ExprOrPatId, path: &Path) {
359        let hygiene = self.body.expr_or_pat_path_hygiene(node);
360        let value_or_partial = self.resolver.resolve_path_in_value_ns(self.db, path, hygiene);
361        if let Some(ResolveValueResult::ValueNs(ValueNs::StaticId(id), _)) = value_or_partial {
362            let static_data = self.db.static_signature(id);
363            if static_data.flags.contains(StaticFlags::MUTABLE) {
364                self.on_unsafe_op(node, UnsafetyReason::MutableStatic);
365            } else if static_data.flags.contains(StaticFlags::EXTERN)
366                && !static_data.flags.contains(StaticFlags::EXPLICIT_SAFE)
367            {
368                self.on_unsafe_op(node, UnsafetyReason::ExternStatic);
369            }
370        }
371    }
