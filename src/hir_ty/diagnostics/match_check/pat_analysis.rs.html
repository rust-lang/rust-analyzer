44/// An index type for enum variants. This ranges from 0 to `variants.len()`, whereas `EnumVariantId`
45/// can take arbitrary large values (and hence mustn't be used with `IndexVec`/`BitSet`).
46#[derive(Copy, Clone, Debug, PartialEq, Eq, Hash)]
47pub(crate) struct EnumVariantContiguousIndex(usize);

49impl EnumVariantContiguousIndex {
50    fn from_enum_variant_id(db: &dyn HirDatabase, target_evid: EnumVariantId) -> Self {
51        // Find the index of this variant in the list of variants.
52        use hir_def::Lookup;
53        let i = target_evid.lookup(db).index as usize;
54        EnumVariantContiguousIndex(i)
55    }
56
57    fn to_enum_variant_id(self, db: &dyn HirDatabase, eid: EnumId) -> EnumVariantId {
58        eid.enum_variants(db).variants[self.0].0
59    }
60}

82    pub(crate) fn new(
83        module: ModuleId,
84        infcx: &'a InferCtxt<'db>,
85        env: Arc<TraitEnvironment<'db>>,
86    ) -> Self {
87        let db = infcx.interner.db;
88        let def_map = module.crate_def_map(db);
89        let exhaustive_patterns = def_map.is_unstable_feature_enabled(&sym::exhaustive_patterns);
90        Self { module, db, exhaustive_patterns, env, infcx }
91    }

93    pub(crate) fn compute_match_usefulness<'b>(
94        &self,
95        arms: &[MatchArm<'a, 'b, 'db>],
96        scrut_ty: Ty<'db>,
97        known_valid_scrutinee: Option<bool>,
98    ) -> Result<UsefulnessReport<'b, Self>, ()> {
99        if scrut_ty.references_non_lt_error() {
100            return Err(());
101        }
102        for arm in arms {
103            if arm.pat.ty().references_non_lt_error() {
104                return Err(());
105            }
106        }
107
108        let place_validity = PlaceValidity::from_bool(known_valid_scrutinee.unwrap_or(true));
109        // Measured to take ~100ms on modern hardware.
110        let complexity_limit = 500000;
111        compute_match_usefulness(self, arms, scrut_ty, place_validity, complexity_limit)
112    }

114    fn is_uninhabited(&self, ty: Ty<'db>) -> bool {
115        is_ty_uninhabited_from(self.infcx, ty, self.module, self.env.clone())
116    }

118    /// Returns whether the given ADT is from another crate declared `#[non_exhaustive]`.
119    fn is_foreign_non_exhaustive(&self, adt: hir_def::AdtId) -> bool {
120        let is_local = adt.krate(self.db) == self.module.krate();
121        !is_local && self.db.attrs(adt.into()).by_key(sym::non_exhaustive).exists()
122    }

124    fn variant_id_for_adt(
125        db: &'db dyn HirDatabase,
126        ctor: &Constructor<Self>,
127        adt: hir_def::AdtId,
128    ) -> Option<VariantId> {
129        match ctor {
130            Variant(id) => {
131                let hir_def::AdtId::EnumId(eid) = adt else {
132                    panic!("bad constructor {ctor:?} for adt {adt:?}")
133                };
134                Some(id.to_enum_variant_id(db, eid).into())
135            }
136            Struct | UnionField => match adt {
137                hir_def::AdtId::EnumId(_) => None,
138                hir_def::AdtId::StructId(id) => Some(id.into()),
139                hir_def::AdtId::UnionId(id) => Some(id.into()),
140            },
141            _ => panic!("bad constructor {ctor:?} for adt {adt:?}"),
142        }
143    }

145    // This lists the fields of a variant along with their types.
146    fn list_variant_fields(
147        &self,
148        ty: Ty<'db>,
149        variant: VariantId,
150    ) -> impl Iterator<Item = (LocalFieldId, Ty<'db>)> {
151        let (_, substs) = ty.as_adt().unwrap();
152
153        let field_tys = self.db.field_types(variant);
154        let fields_len = variant.fields(self.db).fields().len() as u32;
155
156        (0..fields_len).map(|idx| LocalFieldId::from_raw(idx.into())).map(move |fid| {
157            let ty = field_tys[fid].instantiate(self.infcx.interner, substs);
158            let ty = self
159                .infcx
160                .at(&ObligationCause::dummy(), self.env.env)
161                .deeply_normalize(ty)
162                .unwrap_or(ty);
163            (fid, ty)
164        })
165    }

167    pub(crate) fn lower_pat(&self, pat: &Pat<'db>) -> DeconstructedPat<'a, 'db> {
168        let singleton = |pat: DeconstructedPat<'a, 'db>| vec![pat.at_index(0)];
169        let ctor;
170        let mut fields: Vec<_>;
171        let arity;
172
173        match pat.kind.as_ref() {
174            PatKind::Binding { subpattern: Some(subpat), .. } => return self.lower_pat(subpat),
175            PatKind::Binding { subpattern: None, .. } | PatKind::Wild => {
176                ctor = Wildcard;
177                fields = Vec::new();
178                arity = 0;
179            }
180            PatKind::Deref { subpattern } => {
181                ctor = match pat.ty.kind() {
182                    TyKind::Ref(..) => Ref,
183                    _ => {
184                        never!("pattern has unexpected type: pat: {:?}, ty: {:?}", pat, &pat.ty);
185                        Wildcard
186                    }
187                };
188                fields = singleton(self.lower_pat(subpattern));
189                arity = 1;
190            }
191            PatKind::Leaf { subpatterns } | PatKind::Variant { subpatterns, .. } => {
192                fields = subpatterns
193                    .iter()
194                    .map(|pat| {
195                        let idx: u32 = pat.field.into_raw().into();
196                        self.lower_pat(&pat.pattern).at_index(idx as usize)
197                    })
198                    .collect();
199                match pat.ty.kind() {
200                    TyKind::Tuple(substs) => {
201                        ctor = Struct;
202                        arity = substs.len();
203                    }
204                    TyKind::Adt(adt_def, _) => {
205                        let adt = adt_def.def_id().0;
206                        ctor = match pat.kind.as_ref() {
207                            PatKind::Leaf { .. } if matches!(adt, hir_def::AdtId::UnionId(_)) => {
208                                UnionField
209                            }
210                            PatKind::Leaf { .. } => Struct,
211                            PatKind::Variant { enum_variant, .. } => {
212                                Variant(EnumVariantContiguousIndex::from_enum_variant_id(
213                                    self.db,
214                                    *enum_variant,
215                                ))
216                            }
217                            _ => {
218                                never!();
219                                Wildcard
220                            }
221                        };
222                        let variant = Self::variant_id_for_adt(self.db, &ctor, adt).unwrap();
223                        arity = variant.fields(self.db).fields().len();
224                    }
225                    _ => {
226                        never!("pattern has unexpected type: pat: {:?}, ty: {:?}", pat, &pat.ty);
227                        ctor = Wildcard;
228                        fields.clear();
229                        arity = 0;
230                    }
231                }
232            }
233            &PatKind::LiteralBool { value } => {
234                ctor = Bool(value);
235                fields = Vec::new();
236                arity = 0;
237            }
238            PatKind::Never => {
239                ctor = Never;
240                fields = Vec::new();
241                arity = 0;
242            }
243            PatKind::Or { pats } => {
244                ctor = Or;
245                fields = pats
246                    .iter()
247                    .enumerate()
248                    .map(|(i, pat)| self.lower_pat(pat).at_index(i))
249                    .collect();
250                arity = pats.len();
251            }
252        }
253        DeconstructedPat::new(ctor, fields, arity, pat.ty, ())
254    }

256    pub(crate) fn hoist_witness_pat(&self, pat: &WitnessPat<'a, 'db>) -> Pat<'db> {
257        let mut subpatterns = pat.iter_fields().map(|p| self.hoist_witness_pat(p));
258        let kind = match pat.ctor() {
259            &Bool(value) => PatKind::LiteralBool { value },
260            IntRange(_) => unimplemented!(),
261            Struct | Variant(_) | UnionField => match pat.ty().kind() {
262                TyKind::Tuple(..) => PatKind::Leaf {
263                    subpatterns: subpatterns
264                        .zip(0u32..)
265                        .map(|(p, i)| FieldPat {
266                            field: LocalFieldId::from_raw(i.into()),
267                            pattern: p,
268                        })
269                        .collect(),
270                },
271                TyKind::Adt(adt, substs) => {
272                    let variant =
273                        Self::variant_id_for_adt(self.db, pat.ctor(), adt.def_id().0).unwrap();
274                    let subpatterns = self
275                        .list_variant_fields(*pat.ty(), variant)
276                        .zip(subpatterns)
277                        .map(|((field, _ty), pattern)| FieldPat { field, pattern })
278                        .collect();
279
280                    if let VariantId::EnumVariantId(enum_variant) = variant {
281                        PatKind::Variant { substs, enum_variant, subpatterns }
282                    } else {
283                        PatKind::Leaf { subpatterns }
284                    }
285                }
286                _ => {
287                    never!("unexpected ctor for type {:?} {:?}", pat.ctor(), pat.ty());
288                    PatKind::Wild
289                }
290            },
291            // Note: given the expansion of `&str` patterns done in `expand_pattern`, we should
292            // be careful to reconstruct the correct constant pattern here. However a string
293            // literal pattern will never be reported as a non-exhaustiveness witness, so we
294            // ignore this issue.
295            Ref => PatKind::Deref { subpattern: subpatterns.next().unwrap() },
296            Slice(_) => unimplemented!(),
297            DerefPattern(_) => unimplemented!(),
298            &Str(void) => match void {},
299            Wildcard | NonExhaustive | Hidden | PrivateUninhabited => PatKind::Wild,
300            Never => PatKind::Never,
301            Missing | F16Range(..) | F32Range(..) | F64Range(..) | F128Range(..) | Opaque(..)
302            | Or => {
303                never!("can't convert to pattern: {:?}", pat.ctor());
304                PatKind::Wild
305            }
306        };
307        Pat { ty: *pat.ty(), kind: Box::new(kind) }
308    }

323    fn ctor_arity(
324        &self,
325        ctor: &rustc_pattern_analysis::constructor::Constructor<Self>,
326        ty: &Self::Ty,
327    ) -> usize {
328        match ctor {
329            Struct | Variant(_) | UnionField => match ty.kind() {
330                TyKind::Tuple(tys) => tys.len(),
331                TyKind::Adt(adt_def, ..) => {
332                    let variant =
333                        Self::variant_id_for_adt(self.db, ctor, adt_def.def_id().0).unwrap();
334                    variant.fields(self.db).fields().len()
335                }
336                _ => {
337                    never!("Unexpected type for `Single` constructor: {:?}", ty);
338                    0
339                }
340            },
341            Ref => 1,
342            Slice(..) => unimplemented!(),
343            DerefPattern(..) => unimplemented!(),
344            Never | Bool(..) | IntRange(..) | F16Range(..) | F32Range(..) | F64Range(..)
345            | F128Range(..) | Str(..) | Opaque(..) | NonExhaustive | PrivateUninhabited
346            | Hidden | Missing | Wildcard => 0,
347            Or => {
348                never!("The `Or` constructor doesn't have a fixed arity");
349                0
350            }
351        }
352    }
353
354    fn ctor_sub_tys(
355        &self,
356        ctor: &rustc_pattern_analysis::constructor::Constructor<Self>,
357        ty: &Self::Ty,
358    ) -> impl ExactSizeIterator<Item = (Self::Ty, PrivateUninhabitedField)> {
359        let single = |ty| smallvec![(ty, PrivateUninhabitedField(false))];
360        let tys: SmallVec<[_; 2]> = match ctor {
361            Struct | Variant(_) | UnionField => match ty.kind() {
362                TyKind::Tuple(substs) => {
363                    substs.iter().map(|ty| (ty, PrivateUninhabitedField(false))).collect()
364                }
365                TyKind::Ref(_, rty, _) => single(rty),
366                TyKind::Adt(adt_def, ..) => {
367                    let adt = adt_def.def_id().0;
368                    let variant = Self::variant_id_for_adt(self.db, ctor, adt).unwrap();
369
370                    let visibilities = LazyCell::new(|| self.db.field_visibilities(variant));
371
372                    self.list_variant_fields(*ty, variant)
373                        .map(move |(fid, ty)| {
374                            let is_visible = || {
375                                matches!(adt, hir_def::AdtId::EnumId(..))
376                                    || visibilities[fid].is_visible_from(self.db, self.module)
377                            };
378                            let is_uninhabited = self.is_uninhabited(ty);
379                            let private_uninhabited = is_uninhabited && !is_visible();
380                            (ty, PrivateUninhabitedField(private_uninhabited))
381                        })
382                        .collect()
383                }
384                ty_kind => {
385                    never!("Unexpected type for `{:?}` constructor: {:?}", ctor, ty_kind);
386                    single(*ty)
387                }
388            },
389            Ref => match ty.kind() {
390                TyKind::Ref(_, rty, _) => single(rty),
391                ty_kind => {
392                    never!("Unexpected type for `{:?}` constructor: {:?}", ctor, ty_kind);
393                    single(*ty)
394                }
395            },
396            Slice(_) => unreachable!("Found a `Slice` constructor in match checking"),
397            DerefPattern(_) => unreachable!("Found a `DerefPattern` constructor in match checking"),
398            Never | Bool(..) | IntRange(..) | F16Range(..) | F32Range(..) | F64Range(..)
399            | F128Range(..) | Str(..) | Opaque(..) | NonExhaustive | PrivateUninhabited
400            | Hidden | Missing | Wildcard => {
401                smallvec![]
402            }
403            Or => {
404                never!("called `Fields::wildcards` on an `Or` ctor");
405                smallvec![]
406            }
407        };
408        tys.into_iter()
409    }

411    fn ctors_for_ty(
412        &self,
413        ty: &Self::Ty,
414    ) -> Result<rustc_pattern_analysis::constructor::ConstructorSet<Self>, Self::Error> {
415        let cx = self;
416
417        // Unhandled types are treated as non-exhaustive. Being explicit here instead of falling
418        // to catchall arm to ease further implementation.
419        let unhandled = || ConstructorSet::Unlistable;
420
421        // This determines the set of all possible constructors for the type `ty`. For numbers,
422        // arrays and slices we use ranges and variable-length slices when appropriate.
423        //
424        // If the `exhaustive_patterns` feature is enabled, we make sure to omit constructors that
425        // are statically impossible. E.g., for `Option<!>`, we do not include `Some(_)` in the
426        // returned list of constructors.
427        // Invariant: this is empty if and only if the type is uninhabited (as determined by
428        // `cx.is_uninhabited()`).
429        Ok(match ty.kind() {
430            TyKind::Bool => ConstructorSet::Bool,
431            TyKind::Char => unhandled(),
432            TyKind::Int(..) | TyKind::Uint(..) => unhandled(),
433            TyKind::Array(..) | TyKind::Slice(..) => unhandled(),
434            TyKind::Adt(adt_def, subst) => {
435                let adt = adt_def.def_id().0;
436                match adt {
437                    hir_def::AdtId::EnumId(enum_id) => {
438                        let enum_data = enum_id.enum_variants(cx.db);
439                        let is_declared_nonexhaustive = cx.is_foreign_non_exhaustive(adt);
440
441                        if enum_data.variants.is_empty() && !is_declared_nonexhaustive {
442                            ConstructorSet::NoConstructors
443                        } else {
444                            let mut variants = IndexVec::with_capacity(enum_data.variants.len());
445                            for &(variant, _, _) in enum_data.variants.iter() {
446                                let is_uninhabited = is_enum_variant_uninhabited_from(
447                                    cx.infcx,
448                                    variant,
449                                    subst,
450                                    cx.module,
451                                    self.env.clone(),
452                                );
453                                let visibility = if is_uninhabited {
454                                    VariantVisibility::Empty
455                                } else {
456                                    VariantVisibility::Visible
457                                };
458                                variants.push(visibility);
459                            }
460
461                            ConstructorSet::Variants {
462                                variants,
463                                non_exhaustive: is_declared_nonexhaustive,
464                            }
465                        }
466                    }
467                    hir_def::AdtId::UnionId(_) => ConstructorSet::Union,
468                    hir_def::AdtId::StructId(_) => {
469                        ConstructorSet::Struct { empty: cx.is_uninhabited(*ty) }
470                    }
471                }
472            }
473            TyKind::Tuple(..) => ConstructorSet::Struct { empty: cx.is_uninhabited(*ty) },
474            TyKind::Ref(..) => ConstructorSet::Ref,
475            TyKind::Never => ConstructorSet::NoConstructors,
476            // This type is one for which we cannot list constructors, like `str` or `f64`.
477            _ => ConstructorSet::Unlistable,
478        })
479    }
