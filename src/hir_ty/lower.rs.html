98#[derive(Debug, Clone, Copy)]
99pub enum LifetimeElisionKind<'db> {
100    /// Create a new anonymous lifetime parameter and reference it.
101    ///
102    /// If `report_in_path`, report an error when encountering lifetime elision in a path:
103    /// ```compile_fail
104    /// struct Foo<'a> { x: &'a () }
105    /// async fn foo(x: Foo) {}
106    /// ```
107    ///
108    /// Note: the error should not trigger when the elided lifetime is in a pattern or
109    /// expression-position path:
110    /// ```
111    /// struct Foo<'a> { x: &'a () }
112    /// async fn foo(Foo { x: _ }: Foo<'_>) {}
113    /// ```
114    AnonymousCreateParameter { report_in_path: bool },
115
116    /// Replace all anonymous lifetimes by provided lifetime.
117    Elided(Region<'db>),
118
119    /// Give a hard error when either `&` or `'_` is written. Used to
120    /// rule out things like `where T: Foo<'_>`. Does not imply an
121    /// error on default object bounds (e.g., `Box<dyn Foo>`).
122    AnonymousReportError,
123
124    /// Resolves elided lifetimes to `'static` if there are no other lifetimes in scope,
125    /// otherwise give a warning that the previous behavior of introducing a new early-bound
126    /// lifetime is a bug and will be removed (if `only_lint` is enabled).
127    StaticIfNoLifetimeInScope { only_lint: bool },
128
129    /// Signal we cannot find which should be the anonymous lifetime.
130    ElisionFailure,
131
132    /// Infer all elided lifetimes.
133    Infer,
134}

282    pub(crate) fn lower_const(&mut self, const_ref: ConstRef, const_type: Ty<'db>) -> Const<'db> {
283        let const_ref = &self.store[const_ref.expr];
284        match const_ref {
285            hir_def::hir::Expr::Path(path) => {
286                self.path_to_const(path).unwrap_or_else(|| unknown_const(const_type))
287            }
288            hir_def::hir::Expr::Literal(literal) => {
289                intern_const_ref(self.db, literal, const_type, self.resolver.krate())
290            }
291            hir_def::hir::Expr::UnaryOp { expr: inner_expr, op: hir_def::hir::UnaryOp::Neg } => {
292                if let hir_def::hir::Expr::Literal(literal) = &self.store[*inner_expr] {
293                    // Only handle negation for signed integers and floats
294                    match literal {
295                        hir_def::hir::Literal::Int(_, _) | hir_def::hir::Literal::Float(_, _) => {
296                            if let Some(negated_literal) = literal.clone().negate() {
297                                intern_const_ref(
298                                    self.db,
299                                    &negated_literal,
300                                    const_type,
301                                    self.resolver.krate(),
302                                )
303                            } else {
304                                unknown_const(const_type)
305                            }
306                        }
307                        // For unsigned integers, chars, bools, etc., negation is not meaningful
308                        _ => unknown_const(const_type),
309                    }
310                } else {
311                    unknown_const(const_type)
312                }
313            }
314            _ => unknown_const(const_type),
315        }
316    }

318    pub(crate) fn path_to_const(&mut self, path: &Path) -> Option<Const<'db>> {
319        match self.resolver.resolve_path_in_value_ns_fully(self.db, path, HygieneId::ROOT) {
320            Some(ValueNs::GenericParam(p)) => {
321                let args = self.generics();
322                match args.type_or_const_param_idx(p.into()) {
323                    Some(idx) => Some(self.const_param(p, idx as u32)),
324                    None => {
325                        never!(
326                            "Generic list doesn't contain this param: {:?}, {:?}, {:?}",
327                            args,
328                            path,
329                            p
330                        );
331                        None
332                    }
333                }
334            }
335            Some(ValueNs::ConstId(c)) => {
336                let args = GenericArgs::empty(self.interner);
337                Some(Const::new(
338                    self.interner,
339                    rustc_type_ir::ConstKind::Unevaluated(UnevaluatedConst::new(
340                        GeneralConstId::ConstId(c).into(),
341                        args,
342                    )),
343                ))
344            }
345            _ => None,
346        }
347    }

389    #[tracing::instrument(skip(self), ret)]
390    pub fn lower_ty_ext(&mut self, type_ref_id: TypeRefId) -> (Ty<'db>, Option<TypeNs>) {
391        let interner = self.interner;
392        let mut res = None;
393        let type_ref = &self.store[type_ref_id];
394        tracing::debug!(?type_ref);
395        let ty = match type_ref {
396            TypeRef::Never => self.types.types.never,
397            TypeRef::Tuple(inner) => {
398                let inner_tys = inner.iter().map(|&tr| self.lower_ty(tr));
399                Ty::new_tup_from_iter(interner, inner_tys)
400            }
401            TypeRef::Path(path) => {
402                let (ty, res_) =
403                    self.lower_path(path, PathId::from_type_ref_unchecked(type_ref_id));
404                res = res_;
405                ty
406            }
407            &TypeRef::TypeParam(type_param_id) => {
408                res = Some(TypeNs::GenericParam(type_param_id));
409
410                let generics = self.generics();
411                let (idx, _data) =
412                    generics.type_or_const_param(type_param_id.into()).expect("matching generics");
413                self.type_param(type_param_id, idx as u32)
414            }
415            &TypeRef::RawPtr(inner, mutability) => {
416                let inner_ty = self.lower_ty(inner);
417                Ty::new(interner, TyKind::RawPtr(inner_ty, lower_mutability(mutability)))
418            }
419            TypeRef::Array(array) => {
420                let inner_ty = self.lower_ty(array.ty);
421                let const_len = self.lower_const(array.len, Ty::new_usize(interner));
422                Ty::new_array_with_const_len(interner, inner_ty, const_len)
423            }
424            &TypeRef::Slice(inner) => {
425                let inner_ty = self.lower_ty(inner);
426                Ty::new_slice(interner, inner_ty)
427            }
428            TypeRef::Reference(ref_) => {
429                let inner_ty = self.lower_ty(ref_.ty);
430                // FIXME: It should infer the eldided lifetimes instead of stubbing with error
431                let lifetime = ref_
432                    .lifetime
433                    .map_or_else(|| Region::error(interner), |lr| self.lower_lifetime(lr));
434                Ty::new_ref(interner, lifetime, inner_ty, lower_mutability(ref_.mutability))
435            }
436            TypeRef::Placeholder => Ty::new_error(interner, ErrorGuaranteed),
437            TypeRef::Fn(fn_) => self.lower_fn_ptr(fn_),
438            TypeRef::DynTrait(bounds) => self.lower_dyn_trait(bounds),
439            TypeRef::ImplTrait(bounds) => {
440                match self.impl_trait_mode.mode {
441                    ImplTraitLoweringMode::Opaque => {
442                        let origin = match self.resolver.generic_def() {
443                            Some(GenericDefId::FunctionId(it)) => Either::Left(it),
444                            Some(GenericDefId::TypeAliasId(it)) => Either::Right(it),
445                            _ => panic!(
446                                "opaque impl trait lowering must be in function or type alias"
447                            ),
448                        };
449
450                        // this dance is to make sure the data is in the right
451                        // place even if we encounter more opaque types while
452                        // lowering the bounds
453                        let idx = self.impl_trait_mode.opaque_type_data.alloc(ImplTrait {
454                            predicates: Clauses::empty(interner).store(),
455                            assoc_ty_bounds_start: 0,
456                        });
457
458                        let impl_trait_id = origin.either(
459                            |f| ImplTraitId::ReturnTypeImplTrait(f, idx),
460                            |a| ImplTraitId::TypeAliasImplTrait(a, idx),
461                        );
462                        let opaque_ty_id: SolverDefId =
463                            self.db.intern_impl_trait_id(impl_trait_id).into();
464
465                        // We don't want to lower the bounds inside the binders
466                        // we're currently in, because they don't end up inside
467                        // those binders. E.g. when we have `impl Trait<impl
468                        // OtherTrait<T>>`, the `impl OtherTrait<T>` can't refer
469                        // to the self parameter from `impl Trait`, and the
470                        // bounds aren't actually stored nested within each
471                        // other, but separately. So if the `T` refers to a type
472                        // parameter of the outer function, it's just one binder
473                        // away instead of two.
474                        let actual_opaque_type_data = self
475                            .with_debruijn(DebruijnIndex::ZERO, |ctx| {
476                                ctx.lower_impl_trait(opaque_ty_id, bounds)
477                            });
478                        self.impl_trait_mode.opaque_type_data[idx] = actual_opaque_type_data;
479
480                        let args = GenericArgs::identity_for_item(self.interner, opaque_ty_id);
481                        Ty::new_alias(
482                            self.interner,
483                            AliasTyKind::Opaque,
484                            AliasTy::new_from_args(self.interner, opaque_ty_id, args),
485                        )
486                    }
487                    ImplTraitLoweringMode::Disallowed => {
488                        // FIXME: report error
489                        Ty::new_error(self.interner, ErrorGuaranteed)
490                    }
491                }
492            }
493            TypeRef::Error => Ty::new_error(self.interner, ErrorGuaranteed),
494        };
495        (ty, res)
496    }

498    fn lower_fn_ptr(&mut self, fn_: &FnType) -> Ty<'db> {
499        let interner = self.interner;
500        let (params, ret_ty) = fn_.split_params_and_ret();
501        let old_lifetime_elision = self.lifetime_elision;
502        let mut args = Vec::with_capacity(fn_.params.len());
503        self.with_shifted_in(DebruijnIndex::from_u32(1), |ctx: &mut TyLoweringContext<'_, '_>| {
504            ctx.lifetime_elision =
505                LifetimeElisionKind::AnonymousCreateParameter { report_in_path: false };
506            args.extend(params.iter().map(|&(_, tr)| ctx.lower_ty(tr)));
507            ctx.lifetime_elision = LifetimeElisionKind::for_fn_ret(interner);
508            args.push(ctx.lower_ty(ret_ty));
509        });
510        self.lifetime_elision = old_lifetime_elision;
511        Ty::new_fn_ptr(
512            interner,
513            Binder::dummy(FnSig {
514                abi: fn_.abi.as_ref().map_or(FnAbi::Rust, FnAbi::from_symbol),
515                safety: if fn_.is_unsafe { Safety::Unsafe } else { Safety::Safe },
516                c_variadic: fn_.is_varargs,
517                inputs_and_output: Tys::new_from_slice(&args),
518            }),
519        )
520    }

522    /// This is only for `generic_predicates_for_param`, where we can't just
523    /// lower the self types of the predicates since that could lead to cycles.
524    /// So we just check here if the `type_ref` resolves to a generic param, and which.
525    fn lower_ty_only_param(&self, type_ref: TypeRefId) -> Option<TypeOrConstParamId> {
526        let type_ref = &self.store[type_ref];
527        let path = match type_ref {
528            TypeRef::Path(path) => path,
529            &TypeRef::TypeParam(idx) => return Some(idx.into()),
530            _ => return None,
531        };
532        if path.type_anchor().is_some() {
533            return None;
534        }
535        if path.segments().len() > 1 {
536            return None;
537        }
538        let resolution = match self.resolver.resolve_path_in_type_ns(self.db, path) {
539            Some((it, None, _)) => it,
540            _ => return None,
541        };
542        match resolution {
543            TypeNs::GenericParam(param_id) => Some(param_id.into()),
544            _ => None,
545        }
546    }

568    pub(crate) fn lower_path(&mut self, path: &Path, path_id: PathId) -> (Ty<'db>, Option<TypeNs>) {
569        // Resolve the path (in type namespace)
570        if let Some(type_ref) = path.type_anchor() {
571            let (ty, res) = self.lower_ty_ext(type_ref);
572            let mut ctx = self.at_path(path_id);
573            return ctx.lower_ty_relative_path(ty, res, false);
574        }
575
576        let mut ctx = self.at_path(path_id);
577        let (resolution, remaining_index) = match ctx.resolve_path_in_type_ns() {
578            Some(it) => it,
579            None => return (Ty::new_error(self.interner, ErrorGuaranteed), None),
580        };
581
582        if matches!(resolution, TypeNs::TraitId(_)) && remaining_index.is_none() {
583            // trait object type without dyn
584            let bound = TypeBound::Path(path_id, TraitBoundModifier::None);
585            let ty = self.lower_dyn_trait(&[bound]);
586            return (ty, None);
587        }
588
589        ctx.lower_partly_resolved_path(resolution, false)
590    }

614    pub(crate) fn lower_where_predicate<'b>(
615        &'b mut self,
616        where_predicate: &'b WherePredicate,
617        ignore_bindings: bool,
618    ) -> impl Iterator<Item = (Clause<'db>, GenericPredicateSource)> + use<'a, 'b, 'db> {
619        match where_predicate {
620            WherePredicate::ForLifetime { target, bound, .. }
621            | WherePredicate::TypeBound { target, bound } => {

631                            self.lower_lifetime(target),

633                    )),
634                )),
635                GenericPredicateSource::SelfOnly,
636            ))),
637        }
638        .into_iter()
639    }

641    pub(crate) fn lower_type_bound<'b>(
642        &'b mut self,
643        bound: &'b TypeBound,
644        self_ty: Ty<'db>,
645        ignore_bindings: bool,
646    ) -> impl Iterator<Item = (Clause<'db>, GenericPredicateSource)> + use<'b, 'a, 'db> {
647        let interner = self.interner;
648        let meta_sized = self.lang_items.MetaSized;
649        let pointee_sized = self.lang_items.PointeeSized;
650        let mut assoc_bounds = None;
651        let mut clause = None;
652        match bound {
653            &TypeBound::Path(path, TraitBoundModifier::None) | &TypeBound::ForLifetime(_, path) => {
654                // FIXME Don't silently drop the hrtb lifetimes here
655                if let Some((trait_ref, mut ctx)) = self.lower_trait_ref_from_path(path, self_ty) {
656                    // FIXME(sized-hierarchy): Remove this bound modifications once we have implemented
657                    // sized-hierarchy correctly.
658                    if meta_sized.is_some_and(|it| it == trait_ref.def_id.0) {
659                        // Ignore this bound
660                    } else if pointee_sized.is_some_and(|it| it == trait_ref.def_id.0) {
661                        // Regard this as `?Sized` bound
662                        ctx.ty_ctx().unsized_types.insert(self_ty);
663                    } else {
664                        if !ignore_bindings {
665                            assoc_bounds = ctx.assoc_type_bindings_from_type_bound(trait_ref);
666                        }
667                        clause = Some(Clause(Predicate::new(
668                            interner,
669                            Binder::dummy(rustc_type_ir::PredicateKind::Clause(
670                                rustc_type_ir::ClauseKind::Trait(TraitPredicate {
671                                    trait_ref,
672                                    polarity: rustc_type_ir::PredicatePolarity::Positive,
673                                }),
674                            )),
675                        )));
676                    }
677                }
678            }
679            &TypeBound::Path(path, TraitBoundModifier::Maybe) => {
680                let sized_trait = self.lang_items.Sized;
681                // Don't lower associated type bindings as the only possible relaxed trait bound
682                // `?Sized` has no of them.
683                // If we got another trait here ignore the bound completely.
684                let trait_id = self
685                    .lower_trait_ref_from_path(path, self_ty)
686                    .map(|(trait_ref, _)| trait_ref.def_id.0);
687                if trait_id == sized_trait {
688                    self.unsized_types.insert(self_ty);
689                }
690            }
691            &TypeBound::Lifetime(l) => {
692                let lifetime = self.lower_lifetime(l);
693                clause = Some(Clause(Predicate::new(
694                    self.interner,
695                    Binder::dummy(rustc_type_ir::PredicateKind::Clause(
696                        rustc_type_ir::ClauseKind::TypeOutlives(OutlivesPredicate(
697                            self_ty, lifetime,
698                        )),
699                    )),
700                )));
701            }
702            TypeBound::Use(_) | TypeBound::Error => {}
703        }
704        clause
705            .into_iter()
706            .map(|pred| (pred, GenericPredicateSource::SelfOnly))
707            .chain(assoc_bounds.into_iter().flatten())
708    }

710    fn lower_dyn_trait(&mut self, bounds: &[TypeBound]) -> Ty<'db> {
711        let interner = self.interner;
712        let dummy_self_ty = dyn_trait_dummy_self(interner);
713        let mut region = None;
714        // INVARIANT: The principal trait bound, if present, must come first. Others may be in any
715        // order but should be in the same order for the same set but possibly different order of
716        // bounds in the input.
717        // INVARIANT: If this function returns `DynTy`, there should be at least one trait bound.
718        // These invariants are utilized by `TyExt::dyn_trait()` and chalk.
719        let bounds = self.with_shifted_in(DebruijnIndex::from_u32(1), |ctx| {
720            let mut principal = None;
721            let mut auto_traits = SmallVec::<[_; 3]>::new();
722            let mut projections = Vec::new();
723            let mut had_error = false;
724
725            for b in bounds {
726                let db = ctx.db;
727                ctx.lower_type_bound(b, dummy_self_ty, false).for_each(|(b, _)| {
728                    match b.kind().skip_binder() {
729                        rustc_type_ir::ClauseKind::Trait(t) => {
730                            let id = t.def_id();
731                            let is_auto = db.trait_signature(id.0).flags.contains(TraitFlags::AUTO);
732                            if is_auto {
733                                auto_traits.push(t.def_id().0);
734                            } else {
735                                if principal.is_some() {
736                                    // FIXME: Report an error.
737                                    had_error = true;
738                                }
739                                principal = Some(b.kind().rebind(t.trait_ref));
740                            }
741                        }
742                        rustc_type_ir::ClauseKind::Projection(p) => {
743                            projections.push(b.kind().rebind(p));
744                        }
745                        rustc_type_ir::ClauseKind::TypeOutlives(outlives_predicate) => {
746                            if region.is_some() {
747                                // FIXME: Report an error.
748                                had_error = true;
749                            }
750                            region = Some(outlives_predicate.1);
751                        }
752                        rustc_type_ir::ClauseKind::RegionOutlives(_)
753                        | rustc_type_ir::ClauseKind::ConstArgHasType(_, _)
754                        | rustc_type_ir::ClauseKind::WellFormed(_)
755                        | rustc_type_ir::ClauseKind::ConstEvaluatable(_)
756                        | rustc_type_ir::ClauseKind::HostEffect(_)
757                        | rustc_type_ir::ClauseKind::UnstableFeature(_) => unreachable!(),
758                    }
759                })
760            }
761
762            if had_error {
763                return None;
764            }
765
766            if principal.is_none() && auto_traits.is_empty() {
767                // No traits is not allowed.
768                return None;
769            }
770
771            // `Send + Sync` is the same as `Sync + Send`.
772            auto_traits.sort_unstable();
773            // Duplicate auto traits are permitted.
774            auto_traits.dedup();
775
776            // Map the projection bounds onto a key that makes it easy to remove redundant
777            // bounds that are constrained by supertraits of the principal def id.
778            //
779            // Also make sure we detect conflicting bounds from expanding a trait alias and
780            // also specifying it manually, like:
781            // ```
782            // type Alias = Trait<Assoc = i32>;
783            // let _: &dyn Alias<Assoc = u32> = /* ... */;
784            // ```
785            let mut projection_bounds = FxIndexMap::default();
786            for proj in projections {
787                let key = (
788                    proj.skip_binder().def_id().expect_type_alias(),
789                    interner.anonymize_bound_vars(
790                        proj.map_bound(|proj| proj.projection_term.trait_ref(interner)),
791                    ),
792                );
793                if let Some(old_proj) = projection_bounds.insert(key, proj)
794                    && interner.anonymize_bound_vars(proj)
795                        != interner.anonymize_bound_vars(old_proj)
796                {
797                    // FIXME: Report "conflicting associated type" error.
798                }
799            }
800
801            // A stable ordering of associated types from the principal trait and all its
802            // supertraits. We use this to ensure that different substitutions of a trait
803            // don't result in `dyn Trait` types with different projections lists, which
804            // can be unsound: <https://github.com/rust-lang/rust/pull/136458>.
805            // We achieve a stable ordering by walking over the unsubstituted principal
806            // trait ref.
807            let mut ordered_associated_types = vec![];
808
809            if let Some(principal_trait) = principal {
810                // Generally we should not elaborate in lowering as this can lead to cycles, but
811                // here rustc cycles as well.
812                for clause in elaborate::elaborate(
813                    interner,
814                    [Clause::upcast_from(
815                        TraitRef::identity(interner, principal_trait.def_id()),
816                        interner,
817                    )],
818                )
819                .filter_only_self()
820                {
821                    let clause = clause.instantiate_supertrait(interner, principal_trait);
822                    debug!("observing object predicate `{clause:?}`");
823
824                    let bound_predicate = clause.kind();
825                    match bound_predicate.skip_binder() {
826                        ClauseKind::Trait(pred) => {
827                            // FIXME(negative_bounds): Handle this correctly...
828                            let trait_ref = interner
829                                .anonymize_bound_vars(bound_predicate.rebind(pred.trait_ref));
830                            ordered_associated_types.extend(
831                                pred.trait_ref
832                                    .def_id
833                                    .0
834                                    .trait_items(self.db)
835                                    .associated_types()
836                                    .map(|item| (item, trait_ref)),
837                            );
838                        }
839                        ClauseKind::Projection(pred) => {
840                            let pred = bound_predicate.rebind(pred);
841                            // A `Self` within the original bound will be instantiated with a
842                            // `trait_object_dummy_self`, so check for that.
843                            let references_self = match pred.skip_binder().term.kind() {
844                                TermKind::Ty(ty) => {
845                                    ty.walk().any(|arg| arg == dummy_self_ty.into())
846                                }
847                                // FIXME(mgca): We should walk the const instead of not doing anything
848                                TermKind::Const(_) => false,
849                            };
850
851                            // If the projection output contains `Self`, force the user to
852                            // elaborate it explicitly to avoid a lot of complexity.
853                            //
854                            // The "classically useful" case is the following:
855                            // ```
856                            //     trait MyTrait: FnMut() -> <Self as MyTrait>::MyOutput {
857                            //         type MyOutput;
858                            //     }
859                            // ```
860                            //
861                            // Here, the user could theoretically write `dyn MyTrait<MyOutput = X>`,
862                            // but actually supporting that would "expand" to an infinitely-long type
863                            // `fix $   dyn MyTrait<MyOutput = X, Output = < as MyTrait>::MyOutput`.
864                            //
865                            // Instead, we force the user to write
866                            // `dyn MyTrait<MyOutput = X, Output = X>`, which is uglier but works. See
867                            // the discussion in #56288 for alternatives.
868                            if !references_self {
869                                let key = (
870                                    pred.skip_binder().projection_term.def_id.expect_type_alias(),
871                                    interner.anonymize_bound_vars(pred.map_bound(|proj| {
872                                        proj.projection_term.trait_ref(interner)
873                                    })),
874                                );
875                                if !projection_bounds.contains_key(&key) {
876                                    projection_bounds.insert(key, pred);
877                                }
878                            }
879                        }
880                        _ => (),
881                    }
882                }
883            }
884
885            // We compute the list of projection bounds taking the ordered associated types,
886            // and check if there was an entry in the collected `projection_bounds`. Those
887            // are computed by first taking the user-written associated types, then elaborating
888            // the principal trait ref, and only using those if there was no user-written.
889            // See note below about how we handle missing associated types with `Self: Sized`,
890            // which are not required to be provided, but are still used if they are provided.
891            let mut projection_bounds: Vec<_> = ordered_associated_types
892                .into_iter()
893                .filter_map(|key| projection_bounds.get(&key).copied())
894                .collect();
895
896            projection_bounds.sort_unstable_by_key(|proj| proj.skip_binder().def_id());
897
898            let principal = principal.map(|principal| {
899                principal.map_bound(|principal| {
900                    // Verify that `dummy_self` did not leak inside default type parameters.
901                    let args: Vec<_> = principal
902                        .args
903                        .iter()
904                        // Skip `Self`
905                        .skip(1)
906                        .map(|arg| {
907                            if arg.walk().any(|arg| arg == dummy_self_ty.into()) {
908                                // FIXME: Report an error.
909                                Ty::new_error(interner, ErrorGuaranteed).into()
910                            } else {
911                                arg
912                            }
913                        })
914                        .collect();
915
916                    ExistentialPredicate::Trait(ExistentialTraitRef::new(
917                        interner,
918                        principal.def_id,
919                        args,
920                    ))
921                })
922            });
923
924            let projections = projection_bounds.into_iter().map(|proj| {
925                proj.map_bound(|mut proj| {
926                    // Like for trait refs, verify that `dummy_self` did not leak inside default type
927                    // parameters.
928                    let references_self = proj.projection_term.args.iter().skip(1).any(|arg| {
929                        if arg.walk().any(|arg| arg == dummy_self_ty.into()) {
930                            return true;
931                        }
932                        false
933                    });
934                    if references_self {
935                        proj.projection_term =
936                            replace_dummy_self_with_error(interner, proj.projection_term);
937                    }
938
939                    ExistentialPredicate::Projection(ExistentialProjection::erase_self_ty(
940                        interner, proj,
941                    ))
942                })
943            });
944
945            let auto_traits = auto_traits.into_iter().map(|auto_trait| {
946                Binder::dummy(ExistentialPredicate::AutoTrait(auto_trait.into()))
947            });
948
949            // N.b. principal, projections, auto traits
950            Some(BoundExistentialPredicates::new_from_iter(
951                interner,
952                principal.into_iter().chain(projections).chain(auto_traits),
953            ))
954        });
955
956        if let Some(bounds) = bounds {
957            let region = match region {
958                Some(it) => match it.kind() {
959                    rustc_type_ir::RegionKind::ReBound(BoundVarIndexKind::Bound(db), var) => {
960                        Region::new_bound(
961                            self.interner,
962                            db.shifted_out_to_binder(DebruijnIndex::from_u32(2)),
963                            var,
964                        )
965                    }
966                    _ => it,
967                },
968                None => Region::new_static(self.interner),
969            };
970            Ty::new_dynamic(self.interner, bounds, region)
971        } else {
972            // FIXME: report error
973            // (additional non-auto traits, associated type rebound, or no resolved trait)
974            Ty::new_error(self.interner, ErrorGuaranteed)
975        }
976    }

978    fn lower_impl_trait(&mut self, def_id: SolverDefId, bounds: &[TypeBound]) -> ImplTrait {
979        let interner = self.interner;
980        cov_mark::hit!(lower_rpit);
981        let args = GenericArgs::identity_for_item(interner, def_id);
982        let self_ty = Ty::new_alias(
983            self.interner,
984            rustc_type_ir::AliasTyKind::Opaque,
985            AliasTy::new_from_args(interner, def_id, args),
986        );
987        let (predicates, assoc_ty_bounds_start) =
988            self.with_shifted_in(DebruijnIndex::from_u32(1), |ctx| {
989                let mut predicates = Vec::new();
990                let mut assoc_ty_bounds = Vec::new();
991                for b in bounds {
992                    for (pred, source) in ctx.lower_type_bound(b, self_ty, false) {
993                        match source {
994                            GenericPredicateSource::SelfOnly => predicates.push(pred),
995                            GenericPredicateSource::AssocTyBound => assoc_ty_bounds.push(pred),
996                        }
997                    }
998                }
999
1000                if !ctx.unsized_types.contains(&self_ty) {
1001                    let sized_trait = self.lang_items.Sized;
1002                    let sized_clause = sized_trait.map(|trait_id| {
1003                        let trait_ref = TraitRef::new_from_args(
1004                            interner,
1005                            trait_id.into(),
1006                            GenericArgs::new_from_slice(&[self_ty.into()]),
1007                        );
1008                        Clause(Predicate::new(
1009                            interner,
1010                            Binder::dummy(rustc_type_ir::PredicateKind::Clause(
1011                                rustc_type_ir::ClauseKind::Trait(TraitPredicate {
1012                                    trait_ref,
1013                                    polarity: rustc_type_ir::PredicatePolarity::Positive,
1014                                }),
1015                            )),
1016                        ))
1017                    });
1018                    predicates.extend(sized_clause);
1019                }
1020
1021                let assoc_ty_bounds_start = predicates.len() as u32;
1022                predicates.extend(assoc_ty_bounds);
1023                (predicates, assoc_ty_bounds_start)
1024            });
1025
1026        ImplTrait {
1027            predicates: Clauses::new_from_slice(&predicates).store(),
1028            assoc_ty_bounds_start,
1029        }
1030    }

1136    pub fn predicates<'db>(self, db: &'db dyn HirDatabase) -> EarlyBinder<'db, &'db [Clause<'db>]> {

1153        self,
1154        db: &'db dyn HirDatabase,

1157            ImplTraitId::ReturnTypeImplTrait(owner, idx) => {

1160            ImplTraitId::TypeAliasImplTrait(owner, idx) => {
1161                (ImplTraits::type_alias_impl_traits(db, owner), idx)

1163        };

1165            impl_traits.as_deref().expect("owner should have opaque type").get_with(|it| {

1168                    impl_trait.predicates.as_ref().as_slice(),
1169                    impl_trait.assoc_ty_bounds_start as usize,
1170                )
1171            });

1174    }

1606pub(crate) struct SupertraitsInfo {
1607    /// This includes the trait itself.
1608    pub(crate) all_supertraits: Box<[TraitId]>,
1609    pub(crate) direct_supertraits: Box<[TraitId]>,
1610    pub(crate) defined_assoc_types: Box<[(Name, TypeAliasId)]>,

1613impl SupertraitsInfo {
1614    #[inline]
1615    pub(crate) fn query(db: &dyn HirDatabase, trait_: TraitId) -> &Self {
1616        return supertraits_info(db, trait_);
1617
1618        #[salsa::tracked(returns(ref), cycle_result = supertraits_info_cycle)]
1619        fn supertraits_info(db: &dyn HirDatabase, trait_: TraitId) -> SupertraitsInfo {
1620            let mut all_supertraits = FxHashSet::default();
1621            let mut direct_supertraits = FxHashSet::default();
1622            let mut defined_assoc_types = FxHashSet::default();
1623
1624            all_supertraits.insert(trait_);
1625            defined_assoc_types.extend(trait_.trait_items(db).items.iter().filter_map(
1626                |(name, id)| match *id {
1627                    AssocItemId::TypeAliasId(id) => Some((name.clone(), id)),
1628                    _ => None,
1629                },
1630            ));
1631
1632            let resolver = trait_.resolver(db);
1633            let signature = db.trait_signature(trait_);
1634            for pred in signature.generic_params.where_predicates() {
1635                let (WherePredicate::TypeBound { target, bound }
1636                | WherePredicate::ForLifetime { lifetimes: _, target, bound }) = pred
1637                else {
1638                    continue;
1639                };
1640                let (TypeBound::Path(bounded_trait, TraitBoundModifier::None)
1641                | TypeBound::ForLifetime(_, bounded_trait)) = *bound
1642                else {
1643                    continue;
1644                };
1645                let target = &signature.store[*target];
1646                match target {
1647                    TypeRef::TypeParam(param)
1648                        if param.local_id() == GenericParams::SELF_PARAM_ID_IN_SELF => {}
1649                    TypeRef::Path(path) if path.is_self_type() => {}
1650                    _ => continue,
1651                }
1652                let Some(TypeNs::TraitId(bounded_trait)) =
1653                    resolver.resolve_path_in_type_ns_fully(db, &signature.store[bounded_trait])
1654                else {
1655                    continue;
1656                };
1657                let SupertraitsInfo {
1658                    all_supertraits: bounded_trait_all_supertraits,
1659                    direct_supertraits: _,
1660                    defined_assoc_types: bounded_traits_defined_assoc_types,
1661                } = SupertraitsInfo::query(db, bounded_trait);
1662                all_supertraits.extend(bounded_trait_all_supertraits);
1663                direct_supertraits.insert(bounded_trait);
1664                defined_assoc_types.extend(bounded_traits_defined_assoc_types.iter().cloned());
1665            }
1666
1667            SupertraitsInfo {
1668                all_supertraits: Box::from_iter(all_supertraits),
1669                direct_supertraits: Box::from_iter(direct_supertraits),
1670                defined_assoc_types: Box::from_iter(defined_assoc_types),
1671            }
1672        }
1673
1674        fn supertraits_info_cycle(
1675            _db: &dyn HirDatabase,
1676            _: salsa::Id,
1677            _trait_: TraitId,
1678        ) -> SupertraitsInfo {
1679            SupertraitsInfo::default()
1680        }
1681    }

1691/// Predicates for `param_id` of the form `P: SomeTrait`. If
1692/// `assoc_name` is provided, only return predicates referencing traits
1693/// that have an associated type of that name.
1694///
1695/// This query exists only to be used when resolving short-hand associated types
1696/// like `T::Item`.
1697///
1698/// See the analogous query in rustc and its comment:
1699/// <https://github.com/rust-lang/rust/blob/9150f844e2624eb013ec78ca08c1d416e6644026/src/librustc_typeck/astconv.rs#L46>
1700///
1701/// This is a query mostly to handle cycles somewhat gracefully; e.g. the
1702/// following bounds are disallowed: `T: Foo<U::Item>, U: Foo<T::Item>`, but
1703/// these are fine: `T: Foo<U::Item>, U: Foo<()>`.
1704#[tracing::instrument(skip(db), ret)]
1705#[salsa::tracked(returns(ref), cycle_result = resolve_type_param_assoc_type_shorthand_cycle_result)]
1706fn resolve_type_param_assoc_type_shorthand(
1707    db: &dyn HirDatabase,
1708    def: GenericDefId,
1709    param: TypeParamId,
1710    assoc_name: Name,
1711) -> Result<StoredEarlyBinder<(TypeAliasId, StoredGenericArgs)>, TypeParamAssocTypeShorthandError> {
1712    let generics = generics(db, def);
1713    let resolver = def.resolver(db);
1714    let mut ctx = TyLoweringContext::new(
1715        db,
1716        &resolver,
1717        generics.store(),
1718        def,
1719        LifetimeElisionKind::AnonymousReportError,
1720    );
1721    let interner = ctx.interner;
1722    let mut result = None;
1723    let param_ty = Ty::new_param(
1724        interner,
1725        param,
1726        generics.type_or_const_param_idx(param.into()).unwrap() as u32,
1727    );
1728
1729    if let GenericDefId::TraitId(containing_trait) = param.parent()
1730        && param.local_id() == GenericParams::SELF_PARAM_ID_IN_SELF
1731    {
1732        // Add the trait's own associated types.
1733        if let Some(assoc_type) =
1734            containing_trait.trait_items(db).associated_type_by_name(&assoc_name)
1735        {
1736            let args = GenericArgs::identity_for_item(interner, containing_trait.into());
1737            result = Some(StoredEarlyBinder::bind((assoc_type, args.store())));
1738        }
1739    }
1740
1741    for maybe_parent_generics in
1742        std::iter::successors(Some(&generics), |generics| generics.parent_generics())
1743    {
1744        ctx.store = maybe_parent_generics.store();
1745        for pred in maybe_parent_generics.where_predicates() {
1746            let (WherePredicate::TypeBound { target, bound }
1747            | WherePredicate::ForLifetime { lifetimes: _, target, bound }) = pred
1748            else {
1749                continue;
1750            };
1751            let (TypeBound::Path(bounded_trait_path, TraitBoundModifier::None)
1752            | TypeBound::ForLifetime(_, bounded_trait_path)) = *bound
1753            else {
1754                continue;
1755            };
1756            let Some(target) = ctx.lower_ty_only_param(*target) else { continue };
1757            if target != param.into() {
1758                continue;
1759            }
1760            let Some(TypeNs::TraitId(bounded_trait)) =
1761                resolver.resolve_path_in_type_ns_fully(db, &ctx.store[bounded_trait_path])
1762            else {
1763                continue;
1764            };
1765            if !SupertraitsInfo::query(db, bounded_trait)
1766                .defined_assoc_types
1767                .iter()
1768                .any(|(name, _)| *name == assoc_name)
1769            {
1770                continue;
1771            }
1772
1773            let Some((bounded_trait_ref, _)) =
1774                ctx.lower_trait_ref_from_path(bounded_trait_path, param_ty)
1775            else {
1776                continue;
1777            };
1778            // Now, search from the start on the *bounded* trait like if we wrote `Self::Assoc`. Eventually, we'll get
1779            // the correct trait ref (or a cycle).
1780            let lookup_on_bounded_trait = resolve_type_param_assoc_type_shorthand(
1781                db,
1782                bounded_trait.into(),
1783                TypeParamId::trait_self(bounded_trait),
1784                assoc_name.clone(),
1785            );
1786            let lookup_on_bounded_trait = match lookup_on_bounded_trait {
1787                Ok(it) => it,
1788                Err(
1789                    err @ (TypeParamAssocTypeShorthandError::AmbiguousAssocType
1790                    | TypeParamAssocTypeShorthandError::Cycle),
1791                ) => return Err(*err),
1792                Err(TypeParamAssocTypeShorthandError::AssocTypeNotFound) => {
1793                    never!("we checked that the trait defines this assoc type");
1794                    continue;
1795                }
1796            };
1797            let (assoc_type, args) = lookup_on_bounded_trait
1798                .get_with(|(assoc_type, args)| (*assoc_type, args.as_ref()))
1799                .skip_binder();
1800            let args = EarlyBinder::bind(args).instantiate(interner, bounded_trait_ref.args);
1801            let current_result = StoredEarlyBinder::bind((assoc_type, args.store()));
1802            // If we already have a result, this is an ambiguity - unless this is the same result, then we are fine
1803            // (e.g. rustc allows to write the same bound twice without ambiguity).
1804            if let Some(existing_result) = result
1805                && existing_result != current_result
1806            {
1807                return Err(TypeParamAssocTypeShorthandError::AmbiguousAssocType);
1808            }
1809            result = Some(current_result);
1810        }
1811    }
1812
1813    result.ok_or(TypeParamAssocTypeShorthandError::AssocTypeNotFound)
1814}

1850fn type_alias_bounds_with_diagnostics<'db>(
1851    db: &'db dyn HirDatabase,
1852    type_alias: TypeAliasId,
1853) -> (TypeAliasBounds<EarlyBinder<'db, Clauses<'db>>>, Diagnostics) {
1854    let (TypeAliasBounds { predicates, assoc_ty_bounds_start }, diags) =
1855        type_alias_bounds_with_diagnostics_query(db, type_alias);
1856    return (

1864    #[salsa::tracked(returns(ref))]
1865    pub fn type_alias_bounds_with_diagnostics_query<'db>(
1866        db: &'db dyn HirDatabase,
1867        type_alias: TypeAliasId,
1868    ) -> (TypeAliasBounds<StoredEarlyBinder<StoredClauses>>, Diagnostics) {
1869        let type_alias_data = db.type_alias_signature(type_alias);
1870        let resolver = hir_def::resolver::HasResolver::resolver(type_alias, db);
1871        let mut ctx = TyLoweringContext::new(
1872            db,
1873            &resolver,
1874            &type_alias_data.store,
1875            type_alias.into(),
1876            LifetimeElisionKind::AnonymousReportError,
1877        );
1878        let interner = ctx.interner;
1879        let def_id = type_alias.into();
1880
1881        let item_args = GenericArgs::identity_for_item(interner, def_id);
1882        let interner_ty = Ty::new_projection_from_args(interner, def_id, item_args);
1883
1884        let mut bounds = Vec::new();
1885        let mut assoc_ty_bounds = Vec::new();
1886        for bound in &type_alias_data.bounds {
1887            ctx.lower_type_bound(bound, interner_ty, false).for_each(
1888                |(pred, source)| match source {
1889                    GenericPredicateSource::SelfOnly => {
1890                        bounds.push(pred);
1891                    }
1892                    GenericPredicateSource::AssocTyBound => {
1893                        assoc_ty_bounds.push(pred);
1894                    }
1895                },
1896            );
1897        }
1898
1899        if !ctx.unsized_types.contains(&interner_ty) {
1900            let sized_trait = ctx.lang_items.Sized;
1901            if let Some(sized_trait) = sized_trait {
1902                let trait_ref = TraitRef::new_from_args(
1903                    interner,
1904                    sized_trait.into(),
1905                    GenericArgs::new_from_slice(&[interner_ty.into()]),
1906                );
1907                bounds.push(trait_ref.upcast(interner));
1908            };
1909        }
1910
1911        let assoc_ty_bounds_start = bounds.len() as u32;
1912        bounds.extend(assoc_ty_bounds);
1913
1914        (
1915            TypeAliasBounds {
1916                predicates: StoredEarlyBinder::bind(Clauses::new_from_slice(&bounds).store()),
1917                assoc_ty_bounds_start,
1918            },
1919            create_diagnostics(ctx.diagnostics),
1920        )
1921    }
1922}

2087/// Resolve the where clause(s) of an item with generics,
2088/// with a given filter
2089#[tracing::instrument(skip(db), ret)]
2090fn generic_predicates(db: &dyn HirDatabase, def: GenericDefId) -> (GenericPredicates, Diagnostics) {
2091    let generics = generics(db, def);
2092    let resolver = def.resolver(db);
2093    let interner = DbInterner::new_no_crate(db);
2094    let mut ctx = TyLoweringContext::new(
2095        db,
2096        &resolver,
2097        generics.store(),
2098        def,
2099        LifetimeElisionKind::AnonymousReportError,
2100    );
2101    let sized_trait = ctx.lang_items.Sized;
2102
2103    // We need to lower parents and self separately - see the comment below lowering of implicit
2104    // `Sized` predicates for why.
2105    let mut own_predicates = Vec::new();
2106    let mut parent_predicates = Vec::new();
2107    let mut own_assoc_ty_bounds = Vec::new();
2108    let mut parent_assoc_ty_bounds = Vec::new();
2109    let all_generics =
2110        std::iter::successors(Some(&generics), |generics| generics.parent_generics())
2111            .collect::<ArrayVec<_, 2>>();
2112    let own_implicit_trait_predicate = implicit_trait_predicate(interner, def);
2113    let parent_implicit_trait_predicate = if all_generics.len() > 1 {
2114        implicit_trait_predicate(interner, all_generics.last().unwrap().def())
2115    } else {
2116        None
2117    };
2118    for &maybe_parent_generics in all_generics.iter().rev() {
2119        // Collect only diagnostics from the child, not including parents.
2120        ctx.diagnostics.clear();
2121
2122        ctx.store = maybe_parent_generics.store();
2123        for pred in maybe_parent_generics.where_predicates() {
2124            tracing::debug!(?pred);
2125            for (pred, source) in ctx.lower_where_predicate(pred, false) {
2126                match source {
2127                    GenericPredicateSource::SelfOnly => {
2128                        if maybe_parent_generics.def() == def {
2129                            own_predicates.push(pred);
2130                        } else {
2131                            parent_predicates.push(pred);
2132                        }
2133                    }
2134                    GenericPredicateSource::AssocTyBound => {
2135                        if maybe_parent_generics.def() == def {
2136                            own_assoc_ty_bounds.push(pred);
2137                        } else {
2138                            parent_assoc_ty_bounds.push(pred);
2139                        }
2140                    }
2141                }
2142            }
2143        }
2144
2145        if maybe_parent_generics.def() == def {
2146            push_const_arg_has_type_predicates(db, &mut own_predicates, maybe_parent_generics);
2147        } else {
2148            push_const_arg_has_type_predicates(db, &mut parent_predicates, maybe_parent_generics);
2149        }
2150
2151        if let Some(sized_trait) = sized_trait {
2152            let mut add_sized_clause = |param_idx, param_id, param_data| {
2153                let (
2154                    GenericParamId::TypeParamId(param_id),
2155                    GenericParamDataRef::TypeParamData(param_data),
2156                ) = (param_id, param_data)
2157                else {
2158                    return;
2159                };
2160
2161                if param_data.provenance == TypeParamProvenance::TraitSelf {
2162                    return;
2163                }
2164
2165                let param_ty = Ty::new_param(interner, param_id, param_idx);
2166                if ctx.unsized_types.contains(&param_ty) {
2167                    return;
2168                }
2169                let trait_ref = TraitRef::new_from_args(
2170                    interner,
2171                    sized_trait.into(),
2172                    GenericArgs::new_from_slice(&[param_ty.into()]),
2173                );
2174                let clause = Clause(Predicate::new(
2175                    interner,
2176                    Binder::dummy(rustc_type_ir::PredicateKind::Clause(
2177                        rustc_type_ir::ClauseKind::Trait(TraitPredicate {
2178                            trait_ref,
2179                            polarity: rustc_type_ir::PredicatePolarity::Positive,
2180                        }),
2181                    )),
2182                ));
2183                if maybe_parent_generics.def() == def {
2184                    own_predicates.push(clause);
2185                } else {
2186                    parent_predicates.push(clause);
2187                }
2188            };
2189            let parent_params_len = maybe_parent_generics.len_parent();
2190            maybe_parent_generics.iter_self().enumerate().for_each(
2191                |(param_idx, (param_id, param_data))| {
2192                    add_sized_clause((param_idx + parent_params_len) as u32, param_id, param_data);
2193                },
2194            );
2195        }
2196
2197        // We do not clear `ctx.unsized_types`, as the `?Sized` clause of a child (e.g. an associated type) can
2198        // be declared on the parent (e.g. the trait). It is nevertheless fine to register the implicit `Sized`
2199        // predicates before lowering the child, as a child cannot define a `?Sized` predicate for its parent.
2200        // But we do have to lower the parent first.
2201    }
2202
2203    let diagnostics = create_diagnostics(ctx.diagnostics);
2204
2205    // The order is:
2206    //
2207    // 1. parent implicit trait pred
2208    // 2. parent assoc bounds
2209    // 3. parent self only preds
2210    // 4. own self only preds
2211    // 5. own assoc ty bounds
2212    // 6. own implicit trait pred
2213    //
2214    // The purpose of this is to index the slice of the followings, without making extra `Vec`s or
2215    // iterators:
2216    // - explicit self only predicates, of own or own + self
2217    // - explicit predicates, of own or own + self
2218    let predicates = parent_implicit_trait_predicate
2219        .iter()
2220        .chain(parent_assoc_ty_bounds.iter())
2221        .chain(parent_predicates.iter())
2222        .chain(own_predicates.iter())
2223        .chain(own_assoc_ty_bounds.iter())
2224        .chain(own_implicit_trait_predicate.iter())
2225        .copied()
2226        .collect::<Vec<_>>();
2227    let parent_is_trait = parent_implicit_trait_predicate.is_some();
2228    let is_trait = own_implicit_trait_predicate.is_some();
2229    let parent_explicit_self_predicates_start =
2230        parent_is_trait as u32 + parent_assoc_ty_bounds.len() as u32;
2231    let own_predicates_start =
2232        parent_explicit_self_predicates_start + parent_predicates.len() as u32;
2233    let own_assoc_ty_bounds_start = own_predicates_start + own_predicates.len() as u32;
2234
2235    let predicates = GenericPredicates {
2236        parent_explicit_self_predicates_start,
2237        own_predicates_start,
2238        own_assoc_ty_bounds_start,
2239        is_trait,
2240        parent_is_trait,
2241        predicates: StoredEarlyBinder::bind(Clauses::new_from_slice(&predicates).store()),
2242    };
2243    return (predicates, diagnostics);
2244
2245    fn implicit_trait_predicate<'db>(
2246        interner: DbInterner<'db>,
2247        def: GenericDefId,
2248    ) -> Option<Clause<'db>> {
2249        // For traits, add `Self: Trait` predicate. This is
2250        // not part of the predicates that a user writes, but it
2251        // is something that one must prove in order to invoke a
2252        // method or project an associated type.
2253        //
2254        // In the chalk setup, this predicate is not part of the
2255        // "predicates" for a trait item. But it is useful in
2256        // rustc because if you directly (e.g.) invoke a trait
2257        // method like `Trait::method(...)`, you must naturally
2258        // prove that the trait applies to the types that were
2259        // used, and adding the predicate into this list ensures
2260        // that this is done.
2261        if let GenericDefId::TraitId(def_id) = def {
2262            Some(TraitRef::identity(interner, def_id.into()).upcast(interner))
2263        } else {
2264            None
2265        }
2266    }
2267}

2269fn push_const_arg_has_type_predicates<'db>(
2270    db: &'db dyn HirDatabase,
2271    predicates: &mut Vec<Clause<'db>>,
2272    generics: &Generics,
2273) {
2274    let interner = DbInterner::new_no_crate(db);
2275    let const_params_offset = generics.len_parent() + generics.len_lifetimes_self();
2276    for (param_index, (param_idx, param_data)) in generics.iter_self_type_or_consts().enumerate() {
2277        if !matches!(param_data, TypeOrConstParamData::ConstParamData(_)) {
2278            continue;
2279        }
2280
2281        let param_id = ConstParamId::from_unchecked(TypeOrConstParamId {
2282            parent: generics.def(),
2283            local_id: param_idx,
2284        });
2285        predicates.push(Clause(
2286            ClauseKind::ConstArgHasType(
2287                Const::new_param(
2288                    interner,
2289                    ParamConst { id: param_id, index: (param_index + const_params_offset) as u32 },
2290                ),
2291                db.const_param_ty_ns(param_id),
2292            )
2293            .upcast(interner),
2294        ));
2295    }

2312/// Resolve the default type params from generics.
2313///
2314/// Diagnostics are only returned for this `GenericDefId` (returned defaults include parents).
2315pub(crate) fn generic_defaults_with_diagnostics_query(
2316    db: &dyn HirDatabase,
2317    def: GenericDefId,
2318) -> (GenericDefaults, Diagnostics) {
2319    let generic_params = generics(db, def);
2320    if generic_params.is_empty() {
2321        return (GenericDefaults(None), None);
2322    }
2323    let resolver = def.resolver(db);
2324
2325    let mut ctx = TyLoweringContext::new(
2326        db,
2327        &resolver,
2328        generic_params.store(),
2329        def,
2330        LifetimeElisionKind::AnonymousReportError,
2331    )
2332    .with_impl_trait_mode(ImplTraitLoweringMode::Disallowed);
2333    let mut idx = 0;
2334    let mut has_any_default = false;
2335    let mut defaults = generic_params
2336        .iter_parents_with_store()
2337        .map(|((_id, p), store)| {
2338            ctx.store = store;
2339            let (result, has_default) = handle_generic_param(&mut ctx, idx, p);
2340            has_any_default |= has_default;
2341            idx += 1;
2342            result
2343        })
2344        .collect::<Vec<_>>();
2345    ctx.diagnostics.clear(); // Don't include diagnostics from the parent.
2346    defaults.extend(generic_params.iter_self().map(|(_id, p)| {
2347        let (result, has_default) = handle_generic_param(&mut ctx, idx, p);
2348        has_any_default |= has_default;
2349        idx += 1;
2350        result
2351    }));
2352    let diagnostics = create_diagnostics(mem::take(&mut ctx.diagnostics));
2353    let defaults = if has_any_default {
2354        GenericDefaults(Some(Arc::from_iter(defaults)))
2355    } else {
2356        GenericDefaults(None)
2357    };
2358    return (defaults, diagnostics);
2359
2360    fn handle_generic_param<'db>(
2361        ctx: &mut TyLoweringContext<'db, '_>,
2362        idx: usize,
2363        p: GenericParamDataRef<'_>,
2364    ) -> (Option<StoredEarlyBinder<StoredGenericArg>>, bool) {
2365        ctx.lowering_param_default(idx as u32);
2366        match p {
2367            GenericParamDataRef::TypeParamData(p) => {
2368                let ty = p.default.map(|ty| ctx.lower_ty(ty));
2369                (
2370                    ty.map(|ty| StoredEarlyBinder::bind(GenericArg::from(ty).store())),
2371                    p.default.is_some(),
2372                )
2373            }
2374            GenericParamDataRef::ConstParamData(p) => {
2375                let val = p.default.map(|c| {
2376                    let param_ty = ctx.lower_ty(p.ty);
2377                    let c = ctx.lower_const(c, param_ty);
2378                    GenericArg::from(c).store()
2379                });
2380                (val.map(StoredEarlyBinder::bind), p.default.is_some())
2381            }
2382            GenericParamDataRef::LifetimeParamData(_) => (None, false),
2383        }
2384    }
2385}

2415fn fn_sig_for_fn(db: &dyn HirDatabase, def: FunctionId) -> StoredEarlyBinder<StoredPolyFnSig> {
2416    let data = db.function_signature(def);
2417    let resolver = def.resolver(db);
2418    let interner = DbInterner::new_no_crate(db);
2419    let mut ctx_params = TyLoweringContext::new(
2420        db,
2421        &resolver,
2422        &data.store,
2423        def.into(),
2424        LifetimeElisionKind::for_fn_params(&data),
2425    );
2426    let params = data.params.iter().map(|&tr| ctx_params.lower_ty(tr));
2427
2428    let ret = match data.ret_type {
2429        Some(ret_type) => {
2430            let mut ctx_ret = TyLoweringContext::new(
2431                db,
2432                &resolver,
2433                &data.store,
2434                def.into(),
2435                LifetimeElisionKind::for_fn_ret(interner),
2436            )
2437            .with_impl_trait_mode(ImplTraitLoweringMode::Opaque);
2438            ctx_ret.lower_ty(ret_type)
2439        }
2440        None => Ty::new_tup(interner, &[]),
2441    };
2442
2443    let inputs_and_output = Tys::new_from_iter(interner, params.chain(Some(ret)));
2444    // If/when we track late bound vars, we need to switch this to not be `dummy`
2445    StoredEarlyBinder::bind(StoredPolyFnSig::new(Binder::dummy(FnSig {
2446        abi: data.abi.as_ref().map_or(FnAbi::Rust, FnAbi::from_symbol),
2447        c_variadic: data.is_varargs(),
2448        safety: if data.is_unsafe() { Safety::Unsafe } else { Safety::Safe },
2449        inputs_and_output,
2450    })))
2451}

2498pub(crate) fn associated_ty_item_bounds<'db>(
2499    db: &'db dyn HirDatabase,
2500    type_alias: TypeAliasId,
2501) -> EarlyBinder<'db, BoundExistentialPredicates<'db>> {
2502    let type_alias_data = db.type_alias_signature(type_alias);
2503    let resolver = hir_def::resolver::HasResolver::resolver(type_alias, db);
2504    let interner = DbInterner::new_no_crate(db);
2505    let mut ctx = TyLoweringContext::new(
2506        db,
2507        &resolver,
2508        &type_alias_data.store,
2509        type_alias.into(),
2510        LifetimeElisionKind::AnonymousReportError,
2511    );
2512    // FIXME: we should never create non-existential predicates in the first place
2513    // For now, use an error type so we don't run into dummy binder issues
2514    let self_ty = Ty::new_error(interner, ErrorGuaranteed);
2515
2516    let mut bounds = Vec::new();
2517    for bound in &type_alias_data.bounds {
2518        ctx.lower_type_bound(bound, self_ty, false).for_each(|(pred, _)| {
2519            if let Some(bound) = pred
2520                .kind()
2521                .map_bound(|c| match c {
2522                    rustc_type_ir::ClauseKind::Trait(t) => {
2523                        let id = t.def_id();
2524                        let is_auto = db.trait_signature(id.0).flags.contains(TraitFlags::AUTO);
2525                        if is_auto {
2526                            Some(ExistentialPredicate::AutoTrait(t.def_id()))
2527                        } else {
2528                            Some(ExistentialPredicate::Trait(ExistentialTraitRef::new_from_args(
2529                                interner,
2530                                t.def_id(),
2531                                GenericArgs::new_from_slice(&t.trait_ref.args[1..]),
2532                            )))
2533                        }
2534                    }
2535                    rustc_type_ir::ClauseKind::Projection(p) => Some(
2536                        ExistentialPredicate::Projection(ExistentialProjection::new_from_args(
2537                            interner,
2538                            p.def_id(),
2539                            GenericArgs::new_from_slice(&p.projection_term.args[1..]),
2540                            p.term,
2541                        )),
2542                    ),
2543                    rustc_type_ir::ClauseKind::TypeOutlives(_) => None,
2544                    rustc_type_ir::ClauseKind::RegionOutlives(_)
2545                    | rustc_type_ir::ClauseKind::ConstArgHasType(_, _)
2546                    | rustc_type_ir::ClauseKind::WellFormed(_)
2547                    | rustc_type_ir::ClauseKind::ConstEvaluatable(_)
2548                    | rustc_type_ir::ClauseKind::HostEffect(_)
2549                    | rustc_type_ir::ClauseKind::UnstableFeature(_) => unreachable!(),
2550                })
2551                .transpose()
2552            {
2553                bounds.push(bound);
2554            }
2555        });
2556    }
2557
2558    if !ctx.unsized_types.contains(&self_ty)
2559        && let Some(sized_trait) = ctx.lang_items.Sized
2560    {
2561        let sized_clause = Binder::dummy(ExistentialPredicate::Trait(ExistentialTraitRef::new(
2562            interner,
2563            sized_trait.into(),
2564            [] as [GenericArg<'_>; 0],
2565        )));
2566        bounds.push(sized_clause);
2567    }
2568
2569    EarlyBinder::bind(BoundExistentialPredicates::new_from_slice(&bounds))
2570}
