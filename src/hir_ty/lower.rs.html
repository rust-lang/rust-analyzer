224#[derive(Copy, Clone, Debug, PartialEq, Eq, Default)]
225pub enum ImplTraitLoweringMode {
226    /// `impl Trait` gets lowered into an opaque type that doesn't unify with
227    /// anything except itself. This is used in places where values flow 'out',
228    /// i.e. for arguments of the function we're currently checking, and return
229    /// types of functions we're calling.
230    Opaque,
231    /// `impl Trait` gets lowered into a type variable. Used for argument
232    /// position impl Trait when inside the respective function, since it allows
233    /// us to support that without Chalk.
234    Param,
235    /// `impl Trait` gets lowered into a variable that can unify with some
236    /// type. This is used in places where values flow 'in', i.e. for arguments
237    /// of functions we're calling, and the return type of the function we're
238    /// currently checking.
239    Variable,
240    /// `impl Trait` is disallowed and will be an error.
241    #[default]
242    Disallowed,
243}
244
245#[derive(Copy, Clone, Debug, PartialEq, Eq)]
246pub enum ParamLoweringMode {
247    Placeholder,
248    Variable,
249}

277    pub fn lower_ty_ext(&mut self, type_ref_id: TypeRefId) -> (Ty, Option<TypeNs>) {
278        let mut res = None;
279        let type_ref = &self.types_map[type_ref_id];
280        let ty = match type_ref {
281            TypeRef::Never => TyKind::Never.intern(Interner),
282            TypeRef::Tuple(inner) => {
283                let inner_tys = inner.iter().map(|&tr| self.lower_ty(tr));
284                TyKind::Tuple(inner_tys.len(), Substitution::from_iter(Interner, inner_tys))
285                    .intern(Interner)
286            }
287            TypeRef::Path(path) => {
288                let (ty, res_) =
289                    self.lower_path(path, PathId::from_type_ref_unchecked(type_ref_id));
290                res = res_;
291                ty
292            }
293            &TypeRef::RawPtr(inner, mutability) => {
294                let inner_ty = self.lower_ty(inner);
295                TyKind::Raw(lower_to_chalk_mutability(mutability), inner_ty).intern(Interner)
296            }
297            TypeRef::Array(array) => {
298                let inner_ty = self.lower_ty(array.ty);
299                let const_len = self.lower_const(&array.len, TyBuilder::usize());
300                TyKind::Array(inner_ty, const_len).intern(Interner)
301            }
302            &TypeRef::Slice(inner) => {
303                let inner_ty = self.lower_ty(inner);
304                TyKind::Slice(inner_ty).intern(Interner)
305            }
306            TypeRef::Reference(ref_) => {
307                let inner_ty = self.lower_ty(ref_.ty);
308                // FIXME: It should infer the eldided lifetimes instead of stubbing with static
309                let lifetime = ref_
310                    .lifetime
311                    .as_ref()
312                    .map_or_else(error_lifetime, |lr| self.lower_lifetime(lr));
313                TyKind::Ref(lower_to_chalk_mutability(ref_.mutability), lifetime, inner_ty)
314                    .intern(Interner)
315            }
316            TypeRef::Placeholder => TyKind::Error.intern(Interner),
317            TypeRef::Fn(fn_) => {
318                let substs = self.with_shifted_in(DebruijnIndex::ONE, |ctx| {
319                    Substitution::from_iter(
320                        Interner,
321                        fn_.params.iter().map(|&(_, tr)| ctx.lower_ty(tr)),
322                    )
323                });
324                TyKind::Function(FnPointer {
325                    num_binders: 0, // FIXME lower `for<'a> fn()` correctly
326                    sig: FnSig {
327                        abi: fn_.abi.as_ref().map_or(FnAbi::Rust, FnAbi::from_symbol),
328                        safety: if fn_.is_unsafe { Safety::Unsafe } else { Safety::Safe },
329                        variadic: fn_.is_varargs,
330                    },
331                    substitution: FnSubst(substs),
332                })
333                .intern(Interner)
334            }
335            TypeRef::DynTrait(bounds) => self.lower_dyn_trait(bounds),
336            TypeRef::ImplTrait(bounds) => {
337                match self.impl_trait_mode.mode {
338                    ImplTraitLoweringMode::Opaque => {
339                        let origin = match self.resolver.generic_def() {
340                            Some(GenericDefId::FunctionId(it)) => Either::Left(it),
341                            Some(GenericDefId::TypeAliasId(it)) => Either::Right(it),
342                            _ => panic!(
343                                "opaque impl trait lowering must be in function or type alias"
344                            ),
345                        };
346
347                        // this dance is to make sure the data is in the right
348                        // place even if we encounter more opaque types while
349                        // lowering the bounds
350                        let idx = self.impl_trait_mode.opaque_type_data.alloc(ImplTrait {
351                            bounds: crate::make_single_type_binders(Vec::default()),
352                        });
353                        // We don't want to lower the bounds inside the binders
354                        // we're currently in, because they don't end up inside
355                        // those binders. E.g. when we have `impl Trait<impl
356                        // OtherTrait<T>>`, the `impl OtherTrait<T>` can't refer
357                        // to the self parameter from `impl Trait`, and the
358                        // bounds aren't actually stored nested within each
359                        // other, but separately. So if the `T` refers to a type
360                        // parameter of the outer function, it's just one binder
361                        // away instead of two.
362                        let actual_opaque_type_data = self
363                            .with_debruijn(DebruijnIndex::INNERMOST, |ctx| {
364                                ctx.lower_impl_trait(bounds, self.resolver.krate())
365                            });
366                        self.impl_trait_mode.opaque_type_data[idx] = actual_opaque_type_data;
367
368                        let impl_trait_id = origin.either(
369                            |f| ImplTraitId::ReturnTypeImplTrait(f, idx),
370                            |a| ImplTraitId::TypeAliasImplTrait(a, idx),
371                        );
372                        let opaque_ty_id = self.db.intern_impl_trait_id(impl_trait_id).into();
373                        let generics =
374                            generics(self.db.upcast(), origin.either(|f| f.into(), |a| a.into()));
375                        let parameters = generics.bound_vars_subst(self.db, self.in_binders);
376                        TyKind::OpaqueType(opaque_ty_id, parameters).intern(Interner)
377                    }
378                    ImplTraitLoweringMode::Param => {
379                        let idx = self.impl_trait_mode.param_and_variable_counter;
380                        // Count the number of `impl Trait` things that appear within our bounds.
381                        // Since those have been emitted as implicit type args already.
382                        self.impl_trait_mode.param_and_variable_counter =
383                            idx + self.count_impl_traits(type_ref_id) as u16;
384                        let db = self.db;
385                        let kind = self
386                            .generics()
387                            .expect("param impl trait lowering must be in a generic def")
388                            .iter()
389                            .filter_map(|(id, data)| match (id, data) {
390                                (
391                                    GenericParamId::TypeParamId(id),
392                                    GenericParamDataRef::TypeParamData(data),
393                                ) if data.provenance == TypeParamProvenance::ArgumentImplTrait => {
394                                    Some(id)
395                                }
396                                _ => None,
397                            })
398                            .nth(idx as usize)
399                            .map_or(TyKind::Error, |id| {
400                                TyKind::Placeholder(to_placeholder_idx(db, id.into()))
401                            });
402                        kind.intern(Interner)
403                    }
404                    ImplTraitLoweringMode::Variable => {
405                        let idx = self.impl_trait_mode.param_and_variable_counter;
406                        // Count the number of `impl Trait` things that appear within our bounds.
407                        // Since t hose have been emitted as implicit type args already.
408                        self.impl_trait_mode.param_and_variable_counter =
409                            idx + self.count_impl_traits(type_ref_id) as u16;
410                        let debruijn = self.in_binders;
411                        let kind = self
412                            .generics()
413                            .expect("variable impl trait lowering must be in a generic def")
414                            .iter()
415                            .enumerate()
416                            .filter_map(|(i, (id, data))| match (id, data) {
417                                (
418                                    GenericParamId::TypeParamId(_),
419                                    GenericParamDataRef::TypeParamData(data),
420                                ) if data.provenance == TypeParamProvenance::ArgumentImplTrait => {
421                                    Some(i)
422                                }
423                                _ => None,
424                            })
425                            .nth(idx as usize)
426                            .map_or(TyKind::Error, |id| {
427                                TyKind::BoundVar(BoundVar { debruijn, index: id })
428                            });
429                        kind.intern(Interner)
430                    }
431                    ImplTraitLoweringMode::Disallowed => {
432                        // FIXME: report error
433                        TyKind::Error.intern(Interner)
434                    }
435                }
436            }
437            TypeRef::Macro(macro_call) => {
438                let (expander, recursion_start) = {
439                    match &mut self.expander {
440                        // There already is an expander here, this means we are already recursing
441                        Some(expander) => (expander, false),
442                        // No expander was created yet, so we are at the start of the expansion recursion
443                        // and therefore have to create an expander.
444                        None => {
445                            let expander = self.expander.insert(Expander::new(
446                                self.db.upcast(),
447                                macro_call.file_id,
448                                self.resolver.module(),
449                            ));
450                            (expander, true)
451                        }
452                    }
453                };
454                let ty = {
455                    let macro_call = macro_call.to_node(self.db.upcast());
456                    let resolver = |path: &_| {
457                        self.resolver
458                            .resolve_path_as_macro(self.db.upcast(), path, Some(MacroSubNs::Bang))
459                            .map(|(it, _)| it)
460                    };
461                    match expander.enter_expand::<ast::Type>(self.db.upcast(), macro_call, resolver)
462                    {
463                        Ok(ExpandResult { value: Some((mark, expanded)), .. }) => {
464                            let (mut types_map, mut types_source_map) =
465                                (TypesMap::default(), TypesSourceMap::default());
466
467                            let mut ctx = expander.ctx(
468                                self.db.upcast(),
469                                &mut types_map,
470                                &mut types_source_map,
471                            );
472                            // FIXME: Report syntax errors in expansion here
473                            let type_ref = TypeRef::from_ast(&mut ctx, expanded.tree());
474
475                            // Can't mutate `self`, must create a new instance, because of the lifetimes.
476                            let mut inner_ctx = TyLoweringContext {
477                                db: self.db,
478                                resolver: self.resolver,
479                                generics: self.generics.clone(),
480                                types_map: &types_map,
481                                types_source_map: Some(&types_source_map),
482                                in_binders: self.in_binders,
483                                owner: self.owner,
484                                type_param_mode: self.type_param_mode,
485                                impl_trait_mode: mem::take(&mut self.impl_trait_mode),
486                                expander: self.expander.take(),
487                                unsized_types: mem::take(&mut self.unsized_types),
488                                diagnostics: mem::take(&mut self.diagnostics),
489                            };
490
491                            let ty = inner_ctx.lower_ty(type_ref);
492
493                            self.impl_trait_mode = inner_ctx.impl_trait_mode;
494                            self.expander = inner_ctx.expander;
495                            self.unsized_types = inner_ctx.unsized_types;
496                            self.diagnostics = inner_ctx.diagnostics;
497
498                            self.expander.as_mut().unwrap().exit(mark);
499                            Some(ty)
500                        }
501                        _ => None,
502                    }
503                };
504
505                // drop the expander, resetting it to pre-recursion state
506                if recursion_start {
507                    self.expander = None;
508                }
509                ty.unwrap_or_else(|| TyKind::Error.intern(Interner))
510            }
511            TypeRef::Error => TyKind::Error.intern(Interner),
512        };
513        (ty, res)
514    }

516    /// This is only for `generic_predicates_for_param`, where we can't just
517    /// lower the self types of the predicates since that could lead to cycles.
518    /// So we just check here if the `type_ref` resolves to a generic param, and which.
519    fn lower_ty_only_param(&mut self, type_ref_id: TypeRefId) -> Option<TypeOrConstParamId> {
520        let type_ref = &self.types_map[type_ref_id];
521        let path = match type_ref {
522            TypeRef::Path(path) => path,
523            _ => return None,
524        };
525        if path.type_anchor().is_some() {
526            return None;
527        }
528        if path.segments().len() > 1 {
529            return None;
530        }
531        let mut ctx = self.at_path(PathId::from_type_ref_unchecked(type_ref_id));
532        let resolution = match ctx.resolve_path_in_type_ns() {
533            Some((it, None)) => it,
534            _ => return None,
535        };
536        match resolution {
537            TypeNs::GenericParam(param_id) => Some(param_id.into()),
538            _ => None,
539        }
540    }

562    pub(crate) fn lower_path(&mut self, path: &Path, path_id: PathId) -> (Ty, Option<TypeNs>) {
563        // Resolve the path (in type namespace)
564        if let Some(type_ref) = path.type_anchor() {
565            let (ty, res) = self.lower_ty_ext(type_ref);
566            let mut ctx = self.at_path(path_id);
567            return ctx.lower_ty_relative_path(ty, res);
568        }
569
570        let mut ctx = self.at_path(path_id);
571        let (resolution, remaining_index) = match ctx.resolve_path_in_type_ns() {
572            Some(it) => it,
573            None => return (TyKind::Error.intern(Interner), None),
574        };
575
576        if matches!(resolution, TypeNs::TraitId(_)) && remaining_index.is_none() {
577            // trait object type without dyn
578            let bound = TypeBound::Path(path_id, TraitBoundModifier::None);
579            let ty = self.lower_dyn_trait(&[bound]);
580            return (ty, None);
581        }
582
583        ctx.lower_partly_resolved_path(resolution, false)
584    }

589        explicit_self_ty: Ty,
590    ) -> Option<(TraitRef, PathLoweringContext<'_, 'a>)> {
591        let mut ctx = self.at_path(path_id);
592        let resolved = match ctx.resolve_path_in_type_ns_fully()? {
593            // FIXME(trait_alias): We need to handle trait alias here.
594            TypeNs::TraitId(tr) => tr,
595            _ => return None,
596        };
597        Some((ctx.lower_trait_ref_from_resolved_path(resolved, explicit_self_ty), ctx))
598    }

608    pub(crate) fn lower_where_predicate<'b>(
609        &'b mut self,
610        where_predicate: &'b WherePredicate,
611        &def: &GenericDefId,
612        ignore_bindings: bool,
613    ) -> impl Iterator<Item = QuantifiedWhereClause> + use<'a, 'b> {
614        match where_predicate {
615            WherePredicate::ForLifetime { target, bound, .. }
616            | WherePredicate::TypeBound { target, bound } => {
617                let self_ty = match target {
618                    WherePredicateTypeTarget::TypeRef(type_ref) => self.lower_ty(*type_ref),
619                    &WherePredicateTypeTarget::TypeOrConstParam(local_id) => {
620                        let param_id = hir_def::TypeOrConstParamId { parent: def, local_id };
621                        match self.type_param_mode {
622                            ParamLoweringMode::Placeholder => {
623                                TyKind::Placeholder(to_placeholder_idx(self.db, param_id))
624                            }
625                            ParamLoweringMode::Variable => {
626                                let idx = generics(self.db.upcast(), def)
627                                    .type_or_const_param_idx(param_id)
628                                    .expect("matching generics");
629                                TyKind::BoundVar(BoundVar::new(DebruijnIndex::INNERMOST, idx))
630                            }
631                        }
632                        .intern(Interner)
633                    }
634                };
635                Either::Left(self.lower_type_bound(bound, self_ty, ignore_bindings))
636            }
637            WherePredicate::Lifetime { bound, target } => Either::Right(iter::once(
638                crate::wrap_empty_binders(WhereClause::LifetimeOutlives(LifetimeOutlives {
639                    a: self.lower_lifetime(bound),
640                    b: self.lower_lifetime(target),
641                })),
642            )),
643        }
644        .into_iter()
645    }

647    pub(crate) fn lower_type_bound<'b>(
648        &'b mut self,
649        bound: &'b TypeBound,
650        self_ty: Ty,
651        ignore_bindings: bool,
652    ) -> impl Iterator<Item = QuantifiedWhereClause> + use<'b, 'a> {
653        let mut assoc_bounds = None;
654        let mut clause = None;
655        match bound {
656            &TypeBound::Path(path, TraitBoundModifier::None) | &TypeBound::ForLifetime(_, path) => {
657                // FIXME Don't silently drop the hrtb lifetimes here
658                if let Some((trait_ref, ctx)) = self.lower_trait_ref_from_path(path, self_ty) {
659                    if !ignore_bindings {
660                        assoc_bounds =
661                            ctx.assoc_type_bindings_from_type_bound(bound, trait_ref.clone());
662                    }
663                    clause = Some(crate::wrap_empty_binders(WhereClause::Implemented(trait_ref)));
664                }
665            }
666            &TypeBound::Path(path, TraitBoundModifier::Maybe) => {
667                let sized_trait = self
668                    .db
669                    .lang_item(self.resolver.krate(), LangItem::Sized)
670                    .and_then(|lang_item| lang_item.as_trait());
671                // Don't lower associated type bindings as the only possible relaxed trait bound
672                // `?Sized` has no of them.
673                // If we got another trait here ignore the bound completely.
674                let trait_id = self
675                    .lower_trait_ref_from_path(path, self_ty.clone())
676                    .map(|(trait_ref, _)| trait_ref.hir_trait_id());
677                if trait_id == sized_trait {
678                    self.unsized_types.insert(self_ty);
679                }
680            }
681            TypeBound::Lifetime(l) => {
682                let lifetime = self.lower_lifetime(l);
683                clause = Some(crate::wrap_empty_binders(WhereClause::TypeOutlives(TypeOutlives {
684                    ty: self_ty,
685                    lifetime,
686                })));
687            }
688            TypeBound::Use(_) | TypeBound::Error => {}
689        }
690        clause.into_iter().chain(assoc_bounds.into_iter().flatten())
691    }

693    fn lower_dyn_trait(&mut self, bounds: &[TypeBound]) -> Ty {
694        let self_ty = TyKind::BoundVar(BoundVar::new(DebruijnIndex::INNERMOST, 0)).intern(Interner);
695        // INVARIANT: The principal trait bound, if present, must come first. Others may be in any
696        // order but should be in the same order for the same set but possibly different order of
697        // bounds in the input.
698        // INVARIANT: If this function returns `DynTy`, there should be at least one trait bound.
699        // These invariants are utilized by `TyExt::dyn_trait()` and chalk.
700        let mut lifetime = None;
701        let bounds = self.with_shifted_in(DebruijnIndex::ONE, |ctx| {
702            let mut lowered_bounds = Vec::new();
703            for b in bounds {
704                ctx.lower_type_bound(b, self_ty.clone(), false).for_each(|b| {
705                    let filter = match b.skip_binders() {
706                        WhereClause::Implemented(_) | WhereClause::AliasEq(_) => true,
707                        WhereClause::LifetimeOutlives(_) => false,
708                        WhereClause::TypeOutlives(t) => {
709                            lifetime = Some(t.lifetime.clone());
710                            false
711                        }
712                    };
713                    if filter {
714                        lowered_bounds.push(b);
715                    }
716                });
717            }
718
719            let mut multiple_regular_traits = false;
720            let mut multiple_same_projection = false;
721            lowered_bounds.sort_unstable_by(|lhs, rhs| {
722                use std::cmp::Ordering;
723                match (lhs.skip_binders(), rhs.skip_binders()) {
724                    (WhereClause::Implemented(lhs), WhereClause::Implemented(rhs)) => {
725                        let lhs_id = lhs.trait_id;
726                        let lhs_is_auto = ctx
727                            .db
728                            .trait_data(from_chalk_trait_id(lhs_id))
729                            .flags
730                            .contains(TraitFlags::IS_AUTO);
731                        let rhs_id = rhs.trait_id;
732                        let rhs_is_auto = ctx
733                            .db
734                            .trait_data(from_chalk_trait_id(rhs_id))
735                            .flags
736                            .contains(TraitFlags::IS_AUTO);
737
738                        if !lhs_is_auto && !rhs_is_auto {
739                            multiple_regular_traits = true;
740                        }
741                        // Note that the ordering here is important; this ensures the invariant
742                        // mentioned above.
743                        (lhs_is_auto, lhs_id).cmp(&(rhs_is_auto, rhs_id))
744                    }
745                    (WhereClause::Implemented(_), _) => Ordering::Less,
746                    (_, WhereClause::Implemented(_)) => Ordering::Greater,
747                    (WhereClause::AliasEq(lhs), WhereClause::AliasEq(rhs)) => {
748                        match (&lhs.alias, &rhs.alias) {
749                            (AliasTy::Projection(lhs_proj), AliasTy::Projection(rhs_proj)) => {
750                                // We only compare the `associated_ty_id`s. We shouldn't have
751                                // multiple bounds for an associated type in the correct Rust code,
752                                // and if we do, we error out.
753                                if lhs_proj.associated_ty_id == rhs_proj.associated_ty_id {
754                                    multiple_same_projection = true;
755                                }
756                                lhs_proj.associated_ty_id.cmp(&rhs_proj.associated_ty_id)
757                            }
758                            // We don't produce `AliasTy::Opaque`s yet.
759                            _ => unreachable!(),
760                        }
761                    }
762                    // `WhereClause::{TypeOutlives, LifetimeOutlives}` have been filtered out
763                    _ => unreachable!(),
764                }
765            });
766
767            if multiple_regular_traits || multiple_same_projection {
768                return None;
769            }
770
771            lowered_bounds.first().and_then(|b| b.trait_id())?;
772
773            // As multiple occurrences of the same auto traits *are* permitted, we deduplicate the
774            // bounds. We shouldn't have repeated elements besides auto traits at this point.
775            lowered_bounds.dedup();
776
777            Some(QuantifiedWhereClauses::from_iter(Interner, lowered_bounds))
778        });
779
780        if let Some(bounds) = bounds {
781            let bounds = crate::make_single_type_binders(bounds);
782            TyKind::Dyn(DynTy {
783                bounds,
784                lifetime: match lifetime {
785                    Some(it) => match it.bound_var(Interner) {
786                        Some(bound_var) => bound_var
787                            .shifted_out_to(DebruijnIndex::new(2))
788                            .map(|bound_var| LifetimeData::BoundVar(bound_var).intern(Interner))
789                            .unwrap_or(it),
790                        None => it,
791                    },
792                    None => static_lifetime(),
793                },
794            })
795            .intern(Interner)
796        } else {
797            // FIXME: report error
798            // (additional non-auto traits, associated type rebound, or no resolved trait)
799            TyKind::Error.intern(Interner)
800        }
801    }

803    fn lower_impl_trait(&mut self, bounds: &[TypeBound], krate: Crate) -> ImplTrait {
804        cov_mark::hit!(lower_rpit);
805        let self_ty = TyKind::BoundVar(BoundVar::new(DebruijnIndex::INNERMOST, 0)).intern(Interner);
806        let predicates = self.with_shifted_in(DebruijnIndex::ONE, |ctx| {
807            let mut predicates = Vec::new();
808            for b in bounds {
809                predicates.extend(ctx.lower_type_bound(b, self_ty.clone(), false));
810            }
811
812            if !ctx.unsized_types.contains(&self_ty) {
813                let sized_trait = ctx
814                    .db
815                    .lang_item(krate, LangItem::Sized)
816                    .and_then(|lang_item| lang_item.as_trait().map(to_chalk_trait_id));
817                let sized_clause = sized_trait.map(|trait_id| {
818                    let clause = WhereClause::Implemented(TraitRef {
819                        trait_id,
820                        substitution: Substitution::from1(Interner, self_ty.clone()),
821                    });
822                    crate::wrap_empty_binders(clause)
823                });
824                predicates.extend(sized_clause);
825            }
826            predicates.shrink_to_fit();
827            predicates
828        });
829        ImplTrait { bounds: crate::make_single_type_binders(predicates) }
830    }

832    pub fn lower_lifetime(&self, lifetime: &LifetimeRef) -> Lifetime {
833        match self.resolver.resolve_lifetime(lifetime) {
834            Some(resolution) => match resolution {
835                LifetimeNs::Static => static_lifetime(),
836                LifetimeNs::LifetimeParam(id) => match self.type_param_mode {
837                    ParamLoweringMode::Placeholder => {
838                        LifetimeData::Placeholder(lt_to_placeholder_idx(self.db, id))
839                    }
840                    ParamLoweringMode::Variable => {
841                        let generics = self.generics().expect("generics in scope");
842                        let idx = match generics.lifetime_idx(id) {
843                            None => return error_lifetime(),
844                            Some(idx) => idx,
845                        };
846
847                        LifetimeData::BoundVar(BoundVar::new(self.in_binders, idx))
848                    }
849                }
850                .intern(Interner),
851            },
852            None => error_lifetime(),
853        }
854    }

877pub fn associated_type_shorthand_candidates<R>(
878    db: &dyn HirDatabase,
879    def: GenericDefId,
880    res: TypeNs,
881    mut cb: impl FnMut(&Name, TypeAliasId) -> Option<R>,
882) -> Option<R> {
883    named_associated_type_shorthand_candidates(db, def, res, None, |name, _, id| cb(name, id))
884}
885
886fn named_associated_type_shorthand_candidates<R>(
887    db: &dyn HirDatabase,
888    // If the type parameter is defined in an impl and we're in a method, there
889    // might be additional where clauses to consider
890    def: GenericDefId,
891    res: TypeNs,
892    assoc_name: Option<Name>,
893    // Do NOT let `cb` touch `TraitRef` outside of `TyLoweringContext`. Its substitution contains
894    // free `BoundVar`s that need to be shifted and only `TyLoweringContext` knows how to do that
895    // properly (see `TyLoweringContext::select_associated_type()`).
896    mut cb: impl FnMut(&Name, &TraitRef, TypeAliasId) -> Option<R>,
897) -> Option<R> {
898    let mut search = |t| {
899        all_super_trait_refs(db, t, |t| {
900            let data = db.trait_items(t.hir_trait_id());
901
902            for (name, assoc_id) in &data.items {
903                if let AssocItemId::TypeAliasId(alias) = assoc_id {
904                    if let Some(result) = cb(name, &t, *alias) {
905                        return Some(result);
906                    }
907                }
908            }
909            None
910        })
911    };
912
913    match res {
914        TypeNs::SelfType(impl_id) => {
915            // we're _in_ the impl -- the binders get added back later. Correct,
916            // but it would be nice to make this more explicit
917            let trait_ref = db.impl_trait(impl_id)?.into_value_and_skipped_binders().0;
918
919            let impl_id_as_generic_def: GenericDefId = impl_id.into();
920            if impl_id_as_generic_def != def {
921                // `trait_ref` contains `BoundVar`s bound by impl's `Binders`, but here we need
922                // `BoundVar`s from `def`'s point of view.
923                // FIXME: A `HirDatabase` query may be handy if this process is needed in more
924                // places. It'd be almost identical as `impl_trait_query` where `resolver` would be
925                // of `def` instead of `impl_id`.
926                let starting_idx = generics(db.upcast(), def).len_self();
927                let subst = TyBuilder::subst_for_def(db, impl_id, None)
928                    .fill_with_bound_vars(DebruijnIndex::INNERMOST, starting_idx)
929                    .build();
930                let trait_ref = subst.apply(trait_ref, Interner);
931                search(trait_ref)
932            } else {
933                search(trait_ref)
934            }
935        }
936        TypeNs::GenericParam(param_id) => {
937            let predicates = db.generic_predicates_for_param(def, param_id.into(), assoc_name);
938            let res = predicates.iter().find_map(|pred| match pred.skip_binders().skip_binders() {
939                // FIXME: how to correctly handle higher-ranked bounds here?
940                WhereClause::Implemented(tr) => search(
941                    tr.clone()
942                        .shifted_out_to(Interner, DebruijnIndex::ONE)
943                        .expect("FIXME unexpected higher-ranked trait bound"),
944                ),
945                _ => None,
946            });
947            if res.is_some() {
948                return res;
949            }
950            // Handle `Self::Type` referring to own associated type in trait definitions
951            if let GenericDefId::TraitId(trait_id) = param_id.parent() {
952                let trait_generics = generics(db.upcast(), trait_id.into());
953                if trait_generics[param_id.local_id()].is_trait_self() {
954                    let def_generics = generics(db.upcast(), def);
955                    let starting_idx = match def {
956                        GenericDefId::TraitId(_) => 0,
957                        // `def` is an item within trait. We need to substitute `BoundVar`s but
958                        // remember that they are for parent (i.e. trait) generic params so they
959                        // come after our own params.
960                        _ => def_generics.len_self(),
961                    };
962                    let trait_ref = TyBuilder::trait_ref(db, trait_id)
963                        .fill_with_bound_vars(DebruijnIndex::INNERMOST, starting_idx)
964                        .build();
965                    return search(trait_ref);
966                }
967            }
968            None
969        }
970        _ => None,
971    }
972}

1010/// This query exists only to be used when resolving short-hand associated types
1011/// like `T::Item`.
1012///
1013/// See the analogous query in rustc and its comment:
1014/// <https://github.com/rust-lang/rust/blob/9150f844e2624eb013ec78ca08c1d416e6644026/src/librustc_typeck/astconv.rs#L46>
1015/// This is a query mostly to handle cycles somewhat gracefully; e.g. the
1016/// following bounds are disallowed: `T: Foo<U::Item>, U: Foo<T::Item>`, but
1017/// these are fine: `T: Foo<U::Item>, U: Foo<()>`.
1018pub(crate) fn generic_predicates_for_param_query(
1019    db: &dyn HirDatabase,
1020    def: GenericDefId,
1021    param_id: TypeOrConstParamId,
1022    assoc_name: Option<Name>,
1023) -> GenericPredicates {
1024    let resolver = def.resolver(db.upcast());
1025    let mut ctx = if let GenericDefId::FunctionId(_) = def {
1026        TyLoweringContext::new(db, &resolver, TypesMap::EMPTY, def.into())
1027            .with_impl_trait_mode(ImplTraitLoweringMode::Variable)
1028            .with_type_param_mode(ParamLoweringMode::Variable)
1029    } else {
1030        TyLoweringContext::new(db, &resolver, TypesMap::EMPTY, def.into())
1031            .with_type_param_mode(ParamLoweringMode::Variable)
1032    };
1033    let generics = generics(db.upcast(), def);
1034
1035    // we have to filter out all other predicates *first*, before attempting to lower them
1036    let predicate = |pred: &_, def: &_, ctx: &mut TyLoweringContext<'_>| match pred {
1037        WherePredicate::ForLifetime { target, bound, .. }
1038        | WherePredicate::TypeBound { target, bound, .. } => {
1039            let invalid_target = match target {
1040                WherePredicateTypeTarget::TypeRef(type_ref) => {
1041                    ctx.lower_ty_only_param(*type_ref) != Some(param_id)
1042                }
1043                &WherePredicateTypeTarget::TypeOrConstParam(local_id) => {
1044                    let target_id = TypeOrConstParamId { parent: *def, local_id };
1045                    target_id != param_id
1046                }
1047            };
1048            if invalid_target {
1049                // If this is filtered out without lowering, `?Sized` is not gathered into `ctx.unsized_types`
1050                if let TypeBound::Path(_, TraitBoundModifier::Maybe) = bound {
1051                    ctx.lower_where_predicate(pred, def, true).for_each(drop);
1052                }
1053                return false;
1054            }
1055
1056            match bound {
1057                &TypeBound::ForLifetime(_, path) | &TypeBound::Path(path, _) => {
1058                    // Only lower the bound if the trait could possibly define the associated
1059                    // type we're looking for.
1060                    let path = &ctx.types_map[path];
1061
1062                    let Some(assoc_name) = &assoc_name else { return true };
1063                    let Some(TypeNs::TraitId(tr)) =
1064                        resolver.resolve_path_in_type_ns_fully(db.upcast(), path)
1065                    else {
1066                        return false;
1067                    };
1068
1069                    all_super_traits(db.upcast(), tr).iter().any(|tr| {
1070                        db.trait_items(*tr).items.iter().any(|(name, item)| {
1071                            matches!(item, AssocItemId::TypeAliasId(_)) && name == assoc_name
1072                        })
1073                    })
1074                }
1075                TypeBound::Use(_) | TypeBound::Lifetime(_) | TypeBound::Error => false,
1076            }
1077        }
1078        WherePredicate::Lifetime { .. } => false,
1079    };
1080    let mut predicates = Vec::new();
1081    for (params, def) in resolver.all_generic_params() {
1082        ctx.types_map = &params.types_map;
1083        for pred in params.where_predicates() {
1084            if predicate(pred, def, &mut ctx) {
1085                predicates.extend(
1086                    ctx.lower_where_predicate(pred, def, true)
1087                        .map(|p| make_binders(db, &generics, p)),
1088                );
1089            }
1090        }
1091    }
1092
1093    let subst = generics.bound_vars_subst(db, DebruijnIndex::INNERMOST);
1094    if !subst.is_empty(Interner) {
1095        let explicitly_unsized_tys = ctx.unsized_types;
1096        if let Some(implicitly_sized_predicates) = implicitly_sized_clauses(
1097            db,
1098            param_id.parent,
1099            &explicitly_unsized_tys,
1100            &subst,
1101            &resolver,
1102        ) {
1103            predicates.extend(
1104                implicitly_sized_predicates
1105                    .map(|p| make_binders(db, &generics, crate::wrap_empty_binders(p))),
1106            );
1107        };
1108    }
1109    GenericPredicates(predicates.is_empty().not().then(|| predicates.into()))
1110}

1136    def: GenericDefId,
1137) -> Arc<TraitEnvironment> {
1138    let resolver = def.resolver(db.upcast());
1139    let mut ctx = if let GenericDefId::FunctionId(_) = def {
1140        TyLoweringContext::new(db, &resolver, TypesMap::EMPTY, def.into())
1141            .with_impl_trait_mode(ImplTraitLoweringMode::Param)
1142            .with_type_param_mode(ParamLoweringMode::Placeholder)
1143    } else {
1144        TyLoweringContext::new(db, &resolver, TypesMap::EMPTY, def.into())
1145            .with_type_param_mode(ParamLoweringMode::Placeholder)
1146    };
1147    let mut traits_in_scope = Vec::new();
1148    let mut clauses = Vec::new();
1149    for (params, def) in resolver.all_generic_params() {
1150        ctx.types_map = &params.types_map;
1151        for pred in params.where_predicates() {
1152            for pred in ctx.lower_where_predicate(pred, def, false) {
1153                if let WhereClause::Implemented(tr) = pred.skip_binders() {
1154                    traits_in_scope
1155                        .push((tr.self_type_parameter(Interner).clone(), tr.hir_trait_id()));
1156                }
1157                let program_clause: chalk_ir::ProgramClause<Interner> = pred.cast(Interner);
1158                clauses.push(program_clause.into_from_env_clause(Interner));
1159            }
1160        }
1161    }
1162
1163    if let Some(trait_id) = def.assoc_trait_container(db.upcast()) {
1164        // add `Self: Trait<T1, T2, ...>` to the environment in trait
1165        // function default implementations (and speculative code
1166        // inside consts or type aliases)
1167        cov_mark::hit!(trait_self_implements_self);
1168        let substs = TyBuilder::placeholder_subst(db, trait_id);
1169        let trait_ref = TraitRef { trait_id: to_chalk_trait_id(trait_id), substitution: substs };
1170        let pred = WhereClause::Implemented(trait_ref);
1171        clauses.push(pred.cast::<ProgramClause>(Interner).into_from_env_clause(Interner));
1172    }
1173
1174    let subst = generics(db.upcast(), def).placeholder_subst(db);
1175    if !subst.is_empty(Interner) {
1176        let explicitly_unsized_tys = ctx.unsized_types;
1177        if let Some(implicitly_sized_clauses) =
1178            implicitly_sized_clauses(db, def, &explicitly_unsized_tys, &subst, &resolver)
1179        {
1180            clauses.extend(
1181                implicitly_sized_clauses.map(|pred| {
1182                    pred.cast::<ProgramClause>(Interner).into_from_env_clause(Interner)
1183                }),
1184            );
1185        };
1186    }
1187
1188    let env = chalk_ir::Environment::new(Interner).add_clauses(Interner, clauses);
1189
1190    TraitEnvironment::new(resolver.krate(), None, traits_in_scope.into_boxed_slice(), env)
1191}

1228/// Resolve the where clause(s) of an item with generics,
1229/// except the ones inherited from the parent
1230fn generic_predicates_filtered_by<F>(
1231    db: &dyn HirDatabase,
1232    def: GenericDefId,
1233    filter: F,
1234) -> (GenericPredicates, Diagnostics)
1235where
1236    F: Fn(&WherePredicate, &GenericDefId) -> bool,
1237{
1238    let resolver = def.resolver(db.upcast());
1239    let (impl_trait_lowering, param_lowering) = match def {
1240        GenericDefId::FunctionId(_) => {
1241            (ImplTraitLoweringMode::Variable, ParamLoweringMode::Variable)
1242        }
1243        _ => (ImplTraitLoweringMode::Disallowed, ParamLoweringMode::Variable),
1244    };
1245    let mut ctx = TyLoweringContext::new(db, &resolver, TypesMap::EMPTY, def.into())
1246        .with_impl_trait_mode(impl_trait_lowering)
1247        .with_type_param_mode(param_lowering);
1248    let generics = generics(db.upcast(), def);
1249
1250    let mut predicates = Vec::new();
1251    for (params, def) in resolver.all_generic_params() {
1252        ctx.types_map = &params.types_map;
1253        for pred in params.where_predicates() {
1254            if filter(pred, def) {
1255                predicates.extend(
1256                    ctx.lower_where_predicate(pred, def, false)
1257                        .map(|p| make_binders(db, &generics, p)),
1258                );
1259            }
1260        }
1261    }
1262
1263    if generics.len() > 0 {
1264        let subst = generics.bound_vars_subst(db, DebruijnIndex::INNERMOST);
1265        let explicitly_unsized_tys = ctx.unsized_types;
1266        if let Some(implicitly_sized_predicates) =
1267            implicitly_sized_clauses(db, def, &explicitly_unsized_tys, &subst, &resolver)
1268        {
1269            predicates.extend(
1270                implicitly_sized_predicates
1271                    .map(|p| make_binders(db, &generics, crate::wrap_empty_binders(p))),
1272            );
1273        };
1274    }
1275
1276    (
1277        GenericPredicates(predicates.is_empty().not().then(|| predicates.into())),
1278        create_diagnostics(ctx.diagnostics),
1279    )
1280}

1282/// Generate implicit `: Sized` predicates for all generics that has no `?Sized` bound.
1283/// Exception is Self of a trait def.
1284fn implicitly_sized_clauses<'a, 'subst: 'a>(
1285    db: &dyn HirDatabase,
1286    def: GenericDefId,
1287    explicitly_unsized_tys: &'a FxHashSet<Ty>,
1288    substitution: &'subst Substitution,
1289    resolver: &Resolver,
1290) -> Option<impl Iterator<Item = WhereClause> + Captures<'a> + Captures<'subst>> {
1291    let sized_trait = db
1292        .lang_item(resolver.krate(), LangItem::Sized)
1293        .and_then(|lang_item| lang_item.as_trait().map(to_chalk_trait_id))?;
1294
1295    let trait_self_idx = trait_self_param_idx(db.upcast(), def);
1296
1297    Some(
1298        substitution
1299            .iter(Interner)
1300            .enumerate()
1301            .filter_map(
1302                move |(idx, generic_arg)| {
1303                    if Some(idx) == trait_self_idx { None } else { Some(generic_arg) }
1304                },
1305            )
1306            .filter_map(|generic_arg| generic_arg.ty(Interner))
1307            .filter(move |&self_ty| !explicitly_unsized_tys.contains(self_ty))
1308            .map(move |self_ty| {
1309                WhereClause::Implemented(TraitRef {
1310                    trait_id: sized_trait,
1311                    substitution: Substitution::from1(Interner, self_ty.clone()),
1312                })
1313            }),
1314    )
1315}

1332/// Resolve the default type params from generics.
1333///
1334/// Diagnostics are only returned for this `GenericDefId` (returned defaults include parents).
1335pub(crate) fn generic_defaults_with_diagnostics_query(
1336    db: &dyn HirDatabase,
1337    def: GenericDefId,
1338) -> (GenericDefaults, Diagnostics) {
1339    let generic_params = generics(db.upcast(), def);
1340    if generic_params.len() == 0 {
1341        return (GenericDefaults(None), None);
1342    }
1343    let resolver = def.resolver(db.upcast());
1344    let parent_start_idx = generic_params.len_self();
1345
1346    let mut ctx =
1347        TyLoweringContext::new(db, &resolver, generic_params.self_types_map(), def.into())
1348            .with_impl_trait_mode(ImplTraitLoweringMode::Disallowed)
1349            .with_type_param_mode(ParamLoweringMode::Variable);
1350    let mut idx = 0;
1351    let mut defaults = generic_params
1352        .iter_self()
1353        .map(|(id, p)| {
1354            let result =
1355                handle_generic_param(&mut ctx, idx, id, p, parent_start_idx, &generic_params);
1356            idx += 1;
1357            result
1358        })
1359        .collect::<Vec<_>>();
1360    let diagnostics = create_diagnostics(mem::take(&mut ctx.diagnostics));
1361    defaults.extend(generic_params.iter_parents_with_types_map().map(|((id, p), types_map)| {
1362        ctx.types_map = types_map;
1363        let result = handle_generic_param(&mut ctx, idx, id, p, parent_start_idx, &generic_params);
1364        idx += 1;
1365        result
1366    }));
1367    let defaults = GenericDefaults(Some(Arc::from_iter(defaults)));
1368    return (defaults, diagnostics);
1369
1370    fn handle_generic_param(
1371        ctx: &mut TyLoweringContext<'_>,
1372        idx: usize,
1373        id: GenericParamId,
1374        p: GenericParamDataRef<'_>,
1375        parent_start_idx: usize,
1376        generic_params: &Generics,
1377    ) -> Binders<crate::GenericArg> {
1378        match p {
1379            GenericParamDataRef::TypeParamData(p) => {
1380                let ty = p.default.as_ref().map_or(TyKind::Error.intern(Interner), |ty| {
1381                    // Each default can only refer to previous parameters.
1382                    // Type variable default referring to parameter coming
1383                    // after it is forbidden (FIXME: report diagnostic)
1384                    fallback_bound_vars(ctx.lower_ty(*ty), idx, parent_start_idx)
1385                });
1386                crate::make_binders(ctx.db, generic_params, ty.cast(Interner))
1387            }
1388            GenericParamDataRef::ConstParamData(p) => {
1389                let GenericParamId::ConstParamId(id) = id else {
1390                    unreachable!("Unexpected lifetime or type argument")
1391                };
1392
1393                let mut val = p.default.as_ref().map_or_else(
1394                    || unknown_const_as_generic(ctx.db.const_param_ty(id)),
1395                    |c| {
1396                        let param_ty = ctx.lower_ty(p.ty);
1397                        let c = ctx.lower_const(c, param_ty);
1398                        c.cast(Interner)
1399                    },
1400                );
1401                // Each default can only refer to previous parameters, see above.
1402                val = fallback_bound_vars(val, idx, parent_start_idx);
1403                make_binders(ctx.db, generic_params, val)
1404            }
1405            GenericParamDataRef::LifetimeParamData(_) => {
1406                make_binders(ctx.db, generic_params, error_lifetime().cast(Interner))
1407            }
1408        }
1409    }
1410}

1789/// Checks if the provided generic arg matches its expected kind, then lower them via
1790/// provided closures. Use unknown if there was kind mismatch.
1791///
1792pub(crate) fn generic_arg_to_chalk<'a, T>(
1793    db: &dyn HirDatabase,
1794    kind_id: GenericParamId,
1795    arg: &'a GenericArg,
1796    this: &mut T,
1797    types_map: &TypesMap,
1798    for_type: impl FnOnce(&mut T, TypeRefId) -> Ty + 'a,
1799    for_const: impl FnOnce(&mut T, &ConstRef, Ty) -> Const + 'a,
1800    for_lifetime: impl FnOnce(&mut T, &LifetimeRef) -> Lifetime + 'a,
1801) -> crate::GenericArg {
1802    let kind = match kind_id {
1803        GenericParamId::TypeParamId(_) => ParamKind::Type,
1804        GenericParamId::ConstParamId(id) => {
1805            let ty = db.const_param_ty(id);
1806            ParamKind::Const(ty)
1807        }
1808        GenericParamId::LifetimeParamId(_) => ParamKind::Lifetime,
1809    };
1810    match (arg, kind) {
1811        (GenericArg::Type(type_ref), ParamKind::Type) => for_type(this, *type_ref).cast(Interner),
1812        (GenericArg::Const(c), ParamKind::Const(c_ty)) => for_const(this, c, c_ty).cast(Interner),
1813        (GenericArg::Lifetime(lifetime_ref), ParamKind::Lifetime) => {
1814            for_lifetime(this, lifetime_ref).cast(Interner)
1815        }
1816        (GenericArg::Const(_), ParamKind::Type) => TyKind::Error.intern(Interner).cast(Interner),
1817        (GenericArg::Lifetime(_), ParamKind::Type) => TyKind::Error.intern(Interner).cast(Interner),
1818        (GenericArg::Type(t), ParamKind::Const(c_ty)) => {
1819            // We want to recover simple idents, which parser detects them
1820            // as types. Maybe here is not the best place to do it, but
1821            // it works.
1822            if let TypeRef::Path(p) = &types_map[*t] {
1823                if let Some(p) = p.mod_path() {
1824                    if p.kind == PathKind::Plain {
1825                        if let [n] = p.segments() {
1826                            let c = ConstRef::Path(n.clone());
1827                            return for_const(this, &c, c_ty).cast(Interner);
1828                        }
1829                    }
1830                }
1831            }
1832            unknown_const_as_generic(c_ty)
1833        }
1834        (GenericArg::Lifetime(_), ParamKind::Const(c_ty)) => unknown_const_as_generic(c_ty),
1835        (GenericArg::Type(_), ParamKind::Lifetime) => error_lifetime().cast(Interner),
1836        (GenericArg::Const(_), ParamKind::Lifetime) => error_lifetime().cast(Interner),
1837    }
1838}

1841    db: &dyn HirDatabase,
1842    resolver: &Resolver,
1843    owner: TypeOwnerId,

1845    value: &ConstRef,
1846    mode: ParamLoweringMode,
1847    args: impl FnOnce() -> Option<&'g Generics>,
1848    debruijn: DebruijnIndex,
1849) -> Const {
1850    match value {
1851        ConstRef::Scalar(s) => intern_const_ref(db, s, expected_ty, resolver.krate()),
1852        ConstRef::Path(n) => {

1855                db,
1856                resolver,
1857                &Path::from_known_path_with_no_generic(path),
1858                mode,
1859                args,
1860                debruijn,
1861                expected_ty.clone(),

1870                && krate.env(db).get("__ra_is_test_fixture").is_none()
1871            {

1876            let c = db
1877                .intern_in_type_const(InTypeConstLoc {
1878                    id: it,

1880                    expected_ty: Box::new(InTypeConstIdMetadata(expected_ty.clone())),

1884                ConstScalar::UnevaluatedConst(c, Substitution::empty(Interner)),

1887        }
1888    }

1891/// Replaces any 'free' `BoundVar`s in `s` by `TyKind::Error` from the perspective of generic
1892/// parameter whose index is `param_index`. A `BoundVar` is free when it is or (syntactically)
1893/// appears after the generic parameter of `param_index`.
1894fn fallback_bound_vars<T: TypeFoldable<Interner> + HasInterner<Interner = Interner>>(
1895    s: T,
1896    param_index: usize,
1897    parent_start: usize,
1898) -> T {
1899    // Keep in mind that parent generic parameters, if any, come *after* those of the item in
1900    // question. In the diagrams below, `c*` and `p*` represent generic parameters of the item and
1901    // its parent respectively.
1902    let is_allowed = |index| {
1903        if param_index < parent_start {
1904            // The parameter of `param_index` is one from the item in question. Any parent generic
1905            // parameters or the item's generic parameters that come before `param_index` is
1906            // allowed.
1907            // [c1, .., cj, .., ck, p1, .., pl] where cj is `param_index`
1908            //  ^^^^^^              ^^^^^^^^^^ these are allowed
1909            !(param_index..parent_start).contains(&index)
1910        } else {
1911            // The parameter of `param_index` is one from the parent generics. Only parent generic
1912            // parameters that come before `param_index` are allowed.
1913            // [c1, .., ck, p1, .., pj, .., pl] where pj is `param_index`
1914            //              ^^^^^^ these are allowed
1915            (parent_start..param_index).contains(&index)
1916        }
1917    };
1918
1919    crate::fold_free_vars(
1920        s,
1921        |bound, binders| {
1922            if bound.index_if_innermost().is_none_or(is_allowed) {
1923                bound.shifted_in_from(binders).to_ty(Interner)
1924            } else {
1925                TyKind::Error.intern(Interner)
1926            }
1927        },
1928        |ty, bound, binders| {
1929            if bound.index_if_innermost().is_none_or(is_allowed) {
1930                bound.shifted_in_from(binders).to_const(Interner, ty)
1931            } else {
1932                unknown_const(ty)
1933            }
1934        },
1935    )
