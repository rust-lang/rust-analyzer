1//! Methods for lowering the HIR to types. There are two main cases here:
2//!
3//!  - Lowering a type reference like `&usize` or `Option<foo::bar::Baz>` to a
4//!    type: The entry point for this is `TyLoweringContext::lower_ty`.
5//!  - Building the type for an item: This happens through the `ty` query.
6//!
7//! This usually involves resolving names, collecting generic arguments etc.

88#[derive(Debug, Clone)]
89pub enum LifetimeElisionKind {
90    /// Create a new anonymous lifetime parameter and reference it.
91    ///
92    /// If `report_in_path`, report an error when encountering lifetime elision in a path:
93    /// ```compile_fail
94    /// struct Foo<'a> { x: &'a () }
95    /// async fn foo(x: Foo) {}
96    /// ```
97    ///
98    /// Note: the error should not trigger when the elided lifetime is in a pattern or
99    /// expression-position path:
100    /// ```
101    /// struct Foo<'a> { x: &'a () }
102    /// async fn foo(Foo { x: _ }: Foo<'_>) {}
103    /// ```
104    AnonymousCreateParameter { report_in_path: bool },
105
106    /// Replace all anonymous lifetimes by provided lifetime.
107    Elided(Lifetime),
108
109    /// Give a hard error when either `&` or `'_` is written. Used to
110    /// rule out things like `where T: Foo<'_>`. Does not imply an
111    /// error on default object bounds (e.g., `Box<dyn Foo>`).
112    AnonymousReportError,
113
114    /// Resolves elided lifetimes to `'static` if there are no other lifetimes in scope,
115    /// otherwise give a warning that the previous behavior of introducing a new early-bound
116    /// lifetime is a bug and will be removed (if `only_lint` is enabled).
117    StaticIfNoLifetimeInScope { only_lint: bool },
118
119    /// Signal we cannot find which should be the anonymous lifetime.
120    ElisionFailure,
121
122    /// Infer all elided lifetimes.
123    Infer,
124}
125
126impl LifetimeElisionKind {
127    #[inline]
128    pub(crate) fn for_fn_ret() -> LifetimeElisionKind {
129        // FIXME: We should use the elided lifetime here, or `ElisionFailure`.
130        LifetimeElisionKind::Elided(error_lifetime())
131    }
132}

180    pub fn with_debruijn<T>(
181        &mut self,
182        debruijn: DebruijnIndex,
183        f: impl FnOnce(&mut TyLoweringContext<'_>) -> T,
184    ) -> T {
185        let old_debruijn = mem::replace(&mut self.in_binders, debruijn);
186        let result = f(self);
187        self.in_binders = old_debruijn;
188        result
189    }

192        &mut self,
193        debruijn: DebruijnIndex,
194        f: impl FnOnce(&mut TyLoweringContext<'_>) -> T,
195    ) -> T {
196        self.with_debruijn(self.in_binders.shifted_in_from(debruijn), f)
197    }

200        &mut self,
201        lifetime_elision: LifetimeElisionKind,
202        f: impl FnOnce(&mut TyLoweringContext<'_>) -> T,
203    ) -> T {
204        let old_lifetime_elision = mem::replace(&mut self.lifetime_elision, lifetime_elision);
205        let result = f(self);
206        self.lifetime_elision = old_lifetime_elision;

256    pub fn lower_const(&mut self, const_ref: &ConstRef, const_type: Ty) -> Const {
257        let const_ref = &self.store[const_ref.expr];
258        match const_ref {
259            hir_def::hir::Expr::Path(path) => path_to_const(
260                self.db,
261                self.resolver,
262                path,
263                self.type_param_mode,
264                || self.generics(),
265                self.in_binders,
266                const_type.clone(),
267            )
268            .unwrap_or_else(|| unknown_const(const_type)),
269            hir_def::hir::Expr::Literal(literal) => intern_const_ref(
270                self.db,
271                &match *literal {
272                    hir_def::hir::Literal::Float(_, _)
273                    | hir_def::hir::Literal::String(_)
274                    | hir_def::hir::Literal::ByteString(_)
275                    | hir_def::hir::Literal::CString(_) => LiteralConstRef::Unknown,
276                    hir_def::hir::Literal::Char(c) => LiteralConstRef::Char(c),
277                    hir_def::hir::Literal::Bool(b) => LiteralConstRef::Bool(b),
278                    hir_def::hir::Literal::Int(val, _) => LiteralConstRef::Int(val),
279                    hir_def::hir::Literal::Uint(val, _) => LiteralConstRef::UInt(val),
280                },
281                const_type,
282                self.resolver.krate(),
283            ),
284            hir_def::hir::Expr::UnaryOp { expr: inner_expr, op: hir_def::hir::UnaryOp::Neg } => {
285                if let hir_def::hir::Expr::Literal(literal) = &self.store[*inner_expr] {
286                    // Only handle negation for signed integers and floats
287                    match literal {
288                        hir_def::hir::Literal::Int(_, _) | hir_def::hir::Literal::Float(_, _) => {
289                            if let Some(negated_literal) = literal.clone().negate() {
290                                intern_const_ref(
291                                    self.db,
292                                    &negated_literal.into(),
293                                    const_type,
294                                    self.resolver.krate(),
295                                )
296                            } else {
297                                unknown_const(const_type)
298                            }
299                        }
300                        // For unsigned integers, chars, bools, etc., negation is not meaningful
301                        _ => unknown_const(const_type),
302                    }
303                } else {
304                    unknown_const(const_type)
305                }
306            }
307            _ => unknown_const(const_type),
308        }
309    }

328    pub fn lower_ty_ext(&mut self, type_ref_id: TypeRefId) -> (Ty, Option<TypeNs>) {
329        let mut res = None;
330        let type_ref = &self.store[type_ref_id];
331        let ty = match type_ref {
332            TypeRef::Never => TyKind::Never.intern(Interner),
333            TypeRef::Tuple(inner) => {
334                let inner_tys = inner.iter().map(|&tr| self.lower_ty(tr));
335                TyKind::Tuple(inner_tys.len(), Substitution::from_iter(Interner, inner_tys))
336                    .intern(Interner)
337            }
338            TypeRef::Path(path) => {
339                let (ty, res_) =
340                    self.lower_path(path, PathId::from_type_ref_unchecked(type_ref_id));
341                res = res_;
342                ty
343            }
344            &TypeRef::TypeParam(type_param_id) => {
345                res = Some(TypeNs::GenericParam(type_param_id));
346                match self.type_param_mode {
347                    ParamLoweringMode::Placeholder => {
348                        let generics = self.generics();
349                        let idx = generics.type_or_const_param_idx(type_param_id.into()).unwrap();
350                        TyKind::Placeholder(to_placeholder_idx(
351                            self.db,
352                            type_param_id.into(),
353                            idx as u32,
354                        ))
355                    }
356                    ParamLoweringMode::Variable => {
357                        let idx =
358                            self.generics().type_or_const_param_idx(type_param_id.into()).unwrap();
359                        TyKind::BoundVar(BoundVar::new(self.in_binders, idx))
360                    }
361                }
362                .intern(Interner)
363            }
364            &TypeRef::RawPtr(inner, mutability) => {
365                let inner_ty = self.lower_ty(inner);
366                TyKind::Raw(lower_to_chalk_mutability(mutability), inner_ty).intern(Interner)
367            }
368            TypeRef::Array(array) => {
369                let inner_ty = self.lower_ty(array.ty);
370                let const_len = self.lower_const(&array.len, TyBuilder::usize());
371                TyKind::Array(inner_ty, const_len).intern(Interner)
372            }
373            &TypeRef::Slice(inner) => {
374                let inner_ty = self.lower_ty(inner);
375                TyKind::Slice(inner_ty).intern(Interner)
376            }
377            TypeRef::Reference(ref_) => {
378                let inner_ty = self.lower_ty(ref_.ty);
379                // FIXME: It should infer the eldided lifetimes instead of stubbing with static
380                let lifetime = ref_
381                    .lifetime
382                    .as_ref()
383                    .map_or_else(error_lifetime, |&lr| self.lower_lifetime(lr));
384                TyKind::Ref(lower_to_chalk_mutability(ref_.mutability), lifetime, inner_ty)
385                    .intern(Interner)
386            }
387            TypeRef::Placeholder => TyKind::Error.intern(Interner),
388            TypeRef::Fn(fn_) => {
389                let substs = self.with_shifted_in(DebruijnIndex::ONE, |ctx| {
390                    let (params, ret) = fn_.split_params_and_ret();
391                    let mut subst = Vec::with_capacity(fn_.params.len());
392                    ctx.with_lifetime_elision(
393                        LifetimeElisionKind::AnonymousCreateParameter { report_in_path: false },
394                        |ctx| {
395                            subst.extend(params.iter().map(|&(_, tr)| ctx.lower_ty(tr)));
396                        },
397                    );
398                    ctx.with_lifetime_elision(LifetimeElisionKind::for_fn_ret(), |ctx| {
399                        subst.push(ctx.lower_ty(ret));
400                    });
401                    Substitution::from_iter(Interner, subst)
402                });
403                TyKind::Function(FnPointer {
404                    num_binders: 0, // FIXME lower `for<'a> fn()` correctly
405                    sig: FnSig {
406                        abi: fn_.abi.as_ref().map_or(FnAbi::Rust, FnAbi::from_symbol),
407                        safety: if fn_.is_unsafe { Safety::Unsafe } else { Safety::Safe },
408                        variadic: fn_.is_varargs,
409                    },
410                    substitution: FnSubst(substs),
411                })
412                .intern(Interner)
413            }
414            TypeRef::DynTrait(bounds) => self.lower_dyn_trait(bounds),
415            TypeRef::ImplTrait(bounds) => {
416                match self.impl_trait_mode.mode {
417                    ImplTraitLoweringMode::Opaque => {
418                        let origin = match self.def {
419                            GenericDefId::FunctionId(it) => Either::Left(it),
420                            GenericDefId::TypeAliasId(it) => Either::Right(it),
421                            _ => panic!(
422                                "opaque impl trait lowering must be in function or type alias"
423                            ),
424                        };
425
426                        // this dance is to make sure the data is in the right
427                        // place even if we encounter more opaque types while
428                        // lowering the bounds
429                        let idx = self.impl_trait_mode.opaque_type_data.alloc(ImplTrait {
430                            bounds: crate::make_single_type_binders(Vec::default()),
431                        });
432                        // We don't want to lower the bounds inside the binders
433                        // we're currently in, because they don't end up inside
434                        // those binders. E.g. when we have `impl Trait<impl
435                        // OtherTrait<T>>`, the `impl OtherTrait<T>` can't refer
436                        // to the self parameter from `impl Trait`, and the
437                        // bounds aren't actually stored nested within each
438                        // other, but separately. So if the `T` refers to a type
439                        // parameter of the outer function, it's just one binder
440                        // away instead of two.
441                        let actual_opaque_type_data = self
442                            .with_debruijn(DebruijnIndex::INNERMOST, |ctx| {
443                                ctx.lower_impl_trait(bounds, self.resolver.krate())
444                            });
445                        self.impl_trait_mode.opaque_type_data[idx] = actual_opaque_type_data;
446
447                        let impl_trait_id = origin.either(
448                            |f| ImplTraitId::ReturnTypeImplTrait(f, idx),
449                            |a| ImplTraitId::TypeAliasImplTrait(a, idx),
450                        );
451                        let opaque_ty_id = self.db.intern_impl_trait_id(impl_trait_id).into();
452                        let generics = generics(self.db, origin.either(|f| f.into(), |a| a.into()));
453                        let parameters = generics.bound_vars_subst(self.db, self.in_binders);
454                        TyKind::OpaqueType(opaque_ty_id, parameters).intern(Interner)
455                    }
456                    ImplTraitLoweringMode::Disallowed => {
457                        // FIXME: report error
458                        TyKind::Error.intern(Interner)
459                    }
460                }
461            }
462            TypeRef::Error => TyKind::Error.intern(Interner),
463        };
464        (ty, res)
465    }

467    /// This is only for `generic_predicates_for_param`, where we can't just
468    /// lower the self types of the predicates since that could lead to cycles.
469    /// So we just check here if the `type_ref` resolves to a generic param, and which.
470    fn lower_ty_only_param(&mut self, type_ref_id: TypeRefId) -> Option<TypeOrConstParamId> {
471        let type_ref = &self.store[type_ref_id];
472        let path = match type_ref {
473            TypeRef::Path(path) => path,
474            &TypeRef::TypeParam(idx) => return Some(idx.into()),
475            _ => return None,
476        };
477        if path.type_anchor().is_some() {
478            return None;
479        }
480        if path.segments().len() > 1 {
481            return None;
482        }
483        let mut ctx = self.at_path(PathId::from_type_ref_unchecked(type_ref_id));
484        let resolution = match ctx.resolve_path_in_type_ns() {
485            Some((it, None)) => it,
486            _ => return None,
487        };
488        match resolution {
489            TypeNs::GenericParam(param_id) => Some(param_id.into()),
490            _ => None,
491        }
492    }

514    pub(crate) fn lower_path(&mut self, path: &Path, path_id: PathId) -> (Ty, Option<TypeNs>) {
515        // Resolve the path (in type namespace)
516        if let Some(type_ref) = path.type_anchor() {
517            let (ty, res) = self.lower_ty_ext(type_ref);
518            let mut ctx = self.at_path(path_id);
519            return ctx.lower_ty_relative_path(ty, res, false);
520        }
521
522        let mut ctx = self.at_path(path_id);
523        let (resolution, remaining_index) = match ctx.resolve_path_in_type_ns() {
524            Some(it) => it,
525            None => return (TyKind::Error.intern(Interner), None),
526        };
527
528        if matches!(resolution, TypeNs::TraitId(_)) && remaining_index.is_none() {
529            // trait object type without dyn
530            let bound = TypeBound::Path(path_id, TraitBoundModifier::None);
531            let ty = self.lower_dyn_trait(&[bound]);
532            return (ty, None);
533        }
534
535        ctx.lower_partly_resolved_path(resolution, false)
536    }

552    /// When lowering predicates from parents (impl, traits) for children defs (fns, consts, types), `generics` should
553    /// contain the `Generics` for the **child**, while `predicate_owner` should contain the `GenericDefId` of the
554    /// **parent**. This is important so we generate the correct bound var/placeholder.
555    pub(crate) fn lower_where_predicate<'b>(
556        &'b mut self,
557        where_predicate: &'b WherePredicate,
558        ignore_bindings: bool,
559    ) -> impl Iterator<Item = QuantifiedWhereClause> + use<'db, 'b> {
560        match where_predicate {
561            WherePredicate::ForLifetime { target, bound, .. }
562            | WherePredicate::TypeBound { target, bound } => {
563                let self_ty = self.lower_ty(*target);
564                Either::Left(self.lower_type_bound(bound, self_ty, ignore_bindings))
565            }
566            &WherePredicate::Lifetime { bound, target } => Either::Right(iter::once(
567                crate::wrap_empty_binders(WhereClause::LifetimeOutlives(LifetimeOutlives {
568                    a: self.lower_lifetime(bound),
569                    b: self.lower_lifetime(target),
570                })),
571            )),
572        }
573        .into_iter()
574    }

576    pub(crate) fn lower_type_bound<'b>(
577        &'b mut self,
578        bound: &'b TypeBound,
579        self_ty: Ty,
580        ignore_bindings: bool,
581    ) -> impl Iterator<Item = QuantifiedWhereClause> + use<'b, 'db> {
582        let mut assoc_bounds = None;
583        let mut clause = None;
584        match bound {
585            &TypeBound::Path(path, TraitBoundModifier::None) | &TypeBound::ForLifetime(_, path) => {
586                // FIXME Don't silently drop the hrtb lifetimes here
587                if let Some((trait_ref, mut ctx)) =
588                    self.lower_trait_ref_from_path(path, self_ty.clone())
589                {
590                    // FIXME(sized-hierarchy): Remove this bound modifications once we have implemented
591                    // sized-hierarchy correctly.
592                    let meta_sized = LangItem::MetaSized
593                        .resolve_trait(ctx.ty_ctx().db, ctx.ty_ctx().resolver.krate());
594                    let pointee_sized = LangItem::PointeeSized
595                        .resolve_trait(ctx.ty_ctx().db, ctx.ty_ctx().resolver.krate());
596                    let destruct = LangItem::Destruct
597                        .resolve_trait(ctx.ty_ctx().db, ctx.ty_ctx().resolver.krate());
598                    let hir_trait_id = trait_ref.hir_trait_id();
599                    if meta_sized.is_some_and(|it| it == hir_trait_id)
600                        || destruct.is_some_and(|it| it == hir_trait_id)
601                    {
602                        // Ignore this bound
603                    } else if pointee_sized.is_some_and(|it| it == hir_trait_id) {
604                        // Regard this as `?Sized` bound
605                        ctx.ty_ctx().unsized_types.insert(self_ty);
606                    } else {
607                        if !ignore_bindings {
608                            assoc_bounds =
609                                ctx.assoc_type_bindings_from_type_bound(trait_ref.clone());
610                        }
611                        clause =
612                            Some(crate::wrap_empty_binders(WhereClause::Implemented(trait_ref)));
613                    }
614                }
615            }
616            &TypeBound::Path(path, TraitBoundModifier::Maybe) => {
617                let sized_trait = LangItem::Sized.resolve_trait(self.db, self.resolver.krate());
618                // Don't lower associated type bindings as the only possible relaxed trait bound
619                // `?Sized` has no of them.
620                // If we got another trait here ignore the bound completely.
621                let trait_id = self
622                    .lower_trait_ref_from_path(path, self_ty.clone())
623                    .map(|(trait_ref, _)| trait_ref.hir_trait_id());
624                if trait_id == sized_trait {
625                    self.unsized_types.insert(self_ty);
626                }
627            }
628            &TypeBound::Lifetime(l) => {
629                let lifetime = self.lower_lifetime(l);
630                clause = Some(crate::wrap_empty_binders(WhereClause::TypeOutlives(TypeOutlives {
631                    ty: self_ty,
632                    lifetime,
633                })));
634            }
635            TypeBound::Use(_) | TypeBound::Error => {}
636        }
637        clause.into_iter().chain(assoc_bounds.into_iter().flatten())
638    }
639
640    fn lower_dyn_trait(&mut self, bounds: &[TypeBound]) -> Ty {
641        let self_ty = TyKind::BoundVar(BoundVar::new(DebruijnIndex::INNERMOST, 0)).intern(Interner);
642        // INVARIANT: The principal trait bound, if present, must come first. Others may be in any
643        // order but should be in the same order for the same set but possibly different order of
644        // bounds in the input.
645        // INVARIANT: If this function returns `DynTy`, there should be at least one trait bound.
646        // These invariants are utilized by `TyExt::dyn_trait()` and chalk.
647        let mut lifetime = None;
648        let bounds = self.with_shifted_in(DebruijnIndex::ONE, |ctx| {
649            let mut lowered_bounds = Vec::new();
650            for b in bounds {
651                ctx.lower_type_bound(b, self_ty.clone(), false).for_each(|b| {
652                    let filter = match b.skip_binders() {
653                        WhereClause::Implemented(_) | WhereClause::AliasEq(_) => true,
654                        WhereClause::LifetimeOutlives(_) => false,
655                        WhereClause::TypeOutlives(t) => {
656                            lifetime = Some(t.lifetime.clone());
657                            false
658                        }
659                    };
660                    if filter {
661                        lowered_bounds.push(b);
662                    }
663                });
664            }
665
666            let mut multiple_regular_traits = false;
667            let mut multiple_same_projection = false;
668            lowered_bounds.sort_unstable_by(|lhs, rhs| {
669                use std::cmp::Ordering;
670                match (lhs.skip_binders(), rhs.skip_binders()) {
671                    (WhereClause::Implemented(lhs), WhereClause::Implemented(rhs)) => {
672                        let lhs_id = lhs.trait_id;
673                        let lhs_is_auto = ctx
674                            .db
675                            .trait_signature(from_chalk_trait_id(lhs_id))
676                            .flags
677                            .contains(TraitFlags::AUTO);
678                        let rhs_id = rhs.trait_id;
679                        let rhs_is_auto = ctx
680                            .db
681                            .trait_signature(from_chalk_trait_id(rhs_id))
682                            .flags
683                            .contains(TraitFlags::AUTO);
684
685                        if !lhs_is_auto && !rhs_is_auto {
686                            multiple_regular_traits = true;
687                        }
688                        // Note that the ordering here is important; this ensures the invariant
689                        // mentioned above.
690                        (lhs_is_auto, lhs_id).cmp(&(rhs_is_auto, rhs_id))
691                    }
692                    (WhereClause::Implemented(_), _) => Ordering::Less,
693                    (_, WhereClause::Implemented(_)) => Ordering::Greater,
694                    (WhereClause::AliasEq(lhs), WhereClause::AliasEq(rhs)) => {
695                        match (&lhs.alias, &rhs.alias) {
696                            (AliasTy::Projection(lhs_proj), AliasTy::Projection(rhs_proj)) => {
697                                // We only compare the `associated_ty_id`s. We shouldn't have
698                                // multiple bounds for an associated type in the correct Rust code,
699                                // and if we do, we error out.
700                                if lhs_proj.associated_ty_id == rhs_proj.associated_ty_id {
701                                    multiple_same_projection = true;
702                                }
703                                lhs_proj.associated_ty_id.cmp(&rhs_proj.associated_ty_id)
704                            }
705                            // We don't produce `AliasTy::Opaque`s yet.
706                            _ => unreachable!(),
707                        }
708                    }
709                    // `WhereClause::{TypeOutlives, LifetimeOutlives}` have been filtered out
710                    _ => unreachable!(),
711                }
712            });
713
714            if multiple_regular_traits || multiple_same_projection {
715                return None;
716            }
717
718            lowered_bounds.first().and_then(|b| b.trait_id())?;
719
720            // As multiple occurrences of the same auto traits *are* permitted, we deduplicate the
721            // bounds. We shouldn't have repeated elements besides auto traits at this point.
722            lowered_bounds.dedup();
723
724            Some(QuantifiedWhereClauses::from_iter(Interner, lowered_bounds))
725        });
726
727        if let Some(bounds) = bounds {
728            let bounds = crate::make_single_type_binders(bounds);
729            TyKind::Dyn(DynTy {
730                bounds,
731                lifetime: match lifetime {
732                    Some(it) => match it.bound_var(Interner) {
733                        Some(bound_var) => bound_var
734                            .shifted_out_to(DebruijnIndex::new(2))
735                            .map(|bound_var| LifetimeData::BoundVar(bound_var).intern(Interner))
736                            .unwrap_or(it),
737                        None => it,
738                    },
739                    None => error_lifetime(),
740                },
741            })
742            .intern(Interner)
743        } else {
744            // FIXME: report error
745            // (additional non-auto traits, associated type rebound, or no resolved trait)
746            TyKind::Error.intern(Interner)
747        }
748    }
749
750    fn lower_impl_trait(&mut self, bounds: &[TypeBound], krate: Crate) -> ImplTrait {
751        cov_mark::hit!(lower_rpit);
752        let self_ty = TyKind::BoundVar(BoundVar::new(DebruijnIndex::INNERMOST, 0)).intern(Interner);
753        let predicates = self.with_shifted_in(DebruijnIndex::ONE, |ctx| {
754            let mut predicates = Vec::new();
755            for b in bounds {
756                predicates.extend(ctx.lower_type_bound(b, self_ty.clone(), false));
757            }
758
759            if !ctx.unsized_types.contains(&self_ty) {
760                let sized_trait =
761                    LangItem::Sized.resolve_trait(ctx.db, krate).map(to_chalk_trait_id);
762                let sized_clause = sized_trait.map(|trait_id| {
763                    let clause = WhereClause::Implemented(TraitRef {
764                        trait_id,
765                        substitution: Substitution::from1(Interner, self_ty.clone()),
766                    });
767                    crate::wrap_empty_binders(clause)
768                });
769                predicates.extend(sized_clause);
770            }
771            predicates.shrink_to_fit();
772            predicates
773        });
774        ImplTrait { bounds: crate::make_single_type_binders(predicates) }
775    }
776
777    pub fn lower_lifetime(&self, lifetime: LifetimeRefId) -> Lifetime {
778        match self.resolver.resolve_lifetime(&self.store[lifetime]) {
779            Some(resolution) => match resolution {
780                LifetimeNs::Static => static_lifetime(),
781                LifetimeNs::LifetimeParam(id) => match self.type_param_mode {
782                    ParamLoweringMode::Placeholder => {
783                        let generics = self.generics();
784                        let idx = generics.lifetime_idx(id).unwrap();
785                        LifetimeData::Placeholder(lt_to_placeholder_idx(self.db, id, idx as u32))
786                    }
787                    ParamLoweringMode::Variable => {
788                        let idx = match self.generics().lifetime_idx(id) {
789                            None => return error_lifetime(),
790                            Some(idx) => idx,
791                        };
792
793                        LifetimeData::BoundVar(BoundVar::new(self.in_binders, idx))
794                    }
795                }
796                .intern(Interner),
797            },
798            None => error_lifetime(),
799        }
800    }

803fn named_associated_type_shorthand_candidates<R>(
804    db: &dyn HirDatabase,
805    // If the type parameter is defined in an impl and we're in a method, there
806    // might be additional where clauses to consider
807    def: GenericDefId,
808    res: TypeNs,
809    assoc_name: Option<Name>,
810    // Do NOT let `cb` touch `TraitRef` outside of `TyLoweringContext`. Its substitution contains
811    // free `BoundVar`s that need to be shifted and only `TyLoweringContext` knows how to do that
812    // properly (see `TyLoweringContext::select_associated_type()`).
813    mut cb: impl FnMut(&Name, &TraitRef, TypeAliasId) -> Option<R>,
814) -> Option<R> {
815    let mut search = |t| {
816        all_super_trait_refs(db, t, |t| {
817            let data = t.hir_trait_id().trait_items(db);
818
819            for (name, assoc_id) in &data.items {
820                if let AssocItemId::TypeAliasId(alias) = assoc_id
821                    && let Some(result) = cb(name, &t, *alias)
822                {
823                    return Some(result);
824                }
825            }
826            None
827        })
828    };
829
830    let interner = DbInterner::new_with(db, None, None);
831    match res {
832        TypeNs::SelfType(impl_id) => {
833            let trait_ref = db.impl_trait(impl_id)?;
834
835            let impl_id_as_generic_def: GenericDefId = impl_id.into();
836            if impl_id_as_generic_def != def {
837                let subst = TyBuilder::subst_for_def(db, impl_id, None)
838                    .fill_with_bound_vars(DebruijnIndex::INNERMOST, 0)
839                    .build();
840                let args: crate::next_solver::GenericArgs<'_> = subst.to_nextsolver(interner);
841                let trait_ref = trait_ref.instantiate(interner, args).to_chalk(interner);
842                search(trait_ref)
843            } else {
844                search(trait_ref.skip_binder().to_chalk(interner))
845            }
846        }
847        TypeNs::GenericParam(param_id) => {
848            let predicates = db.generic_predicates_for_param(def, param_id.into(), assoc_name);
849            let res = predicates.iter().find_map(|pred| match pred.skip_binders().skip_binders() {
850                // FIXME: how to correctly handle higher-ranked bounds here?
851                WhereClause::Implemented(tr) => search(
852                    tr.clone()
853                        .shifted_out_to(Interner, DebruijnIndex::ONE)
854                        .expect("FIXME unexpected higher-ranked trait bound"),
855                ),
856                _ => None,
857            });
858            if res.is_some() {
859                return res;
860            }
861            // Handle `Self::Type` referring to own associated type in trait definitions
862            if let GenericDefId::TraitId(trait_id) = param_id.parent() {
863                let trait_generics = generics(db, trait_id.into());
864                if trait_generics[param_id.local_id()].is_trait_self() {
865                    let trait_ref = TyBuilder::trait_ref(db, trait_id)
866                        .fill_with_bound_vars(DebruijnIndex::INNERMOST, 0)
867                        .build();
868                    return search(trait_ref);
869                }
870            }
871            None
872        }
873        _ => None,
874    }
875}

890/// Build the type of all specific fields of a struct or enum variant.
891pub(crate) fn field_types_with_diagnostics_query(
892    db: &dyn HirDatabase,
893    variant_id: VariantId,
894) -> (Arc<ArenaMap<LocalFieldId, Binders<Ty>>>, Diagnostics) {
895    let var_data = variant_id.fields(db);
896    let fields = var_data.fields();
897    if fields.is_empty() {
898        return (Arc::new(ArenaMap::default()), None);
899    }
900
901    let (resolver, def): (_, GenericDefId) = match variant_id {
902        VariantId::StructId(it) => (it.resolver(db), it.into()),
903        VariantId::UnionId(it) => (it.resolver(db), it.into()),
904        VariantId::EnumVariantId(it) => (it.resolver(db), it.lookup(db).parent.into()),
905    };
906    let generics = generics(db, def);
907    let mut res = ArenaMap::default();
908    let mut ctx = TyLoweringContext::new(
909        db,
910        &resolver,
911        &var_data.store,
912        def,
913        LifetimeElisionKind::AnonymousReportError,
914    )
915    .with_type_param_mode(ParamLoweringMode::Variable);
916    for (field_id, field_data) in fields.iter() {
917        res.insert(field_id, make_binders(db, &generics, ctx.lower_ty(field_data.type_ref)));
918    }
919    (Arc::new(res), create_diagnostics(ctx.diagnostics))
920}

922/// This query exists only to be used when resolving short-hand associated types
923/// like `T::Item`.
924///
925/// See the analogous query in rustc and its comment:
926/// <https://github.com/rust-lang/rust/blob/9150f844e2624eb013ec78ca08c1d416e6644026/src/librustc_typeck/astconv.rs#L46>
927/// This is a query mostly to handle cycles somewhat gracefully; e.g. the
928/// following bounds are disallowed: `T: Foo<U::Item>, U: Foo<T::Item>`, but
929/// these are fine: `T: Foo<U::Item>, U: Foo<()>`.
930pub(crate) fn generic_predicates_for_param_query(
931    db: &dyn HirDatabase,
932    def: GenericDefId,
933    param_id: TypeOrConstParamId,
934    assoc_name: Option<Name>,
935) -> GenericPredicates {
936    let generics = generics(db, def);
937    if generics.has_no_predicates() && generics.is_empty() {
938        return GenericPredicates(None);
939    }
940
941    let resolver = def.resolver(db);
942    let mut ctx = TyLoweringContext::new(
943        db,
944        &resolver,
945        generics.store(),
946        def,
947        LifetimeElisionKind::AnonymousReportError,
948    )
949    .with_type_param_mode(ParamLoweringMode::Variable);
950
951    // we have to filter out all other predicates *first*, before attempting to lower them
952    let predicate = |pred: &_, ctx: &mut TyLoweringContext<'_>| match pred {
953        WherePredicate::ForLifetime { target, bound, .. }
954        | WherePredicate::TypeBound { target, bound, .. } => {
955            let invalid_target = { ctx.lower_ty_only_param(*target) != Some(param_id) };
956            if invalid_target {
957                // FIXME(sized-hierarchy): Revisit and adjust this properly once we have implemented
958                // sized-hierarchy correctly.
959                // If this is filtered out without lowering, `?Sized` or `PointeeSized` is not gathered into
960                // `ctx.unsized_types`
961                let lower = || -> bool {
962                    match bound {
963                        TypeBound::Path(_, TraitBoundModifier::Maybe) => true,
964                        TypeBound::Path(path, _) | TypeBound::ForLifetime(_, path) => {
965                            let TypeRef::Path(path) = &ctx.store[path.type_ref()] else {
966                                return false;
967                            };
968                            let Some(pointee_sized) =
969                                LangItem::PointeeSized.resolve_trait(ctx.db, ctx.resolver.krate())
970                            else {
971                                return false;
972                            };
973                            // Lower the path directly with `Resolver` instead of PathLoweringContext`
974                            // to prevent diagnostics duplications.
975                            ctx.resolver.resolve_path_in_type_ns_fully(ctx.db, path).is_some_and(
976                                |it| matches!(it, TypeNs::TraitId(tr) if tr == pointee_sized),
977                            )
978                        }
979                        _ => false,
980                    }
981                }();
982                if lower {
983                    ctx.lower_where_predicate(pred, true).for_each(drop);
984                }
985                return false;
986            }
987
988            match bound {
989                &TypeBound::ForLifetime(_, path) | &TypeBound::Path(path, _) => {
990                    // Only lower the bound if the trait could possibly define the associated
991                    // type we're looking for.
992                    let path = &ctx.store[path];
993
994                    let Some(assoc_name) = &assoc_name else { return true };
995                    let Some(TypeNs::TraitId(tr)) =
996                        resolver.resolve_path_in_type_ns_fully(db, path)
997                    else {
998                        return false;
999                    };
1000
1001                    all_super_traits(db, tr).iter().any(|tr| {
1002                        tr.trait_items(db).items.iter().any(|(name, item)| {
1003                            matches!(item, AssocItemId::TypeAliasId(_)) && name == assoc_name
1004                        })
1005                    })
1006                }
1007                TypeBound::Use(_) | TypeBound::Lifetime(_) | TypeBound::Error => false,
1008            }
1009        }
1010        WherePredicate::Lifetime { .. } => false,
1011    };
1012    let mut predicates = Vec::new();
1013    for maybe_parent_generics in
1014        std::iter::successors(Some(&generics), |generics| generics.parent_generics())
1015    {
1016        ctx.store = maybe_parent_generics.store();
1017        for pred in maybe_parent_generics.where_predicates() {
1018            if predicate(pred, &mut ctx) {
1019                predicates.extend(
1020                    ctx.lower_where_predicate(pred, true).map(|p| make_binders(db, &generics, p)),
1021                );
1022            }
1023        }
1024    }
1025
1026    let subst = generics.bound_vars_subst(db, DebruijnIndex::INNERMOST);
1027    if !subst.is_empty(Interner) {
1028        let explicitly_unsized_tys = ctx.unsized_types;
1029        if let Some(implicitly_sized_predicates) = implicitly_sized_clauses(
1030            db,
1031            param_id.parent,
1032            &explicitly_unsized_tys,
1033            &subst,
1034            &resolver,
1035        ) {
1036            predicates.extend(
1037                implicitly_sized_predicates
1038                    .map(|p| make_binders(db, &generics, crate::wrap_empty_binders(p))),
1039            );
1040        };
1041    }
1042    GenericPredicates(predicates.is_empty().not().then(|| predicates.into()))
1043}

1065/// Resolve the where clause(s) of an item with generics.
1066pub(crate) fn generic_predicates_query(
1067    db: &dyn HirDatabase,
1068    def: GenericDefId,
1069) -> GenericPredicates {
1070    generic_predicates_filtered_by(db, def, |_, _| true).0
1071}

1073/// Resolve the where clause(s) of an item with generics,
1074/// with a given filter
1075fn generic_predicates_filtered_by<F>(
1076    db: &dyn HirDatabase,
1077    def: GenericDefId,
1078    filter: F,
1079) -> (GenericPredicates, Diagnostics)
1080where
1081    F: Fn(&WherePredicate, GenericDefId) -> bool,
1082{
1083    let generics = generics(db, def);
1084    if generics.has_no_predicates() && generics.is_empty() {
1085        return (GenericPredicates(None), None);
1086    }
1087
1088    let resolver = def.resolver(db);
1089    let mut ctx = TyLoweringContext::new(
1090        db,
1091        &resolver,
1092        generics.store(),
1093        def,
1094        LifetimeElisionKind::AnonymousReportError,
1095    )
1096    .with_type_param_mode(ParamLoweringMode::Variable);
1097
1098    let mut predicates = Vec::new();
1099    for maybe_parent_generics in
1100        std::iter::successors(Some(&generics), |generics| generics.parent_generics())
1101    {
1102        ctx.store = maybe_parent_generics.store();
1103        for pred in maybe_parent_generics.where_predicates() {
1104            if filter(pred, maybe_parent_generics.def()) {
1105                // We deliberately use `generics` and not `maybe_parent_generics` here. This is not a mistake!
1106                // If we use the parent generics
1107                predicates.extend(
1108                    ctx.lower_where_predicate(pred, false).map(|p| make_binders(db, &generics, p)),
1109                );
1110            }
1111        }
1112    }
1113
1114    if !generics.is_empty() {
1115        let subst = generics.bound_vars_subst(db, DebruijnIndex::INNERMOST);
1116        let explicitly_unsized_tys = ctx.unsized_types;
1117        if let Some(implicitly_sized_predicates) =
1118            implicitly_sized_clauses(db, def, &explicitly_unsized_tys, &subst, &resolver)
1119        {
1120            predicates.extend(
1121                implicitly_sized_predicates
1122                    .map(|p| make_binders(db, &generics, crate::wrap_empty_binders(p))),
1123            );
1124        };
1125    }
1126
1127    (
1128        GenericPredicates(predicates.is_empty().not().then(|| predicates.into())),
1129        create_diagnostics(ctx.diagnostics),
1130    )
1131}

1133/// Generate implicit `: Sized` predicates for all generics that has no `?Sized` bound.
1134/// Exception is Self of a trait def.
1135fn implicitly_sized_clauses<'db, 'a, 'subst: 'a>(
1136    db: &'db dyn HirDatabase,
1137    def: GenericDefId,
1138    explicitly_unsized_tys: &'a FxHashSet<Ty>,
1139    substitution: &'subst Substitution,
1140    resolver: &Resolver<'db>,
1141) -> Option<impl Iterator<Item = WhereClause>> {
1142    let sized_trait = LangItem::Sized.resolve_trait(db, resolver.krate()).map(to_chalk_trait_id)?;
1143
1144    let trait_self_idx = trait_self_param_idx(db, def);
1145
1146    Some(
1147        substitution
1148            .iter(Interner)
1149            .enumerate()
1150            .filter_map(
1151                move |(idx, generic_arg)| {
1152                    if Some(idx) == trait_self_idx { None } else { Some(generic_arg) }
1153                },
1154            )
1155            .filter_map(|generic_arg| generic_arg.ty(Interner))
1156            .filter(move |&self_ty| !explicitly_unsized_tys.contains(self_ty))
1157            .map(move |self_ty| {
1158                WhereClause::Implemented(TraitRef {
1159                    trait_id: sized_trait,
1160                    substitution: Substitution::from1(Interner, self_ty.clone()),
1161                })
1162            }),
1163    )
1164}

1181/// Resolve the default type params from generics.
1182///
1183/// Diagnostics are only returned for this `GenericDefId` (returned defaults include parents).
1184pub(crate) fn generic_defaults_with_diagnostics_query(
1185    db: &dyn HirDatabase,
1186    def: GenericDefId,
1187) -> (GenericDefaults, Diagnostics) {
1188    let generic_params = generics(db, def);
1189    if generic_params.is_empty() {
1190        return (GenericDefaults(None), None);
1191    }
1192    let resolver = def.resolver(db);
1193
1194    let mut ctx = TyLoweringContext::new(
1195        db,
1196        &resolver,
1197        generic_params.store(),
1198        def,
1199        LifetimeElisionKind::AnonymousReportError,
1200    )
1201    .with_impl_trait_mode(ImplTraitLoweringMode::Disallowed)
1202    .with_type_param_mode(ParamLoweringMode::Variable);
1203    let mut idx = 0;
1204    let mut has_any_default = false;
1205    let mut defaults = generic_params
1206        .iter_parents_with_store()
1207        .map(|((id, p), store)| {
1208            ctx.store = store;
1209            let (result, has_default) = handle_generic_param(&mut ctx, idx, id, p, &generic_params);
1210            has_any_default |= has_default;
1211            idx += 1;
1212            result
1213        })
1214        .collect::<Vec<_>>();
1215    ctx.diagnostics.clear(); // Don't include diagnostics from the parent.
1216    defaults.extend(generic_params.iter_self().map(|(id, p)| {
1217        let (result, has_default) = handle_generic_param(&mut ctx, idx, id, p, &generic_params);
1218        has_any_default |= has_default;
1219        idx += 1;
1220        result
1221    }));
1222    let diagnostics = create_diagnostics(mem::take(&mut ctx.diagnostics));
1223    let defaults = if has_any_default {
1224        GenericDefaults(Some(Arc::from_iter(defaults)))
1225    } else {
1226        GenericDefaults(None)
1227    };
1228    return (defaults, diagnostics);
1229
1230    fn handle_generic_param(
1231        ctx: &mut TyLoweringContext<'_>,
1232        idx: usize,
1233        id: GenericParamId,
1234        p: GenericParamDataRef<'_>,
1235        generic_params: &Generics,
1236    ) -> (Binders<crate::GenericArg>, bool) {
1237        let binders = variable_kinds_from_iter(ctx.db, generic_params.iter_id().take(idx));
1238        match p {
1239            GenericParamDataRef::TypeParamData(p) => {
1240                let ty = p.default.as_ref().map_or_else(
1241                    || TyKind::Error.intern(Interner),
1242                    |ty| {
1243                        // Each default can only refer to previous parameters.
1244                        // Type variable default referring to parameter coming
1245                        // after it is forbidden (FIXME: report diagnostic)
1246                        fallback_bound_vars(ctx.lower_ty(*ty), idx)
1247                    },
1248                );
1249                (Binders::new(binders, ty.cast(Interner)), p.default.is_some())
1250            }
1251            GenericParamDataRef::ConstParamData(p) => {
1252                let GenericParamId::ConstParamId(id) = id else {
1253                    unreachable!("Unexpected lifetime or type argument")
1254                };
1255
1256                let mut val = p.default.as_ref().map_or_else(
1257                    || unknown_const_as_generic(ctx.db.const_param_ty(id)),
1258                    |c| {
1259                        let param_ty = ctx.lower_ty(p.ty);
1260                        let c = ctx.lower_const(c, param_ty);
1261                        c.cast(Interner)
1262                    },
1263                );
1264                // Each default can only refer to previous parameters, see above.
1265                val = fallback_bound_vars(val, idx);
1266                (Binders::new(binders, val), p.default.is_some())
1267            }
1268            GenericParamDataRef::LifetimeParamData(_) => {
1269                (Binders::new(binders, error_lifetime().cast(Interner)), false)
1270            }
1271        }
1272    }
1273}

1409/// Replaces any 'free' `BoundVar`s in `s` by `TyKind::Error` from the perspective of generic
1410/// parameter whose index is `param_index`. A `BoundVar` is free when it appears after the
1411/// generic parameter of `param_index`.
1412fn fallback_bound_vars<T: TypeFoldable<Interner> + HasInterner<Interner = Interner>>(
1413    s: T,
1414    param_index: usize,
1415) -> T {
1416    let is_allowed = |index| (0..param_index).contains(&index);
1417
1418    crate::fold_free_vars(
1419        s,
1420        |bound, binders| {
1421            if bound.index_if_innermost().is_none_or(is_allowed) {
1422                bound.shifted_in_from(binders).to_ty(Interner)
1423            } else {
1424                TyKind::Error.intern(Interner)
1425            }
1426        },
1427        |ty, bound, binders| {
1428            if bound.index_if_innermost().is_none_or(is_allowed) {
1429                bound.shifted_in_from(binders).to_const(Interner, ty)
1430            } else {
1431                unknown_const(ty)
1432            }
1433        },
1434    )
