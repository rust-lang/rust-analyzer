57pub(crate) fn associated_ty_data_query(
58    db: &dyn HirDatabase,
59    type_alias: TypeAliasId,
60) -> Arc<AssociatedTyDatum> {
61    debug!("associated_ty_data {:?}", type_alias);
62    let trait_ = match type_alias.lookup(db).container {
63        ItemContainerId::TraitId(t) => t,
64        _ => panic!("associated type not in trait"),
65    };
66
67    // Lower bounds -- we could/should maybe move this to a separate query in `lower`
68    let type_alias_data = db.type_alias_signature(type_alias);
69    let generic_params = generics(db, type_alias.into());
70    let resolver = hir_def::resolver::HasResolver::resolver(type_alias, db);
71    let mut ctx = crate::TyLoweringContext::new(
72        db,
73        &resolver,
74        &type_alias_data.store,
75        type_alias.into(),
76        LifetimeElisionKind::AnonymousReportError,
77    )
78    .with_type_param_mode(crate::lower::ParamLoweringMode::Variable);
79
80    let trait_subst = TyBuilder::subst_for_def(db, trait_, None)
81        .fill_with_bound_vars(crate::DebruijnIndex::INNERMOST, 0)
82        .build();
83    let pro_ty = TyBuilder::assoc_type_projection(db, type_alias, Some(trait_subst))
84        .fill_with_bound_vars(
85            crate::DebruijnIndex::INNERMOST,
86            generic_params.parent_generics().map_or(0, |it| it.len()),
87        )
88        .build();
89    let self_ty = TyKind::Alias(AliasTy::Projection(pro_ty)).intern(Interner);
90
91    let mut bounds = Vec::new();
92    for bound in &type_alias_data.bounds {
93        ctx.lower_type_bound(bound, self_ty.clone(), false).for_each(|pred| {
94            if let Some(pred) = generic_predicate_to_inline_bound(db, &pred, &self_ty) {
95                bounds.push(pred);
96            }
97        });
98    }
99
100    if !ctx.unsized_types.contains(&self_ty) {
101        let sized_trait =
102            LangItem::Sized.resolve_trait(db, resolver.krate()).map(to_chalk_trait_id);
103        let sized_bound = sized_trait.into_iter().map(|sized_trait| {
104            let trait_bound =
105                rust_ir::TraitBound { trait_id: sized_trait, args_no_self: Default::default() };
106            let inline_bound = rust_ir::InlineBound::TraitBound(trait_bound);
107            chalk_ir::Binders::empty(Interner, inline_bound)
108        });
109        bounds.extend(sized_bound);
110        bounds.shrink_to_fit();
111    }
112
113    // FIXME: Re-enable where clauses on associated types when an upstream chalk bug is fixed.
114    //        (rust-analyzer#9052)
115    // let where_clauses = convert_where_clauses(db, type_alias.into(), &bound_vars);
116    let bound_data = rust_ir::AssociatedTyDatumBound { bounds, where_clauses: vec![] };
117    let datum = AssociatedTyDatum {
118        trait_id: to_chalk_trait_id(trait_),
119        id: to_assoc_type_id(type_alias),
120        name: type_alias,
121        binders: make_binders(db, &generic_params, bound_data),
122    };
123    Arc::new(datum)
124}

126pub(crate) fn trait_datum_query(
127    db: &dyn HirDatabase,
128    krate: Crate,
129    trait_id: TraitId,
130) -> Arc<TraitDatum> {
131    debug!("trait_datum {:?}", trait_id);
132    let trait_ = from_chalk_trait_id(trait_id);
133    let trait_data = db.trait_signature(trait_);
134    debug!("trait {:?} = {:?}", trait_id, trait_data.name);
135    let generic_params = generics(db, trait_.into());
136    let bound_vars = generic_params.bound_vars_subst(db, DebruijnIndex::INNERMOST);
137    let flags = rust_ir::TraitFlags {
138        auto: trait_data.flags.contains(TraitFlags::AUTO),
139        upstream: trait_.lookup(db).container.krate() != krate,
140        non_enumerable: true,
141        coinductive: false, // only relevant for Chalk testing
142        // FIXME: set these flags correctly
143        marker: false,
144        fundamental: trait_data.flags.contains(TraitFlags::FUNDAMENTAL),
145    };
146    let where_clauses = convert_where_clauses(db, trait_.into(), &bound_vars);
147    let associated_ty_ids =
148        trait_.trait_items(db).associated_types().map(to_assoc_type_id).collect();
149    let trait_datum_bound = rust_ir::TraitDatumBound { where_clauses };
150    let well_known = db.lang_attr(trait_.into()).and_then(well_known_trait_from_lang_item);
151    let trait_datum = TraitDatum {
152        id: trait_id,
153        binders: make_binders(db, &generic_params, trait_datum_bound),
154        flags,
155        associated_ty_ids,
156        well_known,
157    };
158    Arc::new(trait_datum)
159}

161fn well_known_trait_from_lang_item(item: LangItem) -> Option<WellKnownTrait> {
162    Some(match item {
163        LangItem::Clone => WellKnownTrait::Clone,
164        LangItem::CoerceUnsized => WellKnownTrait::CoerceUnsized,
165        LangItem::Copy => WellKnownTrait::Copy,
166        LangItem::DiscriminantKind => WellKnownTrait::DiscriminantKind,
167        LangItem::DispatchFromDyn => WellKnownTrait::DispatchFromDyn,
168        LangItem::Drop => WellKnownTrait::Drop,
169        LangItem::Fn => WellKnownTrait::Fn,
170        LangItem::FnMut => WellKnownTrait::FnMut,
171        LangItem::FnOnce => WellKnownTrait::FnOnce,
172        LangItem::AsyncFn => WellKnownTrait::AsyncFn,
173        LangItem::AsyncFnMut => WellKnownTrait::AsyncFnMut,
174        LangItem::AsyncFnOnce => WellKnownTrait::AsyncFnOnce,
175        LangItem::Coroutine => WellKnownTrait::Coroutine,
176        LangItem::Sized => WellKnownTrait::Sized,
177        LangItem::Unpin => WellKnownTrait::Unpin,
178        LangItem::Unsize => WellKnownTrait::Unsize,
179        LangItem::Tuple => WellKnownTrait::Tuple,
180        LangItem::PointeeTrait => WellKnownTrait::Pointee,
181        LangItem::FnPtrTrait => WellKnownTrait::FnPtr,
182        LangItem::Future => WellKnownTrait::Future,
183        _ => return None,
184    })
185}

187pub(crate) fn adt_datum_query(
188    db: &dyn HirDatabase,
189    krate: Crate,
190    chalk_ir::AdtId(adt_id): AdtId,
191) -> Arc<AdtDatum> {
192    debug!("adt_datum {:?}", adt_id);
193    let generic_params = generics(db, adt_id.into());
194    let bound_vars_subst = generic_params.bound_vars_subst(db, DebruijnIndex::INNERMOST);
195    let where_clauses = convert_where_clauses(db, adt_id.into(), &bound_vars_subst);
196
197    let (fundamental, phantom_data) = match adt_id {
198        hir_def::AdtId::StructId(s) => {
199            let flags = db.struct_signature(s).flags;
200            (flags.contains(StructFlags::FUNDAMENTAL), flags.contains(StructFlags::IS_PHANTOM_DATA))
201        }
202        // FIXME set fundamental flags correctly
203        hir_def::AdtId::UnionId(_) => (false, false),
204        hir_def::AdtId::EnumId(_) => (false, false),
205    };
206    let flags = rust_ir::AdtFlags {
207        upstream: adt_id.module(db).krate() != krate,
208        fundamental,
209        phantom_data,
210    };
211
212    // this slows down rust-analyzer by quite a bit unfortunately, so enabling this is currently not worth it
213    let _variant_id_to_fields = |id: VariantId| {
214        let variant_data = &id.fields(db);
215        let fields = if variant_data.fields().is_empty() {
216            vec![]
217        } else {
218            let field_types = db.field_types(id);
219            variant_data
220                .fields()
221                .iter()
222                .map(|(idx, _)| field_types[idx].clone().substitute(Interner, &bound_vars_subst))
223                .filter(|it| !it.contains_unknown())
224                .collect()
225        };
226        rust_ir::AdtVariantDatum { fields }
227    };
228    let variant_id_to_fields = |_: VariantId| rust_ir::AdtVariantDatum { fields: vec![] };
229
230    let (kind, variants) = match adt_id {
231        hir_def::AdtId::StructId(id) => {
232            (rust_ir::AdtKind::Struct, vec![variant_id_to_fields(id.into())])
233        }
234        hir_def::AdtId::EnumId(id) => {
235            let variants = id
236                .enum_variants(db)
237                .variants
238                .iter()
239                .map(|&(variant_id, _, _)| variant_id_to_fields(variant_id.into()))
240                .collect();
241            (rust_ir::AdtKind::Enum, variants)
242        }
243        hir_def::AdtId::UnionId(id) => {
244            (rust_ir::AdtKind::Union, vec![variant_id_to_fields(id.into())])
245        }
246    };
247
248    let struct_datum_bound = rust_ir::AdtDatumBound { variants, where_clauses };
249    let struct_datum = AdtDatum {
250        kind,
251        id: chalk_ir::AdtId(adt_id),
252        binders: make_binders(db, &generic_params, struct_datum_bound),
253        flags,
254    };
255    Arc::new(struct_datum)
256}

258pub(crate) fn impl_datum_query(
259    db: &dyn HirDatabase,
260    krate: Crate,
261    impl_id: ImplId,

269fn impl_def_datum(db: &dyn HirDatabase, krate: Crate, impl_id: hir_def::ImplId) -> Arc<ImplDatum> {
270    let trait_ref = db
271        .impl_trait(impl_id)
272        // ImplIds for impls where the trait ref can't be resolved should never reach Chalk
273        .expect("invalid impl passed to Chalk")
274        .into_value_and_skipped_binders()
275        .0;
276    let impl_data = db.impl_signature(impl_id);
277
278    let generic_params = generics(db, impl_id.into());
279    let bound_vars = generic_params.bound_vars_subst(db, DebruijnIndex::INNERMOST);
280    let trait_ = trait_ref.hir_trait_id();
281    let impl_type = if impl_id.lookup(db).container.krate() == krate {
282        rust_ir::ImplType::Local
283    } else {
284        rust_ir::ImplType::External
285    };
286    let where_clauses = convert_where_clauses(db, impl_id.into(), &bound_vars);
287    let negative = impl_data.flags.contains(ImplFlags::NEGATIVE);
288    let polarity = if negative { rust_ir::Polarity::Negative } else { rust_ir::Polarity::Positive };
289
290    let impl_datum_bound = rust_ir::ImplDatumBound { trait_ref, where_clauses };
291    let trait_data = trait_.trait_items(db);
292    let associated_ty_value_ids = impl_id
293        .impl_items(db)
294        .items
295        .iter()
296        .filter_map(|(_, item)| match item {
297            AssocItemId::TypeAliasId(type_alias) => Some(*type_alias),
298            _ => None,
299        })
300        .filter(|&type_alias| {
301            // don't include associated types that don't exist in the trait
302            let name = &db.type_alias_signature(type_alias).name;
303            trait_data.associated_type_by_name(name).is_some()
304        })
305        .map(|type_alias| TypeAliasAsValue(type_alias).to_chalk(db))
306        .collect();
307    debug!("impl_datum: {:?}", impl_datum_bound);
308    let impl_datum = ImplDatum {
309        binders: make_binders(db, &generic_params, impl_datum_bound),
310        impl_type,
311        polarity,
312        associated_ty_value_ids,
313    };
314    Arc::new(impl_datum)
315}

326fn type_alias_associated_ty_value(
327    db: &dyn HirDatabase,
328    _krate: Crate,
329    type_alias: TypeAliasId,
330) -> Arc<AssociatedTyValue> {
331    let type_alias_data = db.type_alias_signature(type_alias);
332    let impl_id = match type_alias.lookup(db).container {
333        ItemContainerId::ImplId(it) => it,
334        _ => panic!("assoc ty value should be in impl"),
335    };
336
337    let trait_ref = db
338        .impl_trait(impl_id)
339        .expect("assoc ty value should not exist")
340        .into_value_and_skipped_binders()
341        .0; // we don't return any assoc ty values if the impl'd trait can't be resolved
342
343    let assoc_ty = trait_ref
344        .hir_trait_id()
345        .trait_items(db)
346        .associated_type_by_name(&type_alias_data.name)
347        .expect("assoc ty value should not exist"); // validated when building the impl data as well
348    let (ty, binders) = db.ty(type_alias.into()).into_value_and_skipped_binders();
349    let value_bound = rust_ir::AssociatedTyValueBound { ty };
350    let value = rust_ir::AssociatedTyValue {
351        impl_id: impl_id.to_chalk(db),
352        associated_ty_id: to_assoc_type_id(assoc_ty),
353        value: chalk_ir::Binders::new(binders, value_bound),
354    };
355    Arc::new(value)
356}

358pub(crate) fn fn_def_datum_query(
359    db: &dyn HirDatabase,
360    callable_def: CallableDefId,
361) -> Arc<FnDefDatum> {
362    let generic_def = GenericDefId::from_callable(db, callable_def);
363    let generic_params = generics(db, generic_def);
364    let (sig, binders) = db.callable_item_signature(callable_def).into_value_and_skipped_binders();
365    let bound_vars = generic_params.bound_vars_subst(db, DebruijnIndex::INNERMOST);
366    let where_clauses = convert_where_clauses(db, generic_def, &bound_vars);
367    let bound = rust_ir::FnDefDatumBound {
368        // Note: Chalk doesn't actually use this information yet as far as I am aware, but we provide it anyway
369        inputs_and_output: chalk_ir::Binders::empty(
370            Interner,
371            rust_ir::FnDefInputsAndOutputDatum {
372                argument_types: sig.params().to_vec(),
373                return_type: sig.ret().clone(),
374            }
375            .shifted_in(Interner),
376        ),
377        where_clauses,
378    };
379    let datum = FnDefDatum {
380        id: callable_def.to_chalk(db),
381        sig: chalk_ir::FnSig {
382            abi: sig.abi,
383            safety: chalk_ir::Safety::Safe,
384            variadic: sig.is_varargs,
385        },
386        binders: chalk_ir::Binders::new(binders, bound),
387    };
388    Arc::new(datum)
389}

391pub(crate) fn fn_def_variance_query(
392    db: &dyn HirDatabase,
393    callable_def: CallableDefId,
394) -> Variances {
395    Variances::from_iter(
396        Interner,
397        db.variances_of(GenericDefId::from_callable(db, callable_def))
398            .as_deref()
399            .unwrap_or_default()
400            .iter()
401            .map(|v| match v {
402                crate::variance::Variance::Covariant => chalk_ir::Variance::Covariant,
403                crate::variance::Variance::Invariant => chalk_ir::Variance::Invariant,
404                crate::variance::Variance::Contravariant => chalk_ir::Variance::Contravariant,
405                crate::variance::Variance::Bivariant => chalk_ir::Variance::Invariant,
406            }),
407    )
408}

422/// Returns instantiated predicates.
423pub(super) fn convert_where_clauses(
424    db: &dyn HirDatabase,
425    def: GenericDefId,
426    substs: &Substitution,
427) -> Vec<chalk_ir::QuantifiedWhereClause<Interner>> {
428    db.generic_predicates(def)
429        .iter()
430        .cloned()
431        .map(|pred| pred.substitute(Interner, substs))
432        .collect()
433}

435pub(super) fn generic_predicate_to_inline_bound(
436    db: &dyn HirDatabase,
437    pred: &QuantifiedWhereClause,
438    self_ty: &Ty,
439) -> Option<chalk_ir::Binders<rust_ir::InlineBound<Interner>>> {
440    // An InlineBound is like a GenericPredicate, except the self type is left out.
441    // We don't have a special type for this, but Chalk does.
442    let self_ty_shifted_in = self_ty.clone().shifted_in_from(Interner, DebruijnIndex::ONE);
443    let (pred, binders) = pred.as_ref().into_value_and_skipped_binders();
444    match pred {
445        WhereClause::Implemented(trait_ref) => {
446            if trait_ref.self_type_parameter(Interner) != self_ty_shifted_in {
447                // we can only convert predicates back to type bounds if they
448                // have the expected self type
449                return None;
450            }
451            let args_no_self = trait_ref.substitution.as_slice(Interner)[1..]
452                .iter()
453                .cloned()
454                .casted(Interner)
455                .collect();
456            let trait_bound = rust_ir::TraitBound { trait_id: trait_ref.trait_id, args_no_self };
457            Some(chalk_ir::Binders::new(binders, rust_ir::InlineBound::TraitBound(trait_bound)))
458        }
459        WhereClause::AliasEq(AliasEq { alias: AliasTy::Projection(projection_ty), ty }) => {
460            let generics = generics(db, from_assoc_type_id(projection_ty.associated_ty_id).into());
461            let parent_len = generics.parent_generics().map_or(0, |g| g.len_self());
462            let (trait_args, assoc_args) =
463                projection_ty.substitution.as_slice(Interner).split_at(parent_len);
464            let (self_ty, args_no_self) =
465                trait_args.split_first().expect("projection without trait self type");
466            if self_ty.assert_ty_ref(Interner) != &self_ty_shifted_in {
467                return None;
468            }
469
470            let args_no_self = args_no_self.iter().cloned().casted(Interner).collect();
471            let parameters = assoc_args.to_vec();
472
473            let alias_eq_bound = rust_ir::AliasEqBound {
474                value: ty.clone(),
475                trait_bound: rust_ir::TraitBound {
476                    trait_id: to_chalk_trait_id(projection_ty.trait_(db)),
477                    args_no_self,
478                },
479                associated_ty_id: projection_ty.associated_ty_id,
480                parameters,
481            };
482            Some(chalk_ir::Binders::new(
483                binders,
484                rust_ir::InlineBound::AliasEqBound(alias_eq_bound),
485            ))
486        }
487        _ => None,
488    }
