76#[derive(Debug, PartialEq, Eq, Clone)]
77pub enum LayoutError {
78    // FIXME: Remove more variants once they get added to LayoutCalculatorError
79    BadCalc(LayoutCalculatorError<()>),
80    HasErrorConst,
81    HasErrorType,
82    HasPlaceholder,
83    InvalidSimdType,
84    NotImplemented,
85    RecursiveTypeWithoutIndirection,
86    TargetLayoutNotAvailable,
87    Unknown,
88    UserReprTooSmall,
89}
90
91impl std::error::Error for LayoutError {}
92impl fmt::Display for LayoutError {
93    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
94        match self {
95            LayoutError::BadCalc(err) => err.fallback_fmt(f),
96            LayoutError::HasErrorConst => write!(f, "type contains an unevaluatable const"),
97            LayoutError::HasErrorType => write!(f, "type contains an error"),
98            LayoutError::HasPlaceholder => write!(f, "type contains placeholders"),
99            LayoutError::InvalidSimdType => write!(f, "invalid simd type definition"),
100            LayoutError::NotImplemented => write!(f, "not implemented"),
101            LayoutError::RecursiveTypeWithoutIndirection => {
102                write!(f, "recursive type without indirection")
103            }
104            LayoutError::TargetLayoutNotAvailable => write!(f, "target layout not available"),
105            LayoutError::Unknown => write!(f, "unknown"),
106            LayoutError::UserReprTooSmall => {
107                write!(f, "the `#[repr]` hint is too small to hold the discriminants of the enum")
108            }
109        }
110    }
111}
112
113impl<F> From<LayoutCalculatorError<F>> for LayoutError {
114    fn from(err: LayoutCalculatorError<F>) -> Self {
115        LayoutError::BadCalc(err.without_payload())
116    }
117}

129// FIXME: move this to the `rustc_abi`.
130fn layout_of_simd_ty<'db>(
131    db: &'db dyn HirDatabase,
132    id: StructId,
133    repr_packed: bool,
134    args: &GenericArgs<'db>,
135    env: ParamEnvAndCrate<'db>,
136    dl: &TargetDataLayout,
137) -> Result<Arc<Layout>, LayoutError> {
138    // Supported SIMD vectors are homogeneous ADTs with exactly one array field:
139    //
140    // * #[repr(simd)] struct S([T; 4])
141    //
142    // where T is a primitive scalar (integer/float/pointer).
143    let fields = db.field_types(id.into());
144    let mut fields = fields.iter();
145    let Some(TyKind::Array(e_ty, e_len)) = fields
146        .next()
147        .filter(|_| fields.next().is_none())
148        .map(|f| (*f.1).get().instantiate(DbInterner::new_no_crate(db), args).kind())
149    else {
150        return Err(LayoutError::InvalidSimdType);
151    };
152
153    let e_len = try_const_usize(db, e_len).ok_or(LayoutError::HasErrorConst)? as u64;
154    let e_ly = db.layout_of_ty(e_ty.store(), env.store())?;
155
156    let cx = LayoutCx::new(dl);
157    Ok(Arc::new(cx.calc.simd_type(e_ly, e_len, repr_packed)?))
158}

160pub fn layout_of_ty_query(
161    db: &dyn HirDatabase,
162    ty: StoredTy,
163    trait_env: StoredParamEnvAndCrate,
164) -> Result<Arc<Layout>, LayoutError> {
165    let krate = trait_env.krate;
166    let interner = DbInterner::new_with(db, krate);
167    let Ok(target) = db.target_data_layout(krate) else {
168        return Err(LayoutError::TargetLayoutNotAvailable);
169    };
170    let dl = &*target;
171    let cx = LayoutCx::new(dl);
172    let infer_ctxt = interner.infer_ctxt().build(TypingMode::PostAnalysis);
173    let cause = ObligationCause::dummy();
174    let ty = infer_ctxt
175        .at(&cause, trait_env.param_env())
176        .deeply_normalize(ty.as_ref())
177        .unwrap_or(ty.as_ref());
178    let result = match ty.kind() {
179        TyKind::Adt(def, args) => {
180            match def.inner().id {
181                hir_def::AdtId::StructId(s) => {
182                    let repr = AttrFlags::repr(db, s.into()).unwrap_or_default();
183                    if repr.simd() {
184                        return layout_of_simd_ty(
185                            db,
186                            s,
187                            repr.packed(),
188                            &args,
189                            trait_env.as_ref(),
190                            &target,
191                        );
192                    }
193                }
194                _ => {}
195            }
196            return db.layout_of_adt(def.inner().id, args.store(), trait_env);
197        }
198        TyKind::Bool => Layout::scalar(
199            dl,
200            Scalar::Initialized {
201                value: Primitive::Int(Integer::I8, false),
202                valid_range: WrappingRange { start: 0, end: 1 },
203            },
204        ),
205        TyKind::Char => Layout::scalar(
206            dl,
207            Scalar::Initialized {
208                value: Primitive::Int(Integer::I32, false),
209                valid_range: WrappingRange { start: 0, end: 0x10FFFF },
210            },
211        ),
212        TyKind::Int(i) => Layout::scalar(
213            dl,
214            scalar_unit(
215                dl,
216                Primitive::Int(
217                    match i {
218                        IntTy::Isize => dl.ptr_sized_integer(),
219                        IntTy::I8 => Integer::I8,
220                        IntTy::I16 => Integer::I16,
221                        IntTy::I32 => Integer::I32,
222                        IntTy::I64 => Integer::I64,
223                        IntTy::I128 => Integer::I128,
224                    },
225                    true,
226                ),
227            ),
228        ),
229        TyKind::Uint(i) => Layout::scalar(
230            dl,
231            scalar_unit(
232                dl,
233                Primitive::Int(
234                    match i {
235                        UintTy::Usize => dl.ptr_sized_integer(),
236                        UintTy::U8 => Integer::I8,
237                        UintTy::U16 => Integer::I16,
238                        UintTy::U32 => Integer::I32,
239                        UintTy::U64 => Integer::I64,
240                        UintTy::U128 => Integer::I128,
241                    },
242                    false,
243                ),
244            ),
245        ),
246        TyKind::Float(f) => Layout::scalar(
247            dl,
248            scalar_unit(
249                dl,
250                Primitive::Float(match f {
251                    FloatTy::F16 => Float::F16,
252                    FloatTy::F32 => Float::F32,
253                    FloatTy::F64 => Float::F64,
254                    FloatTy::F128 => Float::F128,
255                }),
256            ),
257        ),
258        TyKind::Tuple(tys) => {
259            let kind =
260                if tys.is_empty() { StructKind::AlwaysSized } else { StructKind::MaybeUnsized };
261
262            let fields = tys
263                .iter()
264                .map(|k| db.layout_of_ty(k.store(), trait_env.clone()))
265                .collect::<Result<Vec<_>, _>>()?;
266            let fields = fields.iter().map(|it| &**it).collect::<Vec<_>>();
267            let fields = fields.iter().collect::<IndexVec<_, _>>();
268            cx.calc.univariant(&fields, &ReprOptions::default(), kind)?
269        }
270        TyKind::Array(element, count) => {
271            let count = try_const_usize(db, count).ok_or(LayoutError::HasErrorConst)? as u64;
272            let element = db.layout_of_ty(element.store(), trait_env)?;
273            cx.calc.array_like::<_, _, ()>(&element, Some(count))?
274        }
275        TyKind::Slice(element) => {
276            let element = db.layout_of_ty(element.store(), trait_env)?;
277            cx.calc.array_like::<_, _, ()>(&element, None)?
278        }
279        TyKind::Str => {
280            let element = scalar_unit(dl, Primitive::Int(Integer::I8, false));
281            cx.calc.array_like::<_, _, ()>(&Layout::scalar(dl, element), None)?
282        }
283        // Potentially-wide pointers.
284        TyKind::Ref(_, pointee, _) | TyKind::RawPtr(pointee, _) => {
285            let mut data_ptr = scalar_unit(dl, Primitive::Pointer(AddressSpace::ZERO));
286            if matches!(ty.kind(), TyKind::Ref(..)) {
287                data_ptr.valid_range_mut().start = 1;
288            }
289
290            // FIXME(next-solver)
291            // let pointee = tcx.normalize_erasing_regions(param_env, pointee);
292            // if pointee.is_sized(tcx.at(DUMMY_SP), param_env) {
293            //     return Ok(tcx.mk_layout(LayoutS::scalar(cx, data_ptr)));
294            // }
295
296            let unsized_part = struct_tail_erasing_lifetimes(db, pointee);
297            // FIXME(next-solver)
298            /*
299            if let TyKind::AssociatedType(id, subst) = unsized_part.kind(Interner) {
300                unsized_part = TyKind::Alias(chalk_ir::AliasTy::Projection(ProjectionTy {
301                    associated_ty_id: *id,
302                    substitution: subst.clone(),
303                }))
304                .intern(Interner);
305            }
306            unsized_part = normalize(db, trait_env, unsized_part);
307            */
308            let metadata = match unsized_part.kind() {
309                TyKind::Slice(_) | TyKind::Str => {
310                    scalar_unit(dl, Primitive::Int(dl.ptr_sized_integer(), false))
311                }
312                TyKind::Dynamic(..) => {
313                    let mut vtable = scalar_unit(dl, Primitive::Pointer(AddressSpace::ZERO));
314                    vtable.valid_range_mut().start = 1;
315                    vtable
316                }
317                _ => {
318                    // pointee is sized
319                    return Ok(Arc::new(Layout::scalar(dl, data_ptr)));
320                }
321            };
322
323            // Effectively a (ptr, meta) tuple.
324            LayoutData::scalar_pair(dl, data_ptr, metadata)
325        }
326        TyKind::Never => LayoutData::never_type(dl),
327        TyKind::FnDef(..) => LayoutData::unit(dl, true),
328        TyKind::Dynamic(..) | TyKind::Foreign(_) => LayoutData::unit(dl, false),
329        TyKind::FnPtr(..) => {
330            let mut ptr = scalar_unit(dl, Primitive::Pointer(dl.instruction_address_space));
331            ptr.valid_range_mut().start = 1;
332            Layout::scalar(dl, ptr)
333        }
334        TyKind::Closure(id, args) => {
335            let def = db.lookup_intern_closure(id.0);
336            let infer = InferenceResult::for_body(db, def.0);
337            let (captures, _) = infer.closure_info(id.0);
338            let fields = captures
339                .iter()
340                .map(|it| {
341                    let ty = it.ty.get().instantiate(interner, args.as_closure().parent_args());
342                    db.layout_of_ty(ty.store(), trait_env.clone())
343                })
344                .collect::<Result<Vec<_>, _>>()?;
345            let fields = fields.iter().map(|it| &**it).collect::<Vec<_>>();
346            let fields = fields.iter().collect::<IndexVec<_, _>>();
347            cx.calc.univariant(&fields, &ReprOptions::default(), StructKind::AlwaysSized)?
348        }
349
350        TyKind::Coroutine(_, _)
351        | TyKind::CoroutineWitness(_, _)
352        | TyKind::CoroutineClosure(_, _) => {
353            return Err(LayoutError::NotImplemented);
354        }
355
356        TyKind::Pat(_, _) | TyKind::UnsafeBinder(_) => {
357            return Err(LayoutError::NotImplemented);
358        }
359
360        TyKind::Error(_) => return Err(LayoutError::HasErrorType),
361        TyKind::Placeholder(_)
362        | TyKind::Bound(..)
363        | TyKind::Infer(..)
364        | TyKind::Param(..)
365        | TyKind::Alias(..) => {
366            return Err(LayoutError::HasPlaceholder);
367        }
368    };
369    Ok(Arc::new(result))
370}

381fn struct_tail_erasing_lifetimes<'a>(db: &'a dyn HirDatabase, pointee: Ty<'a>) -> Ty<'a> {
382    match pointee.kind() {
383        TyKind::Adt(def, args) => {
384            let struct_id = match def.inner().id {
385                AdtId::StructId(id) => id,
386                _ => return pointee,
387            };
388            let data = struct_id.fields(db);
389            let mut it = data.fields().iter().rev();
390            match it.next() {
391                Some((f, _)) => {
392                    let last_field_ty = field_ty(db, struct_id.into(), f, args);
393                    struct_tail_erasing_lifetimes(db, last_field_ty)
394                }
395                None => pointee,
396            }
397        }
398        TyKind::Tuple(tys) => {
399            if let Some(last_field_ty) = tys.iter().next_back() {
400                struct_tail_erasing_lifetimes(db, last_field_ty)
401            } else {
402                pointee
403            }
404        }
405        _ => pointee,
406    }
407}

409fn field_ty<'a>(
410    db: &'a dyn HirDatabase,
411    def: hir_def::VariantId,
412    fd: LocalFieldId,
413    args: GenericArgs<'a>,
414) -> Ty<'a> {
415    db.field_types(def)[fd].get().instantiate(DbInterner::new_no_crate(db), args)
416}

418fn scalar_unit(dl: &TargetDataLayout, value: Primitive) -> Scalar {
419    Scalar::Initialized { value, valid_range: WrappingRange::full(value.size(dl)) }
420}
