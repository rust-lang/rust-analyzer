70#[derive(Debug, PartialEq, Eq, Clone)]
71pub enum LayoutError {
72    // FIXME: Remove more variants once they get added to LayoutCalculatorError
73    BadCalc(LayoutCalculatorError<()>),
74    HasErrorConst,
75    HasErrorType,
76    HasPlaceholder,
77    InvalidSimdType,
78    NotImplemented,
79    RecursiveTypeWithoutIndirection,
80    TargetLayoutNotAvailable,
81    Unknown,
82    UserReprTooSmall,
83}
84
85impl std::error::Error for LayoutError {}
86impl fmt::Display for LayoutError {
87    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
88        match self {
89            LayoutError::BadCalc(err) => err.fallback_fmt(f),
90            LayoutError::HasErrorConst => write!(f, "type contains an unevaluatable const"),
91            LayoutError::HasErrorType => write!(f, "type contains an error"),
92            LayoutError::HasPlaceholder => write!(f, "type contains placeholders"),
93            LayoutError::InvalidSimdType => write!(f, "invalid simd type definition"),
94            LayoutError::NotImplemented => write!(f, "not implemented"),
95            LayoutError::RecursiveTypeWithoutIndirection => {
96                write!(f, "recursive type without indirection")
97            }
98            LayoutError::TargetLayoutNotAvailable => write!(f, "target layout not available"),
99            LayoutError::Unknown => write!(f, "unknown"),
100            LayoutError::UserReprTooSmall => {
101                write!(f, "the `#[repr]` hint is too small to hold the discriminants of the enum")
102            }
103        }
104    }
105}

107impl<F> From<LayoutCalculatorError<F>> for LayoutError {
108    fn from(err: LayoutCalculatorError<F>) -> Self {
109        LayoutError::BadCalc(err.without_payload())
110    }

123fn layout_of_simd_ty(
124    db: &dyn HirDatabase,
125    id: StructId,
126    repr_packed: bool,
127    subst: &Substitution,
128    env: Arc<TraitEnvironment>,
129    dl: &TargetDataLayout,
130) -> Result<Arc<Layout>, LayoutError> {
131    // Supported SIMD vectors are homogeneous ADTs with exactly one array field:
132    //
133    // * #[repr(simd)] struct S([T; 4])
134    //
135    // where T is a primitive scalar (integer/float/pointer).
136    let fields = db.field_types(id.into());
137    let mut fields = fields.iter();
138    let Some(TyKind::Array(e_ty, e_len)) = fields
139        .next()
140        .filter(|_| fields.next().is_none())
141        .map(|f| f.1.clone().substitute(Interner, subst).kind(Interner).clone())
142    else {
143        return Err(LayoutError::InvalidSimdType);
144    };
145
146    let e_len = try_const_usize(db, &e_len).ok_or(LayoutError::HasErrorConst)? as u64;
147    let e_ly = db.layout_of_ty(e_ty, env)?;
148
149    let cx = LayoutCx::new(dl);
150    Ok(Arc::new(cx.calc.simd_type(e_ly, e_len, repr_packed)?))
151}

153pub fn layout_of_ty_query(
154    db: &dyn HirDatabase,
155    ty: Ty,
156    trait_env: Arc<TraitEnvironment>,
157) -> Result<Arc<Layout>, LayoutError> {
158    let krate = trait_env.krate;
159    let Ok(target) = db.target_data_layout(krate) else {
160        return Err(LayoutError::TargetLayoutNotAvailable);
161    };
162    let dl = &*target;
163    let cx = LayoutCx::new(dl);
164    let ty = normalize(db, trait_env.clone(), ty);
165    let kind = ty.kind(Interner);
166    let result = match kind {
167        TyKind::Adt(AdtId(def), subst) => {
168            if let hir_def::AdtId::StructId(s) = def {
169                let data = db.struct_signature(*s);
170                let repr = data.repr.unwrap_or_default();
171                if repr.simd() {
172                    return layout_of_simd_ty(db, *s, repr.packed(), subst, trait_env, &target);
173                }
174            };
175            return db.layout_of_adt(*def, subst.clone(), trait_env);
176        }
177        TyKind::Scalar(s) => match s {
178            chalk_ir::Scalar::Bool => Layout::scalar(
179                dl,
180                Scalar::Initialized {
181                    value: Primitive::Int(Integer::I8, false),
182                    valid_range: WrappingRange { start: 0, end: 1 },
183                },
184            ),
185            chalk_ir::Scalar::Char => Layout::scalar(
186                dl,
187                Scalar::Initialized {
188                    value: Primitive::Int(Integer::I32, false),
189                    valid_range: WrappingRange { start: 0, end: 0x10FFFF },
190                },
191            ),
192            chalk_ir::Scalar::Int(i) => Layout::scalar(
193                dl,
194                scalar_unit(
195                    dl,
196                    Primitive::Int(
197                        match i {
198                            IntTy::Isize => dl.ptr_sized_integer(),
199                            IntTy::I8 => Integer::I8,
200                            IntTy::I16 => Integer::I16,
201                            IntTy::I32 => Integer::I32,
202                            IntTy::I64 => Integer::I64,
203                            IntTy::I128 => Integer::I128,
204                        },
205                        true,
206                    ),
207                ),
208            ),
209            chalk_ir::Scalar::Uint(i) => Layout::scalar(
210                dl,
211                scalar_unit(
212                    dl,
213                    Primitive::Int(
214                        match i {
215                            UintTy::Usize => dl.ptr_sized_integer(),
216                            UintTy::U8 => Integer::I8,
217                            UintTy::U16 => Integer::I16,
218                            UintTy::U32 => Integer::I32,
219                            UintTy::U64 => Integer::I64,
220                            UintTy::U128 => Integer::I128,
221                        },
222                        false,
223                    ),
224                ),
225            ),
226            chalk_ir::Scalar::Float(f) => Layout::scalar(
227                dl,
228                scalar_unit(
229                    dl,
230                    Primitive::Float(match f {
231                        FloatTy::F16 => Float::F16,
232                        FloatTy::F32 => Float::F32,
233                        FloatTy::F64 => Float::F64,
234                        FloatTy::F128 => Float::F128,
235                    }),
236                ),
237            ),
238        },
239        TyKind::Tuple(len, tys) => {
240            let kind = if *len == 0 { StructKind::AlwaysSized } else { StructKind::MaybeUnsized };
241
242            let fields = tys
243                .iter(Interner)
244                .map(|k| db.layout_of_ty(k.assert_ty_ref(Interner).clone(), trait_env.clone()))
245                .collect::<Result<Vec<_>, _>>()?;
246            let fields = fields.iter().map(|it| &**it).collect::<Vec<_>>();
247            let fields = fields.iter().collect::<IndexVec<_, _>>();
248            cx.calc.univariant(&fields, &ReprOptions::default(), kind)?
249        }
250        TyKind::Array(element, count) => {
251            let count = try_const_usize(db, count).ok_or(LayoutError::HasErrorConst)? as u64;
252            let element = db.layout_of_ty(element.clone(), trait_env)?;
253            cx.calc.array_like::<_, _, ()>(&element, Some(count))?
254        }
255        TyKind::Slice(element) => {
256            let element = db.layout_of_ty(element.clone(), trait_env)?;
257            cx.calc.array_like::<_, _, ()>(&element, None)?
258        }
259        TyKind::Str => {
260            let element = scalar_unit(dl, Primitive::Int(Integer::I8, false));
261            cx.calc.array_like::<_, _, ()>(&Layout::scalar(dl, element), None)?
262        }
263        // Potentially-wide pointers.
264        TyKind::Ref(_, _, pointee) | TyKind::Raw(_, pointee) => {
265            let mut data_ptr = scalar_unit(dl, Primitive::Pointer(AddressSpace::DATA));
266            if matches!(ty.kind(Interner), TyKind::Ref(..)) {
267                data_ptr.valid_range_mut().start = 1;
268            }
269
270            // let pointee = tcx.normalize_erasing_regions(param_env, pointee);
271            // if pointee.is_sized(tcx.at(DUMMY_SP), param_env) {
272            //     return Ok(tcx.mk_layout(LayoutS::scalar(cx, data_ptr)));
273            // }
274
275            let mut unsized_part = struct_tail_erasing_lifetimes(db, pointee.clone());
276            if let TyKind::AssociatedType(id, subst) = unsized_part.kind(Interner) {
277                unsized_part = TyKind::Alias(chalk_ir::AliasTy::Projection(ProjectionTy {
278                    associated_ty_id: *id,
279                    substitution: subst.clone(),
280                }))
281                .intern(Interner);
282            }
283            unsized_part = normalize(db, trait_env, unsized_part);
284            let metadata = match unsized_part.kind(Interner) {
285                TyKind::Slice(_) | TyKind::Str => {
286                    scalar_unit(dl, Primitive::Int(dl.ptr_sized_integer(), false))
287                }
288                TyKind::Dyn(..) => {
289                    let mut vtable = scalar_unit(dl, Primitive::Pointer(AddressSpace::DATA));
290                    vtable.valid_range_mut().start = 1;
291                    vtable
292                }
293                _ => {
294                    // pointee is sized
295                    return Ok(Arc::new(Layout::scalar(dl, data_ptr)));
296                }
297            };
298
299            // Effectively a (ptr, meta) tuple.
300            LayoutData::scalar_pair(dl, data_ptr, metadata)
301        }
302        TyKind::Never => LayoutData::never_type(dl),
303        TyKind::FnDef(..) | TyKind::Dyn(_) | TyKind::Foreign(_) => {
304            let sized = matches!(kind, TyKind::FnDef(..));
305            LayoutData::unit(dl, sized)
306        }
307        TyKind::Function(_) => {
308            let mut ptr = scalar_unit(dl, Primitive::Pointer(dl.instruction_address_space));
309            ptr.valid_range_mut().start = 1;
310            Layout::scalar(dl, ptr)
311        }
312        TyKind::OpaqueType(opaque_ty_id, _) => {
313            let impl_trait_id = db.lookup_intern_impl_trait_id((*opaque_ty_id).into());
314            match impl_trait_id {
315                crate::ImplTraitId::ReturnTypeImplTrait(func, idx) => {
316                    let infer = db.infer(func.into());
317                    return db.layout_of_ty(infer.type_of_rpit[idx].clone(), trait_env);
318                }
319                crate::ImplTraitId::TypeAliasImplTrait(..) => {
320                    return Err(LayoutError::NotImplemented);
321                }
322                crate::ImplTraitId::AsyncBlockTypeImplTrait(_, _) => {
323                    return Err(LayoutError::NotImplemented);
324                }
325            }
326        }
327        TyKind::Closure(c, subst) => {
328            let InternedClosure(def, _) = db.lookup_intern_closure((*c).into());
329            let infer = db.infer(def);
330            let (captures, _) = infer.closure_info(c);
331            let fields = captures
332                .iter()
333                .map(|it| {
334                    db.layout_of_ty(
335                        it.ty.clone().substitute(Interner, ClosureSubst(subst).parent_subst()),
336                        trait_env.clone(),
337                    )
338                })
339                .collect::<Result<Vec<_>, _>>()?;
340            let fields = fields.iter().map(|it| &**it).collect::<Vec<_>>();
341            let fields = fields.iter().collect::<IndexVec<_, _>>();
342            cx.calc.univariant(&fields, &ReprOptions::default(), StructKind::AlwaysSized)?
343        }
344        TyKind::Coroutine(_, _) | TyKind::CoroutineWitness(_, _) => {
345            return Err(LayoutError::NotImplemented);
346        }
347        TyKind::Error => return Err(LayoutError::HasErrorType),
348        TyKind::AssociatedType(id, subst) => {
349            // Try again with `TyKind::Alias` to normalize the associated type.
350            // Usually we should not try to normalize `TyKind::AssociatedType`, but layout calculation is used
351            // in monomorphized MIR where this is okay. If outside monomorphization, this will lead to cycle,
352            // which we will recover from with an error.
353            let ty = TyKind::Alias(chalk_ir::AliasTy::Projection(ProjectionTy {
354                associated_ty_id: *id,
355                substitution: subst.clone(),
356            }))
357            .intern(Interner);
358            return db.layout_of_ty(ty, trait_env);
359        }
360        TyKind::Alias(_)
361        | TyKind::Placeholder(_)
362        | TyKind::BoundVar(_)
363        | TyKind::InferenceVar(_, _) => return Err(LayoutError::HasPlaceholder),
364    };
365    Ok(Arc::new(result))
366}

378fn struct_tail_erasing_lifetimes(db: &dyn HirDatabase, pointee: Ty) -> Ty {
379    match pointee.kind(Interner) {
380        &TyKind::Adt(AdtId(hir_def::AdtId::StructId(i)), ref subst) => {
381            let data = db.variant_fields(i.into());
382            let mut it = data.fields().iter().rev();
383            match it.next() {
384                Some((f, _)) => {
385                    let last_field_ty = field_ty(db, i.into(), f, subst);
386                    struct_tail_erasing_lifetimes(db, last_field_ty)
387                }
388                None => pointee,
389            }
390        }
391        TyKind::Tuple(_, subst) => {
392            if let Some(last_field_ty) =
393                subst.iter(Interner).last().and_then(|arg| arg.ty(Interner))
394            {
395                struct_tail_erasing_lifetimes(db, last_field_ty.clone())
396            } else {
397                pointee
398            }
399        }
400        _ => pointee,
401    }
402}

404fn field_ty(
405    db: &dyn HirDatabase,
406    def: hir_def::VariantId,
407    fd: LocalFieldId,
408    subst: &Substitution,

413fn scalar_unit(dl: &TargetDataLayout, value: Primitive) -> Scalar {
414    Scalar::Initialized { value, valid_range: WrappingRange::full(value.size(dl)) }
