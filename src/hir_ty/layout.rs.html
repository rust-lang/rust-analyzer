74#[derive(Debug, PartialEq, Eq, Clone)]
75pub enum LayoutError {
76    // FIXME: Remove more variants once they get added to LayoutCalculatorError
77    BadCalc(LayoutCalculatorError<()>),
78    HasErrorConst,
79    HasErrorType,
80    HasPlaceholder,
81    InvalidSimdType,
82    NotImplemented,
83    RecursiveTypeWithoutIndirection,
84    TargetLayoutNotAvailable,
85    Unknown,
86    UserReprTooSmall,
87}
88
89impl std::error::Error for LayoutError {}
90impl fmt::Display for LayoutError {
91    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
92        match self {
93            LayoutError::BadCalc(err) => err.fallback_fmt(f),
94            LayoutError::HasErrorConst => write!(f, "type contains an unevaluatable const"),
95            LayoutError::HasErrorType => write!(f, "type contains an error"),
96            LayoutError::HasPlaceholder => write!(f, "type contains placeholders"),
97            LayoutError::InvalidSimdType => write!(f, "invalid simd type definition"),
98            LayoutError::NotImplemented => write!(f, "not implemented"),
99            LayoutError::RecursiveTypeWithoutIndirection => {
100                write!(f, "recursive type without indirection")
101            }
102            LayoutError::TargetLayoutNotAvailable => write!(f, "target layout not available"),
103            LayoutError::Unknown => write!(f, "unknown"),
104            LayoutError::UserReprTooSmall => {
105                write!(f, "the `#[repr]` hint is too small to hold the discriminants of the enum")
106            }
107        }
108    }
109}
110
111impl<F> From<LayoutCalculatorError<F>> for LayoutError {
112    fn from(err: LayoutCalculatorError<F>) -> Self {
113        LayoutError::BadCalc(err.without_payload())
114    }
115}

127// FIXME: move this to the `rustc_abi`.
128fn layout_of_simd_ty<'db>(
129    db: &'db dyn HirDatabase,
130    id: StructId,
131    repr_packed: bool,
132    args: &GenericArgs<'db>,
133    env: Arc<TraitEnvironment<'db>>,
134    dl: &TargetDataLayout,
135) -> Result<Arc<Layout>, LayoutError> {
136    // Supported SIMD vectors are homogeneous ADTs with exactly one array field:
137    //
138    // * #[repr(simd)] struct S([T; 4])
139    //
140    // where T is a primitive scalar (integer/float/pointer).
141    let fields = db.field_types(id.into());
142    let mut fields = fields.iter();
143    let Some(TyKind::Array(e_ty, e_len)) = fields
144        .next()
145        .filter(|_| fields.next().is_none())
146        .map(|f| (*f.1).instantiate(DbInterner::new_with(db, None, None), args).kind())
147    else {
148        return Err(LayoutError::InvalidSimdType);
149    };
150
151    let e_len = try_const_usize(db, e_len).ok_or(LayoutError::HasErrorConst)? as u64;
152    let e_ly = db.layout_of_ty(e_ty, env)?;
153
154    let cx = LayoutCx::new(dl);
155    Ok(Arc::new(cx.calc.simd_type(e_ly, e_len, repr_packed)?))
156}

158pub fn layout_of_ty_query<'db>(
159    db: &'db dyn HirDatabase,
160    ty: Ty<'db>,
161    trait_env: Arc<TraitEnvironment<'db>>,
162) -> Result<Arc<Layout>, LayoutError> {
163    let krate = trait_env.krate;
164    let interner = DbInterner::new_with(db, Some(krate), trait_env.block);
165    let Ok(target) = db.target_data_layout(krate) else {
166        return Err(LayoutError::TargetLayoutNotAvailable);
167    };
168    let dl = &*target;
169    let cx = LayoutCx::new(dl);
170    let infer_ctxt = interner.infer_ctxt().build(TypingMode::PostAnalysis);
171    let cause = ObligationCause::dummy();
172    let ty = infer_ctxt.at(&cause, ParamEnv::empty()).deeply_normalize(ty).unwrap_or(ty);
173    let result = match ty.kind() {
174        TyKind::Adt(def, args) => {
175            match def.inner().id {
176                hir_def::AdtId::StructId(s) => {
177                    let data = db.struct_signature(s);
178                    let repr = data.repr.unwrap_or_default();
179                    if repr.simd() {
180                        return layout_of_simd_ty(db, s, repr.packed(), &args, trait_env, &target);
181                    }
182                }
183                _ => {}
184            }
185            return db.layout_of_adt(def.inner().id, args, trait_env);
186        }
187        TyKind::Bool => Layout::scalar(
188            dl,
189            Scalar::Initialized {
190                value: Primitive::Int(Integer::I8, false),
191                valid_range: WrappingRange { start: 0, end: 1 },
192            },
193        ),
194        TyKind::Char => Layout::scalar(
195            dl,
196            Scalar::Initialized {
197                value: Primitive::Int(Integer::I32, false),
198                valid_range: WrappingRange { start: 0, end: 0x10FFFF },
199            },
200        ),
201        TyKind::Int(i) => Layout::scalar(
202            dl,
203            scalar_unit(
204                dl,
205                Primitive::Int(
206                    match i {
207                        IntTy::Isize => dl.ptr_sized_integer(),
208                        IntTy::I8 => Integer::I8,
209                        IntTy::I16 => Integer::I16,
210                        IntTy::I32 => Integer::I32,
211                        IntTy::I64 => Integer::I64,
212                        IntTy::I128 => Integer::I128,
213                    },
214                    true,
215                ),
216            ),
217        ),
218        TyKind::Uint(i) => Layout::scalar(
219            dl,
220            scalar_unit(
221                dl,
222                Primitive::Int(
223                    match i {
224                        UintTy::Usize => dl.ptr_sized_integer(),
225                        UintTy::U8 => Integer::I8,
226                        UintTy::U16 => Integer::I16,
227                        UintTy::U32 => Integer::I32,
228                        UintTy::U64 => Integer::I64,
229                        UintTy::U128 => Integer::I128,
230                    },
231                    false,
232                ),
233            ),
234        ),
235        TyKind::Float(f) => Layout::scalar(
236            dl,
237            scalar_unit(
238                dl,
239                Primitive::Float(match f {
240                    FloatTy::F16 => Float::F16,
241                    FloatTy::F32 => Float::F32,
242                    FloatTy::F64 => Float::F64,
243                    FloatTy::F128 => Float::F128,
244                }),
245            ),
246        ),
247        TyKind::Tuple(tys) => {
248            let kind =
249                if tys.len() == 0 { StructKind::AlwaysSized } else { StructKind::MaybeUnsized };
250
251            let fields = tys
252                .iter()
253                .map(|k| db.layout_of_ty(k, trait_env.clone()))
254                .collect::<Result<Vec<_>, _>>()?;
255            let fields = fields.iter().map(|it| &**it).collect::<Vec<_>>();
256            let fields = fields.iter().collect::<IndexVec<_, _>>();
257            cx.calc.univariant(&fields, &ReprOptions::default(), kind)?
258        }
259        TyKind::Array(element, count) => {
260            let count = try_const_usize(db, count).ok_or(LayoutError::HasErrorConst)? as u64;
261            let element = db.layout_of_ty(element, trait_env)?;
262            cx.calc.array_like::<_, _, ()>(&element, Some(count))?
263        }
264        TyKind::Slice(element) => {
265            let element = db.layout_of_ty(element, trait_env)?;
266            cx.calc.array_like::<_, _, ()>(&element, None)?
267        }
268        TyKind::Str => {
269            let element = scalar_unit(dl, Primitive::Int(Integer::I8, false));
270            cx.calc.array_like::<_, _, ()>(&Layout::scalar(dl, element), None)?
271        }
272        // Potentially-wide pointers.
273        TyKind::Ref(_, pointee, _) | TyKind::RawPtr(pointee, _) => {
274            let mut data_ptr = scalar_unit(dl, Primitive::Pointer(AddressSpace::ZERO));
275            if matches!(ty.kind(), TyKind::Ref(..)) {
276                data_ptr.valid_range_mut().start = 1;
277            }
278
279            // FIXME(next-solver)
280            // let pointee = tcx.normalize_erasing_regions(param_env, pointee);
281            // if pointee.is_sized(tcx.at(DUMMY_SP), param_env) {
282            //     return Ok(tcx.mk_layout(LayoutS::scalar(cx, data_ptr)));
283            // }
284
285            let unsized_part = struct_tail_erasing_lifetimes(db, pointee);
286            // FIXME(next-solver)
287            /*
288            if let TyKind::AssociatedType(id, subst) = unsized_part.kind(Interner) {
289                unsized_part = TyKind::Alias(chalk_ir::AliasTy::Projection(ProjectionTy {
290                    associated_ty_id: *id,
291                    substitution: subst.clone(),
292                }))
293                .intern(Interner);
294            }
295            unsized_part = normalize(db, trait_env, unsized_part);
296            */
297            let metadata = match unsized_part.kind() {
298                TyKind::Slice(_) | TyKind::Str => {
299                    scalar_unit(dl, Primitive::Int(dl.ptr_sized_integer(), false))
300                }
301                TyKind::Dynamic(..) => {
302                    let mut vtable = scalar_unit(dl, Primitive::Pointer(AddressSpace::ZERO));
303                    vtable.valid_range_mut().start = 1;
304                    vtable
305                }
306                _ => {
307                    // pointee is sized
308                    return Ok(Arc::new(Layout::scalar(dl, data_ptr)));
309                }
310            };
311
312            // Effectively a (ptr, meta) tuple.
313            LayoutData::scalar_pair(dl, data_ptr, metadata)
314        }
315        TyKind::Never => LayoutData::never_type(dl),
316        TyKind::FnDef(..) => LayoutData::unit(dl, true),
317        TyKind::Dynamic(..) | TyKind::Foreign(_) => LayoutData::unit(dl, false),
318        TyKind::FnPtr(..) => {
319            let mut ptr = scalar_unit(dl, Primitive::Pointer(dl.instruction_address_space));
320            ptr.valid_range_mut().start = 1;
321            Layout::scalar(dl, ptr)
322        }
323        TyKind::Closure(id, args) => {
324            let def = db.lookup_intern_closure(id.0);
325            let infer = db.infer(def.0);
326            let (captures, _) = infer.closure_info(id.0);
327            let fields = captures
328                .iter()
329                .map(|it| {
330                    let ty =
331                        it.ty.instantiate(interner, args.split_closure_args_untupled().parent_args);
332                    db.layout_of_ty(ty, trait_env.clone())
333                })
334                .collect::<Result<Vec<_>, _>>()?;
335            let fields = fields.iter().map(|it| &**it).collect::<Vec<_>>();
336            let fields = fields.iter().collect::<IndexVec<_, _>>();
337            cx.calc.univariant(&fields, &ReprOptions::default(), StructKind::AlwaysSized)?
338        }
339
340        TyKind::Coroutine(_, _)
341        | TyKind::CoroutineWitness(_, _)
342        | TyKind::CoroutineClosure(_, _) => {
343            return Err(LayoutError::NotImplemented);
344        }
345
346        TyKind::Pat(_, _) | TyKind::UnsafeBinder(_) => {
347            return Err(LayoutError::NotImplemented);
348        }
349
350        TyKind::Error(_) => return Err(LayoutError::HasErrorType),
351        TyKind::Placeholder(_)
352        | TyKind::Bound(..)
353        | TyKind::Infer(..)
354        | TyKind::Param(..)
355        | TyKind::Alias(..) => {
356            return Err(LayoutError::HasPlaceholder);
357        }
358    };
359    Ok(Arc::new(result))
360}

370fn struct_tail_erasing_lifetimes<'a>(db: &'a dyn HirDatabase, pointee: Ty<'a>) -> Ty<'a> {
371    match pointee.kind() {
372        TyKind::Adt(def, args) => {
373            let struct_id = match def.inner().id {
374                AdtId::StructId(id) => id,
375                _ => return pointee,
376            };
377            let data = struct_id.fields(db);
378            let mut it = data.fields().iter().rev();
379            match it.next() {
380                Some((f, _)) => {
381                    let last_field_ty = field_ty(db, struct_id.into(), f, &args);
382                    struct_tail_erasing_lifetimes(db, last_field_ty)
383                }
384                None => pointee,
385            }
386        }
387        TyKind::Tuple(tys) => {
388            if let Some(last_field_ty) = tys.iter().next_back() {
389                struct_tail_erasing_lifetimes(db, last_field_ty)
390            } else {
391                pointee
392            }
393        }
394        _ => pointee,
395    }
396}

398fn field_ty<'a>(
399    db: &'a dyn HirDatabase,
400    def: hir_def::VariantId,
401    fd: LocalFieldId,
402    args: &GenericArgs<'a>,
403) -> Ty<'a> {
404    db.field_types(def)[fd].instantiate(DbInterner::new_with(db, None, None), args)

407fn scalar_unit(dl: &TargetDataLayout, value: Primitive) -> Scalar {
408    Scalar::Initialized { value, valid_range: WrappingRange::full(value.size(dl)) }
