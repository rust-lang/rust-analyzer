1//! Utilities for working with generics.
2//!
3//! The layout for generics as expected by chalk are as follows:
4//! - Parent parameters
5//! - Optional Self parameter
6//! - Lifetime parameters
7//! - Type or Const parameters
8//!
9//! where parent follows the same scheme.
10
11use arrayvec::ArrayVec;
12use hir_def::{
13    ConstParamId, GenericDefId, GenericParamId, ItemContainerId, LifetimeParamId, Lookup,
14    TypeOrConstParamId, TypeParamId,
15    db::DefDatabase,
16    expr_store::ExpressionStore,
17    hir::generics::{
18        GenericParamDataRef, GenericParams, LifetimeParamData, TypeOrConstParamData,
19        TypeParamProvenance, WherePredicate,
20    },
21};

23pub(crate) fn generics(db: &dyn DefDatabase, def: GenericDefId) -> Generics<'_> {
24    let mut chain = ArrayVec::new();
25    let mut parent_params_len = 0;
26    if let Some(parent_def) = parent_generic_def(db, def) {
27        let (parent_params, parent_store) = GenericParams::with_store(db, parent_def);
28        chain.push(SingleGenerics {
29            def: parent_def,
30            params: parent_params,
31            store: parent_store,
32            preceding_params_len: 0,
33        });
34        parent_params_len = parent_params.len() as u32;
35    }
36    let (params, store) = GenericParams::with_store(db, def);
37    chain.push(SingleGenerics { def, params, store, preceding_params_len: parent_params_len });
38    Generics { chain }
39}

41#[derive(Debug)]
42pub(crate) struct Generics<'db> {
43    chain: ArrayVec<SingleGenerics<'db>, 2>,
44}

46#[derive(Debug)]
47pub(crate) struct SingleGenerics<'db> {
48    def: GenericDefId,
49    preceding_params_len: u32,
50    params: &'db GenericParams,
51    store: &'db ExpressionStore,
52}

54impl<'db> SingleGenerics<'db> {
55    pub(crate) fn def(&self) -> GenericDefId {
56        self.def
57    }
58
59    pub(crate) fn store(&self) -> &'db ExpressionStore {

61    }

63    pub(crate) fn where_predicates(&self) -> impl Iterator<Item = &WherePredicate> {
64        self.params.where_predicates().iter()
65    }
66
67    pub(crate) fn has_no_params(&self) -> bool {
68        self.params.is_empty()
69    }
70
71    pub(crate) fn len_lifetimes(&self) -> usize {
72        self.params.len_lifetimes()
73    }
74
75    pub(crate) fn len(&self) -> usize {
76        self.params.len()
77    }
78
79    fn iter_lifetimes(&self) -> impl Iterator<Item = (LifetimeParamId, &'db LifetimeParamData)> {
80        let parent = self.def;
81        self.params
82            .iter_lt()
83            .map(move |(local_id, data)| (LifetimeParamId { parent, local_id }, data))
84    }
85
86    pub(crate) fn iter_type_or_consts(
87        &self,
88    ) -> impl Iterator<Item = (TypeOrConstParamId, &'db TypeOrConstParamData)> {
89        let parent = self.def;
90        self.params
91            .iter_type_or_consts()
92            .map(move |(local_id, data)| (TypeOrConstParamId { parent, local_id }, data))
93    }
94
95    fn iter_type_or_consts_as_generic(
96        &self,
97    ) -> impl Iterator<Item = (GenericParamId, GenericParamDataRef<'db>)> {
98        self.iter_type_or_consts().map(|(id, data)| match data {
99            TypeOrConstParamData::TypeParamData(data) => (
100                GenericParamId::TypeParamId(TypeParamId::from_unchecked(id)),
101                GenericParamDataRef::TypeParamData(data),
102            ),
103            TypeOrConstParamData::ConstParamData(data) => (
104                GenericParamId::ConstParamId(ConstParamId::from_unchecked(id)),
105                GenericParamDataRef::ConstParamData(data),
106            ),
107        })
108    }
109
110    fn trait_self_and_others(
111        &self,
112    ) -> (
113        Option<(GenericParamId, GenericParamDataRef<'db>)>,
114        impl Iterator<Item = (GenericParamId, GenericParamDataRef<'db>)>,
115    ) {
116        let mut iter = self.iter_type_or_consts_as_generic();
117        let trait_self = if let GenericDefId::TraitId(_) = self.def { iter.next() } else { None };
118        (trait_self, iter)
119    }
120
121    pub(crate) fn iter(&self) -> impl Iterator<Item = (GenericParamId, GenericParamDataRef<'db>)> {
122        let lifetimes = self.iter_lifetimes().map(|(id, data)| {
123            (GenericParamId::LifetimeParamId(id), GenericParamDataRef::LifetimeParamData(data))
124        });
125        let (trait_self, type_and_consts) = self.trait_self_and_others();
126        trait_self.into_iter().chain(lifetimes).chain(type_and_consts)
127    }
128
129    pub(crate) fn iter_with_idx(
130        &self,
131    ) -> impl Iterator<Item = (u32, GenericParamId, GenericParamDataRef<'db>)> {
132        std::iter::zip(self.preceding_params_len.., self.iter())
133            .map(|(index, (id, data))| (index, id, data))
134    }
135
136    pub(crate) fn iter_id(&self) -> impl Iterator<Item = GenericParamId> {
137        self.iter().map(|(id, _)| id)
138    }
139}

201    /// Returns total number of generic parameters in scope, including those from parent.
202    pub(crate) fn len(&self) -> usize {
203        match &*self.chain {
204            [parent, owner] => parent.len() + owner.len(),
205            [owner] => owner.len(),
206            _ => unreachable!(),
207        }
208    }
209
210    #[inline]
211    pub(crate) fn len_parent(&self) -> usize {
212        self.parent().map_or(0, SingleGenerics::len)
213    }

215    pub(crate) fn len_lifetimes_self(&self) -> usize {
216        self.owner().len_lifetimes()
217    }

219    pub(crate) fn provenance_split(&self) -> ProvenanceSplit {
220        let parent_total = self.len_parent();
221
222        let owner = self.owner();
223        let lifetimes = owner.params.len_lifetimes();
224
225        let mut has_self_param = false;
226        let mut non_impl_trait_type_params = 0;
227        let mut impl_trait_type_params = 0;
228        let mut const_params = 0;
229        owner.params.iter_type_or_consts().for_each(|(_, data)| match data {
230            TypeOrConstParamData::TypeParamData(p) => match p.provenance {
231                TypeParamProvenance::TypeParamList => non_impl_trait_type_params += 1,
232                TypeParamProvenance::TraitSelf => has_self_param |= true,
233                TypeParamProvenance::ArgumentImplTrait => impl_trait_type_params += 1,
234            },
235            TypeOrConstParamData::ConstParamData(_) => const_params += 1,
236        });
237
238        ProvenanceSplit {
239            parent_total,
240            has_self_param,
241            non_impl_trait_type_params,
242            const_params,
243            impl_trait_type_params,
244            lifetimes,
245        }
246    }
247
248    fn find_owner(&self, def: GenericDefId) -> &SingleGenerics<'db> {
249        match &*self.chain {
250            [parent, owner] => {
251                if parent.def == def {
252                    parent
253                } else {
254                    debug_assert_eq!(def, owner.def);
255                    owner
256                }
257            }
258            [owner] => {
259                debug_assert_eq!(def, owner.def);
260                owner
261            }
262            _ => unreachable!(),
263        }
264    }
265
266    pub(crate) fn type_or_const_param_idx(&self, param: TypeOrConstParamId) -> u32 {
267        let owner = self.find_owner(param.parent);
268        let has_trait_self = matches!(owner.def, GenericDefId::TraitId(_));
269        if has_trait_self && param.local_id == GenericParams::SELF_PARAM_ID_IN_SELF {
270            owner.preceding_params_len
271        } else {
272            owner.preceding_params_len
273                + owner.len_lifetimes() as u32
274                + param.local_id.into_raw().into_u32()
275        }
276    }
277
278    pub(crate) fn lifetime_param_idx(&self, param: LifetimeParamId) -> u32 {
279        let owner = self.find_owner(param.parent);
280        let has_trait_self = matches!(owner.def, GenericDefId::TraitId(_));
281        owner.preceding_params_len
282            + u32::from(has_trait_self)
283            + param.local_id.into_raw().into_u32()
284    }
285}

287pub(crate) struct ProvenanceSplit {
288    pub(crate) parent_total: usize,
289    // The rest are about self.
290    pub(crate) has_self_param: bool,
291    pub(crate) non_impl_trait_type_params: usize,
292    pub(crate) const_params: usize,
293    pub(crate) impl_trait_type_params: usize,
294    pub(crate) lifetimes: usize,
295}

297fn parent_generic_def(db: &dyn DefDatabase, def: GenericDefId) -> Option<GenericDefId> {
298    let container = match def {
299        GenericDefId::FunctionId(it) => it.lookup(db).container,
300        GenericDefId::TypeAliasId(it) => it.lookup(db).container,
301        GenericDefId::ConstId(it) => it.lookup(db).container,
302        GenericDefId::StaticId(_)
303        | GenericDefId::AdtId(_)
304        | GenericDefId::TraitId(_)
305        | GenericDefId::ImplId(_) => return None,
306    };
307
308    match container {
309        ItemContainerId::ImplId(it) => Some(it.into()),
310        ItemContainerId::TraitId(it) => Some(it.into()),
311        ItemContainerId::ModuleId(_) | ItemContainerId::ExternBlockId(_) => None,
312    }
