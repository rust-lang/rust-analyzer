1//! Utilities for working with generics.
2//!
3//! The layout for generics as expected by chalk are as follows:
4//! - Parent parameters
5//! - Optional Self parameter
6//! - Lifetime parameters
7//! - Type or Const parameters
8//!
9//! where parent follows the same scheme.
10use std::ops;
11
12use hir_def::{
13    ConstParamId, GenericDefId, GenericParamId, ItemContainerId, LifetimeParamId, Lookup,
14    TypeOrConstParamId, TypeParamId,
15    db::DefDatabase,
16    expr_store::ExpressionStore,
17    hir::generics::{
18        GenericParamDataRef, GenericParams, LifetimeParamData, LocalLifetimeParamId,
19        LocalTypeOrConstParamId, TypeOrConstParamData, TypeParamProvenance, WherePredicate,
20    },
21};
22use itertools::chain;
23use triomphe::Arc;
24
25pub fn generics(db: &dyn DefDatabase, def: GenericDefId) -> Generics {
26    let parent_generics = parent_generic_def(db, def).map(|def| Box::new(generics(db, def)));
27    let (params, store) = db.generic_params_and_store(def);
28    let has_trait_self_param = params.trait_self_param().is_some();
29    Generics { def, params, parent_generics, has_trait_self_param, store }
30}
31#[derive(Clone, Debug)]
32pub struct Generics {
33    def: GenericDefId,
34    params: Arc<GenericParams>,
35    store: Arc<ExpressionStore>,
36    parent_generics: Option<Box<Generics>>,
37    has_trait_self_param: bool,
38}
39
40impl<T> ops::Index<T> for Generics
41where
42    GenericParams: ops::Index<T>,
43{
44    type Output = <GenericParams as ops::Index<T>>::Output;
45    fn index(&self, index: T) -> &Self::Output {
46        &self.params[index]
47    }
48}

50impl Generics {
51    pub(crate) fn def(&self) -> GenericDefId {
52        self.def
53    }
54
55    pub(crate) fn store(&self) -> &ExpressionStore {
56        &self.store
57    }
58
59    pub(crate) fn where_predicates(&self) -> impl Iterator<Item = &WherePredicate> {
60        self.params.where_predicates().iter()
61    }
62
63    pub(crate) fn has_no_predicates(&self) -> bool {
64        self.params.has_no_predicates()
65            && self.parent_generics.as_ref().is_none_or(|g| g.params.has_no_predicates())
66    }
67
68    pub(crate) fn is_empty(&self) -> bool {
69        self.params.is_empty() && self.parent_generics.as_ref().is_none_or(|g| g.params.is_empty())
70    }
71
72    pub(crate) fn iter_id(&self) -> impl Iterator<Item = GenericParamId> + '_ {
73        self.iter_parent_id().chain(self.iter_self_id())
74    }
75
76    pub(crate) fn iter_self_id(&self) -> impl Iterator<Item = GenericParamId> + '_ {
77        self.iter_self().map(|(id, _)| id)
78    }
79
80    pub(crate) fn iter_parent_id(&self) -> impl Iterator<Item = GenericParamId> + '_ {
81        self.iter_parent().map(|(id, _)| id)
82    }
83
84    pub(crate) fn iter_self_type_or_consts(
85        &self,
86    ) -> impl DoubleEndedIterator<Item = (LocalTypeOrConstParamId, &TypeOrConstParamData)> + '_
87    {
88        let mut toc = self.params.iter_type_or_consts();
89        let trait_self_param = self.has_trait_self_param.then(|| toc.next()).flatten();
90        chain!(trait_self_param, toc)
91    }
92
93    /// Iterate over the parent params followed by self params.
94    pub(crate) fn iter(
95        &self,
96    ) -> impl DoubleEndedIterator<Item = (GenericParamId, GenericParamDataRef<'_>)> + '_ {
97        self.iter_parent().chain(self.iter_self())
98    }
99
100    pub(crate) fn iter_parents_with_store(
101        &self,
102    ) -> impl Iterator<Item = ((GenericParamId, GenericParamDataRef<'_>), &ExpressionStore)> + '_
103    {
104        self.iter_parent()
105            .zip(self.parent_generics().into_iter().flat_map(|it| std::iter::repeat(&*it.store)))
106    }
107
108    /// Iterate over the params without parent params.
109    pub(crate) fn iter_self(
110        &self,
111    ) -> impl DoubleEndedIterator<Item = (GenericParamId, GenericParamDataRef<'_>)> + '_ {
112        let mut toc = self.params.iter_type_or_consts().map(from_toc_id(self));
113        let trait_self_param = self.has_trait_self_param.then(|| toc.next()).flatten();
114        chain!(trait_self_param, self.params.iter_lt().map(from_lt_id(self)), toc)
115    }
116
117    /// Iterator over types and const params of parent.
118    pub(crate) fn iter_parent(
119        &self,
120    ) -> impl DoubleEndedIterator<Item = (GenericParamId, GenericParamDataRef<'_>)> + '_ {
121        self.parent_generics().into_iter().flat_map(|it| {
122            let mut toc = it.params.iter_type_or_consts().map(from_toc_id(it));
123            let trait_self_param = it.has_trait_self_param.then(|| toc.next()).flatten();
124            chain!(trait_self_param, it.params.iter_lt().map(from_lt_id(it)), toc)
125        })
126    }

128    /// Returns total number of generic parameters in scope, including those from parent.
129    pub(crate) fn len(&self) -> usize {
130        let parent = self.len_parent();
131        let child = self.params.len();
132        parent + child
133    }
134
135    #[inline]
136    pub(crate) fn len_parent(&self) -> usize {
137        self.parent_generics().map_or(0, Generics::len)
138    }
139
140    /// Returns numbers of generic parameters excluding those from parent.
141    pub(crate) fn len_self(&self) -> usize {
142        self.params.len()
143    }
144
145    pub(crate) fn len_lifetimes_self(&self) -> usize {
146        self.params.len_lifetimes()
147    }

149    /// (parent total, self param, type params, const params, impl trait list, lifetimes)
150    pub(crate) fn provenance_split(&self) -> (usize, bool, usize, usize, usize, usize) {
151        let mut self_param = false;
152        let mut type_params = 0;
153        let mut impl_trait_params = 0;
154        let mut const_params = 0;
155        self.params.iter_type_or_consts().for_each(|(_, data)| match data {
156            TypeOrConstParamData::TypeParamData(p) => match p.provenance {
157                TypeParamProvenance::TypeParamList => type_params += 1,
158                TypeParamProvenance::TraitSelf => self_param |= true,
159                TypeParamProvenance::ArgumentImplTrait => impl_trait_params += 1,
160            },
161            TypeOrConstParamData::ConstParamData(_) => const_params += 1,
162        });
163
164        let lifetime_params = self.params.len_lifetimes();
165
166        let parent_len = self.parent_generics().map_or(0, Generics::len);
167        (parent_len, self_param, type_params, const_params, impl_trait_params, lifetime_params)
168    }
169
170    pub(crate) fn type_or_const_param(
171        &self,
172        param: TypeOrConstParamId,
173    ) -> Option<(usize, TypeOrConstParamData)> {
174        let idx = self.find_type_or_const_param(param)?;
175        self.iter().nth(idx).and_then(|p| {
176            let data = match p.1 {
177                GenericParamDataRef::TypeParamData(p) => p.clone().into(),
178                GenericParamDataRef::ConstParamData(p) => p.clone().into(),
179                _ => return None,
180            };
181            Some((idx, data))
182        })
183    }
184
185    pub fn type_or_const_param_idx(&self, param: TypeOrConstParamId) -> Option<usize> {
186        self.find_type_or_const_param(param)
187    }
188
189    fn find_type_or_const_param(&self, param: TypeOrConstParamId) -> Option<usize> {
190        if param.parent == self.def {
191            let idx = param.local_id.into_raw().into_u32() as usize;
192            debug_assert!(
193                idx <= self.params.len_type_or_consts(),
194                "idx: {} len: {}",
195                idx,
196                self.params.len_type_or_consts()
197            );
198            if self.params.trait_self_param() == Some(param.local_id) {
199                return Some(idx);
200            }
201            Some(self.parent_generics().map_or(0, |g| g.len()) + self.params.len_lifetimes() + idx)
202        } else {
203            debug_assert_eq!(self.parent_generics().map(|it| it.def), Some(param.parent));
204            self.parent_generics().and_then(|g| g.find_type_or_const_param(param))
205        }
206    }
207
208    pub fn lifetime_idx(&self, lifetime: LifetimeParamId) -> Option<usize> {
209        self.find_lifetime(lifetime)
210    }
211
212    fn find_lifetime(&self, lifetime: LifetimeParamId) -> Option<usize> {
213        if lifetime.parent == self.def {
214            let idx = lifetime.local_id.into_raw().into_u32() as usize;
215            debug_assert!(idx <= self.params.len_lifetimes());
216            Some(
217                self.parent_generics().map_or(0, |g| g.len())
218                    + self.params.trait_self_param().is_some() as usize
219                    + idx,
220            )
221        } else {
222            debug_assert_eq!(self.parent_generics().map(|it| it.def), Some(lifetime.parent));
223            self.parent_generics().and_then(|g| g.find_lifetime(lifetime))
224        }
225    }
226
227    pub(crate) fn parent_generics(&self) -> Option<&Generics> {
228        self.parent_generics.as_deref()
229    }

232pub(crate) fn trait_self_param_idx(db: &dyn DefDatabase, def: GenericDefId) -> Option<usize> {
233    match def {
234        GenericDefId::TraitId(_) => {
235            let params = db.generic_params(def);
236            params.trait_self_param().map(|idx| idx.into_raw().into_u32() as usize)
237        }
238        GenericDefId::ImplId(_) => None,
239        _ => {
240            let parent_def = parent_generic_def(db, def)?;
241            let parent_params = db.generic_params(parent_def);
242            let parent_self_idx = parent_params.trait_self_param()?.into_raw().into_u32() as usize;
243            Some(parent_self_idx)
244        }
245    }
246}

248pub(crate) fn parent_generic_def(db: &dyn DefDatabase, def: GenericDefId) -> Option<GenericDefId> {
249    let container = match def {
250        GenericDefId::FunctionId(it) => it.lookup(db).container,
251        GenericDefId::TypeAliasId(it) => it.lookup(db).container,
252        GenericDefId::ConstId(it) => it.lookup(db).container,
253        GenericDefId::StaticId(_)
254        | GenericDefId::AdtId(_)
255        | GenericDefId::TraitId(_)
256        | GenericDefId::ImplId(_) => return None,
257    };
258
259    match container {
260        ItemContainerId::ImplId(it) => Some(it.into()),
261        ItemContainerId::TraitId(it) => Some(it.into()),
262        ItemContainerId::ModuleId(_) | ItemContainerId::ExternBlockId(_) => None,
263    }
264}

266fn from_toc_id<'a>(
267    it: &'a Generics,
268) -> impl Fn(
269    (LocalTypeOrConstParamId, &'a TypeOrConstParamData),
270) -> (GenericParamId, GenericParamDataRef<'a>) {
271    move |(local_id, p): (_, _)| {
272        let id = TypeOrConstParamId { parent: it.def, local_id };
273        match p {
274            TypeOrConstParamData::TypeParamData(p) => (
275                GenericParamId::TypeParamId(TypeParamId::from_unchecked(id)),
276                GenericParamDataRef::TypeParamData(p),
277            ),
278            TypeOrConstParamData::ConstParamData(p) => (
279                GenericParamId::ConstParamId(ConstParamId::from_unchecked(id)),
280                GenericParamDataRef::ConstParamData(p),
281            ),
282        }
283    }
284}
285
286fn from_lt_id<'a>(
287    it: &'a Generics,
288) -> impl Fn((LocalLifetimeParamId, &'a LifetimeParamData)) -> (GenericParamId, GenericParamDataRef<'a>)
289{
290    move |(local_id, p): (_, _)| {
291        (
292            GenericParamId::LifetimeParamId(LifetimeParamId { parent: it.def, local_id }),
293            GenericParamDataRef::LifetimeParamData(p),
294        )
295    }
