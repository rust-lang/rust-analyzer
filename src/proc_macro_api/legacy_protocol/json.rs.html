1//! Protocol functions for json.
2use std::io::{self, BufRead, Write};
3
4/// Reads a JSON message from the input stream.
5pub fn read_json<'a>(
6    inp: &mut impl BufRead,
7    buf: &'a mut String,
8) -> io::Result<Option<&'a String>> {
9    loop {
10        buf.clear();
11
12        inp.read_line(buf)?;
13        buf.pop(); // Remove trailing '\n'
14
15        if buf.is_empty() {
16            return Ok(None);
17        }
18
19        // Some ill behaved macro try to use stdout for debugging
20        // We ignore it here
21        if !buf.starts_with('{') {
22            tracing::error!("proc-macro tried to print : {}", buf);
23            continue;
24        }
25
26        return Ok(Some(buf));
27    }
28}
29
30/// Writes a JSON message to the output stream.
31pub fn write_json(out: &mut impl Write, msg: &str) -> io::Result<()> {
32    tracing::debug!("> {}", msg);
33    out.write_all(msg.as_bytes())?;
34    out.write_all(b"\n")?;
35    out.flush()
36}
