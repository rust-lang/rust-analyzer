1//! Postcard encode and decode implementations.
2
3use std::io::{self, BufRead, Write};
4
5use serde::{Serialize, de::DeserializeOwned};
6
7pub(crate) fn read<'a, R: BufRead + ?Sized>(
8    inp: &mut R,
9    buf: &'a mut Vec<u8>,
10) -> io::Result<Option<&'a mut Vec<u8>>> {
11    buf.clear();
12    let n = inp.read_until(0, buf)?;
13    if n == 0 {
14        return Ok(None);
15    }
16    Ok(Some(buf))
17}
18
19pub(crate) fn write<W: Write + ?Sized>(out: &mut W, buf: &[u8]) -> io::Result<()> {
20    out.write_all(buf)?;
21    out.flush()
22}
23
24pub(crate) fn encode<T: Serialize>(msg: &T) -> io::Result<Vec<u8>> {
25    postcard::to_allocvec_cobs(msg).map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e))
26}
27
28pub(crate) fn decode<T: DeserializeOwned>(buf: &mut [u8]) -> io::Result<T> {
29    postcard::from_bytes_cobs(buf).map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e))
30}
