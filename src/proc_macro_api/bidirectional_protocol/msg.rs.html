128impl Message for BidirectionalMessage {
129    type Buf = Vec<u8>;
130
131    fn read(inp: &mut dyn BufRead, buf: &mut Self::Buf) -> io::Result<Option<Self>> {
132        Ok(match postcard::read(inp, buf)? {
133            None => None,
134            Some(buf) => Some(postcard::decode(buf)?),
135        })
136    }
137    fn write(self, out: &mut dyn Write) -> io::Result<()> {
138        let value = postcard::encode(&self)?;
139        postcard::write(out, &value)
140    }
141}
