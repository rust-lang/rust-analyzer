26/// Legacy span type, only defined here as it is still used by the proc-macro server.
27/// While rust-analyzer doesn't use this anymore at all, RustRover relies on the legacy type for
28/// proc-macro expansion.
29#[derive(Clone, Copy, PartialEq, Eq, Hash)]
30pub struct SpanId(pub u32);
31
32impl std::fmt::Debug for SpanId {
33    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
34        self.0.fmt(f)
35    }

38pub(crate) fn version_check(srv: &ProcMacroServerProcess) -> Result<u32, ServerError> {
39    let request = Request::ApiVersionCheck {};
40    let response = send_task(srv, request)?;
41
42    match response {
43        Response::ApiVersionCheck(version) => Ok(version),
44        _ => Err(ServerError { message: "unexpected response".to_owned(), io: None }),
45    }
46}

48/// Enable support for rust-analyzer span mode if the server supports it.
49pub(crate) fn enable_rust_analyzer_spans(
50    srv: &ProcMacroServerProcess,
51) -> Result<SpanMode, ServerError> {
52    let request = Request::SetConfig(ServerConfig { span_mode: SpanMode::RustAnalyzer });
53    let response = send_task(srv, request)?;
54
55    match response {
56        Response::SetConfig(ServerConfig { span_mode }) => Ok(span_mode),
57        _ => Err(ServerError { message: "unexpected response".to_owned(), io: None }),
58    }
59}

65) -> Result<Result<Vec<(String, ProcMacroKind)>, String>, ServerError> {
66    let request = Request::ListMacros { dylib_path: dylib_path.to_path_buf().into() };
67
68    let response = send_task(srv, request)?;
69
70    match response {
71        Response::ListMacros(it) => Ok(it),
72        _ => Err(ServerError { message: "unexpected response".to_owned(), io: None }),
73    }

76pub(crate) fn expand(
77    proc_macro: &ProcMacro,
78    process: &ProcMacroServerProcess,
79    subtree: tt::SubtreeView<'_>,
80    attr: Option<tt::SubtreeView<'_>>,
81    env: Vec<(String, String)>,
82    def_site: Span,
83    call_site: Span,
84    mixed_site: Span,
85    current_dir: String,
86) -> Result<Result<tt::TopSubtree, String>, crate::ServerError> {
87    let version = process.version();
88    let mut span_data_table = SpanDataIndexMap::default();
89    let def_site = span_data_table.insert_full(def_site).0;
90    let call_site = span_data_table.insert_full(call_site).0;
91    let mixed_site = span_data_table.insert_full(mixed_site).0;
92    let task = ExpandMacro {
93        data: ExpandMacroData {
94            macro_body: FlatTree::from_subtree(subtree, version, &mut span_data_table),
95            macro_name: proc_macro.name.to_string(),
96            attributes: attr
97                .map(|subtree| FlatTree::from_subtree(subtree, version, &mut span_data_table)),
98            has_global_spans: ExpnGlobals {
99                serialize: version >= version::HAS_GLOBAL_SPANS,
100                def_site,
101                call_site,
102                mixed_site,
103            },
104            span_data_table: if process.rust_analyzer_spans() {
105                serialize_span_data_index_map(&span_data_table)
106            } else {
107                Vec::new()
108            },
109        },
110        lib: proc_macro.dylib_path.to_path_buf().into(),
111        env,
112        current_dir: Some(current_dir),
113    };
114
115    let response = send_task(process, Request::ExpandMacro(Box::new(task)))?;
116
117    match response {
118        Response::ExpandMacro(it) => Ok(it
119            .map(|tree| {
120                let mut expanded = FlatTree::to_subtree_resolved(tree, version, &span_data_table);
121                if proc_macro.needs_fixup_change() {
122                    proc_macro.change_fixup_to_match_old_server(&mut expanded);
123                }
124                expanded
125            })
126            .map_err(|msg| msg.0)),
127        Response::ExpandMacroExtended(it) => Ok(it
128            .map(|resp| {
129                let mut expanded = FlatTree::to_subtree_resolved(
130                    resp.tree,
131                    version,
132                    &deserialize_span_data_index_map(&resp.span_data_table),
133                );
134                if proc_macro.needs_fixup_change() {
135                    proc_macro.change_fixup_to_match_old_server(&mut expanded);
136                }
137                expanded
138            })
139            .map_err(|msg| msg.0)),
140        _ => Err(ServerError { message: "unexpected response".to_owned(), io: None }),
141    }
142}

144/// Sends a request to the proc-macro server and waits for a response.
145fn send_task(srv: &ProcMacroServerProcess, req: Request) -> Result<Response, ServerError> {
146    if let Some(server_error) = srv.exited() {
147        return Err(server_error.clone());
148    }
149
150    srv.send_task_legacy::<_, _>(send_request, req)
151}
152
153/// Sends a request to the server and reads the response.
154fn send_request(
155    mut writer: &mut dyn Write,
156    mut reader: &mut dyn BufRead,
157    req: Request,
158    buf: &mut String,
159) -> Result<Option<Response>, ServerError> {
160    req.write(&mut writer).map_err(|err| ServerError {
161        message: "failed to write request".into(),
162        io: Some(Arc::new(err)),
163    })?;
164    let res = Response::read(&mut reader, buf).map_err(|err| ServerError {
165        message: "failed to read response".into(),
166        io: Some(Arc::new(err)),
167    })?;
168    Ok(res)
169}
