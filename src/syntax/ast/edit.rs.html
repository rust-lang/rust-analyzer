24impl From<u8> for IndentLevel {
25    fn from(level: u8) -> IndentLevel {
26        IndentLevel(level)
27    }
28}

30impl fmt::Display for IndentLevel {
31    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
32        let spaces = "                                        ";
33        let buf;
34        let len = self.0 as usize * 4;
35        let indent = if len <= spaces.len() {
36            &spaces[..len]
37        } else {
38            buf = " ".repeat(len);
39            &buf
40        };
41        fmt::Display::fmt(indent, f)
42    }
43}

58impl IndentLevel {
59    pub fn zero() -> IndentLevel {
60        IndentLevel(0)
61    }
62    pub fn is_zero(&self) -> bool {
63        self.0 == 0
64    }
65    pub fn from_element(element: &SyntaxElement) -> IndentLevel {
66        match element {
67            rowan::NodeOrToken::Node(it) => IndentLevel::from_node(it),
68            rowan::NodeOrToken::Token(it) => IndentLevel::from_token(it),
69        }
70    }
71
72    pub fn from_node(node: &SyntaxNode) -> IndentLevel {
73        match node.first_token() {
74            Some(it) => Self::from_token(&it),
75            None => IndentLevel(0),
76        }
77    }
78
79    pub fn from_token(token: &SyntaxToken) -> IndentLevel {
80        for ws in prev_tokens(token.clone()).filter_map(ast::Whitespace::cast) {
81            let text = ws.syntax().text();
82            if let Some(pos) = text.rfind('\n') {
83                let level = text[pos + 1..].chars().count() / 4;
84                return IndentLevel(level as u8);
85            }
86        }
87        IndentLevel(0)
88    }
89
90    pub(super) fn clone_increase_indent(self, node: &SyntaxNode) -> SyntaxNode {
91        let (editor, node) = SyntaxEditor::new(node.clone());
92        let tokens = node
93            .preorder_with_tokens()
94            .filter_map(|event| match event {
95                rowan::WalkEvent::Leave(NodeOrToken::Token(it)) => Some(it),
96                _ => None,
97            })
98            .filter_map(ast::Whitespace::cast)
99            .filter(|ws| ws.text().contains('\n'));
100        for ws in tokens {
101            let new_ws = make::tokens::whitespace(&format!("{}{self}", ws.syntax()));
102            editor.replace(ws.syntax(), &new_ws);
103        }
104        editor.finish().new_root().clone()
105    }
106
107    pub(super) fn clone_decrease_indent(self, node: &SyntaxNode) -> SyntaxNode {
108        let (editor, node) = SyntaxEditor::new(node.clone());
109        let tokens = node
110            .preorder_with_tokens()
111            .filter_map(|event| match event {
112                rowan::WalkEvent::Leave(NodeOrToken::Token(it)) => Some(it),
113                _ => None,
114            })
115            .filter_map(ast::Whitespace::cast)
116            .filter(|ws| ws.text().contains('\n'));
117        for ws in tokens {
118            let new_ws =
119                make::tokens::whitespace(&ws.syntax().text().replace(&format!("\n{self}"), "\n"));
120            editor.replace(ws.syntax(), &new_ws);
121        }
122        editor.finish().new_root().clone()
123    }
124}

126fn prev_tokens(token: SyntaxToken) -> impl Iterator<Item = SyntaxToken> {
127    iter::successors(Some(token), |token| token.prev_token())
128}

130pub trait AstNodeEdit: AstNode + Clone + Sized {
131    fn indent_level(&self) -> IndentLevel {
132        IndentLevel::from_node(self.syntax())
133    }
134    #[must_use]
135    fn indent(&self, level: IndentLevel) -> Self {
136        Self::cast(level.clone_increase_indent(self.syntax())).unwrap()
137    }
138    #[must_use]
139    fn indent_with_mapping(&self, level: IndentLevel, make: &SyntaxFactory) -> Self {
140        let new_node = self.indent(level);
141        if let Some(mut mapping) = make.mappings() {
142            let mut builder = SyntaxMappingBuilder::new(new_node.syntax().clone());
143            for (old, new) in self.syntax().children().zip(new_node.syntax().children()) {
144                builder.map_node(old, new);
145            }
146            builder.finish(&mut mapping);
147        }
148        new_node
149    }
150    #[must_use]
151    fn dedent(&self, level: IndentLevel) -> Self {
152        Self::cast(level.clone_decrease_indent(self.syntax())).unwrap()
153    }
154    #[must_use]
155    fn reset_indent(&self) -> Self {
156        let level = IndentLevel::from_node(self.syntax());
157        self.dedent(level)
158    }
159}

163pub trait AttrsOwnerEdit: ast::HasAttrs {
164    fn remove_attrs_and_docs(&self, editor: &SyntaxEditor) {
165        let mut remove_next_ws = false;
166        for child in self.syntax().children_with_tokens() {
167            match child.kind() {
168                ATTR | COMMENT => {
169                    remove_next_ws = true;
170                    editor.delete(child);
171                    continue;
172                }
173                WHITESPACE if remove_next_ws => {
174                    editor.delete(child);
175                }
176                _ => (),
177            }
178            remove_next_ws = false;
179        }
180    }
181}

185impl ast::IdentPat {
186    pub fn set_pat(&self, pat: Option<ast::Pat>, editor: &SyntaxEditor) -> ast::IdentPat {
187        let make = editor.make();
188        match pat {
189            None => {
190                if let Some(at_token) = self.at_token() {
191                    // Remove `@ Pat`
192                    let start = at_token.clone().into();
193                    let end = self
194                        .pat()
195                        .map(|it| it.syntax().clone().into())
196                        .unwrap_or_else(|| at_token.into());
197                    editor.delete_all(start..=end);
198
199                    // Remove any trailing ws
200                    if let Some(last) =
201                        self.syntax().last_token().filter(|it| it.kind() == WHITESPACE)
202                    {
203                        last.detach();
204                    }
205                }
206            }
207            Some(pat) => {
208                if let Some(old_pat) = self.pat() {
209                    // Replace existing pattern
210                    editor.replace(old_pat.syntax(), pat.syntax())
211                } else if let Some(at_token) = self.at_token() {
212                    // Have an `@` token but not a pattern yet
213                    editor.insert(Position::after(at_token), pat.syntax());
214                } else {
215                    // Don't have an `@`, should have a name
216                    let name = self.name().unwrap();
217                    let elements = vec![
218                        make.whitespace(" ").into(),
219                        make.token(T![@]).into(),
220                        make.whitespace(" ").into(),
221                        pat.syntax().clone().into(),
222                    ];
223
224                    if self.syntax().parent().is_none() {
225                        let (local, local_self) = SyntaxEditor::with_ast_node(self);
226                        let local_name = local_self.name().unwrap();
227                        local.insert_all(Position::after(local_name.syntax()), elements);
228                        let edit = local.finish();
229                        return ast::IdentPat::cast(edit.new_root().clone()).unwrap();
230                    } else {
231                        editor.insert_all(Position::after(name.syntax()), elements);
232                    }
233                }
234            }
235        }
236        self.clone()
237    }
238}

240impl ast::UseTree {
241    pub fn wrap_in_tree_list_with_editor(&self) -> Option<ast::UseTree> {
242        if self.use_tree_list().is_some()
243            && self.path().is_none()
244            && self.star_token().is_none()
245            && self.rename().is_none()
246        {
247            return None;
248        }
249
250        let (editor, use_tree) = SyntaxEditor::with_ast_node(self);
251        let make = editor.make();
252        let first_child = use_tree.syntax().first_child_or_token()?;
253        let last_child = use_tree.syntax().last_child_or_token()?;
254        let use_tree_list = make.use_tree_list(once(self.clone()));
255        editor.replace_all(first_child..=last_child, vec![use_tree_list.syntax().clone().into()]);
256
257        let edit = editor.finish();
258        ast::UseTree::cast(edit.new_root().clone())
259    }
260}

266impl ast::GenericParamList {
267    /// Constructs a matching [`ast::GenericArgList`]
268    pub fn to_generic_args(&self, make: &SyntaxFactory) -> ast::GenericArgList {
269        let args = self.generic_params().filter_map(|param| match param {
270            ast::GenericParam::LifetimeParam(it) => {
271                Some(ast::GenericArg::LifetimeArg(make.lifetime_arg(it.lifetime()?)))
272            }
273            ast::GenericParam::TypeParam(it) => {
274                Some(ast::GenericArg::TypeArg(make.type_arg(make.ty_name(it.name()?))))
275            }
276            ast::GenericParam::ConstParam(it) => {
277                // Name-only const params get parsed as `TypeArg`s
278                Some(ast::GenericArg::TypeArg(make.type_arg(make.ty_name(it.name()?))))
279            }
280        });
281
282        make::generic_arg_list(args)
283    }
284}

286impl ast::UseTree {
287    /// Deletes the usetree node represented by the input. Recursively removes parents, including use nodes that become empty.
288    pub fn remove_recursive(self, editor: &SyntaxEditor) {
289        let parent = self.syntax().parent();
290
291        if let Some(u) = parent.clone().and_then(ast::Use::cast) {
292            u.remove(editor);
293        } else if let Some(u) = parent.and_then(ast::UseTreeList::cast) {
294            if u.use_trees().nth(1).is_none()
295                || u.use_trees().all(|use_tree| {
296                    use_tree.syntax() == self.syntax() || editor.deleted(use_tree.syntax())
297                })
298            {
299                u.parent_use_tree().remove_recursive(editor);
300                return;
301            }
302            self.remove(editor);
303            u.remove_unnecessary_braces(editor);
304        }
305    }
306
307    /// Splits off the given prefix, making it the path component of the use tree,
308    /// appending the rest of the path to all UseTreeList items.
309    ///
310    /// # Examples
311    ///
312    /// `prefix$0::suffix` -> `prefix::{suffix}`
313    ///
314    /// `prefix$0` -> `prefix::{self}`
315    ///
316    /// `prefix$0::*` -> `prefix::{*}`
317    pub fn split_prefix_with_editor(&self, editor: &SyntaxEditor, prefix: &ast::Path) {
318        debug_assert_eq!(self.path(), Some(prefix.top_path()));
319
320        let make = editor.make();
321        let path = self.path().unwrap();
322        let suffix = if path == *prefix {
323            if self.use_tree_list().is_some() {
324                return;
325            } else if self.star_token().is_some() {
326                make.use_tree_glob()
327            } else {
328                let self_path = make.path_unqualified(make.path_segment_self());
329                make.use_tree(self_path, None, self.rename(), false)
330            }
331        } else {
332            let suffix_segments = path.segments().skip(prefix.segments().count());
333            let suffix_path = make.path_from_segments(suffix_segments, false);
334            make.use_tree(
335                suffix_path,
336                self.use_tree_list(),
337                self.rename(),
338                self.star_token().is_some(),
339            )
340        };
341        let use_tree_list = make.use_tree_list(once(suffix));
342        let new_use_tree = make.use_tree(prefix.clone(), Some(use_tree_list), None, false);
343
344        editor.replace(self.syntax(), new_use_tree.syntax());
345    }
346}

348impl ast::RecordExprField {
349    /// This will either replace the initializer, or in the case that this is a shorthand convert
350    /// the initializer into the name ref and insert the expr as the new initializer.
351    pub fn replace_expr(&self, editor: &SyntaxEditor, expr: ast::Expr) {
352        if self.name_ref().is_some() {
353            if let Some(prev) = self.expr() {
354                editor.replace(prev.syntax(), expr.syntax());
355            }
356        } else if let Some(ast::Expr::PathExpr(path_expr)) = self.expr()
357            && let Some(path) = path_expr.path()
358            && let Some(name_ref) = path.as_single_name_ref()
359        {
360            // shorthand `{ x }` → expand to `{ x: expr }`
361            let new_field = editor
362                .make()
363                .record_expr_field(editor.make().name_ref(&name_ref.text()), Some(expr));
364            editor.replace(self.syntax(), new_field.syntax());
365        }
366    }
367}

371    let arm_list = {

407#[cfg(test)]
408fn check_split_prefix(before: &str, expected: &str) {
409    let source = crate::SourceFile::parse(before, parser::Edition::CURRENT).tree();
410    let use_tree = source.syntax().descendants().find_map(ast::UseTree::cast).unwrap();
411    let (editor, use_tree) = SyntaxEditor::with_ast_node(&use_tree);
412    let prefix = use_tree.path().unwrap();
413    use_tree.split_prefix_with_editor(&editor, &prefix);
414    let edit = editor.finish();
415    assert_eq!(edit.new_root().to_string(), expected);
