386    pub fn block_expr(
387        &self,
388        statements: impl IntoIterator<Item = ast::Stmt>,
389        tail_expr: Option<ast::Expr>,
390    ) -> ast::BlockExpr {
391        let (statements, mut input) = iterator_input(statements);
392
393        let ast = make::block_expr(statements, tail_expr.clone()).clone_for_update();
394
395        if let Some(mut mapping) = self.mappings() {
396            let stmt_list = ast.stmt_list().unwrap();
397            let mut builder = SyntaxMappingBuilder::new(stmt_list.syntax().clone());
398
399            if let Some(input) = tail_expr {
400                builder.map_node(
401                    input.syntax().clone(),
402                    stmt_list.tail_expr().unwrap().syntax().clone(),
403                );
404            } else if let Some(ast_tail) = stmt_list.tail_expr() {
405                // The parser interpreted the last statement (probably a statement with a block) as an Expr
406                let last_stmt = input.pop().unwrap();
407
408                builder.map_node(last_stmt, ast_tail.syntax().clone());
409            }
410
411            builder.map_children(input, stmt_list.statements().map(|it| it.syntax().clone()));
412
413            builder.finish(&mut mapping);
414        }
415
416        ast
417    }

1364// We need to collect `input` here instead of taking `impl IntoIterator + Clone`,
1365// because if we took `impl IntoIterator + Clone`, that could be something like an
1366// `Iterator::map` with a closure that also makes use of a `SyntaxFactory` constructor.
1367//
1368// In that case, the iterator would be evaluated inside of the call to `map_children`,
1369// and the inner constructor would try to take a mutable borrow of the mappings `RefCell`,
1370// which would panic since it's already being mutably borrowed in the outer constructor.
1371fn iterator_input<N: AstNode>(input: impl IntoIterator<Item = N>) -> (Vec<N>, Vec<SyntaxNode>) {
1372    input
1373        .into_iter()
1374        .map(|it| {
1375            let syntax = it.syntax().clone();
1376            (it, syntax)
1377        })
1378        .collect()
1379}
