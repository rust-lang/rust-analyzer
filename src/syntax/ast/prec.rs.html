11#[derive(Debug, Clone, Copy, PartialEq, PartialOrd)]
12pub enum ExprPrecedence {
13    // return val, break val, yield val, closures
14    Jump,
15    // = += -= *= /= %= &= |= ^= <<= >>=
16    Assign,
17    // .. ..=
18    Range,
19    // ||
20    LOr,
21    // &&
22    LAnd,
23    // == != < > <= >=
24    Compare,
25    // |
26    BitOr,
27    // ^
28    BitXor,
29    // &
30    BitAnd,
31    // << >>
32    Shift,
33    // + -
34    Sum,
35    // * / %
36    Product,
37    // as
38    Cast,
39    // unary - * ! & &mut
40    Prefix,
41    // function calls, array indexing, field expressions, method calls
42    Postfix,
43    // paths, loops,
44    Unambiguous,
45}

47impl ExprPrecedence {
48    pub fn needs_parentheses_in(self, other: ExprPrecedence) -> bool {
49        match other {
50            ExprPrecedence::Unambiguous => false,
51            // postfix ops have higher precedence than any other operator, so we need to wrap
52            // any inner expression that is below
53            ExprPrecedence::Postfix => self < ExprPrecedence::Postfix,
54            // We need to wrap all binary like things, thats everything below prefix except for
55            // jumps (as those are prefix operations as well)
56            ExprPrecedence::Prefix => ExprPrecedence::Jump < self && self < ExprPrecedence::Prefix,
57            parent => self <= parent,
58        }
59    }
60}

72pub fn precedence(expr: &ast::Expr) -> ExprPrecedence {
73    match expr {
74        Expr::ClosureExpr(closure) => match closure.ret_type() {
75            None => ExprPrecedence::Jump,
76            Some(_) => ExprPrecedence::Unambiguous,
77        },
78
79        Expr::BreakExpr(e) if e.expr().is_some() => ExprPrecedence::Jump,
80        Expr::BecomeExpr(e) if e.expr().is_some() => ExprPrecedence::Jump,
81        Expr::ReturnExpr(e) if e.expr().is_some() => ExprPrecedence::Jump,
82        Expr::YeetExpr(e) if e.expr().is_some() => ExprPrecedence::Jump,
83        Expr::YieldExpr(e) if e.expr().is_some() => ExprPrecedence::Jump,
84
85        Expr::BreakExpr(_)
86        | Expr::BecomeExpr(_)
87        | Expr::ReturnExpr(_)
88        | Expr::YeetExpr(_)
89        | Expr::YieldExpr(_)
90        | Expr::ContinueExpr(_) => ExprPrecedence::Unambiguous,
91
92        Expr::RangeExpr(..) => ExprPrecedence::Range,
93
94        Expr::BinExpr(bin_expr) => match bin_expr.op_kind() {
95            Some(it) => match it {
96                BinaryOp::LogicOp(logic_op) => match logic_op {
97                    ast::LogicOp::And => ExprPrecedence::LAnd,
98                    ast::LogicOp::Or => ExprPrecedence::LOr,
99                },
100                BinaryOp::ArithOp(arith_op) => match arith_op {
101                    ast::ArithOp::Add | ast::ArithOp::Sub => ExprPrecedence::Sum,
102                    ast::ArithOp::Div | ast::ArithOp::Rem | ast::ArithOp::Mul => {
103                        ExprPrecedence::Product
104                    }
105                    ast::ArithOp::Shl | ast::ArithOp::Shr => ExprPrecedence::Shift,
106                    ast::ArithOp::BitXor => ExprPrecedence::BitXor,
107                    ast::ArithOp::BitOr => ExprPrecedence::BitOr,
108                    ast::ArithOp::BitAnd => ExprPrecedence::BitAnd,
109                },
110                BinaryOp::CmpOp(_) => ExprPrecedence::Compare,
111                BinaryOp::Assignment { .. } => ExprPrecedence::Assign,
112            },
113            None => ExprPrecedence::Unambiguous,
114        },
115        Expr::CastExpr(_) => ExprPrecedence::Cast,
116
117        Expr::LetExpr(_) | Expr::PrefixExpr(_) | Expr::RefExpr(_) => ExprPrecedence::Prefix,
118
119        Expr::AwaitExpr(_)
120        | Expr::CallExpr(_)
121        | Expr::FieldExpr(_)
122        | Expr::IndexExpr(_)
123        | Expr::MethodCallExpr(_)
124        | Expr::TryExpr(_) => ExprPrecedence::Postfix,
125
126        Expr::ArrayExpr(_)
127        | Expr::AsmExpr(_)
128        | Expr::BlockExpr(_)
129        | Expr::ForExpr(_)
130        | Expr::FormatArgsExpr(_)
131        | Expr::IfExpr(_)
132        | Expr::Literal(_)
133        | Expr::LoopExpr(_)
134        | Expr::MacroExpr(_)
135        | Expr::MatchExpr(_)
136        | Expr::OffsetOfExpr(_)
137        | Expr::ParenExpr(_)
138        | Expr::PathExpr(_)
139        | Expr::RecordExpr(_)
140        | Expr::TupleExpr(_)
141        | Expr::UnderscoreExpr(_)
142        | Expr::WhileExpr(_) => ExprPrecedence::Unambiguous,
143    }
144}

146fn check_ancestry(ancestor: &SyntaxNode, descendent: &SyntaxNode) -> bool {
147    let bail = || always!(false, "{} is not an ancestor of {}", ancestor, descendent);
148
149    if !ancestor.text_range().contains_range(descendent.text_range()) {
150        return bail();
151    }
152
153    for anc in descendent.ancestors() {
154        if anc == *ancestor {
155            return true;
156        }
157    }
158
159    bail()
160}

173    /// Returns `true` if `self` would need to be wrapped in parentheses given that its parent is `parent`.
174    pub fn needs_parens_in(&self, parent: &SyntaxNode) -> bool {
175        self.needs_parens_in_place_of(parent, self.syntax())
176    }

178    /// Returns `true` if `self` would need to be wrapped in parentheses if it replaces `place_of`
179    /// given that `place_of`'s parent is `parent`.
180    pub fn needs_parens_in_place_of(&self, parent: &SyntaxNode, place_of: &SyntaxNode) -> bool {
181        if !check_ancestry(parent, place_of) {
182            return false;
183        }
184
185        match_ast! {
186            match parent {
187                ast::Expr(e) => self.needs_parens_in_expr(&e, place_of),
188                ast::Stmt(e) => self.needs_parens_in_stmt(Some(&e)),
189                ast::StmtList(_) => self.needs_parens_in_stmt(None),
190                ast::ArgList(_) => false,
191                ast::MatchArm(_) => false,
192                _ => false,
193            }
194        }
195    }

197    fn needs_parens_in_expr(&self, parent: &Expr, place_of: &SyntaxNode) -> bool {
198        // Parentheses are necessary when calling a function-like pointer that is a member of a struct or union
199        // (e.g. `(a.f)()`).
200        let is_parent_call_expr = matches!(parent, ast::Expr::CallExpr(_));
201        let is_field_expr = matches!(self, ast::Expr::FieldExpr(_));
202        if is_parent_call_expr && is_field_expr {
203            return true;
204        }
205
206        // Special-case block weirdness
207        if parent.child_is_followed_by_a_block() {
208            use Expr::*;
209            match self {
210                // Cases like `if return {}` (need parens or else `{}` is returned, instead of being `if`'s body)
211                ReturnExpr(e) if e.expr().is_none() => return true,
212                BreakExpr(e) if e.expr().is_none() => return true,
213                YieldExpr(e) if e.expr().is_none() => return true,
214
215                // Same but with `..{}`
216                RangeExpr(e) if matches!(e.end(), Some(BlockExpr(..))) => return true,
217
218                // Similarly with struct literals, e.g. `if S{} == 1 {}`
219                _ if self.contains_exterior_struct_lit() => return true,
220                _ => {}
221            }
222        }
223
224        // Special-case `return.f()`
225        if self.is_ret_like_with_no_value() && parent.is_postfix() {
226            return false;
227        }
228
229        if self.is_paren_like()
230            || parent.is_paren_like()
231            || self.is_prefix()
232                && (parent.is_prefix()
233                    || !self.is_ordered_before_parent_in_place_of(parent, place_of))
234            || self.is_postfix()
235                && (parent.is_postfix()
236                    || self.is_ordered_before_parent_in_place_of(parent, place_of))
237        {
238            return false;
239        }
240
241        let (left, right, inv) = match self.is_ordered_before_parent_in_place_of(parent, place_of) {
242            true => (self, parent, false),
243            false => (parent, self, true),
244        };
245
246        let (_, left_right_bp) = left.binding_power();
247        let (right_left_bp, _) = right.binding_power();
248
249        (left_right_bp < right_left_bp) ^ inv
250    }

252    fn needs_parens_in_stmt(&self, stmt: Option<&ast::Stmt>) -> bool {
253        use Expr::*;
254
255        // Prevent false-positives in cases like `fn x() -> u8 { ({ 0 } + 1) }`,
256        // `{ { 0 } + 1 }` won't parse -- `{ 0 }` would be parsed as a self-contained stmt,
257        // leaving `+ 1` as a parse error.
258        let mut innermost = self.clone();
259        loop {
260            let next = match &innermost {
261                BinExpr(e) => e.lhs(),
262                CallExpr(e) => e.expr(),
263                CastExpr(e) => e.expr(),
264                IndexExpr(e) => e.base(),
265                _ => break,
266            };
267
268            if let Some(next) = next {
269                innermost = next;
270                if !innermost.requires_semi_to_be_stmt() {
271                    return true;
272                }
273            } else {
274                break;
275            }
276        }
277
278        // Not every expression can be followed by `else` in the `let-else`
279        if let Some(ast::Stmt::LetStmt(e)) = stmt
280            && e.let_else().is_some()
281        {
282            match self {
283                BinExpr(e)
284                    if e.op_kind()
285                        .map(|op| matches!(op, BinaryOp::LogicOp(_)))
286                        .unwrap_or(false) =>
287                {
288                    return true;
289                }
290                _ if self.clone().trailing_brace().is_some() => return true,
291                _ => {}
292            }
293        }
294
295        false
296    }

298    /// Returns left and right so-called "binding powers" of this expression.
299    fn binding_power(&self) -> (u8, u8) {
300        use ast::{ArithOp::*, BinaryOp::*, Expr::*, LogicOp::*};
301
302        match self {
303            // (0, 0)   -- paren-like/nullary
304            // (0, N)   -- prefix
305            // (N, 0)   -- postfix
306            // (N, N)   -- infix, requires parens
307            // (N, N+1) -- infix, left to right associative
308            // (N+1, N) -- infix, right to left associative
309            // N is odd
310            //
311            ContinueExpr(_) => (0, 0),
312
313            ClosureExpr(_) | ReturnExpr(_) | BecomeExpr(_) | YieldExpr(_) | YeetExpr(_)
314            | BreakExpr(_) | OffsetOfExpr(_) | FormatArgsExpr(_) | AsmExpr(_) => (0, 1),
315
316            RangeExpr(_) => (5, 5),
317
318            BinExpr(e) => {
319                // Return a dummy value if we don't know the op
320                let Some(op) = e.op_kind() else { return (0, 0) };
321                match op {
322                    Assignment { .. } => (4, 3),
323                    //
324                    // Ranges are here in order :)
325                    //
326                    LogicOp(op) => match op {
327                        Or => (7, 8),
328                        And => (9, 10),
329                    },
330                    CmpOp(_) => (11, 11),
331                    ArithOp(op) => match op {
332                        BitOr => (13, 14),
333                        BitXor => (15, 16),
334                        BitAnd => (17, 18),
335                        Shl | Shr => (19, 20),
336                        Add | Sub => (21, 22),
337                        Mul | Div | Rem => (23, 24),
338                    },
339                }
340            }
341
342            CastExpr(_) => (25, 26),
343
344            RefExpr(_) | LetExpr(_) | PrefixExpr(_) => (0, 27),
345
346            AwaitExpr(_) | CallExpr(_) | MethodCallExpr(_) | IndexExpr(_) | TryExpr(_)
347            | MacroExpr(_) => (29, 0),
348
349            FieldExpr(_) => (31, 32),
350
351            ArrayExpr(_) | TupleExpr(_) | Literal(_) | PathExpr(_) | ParenExpr(_) | IfExpr(_)
352            | WhileExpr(_) | ForExpr(_) | LoopExpr(_) | MatchExpr(_) | BlockExpr(_)
353            | RecordExpr(_) | UnderscoreExpr(_) => (0, 0),
354        }
355    }

357    fn is_paren_like(&self) -> bool {
358        matches!(self.binding_power(), (0, 0))
359    }
360
361    fn is_prefix(&self) -> bool {
362        matches!(self.binding_power(), (0, 1..))
363    }
364
365    fn is_postfix(&self) -> bool {
366        matches!(self.binding_power(), (1.., 0))
367    }

369    /// Returns `true` if this expression can't be a standalone statement.
370    fn requires_semi_to_be_stmt(&self) -> bool {
371        use Expr::*;
372        !matches!(
373            self,
374            IfExpr(..) | MatchExpr(..) | BlockExpr(..) | WhileExpr(..) | LoopExpr(..) | ForExpr(..)
375        )
376    }

378    /// If an expression ends with `}`, returns the innermost expression ending in this `}`.
379    fn trailing_brace(mut self) -> Option<Expr> {
380        use Expr::*;
381
382        loop {
383            let rhs = match self {
384                RefExpr(e) => e.expr(),
385                BinExpr(e) => e.rhs(),
386                BreakExpr(e) => e.expr(),
387                LetExpr(e) => e.expr(),
388                RangeExpr(e) => e.end(),
389                ReturnExpr(e) => e.expr(),
390                PrefixExpr(e) => e.expr(),
391                YieldExpr(e) => e.expr(),
392                ClosureExpr(e) => e.body(),
393
394                BlockExpr(..) | ForExpr(..) | IfExpr(..) | LoopExpr(..) | MatchExpr(..)
395                | RecordExpr(..) | WhileExpr(..) => break Some(self),
396                _ => break None,
397            };
398
399            self = rhs?;
400        }
401    }

403    /// Expressions that syntactically contain an "exterior" struct literal i.e., not surrounded by any
404    /// parens or other delimiters, e.g., `X { y: 1 }`, `X { y: 1 }.method()`, `foo == X { y: 1 }` and
405    /// `X { y: 1 } == foo` all do, but `(X { y: 1 }) == foo` does not.
406    fn contains_exterior_struct_lit(&self) -> bool {
407        return contains_exterior_struct_lit_inner(self).is_some();
408
409        fn contains_exterior_struct_lit_inner(expr: &Expr) -> Option<()> {
410            use Expr::*;
411
412            match expr {
413                RecordExpr(..) => Some(()),
414
415                // X { y: 1 } + X { y: 2 }
416                BinExpr(e) => e
417                    .lhs()
418                    .as_ref()
419                    .and_then(contains_exterior_struct_lit_inner)
420                    .or_else(|| e.rhs().as_ref().and_then(contains_exterior_struct_lit_inner)),
421
422                // `&X { y: 1 }`, `X { y: 1 }.y`, `X { y: 1 }.bar(...)`, etc
423                IndexExpr(e) => contains_exterior_struct_lit_inner(&e.base()?),
424                AwaitExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
425                PrefixExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
426                CastExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
427                FieldExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
428                MethodCallExpr(e) => contains_exterior_struct_lit_inner(&e.receiver()?),
429
430                _ => None,
431            }
432        }
433    }

435    /// Returns true if self is one of `return`, `break`, `continue` or `yield` with **no associated value**.
436    pub fn is_ret_like_with_no_value(&self) -> bool {
437        use Expr::*;
438
439        match self {
440            ReturnExpr(e) => e.expr().is_none(),
441            BreakExpr(e) => e.expr().is_none(),
442            ContinueExpr(_) => true,
443            YieldExpr(e) => e.expr().is_none(),
444            BecomeExpr(e) => e.expr().is_none(),
445            _ => false,
446        }
447    }

449    fn is_ordered_before_parent_in_place_of(&self, parent: &Expr, place_of: &SyntaxNode) -> bool {
450        use Expr::*;
451        use rowan::TextSize;
452
453        let self_range = self.syntax().text_range();
454        let place_of_range = place_of.text_range();
455
456        let self_order_adjusted = order(self) - self_range.start() + place_of_range.start();
457
458        let parent_order = order(parent);
459        let parent_order_adjusted = if parent_order <= place_of_range.start() {
460            parent_order
461        } else if parent_order >= place_of_range.end() {
462            parent_order - place_of_range.len() + self_range.len()
463        } else {
464            return false;
465        };
466
467        return self_order_adjusted < parent_order_adjusted;
468
469        /// Returns text range that can be used to compare two expression for order (which goes first).
470        fn order(this: &Expr) -> TextSize {
471            // For non-paren-like operators: get the operator itself
472            let token = match this {
473                RangeExpr(e) => e.op_token(),
474                BinExpr(e) => e.op_token(),
475                CastExpr(e) => e.as_token(),
476                FieldExpr(e) => e.dot_token(),
477                AwaitExpr(e) => e.dot_token(),
478                BreakExpr(e) => e.break_token(),
479                CallExpr(e) => e.arg_list().and_then(|args| args.l_paren_token()),
480                ClosureExpr(e) => e.param_list().and_then(|params| params.l_paren_token()),
481                ContinueExpr(e) => e.continue_token(),
482                IndexExpr(e) => e.l_brack_token(),
483                MethodCallExpr(e) => e.dot_token(),
484                PrefixExpr(e) => e.op_token(),
485                RefExpr(e) => e.amp_token(),
486                ReturnExpr(e) => e.return_token(),
487                BecomeExpr(e) => e.become_token(),
488                TryExpr(e) => e.question_mark_token(),
489                YieldExpr(e) => e.yield_token(),
490                YeetExpr(e) => e.do_token(),
491                LetExpr(e) => e.let_token(),
492                OffsetOfExpr(e) => e.builtin_token(),
493                FormatArgsExpr(e) => e.builtin_token(),
494                AsmExpr(e) => e.builtin_token(),
495                ArrayExpr(_) | TupleExpr(_) | Literal(_) | PathExpr(_) | ParenExpr(_)
496                | IfExpr(_) | WhileExpr(_) | ForExpr(_) | LoopExpr(_) | MatchExpr(_)
497                | BlockExpr(_) | RecordExpr(_) | UnderscoreExpr(_) | MacroExpr(_) => None,
498            };
499
500            token.map(|t| t.text_range()).unwrap_or_else(|| this.syntax().text_range()).start()
501        }
502    }

504    fn child_is_followed_by_a_block(&self) -> bool {
505        use Expr::*;
506
507        match self {
508            ArrayExpr(_) | AwaitExpr(_) | BlockExpr(_) | CallExpr(_) | CastExpr(_)
509            | ClosureExpr(_) | FieldExpr(_) | IndexExpr(_) | Literal(_) | LoopExpr(_)
510            | MacroExpr(_) | MethodCallExpr(_) | ParenExpr(_) | PathExpr(_) | RecordExpr(_)
511            | TryExpr(_) | TupleExpr(_) | UnderscoreExpr(_) | OffsetOfExpr(_)
512            | FormatArgsExpr(_) | AsmExpr(_) => false,
513
514            // For BinExpr and RangeExpr this is technically wrong -- the child can be on the left...
515            BinExpr(_) | RangeExpr(_) | BreakExpr(_) | ContinueExpr(_) | PrefixExpr(_)
516            | RefExpr(_) | ReturnExpr(_) | BecomeExpr(_) | YieldExpr(_) | YeetExpr(_)
517            | LetExpr(_) => self
518                .syntax()
519                .parent()
520                .and_then(Expr::cast)
521                .map(|e| e.child_is_followed_by_a_block())
522                .unwrap_or(false),
523
524            ForExpr(_) | IfExpr(_) | MatchExpr(_) | WhileExpr(_) => true,
525        }
526    }
