12#[derive(Debug, Clone, Copy, PartialEq, PartialOrd)]
13pub enum ExprPrecedence {
14    // return, break, continue, yield, yeet, become (with or without value)
15    Jump,
16    // = += -= *= /= %= &= |= ^= <<= >>=
17    Assign,
18    // .. ..=
19    Range,
20    // ||
21    LOr,
22    // &&
23    LAnd,
24    // == != < > <= >=
25    Compare,
26    // |
27    BitOr,
28    // ^
29    BitXor,
30    // &
31    BitAnd,
32    // << >>
33    Shift,
34    // + -
35    Sum,
36    // * / %
37    Product,
38    // as
39    Cast,
40    // unary - * ! & &mut
41    Prefix,
42    // function calls, array indexing, field expressions, method calls
43    Postfix,
44    // paths, loops,
45    Unambiguous,
46}

48impl ExprPrecedence {
49    pub fn needs_parentheses_in(self, other: ExprPrecedence) -> bool {
50        match other {
51            ExprPrecedence::Unambiguous => false,
52            // postfix ops have higher precedence than any other operator, so we need to wrap
53            // any inner expression that is below
54            ExprPrecedence::Postfix => self < ExprPrecedence::Postfix,
55            // We need to wrap all binary like things, thats everything below prefix except for
56            // jumps (as those are prefix operations as well)
57            ExprPrecedence::Prefix => ExprPrecedence::Jump < self && self < ExprPrecedence::Prefix,
58            parent => self <= parent,
59        }
60    }
61}

73pub fn precedence(expr: &ast::Expr) -> ExprPrecedence {
74    match expr {
75        Expr::ClosureExpr(closure) => match closure.ret_type() {
76            None => ExprPrecedence::Jump,
77            Some(_) => ExprPrecedence::Unambiguous,
78        },
79
80        Expr::BreakExpr(_)
81        | Expr::BecomeExpr(_)
82        | Expr::ReturnExpr(_)
83        | Expr::YeetExpr(_)
84        | Expr::YieldExpr(_)
85        | Expr::ContinueExpr(_) => ExprPrecedence::Jump,
86
87        Expr::RangeExpr(..) => ExprPrecedence::Range,
88
89        Expr::BinExpr(bin_expr) => match bin_expr.op_kind() {
90            Some(it) => match it {
91                BinaryOp::LogicOp(logic_op) => match logic_op {
92                    ast::LogicOp::And => ExprPrecedence::LAnd,
93                    ast::LogicOp::Or => ExprPrecedence::LOr,
94                },
95                BinaryOp::ArithOp(arith_op) => match arith_op {
96                    ast::ArithOp::Add | ast::ArithOp::Sub => ExprPrecedence::Sum,
97                    ast::ArithOp::Div | ast::ArithOp::Rem | ast::ArithOp::Mul => {
98                        ExprPrecedence::Product
99                    }
100                    ast::ArithOp::Shl | ast::ArithOp::Shr => ExprPrecedence::Shift,
101                    ast::ArithOp::BitXor => ExprPrecedence::BitXor,
102                    ast::ArithOp::BitOr => ExprPrecedence::BitOr,
103                    ast::ArithOp::BitAnd => ExprPrecedence::BitAnd,
104                },
105                BinaryOp::CmpOp(_) => ExprPrecedence::Compare,
106                BinaryOp::Assignment { .. } => ExprPrecedence::Assign,
107            },
108            None => ExprPrecedence::Unambiguous,
109        },
110        Expr::CastExpr(_) => ExprPrecedence::Cast,
111
112        Expr::LetExpr(_) | Expr::PrefixExpr(_) | Expr::RefExpr(_) => ExprPrecedence::Prefix,
113
114        Expr::AwaitExpr(_)
115        | Expr::CallExpr(_)
116        | Expr::FieldExpr(_)
117        | Expr::IndexExpr(_)
118        | Expr::MethodCallExpr(_)
119        | Expr::TryExpr(_) => ExprPrecedence::Postfix,
120
121        Expr::ArrayExpr(_)
122        | Expr::AsmExpr(_)
123        | Expr::BlockExpr(_)
124        | Expr::ForExpr(_)
125        | Expr::FormatArgsExpr(_)
126        | Expr::IfExpr(_)
127        | Expr::Literal(_)
128        | Expr::LoopExpr(_)
129        | Expr::MacroExpr(_)
130        | Expr::MatchExpr(_)
131        | Expr::OffsetOfExpr(_)
132        | Expr::ParenExpr(_)
133        | Expr::PathExpr(_)
134        | Expr::RecordExpr(_)
135        | Expr::TupleExpr(_)
136        | Expr::UnderscoreExpr(_)
137        | Expr::WhileExpr(_) => ExprPrecedence::Unambiguous,
138    }
139}

141fn check_ancestry(ancestor: &SyntaxNode, descendent: &SyntaxNode) -> bool {
142    let bail = || always!(false, "{} is not an ancestor of {}", ancestor, descendent);
143
144    if !ancestor.text_range().contains_range(descendent.text_range()) {
145        return bail();
146    }
147
148    for anc in descendent.ancestors() {
149        if anc == *ancestor {
150            return true;
151        }
152    }
153
154    bail()
155}

157fn next_token_of(node: &SyntaxNode) -> Option<ast::SyntaxToken> {
158    let last = node.last_token()?;
159    skip_trivia_token(last.next_token()?, Direction::Next)
160}

173    /// Returns `true` if `self` would need to be wrapped in parentheses given that its parent is `parent`.
174    pub fn needs_parens_in(&self, parent: &SyntaxNode) -> bool {
175        self.needs_parens_in_place_of(parent, self.syntax())
176    }

178    /// Returns `true` if `self` would need to be wrapped in parentheses if it replaces `place_of`
179    /// given that `place_of`'s parent is `parent`.
180    pub fn needs_parens_in_place_of(&self, parent: &SyntaxNode, place_of: &SyntaxNode) -> bool {
181        if !check_ancestry(parent, place_of) {
182            return false;
183        }
184
185        match_ast! {
186            match parent {
187                ast::Expr(e) => self.needs_parens_in_expr(&e, place_of),
188                ast::Stmt(e) => self.needs_parens_in_stmt(Some(&e)),
189                ast::StmtList(_) => self.needs_parens_in_stmt(None),
190                ast::ArgList(_) => false,
191                ast::MatchArm(_) => false,
192                _ => false,
193            }
194        }
195    }

197    fn needs_parens_in_expr(&self, parent: &Expr, place_of: &SyntaxNode) -> bool {
198        // Parentheses are necessary when calling a function-like pointer that is a member of a struct or union
199        // (e.g. `(a.f)()`).
200        let is_parent_call_expr = matches!(parent, ast::Expr::CallExpr(_));
201        let is_field_expr = matches!(self, ast::Expr::FieldExpr(_));
202        if is_parent_call_expr && is_field_expr {
203            return true;
204        }
205        let place_of_parent =
206            || place_of.ancestors().find(|it| it.parent().is_none_or(|p| &p == parent.syntax()));
207
208        // Special-case block weirdness
209        if parent.child_is_followed_by_a_block() {
210            use Expr::*;
211            match self {
212                // Cases like `if return {}` (need parens or else `{}` is returned, instead of being `if`'s body)
213                ReturnExpr(e) if e.expr().is_none() => return true,
214                BreakExpr(e) if e.expr().is_none() => return true,
215                YieldExpr(e) if e.expr().is_none() => return true,
216
217                // Same but with `..{}`
218                RangeExpr(e) if matches!(e.end(), Some(BlockExpr(..))) => return true,
219
220                // Similarly with struct literals, e.g. `if S{} == 1 {}`
221                _ if self.contains_exterior_struct_lit() => return true,
222                _ => {}
223            }
224        }
225
226        // Special-case `return.f()`
227        if self.is_ret_like_with_no_value() && parent.is_postfix() {
228            return false;
229        }
230
231        // Keep parens when a ret-like expr is followed by `||` or `&&`.
232        // For `||`, removing parens could reparse as `<ret-like> || <closure>`.
233        // For `&&`, we avoid introducing `<ret-like> && <expr>` into a binary chain.
234
235        if self.precedence() == ExprPrecedence::Jump
236            && let Some(node) = place_of_parent()
237            && let Some(next) = next_token_of(&node)
238            && matches!(next.kind(), T![||] | T![&&])
239        {
240            return true;
241        }
242
243        // Special-case `2 as x < 3`
244        if let ast::Expr::CastExpr(it) = self
245            && let Some(ty) = it.ty()
246            && ty.syntax().last_token().and_then(|it| ast::NameLike::cast(it.parent()?)).is_some()
247            && let Some(node) = place_of_parent()
248            && let Some(next) = next_token_of(&node)
249            && matches!(next.kind(), T![<] | T![<<])
250        {
251            return true;
252        }
253
254        if self.is_paren_like()
255            || parent.is_paren_like()
256            || self.is_prefix()
257                && (parent.is_prefix()
258                    || !self.is_ordered_before_parent_in_place_of(parent, place_of))
259            || self.is_postfix()
260                && (parent.is_postfix()
261                    || self.is_ordered_before_parent_in_place_of(parent, place_of))
262        {
263            return false;
264        }
265
266        let (left, right, inv) = match self.is_ordered_before_parent_in_place_of(parent, place_of) {
267            true => (self, parent, false),
268            false => (parent, self, true),
269        };
270
271        let (_, left_right_bp) = left.binding_power();
272        let (right_left_bp, _) = right.binding_power();
273
274        (left_right_bp < right_left_bp) ^ inv
275    }

277    fn needs_parens_in_stmt(&self, stmt: Option<&ast::Stmt>) -> bool {
278        use Expr::*;
279
280        // Prevent false-positives in cases like `fn x() -> u8 { ({ 0 } + 1) }`,
281        // `{ { 0 } + 1 }` won't parse -- `{ 0 }` would be parsed as a self-contained stmt,
282        // leaving `+ 1` as a parse error.
283        let mut innermost = self.clone();
284        loop {
285            let next = match &innermost {
286                BinExpr(e) => e.lhs(),
287                CallExpr(e) => e.expr(),
288                CastExpr(e) => e.expr(),
289                IndexExpr(e) => e.base(),
290                _ => break,
291            };
292
293            if let Some(next) = next {
294                innermost = next;
295                if !innermost.requires_semi_to_be_stmt() {
296                    return true;
297                }
298            } else {
299                break;
300            }
301        }
302
303        // Not every expression can be followed by `else` in the `let-else`
304        if let Some(ast::Stmt::LetStmt(e)) = stmt
305            && e.let_else().is_some()
306        {
307            match self {
308                BinExpr(e)
309                    if e.op_kind()
310                        .map(|op| matches!(op, BinaryOp::LogicOp(_)))
311                        .unwrap_or(false) =>
312                {
313                    return true;
314                }
315                _ if self.clone().trailing_brace().is_some() => return true,
316                _ => {}
317            }
318        }
319
320        false
321    }

323    /// Returns left and right so-called "binding powers" of this expression.
324    fn binding_power(&self) -> (u8, u8) {
325        use ast::{ArithOp::*, BinaryOp::*, Expr::*, LogicOp::*};
326
327        match self {
328            // (0, 0)   -- paren-like/nullary
329            // (0, N)   -- prefix
330            // (N, 0)   -- postfix
331            // (N, N)   -- infix, requires parens
332            // (N, N+1) -- infix, left to right associative
333            // (N+1, N) -- infix, right to left associative
334            // N is odd
335            //
336            ContinueExpr(_) => (0, 0),
337
338            ClosureExpr(_) | ReturnExpr(_) | BecomeExpr(_) | YieldExpr(_) | YeetExpr(_)
339            | BreakExpr(_) | OffsetOfExpr(_) | FormatArgsExpr(_) | AsmExpr(_) => (0, 1),
340
341            RangeExpr(_) => (5, 5),
342
343            BinExpr(e) => {
344                // Return a dummy value if we don't know the op
345                let Some(op) = e.op_kind() else { return (0, 0) };
346                match op {
347                    Assignment { .. } => (4, 3),
348                    //
349                    // Ranges are here in order :)
350                    //
351                    LogicOp(op) => match op {
352                        Or => (7, 8),
353                        And => (9, 10),
354                    },
355                    CmpOp(_) => (11, 11),
356                    ArithOp(op) => match op {
357                        BitOr => (13, 14),
358                        BitXor => (15, 16),
359                        BitAnd => (17, 18),
360                        Shl | Shr => (19, 20),
361                        Add | Sub => (21, 22),
362                        Mul | Div | Rem => (23, 24),
363                    },
364                }
365            }
366
367            CastExpr(_) => (25, 26),
368
369            RefExpr(_) | LetExpr(_) | PrefixExpr(_) => (0, 27),
370
371            AwaitExpr(_) | CallExpr(_) | MethodCallExpr(_) | IndexExpr(_) | TryExpr(_)
372            | MacroExpr(_) => (29, 0),
373
374            FieldExpr(_) => (31, 32),
375
376            ArrayExpr(_) | TupleExpr(_) | Literal(_) | PathExpr(_) | ParenExpr(_) | IfExpr(_)
377            | WhileExpr(_) | ForExpr(_) | LoopExpr(_) | MatchExpr(_) | BlockExpr(_)
378            | RecordExpr(_) | UnderscoreExpr(_) => (0, 0),
379        }
380    }

382    fn is_paren_like(&self) -> bool {
383        matches!(self.binding_power(), (0, 0))
384    }

386    fn is_prefix(&self) -> bool {
387        matches!(self.binding_power(), (0, 1..))
388    }
389
390    fn is_postfix(&self) -> bool {
391        matches!(self.binding_power(), (1.., 0))
392    }

395    fn requires_semi_to_be_stmt(&self) -> bool {
396        use Expr::*;

399            IfExpr(..) | MatchExpr(..) | BlockExpr(..) | WhileExpr(..) | LoopExpr(..) | ForExpr(..)

403    /// If an expression ends with `}`, returns the innermost expression ending in this `}`.
404    fn trailing_brace(mut self) -> Option<Expr> {
405        use Expr::*;
406
407        loop {
408            let rhs = match self {
409                RefExpr(e) => e.expr(),
410                BinExpr(e) => e.rhs(),
411                BreakExpr(e) => e.expr(),
412                LetExpr(e) => e.expr(),
413                RangeExpr(e) => e.end(),
414                ReturnExpr(e) => e.expr(),
415                PrefixExpr(e) => e.expr(),
416                YieldExpr(e) => e.expr(),
417                ClosureExpr(e) => e.body(),
418
419                BlockExpr(..) | ForExpr(..) | IfExpr(..) | LoopExpr(..) | MatchExpr(..)
420                | RecordExpr(..) | WhileExpr(..) => break Some(self),
421                _ => break None,
422            };
423
424            self = rhs?;
425        }
426    }

428    /// Expressions that syntactically contain an "exterior" struct literal i.e., not surrounded by any
429    /// parens or other delimiters, e.g., `X { y: 1 }`, `X { y: 1 }.method()`, `foo == X { y: 1 }` and
430    /// `X { y: 1 } == foo` all do, but `(X { y: 1 }) == foo` does not.
431    fn contains_exterior_struct_lit(&self) -> bool {
432        return contains_exterior_struct_lit_inner(self).is_some();
433
434        fn contains_exterior_struct_lit_inner(expr: &Expr) -> Option<()> {
435            use Expr::*;
436
437            match expr {
438                RecordExpr(..) => Some(()),
439
440                // X { y: 1 } + X { y: 2 }
441                BinExpr(e) => e
442                    .lhs()
443                    .as_ref()
444                    .and_then(contains_exterior_struct_lit_inner)
445                    .or_else(|| e.rhs().as_ref().and_then(contains_exterior_struct_lit_inner)),
446
447                // `&X { y: 1 }`, `X { y: 1 }.y`, `X { y: 1 }.bar(...)`, etc
448                IndexExpr(e) => contains_exterior_struct_lit_inner(&e.base()?),
449                AwaitExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
450                PrefixExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
451                CastExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
452                FieldExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
453                MethodCallExpr(e) => contains_exterior_struct_lit_inner(&e.receiver()?),
454
455                _ => None,
456            }
457        }
458    }

460    /// Returns true if self is one of `return`, `break`, `continue` or `yield` with **no associated value**.
461    pub fn is_ret_like_with_no_value(&self) -> bool {
462        use Expr::*;
463
464        match self {
465            ReturnExpr(e) => e.expr().is_none(),
466            BreakExpr(e) => e.expr().is_none(),
467            ContinueExpr(_) => true,
468            YieldExpr(e) => e.expr().is_none(),
469            BecomeExpr(e) => e.expr().is_none(),
470            _ => false,
471        }
472    }

474    fn is_ordered_before_parent_in_place_of(&self, parent: &Expr, place_of: &SyntaxNode) -> bool {
475        use Expr::*;
476        use rowan::TextSize;
477
478        let self_range = self.syntax().text_range();
479        let place_of_range = place_of.text_range();
480
481        let self_order_adjusted = order(self) - self_range.start() + place_of_range.start();
482
483        let parent_order = order(parent);
484        let parent_order_adjusted = if parent_order <= place_of_range.start() {
485            parent_order
486        } else if parent_order >= place_of_range.end() {
487            parent_order - place_of_range.len() + self_range.len()
488        } else {
489            return false;
490        };
491
492        return self_order_adjusted < parent_order_adjusted;
493
494        /// Returns text range that can be used to compare two expression for order (which goes first).
495        fn order(this: &Expr) -> TextSize {
496            // For non-paren-like operators: get the operator itself
497            let token = match this {
498                RangeExpr(e) => e.op_token(),
499                BinExpr(e) => e.op_token(),
500                CastExpr(e) => e.as_token(),
501                FieldExpr(e) => e.dot_token(),
502                AwaitExpr(e) => e.dot_token(),
503                BreakExpr(e) => e.break_token(),
504                CallExpr(e) => e.arg_list().and_then(|args| args.l_paren_token()),
505                ClosureExpr(e) => e.param_list().and_then(|params| params.l_paren_token()),
506                ContinueExpr(e) => e.continue_token(),
507                IndexExpr(e) => e.l_brack_token(),
508                MethodCallExpr(e) => e.dot_token(),
509                PrefixExpr(e) => e.op_token(),
510                RefExpr(e) => e.amp_token(),
511                ReturnExpr(e) => e.return_token(),
512                BecomeExpr(e) => e.become_token(),
513                TryExpr(e) => e.question_mark_token(),
514                YieldExpr(e) => e.yield_token(),
515                YeetExpr(e) => e.do_token(),
516                LetExpr(e) => e.let_token(),
517                OffsetOfExpr(e) => e.builtin_token(),
518                FormatArgsExpr(e) => e.builtin_token(),
519                AsmExpr(e) => e.builtin_token(),
520                ArrayExpr(_) | TupleExpr(_) | Literal(_) | PathExpr(_) | ParenExpr(_)
521                | IfExpr(_) | WhileExpr(_) | ForExpr(_) | LoopExpr(_) | MatchExpr(_)
522                | BlockExpr(_) | RecordExpr(_) | UnderscoreExpr(_) | MacroExpr(_) => None,
523            };
524
525            token.map(|t| t.text_range()).unwrap_or_else(|| this.syntax().text_range()).start()
526        }
527    }

529    fn child_is_followed_by_a_block(&self) -> bool {
530        use Expr::*;
531
532        match self {
533            ArrayExpr(_) | AwaitExpr(_) | BlockExpr(_) | CallExpr(_) | CastExpr(_)
534            | ClosureExpr(_) | FieldExpr(_) | IndexExpr(_) | Literal(_) | LoopExpr(_)
535            | MacroExpr(_) | MethodCallExpr(_) | ParenExpr(_) | PathExpr(_) | RecordExpr(_)
536            | TryExpr(_) | TupleExpr(_) | UnderscoreExpr(_) | OffsetOfExpr(_)
537            | FormatArgsExpr(_) | AsmExpr(_) => false,
538
539            // For BinExpr and RangeExpr this is technically wrong -- the child can be on the left...
540            BinExpr(_) | RangeExpr(_) | BreakExpr(_) | ContinueExpr(_) | PrefixExpr(_)
541            | RefExpr(_) | ReturnExpr(_) | BecomeExpr(_) | YieldExpr(_) | YeetExpr(_)
542            | LetExpr(_) => self
543                .syntax()
544                .parent()
545                .and_then(Expr::cast)
546                .map(|e| e.child_is_followed_by_a_block())
547                .unwrap_or(false),
548
549            ForExpr(_) | IfExpr(_) | MatchExpr(_) | WhileExpr(_) => true,
550        }
551    }
