24impl GenericParamsOwnerEdit for ast::Fn {

26        match self.generic_param_list() {
27            Some(it) => it,
28            None => {

45            let position = if let Some(ty) = self.ret_type() {
46                ted::Position::after(ty.syntax())
47            } else if let Some(param_list) = self.param_list() {
48                ted::Position::after(param_list.syntax())
49            } else {
50                ted::Position::last_child_of(self.syntax())
51            };
52            create_where_clause(position);
53        }
54        self.where_clause().unwrap()

227pub trait AttrsOwnerEdit: ast::HasAttrs {
228    fn remove_attrs_and_docs(&self) {
229        remove_attrs_and_docs(self.syntax());
230
231        fn remove_attrs_and_docs(node: &SyntaxNode) {
232            let mut remove_next_ws = false;
233            for child in node.children_with_tokens() {
234                match child.kind() {
235                    ATTR | COMMENT => {
236                        remove_next_ws = true;
237                        child.detach();
238                        continue;
239                    }
240                    WHITESPACE if remove_next_ws => {
241                        child.detach();
242                    }
243                    _ => (),
244                }
245                remove_next_ws = false;
246            }
247        }
248    }
249}

253impl ast::GenericParamList {
254    pub fn add_generic_param(&self, generic_param: ast::GenericParam) {
255        match self.generic_params().last() {
256            Some(last_param) => {
257                let position = ted::Position::after(last_param.syntax());
258                let elements = vec![
259                    make::token(T![,]).into(),
260                    make::tokens::single_space().into(),
261                    generic_param.syntax().clone().into(),
262                ];
263                ted::insert_all(position, elements);
264            }
265            None => {
266                let after_l_angle = ted::Position::after(self.l_angle_token().unwrap());
267                ted::insert(after_l_angle, generic_param.syntax());
268            }
269        }
270    }

272    /// Removes the existing generic param
273    pub fn remove_generic_param(&self, generic_param: ast::GenericParam) {
274        if let Some(previous) = generic_param.syntax().prev_sibling() {
275            if let Some(next_token) = previous.next_sibling_or_token() {
276                ted::remove_all(next_token..=generic_param.syntax().clone().into());
277            }
278        } else if let Some(next) = generic_param.syntax().next_sibling() {
279            if let Some(next_token) = next.prev_sibling_or_token() {
280                ted::remove_all(generic_param.syntax().clone().into()..=next_token);
281            }
282        } else {
283            ted::remove(generic_param.syntax());
284        }
285    }

287    /// Find the params corresponded to generic arg
288    pub fn find_generic_arg(&self, generic_arg: &ast::GenericArg) -> Option<GenericParam> {
289        self.generic_params().find_map(move |param| match (&param, &generic_arg) {
290            (ast::GenericParam::LifetimeParam(a), ast::GenericArg::LifetimeArg(b)) => {
291                (a.lifetime()?.lifetime_ident_token()?.text()
292                    == b.lifetime()?.lifetime_ident_token()?.text())
293                .then_some(param)
294            }
295            (ast::GenericParam::TypeParam(a), ast::GenericArg::TypeArg(b)) => {
296                debug_assert_eq!(b.syntax().first_token(), b.syntax().last_token());
297                (a.name()?.text() == b.syntax().first_token()?.text()).then_some(param)
298            }
299            (ast::GenericParam::ConstParam(a), ast::GenericArg::TypeArg(b)) => {
300                debug_assert_eq!(b.syntax().first_token(), b.syntax().last_token());
301                (a.name()?.text() == b.syntax().first_token()?.text()).then_some(param)
302            }
303            _ => None,
304        })
305    }
306
307    /// Removes the corresponding generic arg
308    pub fn remove_generic_arg(&self, generic_arg: &ast::GenericArg) {
309        let param_to_remove = self.find_generic_arg(generic_arg);
310
311        if let Some(param) = &param_to_remove {
312            self.remove_generic_param(param.clone());
313        }
314    }
315
316    /// Constructs a matching [`ast::GenericArgList`]
317    pub fn to_generic_args(&self) -> ast::GenericArgList {
318        let args = self.generic_params().filter_map(|param| match param {
319            ast::GenericParam::LifetimeParam(it) => {
320                Some(ast::GenericArg::LifetimeArg(make::lifetime_arg(it.lifetime()?)))
321            }
322            ast::GenericParam::TypeParam(it) => {
323                Some(ast::GenericArg::TypeArg(make::type_arg(make::ext::ty_name(it.name()?))))
324            }
325            ast::GenericParam::ConstParam(it) => {
326                // Name-only const params get parsed as `TypeArg`s
327                Some(ast::GenericArg::TypeArg(make::type_arg(make::ext::ty_name(it.name()?))))
328            }
329        });
330
331        make::generic_arg_list(args)
332    }

336    pub fn add_predicate(&self, predicate: ast::WherePred) {
337        if let Some(pred) = self.predicates().last()
338            && !pred.syntax().siblings_with_tokens(Direction::Next).any(|it| it.kind() == T![,])
339        {
340            ted::append_child_raw(self.syntax(), make::token(T![,]));

346        if let Some(previous) = predicate.syntax().prev_sibling() {
347            if let Some(next_token) = previous.next_sibling_or_token() {
348                ted::remove_all(next_token..=predicate.syntax().clone().into());
349            }
350        } else if let Some(next) = predicate.syntax().next_sibling() {
351            if let Some(next_token) = next.prev_sibling_or_token() {
352                ted::remove_all(predicate.syntax().clone().into()..=next_token);
353            }
354        } else {
355            ted::remove(predicate.syntax());
356        }

373impl Removable for ast::UseTree {
374    fn remove(&self) {
375        for dir in [Direction::Next, Direction::Prev] {
376            if let Some(next_use_tree) = neighbor(self, dir) {
377                let separators = self
378                    .syntax()
379                    .siblings_with_tokens(dir)
380                    .skip(1)
381                    .take_while(|it| it.as_node() != Some(next_use_tree.syntax()));
382                ted::remove_all_iter(separators);
383                break;
384            }
385        }
386        ted::remove(self.syntax());
387    }

390impl ast::UseTree {
391    /// Deletes the usetree node represented by the input. Recursively removes parents, including use nodes that become empty.
392    pub fn remove_recursive(self) {
393        let parent = self.syntax().parent();
394
395        self.remove();
396
397        if let Some(u) = parent.clone().and_then(ast::Use::cast) {
398            if u.use_tree().is_none() {
399                u.remove();
400            }
401        } else if let Some(u) = parent.and_then(ast::UseTreeList::cast) {
402            if u.use_trees().next().is_none() {
403                let parent = u.syntax().parent().and_then(ast::UseTree::cast);
404                if let Some(u) = parent {
405                    u.remove_recursive();
406                }
407            }
408            u.remove_unnecessary_braces();
409        }
410    }

429    /// Splits off the given prefix, making it the path component of the use tree,
430    /// appending the rest of the path to all UseTreeList items.
431    ///
432    /// # Examples
433    ///
434    /// `prefix$0::suffix` -> `prefix::{suffix}`
435    ///
436    /// `prefix$0` -> `prefix::{self}`
437    ///
438    /// `prefix$0::*` -> `prefix::{*}`
439    pub fn split_prefix(&self, prefix: &ast::Path) {
440        debug_assert_eq!(self.path(), Some(prefix.top_path()));
441        let path = self.path().unwrap();
442        if &path == prefix && self.use_tree_list().is_none() {
443            if self.star_token().is_some() {
444                // path$0::* -> *
445                if let Some(a) = self.coloncolon_token() {
446                    ted::remove(a)
447                }
448                ted::remove(prefix.syntax());
449            } else {
450                // path$0 -> self
451                let self_suffix =
452                    make::path_unqualified(make::path_segment_self()).clone_for_update();
453                ted::replace(path.syntax(), self_suffix.syntax());
454            }
455        } else if split_path_prefix(prefix).is_none() {
456            return;
457        }
458        // At this point, prefix path is detached; _self_ use tree has suffix path.
459        // Next, transform 'suffix' use tree into 'prefix::{suffix}'
460        let subtree = self.clone_subtree().clone_for_update();
461        ted::remove_all_iter(self.syntax().children_with_tokens());
462        ted::insert(ted::Position::first_child_of(self.syntax()), prefix.syntax());
463        self.get_or_create_use_tree_list().add_use_tree(subtree);
464
465        fn split_path_prefix(prefix: &ast::Path) -> Option<()> {
466            let parent = prefix.parent_path()?;
467            let segment = parent.segment()?;
468            if algo::has_errors(segment.syntax()) {
469                return None;
470            }
471            for p in successors(parent.parent_path(), |it| it.parent_path()) {
472                p.segment()?;
473            }
474            if let Some(a) = prefix.parent_path().and_then(|p| p.coloncolon_token()) {
475                ted::remove(a)
476            }
477            ted::remove(prefix.syntax());
478            Some(())
479        }
480    }
481
482    /// Wraps the use tree in use tree list with no top level path (if it isn't already).
483    ///
484    /// # Examples
485    ///
486    /// `foo::bar` -> `{foo::bar}`
487    ///
488    /// `{foo::bar}` -> `{foo::bar}`
489    pub fn wrap_in_tree_list(&self) -> Option<()> {
490        if self.use_tree_list().is_some()
491            && self.path().is_none()
492            && self.star_token().is_none()
493            && self.rename().is_none()
494        {
495            return None;
496        }
497        let subtree = self.clone_subtree().clone_for_update();
498        ted::remove_all_iter(self.syntax().children_with_tokens());
499        ted::append_child(
500            self.syntax(),
501            make::use_tree_list(once(subtree)).clone_for_update().syntax(),
502        );
503        Some(())
504    }
505}

530impl Removable for ast::Use {
531    fn remove(&self) {
532        let next_ws = self
533            .syntax()
534            .next_sibling_or_token()
535            .and_then(|it| it.into_token())
536            .and_then(ast::Whitespace::cast);
537        if let Some(next_ws) = next_ws {
538            let ws_text = next_ws.syntax().text();
539            if let Some(rest) = ws_text.strip_prefix('\n') {
540                if rest.is_empty() {
541                    ted::remove(next_ws.syntax());
542                } else {
543                    ted::replace(next_ws.syntax(), make::tokens::whitespace(rest));
544                }
545            }
546        }
547        let prev_ws = self
548            .syntax()
549            .prev_sibling_or_token()
550            .and_then(|it| it.into_token())
551            .and_then(ast::Whitespace::cast);
552        if let Some(prev_ws) = prev_ws {
553            let ws_text = prev_ws.syntax().text();
554            let prev_newline = ws_text.rfind('\n').map(|x| x + 1).unwrap_or(0);
555            let rest = &ws_text[0..prev_newline];
556            if rest.is_empty() {
557                ted::remove(prev_ws.syntax());
558            } else {
559                ted::replace(prev_ws.syntax(), make::tokens::whitespace(rest));
560            }
561        }
562
563        ted::remove(self.syntax());
564    }
565}

577impl ast::AssocItemList {
578    /// Adds a new associated item after all of the existing associated items.
579    ///
580    /// Attention! This function does align the first line of `item` with respect to `self`,
581    /// but it does _not_ change indentation of other lines (if any).
582    pub fn add_item(&self, item: ast::AssocItem) {
583        let (indent, position, whitespace) = match self.assoc_items().last() {
584            Some(last_item) => (
585                IndentLevel::from_node(last_item.syntax()),
586                ted::Position::after(last_item.syntax()),
587                "\n\n",
588            ),
589            None => match self.l_curly_token() {
590                Some(l_curly) => {
591                    normalize_ws_between_braces(self.syntax());
592                    (IndentLevel::from_token(&l_curly) + 1, ted::Position::after(&l_curly), "\n")
593                }
594                None => (IndentLevel::single(), ted::Position::last_child_of(self.syntax()), "\n"),
595            },
596        };
597        let elements: Vec<SyntaxElement> = vec![
598            make::tokens::whitespace(&format!("{whitespace}{indent}")).into(),
599            item.syntax().clone().into(),
600        ];
601        ted::insert_all(position, elements);
602    }
603}

605impl ast::RecordExprFieldList {
606    pub fn add_field(&self, field: ast::RecordExprField) {
607        let is_multiline = self.syntax().text().contains_char('\n');
608        let whitespace = if is_multiline {
609            let indent = IndentLevel::from_node(self.syntax()) + 1;
610            make::tokens::whitespace(&format!("\n{indent}"))
611        } else {
612            make::tokens::single_space()
613        };
614
615        if is_multiline {
616            normalize_ws_between_braces(self.syntax());
617        }
618
619        let position = match self.fields().last() {
620            Some(last_field) => {
621                let comma = get_or_insert_comma_after(last_field.syntax());
622                ted::Position::after(comma)
623            }
624            None => match self.l_curly_token() {
625                Some(it) => ted::Position::after(it),
626                None => ted::Position::last_child_of(self.syntax()),
627            },
628        };
629
630        ted::insert_all(position, vec![whitespace.into(), field.syntax().clone().into()]);
631        if is_multiline {
632            ted::insert(ted::Position::after(field.syntax()), ast::make::token(T![,]));
633        }
634    }
635}

637impl ast::RecordExprField {
638    /// This will either replace the initializer, or in the case that this is a shorthand convert
639    /// the initializer into the name ref and insert the expr as the new initializer.
640    pub fn replace_expr(&self, expr: ast::Expr) {
641        if self.name_ref().is_some() {
642            match self.expr() {
643                Some(prev) => ted::replace(prev.syntax(), expr.syntax()),
644                None => ted::append_child(self.syntax(), expr.syntax()),
645            }
646            return;
647        }
648        // this is a shorthand
649        if let Some(ast::Expr::PathExpr(path_expr)) = self.expr()
650            && let Some(path) = path_expr.path()
651            && let Some(name_ref) = path.as_single_name_ref()
652        {
653            path_expr.syntax().detach();
654            let children = vec![
655                name_ref.syntax().clone().into(),
656                ast::make::token(T![:]).into(),
657                ast::make::tokens::single_space().into(),
658                expr.syntax().clone().into(),
659            ];
660            ted::insert_all_raw(ted::Position::last_child_of(self.syntax()), children);
661        }
662    }
663}

665impl ast::RecordPatFieldList {
666    pub fn add_field(&self, field: ast::RecordPatField) {
667        let is_multiline = self.syntax().text().contains_char('\n');
668        let whitespace = if is_multiline {
669            let indent = IndentLevel::from_node(self.syntax()) + 1;
670            make::tokens::whitespace(&format!("\n{indent}"))
671        } else {
672            make::tokens::single_space()
673        };
674
675        if is_multiline {
676            normalize_ws_between_braces(self.syntax());
677        }
678
679        let position = match self.fields().last() {
680            Some(last_field) => {
681                let syntax = last_field.syntax();
682                let comma = get_or_insert_comma_after(syntax);
683                ted::Position::after(comma)
684            }
685            None => match self.l_curly_token() {
686                Some(it) => ted::Position::after(it),
687                None => ted::Position::last_child_of(self.syntax()),
688            },
689        };
690
691        ted::insert_all(position, vec![whitespace.into(), field.syntax().clone().into()]);
692        if is_multiline {
693            ted::insert(ted::Position::after(field.syntax()), ast::make::token(T![,]));
694        }
695    }
696}
697
698fn get_or_insert_comma_after(syntax: &SyntaxNode) -> SyntaxToken {
699    match syntax
700        .siblings_with_tokens(Direction::Next)
701        .filter_map(|it| it.into_token())
702        .find(|it| it.kind() == T![,])
703    {
704        Some(it) => it,
705        None => {
706            let comma = ast::make::token(T![,]);
707            ted::insert(ted::Position::after(syntax), &comma);
708            comma
709        }
710    }
711}
712
713fn normalize_ws_between_braces(node: &SyntaxNode) -> Option<()> {
714    let l = node
715        .children_with_tokens()
716        .filter_map(|it| it.into_token())
717        .find(|it| it.kind() == T!['{'])?;
718    let r = node
719        .children_with_tokens()
720        .filter_map(|it| it.into_token())
721        .find(|it| it.kind() == T!['}'])?;
722
723    let indent = IndentLevel::from_node(node);
724
725    match l.next_sibling_or_token() {
726        Some(ws) if ws.kind() == SyntaxKind::WHITESPACE => {
727            if ws.next_sibling_or_token()?.into_token()? == r {
728                ted::replace(ws, make::tokens::whitespace(&format!("\n{indent}")));
729            }
730        }
731        Some(ws) if ws.kind() == T!['}'] => {
732            ted::insert(ted::Position::after(l), make::tokens::whitespace(&format!("\n{indent}")));
733        }
734        _ => (),
735    }
736    Some(())
737}

739impl ast::IdentPat {
740    pub fn set_pat(&self, pat: Option<ast::Pat>) {
741        match pat {
742            None => {
743                if let Some(at_token) = self.at_token() {
744                    // Remove `@ Pat`
745                    let start = at_token.clone().into();
746                    let end = self
747                        .pat()
748                        .map(|it| it.syntax().clone().into())
749                        .unwrap_or_else(|| at_token.into());
750
751                    ted::remove_all(start..=end);
752
753                    // Remove any trailing ws
754                    if let Some(last) =
755                        self.syntax().last_token().filter(|it| it.kind() == WHITESPACE)
756                    {
757                        last.detach();
758                    }
759                }
760            }
761            Some(pat) => {
762                if let Some(old_pat) = self.pat() {
763                    // Replace existing pattern
764                    ted::replace(old_pat.syntax(), pat.syntax())
765                } else if let Some(at_token) = self.at_token() {
766                    // Have an `@` token but not a pattern yet
767                    ted::insert(ted::Position::after(at_token), pat.syntax());
768                } else {
769                    // Don't have an `@`, should have a name
770                    let name = self.name().unwrap();
771
772                    ted::insert_all(
773                        ted::Position::after(name.syntax()),
774                        vec![
775                            make::token(T![@]).into(),
776                            make::tokens::single_space().into(),
777                            pat.syntax().clone().into(),
778                        ],
779                    )
780                }
781            }
782        }
783    }
784
785    pub fn set_pat_with_editor(
786        &self,
787        pat: Option<ast::Pat>,
788        syntax_editor: &mut SyntaxEditor,
789        syntax_factory: &SyntaxFactory,
790    ) {
791        match pat {
792            None => {
793                if let Some(at_token) = self.at_token() {
794                    // Remove `@ Pat`
795                    let start = at_token.clone().into();
796                    let end = self
797                        .pat()
798                        .map(|it| it.syntax().clone().into())
799                        .unwrap_or_else(|| at_token.into());
800                    syntax_editor.delete_all(start..=end);
801
802                    // Remove any trailing ws
803                    if let Some(last) =
804                        self.syntax().last_token().filter(|it| it.kind() == WHITESPACE)
805                    {
806                        last.detach();
807                    }
808                }
809            }
810            Some(pat) => {
811                if let Some(old_pat) = self.pat() {
812                    // Replace existing pattern
813                    syntax_editor.replace(old_pat.syntax(), pat.syntax())
814                } else if let Some(at_token) = self.at_token() {
815                    // Have an `@` token but not a pattern yet
816                    syntax_editor.insert(Position::after(at_token), pat.syntax());
817                } else {
818                    // Don't have an `@`, should have a name
819                    let name = self.name().unwrap();
820
821                    syntax_editor.insert_all(
822                        Position::after(name.syntax()),
823                        vec![
824                            syntax_factory.whitespace(" ").into(),
825                            syntax_factory.token(T![@]).into(),
826                            syntax_factory.whitespace(" ").into(),
827                            pat.syntax().clone().into(),
828                        ],
829                    )
830                }
831            }
832        }
833    }
834}

836pub trait HasVisibilityEdit: ast::HasVisibility {
837    fn set_visibility(&self, visibility: Option<ast::Visibility>) {
838        if let Some(visibility) = visibility {
839            match self.visibility() {
840                Some(current_visibility) => {
841                    ted::replace(current_visibility.syntax(), visibility.syntax())
842                }
843                None => {
844                    let vis_before = self
845                        .syntax()
846                        .children_with_tokens()
847                        .find(|it| !matches!(it.kind(), WHITESPACE | COMMENT | ATTR))
848                        .unwrap_or_else(|| self.syntax().first_child_or_token().unwrap());
849
850                    ted::insert(ted::Position::before(vis_before), visibility.syntax());
851                }
852            }
853        } else if let Some(visibility) = self.visibility() {
854            ted::remove(visibility.syntax());
855        }
856    }
857}

861pub trait Indent: AstNode + Clone + Sized {
862    fn indent_level(&self) -> IndentLevel {
863        IndentLevel::from_node(self.syntax())
864    }
865    fn indent(&self, by: IndentLevel) {
866        by.increase_indent(self.syntax());
867    }
868    fn dedent(&self, by: IndentLevel) {
869        by.decrease_indent(self.syntax());
870    }
871    fn reindent_to(&self, target_level: IndentLevel) {
872        let current_level = IndentLevel::from_node(self.syntax());
873        self.dedent(current_level);
874        self.indent(target_level);
875    }
876}
