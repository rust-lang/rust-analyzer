19pub trait AttrsOwnerEdit: ast::HasAttrs {
20    fn remove_attrs_and_docs(&self) {
21        remove_attrs_and_docs(self.syntax());
22
23        fn remove_attrs_and_docs(node: &SyntaxNode) {
24            let mut remove_next_ws = false;
25            for child in node.children_with_tokens() {
26                match child.kind() {
27                    ATTR | COMMENT => {
28                        remove_next_ws = true;
29                        child.detach();
30                        continue;
31                    }
32                    WHITESPACE if remove_next_ws => {
33                        child.detach();
34                    }
35                    _ => (),
36                }
37                remove_next_ws = false;
38            }
39        }
40    }
41}

45impl ast::GenericParamList {
46    /// Constructs a matching [`ast::GenericArgList`]
47    pub fn to_generic_args(&self, make: &SyntaxFactory) -> ast::GenericArgList {
48        let args = self.generic_params().filter_map(|param| match param {
49            ast::GenericParam::LifetimeParam(it) => {
50                Some(ast::GenericArg::LifetimeArg(make.lifetime_arg(it.lifetime()?)))
51            }
52            ast::GenericParam::TypeParam(it) => {
53                Some(ast::GenericArg::TypeArg(make.type_arg(make.ty_name(it.name()?))))
54            }
55            ast::GenericParam::ConstParam(it) => {
56                // Name-only const params get parsed as `TypeArg`s
57                Some(ast::GenericArg::TypeArg(make.type_arg(make.ty_name(it.name()?))))
58            }
59        });
60
61        make::generic_arg_list(args)
62    }

69impl Removable for ast::UseTree {
70    fn remove(&self) {
71        for dir in [Direction::Next, Direction::Prev] {
72            if let Some(next_use_tree) = neighbor(self, dir) {
73                let separators = self
74                    .syntax()
75                    .siblings_with_tokens(dir)
76                    .skip(1)
77                    .take_while(|it| it.as_node() != Some(next_use_tree.syntax()));
78                ted::remove_all_iter(separators);
79                break;
80            }
81        }
82        ted::remove(self.syntax());
83    }
84}

86impl ast::UseTree {
87    /// Deletes the usetree node represented by the input. Recursively removes parents, including use nodes that become empty.
88    pub fn remove_recursive(self) {
89        let parent = self.syntax().parent();
90
91        self.remove();
92
93        if let Some(u) = parent.clone().and_then(ast::Use::cast) {
94            if u.use_tree().is_none() {
95                u.remove();
96            }
97        } else if let Some(u) = parent.and_then(ast::UseTreeList::cast) {
98            if u.use_trees().next().is_none() {
99                let parent = u.syntax().parent().and_then(ast::UseTree::cast);
100                if let Some(u) = parent {
101                    u.remove_recursive();
102                }
103            }
104            u.remove_unnecessary_braces();
105        }
106    }
107
108    pub fn get_or_create_use_tree_list(&self) -> ast::UseTreeList {
109        match self.use_tree_list() {
110            Some(it) => it,
111            None => {
112                let position = ted::Position::last_child_of(self.syntax());
113                let use_tree_list = make::use_tree_list(empty()).clone_for_update();
114                let mut elements = Vec::with_capacity(2);
115                if self.coloncolon_token().is_none() {
116                    elements.push(make::token(T![::]).into());
117                }
118                elements.push(use_tree_list.syntax().clone().into());
119                ted::insert_all_raw(position, elements);
120                use_tree_list
121            }
122        }
123    }
124
125    /// Splits off the given prefix, making it the path component of the use tree,
126    /// appending the rest of the path to all UseTreeList items.
127    ///
128    /// # Examples
129    ///
130    /// `prefix$0::suffix` -> `prefix::{suffix}`
131    ///
132    /// `prefix$0` -> `prefix::{self}`
133    ///
134    /// `prefix$0::*` -> `prefix::{*}`
135    pub fn split_prefix(&self, prefix: &ast::Path) {
136        debug_assert_eq!(self.path(), Some(prefix.top_path()));
137        let path = self.path().unwrap();
138        if &path == prefix && self.use_tree_list().is_none() {
139            if self.star_token().is_some() {
140                // path$0::* -> *
141                if let Some(a) = self.coloncolon_token() {
142                    ted::remove(a)
143                }
144                ted::remove(prefix.syntax());
145            } else {
146                // path$0 -> self
147                let self_suffix =
148                    make::path_unqualified(make::path_segment_self()).clone_for_update();
149                ted::replace(path.syntax(), self_suffix.syntax());
150            }
151        } else if split_path_prefix(prefix).is_none() {
152            return;
153        }
154        // At this point, prefix path is detached; _self_ use tree has suffix path.
155        // Next, transform 'suffix' use tree into 'prefix::{suffix}'
156        let subtree = self.clone_subtree().clone_for_update();
157        ted::remove_all_iter(self.syntax().children_with_tokens());
158        ted::insert(ted::Position::first_child_of(self.syntax()), prefix.syntax());
159        self.get_or_create_use_tree_list().add_use_tree(subtree);
160
161        fn split_path_prefix(prefix: &ast::Path) -> Option<()> {
162            let parent = prefix.parent_path()?;
163            let segment = parent.segment()?;
164            if algo::has_errors(segment.syntax()) {
165                return None;
166            }
167            for p in successors(parent.parent_path(), |it| it.parent_path()) {
168                p.segment()?;
169            }
170            if let Some(a) = prefix.parent_path().and_then(|p| p.coloncolon_token()) {
171                ted::remove(a)
172            }
173            ted::remove(prefix.syntax());
174            Some(())
175        }
176    }
177
178    /// Wraps the use tree in use tree list with no top level path (if it isn't already).
179    ///
180    /// # Examples
181    ///
182    /// `foo::bar` -> `{foo::bar}`
183    ///
184    /// `{foo::bar}` -> `{foo::bar}`
185    pub fn wrap_in_tree_list(&self) -> Option<()> {
186        if self.use_tree_list().is_some()
187            && self.path().is_none()
188            && self.star_token().is_none()
189            && self.rename().is_none()
190        {
191            return None;
192        }
193        let subtree = self.clone_subtree().clone_for_update();
194        ted::remove_all_iter(self.syntax().children_with_tokens());
195        ted::append_child(
196            self.syntax(),
197            make::use_tree_list(once(subtree)).clone_for_update().syntax(),
198        );
199        Some(())
200    }
201}

203impl ast::UseTreeList {
204    pub fn add_use_tree(&self, use_tree: ast::UseTree) {
205        let (position, elements) = match self.use_trees().last() {
206            Some(last_tree) => (
207                ted::Position::after(last_tree.syntax()),
208                vec![
209                    make::token(T![,]).into(),
210                    make::tokens::single_space().into(),
211                    use_tree.syntax.into(),
212                ],
213            ),
214            None => {
215                let position = match self.l_curly_token() {
216                    Some(l_curly) => ted::Position::after(l_curly),
217                    None => ted::Position::last_child_of(self.syntax()),
218                };
219                (position, vec![use_tree.syntax.into()])
220            }
221        };
222        ted::insert_all_raw(position, elements);
223    }

226impl Removable for ast::Use {
227    fn remove(&self) {
228        let next_ws = self
229            .syntax()
230            .next_sibling_or_token()
231            .and_then(|it| it.into_token())
232            .and_then(ast::Whitespace::cast);
233        if let Some(next_ws) = next_ws {
234            let ws_text = next_ws.syntax().text();
235            if let Some(rest) = ws_text.strip_prefix('\n') {
236                if rest.is_empty() {
237                    ted::remove(next_ws.syntax());
238                } else {
239                    ted::replace(next_ws.syntax(), make::tokens::whitespace(rest));
240                }
241            }
242        }
243        let prev_ws = self
244            .syntax()
245            .prev_sibling_or_token()
246            .and_then(|it| it.into_token())
247            .and_then(ast::Whitespace::cast);
248        if let Some(prev_ws) = prev_ws {
249            let ws_text = prev_ws.syntax().text();
250            let prev_newline = ws_text.rfind('\n').map(|x| x + 1).unwrap_or(0);
251            let rest = &ws_text[0..prev_newline];
252            if rest.is_empty() {
253                ted::remove(prev_ws.syntax());
254            } else {
255                ted::replace(prev_ws.syntax(), make::tokens::whitespace(rest));
256            }
257        }
258
259        ted::remove(self.syntax());
260    }
261}

273impl ast::AssocItemList {
274    /// Adds a new associated item after all of the existing associated items.
275    ///
276    /// Attention! This function does align the first line of `item` with respect to `self`,
277    /// but it does _not_ change indentation of other lines (if any).
278    pub fn add_item(&self, editor: &SyntaxEditor, item: ast::AssocItem) {
279        let make = editor.make();
280        let (indent, position, whitespace) = match self.assoc_items().last() {
281            Some(last_item) => (
282                IndentLevel::from_node(last_item.syntax()),
283                syntax_editor::Position::after(last_item.syntax()),
284                "\n\n",
285            ),
286            None => match self.l_curly_token() {
287                Some(l_curly) => {
288                    normalize_ws_between_braces_with_editor(editor, self.syntax());
289                    (
290                        IndentLevel::from_token(&l_curly) + 1,
291                        syntax_editor::Position::after(&l_curly),
292                        "\n",
293                    )
294                }
295                None => (
296                    IndentLevel::zero(),
297                    syntax_editor::Position::last_child_of(self.syntax()),
298                    "\n",
299                ),
300            },
301        };
302        let elements: Vec<SyntaxElement> = vec![
303            make.whitespace(&format!("{whitespace}{indent}")).into(),
304            item.syntax().clone().into(),
305        ];
306        editor.insert_all(position, elements);
307    }
308}

310impl ast::RecordExprFieldList {
311    pub fn add_field(&self, field: ast::RecordExprField) {
312        let is_multiline = self.syntax().text().contains_char('\n');
313        let whitespace = if is_multiline {
314            let indent = IndentLevel::from_node(self.syntax()) + 1;
315            make::tokens::whitespace(&format!("\n{indent}"))
316        } else {
317            make::tokens::single_space()
318        };
319
320        if is_multiline {
321            normalize_ws_between_braces(self.syntax());
322        }
323
324        let position = match self.fields().last() {
325            Some(last_field) => {
326                let comma = get_or_insert_comma_after(last_field.syntax());
327                ted::Position::after(comma)
328            }
329            None => match self.l_curly_token() {
330                Some(it) => ted::Position::after(it),
331                None => ted::Position::last_child_of(self.syntax()),
332            },
333        };
334
335        ted::insert_all(position, vec![whitespace.into(), field.syntax().clone().into()]);
336        if is_multiline {
337            ted::insert(ted::Position::after(field.syntax()), ast::make::token(T![,]));
338        }
339    }
340}

342impl ast::RecordExprField {
343    /// This will either replace the initializer, or in the case that this is a shorthand convert
344    /// the initializer into the name ref and insert the expr as the new initializer.
345    pub fn replace_expr(&self, editor: &SyntaxEditor, expr: ast::Expr) {
346        if self.name_ref().is_some() {
347            if let Some(prev) = self.expr() {
348                editor.replace(prev.syntax(), expr.syntax());
349            }
350        } else if let Some(ast::Expr::PathExpr(path_expr)) = self.expr()
351            && let Some(path) = path_expr.path()
352            && let Some(name_ref) = path.as_single_name_ref()
353        {
354            // shorthand `{ x }` → expand to `{ x: expr }`
355            let new_field = editor
356                .make()
357                .record_expr_field(editor.make().name_ref(&name_ref.text()), Some(expr));
358            editor.replace(self.syntax(), new_field.syntax());
359        }
360    }
361}

396fn get_or_insert_comma_after(syntax: &SyntaxNode) -> SyntaxToken {
397    match syntax
398        .siblings_with_tokens(Direction::Next)
399        .filter_map(|it| it.into_token())
400        .find(|it| it.kind() == T![,])
401    {
402        Some(it) => it,
403        None => {
404            let comma = ast::make::token(T![,]);
405            ted::insert(ted::Position::after(syntax), &comma);
406            comma
407        }
408    }
409}

411fn normalize_ws_between_braces(node: &SyntaxNode) -> Option<()> {
412    let l = node
413        .children_with_tokens()
414        .filter_map(|it| it.into_token())
415        .find(|it| it.kind() == T!['{'])?;
416    let r = node
417        .children_with_tokens()
418        .filter_map(|it| it.into_token())
419        .find(|it| it.kind() == T!['}'])?;
420
421    let indent = IndentLevel::from_node(node);
422
423    match l.next_sibling_or_token() {
424        Some(ws)
425            if ws.kind() == SyntaxKind::WHITESPACE
426                && ws.next_sibling_or_token()?.into_token()? == r =>
427        {
428            ted::replace(ws, make::tokens::whitespace(&format!("\n{indent}")));
429        }
430        Some(ws) if ws.kind() == T!['}'] => {
431            ted::insert(ted::Position::after(l), make::tokens::whitespace(&format!("\n{indent}")));
432        }
433        _ => (),
434    }
435    Some(())
436}

438fn normalize_ws_between_braces_with_editor(editor: &SyntaxEditor, node: &SyntaxNode) -> Option<()> {
439    let make = editor.make();
440    let l = node
441        .children_with_tokens()
442        .filter_map(|it| it.into_token())
443        .find(|it| it.kind() == T!['{'])?;
444    let r = node
445        .children_with_tokens()
446        .filter_map(|it| it.into_token())
447        .find(|it| it.kind() == T!['}'])?;
448
449    let indent = IndentLevel::from_node(node);
450
451    match l.next_sibling_or_token() {
452        Some(ws)
453            if ws.kind() == SyntaxKind::WHITESPACE
454                && ws.next_sibling_or_token()?.into_token()? == r =>
455        {
456            editor.replace(ws, make.whitespace(&format!("\n{indent}")));
457        }
458        Some(ws) if ws.kind() == T!['}'] => {
459            editor
460                .insert(syntax_editor::Position::after(l), make.whitespace(&format!("\n{indent}")));
461        }
462        _ => (),
463    }
464    Some(())

467pub trait Indent: AstNode + Clone + Sized {
468    fn indent_level(&self) -> IndentLevel {
469        IndentLevel::from_node(self.syntax())
470    }
471    fn indent(&self, by: IndentLevel) {
472        by.increase_indent(self.syntax());
473    }
474    fn dedent(&self, by: IndentLevel) {
475        by.decrease_indent(self.syntax());
476    }
477    fn reindent_to(&self, target_level: IndentLevel) {
478        let current_level = IndentLevel::from_node(self.syntax());
479        self.dedent(current_level);
480        self.indent(target_level);
481    }
482}
