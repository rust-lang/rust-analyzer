20pub(super) fn apply_edits(editor: SyntaxEditor) -> SyntaxEdit {
21    // Algorithm overview:
22    //
23    // - Sort changes by (range, type)
24    //   - Ensures that parent edits are before child edits
25    //   - Ensures that inserts will be guaranteed to be inserted at the right range
26    // - Validate changes
27    //   - Checking for invalid changes is easy since the changes will be sorted by range
28    // - Fixup change targets
29    //   - standalone change? map to original syntax tree
30    //   - dependent change?
31    //     - try to map to parent change (either independent or another dependent)
32    //     - note: need to keep track of a parent change stack, since a change can be a parent of multiple changes
33    // - Apply changes
34    //   - find changes to apply to real tree by applying nested changes first
35    //   - changed nodes become part of the changed node set (useful for the formatter to only change those parts)
36    // - Propagate annotations
37
38    let SyntaxEditor { root, changes, annotations, make } = editor;
39    let mut changes = changes.into_inner();
40    let annotations = annotations.into_inner();
41    let mappings = make.take();
42
43    let mut node_depths = FxHashMap::<SyntaxNode, usize>::default();
44    let mut get_node_depth = |node: SyntaxNode| {
45        *node_depths.entry(node).or_insert_with_key(|node| node.ancestors().count())
46    };
47
48    // Sort changes by range, then depth, then change kind, so that we can:
49    // - ensure that parent edits are ordered before child edits
50    // - ensure that inserts will be guaranteed to be inserted at the right range
51    // - easily check for disjoint replace ranges
52    changes.sort_by(|a, b| {
53        a.target_range()
54            .start()
55            .cmp(&b.target_range().start())
56            .then_with(|| {
57                let a_target = a.target_parent();
58                let b_target = b.target_parent();
59
60                if a_target == b_target {
61                    return Ordering::Equal;
62                }
63
64                get_node_depth(a_target).cmp(&get_node_depth(b_target))
65            })
66            .then(a.change_kind().cmp(&b.change_kind()))
67    });
68
69    let disjoint_replaces_ranges = changes
70        .iter()
71        .zip(changes.iter().skip(1))
72        .filter(|(l, r)| {
73            // We only care about checking for disjoint replace ranges
74            matches!(
75                (l.change_kind(), r.change_kind()),
76                (
77                    ChangeKind::Replace | ChangeKind::ReplaceRange,
78                    ChangeKind::Replace | ChangeKind::ReplaceRange
79                )
80            )
81        })
82        .all(|(l, r)| {
83            get_node_depth(l.target_parent()) != get_node_depth(r.target_parent())
84                || (l.target_range().end() <= r.target_range().start())
85        });
86
87    if !disjoint_replaces_ranges {
88        report_intersecting_changes(&changes, get_node_depth, &root);
89
90        return SyntaxEdit {
91            old_root: root.clone(),
92            new_root: root,
93            annotations: Default::default(),
94            changed_elements: vec![],
95        };
96    }
97
98    #[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
99    struct DependentChange {
100        parent: u32,
101        child: u32,
102    }
103
104    // Build change tree
105    let mut changed_ancestors: VecDeque<ChangedAncestor> = VecDeque::new();
106    let mut dependent_changes = vec![];
107    let mut independent_changes = vec![];
108    let mut outdated_changes = vec![];
109
110    for (change_index, change) in changes.iter().enumerate() {
111        // Check if this change is dependent on another change (i.e. it's contained within another range)
112        if let Some(index) = changed_ancestors
113            .iter()
114            .rev()
115            .position(|ancestor| ancestor.affected_range().contains_range(change.target_range()))
116        {
117            // Pop off any ancestors that aren't applicable
118            changed_ancestors.drain((index + 1)..);
119
120            // FIXME: Resolve changes that depend on a range of elements
121            let ancestor = &changed_ancestors[index];
122
123            if let Change::Replace(_, None) = changes[ancestor.change_index] {
124                outdated_changes.push(change_index as u32);
125            } else {
126                dependent_changes.push(DependentChange {
127                    parent: ancestor.change_index as u32,
128                    child: change_index as u32,
129                });
130            }
131        } else {
132            // This change is independent of any other change
133
134            // Drain the changed ancestors since we're no longer in a set of dependent changes
135            changed_ancestors.drain(..);
136
137            independent_changes.push(change_index as u32);
138        }
139
140        // Add to changed ancestors, if applicable
141        match change {
142            Change::Replace(SyntaxElement::Node(target), _)
143            | Change::ReplaceWithMany(SyntaxElement::Node(target), _) => {
144                changed_ancestors.push_back(ChangedAncestor::single(target, change_index))
145            }
146            Change::ReplaceAll(range, _) => {
147                changed_ancestors.push_back(ChangedAncestor::multiple(range, change_index))
148            }
149            _ => (),
150        }
151    }
152
153    // Map change targets to the correct syntax nodes
154    let tree_mutator = TreeMutator::new(&root);
155    let mut changed_elements = vec![];
156    let mut changed_elements_set = rustc_hash::FxHashSet::default();
157    let mut deduplicate_node = |node_or_token: &mut SyntaxElement| {
158        let node;
159        let node = match node_or_token {
160            SyntaxElement::Token(token) => match token.parent() {
161                None => return,
162                Some(parent) => {
163                    node = parent;
164                    &node
165                }
166            },
167            SyntaxElement::Node(node) => node,
168        };
169        if changed_elements_set.contains(node) {
170            let new_node = node.clone_subtree().clone_for_update();
171            match node_or_token {
172                SyntaxElement::Node(node) => *node = new_node,
173                SyntaxElement::Token(token) => {
174                    *token = new_node
175                        .children_with_tokens()
176                        .filter_map(SyntaxElement::into_token)
177                        .find(|it| it.kind() == token.kind() && it.text() == token.text())
178                        .unwrap();
179                }
180            }
181        } else {
182            changed_elements_set.insert(node.clone());
183        }
184    };
185
186    for index in independent_changes {
187        match &mut changes[index as usize] {
188            Change::Insert(target, _) | Change::InsertAll(target, _) => {
189                match &mut target.repr {
190                    PositionRepr::FirstChild(parent) => {
191                        *parent = tree_mutator.make_syntax_mut(parent);
192                    }
193                    PositionRepr::After(child) => {
194                        *child = tree_mutator.make_element_mut(child);
195                    }
196                };
197            }
198            Change::Replace(SyntaxElement::Node(target), Some(SyntaxElement::Node(_))) => {
199                *target = tree_mutator.make_syntax_mut(target);
200            }
201            Change::Replace(target, _) | Change::ReplaceWithMany(target, _) => {
202                *target = tree_mutator.make_element_mut(target);
203            }
204            Change::ReplaceAll(range, _) => {
205                let start = tree_mutator.make_element_mut(range.start());
206                let end = tree_mutator.make_element_mut(range.end());
207
208                *range = start..=end;
209            }
210        }
211
212        match &mut changes[index as usize] {
213            Change::Insert(_, SyntaxElement::Node(node))
214            | Change::Replace(_, Some(SyntaxElement::Node(node))) => {
215                if node.parent().is_some() {
216                    *node = node.clone_subtree().clone_for_update();
217                } else if !node.is_mutable() {
218                    *node = node.clone_for_update();
219                }
220            }
221            Change::Insert(_, SyntaxElement::Token(token))
222            | Change::Replace(_, Some(SyntaxElement::Token(token))) => {
223                if let Some(parent) = token.parent() {
224                    let idx = token.index();
225                    let new_parent = parent.clone_subtree().clone_for_update();
226                    *token = new_parent
227                        .children_with_tokens()
228                        .nth(idx)
229                        .and_then(SyntaxElement::into_token)
230                        .unwrap();
231                }
232            }
233            Change::InsertAll(_, elements)
234            | Change::ReplaceWithMany(_, elements)
235            | Change::ReplaceAll(_, elements) => {
236                for element in elements {
237                    match element {
238                        SyntaxElement::Node(node) => {
239                            if node.parent().is_some() {
240                                *node = node.clone_subtree().clone_for_update();
241                            } else if !node.is_mutable() {
242                                *node = node.clone_for_update();
243                            }
244                        }
245                        SyntaxElement::Token(token) => {
246                            if let Some(parent) = token.parent() {
247                                let idx = token.index();
248                                let new_parent = parent.clone_subtree().clone_for_update();
249                                *token = new_parent
250                                    .children_with_tokens()
251                                    .nth(idx)
252                                    .and_then(SyntaxElement::into_token)
253                                    .unwrap();
254                            }
255                        }
256                    }
257                }
258            }
259            _ => {}
260        }
261
262        match &mut changes[index as usize] {
263            Change::Insert(_, element) | Change::Replace(_, Some(element)) => {
264                deduplicate_node(element);
265            }
266            Change::InsertAll(_, elements)
267            | Change::ReplaceWithMany(_, elements)
268            | Change::ReplaceAll(_, elements) => {
269                elements.iter_mut().for_each(&mut deduplicate_node);
270            }
271            Change::Replace(_, None) => (),
272        }
273
274        // Collect changed elements
275        match &changes[index as usize] {
276            Change::Insert(_, element) => changed_elements.push(element.clone()),
277            Change::InsertAll(_, elements) => changed_elements.extend(elements.iter().cloned()),
278            Change::Replace(_, Some(element)) => changed_elements.push(element.clone()),
279            Change::Replace(_, None) => {}
280            Change::ReplaceWithMany(_, elements) => {
281                changed_elements.extend(elements.iter().cloned())
282            }
283            Change::ReplaceAll(_, elements) => changed_elements.extend(elements.iter().cloned()),
284        }
285    }
286
287    for DependentChange { parent, child } in dependent_changes.into_iter() {
288        let (input_ancestor, output_ancestor) = match &changes[parent as usize] {
289            // No change will depend on an insert since changes can only depend on nodes in the root tree
290            Change::Insert(_, _) | Change::InsertAll(_, _) => unreachable!(),
291            Change::Replace(target, Some(new_target)) => {
292                (to_owning_node(target), to_owning_node(new_target))
293            }
294            Change::Replace(_, None) => {
295                unreachable!("deletions should not generate dependent changes")
296            }
297            Change::ReplaceAll(_, _) | Change::ReplaceWithMany(_, _) => {
298                unimplemented!("cannot resolve changes that depend on replacing many elements")
299            }
300        };
301
302        let upmap_target_node = |target: &SyntaxNode| match mappings.upmap_child(
303            target,
304            &input_ancestor,
305            &output_ancestor,
306        ) {
307            Ok(it) => it,
308            Err(MissingMapping(current)) => unreachable!(
309                "no mappings exist between {current:?} (ancestor of {input_ancestor:?}) and {output_ancestor:?}"
310            ),
311        };
312
313        let upmap_target = |target: &SyntaxElement| match mappings.upmap_child_element(
314            target,
315            &input_ancestor,
316            &output_ancestor,
317        ) {
318            Ok(it) => it,
319            Err(MissingMapping(current)) => unreachable!(
320                "no mappings exist between {current:?} (ancestor of {input_ancestor:?}) and {output_ancestor:?}"
321            ),
322        };
323
324        match &mut changes[child as usize] {
325            Change::Insert(target, _) | Change::InsertAll(target, _) => match &mut target.repr {
326                PositionRepr::FirstChild(parent) => {
327                    *parent = upmap_target_node(parent);
328                }
329                PositionRepr::After(child) => {
330                    *child = upmap_target(child);
331                }
332            },
333            Change::Replace(target, _) | Change::ReplaceWithMany(target, _) => {
334                *target = upmap_target(target);
335            }
336            Change::ReplaceAll(range, _) => {
337                *range = upmap_target(range.start())..=upmap_target(range.end());
338            }
339        }
340    }
341
342    // We reverse here since we pushed to this in ascending order,
343    // and we want to remove elements in descending order
344    for idx in outdated_changes.into_iter().rev() {
345        changes.remove(idx as usize);
346    }
347
348    // Apply changes
349    let mut root = tree_mutator.mutable_clone;
350
351    for change in changes {
352        match change {
353            Change::Insert(position, element) => {
354                let (parent, index) = position.place();
355                parent.splice_children(index..index, vec![element]);
356            }
357            Change::InsertAll(position, elements) => {
358                let (parent, index) = position.place();
359                parent.splice_children(index..index, elements);
360            }
361            Change::Replace(target, None) => {
362                target.detach();
363            }
364            Change::Replace(SyntaxElement::Node(target), Some(new_target)) if target == root => {
365                root = new_target.into_node().expect("root node replacement should be a node");
366            }
367            Change::Replace(target, Some(new_target)) => {
368                let parent = target.parent().unwrap();
369                parent.splice_children(target.index()..target.index() + 1, vec![new_target]);
370            }
371            Change::ReplaceWithMany(target, elements) => {
372                let parent = target.parent().unwrap();
373                parent.splice_children(target.index()..target.index() + 1, elements);
374            }
375            Change::ReplaceAll(range, elements) => {
376                let start = range.start().index();
377                let end = range.end().index();
378                let parent = range.start().parent().unwrap();
379                parent.splice_children(start..end + 1, elements);
380            }
381        }
382    }
383
384    // Propagate annotations
385    let annotations = annotations.into_iter().filter_map(|(element, annotation)| {
386        match mappings.upmap_element(&element, &root) {
387            // Needed to follow the new tree to find the resulting element
388            Some(Ok(mapped)) => Some((mapped, annotation)),
389            // Element did not need to be mapped
390            None => Some((element, annotation)),
391            // Element did not make it to the final tree
392            Some(Err(_)) => None,
393        }
394    });
395
396    let mut annotation_groups = FxHashMap::default();
397
398    for (element, annotation) in annotations {
399        annotation_groups.entry(annotation).or_insert(vec![]).push(element);
400    }
401
402    SyntaxEdit {
403        old_root: tree_mutator.immutable,
404        new_root: root,
405        changed_elements,
406        annotations: annotation_groups,
407    }
408}

410fn report_intersecting_changes(
411    changes: &[Change],
412    mut get_node_depth: impl FnMut(rowan::SyntaxNode<crate::RustLanguage>) -> usize,
413    root: &rowan::SyntaxNode<crate::RustLanguage>,
414) {
415    let intersecting_changes = changes
416        .iter()
417        .zip(changes.iter().skip(1))
418        .filter(|(l, r)| {
419            // We only care about checking for disjoint replace ranges.
420            matches!(
421                (l.change_kind(), r.change_kind()),
422                (
423                    ChangeKind::Replace | ChangeKind::ReplaceRange,
424                    ChangeKind::Replace | ChangeKind::ReplaceRange
425                )
426            )
427        })
428        .filter(|(l, r)| {
429            get_node_depth(l.target_parent()) == get_node_depth(r.target_parent())
430                && (l.target_range().end() > r.target_range().start())
431        });
432
433    let mut error_msg = String::from("some replace change ranges intersect!\n");
434
435    let parent_str = root.to_string();
436
437    for (l, r) in intersecting_changes {
438        let mut highlighted_str = parent_str.clone();
439        let l_range = l.target_range();
440        let r_range = r.target_range();
441
442        let i_range = l_range.intersect(r_range).unwrap();
443        let i_str = format!("\x1b[46m{}", &parent_str[i_range]);
444
445        let pre_range: Range<usize> = l_range.start().into()..i_range.start().into();
446        let pre_str = format!("\x1b[44m{}", &parent_str[pre_range]);
447
448        let (highlight_range, highlight_str) = if l_range == r_range {
449            format_to!(error_msg, "\x1b[46mleft change:\x1b[0m  {l:?} {l}\n");
450            format_to!(error_msg, "\x1b[46mequals\x1b[0m\n");
451            format_to!(error_msg, "\x1b[46mright change:\x1b[0m {r:?} {r}\n");
452            let i_highlighted = format!("{i_str}\x1b[0m\x1b[K");
453            let total_range: Range<usize> = i_range.into();
454            (total_range, i_highlighted)
455        } else {
456            format_to!(error_msg, "\x1b[44mleft change:\x1b[0m  {l:?} {l}\n");
457            let range_end = if l_range.contains_range(r_range) {
458                format_to!(error_msg, "\x1b[46mcovers\x1b[0m\n");
459                format_to!(error_msg, "\x1b[46mright change:\x1b[0m {r:?} {r}\n");
460                l_range.end()
461            } else {
462                format_to!(error_msg, "\x1b[46mintersects\x1b[0m\n");
463                format_to!(error_msg, "\x1b[42mright change:\x1b[0m {r:?} {r}\n");
464                r_range.end()
465            };
466
467            let post_range: Range<usize> = i_range.end().into()..range_end.into();
468
469            let post_str = format!("\x1b[42m{}", &parent_str[post_range]);
470            let result = format!("{pre_str}{i_str}{post_str}\x1b[0m\x1b[K");
471            let total_range: Range<usize> = l_range.start().into()..range_end.into();
472            (total_range, result)
473        };
474        highlighted_str.replace_range(highlight_range, &highlight_str);
475
476        format_to!(error_msg, "{highlighted_str}\n");
477    }
478
479    stdx::always!(false, "{}", error_msg);
480}

482fn to_owning_node(element: &SyntaxElement) -> SyntaxNode {
483    match element {
484        SyntaxElement::Node(node) => node.clone(),
485        SyntaxElement::Token(token) => token.parent().unwrap(),
486    }
487}

489struct ChangedAncestor {
490    kind: ChangedAncestorKind,
491    change_index: usize,
492}
493
494enum ChangedAncestorKind {
495    Single { node: SyntaxNode },
496    Range { _changed_elements: RangeInclusive<SyntaxElement>, _in_parent: SyntaxNode },
497}
498
499impl ChangedAncestor {
500    fn single(node: &SyntaxNode, change_index: usize) -> Self {
501        let kind = ChangedAncestorKind::Single { node: node.clone() };
502
503        Self { kind, change_index }
504    }
505
506    fn multiple(range: &RangeInclusive<SyntaxElement>, change_index: usize) -> Self {
507        Self {
508            kind: ChangedAncestorKind::Range {
509                _changed_elements: range.clone(),
510                _in_parent: range.start().parent().unwrap(),
511            },
512            change_index,
513        }
514    }
515
516    fn affected_range(&self) -> TextRange {
517        match &self.kind {
518            ChangedAncestorKind::Single { node } => node.text_range(),
519            ChangedAncestorKind::Range { _changed_elements: changed_nodes, _in_parent: _ } => {
520                TextRange::new(
521                    changed_nodes.start().text_range().start(),
522                    changed_nodes.end().text_range().end(),
523                )
524            }
525        }
526    }
527}

529struct TreeMutator {
530    immutable: SyntaxNode,
531    mutable_clone: SyntaxNode,
532}
533
534impl TreeMutator {
535    fn new(immutable: &SyntaxNode) -> TreeMutator {
536        let immutable = immutable.clone();
537        let mutable_clone = immutable.clone_for_update();
538        TreeMutator { immutable, mutable_clone }
539    }
540
541    fn make_element_mut(&self, element: &SyntaxElement) -> SyntaxElement {
542        match element {
543            SyntaxElement::Node(node) => SyntaxElement::Node(self.make_syntax_mut(node)),
544            SyntaxElement::Token(token) => {
545                let parent = self.make_syntax_mut(&token.parent().unwrap());
546                parent.children_with_tokens().nth(token.index()).unwrap()
547            }
548        }
549    }
550
551    fn make_syntax_mut(&self, node: &SyntaxNode) -> SyntaxNode {
552        let ptr = SyntaxNodePtr::new(node);
553        ptr.to_node(&self.mutable_clone)
554    }
