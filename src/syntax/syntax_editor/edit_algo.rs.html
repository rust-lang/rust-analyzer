20pub(super) fn apply_edits(editor: SyntaxEditor) -> SyntaxEdit {
21    // Algorithm overview:
22    //
23    // - Sort changes by (range, type)
24    //   - Ensures that parent edits are before child edits
25    //   - Ensures that inserts will be guaranteed to be inserted at the right range
26    // - Validate changes
27    //   - Checking for invalid changes is easy since the changes will be sorted by range
28    // - Fixup change targets
29    //   - standalone change? map to original syntax tree
30    //   - dependent change?
31    //     - try to map to parent change (either independent or another dependent)
32    //     - note: need to keep track of a parent change stack, since a change can be a parent of multiple changes
33    // - Apply changes
34    //   - find changes to apply to real tree by applying nested changes first
35    //   - changed nodes become part of the changed node set (useful for the formatter to only change those parts)
36    // - Propagate annotations
37
38    let SyntaxEditor { root, mut changes, mappings, annotations } = editor;
39
40    let mut node_depths = FxHashMap::<SyntaxNode, usize>::default();
41    let mut get_node_depth = |node: SyntaxNode| {
42        *node_depths.entry(node).or_insert_with_key(|node| node.ancestors().count())
43    };
44
45    // Sort changes by range, then depth, then change kind, so that we can:
46    // - ensure that parent edits are ordered before child edits
47    // - ensure that inserts will be guaranteed to be inserted at the right range
48    // - easily check for disjoint replace ranges
49    changes.sort_by(|a, b| {
50        a.target_range()
51            .start()
52            .cmp(&b.target_range().start())
53            .then_with(|| {
54                let a_target = a.target_parent();
55                let b_target = b.target_parent();
56
57                if a_target == b_target {
58                    return Ordering::Equal;
59                }
60
61                get_node_depth(a_target).cmp(&get_node_depth(b_target))
62            })
63            .then(a.change_kind().cmp(&b.change_kind()))
64    });
65
66    let disjoint_replaces_ranges = changes
67        .iter()
68        .zip(changes.iter().skip(1))
69        .filter(|(l, r)| {
70            // We only care about checking for disjoint replace ranges
71            matches!(
72                (l.change_kind(), r.change_kind()),
73                (
74                    ChangeKind::Replace | ChangeKind::ReplaceRange,
75                    ChangeKind::Replace | ChangeKind::ReplaceRange
76                )
77            )
78        })
79        .all(|(l, r)| {
80            get_node_depth(l.target_parent()) != get_node_depth(r.target_parent())
81                || (l.target_range().end() <= r.target_range().start())
82        });
83
84    if !disjoint_replaces_ranges {
85        report_intersecting_changes(&changes, get_node_depth, &root);
86
87        return SyntaxEdit {
88            old_root: root.clone(),
89            new_root: root,
90            annotations: Default::default(),
91            changed_elements: vec![],
92        };
93    }
94
95    #[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
96    struct DependentChange {
97        parent: u32,
98        child: u32,
99    }
100
101    // Build change tree
102    let mut changed_ancestors: VecDeque<ChangedAncestor> = VecDeque::new();
103    let mut dependent_changes = vec![];
104    let mut independent_changes = vec![];
105    let mut outdated_changes = vec![];
106
107    for (change_index, change) in changes.iter().enumerate() {
108        // Check if this change is dependent on another change (i.e. it's contained within another range)
109        if let Some(index) = changed_ancestors
110            .iter()
111            .rev()
112            .position(|ancestor| ancestor.affected_range().contains_range(change.target_range()))
113        {
114            // Pop off any ancestors that aren't applicable
115            changed_ancestors.drain((index + 1)..);
116
117            // FIXME: Resolve changes that depend on a range of elements
118            let ancestor = &changed_ancestors[index];
119
120            if let Change::Replace(_, None) = changes[ancestor.change_index] {
121                outdated_changes.push(change_index as u32);
122            } else {
123                dependent_changes.push(DependentChange {
124                    parent: ancestor.change_index as u32,
125                    child: change_index as u32,
126                });
127            }
128        } else {
129            // This change is independent of any other change
130
131            // Drain the changed ancestors since we're no longer in a set of dependent changes
132            changed_ancestors.drain(..);
133
134            independent_changes.push(change_index as u32);
135        }
136
137        // Add to changed ancestors, if applicable
138        match change {
139            Change::Replace(SyntaxElement::Node(target), _)
140            | Change::ReplaceWithMany(SyntaxElement::Node(target), _) => {
141                changed_ancestors.push_back(ChangedAncestor::single(target, change_index))
142            }
143            Change::ReplaceAll(range, _) => {
144                changed_ancestors.push_back(ChangedAncestor::multiple(range, change_index))
145            }
146            _ => (),
147        }
148    }
149
150    // Map change targets to the correct syntax nodes
151    let tree_mutator = TreeMutator::new(&root);
152    let mut changed_elements = vec![];
153
154    for index in independent_changes {
155        match &mut changes[index as usize] {
156            Change::Insert(target, _) | Change::InsertAll(target, _) => {
157                match &mut target.repr {
158                    PositionRepr::FirstChild(parent) => {
159                        *parent = tree_mutator.make_syntax_mut(parent);
160                    }
161                    PositionRepr::After(child) => {
162                        *child = tree_mutator.make_element_mut(child);
163                    }
164                };
165            }
166            Change::Replace(SyntaxElement::Node(target), Some(SyntaxElement::Node(new_target))) => {
167                *target = tree_mutator.make_syntax_mut(target);
168                if new_target.ancestors().any(|node| node == tree_mutator.immutable) {
169                    *new_target = new_target.clone_for_update();
170                }
171            }
172            Change::Replace(target, _) | Change::ReplaceWithMany(target, _) => {
173                *target = tree_mutator.make_element_mut(target);
174            }
175            Change::ReplaceAll(range, _) => {
176                let start = tree_mutator.make_element_mut(range.start());
177                let end = tree_mutator.make_element_mut(range.end());
178
179                *range = start..=end;
180            }
181        }
182
183        // Collect changed elements
184        match &changes[index as usize] {
185            Change::Insert(_, element) => changed_elements.push(element.clone()),
186            Change::InsertAll(_, elements) => changed_elements.extend(elements.iter().cloned()),
187            Change::Replace(_, Some(element)) => changed_elements.push(element.clone()),
188            Change::Replace(_, None) => {}
189            Change::ReplaceWithMany(_, elements) => {
190                changed_elements.extend(elements.iter().cloned())
191            }
192            Change::ReplaceAll(_, elements) => changed_elements.extend(elements.iter().cloned()),
193        }
194    }
195
196    for DependentChange { parent, child } in dependent_changes.into_iter() {
197        let (input_ancestor, output_ancestor) = match &changes[parent as usize] {
198            // No change will depend on an insert since changes can only depend on nodes in the root tree
199            Change::Insert(_, _) | Change::InsertAll(_, _) => unreachable!(),
200            Change::Replace(target, Some(new_target)) => {
201                (to_owning_node(target), to_owning_node(new_target))
202            }
203            Change::Replace(_, None) => {
204                unreachable!("deletions should not generate dependent changes")
205            }
206            Change::ReplaceAll(_, _) | Change::ReplaceWithMany(_, _) => {
207                unimplemented!("cannot resolve changes that depend on replacing many elements")
208            }
209        };
210
211        let upmap_target_node = |target: &SyntaxNode| match mappings.upmap_child(
212            target,
213            &input_ancestor,
214            &output_ancestor,
215        ) {
216            Ok(it) => it,
217            Err(MissingMapping(current)) => unreachable!(
218                "no mappings exist between {current:?} (ancestor of {input_ancestor:?}) and {output_ancestor:?}"
219            ),
220        };
221
222        let upmap_target = |target: &SyntaxElement| match mappings.upmap_child_element(
223            target,
224            &input_ancestor,
225            &output_ancestor,
226        ) {
227            Ok(it) => it,
228            Err(MissingMapping(current)) => unreachable!(
229                "no mappings exist between {current:?} (ancestor of {input_ancestor:?}) and {output_ancestor:?}"
230            ),
231        };
232
233        match &mut changes[child as usize] {
234            Change::Insert(target, _) | Change::InsertAll(target, _) => match &mut target.repr {
235                PositionRepr::FirstChild(parent) => {
236                    *parent = upmap_target_node(parent);
237                }
238                PositionRepr::After(child) => {
239                    *child = upmap_target(child);
240                }
241            },
242            Change::Replace(target, _) | Change::ReplaceWithMany(target, _) => {
243                *target = upmap_target(target);
244            }
245            Change::ReplaceAll(range, _) => {
246                *range = upmap_target(range.start())..=upmap_target(range.end());
247            }
248        }
249    }
250
251    // We reverse here since we pushed to this in ascending order,
252    // and we want to remove elements in descending order
253    for idx in outdated_changes.into_iter().rev() {
254        changes.remove(idx as usize);
255    }
256
257    // Apply changes
258    let mut root = tree_mutator.mutable_clone;
259
260    for change in changes {
261        match change {
262            Change::Insert(position, element) => {
263                let (parent, index) = position.place();
264                parent.splice_children(index..index, vec![element]);
265            }
266            Change::InsertAll(position, elements) => {
267                let (parent, index) = position.place();
268                parent.splice_children(index..index, elements);
269            }
270            Change::Replace(target, None) => {
271                target.detach();
272            }
273            Change::Replace(SyntaxElement::Node(target), Some(new_target)) if target == root => {
274                root = new_target.into_node().expect("root node replacement should be a node");
275            }
276            Change::Replace(target, Some(new_target)) => {
277                let parent = target.parent().unwrap();
278                parent.splice_children(target.index()..target.index() + 1, vec![new_target]);
279            }
280            Change::ReplaceWithMany(target, elements) => {
281                let parent = target.parent().unwrap();
282                parent.splice_children(target.index()..target.index() + 1, elements);
283            }
284            Change::ReplaceAll(range, elements) => {
285                let start = range.start().index();
286                let end = range.end().index();
287                let parent = range.start().parent().unwrap();
288                parent.splice_children(start..end + 1, elements);
289            }
290        }
291    }
292
293    // Propagate annotations
294    let annotations = annotations.into_iter().filter_map(|(element, annotation)| {
295        match mappings.upmap_element(&element, &root) {
296            // Needed to follow the new tree to find the resulting element
297            Some(Ok(mapped)) => Some((mapped, annotation)),
298            // Element did not need to be mapped
299            None => Some((element, annotation)),
300            // Element did not make it to the final tree
301            Some(Err(_)) => None,
302        }
303    });
304
305    let mut annotation_groups = FxHashMap::default();
306
307    for (element, annotation) in annotations {
308        annotation_groups.entry(annotation).or_insert(vec![]).push(element);
309    }
310
311    SyntaxEdit {
312        old_root: tree_mutator.immutable,
313        new_root: root,
314        changed_elements,
315        annotations: annotation_groups,
316    }
317}

319fn report_intersecting_changes(
320    changes: &[Change],
321    mut get_node_depth: impl FnMut(rowan::SyntaxNode<crate::RustLanguage>) -> usize,
322    root: &rowan::SyntaxNode<crate::RustLanguage>,
323) {
324    let intersecting_changes = changes
325        .iter()
326        .zip(changes.iter().skip(1))
327        .filter(|(l, r)| {
328            // We only care about checking for disjoint replace ranges.
329            matches!(
330                (l.change_kind(), r.change_kind()),
331                (
332                    ChangeKind::Replace | ChangeKind::ReplaceRange,
333                    ChangeKind::Replace | ChangeKind::ReplaceRange
334                )
335            )
336        })
337        .filter(|(l, r)| {
338            get_node_depth(l.target_parent()) == get_node_depth(r.target_parent())
339                && (l.target_range().end() > r.target_range().start())
340        });
341
342    let mut error_msg = String::from("some replace change ranges intersect!\n");
343
344    let parent_str = root.to_string();
345
346    for (l, r) in intersecting_changes {
347        let mut highlighted_str = parent_str.clone();
348        let l_range = l.target_range();
349        let r_range = r.target_range();
350
351        let i_range = l_range.intersect(r_range).unwrap();
352        let i_str = format!("\x1b[46m{}", &parent_str[i_range]);
353
354        let pre_range: Range<usize> = l_range.start().into()..i_range.start().into();
355        let pre_str = format!("\x1b[44m{}", &parent_str[pre_range]);
356
357        let (highlight_range, highlight_str) = if l_range == r_range {
358            format_to!(error_msg, "\x1b[46mleft change:\x1b[0m  {l:?} {l}\n");
359            format_to!(error_msg, "\x1b[46mequals\x1b[0m\n");
360            format_to!(error_msg, "\x1b[46mright change:\x1b[0m {r:?} {r}\n");
361            let i_highlighted = format!("{i_str}\x1b[0m\x1b[K");
362            let total_range: Range<usize> = i_range.into();
363            (total_range, i_highlighted)
364        } else {
365            format_to!(error_msg, "\x1b[44mleft change:\x1b[0m  {l:?} {l}\n");
366            let range_end = if l_range.contains_range(r_range) {
367                format_to!(error_msg, "\x1b[46mcovers\x1b[0m\n");
368                format_to!(error_msg, "\x1b[46mright change:\x1b[0m {r:?} {r}\n");
369                l_range.end()
370            } else {
371                format_to!(error_msg, "\x1b[46mintersects\x1b[0m\n");
372                format_to!(error_msg, "\x1b[42mright change:\x1b[0m {r:?} {r}\n");
373                r_range.end()
374            };
375
376            let post_range: Range<usize> = i_range.end().into()..range_end.into();
377
378            let post_str = format!("\x1b[42m{}", &parent_str[post_range]);
379            let result = format!("{pre_str}{i_str}{post_str}\x1b[0m\x1b[K");
380            let total_range: Range<usize> = l_range.start().into()..range_end.into();
381            (total_range, result)
382        };
383        highlighted_str.replace_range(highlight_range, &highlight_str);
384
385        format_to!(error_msg, "{highlighted_str}\n");
386    }
387
388    stdx::always!(false, "{}", error_msg);
389}

391fn to_owning_node(element: &SyntaxElement) -> SyntaxNode {
392    match element {
393        SyntaxElement::Node(node) => node.clone(),
394        SyntaxElement::Token(token) => token.parent().unwrap(),
395    }
396}

398struct ChangedAncestor {
399    kind: ChangedAncestorKind,
400    change_index: usize,
401}
402
403enum ChangedAncestorKind {
404    Single { node: SyntaxNode },
405    Range { _changed_elements: RangeInclusive<SyntaxElement>, _in_parent: SyntaxNode },
406}
407
408impl ChangedAncestor {
409    fn single(node: &SyntaxNode, change_index: usize) -> Self {
410        let kind = ChangedAncestorKind::Single { node: node.clone() };
411
412        Self { kind, change_index }
413    }
414
415    fn multiple(range: &RangeInclusive<SyntaxElement>, change_index: usize) -> Self {
416        Self {
417            kind: ChangedAncestorKind::Range {
418                _changed_elements: range.clone(),
419                _in_parent: range.start().parent().unwrap(),
420            },
421            change_index,
422        }
423    }
424
425    fn affected_range(&self) -> TextRange {
426        match &self.kind {
427            ChangedAncestorKind::Single { node } => node.text_range(),
428            ChangedAncestorKind::Range { _changed_elements: changed_nodes, _in_parent: _ } => {
429                TextRange::new(
430                    changed_nodes.start().text_range().start(),
431                    changed_nodes.end().text_range().end(),
432                )
433            }
434        }
435    }
436}

438struct TreeMutator {
439    immutable: SyntaxNode,
440    mutable_clone: SyntaxNode,
441}
442
443impl TreeMutator {
444    fn new(immutable: &SyntaxNode) -> TreeMutator {
445        let immutable = immutable.clone();
446        let mutable_clone = immutable.clone_for_update();
447        TreeMutator { immutable, mutable_clone }
448    }
449
450    fn make_element_mut(&self, element: &SyntaxElement) -> SyntaxElement {
451        match element {
452            SyntaxElement::Node(node) => SyntaxElement::Node(self.make_syntax_mut(node)),
453            SyntaxElement::Token(token) => {
454                let parent = self.make_syntax_mut(&token.parent().unwrap());
455                parent.children_with_tokens().nth(token.index()).unwrap()
456            }
457        }
458    }
459
460    fn make_syntax_mut(&self, node: &SyntaxNode) -> SyntaxNode {
461        let ptr = SyntaxNodePtr::new(node);
462        ptr.to_node(&self.mutable_clone)
463    }
