41/// The main trait to go from untyped `SyntaxNode`  to a typed ast. The
42/// conversion itself has zero runtime cost: ast and syntax nodes have exactly
43/// the same representation: a pointer to the tree root and a pointer to the
44/// node itself.
45pub trait AstNode {
46    /// This panics if the `SyntaxKind` is not statically known.
47    fn kind() -> SyntaxKind
48    where
49        Self: Sized,
50    {
51        panic!("dynamic `SyntaxKind` for `AstNode::kind()`")
52    }
53
54    fn can_cast(kind: SyntaxKind) -> bool
55    where
56        Self: Sized;
57
58    fn cast(syntax: SyntaxNode) -> Option<Self>
59    where
60        Self: Sized;
61
62    fn syntax(&self) -> &SyntaxNode;
63    fn clone_for_update(&self) -> Self
64    where
65        Self: Sized,
66    {
67        Self::cast(self.syntax().clone_for_update()).unwrap()
68    }
69    fn clone_subtree(&self) -> Self
70    where
71        Self: Sized,
72    {
73        Self::cast(self.syntax().clone_subtree()).unwrap()
74    }
75}

77/// Like `AstNode`, but wraps tokens rather than interior nodes.
78pub trait AstToken {
79    fn can_cast(token: SyntaxKind) -> bool
80    where
81        Self: Sized;
82
83    fn cast(syntax: SyntaxToken) -> Option<Self>
84    where
85        Self: Sized;
86
87    fn syntax(&self) -> &SyntaxToken;
88
89    fn text(&self) -> &str {
90        self.syntax().text()
91    }
92}

94/// An iterator over `SyntaxNode` children of a particular AST type.
95#[derive(Debug, Clone)]
96pub struct AstChildren<N> {
97    inner: SyntaxNodeChildren,
98    ph: PhantomData<N>,
99}
100
101impl<N> AstChildren<N> {
102    fn new(parent: &SyntaxNode) -> Self {
103        AstChildren { inner: parent.children(), ph: PhantomData }
104    }
105}
106
107impl<N: AstNode> Iterator for AstChildren<N> {
108    type Item = N;
109    fn next(&mut self) -> Option<N> {
110        self.inner.find_map(N::cast)
111    }
112}

114impl<L, R> AstNode for Either<L, R>
115where
116    L: AstNode,
117    R: AstNode,
118{
119    fn can_cast(kind: SyntaxKind) -> bool
120    where
121        Self: Sized,
122    {
123        L::can_cast(kind) || R::can_cast(kind)
124    }
125
126    fn cast(syntax: SyntaxNode) -> Option<Self>
127    where
128        Self: Sized,
129    {
130        if L::can_cast(syntax.kind()) {
131            L::cast(syntax).map(Either::Left)
132        } else {
133            R::cast(syntax).map(Either::Right)
134        }
135    }
136
137    fn syntax(&self) -> &SyntaxNode {
138        self.as_ref().either(L::syntax, R::syntax)
139    }
140}

159mod support {
160    use super::{AstChildren, AstNode, SyntaxKind, SyntaxNode, SyntaxToken};
161
162    #[inline]
163    pub(super) fn child<N: AstNode>(parent: &SyntaxNode) -> Option<N> {
164        parent.children().find_map(N::cast)
165    }
166
167    #[inline]
168    pub(super) fn children<N: AstNode>(parent: &SyntaxNode) -> AstChildren<N> {
169        AstChildren::new(parent)
170    }
171
172    #[inline]
173    pub(super) fn token(parent: &SyntaxNode, kind: SyntaxKind) -> Option<SyntaxToken> {
174        parent.children_with_tokens().filter_map(|it| it.into_token()).find(|it| it.kind() == kind)
175    }

368#[test]
369fn test_where_predicates() {
370    fn assert_bound(text: &str, bound: Option<TypeBound>) {
371        assert_eq!(text, bound.unwrap().syntax().text().to_string());
372    }
373
374    let file = SourceFile::parse(
375        r#"
376fn foo()
377where
378   T: Clone + Copy + Debug + 'static,
379   'a: 'b + 'c,
380   Iterator::Item: 'a + Debug,
381   Iterator::Item: Debug + 'a,
382   <T as Iterator>::Item: Debug + 'a,
383   for<'a> F: Fn(&'a str)
384{}
385        "#,
386        parser::Edition::CURRENT,
387    )
388    .ok()
389    .unwrap();
390    let where_clause = file.syntax().descendants().find_map(WhereClause::cast).unwrap();
391
392    let mut predicates = where_clause.predicates();
393
394    let pred = predicates.next().unwrap();
395    let mut bounds = pred.type_bound_list().unwrap().bounds();
396
397    assert!(pred.for_binder().is_none());
398    assert_eq!("T", pred.ty().unwrap().syntax().text().to_string());
399    assert_bound("Clone", bounds.next());
400    assert_bound("Copy", bounds.next());
401    assert_bound("Debug", bounds.next());
402    assert_bound("'static", bounds.next());
403
404    let pred = predicates.next().unwrap();
405    let mut bounds = pred.type_bound_list().unwrap().bounds();
406
407    assert_eq!("'a", pred.lifetime().unwrap().lifetime_ident_token().unwrap().text());
408
409    assert_bound("'b", bounds.next());
410    assert_bound("'c", bounds.next());
411
412    let pred = predicates.next().unwrap();
413    let mut bounds = pred.type_bound_list().unwrap().bounds();
414
415    assert_eq!("Iterator::Item", pred.ty().unwrap().syntax().text().to_string());
416    assert_bound("'a", bounds.next());
417
418    let pred = predicates.next().unwrap();
419    let mut bounds = pred.type_bound_list().unwrap().bounds();
420
421    assert_eq!("Iterator::Item", pred.ty().unwrap().syntax().text().to_string());
422    assert_bound("Debug", bounds.next());
423    assert_bound("'a", bounds.next());
424
425    let pred = predicates.next().unwrap();
426    let mut bounds = pred.type_bound_list().unwrap().bounds();
427
428    assert_eq!("<T as Iterator>::Item", pred.ty().unwrap().syntax().text().to_string());
429    assert_bound("Debug", bounds.next());
430    assert_bound("'a", bounds.next());
431
432    let pred = predicates.next().unwrap();
433    let mut bounds = pred.type_bound_list().unwrap().bounds();
434
435    assert_eq!(
436        "<'a>",
437        pred.for_binder().unwrap().generic_param_list().unwrap().syntax().text().to_string()
438    );
439    assert_eq!("F", pred.ty().unwrap().syntax().text().to_string());
440    assert_bound("Fn(&'a str)", bounds.next());
