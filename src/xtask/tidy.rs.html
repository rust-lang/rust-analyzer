23fn check_lsp_extensions_docs(sh: &Shell) {
24    let expected_hash = {
25        let lsp_ext_rs =
26            sh.read_file(project_root().join("crates/rust-analyzer/src/lsp/ext.rs")).unwrap();
27        stable_hash(lsp_ext_rs.as_str())
28    };
29
30    let actual_hash = {
31        let lsp_extensions_md = sh
32            .read_file(project_root().join("docs/book/src/contributing/lsp-extensions.md"))
33            .unwrap();
34        let text = lsp_extensions_md
35            .lines()
36            .find_map(|line| line.strip_prefix("lsp/ext.rs hash:"))
37            .unwrap()
38            .trim();
39        u64::from_str_radix(text, 16).unwrap()
40    };
41
42    if actual_hash != expected_hash {
43        panic!(
44            "
45lsp/ext.rs was changed without touching lsp-extensions.md.
46
47Expected hash: {expected_hash:x}
48Actual hash:   {actual_hash:x}
49
50Please adjust docs/book/src/contributing/lsp-extensions.md.
51"
52        )
53    }
54}

56fn files_are_tidy(sh: &Shell) {
57    let files = list_files(&project_root().join("crates"));
58
59    let mut tidy_docs = TidyDocs::default();
60    let mut tidy_marks = TidyMarks::default();
61    for path in files {
62        let extension = path.extension().unwrap_or_default().to_str().unwrap_or_default();
63        match extension {
64            "rs" => {
65                let text = sh.read_file(&path).unwrap();
66                check_test_attrs(&path, &text);
67                check_trailing_ws(&path, &text);
68                tidy_docs.visit(&path, &text);
69                tidy_marks.visit(&path, &text);
70            }
71            "toml" => {
72                let text = sh.read_file(&path).unwrap();
73                check_cargo_toml(&path, text);
74            }
75            _ => (),
76        }
77    }
78
79    tidy_docs.finish();
80    tidy_marks.finish();
81}

83fn check_cargo_toml(path: &Path, text: String) {
84    let mut section = None;
85    for (line_no, text) in text.lines().enumerate() {
86        let text = text.trim();
87        if text.starts_with('[') {
88            if !text.ends_with(']') {
89                panic!(
90                    "\nplease don't add comments or trailing whitespace in section lines.\n\
91                        {}:{}\n",
92                    path.display(),
93                    line_no + 1
94                )
95            }
96            section = Some(text);
97            continue;
98        }
99        let text: String = text.split_whitespace().collect();
100        if !text.contains("path=") {
101            continue;
102        }
103        match section {
104            Some(s) if s.contains("dev-dependencies") => {
105                if text.contains("version") {
106                    panic!(
107                        "\ncargo internal dev-dependencies should not have a version.\n\
108                        {}:{}\n",
109                        path.display(),
110                        line_no + 1
111                    );
112                }
113            }
114            Some(s) if s.contains("dependencies") => {
115                if !text.contains("version") {
116                    panic!(
117                        "\ncargo internal dependencies should have a version.\n\
118                        {}:{}\n",
119                        path.display(),
120                        line_no + 1
121                    );
122                }
123            }
124            _ => {}
125        }
126    }
127}

129fn check_licenses(sh: &Shell) {
130    const EXPECTED: [&str; 20] = [
131        "(MIT OR Apache-2.0) AND Unicode-3.0",
132        "0BSD OR MIT OR Apache-2.0",
133        "Apache-2.0",
134        "Apache-2.0 OR BSL-1.0",
135        "Apache-2.0 OR MIT",
136        "Apache-2.0 WITH LLVM-exception",
137        "Apache-2.0 WITH LLVM-exception OR Apache-2.0 OR MIT",
138        "Apache-2.0/MIT",
139        "CC0-1.0",
140        "ISC",
141        "MIT",
142        "MIT / Apache-2.0",
143        "MIT OR Apache-2.0",
144        "MIT OR Zlib OR Apache-2.0",
145        "MIT/Apache-2.0",
146        "MPL-2.0",
147        "Unicode-3.0",
148        "Unlicense OR MIT",
149        "Unlicense/MIT",
150        "Zlib",
151    ];
152
153    let meta = cmd!(sh, "cargo metadata --format-version 1").read().unwrap();
154    let mut licenses = meta
155        .split([',', '{', '}'])
156        .filter(|it| it.contains(r#""license""#))
157        .map(|it| it.trim())
158        .map(|it| it[r#""license":"#.len()..].trim_matches('"'))
159        .collect::<Vec<_>>();
160    licenses.sort_unstable();
161    licenses.dedup();
162    if licenses != EXPECTED {
163        let mut diff = String::new();
164
165        diff.push_str("New Licenses:\n");
166        for &l in licenses.iter() {
167            if !EXPECTED.contains(&l) {
168                diff += &format!("  {l}\n")
169            }
170        }
171
172        diff.push_str("\nMissing Licenses:\n");
173        for l in EXPECTED {
174            if !licenses.contains(&l) {
175                diff += &format!("  {l}\n")
176            }
177        }
178
179        panic!("different set of licenses!\n{diff}");
180    }
181    assert_eq!(licenses, EXPECTED);
182}

184fn check_test_attrs(path: &Path, text: &str) {
185    let panic_rule = "https://github.com/rust-lang/rust-analyzer/blob/master/docs/book/src/contributing/style.md#should_panic";
186    let need_panic: &[&str] = &[
187        // This file.
188        "slow-tests/tidy.rs",
189        "test-utils/src/fixture.rs",
190        // Generated code from lints contains doc tests in string literals.
191        "ide-db/src/generated/lints.rs",
192    ];
193    if need_panic.iter().any(|p| path.ends_with(p)) {
194        return;
195    }
196    if let Some((line, _)) = text
197        .lines()
198        .tuple_windows()
199        .enumerate()
200        .find(|(_, (a, b))| b.contains("#[should_panic") && !a.contains("FIXME"))
201    {
202        panic!(
203            "\ndon't add `#[should_panic]` tests, see:\n\n    {}\n\n   {}:{line}\n",
204            panic_rule,
205            path.display(),
206        )
207    }
208}

210fn check_trailing_ws(path: &Path, text: &str) {
211    if is_exclude_dir(path, &["test_data"]) {
212        return;
213    }
214    for (line_number, line) in text.lines().enumerate() {
215        if line.chars().last().is_some_and(char::is_whitespace) {
216            panic!("Trailing whitespace in {} at line {}", path.display(), line_number + 1)
217        }
218    }
219}

221#[derive(Default)]
222struct TidyDocs {
223    missing_docs: Vec<String>,
224    contains_fixme: Vec<PathBuf>,
225}
226
227impl TidyDocs {
228    fn visit(&mut self, path: &Path, text: &str) {
229        // Tests and diagnostic fixes don't need module level comments.
230        if is_exclude_dir(path, &["tests", "test_data", "fixes", "grammar", "stdx"]) {
231            return;
232        }
233
234        if is_exclude_file(path) {
235            return;
236        }
237
238        if is_ported_from_rustc(path, &["crates/hir-ty/src/next_solver"]) {
239            return;
240        }
241
242        let first_line = match text.lines().next() {
243            Some(it) => it,
244            None => return,
245        };
246
247        if first_line.starts_with("//!") {
248            if first_line.contains("FIXME") {
249                self.contains_fixme.push(path.to_path_buf());
250            }
251        } else {
252            if text.contains("// Feature:")
253                || text.contains("// Assist:")
254                || text.contains("// Diagnostic:")
255            {
256                return;
257            }
258            self.missing_docs.push(path.display().to_string());
259        }
260
261        fn is_exclude_file(d: &Path) -> bool {
262            let file_names = ["tests.rs", "famous_defs_fixture.rs", "frontmatter.rs"];
263
264            d.file_name()
265                .unwrap_or_default()
266                .to_str()
267                .map(|f_n| file_names.contains(&f_n))
268                .unwrap_or(false)
269        }
270    }
271
272    fn finish(self) {
273        if !self.missing_docs.is_empty() {
274            panic!(
275                "\nMissing docs strings\n\n\
276                 modules:\n{}\n\n",
277                self.missing_docs.join("\n")
278            )
279        }
280
281        if let Some(path) = self.contains_fixme.first() {
282            panic!("FIXME doc in a fully-documented crate: {}", path.display())
283        }
284    }
285}

287fn is_exclude_dir(p: &Path, dirs_to_exclude: &[&str]) -> bool {
288    p.strip_prefix(project_root())
289        .unwrap()
290        .components()
291        .rev()
292        .skip(1)
293        .filter_map(|it| it.as_os_str().to_str())
294        .any(|it| dirs_to_exclude.contains(&it))
295}

297fn is_ported_from_rustc(p: &Path, dirs_to_exclude: &[&str]) -> bool {
298    let p = p.strip_prefix(project_root()).unwrap();
299    dirs_to_exclude.iter().any(|exclude| p.starts_with(exclude))
300}

302#[derive(Default)]
303struct TidyMarks {
304    hits: HashSet<String>,
305    checks: HashSet<String>,
306}
307
308impl TidyMarks {
309    fn visit(&mut self, _path: &Path, text: &str) {
310        find_marks(&mut self.hits, text, "hit");
311        find_marks(&mut self.checks, text, "check");
312        find_marks(&mut self.checks, text, "check_count");
313    }
314
315    fn finish(self) {
316        assert!(!self.hits.is_empty());
317
318        let diff: Vec<_> =
319            self.hits.symmetric_difference(&self.checks).map(|it| it.as_str()).collect();
320
321        if !diff.is_empty() {
322            panic!("unpaired marks: {diff:?}")
323        }
324    }
325}

327#[allow(deprecated)]
328fn stable_hash(text: &str) -> u64 {
329    use std::hash::{Hash, Hasher, SipHasher};
330
331    let text = text.replace('\r', "");
332    let mut hasher = SipHasher::default();
333    text.hash(&mut hasher);
334    hasher.finish()
335}

337fn find_marks(set: &mut HashSet<String>, text: &str, mark: &str) {
338    let mut text = text;
339    let mut prev_text = "";
340    while text != prev_text {
341        prev_text = text;
342        if let Some(idx) = text.find(mark) {
343            text = &text[idx + mark.len()..];
344            if let Some(stripped_text) = text.strip_prefix("!(") {
345                text = stripped_text.trim_start();
346                if let Some(idx2) = text.find(|c: char| !(c.is_alphanumeric() || c == '_')) {
347                    let mark_text = &text[..idx2];
348                    set.insert(mark_text.to_owned());
349                    text = &text[idx2..];
350                }
351            }
352        }
353    }
354}
