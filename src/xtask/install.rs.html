13impl flags::Install {
14    pub(crate) fn run(self, sh: &Shell) -> anyhow::Result<()> {
15        if cfg!(target_os = "macos") {
16            fix_path_for_mac(sh).context("Fix path for mac")?;
17        }
18        if let Some(server) = self.server() {
19            install_server(sh, server).context("install server")?;
20        }
21        if let Some(server) = self.proc_macro_server() {
22            install_proc_macro_server(sh, server).context("install proc-macro server")?;
23        }
24        if let Some(client) = self.client() {
25            install_client(sh, client).context("install client")?;
26        }
27        Ok(())
28    }
29}

60fn fix_path_for_mac(sh: &Shell) -> anyhow::Result<()> {
61    let mut vscode_path: Vec<PathBuf> = {
62        const COMMON_APP_PATH: &str =
63            r"/Applications/Visual Studio Code.app/Contents/Resources/app/bin";
64        const ROOT_DIR: &str = "";
65        let home_dir = sh.var("HOME").map_err(|err| {
66            format_err!("Failed getting HOME from environment with error: {}.", err)
67        })?;
68
69        [ROOT_DIR, &home_dir]
70            .into_iter()
71            .map(|dir| dir.to_owned() + COMMON_APP_PATH)
72            .map(PathBuf::from)
73            .filter(|path| path.exists())
74            .collect()
75    };
76
77    if !vscode_path.is_empty() {
78        let vars = sh.var_os("PATH").context("Could not get PATH variable from env.")?;
79
80        let mut paths = env::split_paths(&vars).collect::<Vec<_>>();
81        paths.append(&mut vscode_path);
82        let new_paths = env::join_paths(paths).context("build env PATH")?;
83        sh.set_var("PATH", new_paths);
84    }
85
86    Ok(())
87}

89fn install_client(sh: &Shell, client_opt: ClientOpt) -> anyhow::Result<()> {
90    let _dir = sh.push_dir("./editors/code");
91
92    // Package extension.
93    if cfg!(unix) {
94        cmd!(sh, "npm --version").run().context("`npm` is required to build the VS Code plugin")?;
95        cmd!(sh, "npm ci").run()?;
96
97        cmd!(sh, "npm run package --scripts-prepend-node-path").run()?;
98    } else {
99        cmd!(sh, "cmd.exe /c npm --version")
100            .run()
101            .context("`npm` is required to build the VS Code plugin")?;
102        cmd!(sh, "cmd.exe /c npm ci").run()?;
103
104        cmd!(sh, "cmd.exe /c npm run package").run()?;
105    };
106
107    // Find the appropriate VS Code binary.
108    let lifetime_extender;
109    let candidates: &[&str] = match client_opt.code_bin.as_deref() {
110        Some(it) => {
111            lifetime_extender = [it];
112            &lifetime_extender[..]
113        }
114        None => VS_CODES,
115    };
116    let code = candidates
117        .iter()
118        .copied()
119        .find(|&bin| {
120            if cfg!(unix) {
121                cmd!(sh, "{bin} --version").read().is_ok()
122            } else {
123                cmd!(sh, "cmd.exe /c {bin}.cmd --version").read().is_ok()
124            }
125        })
126        .ok_or_else(|| {
127            format_err!("Can't execute `{} --version`. Perhaps it is not in $PATH?", candidates[0])
128        })?;
129
130    // Install & verify.
131    let installed_extensions = if cfg!(unix) {
132        cmd!(sh, "{code} --install-extension rust-analyzer.vsix --force").run()?;
133        cmd!(sh, "{code} --list-extensions").read()?
134    } else {
135        cmd!(sh, "cmd.exe /c {code}.cmd --install-extension rust-analyzer.vsix --force").run()?;
136        cmd!(sh, "cmd.exe /c {code}.cmd --list-extensions").read()?
137    };
138
139    if !installed_extensions.contains("rust-analyzer") {
140        bail!(
141            "Could not install the Visual Studio Code extension. \
142            Please make sure you have at least NodeJS 16.x together with the latest version of VS Code installed and try again. \
143            Note that installing via xtask install does not work for VS Code Remote, instead you’ll need to install the .vsix manually."
144        );
145    }
146
147    Ok(())
148}

150fn install_server(sh: &Shell, opts: ServerOpt) -> anyhow::Result<()> {
151    let features = &opts.to_features();
152    let profile = if opts.dev_rel { "dev-rel" } else { "release" };
153
154    let mut install_cmd = cmd!(
155        sh,
156        "cargo install --path crates/rust-analyzer --profile={profile} --locked --force --features force-always-assert {features...}"
157    );
158
159    if let Some(train_crate) = opts.pgo {
160        let target = detect_target(sh);
161        let build_cmd = cmd!(
162            sh,
163            "cargo build --manifest-path ./crates/rust-analyzer/Cargo.toml --bin rust-analyzer --target {target} --profile={profile} --locked {features...}"
164        );
165
166        let profile = crate::pgo::gather_pgo_profile(sh, build_cmd, &target, train_crate)?;
167        install_cmd = crate::pgo::apply_pgo_to_cmd(install_cmd, &profile);
168    }
169
170    install_cmd.run()?;
171    Ok(())
172}

174fn install_proc_macro_server(sh: &Shell, opts: ProcMacroServerOpt) -> anyhow::Result<()> {
175    let profile = if opts.dev_rel { "dev-rel" } else { "release" };
176
177    cmd!(
178        sh,
179        "cargo +nightly install --path crates/proc-macro-srv-cli --profile={profile} --locked --force --features sysroot-abi"
180    ).run()?;
181
182    Ok(())
