26/// Syntactical attributes, without filtering of `cfg_attr`s.
27#[derive(Default, Debug, Clone, PartialEq, Eq)]
28pub struct RawAttrs {
29    // FIXME: This can become `Box<[Attr]>` if https://internals.rust-lang.org/t/layout-of-dst-box/21728?u=chrefr is accepted.
30    entries: Option<ThinArc<(), Attr>>,
31}
32
33impl ops::Deref for RawAttrs {
34    type Target = [Attr];
35
36    fn deref(&self) -> &[Attr] {
37        match &self.entries {
38            Some(it) => &it.slice,
39            None => &[],
40        }
41    }
42}
43
44impl RawAttrs {
45    pub const EMPTY: Self = Self { entries: None };
46
47    pub fn new(
48        db: &dyn ExpandDatabase,
49        owner: &dyn ast::HasAttrs,
50        span_map: SpanMapRef<'_>,
51    ) -> Self {
52        let entries: Vec<_> = collect_attrs(owner)
53            .filter_map(|(id, attr)| match attr {
54                Either::Left(attr) => {
55                    attr.meta().and_then(|meta| Attr::from_src(db, meta, span_map, id))
56                }
57                Either::Right(comment) => comment.doc_comment().map(|doc| {
58                    let span = span_map.span_for_range(comment.syntax().text_range());
59                    let (text, kind) =
60                        desugar_doc_comment_text(doc, DocCommentDesugarMode::ProcMacro);
61                    Attr {
62                        id,
63                        input: Some(Box::new(AttrInput::Literal(tt::Literal {
64                            symbol: text,
65                            span,
66                            kind,
67                            suffix: None,
68                        }))),
69                        path: Interned::new(ModPath::from(Name::new_symbol(
70                            sym::doc.clone(),
71                            span.ctx,
72                        ))),
73                        ctxt: span.ctx,
74                    }
75                }),
76            })
77            .collect();
78
79        let entries = if entries.is_empty() {
80            None
81        } else {
82            Some(ThinArc::from_header_and_iter((), entries.into_iter()))
83        };
84
85        RawAttrs { entries }
86    }
87
88    pub fn from_attrs_owner(
89        db: &dyn ExpandDatabase,
90        owner: InFile<&dyn ast::HasAttrs>,
91        span_map: SpanMapRef<'_>,
92    ) -> Self {
93        Self::new(db, owner.value, span_map)
94    }
95
96    pub fn merge(&self, other: Self) -> Self {
97        match (&self.entries, other.entries) {
98            (None, None) => Self::EMPTY,
99            (None, entries @ Some(_)) => Self { entries },
100            (Some(entries), None) => Self { entries: Some(entries.clone()) },
101            (Some(a), Some(b)) => {
102                let last_ast_index = a.slice.last().map_or(0, |it| it.id.ast_index() + 1) as u32;
103                let items = a
104                    .slice
105                    .iter()
106                    .cloned()
107                    .chain(b.slice.iter().map(|it| {
108                        let mut it = it.clone();
109                        it.id.id = (it.id.ast_index() as u32 + last_ast_index)
110                            | ((it.id.cfg_attr_index().unwrap_or(0) as u32)
111                                << AttrId::AST_INDEX_BITS);
112                        it
113                    }))
114                    .collect::<Vec<_>>();
115                Self { entries: Some(ThinArc::from_header_and_iter((), items.into_iter())) }
116            }
117        }
118    }
119
120    /// Processes `cfg_attr`s, returning the resulting semantic `Attrs`.
121    // FIXME: This should return a different type, signaling it was filtered?
122    pub fn filter(self, db: &dyn ExpandDatabase, krate: Crate) -> RawAttrs {
123        let has_cfg_attrs = self
124            .iter()
125            .any(|attr| attr.path.as_ident().is_some_and(|name| *name == sym::cfg_attr.clone()));
126        if !has_cfg_attrs {
127            return self;
128        }
129
130        let cfg_options = krate.cfg_options(db);
131        let new_attrs =
132            self.iter()
133                .flat_map(|attr| -> SmallVec<[_; 1]> {
134                    let is_cfg_attr =
135                        attr.path.as_ident().is_some_and(|name| *name == sym::cfg_attr.clone());
136                    if !is_cfg_attr {
137                        return smallvec![attr.clone()];
138                    }
139
140                    let subtree = match attr.token_tree_value() {
141                        Some(it) => it,
142                        _ => return smallvec![attr.clone()],
143                    };
144
145                    let (cfg, parts) = match parse_cfg_attr_input(subtree) {
146                        Some(it) => it,
147                        None => return smallvec![attr.clone()],
148                    };
149                    let index = attr.id;
150                    let attrs = parts.enumerate().take(1 << AttrId::CFG_ATTR_BITS).filter_map(
151                        |(idx, attr)| Attr::from_tt(db, attr, index.with_cfg_attr(idx)),
152                    );
153
154                    let cfg = TopSubtree::from_token_trees(subtree.top_subtree().delimiter, cfg);
155                    let cfg = CfgExpr::parse(&cfg);
156                    if cfg_options.check(&cfg) == Some(false) {
157                        smallvec![]
158                    } else {
159                        cov_mark::hit!(cfg_attr_active);
160
161                        attrs.collect()
162                    }
163                })
164                .collect::<Vec<_>>();
165        let entries = if new_attrs.is_empty() {
166            None
167        } else {
168            Some(ThinArc::from_header_and_iter((), new_attrs.into_iter()))
169        };
170        RawAttrs { entries }
171    }
172
173    pub fn is_empty(&self) -> bool {
174        self.entries.is_none()
175    }
176}

178#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
179pub struct AttrId {
180    id: u32,
181}
182
183// FIXME: This only handles a single level of cfg_attr nesting
184// that is `#[cfg_attr(all(), cfg_attr(all(), cfg(any())))]` breaks again
185impl AttrId {
186    const CFG_ATTR_BITS: usize = 7;
187    const AST_INDEX_MASK: usize = 0x00FF_FFFF;
188    const AST_INDEX_BITS: usize = Self::AST_INDEX_MASK.count_ones() as usize;
189    const CFG_ATTR_SET_BITS: u32 = 1 << 31;
190
191    pub fn ast_index(&self) -> usize {
192        self.id as usize & Self::AST_INDEX_MASK
193    }
194
195    pub fn cfg_attr_index(&self) -> Option<usize> {
196        if self.id & Self::CFG_ATTR_SET_BITS == 0 {
197            None
198        } else {
199            Some(self.id as usize >> Self::AST_INDEX_BITS)
200        }
201    }
202
203    pub fn with_cfg_attr(self, idx: usize) -> AttrId {
204        AttrId { id: self.id | ((idx as u32) << Self::AST_INDEX_BITS) | Self::CFG_ATTR_SET_BITS }
205    }
206}

208#[derive(Debug, Clone, PartialEq, Eq)]
209pub struct Attr {
210    pub id: AttrId,
211    pub path: Interned<ModPath>,
212    pub input: Option<Box<AttrInput>>,
213    pub ctxt: SyntaxContext,
214}
215
216#[derive(Debug, Clone, PartialEq, Eq, Hash)]
217pub enum AttrInput {
218    /// `#[attr = "string"]`
219    Literal(tt::Literal),
220    /// `#[attr(subtree)]`
221    TokenTree(tt::TopSubtree),
222}
223
224impl fmt::Display for AttrInput {
225    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
226        match self {
227            AttrInput::Literal(lit) => write!(f, " = {lit}"),
228            AttrInput::TokenTree(tt) => tt.fmt(f),
229        }
230    }
231}

233impl Attr {
234    fn from_src(
235        db: &dyn ExpandDatabase,
236        ast: ast::Meta,
237        span_map: SpanMapRef<'_>,
238        id: AttrId,
239    ) -> Option<Attr> {
240        let path = ast.path()?;
241        let range = path.syntax().text_range();
242        let path = Interned::new(ModPath::from_src(db, path, &mut |range| {
243            span_map.span_for_range(range).ctx
244        })?);
245        let span = span_map.span_for_range(range);
246        let input = if let Some(ast::Expr::Literal(lit)) = ast.expr() {
247            let token = lit.token();
248            Some(Box::new(AttrInput::Literal(token_to_literal(token.text(), span))))
249        } else if let Some(tt) = ast.token_tree() {
250            let tree = syntax_node_to_token_tree(
251                tt.syntax(),
252                span_map,
253                span,
254                DocCommentDesugarMode::ProcMacro,
255            );
256            Some(Box::new(AttrInput::TokenTree(tree)))
257        } else {
258            None
259        };
260        Some(Attr { id, path, input, ctxt: span.ctx })
261    }
262
263    fn from_tt(
264        db: &dyn ExpandDatabase,
265        mut tt: tt::TokenTreesView<'_>,
266        id: AttrId,
267    ) -> Option<Attr> {
268        if matches!(tt.flat_tokens(),
269            [tt::TokenTree::Leaf(tt::Leaf::Ident(tt::Ident { sym, .. })), ..]
270            if *sym == sym::unsafe_
271        ) {
272            match tt.iter().nth(1) {
273                Some(tt::TtElement::Subtree(_, iter)) => tt = iter.remaining(),
274                _ => return None,
275            }
276        }
277        let first = tt.flat_tokens().first()?;
278        let ctxt = first.first_span().ctx;
279        let (path, input) = {
280            let mut iter = tt.iter();
281            let start = iter.savepoint();
282            let mut input = tt::TokenTreesView::new(&[]);
283            let mut path = iter.from_savepoint(start);
284            let mut path_split_savepoint = iter.savepoint();
285            while let Some(tt) = iter.next() {
286                path = iter.from_savepoint(start);
287                if !matches!(
288                    tt,
289                    tt::TtElement::Leaf(
290                        tt::Leaf::Punct(tt::Punct { char: ':' | '$', .. }) | tt::Leaf::Ident(_),
291                    )
292                ) {
293                    input = path_split_savepoint.remaining();
294                    break;
295                }
296                path_split_savepoint = iter.savepoint();
297            }
298            (path, input)
299        };
300
301        let path = Interned::new(ModPath::from_tt(db, path)?);
302
303        let input = match (input.flat_tokens().first(), input.try_into_subtree()) {
304            (_, Some(tree)) => {
305                Some(Box::new(AttrInput::TokenTree(tt::TopSubtree::from_subtree(tree))))
306            }
307            (Some(tt::TokenTree::Leaf(tt::Leaf::Punct(tt::Punct { char: '=', .. }))), _) => {
308                match input.flat_tokens().get(1) {
309                    Some(tt::TokenTree::Leaf(tt::Leaf::Literal(lit))) => {
310                        Some(Box::new(AttrInput::Literal(lit.clone())))
311                    }
312                    _ => None,
313                }
314            }
315            _ => None,
316        };
317        Some(Attr { id, path, input, ctxt })
318    }
319
320    pub fn path(&self) -> &ModPath {
321        &self.path
322    }
323}

325impl Attr {
326    /// #[path = "string"]
327    pub fn string_value(&self) -> Option<&Symbol> {
328        match self.input.as_deref()? {
329            AttrInput::Literal(tt::Literal {
330                symbol: text,
331                kind: tt::LitKind::Str | tt::LitKind::StrRaw(_),
332                ..
333            }) => Some(text),
334            _ => None,
335        }
336    }
337
338    /// #[path = "string"]
339    pub fn string_value_with_span(&self) -> Option<(&Symbol, span::Span)> {
340        match self.input.as_deref()? {
341            AttrInput::Literal(tt::Literal {
342                symbol: text,
343                kind: tt::LitKind::Str | tt::LitKind::StrRaw(_),
344                span,
345                suffix: _,
346            }) => Some((text, *span)),
347            _ => None,
348        }
349    }
350
351    pub fn string_value_unescape(&self) -> Option<Cow<'_, str>> {
352        match self.input.as_deref()? {
353            AttrInput::Literal(tt::Literal {
354                symbol: text, kind: tt::LitKind::StrRaw(_), ..
355            }) => Some(Cow::Borrowed(text.as_str())),
356            AttrInput::Literal(tt::Literal { symbol: text, kind: tt::LitKind::Str, .. }) => {
357                unescape(text.as_str())
358            }
359            _ => None,
360        }
361    }
362
363    /// #[path(ident)]
364    pub fn single_ident_value(&self) -> Option<&tt::Ident> {
365        match self.input.as_deref()? {
366            AttrInput::TokenTree(tt) => match tt.token_trees().flat_tokens() {
367                [tt::TokenTree::Leaf(tt::Leaf::Ident(ident))] => Some(ident),
368                _ => None,
369            },
370            _ => None,
371        }
372    }
373
374    /// #[path TokenTree]
375    pub fn token_tree_value(&self) -> Option<&TopSubtree> {
376        match self.input.as_deref()? {
377            AttrInput::TokenTree(tt) => Some(tt),
378            _ => None,
379        }
380    }
381
382    /// Parses this attribute as a token tree consisting of comma separated paths.
383    pub fn parse_path_comma_token_tree<'a>(
384        &'a self,
385        db: &'a dyn ExpandDatabase,
386    ) -> Option<impl Iterator<Item = (ModPath, Span)> + 'a> {
387        let args = self.token_tree_value()?;
388
389        if args.top_subtree().delimiter.kind != DelimiterKind::Parenthesis {
390            return None;
391        }
392        let paths = args
393            .token_trees()
394            .split(|tt| matches!(tt, tt::TtElement::Leaf(tt::Leaf::Punct(Punct { char: ',', .. }))))
395            .filter_map(move |tts| {
396                let span = tts.flat_tokens().first()?.first_span();
397                Some((ModPath::from_tt(db, tts)?, span))
398            });
399
400        Some(paths)
401    }
402
403    pub fn cfg(&self) -> Option<CfgExpr> {
404        if *self.path.as_ident()? == sym::cfg.clone() {
405            self.token_tree_value().map(CfgExpr::parse)
406        } else {
407            None
408        }
409    }
410}

412fn unescape(s: &str) -> Option<Cow<'_, str>> {
413    let mut buf = String::new();
414    let mut prev_end = 0;
415    let mut has_error = false;
416    unescape::unescape_unicode(s, unescape::Mode::Str, &mut |char_range, unescaped_char| match (
417        unescaped_char,
418        buf.capacity() == 0,
419    ) {
420        (Ok(c), false) => buf.push(c),
421        (Ok(_), true) if char_range.len() == 1 && char_range.start == prev_end => {
422            prev_end = char_range.end
423        }
424        (Ok(c), true) => {
425            buf.reserve_exact(s.len());
426            buf.push_str(&s[..prev_end]);
427            buf.push(c);
428        }
429        (Err(_), _) => has_error = true,
430    });
431
432    match (has_error, buf.capacity() == 0) {
433        (true, _) => None,
434        (false, false) => Some(Cow::Owned(buf)),
435        (false, true) => Some(Cow::Borrowed(s)),
436    }
437}

439pub fn collect_attrs(
440    owner: &dyn ast::HasAttrs,
441) -> impl Iterator<Item = (AttrId, Either<ast::Attr, ast::Comment>)> {
442    let inner_attrs = inner_attributes(owner.syntax()).into_iter().flatten();
443    let outer_attrs =
444        ast::AttrDocCommentIter::from_syntax_node(owner.syntax()).filter(|el| match el {
445            Either::Left(attr) => attr.kind().is_outer(),
446            Either::Right(comment) => comment.is_outer(),
447        });
448    outer_attrs.chain(inner_attrs).enumerate().map(|(id, attr)| (AttrId { id: id as u32 }, attr))
449}
450
451fn inner_attributes(
452    syntax: &SyntaxNode,
453) -> Option<impl Iterator<Item = Either<ast::Attr, ast::Comment>>> {
454    let node = match_ast! {
455        match syntax {
456            ast::SourceFile(_) => syntax.clone(),
457            ast::ExternBlock(it) => it.extern_item_list()?.syntax().clone(),
458            ast::Fn(it) => it.body()?.stmt_list()?.syntax().clone(),
459            ast::Impl(it) => it.assoc_item_list()?.syntax().clone(),
460            ast::Module(it) => it.item_list()?.syntax().clone(),
461            ast::BlockExpr(it) => {
462                if !it.may_carry_attributes() {
463                    return None
464                }
465                syntax.clone()
466            },
467            _ => return None,
468        }
469    };
470
471    let attrs = ast::AttrDocCommentIter::from_syntax_node(&node).filter(|el| match el {
472        Either::Left(attr) => attr.kind().is_inner(),
473        Either::Right(comment) => comment.is_inner(),
474    });
475    Some(attrs)
476}

478// Input subtree is: `(cfg, $(attr),+)`
479// Split it up into a `cfg` subtree and the `attr` subtrees.
480fn parse_cfg_attr_input(
481    subtree: &TopSubtree,
482) -> Option<(tt::TokenTreesView<'_>, impl Iterator<Item = tt::TokenTreesView<'_>>)> {
483    let mut parts = subtree
484        .token_trees()
485        .split(|tt| matches!(tt, tt::TtElement::Leaf(tt::Leaf::Punct(Punct { char: ',', .. }))));
486    let cfg = parts.next()?;
487    Some((cfg, parts.filter(|it| !it.is_empty())))
