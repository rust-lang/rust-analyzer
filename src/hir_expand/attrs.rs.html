1//! Defines the basics of attributes lowering.
2//!
3//! The heart and soul of this module is [`expand_cfg_attr()`], alongside its sibling
4//! [`expand_cfg_attr_with_doc_comments()`]. It is used to implement all attribute lowering
5//! in r-a. Its basic job is to list attributes; however, attributes do not necessarily map
6//! into [`ast::Attr`], because `cfg_attr` can map to zero, one, or more attributes
7//! (`#[cfg_attr(predicate, attr1, attr2, ...)]`). To bridge this gap, this module defines
8//! [`Meta`], which represents a desugared attribute. Various bits of r-a need different
9//! things from [`Meta`], therefore it contains many parts. The basic idea is:
10//!
11//!  - There are three kinds of attributes, `path = value`, `path`, and `path(token_tree)`.
12//!  - Most bits of rust-analyzer only need to deal with some paths. Therefore, we keep
13//!    the path only if it has up to 2 segments, or one segment for `path = value`.
14//!    We also only keep the value in `path = value` if it is a literal. However, we always
15//!    save the all relevant ranges of attributes (the path range, and the full attribute range)
16//!    for parts of r-a (e.g. name resolution) that need a faithful representation of the
17//!    attribute.
18//!
19//! [`expand_cfg_attr()`] expands `cfg_attr`s as it goes (as its name implies), to list
20//! all attributes.
21//!
22//! Another thing to note is that we need to be able to map an attribute back to a range
23//! (for diagnostic purposes etc.). This is only ever needed for attributes that participate
24//! in name resolution. An attribute is mapped back by its [`AttrId`], which is just an
25//! index into the item tree attributes list. To minimize the risk of bugs, we have one
26//! place (here) and one function ([`is_item_tree_filtered_attr()`]) that decides whether
27//! an attribute participate in name resolution.

59#[derive(Debug)]
60pub struct AttrPath {
61    /// This can be empty if the path is not of 1 or 2 segments exactly.
62    pub segments: ArrayVec<SyntaxToken, 2>,
63    pub range: TextRange,
64    // FIXME: This shouldn't be textual, `#[test]` needs name resolution.
65    // And if textual, it shouldn't be here, it should be in hir-def/src/attrs.rs. But some macros
66    // fully qualify `test` as `core::prelude::vX::test`, and this is more than 2 segments, so hir-def
67    // attrs can't find it. But this will mean we have to push every up-to-4-segments path, which
68    // may impact perf. So it was easier to just hack it here.
69    pub is_test: bool,
70}
71
72impl AttrPath {
73    #[inline]
74    fn extract(path: &ast::Path) -> Self {
75        let mut is_test = false;
76        let segments = (|| {
77            let mut segments = ArrayVec::new();
78            let segment2 = path.segment()?.name_ref()?.syntax().first_token()?;
79            if segment2.text() == "test" {
80                // `#[test]` or `#[core::prelude::vX::test]`.
81                is_test = true;
82            }
83            let segment1 = path.qualifier();
84            if let Some(segment1) = segment1 {
85                if segment1.qualifier().is_some() {
86                    None
87                } else {
88                    let segment1 = segment1.segment()?.name_ref()?.syntax().first_token()?;
89                    segments.push(segment1);
90                    segments.push(segment2);
91                    Some(segments)
92                }
93            } else {
94                segments.push(segment2);
95                Some(segments)
96            }
97        })();
98        AttrPath {
99            segments: segments.unwrap_or(ArrayVec::new()),
100            range: path.syntax().text_range(),
101            is_test,
102        }
103    }
104
105    #[inline]
106    pub fn is1(&self, segment: &str) -> bool {
107        self.segments.len() == 1 && self.segments[0].text() == segment
108    }

111#[derive(Debug)]
112pub enum Meta {
113    /// `name` is `None` if not a single token. `value` is a literal or `None`.
114    NamedKeyValue {
115        path_range: TextRange,
116        name: Option<SyntaxToken>,
117        value: Option<SyntaxToken>,
118    },
119    TokenTree {
120        path: AttrPath,
121        tt: ast::TokenTree,
122    },
123    Path {
124        path: AttrPath,
125    },
126}
127
128impl Meta {
129    #[inline]
130    pub fn path_range(&self) -> TextRange {
131        match self {
132            Meta::NamedKeyValue { path_range, .. } => *path_range,
133            Meta::TokenTree { path, .. } | Meta::Path { path } => path.range,
134        }
135    }
136
137    fn extract(iter: &mut Peekable<TokenTreeChildren>) -> Option<(Self, TextSize)> {
138        let mut start_offset = None;
139        if let Some(NodeOrToken::Token(colon1)) = iter.peek()
140            && colon1.kind() == T![:]
141        {
142            start_offset = Some(colon1.text_range().start());
143            iter.next();
144            iter.next_if(|it| it.as_token().is_some_and(|it| it.kind() == T![:]));
145        }
146        let first_segment = iter
147            .next_if(|it| it.as_token().is_some_and(|it| it.kind().is_any_identifier()))?
148            .into_token()?;
149        let mut is_test = first_segment.text() == "test";
150        let start_offset = start_offset.unwrap_or_else(|| first_segment.text_range().start());
151
152        let mut segments_len = 1;
153        let mut second_segment = None;
154        let mut path_range = first_segment.text_range();
155        while iter.peek().and_then(NodeOrToken::as_token).is_some_and(|it| it.kind() == T![:])
156            && let _ = iter.next()
157            && iter.peek().and_then(NodeOrToken::as_token).is_some_and(|it| it.kind() == T![:])
158            && let _ = iter.next()
159            && let Some(NodeOrToken::Token(segment)) = iter.peek()
160            && segment.kind().is_any_identifier()
161        {
162            segments_len += 1;
163            is_test = segment.text() == "test";
164            second_segment = Some(segment.clone());
165            path_range = TextRange::new(path_range.start(), segment.text_range().end());
166            iter.next();
167        }
168
169        let segments = |first, second| {
170            let mut segments = ArrayVec::new();
171            if segments_len <= 2 {
172                segments.push(first);
173                if let Some(second) = second {
174                    segments.push(second);
175                }
176            }
177            segments
178        };
179        let meta = match iter.peek() {
180            Some(NodeOrToken::Token(eq)) if eq.kind() == T![=] => {
181                iter.next();
182                let value = match iter.peek() {
183                    Some(NodeOrToken::Token(token)) if token.kind().is_literal() => {
184                        // No need to consume it, it will be consumed by `extract_and_eat_comma()`.
185                        Some(token.clone())
186                    }
187                    _ => None,
188                };
189                let name = if second_segment.is_none() { Some(first_segment) } else { None };
190                Meta::NamedKeyValue { path_range, name, value }
191            }
192            Some(NodeOrToken::Node(tt)) => Meta::TokenTree {
193                path: AttrPath {
194                    segments: segments(first_segment, second_segment),
195                    range: path_range,
196                    is_test,
197                },
198                tt: tt.clone(),
199            },
200            _ => Meta::Path {
201                path: AttrPath {
202                    segments: segments(first_segment, second_segment),
203                    range: path_range,
204                    is_test,
205                },
206            },
207        };
208        Some((meta, start_offset))
209    }
210
211    fn extract_possibly_unsafe(
212        iter: &mut Peekable<TokenTreeChildren>,
213        container: &ast::TokenTree,
214    ) -> Option<(Self, TextRange)> {
215        if iter.peek().is_some_and(|it| it.as_token().is_some_and(|it| it.kind() == T![unsafe])) {
216            iter.next();
217            let tt = iter.next()?.into_node()?;
218            let result = Self::extract(&mut TokenTreeChildren::new(&tt).peekable()).map(
219                |(meta, start_offset)| (meta, TextRange::new(start_offset, tt_end_offset(&tt))),
220            );
221            while iter.next().is_some_and(|it| it.as_token().is_none_or(|it| it.kind() != T![,])) {}
222            result
223        } else {
224            Self::extract(iter).map(|(meta, start_offset)| {
225                let end_offset = 'find_end_offset: {
226                    for it in iter {
227                        if let NodeOrToken::Token(it) = it
228                            && it.kind() == T![,]
229                        {
230                            break 'find_end_offset it.text_range().start();
231                        }
232                    }
233                    tt_end_offset(container)
234                };
235                (meta, TextRange::new(start_offset, end_offset))
236            })
237        }
238    }
239}

245/// The callback is passed a desugared form of the attribute ([`Meta`]), a [`SyntaxNode`] fully containing it
246/// (note: it may not be the direct parent), the range within the [`SyntaxNode`] bounding the attribute,
247/// and the outermost `ast::Attr`. Note that one node may map to multiple [`Meta`]s due to `cfg_attr`.
248#[inline]
249pub fn expand_cfg_attr<'a, BreakValue>(
250    attrs: impl Iterator<Item = ast::Attr>,
251    cfg_options: impl FnMut() -> &'a CfgOptions,
252    mut callback: impl FnMut(Meta, &SyntaxNode, TextRange, &ast::Attr) -> ControlFlow<BreakValue>,
253) -> Option<BreakValue> {
254    expand_cfg_attr_with_doc_comments::<Infallible, _>(
255        attrs.map(Either::Left),
256        cfg_options,
257        move |Either::Left((meta, container, range, top_attr))| {
258            callback(meta, container, range, top_attr)
259        },
260    )
261}
262
263#[inline]
264pub fn expand_cfg_attr_with_doc_comments<'a, DocComment, BreakValue>(
265    mut attrs: impl Iterator<Item = Either<ast::Attr, DocComment>>,
266    mut cfg_options: impl FnMut() -> &'a CfgOptions,
267    mut callback: impl FnMut(
268        Either<(Meta, &SyntaxNode, TextRange, &ast::Attr), DocComment>,
269    ) -> ControlFlow<BreakValue>,
270) -> Option<BreakValue> {
271    let mut stack = SmallVec::<[_; 1]>::new();
272    let result = attrs.try_for_each(|top_attr| {
273        let top_attr = match top_attr {
274            Either::Left(it) => it,
275            Either::Right(comment) => return callback(Either::Right(comment)),
276        };
277        if let Some((attr_name, tt)) = top_attr.as_simple_call()
278            && attr_name == "cfg_attr"
279        {
280            let mut tt_iter = TokenTreeChildren::new(&tt).peekable();
281            let cfg = cfg::CfgExpr::parse_from_ast(&mut tt_iter);
282            if cfg_options().check(&cfg) != Some(false) {
283                stack.push((tt_iter, tt));
284                while let Some((tt_iter, tt)) = stack.last_mut() {
285                    let Some((attr, range)) = Meta::extract_possibly_unsafe(tt_iter, tt) else {
286                        stack.pop();
287                        continue;
288                    };
289                    if let Meta::TokenTree { path, tt: nested_tt } = &attr
290                        && path.is1("cfg_attr")
291                    {
292                        let mut nested_tt_iter = TokenTreeChildren::new(nested_tt).peekable();
293                        let cfg = cfg::CfgExpr::parse_from_ast(&mut nested_tt_iter);
294                        if cfg_options().check(&cfg) != Some(false) {
295                            stack.push((nested_tt_iter, nested_tt.clone()));
296                        }
297                    } else {
298                        callback(Either::Left((attr, tt.syntax(), range, &top_attr)))?;
299                    }
300                }
301            }
302        } else if let Some(ast_meta) = top_attr.meta()
303            && let Some(path) = ast_meta.path()
304        {
305            let path = AttrPath::extract(&path);
306            let meta = if let Some(tt) = ast_meta.token_tree() {
307                Meta::TokenTree { path, tt }
308            } else if let Some(value) = ast_meta.expr() {
309                let value =
310                    if let ast::Expr::Literal(value) = value { Some(value.token()) } else { None };
311                let name =
312                    if path.segments.len() == 1 { Some(path.segments[0].clone()) } else { None };
313                Meta::NamedKeyValue { name, value, path_range: path.range }
314            } else {
315                Meta::Path { path }
316            };
317            callback(Either::Left((
318                meta,
319                ast_meta.syntax(),
320                ast_meta.syntax().text_range(),
321                &top_attr,
322            )))?;
323        }
324        ControlFlow::Continue(())
325    });
326    result.break_value()
327}

329#[inline]
330pub(crate) fn is_item_tree_filtered_attr(name: &str) -> bool {
331    matches!(
332        name,
333        "doc"
334            | "stable"
335            | "unstable"
336            | "target_feature"
337            | "allow"
338            | "expect"
339            | "warn"
340            | "deny"
341            | "forbid"
342            | "repr"
343            | "inline"
344            | "track_caller"
345            | "must_use"
346    )
347}

349/// This collects attributes exactly as the item tree needs them. This is used for the item tree,
350/// as well as for resolving [`AttrId`]s.
351pub fn collect_item_tree_attrs<'a, BreakValue>(
352    owner: &dyn ast::HasAttrs,
353    cfg_options: impl Fn() -> &'a CfgOptions,
354    mut on_attr: impl FnMut(Meta, &SyntaxNode, &ast::Attr, TextRange) -> ControlFlow<BreakValue>,
355) -> Option<Either<BreakValue, CfgExpr>> {
356    let attrs = ast::attrs_including_inner(owner);
357    expand_cfg_attr(
358        attrs,
359        || cfg_options(),
360        |attr, container, range, top_attr| {
361            // We filter builtin attributes that we don't need for nameres, because this saves memory.
362            // I only put the most common attributes, but if some attribute becomes common feel free to add it.
363            // Notice, however: for an attribute to be filtered out, it *must* not be shadowable with a macro!
364            let filter = match &attr {
365                Meta::NamedKeyValue { name: Some(name), .. } => {
366                    is_item_tree_filtered_attr(name.text())
367                }
368                Meta::TokenTree { path, tt } if path.segments.len() == 1 => {
369                    let name = path.segments[0].text();
370                    if name == "cfg" {
371                        let cfg =
372                            CfgExpr::parse_from_ast(&mut TokenTreeChildren::new(tt).peekable());
373                        if cfg_options().check(&cfg) == Some(false) {
374                            return ControlFlow::Break(Either::Right(cfg));
375                        }
376                        true
377                    } else {
378                        is_item_tree_filtered_attr(name)
379                    }
380                }
381                Meta::Path { path } => {
382                    path.segments.len() == 1 && is_item_tree_filtered_attr(path.segments[0].text())
383                }
384                _ => false,
385            };
386            if !filter && let ControlFlow::Break(v) = on_attr(attr, container, top_attr, range) {
387                return ControlFlow::Break(Either::Left(v));
388            }
389            ControlFlow::Continue(())
390        },
391    )
392}

470    /// Parses this attribute as a token tree consisting of comma separated paths.
471    pub fn parse_path_comma_token_tree<'a>(
472        &'a self,
473        db: &'a dyn ExpandDatabase,
474    ) -> Option<impl Iterator<Item = (ModPath, Span, tt::TokenTreesView<'a>)> + 'a> {
475        let args = self.token_tree_value()?;
476
477        if args.top_subtree().delimiter.kind != DelimiterKind::Parenthesis {
478            return None;
479        }
480        Some(parse_path_comma_token_tree(db, args))
481    }

483
484fn parse_path_comma_token_tree<'a>(
485    db: &'a dyn ExpandDatabase,
486    args: &'a tt::TopSubtree,
487) -> impl Iterator<Item = (ModPath, Span, tt::TokenTreesView<'a>)> {
488    args.token_trees()
489        .split(|tt| matches!(tt, tt::TtElement::Leaf(tt::Leaf::Punct(Punct { char: ',', .. }))))
490        .filter_map(move |tts| {
491            let span = tts.first_span()?;
492            Some((ModPath::from_tt(db, tts)?, span, tts))
493        })
494}

496fn unescape(s: &str) -> Option<Cow<'_, str>> {
497    let mut buf = String::new();
498    let mut prev_end = 0;
499    let mut has_error = false;
500    unescape::unescape_str(s, |char_range, unescaped_char| {
501        match (unescaped_char, buf.capacity() == 0) {
502            (Ok(c), false) => buf.push(c),
503            (Ok(_), true) if char_range.len() == 1 && char_range.start == prev_end => {
504                prev_end = char_range.end
505            }
506            (Ok(c), true) => {
507                buf.reserve_exact(s.len());
508                buf.push_str(&s[..prev_end]);
509                buf.push(c);
510            }
511            (Err(_), _) => has_error = true,
512        }
513    });
514
515    match (has_error, buf.capacity() == 0) {
516        (true, _) => None,
517        (false, false) => Some(Cow::Owned(buf)),
518        (false, true) => Some(Cow::Borrowed(s)),
519    }
520}

522/// This is an index of an attribute *that always points to the item tree attributes*.
523///
524/// Outer attributes are counted first, then inner attributes. This does not support
525/// out-of-line modules, which may have attributes spread across 2 files!
526#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
527pub struct AttrId {
528    id: u32,
529}

531impl AttrId {
532    #[inline]
533    pub fn from_item_tree_index(id: u32) -> Self {
534        Self { id }
535    }
536
537    #[inline]
538    pub fn item_tree_index(self) -> u32 {
539        self.id
540    }

542    /// Returns the containing `ast::Attr` (note that it may contain other attributes as well due
543    /// to `cfg_attr`), a `SyntaxNode` guaranteed to contain the attribute, the full range of the
544    /// attribute, and its desugared [`Meta`].
545    pub fn find_attr_range<N: ast::HasAttrs>(
546        self,
547        db: &dyn ExpandDatabase,
548        krate: Crate,
549        owner: AstId<N>,
550    ) -> (ast::Attr, SyntaxNode, TextRange, Meta) {
551        self.find_attr_range_with_source(db, krate, &owner.to_node(db))
552    }
553
554    /// Returns the containing `ast::Attr` (note that it may contain other attributes as well due
555    /// to `cfg_attr`), a `SyntaxNode` guaranteed to contain the attribute, the full range of the
556    /// attribute, and its desugared [`Meta`].
557    pub fn find_attr_range_with_source(
558        self,
559        db: &dyn ExpandDatabase,
560        krate: Crate,
561        owner: &dyn ast::HasAttrs,
562    ) -> (ast::Attr, SyntaxNode, TextRange, Meta) {
563        let cfg_options = OnceCell::new();
564        let mut index = 0;
565        let result = collect_item_tree_attrs(
566            owner,
567            || cfg_options.get_or_init(|| krate.cfg_options(db)),
568            |meta, container, top_attr, range| {
569                if index == self.id {
570                    return ControlFlow::Break((top_attr.clone(), container.clone(), range, meta));
571                }
572                index += 1;
573                ControlFlow::Continue(())
574            },
575        );
576        match result {
577            Some(Either::Left(it)) => it,
578            _ => {
579                panic!("used an incorrect `AttrId`; crate={krate:?}, attr_id={self:?}");
580            }
581        }
582    }
583
584    pub fn find_derive_range(
585        self,
586        db: &dyn ExpandDatabase,
587        krate: Crate,
588        owner: AstId<ast::Adt>,
589        derive_index: u32,
590    ) -> TextRange {
591        let (_, _, derive_attr_range, derive_attr) = self.find_attr_range(db, krate, owner);
592        let Meta::TokenTree { tt, .. } = derive_attr else {
593            return derive_attr_range;
594        };
595        // Fake the span map, as we don't really need spans here, just the offsets of the node in the file.
596        let span_map = RealSpanMap::absolute(span::EditionedFileId::current_edition(
597            span::FileId::from_raw(0),
598        ));
599        let tt = syntax_bridge::syntax_node_to_token_tree(
600            tt.syntax(),
601            SpanMapRef::RealSpanMap(&span_map),
602            span_map.span_for_range(tt.syntax().text_range()),
603            DocCommentDesugarMode::ProcMacro,
604        );
605        let Some((_, _, derive_tts)) =
606            parse_path_comma_token_tree(db, &tt).nth(derive_index as usize)
607        else {
608            return derive_attr_range;
609        };
610        let (Some(first_span), Some(last_span)) = (derive_tts.first_span(), derive_tts.last_span())
611        else {
612            return derive_attr_range;
613        };
614        let start = first_span.range.start();
615        let end = last_span.range.end();
616        TextRange::new(start, end)
617    }
