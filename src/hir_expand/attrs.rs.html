1//! Defines the basics of attributes lowering.
2//!
3//! The heart and soul of this module is [`expand_cfg_attr()`], alongside its sibling
4//! [`expand_cfg_attr_with_doc_comments()`]. It is used to implement all attribute lowering
5//! in r-a. Its basic job is to list attributes; however, attributes do not necessarily map
6//! into [`ast::Attr`], because `cfg_attr` can map to zero, one, or more attributes
7//! (`#[cfg_attr(predicate, attr1, attr2, ...)]`). [`expand_cfg_attr()`] expands `cfg_attr`s
8//! as it goes (as its name implies), to list all attributes.
9//!
10//! Another thing to note is that we need to be able to map an attribute back to a range
11//! (for diagnostic purposes etc.). This is only ever needed for attributes that participate
12//! in name resolution. An attribute is mapped back by its [`AttrId`], which is just an
13//! index into the item tree attributes list. To minimize the risk of bugs, we have one
14//! place (here) and one function ([`is_item_tree_filtered_attr()`]) that decides whether
15//! an attribute participate in name resolution.
16

39pub trait AstPathExt {
40    fn is1(&self, segment: &str) -> bool;
41
42    fn as_one_segment(&self) -> Option<SmolStr>;
43
44    fn as_up_to_two_segment(&self) -> Option<(SmolStr, Option<SmolStr>)>;
45}
46
47impl AstPathExt for ast::Path {
48    fn is1(&self, segment: &str) -> bool {
49        self.as_one_segment().is_some_and(|it| it == segment)
50    }
51
52    fn as_one_segment(&self) -> Option<SmolStr> {
53        Some(self.as_single_name_ref()?.text().into())
54    }
55
56    fn as_up_to_two_segment(&self) -> Option<(SmolStr, Option<SmolStr>)> {
57        let parent = self.qualifier().as_one_segment();
58        let this = self.segment()?.name_ref()?.text().into();
59        if let Some(parent) = parent { Some((parent, Some(this))) } else { Some((this, None)) }
60    }
61}
62
63impl AstPathExt for Option<ast::Path> {
64    fn is1(&self, segment: &str) -> bool {
65        self.as_ref().is_some_and(|it| it.is1(segment))
66    }
67
68    fn as_one_segment(&self) -> Option<SmolStr> {
69        self.as_ref().and_then(|it| it.as_one_segment())
70    }
71
72    fn as_up_to_two_segment(&self) -> Option<(SmolStr, Option<SmolStr>)> {
73        self.as_ref().and_then(|it| it.as_up_to_two_segment())
74    }
75}

94/// The callback is passed the attribute and the outermost `ast::Attr`.
95/// Note that one node may map to multiple [`ast::Meta`]s due to `cfg_attr`.
96///
97/// `unsafe(attr)` are passed the inner attribute for now.
98#[inline]
99pub fn expand_cfg_attr<'a, BreakValue>(
100    attrs: impl Iterator<Item = ast::Attr>,
101    cfg_options: impl FnMut() -> &'a CfgOptions,
102    mut callback: impl FnMut(ast::Meta, ast::Attr) -> ControlFlow<BreakValue>,
103) -> Option<BreakValue> {
104    expand_cfg_attr_with_doc_comments::<Infallible, _>(
105        attrs.map(Either::Left),
106        cfg_options,
107        move |Either::Left((meta, top_attr))| callback(meta, top_attr),
108    )
109}
110
111#[inline]
112pub fn expand_cfg_attr_with_doc_comments<'a, DocComment, BreakValue>(
113    mut attrs: impl Iterator<Item = Either<ast::Attr, DocComment>>,
114    mut cfg_options: impl FnMut() -> &'a CfgOptions,
115    mut callback: impl FnMut(Either<(ast::Meta, ast::Attr), DocComment>) -> ControlFlow<BreakValue>,
116) -> Option<BreakValue> {
117    let mut stack = SmallVec::<[_; 1]>::new();
118    loop {
119        let (mut meta, top_attr) = if let Some(it) = stack.pop() {
120            it
121        } else {
122            let attr = attrs.next()?;
123            match attr {
124                Either::Left(attr) => {
125                    let Some(meta) = attr.meta() else { continue };
126                    stack.push((meta, attr));
127                }
128                Either::Right(doc_comment) => {
129                    if let ControlFlow::Break(break_value) = callback(Either::Right(doc_comment)) {
130                        return Some(break_value);
131                    }
132                }
133            }
134            continue;
135        };
136
137        while let ast::Meta::UnsafeMeta(unsafe_meta) = &meta {
138            let Some(inner) = unsafe_meta.meta() else { continue };
139            meta = inner;
140        }
141
142        if let ast::Meta::CfgAttrMeta(meta) = meta {
143            let Some(cfg_predicate) = meta.cfg_predicate() else { continue };
144            let cfg_predicate = CfgExpr::parse_from_ast(cfg_predicate);
145            if cfg_options().check(&cfg_predicate) != Some(false) {
146                let prev_stack_len = stack.len();
147                stack.extend(meta.metas().map(|meta| (meta, top_attr.clone())));
148                stack[prev_stack_len..].reverse();
149            }
150        } else {
151            if let ControlFlow::Break(break_value) = callback(Either::Left((meta, top_attr))) {
152                return Some(break_value);
153            }
154        }
155    }
156}

158#[inline]
159pub(crate) fn is_item_tree_filtered_attr(name: &str) -> bool {
160    matches!(
161        name,
162        "doc"
163            | "stable"
164            | "unstable"
165            | "target_feature"
166            | "allow"
167            | "expect"
168            | "warn"
169            | "deny"
170            | "forbid"
171            | "repr"
172            | "inline"
173            | "track_caller"
174            | "must_use"
175    )
176}

178/// This collects attributes exactly as the item tree needs them. This is used for the item tree,
179/// as well as for resolving [`AttrId`]s.
180pub fn collect_item_tree_attrs<'a, BreakValue>(
181    owner: &dyn ast::HasAttrs,
182    cfg_options: impl Fn() -> &'a CfgOptions,
183    mut on_attr: impl FnMut(ast::Meta, ast::Attr) -> ControlFlow<BreakValue>,
184) -> Option<Either<BreakValue, CfgExpr>> {
185    let attrs = ast::attrs_including_inner(owner);
186    expand_cfg_attr(
187        attrs,
188        || cfg_options(),
189        |attr, top_attr| {
190            // We filter builtin attributes that we don't need for nameres, because this saves memory.
191            // I only put the most common attributes, but if some attribute becomes common feel free to add it.
192            // Notice, however: for an attribute to be filtered out, it *must* not be shadowable with a macro!
193            let filter = match &attr {
194                ast::Meta::CfgMeta(attr) => {
195                    let Some(cfg_predicate) = attr.cfg_predicate() else {
196                        return ControlFlow::Continue(());
197                    };
198                    let cfg = CfgExpr::parse_from_ast(cfg_predicate);
199                    if cfg_options().check(&cfg) == Some(false) {
200                        return ControlFlow::Break(Either::Right(cfg));
201                    }
202                    true
203                }
204                _ => attr
205                    .path()
206                    .and_then(|path| path.as_one_segment())
207                    .is_some_and(|segment| is_item_tree_filtered_attr(&segment)),
208            };
209            if !filter && let ControlFlow::Break(v) = on_attr(attr, top_attr) {
210                return ControlFlow::Break(Either::Left(v));
211            }
212            ControlFlow::Continue(())
213        },
214    )
215}

220    pub input: Option<Box<AttrInput>>,
221    pub ctxt: SyntaxContext,

224#[derive(Debug, Clone, PartialEq, Eq, Hash)]
225pub enum AttrInput {
226    /// `#[attr = "string"]`
227    Literal(tt::Literal),
228    /// `#[attr(subtree)]`
229    TokenTree(tt::TopSubtree),
230}
231
232impl fmt::Display for AttrInput {
233    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
234        match self {
235            AttrInput::Literal(lit) => write!(f, " = {lit}"),
236            AttrInput::TokenTree(tt) => tt.fmt(f),
237        }
238    }
239}

241impl Attr {
242    /// #[path = "string"]
243    pub fn string_value(&self) -> Option<&str> {
244        match self.input.as_deref()? {
245            AttrInput::Literal(
246                lit @ tt::Literal { kind: tt::LitKind::Str | tt::LitKind::StrRaw(_), .. },
247            ) => Some(lit.text()),
248            _ => None,
249        }
250    }
251
252    /// #[path = "string"]
253    pub fn string_value_with_span(&self) -> Option<(&str, span::Span)> {
254        match self.input.as_deref()? {
255            AttrInput::Literal(
256                lit @ tt::Literal { kind: tt::LitKind::Str | tt::LitKind::StrRaw(_), span, .. },
257            ) => Some((lit.text(), *span)),
258            _ => None,
259        }
260    }
261
262    pub fn string_value_unescape(&self) -> Option<Cow<'_, str>> {
263        match self.input.as_deref()? {
264            AttrInput::Literal(lit @ tt::Literal { kind: tt::LitKind::StrRaw(_), .. }) => {
265                Some(Cow::Borrowed(lit.text()))
266            }
267            AttrInput::Literal(lit @ tt::Literal { kind: tt::LitKind::Str, .. }) => {
268                unescape(lit.text())
269            }
270            _ => None,
271        }
272    }
273
274    /// #[path(ident)]
275    pub fn single_ident_value(&self) -> Option<tt::Ident> {
276        match self.input.as_deref()? {
277            AttrInput::TokenTree(tt) => match tt.token_trees().iter().collect_array() {
278                Some([tt::TtElement::Leaf(tt::Leaf::Ident(ident))]) => Some(ident),
279                _ => None,
280            },
281            _ => None,
282        }
283    }
284
285    /// #[path TokenTree]
286    pub fn token_tree_value(&self) -> Option<&TopSubtree> {
287        match self.input.as_deref()? {
288            AttrInput::TokenTree(tt) => Some(tt),
289            _ => None,
290        }
291    }

293    /// Parses this attribute as a token tree consisting of comma separated paths.
294    pub fn parse_path_comma_token_tree<'a>(
295        &'a self,
296        db: &'a dyn ExpandDatabase,
297    ) -> Option<impl Iterator<Item = (ModPath, Span, tt::TokenTreesView<'a>)> + 'a> {
298        let args = self.token_tree_value()?;
299
300        if args.top_subtree().delimiter.kind != DelimiterKind::Parenthesis {
301            return None;
302        }
303        Some(parse_path_comma_token_tree(db, args))
304    }
305}
306
307fn parse_path_comma_token_tree<'a>(
308    db: &'a dyn ExpandDatabase,
309    args: &'a tt::TopSubtree,
310) -> impl Iterator<Item = (ModPath, Span, tt::TokenTreesView<'a>)> {
311    args.token_trees()
312        .split(|tt| matches!(tt, tt::TtElement::Leaf(tt::Leaf::Punct(Punct { char: ',', .. }))))
313        .filter_map(move |tts| {
314            let span = tts.first_span()?;
315            Some((ModPath::from_tt(db, tts)?, span, tts))
316        })
317}

319fn unescape(s: &str) -> Option<Cow<'_, str>> {
320    let mut buf = String::new();
321    let mut prev_end = 0;
322    let mut has_error = false;
323    unescape::unescape_str(s, |char_range, unescaped_char| {
324        match (unescaped_char, buf.capacity() == 0) {
325            (Ok(c), false) => buf.push(c),
326            (Ok(_), true) if char_range.len() == 1 && char_range.start == prev_end => {
327                prev_end = char_range.end
328            }
329            (Ok(c), true) => {
330                buf.reserve_exact(s.len());
331                buf.push_str(&s[..prev_end]);
332                buf.push(c);
333            }
334            (Err(_), _) => has_error = true,
335        }
336    });
337
338    match (has_error, buf.capacity() == 0) {
339        (true, _) => None,
340        (false, false) => Some(Cow::Owned(buf)),
341        (false, true) => Some(Cow::Borrowed(s)),
342    }
343}

345/// This is an index of an attribute *that always points to the item tree attributes*.
346///
347/// Outer attributes are counted first, then inner attributes. This does not support
348/// out-of-line modules, which may have attributes spread across 2 files!
349#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
350pub struct AttrId {
351    id: u32,
352}
353
354impl AttrId {
355    #[inline]
356    pub fn from_item_tree_index(id: u32) -> Self {
357        Self { id }
358    }
359
360    #[inline]
361    pub fn item_tree_index(self) -> u32 {
362        self.id
363    }

365    /// Returns the containing `ast::Attr` (note that it may contain other attributes as well due
366    /// to `cfg_attr`) and its [`ast::Meta`].
367    pub fn find_attr_range<N: ast::HasAttrs>(
368        self,
369        db: &dyn ExpandDatabase,
370        krate: Crate,
371        owner: AstId<N>,
372    ) -> (ast::Attr, ast::Meta) {
373        self.find_attr_range_with_source(db, krate, &owner.to_node(db))
374    }
375
376    /// Returns the containing `ast::Attr` (note that it may contain other attributes as well due
377    /// to `cfg_attr`) and its [`ast::Meta`].
378    pub fn find_attr_range_with_source(
379        self,
380        db: &dyn ExpandDatabase,
381        krate: Crate,
382        owner: &dyn ast::HasAttrs,
383    ) -> (ast::Attr, ast::Meta) {
384        let cfg_options = OnceCell::new();
385        let mut index = 0;
386        let result = collect_item_tree_attrs(
387            owner,
388            || cfg_options.get_or_init(|| krate.cfg_options(db)),
389            |meta, top_attr| {
390                if index == self.id {
391                    return ControlFlow::Break((top_attr, meta));
392                }
393                index += 1;
394                ControlFlow::Continue(())
395            },
396        );
397        match result {
398            Some(Either::Left(it)) => it,
399            _ => {
400                panic!("used an incorrect `AttrId`; crate={krate:?}, attr_id={self:?}");
401            }
402        }
403    }

405    pub fn find_derive_range(
406        self,
407        db: &dyn ExpandDatabase,
408        krate: Crate,
409        owner: AstId<ast::Adt>,
410        derive_index: u32,
411    ) -> TextRange {
412        let (_, derive_attr) = self.find_attr_range(db, krate, owner);
413        let ast::Meta::TokenTreeMeta(derive_attr) = derive_attr else {
414            return derive_attr.syntax().text_range();
415        };
416        let Some(tt) = derive_attr.token_tree() else {
417            return derive_attr.syntax().text_range();
418        };
419        // Fake the span map, as we don't really need spans here, just the offsets of the node in the file.
420        let span_map = RealSpanMap::absolute(span::EditionedFileId::current_edition(
421            span::FileId::from_raw(0),
422        ));
423        let tt = syntax_bridge::syntax_node_to_token_tree(
424            tt.syntax(),
425            SpanMap::RealSpanMap(&span_map),
426            span_map.span_for_range(tt.syntax().text_range()),
427            DocCommentDesugarMode::ProcMacro,
428        );
429        let Some((_, _, derive_tts)) =
430            parse_path_comma_token_tree(db, &tt).nth(derive_index as usize)
431        else {
432            return derive_attr.syntax().text_range();
433        };
434        let (Some(first_span), Some(last_span)) = (derive_tts.first_span(), derive_tts.last_span())
435        else {
436            return derive_attr.syntax().text_range();
437        };
438        let start = first_span.range.start();
439        let end = last_span.range.end();
440        TextRange::new(start, end)
441    }
