1//! Suggests shortening `Foo { field: field }` to `Foo { field }` in both
2//! expressions and patterns.
3
4use ide_db::text_edit::TextEdit;
5use ide_db::{EditionedFileId, FileRange, source_change::SourceChange};
6use syntax::{AstNode, SyntaxNode, ast, match_ast};
7
8use crate::{Diagnostic, DiagnosticCode, fix};
9
10pub(crate) fn field_shorthand(
11    acc: &mut Vec<Diagnostic>,
12    file_id: EditionedFileId,
13    node: &SyntaxNode,
14) {
15    match_ast! {
16        match node {
17            ast::RecordExpr(it) => check_expr_field_shorthand(acc, file_id, it),
18            ast::RecordPat(it) => check_pat_field_shorthand(acc, file_id, it),
19            _ => ()
20        }
21    };
22}
23
24fn check_expr_field_shorthand(
25    acc: &mut Vec<Diagnostic>,
26    file_id: EditionedFileId,
27    record_expr: ast::RecordExpr,
28) {
29    let record_field_list = match record_expr.record_expr_field_list() {
30        Some(it) => it,
31        None => return,
32    };
33    for record_field in record_field_list.fields() {
34        let (name_ref, expr) = match record_field.name_ref().zip(record_field.expr()) {
35            Some(it) => it,
36            None => continue,
37        };
38
39        let field_name = name_ref.syntax().text().to_string();
40        let field_expr = expr.syntax().text().to_string();
41        let field_name_is_tup_index = name_ref.as_tuple_field().is_some();
42        if field_name != field_expr || field_name_is_tup_index {
43            continue;
44        }
45
46        let mut edit_builder = TextEdit::builder();
47        edit_builder.delete(record_field.syntax().text_range());
48        edit_builder.insert(record_field.syntax().text_range().start(), field_name);
49        let edit = edit_builder.finish();
50
51        let field_range = record_field.syntax().text_range();
52        acc.push(
53            Diagnostic::new(
54                DiagnosticCode::Clippy("redundant_field_names"),
55                "Shorthand struct initialization",
56                FileRange { file_id: file_id.into(), range: field_range },
57            )
58            .with_fixes(Some(vec![fix(
59                "use_expr_field_shorthand",
60                "Use struct shorthand initialization",
61                SourceChange::from_text_edit(file_id, edit),
62                field_range,
63            )])),
64        );
65    }
66}
67
68fn check_pat_field_shorthand(
69    acc: &mut Vec<Diagnostic>,
70    file_id: EditionedFileId,
71    record_pat: ast::RecordPat,
72) {
73    let record_pat_field_list = match record_pat.record_pat_field_list() {
74        Some(it) => it,
75        None => return,
76    };
77    for record_pat_field in record_pat_field_list.fields() {
78        let (name_ref, pat) = match record_pat_field.name_ref().zip(record_pat_field.pat()) {
79            Some(it) => it,
80            None => continue,
81        };
82
83        let field_name = name_ref.syntax().text().to_string();
84        let field_pat = pat.syntax().text().to_string();
85        let field_name_is_tup_index = name_ref.as_tuple_field().is_some();
86        if field_name != field_pat || field_name_is_tup_index {
87            continue;
88        }
89
90        let mut edit_builder = TextEdit::builder();
91        edit_builder.delete(record_pat_field.syntax().text_range());
92        edit_builder.insert(record_pat_field.syntax().text_range().start(), field_name);
93        let edit = edit_builder.finish();
94
95        let field_range = record_pat_field.syntax().text_range();
96        acc.push(
97            Diagnostic::new(
98                DiagnosticCode::Clippy("redundant_field_names"),
99                "Shorthand struct pattern",
100                FileRange { file_id: file_id.into(), range: field_range },
101            )
102            .with_fixes(Some(vec![fix(
103                "use_pat_field_shorthand",
104                "Use struct field shorthand",
105                SourceChange::from_text_edit(file_id, edit),
106                field_range,
107            )])),
108        );
109    }
110}

112#[cfg(test)]
113mod tests {
114    use crate::tests::{check_diagnostics, check_fix};
115
116    #[test]
117    fn test_check_expr_field_shorthand() {
118        check_diagnostics(
119            r#"
120struct A { a: &'static str }
121fn main() { A { a: "hello" }; }
122"#,
123        );
124        check_diagnostics(
125            r#"
126struct A(usize);
127fn main() { A { 0: 0 }; }
128"#,
129        );
130
131        check_fix(
132            r#"
133struct A { a: &'static str }
134fn main() {
135    let a = "haha";
136    A { a$0: a };
137}
138"#,
139            r#"
140struct A { a: &'static str }
141fn main() {
142    let a = "haha";
143    A { a };
144}
145"#,
146        );
147
148        check_fix(
149            r#"
150struct A { a: &'static str, b: &'static str }
151fn main() {
152    let a = "haha";
153    let b = "bb";
154    A { a$0: a, b };
155}
156"#,
157            r#"
158struct A { a: &'static str, b: &'static str }
159fn main() {
160    let a = "haha";
161    let b = "bb";
162    A { a, b };
163}
164"#,
165        );
166    }
