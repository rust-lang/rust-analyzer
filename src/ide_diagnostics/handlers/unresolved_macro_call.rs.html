3// Diagnostic: unresolved-macro-call
4//
5// This diagnostic is triggered if rust-analyzer is unable to resolve the path
6// to a macro in a macro invocation.
7pub(crate) fn unresolved_macro_call(
8    ctx: &DiagnosticsContext<'_>,
9    d: &hir::UnresolvedMacroCall,
10) -> Diagnostic {
11    // Use more accurate position if available.
12    let display_range = ctx.resolve_precise_location(&d.macro_call, d.precise_location);
13    let bang = if d.is_bang { "!" } else { "" };
14    Diagnostic::new(
15        DiagnosticCode::RustcHardError("unresolved-macro-call"),
16        format!("unresolved macro `{}{bang}`", d.path.display(ctx.sema.db, ctx.edition)),
17        display_range,
18    )
19}
