19// Diagnostic: type-mismatch
20//
21// This diagnostic is triggered when the type of an expression or pattern does not match
22// the expected type.
23pub(crate) fn type_mismatch(ctx: &DiagnosticsContext<'_>, d: &hir::TypeMismatch<'_>) -> Diagnostic {
24    let display_range = adjusted_display_range(ctx, d.expr_or_pat, &|node| {
25        let Either::Left(expr) = node else { return None };
26        let salient_token_range = match expr {
27            ast::Expr::IfExpr(it) => it.if_token()?.text_range(),
28            ast::Expr::LoopExpr(it) => it.loop_token()?.text_range(),
29            ast::Expr::ForExpr(it) => it.for_token()?.text_range(),
30            ast::Expr::WhileExpr(it) => it.while_token()?.text_range(),
31            ast::Expr::BlockExpr(it) => it.stmt_list()?.r_curly_token()?.text_range(),
32            ast::Expr::MatchExpr(it) => it.match_token()?.text_range(),
33            ast::Expr::MethodCallExpr(it) => it.name_ref()?.ident_token()?.text_range(),
34            ast::Expr::FieldExpr(it) => it.name_ref()?.ident_token()?.text_range(),
35            ast::Expr::AwaitExpr(it) => it.await_token()?.text_range(),
36            _ => return None,
37        };
38
39        cov_mark::hit!(type_mismatch_range_adjustment);
40        Some(salient_token_range)
41    });
42    Diagnostic::new(
43        DiagnosticCode::RustcHardError("E0308"),
44        format!(
45            "expected {}, found {}",
46            d.expected
47                .display(ctx.sema.db, ctx.display_target)
48                .with_closure_style(ClosureStyle::ClosureWithId),
49            d.actual
50                .display(ctx.sema.db, ctx.display_target)
51                .with_closure_style(ClosureStyle::ClosureWithId),
52        ),
53        display_range,
54    )
55    .stable()
56    .with_fixes(fixes(ctx, d))
57}

59fn fixes(ctx: &DiagnosticsContext<'_>, d: &hir::TypeMismatch<'_>) -> Option<Vec<Assist>> {
60    let mut fixes = Vec::new();
61
62    if let Some(expr_ptr) = d.expr_or_pat.value.cast::<ast::Expr>() {
63        let expr_ptr = &InFile { file_id: d.expr_or_pat.file_id, value: expr_ptr };
64        add_reference(ctx, d, expr_ptr, &mut fixes);
65        add_missing_ok_or_some(ctx, d, expr_ptr, &mut fixes);
66        remove_unnecessary_wrapper(ctx, d, expr_ptr, &mut fixes);
67        remove_semicolon(ctx, d, expr_ptr, &mut fixes);
68        str_ref_to_owned(ctx, d, expr_ptr, &mut fixes);
69    }
70
71    if fixes.is_empty() { None } else { Some(fixes) }
72}

74fn add_reference(
75    ctx: &DiagnosticsContext<'_>,
76    d: &hir::TypeMismatch<'_>,
77    expr_ptr: &InFile<AstPtr<ast::Expr>>,
78    acc: &mut Vec<Assist>,
79) -> Option<()> {
80    let range = ctx.sema.diagnostics_display_range((*expr_ptr).map(|it| it.into()));
81
82    let (_, mutability) = d.expected.as_reference()?;
83    let actual_with_ref = d.actual.add_reference(mutability);
84    if !actual_with_ref.could_coerce_to(ctx.sema.db, &d.expected) {
85        return None;
86    }
87
88    let ampersands = format!("&{}", mutability.as_keyword_for_ref());
89
90    let edit = TextEdit::insert(range.range.start(), ampersands);
91    let source_change = SourceChange::from_text_edit(range.file_id, edit);
92    acc.push(fix("add_reference_here", "Add reference here", source_change, range.range));
93    Some(())
94}

96fn add_missing_ok_or_some(
97    ctx: &DiagnosticsContext<'_>,
98    d: &hir::TypeMismatch<'_>,
99    expr_ptr: &InFile<AstPtr<ast::Expr>>,
100    acc: &mut Vec<Assist>,
101) -> Option<()> {
102    let root = ctx.sema.db.parse_or_expand(expr_ptr.file_id);
103    let expr = expr_ptr.value.to_node(&root);
104    let expr_range = expr.syntax().text_range();
105    let scope = ctx.sema.scope(expr.syntax())?;
106
107    let expected_adt = d.expected.as_adt()?;
108    let expected_enum = expected_adt.as_enum()?;
109
110    let famous_defs = FamousDefs(&ctx.sema, scope.krate());
111    let core_result = famous_defs.core_result_Result();
112    let core_option = famous_defs.core_option_Option();
113
114    if Some(expected_enum) != core_result && Some(expected_enum) != core_option {
115        return None;
116    }
117
118    let variant_name = if Some(expected_enum) == core_result { "Ok" } else { "Some" };
119
120    let wrapped_actual_ty =
121        expected_adt.ty_with_args(ctx.sema.db, std::iter::once(d.actual.clone()));
122
123    if !d.expected.could_unify_with(ctx.sema.db, &wrapped_actual_ty) {
124        return None;
125    }
126
127    if d.actual.is_unit() {
128        if let Expr::BlockExpr(block) = &expr {
129            if block.tail_expr().is_none() {
130                // Fix for forms like `fn foo() -> Result<(), String> {}`
131                let mut builder = TextEdit::builder();
132                let block_indent = block.indent_level();
133
134                if block.statements().count() == 0 {
135                    // Empty block
136                    let indent = block_indent + 1;
137                    builder.insert(
138                        block.syntax().text_range().start() + TextSize::from(1),
139                        format!("\n{indent}{variant_name}(())\n{block_indent}"),
140                    );
141                } else {
142                    let indent = IndentLevel::from(1);
143                    builder.insert(
144                        block.syntax().text_range().end() - TextSize::from(1),
145                        format!("{indent}{variant_name}(())\n{block_indent}"),
146                    );
147                }
148
149                let source_change = SourceChange::from_text_edit(
150                    expr_ptr.file_id.original_file(ctx.sema.db).file_id(ctx.sema.db),
151                    builder.finish(),
152                );
153                let name = format!("Insert {variant_name}(()) as the tail of this block");
154                acc.push(fix("insert_wrapped_unit", &name, source_change, expr_range));
155            }
156            return Some(());
157        } else if let Expr::ReturnExpr(ret_expr) = &expr {
158            // Fix for forms like `fn foo() -> Result<(), String> { return; }`
159            if ret_expr.expr().is_none() {
160                let mut builder = TextEdit::builder();
161                builder
162                    .insert(ret_expr.syntax().text_range().end(), format!(" {variant_name}(())"));
163                let source_change = SourceChange::from_text_edit(
164                    expr_ptr.file_id.original_file(ctx.sema.db).file_id(ctx.sema.db),
165                    builder.finish(),
166                );
167                let name = format!("Insert {variant_name}(()) as the return value");
168                acc.push(fix("insert_wrapped_unit", &name, source_change, expr_range));
169            }
170            return Some(());
171        }
172    }
173
174    let mut builder = TextEdit::builder();
175    builder.insert(expr.syntax().text_range().start(), format!("{variant_name}("));
176    builder.insert(expr.syntax().text_range().end(), ")".to_owned());
177    let source_change = SourceChange::from_text_edit(
178        expr_ptr.file_id.original_file(ctx.sema.db).file_id(ctx.sema.db),
179        builder.finish(),
180    );
181    let name = format!("Wrap in {variant_name}");
182    acc.push(fix("wrap_in_constructor", &name, source_change, expr_range));
183    Some(())
184}

186fn remove_unnecessary_wrapper(
187    ctx: &DiagnosticsContext<'_>,
188    d: &hir::TypeMismatch<'_>,
189    expr_ptr: &InFile<AstPtr<ast::Expr>>,
190    acc: &mut Vec<Assist>,
191) -> Option<()> {
192    let db = ctx.sema.db;
193    let root = db.parse_or_expand(expr_ptr.file_id);
194    let expr = expr_ptr.value.to_node(&root);
195    let expr = ctx.sema.original_ast_node(expr)?;
196
197    let Expr::CallExpr(call_expr) = expr else {
198        return None;
199    };
200
201    let callable = ctx.sema.resolve_expr_as_callable(&call_expr.expr()?)?;
202    let CallableKind::TupleEnumVariant(variant) = callable.kind() else {
203        return None;
204    };
205
206    let actual_enum = d.actual.as_adt()?.as_enum()?;
207    let famous_defs = FamousDefs(&ctx.sema, ctx.sema.scope(call_expr.syntax())?.krate());
208    let core_option = famous_defs.core_option_Option();
209    let core_result = famous_defs.core_result_Result();
210    if Some(actual_enum) != core_option && Some(actual_enum) != core_result {
211        return None;
212    }
213
214    let inner_type = variant.fields(db).first()?.ty_with_args(db, d.actual.type_arguments());
215    if !d.expected.could_unify_with(db, &inner_type) {
216        return None;
217    }
218
219    let inner_arg = call_expr.arg_list()?.args().next()?;
220
221    let file_id = expr_ptr.file_id.original_file(db);
222    let mut builder = SourceChangeBuilder::new(file_id.file_id(ctx.sema.db));
223    let mut editor;
224    match inner_arg {
225        // We're returning `()`
226        Expr::TupleExpr(tup) if tup.fields().next().is_none() => {
227            let parent = call_expr
228                .syntax()
229                .parent()
230                .and_then(Either::<ast::ReturnExpr, ast::StmtList>::cast)?;
231
232            editor = builder.make_editor(parent.syntax());
233            let make = SyntaxFactory::with_mappings();
234
235            match parent {
236                Either::Left(ret_expr) => {
237                    editor.replace(ret_expr.syntax(), make.expr_return(None).syntax());
238                }
239                Either::Right(stmt_list) => {
240                    let new_block = if stmt_list.statements().next().is_none() {
241                        make.expr_empty_block()
242                    } else {
243                        make.block_expr(stmt_list.statements(), None)
244                    };
245
246                    editor.replace(stmt_list.syntax().parent()?, new_block.syntax());
247                }
248            }
249
250            editor.add_mappings(make.finish_with_mappings());
251        }
252        _ => {
253            editor = builder.make_editor(call_expr.syntax());
254            editor.replace(call_expr.syntax(), inner_arg.syntax());
255        }
256    }
257
258    builder.add_file_edits(file_id.file_id(ctx.sema.db), editor);
259    let name = format!("Remove unnecessary {}() wrapper", variant.name(db).as_str());
260    acc.push(fix(
261        "remove_unnecessary_wrapper",
262        &name,
263        builder.finish(),
264        call_expr.syntax().text_range(),
265    ));
266    Some(())
267}

269fn remove_semicolon(
270    ctx: &DiagnosticsContext<'_>,
271    d: &hir::TypeMismatch<'_>,
272    expr_ptr: &InFile<AstPtr<ast::Expr>>,
273    acc: &mut Vec<Assist>,
274) -> Option<()> {
275    let root = ctx.sema.db.parse_or_expand(expr_ptr.file_id);
276    let expr = expr_ptr.value.to_node(&root);
277    if !d.actual.is_unit() {
278        return None;
279    }
280    let block = BlockExpr::cast(expr.syntax().clone())?;
281    let expr_before_semi =
282        block.statements().last().and_then(|s| ExprStmt::cast(s.syntax().clone()))?;
283    let type_before_semi = ctx.sema.type_of_expr(&expr_before_semi.expr()?)?.original();
284    if !type_before_semi.could_coerce_to(ctx.sema.db, &d.expected) {
285        return None;
286    }
287    let semicolon_range = expr_before_semi.semicolon_token()?.text_range();
288
289    let edit = TextEdit::delete(semicolon_range);
290    let source_change = SourceChange::from_text_edit(
291        expr_ptr.file_id.original_file(ctx.sema.db).file_id(ctx.sema.db),
292        edit,
293    );
294
295    acc.push(fix("remove_semicolon", "Remove this semicolon", source_change, semicolon_range));
296    Some(())
297}

299fn str_ref_to_owned(
300    ctx: &DiagnosticsContext<'_>,
301    d: &hir::TypeMismatch<'_>,
302    expr_ptr: &InFile<AstPtr<ast::Expr>>,
303    acc: &mut Vec<Assist>,
304) -> Option<()> {
305    let expected = d.expected.display(ctx.sema.db, ctx.display_target);
306    // FIXME do this properly
307    let is_applicable = d.actual.strip_reference().is_str() && expected.to_string() == "String";
308    if !is_applicable {
309        return None;
310    }
311
312    let root = ctx.sema.db.parse_or_expand(expr_ptr.file_id);
313    let expr = expr_ptr.value.to_node(&root);
314    let expr_range = expr.syntax().text_range();
315
316    let to_owned = ".to_owned()".to_owned();
317
318    let edit = TextEdit::insert(expr.syntax().text_range().end(), to_owned);
319    let source_change = SourceChange::from_text_edit(
320        expr_ptr.file_id.original_file(ctx.sema.db).file_id(ctx.sema.db),
321        edit,
322    );
323    acc.push(fix("str_ref_to_owned", "Add .to_owned() here", source_change, expr_range));
324
325    Some(())
326}
