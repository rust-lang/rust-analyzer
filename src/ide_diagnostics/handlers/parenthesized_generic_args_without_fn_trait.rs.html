1use crate::{Diagnostic, DiagnosticCode, DiagnosticsContext};

3// Diagnostic: parenthesized-generic-args-without-fn-trait
4//
5// This diagnostic is shown when a `Fn`-trait-style generic parameters (`Trait(A, B) -> C`)
6// was used on non-`Fn` trait/type.
7pub(crate) fn parenthesized_generic_args_without_fn_trait(
8    ctx: &DiagnosticsContext<'_>,
9    d: &hir::ParenthesizedGenericArgsWithoutFnTrait,
10) -> Diagnostic {
11    Diagnostic::new_with_syntax_node_ptr(
12        ctx,
13        DiagnosticCode::RustcHardError("E0214"),
14        "parenthesized type parameters may only be used with a `Fn` trait",
15        d.args.map(Into::into),
16    )
17    .stable()
18}
