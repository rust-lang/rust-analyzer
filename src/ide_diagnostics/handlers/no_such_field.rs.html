12// Diagnostic: no-such-field
13//
14// This diagnostic is triggered if created structure does not have field provided in record.
15pub(crate) fn no_such_field(ctx: &DiagnosticsContext<'_>, d: &hir::NoSuchField) -> Diagnostic {
16    let node = d.field.map(Into::into);
17    if d.private {
18        // FIXME: quickfix to add required visibility
19        Diagnostic::new_with_syntax_node_ptr(
20            ctx,
21            DiagnosticCode::RustcHardError("E0451"),
22            "field is private",
23            node,
24        )
25    } else {
26        Diagnostic::new_with_syntax_node_ptr(
27            ctx,
28            match d.variant {
29                VariantId::EnumVariantId(_) => DiagnosticCode::RustcHardError("E0559"),
30                _ => DiagnosticCode::RustcHardError("E0560"),
31            },
32            "no such field",
33            node,
34        )
35        .with_fixes(fixes(ctx, d))
36    }
37}
38
39fn fixes(ctx: &DiagnosticsContext<'_>, d: &hir::NoSuchField) -> Option<Vec<Assist>> {
40    // FIXME: quickfix for pattern
41    let root = ctx.sema.db.parse_or_expand(d.field.file_id);
42    match &d.field.value.to_node(&root) {
43        Either::Left(node) => missing_record_expr_field_fixes(
44            &ctx.sema,
45            d.field.file_id.original_file(ctx.sema.db),
46            node,
47        ),
48        _ => None,
49    }
50}

52fn missing_record_expr_field_fixes(
53    sema: &Semantics<'_, RootDatabase>,
54    usage_file_id: EditionedFileId,
55    record_expr_field: &ast::RecordExprField,
56) -> Option<Vec<Assist>> {
57    let record_lit = ast::RecordExpr::cast(record_expr_field.syntax().parent()?.parent()?)?;
58    let def_id = sema.resolve_variant(record_lit)?;
59    let module;
60    let def_file_id;
61    let record_fields = match def_id {
62        hir::VariantDef::Struct(s) => {
63            module = s.module(sema.db);
64            let source = s.source(sema.db)?;
65            def_file_id = source.file_id;
66            let fields = source.value.field_list()?;
67            record_field_list(fields)?
68        }
69        hir::VariantDef::Union(u) => {
70            module = u.module(sema.db);
71            let source = u.source(sema.db)?;
72            def_file_id = source.file_id;
73            source.value.record_field_list()?
74        }
75        hir::VariantDef::Variant(e) => {
76            module = e.module(sema.db);
77            let source = e.source(sema.db)?;
78            def_file_id = source.file_id;
79            let fields = source.value.field_list()?;
80            record_field_list(fields)?
81        }
82    };
83    let def_file_id = def_file_id.original_file(sema.db);
84
85    let new_field_type = sema.type_of_expr(&record_expr_field.expr()?)?.adjusted();
86    if new_field_type.is_unknown() {
87        return None;
88    }
89    let new_field = make::record_field(
90        None,
91        make::name(record_expr_field.field_name()?.ident_token()?.text()),
92        make::ty(&new_field_type.display_source_code(sema.db, module.into(), true).ok()?),
93    );
94
95    let last_field = record_fields.fields().last()?;
96    let last_field_syntax = last_field.syntax();
97    let indent = IndentLevel::from_node(last_field_syntax);
98
99    let mut new_field = new_field.to_string();
100    if usage_file_id != def_file_id {
101        new_field = format!("pub(crate) {new_field}");
102    }
103    new_field = format!("\n{indent}{new_field}");
104
105    let needs_comma = !last_field_syntax.to_string().ends_with(',');
106    if needs_comma {
107        new_field = format!(",{new_field}");
108    }
109
110    let source_change = SourceChange::from_text_edit(
111        def_file_id,
112        TextEdit::insert(last_field_syntax.text_range().end(), new_field),
113    );
114
115    return Some(vec![fix(
116        "create_field",
117        "Create field",
118        source_change,
119        record_expr_field.syntax().text_range(),
120    )]);
121
122    fn record_field_list(field_def_list: ast::FieldList) -> Option<ast::RecordFieldList> {
123        match field_def_list {
124            ast::FieldList::RecordFieldList(it) => Some(it),
125            ast::FieldList::TupleFieldList(_) => None,
126        }
127    }
