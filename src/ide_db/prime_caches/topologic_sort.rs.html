1//! helper data structure to schedule work for parallel prime caches.
2use std::{collections::VecDeque, hash::Hash};
3
4use crate::FxHashMap;
5
6pub(crate) struct TopologicSortIterBuilder<T> {
7    nodes: FxHashMap<T, Entry<T>>,
8}
9
10// this implementation has different bounds on T than would be implied by #[derive(Default)]
11impl<T> Default for TopologicSortIterBuilder<T>
12where
13    T: Copy + Eq + PartialEq + Hash,
14{
15    fn default() -> Self {
16        Self { nodes: Default::default() }
17    }
18}
19
20impl<T> TopologicSortIterBuilder<T>
21where
22    T: Copy + Eq + PartialEq + Hash,
23{
24    fn get_or_create_entry(&mut self, item: T) -> &mut Entry<T> {
25        self.nodes.entry(item).or_default()
26    }
27
28    pub(crate) fn add(&mut self, item: T, predecessors: impl IntoIterator<Item = T>) {
29        let mut num_predecessors = 0;
30
31        for predecessor in predecessors.into_iter() {
32            self.get_or_create_entry(predecessor).successors.push(item);
33            num_predecessors += 1;
34        }
35
36        let entry = self.get_or_create_entry(item);
37        entry.num_predecessors += num_predecessors;
38    }
39
40    pub(crate) fn build(self) -> TopologicalSortIter<T> {
41        let ready = self
42            .nodes
43            .iter()
44            .filter_map(
45                |(item, entry)| if entry.num_predecessors == 0 { Some(*item) } else { None },
46            )
47            .collect();
48
49        TopologicalSortIter { nodes: self.nodes, ready }
50    }
51}
52
53pub(crate) struct TopologicalSortIter<T> {
54    ready: VecDeque<T>,
55    nodes: FxHashMap<T, Entry<T>>,
56}
57
58impl<T> TopologicalSortIter<T>
59where
60    T: Copy + Eq + PartialEq + Hash,
61{
62    pub(crate) fn builder() -> TopologicSortIterBuilder<T> {
63        TopologicSortIterBuilder::default()
64    }
65
66    pub(crate) fn pending(&self) -> usize {
67        self.nodes.len()
68    }
69
70    pub(crate) fn mark_done(&mut self, item: T) {
71        let entry = self.nodes.remove(&item).expect("invariant: unknown item marked as done");
72
73        for successor in entry.successors {
74            let succ_entry = self
75                .nodes
76                .get_mut(&successor)
77                .expect("invariant: unknown successor referenced by entry");
78
79            succ_entry.num_predecessors -= 1;
80            if succ_entry.num_predecessors == 0 {
81                self.ready.push_back(successor);
82            }
83        }
84    }
85}
86
87impl<T> Iterator for TopologicalSortIter<T> {
88    type Item = T;
89
90    fn next(&mut self) -> Option<Self::Item> {
91        self.ready.pop_front()
92    }
93}
94
95struct Entry<T> {
96    successors: Vec<T>,
97    num_predecessors: usize,
98}
99
100impl<T> Default for Entry<T> {
101    fn default() -> Self {
102        Self { successors: Default::default(), num_predecessors: 0 }
103    }
104}
