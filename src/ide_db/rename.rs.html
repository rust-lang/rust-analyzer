1//! Rename infrastructure for rust-analyzer. It is used primarily for the
2//! literal "rename" in the ide (look for tests there), but it is also available
3//! as a general-purpose service. For example, it is used by the fix for the
4//! "incorrect case" diagnostic.
5//!
6//! It leverages the [`crate::search`] functionality to find what needs to be
7//! renamed. The actual renames are tricky -- field shorthands need special
8//! attention, and, when renaming modules, you also want to rename files on the
9//! file system.
10//!
11//! Another can of worms are macros:
12//!
13//! ```ignore
14//! macro_rules! m { () => { fn f() {} } }
15//! m!();
16//! fn main() {
17//!     f() // <- rename me
18//! }
19//! ```
20//!
21//! The correct behavior in such cases is probably to show a dialog to the user.
22//! Our current behavior is ¯\_(ツ)_/¯.

73impl Definition {
74    pub fn rename(
75        &self,
76        sema: &Semantics<'_, RootDatabase>,
77        new_name: &str,
78    ) -> Result<SourceChange> {
79        // We append `r#` if needed.
80        let new_name = new_name.trim_start_matches("r#");
81
82        // self.krate() returns None if
83        // self is a built-in attr, built-in type or tool module.
84        // it is not allowed for these defs to be renamed.
85        // cases where self.krate() is None is handled below.
86        if let Some(krate) = self.krate(sema.db) {
87            // Can we not rename non-local items?
88            // Then bail if non-local
89            if !krate.origin(sema.db).is_local() {
90                bail!("Cannot rename a non-local definition")
91            }
92        }
93
94        match *self {
95            Definition::Module(module) => rename_mod(sema, module, new_name),
96            Definition::ToolModule(_) => {
97                bail!("Cannot rename a tool module")
98            }
99            Definition::BuiltinType(_) => {
100                bail!("Cannot rename builtin type")
101            }
102            Definition::BuiltinAttr(_) => {
103                bail!("Cannot rename a builtin attr.")
104            }
105            Definition::SelfType(_) => bail!("Cannot rename `Self`"),
106            Definition::Macro(mac) => rename_reference(sema, Definition::Macro(mac), new_name),
107            def => rename_reference(sema, def, new_name),
108        }
109    }
110
111    /// Textual range of the identifier which will change when renaming this
112    /// `Definition`. Note that builtin types can't be
113    /// renamed and extern crate names will report its range, though a rename will introduce
114    /// an alias instead.
115    pub fn range_for_rename(self, sema: &Semantics<'_, RootDatabase>) -> Option<FileRange> {
116        let syn_ctx_is_root = |(range, ctx): (_, SyntaxContext)| ctx.is_root().then_some(range);
117        let res = match self {
118            Definition::Macro(mac) => {
119                let src = sema.source(mac)?;
120                let name = match &src.value {
121                    Either::Left(it) => it.name()?,
122                    Either::Right(it) => it.name()?,
123                };
124                src.with_value(name.syntax())
125                    .original_file_range_opt(sema.db)
126                    .and_then(syn_ctx_is_root)
127            }
128            Definition::Field(field) => {
129                let src = sema.source(field)?;
130                match &src.value {
131                    FieldSource::Named(record_field) => {
132                        let name = record_field.name()?;
133                        src.with_value(name.syntax())
134                            .original_file_range_opt(sema.db)
135                            .and_then(syn_ctx_is_root)
136                    }
137                    FieldSource::Pos(_) => None,
138                }
139            }
140            Definition::Crate(_) => None,
141            Definition::Module(module) => {
142                let src = module.declaration_source(sema.db)?;
143                let name = src.value.name()?;
144                src.with_value(name.syntax())
145                    .original_file_range_opt(sema.db)
146                    .and_then(syn_ctx_is_root)
147            }
148            Definition::Function(it) => name_range(it, sema).and_then(syn_ctx_is_root),
149            Definition::Adt(adt) => match adt {
150                hir::Adt::Struct(it) => name_range(it, sema).and_then(syn_ctx_is_root),
151                hir::Adt::Union(it) => name_range(it, sema).and_then(syn_ctx_is_root),
152                hir::Adt::Enum(it) => name_range(it, sema).and_then(syn_ctx_is_root),
153            },
154            Definition::Variant(it) => name_range(it, sema).and_then(syn_ctx_is_root),
155            Definition::Const(it) => name_range(it, sema).and_then(syn_ctx_is_root),
156            Definition::Static(it) => name_range(it, sema).and_then(syn_ctx_is_root),
157            Definition::Trait(it) => name_range(it, sema).and_then(syn_ctx_is_root),
158            Definition::TraitAlias(it) => name_range(it, sema).and_then(syn_ctx_is_root),
159            Definition::TypeAlias(it) => name_range(it, sema).and_then(syn_ctx_is_root),
160            Definition::Local(it) => {
161                name_range(it.primary_source(sema.db), sema).and_then(syn_ctx_is_root)
162            }
163            Definition::GenericParam(generic_param) => match generic_param {
164                hir::GenericParam::LifetimeParam(lifetime_param) => {
165                    let src = sema.source(lifetime_param)?;
166                    src.with_value(src.value.lifetime()?.syntax())
167                        .original_file_range_opt(sema.db)
168                        .and_then(syn_ctx_is_root)
169                }
170                _ => {
171                    let param = match generic_param {
172                        hir::GenericParam::TypeParam(it) => it.merge(),
173                        hir::GenericParam::ConstParam(it) => it.merge(),
174                        hir::GenericParam::LifetimeParam(_) => return None,
175                    };
176                    let src = sema.source(param)?;
177                    let name = match &src.value {
178                        Either::Left(x) => x.name()?,
179                        Either::Right(_) => return None,
180                    };
181                    src.with_value(name.syntax())
182                        .original_file_range_opt(sema.db)
183                        .and_then(syn_ctx_is_root)
184                }
185            },
186            Definition::Label(label) => {
187                let src = sema.source(label)?;
188                let lifetime = src.value.lifetime()?;
189                src.with_value(lifetime.syntax())
190                    .original_file_range_opt(sema.db)
191                    .and_then(syn_ctx_is_root)
192            }
193            Definition::ExternCrateDecl(it) => {
194                let src = sema.source(it)?;
195                if let Some(rename) = src.value.rename() {
196                    let name = rename.name()?;
197                    src.with_value(name.syntax())
198                        .original_file_range_opt(sema.db)
199                        .and_then(syn_ctx_is_root)
200                } else {
201                    let name = src.value.name_ref()?;
202                    src.with_value(name.syntax())
203                        .original_file_range_opt(sema.db)
204                        .and_then(syn_ctx_is_root)
205                }
206            }
207            Definition::InlineAsmOperand(it) => name_range(it, sema).and_then(syn_ctx_is_root),
208            Definition::BuiltinType(_)
209            | Definition::BuiltinLifetime(_)
210            | Definition::BuiltinAttr(_)
211            | Definition::SelfType(_)
212            | Definition::ToolModule(_)
213            | Definition::TupleField(_)
214            | Definition::InlineAsmRegOrRegClass(_) => return None,
215            // FIXME: This should be doable in theory
216            Definition::DeriveHelper(_) => return None,
217        };
218        return res;
219
220        fn name_range<D>(
221            def: D,
222            sema: &Semantics<'_, RootDatabase>,
223        ) -> Option<(FileRange, SyntaxContext)>
224        where
225            D: hir::HasSource,
226            D::Ast: ast::HasName,
227        {
228            let src = sema.source(def)?;
229            let name = src.value.name()?;
230            src.with_value(name.syntax()).original_file_range_opt(sema.db)
231        }
232    }
233}

235fn rename_mod(
236    sema: &Semantics<'_, RootDatabase>,
237    module: hir::Module,
238    new_name: &str,
239) -> Result<SourceChange> {
240    if IdentifierKind::classify(new_name)? != IdentifierKind::Ident {
241        bail!("Invalid name `{0}`: cannot rename module to {0}", new_name);
242    }
243
244    let mut source_change = SourceChange::default();
245
246    if module.is_crate_root() {
247        return Ok(source_change);
248    }
249
250    let InFile { file_id, value: def_source } = module.definition_source(sema.db);
251    if let ModuleSource::SourceFile(..) = def_source {
252        let anchor = file_id.original_file(sema.db).file_id();
253
254        let is_mod_rs = module.is_mod_rs(sema.db);
255        let has_detached_child = module.children(sema.db).any(|child| !child.is_inline(sema.db));
256
257        // Module exists in a named file
258        if !is_mod_rs {
259            let path = format!("{new_name}.rs");
260            let dst = AnchoredPathBuf { anchor, path };
261            source_change.push_file_system_edit(FileSystemEdit::MoveFile { src: anchor, dst })
262        }
263
264        // Rename the dir if:
265        //  - Module source is in mod.rs
266        //  - Module has submodules defined in separate files
267        let dir_paths = match (is_mod_rs, has_detached_child, module.name(sema.db)) {
268            // Go up one level since the anchor is inside the dir we're trying to rename
269            (true, _, Some(mod_name)) => {
270                Some((format!("../{}", mod_name.as_str()), format!("../{new_name}")))
271            }
272            // The anchor is on the same level as target dir
273            (false, true, Some(mod_name)) => {
274                Some((mod_name.as_str().to_owned(), new_name.to_owned()))
275            }
276            _ => None,
277        };
278
279        if let Some((src, dst)) = dir_paths {
280            let src = AnchoredPathBuf { anchor, path: src };
281            let dst = AnchoredPathBuf { anchor, path: dst };
282            source_change.push_file_system_edit(FileSystemEdit::MoveDir {
283                src,
284                src_id: anchor,
285                dst,
286            })
287        }
288    }
289
290    if let Some(src) = module.declaration_source(sema.db) {
291        let file_id = src.file_id.original_file(sema.db);
292        match src.value.name() {
293            Some(name) => {
294                if let Some(file_range) = src
295                    .with_value(name.syntax())
296                    .original_file_range_opt(sema.db)
297                    .map(TupleExt::head)
298                {
299                    let new_name = if is_raw_identifier(new_name, file_id.edition()) {
300                        format!("r#{new_name}")
301                    } else {
302                        new_name.to_owned()
303                    };
304                    source_change.insert_source_edit(
305                        file_id.file_id(),
306                        TextEdit::replace(file_range.range, new_name),
307                    )
308                };
309            }
310            _ => never!("Module source node is missing a name"),
311        }
312    }
313
314    let def = Definition::Module(module);
315    let usages = def.usages(sema).all();
316    let ref_edits = usages.iter().map(|(file_id, references)| {
317        (
318            EditionedFileId::file_id(file_id),
319            source_edit_from_references(references, def, new_name, file_id.edition()),
320        )
321    });
322    source_change.extend(ref_edits);
323
324    Ok(source_change)
325}

327fn rename_reference(
328    sema: &Semantics<'_, RootDatabase>,
329    def: Definition,
330    new_name: &str,
331) -> Result<SourceChange> {
332    let ident_kind = IdentifierKind::classify(new_name)?;
333
334    if matches!(
335        def,
336        Definition::GenericParam(hir::GenericParam::LifetimeParam(_)) | Definition::Label(_)
337    ) {
338        match ident_kind {
339            IdentifierKind::Underscore => {
340                bail!("Invalid name `{}`: not a lifetime identifier", new_name);
341            }
342            _ => cov_mark::hit!(rename_lifetime),
343        }
344    } else {
345        match ident_kind {
346            IdentifierKind::Lifetime => {
347                cov_mark::hit!(rename_not_an_ident_ref);
348                bail!("Invalid name `{}`: not an identifier", new_name);
349            }
350            IdentifierKind::Ident => cov_mark::hit!(rename_non_local),
351            IdentifierKind::Underscore => (),
352        }
353    }
354
355    let def = convert_to_def_in_trait(sema.db, def);
356    let usages = def.usages(sema).all();
357
358    if !usages.is_empty() && ident_kind == IdentifierKind::Underscore {
359        cov_mark::hit!(rename_underscore_multiple);
360        bail!("Cannot rename reference to `_` as it is being referenced multiple times");
361    }
362    let mut source_change = SourceChange::default();
363    source_change.extend(usages.iter().map(|(file_id, references)| {
364        (
365            EditionedFileId::file_id(file_id),
366            source_edit_from_references(references, def, new_name, file_id.edition()),
367        )
368    }));
369
370    // This needs to come after the references edits, because we change the annotation of existing edits
371    // if a conflict is detected.
372    let (file_id, edit) = source_edit_from_def(sema, def, new_name, &mut source_change)?;
373    source_change.insert_source_edit(file_id, edit);
374    Ok(source_change)
375}

377pub fn source_edit_from_references(
378    references: &[FileReference],
379    def: Definition,
380    new_name: &str,
381    edition: Edition,
382) -> TextEdit {
383    let new_name = if is_raw_identifier(new_name, edition) {
384        format!("r#{new_name}")
385    } else {
386        new_name.to_owned()
387    };
388    let mut edit = TextEdit::builder();
389    // macros can cause multiple refs to occur for the same text range, so keep track of what we have edited so far
390    let mut edited_ranges = Vec::new();
391    for &FileReference { range, ref name, .. } in references {
392        let name_range = name.text_range();
393        if name_range.len() != range.len() {
394            // This usage comes from a different token kind that was downmapped to a NameLike in a macro
395            // Renaming this will most likely break things syntax-wise
396            continue;
397        }
398        let has_emitted_edit = match name {
399            // if the ranges differ then the node is inside a macro call, we can't really attempt
400            // to make special rewrites like shorthand syntax and such, so just rename the node in
401            // the macro input
402            FileReferenceNode::NameRef(name_ref) if name_range == range => {
403                source_edit_from_name_ref(&mut edit, name_ref, &new_name, def)
404            }
405            FileReferenceNode::Name(name) if name_range == range => {
406                source_edit_from_name(&mut edit, name, &new_name)
407            }
408            _ => false,
409        };
410        if !has_emitted_edit && !edited_ranges.contains(&range.start()) {
411            let (range, new_name) = match name {
412                FileReferenceNode::Lifetime(_) => (
413                    TextRange::new(range.start() + syntax::TextSize::from(1), range.end()),
414                    new_name.strip_prefix('\'').unwrap_or(&new_name).to_owned(),
415                ),
416                _ => (range, new_name.to_owned()),
417            };
418
419            edit.replace(range, new_name);
420            edited_ranges.push(range.start());
421        }
422    }
423
424    edit.finish()
425}

427fn source_edit_from_name(edit: &mut TextEditBuilder, name: &ast::Name, new_name: &str) -> bool {
428    if ast::RecordPatField::for_field_name(name).is_some() {
429        if let Some(ident_pat) = name.syntax().parent().and_then(ast::IdentPat::cast) {
430            cov_mark::hit!(rename_record_pat_field_name_split);
431            // Foo { ref mut field } -> Foo { new_name: ref mut field }
432            //      ^ insert `new_name: `
433
434            // FIXME: instead of splitting the shorthand, recursively trigger a rename of the
435            // other name https://github.com/rust-lang/rust-analyzer/issues/6547
436            edit.insert(ident_pat.syntax().text_range().start(), format!("{new_name}: "));
437            return true;
438        }
439    }
440
441    false
442}

444fn source_edit_from_name_ref(
445    edit: &mut TextEditBuilder,
446    name_ref: &ast::NameRef,
447    new_name: &str,
448    def: Definition,
449) -> bool {
450    if name_ref.super_token().is_some() {
451        return true;
452    }
453
454    if let Some(record_field) = ast::RecordExprField::for_name_ref(name_ref) {
455        let rcf_name_ref = record_field.name_ref();
456        let rcf_expr = record_field.expr();
457        match &(rcf_name_ref, rcf_expr.and_then(|it| expr_as_name_ref(&it))) {
458            // field: init-expr, check if we can use a field init shorthand
459            (Some(field_name), Some(init)) => {
460                if field_name == name_ref {
461                    if init.text() == new_name {
462                        cov_mark::hit!(test_rename_field_put_init_shorthand);
463                        // Foo { field: local } -> Foo { local }
464                        //       ^^^^^^^ delete this
465
466                        // same names, we can use a shorthand here instead.
467                        // we do not want to erase attributes hence this range start
468                        let s = field_name.syntax().text_range().start();
469                        let e = init.syntax().text_range().start();
470                        edit.delete(TextRange::new(s, e));
471                        return true;
472                    }
473                } else if init == name_ref && field_name.text() == new_name {
474                    cov_mark::hit!(test_rename_local_put_init_shorthand);
475                    // Foo { field: local } -> Foo { field }
476                    //            ^^^^^^^ delete this
477
478                    // same names, we can use a shorthand here instead.
479                    // we do not want to erase attributes hence this range start
480                    let s = field_name.syntax().text_range().end();
481                    let e = init.syntax().text_range().end();
482                    edit.delete(TextRange::new(s, e));
483                    return true;
484                }
485            }
486            // init shorthand
487            (None, Some(_)) if matches!(def, Definition::Field(_)) => {
488                cov_mark::hit!(test_rename_field_in_field_shorthand);
489                // Foo { field } -> Foo { new_name: field }
490                //       ^ insert `new_name: `
491                let offset = name_ref.syntax().text_range().start();
492                edit.insert(offset, format!("{new_name}: "));
493                return true;
494            }
495            (None, Some(_)) if matches!(def, Definition::Local(_)) => {
496                cov_mark::hit!(test_rename_local_in_field_shorthand);
497                // Foo { field } -> Foo { field: new_name }
498                //            ^ insert `: new_name`
499                let offset = name_ref.syntax().text_range().end();
500                edit.insert(offset, format!(": {new_name}"));
501                return true;
502            }
503            _ => (),
504        }
505    } else if let Some(record_field) = ast::RecordPatField::for_field_name_ref(name_ref) {
506        let rcf_name_ref = record_field.name_ref();
507        let rcf_pat = record_field.pat();
508        match (rcf_name_ref, rcf_pat) {
509            // field: rename
510            (Some(field_name), Some(ast::Pat::IdentPat(pat)))
511                if field_name == *name_ref && pat.at_token().is_none() =>
512            {
513                // field name is being renamed
514                if let Some(name) = pat.name() {
515                    if name.text() == new_name {
516                        cov_mark::hit!(test_rename_field_put_init_shorthand_pat);
517                        // Foo { field: ref mut local } -> Foo { ref mut field }
518                        //       ^^^^^^^ delete this
519                        //                      ^^^^^ replace this with `field`
520
521                        // same names, we can use a shorthand here instead/
522                        // we do not want to erase attributes hence this range start
523                        let s = field_name.syntax().text_range().start();
524                        let e = pat.syntax().text_range().start();
525                        edit.delete(TextRange::new(s, e));
526                        edit.replace(name.syntax().text_range(), new_name.to_owned());
527                        return true;
528                    }
529                }
530            }
531            _ => (),
532        }
533    }
534    false
535}

537fn source_edit_from_def(
538    sema: &Semantics<'_, RootDatabase>,
539    def: Definition,
540    new_name: &str,
541    source_change: &mut SourceChange,
542) -> Result<(FileId, TextEdit)> {
543    let new_name_edition_aware = |new_name: &str, file_id: EditionedFileId| {
544        if is_raw_identifier(new_name, file_id.edition()) {
545            format!("r#{new_name}")
546        } else {
547            new_name.to_owned()
548        }
549    };
550    let mut edit = TextEdit::builder();
551    if let Definition::Local(local) = def {
552        let mut file_id = None;
553
554        let conflict_annotation = if !sema.rename_conflicts(&local, new_name).is_empty() {
555            Some(
556                source_change.insert_annotation(ChangeAnnotation {
557                    label: "This rename will change the program's meaning".to_owned(),
558                    needs_confirmation: true,
559                    description: Some(
560                        "Some variable(s) will shadow the renamed variable \
561                        or be shadowed by it if the rename is performed"
562                            .to_owned(),
563                    ),
564                }),
565            )
566        } else {
567            None
568        };
569
570        for source in local.sources(sema.db) {
571            let source = match source.source.clone().original_ast_node_rooted(sema.db) {
572                Some(source) => source,
573                None => match source
574                    .source
575                    .syntax()
576                    .original_file_range_opt(sema.db)
577                    .map(TupleExt::head)
578                {
579                    Some(FileRange { file_id: file_id2, range }) => {
580                        file_id = Some(file_id2);
581                        edit.replace(range, new_name_edition_aware(new_name, file_id2));
582                        continue;
583                    }
584                    None => {
585                        bail!("Can't rename local that is defined in a macro declaration")
586                    }
587                },
588            };
589            file_id = Some(source.file_id);
590            if let Either::Left(pat) = source.value {
591                let name_range = pat.name().unwrap().syntax().text_range();
592                // special cases required for renaming fields/locals in Record patterns
593                if let Some(pat_field) = pat.syntax().parent().and_then(ast::RecordPatField::cast) {
594                    if let Some(name_ref) = pat_field.name_ref() {
595                        if new_name == name_ref.text().as_str().trim_start_matches("r#")
596                            && pat.at_token().is_none()
597                        {
598                            // Foo { field: ref mut local } -> Foo { ref mut field }
599                            //       ^^^^^^ delete this
600                            //                      ^^^^^ replace this with `field`
601                            cov_mark::hit!(test_rename_local_put_init_shorthand_pat);
602                            edit.delete(
603                                name_ref
604                                    .syntax()
605                                    .text_range()
606                                    .cover_offset(pat.syntax().text_range().start()),
607                            );
608                            edit.replace(name_range, name_ref.text().to_string());
609                        } else {
610                            // Foo { field: ref mut local @ local 2} -> Foo { field: ref mut new_name @ local2 }
611                            // Foo { field: ref mut local } -> Foo { field: ref mut new_name }
612                            //                      ^^^^^ replace this with `new_name`
613                            edit.replace(
614                                name_range,
615                                new_name_edition_aware(new_name, source.file_id),
616                            );
617                        }
618                    } else {
619                        // Foo { ref mut field } -> Foo { field: ref mut new_name }
620                        //   original_ast_node_rootedd: `
621                        //               ^^^^^ replace this with `new_name`
622                        edit.insert(
623                            pat.syntax().text_range().start(),
624                            format!("{}: ", pat_field.field_name().unwrap()),
625                        );
626                        edit.replace(name_range, new_name_edition_aware(new_name, source.file_id));
627                    }
628                } else {
629                    edit.replace(name_range, new_name_edition_aware(new_name, source.file_id));
630                }
631            }
632        }
633        let mut edit = edit.finish();
634
635        for (edit, _) in source_change.source_file_edits.values_mut() {
636            edit.set_annotation(conflict_annotation);
637        }
638        edit.set_annotation(conflict_annotation);
639
640        let Some(file_id) = file_id else { bail!("No file available to rename") };
641        return Ok((EditionedFileId::file_id(file_id), edit));
642    }
643    let FileRange { file_id, range } = def
644        .range_for_rename(sema)
645        .ok_or_else(|| format_err!("No identifier available to rename"))?;
646    let (range, new_name) = match def {
647        Definition::GenericParam(hir::GenericParam::LifetimeParam(_)) | Definition::Label(_) => (
648            TextRange::new(range.start() + syntax::TextSize::from(1), range.end()),
649            new_name.strip_prefix('\'').unwrap_or(new_name).to_owned(),
650        ),
651        Definition::ExternCrateDecl(decl) if decl.alias(sema.db).is_none() => {
652            (TextRange::empty(range.end()), format!(" as {new_name}"))
653        }
654        _ => (range, new_name.to_owned()),
655    };
656    edit.replace(range, new_name_edition_aware(&new_name, file_id));
657    Ok((file_id.file_id(), edit.finish()))
658}

660#[derive(Copy, Clone, Debug, PartialEq)]
661pub enum IdentifierKind {
662    Ident,
663    Lifetime,
664    Underscore,
665}
666
667impl IdentifierKind {
668    pub fn classify(new_name: &str) -> Result<IdentifierKind> {
669        let new_name = new_name.trim_start_matches("r#");
670        match parser::LexedStr::single_token(Edition::LATEST, new_name) {
671            Some(res) => match res {
672                (SyntaxKind::IDENT, _) => {
673                    if let Some(inner) = new_name.strip_prefix("r#") {
674                        if matches!(inner, "self" | "crate" | "super" | "Self") {
675                            bail!("Invalid name: `{}` cannot be a raw identifier", inner);
676                        }
677                    }
678                    Ok(IdentifierKind::Ident)
679                }
680                (T![_], _) => Ok(IdentifierKind::Underscore),
681                (SyntaxKind::LIFETIME_IDENT, _) if new_name != "'static" && new_name != "'_" => {
682                    Ok(IdentifierKind::Lifetime)
683                }
684                _ if is_raw_identifier(new_name, Edition::LATEST) => Ok(IdentifierKind::Ident),
685                (_, Some(syntax_error)) => bail!("Invalid name `{}`: {}", new_name, syntax_error),
686                (_, None) => bail!("Invalid name `{}`: not an identifier", new_name),
687            },
688            None => bail!("Invalid name `{}`: not an identifier", new_name),
689        }
690    }
691}
