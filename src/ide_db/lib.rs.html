79#[salsa_macros::db]
80pub struct RootDatabase {
81    // FIXME: Revisit this commit now that we migrated to the new salsa, given we store arcs in this
82    // db directly now
83    // We use `ManuallyDrop` here because every codegen unit that contains a
84    // `&RootDatabase -> &dyn OtherDatabase` cast will instantiate its drop glue in the vtable,
85    // which duplicates `Weak::drop` and `Arc::drop` tens of thousands of times, which makes
86    // compile times of all `ide_*` and downstream crates suffer greatly.
87    storage: ManuallyDrop<salsa::Storage<Self>>,
88    files: Arc<Files>,
89    crates_map: Arc<CratesMap>,
90    nonce: Nonce,
91}
92
93impl std::panic::RefUnwindSafe for RootDatabase {}
94
95#[salsa_macros::db]
96impl salsa::Database for RootDatabase {}
97
98impl Drop for RootDatabase {
99    fn drop(&mut self) {
100        unsafe { ManuallyDrop::drop(&mut self.storage) };
101    }
102}

104impl Clone for RootDatabase {
105    fn clone(&self) -> Self {
106        Self {
107            storage: self.storage.clone(),
108            files: self.files.clone(),
109            crates_map: self.crates_map.clone(),
110            nonce: Nonce::new(),
111        }
112    }
113}

115impl fmt::Debug for RootDatabase {

187    pub fn new(lru_capacity: Option<u16>) -> RootDatabase {
188        let mut db = RootDatabase {
189            storage: ManuallyDrop::new(salsa::Storage::default()),
190            files: Default::default(),
191            crates_map: Default::default(),
192            nonce: Nonce::new(),
193        };
194        // This needs to be here otherwise `CrateGraphBuilder` will panic.
195        db.set_all_crates(Arc::new(Box::new([])));
196        CrateGraphBuilder::default().set_in_db(&mut db);
197        db.set_proc_macros_with_durability(Default::default(), Durability::MEDIUM);
198        db.set_local_roots_with_durability(Default::default(), Durability::MEDIUM);
199        db.set_library_roots_with_durability(Default::default(), Durability::MEDIUM);

202        db

290impl From<hir::MacroKind> for SymbolKind {
291    fn from(it: hir::MacroKind) -> Self {
292        match it {
293            hir::MacroKind::Declarative | hir::MacroKind::DeclarativeBuiltIn => SymbolKind::Macro,
294            hir::MacroKind::ProcMacro => SymbolKind::ProcMacro,
295            hir::MacroKind::Derive | hir::MacroKind::DeriveBuiltIn => SymbolKind::Derive,
296            hir::MacroKind::Attr | hir::MacroKind::AttrBuiltIn => SymbolKind::Attribute,
297        }
298    }
299}
300
301impl From<hir::ModuleDef> for SymbolKind {
302    fn from(it: hir::ModuleDef) -> Self {
303        match it {
304            hir::ModuleDef::Const(..) => SymbolKind::Const,
305            hir::ModuleDef::Variant(..) => SymbolKind::Variant,
306            hir::ModuleDef::Function(..) => SymbolKind::Function,
307            hir::ModuleDef::Macro(mac) if mac.is_proc_macro() => SymbolKind::ProcMacro,
308            hir::ModuleDef::Macro(..) => SymbolKind::Macro,
309            hir::ModuleDef::Module(..) => SymbolKind::Module,
310            hir::ModuleDef::Static(..) => SymbolKind::Static,
311            hir::ModuleDef::Adt(hir::Adt::Struct(..)) => SymbolKind::Struct,
312            hir::ModuleDef::Adt(hir::Adt::Enum(..)) => SymbolKind::Enum,
313            hir::ModuleDef::Adt(hir::Adt::Union(..)) => SymbolKind::Union,
314            hir::ModuleDef::Trait(..) => SymbolKind::Trait,
315            hir::ModuleDef::TypeAlias(..) => SymbolKind::TypeAlias,
316            hir::ModuleDef::BuiltinType(..) => SymbolKind::TypeAlias,
317        }
318    }
319}

332pub struct Ranker<'a> {
333    pub kind: parser::SyntaxKind,
334    pub text: &'a str,
335    pub ident_kind: bool,
336}
337
338impl<'a> Ranker<'a> {
339    pub const MAX_RANK: usize = 0b1110;
340
341    pub fn from_token(token: &'a syntax::SyntaxToken) -> Self {
342        let kind = token.kind();
343        Ranker { kind, text: token.text(), ident_kind: kind.is_any_identifier() }
344    }
345
346    /// A utility function that ranks a token again a given kind and text, returning a number that
347    /// represents how close the token is to the given kind and text.
348    pub fn rank_token(&self, tok: &syntax::SyntaxToken) -> usize {
349        let tok_kind = tok.kind();
350
351        let exact_same_kind = tok_kind == self.kind;
352        let both_idents = exact_same_kind || (tok_kind.is_any_identifier() && self.ident_kind);
353        let same_text = tok.text() == self.text;
354        // anything that mapped into a token tree has likely no semantic information
355        let no_tt_parent =
356            tok.parent().is_some_and(|it| it.kind() != parser::SyntaxKind::TOKEN_TREE);
357        (both_idents as usize)
358            | ((exact_same_kind as usize) << 1)
359            | ((same_text as usize) << 2)
360            | ((no_tt_parent as usize) << 3)
361    }
362}
