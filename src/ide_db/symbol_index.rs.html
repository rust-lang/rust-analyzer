1//! This module handles fuzzy-searching of functions, structs and other symbols
2//! by name across the whole workspace and dependencies.
3//!
4//! It works by building an incrementally-updated text-search index of all
5//! symbols. The backbone of the index is the **awesome** `fst` crate by
6//! @BurntSushi.
7//!
8//! In a nutshell, you give a set of strings to `fst`, and it builds a
9//! finite state machine describing this set of strings. The strings which
10//! could fuzzy-match a pattern can also be described by a finite state machine.
11//! What is freaking cool is that you can now traverse both state machines in
12//! lock-step to enumerate the strings which are both in the input set and
13//! fuzz-match the query. Or, more formally, given two languages described by
14//! FSTs, one can build a product FST which describes the intersection of the
15//! languages.
16//!
17//! `fst` does not support cheap updating of the index, but it supports unioning
18//! of state machines. So, to account for changing source code, we build an FST
19//! for each library (which is assumed to never change) and an FST for each Rust
20//! file in the current workspace, and run a query against the union of all
21//! those FSTs.
22

99#[query_group::query_group]
100pub trait SymbolsDatabase: HirDatabase + SourceDatabase {
101    /// The symbol index for a given module. These modules should only be in source roots that
102    /// are inside local_roots.
103    fn module_symbols(&self, module: Module) -> Arc<SymbolIndex>;
104
105    /// The symbol index for a given source root within library_roots.
106    fn library_symbols(&self, source_root_id: SourceRootId) -> Arc<SymbolIndex>;
107
108    #[salsa::transparent]
109    /// The symbol indices of modules that make up a given crate.
110    fn crate_symbols(&self, krate: Crate) -> Box<[Arc<SymbolIndex>]>;
111
112    /// The set of "local" (that is, from the current workspace) roots.
113    /// Files in local roots are assumed to change frequently.
114    #[salsa::input]
115    fn local_roots(&self) -> Arc<FxHashSet<SourceRootId>>;
116
117    /// The set of roots for crates.io libraries.
118    /// Files in libraries are assumed to never change.
119    #[salsa::input]
120    fn library_roots(&self) -> Arc<FxHashSet<SourceRootId>>;
121}

123fn library_symbols(db: &dyn SymbolsDatabase, source_root_id: SourceRootId) -> Arc<SymbolIndex> {
124    let _p = tracing::info_span!("library_symbols").entered();
125
126    let mut symbol_collector = SymbolCollector::new(db);
127
128    db.source_root_crates(source_root_id)
129        .iter()
130        .flat_map(|&krate| Crate::from(krate).modules(db))
131        // we specifically avoid calling other SymbolsDatabase queries here, even though they do the same thing,
132        // as the index for a library is not going to really ever change, and we do not want to store each
133        // the module or crate indices for those in salsa unless we need to.
134        .for_each(|module| symbol_collector.collect(module));
135
136    Arc::new(SymbolIndex::new(symbol_collector.finish()))
137}
138
139fn module_symbols(db: &dyn SymbolsDatabase, module: Module) -> Arc<SymbolIndex> {
140    let _p = tracing::info_span!("module_symbols").entered();
141
142    Arc::new(SymbolIndex::new(SymbolCollector::new_module(db, module)))
143}
144
145pub fn crate_symbols(db: &dyn SymbolsDatabase, krate: Crate) -> Box<[Arc<SymbolIndex>]> {
146    let _p = tracing::info_span!("crate_symbols").entered();
147    krate.modules(db).into_iter().map(|module| db.module_symbols(module)).collect()
148}

150// Feature: Workspace Symbol
151//
152// Uses fuzzy-search to find types, modules and functions by name across your
153// project and dependencies. This is **the** most useful feature, which improves code
154// navigation tremendously. It mostly works on top of the built-in LSP
155// functionality, however `#` and `*` symbols can be used to narrow down the
156// search. Specifically,
157//
158// - `Foo` searches for `Foo` type in the current workspace
159// - `foo#` searches for `foo` function in the current workspace
160// - `Foo*` searches for `Foo` type among dependencies, including `stdlib`
161// - `foo#*` searches for `foo` function among dependencies
162//
163// That is, `#` switches from "types" to all symbols, `*` switches from the current
164// workspace to dependencies.
165//
166// Note that filtering does not currently work in VSCode due to the editor never
167// sending the special symbols to the language server. Instead, you can configure
168// the filtering via the `rust-analyzer.workspace.symbol.search.scope` and
169// `rust-analyzer.workspace.symbol.search.kind` settings. Symbols prefixed
170// with `__` are hidden from the search results unless configured otherwise.
171//
172// | Editor  | Shortcut |
173// |---------|-----------|
174// | VS Code | <kbd>Ctrl+T</kbd>
175pub fn world_symbols(db: &RootDatabase, query: Query) -> Vec<FileSymbol> {
176    let _p = tracing::info_span!("world_symbols", query = ?query.query).entered();
177
178    let indices: Vec<_> = if query.libs {
179        db.library_roots()
180            .par_iter()
181            .map_with(db.clone(), |snap, &root| snap.library_symbols(root))
182            .collect()
183    } else {
184        let mut crates = Vec::new();
185
186        for &root in db.local_roots().iter() {
187            crates.extend(db.source_root_crates(root).iter().copied())
188        }
189        let indices: Vec<_> = crates
190            .into_par_iter()
191            .map_with(db.clone(), |snap, krate| snap.crate_symbols(krate.into()))
192            .collect();
193        indices.iter().flat_map(|indices| indices.iter().cloned()).collect()
194    };
195
196    let mut res = vec![];
197    query.search::<()>(&indices, |f| {
198        res.push(f.clone());
199        ControlFlow::Continue(())
200    });
201    res
202}

230impl SymbolIndex {
231    fn new(mut symbols: Box<[FileSymbol]>) -> SymbolIndex {
232        fn cmp(lhs: &FileSymbol, rhs: &FileSymbol) -> Ordering {
233            let lhs_chars = lhs.name.as_str().chars().map(|c| c.to_ascii_lowercase());
234            let rhs_chars = rhs.name.as_str().chars().map(|c| c.to_ascii_lowercase());
235            lhs_chars.cmp(rhs_chars)
236        }
237
238        symbols.par_sort_by(cmp);
239
240        let mut builder = fst::MapBuilder::memory();
241
242        let mut last_batch_start = 0;
243
244        for idx in 0..symbols.len() {
245            if let Some(next_symbol) = symbols.get(idx + 1) {
246                if cmp(&symbols[last_batch_start], next_symbol) == Ordering::Equal {
247                    continue;
248                }
249            }
250
251            let start = last_batch_start;
252            let end = idx + 1;
253            last_batch_start = end;
254
255            let key = symbols[start].name.as_str().to_ascii_lowercase();
256            let value = SymbolIndex::range_to_map_value(start, end);
257
258            builder.insert(key, value).unwrap();
259        }
260
261        let map = builder
262            .into_inner()
263            .and_then(|mut buf| {
264                fst::Map::new({
265                    buf.shrink_to_fit();
266                    buf
267                })
268            })
269            .unwrap();
270        SymbolIndex { symbols, map }
271    }
272
273    pub fn len(&self) -> usize {
274        self.symbols.len()
275    }
276
277    pub fn memory_size(&self) -> usize {
278        self.map.as_fst().size() + self.symbols.len() * size_of::<FileSymbol>()
279    }
280
281    fn range_to_map_value(start: usize, end: usize) -> u64 {
282        debug_assert![start <= (u32::MAX as usize)];
283        debug_assert![end <= (u32::MAX as usize)];
284
285        ((start as u64) << 32) | end as u64
286    }
287
288    fn map_value_to_range(value: u64) -> (usize, usize) {
289        let end = value as u32 as usize;
290        let start = (value >> 32) as usize;
291        (start, end)
292    }
293}

295impl Query {
296    pub(crate) fn search<'sym, T>(
297        self,
298        indices: &'sym [Arc<SymbolIndex>],
299        cb: impl FnMut(&'sym FileSymbol) -> ControlFlow<T>,
300    ) -> Option<T> {
301        let _p = tracing::info_span!("symbol_index::Query::search").entered();
302        let mut op = fst::map::OpBuilder::new();
303        match self.mode {
304            SearchMode::Exact => {
305                let automaton = fst::automaton::Str::new(&self.lowercased);
306
307                for index in indices.iter() {
308                    op = op.add(index.map.search(&automaton));
309                }
310                self.search_maps(indices, op.union(), cb)
311            }
312            SearchMode::Fuzzy => {
313                let automaton = fst::automaton::Subsequence::new(&self.lowercased);
314
315                for index in indices.iter() {
316                    op = op.add(index.map.search(&automaton));
317                }
318                self.search_maps(indices, op.union(), cb)
319            }
320            SearchMode::Prefix => {
321                let automaton = fst::automaton::Str::new(&self.lowercased).starts_with();
322
323                for index in indices.iter() {
324                    op = op.add(index.map.search(&automaton));
325                }
326                self.search_maps(indices, op.union(), cb)
327            }
328        }
329    }
330
331    fn search_maps<'sym, T>(
332        &self,
333        indices: &'sym [Arc<SymbolIndex>],
334        mut stream: fst::map::Union<'_>,
335        mut cb: impl FnMut(&'sym FileSymbol) -> ControlFlow<T>,
336    ) -> Option<T> {
337        let ignore_underscore_prefixed = !self.query.starts_with("__");
338        while let Some((_, indexed_values)) = stream.next() {
339            for &IndexedValue { index, value } in indexed_values {
340                let symbol_index = &indices[index];
341                let (start, end) = SymbolIndex::map_value_to_range(value);
342
343                for symbol in &symbol_index.symbols[start..end] {
344                    let non_type_for_type_only_query = self.only_types
345                        && !matches!(
346                            symbol.def,
347                            hir::ModuleDef::Adt(..)
348                                | hir::ModuleDef::TypeAlias(..)
349                                | hir::ModuleDef::BuiltinType(..)
350                                | hir::ModuleDef::TraitAlias(..)
351                                | hir::ModuleDef::Trait(..)
352                        );
353                    if non_type_for_type_only_query || !self.matches_assoc_mode(symbol.is_assoc) {
354                        continue;
355                    }
356                    // Hide symbols that start with `__` unless the query starts with `__`
357                    let symbol_name = symbol.name.as_str();
358                    if ignore_underscore_prefixed && symbol_name.starts_with("__") {
359                        continue;
360                    }
361                    if self.mode.check(&self.query, self.case_sensitive, symbol_name) {
362                        if let Some(b) = cb(symbol).break_value() {
363                            return Some(b);
364                        }
365                    }
366                }
367            }
368        }
369        None
370    }
371
372    fn matches_assoc_mode(&self, is_trait_assoc_item: bool) -> bool {
373        !matches!(
374            (is_trait_assoc_item, self.assoc_mode),
375            (true, AssocSearchMode::Exclude) | (false, AssocSearchMode::AssocItemsOnly)
376        )
377    }
378}
