7decl_derive!(
8    [TypeFoldable, attributes(type_foldable)] =>
9    /// Derives `TypeFoldable` for the annotated `struct` or `enum` (`union` is not supported).
10    ///
11    /// The fold will produce a value of the same struct or enum variant as the input, with
12    /// each field respectively folded using the `TypeFoldable` implementation for its type.
13    /// However, if a field of a struct or an enum variant is annotated with
14    /// `#[type_foldable(identity)]` then that field will retain its incumbent value (and its
15    /// type is not required to implement `TypeFoldable`).
16    type_foldable_derive
17);
18decl_derive!(
19    [TypeVisitable, attributes(type_visitable)] =>
20    /// Derives `TypeVisitable` for the annotated `struct` or `enum` (`union` is not supported).
21    ///
22    /// Each field of the struct or enum variant will be visited in definition order, using the
23    /// `TypeVisitable` implementation for its type. However, if a field of a struct or an enum
24    /// variant is annotated with `#[type_visitable(ignore)]` then that field will not be
25    /// visited (and its type is not required to implement `TypeVisitable`).
26    type_visitable_derive
27);
28decl_derive!(
29    [GenericTypeVisitable] => generic_type_visitable_derive
30);

32fn type_visitable_derive(mut s: synstructure::Structure<'_>) -> proc_macro2::TokenStream {
33    if let syn::Data::Union(_) = s.ast().data {
34        panic!("cannot derive on union")
35    }
36
37    // ignore fields with #[type_visitable(ignore)]
38    s.filter(|bi| {
39        let mut ignored = false;
40
41        bi.ast().attrs.iter().for_each(|attr| {
42            if !attr.path().is_ident("type_visitable") {
43                return;
44            }
45            let _ = attr.parse_nested_meta(|nested| {
46                if nested.path.is_ident("ignore") {
47                    ignored = true;
48                }
49                Ok(())
50            });
51        });
52
53        !ignored
54    });
55
56    if !s.ast().generics.lifetimes().any(|lt| lt.lifetime.ident == "db") {
57        s.add_impl_generic(parse_quote! { 'db });
58    }
59
60    s.add_bounds(synstructure::AddBounds::Generics);
61    let body_visit = s.each(|bind| {
62        quote! {
63            match ::rustc_type_ir::VisitorResult::branch(
64                ::rustc_type_ir::TypeVisitable::visit_with(#bind, __visitor)
65            ) {
66                ::core::ops::ControlFlow::Continue(()) => {},
67                ::core::ops::ControlFlow::Break(r) => {
68                    return ::rustc_type_ir::VisitorResult::from_residual(r);
69                },
70            }
71        }
72    });
73    s.bind_with(|_| synstructure::BindStyle::Move);
74
75    s.bound_impl(
76        quote!(::rustc_type_ir::TypeVisitable<::hir_ty::next_solver::DbInterner<'db>>),
77        quote! {
78            fn visit_with<__V: ::rustc_type_ir::TypeVisitor<::hir_ty::next_solver::DbInterner<'db>>>(
79                &self,
80                __visitor: &mut __V
81            ) -> __V::Result {
82                match *self { #body_visit }
83                <__V::Result as ::rustc_type_ir::VisitorResult>::output()
84            }
85        },
86    )
87}

89fn type_foldable_derive(mut s: synstructure::Structure<'_>) -> proc_macro2::TokenStream {
90    if let syn::Data::Union(_) = s.ast().data {
91        panic!("cannot derive on union")
92    }
93
94    if !s.ast().generics.lifetimes().any(|lt| lt.lifetime.ident == "db") {
95        s.add_impl_generic(parse_quote! { 'db });
96    }
97
98    s.add_bounds(synstructure::AddBounds::Generics);
99    s.bind_with(|_| synstructure::BindStyle::Move);
100    let try_body_fold = s.each_variant(|vi| {
101        let bindings = vi.bindings();
102        vi.construct(|_, index| {
103            let bind = &bindings[index];
104
105            // retain value of fields with #[type_foldable(identity)]
106            if has_ignore_attr(&bind.ast().attrs, "type_foldable", "identity") {
107                bind.to_token_stream()
108            } else {
109                quote! {
110                    ::rustc_type_ir::TypeFoldable::try_fold_with(#bind, __folder)?
111                }
112            }
113        })
114    });
115
116    let body_fold = s.each_variant(|vi| {
117        let bindings = vi.bindings();
118        vi.construct(|_, index| {
119            let bind = &bindings[index];
120
121            // retain value of fields with #[type_foldable(identity)]
122            if has_ignore_attr(&bind.ast().attrs, "type_foldable", "identity") {
123                bind.to_token_stream()
124            } else {
125                quote! {
126                    ::rustc_type_ir::TypeFoldable::fold_with(#bind, __folder)
127                }
128            }
129        })
130    });
131
132    s.bound_impl(
133        quote!(::rustc_type_ir::TypeFoldable<::hir_ty::next_solver::DbInterner<'db>>),
134        quote! {
135            fn try_fold_with<__F: ::rustc_type_ir::FallibleTypeFolder<::hir_ty::next_solver::DbInterner<'db>>>(
136                self,
137                __folder: &mut __F
138            ) -> Result<Self, __F::Error> {
139                Ok(match self { #try_body_fold })
140            }
141
142            fn fold_with<__F: ::rustc_type_ir::TypeFolder<::hir_ty::next_solver::DbInterner<'db>>>(
143                self,
144                __folder: &mut __F
145            ) -> Self {
146                match self { #body_fold }
147            }
148        },
149    )
150}

152fn has_ignore_attr(attrs: &[syn::Attribute], name: &'static str, meta: &'static str) -> bool {
153    let mut ignored = false;
154    attrs.iter().for_each(|attr| {
155        if !attr.path().is_ident(name) {
156            return;
157        }
158        let _ = attr.parse_nested_meta(|nested| {
159            if nested.path.is_ident(meta) {
160                ignored = true;
161            }
162            Ok(())
163        });
164    });
165
166    ignored
167}

169fn generic_type_visitable_derive(mut s: synstructure::Structure<'_>) -> proc_macro2::TokenStream {
170    if let syn::Data::Union(_) = s.ast().data {
171        panic!("cannot derive on union")
172    }
173
174    s.add_bounds(synstructure::AddBounds::Fields);
175    s.bind_with(|_| synstructure::BindStyle::Move);
176    s.add_impl_generic(parse_quote!(__V: hir_ty::next_solver::interner::WorldExposer));
177    let body_visit = s.each(|bind| {
178        quote! {
179            ::rustc_type_ir::GenericTypeVisitable::<__V>::generic_visit_with(#bind, __visitor);
180        }
181    });
182
183    s.bound_impl(
184        quote!(::rustc_type_ir::GenericTypeVisitable<__V>),
185        quote! {
186            fn generic_visit_with(
187                &self,
188                __visitor: &mut __V
189            ) {
190                match self { #body_visit }
191            }
192        },
193    )
194}

200fn upmap_from_ra_fixture(mut s: synstructure::Structure<'_>) -> proc_macro2::TokenStream {
201    if let syn::Data::Union(_) = s.ast().data {
202        panic!("cannot derive on union")
203    }
204
205    s.add_bounds(synstructure::AddBounds::Generics);
206    s.bind_with(|_| synstructure::BindStyle::Move);
207    let body = s.each_variant(|vi| {
208        let bindings = vi.bindings();
209        vi.construct(|_, index| {
210            let bind = &bindings[index];
211
212            quote! {
213                ::ide_db::ra_fixture::UpmapFromRaFixture::upmap_from_ra_fixture(
214                    #bind, __analysis, __virtual_file_id, __real_file_id,
215                )?
216            }
217        })
218    });
219
220    s.bound_impl(
221        quote!(::ide_db::ra_fixture::UpmapFromRaFixture),
222        quote! {
223            fn upmap_from_ra_fixture(
224                self,
225                __analysis: &::ide_db::ra_fixture::RaFixtureAnalysis,
226                __virtual_file_id: ::ide_db::ra_fixture::FileId,
227                __real_file_id: ::ide_db::ra_fixture::FileId,
228            ) -> Result<Self, ()> {
229                Ok(match self { #body })
