18macro_rules! from_id {
19    ($(($id:path, $ty:path)),* $(,)?) => {$(
20        impl From<$id> for $ty {
21            fn from(id: $id) -> $ty {
22                $ty { id }
23            }
24        }
25        impl From<$ty> for $id {
26            fn from(ty: $ty) -> $id {
27                ty.id
28            }
29        }
30    )*}
31}

123impl TryFrom<ModuleDef> for ModuleDefId {
124    type Error = ();
125    fn try_from(id: ModuleDef) -> Result<Self, Self::Error> {
126        Ok(match id {
127            ModuleDef::Module(it) => ModuleDefId::ModuleId(it.into()),
128            ModuleDef::Function(it) => match it.id {
129                AnyFunctionId::FunctionId(it) => it.into(),
130                AnyFunctionId::BuiltinDeriveImplMethod { .. } => return Err(()),
131            },
132            ModuleDef::Adt(it) => ModuleDefId::AdtId(it.into()),
133            ModuleDef::Variant(it) => ModuleDefId::EnumVariantId(it.into()),
134            ModuleDef::Const(it) => ModuleDefId::ConstId(it.into()),
135            ModuleDef::Static(it) => ModuleDefId::StaticId(it.into()),
136            ModuleDef::Trait(it) => ModuleDefId::TraitId(it.into()),
137            ModuleDef::TypeAlias(it) => ModuleDefId::TypeAliasId(it.into()),
138            ModuleDef::BuiltinType(it) => ModuleDefId::BuiltinType(it.into()),
139            ModuleDef::Macro(it) => ModuleDefId::MacroId(it.into()),
140        })
141    }
142}

280impl TryFrom<ItemInNs> for hir_def::item_scope::ItemInNs {
281    type Error = ();
282    fn try_from(it: ItemInNs) -> Result<Self, Self::Error> {
283        Ok(match it {
284            ItemInNs::Types(it) => Self::Types(it.try_into()?),
285            ItemInNs::Values(it) => Self::Values(it.try_into()?),
286            ItemInNs::Macros(it) => Self::Macros(it.into()),
