68/// `SourceAnalyzer` is a convenience wrapper which exposes HIR API in terms of
69/// original source files. It should not be used inside the HIR itself.

71pub(crate) struct SourceAnalyzer<'db> {

118    pub(crate) fn new_for_body_(
119        db: &'db dyn HirDatabase,
120        def: DefWithBodyId,
121        node @ InFile { file_id, .. }: InFile<&SyntaxNode>,
122        offset: Option<TextSize>,
123        infer: Option<&'db InferenceResult>,
124    ) -> SourceAnalyzer<'db> {
125        let (body, source_map) = db.body_with_source_map(def);
126        let scopes = db.expr_scopes(def);
127        let scope = match offset {
128            None => scope_for(db, &scopes, &source_map, node),
129            Some(offset) => {
130                debug_assert!(
131                    node.text_range().contains_inclusive(offset),
132                    "{:?} not in {:?}",
133                    offset,
134                    node.text_range()
135                );
136                scope_for_offset(db, &scopes, &source_map, node.file_id, offset)
137            }
138        };
139        let resolver = resolver_for_scope(db, def, scope);
140        SourceAnalyzer {
141            resolver,
142            body_or_sig: Some(BodyOrSig::Body { def, body, source_map, infer }),
143            file_id,
144        }
145    }

270    pub(crate) fn type_of_type(
271        &self,
272        db: &'db dyn HirDatabase,
273        ty: &ast::Type,
274    ) -> Option<Type<'db>> {
275        let interner = DbInterner::new_no_crate(db);
276
277        let type_ref = self.type_id(ty)?;
278
279        let mut ty = TyLoweringContext::new(
280            db,
281            &self.resolver,
282            self.store()?,
283            self.resolver.generic_def()?,
284            // FIXME: Is this correct here? Anyway that should impact mostly diagnostics, which we don't emit here
285            // (this can impact the lifetimes generated, e.g. in `const` they won't be `'static`, but this seems like a
286            // small problem).
287            LifetimeElisionKind::Infer,
288        )
289        .lower_ty(type_ref);
290
291        // Try and substitute unknown types using InferenceResult
292        if let Some(infer) = self.infer()
293            && let Some(store) = self.store()
294        {
295            let mut inferred_types = vec![];
296            TypeRef::walk(type_ref, store, &mut |type_ref_id, type_ref| {
297                if matches!(type_ref, TypeRef::Placeholder) {
298                    inferred_types.push(infer.type_of_type_placeholder(type_ref_id));
299                }
300            });
301            let mut inferred_types = inferred_types.into_iter();
302
303            let substituted_ty = hir_ty::next_solver::fold::fold_tys(interner, ty, |ty| {
304                if ty.is_ty_error() { inferred_types.next().flatten().unwrap_or(ty) } else { ty }
305            });
306
307            // Only used the result if the placeholder and unknown type counts matched
308            let success =
309                inferred_types.next().is_none() && !substituted_ty.references_non_lt_error();
310            if success {
311                ty = substituted_ty;
312            }
313        }
314
315        Some(Type::new_with_resolver(db, &self.resolver, ty))
316    }

318    pub(crate) fn type_of_expr(
319        &self,
320        db: &'db dyn HirDatabase,
321        expr: &ast::Expr,
322    ) -> Option<(Type<'db>, Option<Type<'db>>)> {
323        let expr_id = self.expr_id(expr.clone())?;
324        let infer = self.infer()?;
325        let coerced = expr_id
326            .as_expr()
327            .and_then(|expr_id| infer.expr_adjustment(expr_id))
328            .and_then(|adjusts| adjusts.last().map(|adjust| adjust.target.as_ref()));
329        let ty = infer.expr_or_pat_ty(expr_id);
330        let mk_ty = |ty: Ty<'db>| Type::new_with_resolver(db, &self.resolver, ty);
331        Some((mk_ty(ty), coerced.map(mk_ty)))
332    }

334    pub(crate) fn type_of_pat(
335        &self,
336        db: &'db dyn HirDatabase,
337        pat: &ast::Pat,
338    ) -> Option<(Type<'db>, Option<Type<'db>>)> {
339        let expr_or_pat_id = self.pat_id(pat)?;
340        let infer = self.infer()?;
341        let coerced = match expr_or_pat_id {
342            ExprOrPatId::ExprId(idx) => infer
343                .expr_adjustment(idx)
344                .and_then(|adjusts| adjusts.last())
345                .map(|adjust| adjust.target.as_ref()),
346            ExprOrPatId::PatId(idx) => infer
347                .pat_adjustment(idx)
348                .and_then(|adjusts| adjusts.last())
349                .map(|adjust| adjust.as_ref()),
350        };
351
352        let ty = infer.expr_or_pat_ty(expr_or_pat_id);
353        let mk_ty = |ty: Ty<'db>| Type::new_with_resolver(db, &self.resolver, ty);
354        Some((mk_ty(ty), coerced.map(mk_ty)))
355    }

438    pub(crate) fn resolve_method_call_fallback(
439        &self,
440        db: &'db dyn HirDatabase,
441        call: &ast::MethodCallExpr,
442    ) -> Option<(Either<Function, Field>, Option<GenericSubstitution<'db>>)> {
443        let expr_id = self.expr_id(call.clone().into())?.as_expr()?;
444        let inference_result = self.infer()?;
445        match inference_result.method_resolution(expr_id) {
446            Some((f_in_trait, substs)) => {
447                let (fn_, subst) =
448                    self.resolve_impl_method_or_trait_def_with_subst(db, f_in_trait, substs);
449                Some((
450                    Either::Left(fn_),
451                    GenericSubstitution::new_from_fn(fn_, subst, self.trait_environment(db)),
452                ))
453            }
454            None => {
455                inference_result.field_resolution(expr_id).and_then(Either::left).map(|field| {
456                    (Either::Right(field.into()), self.field_subst(expr_id, inference_result, db))
457                })
458            }
459        }
460    }

496    pub(crate) fn resolve_field_fallback(
497        &self,
498        db: &'db dyn HirDatabase,
499        field: &ast::FieldExpr,
500    ) -> Option<(Either<Either<Field, TupleField>, Function>, Option<GenericSubstitution<'db>>)>
501    {
502        let (def, ..) = self.body_()?;
503        let expr_id = self.expr_id(field.clone().into())?.as_expr()?;
504        let inference_result = self.infer()?;
505        match inference_result.field_resolution(expr_id) {
506            Some(field) => match field {
507                Either::Left(field) => Some((
508                    Either::Left(Either::Left(field.into())),
509                    self.field_subst(expr_id, inference_result, db),
510                )),
511                Either::Right(field) => Some((
512                    Either::Left(Either::Right(TupleField {
513                        owner: def,
514                        tuple: field.tuple,
515                        index: field.index,
516                    })),
517                    None,
518                )),
519            },
520            None => inference_result.method_resolution(expr_id).map(|(f, substs)| {
521                let (f, subst) = self.resolve_impl_method_or_trait_def_with_subst(db, f, substs);
522                (
523                    Either::Right(f),
524                    GenericSubstitution::new_from_fn(f, subst, self.trait_environment(db)),
525                )
526            }),
527        }

556    fn resolve_range_struct(
557        &self,
558        db: &'db dyn HirDatabase,
559        op_kind: RangeOp,
560        has_start: bool,
561        has_end: bool,
562    ) -> Option<StructId> {
563        let has_new_range =
564            self.resolver.top_level_def_map().is_unstable_feature_enabled(&sym::new_range);
565        let lang_items = self.lang_items(db);
566        match (op_kind, has_start, has_end) {
567            (RangeOp::Exclusive, false, false) => lang_items.RangeFull,
568            (RangeOp::Exclusive, false, true) => lang_items.RangeTo,
569            (RangeOp::Exclusive, true, false) => {
570                if has_new_range {
571                    lang_items.RangeFromCopy
572                } else {
573                    lang_items.RangeFrom
574                }
575            }
576            (RangeOp::Exclusive, true, true) => {
577                if has_new_range {
578                    lang_items.RangeCopy
579                } else {
580                    lang_items.Range
581                }
582            }
583            (RangeOp::Inclusive, false, true) => {
584                if has_new_range {
585                    lang_items.RangeToInclusiveCopy
586                } else {
587                    lang_items.RangeToInclusive
588                }
589            }
590            (RangeOp::Inclusive, true, true) => {
591                if has_new_range {
592                    lang_items.RangeInclusiveCopy
593                } else {
594                    lang_items.RangeInclusiveStruct
595                }
596            }
597            // [E0586] inclusive ranges must be bounded at the end
598            (RangeOp::Inclusive, false, false) => None,
599            (RangeOp::Inclusive, true, false) => None,
600        }
601    }

606        await_expr: &ast::AwaitExpr,
607    ) -> Option<Function> {
608        let mut ty = self.ty_of_expr(await_expr.expr()?)?;
609
610        let into_future_trait = self
611            .resolver
612            .resolve_known_trait(db, &path![core::future::IntoFuture])
613            .map(Trait::from);
614
615        if let Some(into_future_trait) = into_future_trait {
616            let type_ = Type::new_with_resolver(db, &self.resolver, ty);
617            if type_.impls_trait(db, into_future_trait, &[]) {
618                let items = into_future_trait.items(db);
619                let into_future_type = items.into_iter().find_map(|item| match item {
620                    AssocItem::TypeAlias(alias)
621                        if alias.name(db) == Name::new_symbol_root(sym::IntoFuture) =>
622                    {
623                        Some(alias)
624                    }
625                    _ => None,
626                })?;
627                let future_trait = type_.normalize_trait_assoc_type(db, &[], into_future_type)?;
628                ty = future_trait.ty;
629            }
630        }
631
632        let poll_fn = self.lang_items(db).FuturePoll?;
633        // HACK: subst for `poll()` coincides with that for `Future` because `poll()` itself
634        // doesn't have any generic parameters, so we skip building another subst for `poll()`.
635        let substs = GenericArgs::new_from_slice(&[ty.into()]);
636        Some(self.resolve_impl_method_or_trait_def(db, poll_fn, substs))
637    }

639    pub(crate) fn resolve_prefix_expr(
640        &self,
641        db: &'db dyn HirDatabase,
642        prefix_expr: &ast::PrefixExpr,
643    ) -> Option<Function> {
644        let (_op_trait, op_fn) = match prefix_expr.op_kind()? {
645            ast::UnaryOp::Deref => {
646                // This can be either `Deref::deref` or `DerefMut::deref_mut`.
647                // Since deref kind is inferenced and stored in `InferenceResult.method_resolution`,
648                // use that result to find out which one it is.
649                let (deref_trait, deref) = self.lang_trait_fn(
650                    db,
651                    self.lang_items(db).Deref,
652                    &Name::new_symbol_root(sym::deref),
653                )?;
654                self.infer()
655                    .and_then(|infer| {
656                        let expr = self.expr_id(prefix_expr.clone().into())?.as_expr()?;
657                        let (func, _) = infer.method_resolution(expr)?;
658                        let (deref_mut_trait, deref_mut) = self.lang_trait_fn(
659                            db,
660                            self.lang_items(db).DerefMut,
661                            &Name::new_symbol_root(sym::deref_mut),
662                        )?;
663                        if func == deref_mut { Some((deref_mut_trait, deref_mut)) } else { None }
664                    })
665                    .unwrap_or((deref_trait, deref))
666            }
667            ast::UnaryOp::Not => {
668                self.lang_trait_fn(db, self.lang_items(db).Not, &Name::new_symbol_root(sym::not))?
669            }
670            ast::UnaryOp::Neg => {
671                self.lang_trait_fn(db, self.lang_items(db).Neg, &Name::new_symbol_root(sym::neg))?
672            }
673        };
674
675        let ty = self.ty_of_expr(prefix_expr.expr()?)?;
676
677        // HACK: subst for all methods coincides with that for their trait because the methods
678        // don't have any generic parameters, so we skip building another subst for the methods.
679        let substs = GenericArgs::new_from_slice(&[ty.into()]);
680
681        Some(self.resolve_impl_method_or_trait_def(db, op_fn, substs))
682    }

684    pub(crate) fn resolve_index_expr(
685        &self,
686        db: &'db dyn HirDatabase,
687        index_expr: &ast::IndexExpr,
688    ) -> Option<Function> {
689        let base_ty = self.ty_of_expr(index_expr.base()?)?;
690        let index_ty = self.ty_of_expr(index_expr.index()?)?;
691
692        let (_index_trait, index_fn) =
693            self.lang_trait_fn(db, self.lang_items(db).Index, &Name::new_symbol_root(sym::index))?;
694        let op_fn = self
695            .infer()
696            .and_then(|infer| {
697                let expr = self.expr_id(index_expr.clone().into())?.as_expr()?;
698                let (func, _) = infer.method_resolution(expr)?;
699                let (_index_mut_trait, index_mut_fn) = self.lang_trait_fn(
700                    db,
701                    self.lang_items(db).IndexMut,
702                    &Name::new_symbol_root(sym::index_mut),
703                )?;
704                if func == index_mut_fn { Some(index_mut_fn) } else { None }
705            })
706            .unwrap_or(index_fn);
707        // HACK: subst for all methods coincides with that for their trait because the methods
708        // don't have any generic parameters, so we skip building another subst for the methods.
709        let substs = GenericArgs::new_from_slice(&[base_ty.into(), index_ty.into()]);
710        Some(self.resolve_impl_method_or_trait_def(db, op_fn, substs))
711    }

748    pub(crate) fn resolve_record_field(
749        &self,
750        db: &'db dyn HirDatabase,
751        field: &ast::RecordExprField,
752    ) -> Option<(Field, Option<Local>, Type<'db>, GenericSubstitution<'db>)> {
753        let record_expr = ast::RecordExpr::cast(field.syntax().parent().and_then(|p| p.parent())?)?;
754        let expr = ast::Expr::from(record_expr);
755        let expr_id = self.store_sm()?.node_expr(InFile::new(self.file_id, &expr))?;
756        let interner = DbInterner::new_no_crate(db);
757
758        let ast_name = field.field_name()?;
759        let local_name = ast_name.as_name();
760        let local = if field.name_ref().is_some() {
761            None
762        } else {
763            // Shorthand syntax, resolve to the local
764            let path = Path::from_known_path_with_no_generic(ModPath::from_segments(
765                PathKind::Plain,
766                once(local_name.clone()),
767            ));
768            match self.resolver.resolve_path_in_value_ns_fully(
769                db,
770                &path,
771                name_hygiene(db, InFile::new(self.file_id, ast_name.syntax())),
772            ) {
773                Some(ValueNs::LocalBinding(binding_id)) => {
774                    Some(Local { binding_id, parent: self.resolver.body_owner()? })
775                }
776                _ => None,
777            }
778        };
779        let (adt, subst) = self.infer()?.type_of_expr_or_pat(expr_id)?.as_adt()?;
780        let variant = self.infer()?.variant_resolution_for_expr_or_pat(expr_id)?;
781        let variant_data = variant.fields(db);
782        let field = FieldId { parent: variant, local_id: variant_data.field(&local_name)? };
783        let field_ty =
784            (*db.field_types(variant).get(field.local_id)?).get().instantiate(interner, subst);
785        Some((
786            field.into(),
787            local,
788            Type::new_with_resolver(db, &self.resolver, field_ty),
789            GenericSubstitution::new(adt.into(), subst, self.trait_environment(db)),
790        ))
791    }

850    pub(crate) fn resolve_offset_of_field(
851        &self,
852        db: &'db dyn HirDatabase,
853        name_ref: &ast::NameRef,
854    ) -> Option<(Either<crate::Variant, crate::Field>, GenericSubstitution<'db>)> {
855        let offset_of_expr = ast::OffsetOfExpr::cast(name_ref.syntax().parent()?)?;
856        let container = offset_of_expr.ty()?;
857        let container = self.type_of_type(db, &container)?;
858
859        let trait_env = container.env;
860
861        let interner = DbInterner::new_with(db, trait_env.krate);
862        let infcx = interner.infer_ctxt().build(TypingMode::PostAnalysis);
863
864        let mut container = Either::Right(container.ty);
865        for field_name in offset_of_expr.fields() {
866            if let Either::Right(container) = &mut container {
867                *container = structurally_normalize_ty(&infcx, *container, trait_env.param_env);
868            }
869            let handle_variants = |variant: VariantId,
870                                   subst: GenericArgs<'db>,
871                                   container: &mut _| {
872                let fields = variant.fields(db);
873                let field = fields.field(&field_name.as_name())?;
874                let field_types = db.field_types(variant);
875                *container = Either::Right(field_types[field].get().instantiate(interner, subst));
876                let generic_def = match variant {
877                    VariantId::EnumVariantId(it) => it.loc(db).parent.into(),
878                    VariantId::StructId(it) => it.into(),
879                    VariantId::UnionId(it) => it.into(),
880                };
881                Some((
882                    Either::Right(Field { parent: variant.into(), id: field }),
883                    generic_def,
884                    subst,
885                ))
886            };
887            let temp_ty = Ty::new_error(interner, ErrorGuaranteed);
888            let (field_def, generic_def, subst) =
889                match std::mem::replace(&mut container, Either::Right(temp_ty)) {
890                    Either::Left((variant_id, subst)) => {
891                        handle_variants(VariantId::from(variant_id), subst, &mut container)?
892                    }
893                    Either::Right(container_ty) => match container_ty.kind() {
894                        TyKind::Adt(adt_def, subst) => match adt_def.def_id().0 {
895                            AdtId::StructId(id) => {
896                                handle_variants(id.into(), subst, &mut container)?
897                            }
898                            AdtId::UnionId(id) => {
899                                handle_variants(id.into(), subst, &mut container)?
900                            }
901                            AdtId::EnumId(id) => {
902                                let variants = id.enum_variants(db);
903                                let variant = variants.variant(&field_name.as_name())?;
904                                container = Either::Left((variant, subst));
905                                (Either::Left(Variant { id: variant }), id.into(), subst)
906                            }
907                        },
908                        _ => return None,
909                    },
910                };
911
912            if field_name.syntax().text_range() == name_ref.syntax().text_range() {
913                return Some((field_def, GenericSubstitution::new(generic_def, subst, trait_env)));
914            }
915        }
916        never!("the `NameRef` is a child of the `OffsetOfExpr`, we should've visited it");
917        None
918    }

920    pub(crate) fn resolve_path(
921        &self,
922        db: &'db dyn HirDatabase,
923        path: &ast::Path,
924    ) -> Option<(PathResolution, Option<GenericSubstitution<'db>>)> {
925        let parent = path.syntax().parent();
926        let parent = || parent.clone();
927
928        let mut prefer_value_ns = false;
929        let resolved = (|| {
930            let infer = self.infer()?;
931            if let Some(path_expr) = parent().and_then(ast::PathExpr::cast) {
932                let expr_id = self.expr_id(path_expr.into())?;
933                if let Some((assoc, subs)) = infer.assoc_resolutions_for_expr_or_pat(expr_id) {
934                    let (assoc, subst) = match assoc {
935                        CandidateId::FunctionId(f_in_trait) => {
936                            match infer.type_of_expr_or_pat(expr_id) {
937                                None => {
938                                    let subst = GenericSubstitution::new(
939                                        f_in_trait.into(),
940                                        subs,
941                                        self.trait_environment(db),
942                                    );
943                                    (AssocItem::Function(f_in_trait.into()), Some(subst))
944                                }
945                                Some(func_ty) => {
946                                    if let TyKind::FnDef(_fn_def, subs) = func_ty.kind() {
947                                        let (fn_, subst) = self
948                                            .resolve_impl_method_or_trait_def_with_subst(
949                                                db, f_in_trait, subs,
950                                            );
951                                        let subst = GenericSubstitution::new_from_fn(
952                                            fn_,
953                                            subst,
954                                            self.trait_environment(db),
955                                        );
956                                        (AssocItem::Function(fn_), subst)
957                                    } else {
958                                        let subst = GenericSubstitution::new(
959                                            f_in_trait.into(),
960                                            subs,
961                                            self.trait_environment(db),
962                                        );
963                                        (AssocItem::Function(f_in_trait.into()), Some(subst))
964                                    }
965                                }
966                            }
967                        }
968                        CandidateId::ConstId(const_id) => {
969                            let (konst, subst) =
970                                self.resolve_impl_const_or_trait_def_with_subst(db, const_id, subs);
971                            let subst = GenericSubstitution::new(
972                                konst.into(),
973                                subst,
974                                self.trait_environment(db),
975                            );
976                            (AssocItem::Const(konst.into()), Some(subst))
977                        }
978                    };
979
980                    return Some((PathResolution::Def(assoc.into()), subst));
981                }
982                if let Some(VariantId::EnumVariantId(variant)) =
983                    infer.variant_resolution_for_expr_or_pat(expr_id)
984                {
985                    return Some((PathResolution::Def(ModuleDef::Variant(variant.into())), None));
986                }
987                prefer_value_ns = true;
988            } else if let Some(path_pat) = parent().and_then(ast::PathPat::cast) {
989                let expr_or_pat_id = self.pat_id(&path_pat.into())?;
990                if let Some((assoc, subs)) = infer.assoc_resolutions_for_expr_or_pat(expr_or_pat_id)
991                {
992                    let (assoc, subst) = match assoc {
993                        CandidateId::ConstId(const_id) => {
994                            let (konst, subst) =
995                                self.resolve_impl_const_or_trait_def_with_subst(db, const_id, subs);
996                            let subst = GenericSubstitution::new(
997                                konst.into(),
998                                subst,
999                                self.trait_environment(db),
1000                            );
1001                            (AssocItemId::from(konst), subst)
1002                        }
1003                        CandidateId::FunctionId(function_id) => (
1004                            function_id.into(),
1005                            GenericSubstitution::new(
1006                                function_id.into(),
1007                                subs,
1008                                self.trait_environment(db),
1009                            ),
1010                        ),
1011                    };
1012                    return Some((PathResolution::Def(AssocItem::from(assoc).into()), Some(subst)));
1013                }
1014                if let Some(VariantId::EnumVariantId(variant)) =
1015                    infer.variant_resolution_for_expr_or_pat(expr_or_pat_id)
1016                {
1017                    return Some((PathResolution::Def(ModuleDef::Variant(variant.into())), None));
1018                }
1019            } else if let Some(rec_lit) = parent().and_then(ast::RecordExpr::cast) {
1020                let expr_id = self.expr_id(rec_lit.into())?;
1021                if let Some(VariantId::EnumVariantId(variant)) =
1022                    infer.variant_resolution_for_expr_or_pat(expr_id)
1023                {
1024                    return Some((PathResolution::Def(ModuleDef::Variant(variant.into())), None));
1025                }
1026            } else {
1027                let record_pat = parent().and_then(ast::RecordPat::cast).map(ast::Pat::from);
1028                let tuple_struct_pat =
1029                    || parent().and_then(ast::TupleStructPat::cast).map(ast::Pat::from);
1030                if let Some(pat) = record_pat.or_else(tuple_struct_pat) {
1031                    let pat_id = self.pat_id(&pat)?;
1032                    let variant_res_for_pat = infer.variant_resolution_for_pat(pat_id.as_pat()?);
1033                    if let Some(VariantId::EnumVariantId(variant)) = variant_res_for_pat {
1034                        return Some((
1035                            PathResolution::Def(ModuleDef::Variant(variant.into())),
1036                            None,
1037                        ));
1038                    }
1039                }
1040            }
1041            None
1042        })();
1043        if resolved.is_some() {
1044            return resolved;
1045        }
1046
1047        // FIXME: collectiong here shouldnt be necessary?
1048        let mut collector = ExprCollector::new(db, self.resolver.module(), self.file_id);
1049        let hir_path =
1050            collector.lower_path(path.clone(), &mut ExprCollector::impl_trait_error_allocator)?;
1051        let parent_hir_path = path
1052            .parent_path()
1053            .and_then(|p| collector.lower_path(p, &mut ExprCollector::impl_trait_error_allocator));
1054        let (store, _) = collector.store.finish();
1055
1056        // Case where path is a qualifier of a use tree, e.g. foo::bar::{Baz, Qux} where we are
1057        // trying to resolve foo::bar.
1058        if let Some(use_tree) = parent().and_then(ast::UseTree::cast)
1059            && use_tree.coloncolon_token().is_some()
1060        {
1061            return resolve_hir_path_qualifier(db, &self.resolver, &hir_path, &store)
1062                .map(|it| (it, None));
1063        }
1064
1065        let meta_path = path
1066            .syntax()
1067            .ancestors()
1068            .take_while(|it| {
1069                let kind = it.kind();
1070                ast::Path::can_cast(kind) || ast::Meta::can_cast(kind)
1071            })
1072            .last()
1073            .and_then(ast::Meta::cast);
1074
1075        // Case where path is a qualifier of another path, e.g. foo::bar::Baz where we are
1076        // trying to resolve foo::bar.
1077        if let Some(parent_hir_path) = parent_hir_path {
1078            return match resolve_hir_path_qualifier(db, &self.resolver, &hir_path, &store) {
1079                None if meta_path.is_some() => path
1080                    .first_segment()
1081                    .and_then(|it| it.name_ref())
1082                    .and_then(|name_ref| {
1083                        ToolModule::by_name(db, self.resolver.krate().into(), &name_ref.text())
1084                            .map(PathResolution::ToolModule)
1085                    })
1086                    .map(|it| (it, None)),
1087                // Case the type name conflict with use module,
1088                // e.g.
1089                // ```
1090                // use std::str;
1091                // fn main() {
1092                //     str::from_utf8();  // as module std::str
1093                //     str::len();        // as primitive type str
1094                //     str::no_exist_item(); // as primitive type str
1095                // }
1096                // ```
1097                Some(it) if matches!(it, PathResolution::Def(ModuleDef::BuiltinType(_))) => {
1098                    if let Some(mod_path) = hir_path.mod_path()
1099                        && let Some(ModuleDefId::ModuleId(id)) =
1100                            self.resolver.resolve_module_path_in_items(db, mod_path).take_types()
1101                    {
1102                        let parent_hir_name = parent_hir_path.segments().get(1).map(|it| it.name);
1103                        let module = crate::Module { id };
1104                        if module
1105                            .scope(db, None)
1106                            .into_iter()
1107                            .any(|(name, _)| Some(&name) == parent_hir_name)
1108                        {
1109                            return Some((PathResolution::Def(ModuleDef::Module(module)), None));
1110                        };
1111                    }
1112                    Some((it, None))
1113                }
1114                // FIXME: We do not show substitutions for parts of path, because this is really complex
1115                // due to the interactions with associated items of `impl`s and associated items of associated
1116                // types.
1117                res => res.map(|it| (it, None)),
1118            };
1119        } else if let Some(meta_path) = meta_path {
1120            // Case where we are resolving the final path segment of a path in an attribute
1121            // in this case we have to check for inert/builtin attributes and tools and prioritize
1122            // resolution of attributes over other namespaces
1123            if let Some(name_ref) = path.as_single_name_ref() {
1124                let builtin = BuiltinAttr::builtin(&name_ref.text());
1125                if builtin.is_some() {
1126                    return builtin.map(|it| (PathResolution::BuiltinAttr(it), None));
1127                }
1128
1129                if let Some(attr) = meta_path.parent_attr() {
1130                    let adt = if let Some(field) =
1131                        attr.syntax().parent().and_then(ast::RecordField::cast)
1132                    {
1133                        field.syntax().ancestors().take(4).find_map(ast::Adt::cast)
1134                    } else if let Some(field) =
1135                        attr.syntax().parent().and_then(ast::TupleField::cast)
1136                    {
1137                        field.syntax().ancestors().take(4).find_map(ast::Adt::cast)
1138                    } else if let Some(variant) =
1139                        attr.syntax().parent().and_then(ast::Variant::cast)
1140                    {
1141                        variant.syntax().ancestors().nth(2).and_then(ast::Adt::cast)
1142                    } else {
1143                        None
1144                    };
1145                    if let Some(adt) = adt {
1146                        let ast_id = db.ast_id_map(self.file_id).ast_id(&adt);
1147                        if let Some(helpers) = self
1148                            .resolver
1149                            .def_map()
1150                            .derive_helpers_in_scope(InFile::new(self.file_id, ast_id))
1151                        {
1152                            // FIXME: Multiple derives can have the same helper
1153                            let name_ref = name_ref.as_name();
1154                            for (macro_id, mut helpers) in
1155                                helpers.iter().chunk_by(|(_, macro_id, ..)| macro_id).into_iter()
1156                            {
1157                                if let Some(idx) = helpers.position(|(name, ..)| *name == name_ref)
1158                                {
1159                                    return Some((
1160                                        PathResolution::DeriveHelper(DeriveHelper {
1161                                            derive: *macro_id,
1162                                            idx: idx as u32,
1163                                        }),
1164                                        None,
1165                                    ));
1166                                }
1167                            }
1168                        }
1169                    }
1170                }
1171            }
1172            return match resolve_hir_path_as_attr_macro(db, &self.resolver, &hir_path) {
1173                Some(m) => Some((PathResolution::Def(ModuleDef::Macro(m)), None)),
1174                // this labels any path that starts with a tool module as the tool itself, this is technically wrong
1175                // but there is no benefit in differentiating these two cases for the time being
1176                None => path
1177                    .first_segment()
1178                    .and_then(|it| it.name_ref())
1179                    .and_then(|name_ref| {
1180                        ToolModule::by_name(db, self.resolver.krate().into(), &name_ref.text())
1181                            .map(PathResolution::ToolModule)
1182                    })
1183                    .map(|it| (it, None)),
1184            };
1185        }
1186        if parent().is_some_and(|it| ast::Visibility::can_cast(it.kind())) {
1187            // No substitution because only modules can be inside visibilities, and those have no generics.
1188            resolve_hir_path_qualifier(db, &self.resolver, &hir_path, &store).map(|it| (it, None))
1189        } else {
1190            // Probably a type, no need to show substitutions for those.
1191            let res = resolve_hir_path_(
1192                db,
1193                &self.resolver,
1194                &hir_path,
1195                prefer_value_ns,
1196                name_hygiene(db, InFile::new(self.file_id, path.syntax())),
1197                Some(&store),
1198                false,
1199            )
1200            .any()?;
1201            let subst = (|| {
1202                let parent = parent()?;
1203                let ty = if let Some(expr) = ast::Expr::cast(parent.clone()) {
1204                    let expr_id = self.expr_id(expr)?;
1205                    self.infer()?.type_of_expr_or_pat(expr_id)?
1206                } else if let Some(pat) = ast::Pat::cast(parent) {
1207                    let pat_id = self.pat_id(&pat)?;
1208                    self.infer()?.expr_or_pat_ty(pat_id)
1209                } else {
1210                    return None;
1211                };
1212                let env = self.trait_environment(db);
1213                let (subst, expected_resolution) = match ty.kind() {
1214                    TyKind::Adt(adt_def, subst) => {
1215                        let adt_id = adt_def.def_id().0;
1216                        (
1217                            GenericSubstitution::new(adt_id.into(), subst, env),
1218                            PathResolution::Def(ModuleDef::Adt(adt_id.into())),
1219                        )
1220                    }
1221                    TyKind::Alias(AliasTyKind::Projection, alias) => {
1222                        let assoc_id = alias.def_id.expect_type_alias();
1223                        (
1224                            GenericSubstitution::new(assoc_id.into(), alias.args, env),
1225                            PathResolution::Def(ModuleDef::TypeAlias(assoc_id.into())),
1226                        )
1227                    }
1228                    TyKind::FnDef(fn_id, subst) => {
1229                        let generic_def_id = match fn_id.0 {
1230                            CallableDefId::StructId(id) => id.into(),
1231                            CallableDefId::FunctionId(id) => id.into(),
1232                            CallableDefId::EnumVariantId(_) => return None,
1233                        };
1234                        (
1235                            GenericSubstitution::new(generic_def_id, subst, env),
1236                            PathResolution::Def(ModuleDefId::from(fn_id.0).into()),
1237                        )
1238                    }
1239                    _ => return None,
1240                };
1241                if res != expected_resolution {
1242                    // The user will not understand where we're coming from. This can happen (I think) with type aliases.
1243                    return None;
1244                }
1245                Some(subst)
1246            })();
1247            Some((res, subst))
1248        }
1249    }

1545fn scope_for_offset(
1546    db: &dyn HirDatabase,
1547    scopes: &ExprScopes,
1548    source_map: &BodySourceMap,
1549    from_file: HirFileId,
1550    offset: TextSize,
1551) -> Option<ScopeId> {
1552    scopes
1553        .scope_by_expr()
1554        .iter()
1555        .filter_map(|(id, scope)| {
1556            let InFile { file_id, value } = source_map.expr_syntax(id).ok()?;
1557            if from_file == file_id {
1558                return Some((value.text_range(), scope));
1559            }
1560
1561            // FIXME handle attribute expansion
1562            let source = iter::successors(file_id.macro_file().map(|it| it.call_node(db)), |it| {
1563                Some(it.file_id.macro_file()?.call_node(db))
1564            })
1565            .find(|it| it.file_id == from_file)
1566            .filter(|it| it.kind() == SyntaxKind::MACRO_CALL)?;
1567            Some((source.text_range(), scope))
1568        })
1569        .filter(|(expr_range, _scope)| expr_range.start() <= offset && offset <= expr_range.end())
1570        // find containing scope
1571        .min_by_key(|(expr_range, _scope)| expr_range.len())
1572        .map(|(expr_range, scope)| {
1573            adjust(db, scopes, source_map, expr_range, from_file, offset).unwrap_or(*scope)
1574        })
1575}

1577// XXX: during completion, cursor might be outside of any particular
1578// expression. Try to figure out the correct scope...
1579fn adjust(
1580    db: &dyn HirDatabase,
1581    scopes: &ExprScopes,
1582    source_map: &BodySourceMap,
1583    expr_range: TextRange,
1584    from_file: HirFileId,
1585    offset: TextSize,
1586) -> Option<ScopeId> {
1587    let child_scopes = scopes
1588        .scope_by_expr()
1589        .iter()
1590        .filter_map(|(id, scope)| {
1591            let source = source_map.expr_syntax(id).ok()?;
1592            // FIXME: correctly handle macro expansion
1593            if source.file_id != from_file {
1594                return None;
1595            }
1596            let root = source.file_syntax(db);
1597            let node = source.value.to_node(&root);
1598            Some((node.syntax().text_range(), scope))
1599        })
1600        .filter(|&(range, _)| {
1601            range.start() <= offset && expr_range.contains_range(range) && range != expr_range
1602        });
1603
1604    child_scopes
1605        .max_by(|&(r1, _), &(r2, _)| {
1606            if r1.contains_range(r2) {
1607                std::cmp::Ordering::Greater
1608            } else if r2.contains_range(r1) {
1609                std::cmp::Ordering::Less
1610            } else {
1611                r1.start().cmp(&r2.start())
1612            }
1613        })
1614        .map(|(_ptr, scope)| *scope)
1615}

1640fn resolve_hir_path_(
1641    db: &dyn HirDatabase,
1642    resolver: &Resolver<'_>,
1643    path: &Path,
1644    prefer_value_ns: bool,
1645    hygiene: HygieneId,
1646    store: Option<&ExpressionStore>,
1647    resolve_per_ns: bool,
1648) -> PathResolutionPerNs {
1649    let types = || {
1650        let (ty, unresolved) = match path.type_anchor() {
1651            Some(type_ref) => resolver.generic_def().and_then(|def| {
1652                let (_, res) =
1653                    TyLoweringContext::new(db, resolver, store?, def, LifetimeElisionKind::Infer)
1654                        .lower_ty_ext(type_ref);
1655                res.map(|ty_ns| (ty_ns, path.segments().first()))
1656            }),
1657            None => {
1658                let (ty, remaining_idx, _) = resolver.resolve_path_in_type_ns(db, path)?;
1659                match remaining_idx {
1660                    Some(remaining_idx) => {
1661                        if remaining_idx + 1 == path.segments().len() {
1662                            Some((ty, path.segments().last()))
1663                        } else {
1664                            None
1665                        }
1666                    }
1667                    None => Some((ty, None)),
1668                }
1669            }
1670        }?;
1671
1672        // If we are in a TypeNs for a Trait, and we have an unresolved name, try to resolve it as a type
1673        // within the trait's associated types.
1674        if let (Some(unresolved), &TypeNs::TraitId(trait_id)) = (&unresolved, &ty)
1675            && let Some(type_alias_id) =
1676                trait_id.trait_items(db).associated_type_by_name(unresolved.name)
1677        {
1678            return Some(PathResolution::Def(ModuleDefId::from(type_alias_id).into()));
1679        }
1680
1681        let res = match ty {
1682            TypeNs::SelfType(it) => PathResolution::SelfType(it.into()),
1683            TypeNs::GenericParam(id) => PathResolution::TypeParam(id.into()),
1684            TypeNs::AdtSelfType(it) | TypeNs::AdtId(it) => {
1685                PathResolution::Def(Adt::from(it).into())
1686            }
1687            TypeNs::EnumVariantId(it) => PathResolution::Def(Variant::from(it).into()),
1688            TypeNs::TypeAliasId(it) => PathResolution::Def(TypeAlias::from(it).into()),
1689            TypeNs::BuiltinType(it) => PathResolution::Def(BuiltinType::from(it).into()),
1690            TypeNs::TraitId(it) => PathResolution::Def(Trait::from(it).into()),
1691            TypeNs::ModuleId(it) => PathResolution::Def(ModuleDef::Module(it.into())),
1692        };
1693        match unresolved {
1694            Some(unresolved) => resolver
1695                .generic_def()
1696                .and_then(|def| {
1697                    hir_ty::associated_type_shorthand_candidates(
1698                        db,
1699                        def,
1700                        res.in_type_ns()?,
1701                        |name, _| name == unresolved.name,
1702                    )
1703                })
1704                .map(TypeAlias::from)
1705                .map(Into::into)
1706                .map(PathResolution::Def),
1707            None => Some(res),
1708        }
1709    };
1710
1711    let body_owner = resolver.body_owner();
1712    let values = || resolve_hir_value_path(db, resolver, body_owner, path, hygiene);
1713
1714    let items = || {
1715        resolver
1716            .resolve_module_path_in_items(db, path.mod_path()?)
1717            .take_types()
1718            .map(|it| PathResolution::Def(it.into()))
1719    };
1720
1721    let macros = || {
1722        resolver
1723            .resolve_path_as_macro(db, path.mod_path()?, None)
1724            .map(|(def, _)| PathResolution::Def(ModuleDef::Macro(def.into())))
1725    };
1726
1727    if resolve_per_ns {
1728        PathResolutionPerNs {
1729            type_ns: types().or_else(items),
1730            value_ns: values(),
1731            macro_ns: macros(),
1732        }
1733    } else {
1734        let res = if prefer_value_ns {
1735            values()
1736                .map(|value_ns| PathResolutionPerNs::new(None, Some(value_ns), None))
1737                .unwrap_or_else(|| PathResolutionPerNs::new(types(), None, None))
1738        } else {
1739            types()
1740                .map(|type_ns| PathResolutionPerNs::new(Some(type_ns), None, None))
1741                .unwrap_or_else(|| PathResolutionPerNs::new(None, values(), None))
1742        };
1743
1744        if res.any().is_some() {
1745            res
1746        } else if let Some(type_ns) = items() {
1747            PathResolutionPerNs::new(Some(type_ns), None, None)
1748        } else {
1749            PathResolutionPerNs::new(None, None, macros())
1750        }
1751    }
1752}

1754fn resolve_hir_value_path(
1755    db: &dyn HirDatabase,
1756    resolver: &Resolver<'_>,
1757    body_owner: Option<DefWithBodyId>,
1758    path: &Path,
1759    hygiene: HygieneId,
1760) -> Option<PathResolution> {
1761    resolver.resolve_path_in_value_ns_fully(db, path, hygiene).and_then(|val| {
1762        let res = match val {
1763            ValueNs::LocalBinding(binding_id) => {
1764                let var = Local { parent: body_owner?, binding_id };
1765                PathResolution::Local(var)
1766            }
1767            ValueNs::FunctionId(it) => PathResolution::Def(Function::from(it).into()),
1768            ValueNs::ConstId(it) => PathResolution::Def(Const::from(it).into()),
1769            ValueNs::StaticId(it) => PathResolution::Def(Static::from(it).into()),
1770            ValueNs::StructId(it) => PathResolution::Def(Struct::from(it).into()),
1771            ValueNs::EnumVariantId(it) => PathResolution::Def(Variant::from(it).into()),
1772            ValueNs::ImplSelf(impl_id) => PathResolution::SelfType(impl_id.into()),
1773            ValueNs::GenericParam(id) => PathResolution::ConstParam(id.into()),
1774        };
1775        Some(res)
1776    })
1777}

1779/// Resolves a path where we know it is a qualifier of another path.
1780///
1781/// For example, if we have:
1782/// ```
1783/// mod my {
1784///     pub mod foo {
1785///         struct Bar;
1786///     }
1787///
1788///     pub fn foo() {}
1789/// }
1790/// ```
1791/// then we know that `foo` in `my::foo::Bar` refers to the module, not the function.
1792fn resolve_hir_path_qualifier(
1793    db: &dyn HirDatabase,
1794    resolver: &Resolver<'_>,
1795    path: &Path,
1796    store: &ExpressionStore,
1797) -> Option<PathResolution> {
1798    (|| {
1799        let (ty, unresolved) = match path.type_anchor() {
1800            Some(type_ref) => resolver.generic_def().and_then(|def| {
1801                let (_, res) =
1802                    TyLoweringContext::new(db, resolver, store, def, LifetimeElisionKind::Infer)
1803                        .lower_ty_ext(type_ref);
1804                res.map(|ty_ns| (ty_ns, path.segments().first()))
1805            }),
1806            None => {
1807                let (ty, remaining_idx, _) = resolver.resolve_path_in_type_ns(db, path)?;
1808                match remaining_idx {
1809                    Some(remaining_idx) => {
1810                        if remaining_idx + 1 == path.segments().len() {
1811                            Some((ty, path.segments().last()))
1812                        } else {
1813                            None
1814                        }
1815                    }
1816                    None => Some((ty, None)),
1817                }
1818            }
1819        }?;
1820
1821        // If we are in a TypeNs for a Trait, and we have an unresolved name, try to resolve it as a type
1822        // within the trait's associated types.
1823        if let (Some(unresolved), &TypeNs::TraitId(trait_id)) = (&unresolved, &ty)
1824            && let Some(type_alias_id) =
1825                trait_id.trait_items(db).associated_type_by_name(unresolved.name)
1826        {
1827            return Some(PathResolution::Def(ModuleDefId::from(type_alias_id).into()));
1828        }
1829
1830        let res = match ty {
1831            TypeNs::SelfType(it) => PathResolution::SelfType(it.into()),
1832            TypeNs::GenericParam(id) => PathResolution::TypeParam(id.into()),
1833            TypeNs::AdtSelfType(it) | TypeNs::AdtId(it) => {
1834                PathResolution::Def(Adt::from(it).into())
1835            }
1836            TypeNs::EnumVariantId(it) => PathResolution::Def(Variant::from(it).into()),
1837            TypeNs::TypeAliasId(it) => PathResolution::Def(TypeAlias::from(it).into()),
1838            TypeNs::BuiltinType(it) => PathResolution::Def(BuiltinType::from(it).into()),
1839            TypeNs::TraitId(it) => PathResolution::Def(Trait::from(it).into()),
1840            TypeNs::ModuleId(it) => PathResolution::Def(ModuleDef::Module(it.into())),
1841        };
1842        match unresolved {
1843            Some(unresolved) => resolver
1844                .generic_def()
1845                .and_then(|def| {
1846                    hir_ty::associated_type_shorthand_candidates(
1847                        db,
1848                        def,
1849                        res.in_type_ns()?,
1850                        |name, _| name == unresolved.name,
1851                    )
1852                })
1853                .map(TypeAlias::from)
1854                .map(Into::into)
1855                .map(PathResolution::Def),
1856            None => Some(res),
1857        }
1858    })()
1859    .or_else(|| {
1860        resolver
1861            .resolve_module_path_in_items(db, path.mod_path()?)
1862            .take_types()
1863            .map(|it| PathResolution::Def(it.into()))
1864    })
1865}

1867pub(crate) fn name_hygiene(db: &dyn HirDatabase, name: InFile<&SyntaxNode>) -> HygieneId {
1868    let Some(macro_file) = name.file_id.macro_file() else {
1869        return HygieneId::ROOT;
1870    };
1871    let span_map = db.expansion_span_map(macro_file);
1872    let ctx = span_map.span_at(name.value.text_range().start()).ctx;

1874}

1876fn record_literal_matched_fields(
1877    db: &dyn HirDatabase,
1878    infer: &InferenceResult,
1879    id: ExprId,
1880    expr: &Expr,
1881) -> Option<(VariantId, Vec<LocalFieldId>)> {
1882    let (fields, _spread) = match expr {
1883        Expr::RecordLit { fields, spread, .. } => (fields, spread),
1884        _ => return None,
1885    };
1886
1887    let variant_def = infer.variant_resolution_for_expr(id)?;
1888    if let VariantId::UnionId(_) = variant_def {
1889        return None;
1890    }
1891
1892    let variant_data = variant_def.fields(db);
1893
1894    let specified_fields: FxHashSet<_> = fields.iter().map(|f| &f.name).collect();
1895    // suggest fields if:
1896    // - not in code
1897    let matched_fields: Vec<LocalFieldId> = variant_data
1898        .fields()
1899        .iter()
1900        .filter_map(|(f, d)| (!specified_fields.contains(&d.name)).then_some(f))
1901        .collect();
1902    if matched_fields.is_empty() {
1903        return None;
1904    }
1905    Some((variant_def, matched_fields))
1906}

1908fn record_pattern_matched_fields(
1909    db: &dyn HirDatabase,
1910    infer: &InferenceResult,
1911    id: PatId,
1912    pat: &Pat,
1913) -> Option<(VariantId, Vec<LocalFieldId>)> {
1914    let (fields, _ellipsis) = match pat {
1915        Pat::Record { path: _, args, ellipsis } => (args, *ellipsis),
1916        _ => return None,
1917    };
1918
1919    let variant_def = infer.variant_resolution_for_pat(id)?;
1920    if let VariantId::UnionId(_) = variant_def {
1921        return None;
1922    }
1923
1924    let variant_data = variant_def.fields(db);
1925
1926    let specified_fields: FxHashSet<_> = fields.iter().map(|f| &f.name).collect();
1927    // suggest fields if:
1928    // - not in code
1929    let matched_fields: Vec<LocalFieldId> = variant_data
1930        .fields()
1931        .iter()
1932        .filter_map(|(f, d)| if !specified_fields.contains(&d.name) { Some(f) } else { None })
1933        .collect();
1934    if matched_fields.is_empty() {
1935        return None;
1936    }
1937    Some((variant_def, matched_fields))
