86    pub(crate) fn new_for_body_(
87        db: &dyn HirDatabase,
88        def: DefWithBodyId,
89        node @ InFile { file_id, .. }: InFile<&SyntaxNode>,
90        offset: Option<TextSize>,
91        infer: Option<Arc<InferenceResult>>,
92    ) -> SourceAnalyzer {
93        let (body, source_map) = db.body_with_source_map(def);
94        let scopes = db.expr_scopes(def);
95        let scope = match offset {
96            None => scope_for(db, &scopes, &source_map, node),
97            Some(offset) => {
98                debug_assert!(
99                    node.text_range().contains_inclusive(offset),
100                    "{:?} not in {:?}",
101                    offset,
102                    node.text_range()
103                );
104                scope_for_offset(db, &scopes, &source_map, node.file_id, offset)
105            }
106        };
107        let resolver = resolver_for_scope(db.upcast(), def, scope);
108        SourceAnalyzer { resolver, def: Some((def, body, source_map)), infer, file_id }
109    }

149    pub(crate) fn expr_adjustments(&self, expr: &ast::Expr) -> Option<&[Adjustment]> {

157    pub(crate) fn type_of_expr(
158        &self,
159        db: &dyn HirDatabase,
160        expr: &ast::Expr,
161    ) -> Option<(Type, Option<Type>)> {
162        let expr_id = self.expr_id(expr.clone())?;
163        let infer = self.infer.as_ref()?;
164        let coerced = expr_id
165            .as_expr()
166            .and_then(|expr_id| infer.expr_adjustments.get(&expr_id))
167            .and_then(|adjusts| adjusts.last().map(|adjust| adjust.target.clone()));
168        let ty = infer[expr_id].clone();
169        let mk_ty = |ty| Type::new_with_resolver(db, &self.resolver, ty);
170        Some((mk_ty(ty), coerced.map(mk_ty)))
171    }

173    pub(crate) fn type_of_pat(
174        &self,
175        db: &dyn HirDatabase,
176        pat: &ast::Pat,
177    ) -> Option<(Type, Option<Type>)> {
178        let expr_or_pat_id = self.pat_id(pat)?;
179        let infer = self.infer.as_ref()?;
180        let coerced = match expr_or_pat_id {
181            ExprOrPatId::ExprId(idx) => infer
182                .expr_adjustments
183                .get(&idx)
184                .and_then(|adjusts| adjusts.last().cloned())
185                .map(|adjust| adjust.target),
186            ExprOrPatId::PatId(idx) => {
187                infer.pat_adjustments.get(&idx).and_then(|adjusts| adjusts.last().cloned())
188            }
189        };
190
191        let ty = infer[expr_or_pat_id].clone();
192        let mk_ty = |ty| Type::new_with_resolver(db, &self.resolver, ty);
193        Some((mk_ty(ty), coerced.map(mk_ty)))
194    }

252        db: &dyn HirDatabase,
253        call: &ast::MethodCallExpr,
254    ) -> Option<Callable> {
255        let expr_id = self.expr_id(call.clone().into())?.as_expr()?;
256        let (func, substs) = self.infer.as_ref()?.method_resolution(expr_id)?;
257        let ty = db.value_ty(func.into())?.substitute(Interner, &substs);
258        let ty = Type::new_with_resolver(db, &self.resolver, ty);
259        let mut res = ty.as_callable(db)?;
260        res.is_bound_method = true;
261        Some(res)

275    pub(crate) fn resolve_method_call_fallback(
276        &self,
277        db: &dyn HirDatabase,
278        call: &ast::MethodCallExpr,
279    ) -> Option<(Either<Function, Field>, Option<GenericSubstitution>)> {
280        let expr_id = self.expr_id(call.clone().into())?.as_expr()?;
281        let inference_result = self.infer.as_ref()?;
282        match inference_result.method_resolution(expr_id) {
283            Some((f_in_trait, substs)) => {
284                let (fn_, subst) =
285                    self.resolve_impl_method_or_trait_def_with_subst(db, f_in_trait, substs);
286                Some((
287                    Either::Left(fn_.into()),
288                    Some(GenericSubstitution::new(fn_.into(), subst, self.trait_environment(db))),
289                ))
290            }
291            None => {
292                inference_result.field_resolution(expr_id).and_then(Either::left).map(|field| {
293                    (Either::Right(field.into()), self.field_subst(expr_id, inference_result, db))
294                })
295            }
296        }
297    }

337    pub(crate) fn resolve_field_fallback(
338        &self,
339        db: &dyn HirDatabase,
340        field: &ast::FieldExpr,
341    ) -> Option<(Either<Either<Field, TupleField>, Function>, Option<GenericSubstitution>)> {
342        let &(def, ..) = self.def.as_ref()?;
343        let expr_id = self.expr_id(field.clone().into())?.as_expr()?;
344        let inference_result = self.infer.as_ref()?;
345        match inference_result.field_resolution(expr_id) {
346            Some(field) => match field {
347                Either::Left(field) => Some((
348                    Either::Left(Either::Left(field.into())),
349                    self.field_subst(expr_id, inference_result, db),
350                )),
351                Either::Right(field) => Some((
352                    Either::Left(Either::Right(TupleField {
353                        owner: def,
354                        tuple: field.tuple,
355                        index: field.index,
356                    })),
357                    None,
358                )),
359            },
360            None => inference_result.method_resolution(expr_id).map(|(f, substs)| {
361                let (f, subst) = self.resolve_impl_method_or_trait_def_with_subst(db, f, substs);
362                (
363                    Either::Right(f.into()),
364                    Some(GenericSubstitution::new(f.into(), subst, self.trait_environment(db))),
365                )
366            }),
367        }
368    }

371        &self,
372        db: &dyn HirDatabase,
373        range_pat: &ast::RangePat,
374    ) -> Option<StructId> {
375        let path: ModPath = match (range_pat.op_kind()?, range_pat.start(), range_pat.end()) {
376            (RangeOp::Exclusive, None, Some(_)) => path![core::ops::RangeTo],
377            (RangeOp::Exclusive, Some(_), None) => path![core::ops::RangeFrom],
378            (RangeOp::Exclusive, Some(_), Some(_)) => path![core::ops::Range],
379            (RangeOp::Inclusive, None, Some(_)) => path![core::ops::RangeToInclusive],
380            (RangeOp::Inclusive, Some(_), Some(_)) => path![core::ops::RangeInclusive],
381
382            (RangeOp::Exclusive, None, None) => return None,
383            (RangeOp::Inclusive, None, None) => return None,
384            (RangeOp::Inclusive, Some(_), None) => return None,
385        };
386        self.resolver.resolve_known_struct(db.upcast(), &path)
387    }

409    pub(crate) fn resolve_await_to_poll(
410        &self,
411        db: &dyn HirDatabase,
412        await_expr: &ast::AwaitExpr,
413    ) -> Option<FunctionId> {
414        let mut ty = self.ty_of_expr(await_expr.expr()?)?.clone();
415
416        let into_future_trait = self
417            .resolver
418            .resolve_known_trait(db.upcast(), &path![core::future::IntoFuture])
419            .map(Trait::from);
420
421        if let Some(into_future_trait) = into_future_trait {
422            let type_ = Type::new_with_resolver(db, &self.resolver, ty.clone());
423            if type_.impls_trait(db, into_future_trait, &[]) {
424                let items = into_future_trait.items(db);
425                let into_future_type = items.into_iter().find_map(|item| match item {
426                    AssocItem::TypeAlias(alias)
427                        if alias.name(db) == Name::new_symbol_root(sym::IntoFuture.clone()) =>
428                    {
429                        Some(alias)
430                    }
431                    _ => None,
432                })?;
433                let future_trait = type_.normalize_trait_assoc_type(db, &[], into_future_type)?;
434                ty = future_trait.ty;
435            }
436        }
437
438        let future_trait = db.lang_item(self.resolver.krate(), LangItem::Future)?.as_trait()?;
439        let poll_fn = db.lang_item(self.resolver.krate(), LangItem::FuturePoll)?.as_function()?;
440        // HACK: subst for `poll()` coincides with that for `Future` because `poll()` itself
441        // doesn't have any generic parameters, so we skip building another subst for `poll()`.
442        let substs = hir_ty::TyBuilder::subst_for_def(db, future_trait, None).push(ty).build();
443        Some(self.resolve_impl_method_or_trait_def(db, poll_fn, substs))
444    }

446    pub(crate) fn resolve_prefix_expr(
447        &self,
448        db: &dyn HirDatabase,
449        prefix_expr: &ast::PrefixExpr,
450    ) -> Option<FunctionId> {
451        let (op_trait, op_fn) = match prefix_expr.op_kind()? {
452            ast::UnaryOp::Deref => {
453                // This can be either `Deref::deref` or `DerefMut::deref_mut`.
454                // Since deref kind is inferenced and stored in `InferenceResult.method_resolution`,
455                // use that result to find out which one it is.
456                let (deref_trait, deref) = self.lang_trait_fn(
457                    db,
458                    LangItem::Deref,
459                    &Name::new_symbol_root(sym::deref.clone()),
460                )?;
461                self.infer
462                    .as_ref()
463                    .and_then(|infer| {
464                        let expr = self.expr_id(prefix_expr.clone().into())?.as_expr()?;
465                        let (func, _) = infer.method_resolution(expr)?;
466                        let (deref_mut_trait, deref_mut) = self.lang_trait_fn(
467                            db,
468                            LangItem::DerefMut,
469                            &Name::new_symbol_root(sym::deref_mut.clone()),
470                        )?;
471                        if func == deref_mut { Some((deref_mut_trait, deref_mut)) } else { None }
472                    })
473                    .unwrap_or((deref_trait, deref))
474            }
475            ast::UnaryOp::Not => {
476                self.lang_trait_fn(db, LangItem::Not, &Name::new_symbol_root(sym::not.clone()))?
477            }
478            ast::UnaryOp::Neg => {
479                self.lang_trait_fn(db, LangItem::Neg, &Name::new_symbol_root(sym::neg.clone()))?
480            }
481        };
482
483        let ty = self.ty_of_expr(prefix_expr.expr()?)?;
484
485        // HACK: subst for all methods coincides with that for their trait because the methods
486        // don't have any generic parameters, so we skip building another subst for the methods.
487        let substs = hir_ty::TyBuilder::subst_for_def(db, op_trait, None).push(ty.clone()).build();
488
489        Some(self.resolve_impl_method_or_trait_def(db, op_fn, substs))
490    }

492    pub(crate) fn resolve_index_expr(
493        &self,
494        db: &dyn HirDatabase,
495        index_expr: &ast::IndexExpr,
496    ) -> Option<FunctionId> {
497        let base_ty = self.ty_of_expr(index_expr.base()?)?;
498        let index_ty = self.ty_of_expr(index_expr.index()?)?;
499
500        let (index_trait, index_fn) =
501            self.lang_trait_fn(db, LangItem::Index, &Name::new_symbol_root(sym::index.clone()))?;
502        let (op_trait, op_fn) = self
503            .infer
504            .as_ref()
505            .and_then(|infer| {
506                let expr = self.expr_id(index_expr.clone().into())?.as_expr()?;
507                let (func, _) = infer.method_resolution(expr)?;
508                let (index_mut_trait, index_mut_fn) = self.lang_trait_fn(
509                    db,
510                    LangItem::IndexMut,
511                    &Name::new_symbol_root(sym::index_mut.clone()),
512                )?;
513                if func == index_mut_fn { Some((index_mut_trait, index_mut_fn)) } else { None }
514            })
515            .unwrap_or((index_trait, index_fn));
516        // HACK: subst for all methods coincides with that for their trait because the methods
517        // don't have any generic parameters, so we skip building another subst for the methods.
518        let substs = hir_ty::TyBuilder::subst_for_def(db, op_trait, None)
519            .push(base_ty.clone())
520            .push(index_ty.clone())
521            .build();
522        Some(self.resolve_impl_method_or_trait_def(db, op_fn, substs))
523    }

551        let ty = self.ty_of_expr(try_expr.expr()?)?;
552
553        let op_fn = db.lang_item(self.resolver.krate(), LangItem::TryTraitBranch)?.as_function()?;
554        let op_trait = match op_fn.lookup(db.upcast()).container {
555            ItemContainerId::TraitId(id) => id,
556            _ => return None,
557        };
558        // HACK: subst for `branch()` coincides with that for `Try` because `branch()` itself
559        // doesn't have any generic parameters, so we skip building another subst for `branch()`.
560        let substs = hir_ty::TyBuilder::subst_for_def(db, op_trait, None).push(ty.clone()).build();
561
562        Some(self.resolve_impl_method_or_trait_def(db, op_fn, substs))
563    }

565    pub(crate) fn resolve_record_field(
566        &self,
567        db: &dyn HirDatabase,
568        field: &ast::RecordExprField,
569    ) -> Option<(Field, Option<Local>, Type, GenericSubstitution)> {
570        let record_expr = ast::RecordExpr::cast(field.syntax().parent().and_then(|p| p.parent())?)?;
571        let expr = ast::Expr::from(record_expr);
572        let expr_id = self.body_source_map()?.node_expr(InFile::new(self.file_id, &expr))?;
573
574        let ast_name = field.field_name()?;
575        let local_name = ast_name.as_name();
576        let local = if field.name_ref().is_some() {
577            None
578        } else {
579            // Shorthand syntax, resolve to the local
580            let path = Path::from_known_path_with_no_generic(ModPath::from_segments(
581                PathKind::Plain,
582                once(local_name.clone()),
583            ));
584            match self.resolver.resolve_path_in_value_ns_fully(
585                db.upcast(),
586                &path,
587                name_hygiene(db, InFile::new(self.file_id, ast_name.syntax())),
588            ) {
589                Some(ValueNs::LocalBinding(binding_id)) => {
590                    Some(Local { binding_id, parent: self.resolver.body_owner()? })
591                }
592                _ => None,
593            }
594        };
595        let (adt, subst) = self.infer.as_ref()?.type_of_expr_or_pat(expr_id)?.as_adt()?;
596        let variant = self.infer.as_ref()?.variant_resolution_for_expr_or_pat(expr_id)?;
597        let variant_data = variant.variant_data(db.upcast());
598        let field = FieldId { parent: variant, local_id: variant_data.field(&local_name)? };
599        let field_ty =
600            db.field_types(variant).get(field.local_id)?.clone().substitute(Interner, subst);
601        Some((
602            field.into(),
603            local,
604            Type::new_with_resolver(db, &self.resolver, field_ty),
605            GenericSubstitution::new(adt.into(), subst.clone(), self.trait_environment(db)),
606        ))
607    }

679    pub(crate) fn resolve_path(
680        &self,
681        db: &dyn HirDatabase,
682        path: &ast::Path,
683    ) -> Option<(PathResolution, Option<GenericSubstitution>)> {
684        let parent = path.syntax().parent();
685        let parent = || parent.clone();
686
687        let mut prefer_value_ns = false;
688        let resolved = (|| {
689            let infer = self.infer.as_deref()?;
690            if let Some(path_expr) = parent().and_then(ast::PathExpr::cast) {
691                let expr_id = self.expr_id(path_expr.into())?;
692                if let Some((assoc, subs)) = infer.assoc_resolutions_for_expr_or_pat(expr_id) {
693                    let (assoc, subst) = match assoc {
694                        AssocItemId::FunctionId(f_in_trait) => {
695                            match infer.type_of_expr_or_pat(expr_id) {
696                                None => {
697                                    let subst = GenericSubstitution::new(
698                                        f_in_trait.into(),
699                                        subs,
700                                        self.trait_environment(db),
701                                    );
702                                    (assoc, subst)
703                                }
704                                Some(func_ty) => {
705                                    if let TyKind::FnDef(_fn_def, subs) = func_ty.kind(Interner) {
706                                        let (fn_, subst) = self
707                                            .resolve_impl_method_or_trait_def_with_subst(
708                                                db,
709                                                f_in_trait,
710                                                subs.clone(),
711                                            );
712                                        let subst = GenericSubstitution::new(
713                                            fn_.into(),
714                                            subst,
715                                            self.trait_environment(db),
716                                        );
717                                        (fn_.into(), subst)
718                                    } else {
719                                        let subst = GenericSubstitution::new(
720                                            f_in_trait.into(),
721                                            subs,
722                                            self.trait_environment(db),
723                                        );
724                                        (assoc, subst)
725                                    }
726                                }
727                            }
728                        }
729                        AssocItemId::ConstId(const_id) => {
730                            let (konst, subst) =
731                                self.resolve_impl_const_or_trait_def_with_subst(db, const_id, subs);
732                            let subst = GenericSubstitution::new(
733                                konst.into(),
734                                subst,
735                                self.trait_environment(db),
736                            );
737                            (konst.into(), subst)
738                        }
739                        AssocItemId::TypeAliasId(type_alias) => (
740                            assoc,
741                            GenericSubstitution::new(
742                                type_alias.into(),
743                                subs,
744                                self.trait_environment(db),
745                            ),
746                        ),
747                    };
748
749                    return Some((PathResolution::Def(AssocItem::from(assoc).into()), Some(subst)));
750                }
751                if let Some(VariantId::EnumVariantId(variant)) =
752                    infer.variant_resolution_for_expr_or_pat(expr_id)
753                {
754                    return Some((PathResolution::Def(ModuleDef::Variant(variant.into())), None));
755                }
756                prefer_value_ns = true;
757            } else if let Some(path_pat) = parent().and_then(ast::PathPat::cast) {
758                let expr_or_pat_id = self.pat_id(&path_pat.into())?;
759                if let Some((assoc, subs)) = infer.assoc_resolutions_for_expr_or_pat(expr_or_pat_id)
760                {
761                    let (assoc, subst) = match assoc {
762                        AssocItemId::ConstId(const_id) => {
763                            let (konst, subst) =
764                                self.resolve_impl_const_or_trait_def_with_subst(db, const_id, subs);
765                            let subst = GenericSubstitution::new(
766                                konst.into(),
767                                subst,
768                                self.trait_environment(db),
769                            );
770                            (konst.into(), subst)
771                        }
772                        assoc => (
773                            assoc,
774                            GenericSubstitution::new(
775                                assoc.into(),
776                                subs,
777                                self.trait_environment(db),
778                            ),
779                        ),
780                    };
781                    return Some((PathResolution::Def(AssocItem::from(assoc).into()), Some(subst)));
782                }
783                if let Some(VariantId::EnumVariantId(variant)) =
784                    infer.variant_resolution_for_expr_or_pat(expr_or_pat_id)
785                {
786                    return Some((PathResolution::Def(ModuleDef::Variant(variant.into())), None));
787                }
788            } else if let Some(rec_lit) = parent().and_then(ast::RecordExpr::cast) {
789                let expr_id = self.expr_id(rec_lit.into())?;
790                if let Some(VariantId::EnumVariantId(variant)) =
791                    infer.variant_resolution_for_expr_or_pat(expr_id)
792                {
793                    return Some((PathResolution::Def(ModuleDef::Variant(variant.into())), None));
794                }
795            } else {
796                let record_pat = parent().and_then(ast::RecordPat::cast).map(ast::Pat::from);
797                let tuple_struct_pat =
798                    || parent().and_then(ast::TupleStructPat::cast).map(ast::Pat::from);
799                if let Some(pat) = record_pat.or_else(tuple_struct_pat) {
800                    let pat_id = self.pat_id(&pat)?;
801                    let variant_res_for_pat = infer.variant_resolution_for_pat(pat_id.as_pat()?);
802                    if let Some(VariantId::EnumVariantId(variant)) = variant_res_for_pat {
803                        return Some((
804                            PathResolution::Def(ModuleDef::Variant(variant.into())),
805                            None,
806                        ));
807                    }
808                }
809            }
810            None
811        })();
812        if resolved.is_some() {
813            return resolved;
814        }
815
816        let (mut types_map, mut types_source_map) =
817            (TypesMap::default(), TypesSourceMap::default());
818        let mut ctx =
819            LowerCtx::new(db.upcast(), self.file_id, &mut types_map, &mut types_source_map);
820        let hir_path = Path::from_src(&mut ctx, path.clone())?;
821
822        // Case where path is a qualifier of a use tree, e.g. foo::bar::{Baz, Qux} where we are
823        // trying to resolve foo::bar.
824        if let Some(use_tree) = parent().and_then(ast::UseTree::cast) {
825            if use_tree.coloncolon_token().is_some() {
826                return resolve_hir_path_qualifier(db, &self.resolver, &hir_path, &types_map)
827                    .map(|it| (it, None));
828            }
829        }
830
831        let meta_path = path
832            .syntax()
833            .ancestors()
834            .take_while(|it| {
835                let kind = it.kind();
836                ast::Path::can_cast(kind) || ast::Meta::can_cast(kind)
837            })
838            .last()
839            .and_then(ast::Meta::cast);
840
841        // Case where path is a qualifier of another path, e.g. foo::bar::Baz where we are
842        // trying to resolve foo::bar.
843        if let Some(parent_path) = path.parent_path() {
844            let parent_hir_path = Path::from_src(&mut ctx, parent_path);
845            return match resolve_hir_path_qualifier(db, &self.resolver, &hir_path, &types_map) {
846                None if meta_path.is_some() => path
847                    .first_segment()
848                    .and_then(|it| it.name_ref())
849                    .and_then(|name_ref| {
850                        ToolModule::by_name(db, self.resolver.krate().into(), &name_ref.text())
851                            .map(PathResolution::ToolModule)
852                    })
853                    .map(|it| (it, None)),
854                // Case the type name conflict with use module,
855                // e.g.
856                // ```
857                // use std::str;
858                // fn main() {
859                //     str::from_utf8();  // as module std::str
860                //     str::len();        // as primitive type str
861                //     str::no_exist_item(); // as primitive type str
862                // }
863                // ```
864                Some(it) if matches!(it, PathResolution::Def(ModuleDef::BuiltinType(_))) => {
865                    if let (Some(mod_path), Some(parent_hir_path)) =
866                        (hir_path.mod_path(), parent_hir_path)
867                    {
868                        if let Some(ModuleDefId::ModuleId(id)) = self
869                            .resolver
870                            .resolve_module_path_in_items(db.upcast(), mod_path)
871                            .take_types()
872                        {
873                            let parent_hir_name =
874                                parent_hir_path.segments().get(1).map(|it| it.name);
875                            let module = crate::Module { id };
876                            if module
877                                .scope(db, None)
878                                .into_iter()
879                                .any(|(name, _)| Some(&name) == parent_hir_name)
880                            {
881                                return Some((
882                                    PathResolution::Def(ModuleDef::Module(module)),
883                                    None,
884                                ));
885                            };
886                        }
887                    }
888                    Some((it, None))
889                }
890                // FIXME: We do not show substitutions for parts of path, because this is really complex
891                // due to the interactions with associated items of `impl`s and associated items of associated
892                // types.
893                res => res.map(|it| (it, None)),
894            };
895        } else if let Some(meta_path) = meta_path {
896            // Case where we are resolving the final path segment of a path in an attribute
897            // in this case we have to check for inert/builtin attributes and tools and prioritize
898            // resolution of attributes over other namespaces
899            if let Some(name_ref) = path.as_single_name_ref() {
900                let builtin =
901                    BuiltinAttr::by_name(db, self.resolver.krate().into(), &name_ref.text());
902                if builtin.is_some() {
903                    return builtin.map(|it| (PathResolution::BuiltinAttr(it), None));
904                }
905
906                if let Some(attr) = meta_path.parent_attr() {
907                    let adt = if let Some(field) =
908                        attr.syntax().parent().and_then(ast::RecordField::cast)
909                    {
910                        field.syntax().ancestors().take(4).find_map(ast::Adt::cast)
911                    } else if let Some(field) =
912                        attr.syntax().parent().and_then(ast::TupleField::cast)
913                    {
914                        field.syntax().ancestors().take(4).find_map(ast::Adt::cast)
915                    } else if let Some(variant) =
916                        attr.syntax().parent().and_then(ast::Variant::cast)
917                    {
918                        variant.syntax().ancestors().nth(2).and_then(ast::Adt::cast)
919                    } else {
920                        None
921                    };
922                    if let Some(adt) = adt {
923                        let ast_id = db.ast_id_map(self.file_id).ast_id(&adt);
924                        if let Some(helpers) = self
925                            .resolver
926                            .def_map()
927                            .derive_helpers_in_scope(InFile::new(self.file_id, ast_id))
928                        {
929                            // FIXME: Multiple derives can have the same helper
930                            let name_ref = name_ref.as_name();
931                            for (macro_id, mut helpers) in
932                                helpers.iter().chunk_by(|(_, macro_id, ..)| macro_id).into_iter()
933                            {
934                                if let Some(idx) = helpers.position(|(name, ..)| *name == name_ref)
935                                {
936                                    return Some((
937                                        PathResolution::DeriveHelper(DeriveHelper {
938                                            derive: *macro_id,
939                                            idx: idx as u32,
940                                        }),
941                                        None,
942                                    ));
943                                }
944                            }
945                        }
946                    }
947                }
948            }
949            return match resolve_hir_path_as_attr_macro(db, &self.resolver, &hir_path) {
950                Some(m) => Some((PathResolution::Def(ModuleDef::Macro(m)), None)),
951                // this labels any path that starts with a tool module as the tool itself, this is technically wrong
952                // but there is no benefit in differentiating these two cases for the time being
953                None => path
954                    .first_segment()
955                    .and_then(|it| it.name_ref())
956                    .and_then(|name_ref| {
957                        ToolModule::by_name(db, self.resolver.krate().into(), &name_ref.text())
958                            .map(PathResolution::ToolModule)
959                    })
960                    .map(|it| (it, None)),
961            };
962        }
963        if parent().is_some_and(|it| ast::Visibility::can_cast(it.kind())) {
964            // No substitution because only modules can be inside visibilities, and those have no generics.
965            resolve_hir_path_qualifier(db, &self.resolver, &hir_path, &types_map)
966                .map(|it| (it, None))
967        } else {
968            // Probably a type, no need to show substitutions for those.
969            let res = resolve_hir_path_(
970                db,
971                &self.resolver,
972                &hir_path,
973                prefer_value_ns,
974                name_hygiene(db, InFile::new(self.file_id, path.syntax())),
975                &types_map,
976            )?;
977            let subst = (|| {
978                let parent = parent()?;
979                let ty = if let Some(expr) = ast::Expr::cast(parent.clone()) {
980                    let expr_id = self.expr_id(expr)?;
981                    self.infer.as_ref()?.type_of_expr_or_pat(expr_id)?
982                } else if let Some(pat) = ast::Pat::cast(parent) {
983                    let pat_id = self.pat_id(&pat)?;
984                    &self.infer.as_ref()?[pat_id]
985                } else {
986                    return None;
987                };
988                let env = self.trait_environment(db);
989                let (subst, expected_resolution) = match ty.kind(Interner) {
990                    TyKind::Adt(adt_id, subst) => (
991                        GenericSubstitution::new(adt_id.0.into(), subst.clone(), env),
992                        PathResolution::Def(ModuleDef::Adt(adt_id.0.into())),
993                    ),
994                    TyKind::AssociatedType(assoc_id, subst) => {
995                        let assoc_id = from_assoc_type_id(*assoc_id);
996                        (
997                            GenericSubstitution::new(assoc_id.into(), subst.clone(), env),
998                            PathResolution::Def(ModuleDef::TypeAlias(assoc_id.into())),
999                        )
1000                    }
1001                    TyKind::FnDef(fn_id, subst) => {
1002                        let fn_id = hir_ty::db::InternedCallableDefId::from(*fn_id);
1003                        let fn_id = db.lookup_intern_callable_def(fn_id);
1004                        let generic_def_id = match fn_id {
1005                            CallableDefId::StructId(id) => id.into(),
1006                            CallableDefId::FunctionId(id) => id.into(),
1007                            CallableDefId::EnumVariantId(_) => return None,
1008                        };
1009                        (
1010                            GenericSubstitution::new(generic_def_id, subst.clone(), env),
1011                            PathResolution::Def(ModuleDefId::from(fn_id).into()),
1012                        )
1013                    }
1014                    _ => return None,
1015                };
1016                if res != expected_resolution {
1017                    // The user will not understand where we're coming from. This can happen (I think) with type aliases.
1018                    return None;
1019                }
1020                Some(subst)
1021            })();
1022            Some((res, subst))
1023        }
1024    }

1100    pub(crate) fn resolve_variant(&self, record_lit: ast::RecordExpr) -> Option<VariantId> {
1101        let infer = self.infer.as_ref()?;

1272fn scope_for_offset(
1273    db: &dyn HirDatabase,
1274    scopes: &ExprScopes,
1275    source_map: &BodySourceMap,
1276    from_file: HirFileId,
1277    offset: TextSize,
1278) -> Option<ScopeId> {
1279    scopes
1280        .scope_by_expr()
1281        .iter()
1282        .filter_map(|(id, scope)| {
1283            let InFile { file_id, value } = source_map.expr_syntax(id).ok()?;
1284            if from_file == file_id {
1285                return Some((value.text_range(), scope));
1286            }
1287
1288            // FIXME handle attribute expansion
1289            let source =
1290                iter::successors(file_id.macro_file().map(|it| it.call_node(db.upcast())), |it| {
1291                    Some(it.file_id.macro_file()?.call_node(db.upcast()))
1292                })
1293                .find(|it| it.file_id == from_file)
1294                .filter(|it| it.kind() == SyntaxKind::MACRO_CALL)?;
1295            Some((source.text_range(), scope))
1296        })
1297        .filter(|(expr_range, _scope)| expr_range.start() <= offset && offset <= expr_range.end())
1298        // find containing scope
1299        .min_by_key(|(expr_range, _scope)| expr_range.len())
1300        .map(|(expr_range, scope)| {
1301            adjust(db, scopes, source_map, expr_range, from_file, offset).unwrap_or(*scope)
1302        })
1303}

1305// XXX: during completion, cursor might be outside of any particular
1306// expression. Try to figure out the correct scope...
1307fn adjust(
1308    db: &dyn HirDatabase,
1309    scopes: &ExprScopes,
1310    source_map: &BodySourceMap,
1311    expr_range: TextRange,
1312    from_file: HirFileId,
1313    offset: TextSize,
1314) -> Option<ScopeId> {
1315    let child_scopes = scopes
1316        .scope_by_expr()
1317        .iter()
1318        .filter_map(|(id, scope)| {
1319            let source = source_map.expr_syntax(id).ok()?;
1320            // FIXME: correctly handle macro expansion
1321            if source.file_id != from_file {
1322                return None;
1323            }
1324            let root = source.file_syntax(db.upcast());
1325            let node = source.value.to_node(&root);
1326            Some((node.syntax().text_range(), scope))
1327        })
1328        .filter(|&(range, _)| {
1329            range.start() <= offset && expr_range.contains_range(range) && range != expr_range
1330        });
1331
1332    child_scopes
1333        .max_by(|&(r1, _), &(r2, _)| {
1334            if r1.contains_range(r2) {
1335                std::cmp::Ordering::Greater
1336            } else if r2.contains_range(r1) {
1337                std::cmp::Ordering::Less
1338            } else {
1339                r1.start().cmp(&r2.start())
1340            }
1341        })
1342        .map(|(_ptr, scope)| *scope)
1343}

1368fn resolve_hir_path_(
1369    db: &dyn HirDatabase,
1370    resolver: &Resolver,
1371    path: &Path,
1372    prefer_value_ns: bool,
1373    hygiene: HygieneId,
1374    types_map: &TypesMap,
1375) -> Option<PathResolution> {
1376    let types = || {
1377        let (ty, unresolved) = match path.type_anchor() {
1378            Some(type_ref) => {
1379                let (_, res) = TyLoweringContext::new_maybe_unowned(
1380                    db,
1381                    resolver,
1382                    types_map,
1383                    None,
1384                    resolver.type_owner(),
1385                )
1386                .lower_ty_ext(type_ref);
1387                res.map(|ty_ns| (ty_ns, path.segments().first()))
1388            }
1389            None => {
1390                let (ty, remaining_idx, _) = resolver.resolve_path_in_type_ns(db.upcast(), path)?;
1391                match remaining_idx {
1392                    Some(remaining_idx) => {
1393                        if remaining_idx + 1 == path.segments().len() {
1394                            Some((ty, path.segments().last()))
1395                        } else {
1396                            None
1397                        }
1398                    }
1399                    None => Some((ty, None)),
1400                }
1401            }
1402        }?;
1403
1404        // If we are in a TypeNs for a Trait, and we have an unresolved name, try to resolve it as a type
1405        // within the trait's associated types.
1406        if let (Some(unresolved), &TypeNs::TraitId(trait_id)) = (&unresolved, &ty) {
1407            if let Some(type_alias_id) =
1408                db.trait_items(trait_id).associated_type_by_name(unresolved.name)
1409            {
1410                return Some(PathResolution::Def(ModuleDefId::from(type_alias_id).into()));
1411            }
1412        }
1413
1414        let res = match ty {
1415            TypeNs::SelfType(it) => PathResolution::SelfType(it.into()),
1416            TypeNs::GenericParam(id) => PathResolution::TypeParam(id.into()),
1417            TypeNs::AdtSelfType(it) | TypeNs::AdtId(it) => {
1418                PathResolution::Def(Adt::from(it).into())
1419            }
1420            TypeNs::EnumVariantId(it) => PathResolution::Def(Variant::from(it).into()),
1421            TypeNs::TypeAliasId(it) => PathResolution::Def(TypeAlias::from(it).into()),
1422            TypeNs::BuiltinType(it) => PathResolution::Def(BuiltinType::from(it).into()),
1423            TypeNs::TraitId(it) => PathResolution::Def(Trait::from(it).into()),
1424            TypeNs::TraitAliasId(it) => PathResolution::Def(TraitAlias::from(it).into()),
1425        };
1426        match unresolved {
1427            Some(unresolved) => resolver
1428                .generic_def()
1429                .and_then(|def| {
1430                    hir_ty::associated_type_shorthand_candidates(
1431                        db,
1432                        def,
1433                        res.in_type_ns()?,
1434                        |name, id| (name == unresolved.name).then_some(id),
1435                    )
1436                })
1437                .map(TypeAlias::from)
1438                .map(Into::into)
1439                .map(PathResolution::Def),
1440            None => Some(res),
1441        }
1442    };
1443
1444    let body_owner = resolver.body_owner();
1445    let values = || resolve_hir_value_path(db, resolver, body_owner, path, hygiene);
1446
1447    let items = || {
1448        resolver
1449            .resolve_module_path_in_items(db.upcast(), path.mod_path()?)
1450            .take_types()
1451            .map(|it| PathResolution::Def(it.into()))
1452    };
1453
1454    let macros = || {
1455        resolver
1456            .resolve_path_as_macro(db.upcast(), path.mod_path()?, None)
1457            .map(|(def, _)| PathResolution::Def(ModuleDef::Macro(def.into())))
1458    };
1459
1460    if prefer_value_ns { values().or_else(types) } else { types().or_else(values) }
1461        .or_else(items)
1462        .or_else(macros)
1463}

1465fn resolve_hir_value_path(
1466    db: &dyn HirDatabase,
1467    resolver: &Resolver,
1468    body_owner: Option<DefWithBodyId>,
1469    path: &Path,
1470    hygiene: HygieneId,
1471) -> Option<PathResolution> {
1472    resolver.resolve_path_in_value_ns_fully(db.upcast(), path, hygiene).and_then(|val| {
1473        let res = match val {
1474            ValueNs::LocalBinding(binding_id) => {
1475                let var = Local { parent: body_owner?, binding_id };
1476                PathResolution::Local(var)
1477            }
1478            ValueNs::FunctionId(it) => PathResolution::Def(Function::from(it).into()),
1479            ValueNs::ConstId(it) => PathResolution::Def(Const::from(it).into()),
1480            ValueNs::StaticId(it) => PathResolution::Def(Static::from(it).into()),
1481            ValueNs::StructId(it) => PathResolution::Def(Struct::from(it).into()),
1482            ValueNs::EnumVariantId(it) => PathResolution::Def(Variant::from(it).into()),
1483            ValueNs::ImplSelf(impl_id) => PathResolution::SelfType(impl_id.into()),
1484            ValueNs::GenericParam(id) => PathResolution::ConstParam(id.into()),
1485        };
1486        Some(res)
1487    })
1488}
