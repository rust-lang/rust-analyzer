1//! HIR (previously known as descriptors) provides a high-level object-oriented
2//! access to Rust code.
3//!
4//! The principal difference between HIR and syntax trees is that HIR is bound
5//! to a particular crate instance. That is, it has cfg flags and features
6//! applied. So, the relation between syntax and HIR is many-to-one.
7//!
8//! HIR is the public API of the all of the compiler logic above syntax trees.
9//! It is written in "OO" style. Each type is self contained (as in, it knows its
10//! parents and full context). It should be "clean code".
11//!
12//! `hir_*` crates are the implementation of the compiler logic.
13//! They are written in "ECS" style, with relatively little abstractions.
14//! Many types are not self-contained, and explicitly use local indexes, arenas, etc.
15//!
16//! `hir` is what insulates the "we don't know how to actually write an incremental compiler"
17//! from the ide with completions, hovers, etc. It is a (soft, internal) boundary:
18//! <https://www.tedinski.com/2018/02/06/system-boundaries.html>.
19

435    pub fn diagnostics<'db>(
436        self,
437        db: &'db dyn HirDatabase,
438        style_lints: bool,
439    ) -> Vec<AnyDiagnostic<'db>> {
440        let id = match self {
441            ModuleDef::Adt(it) => match it {
442                Adt::Struct(it) => it.id.into(),
443                Adt::Enum(it) => it.id.into(),
444                Adt::Union(it) => it.id.into(),
445            },
446            ModuleDef::Trait(it) => it.id.into(),
447            ModuleDef::Function(it) => match it.id {
448                AnyFunctionId::FunctionId(it) => it.into(),
449                AnyFunctionId::BuiltinDeriveImplMethod { .. } => return Vec::new(),
450            },
451            ModuleDef::TypeAlias(it) => it.id.into(),
452            ModuleDef::Module(it) => it.id.into(),
453            ModuleDef::Const(it) => it.id.into(),
454            ModuleDef::Static(it) => it.id.into(),
455            ModuleDef::Variant(it) => it.id.into(),
456            ModuleDef::BuiltinType(_) | ModuleDef::Macro(_) => return Vec::new(),
457        };
458
459        let mut acc = Vec::new();
460
461        match self.as_def_with_body() {
462            Some(def) => {
463                def.diagnostics(db, &mut acc, style_lints);
464            }
465            None => {
466                for diag in hir_ty::diagnostics::incorrect_case(db, id) {
467                    acc.push(diag.into())
468                }
469            }
470        }
471
472        if let Some(def) = self.as_self_generic_def() {
473            def.diagnostics(db, &mut acc);
474        }
475
476        acc
477    }

594    /// Finds nearest non-block ancestor `Module` (`self` included).
595    pub fn nearest_non_block_module(self, db: &dyn HirDatabase) -> Module {
596        let mut id = self.id;
597        while id.is_block_module(db) {

603    pub fn path_to_root(self, db: &dyn HirDatabase) -> Vec<Module> {
604        let mut res = vec![self];
605        let mut curr = self;
606        while let Some(next) = curr.parent(db) {
607            res.push(next);
608            curr = next
609        }
610        res

613    pub fn modules_in_scope(&self, db: &dyn HirDatabase, pub_only: bool) -> Vec<(Name, Module)> {
614        let def_map = self.id.def_map(db);
615        let scope = &def_map[self.id].scope;
616
617        let mut res = Vec::new();
618
619        for (name, item) in scope.types() {
620            if let ModuleDefId::ModuleId(m) = item.def
621                && (!pub_only || item.vis == Visibility::Public)
622            {
623                res.push((name.clone(), Module { id: m }));
624            }
625        }
626
627        res

630    /// Returns a `ModuleScope`: a set of items, visible in this module.
631    pub fn scope(
632        self,
633        db: &dyn HirDatabase,
634        visible_from: Option<Module>,
635    ) -> Vec<(Name, ScopeDef)> {
636        self.id.def_map(db)[self.id]
637            .scope
638            .entries()
639            .filter_map(|(name, def)| {
640                if let Some(m) = visible_from {
641                    let filtered = def.filter_visibility(|vis| vis.is_visible_from(db, m.id));
642                    if filtered.is_none() && !def.is_none() { None } else { Some((name, filtered)) }
643                } else {
644                    Some((name, def))
645                }
646            })
647            .flat_map(|(name, def)| {
648                ScopeDef::all_items(def).into_iter().map(move |item| (name.clone(), item))
649            })
650            .collect()
651    }

665    /// Fills `acc` with the module's diagnostics.
666    pub fn diagnostics<'db>(
667        self,
668        db: &'db dyn HirDatabase,
669        acc: &mut Vec<AnyDiagnostic<'db>>,
670        style_lints: bool,
671    ) {
672        let _p = tracing::info_span!("diagnostics", name = ?self.name(db)).entered();
673        let edition = self.id.krate(db).data(db).edition;
674        let def_map = self.id.def_map(db);
675        for diag in def_map.diagnostics() {
676            if diag.in_module != self.id {
677                // FIXME: This is accidentally quadratic.
678                continue;
679            }
680            emit_def_diagnostic(db, acc, diag, edition, def_map.krate());
681        }
682
683        if !self.id.is_block_module(db) {
684            // These are reported by the body of block modules
685            let scope = &def_map[self.id].scope;
686            scope.all_macro_calls().for_each(|it| macro_call_diagnostics(db, it, acc));
687        }
688
689        for def in self.declarations(db) {
690            match def {
691                ModuleDef::Module(m) => {
692                    // Only add diagnostics from inline modules
693                    if def_map[m.id].origin.is_inline() {
694                        m.diagnostics(db, acc, style_lints)
695                    }
696                    acc.extend(def.diagnostics(db, style_lints))
697                }
698                ModuleDef::Trait(t) => {
699                    let krate = t.krate(db);
700                    for diag in TraitItems::query_with_diagnostics(db, t.id).1.iter() {
701                        emit_def_diagnostic(db, acc, diag, edition, krate.id);
702                    }
703
704                    for item in t.items(db) {
705                        item.diagnostics(db, acc, style_lints);
706                    }
707
708                    t.all_macro_calls(db)
709                        .iter()
710                        .for_each(|&(_ast, call_id)| macro_call_diagnostics(db, call_id, acc));
711
712                    acc.extend(def.diagnostics(db, style_lints))
713                }
714                ModuleDef::Adt(adt) => {
715                    match adt {
716                        Adt::Struct(s) => {
717                            let source_map = db.struct_signature_with_source_map(s.id).1;
718                            expr_store_diagnostics(db, acc, &source_map);
719                            let source_map = &s.id.fields_with_source_map(db).1;
720                            expr_store_diagnostics(db, acc, source_map);
721                            push_ty_diagnostics(
722                                db,
723                                acc,
724                                db.field_types_with_diagnostics(s.id.into()).1.clone(),
725                                source_map,
726                            );
727                        }
728                        Adt::Union(u) => {
729                            let source_map = db.union_signature_with_source_map(u.id).1;
730                            expr_store_diagnostics(db, acc, &source_map);
731                            let source_map = &u.id.fields_with_source_map(db).1;
732                            expr_store_diagnostics(db, acc, source_map);
733                            push_ty_diagnostics(
734                                db,
735                                acc,
736                                db.field_types_with_diagnostics(u.id.into()).1.clone(),
737                                source_map,
738                            );
739                        }
740                        Adt::Enum(e) => {
741                            let source_map = db.enum_signature_with_source_map(e.id).1;
742                            expr_store_diagnostics(db, acc, &source_map);
743                            let (variants, diagnostics) = e.id.enum_variants_with_diagnostics(db);
744                            let file = e.id.lookup(db).id.file_id;
745                            let ast_id_map = db.ast_id_map(file);
746                            if let Some(diagnostics) = &diagnostics {
747                                for diag in diagnostics.iter() {
748                                    acc.push(
749                                        InactiveCode {
750                                            node: InFile::new(
751                                                file,
752                                                ast_id_map.get(diag.ast_id).syntax_node_ptr(),
753                                            ),
754                                            cfg: diag.cfg.clone(),
755                                            opts: diag.opts.clone(),
756                                        }
757                                        .into(),
758                                    );
759                                }
760                            }
761                            for &(v, _, _) in &variants.variants {
762                                let source_map = &v.fields_with_source_map(db).1;
763                                push_ty_diagnostics(
764                                    db,
765                                    acc,
766                                    db.field_types_with_diagnostics(v.into()).1.clone(),
767                                    source_map,
768                                );
769                                expr_store_diagnostics(db, acc, source_map);
770                            }
771                        }
772                    }
773                    acc.extend(def.diagnostics(db, style_lints))
774                }
775                ModuleDef::Macro(m) => emit_macro_def_diagnostics(db, acc, m),
776                ModuleDef::TypeAlias(type_alias) => {
777                    let source_map = db.type_alias_signature_with_source_map(type_alias.id).1;
778                    expr_store_diagnostics(db, acc, &source_map);
779                    push_ty_diagnostics(
780                        db,
781                        acc,
782                        db.type_for_type_alias_with_diagnostics(type_alias.id).1,
783                        &source_map,
784                    );
785                    acc.extend(def.diagnostics(db, style_lints));
786                }
787                _ => acc.extend(def.diagnostics(db, style_lints)),
788            }
789        }
790        self.legacy_macros(db).into_iter().for_each(|m| emit_macro_def_diagnostics(db, acc, m));
791
792        let interner = DbInterner::new_with(db, self.id.krate(db));
793        let infcx = interner.infer_ctxt().build(TypingMode::non_body_analysis());
794
795        let mut impl_assoc_items_scratch = vec![];
796        for impl_def in self.impl_defs(db) {
797            GenericDef::Impl(impl_def).diagnostics(db, acc);
798
799            let AnyImplId::ImplId(impl_id) = impl_def.id else {
800                continue;
801            };
802            let loc = impl_id.lookup(db);
803            let (impl_signature, source_map) = db.impl_signature_with_source_map(impl_id);
804            expr_store_diagnostics(db, acc, &source_map);
805
806            let file_id = loc.id.file_id;
807            if file_id.macro_file().is_some_and(|it| it.kind(db) == MacroKind::DeriveBuiltIn) {
808                // these expansion come from us, diagnosing them is a waste of resources
809                // FIXME: Once we diagnose the inputs to builtin derives, we should at least extract those diagnostics somehow
810                continue;
811            }
812            impl_def
813                .all_macro_calls(db)
814                .iter()
815                .for_each(|&(_ast, call_id)| macro_call_diagnostics(db, call_id, acc));
816
817            let ast_id_map = db.ast_id_map(file_id);
818
819            for diag in impl_id.impl_items_with_diagnostics(db).1.iter() {
820                emit_def_diagnostic(db, acc, diag, edition, loc.container.krate(db));
821            }
822
823            let trait_impl = impl_signature.target_trait.is_some();
824            if !trait_impl && !is_inherent_impl_coherent(db, def_map, impl_id) {
825                acc.push(IncoherentImpl { impl_: ast_id_map.get(loc.id.value), file_id }.into())
826            }
827
828            if trait_impl && !impl_def.check_orphan_rules(db) {
829                acc.push(TraitImplOrphan { impl_: ast_id_map.get(loc.id.value), file_id }.into())
830            }
831
832            let trait_ = trait_impl.then(|| impl_def.trait_(db)).flatten();
833            let mut trait_is_unsafe = trait_.is_some_and(|t| t.is_unsafe(db));
834            let impl_is_negative = impl_def.is_negative(db);
835            let impl_is_unsafe = impl_def.is_unsafe(db);
836
837            let trait_is_unresolved = trait_.is_none() && trait_impl;
838            if trait_is_unresolved {
839                // Ignore trait safety errors when the trait is unresolved, as otherwise we'll treat it as safe,
840                // which may not be correct.
841                trait_is_unsafe = impl_is_unsafe;
842            }
843
844            let drop_maybe_dangle = (|| {
845                // FIXME: This can be simplified a lot by exposing hir-ty's utils.rs::Generics helper
846                let trait_ = trait_?;
847                let drop_trait = interner.lang_items().Drop?;
848                if drop_trait != trait_.into() {
849                    return None;
850                }
851                let parent = impl_id.into();
852                let (lifetimes_attrs, type_and_consts_attrs) =
853                    AttrFlags::query_generic_params(db, parent);
854                let res = lifetimes_attrs.values().any(|it| it.contains(AttrFlags::MAY_DANGLE))
855                    || type_and_consts_attrs.values().any(|it| it.contains(AttrFlags::MAY_DANGLE));
856                Some(res)
857            })()
858            .unwrap_or(false);
859
860            match (impl_is_unsafe, trait_is_unsafe, impl_is_negative, drop_maybe_dangle) {
861                // unsafe negative impl
862                (true, _, true, _) |
863                // unsafe impl for safe trait
864                (true, false, _, false) => acc.push(TraitImplIncorrectSafety { impl_: ast_id_map.get(loc.id.value), file_id, should_be_safe: true }.into()),
865                // safe impl for unsafe trait
866                (false, true, false, _) |
867                // safe impl of dangling drop
868                (false, false, _, true) => acc.push(TraitImplIncorrectSafety { impl_: ast_id_map.get(loc.id.value), file_id, should_be_safe: false }.into()),
869                _ => (),
870            };
871
872            // Negative impls can't have items, don't emit missing items diagnostic for them
873            if let (false, Some(trait_)) = (impl_is_negative, trait_) {
874                let items = &trait_.id.trait_items(db).items;
875                let required_items = items.iter().filter(|&(_, assoc)| match *assoc {
876                    AssocItemId::FunctionId(it) => !db.function_signature(it).has_body(),
877                    AssocItemId::ConstId(id) => !db.const_signature(id).has_body(),
878                    AssocItemId::TypeAliasId(it) => db.type_alias_signature(it).ty.is_none(),
879                });
880                impl_assoc_items_scratch.extend(impl_id.impl_items(db).items.iter().cloned());
881
882                let redundant = impl_assoc_items_scratch
883                    .iter()
884                    .filter(|(name, id)| {
885                        !items.iter().any(|(impl_name, impl_item)| {
886                            discriminant(impl_item) == discriminant(id) && impl_name == name
887                        })
888                    })
889                    .map(|(name, item)| (name.clone(), AssocItem::from(*item)));
890                for (name, assoc_item) in redundant {
891                    acc.push(
892                        TraitImplRedundantAssocItems {
893                            trait_,
894                            file_id,
895                            impl_: ast_id_map.get(loc.id.value),
896                            assoc_item: (name, assoc_item),
897                        }
898                        .into(),
899                    )
900                }
901
902                let mut missing: Vec<_> = required_items
903                    .filter(|(name, id)| {
904                        !impl_assoc_items_scratch.iter().any(|(impl_name, impl_item)| {
905                            discriminant(impl_item) == discriminant(id) && impl_name == name
906                        })
907                    })
908                    .map(|(name, item)| (name.clone(), AssocItem::from(*item)))
909                    .collect();
910
911                if !missing.is_empty() {
912                    let self_ty = db.impl_self_ty(impl_id).instantiate_identity();
913                    let self_ty = structurally_normalize_ty(
914                        &infcx,
915                        self_ty,
916                        db.trait_environment(impl_id.into()),
917                    );
918                    let self_ty_is_guaranteed_unsized = matches!(
919                        self_ty.kind(),
920                        TyKind::Dynamic(..) | TyKind::Slice(..) | TyKind::Str
921                    );
922                    if self_ty_is_guaranteed_unsized {
923                        missing.retain(|(_, assoc_item)| {
924                            let assoc_item = match *assoc_item {
925                                AssocItem::Function(it) => match it.id {
926                                    AnyFunctionId::FunctionId(id) => id.into(),
927                                    AnyFunctionId::BuiltinDeriveImplMethod { .. } => {
928                                        never!("should not have an `AnyFunctionId::BuiltinDeriveImplMethod` here");
929                                        return false;
930                                    },
931                                },
932                                AssocItem::Const(it) => it.id.into(),
933                                AssocItem::TypeAlias(it) => it.id.into(),
934                            };
935                            !hir_ty::dyn_compatibility::generics_require_sized_self(db, assoc_item)
936                        });
937                    }
938                }
939
940                // HACK: When specialization is enabled in the current crate, and there exists
941                // *any* blanket impl that provides a default implementation for the missing item,
942                // suppress the missing associated item diagnostic.
943                // This can lead to false negatives when the impl in question does not actually
944                // specialize that blanket impl, but determining the exact specialization
945                // relationship here would be significantly more expensive.
946                if !missing.is_empty() {
947                    let krate = self.krate(db).id;
948                    let def_map = crate_def_map(db, krate);
949                    if def_map.is_unstable_feature_enabled(&sym::specialization)
950                        || def_map.is_unstable_feature_enabled(&sym::min_specialization)
951                    {
952                        missing.retain(|(assoc_name, assoc_item)| {
953                            let AssocItem::Function(_) = assoc_item else {
954                                return true;
955                            };
956
957                            for &impl_ in TraitImpls::for_crate(db, krate).blanket_impls(trait_.id)
958                            {
959                                if impl_ == impl_id {
960                                    continue;
961                                }
962
963                                for (name, item) in &impl_.impl_items(db).items {
964                                    let AssocItemId::FunctionId(fn_) = item else {
965                                        continue;
966                                    };
967                                    if name != assoc_name {
968                                        continue;
969                                    }
970
971                                    if db.function_signature(*fn_).is_default() {
972                                        return false;
973                                    }
974                                }
975                            }
976
977                            true
978                        });
979                    }
980                }
981
982                if !missing.is_empty() {
983                    acc.push(
984                        TraitImplMissingAssocItems {
985                            impl_: ast_id_map.get(loc.id.value),
986                            file_id,
987                            missing,
988                        }
989                        .into(),
990                    )
991                }
992                impl_assoc_items_scratch.clear();
993            }
994
995            push_ty_diagnostics(db, acc, db.impl_self_ty_with_diagnostics(impl_id).1, &source_map);
996            push_ty_diagnostics(
997                db,
998                acc,
999                db.impl_trait_with_diagnostics(impl_id).and_then(|it| it.1),
1000                &source_map,
1001            );
1002
1003            for &(_, item) in impl_id.impl_items(db).items.iter() {
1004                AssocItem::from(item).diagnostics(db, acc, style_lints);
1005            }
1006        }
1007    }

1080fn macro_call_diagnostics<'db>(
1081    db: &'db dyn HirDatabase,
1082    macro_call_id: MacroCallId,
1083    acc: &mut Vec<AnyDiagnostic<'db>>,
1084) {
1085    let Some(e) = db.parse_macro_expansion_error(macro_call_id) else {
1086        return;
1087    };
1088    let ValueResult { value: parse_errors, err } = &*e;
1089    if let Some(err) = err {
1090        let loc = db.lookup_intern_macro_call(macro_call_id);
1091        let file_id = loc.kind.file_id();
1092        let mut range = precise_macro_call_location(&loc.kind, db, loc.krate);
1093        let RenderedExpandError { message, error, kind } = err.render_to_string(db);
1094        if Some(err.span().anchor.file_id) == file_id.file_id().map(|it| it.editioned_file_id(db)) {
1095            range.value = err.span().range
1096                + db.ast_id_map(file_id).get_erased(err.span().anchor.ast_id).text_range().start();
1097        }
1098        acc.push(MacroError { range, message, error, kind }.into());
1099    }
1100
1101    if !parse_errors.is_empty() {
1102        let loc = db.lookup_intern_macro_call(macro_call_id);
1103        let range = precise_macro_call_location(&loc.kind, db, loc.krate);
1104        acc.push(MacroExpansionParseError { range, errors: parse_errors.clone() }.into())
1105    }
1106}

1124            db,

1126            &DefDiagnosticKind::MacroDefError { ast, message: e.to_string() },

1143fn emit_def_diagnostic_<'db>(
1144    db: &'db dyn HirDatabase,
1145    acc: &mut Vec<AnyDiagnostic<'db>>,
1146    diag: &DefDiagnosticKind,
1147    edition: Edition,
1148    krate: base_db::Crate,
1149) {
1150    match diag {
1151        DefDiagnosticKind::UnresolvedModule { ast: declaration, candidates } => {
1152            let decl = declaration.to_ptr(db);
1153            acc.push(
1154                UnresolvedModule {
1155                    decl: InFile::new(declaration.file_id, decl),
1156                    candidates: candidates.clone(),
1157                }
1158                .into(),
1159            )
1160        }
1161        DefDiagnosticKind::UnresolvedExternCrate { ast } => {
1162            let item = ast.to_ptr(db);
1163            acc.push(UnresolvedExternCrate { decl: InFile::new(ast.file_id, item) }.into());
1164        }
1165
1166        DefDiagnosticKind::MacroError { ast, path, err } => {
1167            let item = ast.to_ptr(db);
1168            let RenderedExpandError { message, error, kind } = err.render_to_string(db);
1169            acc.push(
1170                MacroError {
1171                    range: InFile::new(ast.file_id, item.text_range()),
1172                    message: format!("{}: {message}", path.display(db, edition)),
1173                    error,
1174                    kind,
1175                }
1176                .into(),
1177            )
1178        }
1179        DefDiagnosticKind::UnresolvedImport { id, index } => {
1180            let file_id = id.file_id;
1181
1182            let use_tree = hir_def::src::use_tree_to_ast(db, *id, *index);
1183            acc.push(
1184                UnresolvedImport { decl: InFile::new(file_id, AstPtr::new(&use_tree)) }.into(),
1185            );
1186        }
1187
1188        DefDiagnosticKind::UnconfiguredCode { ast_id, cfg, opts } => {
1189            let ast_id_map = db.ast_id_map(ast_id.file_id);
1190            let ptr = ast_id_map.get_erased(ast_id.value);
1191            acc.push(
1192                InactiveCode {
1193                    node: InFile::new(ast_id.file_id, ptr),
1194                    cfg: cfg.clone(),
1195                    opts: opts.clone(),
1196                }
1197                .into(),
1198            );
1199        }
1200        DefDiagnosticKind::UnresolvedMacroCall { ast, path } => {
1201            let location = precise_macro_call_location(ast, db, krate);
1202            acc.push(
1203                UnresolvedMacroCall {
1204                    range: location,
1205                    path: path.clone(),
1206                    is_bang: matches!(ast, MacroCallKind::FnLike { .. }),
1207                }
1208                .into(),
1209            );
1210        }
1211        DefDiagnosticKind::UnimplementedBuiltinMacro { ast } => {
1212            let node = ast.to_node(db);
1213            // Must have a name, otherwise we wouldn't emit it.
1214            let name = node.name().expect("unimplemented builtin macro with no name");
1215            acc.push(
1216                UnimplementedBuiltinMacro {
1217                    node: ast.with_value(SyntaxNodePtr::from(AstPtr::new(&name))),
1218                }
1219                .into(),
1220            );
1221        }
1222        DefDiagnosticKind::InvalidDeriveTarget { ast, id } => {
1223            let derive = id.find_attr_range(db, krate, *ast).3.path_range();
1224            acc.push(InvalidDeriveTarget { range: ast.with_value(derive) }.into());
1225        }
1226        DefDiagnosticKind::MalformedDerive { ast, id } => {
1227            let derive = id.find_attr_range(db, krate, *ast).2;
1228            acc.push(MalformedDerive { range: ast.with_value(derive) }.into());
1229        }
1230        DefDiagnosticKind::MacroDefError { ast, message } => {
1231            let node = ast.to_node(db);
1232            acc.push(
1233                MacroDefError {
1234                    node: InFile::new(ast.file_id, AstPtr::new(&node)),
1235                    name: node.name().map(|it| it.syntax().text_range()),
1236                    message: message.clone(),
1237                }
1238                .into(),
1239            );
1240        }
1241    }
1242}

1244fn precise_macro_call_location(
1245    ast: &MacroCallKind,
1246    db: &dyn HirDatabase,
1247    krate: base_db::Crate,
1248) -> InFile<TextRange> {
1249    // FIXME: maybe we actually want slightly different ranges for the different macro diagnostics
1250    // - e.g. the full attribute for macro errors, but only the name for name resolution
1251    match ast {
1252        MacroCallKind::FnLike { ast_id, .. } => {
1253            let node = ast_id.to_node(db);
1254            let range = node
1255                .path()
1256                .and_then(|it| it.segment())
1257                .and_then(|it| it.name_ref())
1258                .map(|it| it.syntax().text_range());
1259            let range = range.unwrap_or_else(|| node.syntax().text_range());
1260            ast_id.with_value(range)
1261        }
1262        MacroCallKind::Derive { ast_id, derive_attr_index, derive_index, .. } => {
1263            let range = derive_attr_index.find_derive_range(db, krate, *ast_id, *derive_index);
1264            ast_id.with_value(range)
1265        }
1266        MacroCallKind::Attr { ast_id, censored_attr_ids: attr_ids, .. } => {
1267            let attr_range = attr_ids.invoc_attr().find_attr_range(db, krate, *ast_id).2;
1268            ast_id.with_value(attr_range)
1269        }
1270    }
1271}

2390                            Ty::new_param(
2391                                self.interner,
2392                                param.id,
2393                                param.index.checked_add_signed(self.shift_by).unwrap(),
2394                            )
2395                        } else {
2396                            ty.super_fold_with(self)

2400                    fn fold_const(
2401                        &mut self,
2402                        ct: hir_ty::next_solver::Const<'db>,
2403                    ) -> hir_ty::next_solver::Const<'db> {
2404                        if let ConstKind::Param(param) = ct.kind() {
2405                            hir_ty::next_solver::Const::new_param(
2406                                self.interner,
2407                                ParamConst {
2408                                    id: param.id,
2409                                    index: param.index.checked_add_signed(self.shift_by).unwrap(),
2410                                },
2411                            )
2412                        } else {
2413                            ct.super_fold_with(self)
2414                        }
2415                    }
2416
2417                    fn fold_region(&mut self, r: Region<'db>) -> Region<'db> {
2418                        if let RegionKind::ReEarlyParam(param) = r.kind() {
2419                            Region::new_early_param(
2420                                self.interner,
2421                                EarlyParamRegion {
2422                                    id: param.id,
2423                                    index: param.index.checked_add_signed(self.shift_by).unwrap(),
2424                                },
2425                            )
2426                        } else {
2427                            r
2428                        }
2429                    }
2430                }
2431
2432                // Get the type for the trait function, as we can't get the type for the impl function
2433                // because it has not `CallableDefId`.
2434                let krate = impl_.module(db).krate(db);
2435                let interner = DbInterner::new_with(db, krate);
2436                let param_env = hir_ty::builtin_derive::param_env(interner, impl_);
2437                let env = ParamEnvAndCrate { param_env, krate };
2438                let Some(trait_method) = method.trait_method(db, impl_) else {
2439                    return Type { env, ty: Ty::new_error(interner, ErrorGuaranteed) };
2440                };
2441                // The procedure works as follows: the method may have additional generic parameters (e.g. `Hash::hash()`),
2442                // and we want them to be params of the impl method as well. So we start with the trait method identity
2443                // args and extract from them the trait method own args. In parallel, we retrieve the impl trait ref.
2444                // Now we can put our args as [...impl_trait_ref.args, ...trait_method_own_args], but we have one problem:
2445                // the args in `trait_method_own_args` use indices appropriate for the trait method, which are not necessarily
2446                // good for the impl method. So we shift them by `impl_generics_len - trait_generics_len`, which is essentially
2447                // `impl_generics_len - impl_trait_ref.args.len()`.
2448                let trait_method_fn_ptr = Ty::new_fn_ptr(
2449                    interner,
2450                    db.callable_item_signature(trait_method.into()).instantiate_identity(),
2451                );
2452                let impl_trait_ref =
2453                    hir_ty::builtin_derive::impl_trait(interner, impl_).instantiate_identity();
2454                let trait_method_args =
2455                    GenericArgs::identity_for_item(interner, trait_method.into());
2456                let trait_method_own_args = GenericArgs::new_from_iter(
2457                    interner,
2458                    trait_method_args.iter().skip(impl_trait_ref.args.len()),
2459                );
2460                let impl_params_count = hir_ty::builtin_derive::generic_params_count(db, impl_);
2461                let shift_args_by = impl_params_count as i32 - impl_trait_ref.args.len() as i32;
2462                let shifted_trait_method_own_args = trait_method_own_args
2463                    .fold_with(&mut ParamsShifter { interner, shift_by: shift_args_by });
2464                let impl_method_args = GenericArgs::new_from_iter(
2465                    interner,
2466                    impl_trait_ref.args.iter().chain(shifted_trait_method_own_args),
2467                );
2468                let impl_method_fn_ptr =
2469                    EarlyBinder::bind(trait_method_fn_ptr).instantiate(interner, impl_method_args);
2470                Type { env, ty: impl_method_fn_ptr }
2471            }
2472        }
2473    }

3546// users definitely do not want to be completed!

5362        fn is_phantom_data(db: &dyn HirDatabase, adt_id: AdtId) -> bool {
5363            match adt_id {
5364                AdtId::StructId(s) => {
5365                    let flags = db.struct_signature(s).flags;
5366                    flags.contains(StructFlags::IS_PHANTOM_DATA)
5367                }
5368                AdtId::UnionId(_) | AdtId::EnumId(_) => false,
5369            }
5370        }
5371
5372        struct Visitor<'db> {
5373            interner: DbInterner<'db>,
5374        }
5375
5376        impl<'db> TypeVisitor<DbInterner<'db>> for Visitor<'db> {
5377            type Result = ControlFlow<()>;
5378
5379            fn visit_ty(&mut self, ty: Ty<'db>) -> Self::Result {
5380                match ty.kind() {
5381                    // Reference itself
5382                    TyKind::Ref(..) => ControlFlow::Break(()),
5383
5384                    // For non-phantom_data adts we check variants/fields as well as generic parameters
5385                    TyKind::Adt(adt_def, args)
5386                        if !is_phantom_data(self.interner.db(), adt_def.def_id().0) =>
5387                    {
5388                        let _variant_id_to_fields = |id: VariantId| {
5389                            let variant_data = &id.fields(self.interner.db());
5390                            if variant_data.fields().is_empty() {
5391                                vec![]
5392                            } else {
5393                                let field_types = self.interner.db().field_types(id);
5394                                variant_data
5395                                    .fields()
5396                                    .iter()
5397                                    .map(|(idx, _)| {
5398                                        field_types[idx].get().instantiate(self.interner, args)
5399                                    })
5400                                    .filter(|it| !it.references_non_lt_error())
5401                                    .collect()
5402                            }
5403                        };
5404                        let variant_id_to_fields = |_: VariantId| vec![];
5405
5406                        let variants: Vec<Vec<Ty<'db>>> = match adt_def.def_id().0 {
5407                            AdtId::StructId(id) => {
5408                                vec![variant_id_to_fields(id.into())]
5409                            }
5410                            AdtId::EnumId(id) => id
5411                                .enum_variants(self.interner.db())
5412                                .variants
5413                                .iter()
5414                                .map(|&(variant_id, _, _)| variant_id_to_fields(variant_id.into()))
5415                                .collect(),
5416                            AdtId::UnionId(id) => {
5417                                vec![variant_id_to_fields(id.into())]
5418                            }
5419                        };
5420
5421                        variants
5422                            .into_iter()
5423                            .flat_map(|variant| variant.into_iter())
5424                            .try_for_each(|ty| ty.visit_with(self))?;
5425                        args.visit_with(self)
5426                    }
5427                    // And for `PhantomData<T>`, we check `T`.
5428                    _ => ty.super_visit_with(self),
5429                }
5430            }
5431        }
5432    }

6499            layout::FieldsShape::Union(_) => Some(0),
6500            layout::FieldsShape::Array { stride, count } => {
6501                let i = u64::try_from(field).ok()?;
6502                (i < count).then_some((stride * i).bytes())
6503            }

6510    pub fn tail_padding(&self, field_size: &mut impl FnMut(usize) -> Option<u64>) -> Option<u64> {
6511        match self.0.fields {
6512            layout::FieldsShape::Primitive => None,
6513            layout::FieldsShape::Union(_) => None,
6514            layout::FieldsShape::Array { stride, count } => count.checked_sub(1).and_then(|tail| {
6515                let tail_field_size = field_size(tail as usize)?;
6516                let offset = stride.bytes() * tail;
6517                self.0.size.bytes().checked_sub(offset)?.checked_sub(tail_field_size)
6518            }),
6519            layout::FieldsShape::Arbitrary { ref offsets, ref memory_index } => {
6520                let tail = memory_index.last_index()?;
6521                let tail_field_size = field_size(tail.0.into_raw().into_u32() as usize)?;
6522                let offset = offsets.get(tail)?.bytes();
6523                self.0.size.bytes().checked_sub(offset)?.checked_sub(tail_field_size)
6524            }
6525        }
6526    }
6527
6528    pub fn largest_padding(
6529        &self,
6530        field_size: &mut impl FnMut(usize) -> Option<u64>,
6531    ) -> Option<u64> {
6532        match self.0.fields {
6533            layout::FieldsShape::Primitive => None,
6534            layout::FieldsShape::Union(_) => None,
6535            layout::FieldsShape::Array { stride: _, count: 0 } => None,
6536            layout::FieldsShape::Array { stride, .. } => {
6537                let size = field_size(0)?;
6538                stride.bytes().checked_sub(size)
6539            }
6540            layout::FieldsShape::Arbitrary { ref offsets, ref memory_index } => {
6541                let mut reverse_index = vec![None; memory_index.len()];
6542                for (src, (mem, offset)) in memory_index.iter().zip(offsets.iter()).enumerate() {
6543                    reverse_index[*mem as usize] = Some((src, offset.bytes()));
6544                }
6545                if reverse_index.iter().any(|it| it.is_none()) {
6546                    stdx::never!();
6547                    return None;
6548                }
6549                reverse_index
6550                    .into_iter()
6551                    .flatten()
6552                    .chain(std::iter::once((0, self.0.size.bytes())))
6553                    .tuple_windows()
6554                    .filter_map(|((i, start), (_, end))| {
6555                        let size = field_size(i)?;
6556                        end.checked_sub(start)?.checked_sub(size)
6557                    })
6558                    .max()
6559            }
6560        }
6561    }

6567                    TagEncoding::Direct => tag.size(&*self.1).bytes_usize(),

7033pub fn resolve_absolute_path<'a, I: Iterator<Item = Symbol> + Clone + 'a>(
7034    db: &'a dyn HirDatabase,
7035    mut segments: I,
7036) -> impl Iterator<Item = ItemInNs> + use<'a, I> {
7037    segments
7038        .next()
7039        .into_iter()
7040        .flat_map(move |crate_name| {
7041            db.all_crates()
7042                .iter()
7043                .filter(|&krate| {
7044                    krate
7045                        .extra_data(db)
7046                        .display_name
7047                        .as_ref()
7048                        .is_some_and(|name| *name.crate_name().symbol() == crate_name)
7049                })
7050                .filter_map(|&krate| {
7051                    let segments = segments.clone();
7052                    let mut def_map = crate_def_map(db, krate);
7053                    let mut module = &def_map[def_map.root_module_id()];
7054                    let mut segments = segments.with_position().peekable();
7055                    while let Some((_, segment)) = segments.next_if(|&(position, _)| {
7056                        !matches!(position, itertools::Position::Last | itertools::Position::Only)
7057                    }) {
7058                        let res = module
7059                            .scope
7060                            .get(&Name::new_symbol_root(segment))
7061                            .take_types()
7062                            .and_then(|res| match res {
7063                                ModuleDefId::ModuleId(it) => Some(it),
7064                                _ => None,
7065                            })?;
7066                        def_map = res.def_map(db);
7067                        module = &def_map[res];
7068                    }
7069                    let (_, item_name) = segments.next()?;
7070                    let res = module.scope.get(&Name::new_symbol_root(item_name));
7071                    Some(res.iter_items().map(|(item, _)| item.into()))
7072                })
7073                .collect::<Vec<_>>()
7074        })
7075        .flatten()
7076}
