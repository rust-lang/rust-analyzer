34#[derive(Debug, Clone, Copy)]
35pub enum AttrsOwner {
36    AttrDef(AttrDefId),
37    Field(FieldId),
38    LifetimeParam(LifetimeParamId),
39    TypeOrConstParam(TypeOrConstParamId),
40    /// Things that do not have attributes. Used for builtin derives.
41    Dummy,
42}
43
44impl AttrsOwner {
45    #[inline]
46    fn attr_def(&self) -> Option<AttrDefId> {
47        match self {
48            AttrsOwner::AttrDef(it) => Some(*it),
49            _ => None,
50        }
51    }
52}

161pub trait HasAttrs: Sized {
162    #[inline]
163    fn attrs(self, db: &dyn HirDatabase) -> AttrsWithOwner {
164        match self.attr_id(db) {
165            AttrsOwner::AttrDef(it) => AttrsWithOwner::new(db, it),
166            AttrsOwner::Field(it) => AttrsWithOwner::new_field(db, it),
167            AttrsOwner::LifetimeParam(it) => AttrsWithOwner::new_lifetime_param(db, it),
168            AttrsOwner::TypeOrConstParam(it) => AttrsWithOwner::new_type_or_const_param(db, it),
169            AttrsOwner::Dummy => {
170                AttrsWithOwner { attrs: AttrFlags::empty(), owner: AttrsOwner::Dummy }
171            }
172        }
173    }
174
175    #[doc(hidden)]
176    fn attr_id(self, db: &dyn HirDatabase) -> AttrsOwner;
177
178    #[inline]
179    fn hir_docs(self, db: &dyn HirDatabase) -> Option<&Docs> {
180        match self.attr_id(db) {
181            AttrsOwner::AttrDef(it) => AttrFlags::docs(db, it).as_deref(),
182            AttrsOwner::Field(it) => AttrFlags::field_docs(db, it),
183            AttrsOwner::LifetimeParam(_) | AttrsOwner::TypeOrConstParam(_) | AttrsOwner::Dummy => {
184                None
185            }
186        }
187    }
188}

287/// Resolves the item `link` points to in the scope of `def`.
288pub fn resolve_doc_path_on(
289    db: &dyn HirDatabase,
290    def: impl HasAttrs + Copy,
291    link: &str,
292    ns: Option<Namespace>,
293    is_inner_doc: IsInnerDoc,
294) -> Option<DocLinkDef> {
295    resolve_doc_path_on_(db, link, def.attr_id(db), ns, is_inner_doc)
296}
297
298fn resolve_doc_path_on_(
299    db: &dyn HirDatabase,
300    link: &str,
301    attr_id: AttrsOwner,
302    ns: Option<Namespace>,
303    is_inner_doc: IsInnerDoc,
304) -> Option<DocLinkDef> {
305    let resolver = match attr_id {
306        AttrsOwner::AttrDef(AttrDefId::ModuleId(it)) => {
307            if is_inner_doc.yes() {
308                it.resolver(db)
309            } else if let Some(parent) = Module::from(it).parent(db) {
310                parent.id.resolver(db)
311            } else {
312                it.resolver(db)
313            }
314        }
315        AttrsOwner::AttrDef(AttrDefId::AdtId(it)) => it.resolver(db),
316        AttrsOwner::AttrDef(AttrDefId::FunctionId(it)) => it.resolver(db),
317        AttrsOwner::AttrDef(AttrDefId::EnumVariantId(it)) => it.resolver(db),
318        AttrsOwner::AttrDef(AttrDefId::StaticId(it)) => it.resolver(db),
319        AttrsOwner::AttrDef(AttrDefId::ConstId(it)) => it.resolver(db),
320        AttrsOwner::AttrDef(AttrDefId::TraitId(it)) => it.resolver(db),
321        AttrsOwner::AttrDef(AttrDefId::TypeAliasId(it)) => it.resolver(db),
322        AttrsOwner::AttrDef(AttrDefId::ImplId(it)) => it.resolver(db),
323        AttrsOwner::AttrDef(AttrDefId::ExternBlockId(it)) => it.resolver(db),
324        AttrsOwner::AttrDef(AttrDefId::UseId(it)) => it.resolver(db),
325        AttrsOwner::AttrDef(AttrDefId::MacroId(it)) => it.resolver(db),
326        AttrsOwner::AttrDef(AttrDefId::ExternCrateId(it)) => it.resolver(db),
327        AttrsOwner::Field(it) => it.parent.resolver(db),
328        AttrsOwner::LifetimeParam(_) | AttrsOwner::TypeOrConstParam(_) | AttrsOwner::Dummy => {
329            return None;
330        }
331    };
332
333    let mut modpath = doc_modpath_from_str(link)?;
334
335    let resolved = resolver.resolve_module_path_in_items(db, &modpath);
336    if resolved.is_none() {
337        let last_name = modpath.pop_segment()?;
338        resolve_assoc_or_field(db, resolver, modpath, last_name, ns)
339    } else {
340        let def = match ns {
341            Some(Namespace::Types) => resolved.take_types(),
342            Some(Namespace::Values) => resolved.take_values(),
343            Some(Namespace::Macros) => resolved.take_macros().map(ModuleDefId::MacroId),
344            None => resolved.iter_items().next().map(|(it, _)| match it {
345                ItemInNs::Types(it) => it,
346                ItemInNs::Values(it) => it,
347                ItemInNs::Macros(it) => ModuleDefId::MacroId(it),
348            }),
349        };
350        Some(DocLinkDef::ModuleDef(def?.into()))
351    }
352}

354fn resolve_assoc_or_field(
355    db: &dyn HirDatabase,
356    resolver: Resolver<'_>,
357    path: ModPath,
358    name: Name,
359    ns: Option<Namespace>,
360) -> Option<DocLinkDef> {
361    let path = Path::from_known_path_with_no_generic(path);
362    let base_def = resolver.resolve_path_in_type_ns_fully(db, &path)?;
363
364    let handle_trait = |id: TraitId| {
365        // Doc paths in this context may only resolve to an item of this trait
366        // (i.e. no items of its supertraits), so we need to handle them here
367        // independently of others.
368        id.trait_items(db).items.iter().find(|it| it.0 == name).map(|(_, assoc_id)| {
369            let def = match *assoc_id {
370                AssocItemId::FunctionId(it) => ModuleDef::Function(it.into()),
371                AssocItemId::ConstId(it) => ModuleDef::Const(it.into()),
372                AssocItemId::TypeAliasId(it) => ModuleDef::TypeAlias(it.into()),
373            };
374            DocLinkDef::ModuleDef(def)
375        })
376    };
377    let ty = match base_def {
378        TypeNs::SelfType(id) => Impl::from(id).self_ty(db),
379        TypeNs::GenericParam(param) => {
380            let generic_params = db.generic_params(param.parent());
381            if generic_params[param.local_id()].is_trait_self() {
382                // `Self::assoc` in traits should refer to the trait itself.
383                let parent_trait = |container| match container {
384                    ItemContainerId::TraitId(trait_) => handle_trait(trait_),
385                    _ => {
386                        never!("container {container:?} should be a trait");
387                        None
388                    }
389                };
390                return match param.parent() {
391                    GenericDefId::TraitId(trait_) => handle_trait(trait_),
392                    GenericDefId::ConstId(it) => parent_trait(it.loc(db).container),
393                    GenericDefId::FunctionId(it) => parent_trait(it.loc(db).container),
394                    GenericDefId::TypeAliasId(it) => parent_trait(it.loc(db).container),
395                    _ => {
396                        never!("type param {param:?} should belong to a trait");
397                        None
398                    }
399                };
400            }
401
402            // Even if this generic parameter has some trait bounds, rustdoc doesn't
403            // resolve `name` to trait items.
404            return None;
405        }
406        TypeNs::AdtId(id) | TypeNs::AdtSelfType(id) => Adt::from(id).ty(db),
407        TypeNs::EnumVariantId(id) => {
408            // Enum variants don't have path candidates.
409            let variant = Variant::from(id);
410            return resolve_field(db, variant.into(), name, ns);
411        }
412        TypeNs::TypeAliasId(id) => {
413            let alias = TypeAlias::from(id);
414            if alias.as_assoc_item(db).is_some() {
415                // We don't normalize associated type aliases, so we have nothing to
416                // resolve `name` to.
417                return None;
418            }
419            alias.ty(db)
420        }
421        TypeNs::BuiltinType(id) => BuiltinType::from(id).ty(db),
422        TypeNs::TraitId(id) => return handle_trait(id),
423        TypeNs::ModuleId(_) => {
424            return None;
425        }
426    };
427
428    // Resolve inherent items first, then trait items, then fields.
429    if let Some(assoc_item_def) = resolve_assoc_item(db, &ty, &name, ns) {
430        return Some(assoc_item_def);
431    }
432
433    if let Some(impl_trait_item_def) = resolve_impl_trait_item(db, resolver, &ty, &name, ns) {
434        return Some(impl_trait_item_def);
435    }
436
437    let variant_def = match ty.as_adt()? {
438        Adt::Struct(it) => it.into(),
439        Adt::Union(it) => it.into(),
440        Adt::Enum(enum_) => {
441            // Can happen on `Self::Variant` (otherwise would be fully resolved by the resolver).
442            return enum_
443                .id
444                .enum_variants(db)
445                .variant(&name)
446                .map(|variant| DocLinkDef::ModuleDef(ModuleDef::Variant(variant.into())));
447        }
448    };
449    resolve_field(db, variant_def, name, ns)
450}

452fn resolve_assoc_item<'db>(
453    db: &'db dyn HirDatabase,
454    ty: &Type<'db>,
455    name: &Name,
456    ns: Option<Namespace>,
457) -> Option<DocLinkDef> {
458    ty.iterate_assoc_items(db, move |assoc_item| {
459        if assoc_item.name(db)? != *name {
460            return None;
461        }
462        as_module_def_if_namespace_matches(assoc_item, ns)
463    })
464}

466fn resolve_impl_trait_item<'db>(
467    db: &'db dyn HirDatabase,
468    resolver: Resolver<'_>,
469    ty: &Type<'db>,
470    name: &Name,
471    ns: Option<Namespace>,
472) -> Option<DocLinkDef> {
473    let krate = ty.krate(db);
474    let environment = crate::param_env_from_resolver(db, &resolver);
475    let traits_in_scope = resolver.traits_in_scope(db);
476
477    // `ty.iterate_path_candidates()` require a scope, which is not available when resolving
478    // attributes here. Use path resolution directly instead.
479    //
480    // FIXME: resolve type aliases (which are not yielded by iterate_path_candidates)
481    let interner = DbInterner::new_with(db, environment.krate);
482    let infcx = interner.infer_ctxt().build(TypingMode::PostAnalysis);
483    let unstable_features =
484        MethodResolutionUnstableFeatures::from_def_map(resolver.top_level_def_map());
485    let ctx = MethodResolutionContext {
486        infcx: &infcx,
487        resolver: &resolver,
488        param_env: environment.param_env,
489        traits_in_scope: &traits_in_scope,
490        edition: krate.edition(db),
491        unstable_features: &unstable_features,
492    };
493    let resolution = ctx.probe_for_name(method_resolution::Mode::Path, name.clone(), ty.ty);
494    let resolution = match resolution {
495        Ok(resolution) => resolution.item,
496        Err(MethodError::PrivateMatch(resolution)) => resolution.item,
497        _ => return None,
498    };
499    let resolution = match resolution {
500        CandidateId::FunctionId(id) => AssocItem::Function(id.into()),
501        CandidateId::ConstId(id) => AssocItem::Const(id.into()),
502    };
503    as_module_def_if_namespace_matches(resolution, ns)
504}

506fn resolve_field(
507    db: &dyn HirDatabase,
508    def: VariantDef,
509    name: Name,
510    ns: Option<Namespace>,
511) -> Option<DocLinkDef> {
512    if let Some(Namespace::Types | Namespace::Macros) = ns {
513        return None;
514    }
515    def.fields(db).into_iter().find(|f| f.name(db) == name).map(DocLinkDef::Field)
516}
517
518fn as_module_def_if_namespace_matches(
519    assoc_item: AssocItem,
520    ns: Option<Namespace>,
521) -> Option<DocLinkDef> {
522    let (def, expected_ns) = match assoc_item {
523        AssocItem::Function(it) => (ModuleDef::Function(it), Namespace::Values),
524        AssocItem::Const(it) => (ModuleDef::Const(it), Namespace::Values),
525        AssocItem::TypeAlias(it) => (ModuleDef::TypeAlias(it), Namespace::Types),
526    };
527
528    (ns.unwrap_or(expected_ns) == expected_ns).then_some(DocLinkDef::ModuleDef(def))
529}

531fn doc_modpath_from_str(link: &str) -> Option<ModPath> {
532    // FIXME: this is not how we should get a mod path here.
533    let try_get_modpath = |link: &str| {
534        let mut parts = link.split("::");
535        let mut first_segment = None;
536        let kind = match parts.next()? {
537            "" => PathKind::Abs,
538            "crate" => PathKind::Crate,
539            "self" => PathKind::SELF,
540            "super" => {
541                let mut deg = 1;
542                for segment in parts.by_ref() {
543                    if segment == "super" {
544                        deg += 1;
545                    } else {
546                        first_segment = Some(segment);
547                        break;
548                    }
549                }
550                PathKind::Super(deg)
551            }
552            segment => {
553                first_segment = Some(segment);
554                PathKind::Plain
555            }
556        };
557        let parts = first_segment.into_iter().chain(parts).map(|segment| match segment.parse() {
558            Ok(idx) => Name::new_tuple_field(idx),
559            Err(_) => Name::new_root(segment.split_once('<').map_or(segment, |it| it.0)),
560        });
561        Some(ModPath::from_segments(kind, parts))
562    };
563    try_get_modpath(link)
564}
