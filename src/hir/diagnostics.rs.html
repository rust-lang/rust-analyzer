42#[derive(Debug, Clone)]
43pub enum SpanAst {
44    Expr(ast::Expr),
45    Pat(ast::Pat),
46    Type(ast::Type),
47}
48const _: () = {
49    use syntax::ast::*;
50    impl_from!(Expr, Pat, Type for SpanAst);
51};
52
53impl From<Either<ast::Expr, ast::Pat>> for SpanAst {
54    fn from(value: Either<ast::Expr, ast::Pat>) -> Self {
55        match value {
56            Either::Left(it) => it.into(),
57            Either::Right(it) => it.into(),
58        }
59    }
60}
61
62impl ast::AstNode for SpanAst {
63    fn can_cast(kind: syntax::SyntaxKind) -> bool {
64        ast::Expr::can_cast(kind) || ast::Pat::can_cast(kind) || ast::Type::can_cast(kind)
65    }
66
67    fn cast(syntax: syntax::SyntaxNode) -> Option<Self> {
68        ast::Expr::cast(syntax.clone())
69            .map(SpanAst::Expr)
70            .or_else(|| ast::Pat::cast(syntax.clone()).map(SpanAst::Pat))
71            .or_else(|| ast::Type::cast(syntax).map(SpanAst::Type))
72    }
73
74    fn syntax(&self) -> &syntax::SyntaxNode {
75        match self {
76            SpanAst::Expr(it) => it.syntax(),
77            SpanAst::Pat(it) => it.syntax(),
78            SpanAst::Type(it) => it.syntax(),
79        }
80    }
81}

85macro_rules! diagnostics {
86    ($AnyDiagnostic:ident <$db:lifetime> -> $($diag:ident $(<$lt:lifetime>)?,)*) => {
87        #[derive(Debug)]
88        pub enum $AnyDiagnostic<$db> {$(
89            $diag(Box<$diag $(<$lt>)?>),
90        )*}
91
92        $(
93            impl<$db> From<$diag $(<$lt>)?> for $AnyDiagnostic<$db> {
94                fn from(d: $diag $(<$lt>)?) -> $AnyDiagnostic<$db> {
95                    $AnyDiagnostic::$diag(Box::new(d))
96                }
97            }
98        )*
99    };
100}

576impl<'db> AnyDiagnostic<'db> {
577    pub(crate) fn body_validation_diagnostic(
578        db: &'db dyn HirDatabase,
579        diagnostic: BodyValidationDiagnostic,
580        source_map: &hir_def::expr_store::BodySourceMap,
581    ) -> Option<AnyDiagnostic<'db>> {
582        match diagnostic {
583            BodyValidationDiagnostic::RecordMissingFields { record, variant, missed_fields } => {
584                let variant_data = variant.fields(db);
585                let missed_fields = missed_fields
586                    .into_iter()
587                    .map(|idx| {
588                        (
589                            variant_data.fields()[idx].name.clone(),
590                            Field { parent: variant.into(), id: idx },
591                        )
592                    })
593                    .collect();
594
595                let record = match record {
596                    Either::Left(record_expr) => source_map.expr_syntax(record_expr).ok()?,
597                    Either::Right(record_pat) => source_map.pat_syntax(record_pat).ok()?,
598                };
599                let file = record.file_id;
600                let root = record.file_syntax(db);
601                match record.value.to_node(&root) {
602                    Either::Left(ast::Expr::RecordExpr(record_expr))
603                        if record_expr.record_expr_field_list().is_some() =>
604                    {
605                        let field_list_parent_path =
606                            record_expr.path().map(|path| AstPtr::new(&path));
607                        return Some(
608                            MissingFields {
609                                file,
610                                field_list_parent: AstPtr::new(&Either::Left(record_expr)),
611                                field_list_parent_path,
612                                missed_fields,
613                            }
614                            .into(),
615                        );
616                    }
617                    Either::Right(ast::Pat::RecordPat(record_pat))
618                        if record_pat.record_pat_field_list().is_some() =>
619                    {
620                        let field_list_parent_path =
621                            record_pat.path().map(|path| AstPtr::new(&path));
622                        return Some(
623                            MissingFields {
624                                file,
625                                field_list_parent: AstPtr::new(&Either::Right(record_pat)),
626                                field_list_parent_path,
627                                missed_fields,
628                            }
629                            .into(),
630                        );
631                    }
632                    _ => {}
633                }
634            }
635            BodyValidationDiagnostic::ReplaceFilterMapNextWithFindMap { method_call_expr } => {
636                if let Ok(next_source_ptr) = source_map.expr_syntax(method_call_expr) {
637                    return Some(
638                        ReplaceFilterMapNextWithFindMap {
639                            file: next_source_ptr.file_id,
640                            next_expr: next_source_ptr.value.cast()?,
641                        }
642                        .into(),
643                    );
644                }
645            }
646            BodyValidationDiagnostic::MissingMatchArms { match_expr, uncovered_patterns } => {
647                if let Ok(source_ptr) = source_map.expr_syntax(match_expr)
648                    && let root = source_ptr.file_syntax(db)
649                    && let Either::Left(ast::Expr::MatchExpr(match_expr)) =
650                        source_ptr.value.to_node(&root)
651                    && let Some(scrut_expr) = match_expr.expr()
652                    && match_expr.match_arm_list().is_some()
653                {
654                    return Some(
655                        MissingMatchArms {
656                            scrutinee_expr: InFile::new(
657                                source_ptr.file_id,
658                                AstPtr::new(&scrut_expr),
659                            ),
660                            uncovered_patterns,
661                        }
662                        .into(),
663                    );
664                }
665            }
666            BodyValidationDiagnostic::NonExhaustiveLet { pat, uncovered_patterns } => {
667                if let Ok(source_ptr) = source_map.pat_syntax(pat)
668                    && let Some(ast_pat) = source_ptr.value.cast::<ast::Pat>()
669                {
670                    return Some(
671                        NonExhaustiveLet {
672                            pat: InFile::new(source_ptr.file_id, ast_pat),
673                            uncovered_patterns,
674                        }
675                        .into(),
676                    );
677                }
678            }
679            BodyValidationDiagnostic::RemoveTrailingReturn { return_expr } => {
680                if let Ok(source_ptr) = source_map.expr_syntax(return_expr)
681                    // Filters out desugared return expressions (e.g. desugared try operators).
682                    && let Some(ptr) = source_ptr.value.cast::<ast::ReturnExpr>()
683                {
684                    return Some(
685                        RemoveTrailingReturn { return_expr: InFile::new(source_ptr.file_id, ptr) }
686                            .into(),
687                    );
688                }
689            }
690            BodyValidationDiagnostic::RemoveUnnecessaryElse { if_expr } => {
691                if let Ok(source_ptr) = source_map.expr_syntax(if_expr)
692                    && let Some(ptr) = source_ptr.value.cast::<ast::IfExpr>()
693                {
694                    return Some(
695                        RemoveUnnecessaryElse { if_expr: InFile::new(source_ptr.file_id, ptr) }
696                            .into(),
697                    );
698                }
699            }
700            BodyValidationDiagnostic::UnusedMustUse { expr } => {
701                if let Ok(source_ptr) = source_map.expr_syntax(expr) {
702                    return Some(UnusedMustUse { expr: source_ptr }.into());
703                }
704            }
705        }
706        None
707    }

709    pub(crate) fn inference_diagnostic(
710        db: &'db dyn HirDatabase,
711        def: DefWithBodyId,
712        d: &'db InferenceDiagnostic,
713        source_map: &hir_def::expr_store::BodySourceMap,
714        sig_map: &hir_def::expr_store::ExpressionStoreSourceMap,
715        env: ParamEnvAndCrate<'db>,
716    ) -> Option<AnyDiagnostic<'db>> {
717        let expr_syntax = |expr| {
718            source_map
719                .expr_syntax(expr)
720                .inspect_err(|_| stdx::never!("inference diagnostic in desugared expr"))
721                .ok()
722        };
723        let pat_syntax = |pat| {
724            source_map
725                .pat_syntax(pat)
726                .inspect_err(|_| stdx::never!("inference diagnostic in desugared pattern"))
727                .ok()
728        };
729        let type_syntax = |pat| {
730            source_map
731                .type_syntax(pat)
732                .inspect_err(|_| stdx::never!("inference diagnostic in desugared type"))
733                .ok()
734        };
735        let expr_or_pat_syntax = |id| match id {
736            ExprOrPatId::ExprId(expr) => expr_syntax(expr),
737            ExprOrPatId::PatId(pat) => pat_syntax(pat),
738        };
739        let span_syntax = |span| match span {
740            hir_ty::Span::ExprId(idx) => expr_syntax(idx).map(|it| it.upcast()),
741            hir_ty::Span::PatId(idx) => pat_syntax(idx).map(|it| it.upcast()),
742            hir_ty::Span::TypeRefId(idx) => type_syntax(idx).map(|it| it.upcast()),
743            hir_ty::Span::BindingId(idx) => {
744                pat_syntax(source_map.patterns_for_binding(idx)[0]).map(|it| it.upcast())
745            }
746            hir_ty::Span::Dummy => {
747                never!("should never create a diagnostic for dummy spans");
748                None
749            }
750        };
751        Some(match d {
752            &InferenceDiagnostic::NoSuchField { field: expr, private, variant } => {
753                let expr_or_pat = match expr {
754                    ExprOrPatId::ExprId(expr) => {
755                        source_map.field_syntax(expr).map(AstPtr::wrap_left)
756                    }
757                    ExprOrPatId::PatId(pat) => source_map.pat_field_syntax(pat),
758                };
759                let private = private.map(|id| Field { id, parent: variant.into() });
760                NoSuchField { field: expr_or_pat, private, variant }.into()
761            }
762            &InferenceDiagnostic::MismatchedArrayPatLen { pat, expected, found, has_rest } => {
763                let pat = pat_syntax(pat)?.map(Into::into);
764                MismatchedArrayPatLen { pat, expected, found, has_rest }.into()
765            }
766            &InferenceDiagnostic::MismatchedArgCount { call_expr, expected, found } => {
767                MismatchedArgCount { call_expr: expr_syntax(call_expr)?, expected, found }.into()
768            }
769            &InferenceDiagnostic::PrivateField { expr, field } => {
770                let expr = expr_syntax(expr)?;
771                let field = field.into();
772                PrivateField { expr, field }.into()
773            }
774            &InferenceDiagnostic::PrivateAssocItem { id, item } => {
775                let expr_or_pat = expr_or_pat_syntax(id)?;
776                let item = item.into();
777                PrivateAssocItem { expr_or_pat, item }.into()
778            }
779            InferenceDiagnostic::ExpectedFunction { call_expr, found } => {
780                let call_expr = expr_syntax(*call_expr)?;
781                ExpectedFunction { call: call_expr, found: Type::new(db, def, found.as_ref()) }
782                    .into()
783            }
784            InferenceDiagnostic::UnresolvedField {
785                expr,
786                receiver,
787                name,
788                method_with_same_name_exists,
789            } => {
790                let expr = expr_syntax(*expr)?;
791                UnresolvedField {
792                    expr,
793                    name: name.clone(),
794                    receiver: Type::new(db, def, receiver.as_ref()),
795                    method_with_same_name_exists: *method_with_same_name_exists,
796                }
797                .into()
798            }
799            InferenceDiagnostic::UnresolvedMethodCall {
800                expr,
801                receiver,
802                name,
803                field_with_same_name,
804                assoc_func_with_same_name,
805            } => {
806                let expr = expr_syntax(*expr)?;
807                UnresolvedMethodCall {
808                    expr,
809                    name: name.clone(),
810                    receiver: Type::new(db, def, receiver.as_ref()),
811                    field_with_same_name: field_with_same_name
812                        .as_ref()
813                        .map(|ty| Type::new(db, def, ty.as_ref())),
814                    assoc_func_with_same_name: assoc_func_with_same_name.map(Into::into),
815                }
816                .into()
817            }
818            &InferenceDiagnostic::UnresolvedAssocItem { id } => {
819                let expr_or_pat = expr_or_pat_syntax(id)?;
820                UnresolvedAssocItem { expr_or_pat }.into()
821            }
822            &InferenceDiagnostic::UnresolvedIdent { id } => {
823                let node = match id {
824                    ExprOrPatId::ExprId(id) => match source_map.expr_syntax(id) {
825                        Ok(syntax) => syntax.map(|it| (it, None)),
826                        Err(SyntheticSyntax) => source_map
827                            .format_args_implicit_capture(id)?
828                            .map(|(node, range)| (node.wrap_left(), Some(range))),
829                    },
830                    ExprOrPatId::PatId(id) => pat_syntax(id)?.map(|it| (it, None)),
831                };
832                UnresolvedIdent { node }.into()
833            }
834            &InferenceDiagnostic::BreakOutsideOfLoop { expr, is_break, bad_value_break } => {
835                let expr = expr_syntax(expr)?;
836                BreakOutsideOfLoop { expr, is_break, bad_value_break }.into()
837            }
838            &InferenceDiagnostic::NonExhaustiveRecordExpr { expr } => {
839                NonExhaustiveRecordExpr { expr: expr_syntax(expr)? }.into()
840            }
841            InferenceDiagnostic::TypedHole { expr, expected } => {
842                let expr = expr_syntax(*expr)?;
843                TypedHole { expr, expected: Type::new(db, def, expected.as_ref()) }.into()
844            }
845            &InferenceDiagnostic::MismatchedTupleStructPatArgCount { pat, expected, found } => {
846                let InFile { file_id, value } = pat_syntax(pat)?;
847                // cast from Either<Pat, SelfParam> -> Either<_, Pat>
848                let ptr = AstPtr::try_from_raw(value.syntax_node_ptr())?;
849                let expr_or_pat = InFile { file_id, value: ptr };
850                MismatchedTupleStructPatArgCount { expr_or_pat, expected, found }.into()
851            }
852            InferenceDiagnostic::CastToUnsized { expr, cast_ty } => {
853                let expr = expr_syntax(*expr)?;
854                CastToUnsized { expr, cast_ty: Type::new(db, def, cast_ty.as_ref()) }.into()
855            }
856            InferenceDiagnostic::InvalidCast { expr, error, expr_ty, cast_ty } => {
857                let expr = expr_syntax(*expr)?;
858                let expr_ty = Type::new(db, def, expr_ty.as_ref());
859                let cast_ty = Type::new(db, def, cast_ty.as_ref());
860                InvalidCast { expr, error: *error, expr_ty, cast_ty }.into()
861            }
862            InferenceDiagnostic::TyDiagnostic { source, diag } => {
863                let source_map = match source {
864                    InferenceTyDiagnosticSource::Body => source_map,
865                    InferenceTyDiagnosticSource::Signature => sig_map,
866                };
867                Self::ty_diagnostic(diag, source_map, db)?
868            }
869            InferenceDiagnostic::PathDiagnostic { node, diag } => {
870                let source = expr_or_pat_syntax(*node)?;
871                let syntax = source.value.to_node(&db.parse_or_expand(source.file_id));
872                let path = match_ast! {
873                    match (syntax.syntax()) {
874                        ast::RecordExpr(it) => it.path()?,
875                        ast::RecordPat(it) => it.path()?,
876                        ast::TupleStructPat(it) => it.path()?,
877                        ast::PathExpr(it) => it.path()?,
878                        ast::PathPat(it) => it.path()?,
879                        _ => return None,
880                    }
881                };
882                Self::path_diagnostic(diag, source.with_value(path))?
883            }
884            &InferenceDiagnostic::MethodCallIncorrectGenericsLen {
885                expr,
886                provided_count,
887                expected_count,
888                kind,
889                def,
890            } => {
891                let syntax = expr_syntax(expr)?;
892                let file_id = syntax.file_id;
893                let syntax =
894                    syntax.with_value(syntax.value.cast::<ast::MethodCallExpr>()?).to_node(db);
895                let generics_or_name = syntax
896                    .generic_arg_list()
897                    .map(Either::Left)
898                    .or_else(|| syntax.name_ref().map(Either::Right))?;
899                let generics_or_name = InFile::new(file_id, AstPtr::new(&generics_or_name));
900                IncorrectGenericsLen {
901                    generics_or_segment: generics_or_name,
902                    kind,
903                    provided: provided_count,
904                    expected: expected_count,
905                    def: def.into(),
906                }
907                .into()
908            }
909            &InferenceDiagnostic::MethodCallIncorrectGenericsOrder {
910                expr,
911                param_id,
912                arg_idx,
913                has_self_arg,
914            } => {
915                let syntax = expr_syntax(expr)?;
916                let file_id = syntax.file_id;
917                let syntax =
918                    syntax.with_value(syntax.value.cast::<ast::MethodCallExpr>()?).to_node(db);
919                let generic_args = syntax.generic_arg_list()?;
920                let provided_arg = hir_generic_arg_to_ast(&generic_args, arg_idx, has_self_arg)?;
921                let provided_arg = InFile::new(file_id, AstPtr::new(&provided_arg));
922                let expected_kind = GenericArgKind::from_id(param_id);
923                IncorrectGenericsOrder { provided_arg, expected_kind }.into()
924            }
925            &InferenceDiagnostic::InvalidLhsOfAssignment { lhs } => {
926                let lhs = expr_syntax(lhs)?;
927                InvalidLhsOfAssignment { lhs }.into()
928            }
929            &InferenceDiagnostic::TypeMustBeKnown { at_point, ref top_term } => {
930                let at_point = span_syntax(at_point)?;
931                let top_term = top_term.as_ref().map(|top_term| match top_term.as_ref().kind() {
932                    rustc_type_ir::GenericArgKind::Type(ty) => Either::Left(Type {
933                        ty,
934                        env: crate::body_param_env_from_has_crate(db, def),
935                    }),
936                    // FIXME: Printing the const to string is definitely not the correct thing to do here.
937                    rustc_type_ir::GenericArgKind::Const(konst) => Either::Right(
938                        konst.display(db, DisplayTarget::from_crate(db, def.krate(db))).to_string(),
939                    ),
940                    rustc_type_ir::GenericArgKind::Lifetime(_) => {
941                        unreachable!("we currently don't emit TypeMustBeKnown for lifetimes")
942                    }
943                });
944                TypeMustBeKnown { at_point, top_term }.into()
945            }
946            &InferenceDiagnostic::UnionExprMustHaveExactlyOneField { expr } => {
947                let expr = expr_syntax(expr)?;
948                UnionExprMustHaveExactlyOneField { expr }.into()
949            }
950            InferenceDiagnostic::TypeMismatch { node, expected, found } => {
951                let expr_or_pat = expr_or_pat_syntax(*node)?;
952                TypeMismatch {
953                    expr_or_pat,
954                    expected: Type { env, ty: expected.as_ref() },
955                    actual: Type { env, ty: found.as_ref() },
956                }
957                .into()
958            }
959            InferenceDiagnostic::SolverDiagnostic(d) => {
960                let span = span_syntax(d.span)?;
961                Self::solver_diagnostic(db, &d.kind, span, env)?
962            }
963        })
964    }

966    fn solver_diagnostic(
967        db: &'db dyn HirDatabase,
968        d: &'db SolverDiagnosticKind,
969        span: SpanSyntax,
970        env: ParamEnvAndCrate<'db>,
971    ) -> Option<AnyDiagnostic<'db>> {
972        let interner = DbInterner::new_no_crate(db);
973        Some(match d {
974            SolverDiagnosticKind::TraitUnimplemented { trait_predicate, root_trait_predicate } => {
975                let trait_predicate =
976                    crate::TraitPredicate { inner: trait_predicate.get(interner), env };
977                let root_trait_predicate =
978                    root_trait_predicate.as_ref().map(|root_trait_predicate| {
979                        crate::TraitPredicate { inner: root_trait_predicate.get(interner), env }
980                    });
981                UnimplementedTrait { span, trait_predicate, root_trait_predicate }.into()
982            }
983        })
984    }

986    fn path_diagnostic(
987        diag: &PathLoweringDiagnostic,
988        path: InFile<ast::Path>,
989    ) -> Option<AnyDiagnostic<'db>> {
990        Some(match *diag {
991            PathLoweringDiagnostic::GenericArgsProhibited { segment, reason } => {
992                let segment = hir_segment_to_ast_segment(&path.value, segment)?;
993
994                if let Some(rtn) = segment.return_type_syntax() {
995                    // RTN errors are emitted as `GenericArgsProhibited` or `ParenthesizedGenericArgsWithoutFnTrait`.
996                    return Some(BadRtn { rtn: path.with_value(AstPtr::new(&rtn)) }.into());
997                }
998
999                let args = if let Some(generics) = segment.generic_arg_list() {
1000                    AstPtr::new(&generics).wrap_left()
1001                } else {
1002                    AstPtr::new(&segment.parenthesized_arg_list()?).wrap_right()
1003                };
1004                let args = path.with_value(args);
1005                GenericArgsProhibited { args, reason }.into()
1006            }
1007            PathLoweringDiagnostic::ParenthesizedGenericArgsWithoutFnTrait { segment } => {
1008                let segment = hir_segment_to_ast_segment(&path.value, segment)?;
1009
1010                if let Some(rtn) = segment.return_type_syntax() {
1011                    // RTN errors are emitted as `GenericArgsProhibited` or `ParenthesizedGenericArgsWithoutFnTrait`.
1012                    return Some(BadRtn { rtn: path.with_value(AstPtr::new(&rtn)) }.into());
1013                }
1014
1015                let args = AstPtr::new(&segment.parenthesized_arg_list()?);
1016                let args = path.with_value(args);
1017                ParenthesizedGenericArgsWithoutFnTrait { args }.into()
1018            }
1019            PathLoweringDiagnostic::IncorrectGenericsLen {
1020                generics_source,
1021                provided_count,
1022                expected_count,
1023                kind,
1024                def,
1025            } => {
1026                let generics_or_segment =
1027                    path_generics_source_to_ast(&path.value, generics_source)?;
1028                let generics_or_segment = path.with_value(AstPtr::new(&generics_or_segment));
1029                IncorrectGenericsLen {
1030                    generics_or_segment,
1031                    kind,
1032                    provided: provided_count,
1033                    expected: expected_count,
1034                    def: def.into(),
1035                }
1036                .into()
1037            }
1038            PathLoweringDiagnostic::IncorrectGenericsOrder {
1039                generics_source,
1040                param_id,
1041                arg_idx,
1042                has_self_arg,
1043            } => {
1044                let generic_args =
1045                    path_generics_source_to_ast(&path.value, generics_source)?.left()?;
1046                let provided_arg = hir_generic_arg_to_ast(&generic_args, arg_idx, has_self_arg)?;
1047                let provided_arg = path.with_value(AstPtr::new(&provided_arg));
1048                let expected_kind = GenericArgKind::from_id(param_id);
1049                IncorrectGenericsOrder { provided_arg, expected_kind }.into()
1050            }
1051            PathLoweringDiagnostic::MissingLifetime { generics_source, expected_count, def }
1052            | PathLoweringDiagnostic::ElisionFailure { generics_source, expected_count, def } => {
1053                let generics_or_segment =
1054                    path_generics_source_to_ast(&path.value, generics_source)?;
1055                let generics_or_segment = path.with_value(AstPtr::new(&generics_or_segment));
1056                MissingLifetime { generics_or_segment, expected: expected_count, def: def.into() }
1057                    .into()
1058            }
1059            PathLoweringDiagnostic::ElidedLifetimesInPath {
1060                generics_source,
1061                expected_count,
1062                def,
1063                hard_error,
1064            } => {
1065                let generics_or_segment =
1066                    path_generics_source_to_ast(&path.value, generics_source)?;
1067                let generics_or_segment = path.with_value(AstPtr::new(&generics_or_segment));
1068                ElidedLifetimesInPath {
1069                    generics_or_segment,
1070                    expected: expected_count,
1071                    def: def.into(),
1072                    hard_error,
1073                }
1074                .into()
1075            }
1076            PathLoweringDiagnostic::GenericDefaultRefersToSelf { segment } => {
1077                let segment = hir_segment_to_ast_segment(&path.value, segment)?;
1078                let segment = path.with_value(AstPtr::new(&segment));
1079                GenericDefaultRefersToSelf { segment }.into()
1080            }
1081        })
1082    }

1084    pub(crate) fn ty_diagnostic(
1085        diag: &TyLoweringDiagnostic,
1086        source_map: &ExpressionStoreSourceMap,
1087        db: &'db dyn HirDatabase,
1088    ) -> Option<AnyDiagnostic<'db>> {
1089        let Ok(source) = source_map.type_syntax(diag.source) else {
1090            stdx::never!("error on synthetic type syntax");
1091            return None;
1092        };
1093        let syntax = || source.value.to_node(&db.parse_or_expand(source.file_id));
1094        Some(match &diag.kind {
1095            TyLoweringDiagnosticKind::PathDiagnostic(diag) => {
1096                let ast::Type::PathType(syntax) = syntax() else { return None };
1097                Self::path_diagnostic(diag, source.with_value(syntax.path()?))?
1098            }
1099        })
1100    }

1103fn path_generics_source_to_ast(
1104    path: &ast::Path,
1105    generics_source: PathGenericsSource,
1106) -> Option<Either<ast::GenericArgList, ast::NameRef>> {
1107    Some(match generics_source {
1108        PathGenericsSource::Segment(segment) => {
1109            let segment = hir_segment_to_ast_segment(path, segment)?;
1110            segment
1111                .generic_arg_list()
1112                .map(Either::Left)
1113                .or_else(|| segment.name_ref().map(Either::Right))?
1114        }
1115        PathGenericsSource::AssocType { segment, assoc_type } => {
1116            let segment = hir_segment_to_ast_segment(path, segment)?;
1117            let segment_args = segment.generic_arg_list()?;
1118            let assoc = hir_assoc_type_binding_to_ast(&segment_args, assoc_type)?;
1119            assoc
1120                .generic_arg_list()
1121                .map(Either::Left)
1122                .or_else(|| assoc.name_ref().map(Either::Right))?
1123        }
1124    })
