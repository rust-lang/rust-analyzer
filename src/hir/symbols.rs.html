28/// The actual data that is stored in the index. It should be as compact as
29/// possible.
30#[derive(Debug, Clone, PartialEq, Eq, Hash)]
31pub struct FileSymbol<'db> {
32    pub name: Symbol,
33    pub def: ModuleDef,
34    pub loc: DeclarationLocation,
35    pub container_name: Option<Symbol>,
36    /// Whether this symbol is a doc alias for the original symbol.
37    pub is_alias: bool,
38    pub is_assoc: bool,
39    pub is_import: bool,
40    pub do_not_complete: Complete,
41    _marker: PhantomData<&'db ()>,
42}

44#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
45pub struct DeclarationLocation {
46    /// The file id for both the `ptr` and `name_ptr`.
47    pub hir_file_id: HirFileId,
48    /// This points to the whole syntax node of the declaration.
49    pub ptr: SyntaxNodePtr,
50    /// This points to the [`syntax::ast::Name`] identifier of the declaration.
51    pub name_ptr: Option<AstPtr<Either<syntax::ast::Name, syntax::ast::NameRef>>>,
52}

54impl DeclarationLocation {
55    pub fn syntax<DB: HirDatabase>(&self, sema: &Semantics<'_, DB>) -> SyntaxNode {
56        let root = sema.parse_or_expand(self.hir_file_id);
57        self.ptr.to_node(&root)
58    }

61/// Represents an outstanding module that the symbol collector must collect symbols from.
62#[derive(Debug)]
63struct SymbolCollectorWork {
64    module_id: ModuleId,
65    parent: Option<Name>,
66}

68pub struct SymbolCollector<'db> {
69    db: &'db dyn HirDatabase,
70    symbols: FxIndexSet<FileSymbol<'db>>,
71    work: Vec<SymbolCollectorWork>,
72    current_container_name: Option<Symbol>,
73    collect_pub_only: bool,

76/// Given a [`ModuleId`] and a [`HirDatabase`], use the DefMap for the module's crate to collect
77/// all symbols that should be indexed for the given module.
78impl<'a> SymbolCollector<'a> {
79    pub fn new(db: &'a dyn HirDatabase, collect_pub_only: bool) -> Self {
80        SymbolCollector {
81            db,
82            symbols: Default::default(),
83            work: Default::default(),
84            current_container_name: None,
85            collect_pub_only,
86        }
87    }

89    pub fn new_module(
90        db: &'a dyn HirDatabase,
91        module: Module,
92        collect_pub_only: bool,
93    ) -> Box<[FileSymbol<'a>]> {
94        let mut symbol_collector = SymbolCollector::new(db, collect_pub_only);
95        symbol_collector.collect(module);
96        symbol_collector.finish()
97    }

99    pub fn collect(&mut self, module: Module) {
100        let _p = tracing::info_span!("SymbolCollector::collect", ?module).entered();
101        tracing::info!(?module, "SymbolCollector::collect");
102
103        // The initial work is the root module we're collecting, additional work will
104        // be populated as we traverse the module's definitions.
105        self.work.push(SymbolCollectorWork { module_id: module.into(), parent: None });
106
107        while let Some(work) = self.work.pop() {
108            self.do_work(work);
109        }
110    }

112    /// Push a symbol for a crate's root module.
113    /// This allows crate roots to appear in the symbol index for queries like `::` or `::foo`.
114    pub fn push_crate_root(&mut self, krate: Crate) {
115        let Some(display_name) = krate.display_name(self.db) else { return };
116        let crate_name = display_name.crate_name();
117        let canonical_name = display_name.canonical_name();
118
119        let def_map = crate_def_map(self.db, krate.into());
120        let module_data = &def_map[def_map.crate_root(self.db)];
121
122        let definition = module_data.origin.definition_source(self.db);
123        let hir_file_id = definition.file_id;
124        let syntax_node = definition.value.node();
125        let ptr = SyntaxNodePtr::new(&syntax_node);
126
127        let loc = DeclarationLocation { hir_file_id, ptr, name_ptr: None };
128        let root_module = krate.root_module(self.db);
129
130        self.symbols.insert(FileSymbol {
131            name: crate_name.symbol().clone(),
132            def: ModuleDef::Module(root_module),
133            loc,
134            container_name: None,
135            is_alias: false,
136            is_assoc: false,
137            is_import: false,
138            do_not_complete: Complete::Yes,
139            _marker: PhantomData,
140        });
141
142        if canonical_name != crate_name.symbol() {
143            self.symbols.insert(FileSymbol {
144                name: canonical_name.clone(),
145                def: ModuleDef::Module(root_module),
146                loc,
147                container_name: None,
148                is_alias: false,
149                is_assoc: false,
150                is_import: false,
151                do_not_complete: Complete::Yes,
152                _marker: PhantomData,
153            });
154        }
155    }

157    pub fn finish(self) -> Box<[FileSymbol<'a>]> {
158        self.symbols.into_iter().collect()
159    }

161    fn do_work(&mut self, work: SymbolCollectorWork) {
162        let _p = tracing::info_span!("SymbolCollector::do_work", ?work).entered();
163        tracing::info!(?work, "SymbolCollector::do_work");
164        self.db.unwind_if_revision_cancelled();
165
166        let parent_name = work.parent.map(|name| Symbol::intern(name.as_str()));
167        self.with_container_name(parent_name, |s| s.collect_from_module(work.module_id));
168    }

170    fn collect_from_module(&mut self, module_id: ModuleId) {
171        let collect_pub_only = self.collect_pub_only;
172        let is_block_module = module_id.is_block_module(self.db);
173        let push_decl = |this: &mut Self, def: ModuleDefId, name, vis| {
174            if collect_pub_only && vis != Visibility::Public {
175                return;
176            }
177            match def {
178                ModuleDefId::ModuleId(id) => this.push_module(id, name),
179                ModuleDefId::FunctionId(id) => {
180                    this.push_decl(id, name, false, None);
181                    this.collect_from_body(id, Some(name.clone()));
182                }
183                ModuleDefId::AdtId(AdtId::StructId(id)) => {
184                    this.push_decl(id, name, false, None);
185                }
186                ModuleDefId::AdtId(AdtId::EnumId(id)) => {
187                    this.push_decl(id, name, false, None);
188                    let enum_name = Symbol::intern(this.db.enum_signature(id).name.as_str());
189                    this.with_container_name(Some(enum_name), |this| {
190                        let variants = id.enum_variants(this.db);
191                        for (variant_id, variant_name, _) in &variants.variants {
192                            this.push_decl(*variant_id, variant_name, true, None);
193                        }
194                    });
195                }
196                ModuleDefId::AdtId(AdtId::UnionId(id)) => {
197                    this.push_decl(id, name, false, None);
198                }
199                ModuleDefId::ConstId(id) => {
200                    this.push_decl(id, name, false, None);
201                    this.collect_from_body(id, Some(name.clone()));
202                }
203                ModuleDefId::StaticId(id) => {
204                    this.push_decl(id, name, false, None);
205                    this.collect_from_body(id, Some(name.clone()));
206                }
207                ModuleDefId::TraitId(id) => {
208                    let trait_do_not_complete = this.push_decl(id, name, false, None);
209                    this.collect_from_trait(id, trait_do_not_complete);
210                }
211                ModuleDefId::TypeAliasId(id) => {
212                    this.push_decl(id, name, false, None);
213                }
214                ModuleDefId::MacroId(id) => {
215                    match id {
216                        MacroId::Macro2Id(id) => this.push_decl(id, name, false, None),
217                        MacroId::MacroRulesId(id) => this.push_decl(id, name, false, None),
218                        MacroId::ProcMacroId(id) => this.push_decl(id, name, false, None),
219                    };
220                }
221                // Don't index these.
222                ModuleDefId::BuiltinType(_) => {}
223                ModuleDefId::EnumVariantId(_) => {}
224            }
225        };
226
227        // Nested trees are very common, so a cache here will hit a lot.
228        let import_child_source_cache = &mut FxHashMap::default();
229
230        let is_explicit_import = |vis| match vis {
231            Visibility::Public => true,
232            Visibility::PubCrate(_) => true,
233            Visibility::Module(_, VisibilityExplicitness::Explicit) => true,
234            Visibility::Module(_, VisibilityExplicitness::Implicit) => false,
235        };
236
237        let mut push_import = |this: &mut Self, i: ImportId, name: &Name, def: ModuleDefId, vis| {
238            if collect_pub_only && vis != Visibility::Public {
239                return;
240            }
241            let source = import_child_source_cache
242                .entry(i.use_)
243                .or_insert_with(|| i.use_.child_source(this.db));
244            if is_block_module && source.file_id.is_macro() {
245                // Macros tend to generate a lot of imports, the user really won't care about them
246                return;
247            }
248            let Some(use_tree_src) = source.value.get(i.idx) else { return };
249            let rename = use_tree_src.rename().and_then(|rename| rename.name());
250            let name_syntax = match rename {
251                Some(name) => Some(Either::Left(name)),
252                None if is_explicit_import(vis) => {
253                    (|| use_tree_src.path()?.segment()?.name_ref().map(Either::Right))()
254                }
255                None => None,
256            };
257            let Some(name_syntax) = name_syntax else {
258                return;
259            };
260            let dec_loc = DeclarationLocation {
261                hir_file_id: source.file_id,
262                ptr: SyntaxNodePtr::new(use_tree_src.syntax()),
263                name_ptr: Some(AstPtr::new(&name_syntax)),
264            };
265            this.symbols.insert(FileSymbol {
266                name: name.symbol().clone(),
267                def: def.into(),
268                container_name: this.current_container_name.clone(),
269                loc: dec_loc,
270                is_alias: false,
271                is_assoc: false,
272                is_import: true,
273                do_not_complete: Complete::Yes,
274                _marker: PhantomData,
275            });
276        };
277
278        let push_extern_crate =
279            |this: &mut Self, i: ExternCrateId, name: &Name, def: ModuleDefId, vis| {
280                if collect_pub_only && vis != Visibility::Public {
281                    return;
282                }
283                let loc = i.lookup(this.db);
284                if is_block_module && loc.ast_id().file_id.is_macro() {
285                    // Macros (especially derivves) tend to generate renamed extern crate items,
286                    // the user really won't care about them
287                    return;
288                }
289
290                let source = loc.source(this.db);
291                let rename = source.value.rename().and_then(|rename| rename.name());
292
293                let name_syntax = match rename {
294                    Some(name) => Some(Either::Left(name)),
295                    None if is_explicit_import(vis) => None,
296                    None => source.value.name_ref().map(Either::Right),
297                };
298                let Some(name_syntax) = name_syntax else {
299                    return;
300                };
301                let dec_loc = DeclarationLocation {
302                    hir_file_id: source.file_id,
303                    ptr: SyntaxNodePtr::new(source.value.syntax()),
304                    name_ptr: Some(AstPtr::new(&name_syntax)),
305                };
306                this.symbols.insert(FileSymbol {
307                    name: name.symbol().clone(),
308                    def: def.into(),
309                    container_name: this.current_container_name.clone(),
310                    loc: dec_loc,
311                    is_alias: false,
312                    is_assoc: false,
313                    is_import: false,
314                    do_not_complete: Complete::Yes,
315                    _marker: PhantomData,
316                });
317            };
318
319        let def_map = module_id.def_map(self.db);
320        let scope = &def_map[module_id].scope;
321
322        for impl_id in scope.impls() {
323            self.collect_from_impl(impl_id);
324        }
325
326        for (name, Item { def, vis, import }) in scope.types() {
327            if let Some(i) = import {
328                match i {
329                    ImportOrExternCrate::Import(i) => push_import(self, i, name, def, vis),
330                    ImportOrExternCrate::Glob(_) => (),
331                    ImportOrExternCrate::ExternCrate(i) => {
332                        push_extern_crate(self, i, name, def, vis)
333                    }
334                }
335
336                continue;
337            }
338            // self is a declaration
339            push_decl(self, def, name, vis)
340        }
341
342        for (name, Item { def, vis, import }) in scope.macros() {
343            if let Some(i) = import {
344                match i {
345                    ImportOrExternCrate::Import(i) => push_import(self, i, name, def.into(), vis),
346                    ImportOrExternCrate::Glob(_) => (),
347                    ImportOrExternCrate::ExternCrate(_) => (),
348                }
349                continue;
350            }
351            // self is a declaration
352            push_decl(self, ModuleDefId::MacroId(def), name, vis)
353        }
354
355        for (name, Item { def, vis, import }) in scope.values() {
356            if let Some(i) = import {
357                match i {
358                    ImportOrGlob::Import(i) => push_import(self, i, name, def, vis),
359                    ImportOrGlob::Glob(_) => (),
360                }
361                continue;
362            }
363            // self is a declaration
364            push_decl(self, def, name, vis)
365        }
366
367        for const_id in scope.unnamed_consts() {
368            self.collect_from_body(const_id, None);
369        }
370
371        for (name, id) in scope.legacy_macros() {
372            for &id in id {
373                if id.module(self.db) == module_id {
374                    match id {
375                        MacroId::Macro2Id(id) => self.push_decl(id, name, false, None),
376                        MacroId::MacroRulesId(id) => self.push_decl(id, name, false, None),
377                        MacroId::ProcMacroId(id) => self.push_decl(id, name, false, None),
378                    };
379                }
380            }
381        }
382    }

384    fn collect_from_body(&mut self, body_id: impl Into<DefWithBodyId>, name: Option<Name>) {
385        if self.collect_pub_only {
386            return;
387        }
388        let body_id = body_id.into();
389        let body = self.db.body(body_id);
390
391        // Descend into the blocks and enqueue collection of all modules within.
392        for (_, def_map) in body.blocks(self.db) {
393            for (id, _) in def_map.modules() {
394                self.work.push(SymbolCollectorWork { module_id: id, parent: name.clone() });
395            }
396        }
397    }

399    fn collect_from_impl(&mut self, impl_id: ImplId) {
400        let impl_data = self.db.impl_signature(impl_id);
401        let impl_name = Some(
402            hir_display_with_store(impl_data.self_ty, &impl_data.store)
403                .display(
404                    self.db,
405                    crate::Impl::from(impl_id).krate(self.db).to_display_target(self.db),
406                )
407                .to_smolstr(),
408        );
409        self.with_container_name(impl_name.as_deref().map(Symbol::intern), |s| {
410            for &(ref name, assoc_item_id) in &impl_id.impl_items(self.db).items {
411                if s.collect_pub_only && s.db.assoc_visibility(assoc_item_id) != Visibility::Public
412                {
413                    continue;
414                }
415
416                s.push_assoc_item(assoc_item_id, name, None)
417            }
418        })
419    }
420
421    fn collect_from_trait(&mut self, trait_id: TraitId, trait_do_not_complete: Complete) {
422        let trait_data = self.db.trait_signature(trait_id);
423        self.with_container_name(Some(Symbol::intern(trait_data.name.as_str())), |s| {
424            for &(ref name, assoc_item_id) in &trait_id.trait_items(self.db).items {
425                s.push_assoc_item(assoc_item_id, name, Some(trait_do_not_complete));
426            }
427        });
428    }

430    fn with_container_name(&mut self, container_name: Option<Symbol>, f: impl FnOnce(&mut Self)) {
431        if let Some(container_name) = container_name {
432            let prev = self.current_container_name.replace(container_name);
433            f(self);
434            self.current_container_name = prev;
435        } else {
436            f(self);
437        }
438    }

440    fn push_assoc_item(
441        &mut self,
442        assoc_item_id: AssocItemId,
443        name: &Name,
444        trait_do_not_complete: Option<Complete>,
445    ) {
446        match assoc_item_id {
447            AssocItemId::FunctionId(id) => self.push_decl(id, name, true, trait_do_not_complete),
448            AssocItemId::ConstId(id) => self.push_decl(id, name, true, trait_do_not_complete),
449            AssocItemId::TypeAliasId(id) => self.push_decl(id, name, true, trait_do_not_complete),
450        };
451    }

453    fn push_decl<L>(
454        &mut self,
455        id: L,
456        name: &Name,
457        is_assoc: bool,
458        trait_do_not_complete: Option<Complete>,
459    ) -> Complete
460    where
461        L: Lookup<Database = dyn DefDatabase> + Into<ModuleDefId>,
462        <L as Lookup>::Data: HasSource,
463        <<L as Lookup>::Data as HasSource>::Value: HasName,
464    {
465        let loc = id.lookup(self.db);
466        let source = loc.source(self.db);
467        let Some(name_node) = source.value.name() else { return Complete::Yes };
468        let def = ModuleDef::from(id.into());
469        let loc = DeclarationLocation {
470            hir_file_id: source.file_id,
471            ptr: SyntaxNodePtr::new(source.value.syntax()),
472            name_ptr: Some(AstPtr::new(&name_node).wrap_left()),
473        };
474
475        let mut do_not_complete = Complete::Yes;
476
477        if let Some(attrs) = def.attrs(self.db) {
478            do_not_complete = Complete::extract(matches!(def, ModuleDef::Trait(_)), attrs.attrs);
479            if let Some(trait_do_not_complete) = trait_do_not_complete {
480                do_not_complete = Complete::for_trait_item(trait_do_not_complete, do_not_complete);
481            }
482
483            for alias in attrs.doc_aliases(self.db) {
484                self.symbols.insert(FileSymbol {
485                    name: alias.clone(),
486                    def,
487                    loc,
488                    container_name: self.current_container_name.clone(),
489                    is_alias: true,
490                    is_assoc,
491                    is_import: false,
492                    do_not_complete,
493                    _marker: PhantomData,
494                });
495            }
496        }
497
498        self.symbols.insert(FileSymbol {
499            name: name.symbol().clone(),
500            def,
501            container_name: self.current_container_name.clone(),
502            loc,
503            is_alias: false,
504            is_assoc,
505            is_import: false,
506            do_not_complete,
507            _marker: PhantomData,
508        });
509
510        do_not_complete
511    }

513    fn push_module(&mut self, module_id: ModuleId, name: &Name) {
514        let def_map = module_id.def_map(self.db);
515        let module_data = &def_map[module_id];
516        let Some(declaration) = module_data.origin.declaration() else { return };
517        let module = declaration.to_node(self.db);
518        let Some(name_node) = module.name() else { return };
519        let loc = DeclarationLocation {
520            hir_file_id: declaration.file_id,
521            ptr: SyntaxNodePtr::new(module.syntax()),
522            name_ptr: Some(AstPtr::new(&name_node).wrap_left()),
523        };
524
525        let def = ModuleDef::Module(module_id.into());
526
527        let mut do_not_complete = Complete::Yes;
528        if let Some(attrs) = def.attrs(self.db) {
529            do_not_complete = Complete::extract(matches!(def, ModuleDef::Trait(_)), attrs.attrs);
530
531            for alias in attrs.doc_aliases(self.db) {
532                self.symbols.insert(FileSymbol {
533                    name: alias.clone(),
534                    def,
535                    loc,
536                    container_name: self.current_container_name.clone(),
537                    is_alias: true,
538                    is_assoc: false,
539                    is_import: false,
540                    do_not_complete,
541                    _marker: PhantomData,
542                });
543            }
544        }
545
546        self.symbols.insert(FileSymbol {
547            name: name.symbol().clone(),
548            def: ModuleDef::Module(module_id.into()),
549            container_name: self.current_container_name.clone(),
550            loc,
551            is_alias: false,
552            is_assoc: false,
553            is_import: false,
554            do_not_complete,
555            _marker: PhantomData,
556        });
557    }
