84impl PathResolution {
85    pub(crate) fn in_type_ns(&self) -> Option<TypeNs> {
86        match self {
87            PathResolution::Def(ModuleDef::Adt(adt)) => Some(TypeNs::AdtId((*adt).into())),
88            PathResolution::Def(ModuleDef::BuiltinType(builtin)) => {
89                Some(TypeNs::BuiltinType((*builtin).into()))
90            }
91            PathResolution::Def(
92                ModuleDef::Const(_)
93                | ModuleDef::Variant(_)
94                | ModuleDef::Macro(_)
95                | ModuleDef::Function(_)
96                | ModuleDef::Module(_)
97                | ModuleDef::Static(_)
98                | ModuleDef::Trait(_),
99            ) => None,
100            PathResolution::Def(ModuleDef::TypeAlias(alias)) => {
101                Some(TypeNs::TypeAliasId((*alias).into()))
102            }
103            PathResolution::BuiltinAttr(_)
104            | PathResolution::ToolModule(_)
105            | PathResolution::Local(_)
106            | PathResolution::DeriveHelper(_)
107            | PathResolution::ConstParam(_) => None,
108            PathResolution::TypeParam(param) => Some(TypeNs::GenericParam((*param).into())),
109            PathResolution::SelfType(impl_def) => match impl_def.id {
110                AnyImplId::ImplId(id) => Some(TypeNs::SelfType(id)),
111                AnyImplId::BuiltinDeriveImplId(_) => None,
112            },
113        }
114    }

276    pub fn lint_attrs(
277        &self,
278        file_id: FileId,
279        krate: Crate,
280        item: ast::AnyHasAttrs,
281    ) -> impl DoubleEndedIterator<Item = (LintAttr, SmolStr)> {
282        let mut cfg_options = None;
283        let cfg_options = || *cfg_options.get_or_insert_with(|| krate.id.cfg_options(self.db));
284
285        let is_crate_root = file_id == krate.root_file(self.imp.db);
286        let is_source_file = ast::SourceFile::can_cast(item.syntax().kind());
287        let extra_crate_attrs = (is_crate_root && is_source_file)
288            .then(|| {
289                parse_extra_crate_attrs(self.imp.db, krate.id)
290                    .into_iter()
291                    .flat_map(|src| src.attrs())
292            })
293            .into_iter()
294            .flatten();
295
296        let mut result = Vec::new();
297        hir_expand::attrs::expand_cfg_attr::<Infallible>(
298            extra_crate_attrs.chain(ast::attrs_including_inner(&item)),
299            cfg_options,
300            |attr, _, _, _| {
301                let hir_expand::attrs::Meta::TokenTree { path, tt } = attr else {
302                    return ControlFlow::Continue(());
303                };
304                if path.segments.len() != 1 {
305                    return ControlFlow::Continue(());
306                }
307                let lint_attr = match path.segments[0].text() {
308                    "allow" => LintAttr::Allow,
309                    "expect" => LintAttr::Expect,
310                    "warn" => LintAttr::Warn,
311                    "deny" => LintAttr::Deny,
312                    "forbid" => LintAttr::Forbid,
313                    _ => return ControlFlow::Continue(()),
314                };
315                let mut lint = SmolStrBuilder::new();
316                for token in
317                    tt.syntax().children_with_tokens().filter_map(SyntaxElement::into_token)
318                {
319                    match token.kind() {
320                        T![:] | T![::] => lint.push_str(token.text()),
321                        kind if kind.is_any_identifier() => lint.push_str(token.text()),
322                        T![,] => {
323                            let lint = mem::replace(&mut lint, SmolStrBuilder::new()).finish();
324                            if !lint.is_empty() {
325                                result.push((lint_attr, lint));
326                            }
327                        }
328                        _ => {}
329                    }
330                }
331                let lint = lint.finish();
332                if !lint.is_empty() {
333                    result.push((lint_attr, lint));
334                }
335
336                ControlFlow::Continue(())
337            },
338        );
339        result.into_iter()
340    }

500    pub fn find_parent_file(&self, file_id: HirFileId) -> Option<InFile<SyntaxNode>> {
501        match file_id {
502            HirFileId::FileId(file_id) => {
503                let module = self.file_to_module_defs(file_id.file_id(self.db)).next()?;
504                let def_map = crate_def_map(self.db, module.krate(self.db).id);
505                match def_map[module.id].origin {
506                    ModuleOrigin::CrateRoot { .. } => None,
507                    ModuleOrigin::File { declaration, declaration_tree_id, .. } => {
508                        let file_id = declaration_tree_id.file_id();
509                        let in_file = InFile::new(file_id, declaration);
510                        let node = in_file.to_node(self.db);
511                        let root = find_root(node.syntax());
512                        self.cache(root, file_id);
513                        Some(in_file.with_value(node.syntax().clone()))
514                    }
515                    _ => unreachable!("FileId can only belong to a file module"),
516                }
517            }
518            HirFileId::MacroFile(macro_file) => {
519                let node = self.db.lookup_intern_macro_call(macro_file).to_node(self.db);
520                let root = find_root(&node.value);
521                self.cache(root, node.file_id);
522                Some(node)
523            }
524        }
525    }

809    /// Retrieves all the formatting parts of the format_args! (or `asm!`) template string.
810    pub fn as_format_args_parts(
811        &self,
812        string: &ast::String,
813    ) -> Option<Vec<(TextRange, Option<Either<PathResolution, InlineAsmOperand>>)>> {
814        let string_start = string.syntax().text_range().start();
815        let token = self.wrap_token_infile(string.syntax().clone());
816        self.descend_into_macros_breakable(token, |token, _| {
817            (|| {
818                let token = token.value;
819                let string = ast::String::cast(token)?;
820                let literal =
821                    string.syntax().parent().filter(|it| it.kind() == SyntaxKind::LITERAL)?;
822                let parent = literal.parent()?;
823                if let Some(format_args) = ast::FormatArgsExpr::cast(parent.clone()) {
824                    let source_analyzer = self.analyze_no_infer(format_args.syntax())?;
825                    let format_args = self.wrap_node_infile(format_args);
826                    let res = source_analyzer
827                        .as_format_args_parts(self.db, format_args.as_ref())?
828                        .map(|(range, res)| (range + string_start, res.map(Either::Left)))
829                        .collect();
830                    Some(res)
831                } else {
832                    let asm = ast::AsmExpr::cast(parent)?;
833                    let source_analyzer = self.analyze_no_infer(asm.syntax())?;
834                    let line = asm.template().position(|it| *it.syntax() == literal)?;
835                    let asm = self.wrap_node_infile(asm);
836                    let (owner, (expr, asm_parts)) = source_analyzer.as_asm_parts(asm.as_ref())?;
837                    let res = asm_parts
838                        .get(line)?
839                        .iter()
840                        .map(|&(range, index)| {
841                            (
842                                range + string_start,
843                                Some(Either::Right(InlineAsmOperand { owner, expr, index })),
844                            )
845                        })
846                        .collect();
847                    Some(res)
848                }
849            })()
850            .map_or(ControlFlow::Continue(()), ControlFlow::Break)
851        })
852    }

1001    /// Maps a node down by mapping its first and last token down.
1002    pub fn descend_node_into_attributes<N: AstNode>(&self, node: N) -> SmallVec<[N; 1]> {
1003        // This might not be the correct way to do this, but it works for now
1004        let mut res = smallvec![];
1005        let tokens = (|| {
1006            // FIXME: the trivia skipping should not be necessary
1007            let first = skip_trivia_token(node.syntax().first_token()?, Direction::Next)?;
1008            let last = skip_trivia_token(node.syntax().last_token()?, Direction::Prev)?;
1009            Some((first, last))
1010        })();
1011        let (first, last) = match tokens {
1012            Some(it) => it,
1013            None => return res,
1014        };
1015        let file = self.find_file(node.syntax());
1016
1017        if first == last {
1018            // node is just the token, so descend the token
1019            self.descend_into_macros_all(
1020                InFile::new(file.file_id, first),
1021                false,
1022                &mut |InFile { value, .. }, _ctx| {
1023                    if let Some(node) = value
1024                        .parent_ancestors()
1025                        .take_while(|it| it.text_range() == value.text_range())
1026                        .find_map(N::cast)
1027                    {
1028                        res.push(node)
1029                    }
1030                },
1031            );
1032        } else {
1033            // Descend first and last token, then zip them to look for the node they belong to
1034            let mut scratch: SmallVec<[_; 1]> = smallvec![];
1035            self.descend_into_macros_all(
1036                InFile::new(file.file_id, first),
1037                false,
1038                &mut |token, _ctx| scratch.push(token),
1039            );
1040
1041            let mut scratch = scratch.into_iter();
1042            self.descend_into_macros_all(
1043                InFile::new(file.file_id, last),
1044                false,
1045                &mut |InFile { value: last, file_id: last_fid }, _ctx| {
1046                    if let Some(InFile { value: first, file_id: first_fid }) = scratch.next()
1047                        && first_fid == last_fid
1048                        && let Some(p) = first.parent()
1049                    {
1050                        let range = first.text_range().cover(last.text_range());
1051                        let node = find_root(&p)
1052                            .covering_element(range)
1053                            .ancestors()
1054                            .take_while(|it| it.text_range() == range)
1055                            .find_map(N::cast);
1056                        if let Some(node) = node {
1057                            res.push(node);
1058                        }
1059                    }
1060                },
1061            );
1062        }
1063        res
1064    }

1223    fn descend_into_macros_impl<T>(
1224        &self,
1225        InFile { value: token, file_id }: InFile<SyntaxToken>,
1226        always_descend_into_derives: bool,
1227        f: &mut dyn FnMut(InFile<SyntaxToken>, SyntaxContext) -> ControlFlow<T>,
1228    ) -> Option<T> {
1229        let _p = tracing::info_span!("descend_into_macros_impl").entered();
1230
1231        let db = self.db;
1232        let span = db.span_map(file_id).span_for_range(token.text_range());
1233
1234        // Process the expansion of a call, pushing all tokens with our span in the expansion back onto our stack
1235        let process_expansion_for_token =
1236            |ctx: &mut SourceToDefCtx<'_, '_>, stack: &mut Vec<_>, macro_file| {
1237                let InMacroFile { file_id, value: mapped_tokens } = ctx
1238                    .cache
1239                    .get_or_insert_expansion(ctx.db, macro_file)
1240                    .map_range_down(span)?
1241                    .map(SmallVec::<[_; 2]>::from_iter);
1242                // we have found a mapping for the token if the vec is non-empty
1243                let res = mapped_tokens.is_empty().not().then_some(());
1244                // requeue the tokens we got from mapping our current token down
1245                stack.push((HirFileId::from(file_id), mapped_tokens));
1246                res
1247            };
1248
1249        // A stack of tokens to process, along with the file they came from
1250        // These are tracked to know which macro calls we still have to look into
1251        // the tokens themselves aren't that interesting as the span that is being used to map
1252        // things down never changes.
1253        let mut stack: Vec<(_, SmallVec<[_; 2]>)> = vec![];
1254        let include = file_id
1255            .file_id()
1256            .and_then(|file_id| self.s2d_cache.borrow_mut().get_or_insert_include_for(db, file_id));
1257        match include {
1258            Some(include) => {
1259                // include! inputs are always from real files, so they only need to be handled once upfront
1260                self.with_ctx(|ctx| process_expansion_for_token(ctx, &mut stack, include))?;
1261            }
1262            None => {
1263                stack.push((file_id, smallvec![(token, span.ctx)]));
1264            }
1265        }
1266
1267        let mut m_cache = self.macro_call_cache.borrow_mut();
1268
1269        // Filters out all tokens that contain the given range (usually the macro call), any such
1270        // token is redundant as the corresponding macro call has already been processed
1271        let filter_duplicates = |tokens: &mut SmallVec<_>, range: TextRange| {
1272            tokens.retain(|(t, _): &mut (SyntaxToken, _)| !range.contains_range(t.text_range()))
1273        };
1274
1275        while let Some((expansion, ref mut tokens)) = stack.pop() {
1276            // Reverse the tokens so we prefer first tokens (to accommodate for popping from the
1277            // back)
1278            // alternatively we could pop from the front but that would shift the content on every pop
1279            tokens.reverse();
1280            while let Some((token, ctx)) = tokens.pop() {
1281                let was_not_remapped = (|| {
1282                    // First expand into attribute invocations, this is required to be handled
1283                    // upfront as any other macro call within will not semantically resolve unless
1284                    // also descended.
1285                    let res = self.with_ctx(|ctx| {
1286                        token
1287                            .parent_ancestors()
1288                            .filter_map(ast::Item::cast)
1289                            // FIXME: This might work incorrectly when we have a derive, followed by
1290                            // an attribute on an item, like:
1291                            // ```
1292                            // #[derive(Debug$0)]
1293                            // #[my_attr]
1294                            // struct MyStruct;
1295                            // ```
1296                            // here we should not consider the attribute at all, as our cursor
1297                            // technically lies outside of its expansion
1298                            .find_map(|item| {
1299                                // Don't force populate the dyn cache for items that don't have an attribute anyways
1300                                item.attrs().next()?;
1301                                ctx.item_to_macro_call(InFile::new(expansion, &item))
1302                                    .zip(Some(item))
1303                            })
1304                            .map(|(call_id, item)| {
1305                                let item_range = item.syntax().text_range();
1306                                let loc = db.lookup_intern_macro_call(call_id);
1307                                let text_range = match loc.kind {
1308                                    hir_expand::MacroCallKind::Attr {
1309                                        censored_attr_ids: attr_ids,
1310                                        ..
1311                                    } => {
1312                                        // FIXME: here, the attribute's text range is used to strip away all
1313                                        // entries from the start of the attribute "list" up the invoking
1314                                        // attribute. But in
1315                                        // ```
1316                                        // mod foo {
1317                                        //     #![inner]
1318                                        // }
1319                                        // ```
1320                                        // we don't wanna strip away stuff in the `mod foo {` range, that is
1321                                        // here if the id corresponds to an inner attribute we got strip all
1322                                        // text ranges of the outer ones, and then all of the inner ones up
1323                                        // to the invoking attribute so that the inbetween is ignored.
1324                                        // FIXME: Should cfg_attr be handled differently?
1325                                        let (attr, _, _, _) = attr_ids
1326                                            .invoc_attr()
1327                                            .find_attr_range_with_source(db, loc.krate, &item);
1328                                        let start = attr.syntax().text_range().start();
1329                                        TextRange::new(start, item_range.end())
1330                                    }
1331                                    _ => item_range,
1332                                };
1333                                filter_duplicates(tokens, text_range);
1334                                process_expansion_for_token(ctx, &mut stack, call_id)
1335                            })
1336                    });
1337
1338                    if let Some(res) = res {
1339                        return res;
1340                    }
1341
1342                    if always_descend_into_derives {
1343                        let res = self.with_ctx(|ctx| {
1344                            let (derives, adt) = token
1345                                .parent_ancestors()
1346                                .filter_map(ast::Adt::cast)
1347                                .find_map(|adt| {
1348                                    Some((
1349                                        ctx.derive_macro_calls(InFile::new(expansion, &adt))?
1350                                            .map(|(a, b, c)| (a, b, c.to_owned()))
1351                                            .collect::<SmallVec<[_; 2]>>(),
1352                                        adt,
1353                                    ))
1354                                })?;
1355                            for (_, derive_attr, derives) in derives {
1356                                // as there may be multiple derives registering the same helper
1357                                // name, we gotta make sure to call this for all of them!
1358                                // FIXME: We need to call `f` for all of them as well though!
1359                                process_expansion_for_token(ctx, &mut stack, derive_attr);
1360                                for derive in derives.into_iter().flatten() {
1361                                    let Either::Left(derive) = derive else { continue };
1362                                    process_expansion_for_token(ctx, &mut stack, derive);
1363                                }
1364                            }
1365                            // remove all tokens that are within the derives expansion
1366                            filter_duplicates(tokens, adt.syntax().text_range());
1367                            Some(())
1368                        });
1369                        // if we found derives, we can early exit. There is no way we can be in any
1370                        // macro call at this point given we are not in a token tree
1371                        if let Some(()) = res {
1372                            // Note: derives do not remap the original token. Furthermore, we want
1373                            // the original token to be before the derives in the list, because if they
1374                            // upmap to the same token and we deduplicate them (e.g. in rename), we
1375                            // want the original token to remain, not the derive.
1376                            return None;
1377                        }
1378                    }
1379                    // Then check for token trees, that means we are either in a function-like macro or
1380                    // secondary attribute inputs
1381                    let tt = token
1382                        .parent_ancestors()
1383                        .map_while(Either::<ast::TokenTree, ast::Meta>::cast)
1384                        .last()?;
1385
1386                    match tt {
1387                        // function-like macro call
1388                        Either::Left(tt) => {
1389                            let macro_call = tt.syntax().parent().and_then(ast::MacroCall::cast)?;
1390                            if tt.left_delimiter_token().map_or(false, |it| it == token) {
1391                                return None;
1392                            }
1393                            if tt.right_delimiter_token().map_or(false, |it| it == token) {
1394                                return None;
1395                            }
1396                            let mcall = InFile::new(expansion, macro_call);
1397                            let file_id = match m_cache.get(&mcall) {
1398                                Some(&it) => it,
1399                                None => {
1400                                    let it = ast::MacroCall::to_def(self, mcall.as_ref())?;
1401                                    m_cache.insert(mcall, it);
1402                                    it
1403                                }
1404                            };
1405                            let text_range = tt.syntax().text_range();
1406                            filter_duplicates(tokens, text_range);
1407
1408                            self.with_ctx(|ctx| {
1409                                process_expansion_for_token(ctx, &mut stack, file_id).or(file_id
1410                                    .eager_arg(db)
1411                                    .and_then(|arg| {
1412                                        // also descend into eager expansions
1413                                        process_expansion_for_token(ctx, &mut stack, arg)
1414                                    }))
1415                            })
1416                        }
1417                        Either::Right(_) if always_descend_into_derives => None,
1418                        // derive or derive helper
1419                        Either::Right(meta) => {
1420                            // attribute we failed expansion for earlier, this might be a derive invocation
1421                            // or derive helper attribute
1422                            let attr = meta.parent_attr()?;
1423                            let adt = match attr.syntax().parent().and_then(ast::Adt::cast) {
1424                                Some(adt) => {
1425                                    // this might be a derive on an ADT
1426                                    let res = self.with_ctx(|ctx| {
1427                                        // so try downmapping the token into the pseudo derive expansion
1428                                        // see [hir_expand::builtin_attr_macro] for how the pseudo derive expansion works
1429                                        let derive_call = ctx
1430                                            .attr_to_derive_macro_call(
1431                                                InFile::new(expansion, &adt),
1432                                                InFile::new(expansion, attr.clone()),
1433                                            )?
1434                                            .1;
1435
1436                                        // resolved to a derive
1437                                        let text_range = attr.syntax().text_range();
1438                                        // remove any other token in this macro input, all their mappings are the
1439                                        // same as this
1440                                        tokens.retain(|(t, _)| {
1441                                            !text_range.contains_range(t.text_range())
1442                                        });
1443                                        Some(process_expansion_for_token(
1444                                            ctx,
1445                                            &mut stack,
1446                                            derive_call,
1447                                        ))
1448                                    });
1449                                    if let Some(res) = res {
1450                                        return res;
1451                                    }
1452                                    Some(adt)
1453                                }
1454                                None => {
1455                                    // Otherwise this could be a derive helper on a variant or field
1456                                    attr.syntax().ancestors().find_map(ast::Item::cast).and_then(
1457                                        |it| match it {
1458                                            ast::Item::Struct(it) => Some(ast::Adt::Struct(it)),
1459                                            ast::Item::Enum(it) => Some(ast::Adt::Enum(it)),
1460                                            ast::Item::Union(it) => Some(ast::Adt::Union(it)),
1461                                            _ => None,
1462                                        },
1463                                    )
1464                                }
1465                            }?;
1466                            let attr_name =
1467                                attr.path().and_then(|it| it.as_single_name_ref())?.as_name();
1468                            // Not an attribute, nor a derive, so it's either an inert attribute or a derive helper
1469                            // Try to resolve to a derive helper and downmap
1470                            let resolver = &token
1471                                .parent()
1472                                .and_then(|parent| {
1473                                    self.analyze_impl(InFile::new(expansion, &parent), None, false)
1474                                })?
1475                                .resolver;
1476                            let id = db.ast_id_map(expansion).ast_id(&adt);
1477                            let helpers = resolver
1478                                .def_map()
1479                                .derive_helpers_in_scope(InFile::new(expansion, id))?;
1480
1481                            if !helpers.is_empty() {
1482                                let text_range = attr.syntax().text_range();
1483                                filter_duplicates(tokens, text_range);
1484                            }
1485
1486                            let mut res = None;
1487                            self.with_ctx(|ctx| {
1488                                for (.., derive) in
1489                                    helpers.iter().filter(|(helper, ..)| *helper == attr_name)
1490                                {
1491                                    let Either::Left(derive) = *derive else { continue };
1492                                    // as there may be multiple derives registering the same helper
1493                                    // name, we gotta make sure to call this for all of them!
1494                                    // FIXME: We need to call `f` for all of them as well though!
1495                                    res = res
1496                                        .or(process_expansion_for_token(ctx, &mut stack, derive));
1497                                }
1498                                res
1499                            })
1500                        }
1501                    }
1502                })()
1503                .is_none();
1504                if was_not_remapped
1505                    && let ControlFlow::Break(b) = f(InFile::new(expansion, token), ctx)
1506                {
1507                    return Some(b);
1508                }
1509            }
1510        }
1511        None
1512    }

1683        analyzer.expr_adjustments(expr).map(|it| {
1684            it.iter()
1685                .map(|adjust| {
1686                    let target = Type::new_with_resolver(
1687                        self.db,
1688                        &analyzer.resolver,
1689                        adjust.target.as_ref(),
1690                    );
1691                    let kind = match adjust.kind {
1692                        hir_ty::Adjust::NeverToAny => Adjust::NeverToAny,
1693                        hir_ty::Adjust::Deref(Some(hir_ty::OverloadedDeref(m))) => {
1694                            // FIXME: Should we handle unknown mutability better?
1695                            Adjust::Deref(Some(OverloadedDeref(
1696                                m.map(mutability).unwrap_or(Mutability::Shared),
1697                            )))
1698                        }
1699                        hir_ty::Adjust::Deref(None) => Adjust::Deref(None),
1700                        hir_ty::Adjust::Borrow(hir_ty::AutoBorrow::RawPtr(m)) => {
1701                            Adjust::Borrow(AutoBorrow::RawPtr(mutability(m)))
1702                        }
1703                        hir_ty::Adjust::Borrow(hir_ty::AutoBorrow::Ref(m)) => {
1704                            // FIXME: Handle lifetimes here
1705                            Adjust::Borrow(AutoBorrow::Ref(mutability(m.into())))
1706                        }
1707                        hir_ty::Adjust::Pointer(pc) => Adjust::Pointer(pc),
1708                    };
1709
1710                    // Update `source_ty` for the next adjustment
1711                    let source = mem::replace(&mut source_ty, target.clone());
1712
1713                    Adjustment { source, target, kind }
1714                })
1715                .collect()
1716        })
1717    }

2285        element: Either<&ast::Expr, &ast::StmtList>,
2286        text_range: TextRange,

2289        let store = sa.store()?;

2294            !path.mod_path().and_then(|path| path.as_ident()).is_some_and(Name::is_generated)
2295        };

2298            |e| vec![e.clone()],

2328                }

2330            },
2331        );
2332        let mut exprs: Vec<_> =
2333            exprs.into_iter().filter_map(|e| sa.expr_id(e).and_then(|e| e.as_expr())).collect();
2334

2336        let mut add_to_locals_used = |id, parent_expr| {
2337            let path = match id {
2338                ExprOrPatId::ExprId(expr_id) => {

2341                    } else {
2342                        None
2343                    }
2344                }

2348                    } else {

2350                    }
2351                }
2352            };

2360                    if let ValueNs::LocalBinding(id) = value {

2362                    }
2363                });
2364            }
2365        };

2502/// `SemanticsScope` encapsulates the notion of a scope (the set of visible
2503/// names) at a particular program point.
2504///
2505/// It is a bit tricky, as scopes do not really exist inside the compiler.
2506/// Rather, the compiler directly computes for each reference the definition it
2507/// refers to. It might transiently compute the explicit scope map while doing
2508/// so, but, generally, this is not something left after the analysis.
2509///
2510/// However, we do very much need explicit scopes for IDE purposes --
2511/// completion, at its core, lists the contents of the current scope. The notion
2512/// of scope is also useful to answer questions like "what would be the meaning
2513/// of this piece of code if we inserted it into this position?".
2514///
2515/// So `SemanticsScope` is constructed from a specific program point (a syntax
2516/// node or just a raw offset) and provides access to the set of visible names

2678struct RenameConflictsVisitor<'a> {
2679    db: &'a dyn HirDatabase,
2680    owner: DefWithBodyId,
2681    resolver: Resolver<'a>,
2682    body: &'a Body,
2683    to_be_renamed: BindingId,
2684    new_name: Symbol,
2685    old_name: Symbol,
2686    conflicts: FxHashSet<BindingId>,
2687}
2688
2689impl RenameConflictsVisitor<'_> {
2690    fn resolve_path(&mut self, node: ExprOrPatId, path: &Path) {
2691        if let Path::BarePath(path) = path
2692            && let Some(name) = path.as_ident()
2693        {
2694            if *name.symbol() == self.new_name {
2695                if let Some(conflicting) = self.resolver.rename_will_conflict_with_renamed(
2696                    self.db,
2697                    name,
2698                    path,
2699                    self.body.expr_or_pat_path_hygiene(node),
2700                    self.to_be_renamed,
2701                ) {
2702                    self.conflicts.insert(conflicting);
2703                }
2704            } else if *name.symbol() == self.old_name
2705                && let Some(conflicting) = self.resolver.rename_will_conflict_with_another_variable(
2706                    self.db,
2707                    name,
2708                    path,
2709                    self.body.expr_or_pat_path_hygiene(node),
2710                    &self.new_name,
2711                    self.to_be_renamed,
2712                )
2713            {
2714                self.conflicts.insert(conflicting);
2715            }
2716        }
2717    }
2718
2719    fn rename_conflicts(&mut self, expr: ExprId) {
2720        match &self.body[expr] {
2721            Expr::Path(path) => {
2722                let guard = self.resolver.update_to_inner_scope(self.db, self.owner, expr);
2723                self.resolve_path(expr.into(), path);
2724                self.resolver.reset_to_guard(guard);
2725            }
2726            &Expr::Assignment { target, .. } => {
2727                let guard = self.resolver.update_to_inner_scope(self.db, self.owner, expr);
2728                self.body.walk_pats(target, &mut |pat| {
2729                    if let Pat::Path(path) = &self.body[pat] {
2730                        self.resolve_path(pat.into(), path);
2731                    }
2732                });
2733                self.resolver.reset_to_guard(guard);
2734            }
2735            _ => {}
2736        }
2737
2738        self.body.walk_child_exprs(expr, |expr| self.rename_conflicts(expr));
2739    }
