20pub trait HasSource: Sized {
21    type Ast: AstNode;
22    /// Fetches the definition's source node.
23    /// Using [`crate::SemanticsImpl::source`] is preferred when working with [`crate::Semantics`],
24    /// as that caches the parsed file in the semantics' cache.
25    ///
26    /// The current some implementations can return `InFile` instead of `Option<InFile>`.
27    /// But we made this method `Option` to support rlib in the future
28    /// by <https://github.com/rust-lang/rust-analyzer/issues/6913>
29    fn source(self, db: &dyn HirDatabase) -> Option<InFile<Self::Ast>>;
30
31    /// Fetches the source node, along with its full range.
32    ///
33    /// The reason for the separate existence of this method is that some things, notably builtin derive impls,
34    /// do not really have a source node, at least not of the correct type. But we still can trace them
35    /// to source code (the derive producing them). So this method will return the range if it is supported,
36    /// and if the node is supported too it will return it as well.
37    fn source_with_range(
38        self,
39        db: &dyn HirDatabase,
40    ) -> Option<InFile<(TextRange, Option<Self::Ast>)>> {
41        let source = self.source(db)?;
42        Some(source.map(|node| (node.syntax().text_range(), Some(node))))
43    }
44}

46/// NB: Module is !HasSource, because it has two source nodes at the same time:
47/// definition and declaration.
48impl Module {
49    /// Returns a node which defines this module. That is, a file or a `mod foo {}` with items.
50    pub fn definition_source(self, db: &dyn HirDatabase) -> InFile<ModuleSource> {
51        let def_map = self.id.def_map(db);
52        def_map[self.id].definition_source(db)
53    }
54
55    /// Returns a node which defines this module. That is, a file or a `mod foo {}` with items.
56    pub fn definition_source_range(self, db: &dyn HirDatabase) -> InFile<TextRange> {
57        let def_map = self.id.def_map(db);
58        def_map[self.id].definition_source_range(db)
59    }
60
61    pub fn definition_source_file_id(self, db: &dyn HirDatabase) -> HirFileId {
62        let def_map = self.id.def_map(db);
63        def_map[self.id].definition_source_file_id()
64    }
65
66    pub fn is_mod_rs(self, db: &dyn HirDatabase) -> bool {
67        let def_map = self.id.def_map(db);
68        match def_map[self.id].origin {
69            ModuleOrigin::File { is_mod_rs, .. } => is_mod_rs,
70            _ => false,
71        }
72    }
73
74    pub fn as_source_file_id(self, db: &dyn HirDatabase) -> Option<EditionedFileId> {
75        let def_map = self.id.def_map(db);
76        match def_map[self.id].origin {
77            ModuleOrigin::File { definition, .. } | ModuleOrigin::CrateRoot { definition, .. } => {
78                Some(definition)
79            }
80            _ => None,
81        }
82    }
83
84    pub fn is_inline(self, db: &dyn HirDatabase) -> bool {
85        let def_map = self.id.def_map(db);
86        def_map[self.id].origin.is_inline()
87    }
88
89    /// Returns a node which declares this module, either a `mod foo;` or a `mod foo {}`.
90    /// `None` for the crate root.
91    pub fn declaration_source(self, db: &dyn HirDatabase) -> Option<InFile<ast::Module>> {
92        let def_map = self.id.def_map(db);
93        def_map[self.id].declaration_source(db)
94    }
95
96    /// Returns a text range which declares this module, either a `mod foo;` or a `mod foo {}`.
97    /// `None` for the crate root.
98    pub fn declaration_source_range(self, db: &dyn HirDatabase) -> Option<InFile<TextRange>> {
99        let def_map = self.id.def_map(db);
100        def_map[self.id].declaration_source_range(db)
101    }
102}

104impl HasSource for Field {
105    type Ast = FieldSource;
106    fn source(self, db: &dyn HirDatabase) -> Option<InFile<Self::Ast>> {
107        let var = VariantId::from(self.parent);
108        let src = var.child_source(db);
109        let field_source = src.map(|it| match it[self.id].clone() {
110            Either::Left(it) => FieldSource::Pos(it),
111            Either::Right(it) => FieldSource::Named(it),
112        });
113        Some(field_source)
114    }
115}

160impl HasSource for Function {
161    type Ast = ast::Fn;
162    fn source(self, db: &dyn HirDatabase) -> Option<InFile<Self::Ast>> {
163        match self.id {
164            AnyFunctionId::FunctionId(id) => Some(id.loc(db).source(db)),
165            // When calling `source()`, we use the trait method source, but when calling `source_with_range()`,
166            // we return `None` as the syntax node source. This is relying on the assumption that if you are calling
167            // `source_with_range()` (e.g. in navigation) you're prepared to deal with no source node, while if
168            // you call `source()` maybe you don't - therefore we fall back to the trait method, to not lose features.
169            AnyFunctionId::BuiltinDeriveImplMethod { method, impl_ } => method
170                .trait_method(db, impl_)
171                .and_then(|trait_method| Function::from(trait_method).source(db)),
172        }
173    }
174
175    fn source_with_range(
176        self,
177        db: &dyn HirDatabase,
178    ) -> Option<InFile<(TextRange, Option<Self::Ast>)>> {
179        match self.id {
180            AnyFunctionId::FunctionId(id) => Some(
181                id.loc(db).source(db).map(|source| (source.syntax().text_range(), Some(source))),
182            ),
183            AnyFunctionId::BuiltinDeriveImplMethod { impl_, .. } => {
184                Some(impl_.loc(db).source(db).map(|range| (range, None)))
185            }
186        }
187    }
188}

275impl HasSource for Param<'_> {
276    type Ast = Either<ast::SelfParam, ast::Param>;
277
278    fn source(self, db: &dyn HirDatabase) -> Option<InFile<Self::Ast>> {
279        match self.func {
280            Callee::Def(CallableDefId::FunctionId(func)) => {
281                let InFile { file_id, value } = Function::from(func).source(db)?;
282                let params = value.param_list()?;
283                if let Some(self_param) = params.self_param() {
284                    if let Some(idx) = self.idx.checked_sub(1) {
285                        params.params().nth(idx).map(Either::Right)
286                    } else {
287                        Some(Either::Left(self_param))
288                    }
289                } else {
290                    params.params().nth(self.idx).map(Either::Right)
291                }
292                .map(|value| InFile { file_id, value })
293            }
294            Callee::Closure(closure, _) => {
295                let InternedClosure(owner, expr_id) = db.lookup_intern_closure(closure);
296                let (_, source_map) = db.body_with_source_map(owner);
297                let ast @ InFile { file_id, value } = source_map.expr_syntax(expr_id).ok()?;
298                let root = db.parse_or_expand(file_id);
299                match value.to_node(&root) {
300                    Either::Left(ast::Expr::ClosureExpr(it)) => it
301                        .param_list()?
302                        .params()
303                        .nth(self.idx)
304                        .map(Either::Right)
305                        .map(|value| InFile { file_id: ast.file_id, value }),
306                    _ => None,
307                }
308            }
309            _ => None,
310        }
311    }
312}

318        let InFile { file_id, value } = self.func.source(db)?;

326impl HasSource for Label {
327    type Ast = ast::Label;
328
329    fn source(self, db: &dyn HirDatabase) -> Option<InFile<Self::Ast>> {
330        let (_body, source_map) = db.body_with_source_map(self.parent);
331        let src = source_map.label_syntax(self.label_id);
332        let root = src.file_syntax(db);
333        src.map(|ast| ast.to_node(&root).left()).transpose()
334    }
335}

345impl HasSource for InlineAsmOperand {
346    type Ast = ast::AsmOperandNamed;
347    fn source(self, db: &dyn HirDatabase) -> Option<InFile<Self::Ast>> {
348        let source_map = db.body_with_source_map(self.owner).1;
349        if let Ok(src) = source_map.expr_syntax(self.expr) {
350            let root = src.file_syntax(db);
351            return src
352                .map(|ast| match ast.to_node(&root) {
353                    Either::Left(ast::Expr::AsmExpr(asm)) => asm
354                        .asm_pieces()
355                        .filter_map(|it| match it {
356                            ast::AsmPiece::AsmOperandNamed(it) => Some(it),
357                            _ => None,
358                        })
359                        .nth(self.index),
360                    _ => None,
361                })
362                .transpose();
363        }
364        None
365    }
