1//! RA Proc Macro Server
2//!
3//! This library is able to call compiled Rust custom derive dynamic libraries on arbitrary code.
4//! The general idea here is based on <https://github.com/fedochet/rust-proc-macro-expander>.
5//!
6//! But we adapt it to better fit RA needs:
7//!
8//! * We use `tt` for proc-macro `TokenStream` server, it is easier to manipulate and interact with
9//!   RA than `proc-macro2` token stream.
10//! * By **copying** the whole rustc `lib_proc_macro` code, we are able to build this with `stable`
11//!   rustc rather than `unstable`. (Although in general ABI compatibility is still an issue)â€¦

127const EXPANDER_STACK_SIZE: usize = 8 * 1024 * 1024;

129pub enum ExpandError {
130    Panic(PanicMessage),
131    Cancelled { reason: Option<String> },
132    Internal { reason: Option<String> },
133}

135impl ExpandError {
136    pub fn into_string(self) -> Option<String> {
137        match self {
138            ExpandError::Panic(panic_message) => panic_message.into_string(),
139            ExpandError::Cancelled { reason } => reason,
140            ExpandError::Internal { reason } => reason,
141        }
142    }

145impl ProcMacroSrv<'_> {
146    pub fn expand<S: ProcMacroSrvSpan>(
147        &self,
148        lib: impl AsRef<Utf8Path>,
149        env: &[(String, String)],
150        current_dir: Option<impl AsRef<Path>>,
151        macro_name: &str,
152        macro_body: token_stream::TokenStream<S>,
153        attribute: Option<token_stream::TokenStream<S>>,
154        def_site: S,
155        call_site: S,
156        mixed_site: S,
157        callback: Option<ProcMacroClientHandle<'_>>,
158    ) -> Result<token_stream::TokenStream<S>, ExpandError> {
159        let snapped_env = self.env;
160        let expander = self.expander(lib.as_ref()).map_err(|err| ExpandError::Internal {
161            reason: Some(format!("failed to load macro: {err}")),
162        })?;
163
164        let prev_env = EnvChange::apply(snapped_env, env, current_dir.as_ref().map(<_>::as_ref));
165
166        // Note, we spawn a new thread here so that thread locals allocation don't accumulate (this
167        // includes the proc-macro symbol interner)
168        let result = thread::scope(|s| {
169            let thread = thread::Builder::new()
170                .stack_size(EXPANDER_STACK_SIZE)
171                .name(macro_name.to_owned())
172                .spawn_scoped(s, move || {
173                    expander.expand(
174                        macro_name, macro_body, attribute, def_site, call_site, mixed_site,
175                        callback,
176                    )
177                });
178            match thread.unwrap().join() {
179                Ok(res) => res.map_err(ExpandError::Panic),
180
181                Err(payload) => {
182                    if let Some(marker) = payload.downcast_ref::<ProcMacroPanicMarker>() {
183                        return match marker {
184                            ProcMacroPanicMarker::Cancelled { reason } => {
185                                Err(ExpandError::Cancelled { reason: Some(reason.clone()) })
186                            }
187                            ProcMacroPanicMarker::Internal { reason } => {
188                                Err(ExpandError::Internal { reason: Some(reason.clone()) })
189                            }
190                        };
191                    }
192
193                    std::panic::resume_unwind(payload)
194                }
195            }
196        });
197        prev_env.rollback();
198
199        result
200    }
201
202    pub fn list_macros(
203        &self,
204        dylib_path: &Utf8Path,
205    ) -> Result<Vec<(String, ProcMacroKind)>, String> {
206        let expander = self.expander(dylib_path)?;
207        Ok(expander.list_macros().map(|(k, v)| (k.to_owned(), v)).collect())
208    }
209
210    fn expander(&self, path: &Utf8Path) -> Result<Arc<dylib::Expander>, String> {
211        let expander = || {
212            let expander = dylib::Expander::new(&self.temp_dir, path)
213                .map_err(|err| format!("Cannot create expander for {path}: {err}",));
214            expander.map(Arc::new)
215        };
216
217        Ok(
218            match self
219                .expanders
220                .lock()
221                .unwrap_or_else(PoisonError::into_inner)
222                .entry(path.to_path_buf())
223            {
224                Entry::Vacant(v) => v.insert(expander()?).clone(),
225                Entry::Occupied(mut e) => {
226                    let time = fs::metadata(path).and_then(|it| it.modified()).ok();
227                    if Some(e.get().modified_time()) != time {
228                        e.insert(expander()?);
229                    }
230                    e.get().clone()
231                }
232            },
233        )
234    }
235}

309static ENV_LOCK: std::sync::Mutex<()> = std::sync::Mutex::new(());
310
311struct EnvChange<'snap> {
312    changed_vars: Vec<&'snap str>,
313    prev_working_dir: Option<PathBuf>,
314    snap: &'snap EnvSnapshot,
315    _guard: std::sync::MutexGuard<'snap, ()>,
316}
317
318impl<'snap> EnvChange<'snap> {
319    fn apply(
320        snap: &'snap EnvSnapshot,
321        new_vars: &'snap [(String, String)],
322        current_dir: Option<&Path>,
323    ) -> EnvChange<'snap> {
324        let guard = ENV_LOCK.lock().unwrap_or_else(std::sync::PoisonError::into_inner);
325        let prev_working_dir = match current_dir {
326            Some(dir) => {
327                let prev_working_dir = std::env::current_dir().ok();
328                if let Err(err) = std::env::set_current_dir(dir) {
329                    eprintln!(
330                        "Failed to set the current working dir to {}. Error: {err:?}",
331                        dir.display()
332                    )
333                }
334                prev_working_dir
335            }
336            None => None,
337        };
338        EnvChange {
339            snap,
340            changed_vars: new_vars
341                .iter()
342                .map(|(k, v)| {
343                    // SAFETY: We have acquired the environment lock
344                    unsafe { env::set_var(k, v) };
345                    &**k
346                })
347                .collect(),
348            prev_working_dir,
349            _guard: guard,
350        }
351    }
352
353    fn rollback(self) {}
354}
355
356impl Drop for EnvChange<'_> {
357    fn drop(&mut self) {
358        for name in self.changed_vars.drain(..) {
359            // SAFETY: We have acquired the environment lock
360            unsafe {
361                match self.snap.vars.get::<std::ffi::OsStr>(name.as_ref()) {
362                    Some(prev_val) => env::set_var(name, prev_val),
363                    None => env::remove_var(name),
364                }
365            }
366        }
367
368        if let Some(dir) = &self.prev_working_dir
369            && let Err(err) = std::env::set_current_dir(dir)
370        {
371            eprintln!(
372                "Failed to set the current working dir to {}. Error: {:?}",
373                dir.display(),
374                err
375            )
376        }
377    }
378}
