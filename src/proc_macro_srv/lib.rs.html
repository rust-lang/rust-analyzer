1//! RA Proc Macro Server
2//!
3//! This library is able to call compiled Rust custom derive dynamic libraries on arbitrary code.
4//! The general idea here is based on <https://github.com/fedochet/rust-proc-macro-expander>.
5//!
6//! But we adapt it to better fit RA needs:
7//!
8//! * We use `tt` for proc-macro `TokenStream` server, it is easier to manipulate and interact with
9//!   RA than `proc-macro2` token stream.
10//! * By **copying** the whole rustc `lib_proc_macro` code, we are able to build this with `stable`
11//!   rustc rather than `unstable`. (Although in general ABI compatibility is still an issue)â€¦

75const EXPANDER_STACK_SIZE: usize = 8 * 1024 * 1024;
76
77impl ProcMacroSrv<'_> {
78    pub fn expand<S: ProcMacroSrvSpan>(
79        &self,
80        lib: impl AsRef<Utf8Path>,
81        env: &[(String, String)],
82        current_dir: Option<impl AsRef<Path>>,
83        macro_name: &str,
84        macro_body: tt::TopSubtree<S>,
85        attribute: Option<tt::TopSubtree<S>>,
86        def_site: S,
87        call_site: S,
88        mixed_site: S,
89    ) -> Result<Vec<tt::TokenTree<S>>, PanicMessage> {
90        let snapped_env = self.env;
91        let expander = self.expander(lib.as_ref()).map_err(|err| PanicMessage {
92            message: Some(format!("failed to load macro: {err}")),
93        })?;
94
95        let prev_env = EnvChange::apply(snapped_env, env, current_dir.as_ref().map(<_>::as_ref));
96
97        // Note, we spawn a new thread here so that thread locals allocation don't accumulate (this
98        // includes the proc-macro symbol interner)
99        let result = thread::scope(|s| {
100            let thread = thread::Builder::new()
101                .stack_size(EXPANDER_STACK_SIZE)
102                .name(macro_name.to_owned())
103                .spawn_scoped(s, move || {
104                    expander
105                        .expand(
106                            macro_name,
107                            server_impl::TopSubtree(macro_body.0.into_vec()),
108                            attribute.map(|it| server_impl::TopSubtree(it.0.into_vec())),
109                            def_site,
110                            call_site,
111                            mixed_site,
112                        )
113                        .map(|tt| tt.0)
114                });
115            match thread.unwrap().join() {
116                Ok(res) => res,
117                Err(e) => std::panic::resume_unwind(e),
118            }
119        });
120        prev_env.rollback();
121
122        result
123    }
124
125    pub fn list_macros(
126        &self,
127        dylib_path: &Utf8Path,
128    ) -> Result<Vec<(String, ProcMacroKind)>, String> {
129        let expander = self.expander(dylib_path)?;
130        Ok(expander.list_macros().map(|(k, v)| (k.to_owned(), v)).collect())
131    }
132
133    fn expander(&self, path: &Utf8Path) -> Result<Arc<dylib::Expander>, String> {
134        let expander = || {
135            let expander = dylib::Expander::new(&self.temp_dir, path)
136                .map_err(|err| format!("Cannot create expander for {path}: {err}",));
137            expander.map(Arc::new)
138        };
139
140        Ok(
141            match self
142                .expanders
143                .lock()
144                .unwrap_or_else(PoisonError::into_inner)
145                .entry(path.to_path_buf())
146            {
147                Entry::Vacant(v) => v.insert(expander()?).clone(),
148                Entry::Occupied(mut e) => {
149                    let time = fs::metadata(path).and_then(|it| it.modified()).ok();
150                    if Some(e.get().modified_time()) != time {
151                        e.insert(expander()?);
152                    }
153                    e.get().clone()
154                }
155            },
156        )
157    }
158}

206static ENV_LOCK: std::sync::Mutex<()> = std::sync::Mutex::new(());
207
208struct EnvChange<'snap> {
209    changed_vars: Vec<&'snap str>,
210    prev_working_dir: Option<PathBuf>,
211    snap: &'snap EnvSnapshot,
212    _guard: std::sync::MutexGuard<'snap, ()>,
213}
214
215impl<'snap> EnvChange<'snap> {
216    fn apply(
217        snap: &'snap EnvSnapshot,
218        new_vars: &'snap [(String, String)],
219        current_dir: Option<&Path>,
220    ) -> EnvChange<'snap> {
221        let guard = ENV_LOCK.lock().unwrap_or_else(std::sync::PoisonError::into_inner);
222        let prev_working_dir = match current_dir {
223            Some(dir) => {
224                let prev_working_dir = std::env::current_dir().ok();
225                if let Err(err) = std::env::set_current_dir(dir) {
226                    eprintln!(
227                        "Failed to set the current working dir to {}. Error: {err:?}",
228                        dir.display()
229                    )
230                }
231                prev_working_dir
232            }
233            None => None,
234        };
235        EnvChange {
236            snap,
237            changed_vars: new_vars
238                .iter()
239                .map(|(k, v)| {
240                    // SAFETY: We have acquired the environment lock
241                    unsafe { env::set_var(k, v) };
242                    &**k
243                })
244                .collect(),
245            prev_working_dir,
246            _guard: guard,
247        }
248    }
249
250    fn rollback(self) {}
251}
252
253impl Drop for EnvChange<'_> {
254    fn drop(&mut self) {
255        for name in self.changed_vars.drain(..) {
256            // SAFETY: We have acquired the environment lock
257            unsafe {
258                match self.snap.vars.get::<std::ffi::OsStr>(name.as_ref()) {
259                    Some(prev_val) => env::set_var(name, prev_val),
260                    None => env::remove_var(name),
261                }
262            }
263        }
264
265        if let Some(dir) = &self.prev_working_dir
266            && let Err(err) = std::env::set_current_dir(dir)
267        {
268            eprintln!(
269                "Failed to set the current working dir to {}. Error: {:?}",
270                dir.display(),
271                err
272            )
273        }
274    }
275}
