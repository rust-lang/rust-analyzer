17fn try_main() -> io::Result<()> {
18    if env::args().count() != 1 {
19        eprintln!("Usage: ungrammar2json < grammar.ungram > grammar.json");
20        return Ok(());
21    }
22    let grammar = read_stdin()?;
23    let grammar = grammar
24        .parse::<Grammar>()
25        .map_err(|err| io::Error::new(io::ErrorKind::InvalidData, err))?;
26
27    let mut buf = String::new();
28    grammar_to_json(&grammar, write_json::object(&mut buf));
29    println!("{}", buf);
30    Ok(())
31}

33fn read_stdin() -> io::Result<String> {
34    let mut buf = String::new();
35    io::stdin().lock().read_to_string(&mut buf)?;
36    Ok(buf)

39fn grammar_to_json(grammar: &Grammar, mut obj: write_json::Object<'_>) {
40    for node in grammar.iter() {
41        let node = &grammar[node];
42        rule_to_json(grammar, &node.rule, obj.object(&node.name));
43    }
44}
45
46fn rule_to_json(grammar: &Grammar, rule: &Rule, mut obj: write_json::Object<'_>) {
47    match rule {
48        Rule::Labeled { label, rule } => {
49            obj.string("label", label);
50            rule_to_json(grammar, rule, obj.object("rule"))
51        }
52        Rule::Node(node) => {
53            obj.string("node", &grammar[*node].name);
54        }
55        Rule::Token(token) => {
56            obj.string("token", &grammar[*token].name);
57        }
58        Rule::Seq(rules) | Rule::Alt(rules) => {
59            let tag = match rule {
60                Rule::Seq(_) => "seq",
61                Rule::Alt(_) => "alt",
62                _ => unreachable!(),
63            };
64            let mut array = obj.array(tag);
65            for rule in rules {
66                rule_to_json(grammar, rule, array.object());
67            }
68        }
69        Rule::Opt(arg) | Rule::Rep(arg) => {
70            let tag = match rule {
71                Rule::Opt(_) => "opt",
72                Rule::Rep(_) => "rep",
73                _ => unreachable!(),
74            };
75            rule_to_json(grammar, arg, obj.object(tag));
76        }
77    }
78}
