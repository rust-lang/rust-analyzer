34pub trait WithFixture: Default + ExpandDatabase + SourceDatabase + 'static {
35    #[track_caller]
36    fn with_single_file(
37        #[rust_analyzer::rust_fixture] ra_fixture: &str,
38    ) -> (Self, EditionedFileId) {
39        let mut db = Self::default();
40        let fixture = ChangeFixture::parse(&db, ra_fixture);
41        fixture.change.apply(&mut db);
42        assert_eq!(fixture.files.len(), 1, "Multiple file found in the fixture");
43        (db, fixture.files[0])
44    }
45
46    #[track_caller]
47    fn with_many_files(
48        #[rust_analyzer::rust_fixture] ra_fixture: &str,
49    ) -> (Self, Vec<EditionedFileId>) {
50        let mut db = Self::default();
51        let fixture = ChangeFixture::parse(&db, ra_fixture);
52        fixture.change.apply(&mut db);
53        assert!(fixture.file_position.is_none());
54        (db, fixture.files)
55    }
56
57    #[track_caller]
58    fn with_files(#[rust_analyzer::rust_fixture] ra_fixture: &str) -> Self {
59        let mut db = Self::default();
60        let fixture = ChangeFixture::parse(&db, ra_fixture);
61        fixture.change.apply(&mut db);
62        assert!(fixture.file_position.is_none());
63        db
64    }
65
66    #[track_caller]
67    fn with_files_extra_proc_macros(
68        #[rust_analyzer::rust_fixture] ra_fixture: &str,
69        proc_macros: Vec<(String, ProcMacro)>,
70    ) -> Self {
71        let mut db = Self::default();
72        let fixture = ChangeFixture::parse_with_proc_macros(
73            &db,
74            ra_fixture,
75            MiniCore::RAW_SOURCE,
76            proc_macros,
77        );
78        fixture.change.apply(&mut db);
79        assert!(fixture.file_position.is_none());
80        db
81    }
82
83    #[track_caller]
84    fn with_position(#[rust_analyzer::rust_fixture] ra_fixture: &str) -> (Self, FilePosition) {
85        let (db, file_id, range_or_offset) = Self::with_range_or_offset(ra_fixture);
86        let offset = range_or_offset.expect_offset();
87        (db, FilePosition { file_id, offset })
88    }
89
90    #[track_caller]
91    fn with_range(#[rust_analyzer::rust_fixture] ra_fixture: &str) -> (Self, FileRange) {
92        let (db, file_id, range_or_offset) = Self::with_range_or_offset(ra_fixture);
93        let range = range_or_offset.expect_range();
94        (db, FileRange { file_id, range })
95    }
96
97    #[track_caller]
98    fn with_range_or_offset(
99        #[rust_analyzer::rust_fixture] ra_fixture: &str,
100    ) -> (Self, EditionedFileId, RangeOrOffset) {
101        let mut db = Self::default();
102        let fixture = ChangeFixture::parse(&db, ra_fixture);
103        fixture.change.apply(&mut db);
104
105        let (file_id, range_or_offset) = fixture
106            .file_position
107            .expect("Could not find file position in fixture. Did you forget to add an `$0`?");
108        (db, file_id, range_or_offset)
109    }
110
111    fn test_crate(&self) -> Crate {
112        self.all_crates().iter().copied().find(|&krate| !krate.data(self).origin.is_lang()).unwrap()
113    }
114}

128impl ChangeFixture {
129    pub fn parse(
130        db: &dyn salsa::Database,
131        #[rust_analyzer::rust_fixture] ra_fixture: &str,
132    ) -> ChangeFixture {
133        Self::parse_with_proc_macros(db, ra_fixture, MiniCore::RAW_SOURCE, Vec::new())

136    pub fn parse_with_proc_macros(
137        db: &dyn salsa::Database,
138        #[rust_analyzer::rust_fixture] ra_fixture: &str,
139        minicore_raw: &str,
140        mut proc_macro_defs: Vec<(String, ProcMacro)>,
141    ) -> ChangeFixture {
142        let FixtureWithProjectMeta {
143            fixture,
144            mini_core,
145            proc_macro_names,
146            toolchain,
147            target_data_layout,
148            target_arch,
149        } = FixtureWithProjectMeta::parse(ra_fixture);
150        let target_data_layout = target_data_layout.into();
151        let target_arch = parse_target_arch(&target_arch);
152        let target = Ok(TargetData { arch: target_arch, data_layout: target_data_layout });
153        let toolchain = Some({
154            let channel = toolchain.as_deref().unwrap_or("stable");
155            Version::parse(&format!("1.76.0-{channel}")).unwrap()
156        });
157        let mut source_change = FileChange::default();
158
159        let mut files = Vec::new();
160        let mut sysroot_files = Vec::new();
161        let mut file_lines = Vec::new();
162        let mut crate_graph = CrateGraphBuilder::default();
163        let mut crates = FxIndexMap::default();
164        let mut crate_deps = Vec::new();
165        let mut default_crate_root: Option<FileId> = None;
166        let mut default_edition = Edition::CURRENT;
167        let mut default_cfg = CfgOptions::default();
168        let mut default_env = Env::from_iter([(
169            String::from("__ra_is_test_fixture"),
170            String::from("__ra_is_test_fixture"),
171        )]);
172
173        let mut file_set = FileSet::default();
174        let mut current_source_root_kind = SourceRootKind::Local;
175        let mut file_id = FileId::from_raw(0);
176        let mut roots = Vec::new();
177
178        let mut file_position = None;
179
180        let crate_ws_data = Arc::new(CrateWorkspaceData { target, toolchain });
181
182        // FIXME: This is less than ideal
183        let proc_macro_cwd = Arc::new(AbsPathBuf::assert_utf8(std::env::current_dir().unwrap()));
184
185        for entry in fixture {
186            file_lines.push(entry.line);
187
188            let mut range_or_offset = None;
189            let text = if entry.text.contains(CURSOR_MARKER) {
190                if entry.text.contains(ESCAPED_CURSOR_MARKER) {
191                    entry.text.replace(ESCAPED_CURSOR_MARKER, CURSOR_MARKER)
192                } else {
193                    let (roo, text) = extract_range_or_offset(&entry.text);
194                    assert!(file_position.is_none());
195                    range_or_offset = Some(roo);
196                    text
197                }
198            } else {
199                entry.text.as_str().into()
200            };
201
202            let meta = FileMeta::from_fixture(entry, current_source_root_kind);
203            if let Some(range_or_offset) = range_or_offset {
204                file_position =
205                    Some((EditionedFileId::new(db, file_id, meta.edition), range_or_offset));
206            }
207
208            assert!(meta.path.starts_with(SOURCE_ROOT_PREFIX));
209            if !meta.deps.is_empty() {
210                assert!(meta.krate.is_some(), "can't specify deps without naming the crate")
211            }
212
213            if let Some(kind) = meta.introduce_new_source_root {
214                assert!(
215                    meta.krate.is_some(),
216                    "new_source_root meta doesn't make sense without crate meta"
217                );
218                let prev_kind = mem::replace(&mut current_source_root_kind, kind);
219                let prev_root = match prev_kind {
220                    SourceRootKind::Local => SourceRoot::new_local(mem::take(&mut file_set)),
221                    SourceRootKind::Library => SourceRoot::new_library(mem::take(&mut file_set)),
222                };
223                roots.push(prev_root);
224            }
225
226            if let Some((krate, origin, version)) = meta.krate {
227                let crate_name = CrateName::normalize_dashes(&krate);
228                let crate_id = crate_graph.add_crate_root(
229                    file_id,
230                    meta.edition,
231                    Some(crate_name.clone().into()),
232                    version,
233                    meta.cfg.clone(),
234                    Some(meta.cfg),
235                    meta.env,
236                    origin,
237                    false,
238                    proc_macro_cwd.clone(),
239                    crate_ws_data.clone(),
240                );
241                let prev = crates.insert(crate_name.clone(), crate_id);
242                assert!(prev.is_none(), "multiple crates with same name: {crate_name}");
243                for dep in meta.deps {
244                    let prelude = match &meta.extern_prelude {
245                        Some(v) => v.contains(&dep),
246                        None => true,
247                    };
248                    let dep = CrateName::normalize_dashes(&dep);
249                    crate_deps.push((crate_name.clone(), dep, prelude))
250                }
251            } else if meta.path == "/main.rs" || meta.path == "/lib.rs" {
252                assert!(default_crate_root.is_none());
253                default_crate_root = Some(file_id);
254                default_edition = meta.edition;
255                default_cfg.append(meta.cfg);
256                default_env.extend_from_other(&meta.env);
257            }
258
259            source_change.change_file(file_id, Some(text));
260            let path = VfsPath::new_virtual_path(meta.path);
261            file_set.insert(file_id, path);
262            files.push(EditionedFileId::new(db, file_id, meta.edition));
263            file_id = FileId::from_raw(file_id.index() + 1);
264        }
265
266        let mini_core = mini_core.map(|mini_core| {
267            let core_file = file_id;
268            file_id = FileId::from_raw(file_id.index() + 1);
269
270            let mut fs = FileSet::default();
271            fs.insert(core_file, VfsPath::new_virtual_path("/sysroot/core/lib.rs".to_owned()));
272            roots.push(SourceRoot::new_library(fs));
273
274            sysroot_files.push(core_file);
275
276            source_change.change_file(core_file, Some(mini_core.source_code(minicore_raw)));
277
278            let core_crate = crate_graph.add_crate_root(
279                core_file,
280                Edition::CURRENT,
281                Some(CrateDisplayName::from_canonical_name("core")),
282                None,
283                Default::default(),
284                Default::default(),
285                Env::from_iter([(
286                    String::from("__ra_is_test_fixture"),
287                    String::from("__ra_is_test_fixture"),
288                )]),
289                CrateOrigin::Lang(LangCrateOrigin::Core),
290                false,
291                proc_macro_cwd.clone(),
292                crate_ws_data.clone(),
293            );
294
295            (
296                move || {
297                    DependencyBuilder::with_prelude(
298                        CrateName::new("core").unwrap(),
299                        core_crate,
300                        true,
301                        true,
302                    )
303                },
304                core_crate,
305            )
306        });
307
308        if crates.is_empty() {
309            let crate_root = default_crate_root
310                .expect("missing default crate root, specify a main.rs or lib.rs");
311            let root = crate_graph.add_crate_root(
312                crate_root,
313                default_edition,
314                Some(CrateName::new("ra_test_fixture").unwrap().into()),
315                None,
316                default_cfg.clone(),
317                Some(default_cfg),
318                default_env,
319                CrateOrigin::Local { repo: None, name: None },
320                false,
321                proc_macro_cwd.clone(),
322                crate_ws_data.clone(),
323            );
324            if let Some((mini_core, _)) = mini_core {
325                crate_graph.add_dep(root, mini_core()).unwrap();
326            }
327        } else {
328            // Insert minicore first to match with `project-model::workspace`
329            if let Some((mini_core, core_crate)) = mini_core {
330                let all_crates = crate_graph.iter().collect::<Vec<_>>();
331                for krate in all_crates {
332                    if krate == core_crate {
333                        continue;
334                    }
335                    crate_graph.add_dep(krate, mini_core()).unwrap();
336                }
337            }
338
339            for (from, to, prelude) in crate_deps {
340                let from_id = crates[&from];
341                let to_id = crates[&to];
342                let sysroot = crate_graph[to_id].basic.origin.is_lang();
343                crate_graph
344                    .add_dep(
345                        from_id,
346                        DependencyBuilder::with_prelude(to.clone(), to_id, prelude, sysroot),
347                    )
348                    .unwrap();
349            }
350        }
351
352        let mut proc_macros = ProcMacrosBuilder::default();
353        if !proc_macro_names.is_empty() {
354            let proc_lib_file = file_id;
355
356            proc_macro_defs.extend(default_test_proc_macros());
357            let (proc_macro, source) = filter_test_proc_macros(&proc_macro_names, proc_macro_defs);
358            let mut fs = FileSet::default();
359            fs.insert(
360                proc_lib_file,
361                VfsPath::new_virtual_path("/sysroot/proc_macros/lib.rs".to_owned()),
362            );
363            roots.push(SourceRoot::new_library(fs));
364
365            sysroot_files.push(proc_lib_file);
366
367            source_change.change_file(proc_lib_file, Some(source));
368
369            let all_crates = crate_graph.iter().collect::<Vec<_>>();
370
371            let proc_macros_crate = crate_graph.add_crate_root(
372                proc_lib_file,
373                Edition::CURRENT,
374                Some(CrateDisplayName::from_canonical_name("proc_macros")),
375                None,
376                Default::default(),
377                Default::default(),
378                Env::from_iter([(
379                    String::from("__ra_is_test_fixture"),
380                    String::from("__ra_is_test_fixture"),
381                )]),
382                CrateOrigin::Local { repo: None, name: None },
383                true,
384                proc_macro_cwd,
385                crate_ws_data,
386            );
387            proc_macros.insert(proc_macros_crate, Ok(proc_macro));
388
389            for krate in all_crates {
390                crate_graph
391                    .add_dep(
392                        krate,
393                        DependencyBuilder::new(
394                            CrateName::new("proc_macros").unwrap(),
395                            proc_macros_crate,
396                        ),
397                    )
398                    .unwrap();
399            }
400        }
401
402        let _ = file_id;
403
404        let root = match current_source_root_kind {
405            SourceRootKind::Local => SourceRoot::new_local(mem::take(&mut file_set)),
406            SourceRootKind::Library => SourceRoot::new_library(mem::take(&mut file_set)),
407        };
408        roots.push(root);
409
410        let mut change = ChangeWithProcMacros { source_change, proc_macros: Some(proc_macros) };
411
412        change.source_change.set_roots(roots);
413        change.source_change.set_crate_graph(crate_graph);
414
415        ChangeFixture { file_position, file_lines, files, change, sysroot_files }
416    }
417}

598fn filter_test_proc_macros(
599    proc_macro_names: &[String],
600    proc_macro_defs: Vec<(String, ProcMacro)>,
601) -> (Vec<ProcMacro>, String) {
602    // The source here is only required so that paths to the macros exist and are resolvable.
603    let mut source = String::new();
604    let mut proc_macros = Vec::new();
605
606    for (c, p) in proc_macro_defs {
607        if !proc_macro_names.iter().any(|name| name == &stdx::to_lower_snake_case(p.name.as_str()))
608        {
609            continue;
610        }
611        proc_macros.push(p);
612        source += &c;
613    }
614
615    (proc_macros, source)
616}

618#[derive(Debug, Clone, Copy)]
619enum SourceRootKind {
620    Local,
621    Library,
622}
623
624#[derive(Debug)]
625struct FileMeta {
626    path: String,
627    krate: Option<(String, CrateOrigin, Option<String>)>,
628    deps: Vec<String>,
629    extern_prelude: Option<Vec<String>>,
630    cfg: CfgOptions,
631    edition: Edition,
632    env: Env,
633    introduce_new_source_root: Option<SourceRootKind>,
634}
635
636impl FileMeta {
637    fn from_fixture(f: Fixture, current_source_root_kind: SourceRootKind) -> Self {
638        let mut cfg = CfgOptions::default();
639        for (k, v) in f.cfgs {
640            if let Some(v) = v {
641                cfg.insert_key_value(Symbol::intern(&k), Symbol::intern(&v));
642            } else {
643                cfg.insert_atom(Symbol::intern(&k));
644            }
645        }
646
647        let introduce_new_source_root = f.introduce_new_source_root.map(|kind| match &*kind {
648            "local" => SourceRootKind::Local,
649            "library" => SourceRootKind::Library,
650            invalid => panic!("invalid source root kind '{invalid}'"),
651        });
652        let current_source_root_kind =
653            introduce_new_source_root.unwrap_or(current_source_root_kind);
654
655        let deps = f.deps;
656        Self {
657            path: f.path,
658            krate: f.krate.map(|it| parse_crate(it, current_source_root_kind, f.library)),
659            extern_prelude: f.extern_prelude,
660            deps,
661            cfg,
662            edition: f.edition.map_or(Edition::CURRENT, |v| Edition::from_str(&v).unwrap()),
663            env: f.env.into_iter().collect(),
664            introduce_new_source_root,
665        }
666    }
667}

675fn parse_crate(
676    crate_str: String,
677    current_source_root_kind: SourceRootKind,
678    explicit_non_workspace_member: bool,
679) -> (String, CrateOrigin, Option<String>) {
680    let (crate_str, force_non_lang_origin) = if let Some(s) = crate_str.strip_prefix("r#") {
681        (s.to_owned(), ForceNoneLangOrigin::Yes)
682    } else {
683        (crate_str, ForceNoneLangOrigin::No)
684    };
685
686    // syntax:
687    //   "my_awesome_crate"
688    //   "my_awesome_crate@0.0.1,http://example.com"
689    let (name, repo, version) = if let Some((name, remain)) = crate_str.split_once('@') {
690        let (version, repo) =
691            remain.split_once(',').expect("crate meta: found '@' without version and url");
692        (name.to_owned(), Some(repo.to_owned()), Some(version.to_owned()))
693    } else {
694        (crate_str, None, None)
695    };
696
697    let non_workspace_member = explicit_non_workspace_member
698        || matches!(current_source_root_kind, SourceRootKind::Library);
699
700    let origin = if force_non_lang_origin == ForceNoneLangOrigin::Yes {
701        let name = Symbol::intern(&name);
702        if non_workspace_member {
703            CrateOrigin::Library { repo, name }
704        } else {
705            CrateOrigin::Local { repo, name: Some(name) }
706        }
707    } else {
708        match LangCrateOrigin::from(&*name) {
709            LangCrateOrigin::Other => {
710                let name = Symbol::intern(&name);
711                if non_workspace_member {
712                    CrateOrigin::Library { repo, name }
713                } else {
714                    CrateOrigin::Local { repo, name: Some(name) }
715                }
716            }
717            origin => CrateOrigin::Lang(origin),
718        }
719    };
720
721    (name, origin, version)
722}

840#[derive(Debug)]
841struct MirrorProcMacroExpander;
842impl ProcMacroExpander for MirrorProcMacroExpander {
843    fn expand(
844        &self,
845        input: &TopSubtree,
846        _: Option<&TopSubtree>,
847        _: &Env,
848        _: Span,
849        _: Span,
850        _: Span,
851        _: String,
852    ) -> Result<TopSubtree, ProcMacroExpansionError> {
853        fn traverse(builder: &mut TopSubtreeBuilder, iter: TtIter<'_>) {
854            for tt in iter.collect_vec().into_iter().rev() {
855                match tt {
856                    TtElement::Leaf(leaf) => builder.push(leaf.clone()),
857                    TtElement::Subtree(subtree, subtree_iter) => {
858                        builder.open(subtree.delimiter.kind, subtree.delimiter.open);
859                        traverse(builder, subtree_iter);
860                        builder.close(subtree.delimiter.close);
861                    }
862                }
863            }
864        }
865        let mut builder = TopSubtreeBuilder::new(input.top_subtree().delimiter);
866        traverse(&mut builder, input.iter());
867        Ok(builder.build())
868    }
869
870    fn eq_dyn(&self, other: &dyn ProcMacroExpander) -> bool {
871        other.type_id() == TypeId::of::<Self>()
872    }
873}

875// Replaces every literal with an empty string literal and every identifier with its first letter,
876// but retains all tokens' span. Useful for testing we don't assume token hasn't been modified by
877// macros even if it retains its span.
878#[derive(Debug)]
879struct ShortenProcMacroExpander;
880impl ProcMacroExpander for ShortenProcMacroExpander {
881    fn expand(
882        &self,
883        input: &TopSubtree,
884        _: Option<&TopSubtree>,
885        _: &Env,
886        _: Span,
887        _: Span,
888        _: Span,
889        _: String,
890    ) -> Result<TopSubtree, ProcMacroExpansionError> {
891        let mut result = input.0.clone();
892        for it in &mut result {
893            if let TokenTree::Leaf(leaf) = it {
894                modify_leaf(leaf)
895            }
896        }
897        return Ok(tt::TopSubtree(result));
898
899        fn modify_leaf(leaf: &mut Leaf) {
900            match leaf {
901                Leaf::Literal(it) => {
902                    // XXX Currently replaces any literals with an empty string, but supporting
903                    // "shortening" other literals would be nice.
904                    it.symbol = Symbol::empty();
905                }
906                Leaf::Punct(_) => {}
907                Leaf::Ident(it) => {
908                    it.sym = Symbol::intern(&it.sym.as_str().chars().take(1).collect::<String>());
909                }
910            }
911        }
912    }
913
914    fn eq_dyn(&self, other: &dyn ProcMacroExpander) -> bool {
915        other.type_id() == TypeId::of::<Self>()
916    }
917}
