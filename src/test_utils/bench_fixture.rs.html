14pub fn big_struct_n(n: u32) -> String {
15    let mut buf = "pub struct RegisterBlock {".to_owned();
16    for i in 0..n {
17        format_to!(buf, "  /// Doc comment for {}.\n", i);
18        format_to!(buf, "  pub s{}: S{},\n", i, i);
19    }
20    buf.push_str("}\n\n");
21    for i in 0..n {
22        format_to!(
23            buf,
24            "
25
26#[repr(transparent)]
27struct S{} {{
28    field: u32,
29}}",
30            i
31        );
32    }
33
34    buf
35}
