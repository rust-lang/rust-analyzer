1//! Defines `Fixture` -- a convenient way to describe the initial state of
2//! rust-analyzer database from a single string.
3//!
4//! Fixtures are strings containing rust source code with optional metadata.
5//! A fixture without metadata is parsed into a single source file.
6//! Use this to test functionality local to one file.
7//!
8//! Simple Example:
9//!
10//! ```ignore
11//! r#"
12//! fn main() {
13//!     println!("Hello World")
14//! }
15//! "#
16//! ```
17//!
18//! Metadata can be added to a fixture after a `//-` comment.
19//! The basic form is specifying filenames,
20//! which is also how to define multiple files in a single test fixture
21//!
22//! Example using two files in the same crate:
23//!
24//! ```ignore
25//! "
26//! //- /main.rs
27//! mod foo;
28//! fn main() {
29//!     foo::bar();
30//! }
31//!
32//! //- /foo.rs
33//! pub fn bar() {}
34//! "
35//! ```
36//!
37//! Example using two crates with one file each, with one crate depending on the other:
38//!
39//! ```ignore
40//! r#"
41//! //- /main.rs crate:a deps:b
42//! fn main() {
43//!     b::foo();
44//! }
45//! //- /lib.rs crate:b
46//! pub fn b() {
47//!     println!("Hello World")
48//! }
49//! "#
50//! ```
51//!
52//! Metadata allows specifying all settings and variables
53//! that are available in a real rust project. See [`Fixture`]
54//! for the syntax.
55//!
56//! Example using some available metadata:
57//!
58//! ```ignore
59//! "
60//! //- /lib.rs crate:foo deps:bar,baz cfg:foo=a,bar=b env:OUTDIR=path/to,OTHER=foo
61//! fn insert_source_code_here() {}
62//! "
63//! ```

70#[derive(Debug, Eq, PartialEq)]
71pub struct Fixture {
72    /// Specifies the path for this file. It must start with "/".
73    pub path: String,
74    /// Defines a new crate and make this file its root module.
75    ///
76    /// Version and repository URL of the crate can optionally be specified; if
77    /// either one is specified, the other must also be specified.
78    ///
79    /// Syntax:
80    /// - `crate:my_awesome_lib`
81    /// - `crate:my_awesome_lib@0.0.1,https://example.com/repo.git`
82    pub krate: Option<String>,
83    /// Specifies dependencies of this crate. This must be used with `crate` meta.
84    ///
85    /// Syntax: `deps:hir-def,ide-assists`
86    pub deps: Vec<String>,
87    /// Limits crates in the extern prelude. The set of crate names must be a
88    /// subset of `deps`. This must be used with `crate` meta.
89    ///
90    /// If this is not specified, all the dependencies will be in the extern prelude.
91    ///
92    /// Syntax: `extern-prelude:hir-def,ide-assists`
93    pub extern_prelude: Option<Vec<String>>,
94    /// Specifies configuration options to be enabled. Options may have associated
95    /// values.
96    ///
97    /// Syntax: `cfg:test,dbg=false,opt_level=2`
98    pub cfgs: Vec<(String, Option<String>)>,
99    /// Specifies the edition of this crate. This must be used with `crate` meta. If
100    /// this is not specified, ([`base_db::input::Edition::CURRENT`]) will be used.
101    /// This must be used with `crate` meta.
102    ///
103    /// Syntax: `edition:2021`
104    pub edition: Option<String>,
105    /// Specifies environment variables.
106    ///
107    /// Syntax: `env:PATH=/bin,RUST_LOG=debug`
108    pub env: FxHashMap<String, String>,
109    /// Introduces a new [source root](base_db::input::SourceRoot). This file **and
110    /// the following files** will belong the new source root. This must be used
111    /// with `crate` meta.
112    ///
113    /// Use this if you want to test something that uses `SourceRoot::is_library()`
114    /// to check editability.
115    ///
116    /// Note that files before the first fixture with `new_source_root` meta will
117    /// belong to an implicitly defined local source root.
118    ///
119    /// Syntax:
120    /// - `new_source_root:library`
121    /// - `new_source_root:local`
122    pub introduce_new_source_root: Option<String>,
123    /// Explicitly declares this crate as a library outside current workspace. This
124    /// must be used with `crate` meta.
125    ///
126    /// This is implied if this file belongs to a library source root.
127    ///
128    /// Use this if you want to test something that checks if a crate is a workspace
129    /// member via [`CrateOrigin`](base_db::input::CrateOrigin).
130    ///
131    /// Syntax: `library`
132    pub library: bool,
133    /// Actual file contents. All meta comments are stripped.
134    pub text: String,
135    /// The line number in the original fixture of the beginning of this fixture.
136    pub line: usize,
137}

145#[derive(Debug)]
146pub struct FixtureWithProjectMeta {
147    pub fixture: Vec<Fixture>,
148    pub mini_core: Option<MiniCore>,
149    pub proc_macro_names: Vec<String>,
150    pub toolchain: Option<String>,
151    /// Specifies LLVM data layout to be used.
152    ///
153    /// You probably don't want to manually specify this. See LLVM manual for the
154    /// syntax, if you must: <https://llvm.org/docs/LangRef.html#data-layout>
155    pub target_data_layout: String,
156    /// Specifies the target architecture.
157    pub target_arch: String,
158}
159
160impl FixtureWithProjectMeta {
161    /// Parses text which looks like this:
162    ///
163    ///  ```text
164    ///  //- some meta
165    ///  line 1
166    ///  line 2
167    ///  //- other meta
168    ///  ```
169    ///
170    /// Fixture can also start with a proc_macros and minicore declaration (in that order):
171    ///
172    /// ```text
173    /// //- toolchain: nightly
174    /// //- proc_macros: identity
175    /// //- minicore: sized
176    /// ```
177    ///
178    /// That will set toolchain to nightly and include predefined proc macros and a subset of
179    /// `libcore` into the fixture, see `minicore.rs` for what's available. Note that toolchain
180    /// defaults to stable.
181    pub fn parse(#[rust_analyzer::rust_fixture] ra_fixture: &str) -> Self {
182        let fixture = trim_indent(ra_fixture);
183        let mut fixture = fixture.as_str();
184        let mut toolchain = None;
185        let mut target_data_layout =
186            "e-m:e-p270:32:32-p271:32:32-p272:64:64-i64:64-f80:128-n8:16:32:64-S128".to_owned();
187        let mut target_arch = "x86_64".to_owned();
188        let mut mini_core = None;
189        let mut res: Vec<Fixture> = Vec::new();
190        let mut proc_macro_names = vec![];
191        let mut first_row = 0;
192
193        if let Some(meta) = fixture.strip_prefix("//- toolchain:") {
194            first_row += 1;
195            let (meta, remain) = meta.split_once('\n').unwrap();
196            toolchain = Some(meta.trim().to_owned());
197            fixture = remain;
198        }
199
200        if let Some(meta) = fixture.strip_prefix("//- target_data_layout:") {
201            first_row += 1;
202            let (meta, remain) = meta.split_once('\n').unwrap();
203            meta.trim().clone_into(&mut target_data_layout);
204            fixture = remain;
205        }
206
207        if let Some(meta) = fixture.strip_prefix("//- target_arch:") {
208            first_row += 1;
209            let (meta, remain) = meta.split_once('\n').unwrap();
210            meta.trim().clone_into(&mut target_arch);
211            fixture = remain;
212        }
213
214        if let Some(meta) = fixture.strip_prefix("//- proc_macros:") {
215            first_row += 1;
216            let (meta, remain) = meta.split_once('\n').unwrap();
217            proc_macro_names = meta.split(',').map(|it| it.trim().to_owned()).collect();
218            fixture = remain;
219        }
220
221        if let Some(meta) = fixture.strip_prefix("//- minicore:") {
222            first_row += 1;
223            let (meta, remain) = meta.split_once('\n').unwrap();
224            mini_core = Some(MiniCore::parse(meta));
225            fixture = remain;
226        }
227
228        let default =
229            if fixture.contains("//- /") { None } else { Some((first_row - 1, "//- /main.rs")) };
230
231        for (ix, line) in
232            default.into_iter().chain((first_row..).zip(fixture.split_inclusive('\n')))
233        {
234            if line.contains("//-") {
235                assert!(
236                    line.starts_with("//-"),
237                    "Metadata line {ix} has invalid indentation. \
238                     All metadata lines need to have the same indentation.\n\
239                     The offending line: {line:?}"
240                );
241            }
242
243            if let Some(line) = line.strip_prefix("//-") {
244                let meta = Self::parse_meta_line(line, (ix + 1).try_into().unwrap());
245                res.push(meta);
246            } else {
247                if matches!(line.strip_prefix("// "), Some(l) if l.trim().starts_with('/')) {
248                    panic!("looks like invalid metadata line: {line:?}");
249                }
250
251                if let Some(entry) = res.last_mut() {
252                    entry.text.push_str(line);
253                }
254            }
255        }
256
257        Self {
258            fixture: res,
259            mini_core,
260            proc_macro_names,
261            toolchain,
262            target_data_layout,
263            target_arch,
264        }
265    }
266
267    //- /lib.rs crate:foo deps:bar,baz cfg:foo=a,bar=b env:OUTDIR=path/to,OTHER=foo
268    fn parse_meta_line(meta: &str, line: usize) -> Fixture {
269        let meta = meta.trim();
270        let mut components = meta.split_ascii_whitespace();
271
272        let path = components.next().expect("fixture meta must start with a path").to_owned();
273        assert!(path.starts_with('/'), "fixture path does not start with `/`: {path:?}");
274
275        let mut krate = None;
276        let mut deps = Vec::new();
277        let mut extern_prelude = None;
278        let mut edition = None;
279        let mut cfgs = Vec::new();
280        let mut env = FxHashMap::default();
281        let mut introduce_new_source_root = None;
282        let mut library = false;
283        for component in components {
284            if component == "library" {
285                library = true;
286                continue;
287            }
288
289            let (key, value) =
290                component.split_once(':').unwrap_or_else(|| panic!("invalid meta line: {meta:?}"));
291            match key {
292                "crate" => krate = Some(value.to_owned()),
293                "deps" => deps = value.split(',').map(|it| it.to_owned()).collect(),
294                "extern-prelude" => {
295                    if value.is_empty() {
296                        extern_prelude = Some(Vec::new());
297                    } else {
298                        extern_prelude =
299                            Some(value.split(',').map(|it| it.to_owned()).collect::<Vec<_>>());
300                    }
301                }
302                "edition" => edition = Some(value.to_owned()),
303                "cfg" => {
304                    for entry in value.split(',') {
305                        match entry.split_once('=') {
306                            Some((k, v)) => cfgs.push((k.to_owned(), Some(v.to_owned()))),
307                            None => cfgs.push((entry.to_owned(), None)),
308                        }
309                    }
310                }
311                "env" => {
312                    for key in value.split(',') {
313                        if let Some((k, v)) = key.split_once('=') {
314                            env.insert(k.into(), v.into());
315                        }
316                    }
317                }
318                "new_source_root" => introduce_new_source_root = Some(value.to_owned()),
319                _ => panic!("bad component: {component:?}"),
320            }
321        }
322
323        for prelude_dep in extern_prelude.iter().flatten() {
324            assert!(
325                deps.contains(prelude_dep),
326                "extern-prelude {extern_prelude:?} must be a subset of deps {deps:?}"
327            );
328        }
329
330        Fixture {
331            path,
332            text: String::new(),
333            line,
334            krate,
335            deps,
336            extern_prelude,
337            cfgs,
338            edition,
339            env,
340            introduce_new_source_root,
341            library,
342        }
343    }
344}

346impl MiniCore {
347    pub const RAW_SOURCE: &'static str = include_str!("./minicore.rs");
348
349    fn has_flag(&self, flag: &str) -> bool {
350        self.activated_flags.iter().any(|it| it == flag)
351    }
352
353    pub fn from_flags<'a>(flags: impl IntoIterator<Item = &'a str>) -> Self {
354        MiniCore {
355            activated_flags: flags.into_iter().map(|x| x.to_owned()).collect(),
356            valid_flags: Vec::new(),
357        }
358    }
359
360    #[track_caller]
361    fn assert_valid_flag(&self, flag: &str) {
362        if !self.valid_flags.iter().any(|it| it == flag) {
363            panic!("invalid flag: {flag:?}, valid flags: {:?}", self.valid_flags);
364        }
365    }
366
367    fn parse(line: &str) -> MiniCore {
368        let mut res = MiniCore { activated_flags: Vec::new(), valid_flags: Vec::new() };
369
370        for entry in line.trim().split(", ") {
371            if res.has_flag(entry) {
372                panic!("duplicate minicore flag: {entry:?}");
373            }
374            res.activated_flags.push(entry.to_owned());
375        }
376
377        res
378    }
379
380    pub fn available_flags(raw_source: &str) -> impl Iterator<Item = &str> {
381        let lines = raw_source.split_inclusive('\n');
382        lines
383            .map_while(|x| x.strip_prefix("//!"))
384            .skip_while(|line| !line.contains("Available flags:"))
385            .skip(1)
386            .map(|x| x.split_once(':').unwrap().0.trim())
387    }
388
389    /// Strips parts of minicore.rs which are flagged by inactive flags.
390    ///
391    /// This is probably over-engineered to support flags dependencies.
392    pub fn source_code(mut self, raw_source: &str) -> String {
393        let mut buf = String::new();
394        let mut lines = raw_source.split_inclusive('\n');
395
396        let mut implications = Vec::new();
397
398        // Parse `//!` preamble and extract flags and dependencies.
399        let trim_doc: fn(&str) -> Option<&str> = |line| match line.strip_prefix("//!") {
400            Some(it) => Some(it),
401            None => {
402                assert!(line.trim().is_empty(), "expected empty line after minicore header");
403                None
404            }
405        };
406        for line in lines
407            .by_ref()
408            .map_while(trim_doc)
409            .skip_while(|line| !line.contains("Available flags:"))
410            .skip(1)
411        {
412            let (flag, deps) = line.split_once(':').unwrap();
413            let flag = flag.trim();
414
415            self.valid_flags.push(flag.to_owned());
416            implications.extend(
417                iter::repeat(flag)
418                    .zip(deps.split(", ").map(str::trim).filter(|dep| !dep.is_empty())),
419            );
420        }
421
422        for (_, dep) in &implications {
423            self.assert_valid_flag(dep);
424        }
425
426        for flag in &self.activated_flags {
427            self.assert_valid_flag(flag);
428        }
429
430        // Fixed point loop to compute transitive closure of flags.
431        loop {
432            let mut changed = false;
433            for &(u, v) in &implications {
434                if self.has_flag(u) && !self.has_flag(v) {
435                    self.activated_flags.push(v.to_owned());
436                    changed = true;
437                }
438            }
439            if !changed {
440                break;
441            }
442        }
443
444        let mut active_regions = Vec::new();
445        let mut inactive_regions = Vec::new();
446        let mut seen_regions = Vec::new();
447        for line in lines {
448            let trimmed = line.trim();
449            if let Some(region) = trimmed.strip_prefix("// region:") {
450                if let Some(region) = region.strip_prefix('!') {
451                    inactive_regions.push(region);
452                    continue;
453                } else {
454                    active_regions.push(region);
455                    continue;
456                }
457            }
458            if let Some(region) = trimmed.strip_prefix("// endregion:") {
459                let (prev, region) = if let Some(region) = region.strip_prefix('!') {
460                    (inactive_regions.pop().unwrap(), region)
461                } else {
462                    (active_regions.pop().unwrap(), region)
463                };
464                assert_eq!(prev, region, "unbalanced region pairs");
465                continue;
466            }
467
468            let mut active_line_region = 0;
469            let mut inactive_line_region = 0;
470            if let Some(idx) = trimmed.find("// :!") {
471                let regions = trimmed[idx + "// :!".len()..].split(", ");
472                inactive_line_region += regions.clone().count();
473                inactive_regions.extend(regions);
474            } else if let Some(idx) = trimmed.find("// :") {
475                let regions = trimmed[idx + "// :".len()..].split(", ");
476                active_line_region += regions.clone().count();
477                active_regions.extend(regions);
478            }
479
480            let mut keep = true;
481            for &region in &active_regions {
482                assert!(!region.starts_with(' '), "region marker starts with a space: {region:?}");
483                self.assert_valid_flag(region);
484                seen_regions.push(region);
485                keep &= self.has_flag(region);
486            }
487            for &region in &inactive_regions {
488                assert!(!region.starts_with(' '), "region marker starts with a space: {region:?}");
489                self.assert_valid_flag(region);
490                seen_regions.push(region);
491                keep &= !self.has_flag(region);
492            }
493
494            if keep {
495                buf.push_str(line);
496            }
497            if active_line_region > 0 {
498                active_regions.drain(active_regions.len() - active_line_region..);
499            }
500            if inactive_line_region > 0 {
501                inactive_regions.drain(inactive_regions.len() - active_line_region..);
502            }
503        }
504
505        if !active_regions.is_empty() {
506            panic!("unclosed regions: {active_regions:?} Add an `endregion` comment");
507        }
508        if !inactive_regions.is_empty() {
509            panic!("unclosed regions: {inactive_regions:?} Add an `endregion` comment");
510        }
511
512        for flag in &self.valid_flags {
513            if !seen_regions.iter().any(|it| it == flag) {
514                panic!("unused minicore flag: {flag:?}");
515            }
516        }
517        buf
518    }
519}

536#[test]
537fn parse_fixture_gets_full_meta() {
538    let FixtureWithProjectMeta {
539        fixture: parsed,
540        mini_core,
541        proc_macro_names,
542        toolchain,
543        target_data_layout: _,
544        target_arch: _,
545    } = FixtureWithProjectMeta::parse(
546        r#"
547//- toolchain: nightly
548//- proc_macros: identity
549//- minicore: coerce_unsized
550//- /lib.rs crate:foo deps:bar,baz cfg:foo=a,bar=b,atom env:OUTDIR=path/to,OTHER=foo
551mod m;
552"#,
553    );
554    assert_eq!(toolchain, Some("nightly".to_owned()));
555    assert_eq!(proc_macro_names, vec!["identity".to_owned()]);
556    assert_eq!(mini_core.unwrap().activated_flags, vec!["coerce_unsized".to_owned()]);
557    assert_eq!(1, parsed.len());
558
559    let meta = &parsed[0];
560    assert_eq!("mod m;\n", meta.text);
561
562    assert_eq!("foo", meta.krate.as_ref().unwrap());
563    assert_eq!("/lib.rs", meta.path);
564    assert_eq!(2, meta.env.len());
