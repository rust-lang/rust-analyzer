11/// A range in text, represented as a pair of [`TextSize`][struct@TextSize].
12///
13/// It is a logic error for `start` to be greater than `end`.
14#[derive(Default, Copy, Clone, Eq, PartialEq, Hash)]
15pub struct TextRange {
16    // Invariant: start <= end
17    start: TextSize,
18    end: TextSize,
19}
20
21impl fmt::Debug for TextRange {
22    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
23        write!(f, "{}..{}", self.start().raw, self.end().raw)
24    }
25}

27impl TextRange {
28    /// Creates a new `TextRange` with the given `start` and `end` (`start..end`).
29    ///
30    /// # Panics
31    ///
32    /// Panics if `end < start`.
33    ///
34    /// # Examples
35    ///
36    /// ```rust
37    /// # use text_size::*;
38    /// let start = TextSize::from(5);
39    /// let end = TextSize::from(10);
40    /// let range = TextRange::new(start, end);
41    ///
42    /// assert_eq!(range.start(), start);
43    /// assert_eq!(range.end(), end);
44    /// assert_eq!(range.len(), end - start);
45    /// ```
46    #[inline]
47    pub const fn new(start: TextSize, end: TextSize) -> TextRange {
48        assert!(start.raw <= end.raw);
49        TextRange { start, end }
50    }
51
52    /// Create a new `TextRange` with the given `offset` and `len` (`offset..offset + len`).
53    ///
54    /// # Examples
55    ///
56    /// ```rust
57    /// # use text_size::*;
58    /// let text = "0123456789";
59    ///
60    /// let offset = TextSize::from(2);
61    /// let length = TextSize::from(5);
62    /// let range = TextRange::at(offset, length);
63    ///
64    /// assert_eq!(range, TextRange::new(offset, offset + length));
65    /// assert_eq!(&text[range], "23456")
66    /// ```
67    #[inline]
68    pub const fn at(offset: TextSize, len: TextSize) -> TextRange {
69        TextRange::new(offset, TextSize::new(offset.raw + len.raw))
70    }
71
72    /// Create a zero-length range at the specified offset (`offset..offset`).
73    ///
74    /// # Examples
75    ///
76    /// ```rust
77    /// # use text_size::*;
78    /// let point: TextSize;
79    /// # point = TextSize::from(3);
80    /// let range = TextRange::empty(point);
81    /// assert!(range.is_empty());
82    /// assert_eq!(range, TextRange::new(point, point));
83    /// ```
84    #[inline]
85    pub const fn empty(offset: TextSize) -> TextRange {
86        TextRange { start: offset, end: offset }
87    }
88
89    /// Create a range up to the given end (`..end`).
90    ///
91    /// # Examples
92    ///
93    /// ```rust
94    /// # use text_size::*;
95    /// let point: TextSize;
96    /// # point = TextSize::from(12);
97    /// let range = TextRange::up_to(point);
98    ///
99    /// assert_eq!(range.len(), point);
100    /// assert_eq!(range, TextRange::new(0.into(), point));
101    /// assert_eq!(range, TextRange::at(0.into(), point));
102    /// ```
103    #[inline]
104    pub const fn up_to(end: TextSize) -> TextRange {
105        TextRange { start: TextSize::new(0), end }
106    }
107}

109/// Identity methods.
110impl TextRange {
111    /// The start point of this range.
112    #[inline]
113    pub const fn start(self) -> TextSize {
114        self.start
115    }
116
117    /// The end point of this range.
118    #[inline]
119    pub const fn end(self) -> TextSize {
120        self.end
121    }
122
123    /// The size of this range.
124    #[inline]
125    pub const fn len(self) -> TextSize {
126        // HACK for const fn: math on primitives only
127        TextSize { raw: self.end().raw - self.start().raw }
128    }
129
130    /// Check if this range is empty.
131    #[inline]
132    pub const fn is_empty(self) -> bool {
133        // HACK for const fn: math on primitives only
134        self.start().raw == self.end().raw
135    }
136}

138/// Manipulation methods.
139impl TextRange {
140    /// Check if this range contains an offset.
141    ///
142    /// The end index is considered excluded.
143    ///
144    /// # Examples
145    ///
146    /// ```rust
147    /// # use text_size::*;
148    /// let (start, end): (TextSize, TextSize);
149    /// # start = 10.into(); end = 20.into();
150    /// let range = TextRange::new(start, end);
151    /// assert!(range.contains(start));
152    /// assert!(!range.contains(end));
153    /// ```
154    #[inline]
155    pub fn contains(self, offset: TextSize) -> bool {
156        self.start() <= offset && offset < self.end()
157    }
158
159    /// Check if this range contains an offset.
160    ///
161    /// The end index is considered included.
162    ///
163    /// # Examples
164    ///
165    /// ```rust
166    /// # use text_size::*;
167    /// let (start, end): (TextSize, TextSize);
168    /// # start = 10.into(); end = 20.into();
169    /// let range = TextRange::new(start, end);
170    /// assert!(range.contains_inclusive(start));
171    /// assert!(range.contains_inclusive(end));
172    /// ```
173    #[inline]
174    pub fn contains_inclusive(self, offset: TextSize) -> bool {
175        self.start() <= offset && offset <= self.end()
176    }
177
178    /// Check if this range completely contains another range.
179    ///
180    /// # Examples
181    ///
182    /// ```rust
183    /// # use text_size::*;
184    /// let larger = TextRange::new(0.into(), 20.into());
185    /// let smaller = TextRange::new(5.into(), 15.into());
186    /// assert!(larger.contains_range(smaller));
187    /// assert!(!smaller.contains_range(larger));
188    ///
189    /// // a range always contains itself
190    /// assert!(larger.contains_range(larger));
191    /// assert!(smaller.contains_range(smaller));
192    /// ```
193    #[inline]
194    pub fn contains_range(self, other: TextRange) -> bool {
195        self.start() <= other.start() && other.end() <= self.end()
196    }
197
198    /// The range covered by both ranges, if it exists.
199    /// If the ranges touch but do not overlap, the output range is empty.
200    ///
201    /// # Examples
202    ///
203    /// ```rust
204    /// # use text_size::*;
205    /// assert_eq!(
206    ///     TextRange::intersect(
207    ///         TextRange::new(0.into(), 10.into()),
208    ///         TextRange::new(5.into(), 15.into()),
209    ///     ),
210    ///     Some(TextRange::new(5.into(), 10.into())),
211    /// );
212    /// ```
213    #[inline]
214    pub fn intersect(self, other: TextRange) -> Option<TextRange> {
215        let start = cmp::max(self.start(), other.start());
216        let end = cmp::min(self.end(), other.end());
217        if end < start {
218            return None;
219        }
220        Some(TextRange::new(start, end))
221    }
222
223    /// Extends the range to cover `other` as well.
224    ///
225    /// # Examples
226    ///
227    /// ```rust
228    /// # use text_size::*;
229    /// assert_eq!(
230    ///     TextRange::cover(
231    ///         TextRange::new(0.into(), 5.into()),
232    ///         TextRange::new(15.into(), 20.into()),
233    ///     ),
234    ///     TextRange::new(0.into(), 20.into()),
235    /// );
236    /// ```
237    #[inline]
238    pub fn cover(self, other: TextRange) -> TextRange {
239        let start = cmp::min(self.start(), other.start());
240        let end = cmp::max(self.end(), other.end());
241        TextRange::new(start, end)
242    }
243
244    /// Extends the range to cover `other` offsets as well.
245    ///
246    /// # Examples
247    ///
248    /// ```rust
249    /// # use text_size::*;
250    /// assert_eq!(
251    ///     TextRange::empty(0.into()).cover_offset(20.into()),
252    ///     TextRange::new(0.into(), 20.into()),
253    /// )
254    /// ```
255    #[inline]
256    pub fn cover_offset(self, offset: TextSize) -> TextRange {
257        self.cover(TextRange::empty(offset))
258    }
259
260    /// Add an offset to this range.
261    ///
262    /// Note that this is not appropriate for changing where a `TextRange` is
263    /// within some string; rather, it is for changing the reference anchor
264    /// that the `TextRange` is measured against.
265    ///
266    /// The unchecked version (`Add::add`) will _always_ panic on overflow,
267    /// in contrast to primitive integers, which check in debug mode only.
268    #[inline]
269    pub fn checked_add(self, offset: TextSize) -> Option<TextRange> {
270        Some(TextRange {
271            start: self.start.checked_add(offset)?,
272            end: self.end.checked_add(offset)?,
273        })
274    }
275
276    /// Subtract an offset from this range.
277    ///
278    /// Note that this is not appropriate for changing where a `TextRange` is
279    /// within some string; rather, it is for changing the reference anchor
280    /// that the `TextRange` is measured against.
281    ///
282    /// The unchecked version (`Sub::sub`) will _always_ panic on overflow,
283    /// in contrast to primitive integers, which check in debug mode only.
284    #[inline]
285    pub fn checked_sub(self, offset: TextSize) -> Option<TextRange> {
286        Some(TextRange {
287            start: self.start.checked_sub(offset)?,
288            end: self.end.checked_sub(offset)?,
289        })
290    }
291
292    /// Relative order of the two ranges (overlapping ranges are considered
293    /// equal).
294    ///
295    ///
296    /// This is useful when, for example, binary searching an array of disjoint
297    /// ranges.
298    ///
299    /// # Examples
300    ///
301    /// ```
302    /// # use text_size::*;
303    /// # use std::cmp::Ordering;
304    ///
305    /// let a = TextRange::new(0.into(), 3.into());
306    /// let b = TextRange::new(4.into(), 5.into());
307    /// assert_eq!(a.ordering(b), Ordering::Less);
308    ///
309    /// let a = TextRange::new(0.into(), 3.into());
310    /// let b = TextRange::new(3.into(), 5.into());
311    /// assert_eq!(a.ordering(b), Ordering::Less);
312    ///
313    /// let a = TextRange::new(0.into(), 3.into());
314    /// let b = TextRange::new(2.into(), 5.into());
315    /// assert_eq!(a.ordering(b), Ordering::Equal);
316    ///
317    /// let a = TextRange::new(0.into(), 3.into());
318    /// let b = TextRange::new(2.into(), 2.into());
319    /// assert_eq!(a.ordering(b), Ordering::Equal);
320    ///
321    /// let a = TextRange::new(2.into(), 3.into());
322    /// let b = TextRange::new(2.into(), 2.into());
323    /// assert_eq!(a.ordering(b), Ordering::Greater);
324    /// ```
325    #[inline]
326    pub fn ordering(self, other: TextRange) -> Ordering {
327        if self.end() <= other.start() {
328            Ordering::Less
329        } else if other.end() <= self.start() {
330            Ordering::Greater
331        } else {
332            Ordering::Equal
333        }
334    }
