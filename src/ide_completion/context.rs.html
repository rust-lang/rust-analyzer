191impl TypeLocation {
192    pub(crate) fn complete_lifetimes(&self) -> bool {
193        matches!(
194            self,
195            TypeLocation::GenericArg {
196                corresponding_param: Some(ast::GenericParam::LifetimeParam(_)),
197                ..
198            }
199        )
200    }
201
202    pub(crate) fn complete_consts(&self) -> bool {
203        matches!(
204            self,
205            TypeLocation::GenericArg {
206                corresponding_param: Some(ast::GenericParam::ConstParam(_)),
207                ..
208            } | TypeLocation::AssocConstEq
209        )
210    }
211
212    pub(crate) fn complete_types(&self) -> bool {
213        match self {
214            TypeLocation::GenericArg { corresponding_param: Some(param), .. } => {
215                matches!(param, ast::GenericParam::TypeParam(_))
216            }
217            TypeLocation::AssocConstEq => false,
218            TypeLocation::AssocTypeEq => true,
219            TypeLocation::ImplTrait => false,
220            _ => true,
221        }
222    }
223
224    pub(crate) fn complete_self_type(&self) -> bool {
225        self.complete_types() && !matches!(self, TypeLocation::ImplTarget | TypeLocation::ImplTrait)
226    }

500impl CompletionContext<'_> {
501    /// The range of the identifier that is being completed.
502    pub(crate) fn source_range(&self) -> TextRange {
503        let kind = self.original_token.kind();
504        match kind {
505            CHAR => {
506                // assume we are completing a lifetime but the user has only typed the '
507                cov_mark::hit!(completes_if_lifetime_without_idents);
508                TextRange::at(self.original_token.text_range().start(), TextSize::from(1))
509            }
510            LIFETIME_IDENT | UNDERSCORE | INT_NUMBER => self.original_token.text_range(),
511            // We want to consider all keywords in all editions.
512            _ if kind.is_any_identifier() => self.original_token.text_range(),
513            _ => TextRange::empty(self.position.offset),
514        }
515    }

522    pub(crate) fn def_is_visible(&self, item: &ScopeDef) -> Visible {
523        match item {
524            ScopeDef::ModuleDef(def) => match def {
525                hir::ModuleDef::Module(it) => self.is_visible(it),
526                hir::ModuleDef::Function(it) => self.is_visible(it),
527                hir::ModuleDef::Adt(it) => self.is_visible(it),
528                hir::ModuleDef::Variant(it) => self.is_visible(it),
529                hir::ModuleDef::Const(it) => self.is_visible(it),
530                hir::ModuleDef::Static(it) => self.is_visible(it),
531                hir::ModuleDef::Trait(it) => self.is_visible(it),
532                hir::ModuleDef::TypeAlias(it) => self.is_visible(it),
533                hir::ModuleDef::Macro(it) => self.is_visible(it),
534                hir::ModuleDef::BuiltinType(_) => Visible::Yes,
535            },
536            ScopeDef::GenericParam(_)
537            | ScopeDef::ImplSelfType(_)
538            | ScopeDef::AdtSelfType(_)
539            | ScopeDef::Local(_)
540            | ScopeDef::Label(_)
541            | ScopeDef::Unknown => Visible::Yes,

612    pub(crate) fn iterate_path_candidates(
613        &self,
614        ty: &hir::Type<'_>,
615        mut cb: impl FnMut(hir::AssocItem),
616    ) {
617        let mut seen = FxHashSet::default();
618        ty.iterate_path_candidates(
619            self.db,
620            &self.scope,
621            &self.traits_in_scope(),
622            Some(self.module),
623            None,
624            |item| {
625                // We might iterate candidates of a trait multiple times here, so deduplicate
626                // them.
627                if seen.insert(item) {
628                    cb(item)
629                }
630                None::<()>
631            },
632        );
633    }

635    /// A version of [`SemanticsScope::process_all_names`] that filters out `#[doc(hidden)]` items and
636    /// passes all doc-aliases along, to funnel it into [`Completions::add_path_resolution`].
637    pub(crate) fn process_all_names(&self, f: &mut dyn FnMut(Name, ScopeDef, Vec<SmolStr>)) {
638        let _p = tracing::info_span!("CompletionContext::process_all_names").entered();
639        self.scope.process_all_names(&mut |name, def| {
640            if self.is_scope_def_hidden(def) {
641                return;
642            }
643            let doc_aliases = self.doc_aliases_in_scope(def);
644            f(name, def, doc_aliases);
645        });
646    }

649        let _p = tracing::info_span!("CompletionContext::process_all_names_raw").entered();
650        self.scope.process_all_names(f);
651    }

669        }
670
671        if !vis.is_visible_from(self.db, self.module.into()) {
672            if !self.config.enable_private_editable {
673                return Visible::No;
674            }
675            // If the definition location is editable, also show private items
676            return if is_editable_crate(defining_crate, self.db) {
677                Visible::Editable
678            } else {
679                Visible::No
680            };
681        }
682
683        if self.is_doc_hidden(attrs, defining_crate) { Visible::No } else { Visible::Yes }
684    }
685
686    pub(crate) fn is_doc_hidden(&self, attrs: &hir::Attrs, defining_crate: hir::Crate) -> bool {
687        // `doc(hidden)` items are only completed within the defining crate.
688        self.krate != defining_crate && attrs.has_doc_hidden()
689    }
690
691    pub(crate) fn doc_aliases_in_scope(&self, scope_def: ScopeDef) -> Vec<SmolStr> {
692        if let Some(attrs) = scope_def.attrs(self.db) {
693            attrs.doc_aliases().map(|it| it.as_str().into()).collect()
694        } else {
695            vec![]
696        }
697    }
698}

700// CompletionContext construction
701impl<'db> CompletionContext<'db> {
702    pub(crate) fn new(
703        db: &'db RootDatabase,
704        position @ FilePosition { file_id, offset }: FilePosition,
705        config: &'db CompletionConfig<'db>,
706        trigger_character: Option<char>,
707    ) -> Option<(CompletionContext<'db>, CompletionAnalysis<'db>)> {
708        let _p = tracing::info_span!("CompletionContext::new").entered();
709        let sema = Semantics::new(db);
710
711        let editioned_file_id = sema.attach_first_edition(file_id)?;
712        let original_file = sema.parse(editioned_file_id);
713
714        // Insert a fake ident to get a valid parse tree. We will use this file
715        // to determine context, though the original_file will be used for
716        // actual completion.
717        let file_with_fake_ident = {
718            let (_, edition) = editioned_file_id.unpack(db);
719            let parse = db.parse(editioned_file_id);
720            parse.reparse(TextRange::empty(offset), COMPLETION_MARKER, edition).tree()
721        };
722
723        // always pick the token to the immediate left of the cursor, as that is what we are actually
724        // completing on
725        let original_token = original_file.syntax().token_at_offset(offset).left_biased()?;
726
727        // try to skip completions on path with invalid colons
728        // this approach works in normal path and inside token tree
729        if original_token.kind() == T![:] {
730            // return if no prev token before colon
731            let prev_token = original_token.prev_token()?;
732
733            // only has a single colon
734            if prev_token.kind() != T![:] {
735                return None;
736            }
737
738            // has 3 colon or 2 coloncolon in a row
739            // special casing this as per discussion in https://github.com/rust-lang/rust-analyzer/pull/13611#discussion_r1031845205
740            // and https://github.com/rust-lang/rust-analyzer/pull/13611#discussion_r1032812751
741            if prev_token
742                .prev_token()
743                .map(|t| t.kind() == T![:] || t.kind() == T![::])
744                .unwrap_or(false)
745            {
746                return None;
747            }
748        }
749
750        let AnalysisResult {
751            analysis,
752            expected: (expected_type, expected_name),
753            qualifier_ctx,
754            token,
755            original_offset,
756        } = expand_and_analyze(
757            &sema,
758            InFile::new(editioned_file_id.into(), original_file.syntax().clone()),
759            file_with_fake_ident.syntax().clone(),
760            offset,
761            &original_token,
762        )?;
763
764        // adjust for macro input, this still fails if there is no token written yet
765        let scope = sema.scope_at_offset(&token.parent()?, original_offset)?;
766
767        let krate = scope.krate();
768        let module = scope.module();
769        let containing_function = scope.containing_function();
770        let edition = krate.edition(db);
771
772        let toolchain = db.toolchain_channel(krate.into());
773        // `toolchain == None` means we're in some detached files. Since we have no information on
774        // the toolchain being used, let's just allow unstable items to be listed.
775        let is_nightly = matches!(toolchain, Some(base_db::ReleaseChannel::Nightly) | None);
776
777        let mut locals = FxHashMap::default();
778        scope.process_all_names(&mut |name, scope| {
779            if let ScopeDef::Local(local) = scope {
780                // synthetic names currently leak out as we lack synthetic hygiene, so filter them
781                // out here
782                if name.as_str().starts_with('<') {
783                    return;
784                }
785                locals.insert(name, local);
786            }
787        });
788
789        let depth_from_crate_root = iter::successors(Some(module), |m| m.parent(db))
790            // `BlockExpr` modules do not count towards module depth
791            .filter(|m| !matches!(m.definition_source(db).value, ModuleSource::BlockExpr(_)))
792            .count()
793            // exclude `m` itself
794            .saturating_sub(1);
795
796        let exclude_traits: FxHashSet<_> = config
797            .exclude_traits
798            .iter()
799            .filter_map(|path| {
800                hir::resolve_absolute_path(db, path.split("::").map(Symbol::intern)).find_map(
801                    |it| match it {
802                        hir::ItemInNs::Types(ModuleDef::Trait(t)) => Some(t),
803                        _ => None,
804                    },
805                )
806            })
807            .collect();
808
809        let mut exclude_flyimport: FxHashMap<_, _> = config
810            .exclude_flyimport
811            .iter()
812            .flat_map(|(path, kind)| {
813                hir::resolve_absolute_path(db, path.split("::").map(Symbol::intern))
814                    .map(|it| (it.into_module_def(), *kind))
815            })
816            .collect();
817        exclude_flyimport
818            .extend(exclude_traits.iter().map(|&t| (t.into(), AutoImportExclusionType::Always)));
819
820        // FIXME: This should be part of `CompletionAnalysis` / `expand_and_analyze`
821        let complete_semicolon = if config.add_semicolon_to_unit {
822            let inside_closure_ret = token.parent_ancestors().try_for_each(|ancestor| {
823                match_ast! {
824                    match ancestor {
825                        ast::BlockExpr(_) => ControlFlow::Break(false),
826                        ast::ClosureExpr(_) => ControlFlow::Break(true),
827                        _ => ControlFlow::Continue(())
828                    }
829                }
830            });
831
832            if inside_closure_ret == ControlFlow::Break(true) {
833                CompleteSemicolon::DoNotComplete
834            } else {
835                let next_non_trivia_token =
836                    std::iter::successors(token.next_token(), |it| it.next_token())
837                        .find(|it| !it.kind().is_trivia());
838                let in_match_arm = token.parent_ancestors().try_for_each(|ancestor| {
839                    if ast::MatchArm::can_cast(ancestor.kind()) {
840                        ControlFlow::Break(true)
841                    } else if matches!(
842                        ancestor.kind(),
843                        SyntaxKind::EXPR_STMT | SyntaxKind::BLOCK_EXPR
844                    ) {
845                        ControlFlow::Break(false)
846                    } else {
847                        ControlFlow::Continue(())
848                    }
849                });
850                // FIXME: This will assume expr macros are not inside match, we need to somehow go to the "parent" of the root node.
851                let in_match_arm = match in_match_arm {
852                    ControlFlow::Continue(()) => false,
853                    ControlFlow::Break(it) => it,
854                };
855                let complete_token = if in_match_arm { T![,] } else { T![;] };
856                if next_non_trivia_token.map(|it| it.kind()) == Some(complete_token) {
857                    CompleteSemicolon::DoNotComplete
858                } else if in_match_arm {
859                    CompleteSemicolon::CompleteComma
860                } else {
861                    CompleteSemicolon::CompleteSemi
862                }
863            }
864        } else {
865            CompleteSemicolon::DoNotComplete
866        };
867
868        let display_target = krate.to_display_target(db);
869        let ctx = CompletionContext {
870            sema,
871            scope,
872            db,
873            config,
874            position,
875            trigger_character,
876            original_token,
877            token,
878            krate,
879            module,
880            containing_function,
881            is_nightly,
882            edition,
883            expected_name,
884            expected_type,
885            qualifier_ctx,
886            locals,
887            depth_from_crate_root,
888            exclude_flyimport,
889            exclude_traits,
890            complete_semicolon,
891            display_target,
892        };
893        Some((ctx, analysis))
894    }
895}
