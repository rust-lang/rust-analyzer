10pub(crate) fn complete_item_list_in_expr(
11    acc: &mut Completions,
12    ctx: &CompletionContext<'_>,
13    path_ctx: &PathCompletionCtx<'_>,
14    expr_ctx: &PathExprCtx<'_>,
15) {
16    if !expr_ctx.in_block_expr {
17        return;
18    }
19    if !path_ctx.is_trivial_path() {
20        return;
21    }
22    add_keywords(acc, ctx, None);
23}

25pub(crate) fn complete_item_list(
26    acc: &mut Completions,
27    ctx: &CompletionContext<'_>,
28    path_ctx @ PathCompletionCtx { qualified, .. }: &PathCompletionCtx<'_>,
29    kind: &ItemListKind,
30) {
31    let _p = tracing::info_span!("complete_item_list").entered();
32
33    // We handle completions for trait-impls in [`item_list::trait_impl`]
34    if path_ctx.is_trivial_path() && !matches!(kind, ItemListKind::TraitImpl(_)) {
35        add_keywords(acc, ctx, Some(kind));
36    }
37
38    match qualified {
39        Qualified::With {
40            resolution: Some(hir::PathResolution::Def(hir::ModuleDef::Module(module))),
41            super_chain_len,
42            ..
43        } => {
44            for (name, def) in module.scope(ctx.db, Some(ctx.module)) {
45                match def {
46                    hir::ScopeDef::ModuleDef(hir::ModuleDef::Macro(m)) if m.is_fn_like(ctx.db) => {
47                        acc.add_macro(ctx, path_ctx, m, name)
48                    }
49                    hir::ScopeDef::ModuleDef(hir::ModuleDef::Module(m)) => {
50                        acc.add_module(ctx, path_ctx, m, name, vec![])
51                    }
52                    _ => (),
53                }
54            }
55
56            acc.add_super_keyword(ctx, *super_chain_len);
57        }
58        Qualified::Absolute => acc.add_crate_roots(ctx, path_ctx),
59        Qualified::No if ctx.qualifier_ctx.none() => {
60            ctx.process_all_names(&mut |name, def, doc_aliases| match def {
61                hir::ScopeDef::ModuleDef(hir::ModuleDef::Macro(m)) if m.is_fn_like(ctx.db) => {
62                    acc.add_macro(ctx, path_ctx, m, name)
63                }
64                hir::ScopeDef::ModuleDef(hir::ModuleDef::Module(m)) => {
65                    acc.add_module(ctx, path_ctx, m, name, doc_aliases)
66                }
67                _ => (),
68            });
69            acc.add_nameref_keywords_with_colon(ctx);
70        }
71        Qualified::TypeAnchor { .. } | Qualified::No | Qualified::With { .. } => {}
72    }
73}

75fn add_keywords(acc: &mut Completions, ctx: &CompletionContext<'_>, kind: Option<&ItemListKind>) {
76    let mut add_keyword = |kw, snippet| acc.add_keyword_snippet(ctx, kw, snippet);
77
78    let in_item_list = matches!(kind, Some(ItemListKind::SourceFile | ItemListKind::Module) | None);
79    let in_assoc_non_trait_impl = matches!(kind, Some(ItemListKind::Impl | ItemListKind::Trait));
80
81    let in_extern_block = matches!(kind, Some(ItemListKind::ExternBlock { .. }));
82    let in_unsafe_extern_block =
83        matches!(kind, Some(ItemListKind::ExternBlock { is_unsafe: true }));
84
85    let in_trait = matches!(kind, Some(ItemListKind::Trait));
86    let in_inherent_impl = matches!(kind, Some(ItemListKind::Impl));
87    let in_block = kind.is_none();
88
89    let no_vis_qualifiers = ctx.qualifier_ctx.vis_node.is_none();
90    let no_abi_qualifiers = ctx.qualifier_ctx.abi_node.is_none();
91    let has_extern_kw =
92        ctx.qualifier_ctx.abi_node.as_ref().is_some_and(|it| it.string_token().is_none());
93    let has_unsafe_kw = ctx.qualifier_ctx.unsafe_tok.is_some();
94    let has_async_kw = ctx.qualifier_ctx.async_tok.is_some();
95    let has_safe_kw = ctx.qualifier_ctx.safe_tok.is_some();
96
97    // Some keywords are invalid after non-vis qualifiers, so we handle them first.
98    if (has_unsafe_kw || has_safe_kw) && in_extern_block {
99        add_keyword("fn", "fn $1($2);");
100        add_keyword("static", "static $1: $2;");
101        return;
102    }
103
104    if has_unsafe_kw || has_async_kw {
105        if !has_unsafe_kw {
106            add_keyword("unsafe", "unsafe $0");
107        }
108        if !has_async_kw {
109            add_keyword("async", "async $0");
110        }
111
112        if in_item_list || in_assoc_non_trait_impl {
113            add_keyword("fn", "fn $1($2) {\n    $0\n}");
114        }
115
116        if has_unsafe_kw && in_item_list {
117            add_keyword("trait", "trait $1 {\n    $0\n}");
118            if no_vis_qualifiers {
119                add_keyword("impl", "impl $1 {\n    $0\n}");
120                add_keyword("impl for", "impl $1 for $2 {\n    $0\n}");
121            }
122        }
123
124        if !has_async_kw && no_vis_qualifiers && no_abi_qualifiers && in_item_list {
125            add_keyword("extern", "extern $0");
126        }
127
128        return;
129    }
130
131    // ...and the rest deals with cases without any non-vis qualifiers.
132
133    // Visibility qualifiers
134    if !in_trait && !in_block && no_vis_qualifiers {
135        add_keyword("pub(crate)", "pub(crate) $0");
136        add_keyword("pub(super)", "pub(super) $0");
137        add_keyword("pub", "pub $0");
138    }
139
140    // Keywords that are valid in `item_list`
141    if in_item_list {
142        add_keyword("enum", "enum $1 {\n    $0\n}");
143        add_keyword("mod", "mod $0");
144        add_keyword("static", "static $0");
145        add_keyword("struct", "struct $0");
146        add_keyword("trait", "trait $1 {\n    $0\n}");
147        add_keyword("union", "union $1 {\n    $0\n}");
148        add_keyword("use", "use $0;");
149        if no_vis_qualifiers {
150            add_keyword("impl", "impl $1 {\n    $0\n}");
151            add_keyword("impl for", "impl $1 for $2 {\n    $0\n}");
152        }
153    }
154
155    if in_extern_block {
156        add_keyword("unsafe", "unsafe $0");
157        if in_unsafe_extern_block {
158            add_keyword("safe", "safe $0");
159        }
160
161        add_keyword("fn", "fn $1($2);");
162        add_keyword("static", "static $1: $2;");
163    } else {
164        if !in_inherent_impl {
165            if !in_trait && no_abi_qualifiers {
166                add_keyword("extern", "extern $0");
167            }
168            add_keyword("type", "type $0");
169        }
170        if has_extern_kw {
171            add_keyword("crate", "crate $0;");
172        }
173
174        add_keyword("fn", "fn $1($2) {\n    $0\n}");
175        add_keyword("unsafe", "unsafe $0");
176        add_keyword("const", "const $0");
177        add_keyword("async", "async $0");
178    }
