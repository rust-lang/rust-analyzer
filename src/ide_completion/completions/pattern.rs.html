12/// Completes constants and paths in unqualified patterns.
13pub(crate) fn complete_pattern(
14    acc: &mut Completions,
15    ctx: &CompletionContext<'_>,
16    pattern_ctx: &PatternContext,
17) {
18    let mut add_keyword = |kw, snippet| acc.add_keyword_snippet(ctx, kw, snippet);
19
20    match pattern_ctx.parent_pat.as_ref() {
21        Some(Pat::RangePat(_) | Pat::BoxPat(_)) => (),
22        Some(Pat::RefPat(r)) => {
23            if r.mut_token().is_none() {
24                add_keyword("mut", "mut $0");
25            }
26        }
27        _ => {
28            let tok = ctx.token.text_range().start();
29            match (pattern_ctx.ref_token.as_ref(), pattern_ctx.mut_token.as_ref()) {
30                (None, None) => {
31                    add_keyword("ref", "ref $0");
32                    add_keyword("mut", "mut $0");
33                }
34                (None, Some(m)) if tok < m.text_range().start() => {
35                    add_keyword("ref", "ref $0");
36                }
37                (Some(r), None) if tok > r.text_range().end() => {
38                    add_keyword("mut", "mut $0");
39                }
40                _ => (),
41            }
42        }
43    }
44
45    if pattern_ctx.after_if_expr {
46        add_keyword("else", "else {\n    $0\n}");
47        add_keyword("else if", "else if $1 {\n    $0\n}");
48    }
49
50    if pattern_ctx.record_pat.is_some() {
51        return;
52    }
53
54    // Suggest name only in let-stmt and fn param
55    if pattern_ctx.should_suggest_name {
56        let mut name_generator = suggest_name::NameGenerator::default();
57        if let Some(suggested) = ctx
58            .expected_type
59            .as_ref()
60            .map(|ty| ty.strip_references())
61            .and_then(|ty| name_generator.for_type(&ty, ctx.db, ctx.edition))
62        {
63            acc.suggest_name(ctx, &suggested);
64        }
65    }
66
67    let refutable = pattern_ctx.refutability == PatternRefutability::Refutable;
68    let single_variant_enum = |enum_: hir::Enum| enum_.num_variants(ctx.db) == 1;
69
70    if let Some(hir::Adt::Enum(e)) =
71        ctx.expected_type.as_ref().and_then(|ty| ty.strip_references().as_adt())
72        && (refutable || single_variant_enum(e))
73    {
74        super::enum_variants_with_paths(
75            acc,
76            ctx,
77            e,
78            pattern_ctx.impl_or_trait.as_ref().and_then(|it| it.as_ref().left()),
79            |acc, ctx, variant, path| {
80                acc.add_qualified_variant_pat(ctx, pattern_ctx, variant, path);
81            },
82        );
83    }
84
85    // FIXME: ideally, we should look at the type we are matching against and
86    // suggest variants + auto-imports
87    ctx.process_all_names(&mut |name, res, _| {
88        let add_simple_path = match res {
89            hir::ScopeDef::ModuleDef(def) => match def {
90                hir::ModuleDef::Adt(hir::Adt::Struct(strukt)) => {
91                    acc.add_struct_pat(ctx, pattern_ctx, strukt, Some(name.clone()));
92                    true
93                }
94                hir::ModuleDef::Variant(variant)
95                    if refutable || single_variant_enum(variant.parent_enum(ctx.db)) =>
96                {
97                    acc.add_variant_pat(ctx, pattern_ctx, None, variant, Some(name.clone()));
98                    true
99                }
100                hir::ModuleDef::Adt(hir::Adt::Enum(e)) => refutable || single_variant_enum(e),
101                hir::ModuleDef::Const(..) => refutable,
102                hir::ModuleDef::Module(..) => true,
103                hir::ModuleDef::Macro(mac) => mac.is_fn_like(ctx.db),
104                _ => false,
105            },
106            hir::ScopeDef::ImplSelfType(impl_) => match impl_.self_ty(ctx.db).as_adt() {
107                Some(hir::Adt::Struct(strukt)) => {
108                    acc.add_struct_pat(ctx, pattern_ctx, strukt, Some(name.clone()));
109                    true
110                }
111                Some(hir::Adt::Enum(e)) => refutable || single_variant_enum(e),
112                Some(hir::Adt::Union(_)) => true,
113                _ => false,
114            },
115            ScopeDef::GenericParam(hir::GenericParam::ConstParam(_)) => true,
116            ScopeDef::GenericParam(_)
117            | ScopeDef::AdtSelfType(_)
118            | ScopeDef::Local(_)
119            | ScopeDef::Label(_)
120            | ScopeDef::Unknown => false,
121        };
122        if add_simple_path {
123            acc.add_pattern_resolution(ctx, pattern_ctx, name, res);
124        }
125    });
126}

128pub(crate) fn complete_pattern_path(
129    acc: &mut Completions,
130    ctx: &CompletionContext<'_>,
131    path_ctx @ PathCompletionCtx { qualified, .. }: &PathCompletionCtx<'_>,
132) {
133    match qualified {
134        Qualified::With { resolution: Some(resolution), super_chain_len, .. } => {
135            acc.add_super_keyword(ctx, *super_chain_len);
136
137            match resolution {
138                hir::PathResolution::Def(hir::ModuleDef::Module(module)) => {
139                    let module_scope = module.scope(ctx.db, Some(ctx.module));
140                    for (name, def) in module_scope {
141                        let add_resolution = match def {
142                            ScopeDef::ModuleDef(hir::ModuleDef::Macro(mac)) => {
143                                mac.is_fn_like(ctx.db)
144                            }
145                            ScopeDef::ModuleDef(_) => true,
146                            _ => false,
147                        };
148
149                        if add_resolution {
150                            acc.add_path_resolution(ctx, path_ctx, name, def, vec![]);
151                        }
152                    }
153                }
154                res => {
155                    let ty = match res {
156                        hir::PathResolution::TypeParam(param) => param.ty(ctx.db),
157                        hir::PathResolution::SelfType(impl_def) => impl_def.self_ty(ctx.db),
158                        hir::PathResolution::Def(hir::ModuleDef::Adt(hir::Adt::Struct(s))) => {
159                            s.ty(ctx.db)
160                        }
161                        hir::PathResolution::Def(hir::ModuleDef::Adt(hir::Adt::Enum(e))) => {
162                            e.ty(ctx.db)
163                        }
164                        hir::PathResolution::Def(hir::ModuleDef::Adt(hir::Adt::Union(u))) => {
165                            u.ty(ctx.db)
166                        }
167                        hir::PathResolution::Def(hir::ModuleDef::BuiltinType(ty)) => ty.ty(ctx.db),
168                        hir::PathResolution::Def(hir::ModuleDef::TypeAlias(ty)) => ty.ty(ctx.db),
169                        _ => return,
170                    };
171
172                    if let Some(hir::Adt::Enum(e)) = ty.as_adt() {
173                        acc.add_enum_variants(ctx, path_ctx, e);
174                    }
175
176                    ctx.iterate_path_candidates(&ty, |item| match item {
177                        AssocItem::TypeAlias(ta) => acc.add_type_alias(ctx, ta),
178                        AssocItem::Const(c) => acc.add_const(ctx, c),
179                        _ => {}
180                    });
181                }
182            }
183        }
184        Qualified::Absolute => acc.add_crate_roots(ctx, path_ctx),
185        Qualified::No => {
186            // this will only be hit if there are brackets or braces, otherwise this will be parsed as an ident pattern
187            ctx.process_all_names(&mut |name, res, doc_aliases| {
188                // FIXME: we should check what kind of pattern we are in and filter accordingly
189                let add_completion = match res {
190                    ScopeDef::ModuleDef(hir::ModuleDef::Macro(mac)) => mac.is_fn_like(ctx.db),
191                    ScopeDef::ModuleDef(hir::ModuleDef::Adt(_)) => true,
192                    ScopeDef::ModuleDef(hir::ModuleDef::Variant(_)) => true,
193                    ScopeDef::ModuleDef(hir::ModuleDef::Module(_)) => true,
194                    ScopeDef::ImplSelfType(_) => true,
195                    _ => false,
196                };
197                if add_completion {
198                    acc.add_path_resolution(ctx, path_ctx, name, res, doc_aliases);
199                }
200            });
201
202            acc.add_nameref_keywords_with_colon(ctx);
203        }
204        Qualified::TypeAnchor { .. } | Qualified::With { .. } => {}
205    }
206}
