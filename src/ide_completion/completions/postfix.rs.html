28pub(crate) fn complete_postfix(
29    acc: &mut Completions,
30    ctx: &CompletionContext<'_>,
31    dot_access: &DotAccess,
32) {
33    if !ctx.config.enable_postfix_completions {
34        return;
35    }
36
37    let (dot_receiver, receiver_ty, receiver_is_ambiguous_float_literal) = match dot_access {
38        DotAccess { receiver_ty: Some(ty), receiver: Some(it), kind, .. } => (
39            it,
40            &ty.original,
41            match *kind {
42                DotAccessKind::Field { receiver_is_ambiguous_float_literal } => {
43                    receiver_is_ambiguous_float_literal
44                }
45                DotAccessKind::Method { .. } => false,
46            },
47        ),
48        _ => return,
49    };
50    let expr_ctx = &dot_access.ctx;
51
52    let receiver_text =
53        get_receiver_text(&ctx.sema, dot_receiver, receiver_is_ambiguous_float_literal);
54
55    let cap = match ctx.config.snippet_cap {
56        Some(it) => it,
57        None => return,
58    };
59
60    let postfix_snippet = match build_postfix_snippet_builder(ctx, cap, dot_receiver) {
61        Some(it) => it,
62        None => return,
63    };
64
65    let cfg = ctx.config.import_path_config(ctx.is_nightly);
66
67    if let Some(drop_trait) = ctx.famous_defs().core_ops_Drop() {
68        if receiver_ty.impls_trait(ctx.db, drop_trait, &[]) {
69            if let Some(drop_fn) = ctx.famous_defs().core_mem_drop() {
70                if let Some(path) =
71                    ctx.module.find_path(ctx.db, ItemInNs::Values(drop_fn.into()), cfg)
72                {
73                    cov_mark::hit!(postfix_drop_completion);
74                    let mut item = postfix_snippet(
75                        "drop",
76                        "fn drop(&mut self)",
77                        &format!(
78                            "{path}($0{receiver_text})",
79                            path = path.display(ctx.db, ctx.edition)
80                        ),
81                    );
82                    item.set_documentation(drop_fn.docs(ctx.db));
83                    item.add_to(acc, ctx.db);
84                }
85            }
86        }
87    }
88
89    let try_enum = TryEnum::from_ty(&ctx.sema, &receiver_ty.strip_references());
90    if let Some(try_enum) = &try_enum {
91        match try_enum {
92            TryEnum::Result => {
93                postfix_snippet(
94                    "ifl",
95                    "if let Ok {}",
96                    &format!("if let Ok($1) = {receiver_text} {{\n    $0\n}}"),
97                )
98                .add_to(acc, ctx.db);
99
100                postfix_snippet(
101                    "lete",
102                    "let Ok else {}",
103                    &format!("let Ok($1) = {receiver_text} else {{\n    $2\n}};\n$0"),
104                )
105                .add_to(acc, ctx.db);
106
107                postfix_snippet(
108                    "while",
109                    "while let Ok {}",
110                    &format!("while let Ok($1) = {receiver_text} {{\n    $0\n}}"),
111                )
112                .add_to(acc, ctx.db);
113            }
114            TryEnum::Option => {
115                postfix_snippet(
116                    "ifl",
117                    "if let Some {}",
118                    &format!("if let Some($1) = {receiver_text} {{\n    $0\n}}"),
119                )
120                .add_to(acc, ctx.db);
121
122                postfix_snippet(
123                    "lete",
124                    "let Some else {}",
125                    &format!("let Some($1) = {receiver_text} else {{\n    $2\n}};\n$0"),
126                )
127                .add_to(acc, ctx.db);
128
129                postfix_snippet(
130                    "while",
131                    "while let Some {}",
132                    &format!("while let Some($1) = {receiver_text} {{\n    $0\n}}"),
133                )
134                .add_to(acc, ctx.db);
135            }
136        }
137    } else if receiver_ty.is_bool() || receiver_ty.is_unknown() {
138        postfix_snippet("if", "if expr {}", &format!("if {receiver_text} {{\n    $0\n}}"))
139            .add_to(acc, ctx.db);
140        postfix_snippet("while", "while expr {}", &format!("while {receiver_text} {{\n    $0\n}}"))
141            .add_to(acc, ctx.db);
142        postfix_snippet("not", "!expr", &format!("!{receiver_text}")).add_to(acc, ctx.db);
143    } else if let Some(trait_) = ctx.famous_defs().core_iter_IntoIterator() {
144        if receiver_ty.impls_trait(ctx.db, trait_, &[]) {
145            postfix_snippet(
146                "for",
147                "for ele in expr {}",
148                &format!("for ele in {receiver_text} {{\n    $0\n}}"),
149            )
150            .add_to(acc, ctx.db);
151        }
152    }
153
154    postfix_snippet("ref", "&expr", &format!("&{receiver_text}")).add_to(acc, ctx.db);
155    postfix_snippet("refm", "&mut expr", &format!("&mut {receiver_text}")).add_to(acc, ctx.db);
156    postfix_snippet("deref", "*expr", &format!("*{receiver_text}")).add_to(acc, ctx.db);
157
158    let mut block_should_be_wrapped = true;
159    if dot_receiver.syntax().kind() == BLOCK_EXPR {
160        block_should_be_wrapped = false;
161        if let Some(parent) = dot_receiver.syntax().parent() {
162            if matches!(parent.kind(), IF_EXPR | WHILE_EXPR | LOOP_EXPR | FOR_EXPR) {
163                block_should_be_wrapped = true;
164            }
165        }
166    };
167    let unsafe_completion_string = if block_should_be_wrapped {
168        format!("unsafe {{ {receiver_text} }}")
169    } else {
170        format!("unsafe {receiver_text}")
171    };
172    postfix_snippet("unsafe", "unsafe {}", &unsafe_completion_string).add_to(acc, ctx.db);
173
174    let const_completion_string = if block_should_be_wrapped {
175        format!("const {{ {receiver_text} }}")
176    } else {
177        format!("const {receiver_text}")
178    };
179    postfix_snippet("const", "const {}", &const_completion_string).add_to(acc, ctx.db);
180
181    // The rest of the postfix completions create an expression that moves an argument,
182    // so it's better to consider references now to avoid breaking the compilation
183
184    let (dot_receiver_including_refs, prefix) = include_references(dot_receiver);
185    let mut receiver_text =
186        get_receiver_text(&ctx.sema, dot_receiver, receiver_is_ambiguous_float_literal);
187    receiver_text.insert_str(0, &prefix);
188    let postfix_snippet =
189        match build_postfix_snippet_builder(ctx, cap, &dot_receiver_including_refs) {
190            Some(it) => it,
191            None => return,
192        };
193
194    if !ctx.config.snippets.is_empty() {
195        add_custom_postfix_completions(acc, ctx, &postfix_snippet, &receiver_text);
196    }
197
198    match try_enum {
199        Some(try_enum) => match try_enum {
200            TryEnum::Result => {
201                postfix_snippet(
202                    "match",
203                    "match expr {}",
204                    &format!("match {receiver_text} {{\n    Ok(${{1:_}}) => {{$2}},\n    Err(${{3:_}}) => {{$0}},\n}}"),
205                )
206                .add_to(acc, ctx.db);
207            }
208            TryEnum::Option => {
209                postfix_snippet(
210                    "match",
211                    "match expr {}",
212                    &format!(
213                        "match {receiver_text} {{\n    Some(${{1:_}}) => {{$2}},\n    None => {{$0}},\n}}"
214                    ),
215                )
216                .add_to(acc, ctx.db);
217            }
218        },
219        None => {
220            postfix_snippet(
221                "match",
222                "match expr {}",
223                &format!("match {receiver_text} {{\n    ${{1:_}} => {{$0}},\n}}"),
224            )
225            .add_to(acc, ctx.db);
226        }
227    }
228
229    postfix_snippet("box", "Box::new(expr)", &format!("Box::new({receiver_text})"))
230        .add_to(acc, ctx.db);
231    postfix_snippet("dbg", "dbg!(expr)", &format!("dbg!({receiver_text})")).add_to(acc, ctx.db); // fixme
232    postfix_snippet("dbgr", "dbg!(&expr)", &format!("dbg!(&{receiver_text})")).add_to(acc, ctx.db);
233    postfix_snippet("call", "function(expr)", &format!("${{1}}({receiver_text})"))
234        .add_to(acc, ctx.db);
235
236    if let Some(parent) = dot_receiver_including_refs.syntax().parent().and_then(|p| p.parent()) {
237        if matches!(parent.kind(), STMT_LIST | EXPR_STMT) {
238            postfix_snippet("let", "let", &format!("let $0 = {receiver_text};"))
239                .add_to(acc, ctx.db);
240            postfix_snippet("letm", "let mut", &format!("let mut $0 = {receiver_text};"))
241                .add_to(acc, ctx.db);
242        }
243    }
244
245    if let ast::Expr::Literal(literal) = dot_receiver_including_refs.clone() {
246        if let Some(literal_text) = ast::String::cast(literal.token()) {
247            add_format_like_completions(acc, ctx, &dot_receiver_including_refs, cap, &literal_text);
248        }
249    }
250
251    postfix_snippet(
252        "return",
253        "return expr",
254        &format!(
255            "return {receiver_text}{semi}",
256            semi = if expr_ctx.in_block_expr { ";" } else { "" }
257        ),
258    )
259    .add_to(acc, ctx.db);
260
261    if let BreakableKind::Block | BreakableKind::Loop = expr_ctx.in_breakable {
262        postfix_snippet(
263            "break",
264            "break expr",
265            &format!(
266                "break {receiver_text}{semi}",
267                semi = if expr_ctx.in_block_expr { ";" } else { "" }
268            ),
269        )
270        .add_to(acc, ctx.db);
271    }
272}

274fn get_receiver_text(
275    sema: &Semantics<'_, RootDatabase>,
276    receiver: &ast::Expr,
277    receiver_is_ambiguous_float_literal: bool,
278) -> String {
279    // Do not just call `receiver.to_string()`, as that will mess up whitespaces inside macros.
280    let Some(mut range) = sema.original_range_opt(receiver.syntax()) else {
281        return receiver.to_string();
282    };
283    if receiver_is_ambiguous_float_literal {
284        range.range = TextRange::at(range.range.start(), range.range.len() - TextSize::of('.'))
285    }
286    let file_text = sema.db.file_text(range.file_id.file_id());
287    let mut text = file_text.text(sema.db)[range.range].to_owned();
288
289    // The receiver texts should be interpreted as-is, as they are expected to be
290    // normal Rust expressions.
291    escape_snippet_bits(&mut text);
292    text
293}
294
295/// Escapes `\` and `$` so that they don't get interpreted as snippet-specific constructs.
296///
297/// Note that we don't need to escape the other characters that can be escaped,
298/// because they wouldn't be treated as snippet-specific constructs without '$'.
299fn escape_snippet_bits(text: &mut String) {
300    stdx::replace(text, '\\', "\\\\");
301    stdx::replace(text, '$', "\\$");
302}

304fn include_references(initial_element: &ast::Expr) -> (ast::Expr, String) {
305    let mut resulting_element = initial_element.clone();
306
307    while let Some(field_expr) = resulting_element.syntax().parent().and_then(ast::FieldExpr::cast)
308    {
309        resulting_element = ast::Expr::from(field_expr);
310    }
311
312    let mut prefix = String::new();
313
314    while let Some(parent_deref_element) =
315        resulting_element.syntax().parent().and_then(ast::PrefixExpr::cast)
316    {
317        if parent_deref_element.op_kind() != Some(ast::UnaryOp::Deref) {
318            break;
319        }
320
321        resulting_element = ast::Expr::from(parent_deref_element);
322
323        prefix.insert(0, '*');
324    }
325
326    if let Some(first_ref_expr) = resulting_element.syntax().parent().and_then(ast::RefExpr::cast) {
327        if let Some(expr) = first_ref_expr.expr() {
328            resulting_element = expr;
329        }
330
331        while let Some(parent_ref_element) =
332            resulting_element.syntax().parent().and_then(ast::RefExpr::cast)
333        {
334            let exclusive = parent_ref_element.mut_token().is_some();
335            resulting_element = ast::Expr::from(parent_ref_element);
336
337            prefix.insert_str(0, if exclusive { "&mut " } else { "&" });
338        }
339    } else {
340        // If we do not find any ref expressions, restore
341        // all the progress of tree climbing
342        resulting_element = initial_element.clone();
343    }
344
345    (resulting_element, prefix)
346}

348fn build_postfix_snippet_builder<'ctx>(
349    ctx: &'ctx CompletionContext<'_>,
350    cap: SnippetCap,
351    receiver: &'ctx ast::Expr,
352) -> Option<impl Fn(&str, &str, &str) -> Builder + 'ctx> {
353    let receiver_range = ctx.sema.original_range_opt(receiver.syntax())?.range;
354    if ctx.source_range().end() < receiver_range.start() {
355        // This shouldn't happen, yet it does. I assume this might be due to an incorrect token
356        // mapping.
357        never!();
358        return None;
359    }
360    let delete_range = TextRange::new(receiver_range.start(), ctx.source_range().end());
361
362    // Wrapping impl Fn in an option ruins lifetime inference for the parameters in a way that
363    // can't be annotated for the closure, hence fix it by constructing it without the Option first
364    fn build<'ctx>(
365        ctx: &'ctx CompletionContext<'_>,
366        cap: SnippetCap,
367        delete_range: TextRange,
368    ) -> impl Fn(&str, &str, &str) -> Builder + 'ctx {
369        move |label, detail, snippet| {
370            let edit = TextEdit::replace(delete_range, snippet.to_owned());
371            let mut item = CompletionItem::new(
372                CompletionItemKind::Snippet,
373                ctx.source_range(),
374                label,
375                ctx.edition,
376            );
377            item.detail(detail).snippet_edit(cap, edit);
378            let postfix_match = if ctx.original_token.text() == label {
379                cov_mark::hit!(postfix_exact_match_is_high_priority);
380                Some(CompletionRelevancePostfixMatch::Exact)
381            } else {
382                cov_mark::hit!(postfix_inexact_match_is_low_priority);
383                Some(CompletionRelevancePostfixMatch::NonExact)
384            };
385            let relevance = CompletionRelevance { postfix_match, ..Default::default() };
386            item.set_relevance(relevance);
387            item
388        }
389    }
390    Some(build(ctx, cap, delete_range))
391}

393fn add_custom_postfix_completions(
394    acc: &mut Completions,
395    ctx: &CompletionContext<'_>,
396    postfix_snippet: impl Fn(&str, &str, &str) -> Builder,
397    receiver_text: &str,
398) -> Option<()> {
399    ImportScope::find_insert_use_container(&ctx.token.parent()?, &ctx.sema)?;
400    ctx.config.postfix_snippets().filter(|(_, snip)| snip.scope == SnippetScope::Expr).for_each(
401        |(trigger, snippet)| {
402            let imports = match snippet.imports(ctx) {
403                Some(imports) => imports,
404                None => return,
405            };
406            let body = snippet.postfix_snippet(receiver_text);
407            let mut builder =
408                postfix_snippet(trigger, snippet.description.as_deref().unwrap_or_default(), &body);
409            builder.documentation(Documentation::new(format!("```rust\n{body}\n```")));
410            for import in imports.into_iter() {
411                builder.add_import(import);
412            }
413            builder.add_to(acc, ctx.db);
414        },
415    );
416    None
417}

676        check_edit(kind, r#"fn main() { foo.$0 }"#, &format!("fn main() {{ {kind} {{ foo }} }}"));
677        check_edit(
678            kind,

681        );
682        check_edit(
683            kind,
684            r#"fn main() { if x { foo }.$0 }"#,
685            &format!("fn main() {{ {kind} {{ if x {{ foo }} }} }}"),
686        );
687        check_edit(
688            kind,
689            r#"fn main() { loop { foo }.$0 }"#,
690            &format!("fn main() {{ {kind} {{ loop {{ foo }} }} }}"),
691        );
692        check_edit(
693            kind,
694            r#"fn main() { if true {}.$0 }"#,
695            &format!("fn main() {{ {kind} {{ if true {{}} }} }}"),
