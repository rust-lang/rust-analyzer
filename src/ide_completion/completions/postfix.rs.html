32pub(crate) fn complete_postfix(
33    acc: &mut Completions,
34    ctx: &CompletionContext<'_>,
35    dot_access: &DotAccess<'_>,
36) {
37    if !ctx.config.enable_postfix_completions {
38        return;
39    }
40
41    let (dot_receiver, receiver_ty, receiver_is_ambiguous_float_literal) = match dot_access {
42        DotAccess { receiver_ty: Some(ty), receiver: Some(it), kind, .. } => (
43            it,
44            &ty.original,
45            match *kind {
46                DotAccessKind::Field { receiver_is_ambiguous_float_literal } => {
47                    receiver_is_ambiguous_float_literal
48                }
49                DotAccessKind::Method => false,
50            },
51        ),
52        _ => return,
53    };
54    let expr_ctx = &dot_access.ctx;
55
56    let receiver_text =
57        get_receiver_text(&ctx.sema, dot_receiver, receiver_is_ambiguous_float_literal);
58
59    let cap = match ctx.config.snippet_cap {
60        Some(it) => it,
61        None => return,
62    };
63
64    let postfix_snippet = match build_postfix_snippet_builder(ctx, cap, dot_receiver) {
65        Some(it) => it,
66        None => return,
67    };
68
69    let cfg = ctx.config.find_path_config(ctx.is_nightly);
70
71    if let Some(drop_trait) = ctx.famous_defs().core_ops_Drop()
72        && receiver_ty.impls_trait(ctx.db, drop_trait, &[])
73        && let Some(drop_fn) = ctx.famous_defs().core_mem_drop()
74        && let Some(path) = ctx.module.find_path(ctx.db, ItemInNs::Values(drop_fn.into()), cfg)
75    {
76        cov_mark::hit!(postfix_drop_completion);
77        let mut item = postfix_snippet(
78            "drop",
79            "fn drop(&mut self)",
80            &format!("{path}($0{receiver_text})", path = path.display(ctx.db, ctx.edition)),
81        );
82        item.set_documentation(drop_fn.docs(ctx.db));
83        item.add_to(acc, ctx.db);
84    }
85
86    postfix_snippet("ref", "&expr", &format!("&{receiver_text}")).add_to(acc, ctx.db);
87    postfix_snippet("refm", "&mut expr", &format!("&mut {receiver_text}")).add_to(acc, ctx.db);
88    postfix_snippet("deref", "*expr", &format!("*{receiver_text}")).add_to(acc, ctx.db);
89
90    // The rest of the postfix completions create an expression that moves an argument,
91    // so it's better to consider references now to avoid breaking the compilation
92
93    let (dot_receiver_including_refs, prefix) = include_references(dot_receiver);
94    let mut receiver_text = receiver_text;
95    receiver_text.insert_str(0, &prefix);
96    let postfix_snippet =
97        match build_postfix_snippet_builder(ctx, cap, &dot_receiver_including_refs) {
98            Some(it) => it,
99            None => return,
100        };
101
102    if !ctx.config.snippets.is_empty() {
103        add_custom_postfix_completions(acc, ctx, &postfix_snippet, &receiver_text);
104    }
105
106    postfix_snippet("box", "Box::new(expr)", &format!("Box::new({receiver_text})"))
107        .add_to(acc, ctx.db);
108    postfix_snippet("dbg", "dbg!(expr)", &format!("dbg!({receiver_text})")).add_to(acc, ctx.db); // fixme
109    postfix_snippet("dbgr", "dbg!(&expr)", &format!("dbg!(&{receiver_text})")).add_to(acc, ctx.db);
110    postfix_snippet("call", "function(expr)", &format!("${{1}}({receiver_text})"))
111        .add_to(acc, ctx.db);
112
113    let try_enum = TryEnum::from_ty(&ctx.sema, receiver_ty);
114    let mut is_in_cond = false;
115    if let Some(parent) = dot_receiver_including_refs.syntax().parent()
116        && let Some(second_ancestor) = parent.parent()
117    {
118        if let Some(parent_expr) = ast::Expr::cast(parent) {
119            is_in_cond = is_in_condition(&parent_expr);
120        }
121        let placeholder = suggest_receiver_name(dot_receiver, "0", &ctx.sema);
122        match &try_enum {
123            Some(try_enum) if is_in_cond => match try_enum {
124                TryEnum::Result => {
125                    postfix_snippet(
126                        "let",
127                        "let Ok(_)",
128                        &format!("let Ok({placeholder}) = {receiver_text}"),
129                    )
130                    .add_to(acc, ctx.db);
131                    postfix_snippet(
132                        "letm",
133                        "let Ok(mut _)",
134                        &format!("let Ok(mut {placeholder}) = {receiver_text}"),
135                    )
136                    .add_to(acc, ctx.db);
137                }
138                TryEnum::Option => {
139                    postfix_snippet(
140                        "let",
141                        "let Some(_)",
142                        &format!("let Some({placeholder}) = {receiver_text}"),
143                    )
144                    .add_to(acc, ctx.db);
145                    postfix_snippet(
146                        "letm",
147                        "let Some(mut _)",
148                        &format!("let Some(mut {placeholder}) = {receiver_text}"),
149                    )
150                    .add_to(acc, ctx.db);
151                }
152            },
153            _ if is_in_cond => {
154                postfix_snippet("let", "let", &format!("let $1 = {receiver_text}"))
155                    .add_to(acc, ctx.db);
156            }
157            _ if matches!(second_ancestor.kind(), STMT_LIST | EXPR_STMT | BLOCK_EXPR) => {
158                postfix_snippet("let", "let", &format!("let $0 = {receiver_text};"))
159                    .add_to(acc, ctx.db);
160                postfix_snippet("letm", "let mut", &format!("let mut $0 = {receiver_text};"))
161                    .add_to(acc, ctx.db);
162            }
163            _ if ast::MatchArm::can_cast(second_ancestor.kind()) => {
164                postfix_snippet(
165                    "let",
166                    "let",
167                    &format!("{{\n    let $1 = {receiver_text};\n    $0\n}}"),
168                )
169                .add_to(acc, ctx.db);
170                postfix_snippet(
171                    "letm",
172                    "let mut",
173                    &format!("{{\n    let mut $1 = {receiver_text};\n    $0\n}}"),
174                )
175                .add_to(acc, ctx.db);
176            }
177            _ => (),
178        }
179    }
180
181    if !is_in_cond {
182        match try_enum {
183            Some(try_enum) => match try_enum {
184                TryEnum::Result => {
185                    postfix_snippet(
186                    "match",
187                    "match expr {}",
188                    &format!("match {receiver_text} {{\n    Ok(${{1:_}}) => {{$2}},\n    Err(${{3:_}}) => {{$0}},\n}}"),
189                )
190                .add_to(acc, ctx.db);
191                }
192                TryEnum::Option => {
193                    postfix_snippet(
194                    "match",
195                    "match expr {}",
196                    &format!(
197                        "match {receiver_text} {{\n    Some(${{1:_}}) => {{$2}},\n    None => {{$0}},\n}}"
198                    ),
199                )
200                .add_to(acc, ctx.db);
201                }
202            },
203            None => {
204                postfix_snippet(
205                    "match",
206                    "match expr {}",
207                    &format!("match {receiver_text} {{\n    ${{1:_}} => {{$0}},\n}}"),
208                )
209                .add_to(acc, ctx.db);
210            }
211        }
212        if let Some(try_enum) = &try_enum {
213            let placeholder = suggest_receiver_name(dot_receiver, "1", &ctx.sema);
214            match try_enum {
215                TryEnum::Result => {
216                    postfix_snippet(
217                        "ifl",
218                        "if let Ok {}",
219                        &format!("if let Ok({placeholder}) = {receiver_text} {{\n    $0\n}}"),
220                    )
221                    .add_to(acc, ctx.db);
222
223                    postfix_snippet(
224                        "lete",
225                        "let Ok else {}",
226                        &format!(
227                            "let Ok({placeholder}) = {receiver_text} else {{\n    $2\n}};\n$0"
228                        ),
229                    )
230                    .add_to(acc, ctx.db);
231
232                    postfix_snippet(
233                        "while",
234                        "while let Ok {}",
235                        &format!("while let Ok({placeholder}) = {receiver_text} {{\n    $0\n}}"),
236                    )
237                    .add_to(acc, ctx.db);
238                }
239                TryEnum::Option => {
240                    postfix_snippet(
241                        "ifl",
242                        "if let Some {}",
243                        &format!("if let Some({placeholder}) = {receiver_text} {{\n    $0\n}}"),
244                    )
245                    .add_to(acc, ctx.db);
246
247                    postfix_snippet(
248                        "lete",
249                        "let Some else {}",
250                        &format!(
251                            "let Some({placeholder}) = {receiver_text} else {{\n    $2\n}};\n$0"
252                        ),
253                    )
254                    .add_to(acc, ctx.db);
255
256                    postfix_snippet(
257                        "while",
258                        "while let Some {}",
259                        &format!("while let Some({placeholder}) = {receiver_text} {{\n    $0\n}}"),
260                    )
261                    .add_to(acc, ctx.db);
262                }
263            }
264        } else if receiver_ty.is_bool() || receiver_ty.is_unknown() {
265            postfix_snippet("if", "if expr {}", &format!("if {receiver_text} {{\n    $0\n}}"))
266                .add_to(acc, ctx.db);
267            postfix_snippet(
268                "while",
269                "while expr {}",
270                &format!("while {receiver_text} {{\n    $0\n}}"),
271            )
272            .add_to(acc, ctx.db);
273        } else if let Some(trait_) = ctx.famous_defs().core_iter_IntoIterator()
274            && receiver_ty.impls_trait(ctx.db, trait_, &[])
275        {
276            postfix_snippet(
277                "for",
278                "for ele in expr {}",
279                &format!("for ele in {receiver_text} {{\n    $0\n}}"),
280            )
281            .add_to(acc, ctx.db);
282        }
283    }
284
285    if receiver_ty.is_bool() || receiver_ty.is_unknown() {
286        postfix_snippet("not", "!expr", &format!("!{receiver_text}")).add_to(acc, ctx.db);
287    }
288
289    let block_should_be_wrapped = if let ast::Expr::BlockExpr(block) = dot_receiver {
290        block.modifier().is_some() || !block.is_standalone()
291    } else {
292        true
293    };
294    {
295        let (open_brace, close_brace) =
296            if block_should_be_wrapped { ("{ ", " }") } else { ("", "") };
297        // FIXME: Why add parentheses
298        let (open_paren, close_paren) = if is_in_cond { ("(", ")") } else { ("", "") };
299        let unsafe_completion_string =
300            format!("{open_paren}unsafe {open_brace}{receiver_text}{close_brace}{close_paren}");
301        postfix_snippet("unsafe", "unsafe {}", &unsafe_completion_string).add_to(acc, ctx.db);
302
303        let const_completion_string =
304            format!("{open_paren}const {open_brace}{receiver_text}{close_brace}{close_paren}");
305        postfix_snippet("const", "const {}", &const_completion_string).add_to(acc, ctx.db);
306    }
307
308    if let ast::Expr::Literal(literal) = dot_receiver_including_refs.clone()
309        && let Some(literal_text) = ast::String::cast(literal.token())
310    {
311        add_format_like_completions(acc, ctx, &dot_receiver_including_refs, cap, &literal_text);
312    }
313
314    postfix_snippet(
315        "return",
316        "return expr",
317        &format!(
318            "return {receiver_text}{semi}",
319            semi = if expr_ctx.in_block_expr { ";" } else { "" }
320        ),
321    )
322    .add_to(acc, ctx.db);
323
324    if let Some(BreakableKind::Block | BreakableKind::Loop) = expr_ctx.in_breakable {
325        postfix_snippet(
326            "break",
327            "break expr",
328            &format!(
329                "break {receiver_text}{semi}",
330                semi = if expr_ctx.in_block_expr { ";" } else { "" }
331            ),
332        )
333        .add_to(acc, ctx.db);
334    }
335}

337fn suggest_receiver_name(
338    receiver: &ast::Expr,
339    n: &str,
340    sema: &Semantics<'_, RootDatabase>,
341) -> SmolStr {
342    let placeholder = |name| format_smolstr!("${{{n}:{name}}}");
343
344    match receiver {
345        ast::Expr::PathExpr(path) => {
346            if let Some(name) = path.path().and_then(|it| it.as_single_name_ref()) {
347                return placeholder(name.text().as_str());
348            }
349        }
350        ast::Expr::RefExpr(it) => {
351            if let Some(receiver) = it.expr() {
352                return suggest_receiver_name(&receiver, n, sema);
353            }
354        }
355        _ => {}
356    }
357
358    let name = NameGenerator::new_with_names([].into_iter()).try_for_variable(receiver, sema);
359    match name {
360        Some(name) => placeholder(&name),
361        None => format_smolstr!("${n}"),
362    }
363}

365fn get_receiver_text(
366    sema: &Semantics<'_, RootDatabase>,
367    receiver: &ast::Expr,
368    receiver_is_ambiguous_float_literal: bool,
369) -> String {
370    // Do not just call `receiver.to_string()`, as that will mess up whitespaces inside macros.
371    let Some(mut range) = sema.original_range_opt(receiver.syntax()) else {
372        return receiver.to_string();
373    };
374    if receiver_is_ambiguous_float_literal {
375        range.range = TextRange::at(range.range.start(), range.range.len() - TextSize::of('.'))
376    }
377    let file_text = sema.db.file_text(range.file_id.file_id(sema.db));
378    let mut text = file_text.text(sema.db)[range.range].to_owned();
379
380    // The receiver texts should be interpreted as-is, as they are expected to be
381    // normal Rust expressions.
382    escape_snippet_bits(&mut text);
383    text
384}
385
386/// Escapes `\` and `$` so that they don't get interpreted as snippet-specific constructs.
387///
388/// Note that we don't need to escape the other characters that can be escaped,
389/// because they wouldn't be treated as snippet-specific constructs without '$'.
390fn escape_snippet_bits(text: &mut String) {
391    stdx::replace(text, '\\', "\\\\");
392    stdx::replace(text, '$', "\\$");
393}

395fn include_references(initial_element: &ast::Expr) -> (ast::Expr, String) {
396    let mut resulting_element = initial_element.clone();
397
398    while let Some(field_expr) = resulting_element.syntax().parent().and_then(ast::FieldExpr::cast)
399    {
400        resulting_element = ast::Expr::from(field_expr);
401    }
402
403    let mut prefix = String::new();
404
405    let mut found_ref_or_deref = false;
406
407    while let Some(parent_deref_element) =
408        resulting_element.syntax().parent().and_then(ast::PrefixExpr::cast)
409    {
410        if parent_deref_element.op_kind() != Some(ast::UnaryOp::Deref) {
411            break;
412        }
413
414        found_ref_or_deref = true;
415        resulting_element = ast::Expr::from(parent_deref_element);
416
417        prefix.insert(0, '*');
418    }
419
420    while let Some(parent_ref_element) =
421        resulting_element.syntax().parent().and_then(ast::RefExpr::cast)
422    {
423        found_ref_or_deref = true;
424        let last_child_or_token = parent_ref_element.syntax().last_child_or_token();
425        prefix.insert_str(
426            0,
427            parent_ref_element
428                .syntax()
429                .children_with_tokens()
430                .filter(|it| Some(it) != last_child_or_token.as_ref())
431                .format("")
432                .to_smolstr()
433                .as_str(),
434        );
435        resulting_element = ast::Expr::from(parent_ref_element);
436    }
437
438    if !found_ref_or_deref {
439        // If we do not find any ref/deref expressions, restore
440        // all the progress of tree climbing
441        prefix.clear();
442        resulting_element = initial_element.clone();
443    }
444
445    (resulting_element, prefix)
446}

448fn build_postfix_snippet_builder<'ctx>(
449    ctx: &'ctx CompletionContext<'_>,
450    cap: SnippetCap,
451    receiver: &'ctx ast::Expr,
452) -> Option<impl Fn(&str, &str, &str) -> Builder + 'ctx> {
453    let receiver_range = ctx.sema.original_range_opt(receiver.syntax())?.range;
454    if ctx.source_range().end() < receiver_range.start() {
455        // This shouldn't happen, yet it does. I assume this might be due to an incorrect token
456        // mapping.
457        never!();
458        return None;
459    }
460    let delete_range = TextRange::new(receiver_range.start(), ctx.source_range().end());
461
462    // Wrapping impl Fn in an option ruins lifetime inference for the parameters in a way that
463    // can't be annotated for the closure, hence fix it by constructing it without the Option first
464    fn build<'ctx>(
465        ctx: &'ctx CompletionContext<'_>,
466        cap: SnippetCap,
467        delete_range: TextRange,
468    ) -> impl Fn(&str, &str, &str) -> Builder + 'ctx {
469        move |label, detail, snippet| {
470            let edit = TextEdit::replace(delete_range, snippet.to_owned());
471            let mut item = CompletionItem::new(
472                CompletionItemKind::Snippet,
473                ctx.source_range(),
474                label,
475                ctx.edition,
476            );
477            item.detail(detail).snippet_edit(cap, edit);
478            let postfix_match = if ctx.original_token.text() == label {
479                cov_mark::hit!(postfix_exact_match_is_high_priority);
480                Some(CompletionRelevancePostfixMatch::Exact)
481            } else {
482                cov_mark::hit!(postfix_inexact_match_is_low_priority);
483                Some(CompletionRelevancePostfixMatch::NonExact)
484            };
485            let relevance = CompletionRelevance { postfix_match, ..Default::default() };
486            item.set_relevance(relevance);
487            item
488        }
489    }
490    Some(build(ctx, cap, delete_range))
491}

493fn add_custom_postfix_completions(
494    acc: &mut Completions,
495    ctx: &CompletionContext<'_>,
496    postfix_snippet: impl Fn(&str, &str, &str) -> Builder,
497    receiver_text: &str,
498) -> Option<()> {
499    ImportScope::find_insert_use_container(&ctx.token.parent()?, &ctx.sema)?;
500    ctx.config.postfix_snippets().filter(|(_, snip)| snip.scope == SnippetScope::Expr).for_each(
501        |(trigger, snippet)| {
502            let imports = match snippet.imports(ctx) {
503                Some(imports) => imports,
504                None => return,
505            };
506            let body = snippet.postfix_snippet(receiver_text);
507            let mut builder =
508                postfix_snippet(trigger, snippet.description.as_deref().unwrap_or_default(), &body);
509            builder.documentation(Documentation::new_owned(format!("```rust\n{body}\n```")));
510            for import in imports.into_iter() {
511                builder.add_import(import);
512            }
513            builder.add_to(acc, ctx.db);
514        },
515    );
516    None
517}

519pub(crate) fn is_in_condition(it: &ast::Expr) -> bool {
520    it.syntax()
521        .parent()
522        .and_then(|parent| {
523            Some(match_ast! { match parent {
524                ast::IfExpr(expr) => expr.condition()? == *it,
525                ast::WhileExpr(expr) => expr.condition()? == *it,
526                ast::MatchGuard(guard) => guard.condition()? == *it,
527                ast::BinExpr(bin_expr) => (bin_expr.op_token()?.kind() == T![&&])
528                    .then(|| is_in_condition(&bin_expr.into()))?,
529                ast::Expr(expr) => (expr.syntax().text_range().start() == it.syntax().text_range().start())
530                    .then(|| is_in_condition(&expr))?,
531                _ => return None,
532            } })
533        })
534        .unwrap_or(false)
535}

1059            kind,

1061            &format!("fn main() {{ {kind} {{ loop {{ foo }} }} }}"),
1062        );

1064            kind,
1065            r#"fn main() { if true {}.$0 }"#,
1066            &format!("fn main() {{ {kind} {{ if true {{}} }} }}"),
1067        );

1069            kind,
1070            r#"fn main() { while true {}.$0 }"#,
1071            &format!("fn main() {{ {kind} {{ while true {{}} }} }}"),
1072        );
1073        check_edit(
1074            kind,
1075            r#"fn main() { for i in 0..10 {}.$0 }"#,
1076            &format!("fn main() {{ {kind} {{ for i in 0..10 {{}} }} }}"),
1077        );
1078        check_edit(
1079            kind,
1080            r#"fn main() { let x = if true {1} else {2}.$0 }"#,
1081            &format!("fn main() {{ let x = {kind} {{ if true {{1}} else {{2}} }} }}"),
1082        );

1084        if kind == "const" {
1085            check_edit(
1086                kind,
1087                r#"fn main() { unsafe {1}.$0 }"#,

1100            kind,
1101            r#"fn main() { let x = true else {panic!()}.$0}"#,
