98    fn is_deprecated_assoc_item(&self, as_assoc_item: impl AsAssocItem) -> bool {
99        let db = self.db();
100        let assoc = match as_assoc_item.as_assoc_item(db) {
101            Some(assoc) => assoc,
102            None => return false,
103        };
104
105        let is_assoc_deprecated = match assoc {
106            hir::AssocItem::Function(it) => self.is_deprecated(it),
107            hir::AssocItem::Const(it) => self.is_deprecated(it),
108            hir::AssocItem::TypeAlias(it) => self.is_deprecated(it),
109        };
110        is_assoc_deprecated
111            || assoc
112                .container_or_implemented_trait(db)
113                .map(|trait_| self.is_deprecated(trait_))
114                .unwrap_or(false)
115    }
116
117    // FIXME: remove this
118    fn docs(&self, def: impl HasDocs) -> Option<Documentation> {
119        def.docs(self.db())
120    }

123pub(crate) fn render_field(
124    ctx: RenderContext<'_>,
125    dot_access: &DotAccess<'_>,
126    receiver: Option<SmolStr>,
127    field: hir::Field,
128    ty: &hir::Type<'_>,
129) -> CompletionItem {
130    let db = ctx.db();
131    let is_deprecated = ctx.is_deprecated(field);
132    let name = field.name(db);
133    let (name, escaped_name) =
134        (name.as_str().to_smolstr(), name.display_no_db(ctx.completion.edition).to_smolstr());
135    let mut item = CompletionItem::new(
136        SymbolKind::Field,
137        ctx.source_range(),
138        field_with_receiver(receiver.as_deref(), &name),
139        ctx.completion.edition,
140    );
141    item.set_relevance(CompletionRelevance {
142        type_match: compute_type_match(ctx.completion, ty),
143        exact_name_match: compute_exact_name_match(ctx.completion, &name),
144        is_skipping_completion: receiver.is_some(),
145        ..CompletionRelevance::default()
146    });
147    item.detail(ty.display(db, ctx.completion.display_target).to_string())
148        .set_documentation(field.docs(db))
149        .set_deprecated(is_deprecated)
150        .lookup_by(name);
151
152    let is_field_access = matches!(dot_access.kind, DotAccessKind::Field { .. });
153    if !is_field_access || ty.is_fn() || ty.is_closure() {
154        let mut builder = TextEdit::builder();
155        // Using TextEdit, insert '(' before the struct name and ')' before the
156        // dot access, then comes the field name and optionally insert function
157        // call parens.
158
159        builder.replace(
160            ctx.source_range(),
161            field_with_receiver(receiver.as_deref(), &escaped_name).into(),
162        );
163
164        let expected_fn_type =
165            ctx.completion.expected_type.as_ref().is_some_and(|ty| ty.is_fn() || ty.is_closure());
166
167        if !expected_fn_type
168            && let Some(receiver) = &dot_access.receiver
169            && let Some(receiver) = ctx.completion.sema.original_ast_node(receiver.clone())
170        {
171            builder.insert(receiver.syntax().text_range().start(), "(".to_owned());
172            builder.insert(ctx.source_range().end(), ")".to_owned());
173
174            let is_parens_needed = !matches!(dot_access.kind, DotAccessKind::Method);
175
176            if is_parens_needed {
177                builder.insert(ctx.source_range().end(), "()".to_owned());
178            }
179        }
180
181        item.text_edit(builder.finish());
182    } else {
183        item.insert_text(field_with_receiver(receiver.as_deref(), &escaped_name));
184    }
185    if let Some(receiver) = &dot_access.receiver
186        && let Some(original) = ctx.completion.sema.original_ast_node(receiver.clone())
187        && let Some(ref_mode) = compute_ref_match(ctx.completion, ty)
188    {
189        item.ref_match(ref_mode, original.syntax().text_range().start());
190    }
191    item.doc_aliases(ctx.doc_aliases);
192    item.build(db)
193}

280pub(crate) fn render_expr(
281    ctx: &CompletionContext<'_>,
282    expr: &hir::term_search::Expr<'_>,
283) -> Option<Builder> {
284    let mut i = 1;
285    let mut snippet_formatter = |ty: &hir::Type<'_>| {
286        let arg_name = ty
287            .as_adt()
288            .map(|adt| stdx::to_lower_snake_case(adt.name(ctx.db).as_str()))
289            .unwrap_or_else(|| String::from("_"));
290        let res = format!("${{{i}:{arg_name}}}");
291        i += 1;
292        res
293    };
294
295    let mut label_formatter = |ty: &hir::Type<'_>| {
296        ty.as_adt()
297            .map(|adt| stdx::to_lower_snake_case(adt.name(ctx.db).as_str()))
298            .unwrap_or_else(|| String::from("..."))
299    };
300
301    let cfg = ctx.config.find_path_config(ctx.is_nightly);
302
303    let label =
304        expr.gen_source_code(&ctx.scope, &mut label_formatter, cfg, ctx.display_target).ok()?;
305
306    let source_range = match ctx.original_token.parent() {
307        Some(node) => match node.ancestors().find_map(ast::Path::cast) {
308            Some(path) => path.syntax().text_range(),
309            None => node.text_range(),
310        },
311        None => ctx.source_range(),
312    };
313
314    let mut item =
315        CompletionItem::new(CompletionItemKind::Expression, source_range, label, ctx.edition);
316
317    let snippet = format!(
318        "{}$0",
319        expr.gen_source_code(&ctx.scope, &mut snippet_formatter, cfg, ctx.display_target).ok()?
320    );
321    let edit = TextEdit::replace(source_range, snippet);
322    item.snippet_edit(ctx.config.snippet_cap?, edit);
323    item.documentation(Documentation::new(String::from("Autogenerated expression by term search")));
324    item.set_relevance(crate::CompletionRelevance {
325        type_match: compute_type_match(ctx, &expr.ty(ctx.db)),
326        ..Default::default()
327    });
328    for trait_ in expr.traits_used(ctx.db) {
329        let trait_item = hir::ItemInNs::from(hir::ModuleDef::from(trait_));
330        let Some(path) = ctx.module.find_path(ctx.db, trait_item, cfg) else {
331            continue;
332        };
333
334        item.add_import(LocatedImport::new_no_completion(path, trait_item, trait_item));
335    }
336
337    Some(item)
338}

340fn get_import_name(
341    resolution: ScopeDef,
342    ctx: &RenderContext<'_>,
343    import_edit: &LocatedImport,
344) -> Option<hir::Name> {
345    // FIXME: Temporary workaround for handling aliased import.
346    // This should be removed after we have proper support for importing alias.
347    // <https://github.com/rust-lang/rust-analyzer/issues/14079>
348
349    // If `item_to_import` matches `original_item`, we are importing the item itself (not its parent module).
350    // In this case, we can use the last segment of `import_path`, as it accounts for the aliased name.
351    if import_edit.item_to_import == import_edit.original_item {
352        import_edit.import_path.segments().last().cloned()
353    } else {
354        scope_def_to_name(resolution, ctx, import_edit)
355    }
356}

371fn render_resolution_pat(
372    ctx: RenderContext<'_>,
373    pattern_ctx: &PatternContext,
374    local_name: hir::Name,
375    import_to_add: Option<LocatedImport>,
376    resolution: ScopeDef,
377) -> Builder {
378    let _p = tracing::info_span!("render_resolution_pat").entered();
379    use hir::ModuleDef::*;
380
381    if let ScopeDef::ModuleDef(Macro(mac)) = resolution {
382        let ctx = ctx.import_to_add(import_to_add);
383        render_macro_pat(ctx, pattern_ctx, local_name, mac)
384    } else {
385        render_resolution_simple_(ctx, &local_name, import_to_add, resolution)
386    }
387}

389fn render_resolution_path(
390    ctx: RenderContext<'_>,
391    path_ctx: &PathCompletionCtx<'_>,
392    local_name: hir::Name,
393    import_to_add: Option<LocatedImport>,
394    resolution: ScopeDef,
395) -> Builder {
396    let _p = tracing::info_span!("render_resolution_path").entered();
397    use hir::ModuleDef::*;
398
399    let krate = ctx.completion.display_target;
400
401    match resolution {
402        ScopeDef::ModuleDef(Macro(mac)) => {
403            let ctx = ctx.import_to_add(import_to_add);
404            return render_macro(ctx, path_ctx, local_name, mac);
405        }
406        ScopeDef::ModuleDef(Function(func)) => {
407            let ctx = ctx.import_to_add(import_to_add);
408            return render_fn(ctx, path_ctx, Some(local_name), func);
409        }
410        ScopeDef::ModuleDef(Variant(var)) => {
411            let ctx = ctx.clone().import_to_add(import_to_add.clone());
412            if let Some(item) =
413                render_variant_lit(ctx, path_ctx, Some(local_name.clone()), var, None)
414            {
415                return item;
416            }
417        }
418        _ => (),
419    }
420
421    let completion = ctx.completion;
422    let cap = ctx.snippet_cap();
423    let db = completion.db;
424    let config = completion.config;
425    let requires_import = import_to_add.is_some();
426
427    let name = local_name.display_no_db(ctx.completion.edition).to_smolstr();
428    let mut item = render_resolution_simple_(ctx, &local_name, import_to_add, resolution);
429    if local_name.needs_escape(completion.edition) {
430        item.insert_text(local_name.display_no_db(completion.edition).to_smolstr());
431    }
432    // Add `<>` for generic types
433    let type_path_no_ty_args = matches!(
434        path_ctx,
435        PathCompletionCtx { kind: PathKind::Type { .. }, has_type_args: false, .. }
436    ) && config.callable.is_some();
437    if type_path_no_ty_args && let Some(cap) = cap {
438        let has_non_default_type_params = match resolution {
439            ScopeDef::ModuleDef(hir::ModuleDef::Adt(it)) => it.has_non_default_type_params(db),
440            ScopeDef::ModuleDef(hir::ModuleDef::TypeAlias(it)) => {
441                it.has_non_default_type_params(db)
442            }
443            _ => false,
444        };
445
446        if has_non_default_type_params {
447            cov_mark::hit!(inserts_angle_brackets_for_generics);
448            item.lookup_by(name.clone())
449                .label(SmolStr::from_iter([&name, "<…>"]))
450                .trigger_call_info()
451                .insert_snippet(cap, format!("{}<$0>", local_name.display(db, completion.edition)));
452        }
453    }
454
455    let mut set_item_relevance = |ty: Type<'_>| {
456        if !ty.is_unknown() {
457            item.detail(ty.display(db, krate).to_string());
458        }
459
460        item.set_relevance(CompletionRelevance {
461            type_match: compute_type_match(completion, &ty),
462            exact_name_match: compute_exact_name_match(completion, &name),
463            is_local: matches!(resolution, ScopeDef::Local(_)),
464            requires_import,
465            ..CompletionRelevance::default()
466        });
467
468        path_ref_match(completion, path_ctx, &ty, &mut item);
469    };
470
471    match resolution {
472        ScopeDef::Local(local) => set_item_relevance(local.ty(db)),
473        ScopeDef::ModuleDef(ModuleDef::Adt(adt)) | ScopeDef::AdtSelfType(adt) => {
474            set_item_relevance(adt.ty(db))
475        }
476        // Filtered out above
477        ScopeDef::ModuleDef(
478            ModuleDef::Function(_) | ModuleDef::Variant(_) | ModuleDef::Macro(_),
479        ) => (),
480        ScopeDef::ModuleDef(ModuleDef::Const(konst)) => set_item_relevance(konst.ty(db)),
481        ScopeDef::ModuleDef(ModuleDef::Static(stat)) => set_item_relevance(stat.ty(db)),
482        ScopeDef::ModuleDef(ModuleDef::BuiltinType(bt)) => set_item_relevance(bt.ty(db)),
483        ScopeDef::ImplSelfType(imp) => set_item_relevance(imp.self_ty(db)),
484        ScopeDef::GenericParam(_)
485        | ScopeDef::Label(_)
486        | ScopeDef::Unknown
487        | ScopeDef::ModuleDef(
488            ModuleDef::Trait(_) | ModuleDef::Module(_) | ModuleDef::TypeAlias(_),
489        ) => (),
490    };
491
492    item
493}

495fn render_resolution_simple_(
496    ctx: RenderContext<'_>,
497    local_name: &hir::Name,
498    import_to_add: Option<LocatedImport>,
499    resolution: ScopeDef,
500) -> Builder {
501    let _p = tracing::info_span!("render_resolution_simple_").entered();
502
503    let db = ctx.db();
504    let ctx = ctx.import_to_add(import_to_add);
505    let kind = res_to_kind(resolution);
506
507    let mut item = CompletionItem::new(
508        kind,
509        ctx.source_range(),
510        local_name.as_str().to_smolstr(),
511        ctx.completion.edition,
512    );
513    item.set_relevance(ctx.completion_relevance())
514        .set_documentation(scope_def_docs(db, resolution))
515        .set_deprecated(scope_def_is_deprecated(&ctx, resolution));
516
517    if let Some(import_to_add) = ctx.import_to_add {
518        item.add_import(import_to_add);
519    }
520
521    item.doc_aliases(ctx.doc_aliases);
522    item
523}

525fn res_to_kind(resolution: ScopeDef) -> CompletionItemKind {
526    use hir::ModuleDef::*;
527    match resolution {
528        ScopeDef::Unknown => CompletionItemKind::UnresolvedReference,
529        ScopeDef::ModuleDef(Function(_)) => CompletionItemKind::SymbolKind(SymbolKind::Function),
530        ScopeDef::ModuleDef(Variant(_)) => CompletionItemKind::SymbolKind(SymbolKind::Variant),
531        ScopeDef::ModuleDef(Macro(_)) => CompletionItemKind::SymbolKind(SymbolKind::Macro),
532        ScopeDef::ModuleDef(Module(..)) => CompletionItemKind::SymbolKind(SymbolKind::Module),
533        ScopeDef::ModuleDef(Adt(adt)) => CompletionItemKind::SymbolKind(match adt {
534            hir::Adt::Struct(_) => SymbolKind::Struct,
535            hir::Adt::Union(_) => SymbolKind::Union,
536            hir::Adt::Enum(_) => SymbolKind::Enum,
537        }),
538        ScopeDef::ModuleDef(Const(..)) => CompletionItemKind::SymbolKind(SymbolKind::Const),
539        ScopeDef::ModuleDef(Static(..)) => CompletionItemKind::SymbolKind(SymbolKind::Static),
540        ScopeDef::ModuleDef(Trait(..)) => CompletionItemKind::SymbolKind(SymbolKind::Trait),
541        ScopeDef::ModuleDef(TypeAlias(..)) => CompletionItemKind::SymbolKind(SymbolKind::TypeAlias),
542        ScopeDef::ModuleDef(BuiltinType(..)) => CompletionItemKind::BuiltinType,
543        ScopeDef::GenericParam(param) => CompletionItemKind::SymbolKind(match param {
544            hir::GenericParam::TypeParam(_) => SymbolKind::TypeParam,
545            hir::GenericParam::ConstParam(_) => SymbolKind::ConstParam,
546            hir::GenericParam::LifetimeParam(_) => SymbolKind::LifetimeParam,
547        }),
548        ScopeDef::Local(..) => CompletionItemKind::SymbolKind(SymbolKind::Local),
549        ScopeDef::Label(..) => CompletionItemKind::SymbolKind(SymbolKind::Label),
550        ScopeDef::AdtSelfType(..) | ScopeDef::ImplSelfType(..) => {
551            CompletionItemKind::SymbolKind(SymbolKind::SelfParam)
552        }
553    }
554}

565        ScopeDef::ModuleDef(TypeAlias(it)) => it.docs(db),
566        _ => None,
567    }

595    ctx: &CompletionContext<'_>,
596    completion_ty: &hir::Type<'_>,
597) -> Option<CompletionRelevanceTypeMatch> {
598    let expected_type = ctx.expected_type.as_ref()?;
599
600    // We don't ever consider unit type to be an exact type match, since
601    // nearly always this is not meaningful to the user.
602    if expected_type.is_unit() {
603        return None;
604    }
605
606    match_types(ctx, expected_type, completion_ty)
607}
608
609fn compute_exact_name_match(ctx: &CompletionContext<'_>, completion_name: &str) -> bool {
610    ctx.expected_name.as_ref().is_some_and(|name| name.text() == completion_name)
611}
612
613fn compute_ref_match(
614    ctx: &CompletionContext<'_>,
615    completion_ty: &hir::Type<'_>,
616) -> Option<CompletionItemRefMode> {
617    let expected_type = ctx.expected_type.as_ref()?;
618    let expected_without_ref = expected_type.remove_ref();
619    let completion_without_ref = completion_ty.remove_ref();
620    if expected_type.could_unify_with(ctx.db, completion_ty) {
621        return None;
622    }
623    if let Some(expected_without_ref) = &expected_without_ref
624        && completion_ty.autoderef(ctx.db).any(|ty| ty == *expected_without_ref)
625    {
626        cov_mark::hit!(suggest_ref);
627        let mutability = if expected_type.is_mutable_reference() {
628            hir::Mutability::Mut
629        } else {
630            hir::Mutability::Shared
631        };
632        return Some(CompletionItemRefMode::Reference(mutability));
633    }
634
635    if let Some(completion_without_ref) = completion_without_ref
636        && completion_without_ref == *expected_type
637        && completion_without_ref.is_copy(ctx.db)
638    {
639        cov_mark::hit!(suggest_deref);
640        return Some(CompletionItemRefMode::Dereference);
641    }
642
643    None
644}
645
646fn path_ref_match(
647    completion: &CompletionContext<'_>,
648    path_ctx: &PathCompletionCtx<'_>,
649    ty: &hir::Type<'_>,
650    item: &mut Builder,
651) {
652    if let Some(original_path) = &path_ctx.original_path {
653        // At least one char was typed by the user already, in that case look for the original path
654        if let Some(original_path) = completion.sema.original_ast_node(original_path.clone())
655            && let Some(ref_mode) = compute_ref_match(completion, ty)
656        {
657            item.ref_match(ref_mode, original_path.syntax().text_range().start());
658        }
659    } else {
660        // completion requested on an empty identifier, there is no path here yet.
661        // FIXME: This might create inconsistent completions where we show a ref match in macro inputs
662        // as long as nothing was typed yet
663        if let Some(ref_mode) = compute_ref_match(completion, ty) {
664            item.ref_match(ref_mode, completion.position.offset);
665        }
666    }
667}

738    fn check_relevance_(actual: Vec<CompletionItem>, expect: Expect) {
739        let actual = actual
740            .into_iter()
741            .flat_map(|it| {
742                let mut items = vec![];
743
744                let tag = it.kind.tag();
745                let relevance = display_relevance(it.relevance);
746                items.push(format!(
747                    "{tag} {} {} {relevance}\n",
748                    it.label.primary,
749                    it.label.detail_right.clone().unwrap_or_default(),
750                ));
751
752                if let Some((label, _indel, relevance)) = it.ref_match() {
753                    let relevance = display_relevance(relevance);
754
755                    items.push(format!("{tag} {label} {relevance}\n"));
756                }
757
758                items
759            })
760            .collect::<String>();
761
762        expect.assert_eq(&actual);
763
764        fn display_relevance(relevance: CompletionRelevance) -> String {
765            let relevance_factors = vec![
766                (relevance.type_match == Some(CompletionRelevanceTypeMatch::Exact), "type"),
767                (
768                    relevance.type_match == Some(CompletionRelevanceTypeMatch::CouldUnify),
769                    "type_could_unify",
770                ),
771                (relevance.exact_name_match, "name"),
772                (relevance.is_local, "local"),
773                (
774                    relevance.postfix_match == Some(CompletionRelevancePostfixMatch::Exact),
775                    "snippet",
776                ),
777                (relevance.trait_.is_some_and(|it| it.is_op_method), "op_method"),
778                (relevance.requires_import, "requires_import"),
779            ]
780            .into_iter()
781            .filter_map(|(cond, desc)| if cond { Some(desc) } else { None })
782            .join("+");
783
784            format!("[{relevance_factors}]")
785        }
786    }
