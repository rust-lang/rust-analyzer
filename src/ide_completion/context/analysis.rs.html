55pub(super) fn expand_and_analyze<'db>(
56    sema: &Semantics<'db, RootDatabase>,
57    original_file: InFile<SyntaxNode>,
58    speculative_file: SyntaxNode,
59    offset: TextSize,
60    original_token: &SyntaxToken,
61) -> Option<AnalysisResult<'db>> {
62    // as we insert after the offset, right biased will *always* pick the identifier no matter
63    // if there is an ident already typed or not
64    let fake_ident_token = speculative_file.token_at_offset(offset).right_biased()?;
65    // the relative offset between the cursor and the *identifier* token we are completing on
66    let relative_offset = offset - fake_ident_token.text_range().start();
67    // make the offset point to the start of the original token, as that is what the
68    // intermediate offsets calculated in expansion always points to
69    let offset = offset - relative_offset;
70    let expansion = expand_maybe_stop(
71        sema,
72        original_file.clone(),
73        speculative_file.clone(),
74        offset,
75        fake_ident_token.clone(),
76        relative_offset,
77    )
78    .unwrap_or(ExpansionResult {
79        original_file: original_file.value,
80        speculative_file,
81        original_offset: offset,
82        speculative_offset: fake_ident_token.text_range().start(),
83        fake_ident_token,
84        derive_ctx: None,
85    });
86
87    // add the relative offset back, so that left_biased finds the proper token
88    let original_offset = expansion.original_offset + relative_offset;
89    let token = expansion.original_file.token_at_offset(original_offset).left_biased()?;
90
91    hir::attach_db(sema.db, || analyze(sema, expansion, original_token, &token)).map(
92        |(analysis, expected, qualifier_ctx)| AnalysisResult {
93            analysis,
94            expected,
95            qualifier_ctx,
96            token,
97            original_offset,
98        },
99    )
100}

104    algo::skip_whitespace_token(token, Direction::Prev)

107/// Expand attributes and macro calls at the current cursor position for both the original file
108/// and fake file repeatedly. As soon as one of the two expansions fail we stop so the original
109/// and speculative states stay in sync.
110///
111/// We do this by recursively expanding all macros and picking the best possible match. We cannot just
112/// choose the first expansion each time because macros can expand to something that does not include
113/// our completion marker, e.g.:
114///
115/// ```ignore
116/// macro_rules! helper { ($v:ident) => {} }
117/// macro_rules! my_macro {
118///     ($v:ident) => {
119///         helper!($v);
120///         $v
121///     };
122/// }
123///
124/// my_macro!(complete_me_here);
125/// ```
126/// If we would expand the first thing we encounter only (which in fact this method used to do), we would
127/// be unable to complete here, because we would be walking directly into the void. So we instead try
128/// *every* possible path.
129///
130/// This can also creates discrepancies between the speculative and real expansions: because we insert
131/// tokens, we insert characters, which means if we try the second occurrence it may not be at the same
132/// position in the original and speculative file. We take an educated guess here, and for each token
133/// that we check, we subtract `COMPLETION_MARKER.len()`. This may not be accurate because proc macros
134/// can insert the text of the completion marker in other places while removing the span, but this is
135/// the best we can do.
136fn expand_maybe_stop(
137    sema: &Semantics<'_, RootDatabase>,
138    original_file: InFile<SyntaxNode>,
139    speculative_file: SyntaxNode,
140    original_offset: TextSize,
141    fake_ident_token: SyntaxToken,
142    relative_offset: TextSize,
143) -> Option<ExpansionResult> {
144    if let result @ Some(_) = expand(
145        sema,
146        original_file.clone(),
147        speculative_file.clone(),
148        original_offset,
149        fake_ident_token.clone(),
150        relative_offset,
151    ) {
152        return result;
153    }
154
155    // We can't check whether the fake expansion is inside macro call, because that requires semantic info.
156    // But hopefully checking just the real one should be enough.
157    if token_at_offset_ignore_whitespace(&original_file.value, original_offset + relative_offset)
158        .is_some_and(|original_token| {
159            !sema.is_inside_macro_call(original_file.with_value(&original_token))
160        })
161    {
162        // Recursion base case.
163        Some(ExpansionResult {
164            original_file: original_file.value,
165            speculative_file,
166            original_offset,
167            speculative_offset: fake_ident_token.text_range().start(),
168            fake_ident_token,
169            derive_ctx: None,
170        })
171    } else {
172        None
173    }
174}

176fn expand(
177    sema: &Semantics<'_, RootDatabase>,
178    original_file: InFile<SyntaxNode>,
179    speculative_file: SyntaxNode,
180    original_offset: TextSize,
181    fake_ident_token: SyntaxToken,
182    relative_offset: TextSize,
183) -> Option<ExpansionResult> {
184    let _p = tracing::info_span!("CompletionContext::expand").entered();
185
186    let parent_item =
187        |item: &ast::Item| item.syntax().ancestors().skip(1).find_map(ast::Item::cast);
188    let original_node = token_at_offset_ignore_whitespace(&original_file.value, original_offset)
189        .and_then(|token| token.parent_ancestors().find_map(ast::Item::cast));
190    let ancestor_items = iter::successors(
191        Option::zip(
192            original_node,
193            find_node_at_offset::<ast::Item>(
194                &speculative_file,
195                fake_ident_token.text_range().start(),
196            ),
197        ),
198        |(a, b)| parent_item(a).zip(parent_item(b)),
199    );
200
201    // first try to expand attributes as these are always the outermost macro calls
202    'ancestors: for (actual_item, item_with_fake_ident) in ancestor_items {
203        match (
204            sema.expand_attr_macro(&actual_item),
205            sema.speculative_expand_attr_macro(
206                &actual_item,
207                &item_with_fake_ident,
208                fake_ident_token.clone(),
209            ),
210        ) {
211            // maybe parent items have attributes, so continue walking the ancestors
212            (None, None) => continue 'ancestors,
213            // successful expansions
214            (
215                Some(ExpandResult { value: actual_expansion, err: _ }),
216                Some((fake_expansion, fake_mapped_tokens)),
217            ) => {
218                let mut accumulated_offset_from_fake_tokens = 0;
219                let actual_range = actual_expansion.text_range().end();
220                let result = fake_mapped_tokens
221                    .into_iter()
222                    .filter_map(|(fake_mapped_token, rank)| {
223                        let accumulated_offset = accumulated_offset_from_fake_tokens;
224                        if !fake_mapped_token.text().contains(COMPLETION_MARKER) {
225                            // Proc macros can make the same span with different text, we don't
226                            // want them to participate in completion because the macro author probably
227                            // didn't intend them to.
228                            return None;
229                        }
230                        accumulated_offset_from_fake_tokens += COMPLETION_MARKER.len();
231
232                        let new_offset = fake_mapped_token.text_range().start()
233                            - TextSize::new(accumulated_offset as u32);
234                        if new_offset + relative_offset > actual_range {
235                            // offset outside of bounds from the original expansion,
236                            // stop here to prevent problems from happening
237                            return None;
238                        }
239                        let result = expand_maybe_stop(
240                            sema,
241                            actual_expansion.clone(),
242                            fake_expansion.clone(),
243                            new_offset,
244                            fake_mapped_token,
245                            relative_offset,
246                        )?;
247                        Some((result, rank))
248                    })
249                    .min_by_key(|(_, rank)| *rank)
250                    .map(|(result, _)| result);
251                if result.is_some() {
252                    return result;
253                }
254            }
255            // exactly one expansion failed, inconsistent state so stop expanding completely
256            _ => break 'ancestors,
257        }
258    }
259
260    // No attributes have been expanded, so look for macro_call! token trees or derive token trees
261    let orig_tt = ancestors_at_offset(&original_file.value, original_offset)
262        .map_while(Either::<ast::TokenTree, ast::Meta>::cast)
263        .last()?;
264    let spec_tt = ancestors_at_offset(&speculative_file, fake_ident_token.text_range().start())
265        .map_while(Either::<ast::TokenTree, ast::Meta>::cast)
266        .last()?;
267
268    let (tts, attrs) = match (orig_tt, spec_tt) {
269        (Either::Left(orig_tt), Either::Left(spec_tt)) => {
270            let attrs = orig_tt
271                .syntax()
272                .parent()
273                .and_then(ast::Meta::cast)
274                .and_then(|it| it.parent_attr())
275                .zip(
276                    spec_tt
277                        .syntax()
278                        .parent()
279                        .and_then(ast::Meta::cast)
280                        .and_then(|it| it.parent_attr()),
281                );
282            (Some((orig_tt, spec_tt)), attrs)
283        }
284        (Either::Right(orig_path), Either::Right(spec_path)) => {
285            (None, orig_path.parent_attr().zip(spec_path.parent_attr()))
286        }
287        _ => return None,
288    };
289
290    // Expand pseudo-derive expansion aka `derive(Debug$0)`
291    if let Some((orig_attr, spec_attr)) = attrs {
292        if let (Some(actual_expansion), Some((fake_expansion, fake_mapped_tokens))) = (
293            sema.expand_derive_as_pseudo_attr_macro(&orig_attr),
294            sema.speculative_expand_derive_as_pseudo_attr_macro(
295                &orig_attr,
296                &spec_attr,
297                fake_ident_token.clone(),
298            ),
299        ) && let Some((fake_mapped_token, _)) =
300            fake_mapped_tokens.into_iter().min_by_key(|(_, rank)| *rank)
301        {
302            return Some(ExpansionResult {
303                original_file: original_file.value,
304                speculative_file,
305                original_offset,
306                speculative_offset: fake_ident_token.text_range().start(),
307                fake_ident_token,
308                derive_ctx: Some((
309                    actual_expansion,
310                    fake_expansion,
311                    fake_mapped_token.text_range().start(),
312                    orig_attr,
313                )),
314            });
315        }
316
317        if let Some(spec_adt) =
318            spec_attr.syntax().ancestors().find_map(ast::Item::cast).and_then(|it| match it {
319                ast::Item::Struct(it) => Some(ast::Adt::Struct(it)),
320                ast::Item::Enum(it) => Some(ast::Adt::Enum(it)),
321                ast::Item::Union(it) => Some(ast::Adt::Union(it)),
322                _ => None,
323            })
324        {
325            // might be the path of derive helper or a token tree inside of one
326            if let Some(helpers) = sema.derive_helper(&orig_attr) {
327                for (_mac, file) in helpers {
328                    if let Some((fake_expansion, fake_mapped_tokens)) = sema.speculative_expand_raw(
329                        file,
330                        spec_adt.syntax(),
331                        fake_ident_token.clone(),
332                    ) {
333                        // we are inside a derive helper token tree, treat this as being inside
334                        // the derive expansion
335                        let actual_expansion = sema.parse_or_expand(file.into());
336                        let mut accumulated_offset_from_fake_tokens = 0;
337                        let actual_range = actual_expansion.text_range().end();
338                        let result = fake_mapped_tokens
339                            .into_iter()
340                            .filter_map(|(fake_mapped_token, rank)| {
341                                let accumulated_offset = accumulated_offset_from_fake_tokens;
342                                if !fake_mapped_token.text().contains(COMPLETION_MARKER) {
343                                    // Proc macros can make the same span with different text, we don't
344                                    // want them to participate in completion because the macro author probably
345                                    // didn't intend them to.
346                                    return None;
347                                }
348                                accumulated_offset_from_fake_tokens += COMPLETION_MARKER.len();
349
350                                let new_offset = fake_mapped_token.text_range().start()
351                                    - TextSize::new(accumulated_offset as u32);
352                                if new_offset + relative_offset > actual_range {
353                                    // offset outside of bounds from the original expansion,
354                                    // stop here to prevent problems from happening
355                                    return None;
356                                }
357                                let result = expand_maybe_stop(
358                                    sema,
359                                    InFile::new(file.into(), actual_expansion.clone()),
360                                    fake_expansion.clone(),
361                                    new_offset,
362                                    fake_mapped_token,
363                                    relative_offset,
364                                )?;
365                                Some((result, rank))
366                            })
367                            .min_by_key(|(_, rank)| *rank)
368                            .map(|(result, _)| result);
369                        if result.is_some() {
370                            return result;
371                        }
372                    }
373                }
374            }
375        }
376        // at this point we won't have any more successful expansions, so stop
377        return None;
378    }
379
380    // Expand fn-like macro calls
381    let (orig_tt, spec_tt) = tts?;
382    let (actual_macro_call, macro_call_with_fake_ident) = (
383        orig_tt.syntax().parent().and_then(ast::MacroCall::cast)?,
384        spec_tt.syntax().parent().and_then(ast::MacroCall::cast)?,
385    );
386    let mac_call_path0 = actual_macro_call.path().as_ref().map(|s| s.syntax().text());
387    let mac_call_path1 = macro_call_with_fake_ident.path().as_ref().map(|s| s.syntax().text());
388
389    // inconsistent state, stop expanding
390    if mac_call_path0 != mac_call_path1 {
391        return None;
392    }
393    let speculative_args = macro_call_with_fake_ident.token_tree()?;
394
395    match (
396        sema.expand_macro_call(&actual_macro_call),
397        sema.speculative_expand_macro_call(&actual_macro_call, &speculative_args, fake_ident_token),
398    ) {
399        // successful expansions
400        (Some(actual_expansion), Some((fake_expansion, fake_mapped_tokens))) => {
401            let mut accumulated_offset_from_fake_tokens = 0;
402            let actual_range = actual_expansion.text_range().end();
403            fake_mapped_tokens
404                .into_iter()
405                .filter_map(|(fake_mapped_token, rank)| {
406                    let accumulated_offset = accumulated_offset_from_fake_tokens;
407                    if !fake_mapped_token.text().contains(COMPLETION_MARKER) {
408                        // Proc macros can make the same span with different text, we don't
409                        // want them to participate in completion because the macro author probably
410                        // didn't intend them to.
411                        return None;
412                    }
413                    accumulated_offset_from_fake_tokens += COMPLETION_MARKER.len();
414
415                    let new_offset = fake_mapped_token.text_range().start()
416                        - TextSize::new(accumulated_offset as u32);
417                    if new_offset + relative_offset > actual_range {
418                        // offset outside of bounds from the original expansion,
419                        // stop here to prevent problems from happening
420                        return None;
421                    }
422                    let result = expand_maybe_stop(
423                        sema,
424                        actual_expansion.clone(),
425                        fake_expansion.clone(),
426                        new_offset,
427                        fake_mapped_token,
428                        relative_offset,
429                    )?;
430                    Some((result, rank))
431                })
432                .min_by_key(|(_, rank)| *rank)
433                .map(|(result, _)| result)
434        }
435        // at least one expansion failed, we won't have anything to expand from this point
436        // onwards so break out
437        _ => None,
438    }
439}

441/// Fill the completion context, this is what does semantic reasoning about the surrounding context
442/// of the completion location.
443fn analyze<'db>(
444    sema: &Semantics<'db, RootDatabase>,
445    expansion_result: ExpansionResult,
446    original_token: &SyntaxToken,
447    self_token: &SyntaxToken,
448) -> Option<(CompletionAnalysis<'db>, (Option<Type<'db>>, Option<ast::NameOrNameRef>), QualifierCtx)>
449{
450    let _p = tracing::info_span!("CompletionContext::analyze").entered();
451    let ExpansionResult {
452        original_file,
453        speculative_file,
454        original_offset: _,
455        speculative_offset,
456        fake_ident_token,
457        derive_ctx,
458    } = expansion_result;
459
460    if original_token.kind() != self_token.kind()
461        // FIXME: This check can be removed once we use speculative database forking for completions
462        && !(original_token.kind().is_punct() || original_token.kind().is_trivia())
463        && !(SyntaxKind::is_any_identifier(original_token.kind())
464            && SyntaxKind::is_any_identifier(self_token.kind()))
465    {
466        return None;
467    }
468
469    // Overwrite the path kind for derives
470    if let Some((original_file, file_with_fake_ident, offset, origin_attr)) = derive_ctx {
471        if let Some(ast::NameLike::NameRef(name_ref)) =
472            find_node_at_offset(&file_with_fake_ident, offset)
473        {
474            let parent = name_ref.syntax().parent()?;
475            let (mut nameref_ctx, _) =
476                classify_name_ref(sema, &original_file, name_ref, offset, parent)?;
477            if let NameRefKind::Path(path_ctx) = &mut nameref_ctx.kind {
478                path_ctx.kind = PathKind::Derive {
479                    existing_derives: sema
480                        .resolve_derive_macro(&origin_attr)
481                        .into_iter()
482                        .flatten()
483                        .flatten()
484                        .collect(),
485                };
486            }
487            return Some((
488                CompletionAnalysis::NameRef(nameref_ctx),
489                (None, None),
490                QualifierCtx::default(),
491            ));
492        }
493        return None;
494    }
495
496    let Some(name_like) = find_node_at_offset(&speculative_file, speculative_offset) else {
497        let analysis = if let Some(original) = ast::String::cast(original_token.clone()) {
498            CompletionAnalysis::String { original, expanded: ast::String::cast(self_token.clone()) }
499        } else {
500            // Fix up trailing whitespace problem
501            // #[attr(foo = $0
502            let token = syntax::algo::skip_trivia_token(self_token.clone(), Direction::Prev)?;
503            let p = token.parent()?;
504            if p.kind() == SyntaxKind::TOKEN_TREE
505                && p.ancestors().any(|it| it.kind() == SyntaxKind::META)
506            {
507                let colon_prefix = previous_non_trivia_token(self_token.clone())
508                    .is_some_and(|it| T![:] == it.kind());
509
510                CompletionAnalysis::UnexpandedAttrTT {
511                    fake_attribute_under_caret: fake_ident_token
512                        .parent_ancestors()
513                        .find_map(ast::Attr::cast),
514                    colon_prefix,
515                    extern_crate: p.ancestors().find_map(ast::ExternCrate::cast),
516                }
517            } else {
518                return None;
519            }
520        };
521        return Some((analysis, (None, None), QualifierCtx::default()));
522    };
523
524    let expected = expected_type_and_name(sema, self_token, &name_like);
525    let mut qual_ctx = QualifierCtx::default();
526    let analysis = match name_like {
527        ast::NameLike::Lifetime(lifetime) => {
528            CompletionAnalysis::Lifetime(classify_lifetime(sema, &original_file, lifetime)?)
529        }
530        ast::NameLike::NameRef(name_ref) => {
531            let parent = name_ref.syntax().parent()?;
532            let (nameref_ctx, qualifier_ctx) = classify_name_ref(
533                sema,
534                &original_file,
535                name_ref,
536                expansion_result.original_offset,
537                parent,
538            )?;
539
540            if let NameRefContext {
541                kind:
542                    NameRefKind::Path(PathCompletionCtx { kind: PathKind::Expr { .. }, path, .. }, ..),
543                ..
544            } = &nameref_ctx
545                && is_in_token_of_for_loop(path)
546            {
547                // for pat $0
548                // there is nothing to complete here except `in` keyword
549                // don't bother populating the context
550                // Ideally this special casing wouldn't be needed, but the parser recovers
551                return None;
552            }
553
554            qual_ctx = qualifier_ctx;
555            CompletionAnalysis::NameRef(nameref_ctx)
556        }
557        ast::NameLike::Name(name) => {
558            let name_ctx = classify_name(sema, &original_file, name)?;
559            CompletionAnalysis::Name(name_ctx)
560        }
561    };
562    Some((analysis, expected, qual_ctx))
563}

565/// Calculate the expected type and name of the cursor position.
566fn expected_type_and_name<'db>(
567    sema: &Semantics<'db, RootDatabase>,
568    self_token: &SyntaxToken,
569    name_like: &ast::NameLike,
570) -> (Option<Type<'db>>, Option<NameOrNameRef>) {
571    let token = prev_special_biased_token_at_trivia(self_token.clone());
572    let mut node = match token.parent() {
573        Some(it) => it,
574        None => return (None, None),
575    };
576
577    let strip_refs = |mut ty: Type<'db>| match name_like {
578        ast::NameLike::NameRef(n) => {
579            let p = match n.syntax().parent() {
580                Some(it) => it,
581                None => return ty,
582            };
583            let top_syn = match_ast! {
584                match p {
585                    ast::FieldExpr(e) => e
586                        .syntax()
587                        .ancestors()
588                        .take_while(|it| ast::FieldExpr::can_cast(it.kind()))
589                        .last(),
590                    ast::PathSegment(e) => e
591                        .syntax()
592                        .ancestors()
593                        .skip(1)
594                        .take_while(|it| ast::Path::can_cast(it.kind()) || ast::PathExpr::can_cast(it.kind()))
595                        .find(|it| ast::PathExpr::can_cast(it.kind())),
596                    _ => None
597                }
598            };
599            let top_syn = match top_syn {
600                Some(it) => it,
601                None => return ty,
602            };
603            for _ in top_syn.ancestors().skip(1).map_while(ast::RefExpr::cast) {
604                cov_mark::hit!(expected_type_fn_param_ref);
605                ty = ty.strip_reference();
606            }
607            ty
608        }
609        _ => ty,
610    };
611
612    let (ty, name) = loop {
613        break match_ast! {
614            match node {
615                ast::LetStmt(it) => {
616                    cov_mark::hit!(expected_type_let_with_leading_char);
617                    cov_mark::hit!(expected_type_let_without_leading_char);
618                    let ty = it.pat()
619                        .and_then(|pat| sema.type_of_pat(&pat))
620                        .or_else(|| it.initializer().and_then(|it| sema.type_of_expr(&it)))
621                        .map(TypeInfo::original)
622                        .filter(|ty| {
623                            // don't infer the let type if the expr is a function,
624                            // preventing parenthesis from vanishing
625                            it.ty().is_some() || !ty.is_fn()
626                        });
627                    let name = match it.pat() {
628                        Some(ast::Pat::IdentPat(ident)) => ident.name().map(NameOrNameRef::Name),
629                        Some(_) | None => None,
630                    };
631
632                    (ty, name)
633                },
634                ast::LetExpr(it) => {
635                    cov_mark::hit!(expected_type_if_let_without_leading_char);
636                    let ty = it.pat()
637                        .and_then(|pat| sema.type_of_pat(&pat))
638                        .or_else(|| it.expr().and_then(|it| sema.type_of_expr(&it)))
639                        .map(TypeInfo::original);
640                    (ty, None)
641                },
642                ast::BinExpr(it) => {
643                    if let Some(ast::BinaryOp::Assignment { op: None }) = it.op_kind() {
644                        let ty = it.lhs()
645                            .and_then(|lhs| sema.type_of_expr(&lhs))
646                            .or_else(|| it.rhs().and_then(|rhs| sema.type_of_expr(&rhs)))
647                            .map(TypeInfo::original);
648                        (ty, None)
649                    } else if let Some(ast::BinaryOp::LogicOp(_)) = it.op_kind() {
650                        let ty = sema.type_of_expr(&it.clone().into()).map(TypeInfo::original);
651                        (ty, None)
652                    } else {
653                        (None, None)
654                    }
655                },
656                ast::ArgList(_) => {
657                    cov_mark::hit!(expected_type_fn_param);
658                    ActiveParameter::at_token(
659                        sema,
660                       token.clone(),
661                    ).map(|ap| {
662                        let name = ap.ident().map(NameOrNameRef::Name);
663                        (Some(ap.ty), name)
664                    })
665                    .unwrap_or((None, None))
666                },
667                ast::RecordExprFieldList(it) => {
668                    // wouldn't try {} be nice...
669                    (|| {
670                        if token.kind() == T![..]
671                            ||token.prev_token().map(|t| t.kind()) == Some(T![..])
672                        {
673                            cov_mark::hit!(expected_type_struct_func_update);
674                            let record_expr = it.syntax().parent().and_then(ast::RecordExpr::cast)?;
675                            let ty = sema.type_of_expr(&record_expr.into())?;
676                            Some((
677                                Some(ty.original),
678                                None
679                            ))
680                        } else {
681                            cov_mark::hit!(expected_type_struct_field_without_leading_char);
682                            cov_mark::hit!(expected_type_struct_field_followed_by_comma);
683                            let expr_field = previous_non_trivia_token(token.clone())?.parent().and_then(ast::RecordExprField::cast)?;
684                            let (_, _, ty) = sema.resolve_record_field(&expr_field)?;
685                            Some((
686                                Some(ty),
687                                expr_field.field_name().map(NameOrNameRef::NameRef),
688                            ))
689                        }
690                    })().unwrap_or((None, None))
691                },
692                ast::RecordExprField(it) => {
693                    let field_ty = sema.resolve_record_field(&it).map(|(_, _, ty)| ty);
694                    let field_name = it.field_name().map(NameOrNameRef::NameRef);
695                    if let Some(expr) = it.expr() {
696                        cov_mark::hit!(expected_type_struct_field_with_leading_char);
697                        let ty = field_ty
698                            .or_else(|| sema.type_of_expr(&expr).map(TypeInfo::original));
699                        (ty, field_name)
700                    } else {
701                        (field_ty, field_name)
702                    }
703                },
704                // match foo { $0 }
705                // match foo { ..., pat => $0 }
706                ast::MatchExpr(it) => {
707                    let on_arrow = previous_non_trivia_token(token.clone()).is_some_and(|it| T![=>] == it.kind());
708
709                    let ty = if on_arrow {
710                        // match foo { ..., pat => $0 }
711                        cov_mark::hit!(expected_type_match_arm_body_without_leading_char);
712                        cov_mark::hit!(expected_type_match_arm_body_with_leading_char);
713                        sema.type_of_expr(&it.into())
714                    } else {
715                        // match foo { $0 }
716                        cov_mark::hit!(expected_type_match_arm_without_leading_char);
717                        it.expr().and_then(|e| sema.type_of_expr(&e))
718                    }.map(TypeInfo::original);
719                    (ty, None)
720                },
721                ast::IfExpr(it) => {
722                    let ty = if let Some(body) = it.then_branch()
723                        && token.text_range().end() > body.syntax().text_range().start()
724                    {
725                        sema.type_of_expr(&body.into())
726                    } else {
727                        it.condition().and_then(|e| sema.type_of_expr(&e))
728                    }.map(TypeInfo::original);
729                    (ty, None)
730                },
731                ast::IdentPat(it) => {
732                    cov_mark::hit!(expected_type_if_let_with_leading_char);
733                    cov_mark::hit!(expected_type_match_arm_with_leading_char);
734                    let ty = sema.type_of_pat(&ast::Pat::from(it)).map(TypeInfo::original);
735                    (ty, None)
736                },
737                ast::Fn(it) => {
738                    cov_mark::hit!(expected_type_fn_ret_with_leading_char);
739                    cov_mark::hit!(expected_type_fn_ret_without_leading_char);
740                    let def = sema.to_def(&it);
741                    (def.map(|def| def.ret_type(sema.db)), None)
742                },
743                ast::ReturnExpr(it) => {
744                    let fn_ = sema.ancestors_with_macros(it.syntax().clone())
745                        .find_map(Either::<ast::Fn, ast::ClosureExpr>::cast);
746                    let ty = fn_.and_then(|f| match f {
747                        Either::Left(f) => Some(sema.to_def(&f)?.ret_type(sema.db)),
748                        Either::Right(f) => {
749                            let ty = sema.type_of_expr(&f.into())?.original.as_callable(sema.db)?;
750                            Some(ty.return_type())
751                        },
752                    });
753                    (ty, None)
754                },
755                ast::ClosureExpr(it) => {
756                    let ty = sema.type_of_expr(&it.into());
757                    ty.and_then(|ty| ty.original.as_callable(sema.db))
758                        .map(|c| (Some(c.return_type()), None))
759                        .unwrap_or((None, None))
760                },
761                ast::ParamList(it) => {
762                    let closure = it.syntax().parent().and_then(ast::ClosureExpr::cast);
763                    let ty = closure
764                        .filter(|_| it.syntax().text_range().end() <= self_token.text_range().start())
765                        .and_then(|it| sema.type_of_expr(&it.into()));
766                    ty.and_then(|ty| ty.original.as_callable(sema.db))
767                        .map(|c| (Some(c.return_type()), None))
768                        .unwrap_or((None, None))
769                },
770                ast::Stmt(_) => (None, None),
771                ast::Item(_) => (None, None),
772                _ => {
773                    match node.parent() {
774                        Some(n) => {
775                            node = n;
776                            continue;
777                        },
778                        None => (None, None),
779                    }
780                },
781            }
782        };
783    };
784    (ty.map(strip_refs), name)
785}

787fn classify_lifetime(
788    sema: &Semantics<'_, RootDatabase>,
789    original_file: &SyntaxNode,
790    lifetime: ast::Lifetime,
791) -> Option<LifetimeContext> {
792    let parent = lifetime.syntax().parent()?;
793    if parent.kind() == SyntaxKind::ERROR {
794        return None;
795    }
796
797    let lifetime =
798        find_node_at_offset::<ast::Lifetime>(original_file, lifetime.syntax().text_range().start());
799    let kind = match_ast! {
800        match parent {
801            ast::LifetimeParam(_) => LifetimeKind::LifetimeParam,
802            ast::BreakExpr(_) => LifetimeKind::LabelRef,
803            ast::ContinueExpr(_) => LifetimeKind::LabelRef,
804            ast::Label(_) => LifetimeKind::LabelDef,
805            _ => {
806                let def = lifetime.as_ref().and_then(|lt| sema.scope(lt.syntax())?.generic_def());
807                LifetimeKind::Lifetime { in_lifetime_param_bound: ast::TypeBound::can_cast(parent.kind()), def }
808            },
809        }
810    };
811
812    Some(LifetimeContext { kind })
813}

815fn classify_name(
816    sema: &Semantics<'_, RootDatabase>,
817    original_file: &SyntaxNode,
818    name: ast::Name,
819) -> Option<NameContext> {
820    let parent = name.syntax().parent()?;
821    let kind = match_ast! {
822        match parent {
823            ast::Const(_) => NameKind::Const,
824            ast::ConstParam(_) => NameKind::ConstParam,
825            ast::Enum(_) => NameKind::Enum,
826            ast::Fn(_) => NameKind::Function,
827            ast::IdentPat(bind_pat) => {
828                let mut pat_ctx = pattern_context_for(sema, original_file, bind_pat.into());
829                if let Some(record_field) = ast::RecordPatField::for_field_name(&name) {
830                    pat_ctx.record_pat = find_node_in_file_compensated(sema, original_file, &record_field.parent_record_pat());
831                }
832
833                NameKind::IdentPat(pat_ctx)
834            },
835            ast::MacroDef(_) => NameKind::MacroDef,
836            ast::MacroRules(_) => NameKind::MacroRules,
837            ast::Module(module) => NameKind::Module(module),
838            ast::RecordField(_) => NameKind::RecordField,
839            ast::Rename(_) => NameKind::Rename,
840            ast::SelfParam(_) => NameKind::SelfParam,
841            ast::Static(_) => NameKind::Static,
842            ast::Struct(_) => NameKind::Struct,
843            ast::Trait(_) => NameKind::Trait,
844            ast::TypeAlias(_) => NameKind::TypeAlias,
845            ast::TypeParam(_) => NameKind::TypeParam,
846            ast::Union(_) => NameKind::Union,
847            ast::Variant(_) => NameKind::Variant,
848            _ => return None,
849        }
850    };
851    let name = find_node_at_offset(original_file, name.syntax().text_range().start());
852    Some(NameContext { name, kind })
853}
854
855fn classify_name_ref<'db>(
856    sema: &Semantics<'db, RootDatabase>,
857    original_file: &SyntaxNode,
858    name_ref: ast::NameRef,
859    original_offset: TextSize,
860    parent: SyntaxNode,
861) -> Option<(NameRefContext<'db>, QualifierCtx)> {
862    let nameref = find_node_at_offset(original_file, original_offset);
863
864    let make_res = |kind| (NameRefContext { nameref: nameref.clone(), kind }, Default::default());
865
866    if let Some(record_field) = ast::RecordExprField::for_field_name(&name_ref) {
867        let dot_prefix = previous_non_trivia_token(name_ref.syntax().clone())
868            .is_some_and(|it| T![.] == it.kind());
869
870        return find_node_in_file_compensated(
871            sema,
872            original_file,
873            &record_field.parent_record_lit(),
874        )
875        .map(|expr| NameRefKind::RecordExpr { expr, dot_prefix })
876        .map(make_res);
877    }
878    if let Some(record_field) = ast::RecordPatField::for_field_name_ref(&name_ref) {
879        let kind = NameRefKind::Pattern(PatternContext {
880            param_ctx: None,
881            has_type_ascription: false,
882            ref_token: None,
883            mut_token: None,
884            record_pat: find_node_in_file_compensated(
885                sema,
886                original_file,
887                &record_field.parent_record_pat(),
888            ),
889            ..pattern_context_for(sema, original_file, record_field.parent_record_pat().into())
890        });
891        return Some(make_res(kind));
892    }
893
894    let field_expr_handle = |receiver, node| {
895        let receiver = find_opt_node_in_file(original_file, receiver);
896        let receiver_is_ambiguous_float_literal = match &receiver {
897            Some(ast::Expr::Literal(l)) => matches! {
898                l.kind(),
899                ast::LiteralKind::FloatNumber { .. } if l.syntax().last_token().is_some_and(|it| it.text().ends_with('.'))
900            },
901            _ => false,
902        };
903
904        let receiver_is_part_of_indivisible_expression = match &receiver {
905            Some(ast::Expr::IfExpr(_)) => {
906                let next_token_kind =
907                    next_non_trivia_token(name_ref.syntax().clone()).map(|t| t.kind());
908                next_token_kind == Some(SyntaxKind::ELSE_KW)
909            }
910            _ => false,
911        };
912        if receiver_is_part_of_indivisible_expression {
913            return None;
914        }
915
916        let mut receiver_ty = receiver.as_ref().and_then(|it| sema.type_of_expr(it));
917        if receiver_is_ambiguous_float_literal {
918            // `123.|` is parsed as a float but should actually be an integer.
919            always!(receiver_ty.as_ref().is_none_or(|receiver_ty| receiver_ty.original.is_float()));
920            receiver_ty =
921                Some(TypeInfo { original: hir::BuiltinType::i32().ty(sema.db), adjusted: None });
922        }
923
924        let kind = NameRefKind::DotAccess(DotAccess {
925            receiver_ty,
926            kind: DotAccessKind::Field { receiver_is_ambiguous_float_literal },
927            receiver,
928            ctx: DotAccessExprCtx {
929                in_block_expr: is_in_block(node),
930                in_breakable: is_in_breakable(node).unzip().0,
931            },
932        });
933        Some(make_res(kind))
934    };
935
936    let segment = match_ast! {
937        match parent {
938            ast::PathSegment(segment) => segment,
939            ast::FieldExpr(field) => {
940                return field_expr_handle(field.expr(), field.syntax());
941            },
942            ast::ExternCrate(_) => {
943                let kind = NameRefKind::ExternCrate;
944                return Some(make_res(kind));
945            },
946            ast::MethodCallExpr(method) => {
947                let receiver = find_opt_node_in_file(original_file, method.receiver());
948                let has_parens = has_parens(&method);
949                if !has_parens && let Some(res) = field_expr_handle(method.receiver(), method.syntax()) {
950                    return Some(res)
951                }
952                let kind = NameRefKind::DotAccess(DotAccess {
953                    receiver_ty: receiver.as_ref().and_then(|it| sema.type_of_expr(it)),
954                    kind: DotAccessKind::Method,
955                    receiver,
956                    ctx: DotAccessExprCtx { in_block_expr: is_in_block(method.syntax()), in_breakable: is_in_breakable(method.syntax()).unzip().0 }
957                });
958                return Some(make_res(kind));
959            },
960            _ => return None,
961        }
962    };
963
964    let path = segment.parent_path();
965    let original_path = find_node_in_file_compensated(sema, original_file, &path);
966
967    let mut path_ctx = PathCompletionCtx {
968        has_call_parens: false,
969        has_macro_bang: false,
970        qualified: Qualified::No,
971        parent: None,
972        path: path.clone(),
973        original_path,
974        kind: PathKind::Item { kind: ItemListKind::SourceFile },
975        has_type_args: false,
976        use_tree_parent: false,
977    };
978
979    let func_update_record = |syn: &SyntaxNode| {
980        if let Some(record_expr) = syn.ancestors().nth(2).and_then(ast::RecordExpr::cast) {
981            find_node_in_file_compensated(sema, original_file, &record_expr)
982        } else {
983            None
984        }
985    };
986    let prev_expr = |node: SyntaxNode| {
987        let node = match node.parent().and_then(ast::ExprStmt::cast) {
988            Some(stmt) => stmt.syntax().clone(),
989            None => node,
990        };
991        let prev_sibling = non_trivia_sibling(node.into(), Direction::Prev)?.into_node()?;
992
993        match_ast! {
994            match prev_sibling {
995                ast::ExprStmt(stmt) => stmt.expr().filter(|_| stmt.semicolon_token().is_none()),
996                ast::LetStmt(stmt) => stmt.initializer().filter(|_| stmt.semicolon_token().is_none()),
997                ast::Expr(expr) => Some(expr),
998                _ => None,
999            }
1000        }
1001    };
1002    let after_incomplete_let = |node: SyntaxNode| {
1003        prev_expr(node).and_then(|it| it.syntax().parent()).and_then(ast::LetStmt::cast)
1004    };
1005    let before_else_kw = |node: &SyntaxNode| {
1006        node.parent()
1007            .and_then(ast::ExprStmt::cast)
1008            .filter(|stmt| stmt.semicolon_token().is_none())
1009            .and_then(|stmt| non_trivia_sibling(stmt.syntax().clone().into(), Direction::Next))
1010            .and_then(NodeOrToken::into_node)
1011            .filter(|next| next.kind() == SyntaxKind::ERROR)
1012            .and_then(|next| next.first_token())
1013            .is_some_and(|token| token.kind() == SyntaxKind::ELSE_KW)
1014    };
1015
1016    // We do not want to generate path completions when we are sandwiched between an item decl signature and its body.
1017    // ex. trait Foo $0 {}
1018    // in these cases parser recovery usually kicks in for our inserted identifier, causing it
1019    // to either be parsed as an ExprStmt or a ItemRecovery, depending on whether it is in a block
1020    // expression or an item list.
1021    // The following code checks if the body is missing, if it is we either cut off the body
1022    // from the item or it was missing in the first place
1023    let inbetween_body_and_decl_check = |node: SyntaxNode| {
1024        if let Some(NodeOrToken::Node(n)) =
1025            syntax::algo::non_trivia_sibling(node.into(), syntax::Direction::Prev)
1026            && let Some(item) = ast::Item::cast(n)
1027        {
1028            let is_inbetween = match &item {
1029                ast::Item::Const(it) => it.body().is_none() && it.semicolon_token().is_none(),
1030                ast::Item::Enum(it) => it.variant_list().is_none(),
1031                ast::Item::ExternBlock(it) => it.extern_item_list().is_none(),
1032                ast::Item::Fn(it) => it.body().is_none() && it.semicolon_token().is_none(),
1033                ast::Item::Impl(it) => it.assoc_item_list().is_none(),
1034                ast::Item::Module(it) => it.item_list().is_none() && it.semicolon_token().is_none(),
1035                ast::Item::Static(it) => it.body().is_none(),
1036                ast::Item::Struct(it) => {
1037                    it.field_list().is_none() && it.semicolon_token().is_none()
1038                }
1039                ast::Item::Trait(it) => it.assoc_item_list().is_none(),
1040                ast::Item::TypeAlias(it) => it.ty().is_none() && it.semicolon_token().is_none(),
1041                ast::Item::Union(it) => it.record_field_list().is_none(),
1042                _ => false,
1043            };
1044            if is_inbetween {
1045                return Some(item);
1046            }
1047        }
1048        None
1049    };
1050
1051    let generic_arg_location = |arg: ast::GenericArg| {
1052        let mut override_location = None;
1053        let location = find_opt_node_in_file_compensated(
1054            sema,
1055            original_file,
1056            arg.syntax().parent().and_then(ast::GenericArgList::cast),
1057        )
1058        .map(|args| {
1059            let mut in_trait = None;
1060            let param = (|| {
1061                let parent = args.syntax().parent()?;
1062                let params = match_ast! {
1063                    match parent {
1064                        ast::PathSegment(segment) => {
1065                            match sema.resolve_path(&segment.parent_path().top_path())? {
1066                                hir::PathResolution::Def(def) => match def {
1067                                    hir::ModuleDef::Function(func) => {
1068                                         sema.source(func)?.value.generic_param_list()
1069                                    }
1070                                    hir::ModuleDef::Adt(adt) => {
1071                                        sema.source(adt)?.value.generic_param_list()
1072                                    }
1073                                    hir::ModuleDef::Variant(variant) => {
1074                                        sema.source(variant.parent_enum(sema.db))?.value.generic_param_list()
1075                                    }
1076                                    hir::ModuleDef::Trait(trait_) => {
1077                                        if let ast::GenericArg::AssocTypeArg(arg) = &arg {
1078                                            let arg_name = arg.name_ref()?;
1079                                            let arg_name = arg_name.text();
1080                                            for item in trait_.items_with_supertraits(sema.db) {
1081                                                match item {
1082                                                    hir::AssocItem::TypeAlias(assoc_ty) => {
1083                                                        if assoc_ty.name(sema.db).as_str() == arg_name {
1084                                                            override_location = Some(TypeLocation::AssocTypeEq);
1085                                                            return None;
1086                                                        }
1087                                                    },
1088                                                    hir::AssocItem::Const(const_) => {
1089                                                        if const_.name(sema.db)?.as_str() == arg_name {
1090                                                            override_location =  Some(TypeLocation::AssocConstEq);
1091                                                            return None;
1092                                                        }
1093                                                    },
1094                                                    _ => (),
1095                                                }
1096                                            }
1097                                            return None;
1098                                        } else {
1099                                            in_trait = Some(trait_);
1100                                            sema.source(trait_)?.value.generic_param_list()
1101                                        }
1102                                    }
1103                                    hir::ModuleDef::TypeAlias(ty_) => {
1104                                        sema.source(ty_)?.value.generic_param_list()
1105                                    }
1106                                    _ => None,
1107                                },
1108                                _ => None,
1109                            }
1110                        },
1111                        ast::MethodCallExpr(call) => {
1112                            let func = sema.resolve_method_call(&call)?;
1113                            sema.source(func)?.value.generic_param_list()
1114                        },
1115                        ast::AssocTypeArg(arg) => {
1116                            let trait_ = ast::PathSegment::cast(arg.syntax().parent()?.parent()?)?;
1117                            match sema.resolve_path(&trait_.parent_path().top_path())? {
1118                                hir::PathResolution::Def(hir::ModuleDef::Trait(trait_)) =>  {
1119                                        let arg_name = arg.name_ref()?;
1120                                        let arg_name = arg_name.text();
1121                                        let trait_items = trait_.items_with_supertraits(sema.db);
1122                                        let assoc_ty = trait_items.iter().find_map(|item| match item {
1123                                            hir::AssocItem::TypeAlias(assoc_ty) => {
1124                                                (assoc_ty.name(sema.db).as_str() == arg_name)
1125                                                    .then_some(assoc_ty)
1126                                            },
1127                                            _ => None,
1128                                        })?;
1129                                        sema.source(*assoc_ty)?.value.generic_param_list()
1130                                    }
1131                                _ => None,
1132                            }
1133                        },
1134                        _ => None,
1135                    }
1136                }?;
1137                // Determine the index of the argument in the `GenericArgList` and match it with
1138                // the corresponding parameter in the `GenericParamList`. Since lifetime parameters
1139                // are often omitted, ignore them for the purposes of matching the argument with
1140                // its parameter unless a lifetime argument is provided explicitly. That is, for
1141                // `struct S<'a, 'b, T>`, match `S::<$0>` to `T` and `S::<'a, $0, _>` to `'b`.
1142                // FIXME: This operates on the syntax tree and will produce incorrect results when
1143                // generic parameters are disabled by `#[cfg]` directives. It should operate on the
1144                // HIR, but the functionality necessary to do so is not exposed at the moment.
1145                let mut explicit_lifetime_arg = false;
1146                let arg_idx = arg
1147                    .syntax()
1148                    .siblings(Direction::Prev)
1149                    // Skip the node itself
1150                    .skip(1)
1151                    .map(|arg| if ast::LifetimeArg::can_cast(arg.kind()) { explicit_lifetime_arg = true })
1152                    .count();
1153                let param_idx = if explicit_lifetime_arg {
1154                    arg_idx
1155                } else {
1156                    // Lifetimes parameters always precede type and generic parameters,
1157                    // so offset the argument index by the total number of lifetime params
1158                    arg_idx + params.lifetime_params().count()
1159                };
1160                params.generic_params().nth(param_idx)
1161            })();
1162            (args, in_trait, param)
1163        });
1164        let (arg_list, of_trait, corresponding_param) = match location {

1592                    }

1609                    }
1610                }
1611                qualifier_ctx.vis_node = error_node.children().find_map(ast::Visibility::cast);
1612            }
1613

1619                && ![T![;], T!['}'], T!['{']].contains(&prev.kind())
1620            {
1621                // This was inferred to be an item position path, but it seems
1622                // to be part of some other broken node which leaked into an item
1623                // list
1624                return None;
1625            }
1626        }
1627    }
1628    Some((NameRefContext { nameref, kind: NameRefKind::Path(path_ctx) }, qualifier_ctx))
1629}

1631/// When writing in the middle of some code the following situation commonly occurs (`|` denotes the cursor):
1632/// ```ignore
1633/// value.method|
1634/// (1, 2, 3)
1635/// ```
1636/// Here, we want to complete the method parentheses & arguments (if the corresponding settings are on),
1637/// but the thing is parsed as a method call with parentheses. Therefore we use heuristics: if the parentheses
1638/// are on the next line, consider them non-existent.
1639fn has_parens(node: &dyn HasArgList) -> bool {
1640    let Some(arg_list) = node.arg_list() else { return false };
1641    if arg_list.l_paren_token().is_none() {
1642        return false;
1643    }
1644    let prev_siblings = iter::successors(arg_list.syntax().prev_sibling_or_token(), |it| {
1645        it.prev_sibling_or_token()
1646    });
1647    prev_siblings
1648        .take_while(|syntax| syntax.kind().is_trivia())
1649        .filter_map(|syntax| {
1650            syntax.into_token().filter(|token| token.kind() == SyntaxKind::WHITESPACE)
1651        })
1652        .all(|whitespace| !whitespace.text().contains('\n'))
1653}

1655fn pattern_context_for(
1656    sema: &Semantics<'_, RootDatabase>,
1657    original_file: &SyntaxNode,
1658    pat: ast::Pat,
1659) -> PatternContext {
1660    let mut param_ctx = None;
1661
1662    let mut missing_variants = vec![];
1663
1664    let (refutability, has_type_ascription) =
1665    pat
1666        .syntax()
1667        .ancestors()
1668        .find(|it| !ast::Pat::can_cast(it.kind()))
1669        .map_or((PatternRefutability::Irrefutable, false), |node| {
1670            let refutability = match_ast! {
1671                match node {
1672                    ast::LetStmt(let_) => return (PatternRefutability::Refutable, let_.ty().is_some()),
1673                    ast::Param(param) => {
1674                        let has_type_ascription = param.ty().is_some();
1675                        param_ctx = (|| {
1676                            let fake_param_list = param.syntax().parent().and_then(ast::ParamList::cast)?;
1677                            let param_list = find_node_in_file_compensated(sema, original_file, &fake_param_list)?;
1678                            let param_list_owner = param_list.syntax().parent()?;
1679                            let kind = match_ast! {
1680                                match param_list_owner {
1681                                    ast::ClosureExpr(closure) => ParamKind::Closure(closure),
1682                                    ast::Fn(fn_) => ParamKind::Function(fn_),
1683                                    _ => return None,
1684                                }
1685                            };
1686                            Some(ParamContext {
1687                                param_list, param, kind
1688                            })
1689                        })();
1690                        return (PatternRefutability::Irrefutable, has_type_ascription)
1691                    },
1692                    ast::MatchArm(match_arm) => {
1693                       let missing_variants_opt = match_arm
1694                            .syntax()
1695                            .parent()
1696                            .and_then(ast::MatchArmList::cast)
1697                            .and_then(|match_arm_list| {
1698                                match_arm_list
1699                                .syntax()
1700                                .parent()
1701                                .and_then(ast::MatchExpr::cast)
1702                                .and_then(|match_expr| {
1703                                    let expr_opt = find_opt_node_in_file(original_file, match_expr.expr());
1704
1705                                    expr_opt.and_then(|expr| {
1706                                        sema.type_of_expr(&expr)?
1707                                        .adjusted()
1708                                        .autoderef(sema.db)
1709                                        .find_map(|ty| match ty.as_adt() {
1710                                            Some(hir::Adt::Enum(e)) => Some(e),
1711                                            _ => None,
1712                                        }).map(|enum_| enum_.variants(sema.db))
1713                                    })
1714                                }).map(|variants| variants.iter().filter_map(|variant| {
1715                                        let variant_name = variant.name(sema.db);
1716
1717                                        let variant_already_present = match_arm_list.arms().any(|arm| {
1718                                            arm.pat().and_then(|pat| {
1719                                                let pat_already_present = pat.syntax().to_string().contains(variant_name.as_str());
1720                                                pat_already_present.then_some(pat_already_present)
1721                                            }).is_some()
1722                                        });
1723
1724                                        (!variant_already_present).then_some(*variant)
1725                                    }).collect::<Vec<Variant>>())
1726                        });
1727
1728                        if let Some(missing_variants_) = missing_variants_opt {
1729                            missing_variants = missing_variants_;
1730                        };
1731
1732                        PatternRefutability::Refutable
1733                    },
1734                    ast::LetExpr(_) => PatternRefutability::Refutable,
1735                    ast::ForExpr(_) => PatternRefutability::Irrefutable,
1736                    _ => PatternRefutability::Irrefutable,
1737                }
1738            };
1739            (refutability, false)
1740        });
1741    let (ref_token, mut_token) = match &pat {
1742        ast::Pat::IdentPat(it) => (it.ref_token(), it.mut_token()),
1743        _ => (None, None),
1744    };
1745
1746    // Only suggest name in let-stmt or fn param
1747    let should_suggest_name = matches!(
1748            &pat,
1749            ast::Pat::IdentPat(it)
1750                if it.syntax()
1751                .parent().is_some_and(|node| {
1752                    let kind = node.kind();
1753                    ast::LetStmt::can_cast(kind) || ast::Param::can_cast(kind)
1754                })
1755    );
1756
1757    PatternContext {
1758        refutability,
1759        param_ctx,
1760        has_type_ascription,
1761        should_suggest_name,
1762        after_if_expr: is_after_if_expr(pat.syntax().clone()),
1763        parent_pat: pat.syntax().parent().and_then(ast::Pat::cast),
1764        mut_token,
1765        ref_token,
1766        record_pat: None,
1767        impl_or_trait: fetch_immediate_impl_or_trait(sema, original_file, pat.syntax()),
1768        missing_variants,
1769    }
1770}

1772fn fetch_immediate_impl_or_trait(
1773    sema: &Semantics<'_, RootDatabase>,
1774    original_file: &SyntaxNode,
1775    node: &SyntaxNode,
1776) -> Option<Either<ast::Impl, ast::Trait>> {
1777    let mut ancestors = ancestors_in_file_compensated(sema, original_file, node)?
1778        .filter_map(ast::Item::cast)
1779        .filter(|it| !matches!(it, ast::Item::MacroCall(_)));
1780
1781    match ancestors.next()? {
1782        ast::Item::Const(_) | ast::Item::Fn(_) | ast::Item::TypeAlias(_) => (),
1783        ast::Item::Impl(it) => return Some(Either::Left(it)),
1784        ast::Item::Trait(it) => return Some(Either::Right(it)),
1785        _ => return None,
1786    }
1787    match ancestors.next()? {
1788        ast::Item::Impl(it) => Some(Either::Left(it)),
1789        ast::Item::Trait(it) => Some(Either::Right(it)),
1790        _ => None,
1791    }
1792}

1800/// Attempts to find `node` inside `syntax` via `node`'s text range.
1801/// If the fake identifier has been inserted after this node or inside of this node use the `_compensated` version instead.
1802fn find_node_in_file<N: AstNode>(syntax: &SyntaxNode, node: &N) -> Option<N> {
1803    let syntax_range = syntax.text_range();
1804    let range = node.syntax().text_range();
1805    let intersection = range.intersect(syntax_range)?;
1806    syntax.covering_element(intersection).ancestors().find_map(N::cast)
1807}

1810/// for the offset introduced by the fake ident.
1811/// This is wrong if `node` comes before the insertion point! Use `find_node_in_file` instead.
1812fn find_node_in_file_compensated<N: AstNode>(
1813    sema: &Semantics<'_, RootDatabase>,
1814    in_file: &SyntaxNode,
1815    node: &N,
1816) -> Option<N> {

1820fn ancestors_in_file_compensated<'sema>(
1821    sema: &'sema Semantics<'_, RootDatabase>,
1822    in_file: &SyntaxNode,
1823    node: &SyntaxNode,
1824) -> Option<impl Iterator<Item = SyntaxNode> + 'sema> {
1825    let syntax_range = in_file.text_range();
1826    let range = node.text_range();
1827    let end = range.end().checked_sub(TextSize::try_from(COMPLETION_MARKER.len()).ok()?)?;
1828    if end < range.start() {
1829        return None;
1830    }
1831    let range = TextRange::new(range.start(), end);
1832    // our inserted ident could cause `range` to go outside of the original syntax, so cap it
1833    let intersection = range.intersect(syntax_range)?;
1834    let node = match in_file.covering_element(intersection) {
1835        NodeOrToken::Node(node) => node,
1836        NodeOrToken::Token(tok) => tok.parent()?,
1837    };
1838    Some(sema.ancestors_with_macros(node))
1839}

1883fn is_in_breakable(node: &SyntaxNode) -> Option<(BreakableKind, SyntaxNode)> {
1884    node.ancestors()
1885        .take_while(|it| it.kind() != SyntaxKind::FN && it.kind() != SyntaxKind::CLOSURE_EXPR)
1886        .find_map(|it| {
1887            let (breakable, loop_body) = match_ast! {
1888                match it {
1889                    ast::ForExpr(it) => (BreakableKind::For, it.loop_body()?),
1890                    ast::WhileExpr(it) => (BreakableKind::While, it.loop_body()?),
1891                    ast::LoopExpr(it) => (BreakableKind::Loop, it.loop_body()?),
1892                    ast::BlockExpr(it) => return it.label().map(|_| (BreakableKind::Block, it.syntax().clone())),
1893                    _ => return None,
1894                }
1895            };
1896            loop_body.syntax().text_range().contains_range(node.text_range())
1897                .then_some((breakable, it))
1898        })
1899}

1910/// Similar to `has_parens`, heuristic sensing incomplete statement before ambiguous `Expr`
1911///
1912/// Heuristic:
1913///
1914/// If the `PathExpr` is left part of the `Expr` and there is a newline after the `PathExpr`,
1915/// it is considered that the `PathExpr` is not part of the `Expr`.
1916fn has_in_newline_expr_first(node: &SyntaxNode) -> bool {
1917    if ast::PathExpr::can_cast(node.kind())
1918        && let Some(NodeOrToken::Token(next)) = node.next_sibling_or_token()
1919        && next.kind() == SyntaxKind::WHITESPACE
1920        && next.text().contains('\n')
1921        && let Some(stmt_like) = node
1922            .ancestors()
1923            .take_while(|it| it.text_range().start() == node.text_range().start())
1924            .filter_map(Either::<ast::ExprStmt, ast::Expr>::cast)
1925            .last()
1926    {
1927        stmt_like.syntax().parent().and_then(ast::StmtList::cast).is_some()
1928    } else {
1929        false
1930    }
1931}

1933fn is_after_if_expr(node: SyntaxNode) -> bool {
1934    let node = match node.parent().and_then(Either::<ast::ExprStmt, ast::MatchArm>::cast) {
1935        Some(stmt) => stmt.syntax().clone(),
1936        None => node,
1937    };
1938    let prev_sibling =
1939        non_trivia_sibling(node.into(), Direction::Prev).and_then(NodeOrToken::into_node);
1940    iter::successors(prev_sibling, |it| it.last_child_or_token()?.into_node())
1941        .find_map(ast::IfExpr::cast)
1942        .is_some()
1943}

1945fn next_non_trivia_token(e: impl Into<SyntaxElement>) -> Option<SyntaxToken> {
1946    let mut token = match e.into() {
1947        SyntaxElement::Node(n) => n.last_token()?,
1948        SyntaxElement::Token(t) => t,
1949    }
1950    .next_token();
1951    while let Some(inner) = token {
1952        if !inner.kind().is_trivia() {
1953            return Some(inner);
1954        } else {
1955            token = inner.next_token();
1956        }
1957    }
1958    None
1959}

1961fn next_non_trivia_sibling(ele: SyntaxElement) -> Option<SyntaxElement> {
1962    let mut e = ele.next_sibling_or_token();

1965            return Some(inner);
1966        } else {
1967            e = inner.next_sibling_or_token();
1968        }
1969    }
1970    None
