55pub fn load_workspace_at(
56    root: &Path,
57    cargo_config: &CargoConfig,
58    load_config: &LoadCargoConfig,
59    progress: &(dyn Fn(String) + Sync),
60) -> anyhow::Result<(RootDatabase, vfs::Vfs, Option<ProcMacroClient>)> {
61    let root = AbsPathBuf::assert_utf8(std::env::current_dir()?.join(root));
62    let root = ProjectManifest::discover_single(&root)?;
63    let manifest_path = root.manifest_path().clone();
64    let mut workspace = ProjectWorkspace::load(root, cargo_config, progress)?;
65
66    if load_config.load_out_dirs_from_check {
67        let build_scripts = workspace.run_build_scripts(cargo_config, progress)?;
68        if let Some(error) = build_scripts.error() {
69            tracing::debug!(
70                "Errors occurred while running build scripts for {}: {}",
71                manifest_path,
72                error
73            );
74        }
75        workspace.set_build_scripts(build_scripts)
76    }
77
78    load_workspace(workspace, &cargo_config.extra_env, load_config)
79}

81pub fn load_workspace(
82    ws: ProjectWorkspace,
83    extra_env: &FxHashMap<String, Option<String>>,
84    load_config: &LoadCargoConfig,
85) -> anyhow::Result<(RootDatabase, vfs::Vfs, Option<ProcMacroClient>)> {
86    let lru_cap = std::env::var("RA_LRU_CAP").ok().and_then(|it| it.parse::<u16>().ok());
87    let mut db = RootDatabase::new(lru_cap);
88
89    let (vfs, proc_macro_server) = load_workspace_into_db(ws, extra_env, load_config, &mut db)?;
90
91    Ok((db, vfs, proc_macro_server))
92}

94// This variant of `load_workspace` allows deferring the loading of rust-analyzer
95// into an existing database, which is useful in certain third-party scenarios,
96// now that `salsa` supports extending foreign databases (e.g. `RootDatabase`).
97pub fn load_workspace_into_db(
98    ws: ProjectWorkspace,
99    extra_env: &FxHashMap<String, Option<String>>,
100    load_config: &LoadCargoConfig,
101    db: &mut RootDatabase,
102) -> anyhow::Result<(vfs::Vfs, Option<ProcMacroClient>)> {
103    let (sender, receiver) = unbounded();
104    let mut vfs = vfs::Vfs::default();
105    let mut loader = {
106        let loader = vfs_notify::NotifyHandle::spawn(sender);
107        Box::new(loader)
108    };
109
110    tracing::debug!(?load_config, "LoadCargoConfig");
111    let proc_macro_server = match &load_config.with_proc_macro_server {
112        ProcMacroServerChoice::Sysroot => ws.find_sysroot_proc_macro_srv().map(|it| {
113            it.and_then(|it| {
114                ProcMacroClient::spawn(
115                    &it,
116                    extra_env,
117                    ws.toolchain.as_ref(),
118                    load_config.proc_macro_processes,
119                )
120                .map_err(Into::into)
121            })
122            .map_err(|e| ProcMacroLoadingError::ProcMacroSrvError(e.to_string().into_boxed_str()))
123        }),
124        ProcMacroServerChoice::Explicit(path) => Some(
125            ProcMacroClient::spawn(
126                path,
127                extra_env,
128                ws.toolchain.as_ref(),
129                load_config.proc_macro_processes,
130            )
131            .map_err(|e| ProcMacroLoadingError::ProcMacroSrvError(e.to_string().into_boxed_str())),
132        ),
133        ProcMacroServerChoice::None => Some(Err(ProcMacroLoadingError::Disabled)),
134    };
135    match &proc_macro_server {
136        Some(Ok(server)) => {
137            tracing::info!(manifest=%ws.manifest_or_root(), path=%server.server_path(), "Proc-macro server started")
138        }
139        Some(Err(e)) => {
140            tracing::info!(manifest=%ws.manifest_or_root(), %e, "Failed to start proc-macro server")
141        }
142        None => {
143            tracing::info!(manifest=%ws.manifest_or_root(), "No proc-macro server started")
144        }
145    }
146
147    let (crate_graph, proc_macros) = ws.to_crate_graph(
148        &mut |path: &AbsPath| {
149            let contents = loader.load_sync(path);
150            let path = vfs::VfsPath::from(path.to_path_buf());
151            vfs.set_file_contents(path.clone(), contents);
152            vfs.file_id(&path).and_then(|(file_id, excluded)| {
153                (excluded == vfs::FileExcluded::No).then_some(file_id)
154            })
155        },
156        extra_env,
157    );
158    let proc_macros = {
159        let proc_macro_server = match &proc_macro_server {
160            Some(Ok(it)) => Ok(it),
161            Some(Err(e)) => {
162                Err(ProcMacroLoadingError::ProcMacroSrvError(e.to_string().into_boxed_str()))
163            }
164            None => Err(ProcMacroLoadingError::ProcMacroSrvError(
165                "proc-macro-srv is not running, workspace is missing a sysroot".into(),
166            )),
167        };
168        proc_macros
169            .into_iter()
170            .map(|(crate_id, path)| {
171                (
172                    crate_id,
173                    path.map_or_else(Err, |(_, path)| {
174                        proc_macro_server.as_ref().map_err(Clone::clone).and_then(
175                            |proc_macro_server| load_proc_macro(proc_macro_server, &path, &[]),
176                        )
177                    }),
178                )
179            })
180            .collect()
181    };
182
183    let project_folders = ProjectFolders::new(std::slice::from_ref(&ws), &[], None);
184    loader.set_config(vfs::loader::Config {
185        load: project_folders.load,
186        watch: vec![],
187        version: 0,
188    });
189
190    load_crate_graph_into_db(
191        crate_graph,
192        proc_macros,
193        project_folders.source_root_config,
194        &mut vfs,
195        &receiver,
196        db,
197    );
198
199    if load_config.prefill_caches {
200        prime_caches::parallel_prime_caches(db, 1, &|_| ());
201    }
202
203    Ok((vfs, proc_macro_server.and_then(Result::ok)))
204}

206#[derive(Default)]
207pub struct ProjectFolders {
208    pub load: Vec<vfs::loader::Entry>,
209    pub watch: Vec<usize>,
210    pub source_root_config: SourceRootConfig,
211}
212
213impl ProjectFolders {
214    pub fn new(
215        workspaces: &[ProjectWorkspace],
216        global_excludes: &[AbsPathBuf],
217        user_config_dir_path: Option<&AbsPath>,
218    ) -> ProjectFolders {
219        let mut res = ProjectFolders::default();
220        let mut fsc = FileSetConfig::builder();
221        let mut local_filesets = vec![];
222
223        // Dedup source roots
224        // Depending on the project setup, we can have duplicated source roots, or for example in
225        // the case of the rustc workspace, we can end up with two source roots that are almost the
226        // same but not quite, like:
227        // PackageRoot { is_local: false, include: [AbsPathBuf(".../rust/src/tools/miri/cargo-miri")], exclude: [] }
228        // PackageRoot {
229        //     is_local: true,
230        //     include: [AbsPathBuf(".../rust/src/tools/miri/cargo-miri"), AbsPathBuf(".../rust/build/x86_64-pc-windows-msvc/stage0-tools/x86_64-pc-windows-msvc/release/build/cargo-miri-85801cd3d2d1dae4/out")],
231        //     exclude: [AbsPathBuf(".../rust/src/tools/miri/cargo-miri/.git"), AbsPathBuf(".../rust/src/tools/miri/cargo-miri/target")]
232        // }
233        //
234        // The first one comes from the explicit rustc workspace which points to the rustc workspace itself
235        // The second comes from the rustc workspace that we load as the actual project workspace
236        // These `is_local` differing in this kind of way gives us problems, especially when trying to filter diagnostics as we don't report diagnostics for external libraries.
237        // So we need to deduplicate these, usually it would be enough to deduplicate by `include`, but as the rustc example shows here that doesn't work,
238        // so we need to also coalesce the includes if they overlap.
239
240        let mut roots: Vec<_> = workspaces
241            .iter()
242            .flat_map(|ws| ws.to_roots())
243            .update(|root| root.include.sort())
244            .sorted_by(|a, b| a.include.cmp(&b.include))
245            .collect();
246
247        // map that tracks indices of overlapping roots
248        let mut overlap_map = FxHashMap::<_, Vec<_>>::default();
249        let mut done = false;
250
251        while !mem::replace(&mut done, true) {
252            // maps include paths to indices of the corresponding root
253            let mut include_to_idx = FxHashMap::default();
254            // Find and note down the indices of overlapping roots
255            for (idx, root) in roots.iter().enumerate().filter(|(_, it)| !it.include.is_empty()) {
256                for include in &root.include {
257                    match include_to_idx.entry(include) {
258                        Entry::Occupied(e) => {
259                            overlap_map.entry(*e.get()).or_default().push(idx);
260                        }
261                        Entry::Vacant(e) => {
262                            e.insert(idx);
263                        }
264                    }
265                }
266            }
267            for (k, v) in overlap_map.drain() {
268                done = false;
269                for v in v {
270                    let r = mem::replace(
271                        &mut roots[v],
272                        PackageRoot { is_local: false, include: vec![], exclude: vec![] },
273                    );
274                    roots[k].is_local |= r.is_local;
275                    roots[k].include.extend(r.include);
276                    roots[k].exclude.extend(r.exclude);
277                }
278                roots[k].include.sort();
279                roots[k].exclude.sort();
280                roots[k].include.dedup();
281                roots[k].exclude.dedup();
282            }
283        }
284
285        for root in roots.into_iter().filter(|it| !it.include.is_empty()) {
286            let file_set_roots: Vec<VfsPath> =
287                root.include.iter().cloned().map(VfsPath::from).collect();
288
289            let entry = {
290                let mut dirs = vfs::loader::Directories::default();
291                dirs.extensions.push("rs".into());
292                dirs.extensions.push("toml".into());
293                dirs.include.extend(root.include);
294                dirs.exclude.extend(root.exclude);
295                for excl in global_excludes {
296                    if dirs
297                        .include
298                        .iter()
299                        .any(|incl| incl.starts_with(excl) || excl.starts_with(incl))
300                    {
301                        dirs.exclude.push(excl.clone());
302                    }
303                }
304
305                vfs::loader::Entry::Directories(dirs)
306            };
307
308            if root.is_local {
309                res.watch.push(res.load.len());
310            }
311            res.load.push(entry);
312
313            if root.is_local {
314                local_filesets.push(fsc.len() as u64);
315            }
316            fsc.add_file_set(file_set_roots)
317        }
318
319        for ws in workspaces.iter() {
320            let mut file_set_roots: Vec<VfsPath> = vec![];
321            let mut entries = vec![];
322
323            for buildfile in ws.buildfiles() {
324                file_set_roots.push(VfsPath::from(buildfile.to_owned()));
325                entries.push(buildfile.to_owned());
326            }
327
328            if !file_set_roots.is_empty() {
329                let entry = vfs::loader::Entry::Files(entries);
330                res.watch.push(res.load.len());
331                res.load.push(entry);
332                local_filesets.push(fsc.len() as u64);
333                fsc.add_file_set(file_set_roots)
334            }
335        }
336
337        if let Some(user_config_path) = user_config_dir_path {
338            let ratoml_path = {
339                let mut p = user_config_path.to_path_buf();
340                p.push("rust-analyzer.toml");
341                p
342            };
343
344            let file_set_roots = vec![VfsPath::from(ratoml_path.to_owned())];
345            let entry = vfs::loader::Entry::Files(vec![ratoml_path]);
346
347            res.watch.push(res.load.len());
348            res.load.push(entry);
349            local_filesets.push(fsc.len() as u64);
350            fsc.add_file_set(file_set_roots)
351        }
352
353        let fsc = fsc.build();
354        res.source_root_config = SourceRootConfig { fsc, local_filesets };
355
356        res
357    }
358}

360#[derive(Default, Debug)]
361pub struct SourceRootConfig {
362    pub fsc: FileSetConfig,
363    pub local_filesets: Vec<u64>,
364}
365
366impl SourceRootConfig {
367    pub fn partition(&self, vfs: &vfs::Vfs) -> Vec<SourceRoot> {
368        self.fsc
369            .partition(vfs)
370            .into_iter()
371            .enumerate()
372            .map(|(idx, file_set)| {
373                let is_local = self.local_filesets.contains(&(idx as u64));
374                if is_local {
375                    SourceRoot::new_local(file_set)
376                } else {
377                    SourceRoot::new_library(file_set)
378                }
379            })
380            .collect()
381    }
382
383    /// Maps local source roots to their parent source roots by bytewise comparing of root paths .
384    /// If a `SourceRoot` doesn't have a parent and is local then it is not contained in this mapping but it can be asserted that it is a root `SourceRoot`.
385    pub fn source_root_parent_map(&self) -> FxHashMap<SourceRootId, SourceRootId> {
386        let roots = self.fsc.roots();
387
388        let mut map = FxHashMap::default();
389
390        // See https://github.com/rust-lang/rust-analyzer/issues/17409
391        //
392        // We can view the connections between roots as a graph. The problem is
393        // that this graph may contain cycles, so when adding edges, it is necessary
394        // to check whether it will lead to a cycle.
395        //
396        // Since we ensure that each node has at most one outgoing edge (because
397        // each SourceRoot can have only one parent), we can use a disjoint-set to
398        // maintain the connectivity between nodes. If an edgeâ€™s two nodes belong
399        // to the same set, they are already connected.
400        let mut dsu = FxHashMap::default();
401        fn find_parent(dsu: &mut FxHashMap<u64, u64>, id: u64) -> u64 {
402            if let Some(&parent) = dsu.get(&id) {
403                let parent = find_parent(dsu, parent);
404                dsu.insert(id, parent);
405                parent
406            } else {
407                id
408            }
409        }
410
411        for (idx, (root, root_id)) in roots.iter().enumerate() {
412            if !self.local_filesets.contains(root_id)
413                || map.contains_key(&SourceRootId(*root_id as u32))
414            {
415                continue;
416            }
417
418            for (root2, root2_id) in roots[..idx].iter().rev() {
419                if self.local_filesets.contains(root2_id)
420                    && root_id != root2_id
421                    && root.starts_with(root2)
422                {
423                    // check if the edge will create a cycle
424                    if find_parent(&mut dsu, *root_id) != find_parent(&mut dsu, *root2_id) {
425                        map.insert(SourceRootId(*root_id as u32), SourceRootId(*root2_id as u32));
426                        dsu.insert(*root_id, *root2_id);
427                    }
428
429                    break;
430                }
431            }
432        }
433
434        map
435    }
436}

438/// Load the proc-macros for the given lib path, disabling all expanders whose names are in `ignored_macros`.
439pub fn load_proc_macro(
440    server: &ProcMacroClient,
441    path: &AbsPath,
442    ignored_macros: &[Box<str>],
443) -> ProcMacroLoadResult {
444    let res: Result<Vec<_>, _> = (|| {
445        let dylib = MacroDylib::new(path.to_path_buf());
446        let vec = server.load_dylib(dylib).map_err(|e| {
447            ProcMacroLoadingError::ProcMacroSrvError(format!("{e}").into_boxed_str())
448        })?;
449        if vec.is_empty() {
450            return Err(ProcMacroLoadingError::NoProcMacros);
451        }
452        Ok(vec
453            .into_iter()
454            .map(|expander| expander_to_proc_macro(expander, ignored_macros))
455            .collect())
456    })();
457    match res {
458        Ok(proc_macros) => {
459            tracing::info!(
460                "Loaded proc-macros for {path}: {:?}",
461                proc_macros.iter().map(|it| it.name.clone()).collect::<Vec<_>>()
462            );
463            Ok(proc_macros)
464        }
465        Err(e) => {
466            tracing::warn!("proc-macro loading for {path} failed: {e}");
467            Err(e)
468        }
469    }
470}

472fn load_crate_graph_into_db(
473    crate_graph: CrateGraphBuilder,
474    proc_macros: ProcMacrosBuilder,
475    source_root_config: SourceRootConfig,
476    vfs: &mut vfs::Vfs,
477    receiver: &Receiver<vfs::loader::Message>,
478    db: &mut RootDatabase,
479) {
480    let mut analysis_change = ChangeWithProcMacros::default();
481
482    db.enable_proc_attr_macros();
483
484    // wait until Vfs has loaded all roots
485    for task in receiver {
486        match task {
487            vfs::loader::Message::Progress { n_done, .. } => {
488                if n_done == LoadingProgress::Finished {
489                    break;
490                }
491            }
492            vfs::loader::Message::Loaded { files } | vfs::loader::Message::Changed { files } => {
493                let _p =
494                    tracing::info_span!("load_cargo::load_crate_craph/LoadedChanged").entered();
495                for (path, contents) in files {
496                    vfs.set_file_contents(path.into(), contents);
497                }
498            }
499        }
500    }
501    let changes = vfs.take_changes();
502    for (_, file) in changes {
503        if let vfs::Change::Create(v, _) | vfs::Change::Modify(v, _) = file.change
504            && let Ok(text) = String::from_utf8(v)
505        {
506            analysis_change.change_file(file.file_id, Some(text))
507        }
508    }
509    let source_roots = source_root_config.partition(vfs);
510    analysis_change.set_roots(source_roots);
511
512    analysis_change.set_crate_graph(crate_graph);
513    analysis_change.set_proc_macros(proc_macros);
514
515    db.apply_change(analysis_change);
516}

518fn expander_to_proc_macro(
519    expander: proc_macro_api::ProcMacro,
520    ignored_macros: &[Box<str>],
521) -> ProcMacro {
522    let name = expander.name();
523    let kind = match expander.kind() {
524        proc_macro_api::ProcMacroKind::CustomDerive => ProcMacroKind::CustomDerive,
525        proc_macro_api::ProcMacroKind::Bang => ProcMacroKind::Bang,
526        proc_macro_api::ProcMacroKind::Attr => ProcMacroKind::Attr,
527    };
528    let disabled = ignored_macros.iter().any(|replace| **replace == *name);
529    ProcMacro {
530        name: intern::Symbol::intern(name),
531        kind,
532        expander: sync::Arc::new(Expander(expander)),
533        disabled,
534    }
535}

537#[derive(Debug, PartialEq, Eq)]
538struct Expander(proc_macro_api::ProcMacro);
539
540impl ProcMacroExpander for Expander {
541    fn expand(
542        &self,
543        db: &dyn ExpandDatabase,
544        subtree: &tt::TopSubtree,
545        attrs: Option<&tt::TopSubtree>,
546        env: &Env,
547        def_site: Span,
548        call_site: Span,
549        mixed_site: Span,
550        current_dir: String,
551    ) -> Result<tt::TopSubtree, ProcMacroExpansionError> {
552        let cb = |req| match req {
553            SubRequest::LocalFilePath { file_id } => {
554                let file_id = FileId::from_raw(file_id);
555                let source_root_id = db.file_source_root(file_id).source_root_id(db);
556                let source_root = db.source_root(source_root_id).source_root(db);
557                let name = source_root
558                    .path_for_file(&file_id)
559                    .and_then(|path| path.as_path())
560                    .map(|path| path.to_string());
561
562                Ok(SubResponse::LocalFilePathResult { name })
563            }
564            // Not incremental: requires full file text.
565            SubRequest::SourceText { file_id, ast_id, start, end } => {
566                let range = resolve_sub_span(
567                    db,
568                    file_id,
569                    ast_id,
570                    TextRange::new(TextSize::from(start), TextSize::from(end)),
571                );
572                let source = db.file_text(range.file_id.file_id(db)).text(db);
573                let text = source
574                    .get(usize::from(range.range.start())..usize::from(range.range.end()))
575                    .map(ToOwned::to_owned);
576
577                Ok(SubResponse::SourceTextResult { text })
578            }
579            // Not incremental: requires building line index.
580            SubRequest::LineColumn { file_id, ast_id, offset } => {
581                let range =
582                    resolve_sub_span(db, file_id, ast_id, TextRange::empty(TextSize::from(offset)));
583                let source = db.file_text(range.file_id.file_id(db)).text(db);
584                let line_index = ide_db::line_index::LineIndex::new(source);
585                let (line, column) = line_index
586                    .try_line_col(range.range.start())
587                    .map(|lc| (lc.line + 1, lc.col + 1))
588                    .unwrap_or((1, 1));
589                // proc_macro::Span line/column are 1-based
590                Ok(SubResponse::LineColumnResult { line, column })
591            }
592            SubRequest::FilePath { file_id } => {
593                let file_id = FileId::from_raw(file_id);
594                let source_root_id = db.file_source_root(file_id).source_root_id(db);
595                let source_root = db.source_root(source_root_id).source_root(db);
596                let name = source_root
597                    .path_for_file(&file_id)
598                    .and_then(|path| path.as_path())
599                    .map(|path| path.to_string())
600                    .unwrap_or_default();
601
602                Ok(SubResponse::FilePathResult { name })
603            }
604            // Not incremental: requires global span resolution.
605            SubRequest::ByteRange { file_id, ast_id, start, end } => {
606                let range = resolve_sub_span(
607                    db,
608                    file_id,
609                    ast_id,
610                    TextRange::new(TextSize::from(start), TextSize::from(end)),
611                );
612
613                Ok(SubResponse::ByteRangeResult { range: range.range.into() })
614            }
615            SubRequest::SpanSource { file_id, ast_id, start, end, ctx } => {
616                let span = Span {
617                    range: TextRange::new(TextSize::from(start), TextSize::from(end)),
618                    anchor: SpanAnchor {
619                        file_id: span::EditionedFileId::from_raw(file_id),
620                        ast_id: span::ErasedFileAstId::from_raw(ast_id),
621                    },
622                    // SAFETY: We only receive spans from the server. If someone mess up the communication UB can happen,
623                    // but that will be their problem.
624                    ctx: unsafe { SyntaxContext::from_u32(ctx) },
625                };
626
627                let mut current_span = span;
628                let mut current_ctx = span.ctx;
629
630                while let Some(macro_call_id) = current_ctx.outer_expn(db) {
631                    let macro_call_loc = db.lookup_intern_macro_call(macro_call_id.into());
632
633                    let call_site_file = macro_call_loc.kind.file_id();
634
635                    let resolved = db.resolve_span(current_span);
636
637                    current_ctx = macro_call_loc.ctxt;
638                    current_span = Span {
639                        range: resolved.range,
640                        anchor: SpanAnchor {
641                            file_id: resolved.file_id.editioned_file_id(db),
642                            ast_id: span::ROOT_ERASED_FILE_AST_ID,
643                        },
644                        ctx: current_ctx,
645                    };
646
647                    if call_site_file.file_id().is_some() {
648                        break;
649                    }
650                }
651
652                let resolved = db.resolve_span(current_span);
653
654                Ok(SubResponse::SpanSourceResult {
655                    file_id: resolved.file_id.editioned_file_id(db).as_u32(),
656                    ast_id: span::ROOT_ERASED_FILE_AST_ID.into_raw(),
657                    start: u32::from(resolved.range.start()),
658                    end: u32::from(resolved.range.end()),
659                    ctx: current_span.ctx.into_u32(),
660                })
661            }
662            SubRequest::SpanParent { file_id, ast_id, start, end, ctx } => {
663                let span = Span {
664                    range: TextRange::new(TextSize::from(start), TextSize::from(end)),
665                    anchor: SpanAnchor {
666                        file_id: span::EditionedFileId::from_raw(file_id),
667                        ast_id: span::ErasedFileAstId::from_raw(ast_id),
668                    },
669                    // SAFETY: We only receive spans from the server. If someone mess up the communication UB can happen,
670                    // but that will be their problem.
671                    ctx: unsafe { SyntaxContext::from_u32(ctx) },
672                };
673
674                if let Some(macro_call_id) = span.ctx.outer_expn(db) {
675                    let macro_call_loc = db.lookup_intern_macro_call(macro_call_id.into());
676
677                    let call_site_file = macro_call_loc.kind.file_id();
678                    let call_site_ast_id = macro_call_loc.kind.erased_ast_id();
679
680                    if let Some(editioned_file_id) = call_site_file.file_id() {
681                        let range = db
682                            .ast_id_map(editioned_file_id.into())
683                            .get_erased(call_site_ast_id)
684                            .text_range();
685
686                        let parent_span = Some(ParentSpan {
687                            file_id: editioned_file_id.editioned_file_id(db).as_u32(),
688                            ast_id: span::ROOT_ERASED_FILE_AST_ID.into_raw(),
689                            start: u32::from(range.start()),
690                            end: u32::from(range.end()),
691                            ctx: macro_call_loc.ctxt.into_u32(),
692                        });
693
694                        return Ok(SubResponse::SpanParentResult { parent_span });
695                    }
696                }
697
698                Ok(SubResponse::SpanParentResult { parent_span: None })
699            }
700        };
701        match self.0.expand(
702            subtree.view(),
703            attrs.map(|attrs| attrs.view()),
704            env.clone().into(),
705            def_site,
706            call_site,
707            mixed_site,
708            current_dir,
709            Some(&cb),
710        ) {
711            Ok(Ok(subtree)) => Ok(subtree),
712            Ok(Err(err)) => Err(ProcMacroExpansionError::Panic(err)),
713            Err(err) => Err(ProcMacroExpansionError::System(err.to_string())),
714        }
715    }
716
717    fn eq_dyn(&self, other: &dyn ProcMacroExpander) -> bool {
718        (other as &dyn Any).downcast_ref::<Self>() == Some(self)
719    }
720}
721
722fn resolve_sub_span(
723    db: &dyn ExpandDatabase,
724    file_id: u32,
725    ast_id: u32,
726    range: TextRange,
727) -> hir_expand::FileRange {
728    let ast_id = span::ErasedFileAstId::from_raw(ast_id);
729    let editioned_file_id = span::EditionedFileId::from_raw(file_id);
730    let span = Span {
731        range,
732        anchor: SpanAnchor { file_id: editioned_file_id, ast_id },
733        ctx: SyntaxContext::root(editioned_file_id.edition()),
734    };
735    db.resolve_span(span)
736}
