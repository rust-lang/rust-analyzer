1//! Convert macro-by-example tokens which are specific to macro expansion into a
2//! format that works for our parser.
3
4use rustc_hash::FxHashMap;
5use span::{Edition, SyntaxContext};
6use syntax::{SyntaxKind, SyntaxKind::*, T};
7
8pub fn to_parser_input(
9    buffer: tt::TokenTreesView<'_>,
10    span_to_edition: &mut dyn FnMut(SyntaxContext) -> Edition,
11) -> parser::Input {
12    let mut res = parser::Input::with_capacity(buffer.len());
13
14    let mut current = buffer.cursor();
15    let mut syntax_context_to_edition_cache = FxHashMap::default();
16    let mut ctx_edition =
17        |ctx| *syntax_context_to_edition_cache.entry(ctx).or_insert_with(|| span_to_edition(ctx));
18
19    while !current.eof() {
20        let tt = current.token_tree();
21
22        // Check if it is lifetime
23        if let Some(tt::TokenTree::Leaf(tt::Leaf::Punct(punct))) = tt
24            && punct.char == '\''
25        {
26            current.bump();
27            match current.token_tree() {
28                Some(tt::TokenTree::Leaf(tt::Leaf::Ident(ident))) => {
29                    res.push(LIFETIME_IDENT, ctx_edition(ident.span.ctx));
30                    current.bump();
31                    continue;
32                }
33                _ => panic!("Next token must be ident"),
34            }
35        }
36
37        match tt {
38            Some(tt::TokenTree::Leaf(leaf)) => {
39                match leaf {
40                    tt::Leaf::Literal(lit) => {
41                        let kind = match lit.kind {
42                            tt::LitKind::Byte => SyntaxKind::BYTE,
43                            tt::LitKind::Char => SyntaxKind::CHAR,
44                            tt::LitKind::Integer => SyntaxKind::INT_NUMBER,
45                            tt::LitKind::Float => SyntaxKind::FLOAT_NUMBER,
46                            tt::LitKind::Str | tt::LitKind::StrRaw(_) => SyntaxKind::STRING,
47                            tt::LitKind::ByteStr | tt::LitKind::ByteStrRaw(_) => {
48                                SyntaxKind::BYTE_STRING
49                            }
50                            tt::LitKind::CStr | tt::LitKind::CStrRaw(_) => SyntaxKind::C_STRING,
51                            tt::LitKind::Err(_) => SyntaxKind::ERROR,
52                        };
53                        res.push(kind, ctx_edition(lit.span.ctx));
54
55                        if kind == FLOAT_NUMBER && !lit.text().ends_with('.') {
56                            // Tag the token as joint if it is float with a fractional part
57                            // we use this jointness to inform the parser about what token split
58                            // event to emit when we encounter a float literal in a field access
59                            res.was_joint();
60                        }
61                    }
62                    tt::Leaf::Ident(ident) => {
63                        let edition = ctx_edition(ident.span.ctx);
64                        match ident.sym.as_str() {
65                            "_" => res.push(T![_], edition),
66                            i if i.starts_with('\'') => res.push(LIFETIME_IDENT, edition),
67                            _ if ident.is_raw.yes() => res.push(IDENT, edition),
68                            text => match SyntaxKind::from_keyword(text, edition) {
69                                Some(kind) => res.push(kind, edition),
70                                None => {
71                                    let contextual_keyword =
72                                        SyntaxKind::from_contextual_keyword(text, edition)
73                                            .unwrap_or(SyntaxKind::IDENT);
74                                    res.push_ident(contextual_keyword, edition);
75                                }
76                            },
77                        }
78                    }
79                    tt::Leaf::Punct(punct) => {
80                        let kind = SyntaxKind::from_char(punct.char)
81                            .unwrap_or_else(|| panic!("{punct:#?} is not a valid punct"));
82                        res.push(kind, ctx_edition(punct.span.ctx));
83                        if punct.spacing == tt::Spacing::Joint {
84                            res.was_joint();
85                        }
86                    }
87                }
88                current.bump();
89            }
90            Some(tt::TokenTree::Subtree(subtree)) => {
91                if let Some(kind) = match subtree.delimiter.kind {
92                    tt::DelimiterKind::Parenthesis => Some(T!['(']),
93                    tt::DelimiterKind::Brace => Some(T!['{']),
94                    tt::DelimiterKind::Bracket => Some(T!['[']),
95                    tt::DelimiterKind::Invisible => None,
96                } {
97                    res.push(kind, ctx_edition(subtree.delimiter.open.ctx));
98                }
99                current.bump();
100            }
101            None => {
102                let subtree = current.end();
103                if let Some(kind) = match subtree.delimiter.kind {
104                    tt::DelimiterKind::Parenthesis => Some(T![')']),
105                    tt::DelimiterKind::Brace => Some(T!['}']),
106                    tt::DelimiterKind::Bracket => Some(T![']']),
107                    tt::DelimiterKind::Invisible => None,
108                } {
109                    res.push(kind, ctx_edition(subtree.delimiter.close.ctx));
110                }
111            }
112        };
113    }
114
115    res
116}
