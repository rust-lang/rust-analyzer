1//! This module defines a `DynMap` -- a container for heterogeneous maps.
2//!
3//! This means that `DynMap` stores a bunch of hash maps inside, and those maps
4//! can be of different types.
5//!
6//! It is used like this:
7//!
8//! ```ignore
9//! # use hir_def::dyn_map::DynMap;
10//! # use hir_def::dyn_map::Key;
11//! // keys define submaps of a `DynMap`
12//! const STRING_TO_U32: Key<String, u32> = Key::new();
13//! const U32_TO_VEC: Key<u32, Vec<bool>> = Key::new();
14//!
15//! // Note: concrete type, no type params!
16//! let mut map = DynMap::new();
17//!
18//! // To access a specific map, index the `DynMap` by `Key`:
19//! map[STRING_TO_U32].insert("hello".to_string(), 92);
20//! let value = map[U32_TO_VEC].get(92);
21//! assert!(value.is_none());
22//! ```
23//!
24//! This is a work of fiction. Any similarities to Kotlin's `BindingContext` are
25//! a coincidence.
26

80    /// XXX: AST Nodes and SyntaxNodes have identity equality semantics: nodes are
81    /// equal if they point to exactly the same object.
82    ///
83    /// In general, we do not guarantee that we have exactly one instance of a
84    /// syntax tree for each file. We probably should add such guarantee, but, for
85    /// the time being, we will use identity-less AstPtr comparison.
86    pub struct AstPtrPolicy<AST, ID> {
87        _phantom: PhantomData<(AST, ID)>,
88    }
89
90    impl<AST: AstNode + 'static, ID: 'static> Policy for AstPtrPolicy<AST, ID> {
91        type K = AstPtr<AST>;
92        type V = ID;
93        fn insert(map: &mut DynMap, key: AstPtr<AST>, value: ID) {
94            map.map
95                .entry::<FxHashMap<AstPtr<AST>, ID>>()
96                .or_insert_with(Default::default)
97                .insert(key, value);
98        }
99        fn get<'a>(map: &'a DynMap, key: &AstPtr<AST>) -> Option<&'a ID> {
100            map.map.get::<FxHashMap<AstPtr<AST>, ID>>()?.get(key)
101        }
102        fn is_empty(map: &DynMap) -> bool {
103            map.map.get::<FxHashMap<AstPtr<AST>, ID>>().is_none_or(|it| it.is_empty())
104        }
105    }
106}

139pub trait Policy {
140    type K;
141    type V;
142
143    fn insert(map: &mut DynMap, key: Self::K, value: Self::V);
144    fn get<'a>(map: &'a DynMap, key: &Self::K) -> Option<&'a Self::V>;
145    fn is_empty(map: &DynMap) -> bool;
146}
147
148impl<K: Hash + Eq + 'static, V: 'static> Policy for (K, V) {
149    type K = K;
150    type V = V;
151    fn insert(map: &mut DynMap, key: K, value: V) {
152        map.map.entry::<FxHashMap<K, V>>().or_insert_with(Default::default).insert(key, value);
153    }
154    fn get<'a>(map: &'a DynMap, key: &K) -> Option<&'a V> {
155        map.map.get::<FxHashMap<K, V>>()?.get(key)
156    }
157    fn is_empty(map: &DynMap) -> bool {
158        map.map.get::<FxHashMap<K, V>>().is_none_or(|it| it.is_empty())
159    }
160}

162#[derive(Default)]
163pub struct DynMap {
164    pub(crate) map: Map,
165}
166
167#[repr(transparent)]
168pub struct KeyMap<KEY> {
169    map: DynMap,
170    _phantom: PhantomData<KEY>,
171}
172
173impl<P: Policy> KeyMap<Key<P::K, P::V, P>> {
174    pub fn insert(&mut self, key: P::K, value: P::V) {
175        P::insert(&mut self.map, key, value)
176    }
177    pub fn get(&self, key: &P::K) -> Option<&P::V> {
178        P::get(&self.map, key)
179    }
180
181    pub fn is_empty(&self) -> bool {
182        P::is_empty(&self.map)
183    }
184}

186impl<P: Policy> Index<Key<P::K, P::V, P>> for DynMap {
187    type Output = KeyMap<Key<P::K, P::V, P>>;
188    fn index(&self, _key: Key<P::K, P::V, P>) -> &Self::Output {
189        // Safe due to `#[repr(transparent)]`.
190        unsafe { std::mem::transmute::<&DynMap, &KeyMap<Key<P::K, P::V, P>>>(self) }
191    }
192}
193
194impl<P: Policy> IndexMut<Key<P::K, P::V, P>> for DynMap {

196        // Safe due to `#[repr(transparent)]`.

198    }
