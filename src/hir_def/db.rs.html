84#[query_group::query_group]
85pub trait DefDatabase: InternDatabase + ExpandDatabase + SourceDatabase {
86    /// Whether to expand procedural macros during name resolution.
87    #[salsa::input]
88    fn expand_proc_attr_macros(&self) -> bool;
89
90    /// Computes an [`ItemTree`] for the given file or macro expansion.
91    #[salsa::invoke(file_item_tree_query)]
92    #[salsa::transparent]
93    fn file_item_tree(&self, file_id: HirFileId, krate: Crate) -> &ItemTree;
94
95    /// Turns a MacroId into a MacroDefId, describing the macro's definition post name resolution.
96    #[salsa::invoke(macro_def)]
97    fn macro_def(&self, m: MacroId) -> MacroDefId;
98
99    // region:visibilities
100
101    #[salsa::invoke(visibility::field_visibilities_query)]
102    fn field_visibilities(&self, var: VariantId) -> Arc<ArenaMap<LocalFieldId, Visibility>>;
103
104    #[salsa::invoke(visibility::assoc_visibility_query)]
105    fn assoc_visibility(&self, def: AssocItemId) -> Visibility;
106
107    // endregion:visibilities
108
109    #[salsa::invoke(crate::lang_item::crate_notable_traits)]
110    #[salsa::transparent]
111    fn crate_notable_traits(&self, krate: Crate) -> Option<&[TraitId]>;
112
113    #[salsa::invoke(crate_supports_no_std)]
114    fn crate_supports_no_std(&self, crate_id: Crate) -> bool;
115
116    #[salsa::invoke(include_macro_invoc)]
117    fn include_macro_invoc(&self, crate_id: Crate) -> Arc<[(MacroCallId, EditionedFileId)]>;
118}

120// return: macro call id and include file id
121fn include_macro_invoc(
122    db: &dyn DefDatabase,
123    krate: Crate,
124) -> Arc<[(MacroCallId, EditionedFileId)]> {
125    crate_def_map(db, krate)
126        .modules
127        .values()
128        .flat_map(|m| m.scope.iter_macro_invoc())
129        .filter_map(|invoc| {
130            db.lookup_intern_macro_call(*invoc.1)
131                .include_file_id(db, *invoc.1)
132                .map(|x| (*invoc.1, x))
133        })
134        .collect()
135}
136
137fn crate_supports_no_std(db: &dyn DefDatabase, crate_id: Crate) -> bool {
138    let root_module = crate_def_map(db, crate_id).root_module_id();
139    let attrs = AttrFlags::query(db, AttrDefId::ModuleId(root_module));
140    attrs.contains(AttrFlags::IS_NO_STD)
141}

143fn macro_def(db: &dyn DefDatabase, id: MacroId) -> MacroDefId {
144    let kind = |expander, file_id, m| {
145        let in_file = InFile::new(file_id, m);
146        match expander {
147            MacroExpander::Declarative { styles } => MacroDefKind::Declarative(in_file, styles),
148            MacroExpander::BuiltIn(it) => MacroDefKind::BuiltIn(in_file, it),
149            MacroExpander::BuiltInAttr(it) => MacroDefKind::BuiltInAttr(in_file, it),
150            MacroExpander::BuiltInDerive(it) => MacroDefKind::BuiltInDerive(in_file, it),
151            MacroExpander::BuiltInEager(it) => MacroDefKind::BuiltInEager(in_file, it),
152        }
153    };
154
155    match id {
156        MacroId::Macro2Id(it) => {
157            let loc: Macro2Loc = it.lookup(db);
158
159            MacroDefId {
160                krate: loc.container.krate(db),
161                kind: kind(loc.expander, loc.id.file_id, loc.id.value.upcast()),
162                local_inner: false,
163                allow_internal_unsafe: loc.allow_internal_unsafe,
164                edition: loc.edition,
165            }
166        }
167        MacroId::MacroRulesId(it) => {
168            let loc: MacroRulesLoc = it.lookup(db);
169
170            MacroDefId {
171                krate: loc.container.krate(db),
172                kind: kind(loc.expander, loc.id.file_id, loc.id.value.upcast()),
173                local_inner: loc.flags.contains(MacroRulesLocFlags::LOCAL_INNER),
174                allow_internal_unsafe: loc
175                    .flags
176                    .contains(MacroRulesLocFlags::ALLOW_INTERNAL_UNSAFE),
177                edition: loc.edition,
178            }
179        }
180        MacroId::ProcMacroId(it) => {
181            let loc = it.lookup(db);
182
183            MacroDefId {
184                krate: loc.container.krate(db),
185                kind: MacroDefKind::ProcMacro(loc.id, loc.expander, loc.kind),
186                local_inner: false,
187                allow_internal_unsafe: false,
188                edition: loc.edition,
189            }
190        }
191    }
