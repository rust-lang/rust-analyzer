261// return: macro call id and include file id
262fn include_macro_invoc(
263    db: &dyn DefDatabase,
264    krate: Crate,
265) -> Arc<[(MacroCallId, EditionedFileId)]> {
266    crate_def_map(db, krate)
267        .modules
268        .values()
269        .flat_map(|m| m.scope.iter_macro_invoc())
270        .filter_map(|invoc| {
271            db.lookup_intern_macro_call(*invoc.1)
272                .include_file_id(db, *invoc.1)
273                .map(|x| (*invoc.1, x))
274        })
275        .collect()
276}

278fn crate_supports_no_std(db: &dyn DefDatabase, crate_id: Crate) -> bool {
279    let root_module = CrateRootModuleId::from(crate_id).module(db);
280    let attrs = AttrFlags::query(db, AttrDefId::ModuleId(InternedModuleId::new(db, root_module)));
281    attrs.contains(AttrFlags::IS_NO_STD)
282}

284fn macro_def(db: &dyn DefDatabase, id: MacroId) -> MacroDefId {
285    let kind = |expander, file_id, m| {
286        let in_file = InFile::new(file_id, m);
287        match expander {
288            MacroExpander::Declarative { styles } => MacroDefKind::Declarative(in_file, styles),
289            MacroExpander::BuiltIn(it) => MacroDefKind::BuiltIn(in_file, it),
290            MacroExpander::BuiltInAttr(it) => MacroDefKind::BuiltInAttr(in_file, it),
291            MacroExpander::BuiltInDerive(it) => MacroDefKind::BuiltInDerive(in_file, it),
292            MacroExpander::BuiltInEager(it) => MacroDefKind::BuiltInEager(in_file, it),
293        }
294    };
295
296    match id {
297        MacroId::Macro2Id(it) => {
298            let loc: Macro2Loc = it.lookup(db);
299
300            MacroDefId {
301                krate: loc.container.krate,
302                kind: kind(loc.expander, loc.id.file_id, loc.id.value.upcast()),
303                local_inner: false,
304                allow_internal_unsafe: loc.allow_internal_unsafe,
305                edition: loc.edition,
306            }
307        }
308        MacroId::MacroRulesId(it) => {
309            let loc: MacroRulesLoc = it.lookup(db);
310
311            MacroDefId {
312                krate: loc.container.krate,
313                kind: kind(loc.expander, loc.id.file_id, loc.id.value.upcast()),
314                local_inner: loc.flags.contains(MacroRulesLocFlags::LOCAL_INNER),
315                allow_internal_unsafe: loc
316                    .flags
317                    .contains(MacroRulesLocFlags::ALLOW_INTERNAL_UNSAFE),
318                edition: loc.edition,
319            }
320        }
321        MacroId::ProcMacroId(it) => {
322            let loc = it.lookup(db);
323
324            MacroDefId {
325                krate: loc.container.krate,
326                kind: MacroDefKind::ProcMacro(loc.id, loc.expander, loc.kind),
327                local_inner: false,
328                allow_internal_unsafe: false,
329                edition: loc.edition,
330            }
331        }
332    }
