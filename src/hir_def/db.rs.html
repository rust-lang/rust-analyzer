18#[query_group::query_group]
19pub trait DefDatabase: ExpandDatabase + SourceDatabase {
20    /// Computes an [`ItemTree`] for the given file or macro expansion.
21    #[salsa::invoke(file_item_tree)]

25    /// Turns a MacroId into a MacroDefId, describing the macro's definition post name resolution.

29    // region:visibilities

37    #[salsa::transparent]

40    #[salsa::invoke(crate_supports_no_std)]

45}

47/// Whether to expand procedural macros during name resolution.
48///
49/// Note: this struct shouldn't be exposed to ide crates -- consider using
50/// [`set_expand_proc_attr_macros`] instead, if possible.
51#[salsa::input(singleton, debug)]
52pub(crate) struct ExpandProcAttrMacros {
53    #[returns(copy)]
54    pub(crate) enabled: bool,
55}
56
57pub fn set_expand_proc_attr_macros(db: &mut dyn DefDatabase, enabled: bool) {
58    if let Some(expand_proc_attr_macros) = ExpandProcAttrMacros::try_get(db) {
59        if expand_proc_attr_macros.enabled(db) != enabled {
60            expand_proc_attr_macros.set_enabled(db).with_durability(Durability::HIGH).to(enabled);
61        }
62    } else {
63        _ = ExpandProcAttrMacros::builder(enabled).durability(Durability::HIGH).new(db);
64    }
65}

67// return: macro call id and include file id
68fn include_macro_invoc(
69    db: &dyn DefDatabase,
70    krate: Crate,
71) -> Arc<[(MacroCallId, EditionedFileId)]> {
72    crate_def_map(db, krate)
73        .modules
74        .values()
75        .flat_map(|m| m.scope.iter_macro_invoc())
76        .filter_map(|invoc| invoc.1.loc(db).include_file_id(db, *invoc.1).map(|x| (*invoc.1, x)))
77        .collect()
78}

80fn crate_supports_no_std(db: &dyn DefDatabase, crate_id: Crate) -> bool {
81    let root_module = crate_def_map(db, crate_id).root_module_id();
82    let attrs = AttrFlags::query(db, AttrDefId::ModuleId(root_module));
83    attrs.contains(AttrFlags::IS_NO_STD)
84}

86fn macro_def(db: &dyn DefDatabase, id: MacroId) -> MacroDefId {
87    let kind = |expander, file_id, m| {
88        let in_file = InFile::new(file_id, m);
89        match expander {
90            MacroExpander::Declarative { styles } => MacroDefKind::Declarative(in_file, styles),
91            MacroExpander::BuiltIn(it) => MacroDefKind::BuiltIn(in_file, it),
92            MacroExpander::BuiltInAttr(it) => MacroDefKind::BuiltInAttr(in_file, it),
93            MacroExpander::BuiltInDerive(it) => MacroDefKind::BuiltInDerive(in_file, it),
94            MacroExpander::BuiltInEager(it) => MacroDefKind::BuiltInEager(in_file, it),
95            MacroExpander::UnimplementedBuiltIn => MacroDefKind::UnimplementedBuiltIn(in_file),
96        }
97    };
98
99    match id {
100        MacroId::Macro2Id(it) => {
101            let loc = it.lookup(db);
102
103            MacroDefId {
104                krate: loc.container.krate(db),
105                kind: kind(loc.expander, loc.id.file_id, loc.id.value.upcast()),
106                local_inner: false,
107                allow_internal_unsafe: loc.allow_internal_unsafe,
108                edition: loc.edition,
109            }
110        }
111        MacroId::MacroRulesId(it) => {
112            let loc = it.lookup(db);
113
114            MacroDefId {
115                krate: loc.container.krate(db),
116                kind: kind(loc.expander, loc.id.file_id, loc.id.value.upcast()),
117                local_inner: loc.flags.contains(MacroRulesLocFlags::LOCAL_INNER),
118                allow_internal_unsafe: loc
119                    .flags
120                    .contains(MacroRulesLocFlags::ALLOW_INTERNAL_UNSAFE),
121                edition: loc.edition,
122            }
123        }
124        MacroId::ProcMacroId(it) => {
125            let loc = it.lookup(db);
126
127            MacroDefId {
128                krate: loc.container.krate(db),
129                kind: MacroDefKind::ProcMacro(loc.id, loc.expander, loc.kind),
130                local_inner: false,
131                allow_internal_unsafe: false,
132                edition: loc.edition,
133            }
134        }
135    }
136}
