1//! A simplified AST that only contains items.
2//!
3//! This is the primary IR used throughout `hir_def`. It is the input to the name resolution
4//! algorithm, as well as to the queries defined in `adt.rs`, `data.rs`, and most things in
5//! `attr.rs`.
6//!
7//! `ItemTree`s are built per `HirFileId`, from the syntax tree of the parsed file. This means that
8//! they are crate-independent: they don't know which `#[cfg]`s are active or which module they
9//! belong to, since those concepts don't exist at this level (a single `ItemTree` might be part of
10//! multiple crates, or might be included into the same crate twice via `#[path]`).
11//!
12//! One important purpose of this layer is to provide an "invalidation barrier" for incremental
13//! computations: when typing inside an item body, the `ItemTree` of the modified file is typically
14//! unaffected, so we don't have to recompute name resolution results or item data (see `data.rs`).
15//!
16//! The `ItemTree` for the currently open file can be displayed by using the VS Code command
17//! "rust-analyzer: Debug ItemTree".
18//!
19//! Compared to rustc's architecture, `ItemTree` has properties from both rustc's AST and HIR: many
20//! syntax-level Rust features are already desugared to simpler forms in the `ItemTree`, but name
21//! resolution has not yet been performed. `ItemTree`s are per-file, while rustc's AST and HIR are
22//! per-crate, because we are interested in incrementally computing it.
23//!
24//! The representation of items in the `ItemTree` should generally mirror the surface syntax: it is
25//! usually a bad idea to desugar a syntax-level construct to something that is structurally
26//! different here. Name resolution needs to be able to process attributes and expand macros
27//! (including attribute macros), and having a 1-to-1 mapping between syntax and the `ItemTree`
28//! avoids introducing subtle bugs.
29//!
30//! In general, any item in the `ItemTree` stores its `AstId`, which allows mapping it back to its
31//! surface syntax.

76impl RawVisibilityId {
77    const PUB: Self = RawVisibilityId(u32::MAX);
78    const PRIV_IMPLICIT: Self = RawVisibilityId(u32::MAX - 1);
79    const PRIV_EXPLICIT: Self = RawVisibilityId(u32::MAX - 2);
80    const PUB_CRATE: Self = RawVisibilityId(u32::MAX - 3);

83impl fmt::Debug for RawVisibilityId {
84    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
85        let mut f = f.debug_tuple("RawVisibilityId");
86        match *self {
87            Self::PUB => f.field(&"pub"),
88            Self::PRIV_IMPLICIT | Self::PRIV_EXPLICIT => f.field(&"pub(self)"),
89            Self::PUB_CRATE => f.field(&"pub(crate)"),
90            _ => f.field(&self.0),
91        };
92        f.finish()
93    }
94}

96fn lower_extra_crate_attrs<'a>(
97    db: &dyn DefDatabase,
98    crate_attrs_as_src: SourceFile,
99    file_id: span::EditionedFileId,
100    cfg_options: &dyn Fn() -> &'a CfgOptions,
101) -> AttrsOrCfg {
102    #[derive(Copy, Clone)]
103    struct FakeSpanMap {
104        file_id: span::EditionedFileId,
105    }
106    impl syntax_bridge::SpanMapper for FakeSpanMap {
107        fn span_for(&self, range: TextRange) -> Span {
108            Span {
109                range,
110                anchor: SpanAnchor {
111                    file_id: self.file_id,
112                    ast_id: NO_DOWNMAP_ERASED_FILE_AST_ID_MARKER,
113                },
114                ctx: SyntaxContext::root(self.file_id.edition()),
115            }
116        }
117    }
118
119    let span_map = FakeSpanMap { file_id };
120    AttrsOrCfg::lower(db, &crate_attrs_as_src, cfg_options, span_map)
121}

123#[salsa_macros::tracked(returns(deref))]
124pub(crate) fn file_item_tree_query(db: &dyn DefDatabase, file_id: HirFileId) -> Arc<ItemTree> {
125    let _p = tracing::info_span!("file_item_tree_query", ?file_id).entered();
126    static EMPTY: OnceLock<Arc<ItemTree>> = OnceLock::new();
127
128    let ctx = lower::Ctx::new(db, file_id);
129    let syntax = db.parse_or_expand(file_id);
130    let mut item_tree = match_ast! {
131        match syntax {
132            ast::SourceFile(file) => {
133                let krate = file_id.krate(db);
134                let root_file_id = krate.root_file_id(db);
135                let extra_top_attrs = (file_id == root_file_id).then(|| {
136                    parse_extra_crate_attrs(db, krate).map(|crate_attrs| {
137                        let file_id = root_file_id.editioned_file_id(db);
138                        lower_extra_crate_attrs(db, crate_attrs, file_id, &|| ctx.cfg_options())
139                    })
140                }).flatten();
141                let top_attrs = match extra_top_attrs {
142                    Some(attrs @ AttrsOrCfg::Enabled { .. }) => attrs.merge(ctx.lower_attrs(&file)),
143                    Some(attrs @ AttrsOrCfg::CfgDisabled(_)) => attrs,
144                    None => ctx.lower_attrs(&file)
145                };
146                let mut item_tree = ctx.lower_module_items(&file);
147                item_tree.top_attrs = top_attrs;
148                item_tree
149            },
150            ast::MacroItems(items) => {
151                ctx.lower_module_items(&items)
152            },
153            ast::MacroStmts(stmts) => {
154                // The produced statements can include items, which should be added as top-level
155                // items.
156                ctx.lower_macro_stmts(stmts)
157            },
158            _ => {
159                if never!(syntax.kind() == SyntaxKind::ERROR, "{:?} from {:?} {}", file_id, syntax, syntax) {
160                    return Default::default();
161                }
162                panic!("cannot create item tree for file {file_id:?} from {syntax:?} {syntax}");
163            },
164        }
165    };
166    let ItemTree { top_level, top_attrs, attrs, vis, big_data, small_data } = &item_tree;
167    if small_data.is_empty()
168        && big_data.is_empty()
169        && top_level.is_empty()
170        && attrs.is_empty()
171        && top_attrs.is_empty()
172        && vis.arena.is_empty()
173    {
174        EMPTY
175            .get_or_init(|| {
176                Arc::new(ItemTree {
177                    top_level: Box::new([]),
178                    attrs: FxHashMap::default(),
179                    small_data: FxHashMap::default(),
180                    big_data: FxHashMap::default(),
181                    top_attrs: AttrsOrCfg::empty(),
182                    vis: ItemVisibilities { arena: ThinVec::new() },
183                })
184            })
185            .clone()
186    } else {
187        item_tree.shrink_to_fit();
188        Arc::new(item_tree)
189    }
190}

192#[salsa_macros::tracked(returns(deref))]
193pub(crate) fn block_item_tree_query(db: &dyn DefDatabase, block: BlockId) -> Arc<ItemTree> {
194    let _p = tracing::info_span!("block_item_tree_query", ?block).entered();
195    static EMPTY: OnceLock<Arc<ItemTree>> = OnceLock::new();
196
197    let loc = block.lookup(db);
198    let block = loc.ast_id.to_node(db);
199
200    let ctx = lower::Ctx::new(db, loc.ast_id.file_id);
201    let mut item_tree = ctx.lower_block(&block);
202    let ItemTree { top_level, top_attrs, attrs, vis, big_data, small_data } = &item_tree;
203    if small_data.is_empty()
204        && big_data.is_empty()
205        && top_level.is_empty()
206        && attrs.is_empty()
207        && top_attrs.is_empty()
208        && vis.arena.is_empty()
209    {
210        EMPTY
211            .get_or_init(|| {
212                Arc::new(ItemTree {
213                    top_level: Box::new([]),
214                    attrs: FxHashMap::default(),
215                    small_data: FxHashMap::default(),
216                    big_data: FxHashMap::default(),
217                    top_attrs: AttrsOrCfg::empty(),
218                    vis: ItemVisibilities { arena: ThinVec::new() },
219                })
220            })
221            .clone()
222    } else {
223        item_tree.shrink_to_fit();
224        Arc::new(item_tree)
225    }

227/// The item tree of a source file.
228#[derive(Debug, Default, Eq, PartialEq)]
229pub struct ItemTree {
230    top_level: Box<[ModItemId]>,
231    top_attrs: AttrsOrCfg,
232    attrs: FxHashMap<FileAstId<ast::Item>, AttrsOrCfg>,
233    vis: ItemVisibilities,
234    big_data: FxHashMap<FileAstId<ast::Item>, BigModItem>,
235    small_data: FxHashMap<FileAstId<ast::Item>, SmallModItem>,
236}
237
238impl ItemTree {
239    /// Returns an iterator over all items located at the top level of the `HirFileId` this
240    /// `ItemTree` was created from.
241    pub(crate) fn top_level_items(&self) -> &[ModItemId] {
242        &self.top_level
243    }
244
245    /// Returns the inner attributes of the source file.
246    pub(crate) fn top_level_attrs(&self) -> &AttrsOrCfg {
247        &self.top_attrs
248    }
249
250    pub(crate) fn attrs(&self, of: FileAstId<ast::Item>) -> Option<&AttrsOrCfg> {
251        self.attrs.get(&of)
252    }
253
254    /// Returns a count of a few, expensive items.
255    ///
256    /// For more detail, see [`ItemTreeDataStats`].
257    pub fn item_tree_stats(&self) -> ItemTreeDataStats {
258        let mut traits = 0;
259        let mut impls = 0;
260        let mut mods = 0;
261        let mut macro_calls = 0;
262        let mut macro_rules = 0;
263        for item in self.small_data.values() {
264            match item {
265                SmallModItem::Trait(_) => traits += 1,
266                SmallModItem::Impl(_) => impls += 1,
267                SmallModItem::MacroRules(_) => macro_rules += 1,
268                SmallModItem::MacroCall(_) => macro_calls += 1,
269                _ => {}
270            }
271        }
272        for item in self.big_data.values() {
273            match item {
274                BigModItem::Mod(_) => mods += 1,
275                _ => {}
276            }
277        }
278        ItemTreeDataStats { traits, impls, mods, macro_calls, macro_rules }
279    }
280
281    pub fn pretty_print(&self, db: &dyn DefDatabase, edition: Edition) -> String {
282        pretty::print_item_tree(db, self, edition)
283    }
284
285    fn shrink_to_fit(&mut self) {
286        let ItemTree { top_level: _, attrs, big_data, small_data, vis: _, top_attrs: _ } = self;
287        attrs.shrink_to_fit();
288        big_data.shrink_to_fit();
289        small_data.shrink_to_fit();
290    }
291}

376macro_rules! mod_items {
377    ($mod_item:ident -> $( $typ:ident in $fld:ident -> $ast:ty ),+ $(,)? ) => {
378        #[derive(Debug, Copy, Clone, Eq, PartialEq, Hash)]
379        pub(crate) enum $mod_item {
380            $(
381                $typ(FileAstId<$ast>),
382            )+
383        }
384
385        impl $mod_item {
386            pub(crate) fn ast_id(self) -> FileAstId<ast::Item> {
387                match self {
388                    $($mod_item::$typ(it) => it.upcast()),+
389                }
390            }
391        }
392
393        $(
394            impl From<FileAstId<$ast>> for $mod_item {
395                fn from(id: FileAstId<$ast>) -> $mod_item {
396                    ModItemId::$typ(id)
397                }
398            }
399        )+
400
401        $(
402            impl ItemTreeNode for $typ {
403                type Source = $ast;
404            }
405
406            impl Index<FileAstId<$ast>> for ItemTree {
407                type Output = $typ;
408
409                #[allow(unused_imports)]
410                fn index(&self, index: FileAstId<$ast>) -> &Self::Output {
411                    use BigModItem::*;
412                    use SmallModItem::*;
413                    match &self.$fld[&index.upcast()] {
414                        $typ(item) => item,
415                        _ => panic!("expected item of type `{}` at index `{:?}`", stringify!($typ), index),
416                    }
417                }
418            }
419        )+
420    };
421}

443impl Index<RawVisibilityId> for ItemTree {
444    type Output = RawVisibility;
445    fn index(&self, index: RawVisibilityId) -> &Self::Output {
446        static VIS_PUB: RawVisibility = RawVisibility::Public;
447        static VIS_PRIV_IMPLICIT: RawVisibility =
448            RawVisibility::PubSelf(VisibilityExplicitness::Implicit);
449        static VIS_PRIV_EXPLICIT: RawVisibility =
450            RawVisibility::PubSelf(VisibilityExplicitness::Explicit);
451        static VIS_PUB_CRATE: RawVisibility = RawVisibility::PubCrate;
452
453        match index {
454            RawVisibilityId::PRIV_IMPLICIT => &VIS_PRIV_IMPLICIT,
455            RawVisibilityId::PRIV_EXPLICIT => &VIS_PRIV_EXPLICIT,
456            RawVisibilityId::PUB => &VIS_PUB,
457            RawVisibilityId::PUB_CRATE => &VIS_PUB_CRATE,
458            _ => &self.vis.arena[index.0 as usize],
459        }
460    }
461}

677        mut cb: impl FnMut(Idx<ast::UseTree>, ModPath, ImportKind, Option<ImportAlias>),
678    ) {
679        self.use_tree.expand_impl(None, &mut 0, &mut cb)
680    }
681}
682
683impl UseTree {
684    /// The [`UseTreeKind`] of this `UseTree`.
685    pub fn kind(&self) -> &UseTreeKind {
686        &self.kind
687    }
688
689    fn expand_impl(
690        &self,
691        prefix: Option<ModPath>,
692        counting_index: &mut u32,
693        cb: &mut impl FnMut(Idx<ast::UseTree>, ModPath, ImportKind, Option<ImportAlias>),
694    ) {
695        fn concat_mod_paths(
696            prefix: Option<ModPath>,
697            path: &ModPath,
698        ) -> Option<(ModPath, ImportKind)> {
699            match (prefix, path.kind) {
700                (None, _) => Some((path.clone(), ImportKind::Plain)),
701                (Some(mut prefix), PathKind::Plain) => {
702                    prefix.extend(path.segments().iter().cloned());
703                    Some((prefix, ImportKind::Plain))
704                }
705                (Some(mut prefix), PathKind::Super(n)) if n > 0 && prefix.segments().is_empty() => {
706                    // `super::super` + `super::rest`
707                    match &mut prefix.kind {
708                        PathKind::Super(m) => {
709                            cov_mark::hit!(concat_super_mod_paths);
710                            *m += n;
711                            prefix.extend(path.segments().iter().cloned());
712                            Some((prefix, ImportKind::Plain))
713                        }
714                        _ => None,
715                    }
716                }
717                (Some(prefix), PathKind::SELF) if path.segments().is_empty() => {
718                    // `some::path::self` == `some::path`
719                    Some((prefix, ImportKind::TypeOnly))
720                }
721                (Some(_), _) => None,
722            }
723        }
724
725        match &self.kind {
726            UseTreeKind::Single { path, alias } => {
727                if let Some((path, kind)) = concat_mod_paths(prefix, path) {
728                    cb(Idx::from_raw(RawIdx::from_u32(*counting_index)), path, kind, alias.clone());
729                }
730            }
731            UseTreeKind::Glob { path: Some(path) } => {
732                if let Some((path, _)) = concat_mod_paths(prefix, path) {
733                    cb(
734                        Idx::from_raw(RawIdx::from_u32(*counting_index)),
735                        path,
736                        ImportKind::Glob,
737                        None,
738                    );
739                }
740            }
741            UseTreeKind::Glob { path: None } => {
742                if let Some(prefix) = prefix {
743                    cb(
744                        Idx::from_raw(RawIdx::from_u32(*counting_index)),
745                        prefix,
746                        ImportKind::Glob,
747                        None,
748                    );
749                }
750            }
751            UseTreeKind::Prefixed { prefix: additional_prefix, list } => {
752                let prefix = match additional_prefix {
753                    Some(path) => match concat_mod_paths(prefix, path) {
754                        Some((path, ImportKind::Plain)) => Some(path),
755                        _ => return,
756                    },
757                    None => prefix,
758                };
759                for tree in &**list {
760                    *counting_index += 1;
761                    tree.expand_impl(prefix.clone(), counting_index, cb);
762                }
763            }
764        }
765    }
766}
