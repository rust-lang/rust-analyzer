52bitflags! {
53    #[derive(Debug, Copy, Clone, PartialEq, Eq)]
54    pub struct StructFlags: u8 {
55        /// Indicates whether this struct has `#[repr]`.
56        const HAS_REPR = 1 << 0;
57        /// Indicates whether the struct has a `#[rustc_has_incoherent_inherent_impls]` attribute.
58        const RUSTC_HAS_INCOHERENT_INHERENT_IMPLS = 1 << 1;
59        /// Indicates whether the struct has a `#[fundamental]` attribute.
60        const FUNDAMENTAL      = 1 << 2;
61        /// Indicates whether the struct is `PhantomData`.
62        const IS_PHANTOM_DATA  = 1 << 3;
63        /// Indicates whether this struct is `Box`.
64        const IS_BOX           = 1 << 4;
65        /// Indicates whether this struct is `ManuallyDrop`.
66        const IS_MANUALLY_DROP = 1 << 5;
67        /// Indicates whether this struct is `UnsafeCell`.
68        const IS_UNSAFE_CELL   = 1 << 6;
69        /// Indicates whether this struct is `UnsafePinned`.
70        const IS_UNSAFE_PINNED = 1 << 7;
71    }
72}
73
74impl StructSignature {
75    pub fn query(db: &dyn DefDatabase, id: StructId) -> (Arc<Self>, Arc<ExpressionStoreSourceMap>) {
76        let loc = id.lookup(db);
77        let InFile { file_id, value: source } = loc.source(db);
78        let attrs = AttrFlags::query(db, id.into());
79
80        let mut flags = StructFlags::empty();
81        if attrs.contains(AttrFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS) {
82            flags |= StructFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS;
83        }
84        if attrs.contains(AttrFlags::FUNDAMENTAL) {
85            flags |= StructFlags::FUNDAMENTAL;
86        }
87        if attrs.contains(AttrFlags::HAS_REPR) {
88            flags |= StructFlags::HAS_REPR;
89        }
90        if let Some(lang) = attrs.lang_item_with_attrs(db, id.into()) {
91            match lang {
92                _ if lang == sym::phantom_data => flags |= StructFlags::IS_PHANTOM_DATA,
93                _ if lang == sym::owned_box => flags |= StructFlags::IS_BOX,
94                _ if lang == sym::manually_drop => flags |= StructFlags::IS_MANUALLY_DROP,
95                _ if lang == sym::unsafe_cell => flags |= StructFlags::IS_UNSAFE_CELL,
96                _ if lang == sym::unsafe_pinned => flags |= StructFlags::IS_UNSAFE_PINNED,
97                _ => (),
98            }
99        }
100        let shape = adt_shape(source.kind());
101
102        let (store, generic_params, source_map) = lower_generic_params(
103            db,
104            loc.container,
105            id.into(),
106            file_id,
107            source.generic_param_list(),
108            source.where_clause(),
109        );
110        (
111            Arc::new(StructSignature {
112                generic_params,
113                store,
114                flags,
115                shape,
116                name: as_name_opt(source.name()),
117            }),
118            Arc::new(source_map),
119        )
120    }
121
122    #[inline]
123    pub fn repr(&self, db: &dyn DefDatabase, id: StructId) -> Option<ReprOptions> {
124        if self.flags.contains(StructFlags::HAS_REPR) {
125            AttrFlags::repr(db, id.into())
126        } else {
127            None
128        }
129    }
130}

132#[inline]
133fn adt_shape(adt_kind: ast::StructKind) -> FieldsShape {
134    match adt_kind {
135        ast::StructKind::Record(_) => FieldsShape::Record,
136        ast::StructKind::Tuple(_) => FieldsShape::Tuple,
137        ast::StructKind::Unit => FieldsShape::Unit,
138    }

447impl TraitSignature {
448    pub fn query(db: &dyn DefDatabase, id: TraitId) -> (Arc<Self>, Arc<ExpressionStoreSourceMap>) {
449        let loc = id.lookup(db);
450
451        let mut flags = TraitFlags::empty();
452        let attrs = AttrFlags::query(db, id.into());
453        let source = loc.source(db);
454        if source.value.auto_token().is_some() {
455            flags.insert(TraitFlags::AUTO);
456        }
457        if source.value.unsafe_token().is_some() {
458            flags.insert(TraitFlags::UNSAFE);
459        }
460        if source.value.eq_token().is_some() {
461            flags.insert(TraitFlags::ALIAS);
462        }
463        if attrs.contains(AttrFlags::FUNDAMENTAL) {
464            flags |= TraitFlags::FUNDAMENTAL;
465        }
466        if attrs.contains(AttrFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS) {
467            flags |= TraitFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS;
468        }
469        if attrs.contains(AttrFlags::RUSTC_PAREN_SUGAR) {
470            flags |= TraitFlags::RUSTC_PAREN_SUGAR;
471        }
472        if attrs.contains(AttrFlags::RUSTC_COINDUCTIVE) {
473            flags |= TraitFlags::COINDUCTIVE;
474        }
475
476        if attrs.contains(AttrFlags::RUSTC_SKIP_ARRAY_DURING_METHOD_DISPATCH) {
477            flags |= TraitFlags::SKIP_ARRAY_DURING_METHOD_DISPATCH;
478        }
479        if attrs.contains(AttrFlags::RUSTC_SKIP_BOXED_SLICE_DURING_METHOD_DISPATCH) {
480            flags |= TraitFlags::SKIP_BOXED_SLICE_DURING_METHOD_DISPATCH;
481        }
482
483        let name = as_name_opt(source.value.name());
484        let (store, source_map, generic_params) = lower_trait(db, loc.container, source, id);
485
486        (
487            Arc::new(TraitSignature { store: Arc::new(store), generic_params, flags, name }),
488            Arc::new(source_map),
489        )
490    }

495    pub struct FnFlags: u16 {
496        const HAS_BODY = 1 << 1;
497        const DEFAULT = 1 << 2;
498        const CONST = 1 << 3;
499        const ASYNC = 1 << 4;
500        const UNSAFE = 1 << 5;
501        const HAS_VARARGS = 1 << 6;
502        const RUSTC_ALLOW_INCOHERENT_IMPL = 1 << 7;
503        const HAS_SELF_PARAM = 1 << 8;
504        /// The `#[target_feature]` attribute is necessary to check safety (with RFC 2396),
505        /// but keeping it for all functions will consume a lot of memory when there are
506        /// only very few functions with it. So we only encode its existence here, and lookup
507        /// it if needed.
508        const HAS_TARGET_FEATURE = 1 << 9;
509        const DEPRECATED_SAFE_2024 = 1 << 10;
510        const EXPLICIT_SAFE = 1 << 11;
511        const HAS_LEGACY_CONST_GENERICS = 1 << 12;
512        const RUSTC_INTRINSIC = 1 << 13;
513    }

516#[derive(Debug, PartialEq, Eq)]
517pub struct FunctionSignature {
518    pub name: Name,
519    pub generic_params: Arc<GenericParams>,
520    pub store: Arc<ExpressionStore>,
521    pub params: Box<[TypeRefId]>,
522    pub ret_type: Option<TypeRefId>,
523    pub abi: Option<Symbol>,
524    pub flags: FnFlags,
525}
526
527impl FunctionSignature {
528    pub fn query(
529        db: &dyn DefDatabase,
530        id: FunctionId,
531    ) -> (Arc<Self>, Arc<ExpressionStoreSourceMap>) {
532        let loc = id.lookup(db);
533        let module = loc.container.module(db);
534
535        let mut flags = FnFlags::empty();
536        let attrs = AttrFlags::query(db, id.into());
537        if attrs.contains(AttrFlags::RUSTC_ALLOW_INCOHERENT_IMPL) {
538            flags.insert(FnFlags::RUSTC_ALLOW_INCOHERENT_IMPL);
539        }
540
541        if attrs.contains(AttrFlags::HAS_TARGET_FEATURE) {
542            flags.insert(FnFlags::HAS_TARGET_FEATURE);
543        }
544
545        if attrs.contains(AttrFlags::RUSTC_INTRINSIC) {
546            flags.insert(FnFlags::RUSTC_INTRINSIC);
547        }
548        if attrs.contains(AttrFlags::HAS_LEGACY_CONST_GENERICS) {
549            flags.insert(FnFlags::HAS_LEGACY_CONST_GENERICS);
550        }
551
552        let source = loc.source(db);
553
554        if source.value.unsafe_token().is_some() {
555            if attrs.contains(AttrFlags::RUSTC_DEPRECATED_SAFE_2024) {
556                flags.insert(FnFlags::DEPRECATED_SAFE_2024);
557            } else {
558                flags.insert(FnFlags::UNSAFE);
559            }
560        }
561        if source.value.async_token().is_some() {
562            flags.insert(FnFlags::ASYNC);
563        }
564        if source.value.const_token().is_some() {
565            flags.insert(FnFlags::CONST);
566        }
567        if source.value.default_token().is_some() {
568            flags.insert(FnFlags::DEFAULT);
569        }
570        if source.value.safe_token().is_some() {
571            flags.insert(FnFlags::EXPLICIT_SAFE);
572        }
573        if source.value.body().is_some() {
574            flags.insert(FnFlags::HAS_BODY);
575        }
576
577        let name = as_name_opt(source.value.name());
578        let abi = source.value.abi().map(|abi| {
579            abi.abi_string().map_or_else(|| sym::C, |it| Symbol::intern(it.text_without_quotes()))
580        });
581        let (store, source_map, generic_params, params, ret_type, self_param, variadic) =
582            lower_function(db, module, source, id);
583        if self_param {
584            flags.insert(FnFlags::HAS_SELF_PARAM);
585        }
586        if variadic {
587            flags.insert(FnFlags::HAS_VARARGS);
588        }
589        (
590            Arc::new(FunctionSignature {
591                generic_params,
592                store: Arc::new(store),
593                params,
594                ret_type,
595                abi,
596                flags,
597                name,
598            }),
599            Arc::new(source_map),
600        )
601    }

658    pub fn is_intrinsic(db: &dyn DefDatabase, id: FunctionId) -> bool {
659        let data = db.function_signature(id);
660        data.flags.contains(FnFlags::RUSTC_INTRINSIC)
661            // Keep this around for a bit until extern "rustc-intrinsic" abis are no longer used
662            || match &data.abi {
663                Some(abi) => *abi == sym::rust_dash_intrinsic,
664                None => match id.lookup(db).container {
665                    ItemContainerId::ExternBlockId(block) => {
666                        block.abi(db) == Some(sym::rust_dash_intrinsic)
667                    }
668                    _ => false,
669                },
670            }
671    }

683#[derive(Debug, PartialEq, Eq)]
684pub struct TypeAliasSignature {
685    pub name: Name,
686    pub generic_params: Arc<GenericParams>,
687    pub store: Arc<ExpressionStore>,
688    pub bounds: Box<[TypeBound]>,
689    pub ty: Option<TypeRefId>,
690    pub flags: TypeAliasFlags,
691}
692
693impl TypeAliasSignature {
694    pub fn query(
695        db: &dyn DefDatabase,
696        id: TypeAliasId,
697    ) -> (Arc<Self>, Arc<ExpressionStoreSourceMap>) {
698        let loc = id.lookup(db);
699
700        let mut flags = TypeAliasFlags::empty();
701        let attrs = AttrFlags::query(db, id.into());
702        if attrs.contains(AttrFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS) {
703            flags.insert(TypeAliasFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPL);
704        }
705        if attrs.contains(AttrFlags::RUSTC_ALLOW_INCOHERENT_IMPL) {
706            flags.insert(TypeAliasFlags::RUSTC_ALLOW_INCOHERENT_IMPL);
707        }
708        if matches!(loc.container, ItemContainerId::ExternBlockId(_)) {
709            flags.insert(TypeAliasFlags::IS_EXTERN);
710        }
711        let source = loc.source(db);
712        let name = as_name_opt(source.value.name());
713        let (store, source_map, generic_params, bounds, ty) =
714            lower_type_alias(db, loc.container.module(db), source, id);
715
716        (
717            Arc::new(TypeAliasSignature {
718                store: Arc::new(store),
719                generic_params,
720                flags,
721                bounds,
722                name,
723                ty,
724            }),
725            Arc::new(source_map),
726        )
727    }

762#[derive(Debug, Clone, PartialEq, Eq)]
763pub struct VariantFields {
764    fields: Arena<FieldData>,
765    pub store: Arc<ExpressionStore>,
766    pub shape: FieldsShape,
767}
768
769#[salsa::tracked]
770impl VariantFields {
771    #[salsa::tracked(returns(clone))]
772    pub(crate) fn query(
773        db: &dyn DefDatabase,
774        id: VariantId,
775    ) -> (Arc<Self>, Arc<ExpressionStoreSourceMap>) {
776        let (shape, result) = match id {
777            VariantId::EnumVariantId(id) => {
778                let loc = id.lookup(db);
779                let parent = loc.parent.lookup(db);
780                let source = loc.source(db);
781                let shape = adt_shape(source.value.kind());
782                let enum_vis = Some(source.value.parent_enum().visibility());
783                let fields = lower_field_list(
784                    db,
785                    parent.container,
786                    source.map(|src| src.field_list()),
787                    enum_vis,
788                );
789                (shape, fields)
790            }
791            VariantId::StructId(id) => {
792                let loc = id.lookup(db);
793                let source = loc.source(db);
794                let shape = adt_shape(source.value.kind());
795                let fields =
796                    lower_field_list(db, loc.container, source.map(|src| src.field_list()), None);
797                (shape, fields)
798            }
799            VariantId::UnionId(id) => {
800                let loc = id.lookup(db);
801                let source = loc.source(db);
802                let fields = lower_field_list(
803                    db,
804                    loc.container,
805                    source.map(|src| src.record_field_list().map(ast::FieldList::RecordFieldList)),
806                    None,
807                );
808                (FieldsShape::Record, fields)
809            }
810        };
811        match result {
812            Some((fields, store, source_map)) => (
813                Arc::new(VariantFields { fields, store: Arc::new(store), shape }),
814                Arc::new(source_map),
815            ),
816            None => {
817                let (store, source_map) = ExpressionStore::empty_singleton();
818                (Arc::new(VariantFields { fields: Arena::default(), store, shape }), source_map)
819            }
820        }
821    }
822
823    #[salsa::tracked(returns(deref))]
824    pub(crate) fn firewall(db: &dyn DefDatabase, id: VariantId) -> Arc<Self> {
825        Self::query(db, id).0
826    }
827}

843fn lower_field_list(
844    db: &dyn DefDatabase,
845    module: ModuleId,
846    fields: InFile<Option<ast::FieldList>>,
847    override_visibility: Option<Option<ast::Visibility>>,
848) -> Option<(Arena<FieldData>, ExpressionStore, ExpressionStoreSourceMap)> {
849    let file_id = fields.file_id;
850    match fields.value? {
851        ast::FieldList::RecordFieldList(fields) => lower_fields(
852            db,
853            module,
854            InFile::new(file_id, fields.fields().map(|field| (field.ty(), field))),
855            |_, field| as_name_opt(field.name()),
856            override_visibility,
857        ),
858        ast::FieldList::TupleFieldList(fields) => lower_fields(
859            db,
860            module,
861            InFile::new(file_id, fields.fields().map(|field| (field.ty(), field))),
862            |idx, _| Name::new_tuple_field(idx),
863            override_visibility,
864        ),
865    }
866}
867
868fn lower_fields<Field: ast::HasAttrs + ast::HasVisibility>(
869    db: &dyn DefDatabase,
870    module: ModuleId,
871    fields: InFile<impl Iterator<Item = (Option<ast::Type>, Field)>>,
872    mut field_name: impl FnMut(usize, &Field) -> Name,
873    override_visibility: Option<Option<ast::Visibility>>,
874) -> Option<(Arena<FieldData>, ExpressionStore, ExpressionStoreSourceMap)> {
875    let cfg_options = module.krate(db).cfg_options(db);
876    let mut col = ExprCollector::new(db, module, fields.file_id);
877    let override_visibility = override_visibility.map(|vis| {
878        LazyCell::new(|| {
879            let span_map = db.span_map(fields.file_id);
880            visibility_from_ast(db, vis, &mut |range| span_map.span_for_range(range).ctx)
881        })
882    });
883
884    let mut arena = Arena::new();
885    let mut idx = 0;
886    let mut has_fields = false;
887    for (ty, field) in fields.value {
888        has_fields = true;
889        match AttrFlags::is_cfg_enabled_for(&field, cfg_options) {
890            Ok(()) => {
891                let type_ref =
892                    col.lower_type_ref_opt(ty, &mut ExprCollector::impl_trait_error_allocator);
893                let visibility = override_visibility.as_ref().map_or_else(
894                    || {
895                        visibility_from_ast(db, field.visibility(), &mut |range| {
896                            col.span_map().span_for_range(range).ctx
897                        })
898                    },
899                    |it| RawVisibility::clone(it),
900                );
901                let is_unsafe = field
902                    .syntax()
903                    .children_with_tokens()
904                    .filter_map(NodeOrToken::into_token)
905                    .any(|token| token.kind() == T![unsafe]);
906                let name = field_name(idx, &field);
907
908                // Check if field has default value (only for record fields)
909                let default_value = ast::RecordField::cast(field.syntax().clone())
910                    .and_then(|rf| rf.eq_token().is_some().then_some(rf.expr()))
911                    .flatten()
912                    .map(|expr| col.collect_expr_opt(Some(expr)));
913
914                arena.alloc(FieldData { name, type_ref, visibility, is_unsafe, default_value });
915                idx += 1;
916            }
917            Err(cfg) => {
918                col.store.diagnostics.push(
919                    crate::expr_store::ExpressionStoreDiagnostics::InactiveCode {
920                        node: InFile::new(fields.file_id, SyntaxNodePtr::new(field.syntax())),
921                        cfg,
922                        opts: cfg_options.clone(),
923                    },
924                );
925            }
926        }
927    }
928    if !has_fields {
929        return None;
930    }
931    let (store, source_map) = col.store.finish();
932    arena.shrink_to_fit();
933    Some((arena, store, source_map))
934}

943#[derive(Debug, Clone, PartialEq, Eq)]
944pub struct EnumVariants {
945    pub variants: Box<[(EnumVariantId, Name, FieldsShape)]>,
946}
947
948#[salsa::tracked]
949impl EnumVariants {
950    #[salsa::tracked(returns(ref))]
951    pub(crate) fn of(
952        db: &dyn DefDatabase,
953        e: EnumId,
954    ) -> (EnumVariants, Option<ThinVec<InactiveEnumVariantCode>>) {
955        let loc = e.lookup(db);
956        let source = loc.source(db);
957        let ast_id_map = db.ast_id_map(source.file_id);
958
959        let mut diagnostics = ThinVec::new();
960        let cfg_options = loc.container.krate(db).cfg_options(db);
961        let mut index = 0;
962        let Some(variants) = source.value.variant_list() else {
963            return (EnumVariants { variants: Box::default() }, None);
964        };
965        let variants = variants
966            .variants()
967            .filter_map(|variant| {
968                let ast_id = ast_id_map.ast_id(&variant);
969                match AttrFlags::is_cfg_enabled_for(&variant, cfg_options) {
970                    Ok(()) => {
971                        let enum_variant =
972                            EnumVariantLoc { id: source.with_value(ast_id), parent: e, index }
973                                .intern(db);
974                        index += 1;
975                        let name = as_name_opt(variant.name());
976                        let shape = adt_shape(variant.kind());
977                        Some((enum_variant, name, shape))
978                    }
979                    Err(cfg) => {
980                        diagnostics.push(InactiveEnumVariantCode {
981                            ast_id,
982                            cfg,
983                            opts: cfg_options.clone(),
984                        });
985                        None
986                    }
987                }
988            })
989            .collect();
990
991        (EnumVariants { variants }, diagnostics.is_empty().not().then_some(diagnostics))
992    }
993}
994
995impl EnumVariants {
996    pub fn variant(&self, name: &Name) -> Option<EnumVariantId> {
997        self.variants.iter().find_map(|(v, n, _)| if n == name { Some(*v) } else { None })
998    }
999
1000    pub fn variant_name_by_id(&self, variant_id: EnumVariantId) -> Option<Name> {
1001        self.variants
1002            .iter()
1003            .find_map(|(id, name, _)| if *id == variant_id { Some(name.clone()) } else { None })
1004    }
1005
1006    // [Adopted from rustc](https://github.com/rust-lang/rust/blob/bd53aa3bf7a24a70d763182303bd75e5fc51a9af/compiler/rustc_middle/src/ty/adt.rs#L446-L448)
1007    pub fn is_payload_free(&self, db: &dyn DefDatabase) -> bool {
1008        self.variants.iter().all(|&(v, _, _)| {
1009            // The condition check order is slightly modified from rustc
1010            // to improve performance by early returning with relatively fast checks
1011            let variant = v.fields(db);
1012            if !variant.fields().is_empty() {
1013                return false;
1014            }
1015            // The outer if condition is whether this variant has const ctor or not
1016            if !matches!(variant.shape, FieldsShape::Unit) {
1017                let body = db.body(v.into());
1018                // A variant with explicit discriminant
1019                if !matches!(body[body.body_expr], crate::hir::Expr::Missing) {
1020                    return false;
1021                }
1022            }
1023            true
1024        })
1025    }
1026}
