1//! Many kinds of items or constructs can have generic parameters: functions,
2//! structs, impls, traits, etc. This module provides a common HIR for these
3//! generic parameters. See also the `Generics` type and the `generics_of` query
4//! in rustc.
5

23pub(crate) type ImplTraitLowerFn<'l> = &'l mut dyn for<'ec, 'db> FnMut(
24    &'ec mut ExprCollector<'db>,
25    TypePtr,
26    ThinVec<TypeBound>,
27) -> TypeRefId;

29pub(crate) struct GenericParamsCollector {
30    type_or_consts: Arena<TypeOrConstParamData>,
31    lifetimes: Arena<LifetimeParamData>,
32    where_predicates: Vec<WherePredicate>,
33    parent: GenericDefId,
34}
35
36impl GenericParamsCollector {
37    pub(crate) fn new(parent: GenericDefId) -> Self {
38        Self {
39            type_or_consts: Default::default(),
40            lifetimes: Default::default(),
41            where_predicates: Default::default(),
42            parent,
43        }
44    }
45    pub(crate) fn with_self_param(
46        ec: &mut ExprCollector<'_>,
47        parent: GenericDefId,
48        bounds: Option<ast::TypeBoundList>,
49    ) -> Self {
50        let mut this = Self::new(parent);
51        this.fill_self_param(ec, bounds);
52        this
53    }
54
55    pub(crate) fn lower(
56        &mut self,
57        ec: &mut ExprCollector<'_>,
58        generic_param_list: Option<ast::GenericParamList>,
59        where_clause: Option<ast::WhereClause>,
60    ) {
61        if let Some(params) = generic_param_list {
62            self.lower_param_list(ec, params)
63        }
64        if let Some(where_clause) = where_clause {
65            self.lower_where_predicates(ec, where_clause);
66        }
67    }
68
69    pub(crate) fn collect_impl_trait<R>(
70        &mut self,
71        ec: &mut ExprCollector<'_>,
72        cb: impl FnOnce(&mut ExprCollector<'_>, ImplTraitLowerFn<'_>) -> R,
73    ) -> R {
74        cb(
75            ec,
76            &mut Self::lower_argument_impl_trait(
77                &mut self.type_or_consts,
78                &mut self.where_predicates,
79                self.parent,
80            ),
81        )
82    }
83
84    pub(crate) fn finish(self) -> GenericParams {
85        let Self { mut lifetimes, mut type_or_consts, where_predicates, parent: _ } = self;
86
87        lifetimes.shrink_to_fit();
88        type_or_consts.shrink_to_fit();
89        GenericParams {
90            type_or_consts,
91            lifetimes,
92            where_predicates: where_predicates.into_boxed_slice(),
93        }
94    }
95
96    fn lower_param_list(&mut self, ec: &mut ExprCollector<'_>, params: ast::GenericParamList) {
97        for generic_param in params.generic_params() {
98            let enabled = ec.check_cfg(&generic_param);
99            if !enabled {
100                continue;
101            }
102
103            match generic_param {
104                ast::GenericParam::TypeParam(type_param) => {
105                    let name = type_param.name().map_or_else(Name::missing, |it| it.as_name());
106                    let default = type_param.default_type().map(|it| {
107                        ec.lower_type_ref(it, &mut ExprCollector::impl_trait_error_allocator)
108                    });
109                    let param = TypeParamData {
110                        name: Some(name.clone()),
111                        default,
112                        provenance: TypeParamProvenance::TypeParamList,
113                    };
114                    let idx = self.type_or_consts.alloc(param.into());
115                    let type_ref =
116                        TypeRef::TypeParam(TypeParamId::from_unchecked(TypeOrConstParamId {
117                            parent: self.parent,
118                            local_id: idx,
119                        }));
120                    let type_ref = ec.alloc_type_ref_desugared(type_ref);
121                    self.lower_bounds(ec, type_param.type_bound_list(), Either::Left(type_ref));
122                }
123                ast::GenericParam::ConstParam(const_param) => {
124                    let name = const_param.name().map_or_else(Name::missing, |it| it.as_name());
125                    let ty = ec.lower_type_ref_opt(
126                        const_param.ty(),
127                        &mut ExprCollector::impl_trait_error_allocator,
128                    );
129                    let default = const_param.default_val().map(|it| ec.lower_const_arg(it));
130                    let param = ConstParamData { name, ty, default };
131                    let idx = self.type_or_consts.alloc(param.into());
132                    if let Some(default) = default
133                        && let Some(const_expr_origins) = &mut ec.store.inference_roots
134                    {
135                        const_expr_origins.push((
136                            default.expr,
137                            crate::expr_store::RootExprOrigin::ConstParam(idx),
138                        ));
139                    }
140                }
141                ast::GenericParam::LifetimeParam(lifetime_param) => {
142                    let lifetime = ec.lower_lifetime_ref_opt(lifetime_param.lifetime());
143                    if let LifetimeRef::Named(name) = &ec.store.lifetimes[lifetime] {
144                        let param = LifetimeParamData { name: name.clone() };
145                        let _idx = self.lifetimes.alloc(param);
146                        self.lower_bounds(
147                            ec,
148                            lifetime_param.type_bound_list(),
149                            Either::Right(lifetime),
150                        );
151                    }
152                }
153            }
154        }
155    }
156
157    fn lower_where_predicates(
158        &mut self,
159        ec: &mut ExprCollector<'_>,
160        where_clause: ast::WhereClause,
161    ) {
162        for pred in where_clause.predicates() {
163            let target = if let Some(type_ref) = pred.ty() {
164                Either::Left(
165                    ec.lower_type_ref(type_ref, &mut ExprCollector::impl_trait_error_allocator),
166                )
167            } else if let Some(lifetime) = pred.lifetime() {
168                Either::Right(ec.lower_lifetime_ref(lifetime))
169            } else {
170                continue;
171            };
172
173            let lifetimes: Option<Box<_>> =
174                pred.for_binder().and_then(|it| it.generic_param_list()).map(|param_list| {
175                    // Higher-Ranked Trait Bounds
176                    param_list
177                        .lifetime_params()
178                        .map(|lifetime_param| {
179                            lifetime_param
180                                .lifetime()
181                                .map_or_else(Name::missing, |lt| Name::new_lifetime(&lt.text()))
182                        })
183                        .collect()
184                });
185            for bound in pred.type_bound_list().iter().flat_map(|l| l.bounds()) {
186                self.lower_type_bound_as_predicate(ec, bound, lifetimes.as_deref(), target);
187            }
188        }
189    }
190
191    fn lower_bounds(
192        &mut self,
193        ec: &mut ExprCollector<'_>,
194        type_bounds: Option<ast::TypeBoundList>,
195        target: Either<TypeRefId, LifetimeRefId>,
196    ) {
197        for bound in type_bounds.iter().flat_map(|type_bound_list| type_bound_list.bounds()) {
198            self.lower_type_bound_as_predicate(ec, bound, None, target);
199        }
200    }
201
202    fn lower_type_bound_as_predicate(
203        &mut self,
204        ec: &mut ExprCollector<'_>,
205        bound: ast::TypeBound,
206        hrtb_lifetimes: Option<&[Name]>,
207        target: Either<TypeRefId, LifetimeRefId>,
208    ) {
209        let bound = ec.lower_type_bound(
210            bound,
211            &mut Self::lower_argument_impl_trait(
212                &mut self.type_or_consts,
213                &mut self.where_predicates,
214                self.parent,
215            ),
216        );
217        let predicate = match (target, bound) {
218            (_, TypeBound::Error | TypeBound::Use(_)) => return,
219            (Either::Left(type_ref), bound) => match hrtb_lifetimes {
220                Some(hrtb_lifetimes) => WherePredicate::ForLifetime {
221                    lifetimes: ThinVec::from_iter(hrtb_lifetimes.iter().cloned()),
222                    target: type_ref,
223                    bound,
224                },
225                None => WherePredicate::TypeBound { target: type_ref, bound },
226            },
227            (Either::Right(lifetime), TypeBound::Lifetime(bound)) => {
228                WherePredicate::Lifetime { target: lifetime, bound }
229            }
230            (Either::Right(_), TypeBound::ForLifetime(..) | TypeBound::Path(..)) => return,
231        };
232        self.where_predicates.push(predicate);
233    }
234
235    fn lower_argument_impl_trait(
236        type_or_consts: &mut Arena<TypeOrConstParamData>,
237        where_predicates: &mut Vec<WherePredicate>,
238        parent: GenericDefId,
239    ) -> impl for<'ec, 'db> FnMut(&'ec mut ExprCollector<'db>, TypePtr, ThinVec<TypeBound>) -> TypeRefId
240    {
241        move |ec, ptr, impl_trait_bounds| {
242            let param = TypeParamData {
243                name: None,
244                default: None,
245                provenance: TypeParamProvenance::ArgumentImplTrait,
246            };
247            let param_id = TypeRef::TypeParam(TypeParamId::from_unchecked(TypeOrConstParamId {
248                parent,
249                local_id: type_or_consts.alloc(param.into()),
250            }));
251            let type_ref = ec.alloc_type_ref(param_id, ptr);
252            for bound in impl_trait_bounds {
253                where_predicates
254                    .push(WherePredicate::TypeBound { target: type_ref, bound: bound.clone() });
255            }
256            type_ref
257        }
258    }
259
260    fn fill_self_param(&mut self, ec: &mut ExprCollector<'_>, bounds: Option<ast::TypeBoundList>) {
261        let self_ = Name::new_symbol_root(sym::Self_);
262        let idx = self.type_or_consts.alloc(
263            TypeParamData {
264                name: Some(self_),
265                default: None,
266                provenance: TypeParamProvenance::TraitSelf,
267            }
268            .into(),
269        );
270        debug_assert_eq!(idx, GenericParams::SELF_PARAM_ID_IN_SELF);
271        let type_ref = TypeRef::TypeParam(TypeParamId::from_unchecked(TypeOrConstParamId {
272            parent: self.parent,
273            local_id: idx,
274        }));
275        let self_ = ec.alloc_type_ref_desugared(type_ref);
276        if let Some(bounds) = bounds {
277            self.lower_bounds(ec, Some(bounds), Either::Left(self_));
278        }
279    }
