33macro_rules! w {
34    ($dst:expr, $($arg:tt)*) => {
35        { let _ = write!($dst, $($arg)*); }
36    };
37}
38
39macro_rules! wln {
40    ($dst:expr) => {
41        { $dst.newline(); }
42    };
43    ($dst:expr, $($arg:tt)*) => {
44        { let _ = w!($dst, $($arg)*); $dst.newline(); }
45    };
46}

55fn item_name<Id, Loc>(db: &dyn DefDatabase, id: Id, default: &str) -> String
56where
57    Id: Lookup<Database = dyn DefDatabase, Data = Loc>,
58    Loc: HasSource,
59    Loc::Value: ast::HasName,
60{
61    let loc = id.lookup(db);
62    let source = loc.source(db);
63    source.value.name().map_or_else(|| default.to_owned(), |name| name.to_string())
64}

66pub fn print_body_hir(
67    db: &dyn DefDatabase,
68    body: &Body,
69    owner: DefWithBodyId,
70    edition: Edition,
71) -> String {
72    let header = match owner {
73        DefWithBodyId::FunctionId(it) => format!("fn {}", item_name(db, it, "<missing>")),
74        DefWithBodyId::StaticId(it) => format!("static {} = ", item_name(db, it, "<missing>")),
75        DefWithBodyId::ConstId(it) => format!("const {} = ", item_name(db, it, "_")),
76        DefWithBodyId::VariantId(it) => format!(
77            "enum {}::{}",
78            item_name(db, it.lookup(db).parent, "<missing>"),
79            item_name(db, it, "<missing>")
80        ),
81    };
82
83    let mut p = Printer {
84        db,
85        store: body,
86        buf: header,
87        indent_level: 0,
88        line_format: LineFormat::Newline,
89        edition,
90    };
91    if let DefWithBodyId::FunctionId(_) = owner {
92        p.buf.push('(');
93        if let Some(self_param) = body.self_param {
94            p.print_binding(self_param);
95            p.buf.push_str(", ");
96        }
97        body.params.iter().for_each(|param| {
98            p.print_pat(*param);
99            p.buf.push_str(", ");
100        });
101        // remove the last ", " in param list
102        if !body.params.is_empty() {
103            p.buf.truncate(p.buf.len() - 2);
104        }
105        p.buf.push(')');
106        p.buf.push(' ');
107    }
108    p.print_expr(body.body_expr);
109    if matches!(owner, DefWithBodyId::StaticId(_) | DefWithBodyId::ConstId(_)) {
110        p.buf.push(';');
111    }
112    p.buf
113}

115pub fn print_variant_body_hir(db: &dyn DefDatabase, owner: VariantId, edition: Edition) -> String {
116    let header = match owner {
117        VariantId::StructId(it) => format!("struct {}", item_name(db, it, "<missing>")),
118        VariantId::EnumVariantId(it) => format!(
119            "enum {}::{}",
120            item_name(db, it.lookup(db).parent, "<missing>"),
121            item_name(db, it, "<missing>")
122        ),
123        VariantId::UnionId(it) => format!("union {}", item_name(db, it, "<missing>")),
124    };
125
126    let fields = owner.fields(db);
127
128    let mut p = Printer {
129        db,
130        store: &fields.store,
131        buf: header,
132        indent_level: 0,
133        line_format: LineFormat::Newline,
134        edition,
135    };
136    match fields.shape {
137        FieldsShape::Record => wln!(p, " {{"),
138        FieldsShape::Tuple => wln!(p, "("),
139        FieldsShape::Unit => (),
140    }
141
142    for (_, data) in fields.fields().iter() {
143        let FieldData { name, type_ref, visibility, is_unsafe, default_value: _ } = data;
144        match visibility {
145            crate::item_tree::RawVisibility::Module(interned, _visibility_explicitness) => {
146                w!(p, "pub(in {})", interned.display(db, p.edition))
147            }
148            crate::item_tree::RawVisibility::Public => w!(p, "pub "),
149            crate::item_tree::RawVisibility::PubCrate => w!(p, "pub(crate) "),
150            crate::item_tree::RawVisibility::PubSelf(_) => w!(p, "pub(self) "),
151        }
152        if *is_unsafe {
153            w!(p, "unsafe ");
154        }
155        w!(p, "{}: ", name.display(db, p.edition));
156        p.print_type_ref(*type_ref);
157    }
158
159    match fields.shape {
160        FieldsShape::Record => wln!(p, "}}"),
161        FieldsShape::Tuple => wln!(p, ");"),
162        FieldsShape::Unit => wln!(p, ";"),
163    }
164    p.buf
165}

167pub fn print_signature(db: &dyn DefDatabase, owner: GenericDefId, edition: Edition) -> String {
168    match owner {
169        GenericDefId::AdtId(id) => match id {
170            AdtId::StructId(id) => {
171                let signature = db.struct_signature(id);
172                print_struct(db, id, &signature, edition)
173            }
174            AdtId::UnionId(id) => {
175                format!("unimplemented {id:?}")
176            }
177            AdtId::EnumId(id) => {
178                format!("unimplemented {id:?}")
179            }
180        },
181        GenericDefId::ConstId(id) => format!("unimplemented {id:?}"),
182        GenericDefId::FunctionId(id) => {
183            let signature = db.function_signature(id);
184            print_function(db, id, &signature, edition)
185        }
186        GenericDefId::ImplId(id) => format!("unimplemented {id:?}"),
187        GenericDefId::StaticId(id) => format!("unimplemented {id:?}"),
188        GenericDefId::TraitId(id) => format!("unimplemented {id:?}"),
189        GenericDefId::TypeAliasId(id) => format!("unimplemented {id:?}"),
190    }
191}

211pub fn print_struct(
212    db: &dyn DefDatabase,
213    id: StructId,
214    StructSignature { name, generic_params, store, flags, shape }: &StructSignature,
215    edition: Edition,
216) -> String {
217    let mut p = Printer {
218        db,
219        store,
220        buf: String::new(),
221        indent_level: 0,
222        line_format: LineFormat::Newline,
223        edition,
224    };
225    if let Some(repr) = AttrFlags::repr(db, id.into()) {
226        if repr.c() {
227            wln!(p, "#[repr(C)]");
228        }
229        if let Some(align) = repr.align {
230            wln!(p, "#[repr(align({}))]", align.bytes());
231        }
232        if let Some(pack) = repr.pack {
233            wln!(p, "#[repr(pack({}))]", pack.bytes());
234        }
235    }
236    if flags.contains(StructFlags::FUNDAMENTAL) {
237        wln!(p, "#[fundamental]");
238    }
239    w!(p, "struct ");
240    w!(p, "{}", name.display(db, edition));
241    print_generic_params(db, generic_params, &mut p);
242    match shape {
243        FieldsShape::Record => wln!(p, " {{...}}"),
244        FieldsShape::Tuple => wln!(p, "(...)"),
245        FieldsShape::Unit => (),
246    }
247
248    print_where_clauses(db, generic_params, &mut p);
249
250    match shape {
251        FieldsShape::Record => wln!(p),
252        FieldsShape::Tuple => wln!(p, ";"),
253        FieldsShape::Unit => wln!(p, ";"),
254    }
255
256    p.buf
257}

259pub fn print_function(
260    db: &dyn DefDatabase,
261    id: FunctionId,
262    signature @ FunctionSignature {
263        name,
264        generic_params,
265        store,
266        params,
267        ret_type,
268        abi,
269        flags,
270    }: &FunctionSignature,
271    edition: Edition,
272) -> String {
273    let legacy_const_generics_indices = signature.legacy_const_generics_indices(db, id);
274    let mut p = Printer {
275        db,
276        store,
277        buf: String::new(),
278        indent_level: 0,
279        line_format: LineFormat::Newline,
280        edition,
281    };
282    if flags.contains(FnFlags::CONST) {
283        w!(p, "const ");
284    }
285    if flags.contains(FnFlags::ASYNC) {
286        w!(p, "async ");
287    }
288    if flags.contains(FnFlags::UNSAFE) {
289        w!(p, "unsafe ");
290    }
291    if flags.contains(FnFlags::EXPLICIT_SAFE) {
292        w!(p, "safe ");
293    }
294    if let Some(abi) = abi {
295        w!(p, "extern \"{}\" ", abi.as_str());
296    }
297    w!(p, "fn ");
298    w!(p, "{}", name.display(db, edition));
299    print_generic_params(db, generic_params, &mut p);
300    w!(p, "(");
301    for (i, param) in params.iter().enumerate() {
302        if i != 0 {
303            w!(p, ", ");
304        }
305        if legacy_const_generics_indices.is_some_and(|idx| idx.contains(&(i as u32))) {
306            w!(p, "const: ");
307        }
308        p.print_type_ref(*param);
309    }
310    w!(p, ")");
311    if let Some(ret_type) = ret_type {
312        w!(p, " -> ");
313        p.print_type_ref(*ret_type);
314    }
315
316    print_where_clauses(db, generic_params, &mut p);
317    wln!(p, " {{...}}");
318
319    p.buf
320}

322fn print_where_clauses(db: &dyn DefDatabase, generic_params: &GenericParams, p: &mut Printer<'_>) {
323    if !generic_params.where_predicates.is_empty() {
324        w!(p, "\nwhere\n");
325        p.indented(|p| {
326            for (i, pred) in generic_params.where_predicates.iter().enumerate() {
327                if i != 0 {
328                    w!(p, ",\n");
329                }
330                match pred {
331                    WherePredicate::TypeBound { target, bound } => {
332                        p.print_type_ref(*target);
333                        w!(p, ": ");
334                        p.print_type_bounds(std::slice::from_ref(bound));
335                    }
336                    WherePredicate::Lifetime { target, bound } => {
337                        p.print_lifetime_ref(*target);
338                        w!(p, ": ");
339                        p.print_lifetime_ref(*bound);
340                    }
341                    WherePredicate::ForLifetime { lifetimes, target, bound } => {
342                        w!(p, "for<");
343                        for (i, lifetime) in lifetimes.iter().enumerate() {
344                            if i != 0 {
345                                w!(p, ", ");
346                            }
347                            w!(p, "{}", lifetime.display(db, p.edition));
348                        }
349                        w!(p, "> ");
350                        p.print_type_ref(*target);
351                        w!(p, ": ");
352                        p.print_type_bounds(std::slice::from_ref(bound));
353                    }
354                }
355            }
356        });
357        wln!(p);
358    }
359}

361fn print_generic_params(db: &dyn DefDatabase, generic_params: &GenericParams, p: &mut Printer<'_>) {
362    if !generic_params.is_empty() {
363        w!(p, "<");
364        let mut first = true;
365        for (_i, param) in generic_params.iter_lt() {
366            if !first {
367                w!(p, ", ");
368            }
369            first = false;
370            w!(p, "{}", param.name.display(db, p.edition));
371        }
372        for (i, param) in generic_params.iter_type_or_consts() {
373            if !first {
374                w!(p, ", ");
375            }
376            first = false;
377            if let Some(const_param) = param.const_param() {
378                w!(p, "const {}: ", const_param.name.display(db, p.edition));
379                p.print_type_ref(const_param.ty);
380                if let Some(default) = const_param.default {
381                    w!(p, " = ");
382                    p.print_expr(default.expr);
383                }
384            }
385            if let Some(type_param) = param.type_param() {
386                match &type_param.name {
387                    Some(name) => w!(p, "{}", name.display(db, p.edition)),
388                    None => w!(p, "Param[{}]", i.into_raw()),
389                }
390                if let Some(default) = type_param.default {
391                    w!(p, " = ");
392                    p.print_type_ref(default);
393                }
394            }
395        }
396        w!(p, ">");
397    }
398}

439struct Printer<'a> {
440    db: &'a dyn DefDatabase,
441    store: &'a ExpressionStore,
442    buf: String,
443    indent_level: usize,
444    line_format: LineFormat,
445    edition: Edition,
446}
447
448impl Write for Printer<'_> {
449    fn write_str(&mut self, s: &str) -> fmt::Result {
450        for line in s.split_inclusive('\n') {
451            if matches!(self.line_format, LineFormat::Indentation) {
452                match self.buf.chars().rev().find(|ch| *ch != ' ') {
453                    Some('\n') | None => {}
454                    _ => self.buf.push('\n'),
455                }
456                self.buf.push_str(&"    ".repeat(self.indent_level));
457            }
458
459            self.buf.push_str(line);
460
461            if matches!(self.line_format, LineFormat::Newline | LineFormat::Indentation) {
462                self.line_format = if line.ends_with('\n') {
463                    LineFormat::Indentation
464                } else {
465                    LineFormat::Newline
466                };
467            }
468        }
469
470        Ok(())
471    }
472}

474impl Printer<'_> {
475    fn indented(&mut self, f: impl FnOnce(&mut Self)) {
476        self.indent_level += 1;
477        wln!(self);
478        f(self);
479        self.indent_level -= 1;
480        self.buf = self.buf.trim_end_matches('\n').to_owned();
481    }
482
483    fn whitespace(&mut self) {
484        match self.buf.chars().next_back() {
485            None | Some('\n' | ' ') => {}
486            _ => self.buf.push(' '),
487        }
488    }
489
490    // Add a newline if the current line is not empty.
491    // If the current line is empty, add a space instead.
492    //
493    // Do not use [`writeln!()`] or [`wln!()`] here, which will result in
494    // infinite recursive calls to this function.
495    fn newline(&mut self) {
496        if matches!(self.line_format, LineFormat::Oneline) {
497            match self.buf.chars().last() {
498                Some(' ') | None => {}
499                Some(_) => {
500                    w!(self, " ");
501                }
502            }
503        } else {
504            match self.buf.chars().rev().find_position(|ch| *ch != ' ') {
505                Some((_, '\n')) | None => {}
506                Some((idx, _)) => {
507                    if idx != 0 {
508                        self.buf.drain(self.buf.len() - idx..);
509                    }
510                    w!(self, "\n");
511                }
512            }
513        }
514    }
515
516    fn print_expr(&mut self, expr: ExprId) {
517        self.print_expr_in(None, expr);
518    }
519
520    fn print_expr_in(&mut self, prec: Option<ast::prec::ExprPrecedence>, expr: ExprId) {
521        let expr = &self.store[expr];
522        let needs_parens = match (prec, expr.precedence()) {
523            (Some(ast::prec::ExprPrecedence::LOr), ast::prec::ExprPrecedence::LOr) => false,
524            (Some(ast::prec::ExprPrecedence::LAnd), ast::prec::ExprPrecedence::LAnd) => false,
525            (Some(parent), prec) => prec.needs_parentheses_in(parent),
526            (None, _) => false,
527        };
528        let prec = Some(expr.precedence());
529
530        if needs_parens {
531            w!(self, "(");
532        }
533
534        match expr {
535            Expr::Missing => w!(self, ""),
536            Expr::Underscore => w!(self, "_"),
537            Expr::InlineAsm(_) => w!(self, "builtin#asm(_)"),
538            Expr::OffsetOf(offset_of) => {
539                w!(self, "builtin#offset_of(");
540                self.print_type_ref(offset_of.container);
541                let edition = self.edition;
542                w!(
543                    self,
544                    ", {})",
545                    offset_of
546                        .fields
547                        .iter()
548                        .format_with(".", |field, f| f(&field.display(self.db, edition)))
549                );
550            }
551            Expr::Path(path) => self.print_path(path),
552            Expr::If { condition, then_branch, else_branch } => {
553                w!(self, "if ");
554                self.print_expr(*condition);
555                w!(self, " ");
556                self.print_expr(*then_branch);
557                if let Some(els) = *else_branch {
558                    w!(self, " else ");
559                    self.print_expr(els);
560                }
561            }
562            Expr::Let { pat, expr } => {
563                w!(self, "let ");
564                self.print_pat(*pat);
565                w!(self, " = ");
566                self.print_expr_in(prec, *expr);
567            }
568            Expr::Loop { body, label } => {
569                if let Some(lbl) = label {
570                    w!(self, "{}: ", self.store[*lbl].name.display(self.db, self.edition));
571                }
572                w!(self, "loop ");
573                self.print_expr(*body);
574            }
575            Expr::Call { callee, args } => {
576                self.print_expr_in(prec, *callee);
577                w!(self, "(");
578                if !args.is_empty() {
579                    self.indented(|p| {
580                        for arg in &**args {
581                            p.print_expr(*arg);
582                            wln!(p, ",");
583                        }
584                    });
585                }
586                w!(self, ")");
587            }
588            Expr::MethodCall { receiver, method_name, args, generic_args } => {
589                self.print_expr_in(prec, *receiver);
590                w!(self, ".{}", method_name.display(self.db, self.edition));
591                if let Some(args) = generic_args {
592                    w!(self, "::<");
593                    self.print_generic_args(args);
594                    w!(self, ">");
595                }
596                w!(self, "(");
597                if !args.is_empty() {
598                    self.indented(|p| {
599                        for arg in &**args {
600                            p.print_expr(*arg);
601                            wln!(p, ",");
602                        }
603                    });
604                }
605                w!(self, ")");
606            }
607            Expr::Match { expr, arms } => {
608                w!(self, "match ");
609                self.print_expr(*expr);
610                w!(self, " {{");
611                self.indented(|p| {
612                    for arm in &**arms {
613                        p.print_pat(arm.pat);
614                        if let Some(guard) = arm.guard {
615                            w!(p, " if ");
616                            p.print_expr(guard);
617                        }
618                        w!(p, " => ");
619                        p.print_expr(arm.expr);
620                        wln!(p, ",");
621                    }
622                });
623                wln!(self, "}}");
624            }
625            Expr::Continue { label } => {
626                w!(self, "continue");
627                if let Some(lbl) = label {
628                    w!(self, " {}", self.store[*lbl].name.display(self.db, self.edition));
629                }
630            }
631            Expr::Break { expr, label } => {
632                w!(self, "break");
633                if let Some(lbl) = label {
634                    w!(self, " {}", self.store[*lbl].name.display(self.db, self.edition));
635                }
636                if let Some(expr) = expr {
637                    self.whitespace();
638                    self.print_expr_in(prec, *expr);
639                }
640            }
641            Expr::Return { expr } => {
642                w!(self, "return");
643                if let Some(expr) = expr {
644                    self.whitespace();
645                    self.print_expr_in(prec, *expr);
646                }
647            }
648            Expr::Become { expr } => {
649                w!(self, "become");
650                self.whitespace();
651                self.print_expr_in(prec, *expr);
652            }
653            Expr::Yield { expr } => {
654                w!(self, "yield");
655                if let Some(expr) = expr {
656                    self.whitespace();
657                    self.print_expr_in(prec, *expr);
658                }
659            }
660            Expr::Yeet { expr } => {
661                w!(self, "do");
662                self.whitespace();
663                w!(self, "yeet");
664                if let Some(expr) = expr {
665                    self.whitespace();
666                    self.print_expr_in(prec, *expr);
667                }
668            }
669            Expr::RecordLit { path, fields, spread } => {
670                match path {
671                    Some(path) => self.print_path(path),
672                    None => w!(self, ""),
673                }
674
675                w!(self, "{{");
676                let edition = self.edition;
677                self.indented(|p| {
678                    for field in &**fields {
679                        w!(p, "{}: ", field.name.display(self.db, edition));
680                        p.print_expr(field.expr);
681                        wln!(p, ",");
682                    }
683                    match spread {
684                        RecordSpread::None => {}
685                        RecordSpread::FieldDefaults => {
686                            w!(p, "..");
687                            wln!(p);
688                        }
689                        RecordSpread::Expr(spread_expr) => {
690                            w!(p, "..");
691                            p.print_expr(*spread_expr);
692                            wln!(p);
693                        }
694                    }
695                });
696                w!(self, "}}");
697            }
698            Expr::Field { expr, name } => {
699                self.print_expr_in(prec, *expr);
700                w!(self, ".{}", name.display(self.db, self.edition));
701            }
702            Expr::Await { expr } => {
703                self.print_expr_in(prec, *expr);
704                w!(self, ".await");
705            }
706            Expr::Cast { expr, type_ref } => {
707                self.print_expr_in(prec, *expr);
708                w!(self, " as ");
709                self.print_type_ref(*type_ref);
710            }
711            Expr::Ref { expr, rawness, mutability } => {
712                w!(self, "&");
713                if rawness.is_raw() {
714                    w!(self, "raw ");
715                }
716                if mutability.is_mut() {
717                    w!(self, "mut ");
718                }
719                self.print_expr_in(prec, *expr);
720            }
721            Expr::Box { expr } => {
722                w!(self, "box ");
723                self.print_expr_in(prec, *expr);
724            }
725            Expr::UnaryOp { expr, op } => {
726                let op = match op {
727                    ast::UnaryOp::Deref => "*",
728                    ast::UnaryOp::Not => "!",
729                    ast::UnaryOp::Neg => "-",
730                };
731                w!(self, "{}", op);
732                self.print_expr_in(prec, *expr);
733            }
734            Expr::BinaryOp { lhs, rhs, op } => {
735                self.print_expr_in(prec, *lhs);
736                self.whitespace();
737                match op {
738                    Some(op) => w!(self, "{}", op),
739                    None => w!(self, ""), // :)
740                }
741                self.whitespace();
742                self.print_expr_in(prec, *rhs);
743            }
744            Expr::Range { lhs, rhs, range_type } => {
745                if let Some(lhs) = lhs {
746                    self.print_expr_in(prec, *lhs);
747                }
748                match range_type {
749                    RangeOp::Exclusive => w!(self, ".."),
750                    RangeOp::Inclusive => w!(self, "..="),
751                };
752                if let Some(rhs) = rhs {
753                    self.print_expr_in(prec, *rhs);
754                }
755            }
756            Expr::Index { base, index } => {
757                self.print_expr_in(prec, *base);
758                w!(self, "[");
759                self.print_expr(*index);
760                w!(self, "]");
761            }
762            Expr::Closure { args, arg_types, ret_type, body, closure_kind, capture_by } => {
763                match closure_kind {
764                    ClosureKind::Coroutine(Movability::Static) => {
765                        w!(self, "static ");
766                    }
767                    ClosureKind::Async => {
768                        w!(self, "async ");
769                    }
770                    _ => (),
771                }
772                match capture_by {
773                    CaptureBy::Value => {
774                        w!(self, "move ");
775                    }
776                    CaptureBy::Ref => (),
777                }
778                w!(self, "|");
779                for (i, (pat, ty)) in args.iter().zip(arg_types.iter()).enumerate() {
780                    if i != 0 {
781                        w!(self, ", ");
782                    }
783                    self.print_pat(*pat);
784                    if let Some(ty) = ty {
785                        w!(self, ": ");
786                        self.print_type_ref(*ty);
787                    }
788                }
789                w!(self, "|");
790                if let Some(ret_ty) = ret_type {
791                    w!(self, " -> ");
792                    self.print_type_ref(*ret_ty);
793                }
794                self.whitespace();
795                self.print_expr(*body);
796            }
797            Expr::Tuple { exprs } => {
798                w!(self, "(");
799                for expr in exprs.iter() {
800                    self.print_expr(*expr);
801                    w!(self, ", ");
802                }
803                w!(self, ")");
804            }
805            Expr::Array(arr) => {
806                w!(self, "[");
807                if !matches!(arr, Array::ElementList { elements, .. } if elements.is_empty()) {
808                    self.indented(|p| match arr {
809                        Array::ElementList { elements } => {
810                            for elem in elements.iter() {
811                                p.print_expr(*elem);
812                                w!(p, ", ");
813                            }
814                        }
815                        Array::Repeat { initializer, repeat } => {
816                            p.print_expr(*initializer);
817                            w!(p, "; ");
818                            p.print_expr(*repeat);
819                        }
820                    });
821                    self.newline();
822                }
823                w!(self, "]");
824            }
825            Expr::Literal(lit) => self.print_literal(lit),
826            Expr::Block { id: _, statements, tail, label } => {
827                let label = label.map(|lbl| {
828                    format!("{}: ", self.store[lbl].name.display(self.db, self.edition))
829                });
830                self.print_block(label.as_deref(), statements, tail);
831            }
832            Expr::Unsafe { id: _, statements, tail } => {
833                self.print_block(Some("unsafe "), statements, tail);
834            }
835            Expr::Async { id: _, statements, tail } => {
836                self.print_block(Some("async "), statements, tail);
837            }
838            Expr::Const(id) => {
839                w!(self, "const {{ /* {id:?} */ }}");
840            }
841            &Expr::Assignment { target, value } => {
842                self.print_pat(target);
843                w!(self, " = ");
844                self.print_expr_in(prec, value);
845            }
846        }
847
848        if needs_parens {
849            w!(self, ")");
850        }
851    }

853    fn print_block(
854        &mut self,
855        label: Option<&str>,
856        statements: &[Statement],
857        tail: &Option<la_arena::Idx<Expr>>,
858    ) {
859        self.whitespace();
860        if let Some(lbl) = label {
861            w!(self, "{}", lbl);
862        }
863        w!(self, "{{");
864        if !statements.is_empty() || tail.is_some() {
865            self.indented(|p| {
866                for stmt in statements {
867                    p.print_stmt(stmt);
868                }
869                if let Some(tail) = tail {
870                    p.print_expr(*tail);
871                }
872                p.newline();
873            });
874        }
875        w!(self, "}}");
876    }

878    fn print_pat(&mut self, pat: PatId) {
879        let prec = Some(ast::prec::ExprPrecedence::Shift);
880        let pat = &self.store[pat];
881
882        match pat {
883            Pat::Missing => w!(self, ""),
884            Pat::Wild => w!(self, "_"),
885            Pat::Tuple { args, ellipsis } => {
886                w!(self, "(");
887                for (i, pat) in args.iter().enumerate() {
888                    if i != 0 {
889                        w!(self, ", ");
890                    }
891                    if *ellipsis == Some(i as u32) {
892                        w!(self, ".., ");
893                    }
894                    self.print_pat(*pat);
895                }
896                w!(self, ")");
897            }
898            Pat::Or(pats) => {
899                w!(self, "(");
900                for (i, pat) in pats.iter().enumerate() {
901                    if i != 0 {
902                        w!(self, " | ");
903                    }
904                    self.print_pat(*pat);
905                }
906                w!(self, ")");
907            }
908            Pat::Record { path, args, ellipsis } => {
909                match path {
910                    Some(path) => self.print_path(path),
911                    None => w!(self, ""),
912                }
913
914                w!(self, " {{");
915                let edition = self.edition;
916                let oneline = matches!(self.line_format, LineFormat::Oneline);
917                self.indented(|p| {
918                    for (idx, arg) in args.iter().enumerate() {
919                        let field_name = arg.name.display(self.db, edition).to_string();
920
921                        let mut same_name = false;
922                        if let Pat::Bind { id, subpat: None } = &self.store[arg.pat]
923                            && let Binding { name, mode: BindingAnnotation::Unannotated, .. } =
924                                &self.store.assert_expr_only().bindings[*id]
925                            && name.as_str() == field_name
926                        {
927                            same_name = true;
928                        }
929
930                        w!(p, "{}", field_name);
931
932                        if !same_name {
933                            w!(p, ": ");
934                            p.print_pat(arg.pat);
935                        }
936
937                        // Do not print the extra comma if the line format is oneline
938                        if oneline && idx == args.len() - 1 {
939                            w!(p, " ");
940                        } else {
941                            wln!(p, ",");
942                        }
943                    }
944
945                    if *ellipsis {
946                        wln!(p, "..");
947                    }
948                });
949                w!(self, "}}");
950            }
951            Pat::Range { start, end, range_type } => {
952                if let Some(start) = start {
953                    self.print_expr_in(prec, *start);
954                }
955                match range_type {
956                    RangeOp::Inclusive => w!(self, "..="),
957                    RangeOp::Exclusive => w!(self, ".."),
958                }
959                if let Some(end) = end {
960                    self.print_expr_in(prec, *end);
961                }
962            }
963            Pat::Slice { prefix, slice, suffix } => {
964                w!(self, "[");
965                for pat in prefix.iter() {
966                    self.print_pat(*pat);
967                    w!(self, ", ");
968                }
969                if let Some(pat) = slice {
970                    self.print_pat(*pat);
971                    w!(self, ", ");
972                }
973                for pat in suffix.iter() {
974                    self.print_pat(*pat);
975                    w!(self, ", ");
976                }
977                w!(self, "]");
978            }
979            Pat::Path(path) => self.print_path(path),
980            Pat::Lit(expr) => self.print_expr_in(prec, *expr),
981            Pat::Bind { id, subpat } => {
982                self.print_binding(*id);
983                if let Some(pat) = subpat {
984                    self.whitespace();
985                    w!(self, "@ ");
986                    self.print_pat(*pat);
987                }
988            }
989            Pat::TupleStruct { path, args, ellipsis } => {
990                match path {
991                    Some(path) => self.print_path(path),
992                    None => w!(self, ""),
993                }
994                w!(self, "(");
995                for (i, arg) in args.iter().enumerate() {
996                    if i != 0 {
997                        w!(self, ", ");
998                    }
999                    if *ellipsis == Some(i as u32) {
1000                        w!(self, ", ..");
1001                    }
1002                    self.print_pat(*arg);
1003                }
1004                w!(self, ")");
1005            }
1006            Pat::Ref { pat, mutability } => {
1007                w!(self, "&");
1008                if mutability.is_mut() {
1009                    w!(self, "mut ");
1010                }
1011                self.print_pat(*pat);
1012            }
1013            Pat::Box { inner } => {
1014                w!(self, "box ");
1015                self.print_pat(*inner);
1016            }
1017            Pat::ConstBlock(c) => {
1018                w!(self, "const ");
1019                self.print_expr(*c);
1020            }
1021            Pat::Expr(expr) => {
1022                self.print_expr_in(prec, *expr);
1023            }
1024        }
1025    }

1027    fn print_stmt(&mut self, stmt: &Statement) {
1028        match stmt {
1029            Statement::Let { pat, type_ref, initializer, else_branch } => {
1030                w!(self, "let ");
1031                self.print_pat(*pat);
1032                if let Some(ty) = type_ref {
1033                    w!(self, ": ");
1034                    self.print_type_ref(*ty);
1035                }
1036                if let Some(init) = initializer {
1037                    w!(self, " = ");
1038                    self.print_expr(*init);
1039                }
1040                if let Some(els) = else_branch {
1041                    w!(self, " else ");
1042                    self.print_expr(*els);
1043                }
1044                wln!(self, ";");
1045            }
1046            Statement::Expr { expr, has_semi } => {
1047                self.print_expr(*expr);
1048                if *has_semi {
1049                    w!(self, ";");
1050                }
1051                wln!(self);
1052            }
1053            Statement::Item(_) => (),
1054        }
1055    }

1057    fn print_literal(&mut self, literal: &Literal) {
1058        match literal {
1059            Literal::String(it) => w!(self, "{:?}", it),
1060            Literal::ByteString(it) => w!(self, "\"{}\"", it.escape_ascii()),
1061            Literal::CString(it) => w!(self, "\"{}\\0\"", it.escape_ascii()),
1062            Literal::Char(it) => w!(self, "'{}'", it.escape_debug()),
1063            Literal::Bool(it) => w!(self, "{}", it),
1064            Literal::Int(i, suffix) => {
1065                w!(self, "{}", i);
1066                if let Some(suffix) = suffix {
1067                    w!(self, "{}", suffix);
1068                }
1069            }
1070            Literal::Uint(i, suffix) => {
1071                w!(self, "{}", i);
1072                if let Some(suffix) = suffix {
1073                    w!(self, "{}", suffix);
1074                }
1075            }
1076            Literal::Float(f, suffix) => {
1077                w!(self, "{}", f);
1078                if let Some(suffix) = suffix {
1079                    w!(self, "{}", suffix);
1080                }
1081            }
1082        }
1083    }

1085    fn print_binding(&mut self, id: BindingId) {

1089            BindingAnnotation::Mutable => "mut ",

1096    fn print_path(&mut self, path: &Path) {
1097        if let Path::LangItem(it, s) = path {
1098            w!(self, "builtin#lang(");
1099            macro_rules! write_name {
1100                ($it:ident) => {{
1101                    w!(self, "{}", item_name(self.db, $it, "<missing>"));
1102                }};
1103            }
1104            match *it {
1105                LangItemTarget::ImplId(it) => w!(self, "{it:?}"),
1106                LangItemTarget::EnumId(it) => write_name!(it),
1107                LangItemTarget::FunctionId(it) => write_name!(it),
1108                LangItemTarget::StaticId(it) => write_name!(it),
1109                LangItemTarget::StructId(it) => write_name!(it),
1110                LangItemTarget::UnionId(it) => write_name!(it),
1111                LangItemTarget::TypeAliasId(it) => write_name!(it),
1112                LangItemTarget::TraitId(it) => write_name!(it),
1113                LangItemTarget::EnumVariantId(it) => write_name!(it),
1114            }
1115
1116            if let Some(s) = s {
1117                w!(self, "::{}", s.display(self.db, self.edition));
1118            }
1119            return w!(self, ")");
1120        }
1121        match path.type_anchor() {
1122            Some(anchor) => {
1123                w!(self, "<");
1124                self.print_type_ref(anchor);
1125                w!(self, ">::");
1126            }
1127            None => match path.kind() {
1128                PathKind::Plain => {}
1129                &PathKind::SELF => w!(self, "self"),
1130                PathKind::Super(n) => {
1131                    for i in 0..*n {
1132                        if i == 0 {
1133                            w!(self, "super");
1134                        } else {
1135                            w!(self, "::super");
1136                        }
1137                    }
1138                }
1139                PathKind::Crate => w!(self, "crate"),
1140                PathKind::Abs => {}
1141                PathKind::DollarCrate(krate) => w!(
1142                    self,
1143                    "{}",
1144                    krate
1145                        .extra_data(self.db)
1146                        .display_name
1147                        .as_ref()
1148                        .map(|it| it.crate_name().symbol().as_str())
1149                        .unwrap_or("$crate")
1150                ),
1151            },
1152        }
1153
1154        for (i, segment) in path.segments().iter().enumerate() {
1155            if i != 0 || !matches!(path.kind(), PathKind::Plain) {
1156                w!(self, "::");
1157            }
1158
1159            w!(self, "{}", segment.name.display(self.db, self.edition));
1160            if let Some(generics) = segment.args_and_bindings {
1161                w!(self, "::<");
1162                self.print_generic_args(generics);
1163
1164                w!(self, ">");
1165            }
1166        }
1167    }

1169    pub(crate) fn print_generic_args(&mut self, generics: &GenericArgs) {
1170        let mut first = true;
1171        let args = if generics.has_self_type {
1172            let (self_ty, args) = generics.args.split_first().unwrap();
1173            w!(self, "Self=");
1174            self.print_generic_arg(self_ty);
1175            first = false;
1176            args
1177        } else {
1178            &generics.args
1179        };
1180        for arg in args {
1181            if !first {
1182                w!(self, ", ");
1183            }
1184            first = false;
1185            self.print_generic_arg(arg);
1186        }
1187        for binding in generics.bindings.iter() {
1188            if !first {
1189                w!(self, ", ");
1190            }
1191            first = false;
1192            w!(self, "{}", binding.name.display(self.db, self.edition));
1193            if !binding.bounds.is_empty() {
1194                w!(self, ": ");
1195                self.print_type_bounds(&binding.bounds);
1196            }
1197            if let Some(ty) = binding.type_ref {
1198                w!(self, " = ");
1199                self.print_type_ref(ty);
1200            }
1201        }
1202    }

1240    pub(crate) fn print_type_ref(&mut self, type_ref: TypeRefId) {
1241        // FIXME: deduplicate with `HirDisplay` impl
1242        match &self.store[type_ref] {
1243            TypeRef::Never => w!(self, "!"),
1244            &TypeRef::TypeParam(p) => self.print_type_param(p),
1245            TypeRef::Placeholder => w!(self, "_"),
1246            TypeRef::Tuple(fields) => {
1247                w!(self, "(");
1248                for (i, field) in fields.iter().enumerate() {
1249                    if i != 0 {
1250                        w!(self, ", ");
1251                    }
1252                    self.print_type_ref(*field);
1253                }
1254                w!(self, ")");
1255            }
1256            TypeRef::Path(path) => self.print_path(path),
1257            TypeRef::RawPtr(pointee, mtbl) => {
1258                let mtbl = match mtbl {
1259                    Mutability::Shared => "*const",
1260                    Mutability::Mut => "*mut",
1261                };
1262                w!(self, "{mtbl} ");
1263                self.print_type_ref(*pointee);
1264            }
1265            TypeRef::Reference(ref_) => {
1266                let mtbl = match ref_.mutability {
1267                    Mutability::Shared => "",
1268                    Mutability::Mut => "mut ",
1269                };
1270                w!(self, "&");
1271                if let Some(lt) = &ref_.lifetime {
1272                    self.print_lifetime_ref(*lt);
1273                    w!(self, " ");
1274                }
1275                w!(self, "{mtbl}");
1276                self.print_type_ref(ref_.ty);
1277            }
1278            TypeRef::Array(array) => {
1279                w!(self, "[");
1280                self.print_type_ref(array.ty);
1281                w!(self, "; ");
1282                self.print_generic_arg(&GenericArg::Const(array.len));
1283                w!(self, "]");
1284            }
1285            TypeRef::Slice(elem) => {
1286                w!(self, "[");
1287                self.print_type_ref(*elem);
1288                w!(self, "]");
1289            }
1290            TypeRef::Fn(fn_) => {
1291                let ((_, return_type), args) =
1292                    fn_.params.split_last().expect("TypeRef::Fn is missing return type");
1293                if fn_.is_unsafe {
1294                    w!(self, "unsafe ");
1295                }
1296                if let Some(abi) = &fn_.abi {
1297                    w!(self, "extern ");
1298                    w!(self, "{}", abi.as_str());
1299                    w!(self, " ");
1300                }
1301                w!(self, "fn(");
1302                for (i, (_, typeref)) in args.iter().enumerate() {
1303                    if i != 0 {
1304                        w!(self, ", ");
1305                    }
1306                    self.print_type_ref(*typeref);
1307                }
1308                if fn_.is_varargs {
1309                    if !args.is_empty() {
1310                        w!(self, ", ");
1311                    }
1312                    w!(self, "...");
1313                }
1314                w!(self, ") -> ");
1315                self.print_type_ref(*return_type);
1316            }
1317            TypeRef::Error => w!(self, "{{error}}"),
1318            TypeRef::ImplTrait(bounds) => {
1319                w!(self, "impl ");
1320                self.print_type_bounds(bounds);
1321            }
1322            TypeRef::DynTrait(bounds) => {
1323                w!(self, "dyn ");
1324                self.print_type_bounds(bounds);
1325            }
1326        }
1327    }

1329    pub(crate) fn print_type_bounds(&mut self, bounds: &[TypeBound]) {
1330        for (i, bound) in bounds.iter().enumerate() {
1331            if i != 0 {
1332                w!(self, " + ");
1333            }
1334
1335            match bound {
1336                TypeBound::Path(path, modifier) => {
1337                    match modifier {
1338                        TraitBoundModifier::None => (),
1339                        TraitBoundModifier::Maybe => w!(self, "?"),
1340                    }
1341                    self.print_path(&self.store[*path]);
1342                }
1343                TypeBound::ForLifetime(lifetimes, path) => {
1344                    w!(
1345                        self,
1346                        "for<{}> ",
1347                        lifetimes
1348                            .iter()
1349                            .map(|it| it.display(self.db, self.edition))
1350                            .format(", ")
1351                            .to_string()
1352                    );
1353                    self.print_path(&self.store[*path]);
1354                }
1355                TypeBound::Lifetime(lt) => self.print_lifetime_ref(*lt),
1356                TypeBound::Use(args) => {
1357                    w!(self, "use<");
1358                    let mut first = true;
1359                    for arg in args {
1360                        if !mem::take(&mut first) {
1361                            w!(self, ", ");
1362                        }
1363                        match arg {
1364                            UseArgRef::Name(it) => {
1365                                w!(self, "{}", it.display(self.db, self.edition))
1366                            }
1367                            UseArgRef::Lifetime(it) => self.print_lifetime_ref(*it),
1368                        }
1369                    }
1370                    w!(self, ">")
1371                }
1372                TypeBound::Error => w!(self, "{{unknown}}"),
1373            }
1374        }
1375    }
