20#[derive(Debug, Clone, Copy, PartialEq, Eq)]
21pub struct Param<Id> {
22    /// The id of the formal parameter. In analysis, you want to use this: it has a special status as a parameter,
23    /// while [`user_written`][Self::user_written] is just a local variable.
24    pub formal: Id,
25    /// The id that corresponds to the pattern the user wrote. In IDE, you usually want to use this.
26    ///
27    /// It will be different than [`formal`][Self::formal] for coroutine fns (async fn etc.), because we apply
28    /// a desugaring for those that copies the parameter into a local variable.
29    pub user_written: Id,
30}
31
32impl<Id: Copy> Param<Id> {
33    pub(crate) fn new(id: Id) -> Self {
34        Self { formal: id, user_written: id }
35    }
36}

38/// The body of an item (function, const etc.).
39#[derive(Debug, Eq, PartialEq)]
40pub struct Body {
41    pub store: ExpressionStore,
42    /// The patterns for the function's parameters. While the parameter types are
43    /// part of the function signature, the patterns are not (they don't change
44    /// the external type of the function).
45    ///
46    /// If this `Body` is for the body of a constant, this will just be
47    /// empty.
48    pub params: Box<[Param<PatId>]>,
49    pub self_param: Option<Param<BindingId>>,
50}
51
52impl ops::Deref for Body {
53    type Target = ExpressionStore;
54
55    #[inline]
56    fn deref(&self) -> &Self::Target {
57        &self.store
58    }
59}

61/// An item body together with the mapping from syntax nodes to HIR expression
62/// IDs. This is needed to go from e.g. a position in a file to the HIR
63/// expression containing it; but for type inference etc., we want to operate on
64/// a structure that is agnostic to the actual positions of expressions in the
65/// file, so that we don't recompute types whenever some whitespace is typed.
66///
67/// One complication here is that, due to macro expansion, a single `Body` might
68/// be spread across several files. So, for each ExprId and PatId, we record
69/// both the HirFileId and the position inside the file. However, we only store
70/// AST -> ExprId mapping for non-macro files, as it is not clear how to handle
71/// this properly for macros.
72#[derive(Default, Debug, Eq, PartialEq)]
73pub struct BodySourceMap {
74    pub self_param: Option<InFile<SelfParamPtr>>,
75    pub store: ExpressionStoreSourceMap,
76}
77
78impl ops::Deref for BodySourceMap {
79    type Target = ExpressionStoreSourceMap;
80
81    #[inline]
82    fn deref(&self) -> &Self::Target {
83        &self.store
84    }
85}

87#[salsa::tracked]
88impl Body {
89    #[salsa::tracked(lru = 512, returns(ref))]
90    pub fn with_source_map(
91        db: &dyn SourceDatabase,
92        def: DefWithBodyId,
93    ) -> (Arc<Body>, BodySourceMap) {
94        let _p = tracing::info_span!("body_with_source_map_query").entered();
95        let mut params = None;
96
97        let mut is_async_fn = false;
98        let mut is_gen_fn = false;
99        let InFile { file_id, value: body } = {
100            match def {
101                DefWithBodyId::FunctionId(f) => {
102                    let f = f.lookup(db);
103                    let src = f.source(db);
104                    params = src.value.param_list();
105                    is_async_fn = src.value.async_token().is_some();
106                    is_gen_fn = src.value.gen_token().is_some();
107                    src.map(|it| it.body().map(ast::Expr::from))
108                }
109                DefWithBodyId::ConstId(c) => {
110                    let c = c.lookup(db);
111                    let src = c.source(db);
112                    src.map(|it| it.body())
113                }
114                DefWithBodyId::StaticId(s) => {
115                    let s = s.lookup(db);
116                    let src = s.source(db);
117                    src.map(|it| it.body())
118                }
119                DefWithBodyId::VariantId(v) => {
120                    let s = v.lookup(db);
121                    let src = s.source(db);
122                    src.map(|it| it.const_arg()?.expr())
123                }
124            }
125        };
126        let module = def.module(db);
127        let (body, source_map) =
128            lower_body(db, def, file_id, module, params, body, is_async_fn, is_gen_fn);
129
130        (Arc::new(body), source_map)
131    }
132
133    #[salsa::tracked(returns(deref))]
134    pub fn of(db: &dyn SourceDatabase, def: DefWithBodyId) -> Arc<Body> {
135        Self::with_source_map(db, def).0.clone()
136    }
137}

139impl Body {
140    pub fn root_expr(&self) -> ExprId {
141        // A `Body` can also contain root expressions that aren't the body (in the param patterns),
142        // but the body always come last.
143        self.store.expr_roots().next_back().unwrap()
144    }
145
146    /// Returns `true` if this is the formal or user-written self param.
147    #[inline]
148    pub fn is_any_self_param(&self, binding: BindingId) -> bool {
149        self.self_param.is_some_and(|self_param| {
150            self_param.formal == binding || self_param.user_written == binding
151        })
152    }
153
154    pub fn pretty_print(
155        &self,
156        db: &dyn SourceDatabase,
157        owner: DefWithBodyId,
158        edition: Edition,
159    ) -> String {
160        pretty::print_body_hir(db, self, owner, edition)
161    }
162
163    pub fn pretty_print_expr(
164        &self,
165        db: &dyn SourceDatabase,
166        owner: DefWithBodyId,
167        expr: ExprId,
168        edition: Edition,
169    ) -> String {
170        pretty::print_expr_hir(db, self, owner.into(), expr, edition)
171    }
172
173    pub fn pretty_print_pat(
174        &self,
175        db: &dyn SourceDatabase,
176        owner: ExpressionStoreOwnerId,
177        pat: PatId,
178        oneline: bool,
179        edition: Edition,
180    ) -> String {
181        pretty::print_pat_hir(db, self, owner, pat, oneline, edition)
182    }
183}
