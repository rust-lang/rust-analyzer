64pub(super) fn lower_body(
65    db: &dyn DefDatabase,
66    owner: DefWithBodyId,
67    current_file_id: HirFileId,
68    module: ModuleId,
69    parameters: Option<ast::ParamList>,
70    body: Option<ast::Expr>,
71    is_async_fn: bool,
72) -> (Body, BodySourceMap) {
73    // We cannot leave the root span map empty and let any identifier from it be treated as root,
74    // because when inside nested macros `SyntaxContextId`s from the outer macro will be interleaved
75    // with the inner macro, and that will cause confusion because they won't be the same as `ROOT`
76    // even though they should be the same. Also, when the body comes from multiple expansions, their
77    // hygiene is different.
78
79    let mut self_param = None;
80    let mut source_map_self_param = None;
81    let mut params = vec![];
82    let mut collector = ExprCollector::new(db, module, current_file_id);
83
84    let skip_body = AttrFlags::query(
85        db,
86        match owner {
87            DefWithBodyId::FunctionId(it) => it.into(),
88            DefWithBodyId::StaticId(it) => it.into(),
89            DefWithBodyId::ConstId(it) => it.into(),
90            DefWithBodyId::VariantId(it) => it.into(),
91        },
92    )
93    .contains(AttrFlags::RUST_ANALYZER_SKIP);
94    // If #[rust_analyzer::skip] annotated, only construct enough information for the signature
95    // and skip the body.
96    if skip_body {
97        if let Some(param_list) = parameters {
98            if let Some(self_param_syn) =
99                param_list.self_param().filter(|self_param| collector.check_cfg(self_param))
100            {
101                let is_mutable =
102                    self_param_syn.mut_token().is_some() && self_param_syn.amp_token().is_none();
103                let hygiene = self_param_syn
104                    .name()
105                    .map(|name| collector.hygiene_id_for(name.syntax().text_range()))
106                    .unwrap_or(HygieneId::ROOT);
107                let binding_id: la_arena::Idx<Binding> = collector.alloc_binding(
108                    Name::new_symbol_root(sym::self_),
109                    BindingAnnotation::new(is_mutable, false),
110                    hygiene,
111                );
112                self_param = Some(binding_id);
113                source_map_self_param =
114                    Some(collector.expander.in_file(AstPtr::new(&self_param_syn)));
115            }
116            let count = param_list.params().filter(|it| collector.check_cfg(it)).count();
117            params = (0..count).map(|_| collector.missing_pat()).collect();
118        };
119        let body_expr = collector.missing_expr();
120        let (store, source_map) = collector.store.finish();
121        return (
122            Body { store, params: params.into_boxed_slice(), self_param, body_expr },
123            BodySourceMap { self_param: source_map_self_param, store: source_map },
124        );
125    }
126
127    if let Some(param_list) = parameters {
128        if let Some(self_param_syn) = param_list.self_param().filter(|it| collector.check_cfg(it)) {
129            let is_mutable =
130                self_param_syn.mut_token().is_some() && self_param_syn.amp_token().is_none();
131            let hygiene = self_param_syn
132                .name()
133                .map(|name| collector.hygiene_id_for(name.syntax().text_range()))
134                .unwrap_or(HygieneId::ROOT);
135            let binding_id: la_arena::Idx<Binding> = collector.alloc_binding(
136                Name::new_symbol_root(sym::self_),
137                BindingAnnotation::new(is_mutable, false),
138                hygiene,
139            );
140            self_param = Some(binding_id);
141            source_map_self_param = Some(collector.expander.in_file(AstPtr::new(&self_param_syn)));
142        }
143
144        let is_extern = matches!(
145            owner,
146            DefWithBodyId::FunctionId(id)
147                if matches!(id.loc(db).container, ItemContainerId::ExternBlockId(_)),
148        );
149
150        for param in param_list.params() {
151            if collector.check_cfg(&param) {
152                let param_pat = if is_extern {
153                    collector.collect_extern_fn_param(param.pat())
154                } else {
155                    collector.collect_pat_top(param.pat())
156                };
157                params.push(param_pat);
158            }
159        }
160    };
161
162    let body_expr = collector.collect(
163        &mut params,
164        body,
165        if is_async_fn {
166            Awaitable::Yes
167        } else {
168            match owner {
169                DefWithBodyId::FunctionId(..) => Awaitable::No("non-async function"),
170                DefWithBodyId::StaticId(..) => Awaitable::No("static"),
171                DefWithBodyId::ConstId(..) => Awaitable::No("constant"),
172                DefWithBodyId::VariantId(..) => Awaitable::No("enum variant"),
173            }
174        },
175    );
176
177    let (store, source_map) = collector.store.finish();
178    (
179        Body { store, params: params.into_boxed_slice(), self_param, body_expr },
180        BodySourceMap { self_param: source_map_self_param, store: source_map },
181    )
182}

212pub(crate) fn lower_impl(
213    db: &dyn DefDatabase,
214    module: ModuleId,
215    impl_syntax: InFile<ast::Impl>,
216    impl_id: ImplId,
217) -> (ExpressionStore, ExpressionStoreSourceMap, TypeRefId, Option<TraitRef>, Arc<GenericParams>) {
218    let mut expr_collector = ExprCollector::new(db, module, impl_syntax.file_id);
219    let self_ty =
220        expr_collector.lower_type_ref_opt_disallow_impl_trait(impl_syntax.value.self_ty());
221    let trait_ = impl_syntax.value.trait_().and_then(|it| match &it {
222        ast::Type::PathType(path_type) => {
223            let path = expr_collector
224                .lower_path_type(path_type, &mut ExprCollector::impl_trait_allocator)?;
225            Some(TraitRef { path: expr_collector.alloc_path(path, AstPtr::new(&it)) })
226        }
227        _ => None,
228    });
229    let mut collector = generics::GenericParamsCollector::new(impl_id.into());
230    collector.lower(
231        &mut expr_collector,
232        impl_syntax.value.generic_param_list(),
233        impl_syntax.value.where_clause(),
234    );
235    let params = collector.finish();
236    let (store, source_map) = expr_collector.store.finish();
237    (store, source_map, self_ty, trait_, params)
238}

262pub(crate) fn lower_type_alias(
263    db: &dyn DefDatabase,
264    module: ModuleId,
265    alias: InFile<ast::TypeAlias>,
266    type_alias_id: TypeAliasId,
267) -> (
268    ExpressionStore,
269    ExpressionStoreSourceMap,
270    Arc<GenericParams>,
271    Box<[TypeBound]>,
272    Option<TypeRefId>,
273) {
274    let mut expr_collector = ExprCollector::new(db, module, alias.file_id);
275    let bounds = alias
276        .value
277        .type_bound_list()
278        .map(|bounds| {
279            bounds
280                .bounds()
281                .map(|bound| {
282                    expr_collector.lower_type_bound(bound, &mut ExprCollector::impl_trait_allocator)
283                })
284                .collect()
285        })
286        .unwrap_or_default();
287    let mut collector = generics::GenericParamsCollector::new(type_alias_id.into());
288    collector.lower(
289        &mut expr_collector,
290        alias.value.generic_param_list(),
291        alias.value.where_clause(),
292    );
293    let params = collector.finish();
294    let type_ref = alias
295        .value
296        .ty()
297        .map(|ty| expr_collector.lower_type_ref(ty, &mut ExprCollector::impl_trait_allocator));
298    let (store, source_map) = expr_collector.store.finish();
299    (store, source_map, params, bounds, type_ref)
300}

302pub(crate) fn lower_function(
303    db: &dyn DefDatabase,
304    module: ModuleId,
305    fn_: InFile<ast::Fn>,
306    function_id: FunctionId,
307) -> (
308    ExpressionStore,
309    ExpressionStoreSourceMap,
310    Arc<GenericParams>,
311    Box<[TypeRefId]>,
312    Option<TypeRefId>,
313    bool,
314    bool,
315) {
316    let mut expr_collector = ExprCollector::new(db, module, fn_.file_id);
317    let mut collector = generics::GenericParamsCollector::new(function_id.into());
318    collector.lower(&mut expr_collector, fn_.value.generic_param_list(), fn_.value.where_clause());
319    let mut params = vec![];
320    let mut has_self_param = false;
321    let mut has_variadic = false;
322    collector.collect_impl_trait(&mut expr_collector, |collector, mut impl_trait_lower_fn| {
323        if let Some(param_list) = fn_.value.param_list() {
324            if let Some(param) = param_list.self_param() {
325                let enabled = collector.check_cfg(&param);
326                if enabled {
327                    has_self_param = true;
328                    params.push(match param.ty() {
329                        Some(ty) => collector.lower_type_ref(ty, &mut impl_trait_lower_fn),
330                        None => {
331                            let self_type = collector.alloc_type_ref_desugared(TypeRef::Path(
332                                Name::new_symbol_root(sym::Self_).into(),
333                            ));
334                            let lifetime = param
335                                .lifetime()
336                                .map(|lifetime| collector.lower_lifetime_ref(lifetime));
337                            match param.kind() {
338                                ast::SelfParamKind::Owned => self_type,
339                                ast::SelfParamKind::Ref => collector.alloc_type_ref_desugared(
340                                    TypeRef::Reference(Box::new(RefType {
341                                        ty: self_type,
342                                        lifetime,
343                                        mutability: Mutability::Shared,
344                                    })),
345                                ),
346                                ast::SelfParamKind::MutRef => collector.alloc_type_ref_desugared(
347                                    TypeRef::Reference(Box::new(RefType {
348                                        ty: self_type,
349                                        lifetime,
350                                        mutability: Mutability::Mut,
351                                    })),
352                                ),
353                            }
354                        }
355                    });
356                }
357            }
358            let p = param_list
359                .params()
360                .filter(|param| collector.check_cfg(param))
361                .filter(|param| {
362                    let is_variadic = param.dotdotdot_token().is_some();
363                    has_variadic |= is_variadic;
364                    !is_variadic
365                })
366                .map(|param| param.ty())
367                // FIXME
368                .collect::<Vec<_>>();
369            for p in p {
370                params.push(collector.lower_type_ref_opt(p, &mut impl_trait_lower_fn));
371            }
372        }
373    });
374    let generics = collector.finish();
375    let return_type = fn_.value.ret_type().map(|ret_type| {
376        expr_collector.lower_type_ref_opt(ret_type.ty(), &mut ExprCollector::impl_trait_allocator)
377    });
378
379    let return_type = if fn_.value.async_token().is_some() {
380        let path = hir_expand::mod_path::path![core::future::Future];
381        let mut generic_args: Vec<_> =
382            std::iter::repeat_n(None, path.segments().len() - 1).collect();
383        let binding = AssociatedTypeBinding {
384            name: Name::new_symbol_root(sym::Output),
385            args: None,
386            type_ref: Some(
387                return_type
388                    .unwrap_or_else(|| expr_collector.alloc_type_ref_desugared(TypeRef::unit())),
389            ),
390            bounds: Box::default(),
391        };
392        generic_args
393            .push(Some(GenericArgs { bindings: Box::new([binding]), ..GenericArgs::empty() }));
394
395        let path = Path::from_known_path(path, generic_args);
396        let path = PathId::from_type_ref_unchecked(
397            expr_collector.alloc_type_ref_desugared(TypeRef::Path(path)),
398        );
399        let ty_bound = TypeBound::Path(path, TraitBoundModifier::None);
400        Some(
401            expr_collector
402                .alloc_type_ref_desugared(TypeRef::ImplTrait(ThinVec::from_iter([ty_bound]))),
403        )
404    } else {
405        return_type
406    };
407    let (store, source_map) = expr_collector.store.finish();
408    (
409        store,
410        source_map,
411        generics,
412        params.into_boxed_slice(),
413        return_type,
414        has_self_param,
415        has_variadic,
416    )
417}

492#[derive(Debug, Default)]
493struct BindingList {
494    map: FxHashMap<(Name, HygieneId), BindingId>,
495    is_used: FxHashMap<BindingId, bool>,
496    reject_new: bool,
497}
498
499impl BindingList {
500    fn find(
501        &mut self,
502        ec: &mut ExprCollector<'_>,
503        name: Name,
504        hygiene: HygieneId,
505        mode: BindingAnnotation,
506    ) -> BindingId {
507        let id = *self
508            .map
509            .entry((name, hygiene))
510            .or_insert_with_key(|(name, hygiene)| ec.alloc_binding(name.clone(), mode, *hygiene));
511        if ec.store.bindings[id].mode != mode {
512            ec.store.bindings[id].problems = Some(BindingProblems::BoundInconsistently);
513        }
514        self.check_is_used(ec, id);
515        id
516    }
517
518    fn check_is_used(&mut self, ec: &mut ExprCollector<'_>, id: BindingId) {
519        match self.is_used.get(&id) {
520            None => {
521                if self.reject_new {
522                    ec.store.bindings[id].problems = Some(BindingProblems::NotBoundAcrossAll);
523                }
524            }
525            Some(true) => {
526                ec.store.bindings[id].problems = Some(BindingProblems::BoundMoreThanOnce);
527            }
528            Some(false) => {}
529        }
530        self.is_used.insert(id, true);
531    }
532}

598    /// Converts an `ast::TypeRef` to a `hir::TypeRef`.
599    pub fn lower_type_ref(
600        &mut self,
601        node: ast::Type,
602        impl_trait_lower_fn: ImplTraitLowerFn<'_>,
603    ) -> TypeRefId {
604        let ty = match &node {
605            ast::Type::ParenType(inner) => {
606                return self.lower_type_ref_opt(inner.ty(), impl_trait_lower_fn);
607            }
608            ast::Type::TupleType(inner) => TypeRef::Tuple(ThinVec::from_iter(Vec::from_iter(
609                inner.fields().map(|it| self.lower_type_ref(it, impl_trait_lower_fn)),
610            ))),
611            ast::Type::NeverType(..) => TypeRef::Never,
612            ast::Type::PathType(inner) => inner
613                .path()
614                .and_then(|it| self.lower_path(it, impl_trait_lower_fn))
615                .map(TypeRef::Path)
616                .unwrap_or(TypeRef::Error),
617            ast::Type::PtrType(inner) => {
618                let inner_ty = self.lower_type_ref_opt(inner.ty(), impl_trait_lower_fn);
619                let mutability = Mutability::from_mutable(inner.mut_token().is_some());
620                TypeRef::RawPtr(inner_ty, mutability)
621            }
622            ast::Type::ArrayType(inner) => {
623                let len = self.lower_const_arg_opt(inner.const_arg());
624                TypeRef::Array(ArrayType {
625                    ty: self.lower_type_ref_opt(inner.ty(), impl_trait_lower_fn),
626                    len,
627                })
628            }
629            ast::Type::SliceType(inner) => {
630                TypeRef::Slice(self.lower_type_ref_opt(inner.ty(), impl_trait_lower_fn))
631            }
632            ast::Type::RefType(inner) => {
633                let inner_ty = self.lower_type_ref_opt(inner.ty(), impl_trait_lower_fn);
634                let lifetime = inner.lifetime().map(|lt| self.lower_lifetime_ref(lt));
635                let mutability = Mutability::from_mutable(inner.mut_token().is_some());
636                TypeRef::Reference(Box::new(RefType { ty: inner_ty, lifetime, mutability }))
637            }
638            ast::Type::InferType(_inner) => TypeRef::Placeholder,
639            ast::Type::FnPtrType(inner) => {
640                let ret_ty = inner
641                    .ret_type()
642                    .and_then(|rt| rt.ty())
643                    .map(|it| self.lower_type_ref(it, impl_trait_lower_fn))
644                    .unwrap_or_else(|| self.alloc_type_ref_desugared(TypeRef::unit()));
645                let mut is_varargs = false;
646                let mut params = if let Some(pl) = inner.param_list() {
647                    if let Some(param) = pl.params().last() {
648                        is_varargs = param.dotdotdot_token().is_some();
649                    }
650
651                    pl.params()
652                        .map(|it| {
653                            let type_ref = self.lower_type_ref_opt(it.ty(), impl_trait_lower_fn);
654                            let name = match it.pat() {
655                                Some(ast::Pat::IdentPat(it)) => Some(
656                                    it.name().map(|nr| nr.as_name()).unwrap_or_else(Name::missing),
657                                ),
658                                _ => None,
659                            };
660                            (name, type_ref)
661                        })
662                        .collect()
663                } else {
664                    Vec::with_capacity(1)
665                };
666                fn lower_abi(abi: ast::Abi) -> Symbol {
667                    match abi.abi_string() {
668                        Some(tok) => Symbol::intern(tok.text_without_quotes()),
669                        // `extern` default to be `extern "C"`.
670                        _ => sym::C,
671                    }
672                }
673
674                let abi = inner.abi().map(lower_abi);
675                params.push((None, ret_ty));
676                TypeRef::Fn(Box::new(FnType {
677                    is_varargs,
678                    is_unsafe: inner.unsafe_token().is_some(),
679                    abi,
680                    params: params.into_boxed_slice(),
681                }))
682            }
683            // for types are close enough for our purposes to the inner type for now...
684            ast::Type::ForType(inner) => {
685                return self.lower_type_ref_opt(inner.ty(), impl_trait_lower_fn);
686            }
687            ast::Type::ImplTraitType(inner) => {
688                if self.outer_impl_trait {
689                    // Disallow nested impl traits
690                    TypeRef::Error
691                } else {
692                    return self.with_outer_impl_trait_scope(true, |this| {
693                        let type_bounds =
694                            this.type_bounds_from_ast(inner.type_bound_list(), impl_trait_lower_fn);
695                        impl_trait_lower_fn(this, AstPtr::new(&node), type_bounds)
696                    });
697                }
698            }
699            ast::Type::DynTraitType(inner) => TypeRef::DynTrait(
700                self.type_bounds_from_ast(inner.type_bound_list(), impl_trait_lower_fn),
701            ),
702            ast::Type::MacroType(mt) => match mt.macro_call() {
703                Some(mcall) => {
704                    let macro_ptr = AstPtr::new(&mcall);
705                    let src = self.expander.in_file(AstPtr::new(&node));
706                    let id = self.collect_macro_call(mcall, macro_ptr, true, |this, expansion| {
707                        this.lower_type_ref_opt(expansion, impl_trait_lower_fn)
708                    });
709                    self.store.types_map.insert(src, id);
710                    return id;
711                }
712                None => TypeRef::Error,
713            },
714        };
715        self.alloc_type_ref(ty, AstPtr::new(&node))
716    }

811    /// Collect `GenericArgs` from the parts of a fn-like path, i.e. `Fn(X, Y)
812    /// -> Z` (which desugars to `Fn<(X, Y), Output=Z>`).
813    pub fn lower_generic_args_from_fn_path(
814        &mut self,
815        args: Option<ast::ParenthesizedArgList>,
816        ret_type: Option<ast::RetType>,
817        impl_trait_lower_fn: ImplTraitLowerFn<'_>,
818    ) -> Option<GenericArgs> {
819        let params = args?;
820        let mut param_types = Vec::new();
821        for param in params.type_args() {
822            let type_ref = self.lower_type_ref_opt(param.ty(), impl_trait_lower_fn);
823            param_types.push(type_ref);
824        }
825        let args = Box::new([GenericArg::Type(
826            self.alloc_type_ref_desugared(TypeRef::Tuple(ThinVec::from_iter(param_types))),
827        )]);
828        let bindings = if let Some(ret_type) = ret_type {
829            let type_ref = self.lower_type_ref_opt(ret_type.ty(), impl_trait_lower_fn);
830            Box::new([AssociatedTypeBinding {
831                name: Name::new_symbol_root(sym::Output),
832                args: None,
833                type_ref: Some(type_ref),
834                bounds: Box::default(),
835            }])
836        } else {
837            // -> ()
838            let type_ref = self.alloc_type_ref_desugared(TypeRef::unit());
839            Box::new([AssociatedTypeBinding {
840                name: Name::new_symbol_root(sym::Output),
841                args: None,
842                type_ref: Some(type_ref),
843                bounds: Box::default(),
844            }])
845        };
846        Some(GenericArgs {
847            args,
848            has_self_type: false,
849            bindings,
850            parenthesized: GenericArgsParentheses::ParenSugar,
851        })
852    }

854    pub(super) fn lower_generic_args(
855        &mut self,
856        node: ast::GenericArgList,
857        impl_trait_lower_fn: ImplTraitLowerFn<'_>,
858    ) -> Option<GenericArgs> {
859        // This needs to be kept in sync with `hir_generic_arg_to_ast()`.
860        let mut args = Vec::new();
861        let mut bindings = Vec::new();
862        for generic_arg in node.generic_args() {
863            match generic_arg {
864                ast::GenericArg::TypeArg(type_arg) => {
865                    let type_ref = self.lower_type_ref_opt(type_arg.ty(), impl_trait_lower_fn);
866                    args.push(GenericArg::Type(type_ref));
867                }
868                ast::GenericArg::AssocTypeArg(assoc_type_arg) => {
869                    // This needs to be kept in sync with `hir_assoc_type_binding_to_ast()`.
870                    if assoc_type_arg.param_list().is_some() {
871                        // We currently ignore associated return type bounds.
872                        continue;
873                    }
874                    if let Some(name_ref) = assoc_type_arg.name_ref() {
875                        // Nested impl traits like `impl Foo<Assoc = impl Bar>` are allowed
876                        self.with_outer_impl_trait_scope(false, |this| {
877                            let name = name_ref.as_name();
878                            let args = assoc_type_arg
879                                .generic_arg_list()
880                                .and_then(|args| this.lower_generic_args(args, impl_trait_lower_fn))
881                                .or_else(|| {
882                                    assoc_type_arg
883                                        .return_type_syntax()
884                                        .map(|_| GenericArgs::return_type_notation())
885                                });
886                            let type_ref = assoc_type_arg
887                                .ty()
888                                .map(|it| this.lower_type_ref(it, impl_trait_lower_fn));
889                            let bounds = if let Some(l) = assoc_type_arg.type_bound_list() {
890                                l.bounds()
891                                    .map(|it| this.lower_type_bound(it, impl_trait_lower_fn))
892                                    .collect()
893                            } else {
894                                Box::default()
895                            };
896                            bindings.push(AssociatedTypeBinding { name, args, type_ref, bounds });
897                        });
898                    }
899                }
900                ast::GenericArg::LifetimeArg(lifetime_arg) => {
901                    if let Some(lifetime) = lifetime_arg.lifetime() {
902                        let lifetime_ref = self.lower_lifetime_ref(lifetime);
903                        args.push(GenericArg::Lifetime(lifetime_ref))
904                    }
905                }
906                ast::GenericArg::ConstArg(arg) => {
907                    let arg = self.lower_const_arg(arg);
908                    args.push(GenericArg::Const(arg))
909                }
910            }
911        }
912
913        if args.is_empty() && bindings.is_empty() {
914            return None;
915        }
916        Some(GenericArgs {
917            args: args.into_boxed_slice(),
918            has_self_type: false,
919            bindings: bindings.into_boxed_slice(),
920            parenthesized: GenericArgsParentheses::No,
921        })
922    }

924    /// An `async fn` needs to capture all parameters in the generated `async` block, even if they have
925    /// non-captured patterns such as wildcards (to ensure consistent drop order).
926    fn lower_async_fn(&mut self, params: &mut Vec<PatId>, body: ExprId) -> ExprId {
927        let mut statements = Vec::new();
928        for param in params {
929            let name = match self.store.pats[*param] {
930                Pat::Bind { id, .. }
931                    if matches!(
932                        self.store.bindings[id].mode,
933                        BindingAnnotation::Unannotated | BindingAnnotation::Mutable
934                    ) =>
935                {
936                    // If this is a direct binding, we can leave it as-is, as it'll always be captured anyway.
937                    continue;
938                }
939                Pat::Bind { id, .. } => {
940                    // If this is a `ref` binding, we can't leave it as is but we can at least reuse the name, for better display.
941                    self.store.bindings[id].name.clone()
942                }
943                _ => self.generate_new_name(),
944            };
945            let binding_id =
946                self.alloc_binding(name.clone(), BindingAnnotation::Mutable, HygieneId::ROOT);
947            let pat_id = self.alloc_pat_desugared(Pat::Bind { id: binding_id, subpat: None });
948            let expr = self.alloc_expr_desugared(Expr::Path(name.into()));
949            statements.push(Statement::Let {
950                pat: *param,
951                type_ref: None,
952                initializer: Some(expr),
953                else_branch: None,
954            });
955            *param = pat_id;
956        }
957
958        self.alloc_expr_desugared(Expr::Async {
959            id: None,
960            statements: statements.into_boxed_slice(),
961            tail: Some(body),
962        })
963    }

1001    fn lower_type_bound(
1002        &mut self,
1003        node: ast::TypeBound,
1004        impl_trait_lower_fn: ImplTraitLowerFn<'_>,
1005    ) -> TypeBound {
1006        let Some(kind) = node.kind() else { return TypeBound::Error };
1007        match kind {
1008            ast::TypeBoundKind::PathType(binder, path_type) => {
1009                let binder = match binder.and_then(|it| it.generic_param_list()) {
1010                    Some(gpl) => gpl
1011                        .lifetime_params()
1012                        .flat_map(|lp| lp.lifetime().map(|lt| Name::new_lifetime(&lt.text())))
1013                        .collect(),
1014                    None => ThinVec::default(),
1015                };
1016                let m = match node.question_mark_token() {
1017                    Some(_) => TraitBoundModifier::Maybe,
1018                    None => TraitBoundModifier::None,
1019                };
1020                self.lower_path_type(&path_type, impl_trait_lower_fn)
1021                    .map(|p| {
1022                        let path = self.alloc_path(p, AstPtr::new(&path_type).upcast());
1023                        if binder.is_empty() {
1024                            TypeBound::Path(path, m)
1025                        } else {
1026                            TypeBound::ForLifetime(binder, path)
1027                        }
1028                    })
1029                    .unwrap_or(TypeBound::Error)
1030            }
1031            ast::TypeBoundKind::Use(gal) => TypeBound::Use(
1032                gal.use_bound_generic_args()
1033                    .map(|p| match p {
1034                        ast::UseBoundGenericArg::Lifetime(l) => {
1035                            UseArgRef::Lifetime(self.lower_lifetime_ref(l))
1036                        }
1037                        ast::UseBoundGenericArg::NameRef(n) => UseArgRef::Name(n.as_name()),
1038                    })
1039                    .collect(),
1040            ),
1041            ast::TypeBoundKind::Lifetime(lifetime) => {
1042                TypeBound::Lifetime(self.lower_lifetime_ref(lifetime))
1043            }
1044        }
1045    }

1059    /// Returns `None` if and only if the expression is `#[cfg]`d out.
1060    fn maybe_collect_expr(&mut self, expr: ast::Expr) -> Option<ExprId> {
1061        let syntax_ptr = AstPtr::new(&expr);
1062        if !self.check_cfg(&expr) {
1063            return None;
1064        }
1065
1066        // FIXME: Move some of these arms out into separate methods for clarity
1067        Some(match expr {
1068            ast::Expr::IfExpr(e) => {
1069                let then_branch = self.collect_block_opt(e.then_branch());
1070
1071                let else_branch = e.else_branch().map(|b| match b {
1072                    ast::ElseBranch::Block(it) => self.collect_block(it),
1073                    ast::ElseBranch::IfExpr(elif) => {
1074                        let expr: ast::Expr = ast::Expr::cast(elif.syntax().clone()).unwrap();
1075                        self.collect_expr(expr)
1076                    }
1077                });
1078
1079                let condition = self.collect_expr_opt(e.condition());
1080
1081                self.alloc_expr(Expr::If { condition, then_branch, else_branch }, syntax_ptr)
1082            }
1083            ast::Expr::LetExpr(e) => {
1084                let pat = self.collect_pat_top(e.pat());
1085                let expr = self.collect_expr_opt(e.expr());
1086                self.alloc_expr(Expr::Let { pat, expr }, syntax_ptr)
1087            }
1088            ast::Expr::BlockExpr(e) => match e.modifier() {
1089                Some(ast::BlockModifier::Try { try_token: _, bikeshed_token: _, result_type }) => {
1090                    self.desugar_try_block(e, result_type)
1091                }
1092                Some(ast::BlockModifier::Unsafe(_)) => {
1093                    self.collect_block_(e, |id, statements, tail| Expr::Unsafe {
1094                        id,
1095                        statements,
1096                        tail,
1097                    })
1098                }
1099                Some(ast::BlockModifier::Label(label)) => {
1100                    let label_hygiene = self.hygiene_id_for(label.syntax().text_range());
1101                    let label_id = self.collect_label(label);
1102                    self.with_labeled_rib(label_id, label_hygiene, |this| {
1103                        this.collect_block_(e, |id, statements, tail| Expr::Block {
1104                            id,
1105                            statements,
1106                            tail,
1107                            label: Some(label_id),
1108                        })
1109                    })
1110                }
1111                Some(ast::BlockModifier::Async(_)) => {
1112                    self.with_label_rib(RibKind::Closure, |this| {
1113                        this.with_awaitable_block(Awaitable::Yes, |this| {
1114                            this.collect_block_(e, |id, statements, tail| Expr::Async {
1115                                id,
1116                                statements,
1117                                tail,
1118                            })
1119                        })
1120                    })
1121                }
1122                Some(ast::BlockModifier::Const(_)) => {
1123                    self.with_label_rib(RibKind::Constant, |this| {
1124                        this.with_awaitable_block(Awaitable::No("constant block"), |this| {
1125                            this.with_binding_owner(|this| {
1126                                let inner_expr = this.collect_block(e);
1127                                this.alloc_expr(Expr::Const(inner_expr), syntax_ptr)
1128                            })
1129                        })
1130                    })
1131                }
1132                // FIXME
1133                Some(ast::BlockModifier::AsyncGen(_)) => {
1134                    self.with_awaitable_block(Awaitable::Yes, |this| this.collect_block(e))
1135                }
1136                Some(ast::BlockModifier::Gen(_)) => self
1137                    .with_awaitable_block(Awaitable::No("non-async gen block"), |this| {
1138                        this.collect_block(e)
1139                    }),
1140                None => self.collect_block(e),
1141            },
1142            ast::Expr::LoopExpr(e) => {
1143                let label = e.label().map(|label| {
1144                    (self.hygiene_id_for(label.syntax().text_range()), self.collect_label(label))
1145                });
1146                let body = self.collect_labelled_block_opt(label, e.loop_body());
1147                self.alloc_expr(Expr::Loop { body, label: label.map(|it| it.1) }, syntax_ptr)
1148            }
1149            ast::Expr::WhileExpr(e) => self.collect_while_loop(syntax_ptr, e),
1150            ast::Expr::ForExpr(e) => self.collect_for_loop(syntax_ptr, e),
1151            ast::Expr::CallExpr(e) => {
1152                // FIXME(MINIMUM_SUPPORTED_TOOLCHAIN_VERSION): Remove this once we drop support for <1.86, https://github.com/rust-lang/rust/commit/ac9cb908ac4301dfc25e7a2edee574320022ae2c
1153                let is_rustc_box = {
1154                    let attrs = e.attrs();
1155                    attrs.filter_map(|it| it.as_simple_atom()).any(|it| it == "rustc_box")
1156                };
1157                if is_rustc_box {
1158                    let expr = self.collect_expr_opt(e.arg_list().and_then(|it| it.args().next()));
1159                    self.alloc_expr(Expr::Box { expr }, syntax_ptr)
1160                } else {
1161                    let callee = self.collect_expr_opt(e.expr());
1162                    let args = if let Some(arg_list) = e.arg_list() {
1163                        arg_list.args().filter_map(|e| self.maybe_collect_expr(e)).collect()
1164                    } else {
1165                        Box::default()
1166                    };
1167                    self.alloc_expr(Expr::Call { callee, args }, syntax_ptr)
1168                }
1169            }
1170            ast::Expr::MethodCallExpr(e) => {
1171                let receiver = self.collect_expr_opt(e.receiver());
1172                let args = if let Some(arg_list) = e.arg_list() {
1173                    arg_list.args().filter_map(|e| self.maybe_collect_expr(e)).collect()
1174                } else {
1175                    Box::default()
1176                };
1177                let method_name = e.name_ref().map(|nr| nr.as_name()).unwrap_or_else(Name::missing);
1178                let generic_args = e
1179                    .generic_arg_list()
1180                    .and_then(|it| {
1181                        self.lower_generic_args(it, &mut Self::impl_trait_error_allocator)
1182                    })
1183                    .map(Box::new);
1184                self.alloc_expr(
1185                    Expr::MethodCall { receiver, method_name, args, generic_args },
1186                    syntax_ptr,
1187                )
1188            }
1189            ast::Expr::MatchExpr(e) => {
1190                let expr = self.collect_expr_opt(e.expr());
1191                let arms = if let Some(match_arm_list) = e.match_arm_list() {
1192                    match_arm_list
1193                        .arms()
1194                        .filter_map(|arm| {
1195                            if self.check_cfg(&arm) {
1196                                Some(MatchArm {

1528    fn maybe_collect_expr_as_pat(&mut self, expr: &ast::Expr) -> Option<PatId> {
1529        if !self.check_cfg(expr) {
1530            return None;
1531        }
1532        let syntax_ptr = AstPtr::new(expr);
1533
1534        let result = match expr {
1535            ast::Expr::UnderscoreExpr(_) => self.alloc_pat_from_expr(Pat::Wild, syntax_ptr),
1536            ast::Expr::ParenExpr(e) => {
1537                // We special-case `(..)` for consistency with patterns.
1538                if let Some(ast::Expr::RangeExpr(range)) = e.expr()
1539                    && range.is_range_full()
1540                {
1541                    return Some(self.alloc_pat_from_expr(
1542                        Pat::Tuple { args: Box::default(), ellipsis: Some(0) },
1543                        syntax_ptr,
1544                    ));
1545                }
1546                return e.expr().and_then(|expr| self.maybe_collect_expr_as_pat(&expr));
1547            }
1548            ast::Expr::TupleExpr(e) => {
1549                let (ellipsis, args) = collect_tuple(self, e.fields());
1550                self.alloc_pat_from_expr(Pat::Tuple { args, ellipsis }, syntax_ptr)
1551            }
1552            ast::Expr::ArrayExpr(e) => {
1553                if e.semicolon_token().is_some() {
1554                    return None;
1555                }
1556
1557                let mut elements = e.exprs();
1558                let prefix = elements
1559                    .by_ref()
1560                    .map_while(|elem| collect_possibly_rest(self, elem).left())
1561                    .collect();
1562                let suffix = elements.map(|elem| self.collect_expr_as_pat(elem)).collect();
1563                self.alloc_pat_from_expr(Pat::Slice { prefix, slice: None, suffix }, syntax_ptr)
1564            }
1565            ast::Expr::CallExpr(e) => {
1566                let path = collect_path(self, e.expr()?)?;
1567                let path = path
1568                    .path()
1569                    .and_then(|path| self.lower_path(path, &mut Self::impl_trait_error_allocator))
1570                    .map(Box::new);
1571                let (ellipsis, args) = collect_tuple(self, e.arg_list()?.args());
1572                self.alloc_pat_from_expr(Pat::TupleStruct { path, args, ellipsis }, syntax_ptr)
1573            }
1574            ast::Expr::PathExpr(e) => {
1575                let (path, hygiene) = self
1576                    .collect_expr_path(e.clone())
1577                    .map(|(path, hygiene)| (Pat::Path(path), hygiene))
1578                    .unwrap_or((Pat::Missing, HygieneId::ROOT));
1579                let pat_id = self.alloc_pat_from_expr(path, syntax_ptr);
1580                if !hygiene.is_root() {
1581                    self.store.ident_hygiene.insert(pat_id.into(), hygiene);
1582                }
1583                pat_id
1584            }
1585            ast::Expr::MacroExpr(e) => {
1586                let e = e.macro_call()?;
1587                let macro_ptr = AstPtr::new(&e);
1588                let src = self.expander.in_file(AstPtr::new(expr));
1589                let id = self.collect_macro_call(e, macro_ptr, true, |this, expansion| {
1590                    this.collect_expr_as_pat_opt(expansion)
1591                });
1592                self.store.expr_map.insert(src, id.into());
1593                id
1594            }
1595            ast::Expr::RecordExpr(e) => {
1596                let path = e
1597                    .path()
1598                    .and_then(|path| self.lower_path(path, &mut Self::impl_trait_error_allocator))
1599                    .map(Box::new);
1600                let record_field_list = e.record_expr_field_list()?;
1601                let ellipsis = record_field_list.dotdot_token().is_some();
1602                // FIXME: Report an error here if `record_field_list.spread().is_some()`.
1603                let args = record_field_list
1604                    .fields()
1605                    .filter_map(|f| {
1606                        if !self.check_cfg(&f) {
1607                            return None;
1608                        }
1609                        let field_expr = f.expr()?;
1610                        let pat = self.collect_expr_as_pat(field_expr);
1611                        let name = f.field_name()?.as_name();
1612                        let src = self.expander.in_file(AstPtr::new(&f).wrap_left());
1613                        self.store.pat_field_map_back.insert(pat, src);
1614                        Some(RecordFieldPat { name, pat })
1615                    })
1616                    .collect();
1617                self.alloc_pat_from_expr(Pat::Record { path, args, ellipsis }, syntax_ptr)
1618            }
1619            _ => return None,
1620        };
1621        return Some(result);

1629                        let macro_ptr = AstPtr::new(&call);
1630                        this.collect_macro_call(call, macro_ptr, true, |this, expanded_path| {

1639        fn collect_possibly_rest(
1640            this: &mut ExprCollector<'_>,
1641            expr: ast::Expr,
1642        ) -> Either<PatId, ()> {
1643            match &expr {
1644                ast::Expr::RangeExpr(e) if e.is_range_full() => Either::Right(()),
1645                ast::Expr::MacroExpr(mac) => match mac.macro_call() {
1646                    Some(call) => {
1647                        let macro_ptr = AstPtr::new(&call);
1648                        let pat = this.collect_macro_call(
1649                            call,
1650                            macro_ptr,
1651                            true,
1652                            |this, expanded_expr| match expanded_expr {
1653                                Some(expanded_pat) => collect_possibly_rest(this, expanded_pat),
1654                                None => Either::Left(this.missing_pat()),
1655                            },
1656                        );
1657                        if let Either::Left(pat) = pat {
1658                            let src = this.expander.in_file(AstPtr::new(&expr).wrap_left());
1659                            this.store.pat_map_back.insert(pat, src);
1660                        }
1661                        pat
1662                    }
1663                    None => {
1664                        let ptr = AstPtr::new(&expr);
1665                        Either::Left(this.alloc_pat_from_expr(Pat::Missing, ptr))
1666                    }
1667                },
1668                _ => Either::Left(this.collect_expr_as_pat(expr)),
1669            }
1670        }

1705    /// Desugar `try { <stmts>; <expr> }` into `'<new_label>: { <stmts>; ::std::ops::Try::from_output(<expr>) }`,
1706    /// `try { <stmts>; }` into `'<new_label>: { <stmts>; ::std::ops::Try::from_output(()) }`
1707    /// and save the `<new_label>` to use it as a break target for desugaring of the `?` operator.
1708    fn desugar_try_block(&mut self, e: BlockExpr, result_type: Option<ast::Type>) -> ExprId {
1709        let try_from_output = self.lang_path(self.lang_items().TryTraitFromOutput);
1710        let label = self.generate_new_name();
1711        let label = self.alloc_label_desugared(Label { name: label }, AstPtr::new(&e).wrap_right());
1712        let try_block_info = match result_type {
1713            Some(_) => TryBlock::Heterogeneous { label },
1714            None => TryBlock::Homogeneous { label },
1715        };
1716        let old_try_block = self.current_try_block.replace(try_block_info);
1717
1718        let ptr = AstPtr::new(&e).upcast();
1719        let (btail, expr_id) = self.with_labeled_rib(label, HygieneId::ROOT, |this| {
1720            let mut btail = None;
1721            let block = this.collect_block_(e, |id, statements, tail| {
1722                btail = tail;
1723                Expr::Block { id, statements, tail, label: Some(label) }
1724            });
1725            (btail, block)
1726        });
1727
1728        let callee = self
1729            .alloc_expr_desugared_with_ptr(try_from_output.map_or(Expr::Missing, Expr::Path), ptr);
1730        let next_tail = match btail {
1731            Some(tail) => self
1732                .alloc_expr_desugared_with_ptr(Expr::Call { callee, args: Box::new([tail]) }, ptr),
1733            None => {
1734                let unit =
1735                    self.alloc_expr_desugared_with_ptr(Expr::Tuple { exprs: Box::new([]) }, ptr);
1736                self.alloc_expr_desugared_with_ptr(
1737                    Expr::Call { callee, args: Box::new([unit]) },
1738                    ptr,
1739                )
1740            }
1741        };
1742        let Expr::Block { tail, .. } = &mut self.store.exprs[expr_id] else {
1743            unreachable!("block was lowered to non-block");
1744        };
1745        *tail = Some(next_tail);
1746        self.current_try_block = old_try_block;
1747        match result_type {
1748            Some(ty) => {
1749                // `{ let <name>: <ty> = <expr>; <name> }`
1750                let name = self.generate_new_name();
1751                let type_ref = self.lower_type_ref_disallow_impl_trait(ty);
1752                let binding = self.alloc_binding(
1753                    name.clone(),
1754                    BindingAnnotation::Unannotated,
1755                    HygieneId::ROOT,
1756                );
1757                let pat = self.alloc_pat_desugared(Pat::Bind { id: binding, subpat: None });
1758                self.add_definition_to_binding(binding, pat);
1759                let tail_expr =
1760                    self.alloc_expr_desugared_with_ptr(Expr::Path(Path::from(name)), ptr);
1761                self.alloc_expr_desugared_with_ptr(
1762                    Expr::Block {
1763                        id: None,
1764                        statements: Box::new([Statement::Let {
1765                            pat,
1766                            type_ref: Some(type_ref),
1767                            initializer: Some(expr_id),
1768                            else_branch: None,
1769                        }]),
1770                        tail: Some(tail_expr),
1771                        label: None,
1772                    },
1773                    ptr,
1774                )
1775            }
1776            None => expr_id,
1777        }

1781    /// ```ignore (pseudo-rust)
1782    /// [opt_ident]: loop {
1783    ///   if <cond> {
1784    ///     <body>
1785    ///   }
1786    ///   else {
1787    ///     break;
1788    ///   }
1789    /// }
1790    /// ```
1791    /// FIXME: Rustc wraps the condition in a construct equivalent to `{ let _t = <cond>; _t }`
1792    /// to preserve drop semantics. We should probably do the same in future.
1793    fn collect_while_loop(&mut self, syntax_ptr: AstPtr<ast::Expr>, e: ast::WhileExpr) -> ExprId {
1794        let label = e.label().map(|label| {
1795            (self.hygiene_id_for(label.syntax().text_range()), self.collect_label(label))
1796        });
1797        let body = self.collect_labelled_block_opt(label, e.loop_body());
1798
1799        // Labels can also be used in the condition expression, like this:
1800        // ```
1801        // fn main() {
1802        //     let mut optional = Some(0);
1803        //     'my_label: while let Some(a) = match optional {
1804        //         None => break 'my_label,
1805        //         Some(val) => Some(val),
1806        //     } {
1807        //         println!("{}", a);
1808        //         optional = None;
1809        //     }
1810        // }
1811        // ```
1812        let condition = match label {
1813            Some((label_hygiene, label)) => self.with_labeled_rib(label, label_hygiene, |this| {
1814                this.collect_expr_opt(e.condition())
1815            }),
1816            None => self.collect_expr_opt(e.condition()),
1817        };
1818
1819        let break_expr = self.alloc_expr(Expr::Break { expr: None, label: None }, syntax_ptr);
1820        let if_expr = self.alloc_expr(
1821            Expr::If { condition, then_branch: body, else_branch: Some(break_expr) },
1822            syntax_ptr,
1823        );
1824        self.alloc_expr(Expr::Loop { body: if_expr, label: label.map(|it| it.1) }, syntax_ptr)
1825    }

1827    /// Desugar `ast::ForExpr` from: `[opt_ident]: for <pat> in <head> <body>` into:
1828    /// ```ignore (pseudo-rust)
1829    /// match IntoIterator::into_iter(<head>) {
1830    ///     mut iter => {
1831    ///         [opt_ident]: loop {
1832    ///             match Iterator::next(&mut iter) {
1833    ///                 None => break,
1834    ///                 Some(<pat>) => <body>,
1835    ///             };
1836    ///         }
1837    ///     }
1838    /// }
1839    /// ```
1840    fn collect_for_loop(&mut self, syntax_ptr: AstPtr<ast::Expr>, e: ast::ForExpr) -> ExprId {
1841        let lang_items = self.lang_items();
1842        let into_iter_fn = self.lang_path(lang_items.IntoIterIntoIter);
1843        let iter_next_fn = self.lang_path(lang_items.IteratorNext);
1844        let option_some = self.lang_path(lang_items.OptionSome);
1845        let option_none = self.lang_path(lang_items.OptionNone);
1846        let head = self.collect_expr_opt(e.iterable());
1847        let into_iter_fn_expr =
1848            self.alloc_expr(into_iter_fn.map_or(Expr::Missing, Expr::Path), syntax_ptr);
1849        let iterator = self.alloc_expr(
1850            Expr::Call { callee: into_iter_fn_expr, args: Box::new([head]) },
1851            syntax_ptr,
1852        );
1853        let none_arm = MatchArm {
1854            pat: self.alloc_pat_desugared(option_none.map_or(Pat::Missing, Pat::Path)),
1855            guard: None,
1856            expr: self.alloc_expr(Expr::Break { expr: None, label: None }, syntax_ptr),
1857        };
1858        let some_pat = Pat::TupleStruct {
1859            path: option_some.map(Box::new),
1860            args: Box::new([self.collect_pat_top(e.pat())]),
1861            ellipsis: None,
1862        };
1863        let label = e.label().map(|label| {
1864            (self.hygiene_id_for(label.syntax().text_range()), self.collect_label(label))
1865        });
1866        let some_arm = MatchArm {
1867            pat: self.alloc_pat_desugared(some_pat),
1868            guard: None,
1869            expr: self.with_opt_labeled_rib(label, |this| {
1870                this.collect_expr_opt(e.loop_body().map(|it| it.into()))
1871            }),
1872        };
1873        let iter_name = self.generate_new_name();
1874        let iter_expr = self.alloc_expr(Expr::Path(Path::from(iter_name.clone())), syntax_ptr);
1875        let iter_expr_mut = self.alloc_expr(
1876            Expr::Ref { expr: iter_expr, rawness: Rawness::Ref, mutability: Mutability::Mut },
1877            syntax_ptr,
1878        );
1879        let iter_next_fn_expr =
1880            self.alloc_expr(iter_next_fn.map_or(Expr::Missing, Expr::Path), syntax_ptr);
1881        let iter_next_expr = self.alloc_expr(
1882            Expr::Call { callee: iter_next_fn_expr, args: Box::new([iter_expr_mut]) },
1883            syntax_ptr,
1884        );
1885        let loop_inner = self.alloc_expr(
1886            Expr::Match { expr: iter_next_expr, arms: Box::new([none_arm, some_arm]) },
1887            syntax_ptr,
1888        );
1889        let loop_inner = self.alloc_expr(
1890            Expr::Block {
1891                id: None,
1892                statements: Box::default(),
1893                tail: Some(loop_inner),
1894                label: None,
1895            },
1896            syntax_ptr,
1897        );
1898        let loop_outer = self
1899            .alloc_expr(Expr::Loop { body: loop_inner, label: label.map(|it| it.1) }, syntax_ptr);
1900        let iter_binding =
1901            self.alloc_binding(iter_name, BindingAnnotation::Mutable, HygieneId::ROOT);
1902        let iter_pat = self.alloc_pat_desugared(Pat::Bind { id: iter_binding, subpat: None });
1903        self.add_definition_to_binding(iter_binding, iter_pat);
1904        self.alloc_expr(
1905            Expr::Match {
1906                expr: iterator,
1907                arms: Box::new([MatchArm { pat: iter_pat, guard: None, expr: loop_outer }]),
1908            },
1909            syntax_ptr,
1910        )
1911    }

1913    /// Desugar `ast::TryExpr` from: `<expr>?` into:
1914    /// ```ignore (pseudo-rust)
1915    /// match Try::branch(<expr>) {
1916    ///     ControlFlow::Continue(val) => val,
1917    ///     ControlFlow::Break(residual) =>
1918    ///         // If there is an enclosing `try {...}`:
1919    ///         break 'catch_target Residual::into_try_type(residual),
1920    ///         // If there is an enclosing `try bikeshed Ty {...}`:
1921    ///         break 'catch_target Try::from_residual(residual),
1922    ///         // Otherwise:
1923    ///         return Try::from_residual(residual),
1924    /// }
1925    /// ```
1926    fn collect_try_operator(&mut self, syntax_ptr: AstPtr<ast::Expr>, e: ast::TryExpr) -> ExprId {
1927        let lang_items = self.lang_items();
1928        let try_branch = self.lang_path(lang_items.TryTraitBranch);
1929        let cf_continue = self.lang_path(lang_items.ControlFlowContinue);
1930        let cf_break = self.lang_path(lang_items.ControlFlowBreak);
1931        let operand = self.collect_expr_opt(e.expr());
1932        let try_branch = self.alloc_expr(try_branch.map_or(Expr::Missing, Expr::Path), syntax_ptr);
1933        let expr = self
1934            .alloc_expr(Expr::Call { callee: try_branch, args: Box::new([operand]) }, syntax_ptr);
1935        let continue_name = self.generate_new_name();
1936        let continue_binding = self.alloc_binding(
1937            continue_name.clone(),
1938            BindingAnnotation::Unannotated,
1939            HygieneId::ROOT,
1940        );
1941        let continue_bpat =
1942            self.alloc_pat_desugared(Pat::Bind { id: continue_binding, subpat: None });
1943        self.add_definition_to_binding(continue_binding, continue_bpat);
1944        let continue_arm = MatchArm {
1945            pat: self.alloc_pat_desugared(Pat::TupleStruct {
1946                path: cf_continue.map(Box::new),
1947                args: Box::new([continue_bpat]),
1948                ellipsis: None,
1949            }),
1950            guard: None,
1951            expr: self.alloc_expr(Expr::Path(Path::from(continue_name)), syntax_ptr),
1952        };
1953        let break_name = self.generate_new_name();
1954        let break_binding =
1955            self.alloc_binding(break_name.clone(), BindingAnnotation::Unannotated, HygieneId::ROOT);
1956        let break_bpat = self.alloc_pat_desugared(Pat::Bind { id: break_binding, subpat: None });
1957        self.add_definition_to_binding(break_binding, break_bpat);
1958        let break_arm = MatchArm {
1959            pat: self.alloc_pat_desugared(Pat::TupleStruct {
1960                path: cf_break.map(Box::new),
1961                args: Box::new([break_bpat]),
1962                ellipsis: None,
1963            }),
1964            guard: None,
1965            expr: {
1966                let it = self.alloc_expr(Expr::Path(Path::from(break_name)), syntax_ptr);
1967                let convert_fn = match self.current_try_block {
1968                    Some(TryBlock::Homogeneous { .. }) => {
1969                        self.lang_path(lang_items.ResidualIntoTryType)
1970                    }
1971                    Some(TryBlock::Heterogeneous { .. }) | None => {
1972                        self.lang_path(lang_items.TryTraitFromResidual)
1973                    }
1974                };
1975                let callee =
1976                    self.alloc_expr(convert_fn.map_or(Expr::Missing, Expr::Path), syntax_ptr);
1977                let result =
1978                    self.alloc_expr(Expr::Call { callee, args: Box::new([it]) }, syntax_ptr);
1979                self.alloc_expr(
1980                    match self.current_try_block {
1981                        Some(
1982                            TryBlock::Heterogeneous { label } | TryBlock::Homogeneous { label },
1983                        ) => Expr::Break { expr: Some(result), label: Some(label) },
1984                        None => Expr::Return { expr: Some(result) },
1985                    },
1986                    syntax_ptr,
1987                )
1988            },
1989        };
1990        let arms = Box::new([continue_arm, break_arm]);
1991        self.alloc_expr(Expr::Match { expr, arms }, syntax_ptr)
1992    }

1994    fn collect_macro_call<T, U>(
1995        &mut self,
1996        mcall: ast::MacroCall,
1997        syntax_ptr: AstPtr<ast::MacroCall>,
1998        record_diagnostics: bool,
1999        collector: impl FnOnce(&mut Self, Option<T>) -> U,
2000    ) -> U
2001    where
2002        T: ast::AstNode,
2003    {
2004        let macro_call_ptr = self.expander.in_file(syntax_ptr);
2005
2006        let block_call = self.def_map.modules[self.module].scope.macro_invoc(
2007            self.expander.in_file(self.expander.ast_id_map().ast_id_for_ptr(syntax_ptr)),
2008        );
2009        let res = match block_call {
2010            // fast path, macro call is in a block module
2011            Some(call) => Ok(self.expander.enter_expand_id(self.db, call)),
2012            None => {
2013                let resolver = |path: &_| {
2014                    self.def_map
2015                        .resolve_path(
2016                            self.local_def_map,
2017                            self.db,
2018                            self.module,
2019                            path,
2020                            crate::item_scope::BuiltinShadowMode::Other,
2021                            Some(MacroSubNs::Bang),
2022                        )
2023                        .0
2024                        .take_macros()
2025                };
2026                self.expander.enter_expand(
2027                    self.db,
2028                    mcall,
2029                    self.krate,
2030                    resolver,
2031                    &mut |ptr, call| {
2032                        _ = self.store.expansions.insert(ptr.map(|(it, _)| it), call);
2033                    },
2034                )
2035            }
2036        };
2037
2038        let res = match res {
2039            Ok(res) => res,
2040            Err(UnresolvedMacro { path }) => {
2041                if record_diagnostics {
2042                    self.store.diagnostics.push(ExpressionStoreDiagnostics::UnresolvedMacroCall {
2043                        node: self.expander.in_file(syntax_ptr),
2044                        path,
2045                    });
2046                }
2047                return collector(self, None);
2048            }
2049        };
2050        // No need to push macro and parsing errors as they'll be recreated from `macro_calls()`.
2051
2052        match res.value {
2053            Some((mark, expansion)) => {
2054                // Keep collecting even with expansion errors so we can provide completions and
2055                // other services in incomplete macro expressions.
2056                if let Some(macro_file) = self.expander.current_file_id().macro_file() {
2057                    self.store.expansions.insert(macro_call_ptr, macro_file);
2058                }
2059
2060                let id = collector(self, expansion.map(|it| it.tree()));
2061                self.expander.exit(mark);
2062                id
2063            }
2064            None => collector(self, None),
2065        }
2066    }

2106    fn collect_stmt(&mut self, statements: &mut Vec<Statement>, s: ast::Stmt) {
2107        match s {
2108            ast::Stmt::LetStmt(stmt) => {
2109                if !self.check_cfg(&stmt) {
2110                    return;
2111                }
2112                let pat = self.collect_pat_top(stmt.pat());
2113                let type_ref = stmt.ty().map(|it| self.lower_type_ref_disallow_impl_trait(it));
2114                let initializer = stmt.initializer().map(|e| self.collect_expr(e));
2115                let else_branch = stmt
2116                    .let_else()
2117                    .and_then(|let_else| let_else.block_expr())
2118                    .map(|block| self.collect_block(block));
2119                statements.push(Statement::Let { pat, type_ref, initializer, else_branch });
2120            }
2121            ast::Stmt::ExprStmt(stmt) => {
2122                let expr = stmt.expr();
2123                match &expr {
2124                    Some(expr) if !self.check_cfg(expr) => return,
2125                    _ => (),
2126                }
2127                let has_semi = stmt.semicolon_token().is_some();
2128                // Note that macro could be expanded to multiple statements
2129                if let Some(ast::Expr::MacroExpr(mac)) = expr {
2130                    if let Some(expr) = self.collect_macro_as_stmt(statements, mac) {
2131                        statements.push(Statement::Expr { expr, has_semi })
2132                    }
2133                } else {
2134                    let expr = self.collect_expr_opt(expr);
2135                    statements.push(Statement::Expr { expr, has_semi });
2136                }
2137            }
2138            ast::Stmt::Item(ast::Item::MacroDef(macro_)) => {
2139                if !self.check_cfg(&macro_) {
2140                    return;
2141                }
2142                let Some(name) = macro_.name() else {
2143                    statements.push(Statement::Item(Item::Other));
2144                    return;
2145                };
2146                let name = name.as_name();

2148                    self.def_map.modules[self.def_map.root].scope.get(&name).take_macros();

2151            ast::Stmt::Item(ast::Item::MacroRules(macro_)) => {

2153                    return;
2154                }

2156                    statements.push(Statement::Item(Item::Other));
2157                    return;
2158                };

2161                    self.current_block_legacy_macro_defs_count.entry(name.clone()).or_insert(0);
2162                let macro_id = self.def_map.modules[self.def_map.root]
2163                    .scope
2164                    .get_legacy_macro(&name)
2165                    .and_then(|it| it.get(*macro_defs_count))
2166                    .copied();

2168                self.collect_macro_def(statements, macro_id);

2170            ast::Stmt::Item(_item) => statements.push(Statement::Item(Item::Other)),
2171        }
2172    }

2296    fn collect_pat(&mut self, pat: ast::Pat, binding_list: &mut BindingList) -> PatId {
2297        let pattern = match &pat {
2298            ast::Pat::IdentPat(bp) => {
2299                let name = bp.name().map(|nr| nr.as_name()).unwrap_or_else(Name::missing);
2300                let hygiene = bp
2301                    .name()
2302                    .map(|name| self.hygiene_id_for(name.syntax().text_range()))
2303                    .unwrap_or(HygieneId::ROOT);
2304
2305                let annotation =
2306                    BindingAnnotation::new(bp.mut_token().is_some(), bp.ref_token().is_some());
2307                let subpat = bp.pat().map(|subpat| self.collect_pat(subpat, binding_list));
2308
2309                let is_simple_ident_pat =
2310                    annotation == BindingAnnotation::Unannotated && subpat.is_none();
2311                let (binding, pattern) = if is_simple_ident_pat {
2312                    // This could also be a single-segment path pattern. To
2313                    // decide that, we need to try resolving the name.
2314                    let (resolved, _) = self.def_map.resolve_path(
2315                        self.local_def_map,
2316                        self.db,
2317                        self.module,
2318                        &name.clone().into(),
2319                        BuiltinShadowMode::Other,
2320                        None,
2321                    );
2322                    // Funnily enough, record structs/variants *can* be shadowed
2323                    // by pattern bindings (but unit or tuple structs/variants
2324                    // can't).
2325                    match resolved.take_values() {
2326                        Some(ModuleDefId::ConstId(_)) => (None, Pat::Path(name.into())),
2327                        Some(ModuleDefId::EnumVariantId(variant))
2328                        // FIXME: This can cause a cycle if the user is writing invalid code
2329                            if variant.fields(self.db).shape != FieldsShape::Record =>
2330                        {
2331                            (None, Pat::Path(name.into()))
2332                        }
2333                        Some(ModuleDefId::AdtId(AdtId::StructId(s)))
2334                        // FIXME: This can cause a cycle if the user is writing invalid code
2335                            if self.db.struct_signature(s).shape != FieldsShape::Record =>
2336                        {
2337                            (None, Pat::Path(name.into()))
2338                        }
2339                        // shadowing statics is an error as well, so we just ignore that case here
2340                        _ => {
2341                            let id = binding_list.find(self, name, hygiene, annotation);
2342                            (Some(id), Pat::Bind { id, subpat })
2343                        }
2344                    }
2345                } else {
2346                    let id = binding_list.find(self, name, hygiene, annotation);
2347                    (Some(id), Pat::Bind { id, subpat })
2348                };
2349
2350                let ptr = AstPtr::new(&pat);
2351                let pat = self.alloc_pat(pattern, ptr);
2352                if let Some(binding_id) = binding {
2353                    self.add_definition_to_binding(binding_id, pat);
2354                }
2355                return pat;
2356            }
2357            ast::Pat::TupleStructPat(p) => {
2358                let path = p
2359                    .path()
2360                    .and_then(|path| self.lower_path(path, &mut Self::impl_trait_error_allocator))
2361                    .map(Box::new);
2362                let (args, ellipsis) = self.collect_tuple_pat(
2363                    p.fields(),
2364                    comma_follows_token(p.l_paren_token()),
2365                    binding_list,
2366                );
2367                Pat::TupleStruct { path, args, ellipsis }
2368            }
2369            ast::Pat::RefPat(p) => {
2370                let pat = self.collect_pat_opt(p.pat(), binding_list);
2371                let mutability = Mutability::from_mutable(p.mut_token().is_some());
2372                Pat::Ref { pat, mutability }
2373            }
2374            ast::Pat::PathPat(p) => {
2375                let path = p
2376                    .path()
2377                    .and_then(|path| self.lower_path(path, &mut Self::impl_trait_error_allocator));
2378                path.map(Pat::Path).unwrap_or(Pat::Missing)
2379            }
2380            ast::Pat::OrPat(p) => 'b: {
2381                let prev_is_used = mem::take(&mut binding_list.is_used);
2382                let prev_reject_new = mem::take(&mut binding_list.reject_new);
2383                let mut pats = Vec::with_capacity(p.pats().count());
2384                let mut it = p.pats();
2385                let Some(first) = it.next() else {
2386                    break 'b Pat::Or(Box::new([]));
2387                };
2388                pats.push(self.collect_pat(first, binding_list));
2389                binding_list.reject_new = true;
2390                for rest in it {
2391                    for (_, it) in binding_list.is_used.iter_mut() {
2392                        *it = false;
2393                    }
2394                    pats.push(self.collect_pat(rest, binding_list));
2395                    for (&id, &is_used) in binding_list.is_used.iter() {
2396                        if !is_used {
2397                            self.store.bindings[id].problems =
2398                                Some(BindingProblems::NotBoundAcrossAll);
2399                        }
2400                    }
2401                }
2402                binding_list.reject_new = prev_reject_new;
2403                let current_is_used = mem::replace(&mut binding_list.is_used, prev_is_used);
2404                for (id, _) in current_is_used.into_iter() {
2405                    binding_list.check_is_used(self, id);
2406                }
2407                if let &[pat] = &*pats {
2408                    // Leading pipe without real OR pattern. Leaving an one-item OR pattern may confuse later stages.
2409                    return pat;
2410                }
2411                Pat::Or(pats.into())
2412            }
2413            ast::Pat::ParenPat(p) => return self.collect_pat_opt(p.pat(), binding_list),
2414            ast::Pat::TuplePat(p) => {
2415                let (args, ellipsis) = self.collect_tuple_pat(
2416                    p.fields(),
2417                    comma_follows_token(p.l_paren_token()),
2418                    binding_list,
2419                );
2420                Pat::Tuple { args, ellipsis }
2421            }
2422            ast::Pat::WildcardPat(_) => Pat::Wild,
2423            ast::Pat::RecordPat(p) => {
2424                let path = p
2425                    .path()
2426                    .and_then(|path| self.lower_path(path, &mut Self::impl_trait_error_allocator))
2427                    .map(Box::new);
2428                let record_pat_field_list =
2429                    &p.record_pat_field_list().expect("every struct should have a field list");
2430                let args = record_pat_field_list
2431                    .fields()
2432                    .filter_map(|f| {
2433                        if !self.check_cfg(&f) {
2434                            return None;
2435                        }
2436                        let ast_pat = f.pat()?;
2437                        let pat = self.collect_pat(ast_pat, binding_list);
2438                        let name = f.field_name()?.as_name();
2439                        let src = self.expander.in_file(AstPtr::new(&f).wrap_right());
2440                        self.store.pat_field_map_back.insert(pat, src);
2441                        Some(RecordFieldPat { name, pat })
2442                    })
2443                    .collect();
2444
2445                let ellipsis = record_pat_field_list.rest_pat().is_some();
2446
2447                Pat::Record { path, args, ellipsis }
2448            }
2449            ast::Pat::SlicePat(p) => {
2450                let SlicePatComponents { prefix, slice, suffix } = p.components();
2451
2452                Pat::Slice {
2453                    prefix: prefix.into_iter().map(|p| self.collect_pat(p, binding_list)).collect(),
2454                    slice: slice.map(|p| self.collect_pat(p, binding_list)),
2455                    suffix: suffix.into_iter().map(|p| self.collect_pat(p, binding_list)).collect(),
2456                }
2457            }
2458            ast::Pat::LiteralPat(lit) => 'b: {
2459                let Some((hir_lit, ast_lit)) = pat_literal_to_hir(lit) else {
2460                    break 'b Pat::Missing;
2461                };
2462                let expr = Expr::Literal(hir_lit);
2463                let expr_ptr = AstPtr::new(&ast::Expr::Literal(ast_lit));
2464                let expr_id = self.alloc_expr(expr, expr_ptr);
2465                Pat::Lit(expr_id)
2466            }
2467            ast::Pat::RestPat(_) => {
2468                // `RestPat` requires special handling and should not be mapped
2469                // to a Pat. Here we are using `Pat::Missing` as a fallback for
2470                // when `RestPat` is mapped to `Pat`, which can easily happen
2471                // when the source code being analyzed has a malformed pattern
2472                // which includes `..` in a place where it isn't valid.
2473
2474                Pat::Missing
2475            }
2476            ast::Pat::BoxPat(boxpat) => {
2477                let inner = self.collect_pat_opt(boxpat.pat(), binding_list);
2478                Pat::Box { inner }
2479            }
2480            ast::Pat::ConstBlockPat(const_block_pat) => {
2481                if let Some(block) = const_block_pat.block_expr() {
2482                    let expr_id = self.with_label_rib(RibKind::Constant, |this| {
2483                        this.with_binding_owner(|this| this.collect_block(block))
2484                    });
2485                    Pat::ConstBlock(expr_id)
2486                } else {
2487                    Pat::Missing
2488                }
2489            }
2490            ast::Pat::MacroPat(mac) => match mac.macro_call() {
2491                Some(call) => {
2492                    let macro_ptr = AstPtr::new(&call);
2493                    let src = self.expander.in_file(AstPtr::new(&pat));
2494                    let pat =
2495                        self.collect_macro_call(call, macro_ptr, true, |this, expanded_pat| {
2496                            this.collect_pat_opt(expanded_pat, binding_list)
2497                        });
2498                    self.store.pat_map.insert(src, pat.into());
2499                    return pat;
2500                }
2501                None => Pat::Missing,
2502            },
2503            ast::Pat::RangePat(p) => {
2504                let mut range_part_lower = |p: Option<ast::Pat>| -> Option<ExprId> {
2505                    p.and_then(|it| {
2506                        let ptr = PatPtr::new(&it);
2507                        match &it {
2508                            ast::Pat::LiteralPat(it) => Some(self.alloc_expr_from_pat(
2509                                Expr::Literal(pat_literal_to_hir(it)?.0),
2510                                ptr,
2511                            )),
2512                            ast::Pat::IdentPat(ident) if ident.is_simple_ident() => ident
2513                                .name()
2514                                .map(|name| name.as_name())
2515                                .map(Path::from)
2516                                .map(|path| self.alloc_expr_from_pat(Expr::Path(path), ptr)),
2517                            ast::Pat::PathPat(p) => p
2518                                .path()
2519                                .and_then(|path| {
2520                                    self.lower_path(path, &mut Self::impl_trait_error_allocator)
2521                                })
2522                                .map(|parsed| self.alloc_expr_from_pat(Expr::Path(parsed), ptr)),
2523                            // We only need to handle literal, ident (if bare) and path patterns here,
2524                            // as any other pattern as a range pattern operand is semantically invalid.
2525                            _ => None,
2526                        }
2527                    })
2528                };
2529                let start = range_part_lower(p.start());
2530                let end = range_part_lower(p.end());
2531                match p.op_kind() {
2532                    Some(range_type) => Pat::Range { start, end, range_type },
2533                    None => Pat::Missing,
2534                }
2535            }
2536        };
2537        let ptr = AstPtr::new(&pat);
2538        self.alloc_pat(pattern, ptr)
2539    }

2548    fn collect_tuple_pat(
2549        &mut self,
2550        args: AstChildren<ast::Pat>,
2551        has_leading_comma: bool,
2552        binding_list: &mut BindingList,
2553    ) -> (Box<[PatId]>, Option<u32>) {
2554        let args: Vec<_> = args.map(|p| self.collect_pat_possibly_rest(p, binding_list)).collect();
2555        // Find the location of the `..`, if there is one. Note that we do not
2556        // consider the possibility of there being multiple `..` here.
2557        let ellipsis = args.iter().position(|p| p.is_right()).map(|it| it as u32);
2558
2559        // We want to skip the `..` pattern here, since we account for it above.
2560        let mut args: Vec<_> = args.into_iter().filter_map(Either::left).collect();
2561        // if there is a leading comma, the user is most likely to type out a leading pattern
2562        // so we insert a missing pattern at the beginning for IDE features
2563        if has_leading_comma {
2564            args.insert(0, self.missing_pat());
2565        }
2566
2567        (args.into_boxed_slice(), ellipsis)
2568    }

2570    // `collect_pat` rejects `ast::Pat::RestPat`, but it should be handled in some cases that
2571    // it is the macro expansion result of an arg sub-pattern in a slice or tuple pattern.
2572    fn collect_pat_possibly_rest(
2573        &mut self,
2574        pat: ast::Pat,
2575        binding_list: &mut BindingList,
2576    ) -> Either<PatId, ()> {
2577        match &pat {
2578            ast::Pat::RestPat(_) => Either::Right(()),
2579            ast::Pat::MacroPat(mac) => match mac.macro_call() {
2580                Some(call) => {
2581                    let macro_ptr = AstPtr::new(&call);
2582                    let src = self.expander.in_file(AstPtr::new(&pat));
2583                    let pat =
2584                        self.collect_macro_call(call, macro_ptr, true, |this, expanded_pat| {
2585                            if let Some(expanded_pat) = expanded_pat {
2586                                this.collect_pat_possibly_rest(expanded_pat, binding_list)
2587                            } else {
2588                                Either::Left(this.missing_pat())
2589                            }
2590                        });
2591                    if let Some(pat) = pat.left() {
2592                        self.store.pat_map.insert(src, pat.into());
2593                    }
2594                    pat
2595                }
2596                None => {
2597                    let ptr = AstPtr::new(&pat);
2598                    Either::Left(self.alloc_pat(Pat::Missing, ptr))
2599                }
2600            },
2601            _ => Either::Left(self.collect_pat(pat, binding_list)),
2602        }
2603    }

2640    fn resolve_label(
2641        &self,
2642        lifetime: Option<ast::Lifetime>,
2643    ) -> Result<Option<LabelId>, ExpressionStoreDiagnostics> {
2644        let Some(lifetime) = lifetime else { return Ok(None) };
2645        let mut hygiene_id =
2646            self.expander.hygiene_for_range(self.db, lifetime.syntax().text_range());
2647        let mut hygiene_info = if hygiene_id.is_root() {
2648            None
2649        } else {
2650            hygiene_id.syntax_context().outer_expn(self.db).map(|expansion| {
2651                let expansion = self.db.lookup_intern_macro_call(expansion.into());
2652                (hygiene_id.syntax_context().parent(self.db), expansion.def)
2653            })
2654        };
2655        let name = Name::new_lifetime(&lifetime.text());
2656
2657        for (rib_idx, rib) in self.label_ribs.iter().enumerate().rev() {
2658            match &rib.kind {
2659                RibKind::Normal(label_name, id, label_hygiene) => {
2660                    if *label_name == name && *label_hygiene == hygiene_id {
2661                        return if self.is_label_valid_from_rib(rib_idx) {
2662                            Ok(Some(*id))
2663                        } else {
2664                            Err(ExpressionStoreDiagnostics::UnreachableLabel {
2665                                name,
2666                                node: self.expander.in_file(AstPtr::new(&lifetime)),
2667                            })
2668                        };
2669                    }
2670                }
2671                RibKind::MacroDef(macro_id) => {
2672                    if let Some((parent_ctx, label_macro_id)) = hygiene_info
2673                        && label_macro_id == **macro_id
2674                    {
2675                        // A macro is allowed to refer to labels from before its declaration.
2676                        // Therefore, if we got to the rib of its declaration, give up its hygiene
2677                        // and use its parent expansion.
2678
2679                        hygiene_id = HygieneId::new(parent_ctx.opaque_and_semiopaque(self.db));
2680                        hygiene_info = parent_ctx.outer_expn(self.db).map(|expansion| {
2681                            let expansion = self.db.lookup_intern_macro_call(expansion.into());
2682                            (parent_ctx.parent(self.db), expansion.def)
2683                        });
2684                    }
2685                }
2686                _ => {}
2687            }
2688        }
2689
2690        Err(ExpressionStoreDiagnostics::UndeclaredLabel {
2691            name,
2692            node: self.expander.in_file(AstPtr::new(&lifetime)),
2693        })
2694    }
