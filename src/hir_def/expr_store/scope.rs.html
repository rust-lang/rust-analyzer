46#[derive(Debug, PartialEq, Eq)]
47pub struct ScopeData {
48    parent: Option<ScopeId>,
49    block: Option<BlockId>,
50    label: Option<(LabelId, Name)>,

56#[salsa::tracked]
57impl ExprScopes {
58    #[salsa::tracked(returns(ref))]
59    pub fn body_expr_scopes(db: &dyn DefDatabase, def: DefWithBodyId) -> ExprScopes {
60        let body = Body::of(db, def);
61        let mut scopes = ExprScopes::new_body(body);
62        scopes.shrink_to_fit();
63        scopes
64    }
65
66    #[salsa::tracked(returns(ref))]
67    pub fn sig_expr_scopes(db: &dyn DefDatabase, def: GenericDefId) -> ExprScopes {
68        let (_, store) = GenericParams::with_store(db, def);
69        let roots = store.expr_roots();
70        let mut scopes = ExprScopes::new_store(store, roots);
71        scopes.shrink_to_fit();
72        scopes
73    }
74
75    #[salsa::tracked(returns(ref))]
76    pub fn variant_scopes(db: &dyn DefDatabase, def: VariantId) -> ExprScopes {
77        let fields = VariantFields::of(db, def);
78        let roots = fields.store.expr_roots();
79        let mut scopes = ExprScopes::new_store(&fields.store, roots);
80        scopes.shrink_to_fit();
81        scopes
82    }
83}

85impl ExprScopes {
86    #[inline]
87    pub fn of(db: &dyn DefDatabase, def: impl Into<ExpressionStoreOwnerId>) -> &ExprScopes {
88        match def.into() {
89            ExpressionStoreOwnerId::Body(def) => Self::body_expr_scopes(db, def),
90            ExpressionStoreOwnerId::Signature(def) => Self::sig_expr_scopes(db, def),
91            ExpressionStoreOwnerId::VariantFields(variant_id) => {
92                Self::variant_scopes(db, variant_id)
93            }
94        }
95    }
96
97    pub fn entries(&self, scope: ScopeId) -> &[ScopeEntry] {
98        &self.scope_entries[self.scopes[scope].entries.clone()]
99    }
100
101    /// If `scope` refers to a block expression scope, returns the corresponding `BlockId`.
102    pub fn block(&self, scope: ScopeId) -> Option<BlockId> {
103        self.scopes[scope].block
104    }
105
106    /// If `scope` refers to a macro def scope, returns the corresponding `MacroId`.
107    #[allow(clippy::borrowed_box)] // If we return `&MacroDefId` we need to move it, this way we just clone the `Box`.
108    pub fn macro_def(&self, scope: ScopeId) -> Option<&Box<MacroDefId>> {
109        self.scopes[scope].macro_def.as_ref()
110    }
111
112    /// If `scope` refers to a labeled expression scope, returns the corresponding `Label`.
113    pub fn label(&self, scope: ScopeId) -> Option<(LabelId, Name)> {
114        self.scopes[scope].label.clone()
115    }
116
117    /// Returns the scopes in ascending order.
118    pub fn scope_chain(&self, scope: Option<ScopeId>) -> impl Iterator<Item = ScopeId> + '_ {
119        std::iter::successors(scope, move |&scope| self.scopes[scope].parent)
120    }
121
122    pub fn resolve_name_in_scope(&self, scope: ScopeId, name: &Name) -> Option<&ScopeEntry> {
123        self.scope_chain(Some(scope))
124            .find_map(|scope| self.entries(scope).iter().find(|it| it.name == *name))
125    }
126
127    pub fn scope_for(&self, expr: ExprId) -> Option<ScopeId> {
128        self.scope_by_expr.get(expr).copied()
129    }
130
131    pub fn scope_by_expr(&self) -> &ArenaMap<ExprId, ScopeId> {
132        &self.scope_by_expr
133    }
134}

136fn empty_entries(idx: usize) -> IdxRange<ScopeEntry> {
137    IdxRange::new(Idx::from_raw(RawIdx::from(idx as u32))..Idx::from_raw(RawIdx::from(idx as u32)))
138}

140impl ExprScopes {
141    fn new_body(body: &Body) -> ExprScopes {
142        let mut scopes = ExprScopes {
143            scopes: Arena::default(),
144            scope_entries: Arena::default(),
145            scope_by_expr: ArenaMap::with_capacity(
146                body.expr_only.as_ref().map_or(0, |it| it.exprs.len()),
147            ),
148        };
149        let mut root = scopes.root_scope();
150        if let Some(self_param) = body.self_param {
151            scopes.add_bindings(body, root, self_param, body.binding_hygiene(self_param));
152        }
153        scopes.add_params_bindings(body, root, &body.params);
154        compute_expr_scopes(body.root_expr(), body, &mut scopes, &mut root);
155        scopes
156    }
157
158    fn new_store(store: &ExpressionStore, roots: impl IntoIterator<Item = ExprId>) -> ExprScopes {
159        let mut scopes = ExprScopes {
160            scopes: Arena::default(),
161            scope_entries: Arena::default(),
162            scope_by_expr: ArenaMap::with_capacity(
163                store.expr_only.as_ref().map_or(0, |it| it.exprs.len()),
164            ),
165        };
166        let root = scopes.root_scope();
167        for root_expr in roots {
168            let mut scope = scopes.new_scope(root);
169            compute_expr_scopes(root_expr, store, &mut scopes, &mut scope);
170        }
171        scopes
172    }
173
174    fn root_scope(&mut self) -> ScopeId {
175        self.scopes.alloc(ScopeData {
176            parent: None,
177            block: None,
178            label: None,
179            macro_def: None,
180            entries: empty_entries(self.scope_entries.len()),
181        })
182    }

184    fn new_scope(&mut self, parent: ScopeId) -> ScopeId {
185        self.scopes.alloc(ScopeData {
186            parent: Some(parent),
187            block: None,
188            label: None,
189            macro_def: None,
190            entries: empty_entries(self.scope_entries.len()),
191        })
192    }
193
194    fn new_labeled_scope(&mut self, parent: ScopeId, label: Option<(LabelId, Name)>) -> ScopeId {
195        self.scopes.alloc(ScopeData {
196            parent: Some(parent),
197            block: None,
198            label,
199            macro_def: None,
200            entries: empty_entries(self.scope_entries.len()),
201        })
202    }

204    fn new_block_scope(
205        &mut self,
206        parent: ScopeId,
207        block: Option<BlockId>,
208        label: Option<(LabelId, Name)>,
209    ) -> ScopeId {
210        self.scopes.alloc(ScopeData {
211            parent: Some(parent),
212            block,
213            label,
214            macro_def: None,
215            entries: empty_entries(self.scope_entries.len()),
216        })
217    }
218
219    fn new_macro_def_scope(&mut self, parent: ScopeId, macro_id: Box<MacroDefId>) -> ScopeId {
220        self.scopes.alloc(ScopeData {
221            parent: Some(parent),
222            block: None,
223            label: None,
224            macro_def: Some(macro_id),
225            entries: empty_entries(self.scope_entries.len()),
226        })
227    }
228
229    fn add_bindings(
230        &mut self,
231        store: &ExpressionStore,
232        scope: ScopeId,
233        binding: BindingId,
234        hygiene: HygieneId,
235    ) {
236        let Binding { name, .. } = &store[binding];
237        let entry = self.scope_entries.alloc(ScopeEntry { name: name.clone(), binding, hygiene });
238        self.scopes[scope].entries =
239            IdxRange::new_inclusive(self.scopes[scope].entries.start()..=entry);
240    }
241
242    fn add_pat_bindings(&mut self, store: &ExpressionStore, scope: ScopeId, pat: PatId) {
243        let pattern = &store[pat];
244        if let Pat::Bind { id, .. } = *pattern {
245            self.add_bindings(store, scope, id, store.binding_hygiene(id));
246        }
247
248        pattern.walk_child_pats(|pat| self.add_pat_bindings(store, scope, pat));
249    }
250
251    fn add_params_bindings(&mut self, store: &ExpressionStore, scope: ScopeId, params: &[PatId]) {
252        params.iter().for_each(|pat| self.add_pat_bindings(store, scope, *pat));
253    }
254
255    fn set_scope(&mut self, node: ExprId, scope: ScopeId) {
256        self.scope_by_expr.insert(node, scope);
257    }
258
259    fn shrink_to_fit(&mut self) {
260        let ExprScopes { scopes, scope_entries, scope_by_expr } = self;
261        scopes.shrink_to_fit();
262        scope_entries.shrink_to_fit();
263        scope_by_expr.shrink_to_fit();
264    }
265}

267fn compute_block_scopes(
268    statements: &[Statement],
269    tail: Option<ExprId>,
270    store: &ExpressionStore,
271    scopes: &mut ExprScopes,
272    scope: &mut ScopeId,
273) {
274    for stmt in statements {
275        match stmt {
276            Statement::Let { pat, initializer, else_branch, .. } => {
277                if let Some(expr) = initializer {
278                    compute_expr_scopes(*expr, store, scopes, scope);
279                }
280                if let Some(expr) = else_branch {
281                    compute_expr_scopes(*expr, store, scopes, scope);
282                }
283
284                *scope = scopes.new_scope(*scope);
285                scopes.add_pat_bindings(store, *scope, *pat);
286            }
287            Statement::Expr { expr, .. } => {
288                compute_expr_scopes(*expr, store, scopes, scope);
289            }
290            Statement::Item(Item::MacroDef(macro_id)) => {
291                *scope = scopes.new_macro_def_scope(*scope, macro_id.clone());
292            }
293            Statement::Item(Item::Other) => (),
294        }
295    }
296    if let Some(expr) = tail {
297        compute_expr_scopes(expr, store, scopes, scope);
298    }
299}
300
301fn compute_expr_scopes(
302    expr: ExprId,
303    store: &ExpressionStore,
304    scopes: &mut ExprScopes,
305    scope: &mut ScopeId,
306) {
307    let make_label =
308        |label: &Option<LabelId>| label.map(|label| (label, store[label].name.clone()));
309
310    let compute_expr_scopes = |scopes: &mut ExprScopes, expr: ExprId, scope: &mut ScopeId| {
311        compute_expr_scopes(expr, store, scopes, scope)
312    };
313
314    scopes.set_scope(expr, *scope);
315    match &store[expr] {
316        Expr::Block { statements, tail, id, label } => {
317            let mut scope = scopes.new_block_scope(*scope, *id, make_label(label));
318            // Overwrite the old scope for the block expr, so that every block scope can be found
319            // via the block itself (important for blocks that only contain items, no expressions).
320            scopes.set_scope(expr, scope);
321            compute_block_scopes(statements, *tail, store, scopes, &mut scope);
322        }
323        Expr::Const(id) => {
324            let mut scope = scopes.root_scope();
325            compute_expr_scopes(scopes, *id, &mut scope);
326        }
327        Expr::Unsafe { id, statements, tail } => {
328            let mut scope = scopes.new_block_scope(*scope, *id, None);
329            // Overwrite the old scope for the block expr, so that every block scope can be found
330            // via the block itself (important for blocks that only contain items, no expressions).
331            scopes.set_scope(expr, scope);
332            compute_block_scopes(statements, *tail, store, scopes, &mut scope);
333        }
334        Expr::Loop { body: body_expr, label } => {
335            let mut scope = scopes.new_labeled_scope(*scope, make_label(label));
336            compute_expr_scopes(scopes, *body_expr, &mut scope);
337        }
338        Expr::Closure { args, body: body_expr, .. } => {
339            let mut scope = scopes.new_scope(*scope);
340            scopes.add_params_bindings(store, scope, args);
341            compute_expr_scopes(scopes, *body_expr, &mut scope);
342        }
343        Expr::Match { expr, arms } => {
344            compute_expr_scopes(scopes, *expr, scope);
345            for arm in arms.iter() {
346                let mut scope = scopes.new_scope(*scope);
347                scopes.add_pat_bindings(store, scope, arm.pat);
348                if let Some(guard) = arm.guard {
349                    scope = scopes.new_scope(scope);
350                    compute_expr_scopes(scopes, guard, &mut scope);
351                }
352                compute_expr_scopes(scopes, arm.expr, &mut scope);
353            }
354        }
355        &Expr::If { condition, then_branch, else_branch } => {
356            let mut then_branch_scope = scopes.new_scope(*scope);
357            compute_expr_scopes(scopes, condition, &mut then_branch_scope);
358            compute_expr_scopes(scopes, then_branch, &mut then_branch_scope);
359            if let Some(else_branch) = else_branch {
360                compute_expr_scopes(scopes, else_branch, scope);
361            }
362        }
363        &Expr::Let { pat, expr } => {
364            compute_expr_scopes(scopes, expr, scope);
365            *scope = scopes.new_scope(*scope);
366            scopes.add_pat_bindings(store, *scope, pat);
367        }
368        _ => store.walk_child_exprs(expr, |e| compute_expr_scopes(scopes, e, scope)),
369    };
370}

399    fn do_check(#[rust_analyzer::rust_fixture] ra_fixture: &str, expected: &[&str]) {
400        let (offset, code) = extract_offset(ra_fixture);
401        let code = {
402            let mut buf = String::new();
403            let off: usize = offset.into();
404            buf.push_str(&code[..off]);
405            buf.push_str("$0marker");
406            buf.push_str(&code[off..]);
407            buf
408        };
409
410        let (db, position) = TestDB::with_position(&code);
411        let editioned_file_id = position.file_id;
412        let offset = position.offset;
413
414        let (file_id, _) = editioned_file_id.unpack(&db);
415
416        let file_syntax = editioned_file_id.parse(&db).syntax_node();
417        let marker: ast::PathExpr = find_node_at_offset(&file_syntax, offset).unwrap();
418        let function = find_function(&db, file_id);
419
420        let scopes = ExprScopes::of(&db, DefWithBodyId::from(function));
421        let (_body, source_map) = Body::with_source_map(&db, function.into());
422
423        let expr_id = source_map
424            .node_expr(InFile { file_id: editioned_file_id.into(), value: &marker.into() })
425            .unwrap()
426            .as_expr()
427            .unwrap();
428        let scope = scopes.scope_for(expr_id);
429
430        let actual = scopes
431            .scope_chain(scope)
432            .flat_map(|scope| scopes.entries(scope))
433            .map(|it| it.name().as_str())
434            .collect::<Vec<_>>()
435            .join("\n");
436        let expected = expected.join("\n");
437        assert_eq_text!(&expected, &actual);
438    }

565    fn do_check_local_name(#[rust_analyzer::rust_fixture] ra_fixture: &str, expected_offset: u32) {
566        let (db, position) = TestDB::with_position(ra_fixture);
567        let editioned_file_id = position.file_id;
568        let offset = position.offset;
569
570        let (file_id, _) = editioned_file_id.unpack(&db);
571
572        let file = editioned_file_id.parse(&db).ok().unwrap();
573        let expected_name = find_node_at_offset::<ast::Name>(file.syntax(), expected_offset.into())
574            .expect("failed to find a name at the target offset");
575        let name_ref: ast::NameRef = find_node_at_offset(file.syntax(), offset).unwrap();
576
577        let function = find_function(&db, file_id);
578
579        let scopes = ExprScopes::body_expr_scopes(&db, DefWithBodyId::from(function));
580        let (_, source_map) = Body::with_source_map(&db, function.into());
581
582        let expr_scope = {
583            let expr_ast = name_ref.syntax().ancestors().find_map(ast::Expr::cast).unwrap();
584            let expr_id = source_map
585                .node_expr(InFile { file_id: editioned_file_id.into(), value: &expr_ast })
586                .unwrap()
587                .as_expr()
588                .unwrap();
589            scopes.scope_for(expr_id).unwrap()
590        };
591
592        let resolved = scopes.resolve_name_in_scope(expr_scope, &name_ref.as_name()).unwrap();
593        let pat_src =
594            source_map.pat_syntax(source_map.patterns_for_binding(resolved.binding())[0]).unwrap();
595
596        let local_name = pat_src.value.syntax_node_ptr().to_node(file.syntax());
597        assert_eq!(local_name.text_range(), expected_name.syntax().text_range());
598    }
