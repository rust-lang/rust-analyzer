1//! This module implements import-resolution/macro expansion algorithm.
2//!
3//! The result of this module is `DefMap`: a data structure which contains:
4//!
5//!   * a tree of modules for the crate
6//!   * for each module, a set of items visible in the module (directly declared
7//!     or imported)
8//!
9//! Note that `DefMap` contains fully macro expanded code.
10//!
11//! Computing `DefMap` can be partitioned into several logically
12//! independent "phases". The phases are mutually recursive though, there's no
13//! strict ordering.
14//!
15//! ## Collecting RawItems
16//!
17//! This happens in the `raw` module, which parses a single source file into a
18//! set of top-level items. Nested imports are desugared to flat imports in this
19//! phase. Macro calls are represented as a triple of `(Path, Option<Name>,
20//! TokenTree)`.
21//!
22//! ## Collecting Modules
23//!
24//! This happens in the `collector` module. In this phase, we recursively walk
25//! tree of modules, collect raw items from submodules, populate module scopes
26//! with defined items (so, we assign item ids in this phase) and record the set
27//! of unresolved imports and macros.
28//!
29//! While we walk tree of modules, we also record macro_rules definitions and
30//! expand calls to macro_rules defined macros.
31//!
32//! ## Resolving Imports
33//!
34//! We maintain a list of currently unresolved imports. On every iteration, we
35//! try to resolve some imports from this list. If the import is resolved, we
36//! record it, by adding an item to current module scope and, if necessary, by
37//! recursively populating glob imports.
38//!
39//! ## Resolving Macros
40//!
41//! macro_rules from the same crate use a global mutable namespace. We expand
42//! them immediately, when we collect modules.
43//!
44//! Macros from other crates (including proc-macros) can be used with
45//! `foo::bar!` syntax. We handle them similarly to imports. There's a list of
46//! unexpanded macros. On every iteration, we try to resolve each macro call
47//! path and, upon success, we run macro expansion and "collect module" phase on
48//! the result

99/// Parts of the def map that are only needed when analyzing code in the same crate.
100///
101/// There are some data in the def map (e.g. extern prelude) that is only needed when analyzing
102/// things in the same crate (and maybe in the IDE layer), e.g. the extern prelude. If we put
103/// it in the DefMap dependant DefMaps will be invalidated when they change (e.g. when we add
104/// a dependency to the crate). Instead we split them out of the DefMap into a LocalDefMap struct.
105/// `crate_local_def_map()` returns both, and `crate_def_map()` returns only the external-relevant
106/// DefMap.
107#[derive(Debug, PartialEq, Eq, Default)]
108pub struct LocalDefMap {
109    // FIXME: There are probably some other things that could be here, but this is less severe and you
110    // need to be careful with things that block def maps also have.
111    /// The extern prelude which contains all root modules of external crates that are in scope.
112    extern_prelude: FxIndexMap<Name, (CrateRootModuleId, Option<ExternCrateId>)>,
113}
114
115impl LocalDefMap {
116    pub(crate) const EMPTY: &Self =
117        &Self { extern_prelude: FxIndexMap::with_hasher(rustc_hash::FxBuildHasher) };
118
119    fn shrink_to_fit(&mut self) {
120        let Self { extern_prelude } = self;
121        extern_prelude.shrink_to_fit();
122    }
123
124    pub(crate) fn extern_prelude(
125        &self,
126    ) -> impl DoubleEndedIterator<Item = (&Name, (CrateRootModuleId, Option<ExternCrateId>))> + '_
127    {
128        self.extern_prelude.iter().map(|(name, &def)| (name, def))
129    }

132/// Contains the results of (early) name resolution.
133///
134/// A `DefMap` stores the module tree and the definitions that are in scope in every module after
135/// item-level macros have been expanded.
136///
137/// Every crate has a primary `DefMap` whose root is the crate's main file (`main.rs`/`lib.rs`),
138/// computed by the `crate_def_map` query. Additionally, every block expression introduces the
139/// opportunity to write arbitrary item and module hierarchies, and thus gets its own `DefMap` that
140/// is computed by the `block_def_map` query.
141#[derive(Debug, PartialEq, Eq)]
142pub struct DefMap {
143    /// The crate this `DefMap` belongs to.
144    krate: Crate,
145    /// When this is a block def map, this will hold the block id of the block and module that
146    /// contains this block.
147    block: Option<BlockInfo>,
148    /// The modules and their data declared in this crate.
149    pub modules: Arena<ModuleData>,
150    /// The prelude module for this crate. This either comes from an import
151    /// marked with the `prelude_import` attribute, or (in the normal case) from
152    /// a dependency (`std` or `core`).
153    /// The prelude is empty for non-block DefMaps (unless `#[prelude_import]` was used,
154    /// but that attribute is nightly and when used in a block, it affects resolution globally
155    /// so we aren't handling this correctly anyways).
156    prelude: Option<(ModuleId, Option<UseId>)>,
157    /// `macro_use` prelude that contains macros from `#[macro_use]`'d external crates. Note that
158    /// this contains all kinds of macro, not just `macro_rules!` macro.
159    /// ExternCrateId being None implies it being imported from the general prelude import.
160    macro_use_prelude: FxHashMap<Name, (MacroId, Option<ExternCrateId>)>,
161
162    // FIXME: AstId's are fairly unstable
163    /// Tracks which custom derives are in scope for an item, to allow resolution of derive helper
164    /// attributes.
165    // FIXME: Figure out a better way for the IDE layer to resolve these?
166    derive_helpers_in_scope: FxHashMap<AstId<ast::Item>, Vec<(Name, MacroId, MacroCallId)>>,
167    // FIXME: AstId's are fairly unstable
168    /// A mapping from [`hir_expand::MacroDefId`] to [`crate::MacroId`].
169    pub macro_def_to_macro_id: FxHashMap<ErasedAstId, MacroId>,
170
171    /// The diagnostics that need to be emitted for this crate.
172    diagnostics: Vec<DefDiagnostic>,
173
174    /// The crate data that is shared between a crate's def map and all its block def maps.
175    data: Arc<DefMapCrateData>,
176}

178/// Data that belongs to a crate which is shared between a crate's def map and all its block def maps.
179#[derive(Clone, Debug, PartialEq, Eq)]
180struct DefMapCrateData {
181    /// Side table for resolving derive helpers.
182    exported_derives: FxHashMap<MacroDefId, Box<[Name]>>,
183    fn_proc_macro_mapping: FxHashMap<FunctionId, ProcMacroId>,
184
185    /// Custom attributes registered with `#![register_attr]`.
186    registered_attrs: Vec<Symbol>,
187    /// Custom tool modules registered with `#![register_tool]`.
188    registered_tools: Vec<Symbol>,
189    /// Unstable features of Rust enabled with `#![feature(A, B)]`.
190    unstable_features: FxHashSet<Symbol>,
191    /// #[rustc_coherence_is_core]
192    rustc_coherence_is_core: bool,
193    no_core: bool,
194    no_std: bool,
195
196    edition: Edition,
197    recursion_limit: Option<u32>,
198}

200impl DefMapCrateData {

203            exported_derives: FxHashMap::default(),
204            fn_proc_macro_mapping: FxHashMap::default(),
205            registered_attrs: Vec::new(),
206            registered_tools: PREDEFINED_TOOLS.iter().map(|it| Symbol::intern(it)).collect(),
207            unstable_features: FxHashSet::default(),
208            rustc_coherence_is_core: false,
209            no_core: false,
210            no_std: false,
211            edition,
212            recursion_limit: None,
213        }
214    }
215
216    fn shrink_to_fit(&mut self) {
217        let Self {
218            exported_derives,
219            fn_proc_macro_mapping,
220            registered_attrs,
221            registered_tools,
222            unstable_features,
223            rustc_coherence_is_core: _,
224            no_core: _,
225            no_std: _,
226            edition: _,
227            recursion_limit: _,
228        } = self;
229        exported_derives.shrink_to_fit();
230        fn_proc_macro_mapping.shrink_to_fit();
231        registered_attrs.shrink_to_fit();
232        registered_tools.shrink_to_fit();
233        unstable_features.shrink_to_fit();
234    }

273#[derive(Debug, PartialEq, Eq, Clone, Copy, Hash)]
274pub enum ModuleOrigin {
275    CrateRoot {
276        definition: EditionedFileId,
277    },
278    /// Note that non-inline modules, by definition, live inside non-macro file.
279    File {
280        is_mod_rs: bool,
281        declaration: FileAstId<ast::Module>,
282        declaration_tree_id: ItemTreeId<Mod>,
283        definition: EditionedFileId,
284    },
285    Inline {
286        definition_tree_id: ItemTreeId<Mod>,
287        definition: FileAstId<ast::Module>,
288    },
289    /// Pseudo-module introduced by a block scope (contains only inner items).
290    BlockExpr {
291        id: BlockId,
292        block: AstId<ast::BlockExpr>,
293    },
294}
295
296impl ModuleOrigin {
297    pub fn declaration(&self) -> Option<AstId<ast::Module>> {
298        match self {
299            &ModuleOrigin::File { declaration, declaration_tree_id, .. } => {
300                Some(AstId::new(declaration_tree_id.file_id(), declaration))
301            }
302            &ModuleOrigin::Inline { definition, definition_tree_id } => {
303                Some(AstId::new(definition_tree_id.file_id(), definition))
304            }
305            ModuleOrigin::CrateRoot { .. } | ModuleOrigin::BlockExpr { .. } => None,
306        }
307    }
308
309    pub fn file_id(&self) -> Option<EditionedFileId> {
310        match self {
311            ModuleOrigin::File { definition, .. } | ModuleOrigin::CrateRoot { definition } => {
312                Some(*definition)
313            }
314            _ => None,
315        }
316    }
317
318    pub fn is_inline(&self) -> bool {
319        match self {
320            ModuleOrigin::Inline { .. } | ModuleOrigin::BlockExpr { .. } => true,
321            ModuleOrigin::CrateRoot { .. } | ModuleOrigin::File { .. } => false,
322        }
323    }
324
325    /// Returns a node which defines this module.
326    /// That is, a file or a `mod foo {}` with items.
327    pub fn definition_source(&self, db: &dyn DefDatabase) -> InFile<ModuleSource> {
328        match self {
329            &ModuleOrigin::File { definition: editioned_file_id, .. }
330            | &ModuleOrigin::CrateRoot { definition: editioned_file_id } => {
331                let definition = base_db::EditionedFileId::new(db, editioned_file_id);
332
333                let sf = db.parse(definition).tree();
334                InFile::new(editioned_file_id.into(), ModuleSource::SourceFile(sf))
335            }
336            &ModuleOrigin::Inline { definition, definition_tree_id } => InFile::new(
337                definition_tree_id.file_id(),
338                ModuleSource::Module(
339                    AstId::new(definition_tree_id.file_id(), definition).to_node(db),
340                ),
341            ),
342            ModuleOrigin::BlockExpr { block, .. } => {
343                InFile::new(block.file_id, ModuleSource::BlockExpr(block.to_node(db)))
344            }
345        }
346    }
347}

363impl DefMap {
364    /// The module id of a crate or block root.

371    pub(crate) fn crate_def_map_query(db: &dyn DefDatabase, crate_id: Crate) -> Arc<DefMap> {
372        db.crate_local_def_map(crate_id).0

375    pub(crate) fn crate_local_def_map_query(
376        db: &dyn DefDatabase,
377        crate_id: Crate,
378    ) -> (Arc<DefMap>, Arc<LocalDefMap>) {
379        let krate = crate_id.data(db);
380        let _p = tracing::info_span!(
381            "crate_def_map_query",
382            name=?crate_id
383                .extra_data(db)
384                .display_name
385                .as_ref()
386                .map(|it| it.crate_name().to_smolstr())
387                .unwrap_or_default()
388        )
389        .entered();
390
391        let module_data = ModuleData::new(
392            ModuleOrigin::CrateRoot { definition: krate.root_file_id() },
393            Visibility::Public,
394        );
395
396        let def_map = DefMap::empty(
397            crate_id,
398            Arc::new(DefMapCrateData::new(krate.edition)),
399            module_data,
400            None,
401        );
402        let (def_map, local_def_map) = collector::collect_defs(
403            db,
404            def_map,
405            TreeId::new(krate.root_file_id().into(), None),
406            None,
407        );
408
409        (Arc::new(def_map), Arc::new(local_def_map))
410    }
411
412    pub(crate) fn block_def_map_query(db: &dyn DefDatabase, block_id: BlockId) -> Arc<DefMap> {
413        let BlockLoc { ast_id, module } = block_id.lookup(db);
414
415        let visibility = Visibility::Module(
416            ModuleId { krate: module.krate, local_id: Self::ROOT, block: module.block },
417            VisibilityExplicitness::Implicit,
418        );
419        let module_data =
420            ModuleData::new(ModuleOrigin::BlockExpr { block: ast_id, id: block_id }, visibility);
421
422        let (crate_map, crate_local_map) = db.crate_local_def_map(module.krate);
423        let def_map = DefMap::empty(
424            module.krate,
425            crate_map.data.clone(),
426            module_data,
427            Some(BlockInfo {
428                block: block_id,
429                parent: BlockRelativeModuleId { block: module.block, local_id: module.local_id },
430            }),
431        );
432
433        let (def_map, _) = collector::collect_defs(
434            db,
435            def_map,
436            TreeId::new(ast_id.file_id, Some(block_id)),
437            Some(crate_local_map),
438        );
439        Arc::new(def_map)
440    }

442    fn empty(
443        krate: Crate,
444        crate_data: Arc<DefMapCrateData>,
445        module_data: ModuleData,
446        block: Option<BlockInfo>,
447    ) -> DefMap {
448        let mut modules: Arena<ModuleData> = Arena::default();
449        let root = modules.alloc(module_data);
450        assert_eq!(root, Self::ROOT);
451
452        DefMap {
453            block,
454            modules,
455            krate,
456            prelude: None,
457            macro_use_prelude: FxHashMap::default(),
458            derive_helpers_in_scope: FxHashMap::default(),
459            diagnostics: Vec::new(),
460            data: crate_data,
461            macro_def_to_macro_id: FxHashMap::default(),
462        }
463    }
464    fn shrink_to_fit(&mut self) {
465        // Exhaustive match to require handling new fields.
466        let Self {
467            macro_use_prelude,
468            diagnostics,
469            modules,
470            derive_helpers_in_scope,
471            block: _,
472            krate: _,
473            prelude: _,
474            data: _,
475            macro_def_to_macro_id,
476        } = self;
477
478        macro_def_to_macro_id.shrink_to_fit();
479        macro_use_prelude.shrink_to_fit();
480        diagnostics.shrink_to_fit();
481        modules.shrink_to_fit();
482        derive_helpers_in_scope.shrink_to_fit();
483        for (_, module) in modules.iter_mut() {
484            module.children.shrink_to_fit();
485            module.scope.shrink_to_fit();
486        }
487    }
488}

586    // FIXME: this can use some more human-readable format (ideally, an IR
587    // even), as this should be a great debugging aid.
588    pub fn dump(&self, db: &dyn DefDatabase) -> String {
589        let mut buf = String::new();
590        let mut arc;
591        let mut current_map = self;
592        while let Some(block) = current_map.block {
593            go(&mut buf, db, current_map, "block scope", Self::ROOT);
594            buf.push('\n');
595            arc = block.parent.def_map(db, self.krate);
596            current_map = &arc;
597        }
598        go(&mut buf, db, current_map, "crate", Self::ROOT);
599        return buf;
600
601        fn go(
602            buf: &mut String,
603            db: &dyn DefDatabase,
604            map: &DefMap,
605            path: &str,
606            module: LocalModuleId,
607        ) {
608            format_to!(buf, "{}\n", path);
609
610            map.modules[module].scope.dump(db, buf);
611
612            for (name, child) in
613                map.modules[module].children.iter().sorted_by(|a, b| Ord::cmp(&a.0, &b.0))
614            {
615                let path = format!("{path}::{}", name.display(db, Edition::LATEST));
616                buf.push('\n');
617                go(buf, db, map, &path, *child);
618            }
619        }
620    }
621
622    pub fn dump_block_scopes(&self, db: &dyn DefDatabase) -> String {
623        let mut buf = String::new();
624        let mut arc;
625        let mut current_map = self;
626        while let Some(block) = current_map.block {
627            format_to!(buf, "{:?} in {:?}\n", block.block, block.parent);
628            arc = block.parent.def_map(db, self.krate);
629            current_map = &arc;
630        }
631
632        format_to!(buf, "crate scope\n");
633        buf
634    }

693    /// Ascends the `DefMap` hierarchy and calls `f` with every `DefMap` and containing module.
694    ///
695    /// If `f` returns `Some(val)`, iteration is stopped and `Some(val)` is returned. If `f` returns
696    /// `None`, iteration continues.
697    pub(crate) fn with_ancestor_maps<T>(
698        &self,
699        db: &dyn DefDatabase,
700        local_mod: LocalModuleId,
701        f: &mut dyn FnMut(&DefMap, LocalModuleId) -> Option<T>,
702    ) -> Option<T> {
703        if let Some(it) = f(self, local_mod) {
704            return Some(it);
705        }
706        let mut block = self.block;
707        while let Some(block_info) = block {
708            let parent = block_info.parent.def_map(db, self.krate);
709            if let Some(it) = f(&parent, block_info.parent.local_id) {
710                return Some(it);
711            }
712            block = parent.block;
713        }
714
715        None
716    }
717}

795/// See `sub_namespace_match()`.
796#[derive(Clone, Copy, PartialEq, Eq)]
797pub enum MacroSubNs {
798    /// Function-like macros, suffixed with `!`.
799    Bang,
800    /// Macros inside attributes, i.e. attribute macros and derive macros.
801    Attr,
802}
803
804impl MacroSubNs {
805    fn from_id(db: &dyn DefDatabase, macro_id: MacroId) -> Self {
806        let expander = match macro_id {
807            MacroId::Macro2Id(it) => it.lookup(db).expander,
808            MacroId::MacroRulesId(it) => it.lookup(db).expander,
809            MacroId::ProcMacroId(it) => {
810                return match it.lookup(db).kind {
811                    ProcMacroKind::CustomDerive | ProcMacroKind::Attr => Self::Attr,
812                    ProcMacroKind::Bang => Self::Bang,
813                };
814            }
815        };
816
817        // Eager macros aren't *guaranteed* to be bang macros, but they *are* all bang macros currently.
818        match expander {
819            MacroExpander::Declarative
820            | MacroExpander::BuiltIn(_)
821            | MacroExpander::BuiltInEager(_) => Self::Bang,
822            MacroExpander::BuiltInAttr(_) | MacroExpander::BuiltInDerive(_) => Self::Attr,
823        }
824    }
825}

827/// Quoted from [rustc]:
828/// Macro namespace is separated into two sub-namespaces, one for bang macros and
829/// one for attribute-like macros (attributes, derives).
830/// We ignore resolutions from one sub-namespace when searching names in scope for another.
831///
832/// [rustc]: https://github.com/rust-lang/rust/blob/1.69.0/compiler/rustc_resolve/src/macros.rs#L75
833fn sub_namespace_match(candidate: Option<MacroSubNs>, expected: Option<MacroSubNs>) -> bool {
834    match (candidate, expected) {
835        (Some(candidate), Some(expected)) => candidate == expected,
836        _ => true,
837    }
