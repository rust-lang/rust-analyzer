1//! Defines attribute helpers for name resolution.
2//!
3//! Notice we don't preserve all attributes for name resolution, to save space:
4//! for example, we skip doc comments (desugared to `#[doc = "..."]` attributes)
5//! and `#[inline]`. The filtered attributes are listed in [`hir_expand::attrs`].

27#[derive(Debug, PartialEq, Eq)]
28pub(crate) enum AttrsOrCfg {
29    Enabled {
30        attrs: AttrsOwned,
31    },
32    /// This only collects the attributes up to the disabled `cfg` (this is what needed for crate-level attributes.)
33    CfgDisabled(Box<(CfgExpr, AttrsOwned)>),
34}
35
36impl Default for AttrsOrCfg {
37    #[inline]
38    fn default() -> Self {
39        AttrsOrCfg::Enabled { attrs: AttrsOwned(Box::new([])) }
40    }
41}

43impl AttrsOrCfg {
44    pub(crate) fn lower<'a, S>(
45        db: &dyn DefDatabase,
46        owner: &dyn ast::HasAttrs,
47        cfg_options: &dyn Fn() -> &'a CfgOptions,
48        span_map: S,
49    ) -> AttrsOrCfg
50    where
51        S: syntax_bridge::SpanMapper + Copy,
52    {
53        let mut attrs = Vec::new();
54        let result = collect_item_tree_attrs::<Infallible>(owner, cfg_options, |meta, _| {
55            // NOTE: We cannot early return from this function, *every* attribute must be pushed, otherwise we'll mess the `AttrId`
56            // tracking.
57            let path = meta.path();
58            let path_range = path
59                .as_ref()
60                .map(|path| path.syntax().text_range())
61                .unwrap_or_else(|| meta.syntax().text_range());
62            let (span, input) = match &meta {
63                ast::Meta::KeyValueMeta(meta) => {
64                    let span = span_map.span_for(path_range);
65                    let input = meta.expr().and_then(|value| {
66                        if let ast::Expr::Literal(value) = value {
67                            Some(Box::new(AttrInput::Literal(token_to_literal(
68                                value.token().text(),
69                                span_map.span_for(value.syntax().text_range()),
70                            ))))
71                        } else {
72                            None
73                        }
74                    });
75                    (span, input)
76                }
77                ast::Meta::TokenTreeMeta(meta) => {
78                    let span = span_map.span_for(path_range);
79                    let tt = syntax_bridge::syntax_node_to_token_tree(
80                        &meta
81                            .token_tree()
82                            .map(|it| it.syntax().clone())
83                            .unwrap_or_else(|| meta.syntax().clone()),
84                        span_map,
85                        span,
86                        DocCommentDesugarMode::ProcMacro,
87                    );
88                    let input = Some(Box::new(AttrInput::TokenTree(tt)));
89                    (span, input)
90                }
91                ast::Meta::PathMeta(_) => {
92                    let span = span_map.span_for(path_range);
93                    (span, None)
94                }
95                ast::Meta::CfgMeta(_) | ast::Meta::CfgAttrMeta(_) | ast::Meta::UnsafeMeta(_) => {
96                    unreachable!(
97                        "`cfg`, `cfg_attr` and `unsafe(...)` are handled in `collect_item_tree_attrs()`"
98                    )
99                }
100            };
101
102            let path = path.and_then(|path| {
103                ModPath::from_src(db, path, &mut |range| span_map.span_for(range).ctx)
104            });
105            let path = path.unwrap_or_else(|| Name::missing().into());
106
107            attrs.push(Attr { path: Interned::new(path), input, ctxt: span.ctx });
108            ControlFlow::Continue(())
109        });
110        let attrs = AttrsOwned(attrs.into_boxed_slice());
111        match result {
112            Some(Either::Right(cfg)) => AttrsOrCfg::CfgDisabled(Box::new((cfg, attrs))),
113            None => AttrsOrCfg::Enabled { attrs },
114        }
115    }
116
117    // Merges two `AttrsOrCfg`s, assuming `self` is placed before `other` in the source code.
118    // The operation follows these rules:
119    //
120    //   - If `self` and `other` are both `AttrsOrCfg::Enabled`, the result is a new
121    //     `AttrsOrCfg::Enabled`. It contains the concatenation of `self`'s attributes followed by
122    //     `other`'s.
123    //   - If `self` is `AttrsOrCfg::Enabled` but `other` is `AttrsOrCfg::CfgDisabled`, the result
124    //     is a new `AttrsOrCfg::CfgDisabled`. It contains the concatenation of `self`'s attributes
125    //     followed by `other`'s.
126    //   - If `self` is `AttrsOrCfg::CfgDisabled`, return `self` as-is.
127    //
128    // The rationale is that attribute collection is sequential and order-sensitive. This operation
129    // preserves those semantics when combining attributes from two different sources.
130    // `AttrsOrCfg::CfgDisabled` marks a point where collection stops due to a false `#![cfg(...)]`
131    // condition. It acts as a "breakpoint": attributes beyond it are not collected. Therefore,
132    // when merging, an `AttrsOrCfg::CfgDisabled` on the left-hand side short-circuits the
133    // operation, while an `AttrsOrCfg::CfgDisabled` on the right-hand side preserves all
134    // attributes collected up to that point.
135    //
136    // Note that this operation is neither commutative nor associative.
137    pub(crate) fn merge(self, other: AttrsOrCfg) -> AttrsOrCfg {
138        match (self, other) {
139            (AttrsOrCfg::Enabled { attrs }, AttrsOrCfg::Enabled { attrs: other_attrs }) => {
140                let mut v = attrs.0.into_vec();
141                v.extend(other_attrs.0);
142                AttrsOrCfg::Enabled { attrs: AttrsOwned(v.into_boxed_slice()) }
143            }
144            (AttrsOrCfg::Enabled { attrs }, AttrsOrCfg::CfgDisabled(mut other)) => {
145                let other_attrs = &mut other.1;
146                let mut v = attrs.0.into_vec();
147                v.extend(std::mem::take(&mut other_attrs.0));
148                other_attrs.0 = v.into_boxed_slice();
149                AttrsOrCfg::CfgDisabled(other)
150            }
151            (this @ AttrsOrCfg::CfgDisabled(_), _) => this,
152        }
153    }
154}

159#[derive(Debug, Clone, Copy)]
160pub(crate) struct Attrs<'a>(&'a [Attr]);
161
162impl ops::Deref for Attrs<'_> {
163    type Target = [Attr];
164
165    #[inline]
166    fn deref(&self) -> &Self::Target {
167        self.0
168    }
169}

171impl Ctx<'_> {
172    #[inline]
173    pub(super) fn lower_attrs(&self, owner: &dyn ast::HasAttrs) -> AttrsOrCfg {
174        AttrsOrCfg::lower(self.db, owner, &|| self.cfg_options(), self.span_map())
175    }

185impl<'a> Attrs<'a> {
186    pub(crate) const EMPTY: Self = Attrs(&[]);
187
188    #[inline]
189    pub(crate) fn by_key(self, key: Symbol) -> AttrQuery<'a> {
190        AttrQuery { attrs: self, key }
191    }
192
193    #[inline]
194    pub(crate) fn iter(self) -> impl Iterator<Item = (AttrId, &'a Attr)> {
195        self.0.iter().enumerate().map(|(id, attr)| (AttrId::from_item_tree_index(id as u32), attr))
196    }
197
198    #[inline]
199    pub(crate) fn iter_after(
200        self,
201        after: Option<AttrId>,
202    ) -> impl Iterator<Item = (AttrId, &'a Attr)> {
203        let skip = after.map_or(0, |after| after.item_tree_index() + 1);
204        self.0[skip as usize..]
205            .iter()
206            .enumerate()
207            .map(move |(id, attr)| (AttrId::from_item_tree_index(id as u32 + skip), attr))
208    }
209
210    #[inline]
211    pub(crate) fn is_proc_macro(&self) -> bool {
212        self.by_key(sym::proc_macro).exists()
213    }
214
215    #[inline]
216    pub(crate) fn is_proc_macro_attribute(&self) -> bool {
217        self.by_key(sym::proc_macro_attribute).exists()
218    }
219}
220#[derive(Debug, Clone)]
221pub(crate) struct AttrQuery<'attr> {
222    attrs: Attrs<'attr>,
223    key: Symbol,
224}
225
226impl<'attr> AttrQuery<'attr> {
227    #[inline]
228    pub(crate) fn tt_values(self) -> impl Iterator<Item = &'attr crate::tt::TopSubtree> {
229        self.attrs().filter_map(|attr| attr.token_tree_value())
230    }
231
232    #[inline]
233    pub(crate) fn string_value_with_span(self) -> Option<(&'attr str, span::Span)> {
234        self.attrs().find_map(|attr| attr.string_value_with_span())
235    }
236
237    #[inline]
238    pub(crate) fn string_value_unescape(self) -> Option<Cow<'attr, str>> {
239        self.attrs().find_map(|attr| attr.string_value_unescape())
240    }
241
242    #[inline]
243    pub(crate) fn exists(self) -> bool {
244        self.attrs().next().is_some()
245    }
246
247    #[inline]
248    pub(crate) fn attrs(self) -> impl Iterator<Item = &'attr Attr> + Clone {
249        let key = self.key;
250        self.attrs.0.iter().filter(move |attr| attr.path.as_ident().is_some_and(|s| *s == key))
251    }
252}
253
254impl AttrsOrCfg {
255    #[inline]
256    pub(super) fn empty() -> Self {
257        AttrsOrCfg::Enabled { attrs: AttrsOwned(Box::new([])) }
258    }
259
260    #[inline]
261    pub(super) fn is_empty(&self) -> bool {
262        matches!(self, AttrsOrCfg::Enabled { attrs } if attrs.as_ref().is_empty())
263    }
