1//! Defines attribute helpers for name resolution.
2//!
3//! Notice we don't preserve all attributes for name resolution, to save space:
4//! for example, we skip doc comments (desugared to `#[doc = "..."]` attributes)
5//! and `#[inline]`. The filtered attributes are listed in [`hir_expand::attrs`].
6

27#[derive(Debug, PartialEq, Eq)]
28pub(crate) enum AttrsOrCfg {
29    Enabled {
30        attrs: AttrsOwned,
31    },
32    /// This only collects the attributes up to the disabled `cfg` (this is what needed for crate-level attributes.)
33    CfgDisabled(Box<(CfgExpr, AttrsOwned)>),
34}
35
36impl Default for AttrsOrCfg {
37    #[inline]
38    fn default() -> Self {
39        AttrsOrCfg::Enabled { attrs: AttrsOwned(Box::new([])) }
40    }
41}

43impl AttrsOrCfg {
44    pub(crate) fn lower<'a, S>(
45        db: &dyn DefDatabase,
46        owner: &dyn ast::HasAttrs,
47        cfg_options: &dyn Fn() -> &'a CfgOptions,
48        span_map: S,
49    ) -> AttrsOrCfg
50    where
51        S: syntax_bridge::SpanMapper + Copy,
52    {
53        let mut attrs = Vec::new();
54        let result =
55            collect_item_tree_attrs::<Infallible>(owner, cfg_options, |meta, container, _, _| {
56                // NOTE: We cannot early return from this function, *every* attribute must be pushed, otherwise we'll mess the `AttrId`
57                // tracking.
58                let (span, path_range, input) = match meta {
59                    Meta::NamedKeyValue { path_range, name: _, value } => {
60                        let span = span_map.span_for(path_range);
61                        let input = value.map(|value| {
62                            Box::new(AttrInput::Literal(token_to_literal(
63                                value.text(),
64                                span_map.span_for(value.text_range()),
65                            )))
66                        });
67                        (span, path_range, input)
68                    }
69                    Meta::TokenTree { path, tt } => {
70                        let span = span_map.span_for(path.range);
71                        let tt = syntax_bridge::syntax_node_to_token_tree(
72                            tt.syntax(),
73                            span_map,
74                            span,
75                            DocCommentDesugarMode::ProcMacro,
76                        );
77                        let input = Some(Box::new(AttrInput::TokenTree(tt)));
78                        (span, path.range, input)
79                    }
80                    Meta::Path { path } => {
81                        let span = span_map.span_for(path.range);
82                        (span, path.range, None)
83                    }
84                };
85
86                let path = container.token_at_offset(path_range.start()).right_biased().and_then(
87                    |first_path_token| {
88                        let is_abs = matches!(first_path_token.kind(), T![:] | T![::]);
89                        let segments =
90                            std::iter::successors(Some(first_path_token), |it| it.next_token())
91                                .take_while(|it| it.text_range().end() <= path_range.end())
92                                .filter(|it| it.kind().is_any_identifier());
93                        ModPath::from_tokens(
94                            db,
95                            &mut |range| span_map.span_for(range).ctx,
96                            is_abs,
97                            segments,
98                        )
99                    },
100                );
101                let path = path.unwrap_or_else(|| Name::missing().into());
102
103                attrs.push(Attr { path: Interned::new(path), input, ctxt: span.ctx });
104                ControlFlow::Continue(())
105            });
106        let attrs = AttrsOwned(attrs.into_boxed_slice());
107        match result {
108            Some(Either::Right(cfg)) => AttrsOrCfg::CfgDisabled(Box::new((cfg, attrs))),
109            None => AttrsOrCfg::Enabled { attrs },
110        }
111    }
112
113    // Merges two `AttrsOrCfg`s, assuming `self` is placed before `other` in the source code.
114    // The operation follows these rules:
115    //
116    //   - If `self` and `other` are both `AttrsOrCfg::Enabled`, the result is a new
117    //     `AttrsOrCfg::Enabled`. It contains the concatenation of `self`'s attributes followed by
118    //     `other`'s.
119    //   - If `self` is `AttrsOrCfg::Enabled` but `other` is `AttrsOrCfg::CfgDisabled`, the result
120    //     is a new `AttrsOrCfg::CfgDisabled`. It contains the concatenation of `self`'s attributes
121    //     followed by `other`'s.
122    //   - If `self` is `AttrsOrCfg::CfgDisabled`, return `self` as-is.
123    //
124    // The rationale is that attribute collection is sequential and order-sensitive. This operation
125    // preserves those semantics when combining attributes from two different sources.
126    // `AttrsOrCfg::CfgDisabled` marks a point where collection stops due to a false `#![cfg(...)]`
127    // condition. It acts as a "breakpoint": attributes beyond it are not collected. Therefore,
128    // when merging, an `AttrsOrCfg::CfgDisabled` on the left-hand side short-circuits the
129    // operation, while an `AttrsOrCfg::CfgDisabled` on the right-hand side preserves all
130    // attributes collected up to that point.
131    //
132    // Note that this operation is neither commutative nor associative.
133    pub(crate) fn merge(self, other: AttrsOrCfg) -> AttrsOrCfg {
134        match (self, other) {
135            (AttrsOrCfg::Enabled { attrs }, AttrsOrCfg::Enabled { attrs: other_attrs }) => {
136                let mut v = attrs.0.into_vec();
137                v.extend(other_attrs.0);
138                AttrsOrCfg::Enabled { attrs: AttrsOwned(v.into_boxed_slice()) }
139            }
140            (AttrsOrCfg::Enabled { attrs }, AttrsOrCfg::CfgDisabled(mut other)) => {
141                let other_attrs = &mut other.1;
142                let mut v = attrs.0.into_vec();
143                v.extend(std::mem::take(&mut other_attrs.0));
144                other_attrs.0 = v.into_boxed_slice();
145                AttrsOrCfg::CfgDisabled(other)
146            }
147            (this @ AttrsOrCfg::CfgDisabled(_), _) => this,
148        }
149    }
150}

153pub(crate) struct AttrsOwned(Box<[Attr]>);

155#[derive(Debug, Clone, Copy)]
156pub(crate) struct Attrs<'a>(&'a [Attr]);
157
158impl ops::Deref for Attrs<'_> {
159    type Target = [Attr];
160
161    #[inline]
162    fn deref(&self) -> &Self::Target {
163        self.0
164    }
165}

167impl Ctx<'_> {
168    #[inline]
169    pub(super) fn lower_attrs(&self, owner: &dyn ast::HasAttrs) -> AttrsOrCfg {
170        AttrsOrCfg::lower(self.db, owner, &|| self.cfg_options(), self.span_map())
171    }

181impl<'a> Attrs<'a> {
182    pub(crate) const EMPTY: Self = Attrs(&[]);
183
184    #[inline]
185    pub(crate) fn by_key(self, key: Symbol) -> AttrQuery<'a> {
186        AttrQuery { attrs: self, key }
187    }
188
189    #[inline]
190    pub(crate) fn iter(self) -> impl Iterator<Item = (AttrId, &'a Attr)> {
191        self.0.iter().enumerate().map(|(id, attr)| (AttrId::from_item_tree_index(id as u32), attr))
192    }
193
194    #[inline]
195    pub(crate) fn iter_after(
196        self,
197        after: Option<AttrId>,
198    ) -> impl Iterator<Item = (AttrId, &'a Attr)> {
199        let skip = after.map_or(0, |after| after.item_tree_index() + 1);
200        self.0[skip as usize..]
201            .iter()
202            .enumerate()
203            .map(move |(id, attr)| (AttrId::from_item_tree_index(id as u32 + skip), attr))
204    }
205
206    #[inline]
207    pub(crate) fn is_proc_macro(&self) -> bool {
208        self.by_key(sym::proc_macro).exists()
209    }
210
211    #[inline]
212    pub(crate) fn is_proc_macro_attribute(&self) -> bool {
213        self.by_key(sym::proc_macro_attribute).exists()
214    }
215}
216#[derive(Debug, Clone)]
217pub(crate) struct AttrQuery<'attr> {
218    attrs: Attrs<'attr>,
219    key: Symbol,
220}
221
222impl<'attr> AttrQuery<'attr> {
223    #[inline]
224    pub(crate) fn tt_values(self) -> impl Iterator<Item = &'attr crate::tt::TopSubtree> {
225        self.attrs().filter_map(|attr| attr.token_tree_value())
226    }
227
228    #[inline]
229    pub(crate) fn string_value_with_span(self) -> Option<(&'attr str, span::Span)> {
230        self.attrs().find_map(|attr| attr.string_value_with_span())
231    }
232
233    #[inline]
234    pub(crate) fn string_value_unescape(self) -> Option<Cow<'attr, str>> {
235        self.attrs().find_map(|attr| attr.string_value_unescape())
236    }
237
238    #[inline]
239    pub(crate) fn exists(self) -> bool {
240        self.attrs().next().is_some()
241    }
242
243    #[inline]
244    pub(crate) fn attrs(self) -> impl Iterator<Item = &'attr Attr> + Clone {
245        let key = self.key;
246        self.attrs.0.iter().filter(move |attr| attr.path.as_ident().is_some_and(|s| *s == key))
247    }
248}
249
250impl AttrsOrCfg {
251    #[inline]
252    pub(super) fn empty() -> Self {
253        AttrsOrCfg::Enabled { attrs: AttrsOwned(Box::new([])) }
254    }
255
256    #[inline]
257    pub(super) fn is_empty(&self) -> bool {
258        matches!(self, AttrsOrCfg::Enabled { attrs } if attrs.as_ref().is_empty())
259    }
