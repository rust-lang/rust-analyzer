16pub(super) fn print_item_tree(db: &dyn DefDatabase, tree: &ItemTree, edition: Edition) -> String {
17    let mut p =
18        Printer { db, tree, buf: String::new(), indent_level: 0, needs_indent: true, edition };
19
20    p.print_attrs(&tree.top_attrs, true, "\n");
21    p.blank();
22
23    for item in tree.top_level_items() {
24        p.print_mod_item(*item);
25    }
26
27    let mut s = p.buf.trim_end_matches('\n').to_owned();
28    s.push('\n');
29    s
30}
31
32macro_rules! w {
33    ($dst:expr, $($arg:tt)*) => {
34        { let _ = write!($dst, $($arg)*); }
35    };
36}
37
38macro_rules! wln {
39    ($dst:expr) => {
40        { let _ = writeln!($dst); }
41    };
42    ($dst:expr, $($arg:tt)*) => {
43        { let _ = writeln!($dst, $($arg)*); }
44    };
45}
46
47struct Printer<'a> {
48    db: &'a dyn DefDatabase,
49    tree: &'a ItemTree,
50    buf: String,
51    indent_level: usize,
52    needs_indent: bool,
53    edition: Edition,
54}
55
56impl Printer<'_> {
57    fn indented(&mut self, f: impl FnOnce(&mut Self)) {
58        self.indent_level += 1;
59        wln!(self);
60        f(self);
61        self.indent_level -= 1;
62        self.buf = self.buf.trim_end_matches('\n').to_owned();
63    }
64
65    /// Ensures that a blank line is output before the next text.
66    fn blank(&mut self) {
67        let mut iter = self.buf.chars().rev().fuse();
68        match (iter.next(), iter.next()) {
69            (Some('\n'), Some('\n') | None) | (None, None) => {}
70            (Some('\n'), Some(_)) => {
71                self.buf.push('\n');
72            }
73            (Some(_), _) => {
74                self.buf.push('\n');
75                self.buf.push('\n');
76            }
77            (None, Some(_)) => unreachable!(),
78        }
79    }
80
81    fn whitespace(&mut self) {
82        match self.buf.chars().next_back() {
83            None | Some('\n' | ' ') => {}
84            _ => self.buf.push(' '),
85        }
86    }
87
88    fn print_attrs(&mut self, attrs: &RawAttrs, inner: bool, separated_by: &str) {
89        let inner = if inner { "!" } else { "" };
90        for attr in &**attrs {
91            w!(
92                self,
93                "#{}[{}{}]{}",
94                inner,
95                attr.path.display(self.db, self.edition),
96                attr.input.as_ref().map(|it| it.to_string()).unwrap_or_default(),
97                separated_by,
98            );
99        }
100    }
101
102    fn print_attrs_of(&mut self, of: ModItemId, separated_by: &str) {
103        if let Some(attrs) = self.tree.attrs.get(&of.ast_id()) {
104            self.print_attrs(attrs, false, separated_by);
105        }
106    }
107
108    fn print_visibility(&mut self, vis: RawVisibilityId) {
109        match &self.tree[vis] {
110            RawVisibility::Module(path, _expl) => {
111                w!(self, "pub(in {}) ", path.display(self.db, self.edition))
112            }
113            RawVisibility::Public => w!(self, "pub "),
114            RawVisibility::PubCrate => w!(self, "pub(crate) "),
115            RawVisibility::PubSelf(_) => w!(self, "pub(self) "),
116        };
117    }
118
119    fn print_fields(&mut self, kind: FieldsShape) {
120        match kind {
121            FieldsShape::Record => {
122                self.whitespace();
123                w!(self, "{{ ... }}");
124            }
125            FieldsShape::Tuple => {
126                w!(self, "(...)");
127            }
128            FieldsShape::Unit => {}
129        }
130    }
131
132    fn print_use_tree(&mut self, use_tree: &UseTree) {
133        match &use_tree.kind {
134            UseTreeKind::Single { path, alias } => {
135                w!(self, "{}", path.display(self.db, self.edition));
136                if let Some(alias) = alias {
137                    w!(self, " as {}", alias.display(self.edition));
138                }
139            }
140            UseTreeKind::Glob { path } => {
141                if let Some(path) = path {
142                    w!(self, "{}::", path.display(self.db, self.edition));
143                }
144                w!(self, "*");
145            }
146            UseTreeKind::Prefixed { prefix, list } => {
147                if let Some(prefix) = prefix {
148                    w!(self, "{}::", prefix.display(self.db, self.edition));
149                }
150                w!(self, "{{");
151                for (i, tree) in list.iter().enumerate() {
152                    if i != 0 {
153                        w!(self, ", ");
154                    }
155                    self.print_use_tree(tree);
156                }
157                w!(self, "}}");
158            }
159        }
160    }
161
162    fn print_mod_item(&mut self, item: ModItemId) {
163        self.print_attrs_of(item, "\n");
164
165        match item {
166            ModItemId::Use(ast_id) => {
167                let Use { visibility, use_tree } = &self.tree[ast_id];
168                self.print_ast_id(ast_id.erase());
169                self.print_visibility(*visibility);
170                w!(self, "use ");
171                self.print_use_tree(use_tree);
172                wln!(self, ";");
173            }
174            ModItemId::ExternCrate(ast_id) => {
175                let ExternCrate { name, alias, visibility } = &self.tree[ast_id];
176                self.print_ast_id(ast_id.erase());
177                self.print_visibility(*visibility);
178                w!(self, "extern crate {}", name.display(self.db, self.edition));
179                if let Some(alias) = alias {
180                    w!(self, " as {}", alias.display(self.edition));
181                }
182                wln!(self, ";");
183            }
184            ModItemId::ExternBlock(ast_id) => {
185                let ExternBlock { children } = &self.tree[ast_id];
186                self.print_ast_id(ast_id.erase());
187                w!(self, "extern {{");
188                self.indented(|this| {
189                    for child in &**children {
190                        this.print_mod_item(*child);
191                    }
192                });
193                wln!(self, "}}");
194            }
195            ModItemId::Function(ast_id) => {
196                let Function { name, visibility } = &self.tree[ast_id];
197                self.print_ast_id(ast_id.erase());
198                self.print_visibility(*visibility);
199                wln!(self, "fn {};", name.display(self.db, self.edition));
200            }
201            ModItemId::Struct(ast_id) => {
202                let Struct { visibility, name, shape: kind } = &self.tree[ast_id];
203                self.print_ast_id(ast_id.erase());
204                self.print_visibility(*visibility);
205                w!(self, "struct {}", name.display(self.db, self.edition));
206                self.print_fields(*kind);
207                if matches!(kind, FieldsShape::Record) {
208                    wln!(self);
209                } else {
210                    wln!(self, ";");
211                }
212            }
213            ModItemId::Union(ast_id) => {
214                let Union { name, visibility } = &self.tree[ast_id];
215                self.print_ast_id(ast_id.erase());
216                self.print_visibility(*visibility);
217                w!(self, "union {}", name.display(self.db, self.edition));
218                self.print_fields(FieldsShape::Record);
219                wln!(self);
220            }
221            ModItemId::Enum(ast_id) => {
222                let Enum { name, visibility } = &self.tree[ast_id];
223                self.print_ast_id(ast_id.erase());
224                self.print_visibility(*visibility);
225                w!(self, "enum {} {{ ... }}", name.display(self.db, self.edition));
226            }
227            ModItemId::Const(ast_id) => {
228                let Const { name, visibility } = &self.tree[ast_id];
229                self.print_ast_id(ast_id.erase());
230                self.print_visibility(*visibility);
231                w!(self, "const ");
232                match name {
233                    Some(name) => w!(self, "{}", name.display(self.db, self.edition)),
234                    None => w!(self, "_"),
235                }
236                wln!(self, " = _;");
237            }
238            ModItemId::Static(ast_id) => {
239                let Static { name, visibility } = &self.tree[ast_id];
240                self.print_ast_id(ast_id.erase());
241                self.print_visibility(*visibility);
242                w!(self, "static ");
243                w!(self, "{}", name.display(self.db, self.edition));
244                w!(self, " = _;");
245                wln!(self);
246            }
247            ModItemId::Trait(ast_id) => {
248                let Trait { name, visibility } = &self.tree[ast_id];
249                self.print_ast_id(ast_id.erase());
250                self.print_visibility(*visibility);
251                w!(self, "trait {} {{ ... }}", name.display(self.db, self.edition));
252            }
253            ModItemId::Impl(ast_id) => {
254                let Impl {} = &self.tree[ast_id];
255                self.print_ast_id(ast_id.erase());
256                w!(self, "impl {{ ... }}");
257            }
258            ModItemId::TypeAlias(ast_id) => {
259                let TypeAlias { name, visibility } = &self.tree[ast_id];
260                self.print_ast_id(ast_id.erase());
261                self.print_visibility(*visibility);
262                w!(self, "type {}", name.display(self.db, self.edition));
263                w!(self, ";");
264                wln!(self);
265            }
266            ModItemId::Mod(ast_id) => {
267                let Mod { name, visibility, kind } = &self.tree[ast_id];
268                self.print_ast_id(ast_id.erase());
269                self.print_visibility(*visibility);
270                w!(self, "mod {}", name.display(self.db, self.edition));
271                match kind {
272                    ModKind::Inline { items } => {
273                        w!(self, " {{");
274                        self.indented(|this| {
275                            for item in &**items {
276                                this.print_mod_item(*item);
277                            }
278                        });
279                        wln!(self, "}}");
280                    }
281                    ModKind::Outline => {
282                        wln!(self, ";");
283                    }
284                }
285            }
286            ModItemId::MacroCall(ast_id) => {
287                let MacroCall { path, expand_to, ctxt } = &self.tree[ast_id];
288                let _ = writeln!(
289                    self,
290                    "// AstId: {:#?}, SyntaxContextId: {}, ExpandTo: {:?}",
291                    ast_id.erase(),
292                    ctxt,
293                    expand_to
294                );
295                wln!(self, "{}!(...);", path.display(self.db, self.edition));
296            }
297            ModItemId::MacroRules(ast_id) => {
298                let MacroRules { name } = &self.tree[ast_id];
299                self.print_ast_id(ast_id.erase());
300                wln!(self, "macro_rules! {} {{ ... }}", name.display(self.db, self.edition));
301            }
302            ModItemId::Macro2(ast_id) => {
303                let Macro2 { name, visibility } = &self.tree[ast_id];
304                self.print_ast_id(ast_id.erase());
305                self.print_visibility(*visibility);
306                wln!(self, "macro {} {{ ... }}", name.display(self.db, self.edition));
307            }
308        }
309
310        self.blank();
311    }
312
313    fn print_ast_id(&mut self, ast_id: ErasedFileAstId) {
314        wln!(self, "// AstId: {ast_id:#?}");
315    }
316}
317
318impl Write for Printer<'_> {
319    fn write_str(&mut self, s: &str) -> fmt::Result {
320        for line in s.split_inclusive('\n') {
321            if self.needs_indent {
322                match self.buf.chars().last() {
323                    Some('\n') | None => {}
324                    _ => self.buf.push('\n'),
325                }
326                self.buf.push_str(&"    ".repeat(self.indent_level));
327                self.needs_indent = false;
328            }
329
330            self.buf.push_str(line);
331            self.needs_indent = line.ends_with('\n');
332        }
333
334        Ok(())
335    }
336}
