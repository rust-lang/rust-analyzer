1//! Attributes for anything that is not name resolution.
2//!
3//! The fundamental idea of this module stems from the observation that most "interesting"
4//! attributes have a more memory-compact form than storing their full syntax, and
5//! that most of the attributes are flags, and those that are not are rare. Therefore,
6//! this module defines [`AttrFlags`], which is a bitflag enum that contains only a yes/no
7//! answer to whether an attribute is present on an item. For most attributes, that's all
8//! that is interesting us; for the rest of them, we define another query that extracts
9//! their data. A key part is that every one of those queries will have a wrapper method
10//! that queries (or is given) the `AttrFlags` and checks for the presence of the attribute;
11//! if it is not present, we do not call the query, to prevent Salsa from needing to record
12//! its value. This way, queries are only called on items that have the attribute, which is
13//! usually only a few.
14//!
15//! An exception to this model that is also defined in this module is documentation (doc
16//! comments and `#[doc = "..."]` attributes). But it also has a more compact form than
17//! the attribute: a concatenated string of the full docs as well as a source map
18//! to map it back to AST (which is needed for things like resolving links in doc comments
19//! and highlight injection). The lowering and upmapping of doc comments is a bit complicated,
20//! but it is encapsulated in the [`Docs`] struct.

68fn extract_doc_tt_attr(attr_flags: &mut AttrFlags, tt: ast::TokenTree) {
69    for atom in DocAtom::parse(tt) {
70        match atom {
71            DocAtom::Flag(flag) => match &*flag {
72                "notable_trait" => attr_flags.insert(AttrFlags::IS_DOC_NOTABLE_TRAIT),
73                "hidden" => attr_flags.insert(AttrFlags::IS_DOC_HIDDEN),
74                _ => {}
75            },
76            DocAtom::KeyValue { key, value: _ } => match &*key {
77                "alias" => attr_flags.insert(AttrFlags::HAS_DOC_ALIASES),
78                "keyword" => attr_flags.insert(AttrFlags::HAS_DOC_KEYWORD),
79                _ => {}
80            },
81            DocAtom::Alias(_) => attr_flags.insert(AttrFlags::HAS_DOC_ALIASES),
82        }
83    }
84}

86fn extract_ra_completions(attr_flags: &mut AttrFlags, tt: ast::TokenTree) {
87    let tt = TokenTreeChildren::new(&tt);
88    if let Ok(NodeOrToken::Token(option)) = Itertools::exactly_one(tt)
89        && option.kind().is_any_identifier()
90    {
91        match option.text() {
92            "ignore_flyimport" => attr_flags.insert(AttrFlags::COMPLETE_IGNORE_FLYIMPORT),
93            "ignore_methods" => attr_flags.insert(AttrFlags::COMPLETE_IGNORE_METHODS),
94            "ignore_flyimport_methods" => {
95                attr_flags.insert(AttrFlags::COMPLETE_IGNORE_FLYIMPORT_METHODS)
96            }
97            _ => {}
98        }
99    }

102fn extract_ra_macro_style(attr_flags: &mut AttrFlags, tt: ast::TokenTree) {
103    let tt = TokenTreeChildren::new(&tt);
104    if let Ok(NodeOrToken::Token(option)) = Itertools::exactly_one(tt)
105        && option.kind().is_any_identifier()
106    {
107        match option.text() {
108            "braces" => attr_flags.insert(AttrFlags::MACRO_STYLE_BRACES),
109            "brackets" => attr_flags.insert(AttrFlags::MACRO_STYLE_BRACKETS),
110            "parentheses" => attr_flags.insert(AttrFlags::MACRO_STYLE_PARENTHESES),
111            _ => {}
112        }
113    }

116fn extract_rustc_skip_during_method_dispatch(attr_flags: &mut AttrFlags, tt: ast::TokenTree) {
117    let iter = TokenTreeChildren::new(&tt);
118    for kind in iter {
119        if let NodeOrToken::Token(kind) = kind
120            && kind.kind().is_any_identifier()
121        {
122            match kind.text() {
123                "array" => attr_flags.insert(AttrFlags::RUSTC_SKIP_ARRAY_DURING_METHOD_DISPATCH),
124                "boxed_slice" => {
125                    attr_flags.insert(AttrFlags::RUSTC_SKIP_BOXED_SLICE_DURING_METHOD_DISPATCH)
126                }
127                _ => {}
128            }
129        }
130    }

133#[inline]
134fn match_attr_flags(attr_flags: &mut AttrFlags, attr: Meta) -> ControlFlow<Infallible> {
135    match attr {
136        Meta::NamedKeyValue { name: Some(name), value, .. } => match name.text() {
137            "deprecated" => attr_flags.insert(AttrFlags::IS_DEPRECATED),
138            "ignore" => attr_flags.insert(AttrFlags::IS_IGNORE),
139            "lang" => attr_flags.insert(AttrFlags::LANG_ITEM),
140            "path" => attr_flags.insert(AttrFlags::HAS_PATH),
141            "unstable" => attr_flags.insert(AttrFlags::IS_UNSTABLE),
142            "export_name" => {
143                if let Some(value) = value
144                    && let Some(value) = ast::String::cast(value)
145                    && let Ok(value) = value.value()
146                    && *value == *"main"
147                {
148                    attr_flags.insert(AttrFlags::IS_EXPORT_NAME_MAIN);
149                }
150            }
151            _ => {}
152        },
153        Meta::TokenTree { path, tt } => match path.segments.len() {
154            1 => match path.segments[0].text() {
155                "deprecated" => attr_flags.insert(AttrFlags::IS_DEPRECATED),
156                "cfg" => attr_flags.insert(AttrFlags::HAS_CFG),
157                "doc" => extract_doc_tt_attr(attr_flags, tt),
158                "repr" => attr_flags.insert(AttrFlags::HAS_REPR),
159                "target_feature" => attr_flags.insert(AttrFlags::HAS_TARGET_FEATURE),
160                "proc_macro_derive" | "rustc_builtin_macro" => {
161                    attr_flags.insert(AttrFlags::IS_DERIVE_OR_BUILTIN_MACRO)
162                }
163                "unstable" => attr_flags.insert(AttrFlags::IS_UNSTABLE),
164                "rustc_layout_scalar_valid_range_start" | "rustc_layout_scalar_valid_range_end" => {
165                    attr_flags.insert(AttrFlags::RUSTC_LAYOUT_SCALAR_VALID_RANGE)
166                }
167                "rustc_legacy_const_generics" => {
168                    attr_flags.insert(AttrFlags::HAS_LEGACY_CONST_GENERICS)
169                }
170                "rustc_skip_during_method_dispatch" => {
171                    extract_rustc_skip_during_method_dispatch(attr_flags, tt)
172                }
173                "rustc_deprecated_safe_2024" => {
174                    attr_flags.insert(AttrFlags::RUSTC_DEPRECATED_SAFE_2024)
175                }
176                _ => {}
177            },
178            2 => match path.segments[0].text() {
179                "rust_analyzer" => match path.segments[1].text() {
180                    "completions" => extract_ra_completions(attr_flags, tt),
181                    "macro_style" => extract_ra_macro_style(attr_flags, tt),
182                    _ => {}
183                },
184                _ => {}
185            },
186            _ => {}
187        },
188        Meta::Path { path } => {
189            match path.segments.len() {
190                1 => match path.segments[0].text() {
191                    "rustc_has_incoherent_inherent_impls" => {
192                        attr_flags.insert(AttrFlags::RUSTC_HAS_INCOHERENT_INHERENT_IMPLS)
193                    }
194                    "rustc_allow_incoherent_impl" => {
195                        attr_flags.insert(AttrFlags::RUSTC_ALLOW_INCOHERENT_IMPL)
196                    }
197                    "fundamental" => attr_flags.insert(AttrFlags::FUNDAMENTAL),
198                    "no_std" => attr_flags.insert(AttrFlags::IS_NO_STD),
199                    "may_dangle" => attr_flags.insert(AttrFlags::MAY_DANGLE),
200                    "rustc_paren_sugar" => attr_flags.insert(AttrFlags::RUSTC_PAREN_SUGAR),
201                    "rustc_coinductive" => attr_flags.insert(AttrFlags::RUSTC_COINDUCTIVE),
202                    "rustc_force_inline" => attr_flags.insert(AttrFlags::RUSTC_FORCE_INLINE),
203                    "unstable" => attr_flags.insert(AttrFlags::IS_UNSTABLE),
204                    "deprecated" => attr_flags.insert(AttrFlags::IS_DEPRECATED),
205                    "macro_export" => attr_flags.insert(AttrFlags::IS_MACRO_EXPORT),
206                    "no_mangle" => attr_flags.insert(AttrFlags::NO_MANGLE),
207                    "pointee" => attr_flags.insert(AttrFlags::IS_POINTEE),
208                    "non_exhaustive" => attr_flags.insert(AttrFlags::NON_EXHAUSTIVE),
209                    "ignore" => attr_flags.insert(AttrFlags::IS_IGNORE),
210                    "bench" => attr_flags.insert(AttrFlags::IS_BENCH),
211                    "rustc_const_panic_str" => attr_flags.insert(AttrFlags::RUSTC_CONST_PANIC_STR),
212                    "rustc_intrinsic" => attr_flags.insert(AttrFlags::RUSTC_INTRINSIC),
213                    "rustc_safe_intrinsic" => attr_flags.insert(AttrFlags::RUSTC_SAFE_INTRINSIC),
214                    "rustc_intrinsic_must_be_overridden" => {
215                        attr_flags.insert(AttrFlags::RUSTC_INTRINSIC_MUST_BE_OVERRIDDEN)
216                    }
217                    "rustc_allocator" => attr_flags.insert(AttrFlags::RUSTC_ALLOCATOR),
218                    "rustc_deallocator" => attr_flags.insert(AttrFlags::RUSTC_DEALLOCATOR),
219                    "rustc_reallocator" => attr_flags.insert(AttrFlags::RUSTC_REALLOCATOR),
220                    "rustc_allocator_zeroed" => {
221                        attr_flags.insert(AttrFlags::RUSTC_ALLOCATOR_ZEROED)
222                    }
223                    "rustc_reservation_impl" => {
224                        attr_flags.insert(AttrFlags::RUSTC_RESERVATION_IMPL)
225                    }
226                    "rustc_deprecated_safe_2024" => {
227                        attr_flags.insert(AttrFlags::RUSTC_DEPRECATED_SAFE_2024)
228                    }
229                    "rustc_skip_array_during_method_dispatch" => {
230                        attr_flags.insert(AttrFlags::RUSTC_SKIP_ARRAY_DURING_METHOD_DISPATCH)
231                    }
232                    _ => {}
233                },
234                2 => match path.segments[0].text() {
235                    "rust_analyzer" => match path.segments[1].text() {
236                        "skip" => attr_flags.insert(AttrFlags::RUST_ANALYZER_SKIP),
237                        _ => {}
238                    },
239                    _ => {}
240                },
241                _ => {}
242            }
243
244            if path.is_test {
245                attr_flags.insert(AttrFlags::IS_TEST);
246            }
247        }
248        _ => {}
249    };
250    ControlFlow::Continue(())
251}

317pub fn parse_extra_crate_attrs(db: &dyn DefDatabase, krate: Crate) -> Option<SourceFile> {
318    let crate_data = krate.data(db);
319    let crate_attrs = &crate_data.crate_attrs;
320    if crate_attrs.is_empty() {
321        return None;
322    }
323    // All attributes are already enclosed in `#![]`.
324    let combined = crate_attrs.concat();
325    let p = SourceFile::parse(&combined, crate_data.edition);
326
327    let errs = p.errors();
328    if !errs.is_empty() {
329        let base_msg = "Failed to parse extra crate-level attribute";
330        let crate_name =
331            krate.extra_data(db).display_name.as_ref().map_or("{unknown}", |name| name.as_str());
332        let mut errs = errs.iter().peekable();
333        let mut offset = TextSize::from(0);
334        for raw_attr in crate_attrs {
335            let attr_end = offset + TextSize::of(&**raw_attr);
336            if errs.peeking_take_while(|e| e.range().start() < attr_end).count() > 0 {
337                tracing::error!("{base_msg} {raw_attr} for crate {crate_name}");
338            }
339            offset = attr_end
340        }
341        return None;
342    }
343
344    Some(p.tree())
345}

347fn attrs_source(
348    db: &dyn DefDatabase,
349    owner: AttrDefId,
350) -> (InFile<ast::AnyHasAttrs>, Option<InFile<ast::Module>>, Option<SourceFile>, Crate) {
351    let (owner, krate) = match owner {
352        AttrDefId::ModuleId(id) => {
353            let def_map = id.def_map(db);
354            let krate = def_map.krate();
355            let (definition, declaration, extra_crate_attrs) = match def_map[id].origin {
356                ModuleOrigin::CrateRoot { definition } => {
357                    let definition_source = db.parse(definition).tree();
358                    let definition = InFile::new(definition.into(), definition_source.into());
359                    let extra_crate_attrs = parse_extra_crate_attrs(db, krate);
360                    (definition, None, extra_crate_attrs)
361                }
362                ModuleOrigin::File { declaration, declaration_tree_id, definition, .. } => {
363                    let definition_source = db.parse(definition).tree();
364                    let definition = InFile::new(definition.into(), definition_source.into());
365                    let declaration = InFile::new(declaration_tree_id.file_id(), declaration);
366                    let declaration = declaration.with_value(declaration.to_node(db));
367                    (definition, Some(declaration), None)
368                }
369                ModuleOrigin::Inline { definition_tree_id, definition } => {
370                    let definition = InFile::new(definition_tree_id.file_id(), definition);
371                    let definition = definition.with_value(definition.to_node(db).into());
372                    (definition, None, None)
373                }
374                ModuleOrigin::BlockExpr { block, .. } => {
375                    let definition = block.to_node(db);
376                    (block.with_value(definition.into()), None, None)
377                }
378            };
379            return (definition, declaration, extra_crate_attrs, krate);
380        }
381        AttrDefId::AdtId(AdtId::StructId(it)) => attrs_from_ast_id_loc(db, it),
382        AttrDefId::AdtId(AdtId::UnionId(it)) => attrs_from_ast_id_loc(db, it),
383        AttrDefId::AdtId(AdtId::EnumId(it)) => attrs_from_ast_id_loc(db, it),
384        AttrDefId::FunctionId(it) => attrs_from_ast_id_loc(db, it),
385        AttrDefId::EnumVariantId(it) => attrs_from_ast_id_loc(db, it),
386        AttrDefId::StaticId(it) => attrs_from_ast_id_loc(db, it),
387        AttrDefId::ConstId(it) => attrs_from_ast_id_loc(db, it),
388        AttrDefId::TraitId(it) => attrs_from_ast_id_loc(db, it),
389        AttrDefId::TypeAliasId(it) => attrs_from_ast_id_loc(db, it),
390        AttrDefId::MacroId(MacroId::MacroRulesId(it)) => attrs_from_ast_id_loc(db, it),
391        AttrDefId::MacroId(MacroId::Macro2Id(it)) => attrs_from_ast_id_loc(db, it),
392        AttrDefId::MacroId(MacroId::ProcMacroId(it)) => attrs_from_ast_id_loc(db, it),
393        AttrDefId::ImplId(it) => attrs_from_ast_id_loc(db, it),
394        AttrDefId::ExternBlockId(it) => attrs_from_ast_id_loc(db, it),
395        AttrDefId::ExternCrateId(it) => attrs_from_ast_id_loc(db, it),
396        AttrDefId::UseId(it) => attrs_from_ast_id_loc(db, it),
397    };
398    (owner, None, None, krate)
399}

401fn collect_attrs<BreakValue>(
402    db: &dyn DefDatabase,
403    owner: AttrDefId,
404    mut callback: impl FnMut(Meta) -> ControlFlow<BreakValue>,
405) -> Option<BreakValue> {
406    let (source, outer_mod_decl, extra_crate_attrs, krate) = attrs_source(db, owner);
407    let extra_attrs = extra_crate_attrs
408        .into_iter()
409        .flat_map(|src| src.attrs())
410        .chain(outer_mod_decl.into_iter().flat_map(|it| it.value.attrs()));
411
412    let mut cfg_options = None;
413    expand_cfg_attr(
414        extra_attrs.chain(ast::attrs_including_inner(&source.value)),
415        || cfg_options.get_or_insert_with(|| krate.cfg_options(db)),
416        move |meta, _, _, _| callback(meta),
417    )
418}

420fn collect_field_attrs<T>(
421    db: &dyn DefDatabase,
422    variant: VariantId,
423    mut field_attrs: impl FnMut(&CfgOptions, InFile<ast::AnyHasAttrs>) -> T,
424) -> ArenaMap<LocalFieldId, T> {
425    let (variant_syntax, krate) = match variant {
426        VariantId::EnumVariantId(it) => attrs_from_ast_id_loc(db, it),
427        VariantId::StructId(it) => attrs_from_ast_id_loc(db, it),
428        VariantId::UnionId(it) => attrs_from_ast_id_loc(db, it),
429    };
430    let cfg_options = krate.cfg_options(db);
431    let variant_syntax = variant_syntax
432        .with_value(ast::VariantDef::cast(variant_syntax.value.syntax().clone()).unwrap());
433    let fields = match &variant_syntax.value {
434        ast::VariantDef::Struct(it) => it.field_list(),
435        ast::VariantDef::Union(it) => it.record_field_list().map(ast::FieldList::RecordFieldList),
436        ast::VariantDef::Variant(it) => it.field_list(),
437    };
438    let Some(fields) = fields else {
439        return ArenaMap::new();
440    };
441
442    let mut result = ArenaMap::new();
443    let mut idx = 0;
444    match fields {
445        ast::FieldList::RecordFieldList(fields) => {
446            for field in fields.fields() {
447                if AttrFlags::is_cfg_enabled_for(&field, cfg_options).is_ok() {
448                    result.insert(
449                        la_arena::Idx::from_raw(la_arena::RawIdx::from_u32(idx)),
450                        field_attrs(cfg_options, variant_syntax.with_value(field.into())),
451                    );
452                    idx += 1;
453                }
454            }
455        }
456        ast::FieldList::TupleFieldList(fields) => {
457            for field in fields.fields() {
458                if AttrFlags::is_cfg_enabled_for(&field, cfg_options).is_ok() {
459                    result.insert(
460                        la_arena::Idx::from_raw(la_arena::RawIdx::from_u32(idx)),
461                        field_attrs(cfg_options, variant_syntax.with_value(field.into())),
462                    );
463                    idx += 1;
464                }
465            }
466        }
467    }
468    result.shrink_to_fit();
469    result
470}

531    pub fn find_ast_range(
532        &self,
533        mut string_range: TextRange,
534    ) -> Option<(InFile<TextRange>, IsInnerDoc)> {
535        if string_range.start() < self.prefix_len {
536            return None;
537        }
538        string_range -= self.prefix_len;
539
540        let mut file = self.inline_file;
541        let mut inner_docs_start = self.inline_inner_docs_start;
542        // Check whether the range is from the outline, the inline, or both.
543        let source_map = if let Some((outline_mod_file, outline_mod_end)) = self.outline_mod {
544            if let Some(first_inline) = self.docs_source_map.get(outline_mod_end) {
545                if string_range.end() <= first_inline.string_offset {
546                    // The range is completely in the outline.
547                    file = outline_mod_file;
548                    inner_docs_start = self.outline_inner_docs_start;
549                    &self.docs_source_map[..outline_mod_end]
550                } else if string_range.start() >= first_inline.string_offset {
551                    // The range is completely in the inline.
552                    &self.docs_source_map[outline_mod_end..]
553                } else {
554                    // The range is combined from the outline and the inline - cannot map it back.
555                    return None;
556                }
557            } else {
558                // There is no inline.
559                file = outline_mod_file;
560                inner_docs_start = self.outline_inner_docs_start;
561                &self.docs_source_map
562            }
563        } else {
564            // There is no outline.
565            &self.docs_source_map
566        };
567
568        let after_range =
569            source_map.partition_point(|line| line.string_offset <= string_range.start()) - 1;
570        let after_range = &source_map[after_range..];
571        let line = after_range.first()?;
572        if after_range.get(1).is_some_and(|next_line| next_line.string_offset < string_range.end())
573        {
574            // The range is combined from two lines - cannot map it back.
575            return None;
576        }
577        let ast_range = string_range - line.string_offset + line.ast_offset;
578        let is_inner = if inner_docs_start
579            .is_some_and(|inner_docs_start| string_range.start() >= inner_docs_start)
580        {
581            IsInnerDoc::Yes
582        } else {
583            IsInnerDoc::No
584        };
585        Some((InFile::new(file, ast_range), is_inner))
586    }

637    fn extend_with_doc_str(&mut self, doc: &str, mut offset_in_ast: TextSize, indent: &mut usize) {
638        for line in doc.split('\n') {
639            self.docs_source_map.push(DocsSourceMapLine {
640                string_offset: TextSize::of(&self.docs),
641                ast_offset: offset_in_ast,
642            });
643            offset_in_ast += TextSize::of(line) + TextSize::of("\n");
644
645            let line = line.trim_end();
646            if let Some(line_indent) = line.chars().position(|ch| !ch.is_whitespace()) {
647                // Empty lines are handled because `position()` returns `None` for them.
648                *indent = std::cmp::min(*indent, line_indent);
649            }
650            self.docs.push_str(line);
651            self.docs.push('\n');
652        }

655    fn remove_indent(&mut self, indent: usize, start_source_map_index: usize) {
656        /// In case of panics, we want to avoid corrupted UTF-8 in `self.docs`, so we clear it.
657        struct Guard<'a>(&'a mut Docs);
658        impl Drop for Guard<'_> {
659            fn drop(&mut self) {
660                let Docs {
661                    docs,
662                    docs_source_map,
663                    outline_mod,
664                    inline_file: _,
665                    prefix_len: _,
666                    inline_inner_docs_start: _,
667                    outline_inner_docs_start: _,
668                } = self.0;
669                // Don't use `String::clear()` here because it's not guaranteed to not do UTF-8-dependent things,
670                // and we may have temporarily broken the string's encoding.
671                unsafe { docs.as_mut_vec() }.clear();
672                // This is just to avoid panics down the road.
673                docs_source_map.clear();
674                *outline_mod = None;
675            }
676        }
677
678        if self.docs.is_empty() {
679            return;
680        }
681
682        let guard = Guard(self);
683        let source_map = &mut guard.0.docs_source_map[start_source_map_index..];
684        let Some(&DocsSourceMapLine { string_offset: mut copy_into, .. }) = source_map.first()
685        else {
686            return;
687        };
688        // We basically want to remove multiple ranges from a string. Doing this efficiently (without O(N^2)
689        // or allocations) requires unsafe. Basically, for each line, we copy the line minus the indent into
690        // consecutive to the previous line (which may have moved). Then at the end we truncate.
691        let mut accumulated_offset = TextSize::new(0);
692        for idx in 0..source_map.len() {
693            let string_end_offset = source_map
694                .get(idx + 1)
695                .map_or_else(|| TextSize::of(&guard.0.docs), |next_attr| next_attr.string_offset);
696            let line_source = &mut source_map[idx];
697            let line_docs =
698                &guard.0.docs[TextRange::new(line_source.string_offset, string_end_offset)];
699            let line_docs_len = TextSize::of(line_docs);
700            let indent_size = line_docs.char_indices().nth(indent).map_or_else(
701                || TextSize::of(line_docs) - TextSize::of("\n"),
702                |(offset, _)| TextSize::new(offset as u32),
703            );
704            unsafe { guard.0.docs.as_bytes_mut() }.copy_within(
705                Range::<usize>::from(TextRange::new(
706                    line_source.string_offset + indent_size,
707                    string_end_offset,
708                )),
709                copy_into.into(),
710            );
711            copy_into += line_docs_len - indent_size;
712
713            if let Some(inner_attrs_start) = &mut guard.0.inline_inner_docs_start
714                && *inner_attrs_start == line_source.string_offset
715            {
716                *inner_attrs_start -= accumulated_offset;
717            }
718            // The removals in the string accumulate, but in the AST not, because it already points
719            // to the beginning of each attribute.
720            // Also, we need to shift the AST offset of every line, but the string offset of the first
721            // line should not get shifted (in general, the shift for the string offset is by the
722            // number of lines until the current one, excluding the current one).
723            line_source.string_offset -= accumulated_offset;
724            line_source.ast_offset += indent_size;
725
726            accumulated_offset += indent_size;
727        }
728        // Don't use `String::truncate()` here because it's not guaranteed to not do UTF-8-dependent things,
729        // and we may have temporarily broken the string's encoding.
730        unsafe { guard.0.docs.as_mut_vec() }.truncate(copy_into.into());
731
732        std::mem::forget(guard);
733    }
734
735    fn remove_last_newline(&mut self) {
736        self.docs.truncate(self.docs.len().saturating_sub(1));
737    }

788fn extract_docs<'a>(
789    get_cfg_options: &dyn Fn() -> &'a CfgOptions,
790    source: InFile<ast::AnyHasAttrs>,
791    outer_mod_decl: Option<InFile<ast::Module>>,
792    inner_attrs_node: Option<SyntaxNode>,
793) -> Option<Box<Docs>> {
794    let mut result = Docs {
795        docs: String::new(),
796        docs_source_map: Vec::new(),
797        outline_mod: None,
798        inline_file: source.file_id,
799        prefix_len: TextSize::new(0),
800        inline_inner_docs_start: None,
801        outline_inner_docs_start: None,
802    };
803
804    let mut cfg_options = None;
805    let mut extend_with_attrs =
806        |result: &mut Docs, node: &SyntaxNode, expect_inner_attrs, indent: &mut usize| {
807            expand_cfg_attr_with_doc_comments::<_, Infallible>(
808                AttrDocCommentIter::from_syntax_node(node).filter(|attr| match attr {
809                    Either::Left(attr) => attr.kind().is_inner() == expect_inner_attrs,
810                    Either::Right(comment) => comment.kind().doc.is_some_and(|kind| {
811                        (kind == ast::CommentPlacement::Inner) == expect_inner_attrs
812                    }),
813                }),
814                || cfg_options.get_or_insert_with(get_cfg_options),
815                |attr| {
816                    match attr {
817                        Either::Right(doc_comment) => {
818                            result.extend_with_doc_comment(doc_comment, indent)
819                        }
820                        Either::Left((attr, _, _, _)) => match attr {
821                            // FIXME: Handle macros: `#[doc = concat!("foo", "bar")]`.
822                            Meta::NamedKeyValue {
823                                name: Some(name), value: Some(value), ..
824                            } if name.text() == "doc" => {
825                                result.extend_with_doc_attr(value, indent);
826                            }
827                            _ => {}
828                        },
829                    }
830                    ControlFlow::Continue(())
831                },
832            );
833        };
834
835    if let Some(outer_mod_decl) = outer_mod_decl {
836        let mut indent = usize::MAX;
837        extend_with_attrs(&mut result, outer_mod_decl.value.syntax(), false, &mut indent);
838        result.remove_indent(indent, 0);
839        result.outline_mod = Some((outer_mod_decl.file_id, result.docs_source_map.len()));
840    }
841
842    let inline_source_map_start = result.docs_source_map.len();
843    let mut indent = usize::MAX;
844    extend_with_attrs(&mut result, source.value.syntax(), false, &mut indent);
845    if let Some(inner_attrs_node) = &inner_attrs_node {
846        result.inline_inner_docs_start = Some(TextSize::of(&result.docs));
847        extend_with_attrs(&mut result, inner_attrs_node, true, &mut indent);
848    }
849    result.remove_indent(indent, inline_source_map_start);
850
851    result.remove_last_newline();
852
853    result.shrink_to_fit();
854
855    if result.docs.is_empty() { None } else { Some(Box::new(result)) }
856}

1366fn merge_repr(this: &mut ReprOptions, other: ReprOptions) {
1367    let ReprOptions { int, align, pack, flags, field_shuffle_seed: _ } = this;
1368    flags.insert(other.flags);
1369    *align = (*align).max(other.align);
1370    *pack = match (*pack, other.pack) {
1371        (Some(pack), None) | (None, Some(pack)) => Some(pack),
1372        _ => (*pack).min(other.pack),
1373    };
1374    if other.int.is_some() {
1375        *int = other.int;
1376    }
1377}

1379fn parse_repr_tt(tt: &ast::TokenTree) -> Option<ReprOptions> {
1380    use crate::builtin_type::{BuiltinInt, BuiltinUint};
1381    use rustc_abi::{Align, Integer, IntegerType, ReprFlags, ReprOptions};
1382
1383    let mut tts = TokenTreeChildren::new(tt).peekable();
1384
1385    let mut acc = ReprOptions::default();
1386    while let Some(tt) = tts.next() {
1387        let NodeOrToken::Token(ident) = tt else {
1388            continue;
1389        };
1390        if !ident.kind().is_any_identifier() {
1391            continue;
1392        }
1393        let repr = match ident.text() {
1394            "packed" => {
1395                let pack = if let Some(NodeOrToken::Node(tt)) = tts.peek() {
1396                    let tt = tt.clone();
1397                    tts.next();
1398                    let mut tt_iter = TokenTreeChildren::new(&tt);
1399                    if let Some(NodeOrToken::Token(lit)) = tt_iter.next()
1400                        && let Some(lit) = ast::IntNumber::cast(lit)
1401                        && let Ok(lit) = lit.value()
1402                        && let Ok(lit) = lit.try_into()
1403                    {
1404                        lit
1405                    } else {
1406                        0
1407                    }
1408                } else {
1409                    0
1410                };
1411                let pack = Some(Align::from_bytes(pack).unwrap_or(Align::ONE));
1412                ReprOptions { pack, ..Default::default() }
1413            }
1414            "align" => {
1415                let mut align = None;
1416                if let Some(NodeOrToken::Node(tt)) = tts.peek() {
1417                    let tt = tt.clone();
1418                    tts.next();
1419                    let mut tt_iter = TokenTreeChildren::new(&tt);
1420                    if let Some(NodeOrToken::Token(lit)) = tt_iter.next()
1421                        && let Some(lit) = ast::IntNumber::cast(lit)
1422                        && let Ok(lit) = lit.value()
1423                        && let Ok(lit) = lit.try_into()
1424                    {
1425                        align = Align::from_bytes(lit).ok();
1426                    }
1427                }
1428                ReprOptions { align, ..Default::default() }
1429            }
1430            "C" => ReprOptions { flags: ReprFlags::IS_C, ..Default::default() },
1431            "transparent" => ReprOptions { flags: ReprFlags::IS_TRANSPARENT, ..Default::default() },
1432            "simd" => ReprOptions { flags: ReprFlags::IS_SIMD, ..Default::default() },
1433            repr => {
1434                let mut int = None;
1435                if let Some(builtin) = BuiltinInt::from_suffix(repr)
1436                    .map(Either::Left)
1437                    .or_else(|| BuiltinUint::from_suffix(repr).map(Either::Right))
1438                {
1439                    int = Some(match builtin {
1440                        Either::Left(bi) => match bi {
1441                            BuiltinInt::Isize => IntegerType::Pointer(true),
1442                            BuiltinInt::I8 => IntegerType::Fixed(Integer::I8, true),
1443                            BuiltinInt::I16 => IntegerType::Fixed(Integer::I16, true),
1444                            BuiltinInt::I32 => IntegerType::Fixed(Integer::I32, true),
1445                            BuiltinInt::I64 => IntegerType::Fixed(Integer::I64, true),
1446                            BuiltinInt::I128 => IntegerType::Fixed(Integer::I128, true),
1447                        },
1448                        Either::Right(bu) => match bu {
1449                            BuiltinUint::Usize => IntegerType::Pointer(false),
1450                            BuiltinUint::U8 => IntegerType::Fixed(Integer::I8, false),
1451                            BuiltinUint::U16 => IntegerType::Fixed(Integer::I16, false),
1452                            BuiltinUint::U32 => IntegerType::Fixed(Integer::I32, false),
1453                            BuiltinUint::U64 => IntegerType::Fixed(Integer::I64, false),
1454                            BuiltinUint::U128 => IntegerType::Fixed(Integer::I128, false),
1455                        },
1456                    });
1457                }
1458                ReprOptions { int, ..Default::default() }
1459            }
1460        };
1461        merge_repr(&mut acc, repr);
1462    }
1463
1464    Some(acc)
1465}

1475#[derive(Debug)]
1476enum DocAtom {
1477    /// eg. `#[doc(hidden)]`
1478    Flag(SmolStr),
1479    /// eg. `#[doc(alias = "it")]`
1480    ///
1481    /// Note that a key can have multiple values that are all considered "active" at the same time.
1482    /// For example, `#[doc(alias = "x")]` and `#[doc(alias = "y")]`.
1483    KeyValue { key: SmolStr, value: SmolStr },
1484    /// eg. `#[doc(alias("x", "y"))]`
1485    Alias(Vec<SmolStr>),
1486}
1487
1488impl DocAtom {
1489    fn parse(tt: ast::TokenTree) -> SmallVec<[DocAtom; 1]> {
1490        let mut iter = TokenTreeChildren::new(&tt).peekable();
1491        let mut result = SmallVec::new();
1492        while iter.peek().is_some() {
1493            if let Some(expr) = next_doc_expr(&mut iter) {
1494                result.push(expr);
1495            }
1496        }
1497        result
1498    }
1499}
1500
1501fn next_doc_expr(it: &mut Peekable<TokenTreeChildren>) -> Option<DocAtom> {
1502    let name = match it.next() {
1503        Some(NodeOrToken::Token(token)) if token.kind().is_any_identifier() => {
1504            SmolStr::new(token.text())
1505        }
1506        _ => return None,
1507    };
1508
1509    let ret = match it.peek() {
1510        Some(NodeOrToken::Token(eq)) if eq.kind() == T![=] => {
1511            it.next();
1512            if let Some(NodeOrToken::Token(value)) = it.next()
1513                && let Some(value) = ast::String::cast(value)
1514                && let Ok(value) = value.value()
1515            {
1516                DocAtom::KeyValue { key: name, value: SmolStr::new(&*value) }
1517            } else {
1518                return None;
1519            }
1520        }
1521        Some(NodeOrToken::Node(subtree)) => {
1522            if name != "alias" {
1523                return None;
1524            }
1525            let aliases = TokenTreeChildren::new(subtree)
1526                .filter_map(|alias| {
1527                    Some(SmolStr::new(&*ast::String::cast(alias.into_token()?)?.value().ok()?))
1528                })
1529                .collect();
1530            it.next();
1531            DocAtom::Alias(aliases)
1532        }
1533        _ => DocAtom::Flag(name),
1534    };
1535    Some(ret)
1536}
