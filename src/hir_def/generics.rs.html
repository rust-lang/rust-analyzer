76impl TypeOrConstParamData {
77    pub fn name(&self) -> Option<&Name> {
78        match self {
79            TypeOrConstParamData::TypeParamData(it) => it.name.as_ref(),
80            TypeOrConstParamData::ConstParamData(it) => Some(&it.name),
81        }
82    }
83
84    pub fn has_default(&self) -> bool {
85        match self {
86            TypeOrConstParamData::TypeParamData(it) => it.default.is_some(),
87            TypeOrConstParamData::ConstParamData(it) => it.default.is_some(),
88        }
89    }
90
91    pub fn type_param(&self) -> Option<&TypeParamData> {
92        match self {
93            TypeOrConstParamData::TypeParamData(it) => Some(it),
94            TypeOrConstParamData::ConstParamData(_) => None,
95        }
96    }
97
98    pub fn const_param(&self) -> Option<&ConstParamData> {
99        match self {
100            TypeOrConstParamData::TypeParamData(_) => None,
101            TypeOrConstParamData::ConstParamData(it) => Some(it),
102        }
103    }
104
105    pub fn is_trait_self(&self) -> bool {
106        match self {
107            TypeOrConstParamData::TypeParamData(it) => {
108                it.provenance == TypeParamProvenance::TraitSelf
109            }
110            TypeOrConstParamData::ConstParamData(_) => false,
111        }
112    }

186/// A single predicate from a where clause, i.e. `where Type: Trait`. Combined
187/// where clauses like `where T: Foo + Bar` are turned into multiple of these.
188/// It might still result in multiple actual predicates though, because of
189/// associated type bindings like `Iterator<Item = u32>`.
190#[derive(Clone, PartialEq, Eq, Debug, Hash)]
191pub enum WherePredicate {
192    TypeBound { target: WherePredicateTypeTarget, bound: TypeBound },
193    Lifetime { target: LifetimeRef, bound: LifetimeRef },
194    ForLifetime { lifetimes: Box<[Name]>, target: WherePredicateTypeTarget, bound: TypeBound },
195}
196
197#[derive(Clone, PartialEq, Eq, Debug, Hash)]
198pub enum WherePredicateTypeTarget {
199    TypeRef(TypeRefId),
200    /// For desugared where predicates that can directly refer to a type param.
201    TypeOrConstParam(LocalTypeOrConstParamId),
202}

272    #[inline]
273    pub fn trait_self_param(&self) -> Option<LocalTypeOrConstParamId> {
274        if self.type_or_consts.is_empty() {
275            return None;
276        }
277        matches!(
278            self.type_or_consts[SELF_PARAM_ID_IN_SELF],
279            TypeOrConstParamData::TypeParamData(TypeParamData {

284        .then(|| SELF_PARAM_ID_IN_SELF)

304    pub(crate) fn generic_params_with_source_map_query(
305        db: &dyn DefDatabase,
306        def: GenericDefId,
307    ) -> (Arc<GenericParams>, Option<Arc<TypesSourceMap>>) {
308        let _p = tracing::info_span!("generic_params_query").entered();
309
310        let krate = def.krate(db);
311        let cfg_options = &krate.cfg_options(db);
312
313        // Returns the generic parameters that are enabled under the current `#[cfg]` options
314        let enabled_params =
315            |params: &Arc<GenericParams>, item_tree: &ItemTree, parent: GenericModItem| {
316                let enabled = |param| item_tree.attrs(db, krate, param).is_cfg_enabled(cfg_options);
317                let attr_owner_ct = |param| AttrOwner::TypeOrConstParamData(parent, param);
318                let attr_owner_lt = |param| AttrOwner::LifetimeParamData(parent, param);
319
320                // In the common case, no parameters will by disabled by `#[cfg]` attributes.
321                // Therefore, make a first pass to check if all parameters are enabled and, if so,
322                // clone the `Interned<GenericParams>` instead of recreating an identical copy.
323                let all_type_or_consts_enabled =
324                    params.type_or_consts.iter().all(|(idx, _)| enabled(attr_owner_ct(idx)));
325                let all_lifetimes_enabled =
326                    params.lifetimes.iter().all(|(idx, _)| enabled(attr_owner_lt(idx)));
327
328                if all_type_or_consts_enabled && all_lifetimes_enabled {
329                    params.clone()
330                } else {
331                    Arc::new(GenericParams {
332                        type_or_consts: if all_type_or_consts_enabled {
333                            params.type_or_consts.clone()
334                        } else {
335                            {
336                                params
337                                    .type_or_consts
338                                    .iter()
339                                    .filter(|&(idx, _)| enabled(attr_owner_ct(idx)))
340                                    .map(|(_, param)| param.clone())
341                                    .collect()
342                            }
343                        },
344                        lifetimes: if all_lifetimes_enabled {
345                            params.lifetimes.clone()
346                        } else {
347                            {
348                                params
349                                    .lifetimes
350                                    .iter()
351                                    .filter(|&(idx, _)| enabled(attr_owner_lt(idx)))
352                                    .map(|(_, param)| param.clone())
353                                    .collect()
354                            }
355                        },
356                        where_predicates: params.where_predicates.clone(),
357                        types_map: params.types_map.clone(),
358                    })
359                }
360            };
361        fn id_to_generics<Id: GenericsItemTreeNode>(
362            db: &dyn DefDatabase,
363            id: impl Lookup<Database = dyn DefDatabase, Data = impl ItemTreeLoc<Id = Id>>,
364            enabled_params: impl Fn(
365                &Arc<GenericParams>,
366                &ItemTree,
367                GenericModItem,
368            ) -> Arc<GenericParams>,
369        ) -> (Arc<GenericParams>, Option<Arc<TypesSourceMap>>)
370        where
371            FileItemTreeId<Id>: Into<GenericModItem>,
372        {
373            let id = id.lookup(db).item_tree_id();
374            let tree = id.item_tree(db);
375            let item = &tree[id.value];
376
377            (enabled_params(item.generic_params(), &tree, id.value.into()), None)
378        }
379
380        match def {
381            GenericDefId::FunctionId(id) => {
382                let loc = id.lookup(db);
383                let tree = loc.id.item_tree(db);
384                let item = &tree[loc.id.value];
385
386                let enabled_params =
387                    enabled_params(&item.explicit_generic_params, &tree, loc.id.value.into());
388
389                let module = loc.container.module(db);
390                let func_data = db.function_data(id);
391                if func_data.params.is_empty() {
392                    (enabled_params, None)
393                } else {
394                    let source_maps = loc.id.item_tree_with_source_map(db).1;
395                    let item_source_maps = source_maps.function(loc.id.value);
396                    let mut generic_params = GenericParamsCollector {
397                        type_or_consts: enabled_params.type_or_consts.clone(),
398                        lifetimes: enabled_params.lifetimes.clone(),
399                        where_predicates: enabled_params.where_predicates.clone().into(),
400                    };
401
402                    let (mut types_map, mut types_source_maps) =
403                        (enabled_params.types_map.clone(), item_source_maps.generics().clone());
404                    // Don't create an `Expander` if not needed since this
405                    // could cause a reparse after the `ItemTree` has been created due to the spanmap.
406                    let mut expander = None;
407                    for &param in func_data.params.iter() {
408                        generic_params.fill_implicit_impl_trait_args(
409                            db,
410                            &mut types_map,
411                            &mut types_source_maps,
412                            &mut expander,
413                            &mut || {
414                                let (def_map, local_def_map) = module.local_def_map(db);
415                                (
416                                    def_map,
417                                    local_def_map,
418                                    Expander::new(db, loc.id.file_id(), module),
419                                )
420                            },
421                            param,
422                            &item.types_map,
423                            item_source_maps.item(),
424                        );
425                    }
426                    let generics = generic_params.finish(types_map, &mut types_source_maps);
427                    (generics, Some(Arc::new(types_source_maps)))
428                }
429            }
430            GenericDefId::AdtId(AdtId::StructId(id)) => id_to_generics(db, id, enabled_params),
431            GenericDefId::AdtId(AdtId::EnumId(id)) => id_to_generics(db, id, enabled_params),
432            GenericDefId::AdtId(AdtId::UnionId(id)) => id_to_generics(db, id, enabled_params),
433            GenericDefId::TraitId(id) => id_to_generics(db, id, enabled_params),
434            GenericDefId::TraitAliasId(id) => id_to_generics(db, id, enabled_params),
435            GenericDefId::TypeAliasId(id) => id_to_generics(db, id, enabled_params),
436            GenericDefId::ImplId(id) => id_to_generics(db, id, enabled_params),
437            GenericDefId::ConstId(_) | GenericDefId::StaticId(_) => (
438                Arc::new(GenericParams {
439                    type_or_consts: Default::default(),
440                    lifetimes: Default::default(),
441                    where_predicates: Default::default(),
442                    types_map: Default::default(),
443                }),
444                None,
445            ),
446        }
447    }

457impl GenericParamsCollector {
458    pub(crate) fn fill_self_param(&mut self) {
459        self.type_or_consts.alloc(
460            TypeParamData {
461                name: Some(Name::new_symbol_root(sym::Self_.clone())),
462                default: None,
463                provenance: TypeParamProvenance::TraitSelf,
464            }
465            .into(),

469    pub(crate) fn fill(
470        &mut self,
471        lower_ctx: &mut LowerCtx<'_>,
472        node: &dyn HasGenericParams,
473        add_param_attrs: impl FnMut(
474            Either<LocalTypeOrConstParamId, LocalLifetimeParamId>,
475            ast::GenericParam,
476        ),
477    ) {
478        if let Some(params) = node.generic_param_list() {
479            self.fill_params(lower_ctx, params, add_param_attrs)
480        }
481        if let Some(where_clause) = node.where_clause() {
482            self.fill_where_predicates(lower_ctx, where_clause);
483        }
484    }
485
486    pub(crate) fn fill_bounds(
487        &mut self,
488        lower_ctx: &mut LowerCtx<'_>,
489        type_bounds: Option<ast::TypeBoundList>,
490        target: Either<TypeRefId, LifetimeRef>,
491    ) {
492        for bound in type_bounds.iter().flat_map(|type_bound_list| type_bound_list.bounds()) {
493            self.add_where_predicate_from_bound(lower_ctx, bound, None, target.clone());
494        }
495    }
496
497    fn fill_params(
498        &mut self,
499        lower_ctx: &mut LowerCtx<'_>,
500        params: ast::GenericParamList,
501        mut add_param_attrs: impl FnMut(
502            Either<LocalTypeOrConstParamId, LocalLifetimeParamId>,
503            ast::GenericParam,
504        ),
505    ) {
506        for type_or_const_param in params.type_or_const_params() {
507            match type_or_const_param {
508                ast::TypeOrConstParam::Type(type_param) => {
509                    let name = type_param.name().map_or_else(Name::missing, |it| it.as_name());
510                    // FIXME: Use `Path::from_src`
511                    let default =
512                        type_param.default_type().map(|it| TypeRef::from_ast(lower_ctx, it));
513                    let param = TypeParamData {
514                        name: Some(name.clone()),
515                        default,
516                        provenance: TypeParamProvenance::TypeParamList,
517                    };
518                    let idx = self.type_or_consts.alloc(param.into());
519                    let type_ref = lower_ctx.alloc_type_ref_desugared(TypeRef::Path(name.into()));
520                    self.fill_bounds(
521                        lower_ctx,
522                        type_param.type_bound_list(),
523                        Either::Left(type_ref),
524                    );
525                    add_param_attrs(Either::Left(idx), ast::GenericParam::TypeParam(type_param));
526                }
527                ast::TypeOrConstParam::Const(const_param) => {
528                    let name = const_param.name().map_or_else(Name::missing, |it| it.as_name());
529                    let ty = TypeRef::from_ast_opt(lower_ctx, const_param.ty());
530                    let param = ConstParamData {
531                        name,
532                        ty,
533                        default: ConstRef::from_const_param(lower_ctx, &const_param),
534                    };
535                    let idx = self.type_or_consts.alloc(param.into());
536                    add_param_attrs(Either::Left(idx), ast::GenericParam::ConstParam(const_param));
537                }
538            }
539        }
540        for lifetime_param in params.lifetime_params() {
541            let name =
542                lifetime_param.lifetime().map_or_else(Name::missing, |lt| Name::new_lifetime(&lt));
543            let param = LifetimeParamData { name: name.clone() };
544            let idx = self.lifetimes.alloc(param);
545            let lifetime_ref = LifetimeRef::new_name(name);
546            self.fill_bounds(
547                lower_ctx,
548                lifetime_param.type_bound_list(),
549                Either::Right(lifetime_ref),
550            );
551            add_param_attrs(Either::Right(idx), ast::GenericParam::LifetimeParam(lifetime_param));
552        }
553    }
554
555    fn fill_where_predicates(
556        &mut self,
557        lower_ctx: &mut LowerCtx<'_>,
558        where_clause: ast::WhereClause,
559    ) {
560        for pred in where_clause.predicates() {
561            let target = if let Some(type_ref) = pred.ty() {
562                Either::Left(TypeRef::from_ast(lower_ctx, type_ref))
563            } else if let Some(lifetime) = pred.lifetime() {
564                Either::Right(LifetimeRef::new(&lifetime))
565            } else {
566                continue;
567            };
568
569            let lifetimes: Option<Box<_>> = pred.generic_param_list().map(|param_list| {
570                // Higher-Ranked Trait Bounds
571                param_list
572                    .lifetime_params()
573                    .map(|lifetime_param| {
574                        lifetime_param
575                            .lifetime()
576                            .map_or_else(Name::missing, |lt| Name::new_lifetime(&lt))
577                    })
578                    .collect()
579            });
580            for bound in pred.type_bound_list().iter().flat_map(|l| l.bounds()) {
581                self.add_where_predicate_from_bound(
582                    lower_ctx,
583                    bound,
584                    lifetimes.as_deref(),
585                    target.clone(),
586                );
587            }
588        }
589    }
590
591    fn add_where_predicate_from_bound(
592        &mut self,
593        lower_ctx: &mut LowerCtx<'_>,
594        bound: ast::TypeBound,
595        hrtb_lifetimes: Option<&[Name]>,
596        target: Either<TypeRefId, LifetimeRef>,
597    ) {
598        let bound = TypeBound::from_ast(lower_ctx, bound);
599        self.fill_impl_trait_bounds(lower_ctx.take_impl_traits_bounds());
600        let predicate = match (target, bound) {
601            (Either::Left(type_ref), bound) => match hrtb_lifetimes {
602                Some(hrtb_lifetimes) => WherePredicate::ForLifetime {
603                    lifetimes: hrtb_lifetimes.to_vec().into_boxed_slice(),
604                    target: WherePredicateTypeTarget::TypeRef(type_ref),
605                    bound,
606                },
607                None => WherePredicate::TypeBound {
608                    target: WherePredicateTypeTarget::TypeRef(type_ref),
609                    bound,
610                },
611            },
612            (Either::Right(lifetime), TypeBound::Lifetime(bound)) => {
613                WherePredicate::Lifetime { target: lifetime, bound }
614            }
615            _ => return,
616        };
617        self.where_predicates.push(predicate);
618    }
619
620    fn fill_impl_trait_bounds(&mut self, impl_bounds: Vec<ThinVec<TypeBound>>) {
621        for bounds in impl_bounds {
622            let param = TypeParamData {
623                name: None,
624                default: None,
625                provenance: TypeParamProvenance::ArgumentImplTrait,
626            };
627            let param_id = self.type_or_consts.alloc(param.into());
628            for bound in &bounds {
629                self.where_predicates.push(WherePredicate::TypeBound {
630                    target: WherePredicateTypeTarget::TypeOrConstParam(param_id),
631                    bound: bound.clone(),
632                });
633            }
634        }
635    }

639        db: &dyn DefDatabase,
640        generics_types_map: &mut TypesMap,
641        generics_types_source_map: &mut TypesSourceMap,
642        // FIXME: Change this back to `LazyCell` if https://github.com/rust-lang/libs-team/issues/429 is accepted.
643        exp: &mut Option<(Arc<DefMap>, Arc<LocalDefMap>, Expander)>,
644        exp_fill: &mut dyn FnMut() -> (Arc<DefMap>, Arc<LocalDefMap>, Expander),
645        type_ref: TypeRefId,
646        types_map: &TypesMap,
647        types_source_map: &TypesSourceMap,
648    ) {
649        TypeRef::walk(type_ref, types_map, &mut |type_ref| {
650            if let TypeRef::ImplTrait(bounds) = type_ref {
651                let param = TypeParamData {
652                    name: None,
653                    default: None,
654                    provenance: TypeParamProvenance::ArgumentImplTrait,
655                };
656                let param_id = self.type_or_consts.alloc(param.into());
657                for bound in bounds {
658                    let bound = copy_type_bound(
659                        bound,
660                        types_map,
661                        types_source_map,
662                        generics_types_map,
663                        generics_types_source_map,
664                    );
665                    self.where_predicates.push(WherePredicate::TypeBound {
666                        target: WherePredicateTypeTarget::TypeOrConstParam(param_id),
667                        bound,
668                    });
669                }
670            }
671
672            if let TypeRef::Macro(mc) = type_ref {
673                let macro_call = mc.to_node(db.upcast());
674                let (def_map, local_def_map, expander) = exp.get_or_insert_with(&mut *exp_fill);
675
676                let module = expander.module.local_id;
677                let resolver = |path: &_| {
678                    def_map
679                        .resolve_path(
680                            local_def_map,
681                            db,
682                            module,
683                            path,
684                            crate::item_scope::BuiltinShadowMode::Other,
685                            Some(MacroSubNs::Bang),
686                        )
687                        .0
688                        .take_macros()
689                };
690                if let Ok(ExpandResult { value: Some((mark, expanded)), .. }) =
691                    expander.enter_expand(db, macro_call, resolver)
692                {
693                    let (mut macro_types_map, mut macro_types_source_map) =
694                        (TypesMap::default(), TypesSourceMap::default());
695                    let mut ctx =
696                        expander.ctx(db, &mut macro_types_map, &mut macro_types_source_map);
697                    let type_ref = TypeRef::from_ast(&mut ctx, expanded.tree());
698                    self.fill_implicit_impl_trait_args(
699                        db,
700                        generics_types_map,
701                        generics_types_source_map,
702                        &mut *exp,
703                        exp_fill,
704                        type_ref,
705                        &macro_types_map,
706                        &macro_types_source_map,
707                    );
708                    exp.get_or_insert_with(&mut *exp_fill).2.exit(mark);
709                }
710            }
711        });
712    }
713
714    pub(crate) fn finish(
715        self,
716        mut generics_types_map: TypesMap,
717        generics_types_source_map: &mut TypesSourceMap,
718    ) -> Arc<GenericParams> {
719        let Self { mut lifetimes, mut type_or_consts, mut where_predicates } = self;
720
721        if lifetimes.is_empty() && type_or_consts.is_empty() && where_predicates.is_empty() {
722            static EMPTY: LazyLock<Arc<GenericParams>> = LazyLock::new(|| {
723                Arc::new(GenericParams {
724                    lifetimes: Arena::new(),
725                    type_or_consts: Arena::new(),
726                    where_predicates: Box::default(),
727                    types_map: TypesMap::default(),
728                })
729            });
730            return Arc::clone(&EMPTY);
731        }
732
733        lifetimes.shrink_to_fit();
734        type_or_consts.shrink_to_fit();
735        where_predicates.shrink_to_fit();
736        generics_types_map.shrink_to_fit();
737        generics_types_source_map.shrink_to_fit();
738        Arc::new(GenericParams {
739            type_or_consts,
740            lifetimes,
741            where_predicates: where_predicates.into_boxed_slice(),
742            types_map: generics_types_map,
743        })
744    }
745}

747/// Copies a `TypeRef` from a `TypesMap` (accompanied with `TypesSourceMap`) into another `TypesMap`
748/// (and `TypesSourceMap`).
749fn copy_type_ref(
750    type_ref: TypeRefId,
751    from: &TypesMap,
752    from_source_map: &TypesSourceMap,
753    to: &mut TypesMap,
754    to_source_map: &mut TypesSourceMap,
755) -> TypeRefId {
756    let result = match &from[type_ref] {
757        TypeRef::Fn(fn_) => {
758            let params = fn_.params.iter().map(|(name, param_type)| {
759                (name.clone(), copy_type_ref(*param_type, from, from_source_map, to, to_source_map))
760            });
761            TypeRef::Fn(Box::new(FnType {
762                params: params.collect(),
763                is_varargs: fn_.is_varargs,
764                is_unsafe: fn_.is_unsafe,
765                abi: fn_.abi.clone(),
766            }))
767        }
768        TypeRef::Tuple(types) => TypeRef::Tuple(ThinVec::from_iter(
769            types.iter().map(|&t| copy_type_ref(t, from, from_source_map, to, to_source_map)),
770        )),
771        &TypeRef::RawPtr(type_ref, mutbl) => TypeRef::RawPtr(
772            copy_type_ref(type_ref, from, from_source_map, to, to_source_map),
773            mutbl,
774        ),
775        TypeRef::Reference(ref_) => TypeRef::Reference(Box::new(RefType {
776            ty: copy_type_ref(ref_.ty, from, from_source_map, to, to_source_map),
777            lifetime: ref_.lifetime.clone(),
778            mutability: ref_.mutability,
779        })),
780        TypeRef::Array(array) => TypeRef::Array(Box::new(ArrayType {
781            ty: copy_type_ref(array.ty, from, from_source_map, to, to_source_map),
782            len: array.len.clone(),
783        })),
784        &TypeRef::Slice(type_ref) => {
785            TypeRef::Slice(copy_type_ref(type_ref, from, from_source_map, to, to_source_map))
786        }
787        TypeRef::ImplTrait(bounds) => TypeRef::ImplTrait(ThinVec::from_iter(copy_type_bounds(
788            bounds,
789            from,
790            from_source_map,
791            to,
792            to_source_map,
793        ))),
794        TypeRef::DynTrait(bounds) => TypeRef::DynTrait(ThinVec::from_iter(copy_type_bounds(
795            bounds,
796            from,
797            from_source_map,
798            to,
799            to_source_map,
800        ))),
801        TypeRef::Path(path) => {
802            TypeRef::Path(copy_path(path, from, from_source_map, to, to_source_map))
803        }
804        TypeRef::Never => TypeRef::Never,
805        TypeRef::Placeholder => TypeRef::Placeholder,
806        TypeRef::Macro(macro_call) => TypeRef::Macro(*macro_call),
807        TypeRef::Error => TypeRef::Error,
808    };
809    let id = to.types.alloc(result);
810    if let Some(&ptr) = from_source_map.types_map_back.get(id) {
811        to_source_map.types_map_back.insert(id, ptr);
812    }
813    id
814}
815
816fn copy_path(
817    path: &Path,
818    from: &TypesMap,
819    from_source_map: &TypesSourceMap,
820    to: &mut TypesMap,
821    to_source_map: &mut TypesSourceMap,
822) -> Path {
823    match path {
824        Path::BarePath(mod_path) => Path::BarePath(mod_path.clone()),
825        Path::Normal(path) => {
826            let type_anchor = path
827                .type_anchor
828                .map(|type_ref| copy_type_ref(type_ref, from, from_source_map, to, to_source_map));
829            let mod_path = path.mod_path.clone();
830            let generic_args = path.generic_args.iter().map(|generic_args| {
831                copy_generic_args(generic_args, from, from_source_map, to, to_source_map)
832            });
833            Path::Normal(Box::new(NormalPath {
834                generic_args: generic_args.collect(),
835                type_anchor,
836                mod_path,
837            }))
838        }
839        Path::LangItem(lang_item, name) => Path::LangItem(*lang_item, name.clone()),
840    }
841}
842
843fn copy_generic_args(
844    generic_args: &Option<GenericArgs>,
845    from: &TypesMap,
846    from_source_map: &TypesSourceMap,
847    to: &mut TypesMap,
848    to_source_map: &mut TypesSourceMap,
849) -> Option<GenericArgs> {
850    generic_args.as_ref().map(|generic_args| {
851        let args = generic_args
852            .args
853            .iter()
854            .map(|arg| match arg {
855                &GenericArg::Type(ty) => {
856                    GenericArg::Type(copy_type_ref(ty, from, from_source_map, to, to_source_map))
857                }
858                GenericArg::Lifetime(lifetime) => GenericArg::Lifetime(lifetime.clone()),
859                GenericArg::Const(konst) => GenericArg::Const(konst.clone()),
860            })
861            .collect();
862        let bindings = generic_args
863            .bindings
864            .iter()
865            .map(|binding| {
866                let name = binding.name.clone();
867                let args =
868                    copy_generic_args(&binding.args, from, from_source_map, to, to_source_map);
869                let type_ref = binding.type_ref.map(|type_ref| {
870                    copy_type_ref(type_ref, from, from_source_map, to, to_source_map)
871                });
872                let bounds =
873                    copy_type_bounds(&binding.bounds, from, from_source_map, to, to_source_map)
874                        .collect();
875                AssociatedTypeBinding { name, args, type_ref, bounds }
876            })
877            .collect();
878        GenericArgs {
879            args,
880            has_self_type: generic_args.has_self_type,
881            bindings,
882            parenthesized: generic_args.parenthesized,
883        }
884    })
885}
886
887fn copy_type_bounds<'a>(
888    bounds: &'a [TypeBound],
889    from: &'a TypesMap,
890    from_source_map: &'a TypesSourceMap,
891    to: &'a mut TypesMap,
892    to_source_map: &'a mut TypesSourceMap,
893) -> impl Iterator<Item = TypeBound> + 'a {
894    bounds.iter().map(|bound| copy_type_bound(bound, from, from_source_map, to, to_source_map))
895}
896
897fn copy_type_bound(
898    bound: &TypeBound,
899    from: &TypesMap,
900    from_source_map: &TypesSourceMap,
901    to: &mut TypesMap,
902    to_source_map: &mut TypesSourceMap,
903) -> TypeBound {
904    let mut copy_path_id = |path: PathId| {
905        let new_path = copy_path(&from[path], from, from_source_map, to, to_source_map);
906        let new_path_id = to.types.alloc(TypeRef::Path(new_path));
907        if let Some(&ptr) = from_source_map.types_map_back.get(path.type_ref()) {
908            to_source_map.types_map_back.insert(new_path_id, ptr);
909        }
910        PathId::from_type_ref_unchecked(new_path_id)
911    };
912
913    match bound {
914        &TypeBound::Path(path, modifier) => TypeBound::Path(copy_path_id(path), modifier),
915        TypeBound::ForLifetime(lifetimes, path) => {
916            TypeBound::ForLifetime(lifetimes.clone(), copy_path_id(*path))
917        }
918        TypeBound::Lifetime(lifetime) => TypeBound::Lifetime(lifetime.clone()),
919        TypeBound::Use(use_args) => TypeBound::Use(use_args.clone()),
920        TypeBound::Error => TypeBound::Error,
921    }
