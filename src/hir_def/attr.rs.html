72    pub(crate) fn fields_attrs_query(
73        db: &dyn DefDatabase,
74        v: VariantId,
75    ) -> Arc<ArenaMap<LocalFieldId, Attrs>> {
76        let _p = tracing::info_span!("fields_attrs_query").entered();
77        // FIXME: There should be some proper form of mapping between item tree field ids and hir field ids
78        let mut res = ArenaMap::default();
79        let item_tree;
80        let (parent, fields, krate) = match v {
81            VariantId::EnumVariantId(it) => {
82                let loc = it.lookup(db);
83                let krate = loc.parent.lookup(db).container.krate;
84                item_tree = loc.id.item_tree(db);
85                let variant = &item_tree[loc.id.value];
86                (FieldParent::EnumVariant(loc.id.value), &variant.fields, krate)
87            }
88            VariantId::StructId(it) => {
89                let loc = it.lookup(db);
90                let krate = loc.container.krate;
91                item_tree = loc.id.item_tree(db);
92                let struct_ = &item_tree[loc.id.value];
93                (FieldParent::Struct(loc.id.value), &struct_.fields, krate)
94            }
95            VariantId::UnionId(it) => {
96                let loc = it.lookup(db);
97                let krate = loc.container.krate;
98                item_tree = loc.id.item_tree(db);
99                let union_ = &item_tree[loc.id.value];
100                (FieldParent::Union(loc.id.value), &union_.fields, krate)
101            }
102        };
103
104        let cfg_options = krate.cfg_options(db);
105
106        let mut idx = 0;
107        for (id, _field) in fields.iter().enumerate() {
108            let attrs = item_tree.attrs(db, krate, AttrOwner::make_field_indexed(parent, id));
109            if attrs.is_cfg_enabled(cfg_options) {
110                res.insert(Idx::from_raw(RawIdx::from(idx)), attrs);
111                idx += 1;
112            }
113        }
114
115        Arc::new(res)
116    }
117}

244fn parse_rustc_legacy_const_generics(tt: &crate::tt::TopSubtree) -> Box<[u32]> {
245    let mut indices = Vec::new();
246    let mut iter = tt.iter();
247    while let (Some(first), second) = (iter.next(), iter.next()) {
248        match first {
249            TtElement::Leaf(tt::Leaf::Literal(lit)) => match lit.symbol.as_str().parse() {
250                Ok(index) => indices.push(index),
251                Err(_) => break,
252            },
253            _ => break,
254        }
255
256        if let Some(comma) = second {
257            match comma {
258                TtElement::Leaf(tt::Leaf::Punct(punct)) if punct.char == ',' => {}
259                _ => break,
260            }
261        }
262    }
263
264    indices.into_boxed_slice()
265}

267fn merge_repr(this: &mut ReprOptions, other: ReprOptions) {
268    let ReprOptions { int, align, pack, flags, field_shuffle_seed: _ } = this;
269    flags.insert(other.flags);
270    *align = (*align).max(other.align);
271    *pack = match (*pack, other.pack) {
272        (Some(pack), None) | (None, Some(pack)) => Some(pack),
273        _ => (*pack).min(other.pack),
274    };
275    if other.int.is_some() {
276        *int = other.int;
277    }
278}

280fn parse_repr_tt(tt: &crate::tt::TopSubtree) -> Option<ReprOptions> {
281    use crate::builtin_type::{BuiltinInt, BuiltinUint};
282    use rustc_abi::{Align, Integer, IntegerType, ReprFlags, ReprOptions};
283
284    match tt.top_subtree().delimiter {
285        tt::Delimiter { kind: DelimiterKind::Parenthesis, .. } => {}
286        _ => return None,
287    }
288
289    let mut acc = ReprOptions::default();
290    let mut tts = tt.iter();
291    while let Some(tt) = tts.next() {
292        let TtElement::Leaf(tt::Leaf::Ident(ident)) = tt else {
293            continue;
294        };
295        let repr = match &ident.sym {
296            s if *s == sym::packed => {
297                let pack = if let Some(TtElement::Subtree(_, mut tt_iter)) = tts.peek() {
298                    tts.next();
299                    if let Some(TtElement::Leaf(tt::Leaf::Literal(lit))) = tt_iter.next() {
300                        lit.symbol.as_str().parse().unwrap_or_default()
301                    } else {
302                        0
303                    }
304                } else {
305                    0
306                };
307                let pack = Some(Align::from_bytes(pack).unwrap_or(Align::ONE));
308                ReprOptions { pack, ..Default::default() }
309            }
310            s if *s == sym::align => {
311                let mut align = None;
312                if let Some(TtElement::Subtree(_, mut tt_iter)) = tts.peek() {
313                    tts.next();
314                    if let Some(TtElement::Leaf(tt::Leaf::Literal(lit))) = tt_iter.next() {
315                        if let Ok(a) = lit.symbol.as_str().parse() {
316                            align = Align::from_bytes(a).ok();
317                        }
318                    }
319                }
320                ReprOptions { align, ..Default::default() }
321            }
322            s if *s == sym::C => ReprOptions { flags: ReprFlags::IS_C, ..Default::default() },
323            s if *s == sym::transparent => {
324                ReprOptions { flags: ReprFlags::IS_TRANSPARENT, ..Default::default() }
325            }
326            s if *s == sym::simd => ReprOptions { flags: ReprFlags::IS_SIMD, ..Default::default() },
327            repr => {
328                let mut int = None;
329                if let Some(builtin) = BuiltinInt::from_suffix_sym(repr)
330                    .map(Either::Left)
331                    .or_else(|| BuiltinUint::from_suffix_sym(repr).map(Either::Right))
332                {
333                    int = Some(match builtin {
334                        Either::Left(bi) => match bi {
335                            BuiltinInt::Isize => IntegerType::Pointer(true),
336                            BuiltinInt::I8 => IntegerType::Fixed(Integer::I8, true),
337                            BuiltinInt::I16 => IntegerType::Fixed(Integer::I16, true),
338                            BuiltinInt::I32 => IntegerType::Fixed(Integer::I32, true),
339                            BuiltinInt::I64 => IntegerType::Fixed(Integer::I64, true),
340                            BuiltinInt::I128 => IntegerType::Fixed(Integer::I128, true),
341                        },
342                        Either::Right(bu) => match bu {
343                            BuiltinUint::Usize => IntegerType::Pointer(false),
344                            BuiltinUint::U8 => IntegerType::Fixed(Integer::I8, false),
345                            BuiltinUint::U16 => IntegerType::Fixed(Integer::I16, false),
346                            BuiltinUint::U32 => IntegerType::Fixed(Integer::I32, false),
347                            BuiltinUint::U64 => IntegerType::Fixed(Integer::I64, false),
348                            BuiltinUint::U128 => IntegerType::Fixed(Integer::I128, false),
349                        },
350                    });
351                }
352                ReprOptions { int, ..Default::default() }
353            }
354        };
355        merge_repr(&mut acc, repr);
356    }
357
358    Some(acc)
359}

361#[derive(Debug, Clone, PartialEq, Eq, Hash)]
362pub enum DocAtom {
363    /// eg. `#[doc(hidden)]`
364    Flag(Symbol),
365    /// eg. `#[doc(alias = "it")]`
366    ///
367    /// Note that a key can have multiple values that are all considered "active" at the same time.
368    /// For example, `#[doc(alias = "x")]` and `#[doc(alias = "y")]`.
369    KeyValue { key: Symbol, value: Symbol },
370}
371
372#[derive(Debug, Clone, PartialEq, Eq, Hash)]
373pub enum DocExpr {
374    Invalid,
375    /// eg. `#[doc(hidden)]`, `#[doc(alias = "x")]`
376    Atom(DocAtom),
377    /// eg. `#[doc(alias("x", "y"))]`
378    Alias(Vec<Symbol>),
379}
380
381impl From<DocAtom> for DocExpr {
382    fn from(atom: DocAtom) -> Self {
383        DocExpr::Atom(atom)
384    }
385}
386
387impl DocExpr {
388    fn parse<S: Copy>(tt: &tt::TopSubtree<S>) -> DocExpr {
389        next_doc_expr(tt.iter()).unwrap_or(DocExpr::Invalid)
390    }
391
392    pub fn aliases(&self) -> &[Symbol] {
393        match self {
394            DocExpr::Atom(DocAtom::KeyValue { key, value }) if *key == sym::alias => {
395                std::slice::from_ref(value)
396            }
397            DocExpr::Alias(aliases) => aliases,
398            _ => &[],
399        }
400    }
401}

403fn next_doc_expr<S: Copy>(mut it: TtIter<'_, S>) -> Option<DocExpr> {
404    let name = match it.next() {
405        None => return None,
406        Some(TtElement::Leaf(tt::Leaf::Ident(ident))) => ident.sym.clone(),
407        Some(_) => return Some(DocExpr::Invalid),
408    };
409
410    // Peek
411    let ret = match it.peek() {
412        Some(TtElement::Leaf(tt::Leaf::Punct(punct))) if punct.char == '=' => {
413            it.next();
414            match it.next() {
415                Some(TtElement::Leaf(tt::Leaf::Literal(tt::Literal {
416                    symbol: text,
417                    kind: tt::LitKind::Str,
418                    ..
419                }))) => DocAtom::KeyValue { key: name, value: text.clone() }.into(),
420                _ => return Some(DocExpr::Invalid),
421            }
422        }
423        Some(TtElement::Subtree(_, subtree_iter)) => {
424            it.next();
425            let subs = parse_comma_sep(subtree_iter);
426            match &name {
427                s if *s == sym::alias => DocExpr::Alias(subs),
428                _ => DocExpr::Invalid,
429            }
430        }
431        _ => DocAtom::Flag(name).into(),
432    };
433    Some(ret)
434}
435
436fn parse_comma_sep<S>(iter: TtIter<'_, S>) -> Vec<Symbol> {
437    iter.filter_map(|tt| match tt {
438        TtElement::Leaf(tt::Leaf::Literal(tt::Literal {
439            kind: tt::LitKind::Str, symbol, ..
440        })) => Some(symbol.clone()),
441        _ => None,
442    })
443    .collect()
444}

451    pub(crate) fn attrs_query(db: &dyn DefDatabase, def: AttrDefId) -> Attrs {
452        let _p = tracing::info_span!("attrs_query").entered();
453        // FIXME: this should use `Trace` to avoid duplication in `source_map` below
454        let raw_attrs = match def {
455            AttrDefId::ModuleId(module) => {
456                let def_map = module.def_map(db);
457                let mod_data = &def_map[module.local_id];
458
459                match mod_data.origin {
460                    ModuleOrigin::File { definition, declaration_tree_id, .. } => {
461                        let decl_attrs = declaration_tree_id
462                            .item_tree(db)
463                            .raw_attrs(AttrOwner::ModItem(declaration_tree_id.value.into()))
464                            .clone();
465                        let tree = db.file_item_tree(definition.into());
466                        let def_attrs = tree.raw_attrs(AttrOwner::TopLevel).clone();
467                        decl_attrs.merge(def_attrs)
468                    }
469                    ModuleOrigin::CrateRoot { definition } => {
470                        let tree = db.file_item_tree(definition.into());
471                        tree.raw_attrs(AttrOwner::TopLevel).clone()
472                    }
473                    ModuleOrigin::Inline { definition_tree_id, .. } => definition_tree_id
474                        .item_tree(db)
475                        .raw_attrs(AttrOwner::ModItem(definition_tree_id.value.into()))
476                        .clone(),
477                    ModuleOrigin::BlockExpr { id, .. } => {
478                        let tree = db.block_item_tree(id);
479                        tree.raw_attrs(AttrOwner::TopLevel).clone()
480                    }
481                }
482            }
483            AttrDefId::FieldId(it) => {
484                return db.fields_attrs(it.parent)[it.local_id].clone();
485            }
486            AttrDefId::EnumVariantId(it) => attrs_from_item_tree_loc(db, it),
487            AttrDefId::AdtId(it) => match it {
488                AdtId::StructId(it) => attrs_from_item_tree_loc(db, it),
489                AdtId::EnumId(it) => attrs_from_item_tree_loc(db, it),
490                AdtId::UnionId(it) => attrs_from_item_tree_loc(db, it),
491            },
492            AttrDefId::TraitId(it) => attrs_from_item_tree_loc(db, it),
493            AttrDefId::TraitAliasId(it) => attrs_from_item_tree_loc(db, it),
494            AttrDefId::MacroId(it) => match it {
495                MacroId::Macro2Id(it) => attrs_from_item_tree_loc(db, it),
496                MacroId::MacroRulesId(it) => attrs_from_item_tree_loc(db, it),
497                MacroId::ProcMacroId(it) => attrs_from_item_tree_loc(db, it),
498            },
499            AttrDefId::ImplId(it) => attrs_from_item_tree_loc(db, it),
500            AttrDefId::ConstId(it) => attrs_from_item_tree_loc(db, it),
501            AttrDefId::StaticId(it) => attrs_from_item_tree_loc(db, it),
502            AttrDefId::FunctionId(it) => attrs_from_item_tree_loc(db, it),
503            AttrDefId::TypeAliasId(it) => attrs_from_item_tree_loc(db, it),
504            AttrDefId::GenericParamId(it) => match it {
505                GenericParamId::ConstParamId(it) => {
506                    let src = it.parent().child_source(db);
507                    // FIXME: We should be never getting `None` here.
508                    match src.value.get(it.local_id()) {
509                        Some(val) => RawAttrs::from_attrs_owner(
510                            db,
511                            src.with_value(val),
512                            db.span_map(src.file_id).as_ref(),
513                        ),
514                        None => RawAttrs::EMPTY,
515                    }
516                }
517                GenericParamId::TypeParamId(it) => {
518                    let src = it.parent().child_source(db);
519                    // FIXME: We should be never getting `None` here.
520                    match src.value.get(it.local_id()) {
521                        Some(val) => RawAttrs::from_attrs_owner(
522                            db,
523                            src.with_value(val),
524                            db.span_map(src.file_id).as_ref(),
525                        ),
526                        None => RawAttrs::EMPTY,
527                    }
528                }
529                GenericParamId::LifetimeParamId(it) => {
530                    let src = it.parent.child_source(db);
531                    // FIXME: We should be never getting `None` here.
532                    match src.value.get(it.local_id) {
533                        Some(val) => RawAttrs::from_attrs_owner(
534                            db,
535                            src.with_value(val),
536                            db.span_map(src.file_id).as_ref(),
537                        ),
538                        None => RawAttrs::EMPTY,
539                    }
540                }
541            },
542            AttrDefId::ExternBlockId(it) => attrs_from_item_tree_loc(db, it),
543            AttrDefId::ExternCrateId(it) => attrs_from_item_tree_loc(db, it),
544            AttrDefId::UseId(it) => attrs_from_item_tree_loc(db, it),
545        };
546
547        let attrs = raw_attrs.filter(db, def.krate(db));
548        Attrs(attrs)
549    }

551    pub fn source_map(&self, db: &dyn DefDatabase) -> AttrSourceMap {
552        let owner = match self.owner {
553            AttrDefId::ModuleId(module) => {
554                // Modules can have 2 attribute owners (the `mod x;` item, and the module file itself).
555
556                let def_map = module.def_map(db);
557                let mod_data = &def_map[module.local_id];
558                match mod_data.declaration_source(db) {
559                    Some(it) => {
560                        let mut map = AttrSourceMap::new(InFile::new(it.file_id, &it.value));
561                        if let InFile { file_id, value: ModuleSource::SourceFile(file) } =
562                            mod_data.definition_source(db)
563                        {
564                            map.append_module_inline_attrs(AttrSourceMap::new(InFile::new(
565                                file_id, &file,
566                            )));
567                        }
568                        return map;
569                    }
570                    None => {
571                        let InFile { file_id, value } = mod_data.definition_source(db);
572                        let attrs_owner = match &value {
573                            ModuleSource::SourceFile(file) => file as &dyn ast::HasAttrs,
574                            ModuleSource::Module(module) => module as &dyn ast::HasAttrs,
575                            ModuleSource::BlockExpr(block) => block as &dyn ast::HasAttrs,
576                        };
577                        return AttrSourceMap::new(InFile::new(file_id, attrs_owner));
578                    }
579                }
580            }
581            AttrDefId::FieldId(id) => {
582                let map = db.fields_attrs_source_map(id.parent);
583                let file_id = id.parent.file_id(db);
584                let root = db.parse_or_expand(file_id);
585                let owner = ast::AnyHasAttrs::new(map[id.local_id].to_node(&root));
586                InFile::new(file_id, owner)
587            }
588            AttrDefId::AdtId(adt) => match adt {
589                AdtId::StructId(id) => any_has_attrs(db, id),
590                AdtId::UnionId(id) => any_has_attrs(db, id),
591                AdtId::EnumId(id) => any_has_attrs(db, id),
592            },
593            AttrDefId::FunctionId(id) => any_has_attrs(db, id),
594            AttrDefId::EnumVariantId(id) => any_has_attrs(db, id),
595            AttrDefId::StaticId(id) => any_has_attrs(db, id),
596            AttrDefId::ConstId(id) => any_has_attrs(db, id),
597            AttrDefId::TraitId(id) => any_has_attrs(db, id),
598            AttrDefId::TraitAliasId(id) => any_has_attrs(db, id),
599            AttrDefId::TypeAliasId(id) => any_has_attrs(db, id),
600            AttrDefId::MacroId(id) => match id {
601                MacroId::Macro2Id(id) => any_has_attrs(db, id),
602                MacroId::MacroRulesId(id) => any_has_attrs(db, id),
603                MacroId::ProcMacroId(id) => any_has_attrs(db, id),
604            },
605            AttrDefId::ImplId(id) => any_has_attrs(db, id),
606            AttrDefId::GenericParamId(id) => match id {
607                GenericParamId::ConstParamId(id) => id
608                    .parent()
609                    .child_source(db)
610                    .map(|source| ast::AnyHasAttrs::new(source[id.local_id()].clone())),
611                GenericParamId::TypeParamId(id) => id
612                    .parent()
613                    .child_source(db)
614                    .map(|source| ast::AnyHasAttrs::new(source[id.local_id()].clone())),
615                GenericParamId::LifetimeParamId(id) => id
616                    .parent
617                    .child_source(db)
618                    .map(|source| ast::AnyHasAttrs::new(source[id.local_id].clone())),
619            },
620            AttrDefId::ExternBlockId(id) => any_has_attrs(db, id),
621            AttrDefId::ExternCrateId(id) => any_has_attrs(db, id),
622            AttrDefId::UseId(id) => any_has_attrs(db, id),
623        };
624
625        AttrSourceMap::new(owner.as_ref().map(|node| node as &dyn HasAttrs))
626    }
627}

629#[derive(Debug)]
630pub struct AttrSourceMap {
631    source: Vec<Either<ast::Attr, ast::Comment>>,
632    file_id: HirFileId,
633    /// If this map is for a module, this will be the [`HirFileId`] of the module's definition site,
634    /// while `file_id` will be the one of the module declaration site.
635    /// The usize is the index into `source` from which point on the entries reside in the def site
636    /// file.
637    mod_def_site_file_id: Option<(HirFileId, usize)>,
638}
639
640impl AttrSourceMap {
641    fn new(owner: InFile<&dyn ast::HasAttrs>) -> Self {
642        Self {
643            source: collect_attrs(owner.value).map(|(_, it)| it).collect(),
644            file_id: owner.file_id,
645            mod_def_site_file_id: None,
646        }
647    }
648
649    /// Append a second source map to this one, this is required for modules, whose outline and inline
650    /// attributes can reside in different files
651    fn append_module_inline_attrs(&mut self, other: Self) {
652        assert!(self.mod_def_site_file_id.is_none() && other.mod_def_site_file_id.is_none());
653        let len = self.source.len();
654        self.source.extend(other.source);
655        if other.file_id != self.file_id {
656            self.mod_def_site_file_id = Some((other.file_id, len));
657        }
658    }
659
660    /// Maps the lowered `Attr` back to its original syntax node.
661    ///
662    /// `attr` must come from the `owner` used for AttrSourceMap
663    ///
664    /// Note that the returned syntax node might be a `#[cfg_attr]`, or a doc comment, instead of
665    /// the attribute represented by `Attr`.
666    pub fn source_of(&self, attr: &Attr) -> InFile<&Either<ast::Attr, ast::Comment>> {
667        self.source_of_id(attr.id)
668    }
669
670    pub fn source_of_id(&self, id: AttrId) -> InFile<&Either<ast::Attr, ast::Comment>> {
671        let ast_idx = id.ast_index();
672        let file_id = match self.mod_def_site_file_id {
673            Some((file_id, def_site_cut)) if def_site_cut <= ast_idx => file_id,
674            _ => self.file_id,
675        };
676
677        self.source
678            .get(ast_idx)
679            .map(|it| InFile::new(file_id, it))
680            .unwrap_or_else(|| panic!("cannot find attr at index {id:?}"))
681    }
682}

684#[derive(Debug, Clone, Copy)]
685pub struct AttrQuery<'attr> {
686    attrs: &'attr Attrs,
687    key: &'attr Symbol,
688}
689
690impl<'attr> AttrQuery<'attr> {
691    pub fn tt_values(self) -> impl Iterator<Item = &'attr crate::tt::TopSubtree> {
692        self.attrs().filter_map(|attr| attr.token_tree_value())
693    }
694
695    pub fn string_value(self) -> Option<&'attr Symbol> {
696        self.attrs().find_map(|attr| attr.string_value())
697    }
698
699    pub fn string_value_with_span(self) -> Option<(&'attr Symbol, span::Span)> {
700        self.attrs().find_map(|attr| attr.string_value_with_span())
701    }
702
703    pub fn string_value_unescape(self) -> Option<Cow<'attr, str>> {
704        self.attrs().find_map(|attr| attr.string_value_unescape())
705    }
706
707    pub fn exists(self) -> bool {
708        self.attrs().next().is_some()
709    }
710
711    pub fn attrs(self) -> impl Iterator<Item = &'attr Attr> + Clone {
712        let key = self.key;
713        self.attrs.iter().filter(move |attr| attr.path.as_ident().is_some_and(|s| *s == *key))
714    }
715
716    /// Find string value for a specific key inside token tree
717    ///
718    /// ```ignore
719    /// #[doc(html_root_url = "url")]
720    ///       ^^^^^^^^^^^^^ key
721    /// ```
722    pub fn find_string_value_in_tt(self, key: &'attr Symbol) -> Option<&'attr str> {
723        self.tt_values().find_map(|tt| {
724            let name = tt.iter()
725                .skip_while(|tt| !matches!(tt, TtElement::Leaf(tt::Leaf::Ident(tt::Ident { sym, ..} )) if *sym == *key))
726                .nth(2);
727
728            match name {
729                Some(TtElement::Leaf(tt::Leaf::Literal(tt::Literal{  symbol: text, kind: tt::LitKind::Str | tt::LitKind::StrRaw(_) , ..}))) => Some(text.as_str()),
730                _ => None
731            }
732        })
733    }
734}

736fn any_has_attrs<'db>(
737    db: &(dyn DefDatabase + 'db),
738    id: impl Lookup<Database = dyn DefDatabase, Data = impl HasSource<Value = impl ast::HasAttrs>>,
739) -> InFile<ast::AnyHasAttrs> {
740    id.lookup(db).source(db).map(ast::AnyHasAttrs::new)
741}

743fn attrs_from_item_tree_loc<'db, N: ItemTreeNode>(
744    db: &(dyn DefDatabase + 'db),
745    lookup: impl Lookup<Database = dyn DefDatabase, Data = impl ItemTreeLoc<Id = N>>,
746) -> RawAttrs {
747    let id = lookup.lookup(db).item_tree_id();
748    let tree = id.item_tree(db);
749    let attr_owner = N::attr_owner(id.value);
750    tree.raw_attrs(attr_owner).clone()
751}

753pub(crate) fn fields_attrs_source_map(
754    db: &dyn DefDatabase,
755    def: VariantId,
756) -> Arc<ArenaMap<LocalFieldId, AstPtr<Either<ast::TupleField, ast::RecordField>>>> {
757    let mut res = ArenaMap::default();
758    let child_source = def.child_source(db);
759
760    for (idx, variant) in child_source.value.iter() {
761        res.insert(
762            idx,
763            variant
764                .as_ref()
765                .either(|l| AstPtr::new(l).wrap_left(), |r| AstPtr::new(r).wrap_right()),
766        );
767    }
768
769    Arc::new(res)
770}
