178    pub fn resolve_path_in_type_ns_with_prefix_info(
179        &self,
180        db: &dyn DefDatabase,
181        path: &Path,
182    ) -> Option<(TypeNs, Option<usize>, Option<ImportOrExternCrate>, ResolvePathResultPrefixInfo)>
183    {
184        let path = match path {
185            Path::BarePath(mod_path) => mod_path,
186            Path::Normal(it) => &it.mod_path,
187            Path::LangItem(l, seg) => {
188                let type_ns = match *l {
189                    LangItemTarget::UnionId(it) => TypeNs::AdtId(it.into()),
190                    LangItemTarget::TypeAliasId(it) => TypeNs::TypeAliasId(it),
191                    LangItemTarget::StructId(it) => TypeNs::AdtId(it.into()),
192                    LangItemTarget::EnumVariantId(it) => TypeNs::EnumVariantId(it),
193                    LangItemTarget::EnumId(it) => TypeNs::AdtId(it.into()),
194                    LangItemTarget::TraitId(it) => TypeNs::TraitId(it),
195                    LangItemTarget::FunctionId(_)
196                    | LangItemTarget::ImplId(_)
197                    | LangItemTarget::StaticId(_) => return None,
198                };
199                return Some((
200                    type_ns,
201                    seg.as_ref().map(|_| 1),
202                    None,
203                    ResolvePathResultPrefixInfo::default(),
204                ));
205            }
206        };
207        let first_name = path.segments().first()?;
208        let skip_to_mod = path.kind != PathKind::Plain;
209        if skip_to_mod {
210            return self.module_scope.resolve_path_in_type_ns(db, path);
211        }
212
213        let remaining_idx = || {
214            if path.segments().len() == 1 { None } else { Some(1) }
215        };
216
217        for scope in self.scopes() {
218            match scope {
219                Scope::ExprScope(_) | Scope::MacroDefScope(_) => continue,
220                Scope::GenericParams { params, def } => {
221                    if let &GenericDefId::ImplId(impl_) = def {
222                        if *first_name == sym::Self_ {
223                            return Some((
224                                TypeNs::SelfType(impl_),
225                                remaining_idx(),
226                                None,
227                                ResolvePathResultPrefixInfo::default(),
228                            ));
229                        }
230                    } else if let &GenericDefId::AdtId(adt) = def
231                        && *first_name == sym::Self_
232                    {
233                        return Some((
234                            TypeNs::AdtSelfType(adt),
235                            remaining_idx(),
236                            None,
237                            ResolvePathResultPrefixInfo::default(),
238                        ));
239                    }
240                    if let Some(id) = params.find_type_by_name(first_name, *def) {
241                        return Some((
242                            TypeNs::GenericParam(id),
243                            remaining_idx(),
244                            None,
245                            ResolvePathResultPrefixInfo::default(),
246                        ));
247                    }
248                }
249                Scope::BlockScope(m) => {
250                    if let Some(res) = m.resolve_path_in_type_ns(db, path) {
251                        let res = match res.0 {
252                            TypeNs::ModuleId(_) if res.1.is_none() => {
253                                if let Some(ModuleDefId::BuiltinType(builtin)) = BUILTIN_SCOPE
254                                    .get(first_name)
255                                    .and_then(|builtin| builtin.take_types())
256                                {
257                                    (
258                                        TypeNs::BuiltinType(builtin),
259                                        remaining_idx(),
260                                        None,
261                                        ResolvePathResultPrefixInfo::default(),
262                                    )
263                                } else {
264                                    res
265                                }
266                            }
267                            _ => res,
268                        };
269                        return Some(res);
270                    }
271                }
272            }
273        }
274        self.module_scope.resolve_path_in_type_ns(db, path)
275    }

289    pub fn resolve_visibility(
290        &self,
291        db: &dyn DefDatabase,
292        visibility: &RawVisibility,
293    ) -> Option<Visibility> {
294        match visibility {
295            RawVisibility::Module(_, _) => {
296                let (item_map, item_local_map, module) = self.item_scope_();
297                item_map.resolve_visibility(
298                    item_local_map,
299                    db,
300                    module,
301                    visibility,
302                    self.scopes().any(|scope| {
303                        matches!(scope, Scope::GenericParams { def: GenericDefId::ImplId(_), .. })
304                    }),
305                )
306            }
307            RawVisibility::PubSelf(explicitness) => {
308                Some(Visibility::Module(self.module(), *explicitness))
309            }
310            RawVisibility::PubCrate => Some(Visibility::PubCrate(self.krate())),
311            RawVisibility::Public => Some(Visibility::Public),
312        }

324    pub fn resolve_path_in_value_ns_with_prefix_info(
325        &self,
326        db: &dyn DefDatabase,
327        path: &Path,
328        mut hygiene_id: HygieneId,
329    ) -> Option<(ResolveValueResult, ResolvePathResultPrefixInfo)> {
330        let path = match path {
331            Path::BarePath(mod_path) => mod_path,
332            Path::Normal(it) => &it.mod_path,
333            Path::LangItem(l, None) => {
334                return Some((
335                    ResolveValueResult::ValueNs(match *l {
336                        LangItemTarget::FunctionId(it) => ValueNs::FunctionId(it),
337                        LangItemTarget::StaticId(it) => ValueNs::StaticId(it),
338                        LangItemTarget::StructId(it) => ValueNs::StructId(it),
339                        LangItemTarget::EnumVariantId(it) => ValueNs::EnumVariantId(it),
340                        LangItemTarget::UnionId(_)
341                        | LangItemTarget::ImplId(_)
342                        | LangItemTarget::TypeAliasId(_)
343                        | LangItemTarget::TraitId(_)
344                        | LangItemTarget::EnumId(_) => return None,
345                    }),
346                    ResolvePathResultPrefixInfo::default(),
347                ));
348            }
349            Path::LangItem(l, Some(_)) => {
350                let type_ns = match *l {
351                    LangItemTarget::UnionId(it) => TypeNs::AdtId(it.into()),
352                    LangItemTarget::TypeAliasId(it) => TypeNs::TypeAliasId(it),
353                    LangItemTarget::StructId(it) => TypeNs::AdtId(it.into()),
354                    LangItemTarget::EnumVariantId(it) => TypeNs::EnumVariantId(it),
355                    LangItemTarget::EnumId(it) => TypeNs::AdtId(it.into()),
356                    LangItemTarget::TraitId(it) => TypeNs::TraitId(it),
357                    LangItemTarget::FunctionId(_)
358                    | LangItemTarget::ImplId(_)
359                    | LangItemTarget::StaticId(_) => return None,
360                };
361                // Remaining segments start from 0 because lang paths have no segments other than the remaining.
362                return Some((
363                    ResolveValueResult::Partial(type_ns, 0),
364                    ResolvePathResultPrefixInfo::default(),
365                ));
366            }
367        };
368        let n_segments = path.segments().len();
369        let tmp = Name::new_symbol_root(sym::self_);
370        let first_name = if path.is_self() { &tmp } else { path.segments().first()? };
371        let skip_to_mod = path.kind != PathKind::Plain && !path.is_self();
372        if skip_to_mod {
373            return self.module_scope.resolve_path_in_value_ns(db, path);
374        }
375
376        if n_segments <= 1 {
377            let mut hygiene_info = hygiene_info(db, hygiene_id);
378            for scope in self.scopes() {
379                match scope {
380                    Scope::ExprScope(scope) => {
381                        let entry =
382                            scope.expr_scopes.entries(scope.scope_id).iter().find(|entry| {
383                                entry.name() == first_name && entry.hygiene() == hygiene_id
384                            });
385
386                        if let Some(e) = entry {
387                            return Some((
388                                ResolveValueResult::ValueNs(ValueNs::LocalBinding(e.binding())),
389                                ResolvePathResultPrefixInfo::default(),
390                            ));
391                        }
392                    }
393                    Scope::MacroDefScope(macro_id) => {
394                        handle_macro_def_scope(db, &mut hygiene_id, &mut hygiene_info, macro_id)
395                    }
396                    Scope::GenericParams { params, def } => {
397                        if let &GenericDefId::ImplId(impl_) = def
398                            && *first_name == sym::Self_
399                        {
400                            return Some((
401                                ResolveValueResult::ValueNs(ValueNs::ImplSelf(impl_)),
402                                ResolvePathResultPrefixInfo::default(),
403                            ));
404                        }
405                        if let Some(id) = params.find_const_by_name(first_name, *def) {
406                            let val = ValueNs::GenericParam(id);
407                            return Some((
408                                ResolveValueResult::ValueNs(val),
409                                ResolvePathResultPrefixInfo::default(),
410                            ));
411                        }
412                    }
413                    Scope::BlockScope(m) => {
414                        if let Some(def) = m.resolve_path_in_value_ns(db, path) {
415                            return Some(def);
416                        }
417                    }
418                }
419            }
420        } else {
421            for scope in self.scopes() {
422                match scope {
423                    Scope::ExprScope(_) | Scope::MacroDefScope(_) => continue,
424                    Scope::GenericParams { params, def } => {
425                        if let &GenericDefId::ImplId(impl_) = def {
426                            if *first_name == sym::Self_ {
427                                return Some((
428                                    ResolveValueResult::Partial(TypeNs::SelfType(impl_), 1),
429                                    ResolvePathResultPrefixInfo::default(),
430                                ));
431                            }
432                        } else if let &GenericDefId::AdtId(adt) = def
433                            && *first_name == sym::Self_
434                        {
435                            let ty = TypeNs::AdtSelfType(adt);
436                            return Some((
437                                ResolveValueResult::Partial(ty, 1),
438                                ResolvePathResultPrefixInfo::default(),
439                            ));
440                        }
441                        if let Some(id) = params.find_type_by_name(first_name, *def) {
442                            let ty = TypeNs::GenericParam(id);
443                            return Some((
444                                ResolveValueResult::Partial(ty, 1),
445                                ResolvePathResultPrefixInfo::default(),
446                            ));
447                        }
448                    }
449                    Scope::BlockScope(m) => {
450                        if let Some(def) = m.resolve_path_in_value_ns(db, path) {
451                            return Some(def);
452                        }
453                    }
454                }
455            }
456        }
457
458        if let Some(res) = self.module_scope.resolve_path_in_value_ns(db, path) {
459            return Some(res);
460        }
461
462        // If a path of the shape `u16::from_le_bytes` failed to resolve at all, then we fall back
463        // to resolving to the primitive type, to allow this to still work in the presence of
464        // `use core::u16;`.
465        if path.kind == PathKind::Plain
466            && n_segments > 1
467            && let Some(builtin) = BuiltinType::by_name(first_name)
468        {
469            return Some((
470                ResolveValueResult::Partial(TypeNs::BuiltinType(builtin), 1),
471                ResolvePathResultPrefixInfo::default(),
472            ));
473        }
474
475        None
476    }

535    /// Returns a set of names available in the current scope.
536    ///
537    /// Note that this is a somewhat fuzzy concept -- internally, the compiler
538    /// doesn't necessary follow a strict scoping discipline. Rather, it just
539    /// tells for each ident what it resolves to.
540    ///
541    /// A good example is something like `str::from_utf8`. From scopes point of
542    /// view, this code is erroneous -- both `str` module and `str` type occupy
543    /// the same type namespace.
544    ///
545    /// We don't try to model that super-correctly -- this functionality is
546    /// primarily exposed for completions.
547    ///
548    /// Note that in Rust one name can be bound to several items:
549    ///
550    /// ```
551    /// # #![allow(non_camel_case_types)]
552    /// macro_rules! t { () => (()) }
553    /// type t = t!();
554    /// const t: t = t!();
555    /// ```
556    ///
557    /// That's why we return a multimap.
558    ///
559    /// The shadowing is accounted for: in
560    ///
561    /// ```ignore
562    /// let it = 92;
563    /// {
564    ///     let it = 92;
565    ///     $0
566    /// }
567    /// ```
568    ///
569    /// there will be only one entry for `it` in the result.
570    ///
571    /// The result is ordered *roughly* from the innermost scope to the
572    /// outermost: when the name is introduced in two namespaces in two scopes,
573    /// we use the position of the first scope.
574    pub fn names_in_scope(
575        &self,
576        db: &dyn DefDatabase,
577    ) -> FxIndexMap<Name, SmallVec<[ScopeDef; 1]>> {
578        let mut res = ScopeNames::default();
579        for scope in self.scopes() {
580            scope.process_names(&mut res, db);
581        }
582        let ModuleItemMap { def_map, module_id, local_def_map } = self.module_scope;
583        // FIXME: should we provide `self` here?
584        // f(
585        //     Name::self_param(),
586        //     PerNs::types(Resolution::Def {
587        //         def: m.module.into(),
588        //     }),
589        // );
590        def_map[module_id].scope.entries().for_each(|(name, def)| {
591            res.add_per_ns(name, def);
592        });
593
594        def_map[module_id].scope.legacy_macros().for_each(|(name, macs)| {
595            macs.iter().for_each(|&mac| {
596                res.add(name, ScopeDef::ModuleDef(ModuleDefId::MacroId(mac)));
597            })
598        });
599        def_map.macro_use_prelude().iter().sorted_by_key(|&(k, _)| k.clone()).for_each(
600            |(name, &(def, _extern_crate))| {
601                res.add(name, ScopeDef::ModuleDef(def.into()));
602            },
603        );
604        local_def_map.extern_prelude().for_each(|(name, (def, _extern_crate))| {
605            res.add(name, ScopeDef::ModuleDef(ModuleDefId::ModuleId(def)));
606        });
607        BUILTIN_SCOPE.iter().for_each(|(name, &def)| {
608            res.add_per_ns(name, def);
609        });
610        if let Some((prelude, _use)) = def_map.prelude() {
611            let prelude_def_map = prelude.def_map(db);
612            for (name, def) in prelude_def_map[prelude].scope.entries() {
613                res.add_per_ns(name, def)
614            }
615        }
616        res.map
617    }

619    /// Note: Not to be used directly within hir-def/hir-ty
620    pub fn extern_crate_decls_in_scope<'a>(
621        &'a self,
622        db: &'a dyn DefDatabase,
623    ) -> impl Iterator<Item = Name> + 'a {
624        self.module_scope.def_map[self.module_scope.module_id]
625            .scope
626            .extern_crate_decls()
627            .filter_map(|id| {
628                let loc = id.lookup(db);
629                let extern_crate = loc.source(db);
630                // If there is a rename (`as x`), extract the renamed name, or remove the `extern crate`
631                // if it is an underscore.
632                extern_crate
633                    .value
634                    .rename()
635                    .map(|a| a.name().map(|it| it.as_name()))
636                    .unwrap_or_else(|| extern_crate.value.name_ref().map(|it| it.as_name()))
637            })
638    }

647    pub fn traits_in_scope(&self, db: &dyn DefDatabase) -> FxHashSet<TraitId> {
648        // FIXME(trait_alias): Trait alias brings aliased traits in scope! Note that supertraits of
649        // aliased traits are NOT brought in scope (unless also aliased).
650        let mut traits = FxHashSet::default();

652        for scope in self.scopes() {
653            match scope {
654                Scope::BlockScope(m) => traits.extend(m.def_map[m.module_id].scope.traits()),

659                            .resolve_path_in_type_ns_fully(db, &impl_data.store[target_trait.path])

661                        traits.insert(trait_);
662                    }
663                }
664                _ => (),

671            traits.extend(prelude_def_map[prelude].scope.traits());
672        }
673        // Fill in module visible traits
674        traits.extend(self.module_scope.def_map[self.module_scope.module_id].scope.traits());
675        traits
676    }

753    /// Checks if we rename `renamed` (currently named `current_name`) to `new_name`, will the meaning of this reference
754    /// (that contains `current_name` path) change from `renamed` to some another variable (returned as `Some`).
755    pub fn rename_will_conflict_with_another_variable(
756        &self,
757        db: &dyn DefDatabase,
758        current_name: &Name,
759        current_name_as_path: &ModPath,
760        mut hygiene_id: HygieneId,
761        new_name: &Symbol,
762        to_be_renamed: BindingId,
763    ) -> Option<BindingId> {
764        let mut hygiene_info = hygiene_info(db, hygiene_id);
765        let mut will_be_resolved_to = None;
766        for scope in self.scopes() {
767            match scope {
768                Scope::ExprScope(scope) => {
769                    for entry in scope.expr_scopes.entries(scope.scope_id) {
770                        if entry.hygiene() == hygiene_id {
771                            if entry.binding() == to_be_renamed {
772                                // This currently resolves to our renamed variable, now `will_be_resolved_to`
773                                // contains `Some` if the meaning will change or `None` if not.
774                                return will_be_resolved_to;
775                            } else if entry.name().symbol() == new_name {
776                                will_be_resolved_to = Some(entry.binding());
777                            }
778                        }
779                    }
780                }
781                Scope::MacroDefScope(macro_id) => {
782                    handle_macro_def_scope(db, &mut hygiene_id, &mut hygiene_info, macro_id)
783                }
784                Scope::GenericParams { params, def } => {
785                    if params.find_const_by_name(current_name, *def).is_some() {
786                        // It does not resolve to our renamed variable.
787                        return None;
788                    }
789                }
790                Scope::BlockScope(m) => {
791                    if m.resolve_path_in_value_ns(db, current_name_as_path).is_some() {
792                        // It does not resolve to our renamed variable.
793                        return None;
794                    }
795                }
796            }
797        }
798        // It does not resolve to our renamed variable.
799        None
800    }

802    /// Checks if we rename `renamed` to `name`, will the meaning of this reference (that contains `name` path) change
803    /// from some other variable (returned as `Some`) to `renamed`.
804    pub fn rename_will_conflict_with_renamed(
805        &self,
806        db: &dyn DefDatabase,
807        name: &Name,
808        name_as_path: &ModPath,
809        mut hygiene_id: HygieneId,
810        to_be_renamed: BindingId,
811    ) -> Option<BindingId> {
812        let mut hygiene_info = hygiene_info(db, hygiene_id);
813        let mut will_resolve_to_renamed = false;
814        for scope in self.scopes() {
815            match scope {
816                Scope::ExprScope(scope) => {
817                    for entry in scope.expr_scopes.entries(scope.scope_id) {
818                        if entry.binding() == to_be_renamed {
819                            will_resolve_to_renamed = true;
820                        } else if entry.hygiene() == hygiene_id && entry.name() == name {
821                            if will_resolve_to_renamed {
822                                // This will resolve to the renamed variable before it resolves to the original variable.
823                                return Some(entry.binding());
824                            } else {
825                                // This will resolve to the original variable.
826                                return None;
827                            }
828                        }
829                    }
830                }
831                Scope::MacroDefScope(macro_id) => {
832                    handle_macro_def_scope(db, &mut hygiene_id, &mut hygiene_info, macro_id)
833                }
834                Scope::GenericParams { params, def } => {
835                    if params.find_const_by_name(name, *def).is_some() {
836                        // Here and below, it might actually resolve to our renamed variable - in which case it'll
837                        // hide the generic parameter or some other thing (not a variable). We don't check for that
838                        // because due to naming conventions, it is rare that variable will shadow a non-variable.
839                        return None;
840                    }
841                }
842                Scope::BlockScope(m) => {
843                    if m.resolve_path_in_value_ns(db, name_as_path).is_some() {
844                        return None;
845                    }
846                }
847            }
848        }
849        None
850    }

852    /// `expr_id` is required to be an expression id that comes after the top level expression scope in the given resolver
853    #[must_use]
854    pub fn update_to_inner_scope(
855        &mut self,
856        db: &'db dyn DefDatabase,
857        owner: DefWithBodyId,
858        expr_id: ExprId,
859    ) -> UpdateGuard {
860        #[inline(always)]
861        fn append_expr_scope<'db>(
862            db: &'db dyn DefDatabase,
863            resolver: &mut Resolver<'db>,
864            owner: DefWithBodyId,
865            expr_scopes: &Arc<ExprScopes>,
866            scope_id: ScopeId,
867        ) {
868            if let Some(macro_id) = expr_scopes.macro_def(scope_id) {
869                resolver.scopes.push(Scope::MacroDefScope(**macro_id));
870            }
871            resolver.scopes.push(Scope::ExprScope(ExprScope {
872                owner,
873                expr_scopes: expr_scopes.clone(),
874                scope_id,
875            }));
876            if let Some(block) = expr_scopes.block(scope_id) {
877                let def_map = block_def_map(db, block);
878                let local_def_map = block.lookup(db).module.only_local_def_map(db);
879                resolver.scopes.push(Scope::BlockScope(ModuleItemMap {
880                    def_map,
881                    local_def_map,
882                    module_id: def_map.root,
883                }));
884                // FIXME: This adds as many module scopes as there are blocks, but resolving in each
885                // already traverses all parents, so this is O(n¬≤). I think we could only store the
886                // innermost module scope instead?
887            }
888        }
889
890        let start = self.scopes.len();
891        let innermost_scope = self.scopes().find(|scope| !matches!(scope, Scope::MacroDefScope(_)));
892        match innermost_scope {
893            Some(&Scope::ExprScope(ExprScope { scope_id, ref expr_scopes, owner })) => {
894                let expr_scopes = expr_scopes.clone();
895                let scope_chain = expr_scopes
896                    .scope_chain(expr_scopes.scope_for(expr_id))
897                    .take_while(|&it| it != scope_id);
898                for scope_id in scope_chain {
899                    append_expr_scope(db, self, owner, &expr_scopes, scope_id);
900                }
901            }
902            _ => {
903                let expr_scopes = db.expr_scopes(owner);
904                let scope_chain = expr_scopes.scope_chain(expr_scopes.scope_for(expr_id));
905
906                for scope_id in scope_chain {
907                    append_expr_scope(db, self, owner, &expr_scopes, scope_id);
908                }
909            }
910        }
911        self.scopes[start..].reverse();
912        UpdateGuard(start)
913    }
914
915    pub fn reset_to_guard(&mut self, UpdateGuard(start): UpdateGuard) {
916        self.scopes.truncate(start);
917    }
918}

920#[inline]
921fn handle_macro_def_scope(
922    db: &dyn DefDatabase,
923    hygiene_id: &mut HygieneId,
924    hygiene_info: &mut Option<(SyntaxContext, MacroDefId)>,
925    macro_id: &MacroDefId,
926) {
927    if let Some((parent_ctx, label_macro_id)) = hygiene_info
928        && label_macro_id == macro_id
929    {
930        // A macro is allowed to refer to variables from before its declaration.
931        // Therefore, if we got to the rib of its declaration, give up its hygiene
932        // and use its parent expansion.
933        *hygiene_id = HygieneId::new(parent_ctx.opaque_and_semiopaque(db));
934        *hygiene_info = parent_ctx.outer_expn(db).map(|expansion| {
935            let expansion = db.lookup_intern_macro_call(expansion.into());
936            (parent_ctx.parent(db), expansion.def)
937        });
938    }
939}
940
941#[inline]
942fn hygiene_info(
943    db: &dyn DefDatabase,
944    hygiene_id: HygieneId,
945) -> Option<(SyntaxContext, MacroDefId)> {
946    if !hygiene_id.is_root() {
947        let ctx = hygiene_id.syntax_context();
948        ctx.outer_expn(db).map(|expansion| {
949            let expansion = db.lookup_intern_macro_call(expansion.into());
950            (ctx.parent(db), expansion.def)
951        })
952    } else {
953        None
954    }
955}

1006impl<'db> Scope<'db> {
1007    fn process_names(&self, acc: &mut ScopeNames, db: &'db dyn DefDatabase) {
1008        match self {
1009            Scope::BlockScope(m) => {
1010                m.def_map[m.module_id].scope.entries().for_each(|(name, def)| {
1011                    acc.add_per_ns(name, def);
1012                });
1013                m.def_map[m.module_id].scope.legacy_macros().for_each(|(name, macs)| {
1014                    macs.iter().for_each(|&mac| {
1015                        acc.add(name, ScopeDef::ModuleDef(ModuleDefId::MacroId(mac)));
1016                    })
1017                });
1018            }
1019            &Scope::GenericParams { ref params, def: parent } => {
1020                if let GenericDefId::ImplId(impl_) = parent {
1021                    acc.add(&Name::new_symbol_root(sym::Self_), ScopeDef::ImplSelfType(impl_));
1022                } else if let GenericDefId::AdtId(adt) = parent {
1023                    acc.add(&Name::new_symbol_root(sym::Self_), ScopeDef::AdtSelfType(adt));
1024                }
1025
1026                for (local_id, param) in params.iter_type_or_consts() {
1027                    if let Some(name) = &param.name() {
1028                        let id = TypeOrConstParamId { parent, local_id };
1029                        let data = &db.generic_params(parent)[local_id];
1030                        acc.add(
1031                            name,
1032                            ScopeDef::GenericParam(match data {
1033                                TypeOrConstParamData::TypeParamData(_) => {
1034                                    GenericParamId::TypeParamId(TypeParamId::from_unchecked(id))
1035                                }
1036                                TypeOrConstParamData::ConstParamData(_) => {
1037                                    GenericParamId::ConstParamId(ConstParamId::from_unchecked(id))
1038                                }
1039                            }),
1040                        );
1041                    }
1042                }
1043                for (local_id, param) in params.iter_lt() {
1044                    let id = LifetimeParamId { parent, local_id };
1045                    acc.add(&param.name, ScopeDef::GenericParam(id.into()))
1046                }
1047            }
1048            Scope::ExprScope(scope) => {
1049                if let Some((label, name)) = scope.expr_scopes.label(scope.scope_id) {
1050                    acc.add(&name, ScopeDef::Label(label))
1051                }
1052                scope.expr_scopes.entries(scope.scope_id).iter().for_each(|e| {
1053                    acc.add_local(e.name(), e.binding());
1054                });
1055            }
1056            Scope::MacroDefScope(_) => {}
1057        }
1058    }
1059}

1061pub fn resolver_for_scope(
1062    db: &dyn DefDatabase,
1063    owner: DefWithBodyId,
1064    scope_id: Option<ScopeId>,
1065) -> Resolver<'_> {
1066    let r = owner.resolver(db);
1067    let scopes = db.expr_scopes(owner);
1068    resolver_for_scope_(db, scopes, scope_id, r, owner)
1069}
1070
1071fn resolver_for_scope_<'db>(
1072    db: &'db dyn DefDatabase,
1073    scopes: Arc<ExprScopes>,
1074    scope_id: Option<ScopeId>,
1075    mut r: Resolver<'db>,
1076    owner: DefWithBodyId,
1077) -> Resolver<'db> {
1078    let scope_chain = scopes.scope_chain(scope_id).collect::<Vec<_>>();
1079    r.scopes.reserve(scope_chain.len());
1080
1081    for scope in scope_chain.into_iter().rev() {
1082        if let Some(block) = scopes.block(scope) {
1083            let def_map = block_def_map(db, block);
1084            let local_def_map = block.lookup(db).module.only_local_def_map(db);
1085            // Using `DefMap::ROOT` is okay here since inside modules other than the root,
1086            // there can't directly be expressions.
1087            r = r.push_block_scope(def_map, local_def_map, def_map.root);
1088            // FIXME: This adds as many module scopes as there are blocks, but resolving in each
1089            // already traverses all parents, so this is O(n¬≤). I think we could only store the
1090            // innermost module scope instead?
1091        }
1092        if let Some(macro_id) = scopes.macro_def(scope) {
1093            r = r.push_scope(Scope::MacroDefScope(**macro_id));
1094        }
1095
1096        r = r.push_expr_scope(owner, Arc::clone(&scopes), scope);
1097    }
1098    r
1099}

1135impl<'db> ModuleItemMap<'db> {
1136    fn resolve_path_in_value_ns(
1137        &self,
1138        db: &'db dyn DefDatabase,
1139        path: &ModPath,
1140    ) -> Option<(ResolveValueResult, ResolvePathResultPrefixInfo)> {
1141        let (module_def, unresolved_idx, prefix_info) = self.def_map.resolve_path_locally(
1142            self.local_def_map,
1143            db,
1144            self.module_id,
1145            path,
1146            BuiltinShadowMode::Other,
1147        );
1148        match unresolved_idx {
1149            None => {
1150                let value = to_value_ns(module_def, self.def_map)?;
1151                Some((ResolveValueResult::ValueNs(value), prefix_info))
1152            }
1153            Some(unresolved_idx) => {
1154                let def = module_def.take_types()?;
1155                let ty = match def {
1156                    ModuleDefId::AdtId(it) => TypeNs::AdtId(it),
1157                    ModuleDefId::TraitId(it) => TypeNs::TraitId(it),
1158                    ModuleDefId::TypeAliasId(it) => TypeNs::TypeAliasId(it),
1159                    ModuleDefId::BuiltinType(it) => TypeNs::BuiltinType(it),
1160
1161                    ModuleDefId::ModuleId(_)
1162                    | ModuleDefId::FunctionId(_)
1163                    | ModuleDefId::EnumVariantId(_)
1164                    | ModuleDefId::ConstId(_)
1165                    | ModuleDefId::MacroId(_)
1166                    | ModuleDefId::StaticId(_) => return None,
1167                };
1168                Some((ResolveValueResult::Partial(ty, unresolved_idx), prefix_info))
1169            }
1170        }
1171    }
1172
1173    fn resolve_path_in_type_ns(
1174        &self,
1175        db: &dyn DefDatabase,
1176        path: &ModPath,
1177    ) -> Option<(TypeNs, Option<usize>, Option<ImportOrExternCrate>, ResolvePathResultPrefixInfo)>
1178    {
1179        let (module_def, idx, prefix_info) = self.def_map.resolve_path_locally(
1180            self.local_def_map,
1181            db,
1182            self.module_id,
1183            path,
1184            BuiltinShadowMode::Other,
1185        );
1186        let (res, import) = to_type_ns(module_def)?;
1187        Some((res, idx, import, prefix_info))
1188    }
1189}

1191fn to_value_ns(per_ns: PerNs, def_map: &DefMap) -> Option<ValueNs> {
1192    let def = per_ns.take_values().or_else(|| {
1193        let Some(MacroId::ProcMacroId(proc_macro)) = per_ns.take_macros() else { return None };
1194        // If we cannot resolve to value ns, but we can resolve to a proc macro, and this is the crate
1195        // defining this proc macro - inside this crate, we should treat the macro as a function.
1196        def_map.proc_macro_as_fn(proc_macro).map(ModuleDefId::FunctionId)
1197    })?;
1198    let res = match def {
1199        ModuleDefId::FunctionId(it) => ValueNs::FunctionId(it),
1200        ModuleDefId::AdtId(AdtId::StructId(it)) => ValueNs::StructId(it),
1201        ModuleDefId::EnumVariantId(it) => ValueNs::EnumVariantId(it),
1202        ModuleDefId::ConstId(it) => ValueNs::ConstId(it),
1203        ModuleDefId::StaticId(it) => ValueNs::StaticId(it),
1204
1205        ModuleDefId::AdtId(AdtId::EnumId(_) | AdtId::UnionId(_))
1206        | ModuleDefId::TraitId(_)
1207        | ModuleDefId::TypeAliasId(_)
1208        | ModuleDefId::BuiltinType(_)
1209        | ModuleDefId::MacroId(_)
1210        | ModuleDefId::ModuleId(_) => return None,
1211    };
1212    Some(res)
1213}
1214
1215fn to_type_ns(per_ns: PerNs) -> Option<(TypeNs, Option<ImportOrExternCrate>)> {
1216    let def = per_ns.take_types_full()?;
1217    let res = match def.def {
1218        ModuleDefId::AdtId(it) => TypeNs::AdtId(it),
1219        ModuleDefId::EnumVariantId(it) => TypeNs::EnumVariantId(it),
1220
1221        ModuleDefId::TypeAliasId(it) => TypeNs::TypeAliasId(it),
1222        ModuleDefId::BuiltinType(it) => TypeNs::BuiltinType(it),
1223
1224        ModuleDefId::TraitId(it) => TypeNs::TraitId(it),
1225
1226        ModuleDefId::ModuleId(it) => TypeNs::ModuleId(it),
1227
1228        ModuleDefId::FunctionId(_)
1229        | ModuleDefId::ConstId(_)
1230        | ModuleDefId::MacroId(_)
1231        | ModuleDefId::StaticId(_) => return None,
1232    };
1233    Some((res, def.import))
1234}

1236#[derive(Default)]
1237struct ScopeNames {
1238    map: FxIndexMap<Name, SmallVec<[ScopeDef; 1]>>,
1239}
1240
1241impl ScopeNames {
1242    fn add(&mut self, name: &Name, def: ScopeDef) {
1243        let set = self.map.entry(name.clone()).or_default();
1244        if !set.contains(&def) {
1245            set.push(def)
1246        }
1247    }
1248    fn add_per_ns(&mut self, name: &Name, def: PerNs) {
1249        if let Some(ty) = &def.types {
1250            self.add(name, ScopeDef::ModuleDef(ty.def))
1251        }
1252        if let Some(def) = &def.values {
1253            self.add(name, ScopeDef::ModuleDef(def.def))
1254        }
1255        if let Some(mac) = &def.macros {
1256            self.add(name, ScopeDef::ModuleDef(ModuleDefId::MacroId(mac.def)))
1257        }
1258        if def.is_none() {
1259            self.add(name, ScopeDef::Unknown)
1260        }
1261    }
1262    fn add_local(&mut self, name: &Name, binding: BindingId) {
1263        let set = self.map.entry(name.clone()).or_default();
1264        // XXX: hack, account for local (and only local) shadowing.
1265        //
1266        // This should be somewhat more principled and take namespaces into
1267        // accounts, but, alas, scoping rules are a hoax. `str` type and `str`
1268        // module can be both available in the same scope.
1269        if set.iter().any(|it| matches!(it, &ScopeDef::Local(_))) {
1270            cov_mark::hit!(shadowing_shows_single_completion);
1271            return;
1272        }
1273        set.push(ScopeDef::Local(binding))
1274    }
1275}

1277pub trait HasResolver: Copy {
1278    /// Builds a resolver for type references inside this def.
1279    fn resolver(self, db: &dyn DefDatabase) -> Resolver<'_>;
1280}
1281
1282impl HasResolver for ModuleId {
1283    fn resolver(self, db: &dyn DefDatabase) -> Resolver<'_> {
1284        let (mut def_map, local_def_map) = self.local_def_map(db);
1285        let mut module_id = self;
1286
1287        if self.block(db).is_none() {
1288            return Resolver {
1289                scopes: vec![],
1290                module_scope: ModuleItemMap { def_map, local_def_map, module_id },
1291            };
1292        }
1293
1294        let mut modules: SmallVec<[_; 1]> = smallvec![];
1295        while let Some(parent) = def_map.parent() {
1296            let block_def_map = mem::replace(&mut def_map, parent.def_map(db));
1297            let block_module_id = mem::replace(&mut module_id, parent);
1298            modules.push((block_def_map, block_module_id));
1299            if parent.block(db).is_none() {
1300                break;
1301            }
1302        }
1303        let mut resolver = Resolver {
1304            scopes: Vec::with_capacity(modules.len()),
1305            module_scope: ModuleItemMap { def_map, local_def_map, module_id },
1306        };
1307        for (def_map, module_id) in modules.into_iter().rev() {
1308            resolver = resolver.push_block_scope(def_map, local_def_map, module_id);
1309        }
1310        resolver
1311    }
1312}
