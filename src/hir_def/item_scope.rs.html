35pub enum ImportOrExternCrate {
36    Glob(GlobId),
37    Import(ImportId),
38    ExternCrate(ExternCrateId),

41impl From<ImportOrGlob> for ImportOrExternCrate {
42    fn from(value: ImportOrGlob) -> Self {
43        match value {
44            ImportOrGlob::Glob(it) => ImportOrExternCrate::Glob(it),
45            ImportOrGlob::Import(it) => ImportOrExternCrate::Import(it),
46        }
47    }

50impl ImportOrExternCrate {

52        match self {
53            ImportOrExternCrate::Import(it) => Some(ImportOrGlob::Import(it)),
54            ImportOrExternCrate::Glob(it) => Some(ImportOrGlob::Glob(it)),
55            _ => None,
56        }
57    }
58
59    pub fn import(self) -> Option<ImportId> {
60        match self {
61            ImportOrExternCrate::Import(it) => Some(it),
62            _ => None,
63        }
64    }

66    pub fn glob(self) -> Option<GlobId> {
67        match self {
68            ImportOrExternCrate::Glob(id) => Some(id),
69            _ => None,
70        }
71    }

73    pub fn use_(self) -> Option<UseId> {
74        match self {
75            ImportOrExternCrate::Glob(id) => Some(id.use_),
76            ImportOrExternCrate::Import(id) => Some(id.use_),
77            _ => None,
78        }

148#[derive(Debug, Default, PartialEq, Eq)]
149pub struct ItemScope {
150    /// Defs visible in this scope. This includes `declarations`, but also
151    /// imports. The imports belong to this module and can be resolved by using them on
152    /// the `use_imports_*` fields.
153    types: FxIndexMap<Name, TypesItem>,
154    values: FxIndexMap<Name, ValuesItem>,
155    macros: FxIndexMap<Name, MacrosItem>,
156    unresolved: FxHashSet<Name>,
157
158    /// The defs declared in this scope. Each def has a single scope where it is
159    /// declared.
160    declarations: ThinVec<ModuleDefId>,
161
162    impls: ThinVec<(ImplId, /* trait impl */ bool)>,
163    builtin_derive_impls: ThinVec<BuiltinDeriveImplId>,
164    extern_blocks: ThinVec<ExternBlockId>,
165    unnamed_consts: ThinVec<ConstId>,
166    /// Traits imported via `use Trait as _;`.
167    unnamed_trait_imports: ThinVec<(TraitId, Item<()>)>,
168
169    // the resolutions of the imports of this scope
170    use_imports_types: FxHashMap<ImportOrExternCrate, ImportOrDef>,
171    use_imports_values: FxHashMap<ImportOrGlob, ImportOrDef>,
172    use_imports_macros: FxHashMap<ImportOrExternCrate, ImportOrDef>,
173
174    use_decls: ThinVec<UseId>,
175    extern_crate_decls: ThinVec<ExternCrateId>,
176    /// Macros visible in current module in legacy textual scope
177    ///
178    /// For macros invoked by an unqualified identifier like `bar!()`, `legacy_macros` will be searched in first.
179    /// If it yields no result, then it turns to module scoped `macros`.
180    /// It macros with name qualified with a path like `crate::foo::bar!()`, `legacy_macros` will be skipped,
181    /// and only normal scoped `macros` will be searched in.
182    ///
183    /// Note that this automatically inherit macros defined textually before the definition of module itself.
184    ///
185    /// Module scoped macros will be inserted into `items` instead of here.
186    // FIXME: Macro shadowing in one module is not properly handled. Non-item place macros will
187    // be all resolved to the last one defined if shadowing happens.
188    legacy_macros: FxHashMap<Name, SmallVec<[MacroId; 2]>>,
189    /// The attribute macro invocations in this scope.
190    attr_macros: FxHashMap<AstId<ast::Item>, MacroCallId>,
191    /// The macro invocations in this scope.
192    macro_invocations: FxHashMap<AstId<ast::MacroCall>, MacroCallId>,
193    /// The derive macro invocations in this scope, keyed by the owner item over the actual derive attributes
194    /// paired with the derive macro invocations for the specific attribute.
195    derive_macros: FxHashMap<AstId<ast::Adt>, SmallVec<[DeriveMacroInvocation; 1]>>,
196}

252        self.use_imports_types

263    pub fn fully_resolve_import(&self, db: &dyn DefDatabase, mut import: ImportId) -> PerNs {
264        let mut res = PerNs::none();
265
266        let mut scope = self;
267        while let Some(&m) = scope.use_imports_macros.get(&ImportOrExternCrate::Import(import)) {
268            match m {
269                ImportOrDef::Import(i) => {
270                    let module_id = i.use_.lookup(db).container;
271                    scope = &module_id.def_map(db)[module_id].scope;
272                    import = i;
273                }
274                ImportOrDef::Def(ModuleDefId::MacroId(def)) => {
275                    res.macros = Some(Item { def, vis: Visibility::Public, import: None });
276                    break;
277                }
278                _ => break,
279            }
280        }
281        let mut scope = self;
282        while let Some(&m) = scope.use_imports_types.get(&ImportOrExternCrate::Import(import)) {
283            match m {
284                ImportOrDef::Import(i) => {
285                    let module_id = i.use_.lookup(db).container;
286                    scope = &module_id.def_map(db)[module_id].scope;
287                    import = i;
288                }
289                ImportOrDef::Def(def) => {
290                    res.types = Some(Item { def, vis: Visibility::Public, import: None });
291                    break;
292                }
293                _ => break,
294            }
295        }
296        let mut scope = self;
297        while let Some(&m) = scope.use_imports_values.get(&ImportOrGlob::Import(import)) {
298            match m {
299                ImportOrDef::Import(i) => {
300                    let module_id = i.use_.lookup(db).container;
301                    scope = &module_id.def_map(db)[module_id].scope;
302                    import = i;
303                }
304                ImportOrDef::Def(def) => {
305                    res.values = Some(Item { def, vis: Visibility::Public, import: None });
306                    break;
307                }
308                _ => break,
309            }
310        }
311        res
312    }

390    pub(crate) fn name_of(&self, item: ItemInNs) -> Option<(&Name, Visibility, /*declared*/ bool)> {
391        match item {
392            ItemInNs::Macros(def) => self.macros.iter().find_map(|(name, other_def)| {
393                (other_def.def == def).then_some((name, other_def.vis, other_def.import.is_none()))
394            }),
395            ItemInNs::Types(def) => self.types.iter().find_map(|(name, other_def)| {
396                (other_def.def == def).then_some((name, other_def.vis, other_def.import.is_none()))
397            }),
398            ItemInNs::Values(def) => self.values.iter().find_map(|(name, other_def)| {
399                (other_def.def == def).then_some((name, other_def.vis, other_def.import.is_none()))
400            }),
401        }
402    }

404    /// XXX: this is O(N) rather than O(1), try to not introduce new usages.
405    pub(crate) fn names_of<T>(
406        &self,
407        item: ItemInNs,
408        mut cb: impl FnMut(&Name, Visibility, /*declared*/ bool) -> Option<T>,
409    ) -> Option<T> {
410        match item {
411            ItemInNs::Macros(def) => self
412                .macros
413                .iter()
414                .filter_map(|(name, other_def)| {
415                    (other_def.def == def).then_some((
416                        name,
417                        other_def.vis,
418                        other_def.import.is_none(),
419                    ))
420                })
421                .find_map(|(a, b, c)| cb(a, b, c)),
422            ItemInNs::Types(def) => self
423                .types
424                .iter()
425                .filter_map(|(name, other_def)| {
426                    (other_def.def == def).then_some((
427                        name,
428                        other_def.vis,
429                        other_def.import.is_none(),
430                    ))
431                })
432                .find_map(|(a, b, c)| cb(a, b, c)),
433            ItemInNs::Values(def) => self
434                .values
435                .iter()
436                .filter_map(|(name, other_def)| {
437                    (other_def.def == def).then_some((
438                        name,
439                        other_def.vis,
440                        other_def.import.is_none(),
441                    ))
442                })
443                .find_map(|(a, b, c)| cb(a, b, c)),
444        }
445    }

546    /// We are required to set this up front as derive invocation recording happens out of order
547    /// due to the fixed pointer iteration loop being able to record some derives later than others
548    /// independent of their indices.
549    pub(crate) fn init_derive_attribute(
550        &mut self,
551        adt: AstId<ast::Adt>,
552        attr_id: AttrId,
553        attr_call_id: MacroCallId,
554        mut derive_call_ids: SmallVec<[Option<Either<MacroCallId, BuiltinDeriveImplId>>; 4]>,
555    ) {
556        derive_call_ids.shrink_to_fit();
557        self.derive_macros.entry(adt).or_default().push(DeriveMacroInvocation {
558            attr_id,
559            attr_call_id,
560            derive_call_ids,
561        });
562    }

601        import: Option<ImportId>,
602    ) {
603        self.unnamed_trait_imports.push((tr, Item { def: (), vis, import }));
604    }

606    pub(crate) fn push_res_with_import(
607        &mut self,
608        glob_imports: &mut PerNsGlobImports,
609        lookup: (ModuleId, Name),
610        def: PerNs,
611        import: Option<ImportOrExternCrate>,
612    ) -> bool {
613        let mut changed = false;
614
615        // FIXME: Document and simplify this
616
617        if let Some(mut fld) = def.types {
618            let existing = self.types.entry(lookup.1.clone());
619            match existing {
620                Entry::Vacant(entry) => {
621                    match import {
622                        Some(ImportOrExternCrate::Glob(_)) => {
623                            glob_imports.types.insert(lookup.clone());
624                        }
625                        _ => _ = glob_imports.types.remove(&lookup),
626                    }
627                    let prev = std::mem::replace(&mut fld.import, import);
628                    if let Some(import) = import {
629                        self.use_imports_types
630                            .insert(import, prev.map_or(ImportOrDef::Def(fld.def), Into::into));
631                    }
632                    entry.insert(fld);
633                    changed = true;
634                }
635                Entry::Occupied(mut entry) => {
636                    match import {
637                        Some(ImportOrExternCrate::Glob(..)) => {
638                            // Multiple globs may import the same item and they may
639                            // override visibility from previously resolved globs. This is
640                            // currently handled by `DefCollector`, because we need to
641                            // compute the max visibility for items and we need `DefMap`
642                            // for that.
643                        }
644                        _ => {
645                            if glob_imports.types.remove(&lookup) {
646                                let prev = std::mem::replace(&mut fld.import, import);
647                                if let Some(import) = import {
648                                    self.use_imports_types.insert(
649                                        import,
650                                        prev.map_or(ImportOrDef::Def(fld.def), Into::into),
651                                    );
652                                }
653                                cov_mark::hit!(import_shadowed);
654                                entry.insert(fld);
655                                changed = true;
656                            }
657                        }
658                    }
659                }
660            }
661        }
662
663        if let Some(mut fld) = def.values {
664            let existing = self.values.entry(lookup.1.clone());
665            match existing {
666                Entry::Vacant(entry) => {
667                    match import {
668                        Some(ImportOrExternCrate::Glob(_)) => {
669                            glob_imports.values.insert(lookup.clone());
670                        }
671                        _ => _ = glob_imports.values.remove(&lookup),
672                    }
673                    let import = import.and_then(ImportOrExternCrate::import_or_glob);
674                    let prev = std::mem::replace(&mut fld.import, import);
675                    if let Some(import) = import {
676                        self.use_imports_values
677                            .insert(import, prev.map_or(ImportOrDef::Def(fld.def), Into::into));
678                    }
679                    entry.insert(fld);
680                    changed = true;
681                }
682                Entry::Occupied(mut entry)
683                    if !matches!(import, Some(ImportOrExternCrate::Glob(..))) =>
684                {
685                    if glob_imports.values.remove(&lookup) {
686                        cov_mark::hit!(import_shadowed);
687
688                        let import = import.and_then(ImportOrExternCrate::import_or_glob);
689                        let prev = std::mem::replace(&mut fld.import, import);
690                        if let Some(import) = import {
691                            self.use_imports_values
692                                .insert(import, prev.map_or(ImportOrDef::Def(fld.def), Into::into));
693                        }
694                        entry.insert(fld);
695                        changed = true;
696                    }
697                }
698                _ => {}
699            }
700        }
701
702        if let Some(mut fld) = def.macros {
703            let existing = self.macros.entry(lookup.1.clone());
704            match existing {
705                Entry::Vacant(entry) => {
706                    match import {
707                        Some(ImportOrExternCrate::Glob(_)) => {
708                            glob_imports.macros.insert(lookup.clone());
709                        }
710                        _ => _ = glob_imports.macros.remove(&lookup),
711                    }
712                    let prev = std::mem::replace(&mut fld.import, import);
713                    if let Some(import) = import {
714                        self.use_imports_macros.insert(
715                            import,
716                            prev.map_or_else(|| ImportOrDef::Def(fld.def.into()), Into::into),
717                        );
718                    }
719                    entry.insert(fld);
720                    changed = true;
721                }
722                Entry::Occupied(mut entry)
723                    if !matches!(import, Some(ImportOrExternCrate::Glob(..))) =>
724                {
725                    if glob_imports.macros.remove(&lookup) {
726                        cov_mark::hit!(import_shadowed);
727                        let prev = std::mem::replace(&mut fld.import, import);
728                        if let Some(import) = import {
729                            self.use_imports_macros.insert(
730                                import,
731                                prev.map_or_else(|| ImportOrDef::Def(fld.def.into()), Into::into),
732                            );
733                        }
734                        entry.insert(fld);
735                        changed = true;
736                    }
737                }
738                _ => {}
739            }
740        }
741
742        if def.is_none() && self.unresolved.insert(lookup.1) {
743            changed = true;
744        }
745
746        changed
747    }

749    /// Marks everything that is not a procedural macro as private to `this_module`.
750    pub(crate) fn censor_non_proc_macros(&mut self, krate: Crate) {
751        self.types
752            .values_mut()
753            .map(|def| &mut def.vis)
754            .chain(self.values.values_mut().map(|def| &mut def.vis))
755            .chain(self.unnamed_trait_imports.iter_mut().map(|(_, def)| &mut def.vis))
756            .for_each(|vis| *vis = Visibility::PubCrate(krate));
757
758        for mac in self.macros.values_mut() {
759            if matches!(mac.def, MacroId::ProcMacroId(_) if mac.import.is_none()) {
760                continue;
761            }
762            mac.vis = Visibility::PubCrate(krate)
763        }
764    }

766    pub(crate) fn dump(&self, db: &dyn DefDatabase, buf: &mut String) {
767        let mut entries: Vec<_> = self.resolutions().collect();
768        entries.sort_by_key(|(name, _)| name.clone());
769
770        let print_macro_sub_ns = |buf: &mut String, macro_id: MacroId| {
771            let styles = crate::nameres::macro_styles_from_id(db, macro_id);
772            if styles.contains(MacroCallStyles::FN_LIKE) {
773                buf.push('!');
774            }
775            if styles.contains(MacroCallStyles::ATTR) || styles.contains(MacroCallStyles::DERIVE) {
776                buf.push('#');
777            }
778        };
779
780        for (name, def) in entries {
781            let display_name: &dyn fmt::Display = match &name {
782                Some(name) => &name.display(db, Edition::LATEST),
783                None => &"_",
784            };
785            format_to!(buf, "- {display_name} :");
786
787            if let Some(Item { import, .. }) = def.types {
788                buf.push_str(" type");
789                match import {
790                    Some(ImportOrExternCrate::Import(_)) => buf.push_str(" (import)"),
791                    Some(ImportOrExternCrate::Glob(_)) => buf.push_str(" (glob)"),
792                    Some(ImportOrExternCrate::ExternCrate(_)) => buf.push_str(" (extern)"),
793                    None => (),
794                }
795            }
796            if let Some(Item { import, .. }) = def.values {
797                buf.push_str(" value");
798                match import {
799                    Some(ImportOrGlob::Import(_)) => buf.push_str(" (import)"),
800                    Some(ImportOrGlob::Glob(_)) => buf.push_str(" (glob)"),
801                    None => (),
802                }
803            }
804            if let Some(Item { def: macro_id, import, .. }) = def.macros {
805                buf.push_str(" macro");
806                print_macro_sub_ns(buf, macro_id);
807                match import {
808                    Some(ImportOrExternCrate::Import(_)) => buf.push_str(" (import)"),
809                    Some(ImportOrExternCrate::Glob(_)) => buf.push_str(" (glob)"),
810                    Some(ImportOrExternCrate::ExternCrate(_)) => buf.push_str(" (extern)"),
811                    None => (),
812                }
813            }
814            if def.is_none() {
815                buf.push_str(" _");
816            }
817
818            buf.push('\n');
819        }
820
821        // Also dump legacy-textual-scope macros visible at the _end_ of the scope.
822        //
823        // For tests involving a cursor position, this might include macros that
824        // are _not_ visible at the cursor position.
825        let mut legacy_macros = self.legacy_macros().collect::<Vec<_>>();
826        legacy_macros.sort_by(|(a, _), (b, _)| Ord::cmp(a, b));
827        for (name, macros) in legacy_macros {
828            format_to!(buf, "- (legacy) {} :", name.display(db, Edition::LATEST));
829            for &macro_id in macros {
830                buf.push_str(" macro");
831                print_macro_sub_ns(buf, macro_id);
832            }
833            buf.push('\n');
834        }
835    }

921impl PerNs {
922    pub(crate) fn from_def(
923        def: ModuleDefId,
924        v: Visibility,
925        has_constructor: bool,
926        import: Option<ImportOrExternCrate>,
927    ) -> PerNs {
928        match def {
929            ModuleDefId::ModuleId(_) => PerNs::types(def, v, import),
930            ModuleDefId::FunctionId(_) => {
931                PerNs::values(def, v, import.and_then(ImportOrExternCrate::import_or_glob))
932            }
933            ModuleDefId::AdtId(adt) => match adt {
934                AdtId::UnionId(_) => PerNs::types(def, v, import),
935                AdtId::EnumId(_) => PerNs::types(def, v, import),
936                AdtId::StructId(_) => {
937                    if has_constructor {
938                        PerNs::both(def, def, v, import)
939                    } else {
940                        PerNs::types(def, v, import)
941                    }
942                }
943            },
944            ModuleDefId::EnumVariantId(_) => PerNs::both(def, def, v, import),
945            ModuleDefId::ConstId(_) | ModuleDefId::StaticId(_) => {
946                PerNs::values(def, v, import.and_then(ImportOrExternCrate::import_or_glob))
947            }
948            ModuleDefId::TraitId(_) => PerNs::types(def, v, import),
949            ModuleDefId::TypeAliasId(_) => PerNs::types(def, v, import),
950            ModuleDefId::BuiltinType(_) => PerNs::types(def, v, import),
951            ModuleDefId::MacroId(mac) => PerNs::macros(mac, v, import),
952        }
953    }
954}

956#[derive(Clone, Copy, PartialEq, Eq, Debug, Hash)]
957pub enum ItemInNs {
958    Types(ModuleDefId),
959    Values(ModuleDefId),
960    Macros(MacroId),
961}
962
963impl ItemInNs {
964    pub fn as_module_def_id(self) -> Option<ModuleDefId> {
965        match self {
966            ItemInNs::Types(id) | ItemInNs::Values(id) => Some(id),
967            ItemInNs::Macros(_) => None,
968        }
969    }
970
971    /// Returns the crate defining this item (or `None` if `self` is built-in).
972    pub fn krate(&self, db: &dyn DefDatabase) -> Option<Crate> {
973        self.module(db).map(|module_id| module_id.krate(db))
974    }
975
976    pub fn module(&self, db: &dyn DefDatabase) -> Option<ModuleId> {
977        match self {
978            ItemInNs::Types(id) | ItemInNs::Values(id) => id.module(db),
979            ItemInNs::Macros(id) => Some(id.module(db)),
980        }
981    }
