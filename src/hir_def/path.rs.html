25#[derive(Debug, Clone, PartialEq, Eq)]
26pub enum ImportAlias {
27    /// Unnamed alias, as in `use Foo as _;`
28    Underscore,
29    /// Named alias
30    Alias(Name),
31}
32
33impl ImportAlias {
34    pub fn display(&self, edition: Edition) -> impl Display + '_ {
35        ImportAliasDisplay { value: self, edition }
36    }
37}
38
39struct ImportAliasDisplay<'a> {
40    value: &'a ImportAlias,
41    edition: Edition,
42}
43impl Display for ImportAliasDisplay<'_> {
44    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
45        match self.value {
46            ImportAlias::Underscore => f.write_str("_"),
47            ImportAlias::Alias(name) => Display::fmt(&name.display_no_db(self.edition), f),
48        }
49    }
50}

52#[derive(Debug, Clone, PartialEq, Eq, Hash)]
53pub enum Path {
54    /// `BarePath` is used when the path has neither generics nor type anchor, since the vast majority of paths
55    /// are in this category, and splitting `Path` this way allows it to be more thin. When the path has either generics
56    /// or type anchor, it is `Path::Normal` with the generics filled with `None` even if there are none (practically
57    /// this is not a problem since many more paths have generics than a type anchor).
58    BarePath(Interned<ModPath>),
59    /// `Path::Normal` will always have either generics or type anchor.
60    Normal(Box<NormalPath>),
61    /// A link to a lang item. It is used in desugaring of things like `it?`. We can show these
62    /// links via a normal path since they might be private and not accessible in the usage place.
63    LangItem(LangItemTarget, Option<Name>),
64}
65
66// This type is being used a lot, make sure it doesn't grow unintentionally.
67#[cfg(target_arch = "x86_64")]
68const _: () = {
69    assert!(size_of::<Path>() == 16);
70    assert!(size_of::<Option<Path>>() == 16);
71};

77    pub mod_path: Interned<ModPath>,

80#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
81pub enum GenericArgsParentheses {
82    No,
83    /// Bounds of the form `Type::method(..): Send` or `impl Trait<method(..): Send>`,
84    /// aka. Return Type Notation or RTN.
85    ReturnTypeNotation,
86    /// `Fn`-family parenthesized traits, e.g. `impl Fn(u32) -> String`.
87    ///
88    /// This is desugared into one generic argument containing a tuple of all arguments,
89    /// and an associated type binding for `Output` for the return type.
90    ParenSugar,
91}
92
93/// Generic arguments to a path segment (e.g. the `i32` in `Option<i32>`). This
94/// also includes bindings of associated types, like in `Iterator<Item = Foo>`.
95#[derive(Debug, Clone, PartialEq, Eq, Hash)]
96pub struct GenericArgs {
97    pub args: Box<[GenericArg]>,
98    /// This specifies whether the args contain a Self type as the first
99    /// element. This is the case for path segments like `<T as Trait>`, where
100    /// `T` is actually a type parameter for the path `Trait` specifying the
101    /// Self type. Otherwise, when we have a path `Trait<X, Y>`, the Self type
102    /// is left out.
103    pub has_self_type: bool,
104    /// Associated type bindings like in `Iterator<Item = T>`.
105    pub bindings: Box<[AssociatedTypeBinding]>,
106    /// Whether these generic args were written with parentheses and how.
107    pub parenthesized: GenericArgsParentheses,
108}
109
110/// An associated type binding like in `Iterator<Item = T>`.
111#[derive(Debug, Clone, PartialEq, Eq, Hash)]
112pub struct AssociatedTypeBinding {
113    /// The name of the associated type.
114    pub name: Name,
115    /// The generic arguments to the associated type. e.g. For `Trait<Assoc<'a, T> = &'a T>`, this
116    /// would be `['a, T]`.
117    pub args: Option<GenericArgs>,
118    /// The type bound to this associated type (in `Item = T`, this would be the
119    /// `T`). This can be `None` if there are bounds instead.
120    pub type_ref: Option<TypeRefId>,
121    /// Bounds for the associated type, like in `Iterator<Item:
122    /// SomeOtherTrait>`. (This is the unstable `associated_type_bounds`
123    /// feature.)
124    pub bounds: Box<[TypeBound]>,
125}

127/// A single generic argument.
128#[derive(Debug, Clone, PartialEq, Eq, Hash)]
129pub enum GenericArg {
130    Type(TypeRefId),
131    Lifetime(LifetimeRef),
132    Const(ConstRef),

135impl Path {
136    /// Converts an `ast::Path` to `Path`. Works with use trees.
137    /// It correctly handles `$crate` based path from macro call.
138    pub fn from_src(ctx: &mut LowerCtx<'_>, path: ast::Path) -> Option<Path> {
139        lower::lower_path(ctx, path)
140    }
141
142    /// Converts a known mod path to `Path`.
143    pub fn from_known_path(path: ModPath, generic_args: Vec<Option<GenericArgs>>) -> Path {
144        Path::Normal(Box::new(NormalPath {
145            generic_args: generic_args.into_boxed_slice(),
146            type_anchor: None,
147            mod_path: Interned::new(path),
148        }))
149    }
150
151    /// Converts a known mod path to `Path`.
152    pub fn from_known_path_with_no_generic(path: ModPath) -> Path {
153        Path::BarePath(Interned::new(path))
154    }
155
156    #[inline]
157    pub fn kind(&self) -> &PathKind {
158        match self {
159            Path::BarePath(mod_path) => &mod_path.kind,
160            Path::Normal(path) => &path.mod_path.kind,
161            Path::LangItem(..) => &PathKind::Abs,
162        }
163    }
164
165    #[inline]
166    pub fn type_anchor(&self) -> Option<TypeRefId> {
167        match self {
168            Path::Normal(path) => path.type_anchor,
169            Path::LangItem(..) | Path::BarePath(_) => None,
170        }
171    }
172
173    #[inline]
174    pub fn generic_args(&self) -> Option<&[Option<GenericArgs>]> {
175        match self {
176            Path::Normal(path) => Some(&path.generic_args),
177            Path::LangItem(..) | Path::BarePath(_) => None,
178        }
179    }
180
181    pub fn segments(&self) -> PathSegments<'_> {
182        match self {
183            Path::BarePath(mod_path) => {
184                PathSegments { segments: mod_path.segments(), generic_args: None }
185            }
186            Path::Normal(path) => PathSegments {
187                segments: path.mod_path.segments(),
188                generic_args: Some(&path.generic_args),
189            },
190            Path::LangItem(_, seg) => PathSegments { segments: seg.as_slice(), generic_args: None },
191        }
192    }
193
194    pub fn mod_path(&self) -> Option<&ModPath> {
195        match self {
196            Path::BarePath(mod_path) => Some(mod_path),
197            Path::Normal(path) => Some(&path.mod_path),
198            Path::LangItem(..) => None,
199        }
200    }
201
202    pub fn qualifier(&self) -> Option<Path> {
203        match self {
204            Path::BarePath(mod_path) => {
205                if mod_path.is_ident() {
206                    return None;
207                }
208                Some(Path::BarePath(Interned::new(ModPath::from_segments(
209                    mod_path.kind,
210                    mod_path.segments()[..mod_path.segments().len() - 1].iter().cloned(),
211                ))))
212            }
213            Path::Normal(path) => {
214                let mod_path = &path.mod_path;
215                if mod_path.is_ident() {
216                    return None;
217                }
218                let type_anchor = path.type_anchor;
219                let generic_args = &path.generic_args;
220                let qualifier_mod_path = Interned::new(ModPath::from_segments(
221                    mod_path.kind,
222                    mod_path.segments()[..mod_path.segments().len() - 1].iter().cloned(),
223                ));
224                let qualifier_generic_args = &generic_args[..generic_args.len() - 1];
225                if type_anchor.is_none() && qualifier_generic_args.iter().all(|it| it.is_none()) {
226                    Some(Path::BarePath(qualifier_mod_path))
227                } else {
228                    Some(Path::Normal(Box::new(NormalPath {
229                        type_anchor,
230                        mod_path: qualifier_mod_path,
231                        generic_args: qualifier_generic_args.iter().cloned().collect(),
232                    })))
233                }
234            }
235            Path::LangItem(..) => None,
236        }
237    }
238
239    pub fn is_self_type(&self) -> bool {
240        match self {
241            Path::BarePath(mod_path) => mod_path.is_Self(),
242            Path::Normal(path) => {
243                path.type_anchor.is_none()
244                    && path.mod_path.is_Self()
245                    && path.generic_args.iter().all(|args| args.is_none())
246            }
247            Path::LangItem(..) => false,
248        }
249    }
250}

252#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
253pub struct PathSegment<'a> {
254    pub name: &'a Name,
255    pub args_and_bindings: Option<&'a GenericArgs>,
256}
257
258impl PathSegment<'_> {
259    pub const MISSING: PathSegment<'static> =
260        PathSegment { name: &Name::missing(), args_and_bindings: None };
261}
262
263#[derive(Debug, Clone, Copy)]
264pub struct PathSegments<'a> {
265    segments: &'a [Name],
266    generic_args: Option<&'a [Option<GenericArgs>]>,
267}
268
269impl<'a> PathSegments<'a> {
270    pub const EMPTY: PathSegments<'static> = PathSegments { segments: &[], generic_args: None };
271    pub fn is_empty(&self) -> bool {
272        self.len() == 0
273    }
274    pub fn len(&self) -> usize {
275        self.segments.len()
276    }
277    pub fn first(&self) -> Option<PathSegment<'a>> {
278        self.get(0)
279    }
280    pub fn last(&self) -> Option<PathSegment<'a>> {
281        self.get(self.len().checked_sub(1)?)
282    }
283
284    pub fn get(&self, idx: usize) -> Option<PathSegment<'a>> {
285        let res = PathSegment {
286            name: self.segments.get(idx)?,
287            args_and_bindings: self.generic_args.and_then(|it| it.get(idx)?.as_ref()),
288        };
289        Some(res)
290    }
291
292    pub fn skip(&self, len: usize) -> PathSegments<'a> {
293        PathSegments {
294            segments: self.segments.get(len..).unwrap_or(&[]),
295            generic_args: self.generic_args.and_then(|it| it.get(len..)),
296        }
297    }
298
299    pub fn take(&self, len: usize) -> PathSegments<'a> {
300        PathSegments {
301            segments: self.segments.get(..len).unwrap_or(self.segments),
302            generic_args: self.generic_args.map(|it| it.get(..len).unwrap_or(it)),
303        }
304    }
305
306    pub fn strip_last(&self) -> PathSegments<'a> {
307        PathSegments {
308            segments: self.segments.split_last().map_or(&[], |it| it.1),
309            generic_args: self.generic_args.map(|it| it.split_last().map_or(&[][..], |it| it.1)),
310        }
311    }
312
313    pub fn strip_last_two(&self) -> PathSegments<'a> {
314        PathSegments {
315            segments: self.segments.get(..self.segments.len().saturating_sub(2)).unwrap_or(&[]),
316            generic_args: self
317                .generic_args
318                .map(|it| it.get(..it.len().saturating_sub(2)).unwrap_or(&[])),
319        }
320    }
321
322    pub fn iter(&self) -> impl Iterator<Item = PathSegment<'a>> {
323        self.segments
324            .iter()
325            .zip(self.generic_args.into_iter().flatten().chain(iter::repeat(&None)))
326            .map(|(name, args)| PathSegment { name, args_and_bindings: args.as_ref() })
327    }
328}
