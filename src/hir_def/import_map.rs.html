23/// Item import details stored in the `ImportMap`.
24#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord)]
25pub struct ImportInfo {
26    /// A name that can be used to import the item, relative to the container.
27    pub name: Name,
28    /// The module containing this item.
29    pub container: ModuleId,
30    /// Whether this item is annotated with `#[doc(hidden)]`.
31    pub is_doc_hidden: bool,
32    /// Whether this item is annotated with `#[unstable(..)]`.
33    pub is_unstable: bool,
34    /// The value of `#[rust_analyzer::completions(...)]`, if exists.
35    pub complete: Complete,
36}

38/// A map from publicly exported items to its name.
39///
40/// Reexports of items are taken into account.
41#[derive(Default)]
42pub struct ImportMap {
43    /// Maps from `ItemInNs` to information of imports that bring the item into scope.
44    item_to_info_map: ImportMapIndex,
45    /// List of keys stored in [`Self::item_to_info_map`], sorted lexicographically by their
46    /// [`Name`]. Indexed by the values returned by running `fst`.
47    ///
48    /// Since a name can refer to multiple items due to namespacing and import aliases, we store all
49    /// items with the same name right after each other. This allows us to find all items after the
50    /// fst gives us the index of the first one.
51    ///
52    /// The [`u32`] is the index into the smallvec in the value of [`Self::item_to_info_map`].
53    importables: Vec<(ItemInNs, u32)>,
54    fst: fst::Map<Vec<u8>>,
55}
56
57#[derive(Copy, Clone, Debug, PartialEq, Eq, Ord, PartialOrd)]
58enum IsTraitAssocItem {
59    Yes,
60    No,
61}
62
63type ImportMapIndex = FxIndexMap<ItemInNs, (SmallVec<[ImportInfo; 1]>, IsTraitAssocItem)>;

73impl ImportMap {
74    pub fn dump(&self, db: &dyn DefDatabase) -> String {
75        let mut out = String::new();
76        for (k, v) in self.item_to_info_map.iter() {
77            format_to!(out, "{:?} ({:?}) -> ", k, v.1);
78            for v in &v.0 {
79                format_to!(out, "{}:{:?}, ", v.name.display(db, Edition::CURRENT), v.container);
80            }
81            format_to!(out, "\n");
82        }
83        out
84    }

86    fn import_map_query_impl(db: &dyn DefDatabase, krate: Crate) -> Self {
87        let _p = tracing::info_span!("import_map_query").entered();
88
89        let map = Self::collect_import_map(db, krate);
90
91        let mut importables: Vec<_> = map
92            .iter()
93            // We've only collected items, whose name cannot be tuple field so unwrapping is fine.
94            .flat_map(|(&item, (info, _))| {
95                info.iter()
96                    .enumerate()
97                    .map(move |(idx, info)| (item, info.name.as_str(), idx as u32))
98            })
99            .collect();
100        importables.sort_by(|(_, l_info, _), (_, r_info, _)| {
101            let lhs_chars = l_info.chars().map(|c| c.to_ascii_lowercase());
102            let rhs_chars = r_info.chars().map(|c| c.to_ascii_lowercase());
103            lhs_chars.cmp(rhs_chars)
104        });
105        importables.dedup();
106
107        // Build the FST, taking care not to insert duplicate values.
108        let mut builder = fst::MapBuilder::memory();
109        let mut iter = importables
110            .iter()
111            .enumerate()
112            .dedup_by(|&(_, (_, lhs, _)), &(_, (_, rhs, _))| lhs.eq_ignore_ascii_case(rhs));
113
114        let mut insert = |name: &str, start, end| {
115            builder.insert(name.to_ascii_lowercase(), ((start as u64) << 32) | end as u64).unwrap()
116        };
117
118        if let Some((mut last, (_, name, _))) = iter.next() {
119            debug_assert_eq!(last, 0);
120            let mut last_name = name;
121            for (next, (_, next_name, _)) in iter {
122                insert(last_name, last, next);
123                last = next;
124                last_name = next_name;
125            }
126            insert(last_name, last, importables.len());
127        }
128
129        let importables = importables.into_iter().map(|(item, _, idx)| (item, idx)).collect();
130        ImportMap { item_to_info_map: map, fst: builder.into_map(), importables }
131    }

133    pub fn import_info_for(&self, item: ItemInNs) -> Option<&[ImportInfo]> {
134        self.item_to_info_map.get(&item).map(|(info, _)| &**info)

137    fn collect_import_map(db: &dyn DefDatabase, krate: Crate) -> ImportMapIndex {
138        let _p = tracing::info_span!("collect_import_map").entered();
139
140        let def_map = crate_def_map(db, krate);
141        let mut map = FxIndexMap::default();
142
143        // We look only into modules that are public(ly reexported), starting with the crate root.
144        let root = def_map.root_module_id();
145        let mut worklist = vec![root];
146        let mut visited = FxHashSet::default();
147
148        while let Some(module) = worklist.pop() {
149            if !visited.insert(module) {
150                continue;
151            }
152            let mod_data = if module.krate(db) == krate {
153                &def_map[module]
154            } else {
155                // The crate might reexport a module defined in another crate.
156                &module.def_map(db)[module]
157            };
158
159            let visible_items = mod_data.scope.entries().filter_map(|(name, per_ns)| {
160                let per_ns = per_ns.filter_visibility(|vis| vis == Visibility::Public);
161                if per_ns.is_none() { None } else { Some((name, per_ns)) }
162            });
163
164            for (name, per_ns) in visible_items {
165                for (item, import) in per_ns.iter_items() {
166                    let attr_id = if let Some(import) = import {
167                        match import {
168                            ImportOrExternCrate::ExternCrate(id) => Some(id.into()),
169                            ImportOrExternCrate::Import(id) => Some(id.use_.into()),
170                            ImportOrExternCrate::Glob(id) => Some(id.use_.into()),
171                        }
172                    } else {
173                        match item {
174                            ItemInNs::Types(id) | ItemInNs::Values(id) => match id {
175                                ModuleDefId::ModuleId(it) => Some(AttrDefId::ModuleId(it)),
176                                ModuleDefId::FunctionId(it) => Some(it.into()),
177                                ModuleDefId::AdtId(it) => Some(it.into()),
178                                ModuleDefId::EnumVariantId(it) => Some(it.into()),
179                                ModuleDefId::ConstId(it) => Some(it.into()),
180                                ModuleDefId::StaticId(it) => Some(it.into()),
181                                ModuleDefId::TraitId(it) => Some(it.into()),
182                                ModuleDefId::TypeAliasId(it) => Some(it.into()),
183                                ModuleDefId::MacroId(it) => Some(it.into()),
184                                ModuleDefId::BuiltinType(_) => None,
185                            },
186                            ItemInNs::Macros(id) => Some(id.into()),
187                        }
188                    };
189                    let (is_doc_hidden, is_unstable, do_not_complete) = match attr_id {
190                        None => (false, false, Complete::Yes),
191                        Some(attr_id) => {
192                            let attrs = AttrFlags::query(db, attr_id);
193                            let do_not_complete =
194                                Complete::extract(matches!(attr_id, AttrDefId::TraitId(_)), attrs);
195                            (
196                                attrs.contains(AttrFlags::IS_DOC_HIDDEN),
197                                attrs.contains(AttrFlags::IS_UNSTABLE),
198                                do_not_complete,
199                            )
200                        }
201                    };
202
203                    let import_info = ImportInfo {
204                        name: name.clone(),
205                        container: module,
206                        is_doc_hidden,
207                        is_unstable,
208                        complete: do_not_complete,
209                    };
210
211                    if let Some(ModuleDefId::TraitId(tr)) = item.as_module_def_id() {
212                        Self::collect_trait_assoc_items(
213                            db,
214                            &mut map,
215                            tr,
216                            matches!(item, ItemInNs::Types(_)),
217                            &import_info,
218                        );
219                    }
220
221                    let (infos, _) =
222                        map.entry(item).or_insert_with(|| (SmallVec::new(), IsTraitAssocItem::No));
223                    infos.reserve_exact(1);
224                    infos.push(import_info);
225
226                    // If we've just added a module, descend into it.
227                    if let Some(ModuleDefId::ModuleId(mod_id)) = item.as_module_def_id() {
228                        worklist.push(mod_id);
229                    }
230                }
231            }
232        }
233        map.shrink_to_fit();
234        map
235    }

237    fn collect_trait_assoc_items(
238        db: &dyn DefDatabase,
239        map: &mut ImportMapIndex,
240        tr: TraitId,
241        is_type_in_ns: bool,
242        trait_import_info: &ImportInfo,
243    ) {
244        let _p = tracing::info_span!("collect_trait_assoc_items").entered();
245        for &(ref assoc_item_name, item) in &TraitItems::query(db, tr).items {
246            let module_def_id = match item {
247                AssocItemId::FunctionId(f) => ModuleDefId::from(f),
248                AssocItemId::ConstId(c) => ModuleDefId::from(c),
249                // cannot use associated type aliases directly: need a `<Struct as Trait>::TypeAlias`
250                // qualifier, ergo no need to store it for imports in import_map
251                AssocItemId::TypeAliasId(_) => {
252                    cov_mark::hit!(type_aliases_ignored);
253                    continue;
254                }
255            };
256            let assoc_item = if is_type_in_ns {
257                ItemInNs::Types(module_def_id)
258            } else {
259                ItemInNs::Values(module_def_id)
260            };
261
262            let attr_id = item.into();
263            let attrs = AttrFlags::query(db, attr_id);
264            let item_do_not_complete = Complete::extract(false, attrs);
265            let do_not_complete =
266                Complete::for_trait_item(trait_import_info.complete, item_do_not_complete);
267            let assoc_item_info = ImportInfo {
268                container: trait_import_info.container,
269                name: assoc_item_name.clone(),
270                is_doc_hidden: attrs.contains(AttrFlags::IS_DOC_HIDDEN),
271                is_unstable: attrs.contains(AttrFlags::IS_UNSTABLE),
272                complete: do_not_complete,
273            };
274
275            let (infos, _) =
276                map.entry(assoc_item).or_insert_with(|| (SmallVec::new(), IsTraitAssocItem::Yes));
277            infos.reserve_exact(1);
278            infos.push(assoc_item_info);
279        }
280    }

291impl fmt::Debug for ImportMap {
292    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
293        let mut importable_names: Vec<_> = self
294            .item_to_info_map
295            .iter()
296            .map(|(item, (infos, _))| {
297                let l = infos.len();
298                match item {
299                    ItemInNs::Types(it) => format!("- {it:?} (t) [{l}]",),
300                    ItemInNs::Values(it) => format!("- {it:?} (v) [{l}]",),
301                    ItemInNs::Macros(it) => format!("- {it:?} (m) [{l}]",),
302                }
303            })
304            .collect();
305
306        importable_names.sort();
307        f.write_str(&importable_names.join("\n"))
308    }

311/// A way to match import map contents against the search query.
312#[derive(Debug, Copy, Clone, PartialEq, Eq)]
313pub enum SearchMode {
314    /// Import map entry should strictly match the query string.
315    Exact,
316    /// Import map entry should contain all letters from the query string,
317    /// in the same order, but not necessary adjacent.
318    Fuzzy,
319    /// Import map entry should match the query string by prefix.
320    Prefix,
321}
322
323impl SearchMode {
324    pub fn check(self, query: &str, case_sensitive: bool, candidate: &str) -> bool {
325        match self {
326            SearchMode::Exact if case_sensitive => candidate == query,
327            SearchMode::Exact => candidate.eq_ignore_ascii_case(query),
328            SearchMode::Prefix => {
329                query.len() <= candidate.len() && {
330                    let prefix = &candidate[..query.len()];
331                    if case_sensitive {
332                        prefix == query
333                    } else {
334                        prefix.eq_ignore_ascii_case(query)
335                    }
336                }
337            }
338            SearchMode::Fuzzy => {
339                let mut name = candidate;
340                query.chars().all(|query_char| {
341                    let m = if case_sensitive {
342                        name.match_indices(query_char).next()
343                    } else {
344                        name.match_indices([query_char, query_char.to_ascii_uppercase()]).next()
345                    };
346                    match m {
347                        Some((index, _)) => {
348                            name = name[index..].strip_prefix(|_: char| true).unwrap_or_default();
349                            true
350                        }
351                        None => false,
352                    }
353                })
354            }
355        }
356    }
357}

359/// Three possible ways to search for the name in associated and/or other items.
360#[derive(Debug, Clone, Copy)]
361pub enum AssocSearchMode {
362    /// Search for the name in both associated and other items.
363    Include,
364    /// Search for the name in other items only.
365    Exclude,
366    /// Search for the name in the associated items only.
367    AssocItemsOnly,
368}

370#[derive(Debug)]
371pub struct Query {
372    query: String,
373    lowercased: String,
374    search_mode: SearchMode,
375    assoc_mode: AssocSearchMode,
376    case_sensitive: bool,
377}
378
379impl Query {
380    pub fn new(query: String) -> Self {
381        let lowercased = query.to_lowercase();
382        Self {
383            query,
384            lowercased,
385            search_mode: SearchMode::Exact,
386            assoc_mode: AssocSearchMode::Include,
387            case_sensitive: false,
388        }
389    }
390
391    /// Fuzzy finds items instead of exact matching.
392    pub fn fuzzy(self) -> Self {
393        Self { search_mode: SearchMode::Fuzzy, ..self }
394    }
395
396    pub fn prefix(self) -> Self {
397        Self { search_mode: SearchMode::Prefix, ..self }
398    }
399
400    pub fn exact(self) -> Self {
401        Self { search_mode: SearchMode::Exact, ..self }
402    }
403
404    /// Specifies whether we want to include associated items in the result.
405    pub fn assoc_search_mode(self, assoc_mode: AssocSearchMode) -> Self {
406        Self { assoc_mode, ..self }
407    }
408
409    /// Respect casing of the query string when matching.
410    pub fn case_sensitive(self) -> Self {
411        Self { case_sensitive: true, ..self }
412    }
413
414    fn matches_assoc_mode(&self, is_trait_assoc_item: IsTraitAssocItem) -> bool {
415        !matches!(
416            (is_trait_assoc_item, self.assoc_mode),
417            (IsTraitAssocItem::Yes, AssocSearchMode::Exclude)
418                | (IsTraitAssocItem::No, AssocSearchMode::AssocItemsOnly)
419        )
420    }
421}

423/// Searches dependencies of `krate` for an importable name matching `query`.
424///
425/// This returns a list of items that could be imported from dependencies of `krate`.
426pub fn search_dependencies(
427    db: &dyn DefDatabase,
428    krate: Crate,
429    query: &Query,
430) -> FxHashSet<(ItemInNs, Complete)> {
431    let _p = tracing::info_span!("search_dependencies", ?query).entered();
432
433    let import_maps: Vec<_> =
434        krate.data(db).dependencies.iter().map(|dep| ImportMap::of(db, dep.crate_id)).collect();
435
436    let mut op = fst::map::OpBuilder::new();
437
438    match query.search_mode {
439        SearchMode::Exact => {
440            let automaton = fst::automaton::Str::new(&query.lowercased);
441
442            for map in &import_maps {
443                op = op.add(map.fst.search(&automaton));
444            }
445            search_maps(db, &import_maps, op.union(), query)
446        }
447        SearchMode::Fuzzy => {
448            let automaton = fst::automaton::Subsequence::new(&query.lowercased);
449
450            for map in &import_maps {
451                op = op.add(map.fst.search(&automaton));
452            }
453            search_maps(db, &import_maps, op.union(), query)
454        }
455        SearchMode::Prefix => {
456            let automaton = fst::automaton::Str::new(&query.lowercased).starts_with();
457
458            for map in &import_maps {
459                op = op.add(map.fst.search(&automaton));
460            }
461            search_maps(db, &import_maps, op.union(), query)
462        }
463    }
464}

466fn search_maps(
467    _db: &dyn DefDatabase,
468    import_maps: &[&ImportMap],
469    mut stream: fst::map::Union<'_>,
470    query: &Query,
471) -> FxHashSet<(ItemInNs, Complete)> {
472    let mut res = FxHashSet::default();
473    while let Some((_, indexed_values)) = stream.next() {
474        for &IndexedValue { index: import_map_idx, value } in indexed_values {
475            let end = (value & 0xFFFF_FFFF) as usize;
476            let start = (value >> 32) as usize;
477            let ImportMap { item_to_info_map, importables, .. } = import_maps[import_map_idx];
478            let importables = &importables[start..end];
479
480            let iter = importables
481                .iter()
482                .copied()
483                .filter_map(|(item, info_idx)| {
484                    let (import_infos, assoc_mode) = &item_to_info_map[&item];
485                    query
486                        .matches_assoc_mode(*assoc_mode)
487                        .then(|| (item, &import_infos[info_idx as usize]))
488                })
489                .filter(|&(_, info)| {
490                    query.search_mode.check(&query.query, query.case_sensitive, info.name.as_str())
491                })
492                .map(|(item, import_info)| (item, import_info.complete));
493            res.extend(iter);
494        }
495    }
496
497    res
498}

638    fn render_path(db: &dyn DefDatabase, info: &ImportInfo) -> String {
639        let mut module = info.container;
640        let mut segments = vec![&info.name];
641
642        let def_map = module.def_map(db);
643        assert!(def_map.block_id().is_none(), "block local items should not be in `ImportMap`");
644
645        while let Some(parent) = module.containing_module(db) {
646            let parent_data = &def_map[parent];
647            let (name, _) = parent_data.children.iter().find(|(_, id)| **id == module).unwrap();
648            segments.push(name);
649            module = parent;
650        }
651
652        segments.iter().rev().map(|it| it.display(db, Edition::CURRENT)).join("::")
653    }
