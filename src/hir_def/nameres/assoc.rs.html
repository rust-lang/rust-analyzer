21#[derive(Debug, Clone, PartialEq, Eq)]
22pub struct TraitItems {
23    pub items: Box<[(Name, AssocItemId)]>,
24    // box it as the vec is usually empty anyways
25    // FIXME: AstIds are rather unstable...
26    pub macro_calls: Option<Box<Vec<(AstId<ast::Item>, MacroCallId)>>>,
27}
28
29impl TraitItems {
30    #[inline]
31    pub(crate) fn trait_items_query(db: &dyn DefDatabase, tr: TraitId) -> Arc<TraitItems> {
32        db.trait_items_with_diagnostics(tr).0
33    }
34
35    pub(crate) fn trait_items_with_diagnostics_query(
36        db: &dyn DefDatabase,
37        tr: TraitId,
38    ) -> (Arc<TraitItems>, DefDiagnostics) {
39        let ItemLoc { container: module_id, id: tree_id } = tr.lookup(db);
40
41        let collector = AssocItemCollector::new(db, module_id, ItemContainerId::TraitId(tr));
42        let item_tree = tree_id.item_tree(db);
43        let (items, macro_calls, diagnostics) =
44            collector.collect(&item_tree, tree_id.tree_id(), &item_tree[tree_id.value].items);
45
46        (Arc::new(TraitItems { macro_calls, items }), DefDiagnostics::new(diagnostics))
47    }
48
49    pub fn associated_types(&self) -> impl Iterator<Item = TypeAliasId> + '_ {
50        self.items.iter().filter_map(|(_name, item)| match item {
51            AssocItemId::TypeAliasId(t) => Some(*t),
52            _ => None,
53        })
54    }
55
56    pub fn associated_type_by_name(&self, name: &Name) -> Option<TypeAliasId> {
57        self.items.iter().find_map(|(item_name, item)| match item {
58            AssocItemId::TypeAliasId(t) if item_name == name => Some(*t),
59            _ => None,
60        })
61    }
62
63    pub fn method_by_name(&self, name: &Name) -> Option<FunctionId> {
64        self.items.iter().find_map(|(item_name, item)| match item {
65            AssocItemId::FunctionId(t) if item_name == name => Some(*t),
66            _ => None,
67        })
68    }
69
70    pub fn attribute_calls(&self) -> impl Iterator<Item = (AstId<ast::Item>, MacroCallId)> + '_ {
71        self.macro_calls.iter().flat_map(|it| it.iter()).copied()
72    }
73}

109struct AssocItemCollector<'a> {
110    db: &'a dyn DefDatabase,
111    module_id: ModuleId,
112    def_map: Arc<DefMap>,
113    local_def_map: Arc<LocalDefMap>,
114    diagnostics: Vec<DefDiagnostic>,
115    container: ItemContainerId,
116
117    depth: usize,
118    items: Vec<(Name, AssocItemId)>,
119    macro_calls: Vec<(AstId<ast::Item>, MacroCallId)>,
120}
121
122impl<'a> AssocItemCollector<'a> {
123    fn new(db: &'a dyn DefDatabase, module_id: ModuleId, container: ItemContainerId) -> Self {
124        let (def_map, local_def_map) = module_id.local_def_map(db);
125        Self {
126            db,
127            module_id,
128            def_map,
129            local_def_map,
130            container,
131            items: Vec::new(),
132
133            depth: 0,
134            macro_calls: Vec::new(),
135            diagnostics: Vec::new(),
136        }
137    }
138
139    fn collect(
140        mut self,
141        item_tree: &ItemTree,
142        tree_id: TreeId,
143        assoc_items: &[AssocItem],
144    ) -> (
145        Box<[(Name, AssocItemId)]>,
146        Option<Box<Vec<(AstId<ast::Item>, MacroCallId)>>>,
147        Vec<DefDiagnostic>,
148    ) {
149        self.items.reserve(assoc_items.len());
150        for &item in assoc_items {
151            self.collect_item(item_tree, tree_id, item);
152        }
153        (
154            self.items.into_boxed_slice(),
155            if self.macro_calls.is_empty() { None } else { Some(Box::new(self.macro_calls)) },
156            self.diagnostics,
157        )
158    }
159
160    fn collect_item(&mut self, item_tree: &ItemTree, tree_id: TreeId, item: AssocItem) {
161        let attrs = item_tree.attrs(self.db, self.module_id.krate, ModItem::from(item).into());
162        if !attrs.is_cfg_enabled(self.module_id.krate.cfg_options(self.db)) {
163            self.diagnostics.push(DefDiagnostic::unconfigured_code(
164                self.module_id.local_id,
165                tree_id,
166                ModItem::from(item).into(),
167                attrs.cfg().unwrap(),
168                self.module_id.krate.cfg_options(self.db).clone(),
169            ));
170            return;
171        }
172
173        'attrs: for attr in &*attrs {
174            let ast_id = AstId::new(tree_id.file_id(), item.ast_id(item_tree).upcast());
175            let ast_id_with_path = AstIdWithPath { path: attr.path.clone(), ast_id };
176
177            match self.def_map.resolve_attr_macro(
178                &self.local_def_map,
179                self.db,
180                self.module_id.local_id,
181                ast_id_with_path,
182                attr,
183            ) {
184                Ok(ResolvedAttr::Macro(call_id)) => {
185                    let loc = self.db.lookup_intern_macro_call(call_id);
186                    if let MacroDefKind::ProcMacro(_, exp, _) = loc.def.kind {
187                        // If there's no expander for the proc macro (e.g. the
188                        // proc macro is ignored, or building the proc macro
189                        // crate failed), skip expansion like we would if it was
190                        // disabled. This is analogous to the handling in
191                        // `DefCollector::collect_macros`.
192                        if let Some(err) = exp.as_expand_error(self.module_id.krate) {
193                            self.diagnostics.push(DefDiagnostic::macro_error(
194                                self.module_id.local_id,
195                                ast_id,
196                                (*attr.path).clone(),
197                                err,
198                            ));
199                            continue 'attrs;
200                        }
201                    }
202
203                    self.macro_calls.push((ast_id, call_id));
204                    self.collect_macro_items(call_id);
205                    return;
206                }
207                Ok(_) => (),
208                Err(_) => {
209                    self.diagnostics.push(DefDiagnostic::unresolved_macro_call(
210                        self.module_id.local_id,
211                        MacroCallKind::Attr { ast_id, attr_args: None, invoc_attr_index: attr.id },
212                        attr.path().clone(),
213                    ));
214                }
215            }
216        }
217
218        self.record_item(item_tree, tree_id, item);
219    }
220
221    fn record_item(&mut self, item_tree: &ItemTree, tree_id: TreeId, item: AssocItem) {
222        match item {
223            AssocItem::Function(id) => {
224                let item = &item_tree[id];
225                let def =
226                    FunctionLoc { container: self.container, id: ItemTreeId::new(tree_id, id) }
227                        .intern(self.db);
228                self.items.push((item.name.clone(), def.into()));
229            }
230            AssocItem::TypeAlias(id) => {
231                let item = &item_tree[id];
232                let def =
233                    TypeAliasLoc { container: self.container, id: ItemTreeId::new(tree_id, id) }
234                        .intern(self.db);
235                self.items.push((item.name.clone(), def.into()));
236            }
237            AssocItem::Const(id) => {
238                let item = &item_tree[id];
239                let Some(name) = item.name.clone() else { return };
240                let def = ConstLoc { container: self.container, id: ItemTreeId::new(tree_id, id) }
241                    .intern(self.db);
242                self.items.push((name, def.into()));
243            }
244            AssocItem::MacroCall(call) => {
245                let MacroCall { ast_id, expand_to, ctxt, ref path } = item_tree[call];
246
247                let resolver = |path: &_| {
248                    self.def_map
249                        .resolve_path(
250                            &self.local_def_map,
251                            self.db,
252                            self.module_id.local_id,
253                            path,
254                            crate::item_scope::BuiltinShadowMode::Other,
255                            Some(MacroSubNs::Bang),
256                        )
257                        .0
258                        .take_macros()
259                        .map(|it| self.db.macro_def(it))
260                };
261                match macro_call_as_call_id(
262                    self.db,
263                    &AstIdWithPath::new(tree_id.file_id(), ast_id, Clone::clone(path)),
264                    ctxt,
265                    expand_to,
266                    self.module_id.krate(),
267                    resolver,
268                    &mut |ptr, call_id| {
269                        self.macro_calls.push((ptr.map(|(_, it)| it.upcast()), call_id))
270                    },
271                ) {
272                    Ok(Some(call_id)) => {
273                        self.macro_calls
274                            .push((InFile::new(tree_id.file_id(), ast_id.upcast()), call_id));
275                        self.collect_macro_items(call_id);
276                    }
277                    Ok(None) => (),
278                    Err(_) => {
279                        self.diagnostics.push(DefDiagnostic::unresolved_macro_call(
280                            self.module_id.local_id,
281                            MacroCallKind::FnLike {
282                                ast_id: InFile::new(tree_id.file_id(), ast_id),
283                                expand_to,
284                                eager: None,
285                            },
286                            Clone::clone(path),
287                        ));
288                    }
289                }
290            }
291        }
292    }
293
294    fn collect_macro_items(&mut self, macro_call_id: MacroCallId) {
295        if self.depth > self.def_map.recursion_limit() as usize {
296            tracing::warn!("macro expansion is too deep");
297            return;
298        }
299        let file_id = macro_call_id.as_file();
300        let tree_id = TreeId::new(file_id, None);
301        let item_tree = self.db.file_item_tree(file_id);
302
303        self.depth += 1;
304        for item in item_tree.top_level_items().iter().filter_map(ModItem::as_assoc_item) {
305            self.collect_item(&item_tree, tree_id, item);
306        }
307        self.depth -= 1;
308    }
