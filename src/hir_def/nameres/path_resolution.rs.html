1//! This modules implements a function to resolve a path `foo::bar::baz` to a
2//! def, which is used within the name resolution.
3//!
4//! When name resolution is finished, the result of resolving a path is either
5//! `Some(def)` or `None`. However, when we are in process of resolving imports
6//! or macros, there's a third possibility:
7//!
8//!   I can't resolve this path right now, but I might be resolve this path
9//!   later, when more macros are expanded.
10//!
11//! `ReachedFixedPoint` signals about this.
12

46#[derive(Debug, Clone)]
47pub(super) struct ResolvePathResult {
48    pub(super) resolved_def: PerNs,
49    /// The index of the last resolved segment, or `None` if the full path has been resolved.
50    pub(super) segment_index: Option<usize>,
51    pub(super) reached_fixedpoint: ReachedFixedPoint,
52    pub(super) prefix_info: ResolvePathResultPrefixInfo,
53}

56pub struct ResolvePathResultPrefixInfo {

58    /// Path of the form `Enum::Variant` (and not `Variant` alone).

67            None,
68            ResolvePathResultPrefixInfo::default(),

75        segment_index: Option<usize>,

95    pub(crate) fn resolve_visibility(
96        &self,
97        local_def_map: &LocalDefMap,
98        db: &dyn DefDatabase,
99        // module to import to
100        original_module: ModuleId,
101        // pub(path)
102        //     ^^^^ this
103        visibility: &RawVisibility,
104        within_impl: bool,
105    ) -> Option<Visibility> {
106        let vis = match visibility {
107            RawVisibility::Module(path, explicitness) => {
108                let (result, remaining) = self.resolve_path(
109                    local_def_map,
110                    db,
111                    original_module,
112                    path,
113                    BuiltinShadowMode::Module,
114                    None,
115                );
116                if remaining.is_some() {
117                    return None;
118                }
119                let types = result.take_types()?;
120                let mut vis = match types {
121                    ModuleDefId::ModuleId(m) => Visibility::Module(m, *explicitness),
122                    // error: visibility needs to refer to module
123                    _ => {
124                        return None;
125                    }
126                };
127
128                // In block expressions, `self` normally refers to the containing non-block module, and
129                // `super` to its parent (etc.). However, visibilities must only refer to a module in the
130                // DefMap they're written in, so we restrict them when that happens.
131                if let Visibility::Module(m, mv) = vis {
132                    // ...unless we're resolving visibility for an associated item in an impl.
133                    if self.block_id() != m.block(db) && !within_impl {
134                        vis = Visibility::Module(self.root, mv);
135                        tracing::debug!(
136                            "visibility {:?} points outside DefMap, adjusting to {:?}",
137                            m,
138                            vis
139                        );
140                    }
141                }
142                vis
143            }
144            RawVisibility::PubSelf(explicitness) => {
145                Visibility::Module(original_module, *explicitness)
146            }
147            RawVisibility::Public => Visibility::Public,
148            RawVisibility::PubCrate => Visibility::PubCrate(self.krate),
149        };
150        Some(vis)
151    }

153    // Returns Yes if we are sure that additions to `ItemMap` wouldn't change
154    // the result.
155    pub(super) fn resolve_path_fp_with_macro(
156        &self,
157        local_def_map: &LocalDefMap,
158        db: &dyn DefDatabase,
159        mode: ResolveMode,
160        // module to import to
161        mut original_module: ModuleId,
162        path: &ModPath,
163        shadow: BuiltinShadowMode,
164        // Pass `MacroSubNs` if we know we're resolving macro names and which kind of macro we're
165        // resolving them to. Pass `None` otherwise, e.g. when we're resolving import paths.
166        expected_macro_subns: Option<MacroSubNs>,
167    ) -> ResolvePathResult {
168        let mut result = self.resolve_path_fp_with_macro_single(
169            local_def_map,
170            db,
171            mode,
172            original_module,
173            path,
174            shadow,
175            expected_macro_subns,
176        );
177
178        if self.block.is_none() {
179            // If we're in the root `DefMap`, we can resolve the path directly.
180            return result;
181        }
182
183        let mut current_map = self;
184
185        let mut merge = |new: ResolvePathResult| {
186            result.resolved_def = result.resolved_def.or(new.resolved_def);
187            if result.reached_fixedpoint == ReachedFixedPoint::No {
188                result.reached_fixedpoint = new.reached_fixedpoint;
189            }
190            result.prefix_info.differing_crate |= new.prefix_info.differing_crate;
191            result.prefix_info.enum_variant |= new.prefix_info.enum_variant;
192            result.segment_index = match (result.segment_index, new.segment_index) {
193                (Some(idx), None) => Some(idx),
194                (Some(old), Some(new)) => Some(old.max(new)),
195                (None, new) => new,
196            };
197        };
198
199        loop {
200            match current_map.block {
201                Some(block) if original_module == current_map.root => {
202                    // Block modules "inherit" names from its parent module.
203                    original_module = block.parent;
204                    current_map = block.parent.def_map(db);
205                }
206                // Proper (non-block) modules, including those in block `DefMap`s, don't.
207                _ => {
208                    if original_module != current_map.root && current_map.block.is_some() {
209                        // A module inside a block. Do not resolve items declared in upper blocks, but we do need to get
210                        // the prelude items (which are not inserted into blocks because they can be overridden there).
211                        original_module = current_map.root;
212                        current_map = crate_def_map(db, self.krate);
213
214                        let new = current_map.resolve_path_fp_in_all_preludes(
215                            local_def_map,
216                            db,
217                            mode,
218                            original_module,
219                            path,
220                            shadow,
221                        );
222                        merge(new);
223                    }
224
225                    return result;
226                }
227            }
228
229            let new = current_map.resolve_path_fp_with_macro_single(
230                local_def_map,
231                db,
232                mode,
233                original_module,
234                path,
235                shadow,
236                expected_macro_subns,
237            );
238
239            merge(new);
240        }
241    }

243    pub(super) fn resolve_path_fp_with_macro_single(
244        &self,
245        local_def_map: &LocalDefMap,
246        db: &dyn DefDatabase,
247        mode: ResolveMode,
248        original_module: ModuleId,
249        path: &ModPath,
250        shadow: BuiltinShadowMode,
251        expected_macro_subns: Option<MacroSubNs>,
252    ) -> ResolvePathResult {
253        let mut segments = path.segments().iter().enumerate();
254        let curr_per_ns = match path.kind {
255            PathKind::DollarCrate(krate) => {
256                if krate == self.krate {
257                    cov_mark::hit!(macro_dollar_crate_self);
258                    PerNs::types(self.crate_root(db).into(), Visibility::Public, None)
259                } else {
260                    let def_map = crate_def_map(db, krate);
261                    let module = def_map.root;
262                    cov_mark::hit!(macro_dollar_crate_other);
263                    PerNs::types(module.into(), Visibility::Public, None)
264                }
265            }
266            PathKind::Crate => PerNs::types(self.crate_root(db).into(), Visibility::Public, None),
267            // plain import or absolute path in 2015: crate-relative with
268            // fallback to extern prelude (with the simplification in
269            // rust-lang/rust#57745)
270            PathKind::Plain | PathKind::Abs
271                if self.data.edition == Edition::Edition2015
272                    && (path.kind == PathKind::Abs || mode == ResolveMode::Import) =>
273            {
274                let (_, segment) = match segments.next() {
275                    Some((idx, segment)) => (idx, segment),
276                    None => return ResolvePathResult::empty(ReachedFixedPoint::Yes),
277                };
278                tracing::debug!("resolving {:?} in crate root (+ extern prelude)", segment);
279                self.resolve_name_in_crate_root_or_extern_prelude(
280                    local_def_map,
281                    db,
282                    original_module,
283                    segment,
284                )
285            }
286            PathKind::Plain => {
287                let (_, segment) = match segments.next() {
288                    Some((idx, segment)) => (idx, segment),
289                    None => return ResolvePathResult::empty(ReachedFixedPoint::Yes),
290                };
291                // The first segment may be a builtin type. If the path has more
292                // than one segment, we first try resolving it as a module
293                // anyway.
294                // FIXME: If the next segment doesn't resolve in the module and
295                // BuiltinShadowMode wasn't Module, then we need to try
296                // resolving it as a builtin.
297                let prefer_module =
298                    if path.segments().len() == 1 { shadow } else { BuiltinShadowMode::Module };
299
300                tracing::debug!("resolving {:?} in module", segment);
301                self.resolve_name_in_module(
302                    local_def_map,
303                    db,
304                    original_module,
305                    segment,
306                    prefer_module,
307                    expected_macro_subns,
308                )
309            }
310            PathKind::Super(lvl) => {
311                let mut local_id = original_module;
312                let mut def_map = self;
313
314                // Adjust `local_id` to `self`, i.e. the nearest non-block module.
315                (def_map, local_id) = adjust_to_nearest_non_block_module(db, def_map, local_id);
316
317                // Go up the module tree but skip block modules as `super` always refers to the
318                // nearest non-block module.
319                for _ in 0..lvl {
320                    // Loop invariant: at the beginning of each loop, `local_id` must refer to a
321                    // non-block module.
322                    if let Some(parent) = def_map.modules[local_id].parent {
323                        (def_map, local_id) =
324                            adjust_to_nearest_non_block_module(db, def_map, parent);
325                    } else {
326                        stdx::always!(def_map.block.is_none());
327                        tracing::debug!("super path in root module");
328                        return ResolvePathResult::empty(ReachedFixedPoint::Yes);
329                    }
330                }
331
332                if self.block != def_map.block {
333                    // If we have a different `DefMap` from `self` (the original `DefMap` we started
334                    // with), resolve the remaining path segments in that `DefMap`.
335                    let path =
336                        ModPath::from_segments(PathKind::SELF, path.segments().iter().cloned());
337                    // This is the same crate, so the local def map is the same.
338                    return def_map.resolve_path_fp_with_macro(
339                        local_def_map,
340                        db,
341                        mode,
342                        local_id,
343                        &path,
344                        shadow,
345                        expected_macro_subns,
346                    );
347                }
348
349                PerNs::types(local_id.into(), Visibility::Public, None)
350            }
351            PathKind::Abs => match self.resolve_path_abs(local_def_map, &mut segments, path) {
352                Either::Left(it) => it,
353                Either::Right(reached_fixed_point) => {
354                    return ResolvePathResult::empty(reached_fixed_point);
355                }
356            },
357        };
358
359        self.resolve_remaining_segments(
360            db,
361            mode,
362            segments,
363            curr_per_ns,
364            path,
365            shadow,
366            original_module,
367        )
368    }

370    /// Resolves a path only in the preludes, without accounting for item scopes.
371    pub(super) fn resolve_path_fp_in_all_preludes(
372        &self,
373        local_def_map: &LocalDefMap,
374        db: &dyn DefDatabase,
375        mode: ResolveMode,
376        original_module: ModuleId,
377        path: &ModPath,
378        shadow: BuiltinShadowMode,
379    ) -> ResolvePathResult {
380        let mut segments = path.segments().iter().enumerate();
381        let curr_per_ns = match path.kind {
382            // plain import or absolute path in 2015: crate-relative with
383            // fallback to extern prelude (with the simplification in
384            // rust-lang/rust#57745)
385            PathKind::Plain | PathKind::Abs
386                if self.data.edition == Edition::Edition2015
387                    && (path.kind == PathKind::Abs || mode == ResolveMode::Import) =>
388            {
389                let (_, segment) = match segments.next() {
390                    Some((idx, segment)) => (idx, segment),
391                    None => return ResolvePathResult::empty(ReachedFixedPoint::Yes),
392                };
393                tracing::debug!("resolving {:?} in crate root (+ extern prelude)", segment);
394                self.resolve_name_in_extern_prelude(local_def_map, segment)
395            }
396            PathKind::Plain => {
397                let (_, segment) = match segments.next() {
398                    Some((idx, segment)) => (idx, segment),
399                    None => return ResolvePathResult::empty(ReachedFixedPoint::Yes),
400                };
401                tracing::debug!("resolving {:?} in module", segment);
402                self.resolve_name_in_all_preludes(local_def_map, db, segment)
403            }
404            PathKind::Abs => match self.resolve_path_abs(local_def_map, &mut segments, path) {
405                Either::Left(it) => it,
406                Either::Right(reached_fixed_point) => {
407                    return ResolvePathResult::empty(reached_fixed_point);
408                }
409            },
410            PathKind::DollarCrate(_) | PathKind::Crate | PathKind::Super(_) => {
411                return ResolvePathResult::empty(ReachedFixedPoint::Yes);
412            }
413        };
414
415        self.resolve_remaining_segments(
416            db,
417            mode,
418            segments,
419            curr_per_ns,
420            path,
421            shadow,
422            original_module,
423        )
424    }

426    /// 2018-style absolute path -- only extern prelude
427    fn resolve_path_abs<'a>(
428        &self,
429        local_def_map: &LocalDefMap,
430        segments: &mut impl Iterator<Item = (usize, &'a Name)>,
431        path: &ModPath,
432    ) -> Either<PerNs, ReachedFixedPoint> {
433        let segment = match segments.next() {
434            Some((_, segment)) => segment,
435            None => return Either::Right(ReachedFixedPoint::Yes),
436        };
437        if let Some(&(def, extern_crate)) = local_def_map.extern_prelude.get(segment) {
438            tracing::debug!("absolute path {:?} resolved to crate {:?}", path, def);
439            Either::Left(PerNs::types(
440                def.into(),
441                Visibility::Public,
442                extern_crate.map(ImportOrExternCrate::ExternCrate),
443            ))
444        } else {
445            Either::Right(ReachedFixedPoint::No) // extern crate declarations can add to the extern prelude
446        }
447    }

449    fn resolve_remaining_segments<'a>(
450        &self,
451        db: &dyn DefDatabase,
452        mode: ResolveMode,
453        mut segments: impl Iterator<Item = (usize, &'a Name)>,
454        mut curr_per_ns: PerNs,
455        path: &ModPath,
456        shadow: BuiltinShadowMode,
457        original_module: ModuleId,
458    ) -> ResolvePathResult {
459        while let Some((i, segment)) = segments.next() {
460            let curr = match curr_per_ns.take_types_full() {
461                Some(r) => r,
462                None => {
463                    // we still have path segments left, but the path so far
464                    // didn't resolve in the types namespace => no resolution
465                    // (don't break here because `curr_per_ns` might contain
466                    // something in the value namespace, and it would be wrong
467                    // to return that)
468                    return ResolvePathResult::empty(ReachedFixedPoint::No);
469                }
470            };
471            // resolve segment in curr
472
473            curr_per_ns = match curr.def {
474                ModuleDefId::ModuleId(module) => {
475                    if module.krate(db) != self.krate {
476                        // FIXME: Inefficient
477                        let path = ModPath::from_segments(
478                            PathKind::SELF,
479                            path.segments()[i..].iter().cloned(),
480                        );
481                        tracing::debug!("resolving {:?} in other crate", path);
482                        let defp_map = module.def_map(db);
483                        // Macro sub-namespaces only matter when resolving single-segment paths
484                        // because `macro_use` and other preludes should be taken into account. At
485                        // this point, we know we're resolving a multi-segment path so macro kind
486                        // expectation is discarded.
487                        let resolution = defp_map.resolve_path_fp_with_macro(
488                            LocalDefMap::EMPTY,
489                            db,
490                            mode,
491                            module,
492                            &path,
493                            shadow,
494                            None,
495                        );
496                        return ResolvePathResult::new(
497                            resolution.resolved_def,
498                            ReachedFixedPoint::Yes,
499                            resolution.segment_index.map(|s| s + i),
500                            ResolvePathResultPrefixInfo {
501                                differing_crate: true,
502                                enum_variant: resolution.prefix_info.enum_variant,
503                            },
504                        );
505                    }
506
507                    let def_map;
508                    let module_data = if module.block(db) == self.block_id() {
509                        &self[module]
510                    } else {
511                        def_map = module.def_map(db);
512                        &def_map[module]
513                    };
514
515                    // Since it is a qualified path here, it should not contains legacy macros
516                    module_data.scope.get(segment)
517                }
518                ModuleDefId::AdtId(AdtId::EnumId(e)) => {
519                    // enum variant
520                    cov_mark::hit!(can_import_enum_variant);
521
522                    let res = e
523                        .enum_variants(db)
524                        .variants
525                        .iter()
526                        .find(|(_, name, _)| name == segment)
527                        .map(|&(variant, _, shape)| match shape {
528                            FieldsShape::Record => {
529                                PerNs::types(variant.into(), Visibility::Public, None)
530                            }
531                            FieldsShape::Tuple | FieldsShape::Unit => PerNs::both(
532                                variant.into(),
533                                variant.into(),
534                                Visibility::Public,
535                                None,
536                            ),
537                        });
538                    // FIXME: Need to filter visibility here and below? Not sure.
539                    return match res {
540                        Some(res) => {
541                            if segments.next().is_some() {
542                                // Enum variants are in value namespace, segments left => no resolution.
543                                ResolvePathResult::empty(ReachedFixedPoint::No)
544                            } else {
545                                ResolvePathResult::new(
546                                    res,
547                                    ReachedFixedPoint::Yes,
548                                    None,
549                                    ResolvePathResultPrefixInfo {
550                                        enum_variant: true,
551                                        ..ResolvePathResultPrefixInfo::default()
552                                    },
553                                )
554                            }
555                        }
556                        None => ResolvePathResult::new(
557                            PerNs::types(e.into(), curr.vis, curr.import),
558                            ReachedFixedPoint::Yes,
559                            Some(i),
560                            ResolvePathResultPrefixInfo::default(),
561                        ),
562                    };
563                }
564                def @ ModuleDefId::TraitId(t) if mode == ResolveMode::Import => {
565                    // FIXME: Implement this correctly
566                    // We can't actually call `trait_items`, the reason being that if macro calls
567                    // occur, they will call back into the def map which we might be computing right
568                    // now resulting in a cycle.
569                    // To properly implement this, trait item collection needs to be done in def map
570                    // collection...
571                    let item = if true {
572                        None
573                    } else {
574                        TraitItems::query(db, t).assoc_item_by_name(segment)
575                    };
576                    return match item {
577                        Some(item) => ResolvePathResult::new(
578                            match item {
579                                crate::AssocItemId::FunctionId(function_id) => PerNs::values(
580                                    function_id.into(),
581                                    curr.vis,
582                                    curr.import.and_then(|it| it.import_or_glob()),
583                                ),
584                                crate::AssocItemId::ConstId(const_id) => PerNs::values(
585                                    const_id.into(),
586                                    curr.vis,
587                                    curr.import.and_then(|it| it.import_or_glob()),
588                                ),
589                                crate::AssocItemId::TypeAliasId(type_alias_id) => {
590                                    PerNs::types(type_alias_id.into(), curr.vis, curr.import)
591                                }
592                            },
593                            ReachedFixedPoint::Yes,
594                            segments.next().map(TupleExt::head),
595                            ResolvePathResultPrefixInfo::default(),
596                        ),
597                        None => ResolvePathResult::new(
598                            PerNs::types(def, curr.vis, curr.import),
599                            ReachedFixedPoint::Yes,
600                            Some(i),
601                            ResolvePathResultPrefixInfo::default(),
602                        ),
603                    };
604                }
605                s => {
606                    // could be an inherent method call in UFCS form
607                    // (`Struct::method`), or some other kind of associated item
608                    tracing::debug!(
609                        "path segment {:?} resolved to non-module {:?}, but is not last",
610                        segment,
611                        curr,
612                    );
613
614                    return ResolvePathResult::new(
615                        PerNs::types(s, curr.vis, curr.import),
616                        ReachedFixedPoint::Yes,
617                        Some(i),
618                        ResolvePathResultPrefixInfo::default(),
619                    );
620                }
621            };
622
623            curr_per_ns = curr_per_ns
624                .filter_visibility(|vis| vis.is_visible_from_def_map(db, self, original_module));
625        }
626
627        ResolvePathResult::new(
628            curr_per_ns,
629            ReachedFixedPoint::Yes,
630            None,
631            ResolvePathResultPrefixInfo::default(),
632        )
633    }

635    fn resolve_name_in_module(
636        &self,
637        local_def_map: &LocalDefMap,
638        db: &dyn DefDatabase,
639        module: ModuleId,
640        name: &Name,
641        shadow: BuiltinShadowMode,
642        expected_macro_subns: Option<MacroSubNs>,
643    ) -> PerNs {
644        // Resolve in:
645        //  - legacy scope of macro
646        //  - current module / scope
647        //  - extern prelude / macro_use prelude
648        //  - std prelude
649        let from_legacy_macro = self[module]
650            .scope
651            .get_legacy_macro(name)
652            // FIXME: shadowing
653            .and_then(|it| it.last())
654            .copied()
655            .filter(|&id| sub_namespace_match(db, id, expected_macro_subns))
656            .map_or_else(PerNs::none, |m| PerNs::macros(m, Visibility::Public, None));
657        let from_scope = self[module].scope.get(name).filter_macro(db, expected_macro_subns);
658        let from_builtin = match self.block {
659            Some(_) => {
660                // Only resolve to builtins in the root `DefMap`.
661                PerNs::none()
662            }
663            None => BUILTIN_SCOPE.get(name).copied().unwrap_or_else(PerNs::none),
664        };
665        let from_scope_or_builtin = match shadow {
666            BuiltinShadowMode::Module => from_scope.or(from_builtin),
667            BuiltinShadowMode::Other => match from_scope.take_types() {
668                Some(ModuleDefId::ModuleId(_)) => from_builtin.or(from_scope),
669                Some(_) | None => from_scope.or(from_builtin),
670            },
671        };
672
673        let extern_prelude = || {
674            if self.block.is_some() && module == self.root {
675                // Don't resolve extern prelude in pseudo-modules of blocks, because
676                // they might been shadowed by local names.
677                return PerNs::none();
678            }
679            self.resolve_name_in_extern_prelude(local_def_map, name)
680        };
681        let macro_use_prelude = || self.resolve_in_macro_use_prelude(name);
682        let prelude = || {
683            if self.block.is_some() && module == self.root {
684                return PerNs::none();
685            }
686            self.resolve_in_prelude(db, name)
687        };
688
689        from_legacy_macro
690            .or(from_scope_or_builtin)
691            .or_else(extern_prelude)
692            .or_else(macro_use_prelude)
693            .or_else(prelude)
694    }
695
696    fn resolve_name_in_all_preludes(
697        &self,
698        local_def_map: &LocalDefMap,
699        db: &dyn DefDatabase,
700        name: &Name,
701    ) -> PerNs {
702        // Resolve in:
703        //  - extern prelude / macro_use prelude
704        //  - std prelude
705        let extern_prelude = self.resolve_name_in_extern_prelude(local_def_map, name);
706        let macro_use_prelude = || self.resolve_in_macro_use_prelude(name);
707        let prelude = || self.resolve_in_prelude(db, name);
708
709        extern_prelude.or_else(macro_use_prelude).or_else(prelude)
710    }
711
712    fn resolve_name_in_extern_prelude(&self, local_def_map: &LocalDefMap, name: &Name) -> PerNs {
713        local_def_map.extern_prelude.get(name).map_or(PerNs::none(), |&(it, extern_crate)| {
714            PerNs::types(
715                it.into(),
716                Visibility::Public,
717                extern_crate.map(ImportOrExternCrate::ExternCrate),
718            )
719        })
720    }
721
722    fn resolve_in_macro_use_prelude(&self, name: &Name) -> PerNs {
723        self.macro_use_prelude.get(name).map_or(PerNs::none(), |&(it, extern_crate)| {
724            PerNs::macros(
725                it,
726                Visibility::Public,
727                extern_crate.map(ImportOrExternCrate::ExternCrate),
728            )
729        })
730    }
731
732    fn resolve_name_in_crate_root_or_extern_prelude(
733        &self,
734        local_def_map: &LocalDefMap,
735        db: &dyn DefDatabase,
736        module: ModuleId,
737        name: &Name,
738    ) -> PerNs {
739        let from_crate_root = match self.block {
740            Some(_) => {
741                let def_map = self.crate_root(db).def_map(db);
742                def_map[def_map.root].scope.get(name)
743            }
744            None => self[self.root].scope.get(name),
745        };
746        let from_extern_prelude = || {
747            if self.block.is_some() && module == self.root {
748                // Don't resolve extern prelude in pseudo-module of a block.
749                return PerNs::none();
750            }
751            self.resolve_name_in_extern_prelude(local_def_map, name)
752        };
753
754        from_crate_root.or_else(from_extern_prelude)
755    }
756
757    fn resolve_in_prelude(&self, db: &dyn DefDatabase, name: &Name) -> PerNs {
758        if let Some((prelude, _use)) = self.prelude {
759            let keep;
760            let def_map = if prelude.krate(db) == self.krate {
761                self
762            } else {
763                // Extend lifetime
764                keep = prelude.def_map(db);
765                keep
766            };
767            def_map[prelude].scope.get(name)
768        } else {
769            PerNs::none()
770        }
771    }
772}
773
774/// Given a block module, returns its nearest non-block module and the `DefMap` it belongs to.
775#[inline]
776fn adjust_to_nearest_non_block_module<'db>(
777    db: &'db dyn DefDatabase,
778    mut def_map: &'db DefMap,
779    mut local_id: ModuleId,
780) -> (&'db DefMap, ModuleId) {
781    if def_map.root_module_id() != local_id {
782        // if we aren't the root, we are either not a block module, or a non-block module inside a
783        // block def map.
784        return (def_map, local_id);
785    }
786    while let Some(BlockInfo { parent, .. }) = def_map.block {
787        def_map = parent.def_map(db);
788        local_id = parent;
789        if def_map.root_module_id() != local_id {
790            return (def_map, local_id);
791        }
792    }
793    (def_map, local_id)
794}
