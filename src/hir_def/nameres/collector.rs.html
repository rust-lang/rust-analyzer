63pub(super) fn collect_defs(
64    db: &dyn DefDatabase,
65    def_map: DefMap,
66    tree_id: TreeId,
67    crate_local_def_map: Option<&LocalDefMap>,
68) -> (DefMap, LocalDefMap) {
69    let krate = &def_map.krate.data(db);
70    let cfg_options = def_map.krate.cfg_options(db);
71
72    // populate external prelude and dependency list
73    let mut deps =
74        FxIndexMap::with_capacity_and_hasher(krate.dependencies.len(), Default::default());
75    for dep in &krate.dependencies {
76        tracing::debug!("crate dep {:?} -> {:?}", dep.name, dep.crate_id);
77
78        deps.insert(dep.as_name(), dep.clone());
79    }
80
81    let proc_macros = if krate.is_proc_macro {
82        db.proc_macros_for_crate(def_map.krate)
83            .and_then(|proc_macros| {
84                proc_macros.list(db.syntax_context(tree_id.file_id(), krate.edition))
85            })
86            .unwrap_or_default()
87    } else {
88        Default::default()
89    };
90
91    let mut collector = DefCollector {
92        db,
93        def_map,
94        local_def_map: LocalDefMap::default(),
95        crate_local_def_map,
96        deps,
97        glob_imports: FxHashMap::default(),
98        unresolved_imports: Vec::new(),
99        indeterminate_imports: Vec::new(),
100        unresolved_macros: Vec::new(),
101        mod_dirs: FxHashMap::default(),
102        cfg_options,
103        proc_macros,
104        from_glob_import: Default::default(),
105        skip_attrs: Default::default(),
106        prev_active_attrs: Default::default(),
107        unresolved_extern_crates: Default::default(),
108        is_proc_macro: krate.is_proc_macro,
109        deferred_builtin_derives: Default::default(),
110    };
111    if tree_id.is_block() {
112        collector.seed_with_inner(tree_id);
113    } else {
114        collector.seed_with_top_level();
115    }
116    collector.collect();
117    let (mut def_map, mut local_def_map) = collector.finish();
118    def_map.shrink_to_fit();
119    local_def_map.shrink_to_fit();
120    (def_map, local_def_map)
121}

124enum PartialResolvedImport {
125    /// None of any namespaces is resolved
126    Unresolved,
127    /// One of namespaces is resolved
128    Indeterminate(PerNs),
129    /// All namespaces are resolved, OR it comes from other crate

134    fn namespaces(self) -> PerNs {
135        match self {

138        }

232/// Walks the tree of module recursively
233struct DefCollector<'db> {
234    db: &'db dyn DefDatabase,
235    def_map: DefMap,
236    local_def_map: LocalDefMap,
237    /// Set only in case of blocks.
238    crate_local_def_map: Option<&'db LocalDefMap>,
239    // The dependencies of the current crate, including optional deps like `test`.
240    deps: FxIndexMap<Name, BuiltDependency>,
241    glob_imports: FxHashMap<ModuleId, Vec<(ModuleId, Visibility, GlobId)>>,
242    unresolved_imports: Vec<ImportDirective>,
243    indeterminate_imports: Vec<(ImportDirective, PerNs)>,
244    unresolved_macros: Vec<MacroDirective<'db>>,
245    // We'd like to avoid emitting a diagnostics avalanche when some `extern crate` doesn't
246    // resolve. When we emit diagnostics for unresolved imports, we only do so if the import
247    // doesn't start with an unresolved crate's name.
248    unresolved_extern_crates: FxHashSet<Name>,
249    mod_dirs: FxHashMap<ModuleId, ModDir>,
250    cfg_options: &'db CfgOptions,
251    /// List of procedural macros defined by this crate. This is read from the dynamic library
252    /// built by the build system, and is the list of proc-macros we can actually expand. It is
253    /// empty when proc-macro support is disabled (in which case we still do name resolution for
254    /// them). The bool signals whether the proc-macro has been explicitly disabled for name-resolution.
255    proc_macros: Box<[(Name, CustomProcMacroExpander, bool)]>,
256    is_proc_macro: bool,
257    from_glob_import: PerNsGlobImports,
258    /// If we fail to resolve an attribute on a `ModItem`, we fall back to ignoring the attribute.
259    /// This map is used to skip all attributes up to and including the one that failed to resolve,
260    /// in order to not expand them twice.
261    ///
262    /// This also stores the attributes to skip when we resolve derive helpers and non-macro
263    /// non-builtin attributes in general.
264    // FIXME: There has to be a better way to do this
265    skip_attrs: FxHashMap<AstId<ast::Item>, AttrId>,
266    /// When we expand attributes, we need to censor all previous active attributes
267    /// on the same item. Therefore, this holds all active attributes that we already
268    /// expanded.
269    prev_active_attrs: FxHashMap<AstId<ast::Item>, SmallVec<[AttrId; 1]>>,
270    /// To save memory, we do not really expand builtin derives. Instead, we save them as a `BuiltinDeriveImplId`.
271    ///
272    /// However, we can only do that when the derive is directly above the item, and there is no attribute in between.
273    /// Otherwise, all sorts of weird things can happen, like the item name resolving to something else.
274    deferred_builtin_derives: FxHashMap<AstId<ast::Item>, Vec<DeferredBuiltinDerive>>,
275}

278    fn seed_with_top_level(&mut self) {
279        let _p = tracing::info_span!("seed_with_top_level").entered();
280
281        let file_id = self.def_map.krate.root_file_id(self.db);
282        let item_tree = self.db.file_item_tree(file_id.into());
283        let attrs = match item_tree.top_level_attrs() {
284            AttrsOrCfg::Enabled { attrs } => attrs.as_ref(),
285            AttrsOrCfg::CfgDisabled(it) => it.1.as_ref(),
286        };
287        let crate_data = Arc::get_mut(&mut self.def_map.data).unwrap();
288
289        // Process other crate-level attributes.
290        for attr in &*attrs {
291            let Some(attr_name) = attr.path.as_ident() else { continue };
292
293            match () {
294                () if *attr_name == sym::recursion_limit => {
295                    if let Some(limit) = attr.string_value()
296                        && let Ok(limit) = limit.parse()
297                    {
298                        crate_data.recursion_limit = Some(limit);
299                    }
300                }
301                () if *attr_name == sym::crate_type => {
302                    if attr.string_value() == Some("proc-macro") {
303                        self.is_proc_macro = true;
304                    }
305                }
306                () if *attr_name == sym::no_core => crate_data.no_core = true,
307                () if *attr_name == sym::no_std => crate_data.no_std = true,
308                () if *attr_name == sym::rustc_coherence_is_core => {
309                    crate_data.rustc_coherence_is_core = true;
310                }
311                () if *attr_name == sym::feature => {
312                    let features =
313                        attr.parse_path_comma_token_tree(self.db).into_iter().flatten().filter_map(
314                            |(feat, _, _)| match feat.segments() {
315                                [name] => Some(name.symbol().clone()),
316                                _ => None,
317                            },
318                        );
319                    crate_data.unstable_features.extend(features);
320                }
321                () if *attr_name == sym::register_tool => {
322                    if let Some(ident) = attr.single_ident_value() {
323                        crate_data.registered_tools.push(ident.sym.clone());
324                        cov_mark::hit!(register_tool);
325                    }
326                }
327                () => (),
328            }
329        }
330
331        for (name, dep) in &self.deps {
332            // Add all
333            if dep.is_prelude() {
334                // This is a bit confusing but the gist is that `no_core` and `no_std` remove the
335                // sysroot dependence on `core` and `std` respectively. Our `CrateGraph` is eagerly
336                // constructed with them in place no matter what though, since at that point we
337                // don't do pre-configured attribute resolution yet.
338                // So here check if we are no_core / no_std and we are trying to add the
339                // corresponding dep from the sysroot
340
341                // Depending on the crate data of a dependency seems bad for incrementality, but
342                // we only do that for sysroot crates (this is why the order of the `&&` is important)
343                // - which are normally standard library crate, which realistically aren't going
344                // to have their crate ID invalidated, because they stay on the same root file and
345                // they're dependencies of everything else, so if some collision miraculously occurs
346                // we will resolve it by disambiguating the other crate.
347                let skip = dep.is_sysroot()
348                    && match dep.crate_id.data(self.db).origin {
349                        CrateOrigin::Lang(LangCrateOrigin::Core) => crate_data.no_core,
350                        CrateOrigin::Lang(LangCrateOrigin::Std) => {
351                            crate_data.no_core || crate_data.no_std
352                        }
353                        _ => false,
354                    };
355                if skip {
356                    continue;
357                }
358
359                // This eagerly draws a dependency edge between the crate def maps even if the
360                // things the crates are not used. This is not great, but at the same time we would
361                // like to compute our dependency def maps in parallel here anyways in the future
362                // which will have the same effect.
363                self.local_def_map.extern_prelude.insert(
364                    name.clone(),
365                    (crate_def_map(self.db, dep.crate_id).root_module_id(), None),
366                );
367            }
368        }
369
370        self.inject_prelude();
371
372        if matches!(item_tree.top_level_attrs(), AttrsOrCfg::CfgDisabled(_)) {
373            return;
374        }
375
376        let module_id = self.def_map.root;
377        ModCollector {
378            def_collector: self,
379            macro_depth: 0,
380            module_id,
381            tree_id: TreeId::new(file_id.into(), None),
382            item_tree,
383            mod_dir: ModDir::root(),
384        }
385        .collect_in_top_module(item_tree.top_level_items());
386        Arc::get_mut(&mut self.def_map.data).unwrap().shrink_to_fit();
387    }

389    fn seed_with_inner(&mut self, tree_id: TreeId) {
390        let item_tree = tree_id.item_tree(self.db);
391        let is_cfg_enabled = matches!(item_tree.top_level_attrs(), AttrsOrCfg::Enabled { .. });
392        if is_cfg_enabled {
393            self.inject_prelude();

395            let module_id = self.def_map.root;
396            ModCollector {

398                macro_depth: 0,
399                module_id,
400                tree_id,
401                item_tree,
402                mod_dir: ModDir::root(),
403            }
404            .collect_in_top_module(item_tree.top_level_items());
405        }

408    fn resolution_loop(&mut self) {
409        let _p = tracing::info_span!("DefCollector::resolution_loop").entered();
410
411        // main name resolution fixed-point loop.
412        let mut i = 0;
413        'resolve_attr: loop {
414            let _p = tracing::info_span!("resolve_macros loop").entered();
415            'resolve_macros: loop {
416                self.db.unwind_if_revision_cancelled();
417
418                {
419                    let _p = tracing::info_span!("resolve_imports loop").entered();
420
421                    'resolve_imports: loop {
422                        if self.resolve_imports() == ReachedFixedPoint::Yes {
423                            break 'resolve_imports;
424                        }
425                    }
426                }
427                if self.resolve_macros() == ReachedFixedPoint::Yes {
428                    break 'resolve_macros;
429                }
430
431                i += 1;
432                if i > FIXED_POINT_LIMIT {
433                    tracing::error!("name resolution is stuck");
434                    break 'resolve_attr;
435                }
436            }
437
438            if self.reseed_with_unresolved_attribute() == ReachedFixedPoint::Yes {
439                break 'resolve_attr;
440            }
441        }
442    }

444    fn collect(&mut self) {
445        let _p = tracing::info_span!("DefCollector::collect").entered();
446
447        self.resolution_loop();
448
449        let unresolved_imports = mem::take(&mut self.unresolved_imports);
450        // show unresolved imports in completion, etc
451        for directive in &unresolved_imports {
452            self.record_resolved_import(directive);
453        }
454        self.unresolved_imports = unresolved_imports;
455
456        if self.is_proc_macro {
457            // A crate exporting procedural macros is not allowed to export anything else.
458            //
459            // Additionally, while the proc macro entry points must be `pub`, they are not publicly
460            // exported in type/value namespace. This function reduces the visibility of all items
461            // in the crate root that aren't proc macros.
462            let module_id = self.def_map.root_module_id();
463            let root = &mut self.def_map.modules[module_id];
464            root.scope.censor_non_proc_macros(self.def_map.krate);
465        }
466    }

468    /// When the fixed-point loop reaches a stable state, we might still have
469    /// some unresolved attributes left over. This takes one of them, and feeds
470    /// the item it's applied to back into name resolution.
471    ///
472    /// This effectively ignores the fact that the macro is there and just treats the items as
473    /// normal code.
474    ///
475    /// This improves UX for unresolved attributes, and replicates the
476    /// behavior before we supported proc. attribute macros.
477    fn reseed_with_unresolved_attribute(&mut self) -> ReachedFixedPoint {
478        cov_mark::hit!(unresolved_attribute_fallback);
479
480        let unresolved_attr =
481            self.unresolved_macros.iter().enumerate().find_map(|(idx, directive)| match &directive
482                .kind
483            {
484                MacroDirectiveKind::Attr { ast_id, mod_item, attr_id, attr, tree, item_tree } => {
485                    self.def_map.diagnostics.push(DefDiagnostic::unresolved_macro_call(
486                        directive.module_id,
487                        MacroCallKind::Attr {
488                            ast_id: ast_id.ast_id,
489                            attr_args: None,
490                            censored_attr_ids: AttrMacroAttrIds::from_one(*attr_id),
491                        },
492                        (*attr.path).clone(),
493                    ));
494
495                    self.skip_attrs.insert(ast_id.ast_id.with_value(mod_item.ast_id()), *attr_id);
496
497                    Some((idx, directive, *mod_item, *tree, *item_tree))
498                }
499                _ => None,
500            });
501
502        match unresolved_attr {
503            Some((
504                pos,
505                &MacroDirective { module_id, depth, container, .. },
506                mod_item,
507                tree_id,
508                item_tree,
509            )) => {
510                // FIXME: Remove this clone
511                let mod_dir = self.mod_dirs[&module_id].clone();
512                ModCollector {
513                    def_collector: self,
514                    macro_depth: depth,
515                    module_id,
516                    tree_id,
517                    item_tree,
518                    mod_dir,
519                }
520                .collect(&[mod_item], container);
521
522                self.unresolved_macros.swap_remove(pos);
523                // Continue name resolution with the new data.
524                ReachedFixedPoint::No
525            }
526            None => ReachedFixedPoint::Yes,
527        }
528    }

530    fn inject_prelude(&mut self) {
531        // See compiler/rustc_builtin_macros/src/standard_library_imports.rs
532
533        if self.def_map.data.no_core {
534            // libcore does not get a prelude.
535            return;
536        }
537
538        let krate = if self.def_map.data.no_std {
539            Name::new_symbol_root(sym::core)
540        } else if self.local_def_map().extern_prelude().any(|(name, _)| *name == sym::std) {
541            Name::new_symbol_root(sym::std)
542        } else {
543            // If `std` does not exist for some reason, fall back to core. This mostly helps
544            // keep r-a's own tests minimal.
545            Name::new_symbol_root(sym::core)
546        };
547
548        let edition = match self.def_map.data.edition {
549            Edition::Edition2015 => Name::new_symbol_root(sym::rust_2015),
550            Edition::Edition2018 => Name::new_symbol_root(sym::rust_2018),
551            Edition::Edition2021 => Name::new_symbol_root(sym::rust_2021),
552            Edition::Edition2024 => Name::new_symbol_root(sym::rust_2024),
553        };
554
555        let path_kind = match self.def_map.data.edition {
556            Edition::Edition2015 => PathKind::Plain,
557            _ => PathKind::Abs,
558        };
559        let path = ModPath::from_segments(
560            path_kind,
561            [krate, Name::new_symbol_root(sym::prelude), edition],
562        );
563
564        let (per_ns, _) = self.def_map.resolve_path(
565            self.crate_local_def_map.unwrap_or(&self.local_def_map),
566            self.db,
567            self.def_map.root_module_id(),
568            &path,
569            BuiltinShadowMode::Other,
570            None,
571        );
572
573        match per_ns.types {
574            Some(Item { def: ModuleDefId::ModuleId(m), import, .. }) => {
575                self.def_map.prelude = Some((m, import.and_then(ImportOrExternCrate::use_)));
576            }
577            types => {
578                tracing::debug!(
579                    "could not resolve prelude path `{}` to module (resolved to {:?})",
580                    path.display(self.db, Edition::LATEST),
581                    types
582                );
583            }
584        }
585    }

591    /// Adds a definition of procedural macro `name` to the root module.
592    ///
593    /// # Notes on procedural macro resolution
594    ///
595    /// Procedural macro functionality is provided by the build system: It has to build the proc
596    /// macro and pass the resulting dynamic library to rust-analyzer.
597    ///
598    /// When procedural macro support is enabled, the list of proc macros exported by a crate is
599    /// known before we resolve names in the crate. This list is stored in `self.proc_macros` and is
600    /// derived from the dynamic library.
601    ///
602    /// However, we *also* would like to be able to at least *resolve* macros on our own, without
603    /// help by the build system. So, when the macro isn't found in `self.proc_macros`, we instead
604    /// use a dummy expander that always errors. This comes with the drawback of macros potentially
605    /// going out of sync with what the build system sees (since we resolve using VFS state, but
606    /// Cargo builds only on-disk files). We could and probably should add diagnostics for that.
607    fn export_proc_macro(&mut self, def: ProcMacroDef, ast_id: AstId<ast::Fn>, fn_id: FunctionId) {
608        let kind = def.kind.to_basedb_kind();
609        let (expander, kind) = match self.proc_macros.iter().find(|(n, _, _)| n == &def.name) {
610            Some(_)
611                if kind == hir_expand::proc_macro::ProcMacroKind::Attr
612                    && !self.db.expand_proc_attr_macros() =>
613            {
614                (CustomProcMacroExpander::disabled_proc_attr(), kind)
615            }
616            Some(&(_, _, true)) => (CustomProcMacroExpander::disabled(), kind),
617            Some(&(_, expander, false)) => (expander, kind),
618            None => (CustomProcMacroExpander::missing_expander(), kind),
619        };
620
621        let proc_macro_id = ProcMacroLoc {
622            container: self.def_map.root_module_id(),
623            id: ast_id,
624            expander,
625            kind,
626            edition: self.def_map.data.edition,
627        }
628        .intern(self.db);
629
630        self.def_map.macro_def_to_macro_id.insert(ast_id.erase(), proc_macro_id.into());
631        self.define_proc_macro(def.name.clone(), proc_macro_id);
632        let crate_data = Arc::get_mut(&mut self.def_map.data).unwrap();
633        if let ProcMacroKind::Derive { helpers } = def.kind {
634            crate_data.exported_derives.insert(proc_macro_id.into(), helpers);
635        }
636        crate_data.fn_proc_macro_mapping.insert(fn_id, proc_macro_id);
637        crate_data.fn_proc_macro_mapping_back.insert(proc_macro_id, fn_id);
638    }

640    /// Define a macro with `macro_rules`.
641    ///
642    /// It will define the macro in legacy textual scope, and if it has `#[macro_export]`,
643    /// then it is also defined in the root module scope.
644    /// You can `use` or invoke it by `crate::macro_name` anywhere, before or after the definition.
645    ///
646    /// It is surprising that the macro will never be in the current module scope.
647    /// These code fails with "unresolved import/macro",
648    /// ```rust,compile_fail
649    /// mod m { macro_rules! foo { () => {} } }
650    /// use m::foo as bar;
651    /// ```
652    ///
653    /// ```rust,compile_fail
654    /// macro_rules! foo { () => {} }
655    /// self::foo!();
656    /// crate::foo!();
657    /// ```
658    ///
659    /// Well, this code compiles, because the plain path `foo` in `use` is searched
660    /// in the legacy textual scope only.
661    /// ```rust
662    /// macro_rules! foo { () => {} }
663    /// use foo as bar;
664    /// ```
665    fn define_macro_rules(
666        &mut self,
667        module_id: ModuleId,
668        name: Name,
669        macro_: MacroRulesId,
670        export: bool,
671    ) {
672        // Textual scoping
673        self.define_legacy_macro(module_id, name.clone(), macro_.into());
674
675        // Module scoping
676        // In Rust, `#[macro_export]` macros are unconditionally visible at the
677        // crate root, even if the parent modules is **not** visible.
678        if export {
679            let module_id = self.def_map.root;
680            self.def_map.modules[module_id].scope.declare(macro_.into());
681            self.update(
682                module_id,
683                &[(Some(name), PerNs::macros(macro_.into(), Visibility::Public, None))],
684                Visibility::Public,
685                None,
686            );
687        }
688    }

746    /// Import exported macros from another crate. `names`, if `Some(_)`, specifies the name of
747    /// macros to be imported. Otherwise this method imports all exported macros.
748    ///
749    /// Exported macros are just all macros in the root module scope.
750    /// Note that it contains not only all `#[macro_export]` macros, but also all aliases
751    /// created by `use` in the root module, ignoring the visibility of `use`.
752    fn import_macros_from_extern_crate(
753        &mut self,
754        krate: Crate,
755        names: Option<Vec<Name>>,
756        extern_crate: Option<ExternCrateId>,
757    ) {
758        let def_map = crate_def_map(self.db, krate);
759        // `#[macro_use]` brings macros into macro_use prelude. Yes, even non-`macro_rules!`
760        // macros.
761        let root_scope = &def_map[def_map.root].scope;
762        match names {
763            Some(names) => {
764                for name in names {
765                    // FIXME: Report diagnostic on 404.
766                    if let Some(def) = root_scope.get(&name).take_macros() {
767                        self.def_map.macro_use_prelude.insert(name, (def, extern_crate));
768                    }
769                }
770            }
771            None => {
772                for (name, it) in root_scope.macros() {
773                    self.def_map.macro_use_prelude.insert(name.clone(), (it.def, extern_crate));
774                }
775            }
776        }

779    /// Tries to resolve every currently unresolved import.
780    fn resolve_imports(&mut self) -> ReachedFixedPoint {
781        let mut res = ReachedFixedPoint::Yes;
782        let imports = mem::take(&mut self.unresolved_imports);
783
784        self.unresolved_imports = imports
785            .into_iter()
786            .filter_map(|mut directive| {
787                directive.status = self.resolve_import(directive.module_id, &directive.import);
788                match directive.status {
789                    PartialResolvedImport::Indeterminate(resolved) => {
790                        self.record_resolved_import(&directive);
791                        self.indeterminate_imports.push((directive, resolved));
792                        res = ReachedFixedPoint::No;
793                        None
794                    }
795                    PartialResolvedImport::Resolved(_) => {
796                        self.record_resolved_import(&directive);
797                        res = ReachedFixedPoint::No;
798                        None
799                    }
800                    PartialResolvedImport::Unresolved => Some(directive),
801                }
802            })
803            .collect();
804
805        // Resolve all indeterminate resolved imports again
806        // As some of the macros will expand newly import shadowing partial resolved imports
807        // FIXME: We maybe could skip this, if we handle the indeterminate imports in `resolve_imports`
808        // correctly
809        let mut indeterminate_imports = std::mem::take(&mut self.indeterminate_imports);
810        indeterminate_imports.retain_mut(|(directive, partially_resolved)| {
811            let partially_resolved = partially_resolved.availability();
812            directive.status = self.resolve_import(directive.module_id, &directive.import);
813            match directive.status {
814                PartialResolvedImport::Indeterminate(import)
815                    if partially_resolved != import.availability() =>
816                {
817                    self.record_resolved_import(directive);
818                    res = ReachedFixedPoint::No;
819                    false
820                }
821                PartialResolvedImport::Resolved(_) => {
822                    self.record_resolved_import(directive);
823                    res = ReachedFixedPoint::No;
824                    false
825                }
826                _ => true,
827            }
828        });
829        self.indeterminate_imports = indeterminate_imports;
830
831        res
832    }

834    fn resolve_import(&self, module_id: ModuleId, import: &Import) -> PartialResolvedImport {
835        let _p = tracing::info_span!("resolve_import", import_path = %import.path.display(self.db, Edition::LATEST))
836            .entered();
837        tracing::debug!("resolving import: {:?} ({:?})", import, self.def_map.data.edition);
838        let ResolvePathResult { resolved_def, segment_index, reached_fixedpoint, prefix_info } =
839            self.def_map.resolve_path_fp_with_macro(
840                self.crate_local_def_map.unwrap_or(&self.local_def_map),
841                self.db,
842                ResolveMode::Import,
843                module_id,
844                &import.path,
845                BuiltinShadowMode::Module,
846                None, // An import may resolve to any kind of macro.
847            );
848
849        if reached_fixedpoint == ReachedFixedPoint::No
850            || resolved_def.is_none()
851            || segment_index.is_some()
852        {
853            return PartialResolvedImport::Unresolved;
854        }
855
856        if prefix_info.differing_crate {
857            return PartialResolvedImport::Resolved(
858                resolved_def.filter_visibility(|v| matches!(v, Visibility::Public)),
859            );
860        }
861
862        // Check whether all namespaces are resolved.
863        if resolved_def.is_full() {
864            PartialResolvedImport::Resolved(resolved_def)
865        } else {
866            PartialResolvedImport::Indeterminate(resolved_def)
867        }
868    }

870    fn record_resolved_import(&mut self, directive: &ImportDirective) {
871        let _p = tracing::info_span!("record_resolved_import").entered();
872
873        let module_id = directive.module_id;
874        let import = &directive.import;
875        let mut def = directive.status.namespaces();
876        let vis = self
877            .def_map
878            .resolve_visibility(
879                self.crate_local_def_map.unwrap_or(&self.local_def_map),
880                self.db,
881                module_id,
882                &directive.import.visibility,
883                false,
884            )
885            .unwrap_or(Visibility::Public);
886
887        match import.source {
888            ImportSource {
889                kind: kind @ (ImportKind::Plain | ImportKind::TypeOnly),
890                id,
891                use_tree,
892                ..
893            } => {
894                let name = match &import.alias {
895                    Some(ImportAlias::Alias(name)) => Some(name),
896                    Some(ImportAlias::Underscore) => None,
897                    None => match import.path.segments().last() {
898                        Some(last_segment) => Some(last_segment),
899                        None => {
900                            cov_mark::hit!(bogus_paths);
901                            return;
902                        }
903                    },
904                };
905
906                if kind == ImportKind::TypeOnly {
907                    def.values = None;
908                    def.macros = None;
909                }
910                let imp = ImportOrExternCrate::Import(ImportId { use_: id, idx: use_tree });
911                tracing::debug!("resolved import {:?} ({:?}) to {:?}", name, import, def);
912
913                // `extern crate crate_name` things can be re-exported as `pub use crate_name`.
914                // But they cannot be re-exported as `pub use self::crate_name`, `pub use crate::crate_name`
915                // or `pub use ::crate_name`.
916                //
917                // This has been historically allowed, but may be not allowed in future
918                // https://github.com/rust-lang/rust/issues/127909
919                if let Some(def) = def.types.as_mut() {
920                    let is_extern_crate_reimport_without_prefix = || {
921                        let Some(ImportOrExternCrate::ExternCrate(_)) = def.import else {
922                            return false;
923                        };
924                        if kind == ImportKind::Glob {
925                            return false;
926                        }
927                        matches!(import.path.kind, PathKind::Plain | PathKind::SELF)
928                            && import.path.segments().len() < 2
929                    };
930                    if is_extern_crate_reimport_without_prefix() {
931                        def.vis = vis;
932                    }
933                }
934
935                self.update(module_id, &[(name.cloned(), def)], vis, Some(imp));
936            }
937            ImportSource { kind: ImportKind::Glob, id, is_prelude, use_tree, .. } => {
938                tracing::debug!("glob import: {:?}", import);
939                let glob = GlobId { use_: id, idx: use_tree };
940                match def.take_types() {
941                    Some(ModuleDefId::ModuleId(m)) => {
942                        if is_prelude {
943                            // Note: This dodgily overrides the injected prelude. The rustc
944                            // implementation seems to work the same though.
945                            cov_mark::hit!(std_prelude);
946                            self.def_map.prelude = Some((m, Some(id)));
947                        } else if m.krate(self.db) != self.def_map.krate {
948                            cov_mark::hit!(glob_across_crates);
949                            // glob import from other crate => we can just import everything once
950                            let scope = &m.def_map(self.db)[m].scope;
951
952                            // Module scoped macros is included
953                            let items = scope
954                                .resolutions()
955                                // only keep visible names...
956                                .map(|(n, res)| {
957                                    (n, res.filter_visibility(|v| v.is_visible_from_other_crate()))
958                                })
959                                .filter(|(_, res)| !res.is_none())
960                                .collect::<Vec<_>>();
961
962                            self.update(
963                                module_id,
964                                &items,
965                                vis,
966                                Some(ImportOrExternCrate::Glob(glob)),
967                            );
968                        } else {
969                            // glob import from same crate => we do an initial
970                            // import, and then need to propagate any further
971                            // additions
972                            let scope = if m.block(self.db) == self.def_map.block_id() {
973                                &self.def_map[m].scope
974                            } else {
975                                &m.def_map(self.db)[m].scope
976                            };
977
978                            // Module scoped macros is included
979                            let items = scope
980                                .resolutions()
981                                // only keep visible names...
982                                .map(|(n, res)| {
983                                    (
984                                        n,
985                                        res.filter_visibility(|v| {
986                                            v.is_visible_from_def_map(
987                                                self.db,
988                                                &self.def_map,
989                                                module_id,
990                                            )
991                                        }),
992                                    )
993                                })
994                                .filter(|(_, res)| !res.is_none())
995                                .collect::<Vec<_>>();
996
997                            self.update(
998                                module_id,
999                                &items,
1000                                vis,
1001                                Some(ImportOrExternCrate::Glob(glob)),
1002                            );
1003                            // record the glob import in case we add further items
1004                            let glob_imports = self.glob_imports.entry(m).or_default();
1005                            match glob_imports.iter_mut().find(|(mid, _, _)| *mid == module_id) {
1006                                None => glob_imports.push((module_id, vis, glob)),
1007                                Some((_, old_vis, _)) => {
1008                                    if let Some(new_vis) = old_vis.max(self.db, vis, &self.def_map)
1009                                    {
1010                                        *old_vis = new_vis;
1011                                    }
1012                                }
1013                            }
1014                        }
1015                    }
1016                    Some(ModuleDefId::AdtId(AdtId::EnumId(e))) => {
1017                        cov_mark::hit!(glob_enum);
1018                        // glob import from enum => just import all the variants
1019                        let resolutions = e
1020                            .enum_variants(self.db)
1021                            .variants
1022                            .iter()
1023                            .map(|&(variant, ref name, _)| {
1024                                let res = PerNs::both(variant.into(), variant.into(), vis, None);
1025                                (Some(name.clone()), res)
1026                            })
1027                            .collect::<Vec<_>>();
1028                        self.update(
1029                            module_id,
1030                            &resolutions,
1031                            vis,
1032                            Some(ImportOrExternCrate::Glob(glob)),
1033                        );
1034                    }
1035                    Some(ModuleDefId::TraitId(it)) => {
1036                        // FIXME: Implement this correctly
1037                        // We can't actually call `trait_items`, the reason being that if macro calls
1038                        // occur, they will call back into the def map which we might be computing right
1039                        // now resulting in a cycle.
1040                        // To properly implement this, trait item collection needs to be done in def map
1041                        // collection...
1042                        let resolutions = if true {
1043                            vec![]
1044                        } else {
1045                            TraitItems::query(self.db, it)
1046                                .items
1047                                .iter()
1048                                .map(|&(ref name, variant)| {
1049                                    let res = match variant {
1050                                        AssocItemId::FunctionId(it) => {
1051                                            PerNs::values(it.into(), vis, None)
1052                                        }
1053                                        AssocItemId::ConstId(it) => {
1054                                            PerNs::values(it.into(), vis, None)
1055                                        }
1056                                        AssocItemId::TypeAliasId(it) => {
1057                                            PerNs::types(it.into(), vis, None)
1058                                        }
1059                                    };
1060                                    (Some(name.clone()), res)
1061                                })
1062                                .collect::<Vec<_>>()
1063                        };
1064                        self.update(
1065                            module_id,
1066                            &resolutions,
1067                            vis,
1068                            Some(ImportOrExternCrate::Glob(glob)),
1069                        );
1070                    }
1071                    Some(d) => {
1072                        tracing::debug!("glob import {:?} from non-module/enum {:?}", import, d);
1073                    }
1074                    None => {
1075                        tracing::debug!("glob import {:?} didn't resolve as type", import);
1076                    }
1077                }
1078            }
1079        }
1080    }

1083        &mut self,
1084        // The module for which `resolutions` have been resolve
1085        module_id: ModuleId,
1086        resolutions: &[(Option<Name>, PerNs)],
1087        // Visibility this import will have

1090    ) {

1094    fn update_recursive(
1095        &mut self,
1096        // The module for which `resolutions` have been resolved.
1097        module_id: ModuleId,
1098        resolutions: &[(Option<Name>, PerNs)],
1099        // All resolutions are imported with this visibility; the visibilities in
1100        // the `PerNs` values are ignored and overwritten
1101        vis: Visibility,
1102        import: Option<ImportOrExternCrate>,
1103        depth: usize,
1104    ) {
1105        if depth > GLOB_RECURSION_LIMIT {
1106            // prevent stack overflows (but this shouldn't be possible)
1107            panic!("infinite recursion in glob imports!");
1108        }
1109        let mut changed = false;
1110
1111        for (name, res) in resolutions {
1112            match name {
1113                Some(name) => {
1114                    changed |=
1115                        self.push_res_and_update_glob_vis(module_id, name, *res, vis, import);
1116                }
1117                None => {
1118                    let (tr, import) = match res.take_types_full() {
1119                        Some(Item { def: ModuleDefId::TraitId(tr), vis: _, import }) => {
1120                            (tr, import)
1121                        }
1122                        Some(other) => {
1123                            tracing::debug!("non-trait `_` import of {:?}", other);
1124                            continue;
1125                        }
1126                        None => continue,
1127                    };
1128                    let old_vis = self.def_map.modules[module_id].scope.unnamed_trait_vis(tr);
1129                    let should_update = match old_vis {
1130                        None => true,
1131                        Some(old_vis) => {
1132                            let max_vis = old_vis.max(self.db, vis, &self.def_map).unwrap_or_else(|| {
1133                                panic!("`Tr as _` imports with unrelated visibilities {old_vis:?} and {vis:?} (trait {tr:?})");
1134                            });
1135
1136                            if max_vis == old_vis {
1137                                false
1138                            } else {
1139                                cov_mark::hit!(upgrade_underscore_visibility);
1140                                true
1141                            }
1142                        }
1143                    };
1144
1145                    if should_update {
1146                        changed = true;
1147                        self.def_map.modules[module_id].scope.push_unnamed_trait(
1148                            tr,
1149                            vis,
1150                            import.and_then(ImportOrExternCrate::import),
1151                        );
1152                    }
1153                }
1154            }
1155        }
1156
1157        if !changed {
1158            return;
1159        }
1160        let glob_imports = self
1161            .glob_imports
1162            .get(&module_id)
1163            .into_iter()
1164            .flatten()
1165            .filter(|(glob_importing_module, _, _)| {
1166                // we know all resolutions have the same visibility (`vis`), so we
1167                // just need to check that once
1168                vis.is_visible_from_def_map(self.db, &self.def_map, *glob_importing_module)
1169            })
1170            .cloned()
1171            .collect::<Vec<_>>();
1172
1173        for (glob_importing_module, glob_import_vis, glob) in glob_imports {
1174            let vis = glob_import_vis.min(self.db, vis, &self.def_map).unwrap_or(glob_import_vis);
1175            self.update_recursive(
1176                glob_importing_module,
1177                resolutions,
1178                vis,
1179                Some(ImportOrExternCrate::Glob(glob)),
1180                depth + 1,
1181            );
1182        }
1183    }

1185    fn push_res_and_update_glob_vis(
1186        &mut self,
1187        module_id: ModuleId,
1188        name: &Name,
1189        mut defs: PerNs,
1190        vis: Visibility,
1191        def_import_type: Option<ImportOrExternCrate>,
1192    ) -> bool {
1193        if let Some(def) = defs.types.as_mut() {
1194            def.vis = def.vis.min(self.db, vis, &self.def_map).unwrap_or(vis);
1195        }
1196        if let Some(def) = defs.values.as_mut() {
1197            def.vis = def.vis.min(self.db, vis, &self.def_map).unwrap_or(vis);
1198        }
1199        if let Some(def) = defs.macros.as_mut() {
1200            def.vis = def.vis.min(self.db, vis, &self.def_map).unwrap_or(vis);
1201        }
1202
1203        let mut changed = false;
1204
1205        if let Some(ImportOrExternCrate::Glob(_)) = def_import_type {
1206            let prev_defs = self.def_map[module_id].scope.get(name);
1207
1208            // Multiple globs may import the same item and they may override visibility from
1209            // previously resolved globs. Handle overrides here and leave the rest to
1210            // `ItemScope::push_res_with_import()`.
1211            if let Some(def) = defs.types
1212                && let Some(prev_def) = prev_defs.types
1213            {
1214                if def.def == prev_def.def
1215                    && self.from_glob_import.contains_type(module_id, name.clone())
1216                    && def.vis != prev_def.vis
1217                    && def.vis.max(self.db, prev_def.vis, &self.def_map) == Some(def.vis)
1218                {
1219                    changed = true;
1220                    // This import is being handled here, don't pass it down to
1221                    // `ItemScope::push_res_with_import()`.
1222                    defs.types = None;
1223                    self.def_map.modules[module_id].scope.update_visibility_types(name, def.vis);
1224                }
1225                // When the source module's definition changed (e.g., due to an explicit import
1226                // shadowing a glob), propagate the new definition to modules that glob-import from it.
1227                // We check that the previous definition came from the same glob import to avoid
1228                // incorrectly overwriting definitions from different glob sources.
1229                //
1230                // Note this is not a perfect fix, but it makes
1231                // https://github.com/rust-lang/rust-analyzer/issues/19224 work for now until we
1232                // implement a proper glob graph
1233                else if def.def != prev_def.def && prev_def.import == def_import_type {
1234                    changed = true;
1235                    defs.types = None;
1236                    self.def_map.modules[module_id].scope.update_def_types(name, def.def, def.vis);
1237                }
1238            }
1239
1240            if let Some(def) = defs.values
1241                && let Some(prev_def) = prev_defs.values
1242            {
1243                if def.def == prev_def.def
1244                    && self.from_glob_import.contains_value(module_id, name.clone())
1245                    && def.vis != prev_def.vis
1246                    && def.vis.max(self.db, prev_def.vis, &self.def_map) == Some(def.vis)
1247                {
1248                    changed = true;
1249                    defs.values = None;
1250                    self.def_map.modules[module_id].scope.update_visibility_values(name, def.vis);
1251                } else if def.def != prev_def.def
1252                    && prev_def.import.map(ImportOrExternCrate::from) == def_import_type
1253                {
1254                    changed = true;
1255                    defs.values = None;
1256                    self.def_map.modules[module_id].scope.update_def_values(name, def.def, def.vis);
1257                }
1258            }
1259
1260            if let Some(def) = defs.macros
1261                && let Some(prev_def) = prev_defs.macros
1262            {
1263                if def.def == prev_def.def
1264                    && self.from_glob_import.contains_macro(module_id, name.clone())
1265                    && def.vis != prev_def.vis
1266                    && def.vis.max(self.db, prev_def.vis, &self.def_map) == Some(def.vis)
1267                {
1268                    changed = true;
1269                    defs.macros = None;
1270                    self.def_map.modules[module_id].scope.update_visibility_macros(name, def.vis);
1271                } else if def.def != prev_def.def && prev_def.import == def_import_type {
1272                    changed = true;
1273                    defs.macros = None;
1274                    self.def_map.modules[module_id].scope.update_def_macros(name, def.def, def.vis);
1275                }
1276            }
1277        }
1278
1279        changed |= self.def_map.modules[module_id].scope.push_res_with_import(
1280            &mut self.from_glob_import,
1281            (module_id, name.clone()),
1282            defs,
1283            def_import_type,
1284        );
1285
1286        changed
1287    }

1291        let mut resolved = Vec::new();

1293            let attr_macro_item = match &directive.kind {
1294                MacroDirectiveKind::Attr { ast_id, .. } => Some(ast_id.ast_id),
1295                MacroDirectiveKind::FnLike { .. } | MacroDirectiveKind::Derive { .. } => None,
1296            };
1297            resolved.push((
1298                directive.module_id,
1299                directive.depth,
1300                directive.container,
1301                call_id,
1302                attr_macro_item,
1303            ));
1304        };
1305

1307        enum Resolved {
1308            Yes,
1309            No,
1310        }
1311

1313        let mut res = ReachedFixedPoint::Yes;

1315        let mut retain = |directive: &MacroDirective<'db>| {
1316            let subns = match &directive.kind {
1317                MacroDirectiveKind::FnLike { .. } => MacroSubNs::Bang,
1318                MacroDirectiveKind::Attr { .. } | MacroDirectiveKind::Derive { .. } => {
1319                    MacroSubNs::Attr
1320                }
1321            };

1323                let resolved_res = def_map.resolve_path_fp_with_macro(
1324                    self.crate_local_def_map.unwrap_or(&self.local_def_map),
1325                    self.db,
1326                    ResolveMode::Other,
1327                    directive.module_id,
1328                    path,
1329                    BuiltinShadowMode::Module,
1330                    Some(subns),
1331                );
1332                resolved_res.resolved_def.take_macros().map(|it| (it, self.db.macro_def(it)))
1333            };

1335

1337                MacroDirectiveKind::FnLike { ast_id, expand_to, ctxt: call_site } => {
1338                    let call_id = macro_call_as_call_id(
1339                        self.db,
1340                        ast_id.ast_id,
1341                        &ast_id.path,
1342                        *call_site,
1343                        *expand_to,
1344                        self.def_map.krate,
1345                        resolver_def_id,
1346                        &mut |ptr, call_id| {
1347                            eager_callback_buffer.push((directive.module_id, ptr, call_id));
1348                        },
1349                    );
1350                    if let Ok(call_id) = call_id {
1351                        // FIXME: Expansion error
1352                        if let Some(call_id) = call_id.value {
1353                            self.def_map.modules[directive.module_id]
1354                                .scope
1355                                .add_macro_invoc(ast_id.ast_id, call_id);
1356
1357                            push_resolved(&mut resolved, directive, call_id);
1358
1359                            res = ReachedFixedPoint::No;
1360                            return Resolved::Yes;
1361                        }
1362                    }
1363                }
1364                MacroDirectiveKind::Derive {
1365                    ast_id,
1366                    derive_attr,
1367                    derive_pos,
1368                    ctxt: call_site,
1369                    derive_macro_id,
1370                } => {
1371                    // FIXME: This code is almost duplicate below.
1372                    let id = derive_macro_as_call_id(
1373                        self.db,
1374                        ast_id,
1375                        *derive_attr,
1376                        *derive_pos as u32,
1377                        *call_site,
1378                        self.def_map.krate,
1379                        |path| resolver(&self.def_map, path),
1380                        *derive_macro_id,
1381                    );
1382
1383                    if let Ok((macro_id, def_id, call_id)) = id {
1384                        self.def_map.modules[directive.module_id].scope.set_derive_macro_invoc(
1385                            ast_id.ast_id,
1386                            Either::Left(call_id),
1387                            *derive_attr,
1388                            *derive_pos,
1389                        );
1390                        // Record its helper attributes.
1391                        if def_id.krate != self.def_map.krate {
1392                            let def_map = crate_def_map(self.db, def_id.krate);
1393                            if let Some(helpers) = def_map.data.exported_derives.get(&macro_id) {
1394                                self.def_map
1395                                    .derive_helpers_in_scope
1396                                    .entry(ast_id.ast_id.map(|it| it.upcast()))
1397                                    .or_default()
1398                                    .extend(izip!(
1399                                        helpers.iter().cloned(),
1400                                        iter::repeat(macro_id),
1401                                        iter::repeat(Either::Left(call_id)),
1402                                    ));
1403                            }
1404                        }
1405
1406                        push_resolved(&mut resolved, directive, call_id);
1407
1408                        res = ReachedFixedPoint::No;
1409                        return Resolved::Yes;
1410                    }
1411                }
1412                MacroDirectiveKind::Attr {
1413                    ast_id: file_ast_id,
1414                    mod_item,
1415                    attr_id,
1416                    attr,
1417                    tree,
1418                    item_tree,
1419                } => {
1420                    let &AstIdWithPath { ast_id, ref path } = file_ast_id;
1421                    let file_id = ast_id.file_id;
1422
1423                    let mut recollect_without = |collector: &mut Self| {
1424                        // Remove the original directive since we resolved it.
1425                        let mod_dir = collector.mod_dirs[&directive.module_id].clone();
1426                        collector
1427                            .skip_attrs
1428                            .insert(InFile::new(file_id, mod_item.ast_id()), *attr_id);
1429
1430                        ModCollector {
1431                            def_collector: collector,
1432                            macro_depth: directive.depth,
1433                            module_id: directive.module_id,
1434                            tree_id: *tree,
1435                            item_tree,
1436                            mod_dir,
1437                        }
1438                        .collect(&[*mod_item], directive.container);
1439                        res = ReachedFixedPoint::No;
1440                        Resolved::Yes
1441                    };
1442
1443                    if let Some(ident) = path.as_ident()
1444                        && let Some(helpers) = self.def_map.derive_helpers_in_scope.get(&ast_id)
1445                        && helpers.iter().any(|(it, ..)| it == ident)
1446                    {
1447                        cov_mark::hit!(resolved_derive_helper);
1448                        // Resolved to derive helper. Collect the item's attributes again,
1449                        // starting after the derive helper.
1450                        return recollect_without(self);
1451                    }
1452
1453                    let def = match resolver_def_id(path) {
1454                        Some(def) if def.is_attribute() => def,
1455                        _ => return Resolved::No,
1456                    };
1457
1458                    // Skip #[test]/#[bench]/#[test_case] expansion, which would merely result in more memory usage
1459                    // due to duplicating functions into macro expansions, but only if `cfg(test)` is active,
1460                    // otherwise they are expanded to nothing and this can impact e.g. diagnostics (due to things
1461                    // being cfg'ed out).
1462                    // Ideally we will just expand them to nothing here. But we are only collecting macro calls,
1463                    // not expanding them, so we have no way to do that.
1464                    if matches!(
1465                        def.kind,
1466                        MacroDefKind::BuiltInAttr(_, expander)
1467                        if expander.is_test() || expander.is_bench() || expander.is_test_case()
1468                    ) {
1469                        let test_is_active = self.cfg_options.check_atom(&CfgAtom::Flag(sym::test));
1470                        if test_is_active {
1471                            return recollect_without(self);
1472                        }
1473                    }
1474
1475                    let mut call_id = || {
1476                        let active_attrs = self.prev_active_attrs.entry(ast_id).or_default();
1477                        active_attrs.push(*attr_id);
1478
1479                        attr_macro_as_call_id(
1480                            self.db,
1481                            file_ast_id,
1482                            attr,
1483                            AttrMacroAttrIds::from_many(active_attrs),
1484                            self.def_map.krate,
1485                            def,
1486                        )
1487                    };
1488                    if matches!(def,
1489                        MacroDefId { kind: MacroDefKind::BuiltInAttr(_, exp), .. }
1490                        if exp.is_derive()
1491                    ) {
1492                        // Resolved to `#[derive]`, we don't actually expand this attribute like
1493                        // normal (as that would just be an identity expansion with extra output)
1494                        // Instead we treat derive attributes special and apply them separately.
1495
1496                        let ast_adt_id: FileAstId<ast::Adt> = match *mod_item {
1497                            ModItemId::Struct(ast_id) => ast_id.upcast(),
1498                            ModItemId::Union(ast_id) => ast_id.upcast(),
1499                            ModItemId::Enum(ast_id) => ast_id.upcast(),
1500                            _ => {
1501                                let diag = DefDiagnostic::invalid_derive_target(
1502                                    directive.module_id,
1503                                    ast_id,
1504                                    *attr_id,
1505                                );
1506                                self.def_map.diagnostics.push(diag);
1507                                return recollect_without(self);
1508                            }
1509                        };
1510
1511                        let ast_id = ast_id.with_value(ast_adt_id);
1512
1513                        let mut derive_call_ids = SmallVec::new();
1514                        match attr.parse_path_comma_token_tree(self.db) {
1515                            Some(derive_macros) => {
1516                                let call_id = call_id();
1517                                for (idx, (path, call_site, _)) in derive_macros.enumerate() {
1518                                    let ast_id = AstIdWithPath::new(
1519                                        file_id,
1520                                        ast_id.value,
1521                                        Interned::new(path),
1522                                    );
1523
1524                                    derive_call_ids.push(None);
1525
1526                                    // Try to resolve the derive immediately. If we succeed, we can also use the fast path
1527                                    // for builtin derives. If not, we cannot use it, as it can cause the ADT to become
1528                                    // interned while the derive is still unresolved, which will cause it to get forgotten.
1529                                    let id = derive_macro_as_call_id(
1530                                        self.db,
1531                                        &ast_id,
1532                                        *attr_id,
1533                                        idx as u32,
1534                                        call_site.ctx,
1535                                        self.def_map.krate,
1536                                        |path| resolver(&self.def_map, path),
1537                                        call_id,
1538                                    );
1539
1540                                    let ast_id_without_path = ast_id.ast_id;
1541                                    let directive = MacroDirective {
1542                                        module_id: directive.module_id,
1543                                        depth: directive.depth + 1,
1544                                        kind: MacroDirectiveKind::Derive {
1545                                            ast_id,
1546                                            derive_attr: *attr_id,
1547                                            derive_pos: idx,
1548                                            ctxt: call_site.ctx,
1549                                            derive_macro_id: call_id,
1550                                        },
1551                                        container: directive.container,
1552                                    };
1553
1554                                    if let Ok((macro_id, def_id, call_id)) = id {
1555                                        let (mut helpers_start, mut helpers_end) = (0, 0);
1556                                        // Record its helper attributes.
1557                                        if def_id.krate != self.def_map.krate {
1558                                            let def_map = crate_def_map(self.db, def_id.krate);
1559                                            if let Some(helpers) =
1560                                                def_map.data.exported_derives.get(&macro_id)
1561                                            {
1562                                                let derive_helpers = self
1563                                                    .def_map
1564                                                    .derive_helpers_in_scope
1565                                                    .entry(
1566                                                        ast_id_without_path.map(|it| it.upcast()),
1567                                                    )
1568                                                    .or_default();
1569                                                helpers_start = derive_helpers.len();
1570                                                derive_helpers.extend(izip!(
1571                                                    helpers.iter().cloned(),
1572                                                    iter::repeat(macro_id),
1573                                                    iter::repeat(Either::Left(call_id)),
1574                                                ));
1575                                                helpers_end = derive_helpers.len();
1576                                            }
1577                                        }
1578
1579                                        if super::enable_builtin_derive_fast_path()
1580                                            && let MacroDefKind::BuiltInDerive(_, builtin_derive) =
1581                                                def_id.kind
1582                                        {
1583                                            self.deferred_builtin_derives
1584                                                .entry(ast_id_without_path.upcast())
1585                                                .or_default()
1586                                                .push(DeferredBuiltinDerive {
1587                                                    call_id,
1588                                                    derive: builtin_derive,
1589                                                    module_id: directive.module_id,
1590                                                    container: directive.container,
1591                                                    depth: directive.depth,
1592                                                    derive_attr_id: *attr_id,
1593                                                    derive_index: idx as u32,
1594                                                    helpers_range: helpers_start..helpers_end,
1595                                                });
1596                                        } else {
1597                                            push_resolved(&mut resolved, &directive, call_id);
1598                                            *derive_call_ids.last_mut().unwrap() =
1599                                                Some(Either::Left(call_id));
1600                                        }
1601                                    } else {
1602                                        self.unresolved_macros.push(directive);
1603                                    }
1604                                }
1605
1606                                // We treat the #[derive] macro as an attribute call, but we do not resolve it for nameres collection.
1607                                // This is just a trick to be able to resolve the input to derives
1608                                // as proper paths in `Semantics`.
1609                                // Check the comment in [`builtin_attr_macro`].
1610                                self.def_map.modules[directive.module_id]
1611                                    .scope
1612                                    .init_derive_attribute(
1613                                        ast_id,
1614                                        *attr_id,
1615                                        call_id,
1616                                        derive_call_ids,
1617                                    );
1618                            }
1619                            None => {
1620                                let diag = DefDiagnostic::malformed_derive(
1621                                    directive.module_id,
1622                                    ast_id,
1623                                    *attr_id,
1624                                );
1625                                self.def_map.diagnostics.push(diag);
1626                            }
1627                        }
1628
1629                        return recollect_without(self);
1630                    }
1631
1632                    if let MacroDefKind::ProcMacro(_, exp, _) = def.kind {
1633                        // If there's no expander for the proc macro (e.g.
1634                        // because proc macros are disabled, or building the
1635                        // proc macro crate failed), report this and skip
1636                        // expansion like we would if it was disabled
1637                        if let Some(err) = exp.as_expand_error(def.krate) {
1638                            self.def_map.diagnostics.push(DefDiagnostic::macro_error(
1639                                directive.module_id,
1640                                ast_id,
1641                                (**path).clone(),
1642                                err,
1643                            ));
1644                            return recollect_without(self);
1645                        }
1646                    }
1647
1648                    // Clear deferred derives for this item, unfortunately we cannot use them due to the attribute.
1649                    if let Some(deferred_derives) = self.deferred_builtin_derives.remove(&ast_id) {
1650                        resolved.extend(deferred_derives.into_iter().map(|derive| {
1651                            (
1652                                derive.module_id,
1653                                derive.depth,
1654                                derive.container,
1655                                derive.call_id,
1656                                Some(ast_id),
1657                            )
1658                        }));
1659                    }
1660
1661                    let call_id = call_id();
1662                    self.def_map.modules[directive.module_id]
1663                        .scope
1664                        .add_attr_macro_invoc(ast_id, call_id);
1665
1666                    push_resolved(&mut resolved, directive, call_id);
1667                    res = ReachedFixedPoint::No;
1668                    return Resolved::Yes;
1669                }
1670            }
1671
1672            Resolved::No
1673        };
1674        macros.retain(|it| retain(it) == Resolved::No);
1675        // Attribute resolution can add unresolved macro invocations, so concatenate the lists.
1676        macros.extend(mem::take(&mut self.unresolved_macros));
1677        self.unresolved_macros = macros;
1678
1679        for (module_id, ptr, call_id) in eager_callback_buffer {
1680            self.def_map.modules[module_id].scope.add_macro_invoc(ptr.map(|(_, it)| it), call_id);
1681        }
1682
1683        for (module_id, depth, container, macro_call_id, attr_macro_item) in resolved {
1684            self.collect_macro_expansion(
1685                module_id,
1686                macro_call_id,
1687                depth,
1688                container,
1689                attr_macro_item,
1690            );
1691        }
1692
1693        res
1694    }

1696    fn collect_macro_expansion(
1697        &mut self,
1698        module_id: ModuleId,
1699        macro_call_id: MacroCallId,
1700        depth: usize,
1701        container: ItemContainerId,
1702        attr_macro_item: Option<AstId<ast::Item>>,
1703    ) {
1704        if depth > self.def_map.recursion_limit() as usize {
1705            cov_mark::hit!(macro_expansion_overflow);
1706            tracing::warn!("macro expansion is too deep");
1707            return;
1708        }
1709        let file_id = macro_call_id.into();
1710
1711        let item_tree = self.db.file_item_tree(file_id);
1712
1713        // Derive helpers that are in scope for an item are also in scope for attribute macro expansions
1714        // of that item (but not derive or fn like macros).
1715        // FIXME: This is a hack. The proper way to do this is by having a chain of derive helpers scope,
1716        // where the next scope in the chain is the parent hygiene context of the span. Unfortunately
1717        // it's difficult to implement with our current name resolution and hygiene system.
1718        // This hack is also incorrect since it ignores item in blocks. But the main reason to bring derive
1719        // helpers into scope in this case is to help with:
1720        // ```
1721        // #[derive(DeriveWithHelper)]
1722        // #[helper]
1723        // #[attr_macro]
1724        // struct Foo;
1725        // ```
1726        // Where `attr_macro`'s input will include `#[helper]` but not the derive, and it will likely therefore
1727        // also include it in its output. Therefore I hope not supporting blocks is fine at least for now.
1728        if let Some(attr_macro_item) = attr_macro_item
1729            && let Some(derive_helpers) = self.def_map.derive_helpers_in_scope.get(&attr_macro_item)
1730        {
1731            let derive_helpers = derive_helpers.clone();
1732            for item in item_tree.top_level_items() {
1733                self.def_map
1734                    .derive_helpers_in_scope
1735                    .entry(InFile::new(file_id, item.ast_id()))
1736                    .or_default()
1737                    .extend(derive_helpers.iter().cloned());
1738            }
1739        }
1740
1741        let mod_dir = if macro_call_id.is_include_macro(self.db) {
1742            ModDir::root()
1743        } else {
1744            self.mod_dirs[&module_id].clone()
1745        };
1746
1747        ModCollector {
1748            def_collector: &mut *self,
1749            macro_depth: depth,
1750            tree_id: TreeId::new(file_id, None),
1751            module_id,
1752            item_tree,
1753            mod_dir,
1754        }
1755        .collect(item_tree.top_level_items(), container);
1756    }

1758    fn finish(mut self) -> (DefMap, LocalDefMap) {
1759        // Emit diagnostics for all remaining unexpanded macros.
1760        let _p = tracing::info_span!("DefCollector::finish").entered();
1761
1762        for directive in &self.unresolved_macros {
1763            match &directive.kind {
1764                MacroDirectiveKind::FnLike { ast_id, expand_to, ctxt: call_site } => {
1765                    // FIXME: we shouldn't need to re-resolve the macro here just to get the unresolved error!
1766                    let macro_call_as_call_id = macro_call_as_call_id(
1767                        self.db,
1768                        ast_id.ast_id,
1769                        &ast_id.path,
1770                        *call_site,
1771                        *expand_to,
1772                        self.def_map.krate,
1773                        |path| {
1774                            let resolved_res = self.def_map.resolve_path_fp_with_macro(
1775                                self.crate_local_def_map.unwrap_or(&self.local_def_map),
1776                                self.db,
1777                                ResolveMode::Other,
1778                                directive.module_id,
1779                                path,
1780                                BuiltinShadowMode::Module,
1781                                Some(MacroSubNs::Bang),
1782                            );
1783                            resolved_res.resolved_def.take_macros().map(|it| self.db.macro_def(it))
1784                        },
1785                        &mut |_, _| (),
1786                    );
1787                    if let Err(UnresolvedMacro { path }) = macro_call_as_call_id {
1788                        self.def_map.diagnostics.push(DefDiagnostic::unresolved_macro_call(
1789                            directive.module_id,
1790                            MacroCallKind::FnLike {
1791                                ast_id: ast_id.ast_id,
1792                                expand_to: *expand_to,
1793                                eager: None,
1794                            },
1795                            path,
1796                        ));
1797                    }
1798                }
1799                MacroDirectiveKind::Derive {
1800                    ast_id,
1801                    derive_attr,
1802                    derive_pos,
1803                    derive_macro_id,
1804                    ..
1805                } => {
1806                    self.def_map.diagnostics.push(DefDiagnostic::unresolved_macro_call(
1807                        directive.module_id,
1808                        MacroCallKind::Derive {
1809                            ast_id: ast_id.ast_id,
1810                            derive_attr_index: *derive_attr,
1811                            derive_index: *derive_pos as u32,
1812                            derive_macro_id: *derive_macro_id,
1813                        },
1814                        (*ast_id.path).clone(),
1815                    ));
1816                }
1817                // These are diagnosed by `reseed_with_unresolved_attribute`, as that function consumes them
1818                MacroDirectiveKind::Attr { .. } => {}
1819            }
1820        }
1821
1822        // Emit diagnostics for all remaining unresolved imports.
1823        for import in &self.unresolved_imports {
1824            let &ImportDirective {
1825                module_id,
1826                import:
1827                    Import {
1828                        ref path,
1829                        source: ImportSource { use_tree, id, is_prelude: _, kind: _ },
1830                        ..
1831                    },
1832                ..
1833            } = import;
1834            if matches!(
1835                (path.segments().first(), &path.kind),
1836                (Some(krate), PathKind::Plain | PathKind::Abs) if self.unresolved_extern_crates.contains(krate)
1837            ) {
1838                continue;
1839            }
1840            let item_tree_id = id.lookup(self.db).id;
1841            self.def_map.diagnostics.push(DefDiagnostic::unresolved_import(
1842                module_id,
1843                item_tree_id,
1844                use_tree,
1845            ));
1846        }
1847
1848        always!(
1849            self.deferred_builtin_derives.is_empty(),
1850            "self.deferred_builtin_derives={:#?}",
1851            self.deferred_builtin_derives,
1852        );
1853
1854        (self.def_map, self.local_def_map)
1855    }
1856}

1873    fn collect(&mut self, items: &[ModItemId], container: ItemContainerId) {

1875        let is_crate_root = self.module_id == self.def_collector.def_map.root
1876            && self.def_collector.def_map.block.is_none();
1877
1878        // Note: don't assert that inserted value is fresh: it's simply not true
1879        // for macros.
1880        self.def_collector.mod_dirs.insert(self.module_id, self.mod_dir.clone());
1881
1882        // Prelude module is always considered to be `#[macro_use]`.
1883        if let Some((prelude_module, _use)) = self.def_collector.def_map.prelude {
1884            // Don't insert macros from the prelude into blocks, as they can be shadowed by other macros.
1885            if is_crate_root && prelude_module.krate(self.def_collector.db) != krate {
1886                cov_mark::hit!(prelude_is_macro_use);
1887                self.def_collector.import_macros_from_extern_crate(
1888                    prelude_module.krate(self.def_collector.db),
1889                    None,
1890                    None,
1891                );
1892            }
1893        }
1894        let db = self.def_collector.db;
1895        let module_id = self.module_id;
1896        let consider_deferred_derives =
1897            |file_id: HirFileId,
1898             deferred_derives: &mut FxHashMap<_, Vec<DeferredBuiltinDerive>>,
1899             ast_id: FileAstId<ast::Adt>,
1900             id: AdtId,
1901             def_map: &mut DefMap| {
1902                let ast_id = InFile::new(file_id, ast_id.upcast());
1903                let Some(deferred_derives) = deferred_derives.remove(&ast_id.upcast()) else {
1904                    return;
1905                };
1906                let module = &mut def_map.modules[module_id];
1907                for deferred_derive in deferred_derives {
1908                    crate::builtin_derive::with_derive_traits(deferred_derive.derive, |trait_| {
1909                        let impl_id = BuiltinDeriveImplId::new(
1910                            db,
1911                            BuiltinDeriveImplLoc {
1912                                adt: id,
1913                                trait_,
1914                                derive_attr_id: deferred_derive.derive_attr_id,
1915                                derive_index: deferred_derive.derive_index,
1916                            },
1917                        );
1918                        module.scope.define_builtin_derive_impl(impl_id);
1919                        module.scope.set_derive_macro_invoc(
1920                            ast_id,
1921                            Either::Right(impl_id),
1922                            deferred_derive.derive_attr_id,
1923                            deferred_derive.derive_index as usize,
1924                        );
1925                        // Change its helper attributes to the new id.
1926                        if let Some(derive_helpers) =
1927                            def_map.derive_helpers_in_scope.get_mut(&ast_id.map(|it| it.upcast()))
1928                        {
1929                            for (_, _, call_id) in
1930                                &mut derive_helpers[deferred_derive.helpers_range.clone()]
1931                            {
1932                                *call_id = Either::Right(impl_id);
1933                            }
1934                        }
1935                    });
1936                }
1937            };
1938        let update_def =
1939            |def_collector: &mut DefCollector<'_>, id, name: &Name, vis, has_constructor| {
1940                def_collector.def_map.modules[module_id].scope.declare(id);
1941                def_collector.update(
1942                    module_id,
1943                    &[(Some(name.clone()), PerNs::from_def(id, vis, has_constructor, None))],
1944                    vis,
1945                    None,
1946                )
1947            };
1948        let resolve_vis = |def_map: &DefMap, local_def_map: &LocalDefMap, visibility| {
1949            def_map
1950                .resolve_visibility(local_def_map, db, module_id, visibility, false)
1951                .unwrap_or(Visibility::Public)
1952        };
1953
1954        let mut process_mod_item = |item: ModItemId| {
1955            let attrs = match self.item_tree.attrs(item.ast_id()) {
1956                Some(AttrsOrCfg::Enabled { attrs }) => attrs.as_ref(),
1957                None => Attrs::EMPTY,
1958                Some(AttrsOrCfg::CfgDisabled(cfg)) => {
1959                    let ast_id = item.ast_id().erase();
1960                    self.emit_unconfigured_diagnostic(InFile::new(self.file_id(), ast_id), &cfg.0);
1961                    return;
1962                }
1963            };
1964
1965            if let Err(()) = self.resolve_attributes(attrs, item, container) {
1966                // Do not process the item. It has at least one non-builtin attribute, so the
1967                // fixed-point algorithm is required to resolve the rest of them.
1968                return;
1969            }
1970
1971            let def_map = &mut self.def_collector.def_map;
1972            let local_def_map =
1973                self.def_collector.crate_local_def_map.unwrap_or(&self.def_collector.local_def_map);
1974
1975            match item {
1976                ModItemId::Mod(m) => self.collect_module(m, attrs),
1977                ModItemId::Use(item_tree_id) => {

1979                        container: module_id,
1980                        id: InFile::new(self.file_id(), item_tree_id),
1981                    }
1982                    .intern(db);
1983                    let is_prelude = attrs.by_key(sym::prelude_import).exists();

1985                        self.def_collector.unresolved_imports.push(ImportDirective {
1986                            module_id: self.module_id,
1987                            import,
1988                            status: PartialResolvedImport::Unresolved,
1989                        });
1990                    })
1991                }
1992                ModItemId::ExternCrate(item_tree_id) => {
1993                    let item_tree::ExternCrate { name, visibility, alias } =
1994                        &self.item_tree[item_tree_id];
1995

1997                        container: module_id,
1998                        id: InFile::new(self.tree_id.file_id(), item_tree_id),
1999                    }
2000                    .intern(db);
2001                    def_map.modules[self.module_id].scope.define_extern_crate_decl(id);

2005                        cov_mark::hit!(extern_crate_self_as);

2009                            .deps
2010                            .get(name)
2011                            .map(|dep| crate_def_map(db, dep.crate_id).root_module_id())

2015                        Some(ImportAlias::Alias(name)) => Some(name),
2016                        Some(ImportAlias::Underscore) => None,
2017                        None => Some(name),
2018                    };

2021                        let vis = resolve_vis(def_map, local_def_map, &self.item_tree[*visibility]);
2022

2024                            // extern crates in the crate root are special-cased to insert entries into the extern prelude: rust-lang/rust#54658
2025                            if let Some(name) = name {
2026                                self.def_collector
2027                                    .local_def_map
2028                                    .extern_prelude

2030                            }
2031                            // they also allow `#[macro_use]`
2032                            if !is_self {
2033                                self.process_macro_use_extern_crate(

2035                                    attrs.by_key(sym::macro_use).attrs(),

2038                            }
2039                        }
2040
2041                        self.def_collector.update(
2042                            module_id,
2043                            &[(
2044                                name.cloned(),
2045                                PerNs::types(
2046                                    resolved.into(),
2047                                    vis,
2048                                    Some(ImportOrExternCrate::ExternCrate(id)),
2049                                ),
2050                            )],
2051                            vis,
2052                            Some(ImportOrExternCrate::ExternCrate(id)),
2053                        );
2054                    } else {

2056                            self.def_collector.unresolved_extern_crates.insert(name.clone());

2058                        self.def_collector.def_map.diagnostics.push(
2059                            DefDiagnostic::unresolved_extern_crate(
2060                                module_id,
2061                                InFile::new(self.file_id(), item_tree_id),

2063                        );
2064                    }
2065                }
2066                ModItemId::ExternBlock(block) => {

2068                        container: module_id,
2069                        id: InFile::new(self.file_id(), block),
2070                    }
2071                    .intern(db);

2073                        .scope
2074                        .define_extern_block(extern_block_id);

2076                        &self.item_tree[block].children,
2077                        ItemContainerId::ExternBlockId(extern_block_id),

2079                }
2080                ModItemId::MacroCall(mac) => self.collect_macro_call(mac, container),
2081                ModItemId::MacroRules(id) => self.collect_macro_rules(id, module_id),
2082                ModItemId::Macro2(id) => self.collect_macro_def(id, module_id),
2083                ModItemId::Impl(imp) => {

2085                        ImplLoc { container: module_id, id: InFile::new(self.file_id(), imp) }
2086                            .intern(db);

2088                        .scope
2089                        .define_impl(impl_id, self.item_tree[imp].is_trait_impl)
2090                }
2091                ModItemId::Function(id) => {
2092                    let it = &self.item_tree[id];
2093                    let fn_id =
2094                        FunctionLoc { container, id: InFile::new(self.tree_id.file_id(), id) }
2095                            .intern(db);
2096
2097                    let vis = resolve_vis(def_map, local_def_map, &self.item_tree[it.visibility]);
2098
2099                    update_def(self.def_collector, fn_id.into(), &it.name, vis, false);
2100
2101                    if self.def_collector.def_map.block.is_none()
2102                        && self.def_collector.is_proc_macro

2106                        self.def_collector.export_proc_macro(
2107                            proc_macro,
2108                            InFile::new(self.file_id(), id),

2110                        );
2111

2113                        // You cannot call it like a function, for example, except in its defining crate.

2116                            .scope
2117                            .remove_from_value_ns(&it.name, fn_id.into());

2122

2125                        container: module_id,
2126                        id: InFile::new(self.tree_id.file_id(), id),

2130                        self.tree_id.file_id(),
2131                        &mut self.def_collector.deferred_builtin_derives,
2132                        id.upcast(),
2133                        interned.into(),
2134                        def_map,

2137                        self.def_collector,
2138                        interned.into(),

2149                        container: module_id,

2152                    .intern(db);

2154                        self.tree_id.file_id(),
2155                        &mut self.def_collector.deferred_builtin_derives,
2156                        id.upcast(),
2157                        interned.into(),
2158                        def_map,
2159                    );

2165                        container: module_id,

2171                        self.tree_id.file_id(),
2172                        &mut self.def_collector.deferred_builtin_derives,

2179                }

2187                        Some(name) => {

2189                                resolve_vis(def_map, local_def_map, &self.item_tree[it.visibility]);

2199                }

2201                    let it = &self.item_tree[id];

2204                    update_def(
2205                        self.def_collector,
2206                        StaticLoc { container, id: InFile::new(self.file_id(), id) }
2207                            .intern(db)
2208                            .into(),
2209                        &it.name,
2210                        vis,
2211                        false,
2212                    );
2213                }
2214                ModItemId::Trait(id) => {
2215                    let it = &self.item_tree[id];
2216
2217                    let vis = resolve_vis(def_map, local_def_map, &self.item_tree[it.visibility]);
2218                    update_def(
2219                        self.def_collector,
2220                        TraitLoc { container: module_id, id: InFile::new(self.file_id(), id) }
2221                            .intern(db)
2222                            .into(),
2223                        &it.name,
2224                        vis,
2225                        false,
2226                    );
2227                }

2231                    let vis = resolve_vis(def_map, local_def_map, &self.item_tree[it.visibility]);

2242            }
2243        };

2255                .iter()
2256                .filter(|it| !matches!(it, ModItemId::ExternCrate(..)))

2260            items.iter().copied().for_each(process_mod_item);
2261        }

2264    fn process_macro_use_extern_crate<'a>(
2265        &mut self,
2266        extern_crate_id: ExternCrateId,
2267        macro_use_attrs: impl Iterator<Item = &'a Attr>,
2268        target_crate: Crate,

2297    fn collect_module(&mut self, module_ast_id: ItemTreeAstId<Mod>, attrs: Attrs<'_>) {
2298        let path_attr = attrs.by_key(sym::path).string_value_unescape();
2299        let is_macro_use = attrs.by_key(sym::macro_use).exists();
2300        let module = &self.item_tree[module_ast_id];
2301        match &module.kind {
2302            // inline module, just recurse
2303            ModKind::Inline { items } => {
2304                let module_id = self.push_child_module(
2305                    module.name.clone(),
2306                    module_ast_id,
2307                    None,
2308                    &self.item_tree[module.visibility],
2309                );
2310
2311                let Some(mod_dir) =
2312                    self.mod_dir.descend_into_definition(&module.name, path_attr.as_deref())
2313                else {
2314                    return;
2315                };
2316                ModCollector {
2317                    def_collector: &mut *self.def_collector,
2318                    macro_depth: self.macro_depth,
2319                    module_id,
2320                    tree_id: self.tree_id,
2321                    item_tree: self.item_tree,
2322                    mod_dir,
2323                }
2324                .collect_in_top_module(items);
2325                if is_macro_use {
2326                    self.import_all_legacy_macros(module_id);
2327                }
2328            }
2329            // out of line module, resolve, parse and recurse
2330            ModKind::Outline => {
2331                let ast_id = AstId::new(self.file_id(), module_ast_id);
2332                let db = self.def_collector.db;
2333                match self.mod_dir.resolve_declaration(
2334                    db,
2335                    self.file_id(),
2336                    &module.name,
2337                    path_attr.as_deref(),
2338                    self.def_collector.def_map.krate,
2339                ) {
2340                    Ok((file_id, is_mod_rs, mod_dir)) => {
2341                        let item_tree = db.file_item_tree(file_id.into());
2342                        match item_tree.top_level_attrs() {
2343                            AttrsOrCfg::CfgDisabled(cfg) => {
2344                                self.emit_unconfigured_diagnostic(
2345                                    InFile::new(self.file_id(), module_ast_id.erase()),
2346                                    &cfg.0,
2347                                );
2348                            }
2349                            AttrsOrCfg::Enabled { attrs } => {
2350                                let module_id = self.push_child_module(
2351                                    module.name.clone(),
2352                                    ast_id.value,
2353                                    Some((file_id, is_mod_rs)),
2354                                    &self.item_tree[module.visibility],
2355                                );
2356                                ModCollector {
2357                                    def_collector: self.def_collector,
2358                                    macro_depth: self.macro_depth,
2359                                    module_id,
2360                                    tree_id: TreeId::new(file_id.into(), None),
2361                                    item_tree,
2362                                    mod_dir,
2363                                }
2364                                .collect_in_top_module(item_tree.top_level_items());
2365                                let is_macro_use =
2366                                    is_macro_use || attrs.as_ref().by_key(sym::macro_use).exists();
2367                                if is_macro_use {
2368                                    self.import_all_legacy_macros(module_id);
2369                                }
2370                            }
2371                        }
2372                    }
2373                    Err(candidates) => {
2374                        self.push_child_module(
2375                            module.name.clone(),
2376                            ast_id.value,
2377                            None,
2378                            &self.item_tree[module.visibility],
2379                        );
2380                        self.def_collector.def_map.diagnostics.push(
2381                            DefDiagnostic::unresolved_module(self.module_id, ast_id, candidates),
2382                        );
2383                    }
2384                };
2385            }
2386        }
2387    }

2389    fn push_child_module(
2390        &mut self,
2391        name: Name,
2392        declaration: FileAstId<ast::Module>,
2393        definition: Option<(EditionedFileId, bool)>,
2394        visibility: &crate::visibility::RawVisibility,
2395    ) -> ModuleId {
2396        let vis = self
2397            .def_collector
2398            .def_map
2399            .resolve_visibility(
2400                self.def_collector.crate_local_def_map.unwrap_or(&self.def_collector.local_def_map),
2401                self.def_collector.db,
2402                self.module_id,
2403                visibility,
2404                false,
2405            )
2406            .unwrap_or(Visibility::Public);
2407        let origin = match definition {
2408            None => {
2409                ModuleOrigin::Inline { definition: declaration, definition_tree_id: self.tree_id }
2410            }
2411            Some((definition, is_mod_rs)) => ModuleOrigin::File {
2412                declaration,
2413                definition,
2414                is_mod_rs,
2415                declaration_tree_id: self.tree_id,
2416            },
2417        };
2418
2419        let module = unsafe {
2420            crate::ModuleIdLt::new(
2421                self.def_collector.db,
2422                self.def_collector.def_map.krate,
2423                self.def_collector.def_map.block_id(),
2424            )
2425            .to_static()
2426        };
2427        let def_map = &mut self.def_collector.def_map;
2428        let modules = &mut def_map.modules;
2429        let mut data = ModuleData::new(origin, vis, Some(self.module_id));
2430        for (name, macs) in modules[self.module_id].scope.legacy_macros() {
2431            for &mac in macs {
2432                data.scope.define_legacy_macro(name.clone(), mac);
2433            }
2434        }
2435        modules[self.module_id].children.insert(name.clone(), module);
2436        modules.insert(module, data);
2437
2438        let def = ModuleDefId::from(module);
2439
2440        def_map.modules[self.module_id].scope.declare(def);
2441        self.def_collector.update(
2442            self.module_id,
2443            &[(Some(name), PerNs::from_def(def, vis, false, None))],
2444            vis,
2445            None,
2446        );
2447        module
2448    }

2450    /// Resolves attributes on an item.
2451    ///
2452    /// Returns `Err` when some attributes could not be resolved to builtins and have been
2453    /// registered as unresolved.
2454    ///
2455    /// If `ignore_up_to` is `Some`, attributes preceding and including that attribute will be
2456    /// assumed to be resolved already.
2457    fn resolve_attributes(
2458        &mut self,
2459        attrs: Attrs<'_>,
2460        mod_item: ModItemId,
2461        container: ItemContainerId,
2462    ) -> Result<(), ()> {
2463        let ignore_up_to = self
2464            .def_collector
2465            .skip_attrs
2466            .get(&InFile::new(self.file_id(), mod_item.ast_id()))
2467            .copied();
2468        for (attr_id, attr) in attrs.iter_after(ignore_up_to) {
2469            if self.def_collector.def_map.is_builtin_or_registered_attr(&attr.path) {
2470                continue;
2471            }
2472            tracing::debug!(
2473                "non-builtin attribute {}",
2474                attr.path.display(self.def_collector.db, Edition::LATEST)
2475            );
2476
2477            let ast_id = AstIdWithPath::new(self.file_id(), mod_item.ast_id(), attr.path.clone());
2478            self.def_collector.unresolved_macros.push(MacroDirective {
2479                module_id: self.module_id,
2480                depth: self.macro_depth + 1,
2481                kind: MacroDirectiveKind::Attr {
2482                    ast_id,
2483                    attr_id,
2484                    attr: attr.clone(),
2485                    mod_item,
2486                    tree: self.tree_id,
2487                    item_tree: self.item_tree,
2488                },
2489                container,
2490            });
2491
2492            return Err(());
2493        }
2494
2495        Ok(())
2496    }

2498    fn collect_macro_rules(&mut self, ast_id: ItemTreeAstId<MacroRules>, module: ModuleId) {
2499        let krate = self.def_collector.def_map.krate;
2500        let mac = &self.item_tree[ast_id];
2501        let attrs = match self.item_tree.attrs(ast_id.upcast()) {
2502            Some(AttrsOrCfg::Enabled { attrs }) => attrs.as_ref(),
2503            None => Attrs::EMPTY,
2504            Some(AttrsOrCfg::CfgDisabled(_)) => {
2505                unreachable!("we only get here if the macro is not cfg'ed out")
2506            }
2507        };
2508        let f_ast_id = InFile::new(self.file_id(), ast_id.upcast());
2509
2510        let export_attr = || attrs.by_key(sym::macro_export);
2511
2512        let is_export = export_attr().exists();
2513        let local_inner = if is_export {
2514            export_attr().tt_values().flat_map(|it| it.iter()).any(|it| match it {
2515                tt::TtElement::Leaf(tt::Leaf::Ident(ident)) => ident.sym == sym::local_inner_macros,
2516                _ => false,
2517            })
2518        } else {
2519            false
2520        };
2521
2522        // Case 1: builtin macros
2523        let expander = if attrs.by_key(sym::rustc_builtin_macro).exists() {
2524            // `#[rustc_builtin_macro = "builtin_name"]` overrides the `macro_rules!` name.
2525            let name;
2526            let name = match attrs.by_key(sym::rustc_builtin_macro).string_value_with_span() {
2527                Some((it, span)) => {
2528                    name = Name::new_symbol(Symbol::intern(it), span.ctx);
2529                    &name
2530                }
2531                None => {
2532                    let explicit_name =
2533                        attrs.by_key(sym::rustc_builtin_macro).tt_values().next().and_then(|tt| {
2534                            match tt.token_trees().iter().next() {
2535                                Some(tt::TtElement::Leaf(tt::Leaf::Ident(name))) => Some(name),
2536                                _ => None,
2537                            }
2538                        });
2539                    match explicit_name {
2540                        Some(ident) => {
2541                            name = ident.as_name();
2542                            &name
2543                        }
2544                        None => &mac.name,
2545                    }
2546                }
2547            };
2548            match find_builtin_macro(name) {
2549                Some(Either::Left(it)) => MacroExpander::BuiltIn(it),
2550                Some(Either::Right(it)) => MacroExpander::BuiltInEager(it),
2551                None => {
2552                    self.def_collector
2553                        .def_map
2554                        .diagnostics
2555                        .push(DefDiagnostic::unimplemented_builtin_macro(self.module_id, f_ast_id));
2556                    return;
2557                }
2558            }
2559        } else {
2560            // Case 2: normal `macro_rules!` macro
2561            let id = InFile::new(self.file_id(), ast_id);
2562            let decl_expander = self.def_collector.db.decl_macro_expander(krate, id.upcast());
2563            let styles = decl_expander.mac.rule_styles();
2564            MacroExpander::Declarative { styles }
2565        };
2566        let allow_internal_unsafe = attrs.by_key(sym::allow_internal_unsafe).exists();
2567
2568        let mut flags = MacroRulesLocFlags::empty();
2569        flags.set(MacroRulesLocFlags::LOCAL_INNER, local_inner);
2570        flags.set(MacroRulesLocFlags::ALLOW_INTERNAL_UNSAFE, allow_internal_unsafe);
2571
2572        let macro_id = MacroRulesLoc {
2573            container: module,
2574            id: InFile::new(self.file_id(), ast_id),
2575            flags,
2576            expander,
2577            edition: self.def_collector.def_map.data.edition,
2578        }
2579        .intern(self.def_collector.db);
2580        self.def_collector.def_map.macro_def_to_macro_id.insert(f_ast_id.erase(), macro_id.into());
2581        self.def_collector.define_macro_rules(
2582            self.module_id,
2583            mac.name.clone(),
2584            macro_id,
2585            is_export,
2586        );
2587    }

2589    fn collect_macro_def(&mut self, ast_id: ItemTreeAstId<Macro2>, module: ModuleId) {
2590        let krate = self.def_collector.def_map.krate;
2591        let mac = &self.item_tree[ast_id];
2592        let attrs = match self.item_tree.attrs(ast_id.upcast()) {
2593            Some(AttrsOrCfg::Enabled { attrs }) => attrs.as_ref(),
2594            None => Attrs::EMPTY,
2595            Some(AttrsOrCfg::CfgDisabled(_)) => {
2596                unreachable!("we only get here if the macro is not cfg'ed out")
2597            }
2598        };
2599        let f_ast_id = InFile::new(self.file_id(), ast_id.upcast());
2600
2601        // Case 1: builtin macros
2602        let mut helpers_opt = None;
2603        let expander = if attrs.by_key(sym::rustc_builtin_macro).exists() {
2604            if let Some(expander) = find_builtin_macro(&mac.name) {
2605                match expander {
2606                    Either::Left(it) => MacroExpander::BuiltIn(it),
2607                    Either::Right(it) => MacroExpander::BuiltInEager(it),
2608                }
2609            } else if let Some(expander) = find_builtin_derive(&mac.name) {
2610                if let Some(attr) = attrs.by_key(sym::rustc_builtin_macro).tt_values().next() {
2611                    // NOTE: The item *may* have both `#[rustc_builtin_macro]` and `#[proc_macro_derive]`,
2612                    // in which case rustc ignores the helper attributes from the latter, but it
2613                    // "doesn't make sense in practice" (see rust-lang/rust#87027).
2614                    if let Some((name, helpers)) = parse_macro_name_and_helper_attrs(attr) {
2615                        // NOTE: rustc overrides the name if the macro name if it's different from the
2616                        // macro name, but we assume it isn't as there's no such case yet. FIXME if
2617                        // the following assertion fails.
2618                        stdx::always!(
2619                            name == mac.name,
2620                            "built-in macro {} has #[rustc_builtin_macro] which declares different name {}",
2621                            mac.name.display(self.def_collector.db, Edition::LATEST),
2622                            name.display(self.def_collector.db, Edition::LATEST),
2623                        );
2624                        helpers_opt = Some(helpers);
2625                    }
2626                }
2627                MacroExpander::BuiltInDerive(expander)
2628            } else if let Some(expander) = find_builtin_attr(&mac.name) {
2629                MacroExpander::BuiltInAttr(expander)
2630            } else {
2631                self.def_collector
2632                    .def_map
2633                    .diagnostics
2634                    .push(DefDiagnostic::unimplemented_builtin_macro(self.module_id, f_ast_id));
2635                return;
2636            }
2637        } else {
2638            // Case 2: normal `macro`
2639            let id = InFile::new(self.file_id(), ast_id);
2640            let decl_expander = self.def_collector.db.decl_macro_expander(krate, id.upcast());
2641            let styles = decl_expander.mac.rule_styles();
2642            MacroExpander::Declarative { styles }
2643        };
2644        let allow_internal_unsafe = attrs.by_key(sym::allow_internal_unsafe).exists();
2645
2646        let macro_id = Macro2Loc {
2647            container: module,
2648            id: InFile::new(self.file_id(), ast_id),
2649            expander,
2650            allow_internal_unsafe,
2651            edition: self.def_collector.def_map.data.edition,
2652        }
2653        .intern(self.def_collector.db);
2654        self.def_collector.def_map.macro_def_to_macro_id.insert(f_ast_id.erase(), macro_id.into());

2656            self.module_id,
2657            mac.name.clone(),
2658            macro_id,
2659            &self.item_tree[mac.visibility],
2660        );
2661        if let Some(helpers) = helpers_opt
2662            && self.def_collector.def_map.block.is_none()
2663        {
2664            Arc::get_mut(&mut self.def_collector.def_map.data)
2665                .unwrap()
2666                .exported_derives
2667                .insert(macro_id.into(), helpers);
2668        }
2669    }

2671    fn collect_macro_call(
2672        &mut self,
2673        ast_id: FileAstId<ast::MacroCall>,
2674        container: ItemContainerId,
2675    ) {
2676        let &MacroCall { ref path, expand_to, ctxt } = &self.item_tree[ast_id];
2677        let ast_id = AstIdWithPath::new(self.file_id(), ast_id, path.clone());
2678        let db = self.def_collector.db;
2679
2680        // FIXME: Immediately expanding in "Case 1" is insufficient since "Case 2" may also define
2681        // new legacy macros that create textual scopes. We need a way to resolve names in textual
2682        // scopes without eager expansion.
2683
2684        let mut eager_callback_buffer = vec![];
2685        // Case 1: try to resolve macro calls with single-segment name and expand macro_rules
2686        if let Ok(res) = macro_call_as_call_id(
2687            db,
2688            ast_id.ast_id,
2689            &ast_id.path,
2690            ctxt,
2691            expand_to,
2692            self.def_collector.def_map.krate,
2693            |path| {
2694                path.as_ident().and_then(|name| {
2695                    let def_map = &self.def_collector.def_map;
2696                    def_map
2697                        .with_ancestor_maps(db, self.module_id, &mut |map, module| {
2698                            map[module].scope.get_legacy_macro(name)?.last().copied()
2699                        })
2700                        .or_else(|| def_map[self.module_id].scope.get(name).take_macros())
2701                        .or_else(|| Some(def_map.macro_use_prelude.get(name).copied()?.0))
2702                        .filter(|&id| sub_namespace_match(db, id, Some(MacroSubNs::Bang)))
2703                        .map(|it| self.def_collector.db.macro_def(it))
2704                })
2705            },
2706            &mut |ptr, call_id| eager_callback_buffer.push((ptr, call_id)),
2707        ) {
2708            for (ptr, call_id) in eager_callback_buffer {
2709                self.def_collector.def_map.modules[self.module_id]
2710                    .scope
2711                    .add_macro_invoc(ptr.map(|(_, it)| it), call_id);
2712            }
2713            // FIXME: if there were errors, this might've been in the eager expansion from an
2714            // unresolved macro, so we need to push this into late macro resolution. see fixme above
2715            if res.err.is_none() {
2716                // Legacy macros need to be expanded immediately, so that any macros they produce
2717                // are in scope.
2718                if let Some(call_id) = res.value {
2719                    self.def_collector.def_map.modules[self.module_id]
2720                        .scope
2721                        .add_macro_invoc(ast_id.ast_id, call_id);
2722                    self.def_collector.collect_macro_expansion(
2723                        self.module_id,
2724                        call_id,
2725                        self.macro_depth + 1,
2726                        container,
2727                        None,
2728                    );
2729                }
2730
2731                return;
2732            }
2733        }
2734
2735        // Case 2: resolve in module scope, expand during name resolution.
2736        self.def_collector.unresolved_macros.push(MacroDirective {
2737            module_id: self.module_id,
2738            depth: self.macro_depth + 1,
2739            kind: MacroDirectiveKind::FnLike { ast_id, expand_to, ctxt },
2740            container,
2741        });
2742    }
