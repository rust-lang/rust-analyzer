62pub(super) fn collect_defs(
63    db: &dyn DefDatabase,
64    def_map: DefMap,
65    tree_id: TreeId,
66    crate_local_def_map: Option<&LocalDefMap>,
67) -> (DefMap, LocalDefMap) {
68    let krate = &def_map.krate.data(db);
69    let cfg_options = def_map.krate.cfg_options(db);
70
71    // populate external prelude and dependency list
72    let mut deps =
73        FxIndexMap::with_capacity_and_hasher(krate.dependencies.len(), Default::default());
74    for dep in &krate.dependencies {
75        tracing::debug!("crate dep {:?} -> {:?}", dep.name, dep.crate_id);
76
77        deps.insert(dep.as_name(), dep.clone());
78    }
79
80    let proc_macros = if krate.is_proc_macro {
81        db.proc_macros_for_crate(def_map.krate)
82            .and_then(|proc_macros| {
83                proc_macros.list(db.syntax_context(tree_id.file_id(), krate.edition))
84            })
85            .unwrap_or_default()
86    } else {
87        Default::default()
88    };
89
90    let mut collector = DefCollector {
91        db,
92        def_map,
93        local_def_map: LocalDefMap::default(),
94        crate_local_def_map,
95        deps,
96        glob_imports: FxHashMap::default(),
97        unresolved_imports: Vec::new(),
98        indeterminate_imports: Vec::new(),
99        unresolved_macros: Vec::new(),
100        mod_dirs: FxHashMap::default(),
101        cfg_options,
102        proc_macros,
103        from_glob_import: Default::default(),
104        skip_attrs: Default::default(),
105        unresolved_extern_crates: Default::default(),
106        is_proc_macro: krate.is_proc_macro,
107    };
108    if tree_id.is_block() {
109        collector.seed_with_inner(tree_id);
110    } else {
111        collector.seed_with_top_level();
112    }
113    collector.collect();
114    let (mut def_map, mut local_def_map) = collector.finish();
115    def_map.shrink_to_fit();
116    local_def_map.shrink_to_fit();
117    (def_map, local_def_map)
118}

121enum PartialResolvedImport {
122    /// None of any namespaces is resolved
123    Unresolved,
124    /// One of namespaces is resolved
125    Indeterminate(PerNs),
126    /// All namespaces are resolved, OR it comes from other crate
127    Resolved(PerNs),

130impl PartialResolvedImport {
131    fn namespaces(self) -> PerNs {
132        match self {
133            PartialResolvedImport::Unresolved => PerNs::none(),
134            PartialResolvedImport::Indeterminate(ns) | PartialResolvedImport::Resolved(ns) => ns,
135        }
136    }

216/// Walks the tree of module recursively
217struct DefCollector<'db> {
218    db: &'db dyn DefDatabase,
219    def_map: DefMap,
220    local_def_map: LocalDefMap,
221    /// Set only in case of blocks.
222    crate_local_def_map: Option<&'db LocalDefMap>,
223    // The dependencies of the current crate, including optional deps like `test`.
224    deps: FxIndexMap<Name, BuiltDependency>,
225    glob_imports: FxHashMap<LocalModuleId, Vec<(LocalModuleId, Visibility, GlobId)>>,
226    unresolved_imports: Vec<ImportDirective>,
227    indeterminate_imports: Vec<(ImportDirective, PerNs)>,
228    unresolved_macros: Vec<MacroDirective<'db>>,
229    // We'd like to avoid emitting a diagnostics avalanche when some `extern crate` doesn't
230    // resolve. When we emit diagnostics for unresolved imports, we only do so if the import
231    // doesn't start with an unresolved crate's name.
232    unresolved_extern_crates: FxHashSet<Name>,
233    mod_dirs: FxHashMap<LocalModuleId, ModDir>,
234    cfg_options: &'db CfgOptions,
235    /// List of procedural macros defined by this crate. This is read from the dynamic library
236    /// built by the build system, and is the list of proc-macros we can actually expand. It is
237    /// empty when proc-macro support is disabled (in which case we still do name resolution for
238    /// them). The bool signals whether the proc-macro has been explicitly disabled for name-resolution.
239    proc_macros: Box<[(Name, CustomProcMacroExpander, bool)]>,
240    is_proc_macro: bool,
241    from_glob_import: PerNsGlobImports,
242    /// If we fail to resolve an attribute on a `ModItem`, we fall back to ignoring the attribute.
243    /// This map is used to skip all attributes up to and including the one that failed to resolve,
244    /// in order to not expand them twice.
245    ///
246    /// This also stores the attributes to skip when we resolve derive helpers and non-macro
247    /// non-builtin attributes in general.
248    // FIXME: There has to be a better way to do this
249    skip_attrs: FxHashMap<InFile<FileAstId<ast::Item>>, AttrId>,
250}

252impl<'db> DefCollector<'db> {
253    fn seed_with_top_level(&mut self) {
254        let _p = tracing::info_span!("seed_with_top_level").entered();
255
256        let file_id = self.def_map.krate.data(self.db).root_file_id(self.db);
257        let item_tree = self.db.file_item_tree(file_id.into());
258        let attrs = item_tree.top_level_attrs(self.db, self.def_map.krate);
259        let crate_data = Arc::get_mut(&mut self.def_map.data).unwrap();
260
261        let mut process = true;
262
263        // Process other crate-level attributes.
264        for attr in &*attrs {
265            if let Some(cfg) = attr.cfg()
266                && self.cfg_options.check(&cfg) == Some(false)
267            {
268                process = false;
269                break;
270            }
271            let Some(attr_name) = attr.path.as_ident() else { continue };
272
273            match () {
274                () if *attr_name == sym::recursion_limit => {
275                    if let Some(limit) = attr.string_value()
276                        && let Ok(limit) = limit.as_str().parse()
277                    {
278                        crate_data.recursion_limit = Some(limit);
279                    }
280                }
281                () if *attr_name == sym::crate_type => {
282                    if attr.string_value() == Some(&sym::proc_dash_macro) {
283                        self.is_proc_macro = true;
284                    }
285                }
286                () if *attr_name == sym::no_core => crate_data.no_core = true,
287                () if *attr_name == sym::no_std => crate_data.no_std = true,
288                () if *attr_name == sym::rustc_coherence_is_core => {
289                    crate_data.rustc_coherence_is_core = true;
290                }
291                () if *attr_name == sym::feature => {
292                    let features =
293                        attr.parse_path_comma_token_tree(self.db).into_iter().flatten().filter_map(
294                            |(feat, _)| match feat.segments() {
295                                [name] => Some(name.symbol().clone()),
296                                _ => None,
297                            },
298                        );
299                    crate_data.unstable_features.extend(features);
300                }
301                () if *attr_name == sym::register_tool => {
302                    if let Some(ident) = attr.single_ident_value() {
303                        crate_data.registered_tools.push(ident.sym.clone());
304                        cov_mark::hit!(register_tool);
305                    }
306                }
307                () => (),
308            }
309        }
310
311        for (name, dep) in &self.deps {
312            // Add all
313            if dep.is_prelude() {
314                // This is a bit confusing but the gist is that `no_core` and `no_std` remove the
315                // sysroot dependence on `core` and `std` respectively. Our `CrateGraph` is eagerly
316                // constructed with them in place no matter what though, since at that point we
317                // don't do pre-configured attribute resolution yet.
318                // So here check if we are no_core / no_std and we are trying to add the
319                // corresponding dep from the sysroot
320
321                // Depending on the crate data of a dependency seems bad for incrementality, but
322                // we only do that for sysroot crates (this is why the order of the `&&` is important)
323                // - which are normally standard library crate, which realistically aren't going
324                // to have their crate ID invalidated, because they stay on the same root file and
325                // they're dependencies of everything else, so if some collision miraculously occurs
326                // we will resolve it by disambiguating the other crate.
327                let skip = dep.is_sysroot()
328                    && match dep.crate_id.data(self.db).origin {
329                        CrateOrigin::Lang(LangCrateOrigin::Core) => crate_data.no_core,
330                        CrateOrigin::Lang(LangCrateOrigin::Std) => {
331                            crate_data.no_core || crate_data.no_std
332                        }
333                        _ => false,
334                    };
335                if skip {
336                    continue;
337                }
338
339                self.local_def_map
340                    .extern_prelude
341                    .insert(name.clone(), (CrateRootModuleId { krate: dep.crate_id }, None));
342            }
343        }
344
345        self.inject_prelude();
346
347        if !process {
348            return;
349        }
350
351        ModCollector {
352            def_collector: self,
353            macro_depth: 0,
354            module_id: DefMap::ROOT,
355            tree_id: TreeId::new(file_id.into(), None),
356            item_tree,
357            mod_dir: ModDir::root(),
358        }
359        .collect_in_top_module(item_tree.top_level_items());
360        Arc::get_mut(&mut self.def_map.data).unwrap().shrink_to_fit();
361    }

363    fn seed_with_inner(&mut self, tree_id: TreeId) {
364        let item_tree = tree_id.item_tree(self.db);
365        let is_cfg_enabled = item_tree
366            .top_level_attrs(self.db, self.def_map.krate)
367            .cfg()
368            .is_none_or(|cfg| self.cfg_options.check(&cfg) != Some(false));
369        if is_cfg_enabled {
370            self.inject_prelude();
371
372            ModCollector {
373                def_collector: self,
374                macro_depth: 0,
375                module_id: DefMap::ROOT,
376                tree_id,
377                item_tree,
378                mod_dir: ModDir::root(),
379            }
380            .collect_in_top_module(item_tree.top_level_items());
381        }
382    }

384    fn resolution_loop(&mut self) {
385        let _p = tracing::info_span!("DefCollector::resolution_loop").entered();
386
387        // main name resolution fixed-point loop.
388        let mut i = 0;
389        'resolve_attr: loop {
390            let _p = tracing::info_span!("resolve_macros loop").entered();
391            'resolve_macros: loop {
392                self.db.unwind_if_revision_cancelled();
393
394                {
395                    let _p = tracing::info_span!("resolve_imports loop").entered();
396
397                    'resolve_imports: loop {
398                        if self.resolve_imports() == ReachedFixedPoint::Yes {
399                            break 'resolve_imports;
400                        }
401                    }
402                }
403                if self.resolve_macros() == ReachedFixedPoint::Yes {
404                    break 'resolve_macros;
405                }
406
407                i += 1;
408                if i > FIXED_POINT_LIMIT {
409                    tracing::error!("name resolution is stuck");
410                    break 'resolve_attr;
411                }
412            }
413
414            if self.reseed_with_unresolved_attribute() == ReachedFixedPoint::Yes {
415                break 'resolve_attr;
416            }
417        }
418    }

420    fn collect(&mut self) {
421        let _p = tracing::info_span!("DefCollector::collect").entered();
422
423        self.resolution_loop();
424
425        let unresolved_imports = mem::take(&mut self.unresolved_imports);
426        // show unresolved imports in completion, etc
427        for directive in &unresolved_imports {
428            self.record_resolved_import(directive);
429        }
430        self.unresolved_imports = unresolved_imports;
431
432        if self.is_proc_macro {
433            // A crate exporting procedural macros is not allowed to export anything else.
434            //
435            // Additionally, while the proc macro entry points must be `pub`, they are not publicly
436            // exported in type/value namespace. This function reduces the visibility of all items
437            // in the crate root that aren't proc macros.
438            let root = &mut self.def_map.modules[DefMap::ROOT];
439            root.scope.censor_non_proc_macros(self.def_map.krate);
440        }
441    }

443    /// When the fixed-point loop reaches a stable state, we might still have
444    /// some unresolved attributes left over. This takes one of them, and feeds
445    /// the item it's applied to back into name resolution.
446    ///
447    /// This effectively ignores the fact that the macro is there and just treats the items as
448    /// normal code.
449    ///
450    /// This improves UX for unresolved attributes, and replicates the
451    /// behavior before we supported proc. attribute macros.
452    fn reseed_with_unresolved_attribute(&mut self) -> ReachedFixedPoint {
453        cov_mark::hit!(unresolved_attribute_fallback);
454
455        let unresolved_attr =
456            self.unresolved_macros.iter().enumerate().find_map(|(idx, directive)| match &directive
457                .kind
458            {
459                MacroDirectiveKind::Attr { ast_id, mod_item, attr, tree, item_tree } => {
460                    self.def_map.diagnostics.push(DefDiagnostic::unresolved_macro_call(
461                        directive.module_id,
462                        MacroCallKind::Attr {
463                            ast_id: ast_id.ast_id,
464                            attr_args: None,
465                            invoc_attr_index: attr.id,
466                        },
467                        attr.path().clone(),
468                    ));
469
470                    self.skip_attrs.insert(ast_id.ast_id.with_value(mod_item.ast_id()), attr.id);
471
472                    Some((idx, directive, *mod_item, *tree, *item_tree))
473                }
474                _ => None,
475            });
476
477        match unresolved_attr {
478            Some((
479                pos,
480                &MacroDirective { module_id, depth, container, .. },
481                mod_item,
482                tree_id,
483                item_tree,
484            )) => {
485                // FIXME: Remove this clone
486                let mod_dir = self.mod_dirs[&module_id].clone();
487                ModCollector {
488                    def_collector: self,
489                    macro_depth: depth,
490                    module_id,
491                    tree_id,
492                    item_tree,
493                    mod_dir,
494                }
495                .collect(&[mod_item], container);
496
497                self.unresolved_macros.swap_remove(pos);
498                // Continue name resolution with the new data.
499                ReachedFixedPoint::No
500            }
501            None => ReachedFixedPoint::Yes,
502        }
503    }

505    fn inject_prelude(&mut self) {
506        // See compiler/rustc_builtin_macros/src/standard_library_imports.rs
507
508        if self.def_map.data.no_core {
509            // libcore does not get a prelude.
510            return;
511        }
512
513        let krate = if self.def_map.data.no_std {
514            Name::new_symbol_root(sym::core)
515        } else if self.local_def_map().extern_prelude().any(|(name, _)| *name == sym::std) {
516            Name::new_symbol_root(sym::std)
517        } else {
518            // If `std` does not exist for some reason, fall back to core. This mostly helps
519            // keep r-a's own tests minimal.
520            Name::new_symbol_root(sym::core)
521        };
522
523        let edition = match self.def_map.data.edition {
524            Edition::Edition2015 => Name::new_symbol_root(sym::rust_2015),
525            Edition::Edition2018 => Name::new_symbol_root(sym::rust_2018),
526            Edition::Edition2021 => Name::new_symbol_root(sym::rust_2021),
527            Edition::Edition2024 => Name::new_symbol_root(sym::rust_2024),
528        };
529
530        let path_kind = match self.def_map.data.edition {
531            Edition::Edition2015 => PathKind::Plain,
532            _ => PathKind::Abs,
533        };
534        let path = ModPath::from_segments(
535            path_kind,
536            [krate, Name::new_symbol_root(sym::prelude), edition],
537        );
538
539        let (per_ns, _) = self.def_map.resolve_path(
540            self.crate_local_def_map.unwrap_or(&self.local_def_map),
541            self.db,
542            DefMap::ROOT,
543            &path,
544            BuiltinShadowMode::Other,
545            None,
546        );
547
548        match per_ns.types {
549            Some(Item { def: ModuleDefId::ModuleId(m), import, .. }) => {
550                self.def_map.prelude = Some((m, import.and_then(ImportOrExternCrate::use_)));
551            }
552            types => {
553                tracing::debug!(
554                    "could not resolve prelude path `{}` to module (resolved to {:?})",
555                    path.display(self.db, Edition::LATEST),
556                    types
557                );
558            }
559        }
560    }

566    /// Adds a definition of procedural macro `name` to the root module.
567    ///
568    /// # Notes on procedural macro resolution
569    ///
570    /// Procedural macro functionality is provided by the build system: It has to build the proc
571    /// macro and pass the resulting dynamic library to rust-analyzer.
572    ///
573    /// When procedural macro support is enabled, the list of proc macros exported by a crate is
574    /// known before we resolve names in the crate. This list is stored in `self.proc_macros` and is
575    /// derived from the dynamic library.
576    ///
577    /// However, we *also* would like to be able to at least *resolve* macros on our own, without
578    /// help by the build system. So, when the macro isn't found in `self.proc_macros`, we instead
579    /// use a dummy expander that always errors. This comes with the drawback of macros potentially
580    /// going out of sync with what the build system sees (since we resolve using VFS state, but
581    /// Cargo builds only on-disk files). We could and probably should add diagnostics for that.
582    fn export_proc_macro(&mut self, def: ProcMacroDef, ast_id: AstId<ast::Fn>, fn_id: FunctionId) {
583        let kind = def.kind.to_basedb_kind();
584        let (expander, kind) = match self.proc_macros.iter().find(|(n, _, _)| n == &def.name) {
585            Some(_)
586                if kind == hir_expand::proc_macro::ProcMacroKind::Attr
587                    && !self.db.expand_proc_attr_macros() =>
588            {
589                (CustomProcMacroExpander::disabled_proc_attr(), kind)
590            }
591            Some(&(_, _, true)) => (CustomProcMacroExpander::disabled(), kind),
592            Some(&(_, expander, false)) => (expander, kind),
593            None => (CustomProcMacroExpander::missing_expander(), kind),
594        };
595
596        let proc_macro_id = ProcMacroLoc {
597            container: self.def_map.crate_root(),
598            id: ast_id,
599            expander,
600            kind,
601            edition: self.def_map.data.edition,
602        }
603        .intern(self.db);
604
605        self.def_map.macro_def_to_macro_id.insert(ast_id.erase(), proc_macro_id.into());
606        self.define_proc_macro(def.name.clone(), proc_macro_id);
607        let crate_data = Arc::get_mut(&mut self.def_map.data).unwrap();
608        if let ProcMacroKind::Derive { helpers } = def.kind {
609            crate_data.exported_derives.insert(proc_macro_id.into(), helpers);
610        }
611        crate_data.fn_proc_macro_mapping.insert(fn_id, proc_macro_id);
612    }

614    /// Define a macro with `macro_rules`.
615    ///
616    /// It will define the macro in legacy textual scope, and if it has `#[macro_export]`,
617    /// then it is also defined in the root module scope.
618    /// You can `use` or invoke it by `crate::macro_name` anywhere, before or after the definition.
619    ///
620    /// It is surprising that the macro will never be in the current module scope.
621    /// These code fails with "unresolved import/macro",
622    /// ```rust,compile_fail
623    /// mod m { macro_rules! foo { () => {} } }
624    /// use m::foo as bar;
625    /// ```
626    ///
627    /// ```rust,compile_fail
628    /// macro_rules! foo { () => {} }
629    /// self::foo!();
630    /// crate::foo!();
631    /// ```
632    ///
633    /// Well, this code compiles, because the plain path `foo` in `use` is searched
634    /// in the legacy textual scope only.
635    /// ```rust
636    /// macro_rules! foo { () => {} }
637    /// use foo as bar;
638    /// ```
639    fn define_macro_rules(
640        &mut self,
641        module_id: LocalModuleId,
642        name: Name,
643        macro_: MacroRulesId,
644        export: bool,
645    ) {
646        // Textual scoping
647        self.define_legacy_macro(module_id, name.clone(), macro_.into());
648
649        // Module scoping
650        // In Rust, `#[macro_export]` macros are unconditionally visible at the
651        // crate root, even if the parent modules is **not** visible.
652        if export {
653            let module_id = DefMap::ROOT;
654            self.def_map.modules[module_id].scope.declare(macro_.into());
655            self.update(
656                module_id,
657                &[(Some(name), PerNs::macros(macro_.into(), Visibility::Public, None))],
658                Visibility::Public,
659                None,
660            );
661        }
662    }

666    /// We use a map `legacy_macros` to store all legacy textual scoped macros visible per module.
667    /// It will clone all macros from parent legacy scope, whose definition is prior to
668    /// the definition of current module.

720    /// Import exported macros from another crate. `names`, if `Some(_)`, specifies the name of
721    /// macros to be imported. Otherwise this method imports all exported macros.
722    ///
723    /// Exported macros are just all macros in the root module scope.
724    /// Note that it contains not only all `#[macro_export]` macros, but also all aliases
725    /// created by `use` in the root module, ignoring the visibility of `use`.
726    fn import_macros_from_extern_crate(
727        &mut self,
728        krate: Crate,
729        names: Option<Vec<Name>>,
730        extern_crate: Option<ExternCrateId>,
731    ) {
732        let def_map = crate_def_map(self.db, krate);
733        // `#[macro_use]` brings macros into macro_use prelude. Yes, even non-`macro_rules!`
734        // macros.
735        let root_scope = &def_map[DefMap::ROOT].scope;
736        match names {
737            Some(names) => {
738                for name in names {
739                    // FIXME: Report diagnostic on 404.
740                    if let Some(def) = root_scope.get(&name).take_macros() {
741                        self.def_map.macro_use_prelude.insert(name, (def, extern_crate));
742                    }
743                }
744            }
745            None => {
746                for (name, it) in root_scope.macros() {
747                    self.def_map.macro_use_prelude.insert(name.clone(), (it.def, extern_crate));
748                }
749            }
750        }
751    }

753    /// Tries to resolve every currently unresolved import.
754    fn resolve_imports(&mut self) -> ReachedFixedPoint {
755        let mut res = ReachedFixedPoint::Yes;
756        let imports = mem::take(&mut self.unresolved_imports);
757
758        self.unresolved_imports = imports
759            .into_iter()
760            .filter_map(|mut directive| {
761                directive.status = self.resolve_import(directive.module_id, &directive.import);
762                match directive.status {
763                    PartialResolvedImport::Indeterminate(resolved) => {
764                        self.record_resolved_import(&directive);
765                        self.indeterminate_imports.push((directive, resolved));
766                        res = ReachedFixedPoint::No;
767                        None
768                    }
769                    PartialResolvedImport::Resolved(_) => {
770                        self.record_resolved_import(&directive);
771                        res = ReachedFixedPoint::No;
772                        None
773                    }
774                    PartialResolvedImport::Unresolved => Some(directive),
775                }
776            })
777            .collect();
778
779        // Resolve all indeterminate resolved imports again
780        // As some of the macros will expand newly import shadowing partial resolved imports
781        // FIXME: We maybe could skip this, if we handle the indeterminate imports in `resolve_imports`
782        // correctly
783        let mut indeterminate_imports = std::mem::take(&mut self.indeterminate_imports);
784        indeterminate_imports.retain_mut(|(directive, partially_resolved)| {
785            let partially_resolved = partially_resolved.availability();
786            directive.status = self.resolve_import(directive.module_id, &directive.import);
787            match directive.status {
788                PartialResolvedImport::Indeterminate(import)
789                    if partially_resolved != import.availability() =>
790                {
791                    self.record_resolved_import(directive);
792                    res = ReachedFixedPoint::No;
793                    false
794                }
795                PartialResolvedImport::Resolved(_) => {
796                    self.record_resolved_import(directive);
797                    res = ReachedFixedPoint::No;
798                    false
799                }
800                _ => true,
801            }
802        });
803        self.indeterminate_imports = indeterminate_imports;
804
805        res
806    }

808    fn resolve_import(&self, module_id: LocalModuleId, import: &Import) -> PartialResolvedImport {
809        let _p = tracing::info_span!("resolve_import", import_path = %import.path.display(self.db, Edition::LATEST))
810            .entered();
811        tracing::debug!("resolving import: {:?} ({:?})", import, self.def_map.data.edition);
812        let ResolvePathResult { resolved_def, segment_index, reached_fixedpoint, prefix_info } =
813            self.def_map.resolve_path_fp_with_macro(
814                self.crate_local_def_map.unwrap_or(&self.local_def_map),
815                self.db,
816                ResolveMode::Import,
817                module_id,
818                &import.path,
819                BuiltinShadowMode::Module,
820                None, // An import may resolve to any kind of macro.
821            );
822
823        if reached_fixedpoint == ReachedFixedPoint::No
824            || resolved_def.is_none()
825            || segment_index.is_some()
826        {
827            return PartialResolvedImport::Unresolved;
828        }
829
830        if prefix_info.differing_crate {
831            return PartialResolvedImport::Resolved(
832                resolved_def.filter_visibility(|v| matches!(v, Visibility::Public)),
833            );
834        }
835
836        // Check whether all namespaces are resolved.
837        if resolved_def.is_full() {
838            PartialResolvedImport::Resolved(resolved_def)
839        } else {
840            PartialResolvedImport::Indeterminate(resolved_def)
841        }
842    }

844    fn record_resolved_import(&mut self, directive: &ImportDirective) {
845        let _p = tracing::info_span!("record_resolved_import").entered();
846
847        let module_id = directive.module_id;
848        let import = &directive.import;
849        let mut def = directive.status.namespaces();
850        let vis = self
851            .def_map
852            .resolve_visibility(
853                self.crate_local_def_map.unwrap_or(&self.local_def_map),
854                self.db,
855                module_id,
856                &directive.import.visibility,
857                false,
858            )
859            .unwrap_or(Visibility::Public);
860
861        match import.source {
862            ImportSource {
863                kind: kind @ (ImportKind::Plain | ImportKind::TypeOnly),
864                id,
865                use_tree,
866                ..
867            } => {
868                let name = match &import.alias {
869                    Some(ImportAlias::Alias(name)) => Some(name),
870                    Some(ImportAlias::Underscore) => None,
871                    None => match import.path.segments().last() {
872                        Some(last_segment) => Some(last_segment),
873                        None => {
874                            cov_mark::hit!(bogus_paths);
875                            return;
876                        }
877                    },
878                };
879
880                if kind == ImportKind::TypeOnly {
881                    def.values = None;
882                    def.macros = None;
883                }
884                let imp = ImportOrExternCrate::Import(ImportId { use_: id, idx: use_tree });
885                tracing::debug!("resolved import {:?} ({:?}) to {:?}", name, import, def);
886
887                // `extern crate crate_name` things can be re-exported as `pub use crate_name`.
888                // But they cannot be re-exported as `pub use self::crate_name`, `pub use crate::crate_name`
889                // or `pub use ::crate_name`.
890                //
891                // This has been historically allowed, but may be not allowed in future
892                // https://github.com/rust-lang/rust/issues/127909
893                if let Some(def) = def.types.as_mut() {
894                    let is_extern_crate_reimport_without_prefix = || {
895                        let Some(ImportOrExternCrate::ExternCrate(_)) = def.import else {
896                            return false;
897                        };
898                        if kind == ImportKind::Glob {
899                            return false;
900                        }
901                        matches!(import.path.kind, PathKind::Plain | PathKind::SELF)
902                            && import.path.segments().len() < 2
903                    };
904                    if is_extern_crate_reimport_without_prefix() {
905                        def.vis = vis;
906                    }
907                }
908
909                self.update(module_id, &[(name.cloned(), def)], vis, Some(imp));
910            }
911            ImportSource { kind: ImportKind::Glob, id, is_prelude, use_tree, .. } => {
912                tracing::debug!("glob import: {:?}", import);
913                let glob = GlobId { use_: id, idx: use_tree };
914                match def.take_types() {
915                    Some(ModuleDefId::ModuleId(m)) => {
916                        if is_prelude {
917                            // Note: This dodgily overrides the injected prelude. The rustc
918                            // implementation seems to work the same though.
919                            cov_mark::hit!(std_prelude);
920                            self.def_map.prelude = Some((m, Some(id)));
921                        } else if m.krate != self.def_map.krate {
922                            cov_mark::hit!(glob_across_crates);
923                            // glob import from other crate => we can just import everything once
924                            let item_map = m.def_map(self.db);
925                            let scope = &item_map[m.local_id].scope;
926
927                            // Module scoped macros is included
928                            let items = scope
929                                .resolutions()
930                                // only keep visible names...
931                                .map(|(n, res)| {
932                                    (n, res.filter_visibility(|v| v.is_visible_from_other_crate()))
933                                })
934                                .filter(|(_, res)| !res.is_none())
935                                .collect::<Vec<_>>();
936
937                            self.update(
938                                module_id,
939                                &items,
940                                vis,
941                                Some(ImportOrExternCrate::Glob(glob)),
942                            );
943                        } else {
944                            // glob import from same crate => we do an initial
945                            // import, and then need to propagate any further
946                            // additions
947                            let def_map;
948                            let scope = if m.block == self.def_map.block_id() {
949                                &self.def_map[m.local_id].scope
950                            } else {
951                                def_map = m.def_map(self.db);
952                                &def_map[m.local_id].scope
953                            };
954
955                            // Module scoped macros is included
956                            let items = scope
957                                .resolutions()
958                                // only keep visible names...
959                                .map(|(n, res)| {
960                                    (
961                                        n,
962                                        res.filter_visibility(|v| {
963                                            v.is_visible_from_def_map(
964                                                self.db,
965                                                &self.def_map,
966                                                module_id,
967                                            )
968                                        }),
969                                    )
970                                })
971                                .filter(|(_, res)| !res.is_none())
972                                .collect::<Vec<_>>();
973
974                            self.update(
975                                module_id,
976                                &items,
977                                vis,
978                                Some(ImportOrExternCrate::Glob(glob)),
979                            );
980                            // record the glob import in case we add further items
981                            let glob_imports = self.glob_imports.entry(m.local_id).or_default();
982                            match glob_imports.iter_mut().find(|(mid, _, _)| *mid == module_id) {
983                                None => glob_imports.push((module_id, vis, glob)),
984                                Some((_, old_vis, _)) => {
985                                    if let Some(new_vis) = old_vis.max(vis, &self.def_map) {
986                                        *old_vis = new_vis;
987                                    }
988                                }
989                            }
990                        }
991                    }
992                    Some(ModuleDefId::AdtId(AdtId::EnumId(e))) => {
993                        cov_mark::hit!(glob_enum);
994                        // glob import from enum => just import all the variants
995                        let resolutions = e
996                            .enum_variants(self.db)
997                            .variants
998                            .iter()
999                            .map(|&(variant, ref name, _)| {
1000                                let res = PerNs::both(variant.into(), variant.into(), vis, None);
1001                                (Some(name.clone()), res)
1002                            })
1003                            .collect::<Vec<_>>();
1004                        self.update(
1005                            module_id,
1006                            &resolutions,
1007                            vis,
1008                            Some(ImportOrExternCrate::Glob(glob)),
1009                        );
1010                    }
1011                    Some(ModuleDefId::TraitId(it)) => {
1012                        // FIXME: Implement this correctly
1013                        // We can't actually call `trait_items`, the reason being that if macro calls
1014                        // occur, they will call back into the def map which we might be computing right
1015                        // now resulting in a cycle.
1016                        // To properly implement this, trait item collection needs to be done in def map
1017                        // collection...
1018                        let resolutions = if true {
1019                            vec![]
1020                        } else {
1021                            TraitItems::query(self.db, it)
1022                                .items
1023                                .iter()
1024                                .map(|&(ref name, variant)| {
1025                                    let res = match variant {
1026                                        AssocItemId::FunctionId(it) => {
1027                                            PerNs::values(it.into(), vis, None)
1028                                        }
1029                                        AssocItemId::ConstId(it) => {
1030                                            PerNs::values(it.into(), vis, None)
1031                                        }
1032                                        AssocItemId::TypeAliasId(it) => {
1033                                            PerNs::types(it.into(), vis, None)
1034                                        }
1035                                    };
1036                                    (Some(name.clone()), res)
1037                                })
1038                                .collect::<Vec<_>>()
1039                        };
1040                        self.update(
1041                            module_id,
1042                            &resolutions,
1043                            vis,
1044                            Some(ImportOrExternCrate::Glob(glob)),
1045                        );
1046                    }
1047                    Some(d) => {
1048                        tracing::debug!("glob import {:?} from non-module/enum {:?}", import, d);
1049                    }
1050                    None => {
1051                        tracing::debug!("glob import {:?} didn't resolve as type", import);
1052                    }
1053                }
1054            }
1055        }
1056    }

1058    fn update(
1059        &mut self,
1060        // The module for which `resolutions` have been resolve
1061        module_id: LocalModuleId,
1062        resolutions: &[(Option<Name>, PerNs)],
1063        // Visibility this import will have
1064        vis: Visibility,
1065        import: Option<ImportOrExternCrate>,
1066    ) {
1067        self.db.unwind_if_revision_cancelled();
1068        self.update_recursive(module_id, resolutions, vis, import, 0)
1069    }

1071    fn update_recursive(
1072        &mut self,
1073        // The module for which `resolutions` have been resolved.
1074        module_id: LocalModuleId,
1075        resolutions: &[(Option<Name>, PerNs)],
1076        // All resolutions are imported with this visibility; the visibilities in
1077        // the `PerNs` values are ignored and overwritten
1078        vis: Visibility,
1079        import: Option<ImportOrExternCrate>,
1080        depth: usize,
1081    ) {
1082        if depth > GLOB_RECURSION_LIMIT {
1083            // prevent stack overflows (but this shouldn't be possible)
1084            panic!("infinite recursion in glob imports!");
1085        }
1086        let mut changed = false;
1087
1088        for (name, res) in resolutions {
1089            match name {
1090                Some(name) => {
1091                    changed |=
1092                        self.push_res_and_update_glob_vis(module_id, name, *res, vis, import);
1093                }
1094                None => {
1095                    let (tr, import) = match res.take_types_full() {
1096                        Some(Item { def: ModuleDefId::TraitId(tr), vis: _, import }) => {
1097                            (tr, import)
1098                        }
1099                        Some(other) => {
1100                            tracing::debug!("non-trait `_` import of {:?}", other);
1101                            continue;
1102                        }
1103                        None => continue,
1104                    };
1105                    let old_vis = self.def_map.modules[module_id].scope.unnamed_trait_vis(tr);
1106                    let should_update = match old_vis {
1107                        None => true,
1108                        Some(old_vis) => {
1109                            let max_vis = old_vis.max(vis, &self.def_map).unwrap_or_else(|| {
1110                                panic!("`Tr as _` imports with unrelated visibilities {old_vis:?} and {vis:?} (trait {tr:?})");
1111                            });
1112
1113                            if max_vis == old_vis {
1114                                false
1115                            } else {
1116                                cov_mark::hit!(upgrade_underscore_visibility);
1117                                true
1118                            }
1119                        }
1120                    };
1121
1122                    if should_update {
1123                        changed = true;
1124                        self.def_map.modules[module_id].scope.push_unnamed_trait(
1125                            tr,
1126                            vis,
1127                            import.and_then(ImportOrExternCrate::import),
1128                        );
1129                    }
1130                }
1131            }
1132        }
1133
1134        if !changed {
1135            return;
1136        }
1137        let glob_imports = self
1138            .glob_imports
1139            .get(&module_id)
1140            .into_iter()
1141            .flatten()
1142            .filter(|(glob_importing_module, _, _)| {
1143                // we know all resolutions have the same visibility (`vis`), so we
1144                // just need to check that once
1145                vis.is_visible_from_def_map(self.db, &self.def_map, *glob_importing_module)
1146            })
1147            .cloned()
1148            .collect::<Vec<_>>();
1149
1150        for (glob_importing_module, glob_import_vis, glob) in glob_imports {
1151            let vis = glob_import_vis.min(vis, &self.def_map).unwrap_or(glob_import_vis);
1152            self.update_recursive(
1153                glob_importing_module,
1154                resolutions,
1155                vis,
1156                Some(ImportOrExternCrate::Glob(glob)),
1157                depth + 1,
1158            );
1159        }
1160    }

1162    fn push_res_and_update_glob_vis(
1163        &mut self,
1164        module_id: LocalModuleId,
1165        name: &Name,
1166        mut defs: PerNs,
1167        vis: Visibility,
1168        def_import_type: Option<ImportOrExternCrate>,
1169    ) -> bool {
1170        if let Some(def) = defs.types.as_mut() {
1171            def.vis = def.vis.min(vis, &self.def_map).unwrap_or(vis);
1172        }
1173        if let Some(def) = defs.values.as_mut() {
1174            def.vis = def.vis.min(vis, &self.def_map).unwrap_or(vis);
1175        }
1176        if let Some(def) = defs.macros.as_mut() {
1177            def.vis = def.vis.min(vis, &self.def_map).unwrap_or(vis);
1178        }
1179
1180        let mut changed = false;
1181
1182        if let Some(ImportOrExternCrate::Glob(_)) = def_import_type {
1183            let prev_defs = self.def_map[module_id].scope.get(name);
1184
1185            // Multiple globs may import the same item and they may override visibility from
1186            // previously resolved globs. Handle overrides here and leave the rest to
1187            // `ItemScope::push_res_with_import()`.
1188            if let Some(def) = defs.types
1189                && let Some(prev_def) = prev_defs.types
1190                && def.def == prev_def.def
1191                && self.from_glob_import.contains_type(module_id, name.clone())
1192                && def.vis != prev_def.vis
1193                && def.vis.max(prev_def.vis, &self.def_map) == Some(def.vis)
1194            {
1195                changed = true;
1196                // This import is being handled here, don't pass it down to
1197                // `ItemScope::push_res_with_import()`.
1198                defs.types = None;
1199                self.def_map.modules[module_id].scope.update_visibility_types(name, def.vis);
1200            }
1201
1202            if let Some(def) = defs.values
1203                && let Some(prev_def) = prev_defs.values
1204                && def.def == prev_def.def
1205                && self.from_glob_import.contains_value(module_id, name.clone())
1206                && def.vis != prev_def.vis
1207                && def.vis.max(prev_def.vis, &self.def_map) == Some(def.vis)
1208            {
1209                changed = true;
1210                // See comment above.
1211                defs.values = None;
1212                self.def_map.modules[module_id].scope.update_visibility_values(name, def.vis);
1213            }
1214
1215            if let Some(def) = defs.macros
1216                && let Some(prev_def) = prev_defs.macros
1217                && def.def == prev_def.def
1218                && self.from_glob_import.contains_macro(module_id, name.clone())
1219                && def.vis != prev_def.vis
1220                && def.vis.max(prev_def.vis, &self.def_map) == Some(def.vis)
1221            {
1222                changed = true;
1223                // See comment above.
1224                defs.macros = None;
1225                self.def_map.modules[module_id].scope.update_visibility_macros(name, def.vis);
1226            }
1227        }
1228
1229        changed |= self.def_map.modules[module_id].scope.push_res_with_import(
1230            &mut self.from_glob_import,
1231            (module_id, name.clone()),
1232            defs,
1233            def_import_type,
1234        );
1235
1236        changed
1237    }

1239    fn resolve_macros(&mut self) -> ReachedFixedPoint {
1240        let mut macros = mem::take(&mut self.unresolved_macros);
1241        let mut resolved = Vec::new();
1242        let mut push_resolved = |directive: &MacroDirective<'_>, call_id| {
1243            resolved.push((directive.module_id, directive.depth, directive.container, call_id));
1244        };
1245
1246        #[derive(PartialEq, Eq)]
1247        enum Resolved {
1248            Yes,
1249            No,
1250        }
1251
1252        let mut eager_callback_buffer = vec![];
1253        let mut res = ReachedFixedPoint::Yes;
1254        // Retain unresolved macros after this round of resolution.
1255        let mut retain = |directive: &MacroDirective<'db>| {
1256            let subns = match &directive.kind {
1257                MacroDirectiveKind::FnLike { .. } => MacroSubNs::Bang,
1258                MacroDirectiveKind::Attr { .. } | MacroDirectiveKind::Derive { .. } => {
1259                    MacroSubNs::Attr
1260                }
1261            };
1262            let resolver = |path: &_| {
1263                let resolved_res = self.def_map.resolve_path_fp_with_macro(
1264                    self.crate_local_def_map.unwrap_or(&self.local_def_map),
1265                    self.db,
1266                    ResolveMode::Other,
1267                    directive.module_id,
1268                    path,
1269                    BuiltinShadowMode::Module,
1270                    Some(subns),
1271                );
1272                resolved_res.resolved_def.take_macros().map(|it| (it, self.db.macro_def(it)))
1273            };
1274            let resolver_def_id = |path: &_| resolver(path).map(|(_, it)| it);
1275
1276            match &directive.kind {
1277                MacroDirectiveKind::FnLike { ast_id, expand_to, ctxt: call_site } => {
1278                    let call_id = macro_call_as_call_id(
1279                        self.db,
1280                        ast_id.ast_id,
1281                        &ast_id.path,
1282                        *call_site,
1283                        *expand_to,
1284                        self.def_map.krate,
1285                        resolver_def_id,
1286                        &mut |ptr, call_id| {
1287                            eager_callback_buffer.push((directive.module_id, ptr, call_id));
1288                        },
1289                    );
1290                    if let Ok(call_id) = call_id {
1291                        // FIXME: Expansion error
1292                        if let Some(call_id) = call_id.value {
1293                            self.def_map.modules[directive.module_id]
1294                                .scope
1295                                .add_macro_invoc(ast_id.ast_id, call_id);
1296
1297                            push_resolved(directive, call_id);
1298
1299                            res = ReachedFixedPoint::No;
1300                            return Resolved::Yes;
1301                        }
1302                    }
1303                }
1304                MacroDirectiveKind::Derive {
1305                    ast_id,
1306                    derive_attr,
1307                    derive_pos,
1308                    ctxt: call_site,
1309                    derive_macro_id,
1310                } => {
1311                    let id = derive_macro_as_call_id(
1312                        self.db,
1313                        ast_id,
1314                        *derive_attr,
1315                        *derive_pos as u32,
1316                        *call_site,
1317                        self.def_map.krate,
1318                        resolver,
1319                        *derive_macro_id,
1320                    );
1321
1322                    if let Ok((macro_id, def_id, call_id)) = id {
1323                        self.def_map.modules[directive.module_id].scope.set_derive_macro_invoc(
1324                            ast_id.ast_id,
1325                            call_id,
1326                            *derive_attr,
1327                            *derive_pos,
1328                        );
1329                        // Record its helper attributes.
1330                        if def_id.krate != self.def_map.krate {
1331                            let def_map = crate_def_map(self.db, def_id.krate);
1332                            if let Some(helpers) = def_map.data.exported_derives.get(&macro_id) {
1333                                self.def_map
1334                                    .derive_helpers_in_scope
1335                                    .entry(ast_id.ast_id.map(|it| it.upcast()))
1336                                    .or_default()
1337                                    .extend(izip!(
1338                                        helpers.iter().cloned(),
1339                                        iter::repeat(macro_id),
1340                                        iter::repeat(call_id),
1341                                    ));
1342                            }
1343                        }
1344
1345                        push_resolved(directive, call_id);
1346                        res = ReachedFixedPoint::No;
1347                        return Resolved::Yes;
1348                    }
1349                }
1350                MacroDirectiveKind::Attr {
1351                    ast_id: file_ast_id,
1352                    mod_item,
1353                    attr,
1354                    tree,
1355                    item_tree,
1356                } => {
1357                    let &AstIdWithPath { ast_id, ref path } = file_ast_id;
1358                    let file_id = ast_id.file_id;
1359
1360                    let mut recollect_without = |collector: &mut Self| {
1361                        // Remove the original directive since we resolved it.
1362                        let mod_dir = collector.mod_dirs[&directive.module_id].clone();
1363                        collector
1364                            .skip_attrs
1365                            .insert(InFile::new(file_id, mod_item.ast_id()), attr.id);
1366
1367                        ModCollector {
1368                            def_collector: collector,
1369                            macro_depth: directive.depth,
1370                            module_id: directive.module_id,
1371                            tree_id: *tree,
1372                            item_tree,
1373                            mod_dir,
1374                        }
1375                        .collect(&[*mod_item], directive.container);
1376                        res = ReachedFixedPoint::No;
1377                        Resolved::Yes
1378                    };
1379
1380                    if let Some(ident) = path.as_ident()
1381                        && let Some(helpers) = self.def_map.derive_helpers_in_scope.get(&ast_id)
1382                        && helpers.iter().any(|(it, ..)| it == ident)
1383                    {
1384                        cov_mark::hit!(resolved_derive_helper);
1385                        // Resolved to derive helper. Collect the item's attributes again,
1386                        // starting after the derive helper.
1387                        return recollect_without(self);
1388                    }
1389
1390                    let def = match resolver_def_id(path) {
1391                        Some(def) if def.is_attribute() => def,
1392                        _ => return Resolved::No,
1393                    };
1394
1395                    // Skip #[test]/#[bench]/#[test_case] expansion, which would merely result in more memory usage
1396                    // due to duplicating functions into macro expansions, but only if `cfg(test)` is active,
1397                    // otherwise they are expanded to nothing and this can impact e.g. diagnostics (due to things
1398                    // being cfg'ed out).
1399                    // Ideally we will just expand them to nothing here. But we are only collecting macro calls,
1400                    // not expanding them, so we have no way to do that.
1401                    // If you add an ignored attribute here, also add it to `Semantics::might_be_inside_macro_call()`.
1402                    if matches!(
1403                        def.kind,
1404                        MacroDefKind::BuiltInAttr(_, expander)
1405                        if expander.is_test() || expander.is_bench() || expander.is_test_case()
1406                    ) {
1407                        let test_is_active = self.cfg_options.check_atom(&CfgAtom::Flag(sym::test));
1408                        if test_is_active {
1409                            return recollect_without(self);
1410                        }
1411                    }
1412
1413                    let call_id = || {
1414                        attr_macro_as_call_id(self.db, file_ast_id, attr, self.def_map.krate, def)
1415                    };
1416                    if matches!(def,
1417                        MacroDefId { kind: MacroDefKind::BuiltInAttr(_, exp), .. }
1418                        if exp.is_derive()
1419                    ) {
1420                        // Resolved to `#[derive]`, we don't actually expand this attribute like
1421                        // normal (as that would just be an identity expansion with extra output)
1422                        // Instead we treat derive attributes special and apply them separately.
1423
1424                        let ast_adt_id: FileAstId<ast::Adt> = match *mod_item {
1425                            ModItemId::Struct(ast_id) => ast_id.upcast(),
1426                            ModItemId::Union(ast_id) => ast_id.upcast(),
1427                            ModItemId::Enum(ast_id) => ast_id.upcast(),
1428                            _ => {
1429                                let diag = DefDiagnostic::invalid_derive_target(
1430                                    directive.module_id,
1431                                    ast_id,
1432                                    attr.id,
1433                                );
1434                                self.def_map.diagnostics.push(diag);
1435                                return recollect_without(self);
1436                            }
1437                        };
1438
1439                        let ast_id = ast_id.with_value(ast_adt_id);
1440
1441                        match attr.parse_path_comma_token_tree(self.db) {
1442                            Some(derive_macros) => {
1443                                let call_id = call_id();
1444                                let mut len = 0;
1445                                for (idx, (path, call_site)) in derive_macros.enumerate() {
1446                                    let ast_id = AstIdWithPath::new(
1447                                        file_id,
1448                                        ast_id.value,
1449                                        Interned::new(path),
1450                                    );
1451                                    self.unresolved_macros.push(MacroDirective {
1452                                        module_id: directive.module_id,
1453                                        depth: directive.depth + 1,
1454                                        kind: MacroDirectiveKind::Derive {
1455                                            ast_id,
1456                                            derive_attr: attr.id,
1457                                            derive_pos: idx,
1458                                            ctxt: call_site.ctx,
1459                                            derive_macro_id: call_id,
1460                                        },
1461                                        container: directive.container,
1462                                    });
1463                                    len = idx;
1464                                }
1465
1466                                // We treat the #[derive] macro as an attribute call, but we do not resolve it for nameres collection.
1467                                // This is just a trick to be able to resolve the input to derives
1468                                // as proper paths in `Semantics`.
1469                                // Check the comment in [`builtin_attr_macro`].
1470                                self.def_map.modules[directive.module_id]
1471                                    .scope
1472                                    .init_derive_attribute(ast_id, attr.id, call_id, len + 1);
1473                            }
1474                            None => {
1475                                let diag = DefDiagnostic::malformed_derive(
1476                                    directive.module_id,
1477                                    ast_id,
1478                                    attr.id,
1479                                );
1480                                self.def_map.diagnostics.push(diag);
1481                            }
1482                        }
1483
1484                        return recollect_without(self);
1485                    }
1486
1487                    if let MacroDefKind::ProcMacro(_, exp, _) = def.kind {
1488                        // If there's no expander for the proc macro (e.g.
1489                        // because proc macros are disabled, or building the
1490                        // proc macro crate failed), report this and skip
1491                        // expansion like we would if it was disabled
1492                        if let Some(err) = exp.as_expand_error(def.krate) {
1493                            self.def_map.diagnostics.push(DefDiagnostic::macro_error(
1494                                directive.module_id,
1495                                ast_id,
1496                                (**path).clone(),
1497                                err,
1498                            ));
1499                            return recollect_without(self);
1500                        }
1501                    }
1502
1503                    let call_id = call_id();
1504                    self.def_map.modules[directive.module_id]
1505                        .scope
1506                        .add_attr_macro_invoc(ast_id, call_id);
1507
1508                    push_resolved(directive, call_id);
1509                    res = ReachedFixedPoint::No;
1510                    return Resolved::Yes;
1511                }
1512            }
1513
1514            Resolved::No
1515        };
1516        macros.retain(|it| retain(it) == Resolved::No);
1517        // Attribute resolution can add unresolved macro invocations, so concatenate the lists.
1518        macros.extend(mem::take(&mut self.unresolved_macros));
1519        self.unresolved_macros = macros;
1520
1521        for (module_id, ptr, call_id) in eager_callback_buffer {
1522            self.def_map.modules[module_id].scope.add_macro_invoc(ptr.map(|(_, it)| it), call_id);
1523        }
1524
1525        for (module_id, depth, container, macro_call_id) in resolved {
1526            self.collect_macro_expansion(module_id, macro_call_id, depth, container);
1527        }
1528
1529        res
1530    }
1531
1532    fn collect_macro_expansion(
1533        &mut self,
1534        module_id: LocalModuleId,
1535        macro_call_id: MacroCallId,
1536        depth: usize,
1537        container: ItemContainerId,
1538    ) {
1539        if depth > self.def_map.recursion_limit() as usize {
1540            cov_mark::hit!(macro_expansion_overflow);
1541            tracing::warn!("macro expansion is too deep");
1542            return;
1543        }
1544        let file_id = macro_call_id.into();
1545
1546        let item_tree = self.db.file_item_tree(file_id);
1547
1548        let mod_dir = if macro_call_id.is_include_macro(self.db) {
1549            ModDir::root()
1550        } else {
1551            self.mod_dirs[&module_id].clone()
1552        };
1553
1554        ModCollector {
1555            def_collector: &mut *self,
1556            macro_depth: depth,
1557            tree_id: TreeId::new(file_id, None),
1558            module_id,
1559            item_tree,
1560            mod_dir,
1561        }
1562        .collect(item_tree.top_level_items(), container);
1563    }

1565    fn finish(mut self) -> (DefMap, LocalDefMap) {
1566        // Emit diagnostics for all remaining unexpanded macros.
1567        let _p = tracing::info_span!("DefCollector::finish").entered();
1568
1569        for directive in &self.unresolved_macros {
1570            match &directive.kind {
1571                MacroDirectiveKind::FnLike { ast_id, expand_to, ctxt: call_site } => {
1572                    // FIXME: we shouldn't need to re-resolve the macro here just to get the unresolved error!
1573                    let macro_call_as_call_id = macro_call_as_call_id(
1574                        self.db,
1575                        ast_id.ast_id,
1576                        &ast_id.path,
1577                        *call_site,
1578                        *expand_to,
1579                        self.def_map.krate,
1580                        |path| {
1581                            let resolved_res = self.def_map.resolve_path_fp_with_macro(
1582                                self.crate_local_def_map.unwrap_or(&self.local_def_map),
1583                                self.db,
1584                                ResolveMode::Other,
1585                                directive.module_id,
1586                                path,
1587                                BuiltinShadowMode::Module,
1588                                Some(MacroSubNs::Bang),
1589                            );
1590                            resolved_res.resolved_def.take_macros().map(|it| self.db.macro_def(it))
1591                        },
1592                        &mut |_, _| (),
1593                    );
1594                    if let Err(UnresolvedMacro { path }) = macro_call_as_call_id {
1595                        self.def_map.diagnostics.push(DefDiagnostic::unresolved_macro_call(
1596                            directive.module_id,
1597                            MacroCallKind::FnLike {
1598                                ast_id: ast_id.ast_id,
1599                                expand_to: *expand_to,
1600                                eager: None,
1601                            },
1602                            path,
1603                        ));
1604                    }
1605                }
1606                MacroDirectiveKind::Derive {
1607                    ast_id,
1608                    derive_attr,
1609                    derive_pos,
1610                    derive_macro_id,
1611                    ..
1612                } => {
1613                    self.def_map.diagnostics.push(DefDiagnostic::unresolved_macro_call(
1614                        directive.module_id,
1615                        MacroCallKind::Derive {
1616                            ast_id: ast_id.ast_id,
1617                            derive_attr_index: *derive_attr,
1618                            derive_index: *derive_pos as u32,
1619                            derive_macro_id: *derive_macro_id,
1620                        },
1621                        ast_id.path.as_ref().clone(),
1622                    ));
1623                }
1624                // These are diagnosed by `reseed_with_unresolved_attribute`, as that function consumes them
1625                MacroDirectiveKind::Attr { .. } => {}
1626            }
1627        }
1628
1629        // Emit diagnostics for all remaining unresolved imports.
1630        for import in &self.unresolved_imports {
1631            let &ImportDirective {
1632                module_id,
1633                import:
1634                    Import {
1635                        ref path,
1636                        source: ImportSource { use_tree, id, is_prelude: _, kind: _ },
1637                        ..
1638                    },
1639                ..
1640            } = import;
1641            if matches!(
1642                (path.segments().first(), &path.kind),
1643                (Some(krate), PathKind::Plain | PathKind::Abs) if self.unresolved_extern_crates.contains(krate)
1644            ) {
1645                continue;
1646            }
1647            let item_tree_id = id.lookup(self.db).id;
1648            self.def_map.diagnostics.push(DefDiagnostic::unresolved_import(
1649                module_id,
1650                item_tree_id,
1651                use_tree,
1652            ));
1653        }
1654
1655        (self.def_map, self.local_def_map)
1656    }

1659/// Walks a single module, populating defs, imports and macros
1660struct ModCollector<'a, 'db> {
1661    def_collector: &'a mut DefCollector<'db>,
1662    macro_depth: usize,
1663    module_id: LocalModuleId,
1664    tree_id: TreeId,
1665    item_tree: &'db ItemTree,
1666    mod_dir: ModDir,
1667}
1668
1669impl ModCollector<'_, '_> {
1670    fn collect_in_top_module(&mut self, items: &[ModItemId]) {
1671        let module = self.def_collector.def_map.module_id(self.module_id);
1672        self.collect(items, module.into())
1673    }
1674
1675    fn collect(&mut self, items: &[ModItemId], container: ItemContainerId) {
1676        let krate = self.def_collector.def_map.krate;
1677        let is_crate_root =
1678            self.module_id == DefMap::ROOT && self.def_collector.def_map.block.is_none();
1679
1680        // Note: don't assert that inserted value is fresh: it's simply not true
1681        // for macros.
1682        self.def_collector.mod_dirs.insert(self.module_id, self.mod_dir.clone());
1683
1684        // Prelude module is always considered to be `#[macro_use]`.
1685        if let Some((prelude_module, _use)) = self.def_collector.def_map.prelude {
1686            // Don't insert macros from the prelude into blocks, as they can be shadowed by other macros.
1687            if prelude_module.krate != krate && is_crate_root {
1688                cov_mark::hit!(prelude_is_macro_use);
1689                self.def_collector.import_macros_from_extern_crate(
1690                    prelude_module.krate,
1691                    None,
1692                    None,
1693                );
1694            }
1695        }
1696        let db = self.def_collector.db;
1697        let module_id = self.module_id;
1698        let update_def =
1699            |def_collector: &mut DefCollector<'_>, id, name: &Name, vis, has_constructor| {
1700                def_collector.def_map.modules[module_id].scope.declare(id);
1701                def_collector.update(
1702                    module_id,
1703                    &[(Some(name.clone()), PerNs::from_def(id, vis, has_constructor, None))],
1704                    vis,
1705                    None,
1706                )
1707            };
1708        let resolve_vis = |def_map: &DefMap, local_def_map: &LocalDefMap, visibility| {
1709            def_map
1710                .resolve_visibility(local_def_map, db, module_id, visibility, false)
1711                .unwrap_or(Visibility::Public)
1712        };
1713
1714        let mut process_mod_item = |item: ModItemId| {
1715            let attrs = self.item_tree.attrs(db, krate, item.ast_id());
1716            if let Some(cfg) = attrs.cfg()
1717                && !self.is_cfg_enabled(&cfg)
1718            {
1719                let ast_id = item.ast_id().erase();
1720                self.emit_unconfigured_diagnostic(InFile::new(self.file_id(), ast_id), &cfg);
1721                return;
1722            }
1723
1724            if let Err(()) = self.resolve_attributes(&attrs, item, container) {
1725                // Do not process the item. It has at least one non-builtin attribute, so the
1726                // fixed-point algorithm is required to resolve the rest of them.
1727                return;
1728            }
1729
1730            let module = self.def_collector.def_map.module_id(module_id);
1731            let def_map = &mut self.def_collector.def_map;
1732            let local_def_map =
1733                self.def_collector.crate_local_def_map.unwrap_or(&self.def_collector.local_def_map);
1734
1735            match item {
1736                ModItemId::Mod(m) => self.collect_module(m, &attrs),
1737                ModItemId::Use(item_tree_id) => {
1738                    let id =
1739                        UseLoc { container: module, id: InFile::new(self.file_id(), item_tree_id) }
1740                            .intern(db);
1741                    let is_prelude = attrs.by_key(sym::prelude_import).exists();
1742                    Import::from_use(self.item_tree, item_tree_id, id, is_prelude, |import| {
1743                        self.def_collector.unresolved_imports.push(ImportDirective {
1744                            module_id: self.module_id,
1745                            import,
1746                            status: PartialResolvedImport::Unresolved,
1747                        });
1748                    })
1749                }
1750                ModItemId::ExternCrate(item_tree_id) => {
1751                    let item_tree::ExternCrate { name, visibility, alias } =
1752                        &self.item_tree[item_tree_id];
1753
1754                    let id = ExternCrateLoc {
1755                        container: module,
1756                        id: InFile::new(self.tree_id.file_id(), item_tree_id),
1757                    }
1758                    .intern(db);
1759                    def_map.modules[self.module_id].scope.define_extern_crate_decl(id);
1760
1761                    let is_self = *name == sym::self_;
1762                    let resolved = if is_self {
1763                        cov_mark::hit!(extern_crate_self_as);
1764                        Some(def_map.crate_root())
1765                    } else {
1766                        self.def_collector
1767                            .deps
1768                            .get(name)
1769                            .map(|dep| CrateRootModuleId { krate: dep.crate_id })
1770                    };
1771
1772                    let name = match alias {
1773                        Some(ImportAlias::Alias(name)) => Some(name),
1774                        Some(ImportAlias::Underscore) => None,
1775                        None => Some(name),
1776                    };
1777
1778                    if let Some(resolved) = resolved {
1779                        let vis = resolve_vis(def_map, local_def_map, &self.item_tree[*visibility]);
1780
1781                        if is_crate_root {
1782                            // extern crates in the crate root are special-cased to insert entries into the extern prelude: rust-lang/rust#54658
1783                            if let Some(name) = name {
1784                                self.def_collector
1785                                    .local_def_map
1786                                    .extern_prelude
1787                                    .insert(name.clone(), (resolved, Some(id)));
1788                            }
1789                            // they also allow `#[macro_use]`
1790                            if !is_self {
1791                                self.process_macro_use_extern_crate(
1792                                    id,
1793                                    attrs.by_key(sym::macro_use).attrs(),
1794                                    resolved.krate,
1795                                );
1796                            }
1797                        }
1798
1799                        self.def_collector.update(
1800                            module_id,
1801                            &[(
1802                                name.cloned(),
1803                                PerNs::types(
1804                                    resolved.into(),
1805                                    vis,
1806                                    Some(ImportOrExternCrate::ExternCrate(id)),
1807                                ),
1808                            )],
1809                            vis,
1810                            Some(ImportOrExternCrate::ExternCrate(id)),
1811                        );
1812                    } else {
1813                        if let Some(name) = name {
1814                            self.def_collector.unresolved_extern_crates.insert(name.clone());
1815                        }
1816                        self.def_collector.def_map.diagnostics.push(
1817                            DefDiagnostic::unresolved_extern_crate(
1818                                module_id,
1819                                InFile::new(self.file_id(), item_tree_id),
1820                            ),
1821                        );
1822                    }
1823                }
1824                ModItemId::ExternBlock(block) => {
1825                    let extern_block_id = ExternBlockLoc {
1826                        container: module,
1827                        id: InFile::new(self.file_id(), block),
1828                    }
1829                    .intern(db);
1830                    self.def_collector.def_map.modules[self.module_id]
1831                        .scope
1832                        .define_extern_block(extern_block_id);
1833                    self.collect(
1834                        &self.item_tree[block].children,
1835                        ItemContainerId::ExternBlockId(extern_block_id),
1836                    )
1837                }

2427                        .with_ancestor_maps(db, self.module_id, &mut |map, module| {
2428                            map[module].scope.get_legacy_macro(name)?.last().copied()
2429                        })
2430                        .or_else(|| def_map[self.module_id].scope.get(name).take_macros())
2431                        .or_else(|| Some(def_map.macro_use_prelude.get(name).copied()?.0))
2432                        .filter(|&id| {
2433                            sub_namespace_match(
2434                                Some(MacroSubNs::from_id(db, id)),
2435                                Some(MacroSubNs::Bang),
2436                            )
2437                        })
2438                        .map(|it| self.def_collector.db.macro_def(it))
2439                })
2440            },
2441            &mut |ptr, call_id| eager_callback_buffer.push((ptr, call_id)),
2442        ) {
2443            for (ptr, call_id) in eager_callback_buffer {
2444                self.def_collector.def_map.modules[self.module_id]
2445                    .scope
2446                    .add_macro_invoc(ptr.map(|(_, it)| it), call_id);
2447            }
2448            // FIXME: if there were errors, this might've been in the eager expansion from an
2449            // unresolved macro, so we need to push this into late macro resolution. see fixme above
2450            if res.err.is_none() {
2451                // Legacy macros need to be expanded immediately, so that any macros they produce
2452                // are in scope.
2453                if let Some(call_id) = res.value {
2454                    self.def_collector.def_map.modules[self.module_id]
2455                        .scope
2456                        .add_macro_invoc(ast_id.ast_id, call_id);
2457                    self.def_collector.collect_macro_expansion(
2458                        self.module_id,
2459                        call_id,
2460                        self.macro_depth + 1,
2461                        container,
2462                    );
2463                }
2464
2465                return;
2466            }
2467        }
2468
2469        // Case 2: resolve in module scope, expand during name resolution.
2470        self.def_collector.unresolved_macros.push(MacroDirective {
2471            module_id: self.module_id,
2472            depth: self.macro_depth + 1,
2473            kind: MacroDirectiveKind::FnLike { ast_id, expand_to, ctxt },
2474            container,
2475        });
2476    }

2495    fn borrow_modules(
2496        modules: &mut [ModuleData],
2497        a: LocalModuleId,
2498        b: LocalModuleId,
2499    ) -> Option<(&mut ModuleData, &mut ModuleData)> {
2500        let a = a.into_raw().into_u32() as usize;
2501        let b = b.into_raw().into_u32() as usize;
2502
2503        let (a, b) = match a.cmp(&b) {
2504            Ordering::Equal => return None,
2505            Ordering::Less => {
2506                let (prefix, b) = modules.split_at_mut(b);
2507                (&mut prefix[a], &mut b[0])
2508            }
2509            Ordering::Greater => {
2510                let (prefix, a) = modules.split_at_mut(a);
2511                (&mut a[0], &mut prefix[b])
2512            }
2513        };
2514        Some((a, b))
2515    }
