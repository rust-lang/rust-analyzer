16#[derive(Debug, PartialEq, Eq)]
17pub enum DefDiagnosticKind {
18    UnresolvedModule { ast: AstId<ast::Module>, candidates: Box<[String]> },
19    UnresolvedExternCrate { ast: AstId<ast::ExternCrate> },
20    UnresolvedImport { id: ItemTreeId<item_tree::Use>, index: Idx<ast::UseTree> },
21    UnconfiguredCode { tree: TreeId, item: AttrOwner, cfg: CfgExpr, opts: CfgOptions },
22    UnresolvedMacroCall { ast: MacroCallKind, path: ModPath },
23    UnimplementedBuiltinMacro { ast: AstId<ast::Macro> },
24    InvalidDeriveTarget { ast: AstId<ast::Item>, id: usize },
25    MalformedDerive { ast: AstId<ast::Adt>, id: usize },
26    MacroDefError { ast: AstId<ast::Macro>, message: String },
27    MacroError { ast: AstId<ast::Item>, path: ModPath, err: ExpandErrorKind },
28}
29
30#[derive(Clone, Debug, PartialEq, Eq)]
31pub struct DefDiagnostics(Option<triomphe::Arc<Box<[DefDiagnostic]>>>);
32
33impl DefDiagnostics {
34    pub fn new(diagnostics: Vec<DefDiagnostic>) -> Self {
35        Self(
36            diagnostics
37                .is_empty()
38                .not()
39                .then(|| triomphe::Arc::new(diagnostics.into_boxed_slice())),
40        )
41    }
42
43    pub fn iter(&self) -> impl Iterator<Item = &DefDiagnostic> {
44        self.0.as_ref().into_iter().flat_map(|it| &***it)
45    }
46}
47
48#[derive(Debug, PartialEq, Eq)]
49pub struct DefDiagnostic {
50    pub in_module: LocalModuleId,
51    pub kind: DefDiagnosticKind,
52}
53
54impl DefDiagnostic {
55    pub(super) fn unresolved_module(
56        container: LocalModuleId,
57        declaration: AstId<ast::Module>,
58        candidates: Box<[String]>,
59    ) -> Self {
60        Self {
61            in_module: container,
62            kind: DefDiagnosticKind::UnresolvedModule { ast: declaration, candidates },
63        }
64    }
65
66    pub(super) fn unresolved_extern_crate(
67        container: LocalModuleId,
68        declaration: AstId<ast::ExternCrate>,
69    ) -> Self {
70        Self {
71            in_module: container,
72            kind: DefDiagnosticKind::UnresolvedExternCrate { ast: declaration },
73        }
74    }
75
76    pub(super) fn unresolved_import(
77        container: LocalModuleId,
78        id: ItemTreeId<item_tree::Use>,
79        index: Idx<ast::UseTree>,
80    ) -> Self {
81        Self { in_module: container, kind: DefDiagnosticKind::UnresolvedImport { id, index } }
82    }
83
84    pub fn macro_error(
85        container: LocalModuleId,
86        ast: AstId<ast::Item>,
87        path: ModPath,
88        err: ExpandErrorKind,
89    ) -> Self {
90        Self { in_module: container, kind: DefDiagnosticKind::MacroError { ast, path, err } }
91    }
92
93    pub fn unconfigured_code(
94        container: LocalModuleId,
95        tree: TreeId,
96        item: AttrOwner,
97        cfg: CfgExpr,
98        opts: CfgOptions,
99    ) -> Self {
100        Self {
101            in_module: container,
102            kind: DefDiagnosticKind::UnconfiguredCode { tree, item, cfg, opts },
103        }
104    }
105
106    // FIXME: Whats the difference between this and unresolved_proc_macro
107    pub(crate) fn unresolved_macro_call(
108        container: LocalModuleId,
109        ast: MacroCallKind,
110        path: ModPath,
111    ) -> Self {
112        Self { in_module: container, kind: DefDiagnosticKind::UnresolvedMacroCall { ast, path } }
113    }
114
115    pub(super) fn unimplemented_builtin_macro(
116        container: LocalModuleId,
117        ast: AstId<ast::Macro>,
118    ) -> Self {
119        Self { in_module: container, kind: DefDiagnosticKind::UnimplementedBuiltinMacro { ast } }
120    }
121
122    pub(super) fn invalid_derive_target(
123        container: LocalModuleId,
124        ast: AstId<ast::Item>,
125        id: AttrId,
126    ) -> Self {
127        Self {
128            in_module: container,
129            kind: DefDiagnosticKind::InvalidDeriveTarget { ast, id: id.ast_index() },
130        }
131    }
132
133    pub(super) fn malformed_derive(
134        container: LocalModuleId,
135        ast: AstId<ast::Adt>,
136        id: AttrId,
137    ) -> Self {
138        Self {
139            in_module: container,
140            kind: DefDiagnosticKind::MalformedDerive { ast, id: id.ast_index() },
141        }
142    }
