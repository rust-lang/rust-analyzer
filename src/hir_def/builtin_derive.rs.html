1//! Definition of builtin derive impls.
2//!
3//! To save time and memory, builtin derives are not really expanded. Instead, we record them
4//! and create their impls based on lowered data, see crates/hir-ty/src/builtin_derive.rs.
5

13};

15macro_rules! declare_enum {
16    ( $( $trait:ident => [ $( $method:ident ),* ] ),* $(,)? ) => {
17        #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
18        pub enum BuiltinDeriveImplTrait {
19            $( $trait, )*
20        }
21
22        #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
23        #[allow(non_camel_case_types)]
24        pub enum BuiltinDeriveImplMethod {
25            $( $( $method, )* )*
26        }
27
28        impl BuiltinDeriveImplTrait {
29            #[inline]
30            pub fn name(self) -> Symbol {
31                match self {
32                    $( Self::$trait => sym::$trait, )*
33                }
34            }
35
36            #[inline]
37            pub fn get_id(self, lang_items: &crate::lang_item::LangItems) -> Option<crate::TraitId> {
38                match self {
39                    $( Self::$trait => lang_items.$trait, )*
40                }
41            }
42
43            #[inline]
44            pub fn get_method(self, method_name: &Symbol) -> Option<BuiltinDeriveImplMethod> {
45                match self {
46                    $(
47                        Self::$trait => {
48                            match method_name {
49                                $( _ if *method_name == sym::$method => Some(BuiltinDeriveImplMethod::$method), )*
50                                _ => None,
51                            }
52                        }
53                    )*
54                }
55            }
56
57            #[inline]
58            pub fn all_methods(self) -> &'static [BuiltinDeriveImplMethod] {
59                match self {
60                    $( Self::$trait => &[ $(BuiltinDeriveImplMethod::$method),* ], )*
61                }
62            }
63        }
64
65        impl BuiltinDeriveImplMethod {
66            #[inline]
67            pub fn name(self) -> Symbol {
68                match self {
69                    $( $( BuiltinDeriveImplMethod::$method => sym::$method, )* )*
70                }
71            }
72        }
73    };
74}

90impl BuiltinDeriveImplTrait {
91    pub fn derive_macro(self, lang_items: &LangItems) -> Option<MacroId> {
92        match self {
93            BuiltinDeriveImplTrait::Copy => lang_items.CopyDerive,
94            BuiltinDeriveImplTrait::Clone => lang_items.CloneDerive,
95            BuiltinDeriveImplTrait::Default => lang_items.DefaultDerive,
96            BuiltinDeriveImplTrait::Debug => lang_items.DebugDerive,
97            BuiltinDeriveImplTrait::Hash => lang_items.HashDerive,
98            BuiltinDeriveImplTrait::Ord => lang_items.OrdDerive,
99            BuiltinDeriveImplTrait::PartialOrd => lang_items.PartialOrdDerive,
100            BuiltinDeriveImplTrait::Eq => lang_items.EqDerive,
101            BuiltinDeriveImplTrait::PartialEq => lang_items.PartialEqDerive,
102            BuiltinDeriveImplTrait::CoerceUnsized | BuiltinDeriveImplTrait::DispatchFromDyn => {
103                lang_items.CoercePointeeDerive
104            }
105        }
106    }
107}

109impl BuiltinDeriveImplMethod {
110    pub fn trait_method(
111        self,
112        db: &dyn DefDatabase,
113        impl_: BuiltinDeriveImplId,
114    ) -> Option<FunctionId> {
115        let loc = impl_.loc(db);
116        let lang_items = crate::lang_item::lang_items(db, loc.krate(db));
117        let trait_ = impl_.loc(db).trait_.get_id(lang_items)?;
118        trait_.trait_items(db).method_by_name(&Name::new_symbol_root(self.name()))
119    }
120}

122pub(crate) fn with_derive_traits(
123    derive: BuiltinDeriveExpander,
124    mut f: impl FnMut(BuiltinDeriveImplTrait),
125) {
126    let trait_ = match derive {
127        BuiltinDeriveExpander::Copy => BuiltinDeriveImplTrait::Copy,
128        BuiltinDeriveExpander::Clone => BuiltinDeriveImplTrait::Clone,
129        BuiltinDeriveExpander::Default => BuiltinDeriveImplTrait::Default,
130        BuiltinDeriveExpander::Debug => BuiltinDeriveImplTrait::Debug,
131        BuiltinDeriveExpander::Hash => BuiltinDeriveImplTrait::Hash,
132        BuiltinDeriveExpander::Ord => BuiltinDeriveImplTrait::Ord,
133        BuiltinDeriveExpander::PartialOrd => BuiltinDeriveImplTrait::PartialOrd,
134        BuiltinDeriveExpander::Eq => BuiltinDeriveImplTrait::Eq,
135        BuiltinDeriveExpander::PartialEq => BuiltinDeriveImplTrait::PartialEq,
136        BuiltinDeriveExpander::CoercePointee => {
137            f(BuiltinDeriveImplTrait::CoerceUnsized);
138            f(BuiltinDeriveImplTrait::DispatchFromDyn);
139            return;
140        }
141    };
142    f(trait_);
143}

145impl BuiltinDeriveImplLoc {
146    pub fn source(&self, db: &dyn DefDatabase) -> InFile<TextRange> {
147        let (adt_ast_id, module) = match self.adt {
148            AdtId::StructId(adt) => {
149                let adt_loc = adt.loc(db);
150                (adt_loc.id.upcast(), adt_loc.container)
151            }
152            AdtId::UnionId(adt) => {
153                let adt_loc = adt.loc(db);
154                (adt_loc.id.upcast(), adt_loc.container)
155            }
156            AdtId::EnumId(adt) => {
157                let adt_loc = adt.loc(db);
158                (adt_loc.id.upcast(), adt_loc.container)
159            }
160        };
161        let derive_range = self.derive_attr_id.find_derive_range(
162            db,
163            module.krate(db),
164            adt_ast_id,
165            self.derive_index,
166        );
167        adt_ast_id.with_value(derive_range)
168    }
