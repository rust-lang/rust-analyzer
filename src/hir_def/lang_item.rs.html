34/// Salsa query. This will look for lang items in a specific crate.
35#[salsa_macros::tracked(returns(as_deref))]
36pub fn crate_lang_items(db: &dyn DefDatabase, krate: Crate) -> Option<Box<LangItems>> {
37    let _p = tracing::info_span!("crate_lang_items_query").entered();
38
39    let mut lang_items = LangItems::default();
40
41    let crate_def_map = crate_def_map(db, krate);
42
43    if !crate_def_map.is_unstable_feature_enabled(&sym::lang_items) {
44        return None;
45    }
46
47    for (_, module_data) in crate_def_map.modules() {
48        for impl_def in module_data.scope.inherent_impls() {
49            lang_items.collect_lang_item(db, impl_def);
50            for &(_, assoc) in impl_def.impl_items(db).items.iter() {
51                match assoc {
52                    AssocItemId::FunctionId(f) => lang_items.collect_lang_item(db, f),
53                    AssocItemId::TypeAliasId(t) => lang_items.collect_lang_item(db, t),
54                    AssocItemId::ConstId(_) => (),
55                }
56            }
57        }
58
59        for def in module_data.scope.declarations() {
60            match def {
61                ModuleDefId::TraitId(trait_) => {
62                    lang_items.collect_lang_item(db, trait_);
63                    TraitItems::query(db, trait_).items.iter().for_each(|&(_, assoc_id)| {
64                        match assoc_id {
65                            AssocItemId::FunctionId(f) => {
66                                lang_items.collect_lang_item(db, f);
67                            }
68                            AssocItemId::TypeAliasId(alias) => {
69                                lang_items.collect_lang_item(db, alias)
70                            }
71                            AssocItemId::ConstId(_) => {}
72                        }
73                    });
74                }
75                ModuleDefId::AdtId(AdtId::EnumId(e)) => {
76                    lang_items.collect_lang_item(db, e);
77                    e.enum_variants(db).variants.iter().for_each(|&(id, _, _)| {
78                        lang_items.collect_lang_item(db, id);
79                    });
80                }
81                ModuleDefId::AdtId(AdtId::StructId(s)) => {
82                    lang_items.collect_lang_item(db, s);
83                }
84                ModuleDefId::AdtId(AdtId::UnionId(u)) => {
85                    lang_items.collect_lang_item(db, u);
86                }
87                ModuleDefId::FunctionId(f) => {
88                    lang_items.collect_lang_item(db, f);
89                }
90                ModuleDefId::StaticId(s) => {
91                    lang_items.collect_lang_item(db, s);
92                }
93                ModuleDefId::TypeAliasId(t) => {
94                    lang_items.collect_lang_item(db, t);
95                }
96                _ => {}
97            }
98        }
99    }
100
101    if matches!(krate.data(db).origin, base_db::CrateOrigin::Lang(base_db::LangCrateOrigin::Core)) {
102        lang_items.fill_non_lang_core_items(db, crate_def_map);
103    }
104
105    if lang_items.is_empty() { None } else { Some(Box::new(lang_items)) }
106}

108/// Salsa query. Look for a lang items, starting from the specified crate and recursively
109/// traversing its dependencies.
110#[salsa_macros::tracked(returns(ref))]
111pub fn lang_items(db: &dyn DefDatabase, start_crate: Crate) -> LangItems {
112    let _p = tracing::info_span!("lang_items_query").entered();
113
114    let mut result = crate_lang_items(db, start_crate).cloned().unwrap_or_default();
115
116    // Our `CrateGraph` eagerly inserts sysroot dependencies like `core` or `std` into dependencies
117    // even if the target crate has `#![no_std]`, `#![no_core]` or shadowed sysroot dependencies
118    // like `dependencies.std.path = ".."`. So we use `extern_prelude()` instead of
119    // `CrateData.dependencies` here, which has already come through such sysroot complexities
120    // while nameres.
121    //
122    // See https://github.com/rust-lang/rust-analyzer/pull/20475 for details.
123    for (_, (module, _)) in crate_local_def_map(db, start_crate).local(db).extern_prelude() {
124        // Some crates declares themselves as extern crate like `extern crate self as core`.
125        // Ignore these to prevent cycles.
126        let krate = module.krate(db);
127        if krate != start_crate {
128            result.merge_prefer_self(lang_items(db, krate));
129        }
130    }
131
132    result
133}

135impl LangItems {
136    fn collect_lang_item<T>(&mut self, db: &dyn DefDatabase, item: T)
137    where
138        T: Into<AttrDefId> + Into<LangItemTarget> + Copy,
139    {
140        let _p = tracing::info_span!("collect_lang_item").entered();
141        if let Some(lang_item) = AttrFlags::lang_item(db, item.into()) {
142            self.assign_lang_item(lang_item, item.into());
143        }
144    }
145}

147fn resolve_core_trait(
148    db: &dyn DefDatabase,
149    core_def_map: &DefMap,
150    modules: &[Symbol],
151    name: Symbol,
152) -> Option<TraitId> {
153    let mut current = &core_def_map[core_def_map.root];
154    for module in modules {
155        let Some((ModuleDefId::ModuleId(cur), _)) =
156            current.scope.type_(&Name::new_symbol_root(module.clone()))
157        else {
158            return None;
159        };
160        if cur.krate(db) != core_def_map.krate() || cur.block(db) != core_def_map.block_id() {
161            return None;
162        }
163        current = &core_def_map[cur];
164    }
165    let Some((ModuleDefId::TraitId(trait_), _)) = current.scope.type_(&Name::new_symbol_root(name))
166    else {
167        return None;
168    };
169    Some(trait_)
170}

172fn resolve_core_macro(
173    db: &dyn DefDatabase,
174    core_def_map: &DefMap,
175    modules: &[Symbol],
176    name: Symbol,
177) -> Option<MacroId> {
178    let mut current = &core_def_map[core_def_map.root];
179    for module in modules {
180        let Some((ModuleDefId::ModuleId(cur), _)) =
181            current.scope.type_(&Name::new_symbol_root(module.clone()))
182        else {
183            return None;
184        };
185        if cur.krate(db) != core_def_map.krate() || cur.block(db) != core_def_map.block_id() {
186            return None;
187        }
188        current = &core_def_map[cur];
189    }
190    current.scope.makro(&Name::new_symbol_root(name))
191}

193#[salsa::tracked(returns(as_deref))]
194pub(crate) fn crate_notable_traits(db: &dyn DefDatabase, krate: Crate) -> Option<Box<[TraitId]>> {
195    let mut traits = Vec::new();
196
197    let crate_def_map = crate_def_map(db, krate);
198
199    for (_, module_data) in crate_def_map.modules() {
200        for def in module_data.scope.declarations() {
201            if let ModuleDefId::TraitId(trait_) = def
202                && AttrFlags::query(db, trait_.into()).contains(AttrFlags::IS_DOC_NOTABLE_TRAIT)
203            {
204                traits.push(trait_);
205            }
206        }
207    }
208
209    if traits.is_empty() { None } else { Some(traits.into_iter().collect()) }
210}

212macro_rules! language_item_table {
213    (
214        $LangItems:ident =>
215        $( $(#[$attr:meta])* $lang_item:ident, $module:ident :: $name:ident, $target:ident; )*
216
217        @non_lang_core_traits:
218
219        $( core::$($non_lang_trait_module:ident)::*, $non_lang_trait:ident; )*
220
221        @non_lang_core_macros:
222
223        $( core::$($non_lang_macro_module:ident)::*, $non_lang_macro:ident, $non_lang_macro_field:ident; )*
224    ) => {
225        #[allow(non_snake_case)] // FIXME: Should we remove this?
226        #[derive(Debug, Default, Clone, PartialEq, Eq, Hash)]
227        pub struct $LangItems {
228            $(
229                $(#[$attr])*
230                pub $lang_item: Option<$target>,
231            )*
232            $(
233                pub $non_lang_trait: Option<TraitId>,
234            )*
235            $(
236                pub $non_lang_macro_field: Option<MacroId>,
237            )*
238        }
239
240        impl LangItems {
241            fn is_empty(&self) -> bool {
242                $( self.$lang_item.is_none() )&&*
243            }
244
245            /// Merges `self` with `other`, with preference to `self` items.
246            fn merge_prefer_self(&mut self, other: &Self) {
247                $( self.$lang_item = self.$lang_item.or(other.$lang_item); )*
248                $( self.$non_lang_trait = self.$non_lang_trait.or(other.$non_lang_trait); )*
249                $( self.$non_lang_macro_field = self.$non_lang_macro_field.or(other.$non_lang_macro_field); )*
250            }
251
252            fn assign_lang_item(&mut self, name: Symbol, target: LangItemTarget) {
253                match name {
254                    $(
255                        _ if name == $module::$name => {
256                            if let LangItemTarget::$target(target) = target {
257                                self.$lang_item = Some(target);
258                            }
259                        }
260                    )*
261                    _ => {}
262                }
263            }
264
265            fn fill_non_lang_core_items(&mut self, db: &dyn DefDatabase, core_def_map: &DefMap) {
266                $( self.$non_lang_trait = resolve_core_trait(db, core_def_map, &[ $(sym::$non_lang_trait_module),* ], sym::$non_lang_trait); )*
267                $( self.$non_lang_macro_field = resolve_core_macro(db, core_def_map, &[ $(sym::$non_lang_macro_module),* ], sym::$non_lang_macro); )*
268            }
269        }
270
271        #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
272        pub enum LangItemEnum {
273            $(
274                $(#[$attr])*
275                $lang_item,
276            )*
277        }
278
279        impl LangItemEnum {
280            #[inline]
281            pub fn from_lang_items(self, lang_items: &LangItems) -> Option<LangItemTarget> {
282                match self {
283                    $( LangItemEnum::$lang_item => lang_items.$lang_item.map(Into::into), )*
284                }
285            }
286
287            #[inline]
288            pub fn from_symbol(symbol: &Symbol) -> Option<Self> {
289                match symbol {
290                    $( _ if *symbol == $module::$name => Some(Self::$lang_item), )*
291                    _ => None,
292                }
293            }
294        }
295    }
296}
