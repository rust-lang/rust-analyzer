41pub(crate) fn unwrap_trivial_block(block_expr: ast::BlockExpr) -> ast::Expr {
42    extract_trivial_expression(&block_expr)
43        .filter(|expr| !expr.syntax().text().contains_char('\n'))
44        .unwrap_or_else(|| block_expr.into())
45}

47pub fn extract_trivial_expression(block_expr: &ast::BlockExpr) -> Option<ast::Expr> {
48    if block_expr.modifier().is_some() {
49        return None;
50    }
51    let stmt_list = block_expr.stmt_list()?;
52    let has_anything_else = |thing: &SyntaxNode| -> bool {
53        let mut non_trivial_children =
54            stmt_list.syntax().children_with_tokens().filter(|it| match it.kind() {
55                WHITESPACE | T!['{'] | T!['}'] => false,
56                _ => it.as_node() != Some(thing),
57            });
58        non_trivial_children.next().is_some()
59    };
60    if stmt_list
61        .syntax()
62        .children_with_tokens()
63        .filter_map(NodeOrToken::into_token)
64        .any(|token| token.kind() == syntax::SyntaxKind::COMMENT)
65    {
66        return None;
67    }
68
69    if let Some(expr) = stmt_list.tail_expr() {
70        if has_anything_else(expr.syntax()) {
71            return None;
72        }
73        return Some(expr);
74    }
75    // Unwrap `{ continue; }`
76    let stmt = stmt_list.statements().next()?;
77    if let ast::Stmt::ExprStmt(expr_stmt) = stmt {
78        if has_anything_else(expr_stmt.syntax()) {
79            return None;
80        }
81        let expr = expr_stmt.expr()?;
82        if matches!(expr.syntax().kind(), CONTINUE_EXPR | BREAK_EXPR | RETURN_EXPR) {
83            return Some(expr);
84        }
85    }
86    None
87}

89pub(crate) fn wrap_block(expr: &ast::Expr) -> ast::BlockExpr {
90    if let ast::Expr::BlockExpr(block) = expr
91        && let Some(first) = block.syntax().first_token()
92        && first.kind() == T!['{']
93    {
94        block.reset_indent()
95    } else {
96        make::block_expr(None, Some(expr.reset_indent().indent(1.into())))
97    }
98}

100/// This is a method with a heuristics to support test methods annotated with custom test annotations, such as
101/// `#[test_case(...)]`, `#[tokio::test]` and similar.
102/// Also a regular `#[test]` annotation is supported.
103///
104/// It may produce false positives, for example, `#[wasm_bindgen_test]` requires a different command to run the test,
105/// but it's better than not to have the runnables for the tests at all.
106pub fn test_related_attribute_syn(fn_def: &ast::Fn) -> Option<ast::Attr> {
107    fn_def.attrs().find_map(|attr| {
108        let path = attr.path()?;
109        let text = path.syntax().text().to_string();
110        if text.starts_with("test") || text.ends_with("test") { Some(attr) } else { None }
111    })
112}

121    No,

126    Only,
127    No,

130pub fn filter_assoc_items(
131    sema: &Semantics<'_, RootDatabase>,
132    items: &[hir::AssocItem],
133    default_methods: DefaultMethods,
134    ignore_items: IgnoreAssocItems,
135) -> Vec<InFile<ast::AssocItem>> {
136    return items
137        .iter()
138        .copied()
139        .filter(|assoc_item| {
140            if ignore_items == IgnoreAssocItems::DocHiddenAttrPresent
141                && assoc_item.attrs(sema.db).is_doc_hidden()
142            {
143                if let hir::AssocItem::Function(f) = assoc_item
144                    && !f.has_body(sema.db)
145                {
146                    return true;
147                }
148                return false;
149            }
150
151            true
152        })
153        // Note: This throws away items with no source.
154        .filter_map(|assoc_item| {
155            let item = match assoc_item {
156                hir::AssocItem::Function(it) => sema.source(it)?.map(ast::AssocItem::Fn),
157                hir::AssocItem::TypeAlias(it) => sema.source(it)?.map(ast::AssocItem::TypeAlias),
158                hir::AssocItem::Const(it) => sema.source(it)?.map(ast::AssocItem::Const),
159            };
160            Some(item)
161        })
162        .filter(has_def_name)
163        .filter(|it| match &it.value {
164            ast::AssocItem::Fn(def) => matches!(
165                (default_methods, def.body()),
166                (DefaultMethods::Only, Some(_)) | (DefaultMethods::No, None)
167            ),
168            ast::AssocItem::Const(def) => matches!(
169                (default_methods, def.body()),
170                (DefaultMethods::Only, Some(_)) | (DefaultMethods::No, None)
171            ),
172            _ => default_methods == DefaultMethods::No,
173        })
174        .collect();
175
176    fn has_def_name(item: &InFile<ast::AssocItem>) -> bool {
177        match &item.value {
178            ast::AssocItem::Fn(def) => def.name(),
179            ast::AssocItem::TypeAlias(def) => def.name(),
180            ast::AssocItem::Const(def) => def.name(),
181            ast::AssocItem::MacroCall(_) => None,
182        }
183        .is_some()
184    }
185}

187/// Given `original_items` retrieved from the trait definition (usually by
188/// [`filter_assoc_items()`]), clones each item for update and applies path transformation to it,
189/// then inserts into `impl_`. Returns the modified `impl_` and the first associated item that got
190/// inserted.
191#[must_use]
192pub fn add_trait_assoc_items_to_impl(
193    sema: &Semantics<'_, RootDatabase>,
194    config: &AssistConfig,
195    original_items: &[InFile<ast::AssocItem>],
196    trait_: hir::Trait,
197    impl_: &ast::Impl,
198    target_scope: &hir::SemanticsScope<'_>,
199) -> Vec<ast::AssocItem> {
200    let new_indent_level = IndentLevel::from_node(impl_.syntax()) + 1;
201    original_items
202        .iter()
203        .map(|InFile { file_id, value: original_item }| {
204            let mut cloned_item = {
205                if let Some(macro_file) = file_id.macro_file() {
206                    let span_map = sema.db.expansion_span_map(macro_file);
207                    let item_prettified = prettify_macro_expansion(
208                        sema.db,
209                        original_item.syntax().clone(),
210                        &span_map,
211                        target_scope.krate().into(),
212                    );
213                    if let Some(formatted) = ast::AssocItem::cast(item_prettified) {
214                        return formatted;
215                    } else {
216                        stdx::never!("formatted `AssocItem` could not be cast back to `AssocItem`");
217                    }
218                }
219                original_item
220            }
221            .reset_indent();
222
223            if let Some(source_scope) = sema.scope(original_item.syntax()) {
224                // FIXME: Paths in nested macros are not handled well. See
225                // `add_missing_impl_members::paths_in_nested_macro_should_get_transformed` test.
226                let transform =
227                    PathTransform::trait_impl(target_scope, &source_scope, trait_, impl_.clone());
228                cloned_item = ast::AssocItem::cast(transform.apply(cloned_item.syntax())).unwrap();
229            }
230            cloned_item.remove_attrs_and_docs();
231            cloned_item
232        })
233        .filter_map(|item| match item {
234            ast::AssocItem::Fn(fn_) if fn_.body().is_none() => {
235                let fn_ = fn_.clone_subtree();
236                let new_body = &make::block_expr(
237                    None,
238                    Some(match config.expr_fill_default {
239                        ExprFillDefaultMode::Todo => make::ext::expr_todo(),
240                        ExprFillDefaultMode::Underscore => make::ext::expr_underscore(),
241                        ExprFillDefaultMode::Default => make::ext::expr_todo(),
242                    }),
243                );
244                let new_body = AstNodeEdit::indent(new_body, IndentLevel::single());
245                let mut fn_editor = SyntaxEditor::new(fn_.syntax().clone());
246                fn_.replace_or_insert_body(&mut fn_editor, new_body);
247                let new_fn_ = fn_editor.finish().new_root().clone();
248                ast::AssocItem::cast(new_fn_)
249            }
250            ast::AssocItem::TypeAlias(type_alias) => {
251                let type_alias = type_alias.clone_subtree();
252                if let Some(type_bound_list) = type_alias.type_bound_list() {
253                    let mut type_alias_editor = SyntaxEditor::new(type_alias.syntax().clone());
254                    type_bound_list.remove(&mut type_alias_editor);
255                    let type_alias = type_alias_editor.finish().new_root().clone();
256                    ast::AssocItem::cast(type_alias)
257                } else {
258                    Some(ast::AssocItem::TypeAlias(type_alias))
259                }
260            }
261            item => Some(item),
262        })
263        .map(|item| AstNodeEdit::indent(&item, new_indent_level))
264        .collect()
265}

283fn invert_special_case(make: &SyntaxFactory, expr: &ast::Expr) -> Option<ast::Expr> {
284    match expr {
285        ast::Expr::BinExpr(bin) => {
286            let op_kind = bin.op_kind()?;
287            let rev_kind = match op_kind {
288                ast::BinaryOp::CmpOp(ast::CmpOp::Eq { negated }) => {
289                    ast::BinaryOp::CmpOp(ast::CmpOp::Eq { negated: !negated })
290                }
291                ast::BinaryOp::CmpOp(ast::CmpOp::Ord { ordering: ast::Ordering::Less, strict }) => {
292                    ast::BinaryOp::CmpOp(ast::CmpOp::Ord {
293                        ordering: ast::Ordering::Greater,
294                        strict: !strict,
295                    })
296                }
297                ast::BinaryOp::CmpOp(ast::CmpOp::Ord {
298                    ordering: ast::Ordering::Greater,
299                    strict,
300                }) => ast::BinaryOp::CmpOp(ast::CmpOp::Ord {
301                    ordering: ast::Ordering::Less,
302                    strict: !strict,
303                }),
304                // Parenthesize other expressions before prefixing `!`
305                _ => {
306                    return Some(
307                        make.expr_prefix(T![!], make.expr_paren(expr.clone()).into()).into(),
308                    );
309                }
310            };
311
312            Some(make.expr_bin(bin.lhs()?, rev_kind, bin.rhs()?).into())
313        }
314        ast::Expr::MethodCallExpr(mce) => {
315            let receiver = mce.receiver()?;
316            let method = mce.name_ref()?;
317            let arg_list = mce.arg_list()?;
318
319            let method = match method.text().as_str() {
320                "is_some" => "is_none",
321                "is_none" => "is_some",
322                "is_ok" => "is_err",
323                "is_err" => "is_ok",
324                _ => return None,
325            };
326
327            Some(make.expr_method_call(receiver, make.name_ref(method), arg_list).into())
328        }
329        ast::Expr::PrefixExpr(pe) if pe.op_kind()? == ast::UnaryOp::Not => match pe.expr()? {
330            ast::Expr::ParenExpr(parexpr) => {
331                parexpr.expr().map(|e| e.clone_subtree().clone_for_update())
332            }
333            _ => pe.expr().map(|e| e.clone_subtree().clone_for_update()),
334        },
335        ast::Expr::Literal(lit) => match lit.kind() {
336            ast::LiteralKind::Bool(b) => match b {
337                true => Some(ast::Expr::Literal(make.expr_literal("false"))),
338                false => Some(ast::Expr::Literal(make.expr_literal("true"))),
339            },
340            _ => None,
341        },
342        _ => None,
343    }
344}

346fn invert_special_case_legacy(expr: &ast::Expr) -> Option<ast::Expr> {
347    match expr {
348        ast::Expr::BinExpr(bin) => {
349            let bin = bin.clone_subtree();
350            let op_token = bin.op_token()?;
351            let rev_token = match op_token.kind() {
352                T![==] => T![!=],
353                T![!=] => T![==],
354                T![<] => T![>=],
355                T![<=] => T![>],
356                T![>] => T![<=],
357                T![>=] => T![<],
358                // Parenthesize other expressions before prefixing `!`
359                _ => {
360                    return Some(
361                        make::expr_prefix(T![!], make::expr_paren(expr.clone()).into()).into(),
362                    );
363                }
364            };
365            let mut bin_editor = SyntaxEditor::new(bin.syntax().clone());
366            bin_editor.replace(op_token, make::token(rev_token));
367            ast::Expr::cast(bin_editor.finish().new_root().clone())
368        }
369        ast::Expr::MethodCallExpr(mce) => {
370            let receiver = mce.receiver()?;
371            let method = mce.name_ref()?;
372            let arg_list = mce.arg_list()?;
373
374            let method = match method.text().as_str() {
375                "is_some" => "is_none",
376                "is_none" => "is_some",
377                "is_ok" => "is_err",
378                "is_err" => "is_ok",
379                _ => return None,
380            };
381            Some(make::expr_method_call(receiver, make::name_ref(method), arg_list).into())
382        }
383        ast::Expr::PrefixExpr(pe) if pe.op_kind()? == ast::UnaryOp::Not => match pe.expr()? {
384            ast::Expr::ParenExpr(parexpr) => parexpr.expr(),
385            _ => pe.expr(),
386        },
387        ast::Expr::Literal(lit) => match lit.kind() {
388            ast::LiteralKind::Bool(b) => match b {
389                true => Some(ast::Expr::Literal(make::expr_literal("false"))),
390                false => Some(ast::Expr::Literal(make::expr_literal("true"))),
391            },
392            _ => None,
393        },
394        _ => None,
395    }
396}

398pub(crate) fn insert_attributes(
399    before: impl Element,
400    edit: &mut SyntaxEditor,
401    attrs: impl IntoIterator<Item = ast::Attr>,
402) {
403    let mut attrs = attrs.into_iter().peekable();
404    if attrs.peek().is_none() {
405        return;
406    }
407    let elem = before.syntax_element();
408    let indent = IndentLevel::from_element(&elem);
409    let whitespace = format!("\n{indent}");
410    edit.insert_all(
411        syntax::syntax_editor::Position::before(elem),
412        attrs
413            .flat_map(|attr| {
414                [attr.syntax().clone().into(), make::tokens::whitespace(&whitespace).into()]
415            })
416            .collect(),
417    );
418}

424pub(crate) fn does_pat_match_variant(pat: &ast::Pat, var: &ast::Pat) -> bool {
425    let first_node_text = |pat: &ast::Pat| pat.syntax().first_child().map(|node| node.text());
426
427    let pat_head = match pat {
428        ast::Pat::IdentPat(bind_pat) => match bind_pat.pat() {
429            Some(p) => first_node_text(&p),
430            None => return pat.syntax().text() == var.syntax().text(),
431        },
432        pat => first_node_text(pat),
433    };
434
435    let var_head = first_node_text(var);
436
437    pat_head == var_head
438}

440pub(crate) fn does_pat_variant_nested_or_literal(ctx: &AssistContext<'_>, pat: &ast::Pat) -> bool {

444fn check_pat_variant_from_enum(ctx: &AssistContext<'_>, pat: &ast::Pat) -> bool {
445    ctx.sema.type_of_pat(pat).is_none_or(|ty: hir::TypeInfo<'_>| {
446        ty.adjusted().as_adt().is_some_and(|adt| matches!(adt, hir::Adt::Enum(_)))
447    })

450fn check_pat_variant_nested_or_literal_with_depth(
451    ctx: &AssistContext<'_>,
452    pat: &ast::Pat,
453    depth_after_refutable: usize,
454) -> bool {
455    if depth_after_refutable > 1 {
456        return true;
457    }
458
459    match pat {
460        ast::Pat::RestPat(_) | ast::Pat::WildcardPat(_) | ast::Pat::RefPat(_) => false,
461
462        ast::Pat::LiteralPat(_)
463        | ast::Pat::RangePat(_)
464        | ast::Pat::MacroPat(_)
465        | ast::Pat::PathPat(_)
466        | ast::Pat::BoxPat(_)
467        | ast::Pat::ConstBlockPat(_) => true,
468
469        ast::Pat::IdentPat(ident_pat) => ident_pat.pat().is_some_and(|pat| {
470            check_pat_variant_nested_or_literal_with_depth(ctx, &pat, depth_after_refutable)
471        }),
472        ast::Pat::ParenPat(paren_pat) => paren_pat.pat().is_none_or(|pat| {
473            check_pat_variant_nested_or_literal_with_depth(ctx, &pat, depth_after_refutable)
474        }),
475        ast::Pat::TuplePat(tuple_pat) => tuple_pat.fields().any(|pat| {
476            check_pat_variant_nested_or_literal_with_depth(ctx, &pat, depth_after_refutable)
477        }),
478        ast::Pat::RecordPat(record_pat) => {
479            let adjusted_next_depth =
480                depth_after_refutable + if check_pat_variant_from_enum(ctx, pat) { 1 } else { 0 };
481            record_pat.record_pat_field_list().is_none_or(|pat| {
482                pat.fields().any(|pat| {
483                    pat.pat().is_none_or(|pat| {
484                        check_pat_variant_nested_or_literal_with_depth(
485                            ctx,
486                            &pat,
487                            adjusted_next_depth,
488                        )
489                    })
490                })
491            })
492        }
493        ast::Pat::OrPat(or_pat) => or_pat.pats().any(|pat| {
494            check_pat_variant_nested_or_literal_with_depth(ctx, &pat, depth_after_refutable)
495        }),
496        ast::Pat::TupleStructPat(tuple_struct_pat) => {
497            let adjusted_next_depth =
498                depth_after_refutable + if check_pat_variant_from_enum(ctx, pat) { 1 } else { 0 };
499            tuple_struct_pat.fields().any(|pat| {
500                check_pat_variant_nested_or_literal_with_depth(ctx, &pat, adjusted_next_depth)
501            })
502        }
503        ast::Pat::SlicePat(slice_pat) => {
504            let mut pats = slice_pat.pats();
505            pats.next()
506                .is_none_or(|pat| !matches!(pat, ast::Pat::RestPat(_)) || pats.next().is_some())
507        }
508    }
509}

511// Uses a syntax-driven approach to find any impl blocks for the struct that
512// exist within the module/file
513//
514// Returns `None` if we've found an existing fn
515//
516// FIXME: change the new fn checking to a more semantic approach when that's more
517// viable (e.g. we process proc macros, etc)
518// FIXME: this partially overlaps with `find_impl_block_*`
519
520/// `find_struct_impl` looks for impl of a struct, but this also has additional feature
521/// where it takes a list of function names and check if they exist inside impl_, if
522/// even one match is found, it returns None.
523///
524/// That means this function can have 3 potential return values:
525///  - `None`: an impl exists, but one of the function names within the impl matches one of the provided names.
526///  - `Some(None)`: no impl exists.
527///  - `Some(Some(_))`: an impl exists, with no matching function names.
528pub(crate) fn find_struct_impl(
529    ctx: &AssistContext<'_>,
530    adt: &ast::Adt,
531    names: &[String],
532) -> Option<Option<ast::Impl>> {
533    let db = ctx.db();
534    let module = adt.syntax().parent()?;
535
536    let struct_def = ctx.sema.to_def(adt)?;
537
538    let block = module.descendants().filter_map(ast::Impl::cast).find_map(|impl_blk| {
539        let blk = ctx.sema.to_def(&impl_blk)?;
540
541        // FIXME: handle e.g. `struct S<T>; impl<U> S<U> {}`
542        // (we currently use the wrong type parameter)
543        // also we wouldn't want to use e.g. `impl S<u32>`
544
545        let same_ty = match blk.self_ty(db).as_adt() {
546            Some(def) => def == struct_def,
547            None => false,
548        };
549        let not_trait_impl = blk.trait_(db).is_none();
550
551        if !(same_ty && not_trait_impl) { None } else { Some(impl_blk) }
552    });
553
554    if let Some(ref impl_blk) = block
555        && has_any_fn(impl_blk, names)
556    {
557        return None;
558    }
559
560    Some(block)
561}
562
563fn has_any_fn(imp: &ast::Impl, names: &[String]) -> bool {
564    if let Some(il) = imp.assoc_item_list() {
565        for item in il.assoc_items() {
566            if let ast::AssocItem::Fn(f) = item
567                && let Some(name) = f.name()
568                && names.iter().any(|n| n.eq_ignore_ascii_case(&name.text()))
569            {
570                return true;
571            }
572        }
573    }
574
575    false
576}

581pub(crate) fn find_impl_block_end(impl_def: ast::Impl, buf: &mut String) -> Option<TextSize> {
582    buf.push('\n');
583    let end = impl_def
584        .assoc_item_list()
585        .and_then(|it| it.r_curly_token())?
586        .prev_sibling_or_token()?
587        .text_range()
588        .end();
589    Some(end)
590}

622fn generate_impl_text_inner(
623    adt: &ast::Adt,
624    trait_text: Option<&str>,
625    trait_is_transitive: bool,
626    code: &str,
627) -> String {
628    // Ensure lifetime params are before type & const params
629    let generic_params = adt.generic_param_list().map(|generic_params| {
630        let lifetime_params =
631            generic_params.lifetime_params().map(ast::GenericParam::LifetimeParam);
632        let ty_or_const_params = generic_params.type_or_const_params().filter_map(|param| {
633            let param = match param {
634                ast::TypeOrConstParam::Type(param) => {
635                    // remove defaults since they can't be specified in impls
636                    let mut bounds =
637                        param.type_bound_list().map_or_else(Vec::new, |it| it.bounds().collect());
638                    if let Some(trait_) = trait_text {
639                        // Add the current trait to `bounds` if the trait is transitive,
640                        // meaning `impl<T> Trait for U<T>` requires `T: Trait`.
641                        if trait_is_transitive {
642                            bounds.push(make::type_bound_text(trait_));
643                        }
644                    };
645                    // `{ty_param}: {bounds}`
646                    let param = make::type_param(param.name()?, make::type_bound_list(bounds));
647                    ast::GenericParam::TypeParam(param)
648                }
649                ast::TypeOrConstParam::Const(param) => {
650                    // remove defaults since they can't be specified in impls
651                    let param = make::const_param(param.name()?, param.ty()?);
652                    ast::GenericParam::ConstParam(param)
653                }
654            };
655            Some(param)
656        });
657
658        make::generic_param_list(itertools::chain(lifetime_params, ty_or_const_params))
659    });
660
661    // FIXME: use syntax::make & mutable AST apis instead
662    // `trait_text` and `code` can't be opaque blobs of text
663    let mut buf = String::with_capacity(code.len());
664
665    // Copy any cfg attrs from the original adt
666    buf.push_str("\n\n");
667    let cfg_attrs = adt
668        .attrs()
669        .filter(|attr| attr.as_simple_call().map(|(name, _arg)| name == "cfg").unwrap_or(false));
670    cfg_attrs.for_each(|attr| buf.push_str(&format!("{attr}\n")));
671
672    // `impl{generic_params} {trait_text} for {name}{generic_params.to_generic_args()}`
673    buf.push_str("impl");
674    if let Some(generic_params) = &generic_params {
675        format_to!(buf, "{generic_params}");
676    }
677    buf.push(' ');
678    if let Some(trait_text) = trait_text {
679        buf.push_str(trait_text);
680        buf.push_str(" for ");
681    }
682    buf.push_str(&adt.name().unwrap().text());
683    if let Some(generic_params) = generic_params {
684        format_to!(buf, "{}", generic_params.to_generic_args());
685    }
686
687    match adt.where_clause() {
688        Some(where_clause) => {
689            format_to!(buf, "\n{where_clause}\n{{\n{code}\n}}");
690        }
691        None => {
692            format_to!(buf, " {{\n{code}\n}}");
693        }
694    }
695
696    buf
697}

728fn generate_impl_inner(
729    is_unsafe: bool,
730    adt: &ast::Adt,
731    trait_: Option<ast::Type>,
732    trait_is_transitive: bool,
733    body: Option<ast::AssocItemList>,
734) -> ast::Impl {
735    // Ensure lifetime params are before type & const params
736    let generic_params = adt.generic_param_list().map(|generic_params| {
737        let lifetime_params =
738            generic_params.lifetime_params().map(ast::GenericParam::LifetimeParam);
739        let ty_or_const_params = generic_params.type_or_const_params().filter_map(|param| {
740            let param = match param {
741                ast::TypeOrConstParam::Type(param) => {
742                    // remove defaults since they can't be specified in impls
743                    let mut bounds =
744                        param.type_bound_list().map_or_else(Vec::new, |it| it.bounds().collect());
745                    if let Some(trait_) = &trait_ {
746                        // Add the current trait to `bounds` if the trait is transitive,
747                        // meaning `impl<T> Trait for U<T>` requires `T: Trait`.
748                        if trait_is_transitive {
749                            bounds.push(make::type_bound(trait_.clone()));
750                        }
751                    };
752                    // `{ty_param}: {bounds}`
753                    let param = make::type_param(param.name()?, make::type_bound_list(bounds));
754                    ast::GenericParam::TypeParam(param)
755                }
756                ast::TypeOrConstParam::Const(param) => {
757                    // remove defaults since they can't be specified in impls
758                    let param = make::const_param(param.name()?, param.ty()?);
759                    ast::GenericParam::ConstParam(param)
760                }
761            };
762            Some(param)
763        });
764
765        make::generic_param_list(itertools::chain(lifetime_params, ty_or_const_params))
766    });
767    let generic_args =
768        generic_params.as_ref().map(|params| params.to_generic_args().clone_for_update());
769    let adt_assoc_bounds = trait_
770        .as_ref()
771        .zip(generic_params.as_ref())
772        .and_then(|(trait_, params)| generic_param_associated_bounds(adt, trait_, params));
773
774    let ty = make::ty_path(make::ext::ident_path(&adt.name().unwrap().text()));
775
776    let cfg_attrs =
777        adt.attrs().filter(|attr| attr.as_simple_call().is_some_and(|(name, _arg)| name == "cfg"));
778    match trait_ {
779        Some(trait_) => make::impl_trait(
780            cfg_attrs,
781            is_unsafe,
782            None,
783            None,
784            generic_params,
785            generic_args,
786            false,
787            trait_,
788            ty,
789            adt_assoc_bounds,
790            adt.where_clause(),
791            body,
792        ),
793        None => make::impl_(cfg_attrs, generic_params, generic_args, ty, adt.where_clause(), body),
794    }
795    .clone_for_update()
796}

798fn generic_param_associated_bounds(
799    adt: &ast::Adt,
800    trait_: &ast::Type,
801    generic_params: &ast::GenericParamList,
802) -> Option<ast::WhereClause> {
803    let in_type_params = |name: &ast::NameRef| {
804        generic_params
805            .generic_params()
806            .filter_map(|param| match param {
807                ast::GenericParam::TypeParam(type_param) => type_param.name(),
808                _ => None,
809            })
810            .any(|param| param.text() == name.text())
811    };
812    let adt_body = match adt {
813        ast::Adt::Enum(e) => e.variant_list().map(|it| it.syntax().clone()),
814        ast::Adt::Struct(s) => s.field_list().map(|it| it.syntax().clone()),
815        ast::Adt::Union(u) => u.record_field_list().map(|it| it.syntax().clone()),
816    };
817    let mut trait_where_clause = adt_body
818        .into_iter()
819        .flat_map(|it| it.descendants())
820        .filter_map(ast::Path::cast)
821        .filter_map(|path| {
822            let qualifier = path.qualifier()?.as_single_segment()?;
823            let qualifier = qualifier
824                .name_ref()
825                .or_else(|| match qualifier.type_anchor()?.ty()? {
826                    ast::Type::PathType(path_type) => path_type.path()?.as_single_name_ref(),
827                    _ => None,
828                })
829                .filter(in_type_params)?;
830            Some((qualifier, path.segment()?.name_ref()?))
831        })
832        .map(|(qualifier, assoc_name)| {
833            let segments = [qualifier, assoc_name].map(make::path_segment);
834            let path = make::path_from_segments(segments, false);
835            let bounds = Some(make::type_bound(trait_.clone()));
836            make::where_pred(either::Either::Right(make::ty_path(path)), bounds)
837        })
838        .unique_by(|it| it.syntax().to_string())
839        .peekable();
840    trait_where_clause.peek().is_some().then(|| make::where_clause(trait_where_clause))
841}

844    builder: &mut SourceChangeBuilder,
845    adt: &ast::Adt,
846    impl_def: Option<ast::Impl>,
847    method: &str,
848) {
849    let mut buf = String::with_capacity(method.len() + 2);
850    if impl_def.is_some() {

852    }

865#[derive(Debug)]
866pub(crate) struct ReferenceConversion<'db> {
867    conversion: ReferenceConversionType,
868    ty: hir::Type<'db>,
869    impls_deref: bool,
870}
871
872#[derive(Debug)]
873enum ReferenceConversionType {
874    // reference can be stripped if the type is Copy
875    Copy,
876    // &String -> &str
877    AsRefStr,
878    // &Vec<T> -> &[T]
879    AsRefSlice,
880    // &Box<T> -> &T
881    Dereferenced,
882    // &Option<T> -> Option<&T>
883    Option,
884    // &Result<T, E> -> Result<&T, &E>
885    Result,
886}
887
888impl<'db> ReferenceConversion<'db> {
889    pub(crate) fn convert_type(&self, db: &'db dyn HirDatabase, module: hir::Module) -> ast::Type {
890        let ty = match self.conversion {
891            ReferenceConversionType::Copy => self
892                .ty
893                .display_source_code(db, module.into(), true)
894                .unwrap_or_else(|_| "_".to_owned()),
895            ReferenceConversionType::AsRefStr => "&str".to_owned(),
896            ReferenceConversionType::AsRefSlice => {
897                let type_argument_name = self
898                    .ty
899                    .type_arguments()
900                    .next()
901                    .unwrap()
902                    .display_source_code(db, module.into(), true)
903                    .unwrap_or_else(|_| "_".to_owned());
904                format!("&[{type_argument_name}]")
905            }
906            ReferenceConversionType::Dereferenced => {
907                let type_argument_name = self
908                    .ty
909                    .type_arguments()
910                    .next()
911                    .unwrap()
912                    .display_source_code(db, module.into(), true)
913                    .unwrap_or_else(|_| "_".to_owned());
914                format!("&{type_argument_name}")
915            }
916            ReferenceConversionType::Option => {
917                let type_argument_name = self
918                    .ty
919                    .type_arguments()
920                    .next()
921                    .unwrap()
922                    .display_source_code(db, module.into(), true)
923                    .unwrap_or_else(|_| "_".to_owned());
924                format!("Option<&{type_argument_name}>")
925            }
926            ReferenceConversionType::Result => {
927                let mut type_arguments = self.ty.type_arguments();
928                let first_type_argument_name = type_arguments
929                    .next()
930                    .unwrap()
931                    .display_source_code(db, module.into(), true)
932                    .unwrap_or_else(|_| "_".to_owned());
933                let second_type_argument_name = type_arguments
934                    .next()
935                    .unwrap()
936                    .display_source_code(db, module.into(), true)
937                    .unwrap_or_else(|_| "_".to_owned());
938                format!("Result<&{first_type_argument_name}, &{second_type_argument_name}>")
939            }
940        };
941
942        make::ty(&ty)
943    }
944
945    pub(crate) fn getter(&self, field_name: String) -> ast::Expr {
946        let expr = make::expr_field(make::ext::expr_self(), &field_name);
947
948        match self.conversion {
949            ReferenceConversionType::Copy => expr,
950            ReferenceConversionType::AsRefStr
951            | ReferenceConversionType::AsRefSlice
952            | ReferenceConversionType::Dereferenced
953            | ReferenceConversionType::Option
954            | ReferenceConversionType::Result => {
955                if self.impls_deref {
956                    make::expr_ref(expr, false)
957                } else {
958                    make::expr_method_call(expr, make::name_ref("as_ref"), make::arg_list([]))
959                        .into()
960                }
961            }
962        }
963    }
964}

1070/// Trim(remove leading and trailing whitespace) `initial_range` in `source_file`, return the trimmed range.
1071pub(crate) fn trimmed_text_range(source_file: &SourceFile, initial_range: TextRange) -> TextRange {
1072    let mut trimmed_range = initial_range;
1073    while source_file
1074        .syntax()
1075        .token_at_offset(trimmed_range.start())
1076        .find_map(Whitespace::cast)
1077        .is_some()
1078        && trimmed_range.start() < trimmed_range.end()
1079    {
1080        let start = trimmed_range.start() + TextSize::from(1);
1081        trimmed_range = TextRange::new(start, trimmed_range.end());
1082    }
1083    while source_file
1084        .syntax()
1085        .token_at_offset(trimmed_range.end())
1086        .find_map(Whitespace::cast)
1087        .is_some()
1088        && trimmed_range.start() < trimmed_range.end()
1089    {
1090        let end = trimmed_range.end() - TextSize::from(1);
1091        trimmed_range = TextRange::new(trimmed_range.start(), end);
1092    }
1093    trimmed_range
1094}

1097/// into a function call.

1112/// Calculate the number of hashes required for a raw string containing `s`
1113pub(crate) fn required_hashes(s: &str) -> usize {
1114    let mut res = 0usize;
1115    for idx in s.match_indices('"').map(|(i, _)| i) {
1116        let (_, sub) = s.split_at(idx + 1);
1117        let n_hashes = sub.chars().take_while(|c| *c == '#').count();
1118        res = res.max(n_hashes + 1)
1119    }
1120    res
1121}

1124    assert_eq!(0, required_hashes("abc"));
1125    assert_eq!(0, required_hashes("###"));
1126    assert_eq!(1, required_hashes("\""));
1127    assert_eq!(2, required_hashes("\"#abc"));
1128    assert_eq!(0, required_hashes("#abc"));
1129    assert_eq!(3, required_hashes("#ab\"##c"));
1130    assert_eq!(5, required_hashes("#ab\"##\"####c"));
1131}

1192/// Creates a token tree list from a syntax node, creating the needed delimited sub token trees.
1193/// Assumes that the input syntax node is a valid syntax tree.
1194pub(crate) fn tt_from_syntax(node: SyntaxNode) -> Vec<NodeOrToken<ast::TokenTree, SyntaxToken>> {
1195    let mut tt_stack = vec![(None, vec![])];
1196
1197    for element in node.descendants_with_tokens() {
1198        let NodeOrToken::Token(token) = element else { continue };
1199
1200        match token.kind() {
1201            T!['('] | T!['{'] | T!['['] => {
1202                // Found an opening delimiter, start a new sub token tree
1203                tt_stack.push((Some(token.kind()), vec![]));
1204            }
1205            T![')'] | T!['}'] | T![']'] => {
1206                // Closing a subtree
1207                let (delimiter, tt) = tt_stack.pop().expect("unbalanced delimiters");
1208                let (_, parent_tt) = tt_stack
1209                    .last_mut()
1210                    .expect("parent token tree was closed before it was completed");
1211                let closing_delimiter = delimiter.map(|it| match it {
1212                    T!['('] => T![')'],
1213                    T!['{'] => T!['}'],
1214                    T!['['] => T![']'],
1215                    _ => unreachable!(),
1216                });
1217                stdx::always!(
1218                    closing_delimiter == Some(token.kind()),
1219                    "mismatched opening and closing delimiters"
1220                );
1221
1222                let sub_tt = make::token_tree(delimiter.expect("unbalanced delimiters"), tt);
1223                parent_tt.push(NodeOrToken::Node(sub_tt));
1224            }
1225            _ => {
1226                let (_, current_tt) = tt_stack.last_mut().expect("unmatched delimiters");
1227                current_tt.push(NodeOrToken::Token(token))
1228            }
1229        }
1230    }
1231
1232    tt_stack.pop().expect("parent token tree was closed before it was completed").1
1233}

1235pub(crate) fn cover_let_chain(mut expr: ast::Expr, range: TextRange) -> Option<ast::Expr> {
1236    if !expr.syntax().text_range().contains_range(range) {
1237        return None;
1238    }
1239    loop {
1240        let (chain_expr, rest) = if let ast::Expr::BinExpr(bin_expr) = &expr
1241            && bin_expr.op_kind() == Some(ast::BinaryOp::LogicOp(ast::LogicOp::And))
1242        {
1243            (bin_expr.rhs(), bin_expr.lhs())
1244        } else {
1245            (Some(expr), None)
1246        };
1247
1248        if let Some(chain_expr) = chain_expr
1249            && chain_expr.syntax().text_range().contains_range(range)
1250        {
1251            break Some(chain_expr);
1252        }
1253        expr = rest?;
1254    }
1255}

1266pub fn is_body_const(sema: &Semantics<'_, RootDatabase>, expr: &ast::Expr) -> bool {
1267    let mut is_const = true;
1268    preorder_expr(expr, &mut |ev| {
1269        let expr = match ev {
1270            WalkEvent::Enter(_) if !is_const => return true,
1271            WalkEvent::Enter(expr) => expr,
1272            WalkEvent::Leave(_) => return false,
1273        };
1274        match expr {
1275            ast::Expr::CallExpr(call) => {
1276                if let Some(ast::Expr::PathExpr(path_expr)) = call.expr()
1277                    && let Some(PathResolution::Def(ModuleDef::Function(func))) =
1278                        path_expr.path().and_then(|path| sema.resolve_path(&path))
1279                {
1280                    is_const &= func.is_const(sema.db);
1281                }
1282            }
1283            ast::Expr::MethodCallExpr(call) => {
1284                is_const &=
1285                    sema.resolve_method_call(&call).map(|it| it.is_const(sema.db)).unwrap_or(true)
1286            }
1287            ast::Expr::ForExpr(_)
1288            | ast::Expr::ReturnExpr(_)
1289            | ast::Expr::TryExpr(_)
1290            | ast::Expr::YieldExpr(_)
1291            | ast::Expr::AwaitExpr(_) => is_const = false,
1292            _ => (),
1293        }
1294        !is_const
1295    });
1296    is_const
1297}
