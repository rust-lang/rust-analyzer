41pub(crate) fn unwrap_trivial_block(block_expr: ast::BlockExpr) -> ast::Expr {
42    extract_trivial_expression(&block_expr)
43        .filter(|expr| !expr.syntax().text().contains_char('\n'))
44        .unwrap_or_else(|| block_expr.into())
45}
46
47pub fn extract_trivial_expression(block_expr: &ast::BlockExpr) -> Option<ast::Expr> {
48    if block_expr.modifier().is_some() {
49        return None;
50    }
51    let stmt_list = block_expr.stmt_list()?;
52    let has_anything_else = |thing: &SyntaxNode| -> bool {
53        let mut non_trivial_children =
54            stmt_list.syntax().children_with_tokens().filter(|it| match it.kind() {
55                WHITESPACE | T!['{'] | T!['}'] => false,
56                _ => it.as_node() != Some(thing),
57            });
58        non_trivial_children.next().is_some()
59    };
60
61    if let Some(expr) = stmt_list.tail_expr() {
62        if has_anything_else(expr.syntax()) {
63            return None;
64        }
65        return Some(expr);
66    }
67    // Unwrap `{ continue; }`
68    let stmt = stmt_list.statements().next()?;
69    if let ast::Stmt::ExprStmt(expr_stmt) = stmt {
70        if has_anything_else(expr_stmt.syntax()) {
71            return None;
72        }
73        let expr = expr_stmt.expr()?;
74        if matches!(expr.syntax().kind(), CONTINUE_EXPR | BREAK_EXPR | RETURN_EXPR) {
75            return Some(expr);
76        }
77    }
78    None
79}

81/// This is a method with a heuristics to support test methods annotated with custom test annotations, such as
82/// `#[test_case(...)]`, `#[tokio::test]` and similar.
83/// Also a regular `#[test]` annotation is supported.
84///
85/// It may produce false positives, for example, `#[wasm_bindgen_test]` requires a different command to run the test,
86/// but it's better than not to have the runnables for the tests at all.
87pub fn test_related_attribute_syn(fn_def: &ast::Fn) -> Option<ast::Attr> {
88    fn_def.attrs().find_map(|attr| {
89        let path = attr.path()?;
90        let text = path.syntax().text().to_string();
91        if text.starts_with("test") || text.ends_with("test") { Some(attr) } else { None }
92    })
93}

95pub fn has_test_related_attribute(attrs: &hir::AttrsWithOwner) -> bool {
96    attrs.iter().any(|attr| {
97        let path = attr.path();
98        (|| {
99            Some(
100                path.segments().first()?.as_str().starts_with("test")
101                    || path.segments().last()?.as_str().ends_with("test"),
102            )
103        })()
104        .unwrap_or_default()
105    })
106}

120pub fn filter_assoc_items(
121    sema: &Semantics<'_, RootDatabase>,
122    items: &[hir::AssocItem],
123    default_methods: DefaultMethods,
124    ignore_items: IgnoreAssocItems,
125) -> Vec<InFile<ast::AssocItem>> {
126    return items
127        .iter()
128        .copied()
129        .filter(|assoc_item| {
130            if ignore_items == IgnoreAssocItems::DocHiddenAttrPresent
131                && assoc_item.attrs(sema.db).has_doc_hidden()
132            {
133                if let hir::AssocItem::Function(f) = assoc_item
134                    && !f.has_body(sema.db)
135                {
136                    return true;
137                }
138                return false;
139            }
140
141            true
142        })
143        // Note: This throws away items with no source.
144        .filter_map(|assoc_item| {
145            let item = match assoc_item {
146                hir::AssocItem::Function(it) => sema.source(it)?.map(ast::AssocItem::Fn),
147                hir::AssocItem::TypeAlias(it) => sema.source(it)?.map(ast::AssocItem::TypeAlias),
148                hir::AssocItem::Const(it) => sema.source(it)?.map(ast::AssocItem::Const),
149            };
150            Some(item)
151        })
152        .filter(has_def_name)
153        .filter(|it| match &it.value {
154            ast::AssocItem::Fn(def) => matches!(
155                (default_methods, def.body()),
156                (DefaultMethods::Only, Some(_)) | (DefaultMethods::No, None)
157            ),
158            ast::AssocItem::Const(def) => matches!(
159                (default_methods, def.body()),
160                (DefaultMethods::Only, Some(_)) | (DefaultMethods::No, None)
161            ),
162            _ => default_methods == DefaultMethods::No,
163        })
164        .collect();
165
166    fn has_def_name(item: &InFile<ast::AssocItem>) -> bool {
167        match &item.value {
168            ast::AssocItem::Fn(def) => def.name(),
169            ast::AssocItem::TypeAlias(def) => def.name(),
170            ast::AssocItem::Const(def) => def.name(),
171            ast::AssocItem::MacroCall(_) => None,
172        }
173        .is_some()
174    }
175}

177/// Given `original_items` retrieved from the trait definition (usually by
178/// [`filter_assoc_items()`]), clones each item for update and applies path transformation to it,
179/// then inserts into `impl_`. Returns the modified `impl_` and the first associated item that got
180/// inserted.
181#[must_use]
182pub fn add_trait_assoc_items_to_impl(
183    sema: &Semantics<'_, RootDatabase>,
184    config: &AssistConfig,
185    original_items: &[InFile<ast::AssocItem>],
186    trait_: hir::Trait,
187    impl_: &ast::Impl,
188    target_scope: &hir::SemanticsScope<'_>,
189) -> Vec<ast::AssocItem> {
190    let new_indent_level = IndentLevel::from_node(impl_.syntax()) + 1;
191    original_items
192        .iter()
193        .map(|InFile { file_id, value: original_item }| {
194            let mut cloned_item = {
195                if let Some(macro_file) = file_id.macro_file() {
196                    let span_map = sema.db.expansion_span_map(macro_file);
197                    let item_prettified = prettify_macro_expansion(
198                        sema.db,
199                        original_item.syntax().clone(),
200                        &span_map,
201                        target_scope.krate().into(),
202                    );
203                    if let Some(formatted) = ast::AssocItem::cast(item_prettified) {
204                        return formatted;
205                    } else {
206                        stdx::never!("formatted `AssocItem` could not be cast back to `AssocItem`");
207                    }
208                }
209                original_item
210            }
211            .reset_indent();
212
213            if let Some(source_scope) = sema.scope(original_item.syntax()) {
214                // FIXME: Paths in nested macros are not handled well. See
215                // `add_missing_impl_members::paths_in_nested_macro_should_get_transformed` test.
216                let transform =
217                    PathTransform::trait_impl(target_scope, &source_scope, trait_, impl_.clone());
218                cloned_item = ast::AssocItem::cast(transform.apply(cloned_item.syntax())).unwrap();
219            }
220            cloned_item.remove_attrs_and_docs();
221            cloned_item
222        })
223        .filter_map(|item| match item {
224            ast::AssocItem::Fn(fn_) if fn_.body().is_none() => {
225                let fn_ = fn_.clone_subtree();
226                let new_body = &make::block_expr(
227                    None,
228                    Some(match config.expr_fill_default {
229                        ExprFillDefaultMode::Todo => make::ext::expr_todo(),
230                        ExprFillDefaultMode::Underscore => make::ext::expr_underscore(),
231                        ExprFillDefaultMode::Default => make::ext::expr_todo(),
232                    }),
233                );
234                let new_body = AstNodeEdit::indent(new_body, IndentLevel::single());
235                let mut fn_editor = SyntaxEditor::new(fn_.syntax().clone());
236                fn_.replace_or_insert_body(&mut fn_editor, new_body);
237                let new_fn_ = fn_editor.finish().new_root().clone();
238                ast::AssocItem::cast(new_fn_)
239            }
240            ast::AssocItem::TypeAlias(type_alias) => {
241                let type_alias = type_alias.clone_subtree();
242                if let Some(type_bound_list) = type_alias.type_bound_list() {
243                    let mut type_alias_editor = SyntaxEditor::new(type_alias.syntax().clone());
244                    type_bound_list.remove(&mut type_alias_editor);
245                    let type_alias = type_alias_editor.finish().new_root().clone();
246                    ast::AssocItem::cast(type_alias)
247                } else {
248                    Some(ast::AssocItem::TypeAlias(type_alias))
249                }
250            }
251            item => Some(item),
252        })
253        .map(|item| AstNodeEdit::indent(&item, new_indent_level))
254        .collect()
255}

273fn invert_special_case(make: &SyntaxFactory, expr: &ast::Expr) -> Option<ast::Expr> {
274    match expr {
275        ast::Expr::BinExpr(bin) => {
276            let op_kind = bin.op_kind()?;
277            let rev_kind = match op_kind {
278                ast::BinaryOp::CmpOp(ast::CmpOp::Eq { negated }) => {
279                    ast::BinaryOp::CmpOp(ast::CmpOp::Eq { negated: !negated })
280                }
281                ast::BinaryOp::CmpOp(ast::CmpOp::Ord { ordering: ast::Ordering::Less, strict }) => {
282                    ast::BinaryOp::CmpOp(ast::CmpOp::Ord {
283                        ordering: ast::Ordering::Greater,
284                        strict: !strict,
285                    })
286                }
287                ast::BinaryOp::CmpOp(ast::CmpOp::Ord {
288                    ordering: ast::Ordering::Greater,
289                    strict,
290                }) => ast::BinaryOp::CmpOp(ast::CmpOp::Ord {
291                    ordering: ast::Ordering::Less,
292                    strict: !strict,
293                }),
294                // Parenthesize other expressions before prefixing `!`
295                _ => {
296                    return Some(
297                        make.expr_prefix(T![!], make.expr_paren(expr.clone()).into()).into(),
298                    );
299                }
300            };
301
302            Some(make.expr_bin(bin.lhs()?, rev_kind, bin.rhs()?).into())
303        }
304        ast::Expr::MethodCallExpr(mce) => {
305            let receiver = mce.receiver()?;
306            let method = mce.name_ref()?;
307            let arg_list = mce.arg_list()?;
308
309            let method = match method.text().as_str() {
310                "is_some" => "is_none",
311                "is_none" => "is_some",
312                "is_ok" => "is_err",
313                "is_err" => "is_ok",
314                _ => return None,
315            };
316
317            Some(make.expr_method_call(receiver, make.name_ref(method), arg_list).into())
318        }
319        ast::Expr::PrefixExpr(pe) if pe.op_kind()? == ast::UnaryOp::Not => match pe.expr()? {
320            ast::Expr::ParenExpr(parexpr) => {
321                parexpr.expr().map(|e| e.clone_subtree().clone_for_update())
322            }
323            _ => pe.expr().map(|e| e.clone_subtree().clone_for_update()),
324        },
325        ast::Expr::Literal(lit) => match lit.kind() {
326            ast::LiteralKind::Bool(b) => match b {
327                true => Some(ast::Expr::Literal(make.expr_literal("false"))),
328                false => Some(ast::Expr::Literal(make.expr_literal("true"))),
329            },
330            _ => None,
331        },
332        _ => None,
333    }
334}

336fn invert_special_case_legacy(expr: &ast::Expr) -> Option<ast::Expr> {
337    match expr {
338        ast::Expr::BinExpr(bin) => {
339            let bin = bin.clone_subtree();
340            let op_token = bin.op_token()?;
341            let rev_token = match op_token.kind() {
342                T![==] => T![!=],
343                T![!=] => T![==],
344                T![<] => T![>=],
345                T![<=] => T![>],
346                T![>] => T![<=],
347                T![>=] => T![<],
348                // Parenthesize other expressions before prefixing `!`
349                _ => {
350                    return Some(
351                        make::expr_prefix(T![!], make::expr_paren(expr.clone()).into()).into(),
352                    );
353                }
354            };
355            let mut bin_editor = SyntaxEditor::new(bin.syntax().clone());
356            bin_editor.replace(op_token, make::token(rev_token));
357            ast::Expr::cast(bin_editor.finish().new_root().clone())
358        }
359        ast::Expr::MethodCallExpr(mce) => {
360            let receiver = mce.receiver()?;
361            let method = mce.name_ref()?;
362            let arg_list = mce.arg_list()?;
363
364            let method = match method.text().as_str() {
365                "is_some" => "is_none",
366                "is_none" => "is_some",
367                "is_ok" => "is_err",
368                "is_err" => "is_ok",
369                _ => return None,
370            };
371            Some(make::expr_method_call(receiver, make::name_ref(method), arg_list).into())
372        }
373        ast::Expr::PrefixExpr(pe) if pe.op_kind()? == ast::UnaryOp::Not => match pe.expr()? {
374            ast::Expr::ParenExpr(parexpr) => parexpr.expr(),
375            _ => pe.expr(),
376        },
377        ast::Expr::Literal(lit) => match lit.kind() {
378            ast::LiteralKind::Bool(b) => match b {
379                true => Some(ast::Expr::Literal(make::expr_literal("false"))),
380                false => Some(ast::Expr::Literal(make::expr_literal("true"))),
381            },
382            _ => None,
383        },
384        _ => None,
385    }
386}

392pub(crate) fn does_pat_match_variant(pat: &ast::Pat, var: &ast::Pat) -> bool {
393    let first_node_text = |pat: &ast::Pat| pat.syntax().first_child().map(|node| node.text());
394
395    let pat_head = match pat {
396        ast::Pat::IdentPat(bind_pat) => match bind_pat.pat() {
397            Some(p) => first_node_text(&p),
398            None => return pat.syntax().text() == var.syntax().text(),
399        },
400        pat => first_node_text(pat),
401    };
402
403    let var_head = first_node_text(var);
404
405    pat_head == var_head
406}

408pub(crate) fn does_pat_variant_nested_or_literal(ctx: &AssistContext<'_>, pat: &ast::Pat) -> bool {

412fn check_pat_variant_from_enum(ctx: &AssistContext<'_>, pat: &ast::Pat) -> bool {
413    ctx.sema.type_of_pat(pat).is_none_or(|ty: hir::TypeInfo<'_>| {
414        ty.adjusted().as_adt().is_some_and(|adt| matches!(adt, hir::Adt::Enum(_)))
415    })
416}

418fn check_pat_variant_nested_or_literal_with_depth(
419    ctx: &AssistContext<'_>,
420    pat: &ast::Pat,
421    depth_after_refutable: usize,
422) -> bool {
423    if depth_after_refutable > 1 {
424        return true;
425    }
426
427    match pat {
428        ast::Pat::RestPat(_) | ast::Pat::WildcardPat(_) | ast::Pat::RefPat(_) => false,
429
430        ast::Pat::LiteralPat(_)
431        | ast::Pat::RangePat(_)
432        | ast::Pat::MacroPat(_)
433        | ast::Pat::PathPat(_)
434        | ast::Pat::BoxPat(_)
435        | ast::Pat::ConstBlockPat(_) => true,
436
437        ast::Pat::IdentPat(ident_pat) => ident_pat.pat().is_some_and(|pat| {
438            check_pat_variant_nested_or_literal_with_depth(ctx, &pat, depth_after_refutable)
439        }),
440        ast::Pat::ParenPat(paren_pat) => paren_pat.pat().is_none_or(|pat| {
441            check_pat_variant_nested_or_literal_with_depth(ctx, &pat, depth_after_refutable)
442        }),
443        ast::Pat::TuplePat(tuple_pat) => tuple_pat.fields().any(|pat| {
444            check_pat_variant_nested_or_literal_with_depth(ctx, &pat, depth_after_refutable)
445        }),
446        ast::Pat::RecordPat(record_pat) => {
447            let adjusted_next_depth =
448                depth_after_refutable + if check_pat_variant_from_enum(ctx, pat) { 1 } else { 0 };
449            record_pat.record_pat_field_list().is_none_or(|pat| {
450                pat.fields().any(|pat| {
451                    pat.pat().is_none_or(|pat| {
452                        check_pat_variant_nested_or_literal_with_depth(
453                            ctx,
454                            &pat,
455                            adjusted_next_depth,
456                        )
457                    })
458                })
459            })
460        }
461        ast::Pat::OrPat(or_pat) => or_pat.pats().any(|pat| {
462            check_pat_variant_nested_or_literal_with_depth(ctx, &pat, depth_after_refutable)
463        }),
464        ast::Pat::TupleStructPat(tuple_struct_pat) => {
465            let adjusted_next_depth =
466                depth_after_refutable + if check_pat_variant_from_enum(ctx, pat) { 1 } else { 0 };
467            tuple_struct_pat.fields().any(|pat| {
468                check_pat_variant_nested_or_literal_with_depth(ctx, &pat, adjusted_next_depth)
469            })
470        }
471        ast::Pat::SlicePat(slice_pat) => {
472            let mut pats = slice_pat.pats();
473            pats.next()
474                .is_none_or(|pat| !matches!(pat, ast::Pat::RestPat(_)) || pats.next().is_some())
475        }
476    }
477}

479// Uses a syntax-driven approach to find any impl blocks for the struct that
480// exist within the module/file
481//
482// Returns `None` if we've found an existing fn
483//
484// FIXME: change the new fn checking to a more semantic approach when that's more
485// viable (e.g. we process proc macros, etc)
486// FIXME: this partially overlaps with `find_impl_block_*`
487
488/// `find_struct_impl` looks for impl of a struct, but this also has additional feature
489/// where it takes a list of function names and check if they exist inside impl_, if
490/// even one match is found, it returns None.
491///
492/// That means this function can have 3 potential return values:
493///  - `None`: an impl exists, but one of the function names within the impl matches one of the provided names.
494///  - `Some(None)`: no impl exists.
495///  - `Some(Some(_))`: an impl exists, with no matching function names.
496pub(crate) fn find_struct_impl(
497    ctx: &AssistContext<'_>,
498    adt: &ast::Adt,
499    names: &[String],
500) -> Option<Option<ast::Impl>> {
501    let db = ctx.db();
502    let module = adt.syntax().parent()?;
503
504    let struct_def = ctx.sema.to_def(adt)?;
505
506    let block = module.descendants().filter_map(ast::Impl::cast).find_map(|impl_blk| {
507        let blk = ctx.sema.to_def(&impl_blk)?;
508
509        // FIXME: handle e.g. `struct S<T>; impl<U> S<U> {}`
510        // (we currently use the wrong type parameter)
511        // also we wouldn't want to use e.g. `impl S<u32>`
512
513        let same_ty = match blk.self_ty(db).as_adt() {
514            Some(def) => def == struct_def,
515            None => false,
516        };
517        let not_trait_impl = blk.trait_(db).is_none();
518
519        if !(same_ty && not_trait_impl) { None } else { Some(impl_blk) }
520    });
521
522    if let Some(ref impl_blk) = block
523        && has_any_fn(impl_blk, names)
524    {
525        return None;
526    }
527
528    Some(block)
529}
530
531fn has_any_fn(imp: &ast::Impl, names: &[String]) -> bool {
532    if let Some(il) = imp.assoc_item_list() {
533        for item in il.assoc_items() {
534            if let ast::AssocItem::Fn(f) = item
535                && let Some(name) = f.name()
536                && names.iter().any(|n| n.eq_ignore_ascii_case(&name.text()))
537            {
538                return true;
539            }
540        }
541    }
542
543    false
544}

546/// Find the end of the `impl` block for the given `ast::Impl`.
547//
548// FIXME: this partially overlaps with `find_struct_impl`
549pub(crate) fn find_impl_block_end(impl_def: ast::Impl, buf: &mut String) -> Option<TextSize> {
550    buf.push('\n');
551    let end = impl_def
552        .assoc_item_list()
553        .and_then(|it| it.r_curly_token())?
554        .prev_sibling_or_token()?
555        .text_range()
556        .end();
557    Some(end)
558}

591    adt: &ast::Adt,
592    trait_text: Option<&str>,
593    trait_is_transitive: bool,
594    code: &str,
595) -> String {
596    // Ensure lifetime params are before type & const params
597    let generic_params = adt.generic_param_list().map(|generic_params| {
598        let lifetime_params =
599            generic_params.lifetime_params().map(ast::GenericParam::LifetimeParam);
600        let ty_or_const_params = generic_params.type_or_const_params().filter_map(|param| {
601            let param = match param {
602                ast::TypeOrConstParam::Type(param) => {
603                    // remove defaults since they can't be specified in impls
604                    let mut bounds =
605                        param.type_bound_list().map_or_else(Vec::new, |it| it.bounds().collect());
606                    if let Some(trait_) = trait_text {
607                        // Add the current trait to `bounds` if the trait is transitive,
608                        // meaning `impl<T> Trait for U<T>` requires `T: Trait`.
609                        if trait_is_transitive {
610                            bounds.push(make::type_bound_text(trait_));
611                        }
612                    };
613                    // `{ty_param}: {bounds}`
614                    let param = make::type_param(param.name()?, make::type_bound_list(bounds));
615                    ast::GenericParam::TypeParam(param)
616                }
617                ast::TypeOrConstParam::Const(param) => {
618                    // remove defaults since they can't be specified in impls
619                    let param = make::const_param(param.name()?, param.ty()?);
620                    ast::GenericParam::ConstParam(param)
621                }
622            };
623            Some(param)
624        });
625
626        make::generic_param_list(itertools::chain(lifetime_params, ty_or_const_params))
627    });
628
629    // FIXME: use syntax::make & mutable AST apis instead
630    // `trait_text` and `code` can't be opaque blobs of text
631    let mut buf = String::with_capacity(code.len());
632
633    // Copy any cfg attrs from the original adt
634    buf.push_str("\n\n");
635    let cfg_attrs = adt
636        .attrs()
637        .filter(|attr| attr.as_simple_call().map(|(name, _arg)| name == "cfg").unwrap_or(false));
638    cfg_attrs.for_each(|attr| buf.push_str(&format!("{attr}\n")));
639
640    // `impl{generic_params} {trait_text} for {name}{generic_params.to_generic_args()}`
641    buf.push_str("impl");
642    if let Some(generic_params) = &generic_params {
643        format_to!(buf, "{generic_params}");
644    }
645    buf.push(' ');
646    if let Some(trait_text) = trait_text {
647        buf.push_str(trait_text);
648        buf.push_str(" for ");
649    }
650    buf.push_str(&adt.name().unwrap().text());
651    if let Some(generic_params) = generic_params {
652        format_to!(buf, "{}", generic_params.to_generic_args());
653    }
654
655    match adt.where_clause() {
656        Some(where_clause) => {
657            format_to!(buf, "\n{where_clause}\n{{\n{code}\n}}");
658        }
659        None => {
660            format_to!(buf, " {{\n{code}\n}}");
661        }
662    }
663
664    buf
665}

696fn generate_impl_inner(
697    is_unsafe: bool,
698    adt: &ast::Adt,
699    trait_: Option<ast::Type>,
700    trait_is_transitive: bool,
701    body: Option<ast::AssocItemList>,
702) -> ast::Impl {
703    // Ensure lifetime params are before type & const params
704    let generic_params = adt.generic_param_list().map(|generic_params| {
705        let lifetime_params =
706            generic_params.lifetime_params().map(ast::GenericParam::LifetimeParam);
707        let ty_or_const_params = generic_params.type_or_const_params().filter_map(|param| {
708            let param = match param {
709                ast::TypeOrConstParam::Type(param) => {
710                    // remove defaults since they can't be specified in impls
711                    let mut bounds =
712                        param.type_bound_list().map_or_else(Vec::new, |it| it.bounds().collect());
713                    if let Some(trait_) = &trait_ {
714                        // Add the current trait to `bounds` if the trait is transitive,
715                        // meaning `impl<T> Trait for U<T>` requires `T: Trait`.
716                        if trait_is_transitive {
717                            bounds.push(make::type_bound(trait_.clone()));
718                        }
719                    };
720                    // `{ty_param}: {bounds}`
721                    let param = make::type_param(param.name()?, make::type_bound_list(bounds));
722                    ast::GenericParam::TypeParam(param)
723                }
724                ast::TypeOrConstParam::Const(param) => {
725                    // remove defaults since they can't be specified in impls
726                    let param = make::const_param(param.name()?, param.ty()?);
727                    ast::GenericParam::ConstParam(param)
728                }
729            };
730            Some(param)
731        });
732
733        make::generic_param_list(itertools::chain(lifetime_params, ty_or_const_params))
734    });
735    let generic_args =
736        generic_params.as_ref().map(|params| params.to_generic_args().clone_for_update());
737    let ty = make::ty_path(make::ext::ident_path(&adt.name().unwrap().text()));
738
739    let cfg_attrs =
740        adt.attrs().filter(|attr| attr.as_simple_call().is_some_and(|(name, _arg)| name == "cfg"));
741    match trait_ {
742        Some(trait_) => make::impl_trait(
743            cfg_attrs,
744            is_unsafe,
745            None,
746            None,
747            generic_params,
748            generic_args,
749            false,
750            trait_,
751            ty,
752            None,
753            adt.where_clause(),
754            body,
755        ),
756        None => make::impl_(cfg_attrs, generic_params, generic_args, ty, adt.where_clause(), body),
757    }
758    .clone_for_update()
759}

761pub(crate) fn add_method_to_adt(
762    builder: &mut SourceChangeBuilder,
763    adt: &ast::Adt,
764    impl_def: Option<ast::Impl>,
765    method: &str,
766) {
767    let mut buf = String::with_capacity(method.len() + 2);
768    if impl_def.is_some() {
769        buf.push('\n');
770    }

778        });

781}

783#[derive(Debug)]
784pub(crate) struct ReferenceConversion<'db> {
785    conversion: ReferenceConversionType,
786    ty: hir::Type<'db>,
787    impls_deref: bool,
788}
789
790#[derive(Debug)]
791enum ReferenceConversionType {
792    // reference can be stripped if the type is Copy
793    Copy,
794    // &String -> &str
795    AsRefStr,
796    // &Vec<T> -> &[T]
797    AsRefSlice,
798    // &Box<T> -> &T
799    Dereferenced,
800    // &Option<T> -> Option<&T>
801    Option,
802    // &Result<T, E> -> Result<&T, &E>
803    Result,
804}
805
806impl<'db> ReferenceConversion<'db> {
807    pub(crate) fn convert_type(
808        &self,
809        db: &'db dyn HirDatabase,
810        display_target: DisplayTarget,
811    ) -> ast::Type {
812        let ty = match self.conversion {
813            ReferenceConversionType::Copy => self.ty.display(db, display_target).to_string(),
814            ReferenceConversionType::AsRefStr => "&str".to_owned(),
815            ReferenceConversionType::AsRefSlice => {
816                let type_argument_name = self
817                    .ty
818                    .type_arguments()
819                    .next()
820                    .unwrap()
821                    .display(db, display_target)
822                    .to_string();
823                format!("&[{type_argument_name}]")
824            }
825            ReferenceConversionType::Dereferenced => {
826                let type_argument_name = self
827                    .ty
828                    .type_arguments()
829                    .next()
830                    .unwrap()
831                    .display(db, display_target)
832                    .to_string();
833                format!("&{type_argument_name}")
834            }
835            ReferenceConversionType::Option => {
836                let type_argument_name = self
837                    .ty
838                    .type_arguments()
839                    .next()
840                    .unwrap()
841                    .display(db, display_target)
842                    .to_string();
843                format!("Option<&{type_argument_name}>")
844            }
845            ReferenceConversionType::Result => {
846                let mut type_arguments = self.ty.type_arguments();
847                let first_type_argument_name =
848                    type_arguments.next().unwrap().display(db, display_target).to_string();
849                let second_type_argument_name =
850                    type_arguments.next().unwrap().display(db, display_target).to_string();
851                format!("Result<&{first_type_argument_name}, &{second_type_argument_name}>")
852            }
853        };
854
855        make::ty(&ty)
856    }
857
858    pub(crate) fn getter(&self, field_name: String) -> ast::Expr {
859        let expr = make::expr_field(make::ext::expr_self(), &field_name);
860
861        match self.conversion {
862            ReferenceConversionType::Copy => expr,
863            ReferenceConversionType::AsRefStr
864            | ReferenceConversionType::AsRefSlice
865            | ReferenceConversionType::Dereferenced
866            | ReferenceConversionType::Option
867            | ReferenceConversionType::Result => {
868                if self.impls_deref {
869                    make::expr_ref(expr, false)
870                } else {
871                    make::expr_method_call(expr, make::name_ref("as_ref"), make::arg_list([]))
872                        .into()
873                }
874            }
875        }
876    }
877}

879// FIXME: It should return a new hir::Type, but currently constructing new types is too cumbersome
880//        and all users of this function operate on string type names, so they can do the conversion
881//        itself themselves.
882pub(crate) fn convert_reference_type<'db>(
883    ty: hir::Type<'db>,
884    db: &'db RootDatabase,
885    famous_defs: &FamousDefs<'_, 'db>,
886) -> Option<ReferenceConversion<'db>> {
887    handle_copy(&ty, db)
888        .or_else(|| handle_as_ref_str(&ty, db, famous_defs))
889        .or_else(|| handle_as_ref_slice(&ty, db, famous_defs))
890        .or_else(|| handle_dereferenced(&ty, db, famous_defs))
891        .or_else(|| handle_option_as_ref(&ty, db, famous_defs))
892        .or_else(|| handle_result_as_ref(&ty, db, famous_defs))
893        .map(|(conversion, impls_deref)| ReferenceConversion { ty, conversion, impls_deref })
894}

896fn could_deref_to_target(ty: &hir::Type<'_>, target: &hir::Type<'_>, db: &dyn HirDatabase) -> bool {
897    let ty_ref = ty.add_reference(hir::Mutability::Shared);
898    let target_ref = target.add_reference(hir::Mutability::Shared);
899    ty_ref.could_coerce_to(db, &target_ref)
900}

904    db: &dyn HirDatabase,
905) -> Option<(ReferenceConversionType, bool)> {
906    ty.is_copy(db).then_some((ReferenceConversionType::Copy, true))

911    db: &dyn HirDatabase,

917        .then_some((ReferenceConversionType::AsRefStr, could_deref_to_target(ty, &str_type, db)))

926    let slice_type = hir::Type::new_slice(type_argument);

928    ty.impls_trait(db, famous_defs.core_convert_AsRef()?, slice::from_ref(&slice_type)).then_some((
929        ReferenceConversionType::AsRefSlice,
930        could_deref_to_target(ty, &slice_type, db),

983/// Trim(remove leading and trailing whitespace) `initial_range` in `source_file`, return the trimmed range.
984pub(crate) fn trimmed_text_range(source_file: &SourceFile, initial_range: TextRange) -> TextRange {
985    let mut trimmed_range = initial_range;
986    while source_file
987        .syntax()
988        .token_at_offset(trimmed_range.start())
989        .find_map(Whitespace::cast)
990        .is_some()
991        && trimmed_range.start() < trimmed_range.end()
992    {
993        let start = trimmed_range.start() + TextSize::from(1);
994        trimmed_range = TextRange::new(start, trimmed_range.end());
995    }
996    while source_file
997        .syntax()
998        .token_at_offset(trimmed_range.end())
999        .find_map(Whitespace::cast)
1000        .is_some()
1001        && trimmed_range.start() < trimmed_range.end()
1002    {
1003        let end = trimmed_range.end() - TextSize::from(1);
1004        trimmed_range = TextRange::new(trimmed_range.start(), end);
1005    }
1006    trimmed_range
1007}

1025/// Calculate the number of hashes required for a raw string containing `s`
1026pub(crate) fn required_hashes(s: &str) -> usize {
1027    let mut res = 0usize;
1028    for idx in s.match_indices('"').map(|(i, _)| i) {
1029        let (_, sub) = s.split_at(idx + 1);
1030        let n_hashes = sub.chars().take_while(|c| *c == '#').count();
1031        res = res.max(n_hashes + 1)
1032    }
1033    res
1034}
1035#[test]
1036fn test_required_hashes() {
1037    assert_eq!(0, required_hashes("abc"));
1038    assert_eq!(0, required_hashes("###"));
1039    assert_eq!(1, required_hashes("\""));
1040    assert_eq!(2, required_hashes("\"#abc"));
1041    assert_eq!(0, required_hashes("#abc"));
1042    assert_eq!(3, required_hashes("#ab\"##c"));
1043    assert_eq!(5, required_hashes("#ab\"##\"####c"));
1044}

1093/// Creates a token tree list from a syntax node, creating the needed delimited sub token trees.
1094/// Assumes that the input syntax node is a valid syntax tree.
1095pub(crate) fn tt_from_syntax(node: SyntaxNode) -> Vec<NodeOrToken<ast::TokenTree, SyntaxToken>> {
1096    let mut tt_stack = vec![(None, vec![])];
1097
1098    for element in node.descendants_with_tokens() {
1099        let NodeOrToken::Token(token) = element else { continue };
1100
1101        match token.kind() {
1102            T!['('] | T!['{'] | T!['['] => {
1103                // Found an opening delimiter, start a new sub token tree
1104                tt_stack.push((Some(token.kind()), vec![]));
1105            }
1106            T![')'] | T!['}'] | T![']'] => {
1107                // Closing a subtree
1108                let (delimiter, tt) = tt_stack.pop().expect("unbalanced delimiters");
1109                let (_, parent_tt) = tt_stack
1110                    .last_mut()
1111                    .expect("parent token tree was closed before it was completed");
1112                let closing_delimiter = delimiter.map(|it| match it {
1113                    T!['('] => T![')'],
1114                    T!['{'] => T!['}'],
1115                    T!['['] => T![']'],
1116                    _ => unreachable!(),
1117                });
1118                stdx::always!(
1119                    closing_delimiter == Some(token.kind()),
1120                    "mismatched opening and closing delimiters"
1121                );
1122
1123                let sub_tt = make::token_tree(delimiter.expect("unbalanced delimiters"), tt);
1124                parent_tt.push(NodeOrToken::Node(sub_tt));
1125            }
1126            _ => {
1127                let (_, current_tt) = tt_stack.last_mut().expect("unmatched delimiters");
1128                current_tt.push(NodeOrToken::Token(token))
1129            }
1130        }
1131    }
1132
1133    tt_stack.pop().expect("parent token tree was closed before it was completed").1
1134}

1136pub(crate) fn cover_let_chain(mut expr: ast::Expr, range: TextRange) -> Option<ast::Expr> {
1137    if !expr.syntax().text_range().contains_range(range) {
1138        return None;
1139    }
1140    loop {
1141        let (chain_expr, rest) = if let ast::Expr::BinExpr(bin_expr) = &expr
1142            && bin_expr.op_kind() == Some(ast::BinaryOp::LogicOp(ast::LogicOp::And))
1143        {
1144            (bin_expr.rhs(), bin_expr.lhs())
1145        } else {
1146            (Some(expr), None)
1147        };
1148
1149        if let Some(chain_expr) = chain_expr
1150            && chain_expr.syntax().text_range().contains_range(range)
1151        {
1152            break Some(chain_expr);
1153        }
1154        expr = rest?;
1155    }
1156}

1158pub fn is_body_const(sema: &Semantics<'_, RootDatabase>, expr: &ast::Expr) -> bool {
1159    let mut is_const = true;
1160    preorder_expr(expr, &mut |ev| {
1161        let expr = match ev {
1162            WalkEvent::Enter(_) if !is_const => return true,
1163            WalkEvent::Enter(expr) => expr,
1164            WalkEvent::Leave(_) => return false,
1165        };
1166        match expr {
1167            ast::Expr::CallExpr(call) => {
1168                if let Some(ast::Expr::PathExpr(path_expr)) = call.expr()
1169                    && let Some(PathResolution::Def(ModuleDef::Function(func))) =
1170                        path_expr.path().and_then(|path| sema.resolve_path(&path))
1171                {
1172                    is_const &= func.is_const(sema.db);
1173                }
1174            }
1175            ast::Expr::MethodCallExpr(call) => {
1176                is_const &=
1177                    sema.resolve_method_call(&call).map(|it| it.is_const(sema.db)).unwrap_or(true)
1178            }
1179            ast::Expr::ForExpr(_)
1180            | ast::Expr::ReturnExpr(_)
1181            | ast::Expr::TryExpr(_)
1182            | ast::Expr::YieldExpr(_)
1183            | ast::Expr::AwaitExpr(_) => is_const = false,
1184            _ => (),
1185        }
1186        !is_const
1187    });
1188    is_const
1189}
