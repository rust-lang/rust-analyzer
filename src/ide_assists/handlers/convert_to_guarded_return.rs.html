23// Assist: convert_to_guarded_return
24//
25// Replace a large conditional with a guarded return.
26//
27// ```
28// fn main() {
29//     $0if cond {
30//         foo();
31//         bar();
32//     }
33// }
34// ```
35// ->
36// ```
37// fn main() {
38//     if !cond {
39//         return;
40//     }
41//     foo();
42//     bar();
43// }
44// ```
45pub(crate) fn convert_to_guarded_return(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
46    match ctx.find_node_at_offset::<Either<ast::LetStmt, ast::IfExpr>>()? {
47        Either::Left(let_stmt) => let_stmt_to_guarded_return(let_stmt, acc, ctx),
48        Either::Right(if_expr) => if_expr_to_guarded_return(if_expr, acc, ctx),
49    }
50}
51
52fn if_expr_to_guarded_return(
53    if_expr: ast::IfExpr,
54    acc: &mut Assists,
55    ctx: &AssistContext<'_>,
56) -> Option<()> {
57    let else_block = match if_expr.else_branch() {
58        Some(ast::ElseBranch::Block(block_expr)) if is_never_block(&ctx.sema, &block_expr) => {
59            Some(block_expr)
60        }
61        Some(_) => return None,
62        _ => None,
63    };
64
65    let cond = if_expr.condition()?;
66
67    let if_token_range = if_expr.if_token()?.text_range();
68    let if_cond_range = cond.syntax().text_range();
69
70    let cursor_in_range =
71        if_token_range.cover(if_cond_range).contains_range(ctx.selection_trimmed());
72    if !cursor_in_range {
73        return None;
74    }
75
76    let let_chains = flat_let_chain(cond);
77
78    let then_block = if_expr.then_branch()?;
79    let then_block = then_block.stmt_list()?;
80
81    let parent_block = if_expr.syntax().parent()?.ancestors().find_map(ast::BlockExpr::cast)?;
82
83    if parent_block.tail_expr()? != if_expr.clone().into() {
84        return None;
85    }
86
87    // FIXME: This relies on untyped syntax tree and casts to much. It should be
88    // rewritten to use strongly-typed APIs.
89
90    // check for early return and continue
91    let first_in_then_block = then_block.syntax().first_child()?;
92    if ast::ReturnExpr::can_cast(first_in_then_block.kind())
93        || ast::ContinueExpr::can_cast(first_in_then_block.kind())
94        || first_in_then_block
95            .children()
96            .any(|x| ast::ReturnExpr::can_cast(x.kind()) || ast::ContinueExpr::can_cast(x.kind()))
97    {
98        return None;
99    }
100
101    let parent_container = parent_block.syntax().parent()?;
102
103    let early_expression = else_block
104        .or_else(|| {
105            early_expression(parent_container, &ctx.sema).map(ast::make::tail_only_block_expr)
106        })?
107        .reset_indent();
108
109    then_block.syntax().first_child_or_token().map(|t| t.kind() == T!['{'])?;
110
111    then_block.syntax().last_child_or_token().filter(|t| t.kind() == T!['}'])?;
112
113    let then_block_items = then_block.dedent(IndentLevel(1));
114
115    let end_of_then = then_block_items.syntax().last_child_or_token()?;
116    let end_of_then = if end_of_then.prev_sibling_or_token().map(|n| n.kind()) == Some(WHITESPACE) {
117        end_of_then.prev_sibling_or_token()?
118    } else {
119        end_of_then
120    };
121
122    let target = if_expr.syntax().text_range();
123    acc.add(
124        AssistId::refactor_rewrite("convert_to_guarded_return"),
125        "Convert to guarded return",
126        target,
127        |edit| {
128            let if_indent_level = IndentLevel::from_node(if_expr.syntax());
129            let replacement = let_chains.into_iter().map(|expr| {
130                if let ast::Expr::LetExpr(let_expr) = &expr
131                    && let (Some(pat), Some(expr)) = (let_expr.pat(), let_expr.expr())
132                {
133                    // If-let.
134                    let let_else_stmt =
135                        make::let_else_stmt(pat, None, expr, early_expression.clone());
136                    let let_else_stmt = let_else_stmt.indent(if_indent_level);
137                    let_else_stmt.syntax().clone()
138                } else {
139                    // If.
140                    let new_expr = {
141                        let then_branch = clean_stmt_block(&early_expression);
142                        let cond = invert_boolean_expression_legacy(expr);
143                        make::expr_if(cond, then_branch, None).indent(if_indent_level)
144                    };
145                    new_expr.syntax().clone()
146                }
147            });
148
149            let newline = &format!("\n{if_indent_level}");
150            let then_statements = replacement
151                .enumerate()
152                .flat_map(|(i, node)| {
153                    (i != 0)
154                        .then(|| make::tokens::whitespace(newline).into())
155                        .into_iter()
156                        .chain(node.children_with_tokens())
157                })
158                .chain(
159                    then_block_items
160                        .syntax()
161                        .children_with_tokens()
162                        .skip(1)
163                        .take_while(|i| *i != end_of_then),
164                )
165                .collect();
166            let mut editor = edit.make_editor(if_expr.syntax());
167            editor.replace_with_many(if_expr.syntax(), then_statements);
168            edit.add_file_edits(ctx.vfs_file_id(), editor);
169        },
170    )
171}

173fn let_stmt_to_guarded_return(
174    let_stmt: ast::LetStmt,
175    acc: &mut Assists,
176    ctx: &AssistContext<'_>,
177) -> Option<()> {
178    let pat = let_stmt.pat()?;
179    let expr = let_stmt.initializer()?;
180
181    let let_token_range = let_stmt.let_token()?.text_range();
182    let let_pattern_range = pat.syntax().text_range();
183    let cursor_in_range =
184        let_token_range.cover(let_pattern_range).contains_range(ctx.selection_trimmed());
185
186    if !cursor_in_range || let_stmt.let_else().is_some() {
187        return None;
188    }
189
190    let try_enum =
191        ctx.sema.type_of_expr(&expr).and_then(|ty| TryEnum::from_ty(&ctx.sema, &ty.adjusted()))?;
192
193    let happy_pattern = try_enum.happy_pattern(pat);
194    let target = let_stmt.syntax().text_range();
195
196    let early_expression: ast::Expr = {
197        let parent_block =
198            let_stmt.syntax().parent()?.ancestors().find_map(ast::BlockExpr::cast)?;
199        let parent_container = parent_block.syntax().parent()?;
200
201        early_expression(parent_container, &ctx.sema)?
202    };
203
204    acc.add(
205        AssistId::refactor_rewrite("convert_to_guarded_return"),
206        "Convert to guarded return",
207        target,
208        |edit| {
209            let let_indent_level = IndentLevel::from_node(let_stmt.syntax());
210
211            let replacement = {
212                let let_else_stmt = make::let_else_stmt(
213                    happy_pattern,
214                    let_stmt.ty(),
215                    expr,
216                    ast::make::tail_only_block_expr(early_expression),
217                );
218                let let_else_stmt = let_else_stmt.indent(let_indent_level);
219                let_else_stmt.syntax().clone()
220            };
221            let mut editor = edit.make_editor(let_stmt.syntax());
222            editor.replace(let_stmt.syntax(), replacement);
223            edit.add_file_edits(ctx.vfs_file_id(), editor);
224        },
225    )
226}

228fn early_expression(
229    parent_container: SyntaxNode,
230    sema: &Semantics<'_, RootDatabase>,
231) -> Option<ast::Expr> {
232    let return_none_expr = || {
233        let none_expr = make::expr_path(make::ext::ident_path("None"));
234        make::expr_return(Some(none_expr))
235    };
236    if let Some(fn_) = ast::Fn::cast(parent_container.clone())
237        && let Some(fn_def) = sema.to_def(&fn_)
238        && let Some(TryEnum::Option) = TryEnum::from_ty(sema, &fn_def.ret_type(sema.db))
239    {
240        return Some(return_none_expr());
241    }
242    if let Some(body) = ast::ClosureExpr::cast(parent_container.clone()).and_then(|it| it.body())
243        && let Some(ret_ty) = sema.type_of_expr(&body).map(TypeInfo::original)
244        && let Some(TryEnum::Option) = TryEnum::from_ty(sema, &ret_ty)
245    {
246        return Some(return_none_expr());
247    }
248
249    Some(match parent_container.kind() {
250        WHILE_EXPR | LOOP_EXPR | FOR_EXPR => make::expr_continue(None),
251        FN | CLOSURE_EXPR => make::expr_return(None),
252        _ => return None,
253    })
254}

256fn flat_let_chain(mut expr: ast::Expr) -> Vec<ast::Expr> {
257    let mut chains = vec![];
258
259    while let ast::Expr::BinExpr(bin_expr) = &expr
260        && bin_expr.op_kind() == Some(ast::BinaryOp::LogicOp(ast::LogicOp::And))
261        && let (Some(lhs), Some(rhs)) = (bin_expr.lhs(), bin_expr.rhs())
262    {
263        if let Some(last) = chains.pop_if(|last| !matches!(last, ast::Expr::LetExpr(_))) {
264            chains.push(make::expr_bin_op(rhs, ast::BinaryOp::LogicOp(ast::LogicOp::And), last));
265        } else {
266            chains.push(rhs);
267        }
268        expr = lhs;
269    }
270
271    chains.push(expr);
272    chains.reverse();
273    chains
274}

276fn clean_stmt_block(block: &ast::BlockExpr) -> ast::BlockExpr {
277    if block.statements().next().is_none()
278        && let Some(tail_expr) = block.tail_expr()
279        && block.modifier().is_none()
280    {
281        make::block_expr(once(make::expr_stmt(tail_expr).into()), None)
282    } else {
283        block.clone()
284    }
285}
