25// Assist: convert_to_guarded_return
26//
27// Replace a large conditional with a guarded return.
28//
29// ```
30// fn main() {
31//     $0if cond {
32//         foo();
33//         bar();
34//     }
35// }
36// ```
37// ->
38// ```
39// fn main() {
40//     if !cond {
41//         return;
42//     }
43//     foo();
44//     bar();
45// }
46// ```
47pub(crate) fn convert_to_guarded_return(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
48    if let Some(let_stmt) = ctx.find_node_at_offset() {
49        let_stmt_to_guarded_return(let_stmt, acc, ctx)
50    } else if let Some(if_expr) = ctx.find_node_at_offset() {
51        if_expr_to_guarded_return(if_expr, acc, ctx)
52    } else {
53        None
54    }
55}
56
57fn if_expr_to_guarded_return(
58    if_expr: ast::IfExpr,
59    acc: &mut Assists,
60    ctx: &AssistContext<'_>,
61) -> Option<()> {
62    if if_expr.else_branch().is_some() {
63        return None;
64    }
65
66    let cond = if_expr.condition()?;
67
68    let if_token_range = if_expr.if_token()?.text_range();
69    let if_cond_range = cond.syntax().text_range();
70
71    let cursor_in_range =
72        if_token_range.cover(if_cond_range).contains_range(ctx.selection_trimmed());
73    if !cursor_in_range {
74        return None;
75    }
76
77    // Check if there is an IfLet that we can handle.
78    let (if_let_pat, cond_expr) = if is_pattern_cond(cond.clone()) {
79        let let_ = single_let(cond)?;
80        (Some(let_.pat()?), let_.expr()?)
81    } else {
82        (None, cond)
83    };
84
85    let then_block = if_expr.then_branch()?;
86    let then_block = then_block.stmt_list()?;
87
88    let parent_block = if_expr.syntax().parent()?.ancestors().find_map(ast::BlockExpr::cast)?;
89
90    if parent_block.tail_expr()? != if_expr.clone().into() {
91        return None;
92    }
93
94    // FIXME: This relies on untyped syntax tree and casts to much. It should be
95    // rewritten to use strongly-typed APIs.
96
97    // check for early return and continue
98    let first_in_then_block = then_block.syntax().first_child()?;
99    if ast::ReturnExpr::can_cast(first_in_then_block.kind())
100        || ast::ContinueExpr::can_cast(first_in_then_block.kind())
101        || first_in_then_block
102            .children()
103            .any(|x| ast::ReturnExpr::can_cast(x.kind()) || ast::ContinueExpr::can_cast(x.kind()))
104    {
105        return None;
106    }
107
108    let parent_container = parent_block.syntax().parent()?;
109
110    let early_expression: ast::Expr = match parent_container.kind() {
111        WHILE_EXPR | LOOP_EXPR | FOR_EXPR => make::expr_continue(None),
112        FN => make::expr_return(None),
113        _ => return None,
114    };
115
116    then_block.syntax().first_child_or_token().map(|t| t.kind() == T!['{'])?;
117
118    then_block.syntax().last_child_or_token().filter(|t| t.kind() == T!['}'])?;
119
120    let then_block_items = then_block.dedent(IndentLevel(1)).clone_for_update();
121
122    let end_of_then = then_block_items.syntax().last_child_or_token()?;
123    let end_of_then = if end_of_then.prev_sibling_or_token().map(|n| n.kind()) == Some(WHITESPACE) {
124        end_of_then.prev_sibling_or_token()?
125    } else {
126        end_of_then
127    };
128
129    let target = if_expr.syntax().text_range();
130    acc.add(
131        AssistId::refactor_rewrite("convert_to_guarded_return"),
132        "Convert to guarded return",
133        target,
134        |edit| {
135            let if_expr = edit.make_mut(if_expr);
136            let if_indent_level = IndentLevel::from_node(if_expr.syntax());
137            let replacement = match if_let_pat {
138                None => {
139                    // If.
140                    let new_expr = {
141                        let then_branch =
142                            make::block_expr(once(make::expr_stmt(early_expression).into()), None);
143                        let cond = invert_boolean_expression_legacy(cond_expr);
144                        make::expr_if(cond, then_branch, None).indent(if_indent_level)
145                    };
146                    new_expr.syntax().clone_for_update()
147                }
148                Some(pat) => {
149                    // If-let.
150                    let let_else_stmt = make::let_else_stmt(
151                        pat,
152                        None,
153                        cond_expr,
154                        ast::make::tail_only_block_expr(early_expression),
155                    );
156                    let let_else_stmt = let_else_stmt.indent(if_indent_level);
157                    let_else_stmt.syntax().clone_for_update()
158                }
159            };
160
161            let then_statements = replacement
162                .children_with_tokens()
163                .chain(
164                    then_block_items
165                        .syntax()
166                        .children_with_tokens()
167                        .skip(1)
168                        .take_while(|i| *i != end_of_then),
169                )
170                .collect();
171
172            ted::replace_with_many(if_expr.syntax(), then_statements)
173        },
174    )
175}

177fn let_stmt_to_guarded_return(
178    let_stmt: ast::LetStmt,
179    acc: &mut Assists,
180    ctx: &AssistContext<'_>,
181) -> Option<()> {
182    let pat = let_stmt.pat()?;
183    let expr = let_stmt.initializer()?;
184
185    let let_token_range = let_stmt.let_token()?.text_range();
186    let let_pattern_range = pat.syntax().text_range();
187    let cursor_in_range =
188        let_token_range.cover(let_pattern_range).contains_range(ctx.selection_trimmed());
189
190    if !cursor_in_range {
191        return None;
192    }
193
194    let try_enum =
195        ctx.sema.type_of_expr(&expr).and_then(|ty| TryEnum::from_ty(&ctx.sema, &ty.adjusted()))?;
196
197    let happy_pattern = try_enum.happy_pattern(pat);
198    let target = let_stmt.syntax().text_range();
199
200    let early_expression: ast::Expr = {
201        let parent_block =
202            let_stmt.syntax().parent()?.ancestors().find_map(ast::BlockExpr::cast)?;
203        let parent_container = parent_block.syntax().parent()?;
204
205        match parent_container.kind() {
206            WHILE_EXPR | LOOP_EXPR | FOR_EXPR => make::expr_continue(None),
207            FN => make::expr_return(None),
208            _ => return None,
209        }
210    };
211
212    acc.add(
213        AssistId::refactor_rewrite("convert_to_guarded_return"),
214        "Convert to guarded return",
215        target,
216        |edit| {
217            let let_stmt = edit.make_mut(let_stmt);
218            let let_indent_level = IndentLevel::from_node(let_stmt.syntax());
219
220            let replacement = {
221                let let_else_stmt = make::let_else_stmt(
222                    happy_pattern,
223                    let_stmt.ty(),
224                    expr,
225                    ast::make::tail_only_block_expr(early_expression),
226                );
227                let let_else_stmt = let_else_stmt.indent(let_indent_level);
228                let_else_stmt.syntax().clone_for_update()
229            };
230
231            ted::replace(let_stmt.syntax(), replacement)
232        },
233    )
234}
