14// Assist: unwrap_block
15//
16// This assist removes if...else, for, while and loop control statements to just keep the body.
17//
18// ```
19// fn foo() {
20//     if true {$0
21//         println!("foo");
22//     }
23// }
24// ```
25// ->
26// ```
27// fn foo() {
28//     println!("foo");
29// }
30// ```
31pub(crate) fn unwrap_block(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
32    let l_curly_token = ctx.find_token_syntax_at_offset(T!['{'])?;
33    let block = l_curly_token.parent_ancestors().nth(1).and_then(ast::BlockExpr::cast)?;
34    let target = block.syntax().text_range();
35    let mut container = block.syntax().clone();
36    let mut replacement = block.clone();
37    let mut prefer_container = None;
38
39    let from_indent = block.indent_level();
40    let into_indent = loop {
41        let parent = container.parent()?;
42        container = match_ast! {
43            match parent {
44                ast::ForExpr(it) => it.syntax().clone(),
45                ast::LoopExpr(it) => it.syntax().clone(),
46                ast::WhileExpr(it) => it.syntax().clone(),
47                ast::MatchArm(it) => it.parent_match().syntax().clone(),
48                ast::LetStmt(it) => {
49                    replacement = wrap_let(&it, replacement);
50                    prefer_container = Some(it.syntax().clone());
51                    it.syntax().clone()
52                },
53                ast::IfExpr(it) => {
54                    prefer_container.get_or_insert_with(|| {
55                        if let Some(else_branch) = it.else_branch()
56                            && *else_branch.syntax() == container
57                        {
58                            else_branch.syntax().clone()
59                        } else {
60                            it.syntax().clone()
61                        }
62                    });
63                    it.syntax().clone()
64                },
65                ast::ExprStmt(it) => it.syntax().clone(),
66                ast::StmtList(it) => break it.indent_level(),
67                _ => return None,
68            }
69        };
70    };
71    let replacement = replacement.stmt_list()?;
72
73    acc.add(AssistId::refactor_rewrite("unwrap_block"), "Unwrap block", target, |builder| {
74        let mut edit = builder.make_editor(block.syntax());
75        let replacement = replacement.dedent(from_indent).indent(into_indent);
76        let container = prefer_container.unwrap_or(container);
77
78        edit.replace_with_many(&container, extract_statements(replacement));
79        delete_else_before(container, &mut edit);
80
81        builder.add_file_edits(ctx.vfs_file_id(), edit);
82    })
83}

85fn delete_else_before(container: SyntaxNode, edit: &mut SyntaxEditor) {
86    let Some(else_token) = container
87        .siblings_with_tokens(syntax::Direction::Prev)
88        .skip(1)
89        .map_while(|it| it.into_token())
90        .find(|it| it.kind() == T![else])
91    else {
92        return;
93    };
94    itertools::chain(else_token.prev_token(), else_token.next_token())
95        .filter(|it| it.kind() == SyntaxKind::WHITESPACE)
96        .for_each(|it| edit.delete(it));
97    let indent = IndentLevel::from_node(&container);
98    let newline = make::tokens::whitespace(&format!("\n{indent}"));
99    edit.replace(else_token, newline);
100}

102fn wrap_let(assign: &ast::LetStmt, replacement: ast::BlockExpr) -> ast::BlockExpr {
103    let try_wrap_assign = || {
104        let initializer = assign.initializer()?.syntax().syntax_element();
105        let replacement = replacement.clone_subtree();
106        let assign = assign.clone_for_update();
107        let tail_expr = replacement.tail_expr()?;
108        let before =
109            assign.syntax().children_with_tokens().take_while(|it| *it != initializer).collect();
110        let after = assign
111            .syntax()
112            .children_with_tokens()
113            .skip_while(|it| *it != initializer)
114            .skip(1)
115            .collect();
116
117        let mut edit = SyntaxEditor::new(replacement.syntax().clone());
118        edit.insert_all(Position::before(tail_expr.syntax()), before);
119        edit.insert_all(Position::after(tail_expr.syntax()), after);
120        ast::BlockExpr::cast(edit.finish().new_root().clone())
121    };
122    try_wrap_assign().unwrap_or(replacement)
123}
124
125fn extract_statements(stmt_list: ast::StmtList) -> Vec<SyntaxElement> {
126    let mut elements = stmt_list
127        .syntax()
128        .children_with_tokens()
129        .filter(|it| !matches!(it.kind(), T!['{'] | T!['}']))
130        .skip_while(|it| it.kind() == SyntaxKind::WHITESPACE)
131        .collect::<Vec<_>>();
132    while elements.pop_if(|it| it.kind() == SyntaxKind::WHITESPACE).is_some() {}
133    elements
134}
