12// Assist: replace_turbofish_with_explicit_type
13//
14// Converts `::<_>` to an explicit type assignment.
15//
16// ```
17// fn make<T>() -> T { ) }
18// fn main() {
19//     let a = make$0::<i32>();
20// }
21// ```
22// ->
23// ```
24// fn make<T>() -> T { ) }
25// fn main() {
26//     let a: i32 = make();
27// }
28// ```
29pub(crate) fn replace_turbofish_with_explicit_type(
30    acc: &mut Assists,
31    ctx: &AssistContext<'_>,
32) -> Option<()> {
33    let let_stmt = ctx.find_node_at_offset::<LetStmt>()?;
34
35    let initializer = let_stmt.initializer()?;
36
37    let generic_args = generic_arg_list(&initializer)?;
38
39    // Find range of ::<_>
40    let colon2 = generic_args.coloncolon_token()?;
41    let r_angle = generic_args.r_angle_token()?;
42    let turbofish_range = TextRange::new(colon2.text_range().start(), r_angle.text_range().end());
43
44    let turbofish_args: Vec<GenericArg> = generic_args.generic_args().collect();
45
46    // Find type of ::<_>
47    if turbofish_args.len() != 1 {
48        cov_mark::hit!(not_applicable_if_not_single_arg);
49        return None;
50    }
51
52    // An improvement would be to check that this is correctly part of the return value of the
53    // function call, or sub in the actual return type.
54    let returned_type = match ctx.sema.type_of_expr(&initializer) {
55        Some(returned_type) if !returned_type.original.contains_unknown() => {
56            let module = ctx.sema.scope(let_stmt.syntax())?.module();
57            returned_type.original.display_source_code(ctx.db(), module.into(), false).ok()?
58        }
59        _ => {
60            cov_mark::hit!(fallback_to_turbofish_type_if_type_info_not_available);
61            turbofish_args[0].to_string()
62        }
63    };
64
65    let initializer_start = initializer.syntax().text_range().start();
66    if ctx.offset() > turbofish_range.end() || ctx.offset() < initializer_start {
67        cov_mark::hit!(not_applicable_outside_turbofish);
68        return None;
69    }
70
71    if let_stmt.colon_token().is_none() {
72        // If there's no colon in a let statement, then there is no explicit type.
73        // let x = fn::<...>();
74        let ident_range = let_stmt.pat()?.syntax().text_range();
75
76        return acc.add(
77            AssistId::refactor_rewrite("replace_turbofish_with_explicit_type"),
78            "Replace turbofish with explicit type",
79            TextRange::new(initializer_start, turbofish_range.end()),
80            |builder| {
81                builder.insert(ident_range.end(), format!(": {returned_type}"));
82                builder.delete(turbofish_range);
83            },
84        );
85    } else if let Some(InferType(t)) = let_stmt.ty() {
86        // If there's a type inference underscore, we can offer to replace it with the type in
87        // the turbofish.
88        // let x: _ = fn::<...>();
89        let underscore_range = t.syntax().text_range();
90
91        return acc.add(
92            AssistId::refactor_rewrite("replace_turbofish_with_explicit_type"),
93            "Replace `_` with turbofish type",
94            turbofish_range,
95            |builder| {
96                builder.replace(underscore_range, returned_type);
97                builder.delete(turbofish_range);
98            },
99        );
100    }
101
102    None
103}

105fn generic_arg_list(expr: &Expr) -> Option<GenericArgList> {
106    match expr {
107        Expr::MethodCallExpr(expr) => expr.generic_arg_list(),
108        Expr::CallExpr(expr) => {
109            if let Expr::PathExpr(pe) = expr.expr()? {
110                pe.path()?.segment()?.generic_arg_list()
111            } else {
112                cov_mark::hit!(not_applicable_if_non_path_function_call);
113                None
114            }
115        }
116        Expr::AwaitExpr(expr) => generic_arg_list(&expr.expr()?),
117        Expr::TryExpr(expr) => generic_arg_list(&expr.expr()?),
118        _ => {
119            cov_mark::hit!(not_applicable_if_non_function_call_initializer);
120            None
121        }
122    }
123}
