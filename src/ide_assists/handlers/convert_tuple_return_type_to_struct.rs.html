26// Assist: convert_tuple_return_type_to_struct
27//
28// This converts the return type of a function from a tuple type
29// into a tuple struct and updates the body accordingly.
30//
31// ```
32// fn bar() {
33//     let (a, b, c) = foo();
34// }
35//
36// fn foo() -> ($0u32, u32, u32) {
37//     (1, 2, 3)
38// }
39// ```
40// ->
41// ```
42// fn bar() {
43//     let FooResult(a, b, c) = foo();
44// }
45//
46// struct FooResult(u32, u32, u32);
47//
48// fn foo() -> FooResult {
49//     FooResult(1, 2, 3)
50// }
51// ```
52pub(crate) fn convert_tuple_return_type_to_struct(
53    acc: &mut Assists,
54    ctx: &AssistContext<'_>,
55) -> Option<()> {
56    let ret_type = ctx.find_node_at_offset::<ast::RetType>()?;
57    let type_ref = ret_type.ty()?;
58
59    let ast::Type::TupleType(tuple_ty) = &type_ref else { return None };
60    if tuple_ty.fields().any(|field| matches!(field, ast::Type::ImplTraitType(_))) {
61        return None;
62    }
63
64    let fn_ = ret_type.syntax().parent().and_then(ast::Fn::cast)?;
65    let fn_def = ctx.sema.to_def(&fn_)?;
66    let fn_name = fn_.name()?;
67    let target_module = ctx.sema.scope(fn_.syntax())?.module().nearest_non_block_module(ctx.db());
68
69    let target = type_ref.syntax().text_range();
70    acc.add(
71        AssistId::refactor_rewrite("convert_tuple_return_type_to_struct"),
72        "Convert tuple return type to tuple struct",
73        target,
74        move |edit| {
75            let mut syntax_editor = edit.make_editor(ret_type.syntax());
76            let syntax_factory = SyntaxFactory::with_mappings();
77
78            let usages = Definition::Function(fn_def).usages(&ctx.sema).all();
79            let struct_name = format!("{}Result", stdx::to_camel_case(&fn_name.to_string()));
80            let parent = fn_.syntax().ancestors().find_map(<Either<ast::Impl, ast::Trait>>::cast);
81            add_tuple_struct_def(
82                edit,
83                &syntax_factory,
84                ctx,
85                &usages,
86                parent.as_ref().map(|it| it.syntax()).unwrap_or(fn_.syntax()),
87                tuple_ty,
88                &struct_name,
89                &target_module,
90            );
91
92            syntax_editor.replace(
93                ret_type.syntax(),
94                syntax_factory.ret_type(syntax_factory.ty(&struct_name)).syntax(),
95            );
96
97            if let Some(fn_body) = fn_.body() {
98                replace_body_return_values(
99                    &mut syntax_editor,
100                    &syntax_factory,
101                    ast::Expr::BlockExpr(fn_body),
102                    &struct_name,
103                );
104            }
105
106            syntax_editor.add_mappings(syntax_factory.finish_with_mappings());
107            edit.add_file_edits(ctx.vfs_file_id(), syntax_editor);
108
109            replace_usages(edit, ctx, &usages, &struct_name, &target_module);
110        },
111    )
112}

114/// Replaces tuple usages with the corresponding tuple struct pattern.
115fn replace_usages(
116    edit: &mut SourceChangeBuilder,
117    ctx: &AssistContext<'_>,
118    usages: &UsageSearchResult,
119    struct_name: &str,
120    target_module: &hir::Module,
121) {
122    for (file_id, references) in usages.iter() {
123        let Some(first_ref) = references.first() else { continue };
124
125        let mut editor = edit.make_editor(first_ref.name.syntax().as_node().unwrap());
126        let syntax_factory = SyntaxFactory::with_mappings();
127
128        let refs_with_imports = augment_references_with_imports(
129            &syntax_factory,
130            ctx,
131            references,
132            struct_name,
133            target_module,
134        );
135
136        refs_with_imports.into_iter().rev().for_each(|(name, import_data)| {
137            if let Some(fn_) = name.syntax().parent().and_then(ast::Fn::cast) {
138                cov_mark::hit!(replace_trait_impl_fns);
139
140                if let Some(ret_type) = fn_.ret_type() {
141                    editor.replace(
142                        ret_type.syntax(),
143                        syntax_factory.ret_type(syntax_factory.ty(struct_name)).syntax(),
144                    );
145                }
146
147                if let Some(fn_body) = fn_.body() {
148                    replace_body_return_values(
149                        &mut editor,
150                        &syntax_factory,
151                        ast::Expr::BlockExpr(fn_body),
152                        struct_name,
153                    );
154                }
155            } else {
156                // replace tuple patterns
157                let pats = name
158                    .syntax()
159                    .ancestors()
160                    .find(|node| {
161                        ast::CallExpr::can_cast(node.kind())
162                            || ast::MethodCallExpr::can_cast(node.kind())
163                    })
164                    .and_then(|node| node.parent())
165                    .and_then(node_to_pats)
166                    .unwrap_or(Vec::new());
167
168                let tuple_pats = pats.iter().filter_map(|pat| match pat {
169                    ast::Pat::TuplePat(tuple_pat) => Some(tuple_pat),
170                    _ => None,
171                });
172                for tuple_pat in tuple_pats {
173                    editor.replace(
174                        tuple_pat.syntax(),
175                        syntax_factory
176                            .tuple_struct_pat(
177                                syntax_factory.path_from_text(struct_name),
178                                tuple_pat.fields(),
179                            )
180                            .syntax(),
181                    );
182                }
183            }
184            if let Some((import_scope, path)) = import_data {
185                insert_use_with_editor(
186                    &import_scope,
187                    path,
188                    &ctx.config.insert_use,
189                    &mut editor,
190                    &syntax_factory,
191                );
192            }
193        });
194
195        editor.add_mappings(syntax_factory.finish_with_mappings());
196        edit.add_file_edits(file_id.file_id(ctx.db()), editor);
197    }
198}

200fn node_to_pats(node: SyntaxNode) -> Option<Vec<ast::Pat>> {
201    match_ast! {
202        match node {
203            ast::LetStmt(it) => it.pat().map(|pat| vec![pat]),
204            ast::LetExpr(it) => it.pat().map(|pat| vec![pat]),
205            ast::MatchExpr(it) => it.match_arm_list().map(|arm_list| {
206                arm_list.arms().filter_map(|arm| arm.pat()).collect()
207            }),
208            _ => None,
209        }
210    }
211}

213fn augment_references_with_imports(
214    syntax_factory: &SyntaxFactory,
215    ctx: &AssistContext<'_>,
216    references: &[FileReference],
217    struct_name: &str,
218    target_module: &hir::Module,
219) -> Vec<(ast::NameLike, Option<(ImportScope, ast::Path)>)> {
220    let mut visited_modules = FxHashSet::default();
221
222    references
223        .iter()
224        .filter_map(|FileReference { name, .. }| {
225            let name = name.clone().into_name_like()?;
226            ctx.sema.scope(name.syntax()).map(|scope| (name, scope.module()))
227        })
228        .map(|(name, ref_module)| {
229            // if the referenced module is not the same as the target one and has not been seen before, add an import
230            let import_data = if ref_module.nearest_non_block_module(ctx.db()) != *target_module
231                && !visited_modules.contains(&ref_module)
232            {
233                visited_modules.insert(ref_module);
234
235                let cfg =
236                    ctx.config.find_path_config(ctx.sema.is_nightly(ref_module.krate(ctx.sema.db)));
237                let import_scope = ImportScope::find_insert_use_container(name.syntax(), &ctx.sema);
238                let path = ref_module
239                    .find_use_path(
240                        ctx.sema.db,
241                        ModuleDef::Module(*target_module),
242                        ctx.config.insert_use.prefix_kind,
243                        cfg,
244                    )
245                    .map(|mod_path| {
246                        syntax_factory.path_concat(
247                            mod_path_to_ast(
248                                &mod_path,
249                                target_module.krate(ctx.db()).edition(ctx.db()),
250                            ),
251                            syntax_factory.path_from_text(struct_name),
252                        )
253                    });
254
255                import_scope.zip(path)
256            } else {
257                None
258            };
259
260            (name, import_data)
261        })
262        .collect()
263}

265// Adds the definition of the tuple struct before the parent function.
266fn add_tuple_struct_def(
267    edit: &mut SourceChangeBuilder,
268    syntax_factory: &SyntaxFactory,
269    ctx: &AssistContext<'_>,
270    usages: &UsageSearchResult,
271    parent: &SyntaxNode,
272    tuple_ty: &ast::TupleType,
273    struct_name: &str,
274    target_module: &hir::Module,
275) {
276    let make_struct_pub = usages
277        .iter()
278        .flat_map(|(_, refs)| refs)
279        .filter_map(|FileReference { name, .. }| {
280            let name = name.clone().into_name_like()?;
281            ctx.sema.scope(name.syntax()).map(|scope| scope.module())
282        })
283        .any(|module| module.nearest_non_block_module(ctx.db()) != *target_module);
284    let visibility = if make_struct_pub { Some(syntax_factory.visibility_pub()) } else { None };
285
286    let field_list = ast::FieldList::TupleFieldList(syntax_factory.tuple_field_list(
287        tuple_ty.fields().map(|ty| syntax_factory.tuple_field(visibility.clone(), ty)),
288    ));
289    let struct_name = syntax_factory.name(struct_name);
290    let struct_def = syntax_factory.struct_(visibility, struct_name, None, field_list);
291
292    let indent = IndentLevel::from_node(parent);
293    let struct_def = struct_def.indent(indent);
294
295    edit.insert(parent.text_range().start(), format!("{struct_def}\n\n{indent}"));
296}

298/// Replaces each returned tuple in `body` with the constructor of the tuple struct named `struct_name`.
299fn replace_body_return_values(
300    syntax_editor: &mut SyntaxEditor,
301    syntax_factory: &SyntaxFactory,
302    body: ast::Expr,
303    struct_name: &str,
304) {
305    let mut exprs_to_wrap = Vec::new();
306
307    let tail_cb = &mut |e: &_| tail_cb_impl(&mut exprs_to_wrap, e);
308    walk_expr(&body, &mut |expr| {
309        if let ast::Expr::ReturnExpr(ret_expr) = expr
310            && let Some(ret_expr_arg) = &ret_expr.expr()
311        {
312            for_each_tail_expr(ret_expr_arg, tail_cb);
313        }
314    });
315    for_each_tail_expr(&body, tail_cb);
316
317    for ret_expr in exprs_to_wrap {
318        if let ast::Expr::TupleExpr(tuple_expr) = &ret_expr {
319            let struct_constructor = syntax_factory.expr_call(
320                syntax_factory.expr_path(syntax_factory.ident_path(struct_name)),
321                syntax_factory.arg_list(tuple_expr.fields()),
322            );
323            syntax_editor.replace(ret_expr.syntax(), struct_constructor.syntax());
324        }
325    }
326}
327
328fn tail_cb_impl(acc: &mut Vec<ast::Expr>, e: &ast::Expr) {
329    match e {
330        ast::Expr::BreakExpr(break_expr) => {
331            if let Some(break_expr_arg) = break_expr.expr() {
332                for_each_tail_expr(&break_expr_arg, &mut |e| tail_cb_impl(acc, e))
333            }
334        }
335        ast::Expr::ReturnExpr(_) => {
336            // all return expressions have already been handled by the walk loop
337        }
338        e => acc.push(e.clone()),
339    }
340}
