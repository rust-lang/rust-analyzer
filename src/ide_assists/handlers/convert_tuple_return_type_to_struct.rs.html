21// Assist: convert_tuple_return_type_to_struct
22//
23// This converts the return type of a function from a tuple type
24// into a tuple struct and updates the body accordingly.
25//
26// ```
27// fn bar() {
28//     let (a, b, c) = foo();
29// }
30//
31// fn foo() -> ($0u32, u32, u32) {
32//     (1, 2, 3)
33// }
34// ```
35// ->
36// ```
37// fn bar() {
38//     let FooResult(a, b, c) = foo();
39// }
40//
41// struct FooResult(u32, u32, u32);
42//
43// fn foo() -> FooResult {
44//     FooResult(1, 2, 3)
45// }
46// ```
47pub(crate) fn convert_tuple_return_type_to_struct(
48    acc: &mut Assists,
49    ctx: &AssistContext<'_>,
50) -> Option<()> {
51    let ret_type = ctx.find_node_at_offset::<ast::RetType>()?;
52    let type_ref = ret_type.ty()?;
53
54    let ast::Type::TupleType(tuple_ty) = &type_ref else { return None };
55    if tuple_ty.fields().any(|field| matches!(field, ast::Type::ImplTraitType(_))) {
56        return None;
57    }
58
59    let fn_ = ret_type.syntax().parent().and_then(ast::Fn::cast)?;
60    let fn_def = ctx.sema.to_def(&fn_)?;
61    let fn_name = fn_.name()?;
62    let target_module = ctx.sema.scope(fn_.syntax())?.module().nearest_non_block_module(ctx.db());
63
64    let target = type_ref.syntax().text_range();
65    acc.add(
66        AssistId::refactor_rewrite("convert_tuple_return_type_to_struct"),
67        "Convert tuple return type to tuple struct",
68        target,
69        move |edit| {
70            let ret_type = edit.make_mut(ret_type);
71            let fn_ = edit.make_mut(fn_);
72
73            let usages = Definition::Function(fn_def).usages(&ctx.sema).all();
74            let struct_name = format!("{}Result", stdx::to_camel_case(&fn_name.to_string()));
75            let parent = fn_.syntax().ancestors().find_map(<Either<ast::Impl, ast::Trait>>::cast);
76            add_tuple_struct_def(
77                edit,
78                ctx,
79                &usages,
80                parent.as_ref().map(|it| it.syntax()).unwrap_or(fn_.syntax()),
81                tuple_ty,
82                &struct_name,
83                &target_module,
84            );
85
86            ted::replace(
87                ret_type.syntax(),
88                make::ret_type(make::ty(&struct_name)).syntax().clone_for_update(),
89            );
90
91            if let Some(fn_body) = fn_.body() {
92                replace_body_return_values(ast::Expr::BlockExpr(fn_body), &struct_name);
93            }
94
95            replace_usages(edit, ctx, &usages, &struct_name, &target_module);
96        },
97    )
98}

100/// Replaces tuple usages with the corresponding tuple struct pattern.
101fn replace_usages(
102    edit: &mut SourceChangeBuilder,
103    ctx: &AssistContext<'_>,
104    usages: &UsageSearchResult,
105    struct_name: &str,
106    target_module: &hir::Module,
107) {
108    for (file_id, references) in usages.iter() {
109        edit.edit_file(file_id.file_id(ctx.db()));
110
111        let refs_with_imports =
112            augment_references_with_imports(edit, ctx, references, struct_name, target_module);
113
114        refs_with_imports.into_iter().rev().for_each(|(name, import_data)| {
115            if let Some(fn_) = name.syntax().parent().and_then(ast::Fn::cast) {
116                cov_mark::hit!(replace_trait_impl_fns);
117
118                if let Some(ret_type) = fn_.ret_type() {
119                    ted::replace(
120                        ret_type.syntax(),
121                        make::ret_type(make::ty(struct_name)).syntax().clone_for_update(),
122                    );
123                }
124
125                if let Some(fn_body) = fn_.body() {
126                    replace_body_return_values(ast::Expr::BlockExpr(fn_body), struct_name);
127                }
128            } else {
129                // replace tuple patterns
130                let pats = name
131                    .syntax()
132                    .ancestors()
133                    .find(|node| {
134                        ast::CallExpr::can_cast(node.kind())
135                            || ast::MethodCallExpr::can_cast(node.kind())
136                    })
137                    .and_then(|node| node.parent())
138                    .and_then(node_to_pats)
139                    .unwrap_or(Vec::new());
140
141                let tuple_pats = pats.iter().filter_map(|pat| match pat {
142                    ast::Pat::TuplePat(tuple_pat) => Some(tuple_pat),
143                    _ => None,
144                });
145                for tuple_pat in tuple_pats {
146                    ted::replace(
147                        tuple_pat.syntax(),
148                        make::tuple_struct_pat(
149                            make::path_from_text(struct_name),
150                            tuple_pat.fields(),
151                        )
152                        .clone_for_update()
153                        .syntax(),
154                    );
155                }
156            }
157            // add imports across modules where needed
158            if let Some((import_scope, path)) = import_data {
159                insert_use(&import_scope, path, &ctx.config.insert_use);
160            }
161        })
162    }
163}
164
165fn node_to_pats(node: SyntaxNode) -> Option<Vec<ast::Pat>> {
166    match_ast! {
167        match node {
168            ast::LetStmt(it) => it.pat().map(|pat| vec![pat]),
169            ast::LetExpr(it) => it.pat().map(|pat| vec![pat]),
170            ast::MatchExpr(it) => it.match_arm_list().map(|arm_list| {
171                arm_list.arms().filter_map(|arm| arm.pat()).collect()
172            }),
173            _ => None,
174        }
175    }
176}

178fn augment_references_with_imports(
179    edit: &mut SourceChangeBuilder,
180    ctx: &AssistContext<'_>,
181    references: &[FileReference],
182    struct_name: &str,
183    target_module: &hir::Module,
184) -> Vec<(ast::NameLike, Option<(ImportScope, ast::Path)>)> {
185    let mut visited_modules = FxHashSet::default();
186
187    references
188        .iter()
189        .filter_map(|FileReference { name, .. }| {
190            let name = name.clone().into_name_like()?;
191            ctx.sema.scope(name.syntax()).map(|scope| (name, scope.module()))
192        })
193        .map(|(name, ref_module)| {
194            let new_name = edit.make_mut(name);
195
196            // if the referenced module is not the same as the target one and has not been seen before, add an import
197            let import_data = if ref_module.nearest_non_block_module(ctx.db()) != *target_module
198                && !visited_modules.contains(&ref_module)
199            {
200                visited_modules.insert(ref_module);
201
202                let cfg = ctx.config.find_path_config(ctx.sema.is_nightly(ref_module.krate()));
203                let import_scope =
204                    ImportScope::find_insert_use_container(new_name.syntax(), &ctx.sema);
205                let path = ref_module
206                    .find_use_path(
207                        ctx.sema.db,
208                        ModuleDef::Module(*target_module),
209                        ctx.config.insert_use.prefix_kind,
210                        cfg,
211                    )
212                    .map(|mod_path| {
213                        make::path_concat(
214                            mod_path_to_ast(&mod_path, target_module.krate().edition(ctx.db())),
215                            make::path_from_text(struct_name),
216                        )
217                    });
218
219                import_scope.zip(path)
220            } else {
221                None
222            };
223
224            (new_name, import_data)
225        })
226        .collect()
227}

229// Adds the definition of the tuple struct before the parent function.
230fn add_tuple_struct_def(
231    edit: &mut SourceChangeBuilder,
232    ctx: &AssistContext<'_>,
233    usages: &UsageSearchResult,
234    parent: &SyntaxNode,
235    tuple_ty: &ast::TupleType,
236    struct_name: &str,
237    target_module: &hir::Module,
238) {
239    let make_struct_pub = usages
240        .iter()
241        .flat_map(|(_, refs)| refs)
242        .filter_map(|FileReference { name, .. }| {
243            let name = name.clone().into_name_like()?;
244            ctx.sema.scope(name.syntax()).map(|scope| scope.module())
245        })
246        .any(|module| module.nearest_non_block_module(ctx.db()) != *target_module);
247    let visibility = if make_struct_pub { Some(make::visibility_pub()) } else { None };
248
249    let field_list = ast::FieldList::TupleFieldList(make::tuple_field_list(
250        tuple_ty.fields().map(|ty| make::tuple_field(visibility.clone(), ty)),
251    ));
252    let struct_name = make::name(struct_name);
253    let struct_def = make::struct_(visibility, struct_name, None, field_list).clone_for_update();
254
255    let indent = IndentLevel::from_node(parent);
256    struct_def.reindent_to(indent);
257
258    edit.insert(parent.text_range().start(), format!("{struct_def}\n\n{indent}"));
259}

261/// Replaces each returned tuple in `body` with the constructor of the tuple struct named `struct_name`.
262fn replace_body_return_values(body: ast::Expr, struct_name: &str) {
263    let mut exprs_to_wrap = Vec::new();
264
265    let tail_cb = &mut |e: &_| tail_cb_impl(&mut exprs_to_wrap, e);
266    walk_expr(&body, &mut |expr| {
267        if let ast::Expr::ReturnExpr(ret_expr) = expr
268            && let Some(ret_expr_arg) = &ret_expr.expr()
269        {
270            for_each_tail_expr(ret_expr_arg, tail_cb);
271        }
272    });
273    for_each_tail_expr(&body, tail_cb);
274
275    for ret_expr in exprs_to_wrap {
276        if let ast::Expr::TupleExpr(tuple_expr) = &ret_expr {
277            let struct_constructor = make::expr_call(
278                make::expr_path(make::ext::ident_path(struct_name)),
279                make::arg_list(tuple_expr.fields()),
280            )
281            .clone_for_update();
282            ted::replace(ret_expr.syntax(), struct_constructor.syntax());
283        }
284    }
285}
286
287fn tail_cb_impl(acc: &mut Vec<ast::Expr>, e: &ast::Expr) {
288    match e {
289        ast::Expr::BreakExpr(break_expr) => {
290            if let Some(break_expr_arg) = break_expr.expr() {
291                for_each_tail_expr(&break_expr_arg, &mut |e| tail_cb_impl(acc, e))
292            }
293        }
294        ast::Expr::ReturnExpr(_) => {
295            // all return expressions have already been handled by the walk loop
296        }
297        e => acc.push(e.clone()),
298    }
299}
