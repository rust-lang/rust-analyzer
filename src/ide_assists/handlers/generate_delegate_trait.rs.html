28// Assist: generate_delegate_trait
29//
30// Generate delegate trait implementation for `StructField`s.
31//
32// ```
33// trait SomeTrait {
34//     type T;
35//     fn fn_(arg: u32) -> u32;
36//     fn method_(&mut self) -> bool;
37// }
38// struct A;
39// impl SomeTrait for A {
40//     type T = u32;
41//
42//     fn fn_(arg: u32) -> u32 {
43//         42
44//     }
45//
46//     fn method_(&mut self) -> bool {
47//         false
48//     }
49// }
50// struct B {
51//     a$0: A,
52// }
53// ```
54// ->
55// ```
56// trait SomeTrait {
57//     type T;
58//     fn fn_(arg: u32) -> u32;
59//     fn method_(&mut self) -> bool;
60// }
61// struct A;
62// impl SomeTrait for A {
63//     type T = u32;
64//
65//     fn fn_(arg: u32) -> u32 {
66//         42
67//     }
68//
69//     fn method_(&mut self) -> bool {
70//         false
71//     }
72// }
73// struct B {
74//     a: A,
75// }
76//
77// impl SomeTrait for B {
78//     type T = <A as SomeTrait>::T;
79//
80//     fn fn_(arg: u32) -> u32 {
81//         <A as SomeTrait>::fn_(arg)
82//     }
83//
84//     fn method_(&mut self) -> bool {
85//         <A as SomeTrait>::method_(&mut self.a)
86//     }
87// }
88// ```
89pub(crate) fn generate_delegate_trait(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
90    if !ctx.config.code_action_grouping {
91        return None;
92    }
93
94    let strukt = Struct::new(ctx.find_node_at_offset::<ast::Struct>()?)?;
95
96    let field: Field = match ctx.find_node_at_offset::<ast::RecordField>() {
97        Some(field) => Field::new(ctx, Either::Left(field))?,
98        None => {
99            let field = ctx.find_node_at_offset::<ast::TupleField>()?;
100            let field_list = ctx.find_node_at_offset::<ast::TupleFieldList>()?;
101            Field::new(ctx, either::Right((field, field_list)))?
102        }
103    };
104
105    strukt.delegate(field, acc, ctx);
106    Some(())
107}

109/// A utility object that represents a struct's field.
110#[derive(Debug)]
111struct Field {
112    name: String,
113    ty: ast::Type,
114    range: syntax::TextRange,
115    impls: Vec<Delegee>,
116    edition: Edition,
117}
118
119impl Field {
120    pub(crate) fn new(
121        ctx: &AssistContext<'_>,
122        f: Either<ast::RecordField, (ast::TupleField, ast::TupleFieldList)>,
123    ) -> Option<Field> {
124        let db = ctx.sema.db;
125
126        let module = ctx.sema.file_to_module_def(ctx.vfs_file_id())?;
127        let edition = module.krate().edition(ctx.db());
128
129        let (name, range, ty) = match f {
130            Either::Left(f) => {
131                let name = f.name()?.to_string();
132                (name, f.syntax().text_range(), f.ty()?)
133            }
134            Either::Right((f, l)) => {
135                let name = l.fields().position(|it| it == f)?.to_string();
136                (name, f.syntax().text_range(), f.ty()?)
137            }
138        };
139
140        let hir_ty = ctx.sema.resolve_type(&ty)?;
141        let type_impls = hir::Impl::all_for_type(db, hir_ty.clone());
142        let mut impls = Vec::with_capacity(type_impls.len());
143
144        if let Some(tp) = hir_ty.as_type_param(db) {
145            for tb in tp.trait_bounds(db) {
146                impls.push(Delegee::Bound(tb));
147            }
148        };
149
150        for imp in type_impls {
151            if let Some(tr) = imp.trait_(db).filter(|tr| tr.is_visible_from(db, module)) {
152                impls.push(Delegee::Impls(tr, imp))
153            }
154        }
155
156        Some(Field { name, ty, range, impls, edition })
157    }
158}

160/// A field that we want to delegate can offer the enclosing struct
161/// trait to implement in two ways. The first way is when the field
162/// actually implements the trait and the second way is when the field
163/// has a bound type parameter. We handle these cases in different ways
164/// hence the enum.
165#[derive(Debug)]
166enum Delegee {
167    Bound(hir::Trait),
168    Impls(hir::Trait, hir::Impl),
169}
170
171impl Delegee {
172    fn signature(&self, db: &dyn HirDatabase, edition: Edition) -> String {
173        let mut s = String::new();
174
175        let (Delegee::Bound(it) | Delegee::Impls(it, _)) = self;
176
177        for m in it.module(db).path_to_root(db).iter().rev() {
178            if let Some(name) = m.name(db) {
179                s.push_str(&format!("{}::", name.display_no_db(edition).to_smolstr()));
180            }
181        }
182
183        s.push_str(&it.name(db).display_no_db(edition).to_smolstr());
184        s
185    }
186}

188/// A utility struct that is used for the enclosing struct.
189struct Struct {
190    strukt: ast::Struct,
191    name: ast::Name,
192}
193
194impl Struct {
195    pub(crate) fn new(s: ast::Struct) -> Option<Self> {
196        let name = s.name()?;
197        Some(Struct { name, strukt: s })
198    }
199
200    pub(crate) fn delegate(&self, field: Field, acc: &mut Assists, ctx: &AssistContext<'_>) {
201        let db = ctx.db();
202
203        for (index, delegee) in field.impls.iter().enumerate() {
204            let trait_ = match delegee {
205                Delegee::Bound(b) => b,
206                Delegee::Impls(i, _) => i,
207            };
208
209            // Skip trait that has `Self` type, which cannot be delegated
210            //
211            // See [`test_self_ty`]
212            if has_self_type(*trait_, ctx).is_some() {
213                continue;
214            }
215
216            // FIXME :  We can omit already implemented impl_traits
217            // But we don't know what the &[hir::Type] argument should look like.
218            // if self.hir_ty.impls_trait(db, trait_, &[]) {
219            //     continue;
220            // }
221            let signature = delegee.signature(db, field.edition);
222
223            let Some(delegate) =
224                generate_impl(ctx, self, &field.ty, &field.name, delegee, field.edition)
225            else {
226                continue;
227            };
228
229            acc.add_group(
230                &GroupLabel(format!("Generate delegate trait impls for field `{}`", field.name)),
231                AssistId(
232                    "generate_delegate_trait",
233                    ide_db::assists::AssistKind::Generate,
234                    Some(index),
235                ),
236                format!("Generate delegate trait impl `{}` for `{}`", signature, field.name),
237                field.range,
238                |builder| {
239                    builder.insert(
240                        self.strukt.syntax().text_range().end(),
241                        format!("\n\n{}", delegate.syntax()),
242                    );
243                },
244            );
245        }
246    }
247}

249fn generate_impl(
250    ctx: &AssistContext<'_>,
251    strukt: &Struct,
252    field_ty: &ast::Type,
253    field_name: &str,
254    delegee: &Delegee,
255    edition: Edition,
256) -> Option<ast::Impl> {
257    let db = ctx.db();
258    let ast_strukt = &strukt.strukt;
259    let strukt_ty = make::ty_path(make::ext::ident_path(&strukt.name.to_string()));
260    let strukt_params = ast_strukt.generic_param_list();
261
262    match delegee {
263        Delegee::Bound(delegee) => {
264            let bound_def = ctx.sema.source(delegee.to_owned())?.value;
265            let bound_params = bound_def.generic_param_list();
266
267            let delegate = make::impl_trait(
268                None,
269                delegee.is_unsafe(db),
270                bound_params.clone(),
271                bound_params.map(|params| params.to_generic_args()),
272                strukt_params.clone(),
273                strukt_params.map(|params| params.to_generic_args()),
274                delegee.is_auto(db),
275                make::ty(&delegee.name(db).display_no_db(edition).to_smolstr()),
276                strukt_ty,
277                bound_def.where_clause(),
278                ast_strukt.where_clause(),
279                None,
280            )
281            .clone_for_update();
282
283            // Goto link : https://doc.rust-lang.org/reference/paths.html#qualified-paths
284            let qualified_path_type =
285                make::path_from_text(&format!("<{} as {}>", field_ty, delegate.trait_()?));
286
287            let delegate_assoc_items = delegate.get_or_create_assoc_item_list();
288            if let Some(ai) = bound_def.assoc_item_list() {
289                ai.assoc_items()
290                    .filter(|item| matches!(item, AssocItem::MacroCall(_)).not())
291                    .for_each(|item| {
292                        let assoc = process_assoc_item(
293                            item.clone_for_update(),
294                            qualified_path_type.clone(),
295                            field_name,
296                        );
297                        if let Some(assoc) = assoc {
298                            delegate_assoc_items.add_item(assoc);
299                        }
300                    });
301            };
302
303            let target_scope = ctx.sema.scope(strukt.strukt.syntax())?;
304            let source_scope = ctx.sema.scope(bound_def.syntax())?;
305            let transform = PathTransform::generic_transformation(&target_scope, &source_scope);
306            ast::Impl::cast(transform.apply(delegate.syntax()))
307        }
308        Delegee::Impls(trait_, old_impl) => {
309            let old_impl = ctx.sema.source(old_impl.to_owned())?.value;
310            let old_impl_params = old_impl.generic_param_list();
311
312            // 1) Resolve conflicts between generic parameters in old_impl and
313            // those in strukt.
314            //
315            // These generics parameters will also be used in `field_ty` and
316            // `where_clauses`, so we should substitute arguments in them as well.
317            let strukt_params = resolve_name_conflicts(strukt_params, &old_impl_params);
318            let (field_ty, ty_where_clause) = match &strukt_params {
319                Some(strukt_params) => {
320                    let args = strukt_params.to_generic_args();
321                    let field_ty = rename_strukt_args(ctx, ast_strukt, field_ty, &args)?;
322                    let where_clause = ast_strukt
323                        .where_clause()
324                        .and_then(|wc| rename_strukt_args(ctx, ast_strukt, &wc, &args));
325                    (field_ty, where_clause)
326                }
327                None => (field_ty.clone_for_update(), None),
328            };
329
330            // 2) Handle instantiated generics in `field_ty`.
331
332            // 2.1) Some generics used in `self_ty` may be instantiated, so they
333            // are no longer generics, we should remove and instantiate those
334            // generics in advance.
335
336            // `old_trait_args` contains names of generic args for trait in `old_impl`
337            let old_impl_trait_args = old_impl
338                .trait_()?
339                .generic_arg_list()
340                .map(|l| l.generic_args().map(|arg| arg.to_string()))
341                .map_or_else(FxHashSet::default, |it| it.collect());
342
343            let trait_gen_params = remove_instantiated_params(
344                &old_impl.self_ty()?,
345                old_impl_params.clone(),
346                &old_impl_trait_args,
347            );
348
349            // 2.2) Generate generic args applied on impl.
350            let transform_args = generate_args_for_impl(
351                old_impl_params,
352                &old_impl.self_ty()?,
353                &field_ty,
354                &trait_gen_params,
355                &old_impl_trait_args,
356            );
357
358            // 2.3) Instantiate generics with `transform_impl`, this step also
359            // remove unused params.
360            let trait_gen_args = old_impl.trait_()?.generic_arg_list().and_then(|trait_args| {
361                let trait_args = &mut trait_args.clone_for_update();
362                if let Some(new_args) = transform_impl(
363                    ctx,
364                    ast_strukt,
365                    &old_impl,
366                    &transform_args,
367                    trait_args.clone_subtree(),
368                ) {
369                    *trait_args = new_args.clone_subtree();
370                    Some(new_args)
371                } else {
372                    None
373                }
374            });
375
376            let type_gen_args = strukt_params.clone().map(|params| params.to_generic_args());
377            let path_type =
378                make::ty(&trait_.name(db).display_no_db(edition).to_smolstr()).clone_for_update();
379            let path_type = transform_impl(ctx, ast_strukt, &old_impl, &transform_args, path_type)?;
380            // 3) Generate delegate trait impl
381            let delegate = make::impl_trait(
382                None,
383                trait_.is_unsafe(db),
384                trait_gen_params,
385                trait_gen_args,
386                strukt_params,
387                type_gen_args,
388                trait_.is_auto(db),
389                path_type,
390                strukt_ty,
391                old_impl.where_clause().map(|wc| wc.clone_for_update()),
392                ty_where_clause,
393                None,
394            )
395            .clone_for_update();
396            // Goto link : https://doc.rust-lang.org/reference/paths.html#qualified-paths
397            let qualified_path_type =
398                make::path_from_text(&format!("<{} as {}>", field_ty, delegate.trait_()?));
399
400            // 4) Transform associated items in delegte trait impl
401            let delegate_assoc_items = delegate.get_or_create_assoc_item_list();
402            for item in old_impl
403                .get_or_create_assoc_item_list()
404                .assoc_items()
405                .filter(|item| matches!(item, AssocItem::MacroCall(_)).not())
406            {
407                let item = item.clone_for_update();
408                let item = transform_impl(ctx, ast_strukt, &old_impl, &transform_args, item)?;
409
410                let assoc = process_assoc_item(item, qualified_path_type.clone(), field_name)?;
411                delegate_assoc_items.add_item(assoc);
412            }
413
414            // 5) Remove useless where clauses
415            if let Some(wc) = delegate.where_clause() {
416                remove_useless_where_clauses(&delegate.trait_()?, &delegate.self_ty()?, wc);
417            }
418            Some(delegate)
419        }
420    }
421}

423fn transform_impl<N: ast::AstNode>(
424    ctx: &AssistContext<'_>,
425    strukt: &ast::Struct,
426    old_impl: &ast::Impl,
427    args: &Option<GenericArgList>,
428    syntax: N,
429) -> Option<N> {
430    let source_scope = ctx.sema.scope(old_impl.self_ty()?.syntax())?;
431    let target_scope = ctx.sema.scope(strukt.syntax())?;
432    let hir_old_impl = ctx.sema.to_impl_def(old_impl)?;
433
434    let transform = args.as_ref().map_or_else(
435        || PathTransform::generic_transformation(&target_scope, &source_scope),
436        |args| {
437            PathTransform::impl_transformation(
438                &target_scope,
439                &source_scope,
440                hir_old_impl,
441                args.clone(),
442            )
443        },
444    );
445
446    N::cast(transform.apply(syntax.syntax()))
447}

449fn remove_instantiated_params(
450    self_ty: &ast::Type,
451    old_impl_params: Option<GenericParamList>,
452    old_trait_args: &FxHashSet<String>,
453) -> Option<GenericParamList> {
454    match self_ty {
455        ast::Type::PathType(path_type) => {
456            old_impl_params.and_then(|gpl| {
457                // Remove generic parameters in field_ty (which is instantiated).
458                let new_gpl = gpl.clone_for_update();
459
460                path_type
461                    .path()?
462                    .segments()
463                    .filter_map(|seg| seg.generic_arg_list())
464                    .flat_map(|it| it.generic_args())
465                    // However, if the param is also used in the trait arguments,
466                    // it shouldn't be removed now, which will be instantiated in
467                    // later `path_transform`
468                    .filter(|arg| !old_trait_args.contains(&arg.to_string()))
469                    .for_each(|arg| new_gpl.remove_generic_arg(&arg));
470                (new_gpl.generic_params().count() > 0).then_some(new_gpl)
471            })
472        }
473        _ => old_impl_params,
474    }
475}

477fn remove_useless_where_clauses(trait_ty: &ast::Type, self_ty: &ast::Type, wc: ast::WhereClause) {
478    let live_generics = [trait_ty, self_ty]
479        .into_iter()
480        .flat_map(|ty| ty.generic_arg_list())
481        .flat_map(|gal| gal.generic_args())
482        .map(|x| x.to_string())
483        .collect::<FxHashSet<_>>();
484
485    // Keep where-clauses that have generics after substitution, and remove the
486    // rest.
487    let has_live_generics = |pred: &WherePred| {
488        pred.syntax()
489            .descendants_with_tokens()
490            .filter_map(|e| e.into_token())
491            .any(|e| e.kind() == SyntaxKind::IDENT && live_generics.contains(&e.to_string()))
492            .not()
493    };
494    wc.predicates().filter(has_live_generics).for_each(|pred| wc.remove_predicate(pred));
495
496    if wc.predicates().count() == 0 {
497        // Remove useless whitespaces
498        [syntax::Direction::Prev, syntax::Direction::Next]
499            .into_iter()
500            .flat_map(|dir| {
501                wc.syntax()
502                    .siblings_with_tokens(dir)
503                    .skip(1)
504                    .take_while(|node_or_tok| node_or_tok.kind() == SyntaxKind::WHITESPACE)
505            })
506            .for_each(ted::remove);
507
508        ted::insert(
509            ted::Position::after(wc.syntax()),
510            NodeOrToken::Token(make::token(SyntaxKind::WHITESPACE)),
511        );
512        // Remove where clause
513        ted::remove(wc.syntax());
514    }
515}

517// Generate generic args that should be apply to current impl.
518//
519// For example, say we have implementation `impl<A, B, C> Trait for B<A>`,
520// and `b: B<T>` in struct `S<T>`. Then the `A` should be instantiated to `T`.
521// While the last two generic args `B` and `C` doesn't change, it remains
522// `<B, C>`. So we apply `<T, B, C>` as generic arguments to impl.
523fn generate_args_for_impl(
524    old_impl_gpl: Option<GenericParamList>,
525    self_ty: &ast::Type,
526    field_ty: &ast::Type,
527    trait_params: &Option<GenericParamList>,
528    old_trait_args: &FxHashSet<String>,
529) -> Option<ast::GenericArgList> {
530    let old_impl_args = old_impl_gpl.map(|gpl| gpl.to_generic_args().generic_args())?;
531    // Create pairs of the args of `self_ty` and corresponding `field_ty` to
532    // form the substitution list
533    let mut arg_substs = FxHashMap::default();
534
535    if let field_ty @ ast::Type::PathType(_) = field_ty {
536        let field_args = field_ty.generic_arg_list().map(|gal| gal.generic_args());
537        let self_ty_args = self_ty.generic_arg_list().map(|gal| gal.generic_args());
538        if let (Some(field_args), Some(self_ty_args)) = (field_args, self_ty_args) {
539            self_ty_args.zip(field_args).for_each(|(self_ty_arg, field_arg)| {
540                arg_substs.entry(self_ty_arg.to_string()).or_insert(field_arg);
541            })
542        }
543    }
544
545    let args = old_impl_args
546        .map(|old_arg| {
547            arg_substs.get(&old_arg.to_string()).map_or_else(
548                || old_arg.clone(),
549                |replace_with| {
550                    // The old_arg will be replaced, so it becomes redundant
551                    if trait_params.is_some() && old_trait_args.contains(&old_arg.to_string()) {
552                        trait_params.as_ref().unwrap().remove_generic_arg(&old_arg)
553                    }
554                    replace_with.clone()
555                },
556            )
557        })
558        .collect_vec();
559    args.is_empty().not().then(|| make::generic_arg_list(args))
560}

562fn rename_strukt_args<N>(
563    ctx: &AssistContext<'_>,
564    strukt: &ast::Struct,
565    item: &N,
566    args: &GenericArgList,
567) -> Option<N>
568where
569    N: ast::AstNode,
570{
571    let hir_strukt = ctx.sema.to_struct_def(strukt)?;
572    let hir_adt = hir::Adt::from(hir_strukt);
573
574    let item = item.clone_for_update();
575    let scope = ctx.sema.scope(item.syntax())?;
576
577    let transform = PathTransform::adt_transformation(&scope, &scope, hir_adt, args.clone());
578    N::cast(transform.apply(item.syntax()))
579}

581fn has_self_type(trait_: hir::Trait, ctx: &AssistContext<'_>) -> Option<()> {
582    let trait_source = ctx.sema.source(trait_)?.value;
583    trait_source
584        .syntax()
585        .descendants_with_tokens()
586        .filter_map(|e| e.into_token())
587        .find(|e| e.kind() == SyntaxKind::SELF_TYPE_KW)
588        .map(|_| ())
589}

591fn resolve_name_conflicts(
592    strukt_params: Option<ast::GenericParamList>,
593    old_impl_params: &Option<ast::GenericParamList>,
594) -> Option<ast::GenericParamList> {
595    match (strukt_params, old_impl_params) {
596        (Some(old_strukt_params), Some(old_impl_params)) => {
597            let params = make::generic_param_list(std::iter::empty()).clone_for_update();
598
599            for old_strukt_param in old_strukt_params.generic_params() {
600                // Get old name from `strukt`
601                let name = SmolStr::from(match &old_strukt_param {
602                    ast::GenericParam::ConstParam(c) => c.name()?.to_string(),
603                    ast::GenericParam::LifetimeParam(l) => {
604                        l.lifetime()?.lifetime_ident_token()?.to_string()
605                    }
606                    ast::GenericParam::TypeParam(t) => t.name()?.to_string(),
607                });
608
609                // The new name cannot be conflicted with generics in trait, and the renamed names.
610                let param_list_to_names = |param_list: &GenericParamList| {
611                    param_list.generic_params().flat_map(|param| match param {
612                        ast::GenericParam::TypeParam(t) => t.name().map(|name| name.to_string()),
613                        p => Some(p.to_string()),
614                    })
615                };
616                let existing_names = param_list_to_names(old_impl_params)
617                    .chain(param_list_to_names(&params))
618                    .collect_vec();
619                let mut name_generator = suggest_name::NameGenerator::new_with_names(
620                    existing_names.iter().map(|s| s.as_str()),
621                );
622                let name = name_generator.suggest_name(&name);
623                match old_strukt_param {
624                    ast::GenericParam::ConstParam(c) => {
625                        if let Some(const_ty) = c.ty() {
626                            let const_param = make::const_param(make::name(&name), const_ty);
627                            params.add_generic_param(ast::GenericParam::ConstParam(
628                                const_param.clone_for_update(),
629                            ));
630                        }
631                    }
632                    p @ ast::GenericParam::LifetimeParam(_) => {
633                        params.add_generic_param(p.clone_for_update());
634                    }
635                    ast::GenericParam::TypeParam(t) => {
636                        let type_bounds = t.type_bound_list();
637                        let type_param = make::type_param(make::name(&name), type_bounds);
638                        params.add_generic_param(ast::GenericParam::TypeParam(
639                            type_param.clone_for_update(),
640                        ));
641                    }
642                }
643            }
644            Some(params)
645        }
646        (Some(old_strukt_gpl), None) => Some(old_strukt_gpl),
647        _ => None,
648    }
649}

651fn process_assoc_item(
652    item: syntax::ast::AssocItem,
653    qual_path_ty: ast::Path,
654    base_name: &str,
655) -> Option<ast::AssocItem> {
656    match item {
657        AssocItem::Const(c) => const_assoc_item(c, qual_path_ty),
658        AssocItem::Fn(f) => func_assoc_item(f, qual_path_ty, base_name),
659        AssocItem::MacroCall(_) => {
660            // FIXME : Handle MacroCall case.
661            // macro_assoc_item(mac, qual_path_ty)
662            None
663        }
664        AssocItem::TypeAlias(ta) => ty_assoc_item(ta, qual_path_ty),
665    }
666}
667
668fn const_assoc_item(item: syntax::ast::Const, qual_path_ty: ast::Path) -> Option<AssocItem> {
669    let path_expr_segment = make::path_from_text(item.name()?.to_string().as_str());
670
671    // We want rhs of the const assignment to be a qualified path
672    // The general case for const assignment can be found [here](`https://doc.rust-lang.org/reference/items/constant-items.html`)
673    // The qualified will have the following generic syntax :
674    // <Base as Trait<GenArgs>>::ConstName;
675    // FIXME : We can't rely on `make::path_qualified` for now but it would be nice to replace the following with it.
676    // make::path_qualified(qual_path_ty, path_expr_segment.as_single_segment().unwrap());
677    let qualified_path = qualified_path(qual_path_ty, path_expr_segment);
678    let inner = make::item_const(
679        item.attrs(),
680        item.visibility(),
681        item.name()?,
682        item.ty()?,
683        make::expr_path(qualified_path),
684    )
685    .clone_for_update();
686
687    Some(AssocItem::Const(inner))
688}
689
690fn func_assoc_item(
691    item: syntax::ast::Fn,
692    qual_path_ty: Path,
693    base_name: &str,
694) -> Option<AssocItem> {
695    let path_expr_segment = make::path_from_text(item.name()?.to_string().as_str());
696    let qualified_path = qualified_path(qual_path_ty, path_expr_segment);
697
698    let call = match item.param_list() {
699        // Methods and funcs should be handled separately.
700        // We ask if the func has a `self` param.
701        Some(l) => match l.self_param() {
702            Some(slf) => {
703                let mut self_kw = make::expr_path(make::path_from_text("self"));
704                self_kw = make::expr_field(self_kw, base_name);
705
706                let tail_expr_self = match slf.kind() {
707                    ast::SelfParamKind::Owned => self_kw,
708                    ast::SelfParamKind::Ref => make::expr_ref(self_kw, false),
709                    ast::SelfParamKind::MutRef => make::expr_ref(self_kw, true),
710                };
711
712                let param_count = l.params().count();
713                let args = convert_param_list_to_arg_list(l).clone_for_update();
714                let pos_after_l_paren = Position::after(args.l_paren_token()?);
715                if param_count > 0 {
716                    // Add SelfParam and a TOKEN::COMMA
717                    ted::insert_all_raw(
718                        pos_after_l_paren,
719                        vec![
720                            NodeOrToken::Node(tail_expr_self.syntax().clone_for_update()),
721                            NodeOrToken::Token(make::token(SyntaxKind::COMMA)),
722                            NodeOrToken::Token(make::token(SyntaxKind::WHITESPACE)),
723                        ],
724                    );
725                } else {
726                    // Add SelfParam only
727                    ted::insert_raw(
728                        pos_after_l_paren,
729                        NodeOrToken::Node(tail_expr_self.syntax().clone_for_update()),
730                    );
731                }
732
733                make::expr_call(make::expr_path(qualified_path), args)
734            }
735            None => {
736                make::expr_call(make::expr_path(qualified_path), convert_param_list_to_arg_list(l))
737            }
738        },
739        None => make::expr_call(
740            make::expr_path(qualified_path),
741            convert_param_list_to_arg_list(make::param_list(None, Vec::new())),
742        ),
743    }
744    .clone_for_update();
745
746    let body = make::block_expr(vec![], Some(call.into())).clone_for_update();
747    let func = make::fn_(
748        item.attrs(),
749        item.visibility(),
750        item.name()?,
751        item.generic_param_list(),
752        item.where_clause(),
753        item.param_list()?,
754        body,
755        item.ret_type(),
756        item.async_token().is_some(),
757        item.const_token().is_some(),
758        item.unsafe_token().is_some(),
759        item.gen_token().is_some(),
760    )
761    .clone_for_update();
762
763    Some(AssocItem::Fn(func.indent(edit::IndentLevel(1))))
764}
765
766fn ty_assoc_item(item: syntax::ast::TypeAlias, qual_path_ty: Path) -> Option<AssocItem> {
767    let path_expr_segment = make::path_from_text(item.name()?.to_string().as_str());
768    let qualified_path = qualified_path(qual_path_ty, path_expr_segment);
769    let ty = make::ty_path(qualified_path);
770    let ident = item.name()?.to_string();
771
772    let alias = make::ty_alias(
773        item.attrs(),
774        ident.as_str(),
775        item.generic_param_list(),
776        None,
777        item.where_clause(),
778        Some((ty, None)),
779    )
780    .indent(edit::IndentLevel(1));
781
782    Some(AssocItem::TypeAlias(alias))
783}

787}
