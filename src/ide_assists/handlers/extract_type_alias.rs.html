10// Assist: extract_type_alias
11//
12// Extracts the selected type as a type alias.
13//
14// ```
15// struct S {
16//     field: $0(u8, u8, u8)$0,
17// }
18// ```
19// ->
20// ```
21// type $0Type = (u8, u8, u8);
22//
23// struct S {
24//     field: Type,
25// }
26// ```
27pub(crate) fn extract_type_alias(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
28    if ctx.has_empty_selection() {
29        return None;
30    }
31
32    let ty = ctx.find_node_at_range::<ast::Type>()?;
33    let item = ty.syntax().ancestors().find_map(ast::Item::cast)?;
34    let assoc_owner =
35        item.syntax().ancestors().nth(2).and_then(Either::<ast::Trait, ast::Impl>::cast);
36    let node = assoc_owner.as_ref().map_or_else(
37        || item.syntax(),
38        |impl_| impl_.as_ref().either(AstNode::syntax, AstNode::syntax),
39    );
40    let target = ty.syntax().text_range();
41
42    acc.add(
43        AssistId::refactor_extract("extract_type_alias"),
44        "Extract type as type alias",
45        target,
46        |builder| {
47            let mut edit = builder.make_editor(node);
48
49            let mut known_generics = match item.generic_param_list() {
50                Some(it) => it.generic_params().collect(),
51                None => Vec::new(),
52            };
53            if let Some(it) = assoc_owner.as_ref().and_then(|it| match it {
54                Either::Left(it) => it.generic_param_list(),
55                Either::Right(it) => it.generic_param_list(),
56            }) {
57                known_generics.extend(it.generic_params());
58            }
59            let generics = collect_used_generics(&ty, &known_generics);
60            let generic_params =
61                generics.map(|it| make::generic_param_list(it.into_iter().cloned()));
62
63            // Replace original type with the alias
64            let ty_args = generic_params.as_ref().map(|it| it.to_generic_args().generic_args());
65            let new_ty = if let Some(ty_args) = ty_args {
66                make::generic_ty_path_segment(make::name_ref("Type"), ty_args)
67            } else {
68                make::path_segment(make::name_ref("Type"))
69            }
70            .clone_for_update();
71            edit.replace(ty.syntax(), new_ty.syntax());
72
73            // Insert new alias
74            let ty_alias =
75                make::ty_alias(None, "Type", generic_params, None, None, Some((ty, None)))
76                    .clone_for_update();
77
78            if let Some(cap) = ctx.config.snippet_cap
79                && let Some(name) = ty_alias.name()
80            {
81                edit.add_annotation(name.syntax(), builder.make_tabstop_before(cap));
82            }
83
84            let indent = IndentLevel::from_node(node);
85            edit.insert_all(
86                syntax_editor::Position::before(node),
87                vec![
88                    ty_alias.syntax().clone().into(),
89                    make::tokens::whitespace(&format!("\n\n{indent}")).into(),
90                ],
91            );
92
93            builder.add_file_edits(ctx.vfs_file_id(), edit);
94        },
95    )
96}

98fn collect_used_generics<'gp>(
99    ty: &ast::Type,
100    known_generics: &'gp [ast::GenericParam],
101) -> Option<Vec<&'gp ast::GenericParam>> {
102    // can't use a closure -> closure here cause lifetime inference fails for that
103    fn find_lifetime(text: &str) -> impl Fn(&&ast::GenericParam) -> bool + '_ {
104        move |gp: &&ast::GenericParam| match gp {
105            ast::GenericParam::LifetimeParam(lp) => {
106                lp.lifetime().is_some_and(|lt| lt.text() == text)
107            }
108            _ => false,
109        }
110    }
111
112    let mut generics = Vec::new();
113    walk_ty(ty, &mut |ty| {
114        match ty {
115            ast::Type::PathType(ty) => {
116                if let Some(path) = ty.path() {
117                    if let Some(name_ref) = path.as_single_name_ref()
118                        && let Some(param) = known_generics.iter().find(|gp| {
119                            match gp {
120                                ast::GenericParam::ConstParam(cp) => cp.name(),
121                                ast::GenericParam::TypeParam(tp) => tp.name(),
122                                _ => None,
123                            }
124                            .is_some_and(|n| n.text() == name_ref.text())
125                        })
126                    {
127                        generics.push(param);
128                    }
129                    generics.extend(
130                        path.segments()
131                            .filter_map(|seg| seg.generic_arg_list())
132                            .flat_map(|it| it.generic_args())
133                            .filter_map(|it| match it {
134                                ast::GenericArg::LifetimeArg(lt) => {
135                                    let lt = lt.lifetime()?;
136                                    known_generics.iter().find(find_lifetime(&lt.text()))
137                                }
138                                _ => None,
139                            }),
140                    );
141                }
142            }
143            ast::Type::ImplTraitType(impl_ty) => {
144                if let Some(it) = impl_ty.type_bound_list() {
145                    generics.extend(
146                        it.bounds()
147                            .filter_map(|it| it.lifetime())
148                            .filter_map(|lt| known_generics.iter().find(find_lifetime(&lt.text()))),
149                    );
150                }
151            }
152            ast::Type::DynTraitType(dyn_ty) => {
153                if let Some(it) = dyn_ty.type_bound_list() {
154                    generics.extend(
155                        it.bounds()
156                            .filter_map(|it| it.lifetime())
157                            .filter_map(|lt| known_generics.iter().find(find_lifetime(&lt.text()))),
158                    );
159                }
160            }
161            ast::Type::RefType(ref_) => generics.extend(
162                ref_.lifetime()
163                    .and_then(|lt| known_generics.iter().find(find_lifetime(&lt.text()))),
164            ),
165            ast::Type::ArrayType(ar) => {
166                if let Some(ast::Expr::PathExpr(p)) = ar.const_arg().and_then(|x| x.expr())
167                    && let Some(path) = p.path()
168                    && let Some(name_ref) = path.as_single_name_ref()
169                    && let Some(param) = known_generics.iter().find(|gp| {
170                        if let ast::GenericParam::ConstParam(cp) = gp {
171                            cp.name().is_some_and(|n| n.text() == name_ref.text())
172                        } else {
173                            false
174                        }
175                    })
176                {
177                    generics.push(param);
178                }
179            }
180            _ => (),
181        };
182        false
183    });
184    // stable resort to lifetime, type, const
185    generics.sort_by_key(|gp| match gp {
186        ast::GenericParam::ConstParam(_) => 2,
187        ast::GenericParam::LifetimeParam(_) => 0,
188        ast::GenericParam::TypeParam(_) => 1,
189    });
190
191    Some(generics).filter(|it| !it.is_empty())
192}
