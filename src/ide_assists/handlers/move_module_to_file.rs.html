15// Assist: move_module_to_file
16//
17// Moves inline module's contents to a separate file.
18//
19// ```
20// mod $0foo {
21//     fn t() {}
22// }
23// ```
24// ->
25// ```
26// mod foo;
27// ```
28pub(crate) fn move_module_to_file(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
29    let module_ast = ctx.find_node_at_offset::<ast::Module>()?;
30    let module_items = module_ast.item_list()?;
31
32    let l_curly_offset = module_items.syntax().text_range().start();
33    if l_curly_offset <= ctx.offset() {
34        cov_mark::hit!(available_before_curly);
35        return None;
36    }
37    let target = TextRange::new(module_ast.syntax().text_range().start(), l_curly_offset);
38
39    let module_name = module_ast.name()?;
40
41    // get to the outermost module syntax so we can grab the module of file we are in
42    let outermost_mod_decl =
43        iter::successors(Some(module_ast.clone()), |module| module.parent()).last()?;
44    let module_def = ctx.sema.to_def(&outermost_mod_decl)?;
45    let parent_module = module_def.parent(ctx.db())?;
46
47    acc.add(
48        AssistId::refactor_extract("move_module_to_file"),
49        "Extract module to file",
50        target,
51        |builder| {
52            let path = {
53                let mut buf = String::from("./");
54                let db = ctx.db();
55                match parent_module.name(db) {
56                    Some(name)
57                        if !parent_module.is_mod_rs(db)
58                            && parent_module
59                                .attrs(db)
60                                .by_key(sym::path)
61                                .string_value_unescape()
62                                .is_none() =>
63                    {
64                        format_to!(buf, "{}/", name.as_str())
65                    }
66                    _ => (),
67                }
68                let segments = iter::successors(Some(module_ast.clone()), |module| module.parent())
69                    .filter_map(|it| it.name())
70                    .map(|name| SmolStr::from(name.text().trim_start_matches("r#")))
71                    .collect::<Vec<_>>();
72
73                format_to!(buf, "{}", segments.into_iter().rev().format("/"));
74
75                // We need to special case mod named `r#mod` and place the file in a
76                // subdirectory as "mod.rs" would be of its parent module otherwise.
77                if module_name.text() == "r#mod" {
78                    format_to!(buf, "/mod.rs");
79                } else {
80                    format_to!(buf, ".rs");
81                }
82                buf
83            };
84            let contents = {
85                let items = module_items.dedent(IndentLevel(1)).to_string();
86                let mut items =
87                    items.trim_start_matches('{').trim_end_matches('}').trim().to_owned();
88                if !items.is_empty() {
89                    items.push('\n');
90                }
91                items
92            };
93
94            let buf = format!("mod {module_name};");
95
96            let replacement_start = match module_ast.mod_token() {
97                Some(mod_token) => mod_token.text_range(),
98                None => module_ast.syntax().text_range(),
99            }
100            .start();
101
102            builder.replace(
103                TextRange::new(replacement_start, module_ast.syntax().text_range().end()),
104                buf,
105            );
106
107            let dst = AnchoredPathBuf { anchor: ctx.vfs_file_id(), path };
108            builder.create_file(dst, contents);
109        },
110    )
111}
