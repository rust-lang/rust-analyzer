13// Assist: replace_let_with_if_let
14//
15// Replaces `let` with an `if let`.
16//
17// ```
18// # enum Option<T> { Some(T), None }
19//
20// fn main(action: Action) {
21//     $0let x = compute();
22// }
23//
24// fn compute() -> Option<i32> { None }
25// ```
26// ->
27// ```
28// # enum Option<T> { Some(T), None }
29//
30// fn main(action: Action) {
31//     if let Some(x) = compute() {
32//     }
33// }
34//
35// fn compute() -> Option<i32> { None }
36// ```
37pub(crate) fn replace_let_with_if_let(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
38    let let_kw = ctx.find_token_syntax_at_offset(T![let])?;
39    let let_stmt = let_kw.parent().and_then(ast::LetStmt::cast)?;
40    let init = let_stmt.initializer()?;
41    let original_pat = let_stmt.pat()?;
42
43    let target = let_kw.text_range();
44    acc.add(
45        AssistId::refactor_rewrite("replace_let_with_if_let"),
46        "Replace let with if let",
47        target,
48        |builder| {
49            let mut editor = builder.make_editor(let_stmt.syntax());
50            let make = SyntaxFactory::with_mappings();
51            let ty = ctx.sema.type_of_expr(&init);
52            let pat = if let_stmt.let_else().is_some() {
53                // Do not add the wrapper type that implements `Try`,
54                // since the statement already wraps the pattern.
55                original_pat
56            } else {
57                let happy_variant = ty
58                    .and_then(|ty| TryEnum::from_ty(&ctx.sema, &ty.adjusted()))
59                    .map(|it| it.happy_case());
60                match happy_variant {
61                    None => original_pat,
62                    Some(var_name) => {
63                        make.tuple_struct_pat(make.ident_path(var_name), [original_pat]).into()
64                    }
65                }
66            };
67            let init_expr =
68                if let_expr_needs_paren(&init) { make.expr_paren(init).into() } else { init };
69
70            let block = make.block_expr([], None);
71            let block = block.indent(IndentLevel::from_node(let_stmt.syntax()));
72            let if_expr = make.expr_if(
73                make.expr_let(pat, init_expr).into(),
74                block,
75                let_stmt
76                    .let_else()
77                    .and_then(|let_else| let_else.block_expr().map(ast::ElseBranch::from)),
78            );
79            let if_stmt = make.expr_stmt(if_expr.into());
80
81            editor.replace(let_stmt.syntax(), if_stmt.syntax());
82            editor.add_mappings(make.finish_with_mappings());
83            builder.add_file_edits(ctx.vfs_file_id(), editor);
84        },
85    )
86}
87
88fn let_expr_needs_paren(expr: &ast::Expr) -> bool {
89    let fake_expr_let =
90        ast::make::expr_let(ast::make::tuple_pat(None).into(), ast::make::ext::expr_unit());
91    let Some(fake_expr) = fake_expr_let.expr() else {
92        stdx::never!();
93        return false;
94    };
95    expr.needs_parens_in_place_of(fake_expr_let.syntax(), fake_expr.syntax())
96}
