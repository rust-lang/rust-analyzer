14// Assist: move_guard_to_arm_body
15//
16// Moves match guard into match arm body.
17//
18// ```
19// enum Action { Move { distance: u32 }, Stop }
20//
21// fn handle(action: Action) {
22//     match action {
23//         Action::Move { distance } $0if distance > 10 => foo(),
24//         _ => (),
25//     }
26// }
27// ```
28// ->
29// ```
30// enum Action { Move { distance: u32 }, Stop }
31//
32// fn handle(action: Action) {
33//     match action {
34//         Action::Move { distance } => if distance > 10 {
35//             foo()
36//         },
37//         _ => (),
38//     }
39// }
40// ```
41pub(crate) fn move_guard_to_arm_body(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
42    let match_arm = ctx.find_node_at_offset::<MatchArm>()?;
43    let guard = match_arm.guard()?;
44    if ctx.offset() > guard.syntax().text_range().end() {
45        cov_mark::hit!(move_guard_inapplicable_in_arm_body);
46        return None;
47    }
48    let rest_arms = rest_arms(&match_arm, ctx.selection_trimmed())?;
49    let space_before_delete = chain(
50        guard.syntax().prev_sibling_or_token(),
51        rest_arms.iter().filter_map(|it| it.syntax().prev_sibling_or_token()),
52    );
53    let space_after_arrow = match_arm.fat_arrow_token()?.next_sibling_or_token();
54
55    let arm_expr = match_arm.expr()?;
56    let if_branch = chain([&match_arm], &rest_arms)
57        .rfold(None, |else_branch, arm| {
58            if let Some(guard) = arm.guard() {
59                let then_branch = crate::utils::wrap_block(&arm.expr()?);
60                let guard_condition = guard.condition()?.reset_indent();
61                Some(make::expr_if(guard_condition, then_branch, else_branch).into())
62            } else {
63                arm.expr().map(|it| crate::utils::wrap_block(&it).into())
64            }
65        })?
66        .indent(arm_expr.indent_level());
67    let ElseBranch::IfExpr(if_expr) = if_branch else { return None };
68
69    let target = guard.syntax().text_range();
70    acc.add(
71        AssistId::refactor_rewrite("move_guard_to_arm_body"),
72        "Move guard to arm body",
73        target,
74        |builder| {
75            let mut edit = builder.make_editor(match_arm.syntax());
76            for element in space_before_delete {
77                if element.kind() == WHITESPACE {
78                    edit.delete(element);
79                }
80            }
81            for rest_arm in &rest_arms {
82                edit.delete(rest_arm.syntax());
83            }
84            if let Some(element) = space_after_arrow
85                && element.kind() == WHITESPACE
86            {
87                edit.replace(element, make::tokens::single_space());
88            }
89
90            edit.delete(guard.syntax());
91            edit.replace(arm_expr.syntax(), if_expr.syntax());
92            builder.add_file_edits(ctx.vfs_file_id(), edit);
93        },
94    )
95}

97// Assist: move_arm_cond_to_match_guard
98//
99// Moves if expression from match arm body into a guard.
100//
101// ```
102// enum Action { Move { distance: u32 }, Stop }
103//
104// fn handle(action: Action) {
105//     match action {
106//         Action::Move { distance } => $0if distance > 10 { foo() },
107//         _ => (),
108//     }
109// }
110// ```
111// ->
112// ```
113// enum Action { Move { distance: u32 }, Stop }
114//
115// fn handle(action: Action) {
116//     match action {
117//         Action::Move { distance } if distance > 10 => foo(),
118//         _ => (),
119//     }
120// }
121// ```
122pub(crate) fn move_arm_cond_to_match_guard(
123    acc: &mut Assists,
124    ctx: &AssistContext<'_>,
125) -> Option<()> {
126    let match_arm: MatchArm = ctx.find_node_at_offset::<MatchArm>()?;
127    let match_pat = match_arm.pat()?;
128    let arm_body = match_arm.expr()?;
129    let arm_guard = match_arm.guard().and_then(|it| it.condition());
130
131    let mut replace_node = None;
132    let if_expr: IfExpr = IfExpr::cast(arm_body.syntax().clone()).or_else(|| {
133        let block_expr = BlockExpr::cast(arm_body.syntax().clone())?;
134        if block_expr.statements().next().is_some() {
135            cov_mark::hit!(move_guard_non_naked_if);
136            return None;
137        }
138        if let Expr::IfExpr(e) = block_expr.tail_expr()? {
139            replace_node = Some(block_expr.syntax().clone());
140            Some(e)
141        } else {
142            None
143        }
144    })?;
145    if ctx.offset() > if_expr.then_branch()?.syntax().text_range().start() {
146        return None;
147    }
148
149    let replace_node = replace_node.unwrap_or_else(|| if_expr.syntax().clone());
150    let needs_dedent = replace_node != *if_expr.syntax();
151    let (conds_blocks, tail) = parse_if_chain(if_expr)?;
152
153    acc.add(
154        AssistId::refactor_rewrite("move_arm_cond_to_match_guard"),
155        "Move condition to match guard",
156        replace_node.text_range(),
157        |builder| {
158            let make = SyntaxFactory::without_mappings();
159            let mut replace_arms = vec![];
160
161            // Dedent if if_expr is in a BlockExpr
162            let dedent = if needs_dedent {
163                cov_mark::hit!(move_guard_ifelse_in_block);
164                1
165            } else {
166                cov_mark::hit!(move_guard_ifelse_else_block);
167                0
168            };
169            let indent_level = match_arm.indent_level();
170            let make_guard = |cond: Option<Expr>| {
171                let condition = match (arm_guard.clone(), cond) {
172                    (None, None) => return None,
173                    (None, Some(it)) | (Some(it), None) => it,
174                    (Some(lhs), Some(rhs)) => {
175                        let op_expr = |expr: Expr| {
176                            if expr.precedence().needs_parentheses_in(ExprPrecedence::LAnd) {
177                                make.expr_paren(expr).into()
178                            } else {
179                                expr
180                            }
181                        };
182                        let op = syntax::ast::BinaryOp::LogicOp(syntax::ast::LogicOp::And);
183                        let expr_bin = make.expr_bin(op_expr(lhs), op, op_expr(rhs));
184                        expr_bin.into()
185                    }
186                };
187                Some(make.match_guard(condition))
188            };
189
190            for (cond, block) in conds_blocks {
191                let only_expr = block.statements().next().is_none();
192                let expr = match block.tail_expr() {
193                    Some(then_expr) if only_expr => then_expr,
194                    _ => block.dedent(dedent.into()).into(),
195                };
196                let new_arm = make.match_arm(match_pat.clone(), make_guard(Some(cond)), expr);
197                replace_arms.push(new_arm);
198            }
199            if let Some(block) = tail {
200                cov_mark::hit!(move_guard_ifelse_else_tail);
201                let only_expr = block.statements().next().is_none();
202                let expr = match block.tail_expr() {
203                    Some(expr) if only_expr => {
204                        cov_mark::hit!(move_guard_ifelse_expr_only);
205                        expr
206                    }
207                    _ => block.dedent(dedent.into()).into(),
208                };
209                let new_arm = make.match_arm(match_pat, make_guard(None), expr);
210                replace_arms.push(new_arm);
211            } else {
212                // There's no else branch. Add a pattern without guard, unless the following match
213                // arm is `_ => ...`
214                cov_mark::hit!(move_guard_ifelse_notail);
215                match match_arm.syntax().next_sibling().and_then(MatchArm::cast) {
216                    Some(next_arm)
217                        if matches!(next_arm.pat(), Some(Pat::WildcardPat(_)))
218                            && next_arm.guard().is_none() =>
219                    {
220                        cov_mark::hit!(move_guard_ifelse_has_wildcard);
221                    }
222                    _ => {
223                        let block_expr = make.expr_empty_block().into();
224                        replace_arms.push(make.match_arm(match_pat, make_guard(None), block_expr));
225                    }
226                }
227            }
228
229            let mut edit = builder.make_editor(match_arm.syntax());
230
231            let newline = make.whitespace(&format!("\n{indent_level}"));
232            let replace_arms = replace_arms.iter().map(|it| it.syntax().syntax_element());
233            let replace_arms = Itertools::intersperse(replace_arms, newline.syntax_element());
234            edit.replace_with_many(match_arm.syntax(), replace_arms.collect());
235
236            builder.add_file_edits(ctx.vfs_file_id(), edit);
237        },
238    )
239}

241fn rest_arms(match_arm: &MatchArm, selection: TextRange) -> Option<Vec<MatchArm>> {
242    match_arm
243        .parent_match()
244        .match_arm_list()?
245        .arms()
246        .skip_while(|it| it != match_arm)
247        .skip(1)
248        .take_while(move |it| {
249            selection.is_empty() || crate::utils::is_selected(it, selection, false)
250        })
251        .take_while(move |it| {
252            it.pat()
253                .zip(match_arm.pat())
254                .is_some_and(|(a, b)| a.syntax().text() == b.syntax().text())
255        })
256        .collect::<Vec<_>>()
257        .into()
258}

260// Parses an if-else-if chain to get the conditions and the then branches until we encounter an else
261// branch or the end.
262fn parse_if_chain(if_expr: IfExpr) -> Option<(Vec<(Expr, BlockExpr)>, Option<BlockExpr>)> {
263    let mut conds_blocks = Vec::new();
264    let mut curr_if = if_expr;
265    let tail = loop {
266        let cond = curr_if.condition()?;
267        conds_blocks.push((cond, curr_if.then_branch()?));
268        match curr_if.else_branch() {
269            Some(ElseBranch::IfExpr(e)) => {
270                curr_if = e;
271            }
272            Some(ElseBranch::Block(b)) => {
273                break Some(b);
274            }
275            None => break None,
276        }
277    };
278    Some((conds_blocks, tail))
279}
