12// Assist: convert_named_struct_to_tuple_struct
13//
14// Converts struct with named fields to tuple struct, and analogously for enum variants with named
15// fields.
16//
17// ```
18// struct Point$0 { x: f32, y: f32 }
19//
20// impl Point {
21//     pub fn new(x: f32, y: f32) -> Self {
22//         Point { x, y }
23//     }
24//
25//     pub fn x(&self) -> f32 {
26//         self.x
27//     }
28//
29//     pub fn y(&self) -> f32 {
30//         self.y
31//     }
32// }
33// ```
34// ->
35// ```
36// struct Point(f32, f32);
37//
38// impl Point {
39//     pub fn new(x: f32, y: f32) -> Self {
40//         Point(x, y)
41//     }
42//
43//     pub fn x(&self) -> f32 {
44//         self.0
45//     }
46//
47//     pub fn y(&self) -> f32 {
48//         self.1
49//     }
50// }
51// ```
52pub(crate) fn convert_named_struct_to_tuple_struct(
53    acc: &mut Assists,
54    ctx: &AssistContext<'_>,
55) -> Option<()> {
56    // XXX: We don't currently provide this assist for struct definitions inside macros, but if we
57    // are to lift this limitation, don't forget to make `edit_struct_def()` consider macro files
58    // too.
59    let strukt_or_variant = ctx
60        .find_node_at_offset::<ast::Struct>()
61        .map(Either::Left)
62        .or_else(|| ctx.find_node_at_offset::<ast::Variant>().map(Either::Right))?;
63    let field_list = strukt_or_variant.as_ref().either(|s| s.field_list(), |v| v.field_list())?;
64
65    if ctx.offset() > field_list.syntax().text_range().start() {
66        // Assist could be distracting after the braces
67        return None;
68    }
69
70    let record_fields = match field_list {
71        ast::FieldList::RecordFieldList(it) => it,
72        ast::FieldList::TupleFieldList(_) => return None,
73    };
74    let strukt_def = match &strukt_or_variant {
75        Either::Left(s) => Either::Left(ctx.sema.to_def(s)?),
76        Either::Right(v) => Either::Right(ctx.sema.to_def(v)?),
77    };
78
79    acc.add(
80        AssistId::refactor_rewrite("convert_named_struct_to_tuple_struct"),
81        "Convert to tuple struct",
82        strukt_or_variant.syntax().text_range(),
83        |edit| {
84            edit_field_references(ctx, edit, record_fields.fields());
85            edit_struct_references(ctx, edit, strukt_def);
86            edit_struct_def(ctx, edit, &strukt_or_variant, record_fields);
87        },
88    )
89}

91fn edit_struct_def(
92    ctx: &AssistContext<'_>,
93    edit: &mut SourceChangeBuilder,
94    strukt: &Either<ast::Struct, ast::Variant>,
95    record_fields: ast::RecordFieldList,
96) {
97    // Note that we don't need to consider macro files in this function because this is
98    // currently not triggered for struct definitions inside macro calls.
99    let tuple_fields = record_fields.fields().filter_map(|f| {
100        let field = ast::make::tuple_field(f.visibility(), f.ty()?).clone_for_update();
101        ted::insert_all(
102            ted::Position::first_child_of(field.syntax()),
103            f.attrs().map(|attr| attr.syntax().clone_subtree().clone_for_update().into()).collect(),
104        );
105        Some(field)
106    });
107    let tuple_fields = ast::make::tuple_field_list(tuple_fields);
108    let record_fields_text_range = record_fields.syntax().text_range();
109
110    edit.edit_file(ctx.vfs_file_id());
111    edit.replace(record_fields_text_range, tuple_fields.syntax().text());
112
113    if let Either::Left(strukt) = strukt {
114        if let Some(w) = strukt.where_clause() {
115            let mut where_clause = w.to_string();
116            if where_clause.ends_with(',') {
117                where_clause.pop();
118            }
119            where_clause.push(';');
120
121            edit.delete(w.syntax().text_range());
122            edit.insert(record_fields_text_range.end(), ast::make::tokens::single_newline().text());
123            edit.insert(record_fields_text_range.end(), where_clause);
124            edit.insert(record_fields_text_range.end(), ast::make::tokens::single_newline().text());
125
126            if let Some(tok) = strukt
127                .generic_param_list()
128                .and_then(|l| l.r_angle_token())
129                .and_then(|tok| tok.next_token())
130                .filter(|tok| tok.kind() == SyntaxKind::WHITESPACE)
131            {
132                edit.delete(tok.text_range());
133            }
134        } else {
135            edit.insert(record_fields_text_range.end(), ";");
136        }
137    }
138
139    if let Some(tok) = record_fields
140        .l_curly_token()
141        .and_then(|tok| tok.prev_token())
142        .filter(|tok| tok.kind() == SyntaxKind::WHITESPACE)
143    {
144        edit.delete(tok.text_range())
145    }
146}

148fn edit_struct_references(
149    ctx: &AssistContext<'_>,
150    edit: &mut SourceChangeBuilder,
151    strukt: Either<hir::Struct, hir::Variant>,
152) {
153    let strukt_def = match strukt {
154        Either::Left(s) => Definition::Adt(hir::Adt::Struct(s)),
155        Either::Right(v) => Definition::Variant(v),
156    };
157    let usages = strukt_def.usages(&ctx.sema).include_self_refs().all();
158
159    for (file_id, refs) in usages {
160        edit.edit_file(file_id.file_id(ctx.db()));
161        for r in refs {
162            process_struct_name_reference(ctx, r, edit);
163        }
164    }
165}

167fn process_struct_name_reference(
168    ctx: &AssistContext<'_>,
169    r: FileReference,
170    edit: &mut SourceChangeBuilder,
171) -> Option<()> {
172    // First check if it's the last semgnet of a path that directly belongs to a record
173    // expression/pattern.
174    let name_ref = r.name.as_name_ref()?;
175    let path_segment = name_ref.syntax().parent().and_then(ast::PathSegment::cast)?;
176    // A `PathSegment` always belongs to a `Path`, so there's at least one `Path` at this point.
177    let full_path =
178        path_segment.syntax().parent()?.ancestors().map_while(ast::Path::cast).last()?;
179
180    if full_path.segment()?.name_ref()? != *name_ref {
181        // `name_ref` isn't the last segment of the path, so `full_path` doesn't point to the
182        // struct we want to edit.
183        return None;
184    }
185
186    let parent = full_path.syntax().parent()?;
187    match_ast! {
188        match parent {
189            ast::RecordPat(record_struct_pat) => {
190                // When we failed to get the original range for the whole struct expression node,
191                // we can't provide any reasonable edit. Leave it untouched.
192                let file_range = ctx.sema.original_range_opt(record_struct_pat.syntax())?;
193                edit.replace(
194                    file_range.range,
195                    ast::make::tuple_struct_pat(
196                        record_struct_pat.path()?,
197                        record_struct_pat
198                            .record_pat_field_list()?
199                            .fields()
200                            .filter_map(|pat| pat.pat())
201                    )
202                    .to_string()
203                );
204            },
205            ast::RecordExpr(record_expr) => {
206                // When we failed to get the original range for the whole struct pattern node,
207                // we can't provide any reasonable edit. Leave it untouched.
208                let file_range = ctx.sema.original_range_opt(record_expr.syntax())?;
209                let path = record_expr.path()?;
210                let args = record_expr
211                    .record_expr_field_list()?
212                    .fields()
213                    .filter_map(|f| f.expr())
214                    .join(", ");
215
216                edit.replace(file_range.range, format!("{path}({args})"));
217            },
218            _ => {}
219        }
220    }
221
222    Some(())
223}

225fn edit_field_references(
226    ctx: &AssistContext<'_>,
227    edit: &mut SourceChangeBuilder,
228    fields: impl Iterator<Item = ast::RecordField>,
229) {
230    for (index, field) in fields.enumerate() {
231        let field = match ctx.sema.to_def(&field) {
232            Some(it) => it,
233            None => continue,
234        };
235        let def = Definition::Field(field);
236        let usages = def.usages(&ctx.sema).all();
237        for (file_id, refs) in usages {
238            edit.edit_file(file_id.file_id(ctx.db()));
239            for r in refs {
240                if let Some(name_ref) = r.name.as_name_ref() {
241                    // Only edit the field reference if it's part of a `.field` access
242                    if name_ref.syntax().parent().and_then(ast::FieldExpr::cast).is_some() {
243                        edit.replace(r.range, index.to_string());
244                    }
245                }
246            }
247        }
248    }
249}
