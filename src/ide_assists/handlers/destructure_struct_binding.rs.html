23// Assist: destructure_struct_binding
24//
25// Destructures a struct binding in place.
26//
27// ```
28// struct Foo {
29//     bar: i32,
30//     baz: i32,
31// }
32// fn main() {
33//     let $0foo = Foo { bar: 1, baz: 2 };
34//     let bar2 = foo.bar;
35//     let baz2 = &foo.baz;
36// }
37// ```
38// ->
39// ```
40// struct Foo {
41//     bar: i32,
42//     baz: i32,
43// }
44// fn main() {
45//     let Foo { bar, baz } = Foo { bar: 1, baz: 2 };
46//     let bar2 = bar;
47//     let baz2 = &baz;
48// }
49// ```
50pub(crate) fn destructure_struct_binding(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
51    let target = ctx.find_node_at_offset::<Target>()?;
52    let data = collect_data(target, ctx)?;
53
54    acc.add(
55        AssistId::refactor_rewrite("destructure_struct_binding"),
56        "Destructure struct binding",
57        data.target.syntax().text_range(),
58        |edit| destructure_struct_binding_impl(ctx, edit, &data),
59    );
60
61    Some(())
62}
63
64enum Target {
65    IdentPat(ast::IdentPat),
66    SelfParam { param: ast::SelfParam, insert_after: SyntaxToken },
67}
68
69impl Target {
70    fn ty<'db>(&self, sema: &Semantics<'db, RootDatabase>) -> Option<hir::Type<'db>> {
71        match self {
72            Target::IdentPat(pat) => sema.type_of_binding_in_pat(pat),
73            Target::SelfParam { param, .. } => sema.type_of_self(param),
74        }
75    }
76
77    fn is_ref(&self) -> bool {
78        match self {
79            Target::IdentPat(ident_pat) => ident_pat.ref_token().is_some(),
80            Target::SelfParam { .. } => false,
81        }
82    }
83
84    fn is_mut(&self) -> bool {
85        match self {
86            Target::IdentPat(ident_pat) => ident_pat.mut_token().is_some(),
87            Target::SelfParam { param, .. } => {
88                param.mut_token().is_some() && param.amp_token().is_none()
89            }
90        }
91    }
92}
93
94impl HasName for Target {}
95
96impl AstNode for Target {
97    fn cast(node: SyntaxNode) -> Option<Self> {
98        if ast::IdentPat::can_cast(node.kind()) {
99            ast::IdentPat::cast(node).map(Self::IdentPat)
100        } else {
101            let param = ast::SelfParam::cast(node)?;
102            let param_list = param.syntax().parent().and_then(ast::ParamList::cast)?;
103            let block = param_list.syntax().parent()?.children().find_map(ast::BlockExpr::cast)?;
104            let insert_after = block.stmt_list()?.l_curly_token()?;
105            Some(Self::SelfParam { param, insert_after })
106        }
107    }
108
109    fn can_cast(kind: syntax::SyntaxKind) -> bool {
110        ast::IdentPat::can_cast(kind) || ast::SelfParam::can_cast(kind)
111    }
112
113    fn syntax(&self) -> &SyntaxNode {
114        match self {
115            Target::IdentPat(ident_pat) => ident_pat.syntax(),
116            Target::SelfParam { param, .. } => param.syntax(),
117        }
118    }
119}

121fn destructure_struct_binding_impl(
122    ctx: &AssistContext<'_>,
123    builder: &mut SourceChangeBuilder,
124    data: &StructEditData,
125) {
126    let field_names = generate_field_names(ctx, data);
127    let mut editor = builder.make_editor(data.target.syntax());
128    destructure_pat(ctx, &mut editor, data, &field_names);
129    update_usages(ctx, &mut editor, data, &field_names.into_iter().collect());
130    builder.add_file_edits(ctx.vfs_file_id(), editor);
131}

133struct StructEditData {
134    target: Target,
135    name: ast::Name,
136    kind: hir::StructKind,
137    struct_def_path: hir::ModPath,
138    visible_fields: Vec<hir::Field>,
139    usages: Vec<FileReference>,
140    names_in_scope: FxHashSet<SmolStr>,
141    has_private_members: bool,
142    need_record_field_name: bool,
143    is_ref: bool,
144    edition: Edition,
145}
146
147impl StructEditData {
148    fn apply_to_destruct(
149        &self,
150        new_pat: ast::Pat,
151        editor: &mut SyntaxEditor,
152        make: &SyntaxFactory,
153    ) {
154        match &self.target {
155            Target::IdentPat(pat) => {
156                // If the binding is nested inside a record, we need to wrap the new
157                // destructured pattern in a non-shorthand record field
158                if self.need_record_field_name {
159                    let new_pat =
160                        make.record_pat_field(make.name_ref(&self.name.to_string()), new_pat);
161                    editor.replace(pat.syntax(), new_pat.syntax())
162                } else {
163                    editor.replace(pat.syntax(), new_pat.syntax())
164                }
165            }
166            Target::SelfParam { insert_after, .. } => {
167                let indent = IndentLevel::from_token(insert_after) + 1;
168                let newline = make.whitespace(&format!("\n{indent}"));
169                let initializer = make.expr_path(make.ident_path("self"));
170                let let_stmt = make.let_stmt(new_pat, None, Some(initializer));
171                editor.insert_all(
172                    Position::after(insert_after),
173                    vec![newline.into(), let_stmt.syntax().clone().into()],
174                );
175            }
176        }
177    }
178}

180fn collect_data(target: Target, ctx: &AssistContext<'_>) -> Option<StructEditData> {
181    let ty = target.ty(&ctx.sema)?;
182    let hir::Adt::Struct(struct_type) = ty.strip_references().as_adt()? else { return None };
183
184    let module = ctx.sema.scope(target.syntax())?.module();
185    let cfg = ctx.config.find_path_config(ctx.sema.is_nightly(module.krate(ctx.db())));
186    let struct_def = hir::ModuleDef::from(struct_type);
187    let kind = struct_type.kind(ctx.db());
188    let struct_def_path = module.find_path(ctx.db(), struct_def, cfg)?;
189
190    let is_non_exhaustive = struct_def.attrs(ctx.db())?.is_non_exhaustive();
191    let is_foreign_crate =
192        struct_def.module(ctx.db()).is_some_and(|m| m.krate(ctx.db()) != module.krate(ctx.db()));
193
194    let fields = struct_type.fields(ctx.db());
195    let n_fields = fields.len();
196
197    let visible_fields =
198        fields.into_iter().filter(|field| field.is_visible_from(ctx.db(), module)).collect_vec();
199
200    if visible_fields.is_empty() {
201        return None;
202    }
203
204    let has_private_members =
205        (is_non_exhaustive && is_foreign_crate) || visible_fields.len() < n_fields;
206
207    // If private members are present, we can only destructure records
208    if !matches!(kind, hir::StructKind::Record) && has_private_members {
209        return None;
210    }
211
212    let is_ref = ty.is_reference();
213    let need_record_field_name = target
214        .syntax()
215        .parent()
216        .and_then(ast::RecordPatField::cast)
217        .is_some_and(|field| field.colon_token().is_none());
218
219    let def = match &target {
220        Target::IdentPat(pat) => ctx.sema.to_def(pat),
221        Target::SelfParam { param, .. } => ctx.sema.to_def(param),
222    };
223    let usages = def
224        .and_then(|def| {
225            Definition::Local(def)
226                .usages(&ctx.sema)
227                .in_scope(&SearchScope::single_file(ctx.file_id()))
228                .all()
229                .iter()
230                .next()
231                .map(|(_, refs)| refs.to_vec())
232        })
233        .unwrap_or_default();
234
235    let names_in_scope = get_names_in_scope(ctx, &target, &usages).unwrap_or_default();
236
237    Some(StructEditData {
238        name: target.name()?,
239        target,
240        kind,
241        struct_def_path,
242        usages,
243        has_private_members,
244        visible_fields,
245        names_in_scope,
246        need_record_field_name,
247        is_ref,
248        edition: module.krate(ctx.db()).edition(ctx.db()),
249    })
250}
251
252fn get_names_in_scope(
253    ctx: &AssistContext<'_>,
254    target: &Target,
255    usages: &[FileReference],
256) -> Option<FxHashSet<SmolStr>> {
257    fn last_usage(usages: &[FileReference]) -> Option<SyntaxNode> {
258        usages.last()?.name.syntax().into_node()
259    }
260
261    // If available, find names visible to the last usage of the binding
262    // else, find names visible to the binding itself
263    let last_usage = last_usage(usages);
264    let node = last_usage.as_ref().unwrap_or(target.syntax());
265    let scope = ctx.sema.scope(node)?;
266
267    let mut names = FxHashSet::default();
268    scope.process_all_names(&mut |name, scope| {
269        if let hir::ScopeDef::Local(_) = scope {
270            names.insert(name.as_str().into());
271        }
272    });
273    Some(names)
274}

276fn destructure_pat(
277    _ctx: &AssistContext<'_>,
278    editor: &mut SyntaxEditor,
279    data: &StructEditData,
280    field_names: &[(SmolStr, SmolStr)],
281) {
282    let struct_path = mod_path_to_ast(&data.struct_def_path, data.edition);
283    let is_ref = data.target.is_ref();
284    let is_mut = data.target.is_mut();
285
286    let make = SyntaxFactory::with_mappings();
287    let new_pat = match data.kind {
288        hir::StructKind::Tuple => {
289            let ident_pats = field_names.iter().map(|(_, new_name)| {
290                let name = make.name(new_name);
291                ast::Pat::from(make.ident_pat(is_ref, is_mut, name))
292            });
293            ast::Pat::TupleStructPat(make.tuple_struct_pat(struct_path, ident_pats))
294        }
295        hir::StructKind::Record => {
296            let fields = field_names.iter().map(|(old_name, new_name)| {
297                // Use shorthand syntax if possible
298                if old_name == new_name {
299                    make.record_pat_field_shorthand(
300                        make.ident_pat(is_ref, is_mut, make.name(old_name)).into(),
301                    )
302                } else {
303                    make.record_pat_field(
304                        make.name_ref(old_name),
305                        ast::Pat::IdentPat(make.ident_pat(is_ref, is_mut, make.name(new_name))),
306                    )
307                }
308            });
309            let field_list = make
310                .record_pat_field_list(fields, data.has_private_members.then_some(make.rest_pat()));
311
312            ast::Pat::RecordPat(make.record_pat_with_fields(struct_path, field_list))
313        }
314        hir::StructKind::Unit => make.path_pat(struct_path),
315    };
316
317    data.apply_to_destruct(new_pat, editor, &make);
318    editor.add_mappings(make.finish_with_mappings());
319}

321fn generate_field_names(ctx: &AssistContext<'_>, data: &StructEditData) -> Vec<(SmolStr, SmolStr)> {
322    match data.kind {
323        hir::StructKind::Tuple => data
324            .visible_fields
325            .iter()
326            .enumerate()
327            .map(|(index, _)| {
328                let new_name = new_field_name((format!("_{index}")).into(), &data.names_in_scope);
329                (index.to_string().into(), new_name)
330            })
331            .collect(),
332        hir::StructKind::Record => data
333            .visible_fields
334            .iter()
335            .map(|field| {
336                let field_name = field.name(ctx.db()).display_no_db(data.edition).to_smolstr();
337                let new_name = new_field_name(field_name.clone(), &data.names_in_scope);
338                (field_name, new_name)
339            })
340            .collect(),
341        hir::StructKind::Unit => Vec::new(),
342    }
343}
344
345fn new_field_name(base_name: SmolStr, names_in_scope: &FxHashSet<SmolStr>) -> SmolStr {
346    let mut name = base_name.clone();
347    let mut i = 1;
348    while names_in_scope.contains(&name) {
349        name = format!("{base_name}_{i}").into();
350        i += 1;
351    }
352    name
353}

355fn update_usages(
356    ctx: &AssistContext<'_>,
357    editor: &mut SyntaxEditor,
358    data: &StructEditData,
359    field_names: &FxHashMap<SmolStr, SmolStr>,
360) {
361    let make = SyntaxFactory::with_mappings();
362    let edits = data
363        .usages
364        .iter()
365        .filter_map(|r| build_usage_edit(ctx, &make, data, r, field_names))
366        .collect_vec();
367    editor.add_mappings(make.finish_with_mappings());
368    for (old, new) in edits {
369        editor.replace(old, new);
370    }
371}

373fn build_usage_edit(
374    ctx: &AssistContext<'_>,
375    make: &SyntaxFactory,
376    data: &StructEditData,
377    usage: &FileReference,
378    field_names: &FxHashMap<SmolStr, SmolStr>,
379) -> Option<(SyntaxNode, SyntaxNode)> {
380    match usage.name.syntax().ancestors().find_map(ast::FieldExpr::cast) {
381        Some(field_expr) => Some({
382            let field_name: SmolStr = field_expr.name_ref()?.to_string().into();
383            let new_field_name = field_names.get(&field_name)?;
384            let new_expr = ast::make::expr_path(ast::make::ext::ident_path(new_field_name));
385
386            // If struct binding is a reference, we might need to deref field usages
387            if data.is_ref {
388                let (replace_expr, ref_data) = determine_ref_and_parens(ctx, &field_expr);
389                (
390                    replace_expr.syntax().clone_for_update(),
391                    ref_data.wrap_expr(new_expr).syntax().clone_for_update(),
392                )
393            } else {
394                (field_expr.syntax().clone(), new_expr.syntax().clone_for_update())
395            }
396        }),
397        None => Some((
398            usage.name.syntax().as_node().unwrap().clone(),
399            make.expr_macro(
400                ast::make::ext::ident_path("todo"),
401                make.token_tree(syntax::SyntaxKind::L_PAREN, []),
402            )
403            .syntax()
404            .clone(),
405        )),
406    }
407}
