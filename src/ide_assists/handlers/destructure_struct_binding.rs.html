19// Assist: destructure_struct_binding
20//
21// Destructures a struct binding in place.
22//
23// ```
24// struct Foo {
25//     bar: i32,
26//     baz: i32,
27// }
28// fn main() {
29//     let $0foo = Foo { bar: 1, baz: 2 };
30//     let bar2 = foo.bar;
31//     let baz2 = &foo.baz;
32// }
33// ```
34// ->
35// ```
36// struct Foo {
37//     bar: i32,
38//     baz: i32,
39// }
40// fn main() {
41//     let Foo { bar, baz } = Foo { bar: 1, baz: 2 };
42//     let bar2 = bar;
43//     let baz2 = &baz;
44// }
45// ```
46pub(crate) fn destructure_struct_binding(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
47    let ident_pat = ctx.find_node_at_offset::<ast::IdentPat>()?;
48    let data = collect_data(ident_pat, ctx)?;
49
50    acc.add(
51        AssistId::refactor_rewrite("destructure_struct_binding"),
52        "Destructure struct binding",
53        data.ident_pat.syntax().text_range(),
54        |edit| destructure_struct_binding_impl(ctx, edit, &data),
55    );
56
57    Some(())
58}
59
60fn destructure_struct_binding_impl(
61    ctx: &AssistContext<'_>,
62    builder: &mut SourceChangeBuilder,
63    data: &StructEditData,
64) {
65    let field_names = generate_field_names(ctx, data);
66    let mut editor = builder.make_editor(data.ident_pat.syntax());
67    destructure_pat(ctx, &mut editor, data, &field_names);
68    update_usages(ctx, &mut editor, data, &field_names.into_iter().collect());
69    builder.add_file_edits(ctx.vfs_file_id(), editor);
70}

78    usages: Vec<FileReference>,

86fn collect_data(ident_pat: ast::IdentPat, ctx: &AssistContext<'_>) -> Option<StructEditData> {
87    let ty = ctx.sema.type_of_binding_in_pat(&ident_pat)?;
88    let hir::Adt::Struct(struct_type) = ty.strip_references().as_adt()? else { return None };
89
90    let module = ctx.sema.scope(ident_pat.syntax())?.module();
91    let cfg = ctx.config.find_path_config(ctx.sema.is_nightly(module.krate()));
92    let struct_def = hir::ModuleDef::from(struct_type);
93    let kind = struct_type.kind(ctx.db());
94    let struct_def_path = module.find_path(ctx.db(), struct_def, cfg)?;
95
96    let is_non_exhaustive = struct_def.attrs(ctx.db())?.by_key(sym::non_exhaustive).exists();
97    let is_foreign_crate = struct_def.module(ctx.db()).is_some_and(|m| m.krate() != module.krate());
98
99    let fields = struct_type.fields(ctx.db());
100    let n_fields = fields.len();
101
102    let visible_fields =
103        fields.into_iter().filter(|field| field.is_visible_from(ctx.db(), module)).collect_vec();
104
105    if visible_fields.is_empty() {
106        return None;
107    }
108
109    let has_private_members =
110        (is_non_exhaustive && is_foreign_crate) || visible_fields.len() < n_fields;
111
112    // If private members are present, we can only destructure records
113    if !matches!(kind, hir::StructKind::Record) && has_private_members {
114        return None;
115    }
116
117    let is_ref = ty.is_reference();
118    let need_record_field_name = ident_pat
119        .syntax()
120        .parent()
121        .and_then(ast::RecordPatField::cast)
122        .is_some_and(|field| field.colon_token().is_none());
123
124    let usages = ctx
125        .sema
126        .to_def(&ident_pat)
127        .and_then(|def| {
128            Definition::Local(def)
129                .usages(&ctx.sema)
130                .in_scope(&SearchScope::single_file(ctx.file_id()))
131                .all()
132                .iter()
133                .next()
134                .map(|(_, refs)| refs.to_vec())
135        })
136        .unwrap_or_default();
137
138    let names_in_scope = get_names_in_scope(ctx, &ident_pat, &usages).unwrap_or_default();
139
140    Some(StructEditData {
141        name: ident_pat.name()?,
142        ident_pat,
143        kind,
144        struct_def_path,
145        usages,
146        has_private_members,
147        visible_fields,
148        names_in_scope,
149        need_record_field_name,
150        is_ref,
151        edition: module.krate().edition(ctx.db()),
152    })
153}

155fn get_names_in_scope(
156    ctx: &AssistContext<'_>,
157    ident_pat: &ast::IdentPat,
158    usages: &[FileReference],
159) -> Option<FxHashSet<SmolStr>> {
160    fn last_usage(usages: &[FileReference]) -> Option<SyntaxNode> {
161        usages.last()?.name.syntax().into_node()
162    }
163
164    // If available, find names visible to the last usage of the binding
165    // else, find names visible to the binding itself
166    let last_usage = last_usage(usages);
167    let node = last_usage.as_ref().unwrap_or(ident_pat.syntax());
168    let scope = ctx.sema.scope(node)?;
169
170    let mut names = FxHashSet::default();
171    scope.process_all_names(&mut |name, scope| {
172        if let hir::ScopeDef::Local(_) = scope {
173            names.insert(name.as_str().into());
174        }
175    });
176    Some(names)
177}

179fn destructure_pat(
180    _ctx: &AssistContext<'_>,
181    editor: &mut SyntaxEditor,
182    data: &StructEditData,
183    field_names: &[(SmolStr, SmolStr)],
184) {
185    let ident_pat = &data.ident_pat;
186    let name = &data.name;
187
188    let struct_path = mod_path_to_ast(&data.struct_def_path, data.edition);
189    let is_ref = ident_pat.ref_token().is_some();
190    let is_mut = ident_pat.mut_token().is_some();
191
192    let make = SyntaxFactory::with_mappings();
193    let new_pat = match data.kind {
194        hir::StructKind::Tuple => {
195            let ident_pats = field_names.iter().map(|(_, new_name)| {
196                let name = make.name(new_name);
197                ast::Pat::from(make.ident_pat(is_ref, is_mut, name))
198            });
199            ast::Pat::TupleStructPat(make.tuple_struct_pat(struct_path, ident_pats))
200        }
201        hir::StructKind::Record => {
202            let fields = field_names.iter().map(|(old_name, new_name)| {
203                // Use shorthand syntax if possible
204                if old_name == new_name {
205                    make.record_pat_field_shorthand(
206                        make.ident_pat(is_ref, is_mut, make.name(old_name)).into(),
207                    )
208                } else {
209                    make.record_pat_field(
210                        make.name_ref(old_name),
211                        ast::Pat::IdentPat(make.ident_pat(is_ref, is_mut, make.name(new_name))),
212                    )
213                }
214            });
215            let field_list = make
216                .record_pat_field_list(fields, data.has_private_members.then_some(make.rest_pat()));
217
218            ast::Pat::RecordPat(make.record_pat_with_fields(struct_path, field_list))
219        }
220        hir::StructKind::Unit => make.path_pat(struct_path),
221    };
222
223    // If the binding is nested inside a record, we need to wrap the new
224    // destructured pattern in a non-shorthand record field
225    let destructured_pat = if data.need_record_field_name {
226        make.record_pat_field(make.name_ref(&name.to_string()), new_pat).syntax().clone()
227    } else {
228        new_pat.syntax().clone()
229    };
230
231    editor.add_mappings(make.finish_with_mappings());
232    editor.replace(data.ident_pat.syntax(), destructured_pat);
233}

235fn generate_field_names(ctx: &AssistContext<'_>, data: &StructEditData) -> Vec<(SmolStr, SmolStr)> {
236    match data.kind {
237        hir::StructKind::Tuple => data
238            .visible_fields
239            .iter()
240            .enumerate()
241            .map(|(index, _)| {
242                let new_name = new_field_name((format!("_{index}")).into(), &data.names_in_scope);
243                (index.to_string().into(), new_name)
244            })
245            .collect(),
246        hir::StructKind::Record => data
247            .visible_fields
248            .iter()
249            .map(|field| {
250                let field_name = field.name(ctx.db()).display_no_db(data.edition).to_smolstr();
251                let new_name = new_field_name(field_name.clone(), &data.names_in_scope);
252                (field_name, new_name)
253            })
254            .collect(),
255        hir::StructKind::Unit => Vec::new(),
256    }
257}
258
259fn new_field_name(base_name: SmolStr, names_in_scope: &FxHashSet<SmolStr>) -> SmolStr {
260    let mut name = base_name.clone();
261    let mut i = 1;
262    while names_in_scope.contains(&name) {
263        name = format!("{base_name}_{i}").into();
264        i += 1;
265    }
266    name
267}

269fn update_usages(
270    ctx: &AssistContext<'_>,
271    editor: &mut SyntaxEditor,
272    data: &StructEditData,
273    field_names: &FxHashMap<SmolStr, SmolStr>,
274) {
275    let make = SyntaxFactory::with_mappings();
276    let edits = data
277        .usages
278        .iter()
279        .filter_map(|r| build_usage_edit(ctx, &make, data, r, field_names))
280        .collect_vec();
281    editor.add_mappings(make.finish_with_mappings());
282    for (old, new) in edits {
283        editor.replace(old, new);
284    }
285}

287fn build_usage_edit(
288    ctx: &AssistContext<'_>,
289    make: &SyntaxFactory,
290    data: &StructEditData,
291    usage: &FileReference,
292    field_names: &FxHashMap<SmolStr, SmolStr>,
293) -> Option<(SyntaxNode, SyntaxNode)> {
294    match usage.name.syntax().ancestors().find_map(ast::FieldExpr::cast) {
295        Some(field_expr) => Some({
296            let field_name: SmolStr = field_expr.name_ref()?.to_string().into();
297            let new_field_name = field_names.get(&field_name)?;
298            let new_expr = ast::make::expr_path(ast::make::ext::ident_path(new_field_name));
299
300            // If struct binding is a reference, we might need to deref field usages
301            if data.is_ref {
302                let (replace_expr, ref_data) = determine_ref_and_parens(ctx, &field_expr);
303                (
304                    replace_expr.syntax().clone_for_update(),
305                    ref_data.wrap_expr(new_expr).syntax().clone_for_update(),
306                )
307            } else {
308                (field_expr.syntax().clone(), new_expr.syntax().clone_for_update())
309            }
310        }),
311        None => Some((
312            usage.name.syntax().as_node().unwrap().clone(),
313            make.expr_macro(
314                ast::make::ext::ident_path("todo"),
315                make.token_tree(syntax::SyntaxKind::L_PAREN, []),
316            )
317            .syntax()
318            .clone(),
319        )),
320    }
