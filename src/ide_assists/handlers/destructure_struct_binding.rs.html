18// Assist: destructure_struct_binding
19//
20// Destructures a struct binding in place.
21//
22// ```
23// struct Foo {
24//     bar: i32,
25//     baz: i32,
26// }
27// fn main() {
28//     let $0foo = Foo { bar: 1, baz: 2 };
29//     let bar2 = foo.bar;
30//     let baz2 = &foo.baz;
31// }
32// ```
33// ->
34// ```
35// struct Foo {
36//     bar: i32,
37//     baz: i32,
38// }
39// fn main() {
40//     let Foo { bar, baz } = Foo { bar: 1, baz: 2 };
41//     let bar2 = bar;
42//     let baz2 = &baz;
43// }
44// ```
45pub(crate) fn destructure_struct_binding(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
46    let ident_pat = ctx.find_node_at_offset::<ast::IdentPat>()?;
47    let data = collect_data(ident_pat, ctx)?;
48
49    acc.add(
50        AssistId::refactor_rewrite("destructure_struct_binding"),
51        "Destructure struct binding",
52        data.ident_pat.syntax().text_range(),
53        |edit| destructure_struct_binding_impl(ctx, edit, &data),
54    );
55
56    Some(())
57}
58
59fn destructure_struct_binding_impl(
60    ctx: &AssistContext<'_>,
61    builder: &mut SourceChangeBuilder,
62    data: &StructEditData,
63) {
64    let field_names = generate_field_names(ctx, data);
65    let assignment_edit = build_assignment_edit(ctx, builder, data, &field_names);
66    let usage_edits = build_usage_edits(ctx, builder, data, &field_names.into_iter().collect());
67
68    assignment_edit.apply();
69    for edit in usage_edits {
70        edit.apply(builder);
71    }
72}

75    ident_pat: ast::IdentPat,
76    kind: hir::StructKind,
77    struct_def_path: hir::ModPath,
78    visible_fields: Vec<hir::Field>,
79    usages: Vec<FileReference>,
80    names_in_scope: FxHashSet<SmolStr>,

87fn collect_data(ident_pat: ast::IdentPat, ctx: &AssistContext<'_>) -> Option<StructEditData> {
88    let ty = ctx.sema.type_of_binding_in_pat(&ident_pat)?;
89    let hir::Adt::Struct(struct_type) = ty.strip_references().as_adt()? else { return None };
90
91    let cfg = ctx.config.import_path_config();
92
93    let module = ctx.sema.scope(ident_pat.syntax())?.module();
94    let struct_def = hir::ModuleDef::from(struct_type);
95    let kind = struct_type.kind(ctx.db());
96    let struct_def_path = module.find_path(ctx.db(), struct_def, cfg)?;
97
98    let is_non_exhaustive = struct_def.attrs(ctx.db())?.by_key(&sym::non_exhaustive).exists();
99    let is_foreign_crate = struct_def.module(ctx.db()).is_some_and(|m| m.krate() != module.krate());
100
101    let fields = struct_type.fields(ctx.db());
102    let n_fields = fields.len();
103
104    let visible_fields =
105        fields.into_iter().filter(|field| field.is_visible_from(ctx.db(), module)).collect_vec();
106
107    if visible_fields.is_empty() {
108        return None;
109    }
110
111    let has_private_members =
112        (is_non_exhaustive && is_foreign_crate) || visible_fields.len() < n_fields;
113
114    // If private members are present, we can only destructure records
115    if !matches!(kind, hir::StructKind::Record) && has_private_members {
116        return None;
117    }
118
119    let is_ref = ty.is_reference();
120    let is_nested = ident_pat.syntax().parent().and_then(ast::RecordPatField::cast).is_some();
121
122    let usages = ctx
123        .sema
124        .to_def(&ident_pat)
125        .and_then(|def| {
126            Definition::Local(def)
127                .usages(&ctx.sema)
128                .in_scope(&SearchScope::single_file(ctx.file_id()))
129                .all()
130                .iter()
131                .next()
132                .map(|(_, refs)| refs.to_vec())
133        })
134        .unwrap_or_default();
135
136    let names_in_scope = get_names_in_scope(ctx, &ident_pat, &usages).unwrap_or_default();
137
138    Some(StructEditData {
139        ident_pat,
140        kind,
141        struct_def_path,
142        usages,
143        has_private_members,
144        visible_fields,
145        names_in_scope,
146        is_nested,
147        is_ref,
148        edition: module.krate().edition(ctx.db()),
149    })
150}

152fn get_names_in_scope(
153    ctx: &AssistContext<'_>,
154    ident_pat: &ast::IdentPat,
155    usages: &[FileReference],
156) -> Option<FxHashSet<SmolStr>> {
157    fn last_usage(usages: &[FileReference]) -> Option<SyntaxNode> {
158        usages.last()?.name.syntax().into_node()
159    }
160
161    // If available, find names visible to the last usage of the binding
162    // else, find names visible to the binding itself
163    let last_usage = last_usage(usages);
164    let node = last_usage.as_ref().unwrap_or(ident_pat.syntax());
165    let scope = ctx.sema.scope(node)?;
166
167    let mut names = FxHashSet::default();
168    scope.process_all_names(&mut |name, scope| {
169        if let hir::ScopeDef::Local(_) = scope {
170            names.insert(name.as_str().into());
171        }
172    });
173    Some(names)
174}

176fn build_assignment_edit(
177    _ctx: &AssistContext<'_>,
178    builder: &mut SourceChangeBuilder,
179    data: &StructEditData,
180    field_names: &[(SmolStr, SmolStr)],
181) -> AssignmentEdit {
182    let ident_pat = builder.make_mut(data.ident_pat.clone());
183
184    let struct_path = mod_path_to_ast(&data.struct_def_path, data.edition);
185    let is_ref = ident_pat.ref_token().is_some();
186    let is_mut = ident_pat.mut_token().is_some();
187
188    let new_pat = match data.kind {
189        hir::StructKind::Tuple => {
190            let ident_pats = field_names.iter().map(|(_, new_name)| {
191                let name = ast::make::name(new_name);
192                ast::Pat::from(ast::make::ident_pat(is_ref, is_mut, name))
193            });
194            ast::Pat::TupleStructPat(ast::make::tuple_struct_pat(struct_path, ident_pats))
195        }
196        hir::StructKind::Record => {
197            let fields = field_names.iter().map(|(old_name, new_name)| {
198                // Use shorthand syntax if possible
199                if old_name == new_name && !is_mut {
200                    ast::make::record_pat_field_shorthand(ast::make::name_ref(old_name))
201                } else {
202                    ast::make::record_pat_field(
203                        ast::make::name_ref(old_name),
204                        ast::Pat::IdentPat(ast::make::ident_pat(
205                            is_ref,
206                            is_mut,
207                            ast::make::name(new_name),
208                        )),
209                    )
210                }
211            });
212
213            let field_list = ast::make::record_pat_field_list(
214                fields,
215                data.has_private_members.then_some(ast::make::rest_pat()),
216            );
217            ast::Pat::RecordPat(ast::make::record_pat_with_fields(struct_path, field_list))
218        }
219        hir::StructKind::Unit => ast::make::path_pat(struct_path),
220    };
221
222    // If the binding is nested inside a record, we need to wrap the new
223    // destructured pattern in a non-shorthand record field
224    let new_pat = if data.is_nested {
225        let record_pat_field =
226            ast::make::record_pat_field(ast::make::name_ref(&ident_pat.to_string()), new_pat)
227                .clone_for_update();
228        NewPat::RecordPatField(record_pat_field)
229    } else {
230        NewPat::Pat(new_pat.clone_for_update())
231    };
232
233    AssignmentEdit { old_pat: ident_pat, new_pat }
234}

236fn generate_field_names(ctx: &AssistContext<'_>, data: &StructEditData) -> Vec<(SmolStr, SmolStr)> {
237    match data.kind {
238        hir::StructKind::Tuple => data
239            .visible_fields
240            .iter()
241            .enumerate()
242            .map(|(index, _)| {
243                let new_name = new_field_name((format!("_{index}")).into(), &data.names_in_scope);
244                (index.to_string().into(), new_name)
245            })
246            .collect(),
247        hir::StructKind::Record => data
248            .visible_fields
249            .iter()
250            .map(|field| {
251                let field_name = field.name(ctx.db()).display_no_db(data.edition).to_smolstr();
252                let new_name = new_field_name(field_name.clone(), &data.names_in_scope);
253                (field_name, new_name)
254            })
255            .collect(),
256        hir::StructKind::Unit => Vec::new(),
257    }
258}

260fn new_field_name(base_name: SmolStr, names_in_scope: &FxHashSet<SmolStr>) -> SmolStr {
261    let mut name = base_name.clone();
262    let mut i = 1;
263    while names_in_scope.contains(&name) {
264        name = format!("{base_name}_{i}").into();
265        i += 1;
266    }
267    name
268}

270struct AssignmentEdit {
271    old_pat: ast::IdentPat,
272    new_pat: NewPat,
273}
274
275enum NewPat {
276    Pat(ast::Pat),
277    RecordPatField(ast::RecordPatField),
278}
279
280impl AssignmentEdit {
281    fn apply(self) {
282        match self.new_pat {
283            NewPat::Pat(pat) => ted::replace(self.old_pat.syntax(), pat.syntax()),
284            NewPat::RecordPatField(record_pat_field) => {
285                ted::replace(self.old_pat.syntax(), record_pat_field.syntax())
286            }
287        }
288    }
289}

291fn build_usage_edits(
292    ctx: &AssistContext<'_>,
293    builder: &mut SourceChangeBuilder,
294    data: &StructEditData,
295    field_names: &FxHashMap<SmolStr, SmolStr>,
296) -> Vec<StructUsageEdit> {
297    data.usages
298        .iter()
299        .filter_map(|r| build_usage_edit(ctx, builder, data, r, field_names))
300        .collect_vec()
301}
302
303fn build_usage_edit(
304    ctx: &AssistContext<'_>,
305    builder: &mut SourceChangeBuilder,
306    data: &StructEditData,
307    usage: &FileReference,
308    field_names: &FxHashMap<SmolStr, SmolStr>,
309) -> Option<StructUsageEdit> {
310    match usage.name.syntax().ancestors().find_map(ast::FieldExpr::cast) {
311        Some(field_expr) => Some({
312            let field_name: SmolStr = field_expr.name_ref()?.to_string().into();
313            let new_field_name = field_names.get(&field_name)?;
314            let new_expr = ast::make::expr_path(ast::make::ext::ident_path(new_field_name));
315
316            // If struct binding is a reference, we might need to deref field usages
317            if data.is_ref {
318                let (replace_expr, ref_data) = determine_ref_and_parens(ctx, &field_expr);
319                StructUsageEdit::IndexField(
320                    builder.make_mut(replace_expr),
321                    ref_data.wrap_expr(new_expr).clone_for_update(),
322                )
323            } else {
324                StructUsageEdit::IndexField(
325                    builder.make_mut(field_expr).into(),
326                    new_expr.clone_for_update(),
327                )
328            }
329        }),
330        None => Some(StructUsageEdit::Path(usage.range)),
331    }
332}
333
334enum StructUsageEdit {
335    Path(TextRange),
336    IndexField(ast::Expr, ast::Expr),
337}
338
339impl StructUsageEdit {
340    fn apply(self, edit: &mut SourceChangeBuilder) {
341        match self {
342            StructUsageEdit::Path(target_expr) => {
343                edit.replace(target_expr, "todo!()");
344            }
345            StructUsageEdit::IndexField(target_expr, replace_with) => {
346                ted::replace(target_expr.syntax(), replace_with.syntax())
347            }
348        }
349    }
350}
