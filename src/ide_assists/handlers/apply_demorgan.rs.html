22// Assist: apply_demorgan
23//
24// Apply [De Morgan's law](https://en.wikipedia.org/wiki/De_Morgan%27s_laws).
25// This transforms expressions of the form `!l || !r` into `!(l && r)`.
26// This also works with `&&`. This assist can only be applied with the cursor
27// on either `||` or `&&`.
28//
29// ```
30// fn main() {
31//     if x != 4 ||$0 y < 3.14 {}
32// }
33// ```
34// ->
35// ```
36// fn main() {
37//     if !(x == 4 && y >= 3.14) {}
38// }
39// ```
40pub(crate) fn apply_demorgan(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
41    let mut bin_expr = if let Some(not) = ctx.find_token_syntax_at_offset(T![!])
42        && let Some(NodeOrToken::Node(next)) = not.next_sibling_or_token()
43        && let Some(paren) = ast::ParenExpr::cast(next)
44        && let Some(ast::Expr::BinExpr(bin_expr)) = paren.expr()
45    {
46        bin_expr
47    } else {
48        let bin_expr = ctx.find_node_at_offset::<ast::BinExpr>()?;
49        let op_range = bin_expr.op_token()?.text_range();
50
51        // Is the cursor on the expression's logical operator?
52        if !op_range.contains_range(ctx.selection_trimmed()) {
53            return None;
54        }
55
56        bin_expr
57    };
58
59    let op = bin_expr.op_kind()?;
60    let op_range = bin_expr.op_token()?.text_range();
61
62    // Walk up the tree while we have the same binary operator
63    while let Some(parent_expr) = bin_expr.syntax().parent().and_then(ast::BinExpr::cast) {
64        match parent_expr.op_kind() {
65            Some(parent_op) if parent_op == op => {
66                bin_expr = parent_expr;
67            }
68            _ => break,
69        }
70    }
71
72    if is_pattern_cond(bin_expr.clone().into()) {
73        return None;
74    }
75
76    let op = bin_expr.op_kind()?;
77    let (inv_token, prec) = match op {
78        ast::BinaryOp::LogicOp(ast::LogicOp::And) => (SyntaxKind::PIPE2, ExprPrecedence::LOr),
79        ast::BinaryOp::LogicOp(ast::LogicOp::Or) => (SyntaxKind::AMP2, ExprPrecedence::LAnd),
80        _ => return None,
81    };
82
83    let make = SyntaxFactory::with_mappings();
84
85    let demorganed = bin_expr.clone_subtree();
86    let mut editor = SyntaxEditor::new(demorganed.syntax().clone());
87    editor.replace(demorganed.op_token()?, make.token(inv_token));
88
89    let mut exprs = VecDeque::from([
90        (bin_expr.lhs()?, demorganed.lhs()?, prec),
91        (bin_expr.rhs()?, demorganed.rhs()?, prec),
92    ]);
93
94    while let Some((expr, demorganed, prec)) = exprs.pop_front() {
95        if let BinExpr(bin_expr) = &expr {
96            if let BinExpr(cbin_expr) = &demorganed {
97                if op == bin_expr.op_kind()? {
98                    editor.replace(cbin_expr.op_token()?, make.token(inv_token));
99                    exprs.push_back((bin_expr.lhs()?, cbin_expr.lhs()?, prec));
100                    exprs.push_back((bin_expr.rhs()?, cbin_expr.rhs()?, prec));
101                } else {
102                    let mut inv = invert_boolean_expression(&make, expr);
103                    if precedence(&inv).needs_parentheses_in(prec) {
104                        inv = make.expr_paren(inv).into();
105                    }
106                    editor.replace(demorganed.syntax(), inv.syntax());
107                }
108            } else {
109                return None;
110            }
111        } else {
112            let mut inv = invert_boolean_expression(&make, demorganed.clone());
113            if precedence(&inv).needs_parentheses_in(prec) {
114                inv = make.expr_paren(inv).into();
115            }
116            editor.replace(demorganed.syntax(), inv.syntax());
117        }
118    }
119
120    editor.add_mappings(make.finish_with_mappings());
121    let edit = editor.finish();
122    let demorganed = ast::Expr::cast(edit.new_root().clone())?;
123
124    acc.add_group(
125        &GroupLabel("Apply De Morgan's law".to_owned()),
126        AssistId::refactor_rewrite("apply_demorgan"),
127        "Apply De Morgan's law",
128        op_range,
129        |builder| {
130            let make = SyntaxFactory::with_mappings();
131            let (target_node, result_expr) = if let Some(neg_expr) = bin_expr
132                .syntax()
133                .parent()
134                .and_then(ast::ParenExpr::cast)
135                .and_then(|paren_expr| paren_expr.syntax().parent())
136                .and_then(ast::PrefixExpr::cast)
137                .filter(|prefix_expr| matches!(prefix_expr.op_kind(), Some(ast::UnaryOp::Not)))
138            {
139                cov_mark::hit!(demorgan_double_negation);
140                (ast::Expr::from(neg_expr).syntax().clone(), demorganed)
141            } else if let Some(paren_expr) =
142                bin_expr.syntax().parent().and_then(ast::ParenExpr::cast)
143            {
144                cov_mark::hit!(demorgan_double_parens);
145                (paren_expr.syntax().clone(), add_bang_paren(&make, demorganed))
146            } else {
147                (bin_expr.syntax().clone(), add_bang_paren(&make, demorganed))
148            };
149
150            let final_expr = if target_node
151                .parent()
152                .is_some_and(|p| result_expr.needs_parens_in_place_of(&p, &target_node))
153            {
154                cov_mark::hit!(demorgan_keep_parens_for_op_precedence2);
155                make.expr_paren(result_expr).into()
156            } else {
157                result_expr
158            };
159
160            let mut editor = builder.make_editor(&target_node);
161            editor.replace(&target_node, final_expr.syntax());
162            editor.add_mappings(make.finish_with_mappings());
163            builder.add_file_edits(ctx.vfs_file_id(), editor);
164        },
165    )
166}

168// Assist: apply_demorgan_iterator
169//
170// Apply [De Morgan's law](https://en.wikipedia.org/wiki/De_Morgan%27s_laws) to
171// `Iterator::all` and `Iterator::any`.
172//
173// This transforms expressions of the form `!iter.any(|x| predicate(x))` into
174// `iter.all(|x| !predicate(x))` and vice versa. This also works the other way for
175// `Iterator::all` into `Iterator::any`.
176//
177// ```
178// # //- minicore: iterator
179// fn main() {
180//     let arr = [1, 2, 3];
181//     if !arr.into_iter().$0any(|num| num == 4) {
182//         println!("foo");
183//     }
184// }
185// ```
186// ->
187// ```
188// fn main() {
189//     let arr = [1, 2, 3];
190//     if arr.into_iter().all(|num| num != 4) {
191//         println!("foo");
192//     }
193// }
194// ```
195pub(crate) fn apply_demorgan_iterator(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
196    let method_call: ast::MethodCallExpr = ctx.find_node_at_offset()?;
197    let (name, arg_expr) = validate_method_call_expr(ctx, &method_call)?;
198
199    let ast::Expr::ClosureExpr(closure_expr) = arg_expr else { return None };
200    let closure_body = closure_expr.body()?.clone_for_update();
201
202    let op_range = method_call.syntax().text_range();
203    let label = format!("Apply De Morgan's law to `Iterator::{}`", name.text().as_str());
204    acc.add_group(
205        &GroupLabel("Apply De Morgan's law".to_owned()),
206        AssistId::refactor_rewrite("apply_demorgan_iterator"),
207        label,
208        op_range,
209        |builder| {
210            let make = SyntaxFactory::with_mappings();
211            let mut editor = builder.make_editor(method_call.syntax());
212            // replace the method name
213            let new_name = match name.text().as_str() {
214                "all" => make.name_ref("any"),
215                "any" => make.name_ref("all"),
216                _ => unreachable!(),
217            };
218            editor.replace(name.syntax(), new_name.syntax());
219
220            // negate all tail expressions in the closure body
221            let tail_cb = &mut |e: &_| tail_cb_impl(&mut editor, &make, e);
222            walk_expr(&closure_body, &mut |expr| {
223                if let ast::Expr::ReturnExpr(ret_expr) = expr
224                    && let Some(ret_expr_arg) = &ret_expr.expr()
225                {
226                    for_each_tail_expr(ret_expr_arg, tail_cb);
227                }
228            });
229            for_each_tail_expr(&closure_body, tail_cb);
230
231            // negate the whole method call
232            if let Some(prefix_expr) = method_call
233                .syntax()
234                .parent()
235                .and_then(ast::PrefixExpr::cast)
236                .filter(|prefix_expr| matches!(prefix_expr.op_kind(), Some(ast::UnaryOp::Not)))
237            {
238                editor.delete(
239                    prefix_expr.op_token().expect("prefix expression always has an operator"),
240                );
241            } else {
242                editor.insert(Position::before(method_call.syntax()), make.token(SyntaxKind::BANG));
243            }
244
245            editor.add_mappings(make.finish_with_mappings());
246            builder.add_file_edits(ctx.vfs_file_id(), editor);
247        },
248    )
249}

251/// Ensures that the method call is to `Iterator::all` or `Iterator::any`.
252fn validate_method_call_expr(
253    ctx: &AssistContext<'_>,
254    method_call: &ast::MethodCallExpr,
255) -> Option<(ast::NameRef, ast::Expr)> {
256    let name_ref = method_call.name_ref()?;
257    if name_ref.text() != "all" && name_ref.text() != "any" {
258        return None;
259    }
260    let arg_expr = method_call.arg_list()?.args().next()?;
261
262    let sema = &ctx.sema;
263
264    let receiver = method_call.receiver()?;
265    let it_type = sema.type_of_expr(&receiver)?.adjusted();
266    let module = sema.scope(receiver.syntax())?.module();
267    let krate = module.krate(ctx.db());
268
269    let iter_trait = FamousDefs(sema, krate).core_iter_Iterator()?;
270    it_type.impls_trait(sema.db, iter_trait, &[]).then_some((name_ref, arg_expr))
271}
272
273fn tail_cb_impl(editor: &mut SyntaxEditor, make: &SyntaxFactory, e: &ast::Expr) {
274    match e {
275        ast::Expr::BreakExpr(break_expr) => {
276            if let Some(break_expr_arg) = break_expr.expr() {
277                for_each_tail_expr(&break_expr_arg, &mut |e| tail_cb_impl(editor, make, e))
278            }
279        }
280        ast::Expr::ReturnExpr(_) => {
281            // all return expressions have already been handled by the walk loop
282        }
283        e => {
284            let inverted_body = invert_boolean_expression(make, e.clone());
285            editor.replace(e.syntax(), inverted_body.syntax());
286        }
287    }
288}

290/// Add bang and parentheses to the expression.
291fn add_bang_paren(make: &SyntaxFactory, expr: ast::Expr) -> ast::Expr {
292    make.expr_prefix(T![!], make.expr_paren(expr).into()).into()
293}
