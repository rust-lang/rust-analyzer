30// Assist: generate_function
31//
32// Adds a stub function with a signature matching the function under the cursor.
33//
34// ```
35// struct Baz;
36// fn baz() -> Baz { Baz }
37// fn foo() {
38//     bar$0("", baz());
39// }
40//
41// ```
42// ->
43// ```
44// struct Baz;
45// fn baz() -> Baz { Baz }
46// fn foo() {
47//     bar("", baz());
48// }
49//
50// fn bar(arg: &'static str, baz: Baz) ${0:-> _} {
51//     todo!()
52// }
53//
54// ```
55pub(crate) fn generate_function(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
56    gen_fn(acc, ctx).or_else(|| gen_method(acc, ctx))
57}

59fn gen_fn(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
60    let path_expr: ast::PathExpr = ctx.find_node_at_offset()?;
61    let call = path_expr.syntax().parent().and_then(ast::CallExpr::cast)?;
62    let path = path_expr.path()?;
63    let name_ref = path.segment()?.name_ref()?;
64    if ctx.sema.resolve_path(&path).is_some() {
65        // The function call already resolves, no need to add a function
66        return None;
67    }
68
69    let fn_name = &*name_ref.text();
70    let TargetInfo { target_module, adt_info, target, file } =
71        fn_target_info(ctx, path, &call, fn_name)?;
72
73    if let Some(m) = target_module
74        && !is_editable_crate(m.krate(ctx.db()), ctx.db())
75    {
76        return None;
77    }
78
79    let function_builder =
80        FunctionBuilder::from_call(ctx, &call, fn_name, target_module, target, &adt_info)?;
81    let text_range = call.syntax().text_range();
82    let label = format!("Generate {} function", function_builder.fn_name);
83    add_func_to_accumulator(acc, ctx, text_range, function_builder, file, adt_info, label)
84}

104fn fn_target_info(
105    ctx: &AssistContext<'_>,
106    path: ast::Path,
107    call: &CallExpr,
108    fn_name: &str,
109) -> Option<TargetInfo> {
110    match path.qualifier() {
111        Some(qualifier) => match ctx.sema.resolve_path(&qualifier) {
112            Some(hir::PathResolution::Def(hir::ModuleDef::Module(module))) => {
113                get_fn_target_info(ctx, Some(module), call.clone())
114            }
115            Some(hir::PathResolution::Def(hir::ModuleDef::Adt(adt))) => {
116                if let hir::Adt::Enum(_) = adt {
117                    // Don't suggest generating function if the name starts with an uppercase letter
118                    if fn_name.starts_with(char::is_uppercase) {
119                        return None;
120                    }
121                }
122
123                assoc_fn_target_info(ctx, call, adt, fn_name)
124            }
125            Some(hir::PathResolution::SelfType(impl_)) => {
126                let adt = impl_.self_ty(ctx.db()).as_adt()?;
127                assoc_fn_target_info(ctx, call, adt, fn_name)
128            }
129            _ => None,
130        },
131        _ => get_fn_target_info(ctx, None, call.clone()),
132    }
133}

135fn gen_method(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
136    let call: ast::MethodCallExpr = ctx.find_node_at_offset()?;
137    if ctx.sema.resolve_method_call(&call).is_some() {
138        return None;
139    }
140
141    let fn_name = call.name_ref()?;
142    let receiver_ty = ctx.sema.type_of_expr(&call.receiver()?)?.original().strip_references();
143    let adt = receiver_ty.as_adt()?;
144
145    let target_module = adt.module(ctx.sema.db);
146    if !is_editable_crate(target_module.krate(ctx.db()), ctx.db()) {
147        return None;
148    }
149
150    let enclosing_impl = ctx.find_node_at_offset::<ast::Impl>();
151    let cursor_impl = enclosing_impl.filter(|impl_| {
152        ctx.sema.to_def(impl_).map_or(false, |def| def.self_ty(ctx.sema.db).as_adt() == Some(adt))
153    });
154
155    let (impl_, file) = if let Some(impl_) = cursor_impl {
156        (Some(impl_), ctx.vfs_file_id())
157    } else {
158        get_adt_source(ctx, &adt, fn_name.text().as_str())?
159    };
160    let target = get_method_target(ctx, &impl_, &adt)?;
161
162    let function_builder = FunctionBuilder::from_method_call(
163        ctx,
164        &call,
165        &fn_name,
166        receiver_ty,
167        target_module,
168        target,
169    )?;
170    let text_range = call.syntax().text_range();
171    let adt_info = AdtInfo::new(adt, impl_.is_some());
172    let label = format!("Generate {} method", function_builder.fn_name);
173    add_func_to_accumulator(acc, ctx, text_range, function_builder, file, Some(adt_info), label)
174}

176fn add_func_to_accumulator(
177    acc: &mut Assists,
178    ctx: &AssistContext<'_>,
179    text_range: TextRange,
180    function_builder: FunctionBuilder,
181    file: FileId,
182    adt_info: Option<AdtInfo>,
183    label: String,
184) -> Option<()> {
185    acc.add(AssistId::generate("generate_function"), label, text_range, |edit| {
186        edit.edit_file(file);
187
188        let target = function_builder.target.clone();
189        let edition = function_builder.target_edition;
190        let func = function_builder.render(ctx.config.snippet_cap, edit);
191
192        if let Some(adt) = adt_info
193            .and_then(|adt_info| if adt_info.impl_exists { None } else { Some(adt_info.adt) })
194        {
195            let name = make::ty_path(make::ext::ident_path(&format!(
196                "{}",
197                adt.name(ctx.db()).display(ctx.db(), edition)
198            )));
199
200            // FIXME: adt may have generic params.
201            let impl_ = make::impl_(None, None, None, name, None, None).clone_for_update();
202
203            func.indent(IndentLevel(1));
204            impl_.get_or_create_assoc_item_list().add_item(func.into());
205            target.insert_impl_at(edit, impl_);
206        } else {
207            target.insert_fn_at(edit, func);
208        }
209    })
210}

212fn get_adt_source(
213    ctx: &AssistContext<'_>,
214    adt: &hir::Adt,
215    fn_name: &str,
216) -> Option<(Option<ast::Impl>, FileId)> {
217    let range = adt.source(ctx.sema.db)?.syntax().original_file_range_rooted(ctx.sema.db);
218
219    let file = ctx.sema.parse(range.file_id);
220    let adt_source =
221        ctx.sema.find_node_at_offset_with_macros(file.syntax(), range.range.start())?;
222    find_struct_impl(ctx, &adt_source, &[fn_name.to_owned()])
223        .map(|impl_| (impl_, range.file_id.file_id(ctx.db())))
224}

241    /// Prepares a generated function that matches `call`.
242    /// The function is generated in `target_module` or next to `call`
243    fn from_call(
244        ctx: &AssistContext<'_>,
245        call: &ast::CallExpr,
246        fn_name: &str,
247        target_module: Option<Module>,
248        target: GeneratedFunctionTarget,
249        adt_info: &Option<AdtInfo>,
250    ) -> Option<Self> {
251        let target_module =
252            target_module.or_else(|| ctx.sema.scope(target.syntax()).map(|it| it.module()))?;
253        let target_edition = target_module.krate(ctx.db()).edition(ctx.db());
254
255        let current_module = ctx.sema.scope(call.syntax())?.module();
256        let visibility = calculate_necessary_visibility(current_module, target_module, ctx);
257        let fn_name = make::name(fn_name);
258        let mut necessary_generic_params = FxHashSet::default();
259        let params = fn_args(
260            ctx,
261            target_module,
262            ast::CallableExpr::Call(call.clone()),
263            &mut necessary_generic_params,
264        )?;
265
266        let await_expr = call.syntax().parent().and_then(ast::AwaitExpr::cast);
267        let is_async = await_expr.is_some();
268
269        let ret_type;
270        let should_focus_return_type;
271        let fn_body;
272
273        // If generated function has the name "new" and is an associated function, we generate fn body
274        // as a constructor and assume a "Self" return type.
275        if let Some(body) =
276            make_fn_body_as_new_function(ctx, &fn_name.text(), adt_info, target_edition)
277        {
278            ret_type = Some(make::ret_type(make::ty_path(make::ext::ident_path("Self"))));
279            should_focus_return_type = false;
280            fn_body = body;
281        } else {
282            let expr_for_ret_ty = await_expr.map_or_else(|| call.clone().into(), |it| it.into());
283            (ret_type, should_focus_return_type) = make_return_type(
284                ctx,
285                &expr_for_ret_ty,
286                target_module,
287                &mut necessary_generic_params,
288            );
289            let placeholder_expr = match ctx.config.expr_fill_default {
290                ExprFillDefaultMode::Todo => make::ext::expr_todo(),
291                ExprFillDefaultMode::Underscore => make::ext::expr_underscore(),
292                ExprFillDefaultMode::Default => make::ext::expr_todo(),
293            };
294            fn_body = make::block_expr(vec![], Some(placeholder_expr));
295        };
296
297        let (generic_param_list, where_clause) =
298            fn_generic_params(ctx, necessary_generic_params, &target)?;
299
300        Some(Self {
301            target,
302            fn_name,
303            generic_param_list,
304            where_clause,
305            params,
306            fn_body,
307            ret_type,
308            should_focus_return_type,
309            visibility,
310            is_async,
311            target_edition,
312        })
313    }

370    fn render(self, cap: Option<SnippetCap>, edit: &mut SourceChangeBuilder) -> ast::Fn {
371        let visibility = match self.visibility {
372            Visibility::None => None,
373            Visibility::Crate => Some(make::visibility_pub_crate()),
374            Visibility::Pub => Some(make::visibility_pub()),
375        };
376        let type_params =
377            self.generic_param_list.filter(|list| list.generic_params().next().is_some());
378        let fn_def = make::fn_(
379            None,
380            visibility,
381            self.fn_name,
382            type_params,
383            self.where_clause,
384            self.params,
385            self.fn_body,
386            self.ret_type,
387            self.is_async,
388            false, // FIXME : const and unsafe are not handled yet.
389            false,
390            false,
391        )
392        .clone_for_update();
393
394        let ret_type = fn_def.ret_type();
395        // PANIC: we guarantee we always create a function body with a tail expr
396        let tail_expr = fn_def
397            .body()
398            .expect("generated function should have a body")
399            .tail_expr()
400            .expect("function body should have a tail expression");
401
402        if let Some(cap) = cap {
403            if self.should_focus_return_type {
404                // Focus the return type if there is one
405                match ret_type {
406                    Some(ret_type) => {
407                        edit.add_placeholder_snippet(cap, ret_type);
408                    }
409                    None => {
410                        edit.add_placeholder_snippet(cap, tail_expr);
411                    }
412                }
413            } else {
414                edit.add_placeholder_snippet(cap, tail_expr);
415            }
416        }
417
418        fn_def
419    }

422/// Makes an optional return type along with whether the return type should be focused by the cursor.
423/// If we cannot infer what the return type should be, we create a placeholder type.
424///
425/// The rule for whether we focus a return type or not (and thus focus the function body),
426/// is rather simple:
427/// * If we could *not* infer what the return type should be, focus it (so the user can fill-in
428///   the correct return type).
429/// * If we could infer the return type, don't focus it (and thus focus the function body) so the
430///   user can change the `todo!` function body.
431fn make_return_type(
432    ctx: &AssistContext<'_>,
433    expr: &ast::Expr,
434    target_module: Module,
435    necessary_generic_params: &mut FxHashSet<hir::GenericParam>,
436) -> (Option<ast::RetType>, bool) {
437    let (ret_ty, should_focus_return_type) = {
438        match ctx.sema.type_of_expr(expr).map(TypeInfo::original) {
439            Some(ty) if ty.is_unknown() => (Some(make::ty_placeholder()), true),
440            None => (Some(make::ty_placeholder()), true),
441            Some(ty) if ty.is_unit() => (None, false),
442            Some(ty) => {
443                necessary_generic_params.extend(ty.generic_params(ctx.db()));
444                let rendered = ty.display_source_code(ctx.db(), target_module.into(), true);
445                match rendered {
446                    Ok(rendered) => (Some(make::ty(&rendered)), false),
447                    Err(_) => (Some(make::ty_placeholder()), true),
448                }
449            }
450        }
451    };
452    let ret_type = ret_ty.map(make::ret_type);
453    (ret_type, should_focus_return_type)
454}

456fn make_fn_body_as_new_function(
457    ctx: &AssistContext<'_>,
458    fn_name: &str,
459    adt_info: &Option<AdtInfo>,
460    edition: Edition,
461) -> Option<ast::BlockExpr> {
462    if fn_name != "new" {
463        return None;
464    };
465    let adt_info = adt_info.as_ref()?;
466
467    let path_self = make::ext::ident_path("Self");
468    let placeholder_expr = match ctx.config.expr_fill_default {
469        ExprFillDefaultMode::Todo => make::ext::expr_todo(),
470        ExprFillDefaultMode::Underscore => make::ext::expr_underscore(),
471        ExprFillDefaultMode::Default => make::ext::expr_todo(),
472    };
473    let tail_expr = if let Some(strukt) = adt_info.adt.as_struct() {
474        match strukt.kind(ctx.db()) {
475            StructKind::Record => {
476                let fields = strukt
477                    .fields(ctx.db())
478                    .iter()
479                    .map(|field| {
480                        make::record_expr_field(
481                            make::name_ref(&format!(
482                                "{}",
483                                field.name(ctx.db()).display(ctx.db(), edition)
484                            )),
485                            Some(placeholder_expr.clone()),
486                        )
487                    })
488                    .collect::<Vec<_>>();
489
490                make::record_expr(path_self, make::record_expr_field_list(fields)).into()
491            }
492            StructKind::Tuple => {
493                let args = strukt
494                    .fields(ctx.db())
495                    .iter()
496                    .map(|_| placeholder_expr.clone())
497                    .collect::<Vec<_>>();
498
499                make::expr_call(make::expr_path(path_self), make::arg_list(args)).into()
500            }
501            StructKind::Unit => make::expr_path(path_self),
502        }
503    } else {
504        placeholder_expr
505    };
506
507    let fn_body = make::block_expr(vec![], Some(tail_expr));
508    Some(fn_body)
509}

549fn assoc_fn_target_info(
550    ctx: &AssistContext<'_>,
551    call: &CallExpr,
552    adt: hir::Adt,
553    fn_name: &str,
554) -> Option<TargetInfo> {
555    let current_module = ctx.sema.scope(call.syntax())?.module();
556    let module = adt.module(ctx.sema.db);
557    let target_module = if current_module == module { None } else { Some(module) };
558    if current_module.krate(ctx.db()) != module.krate(ctx.db()) {
559        return None;
560    }

562    let target = get_method_target(ctx, &impl_, &adt)?;
563    let adt_info = AdtInfo::new(adt, impl_.is_some());
564    Some(TargetInfo::new(target_module, Some(adt_info), target, file))
565}

567#[derive(Clone)]
568enum GeneratedFunctionTarget {
569    AfterItem(SyntaxNode),
570    InEmptyItemList(SyntaxNode),
571    InImpl(ast::Impl),
572}
573
574impl GeneratedFunctionTarget {
575    fn syntax(&self) -> &SyntaxNode {
576        match self {
577            GeneratedFunctionTarget::AfterItem(it) => it,
578            GeneratedFunctionTarget::InEmptyItemList(it) => it,
579            GeneratedFunctionTarget::InImpl(it) => it.syntax(),
580        }
581    }
582
583    fn parent(&self) -> SyntaxNode {
584        match self {
585            GeneratedFunctionTarget::AfterItem(it) => it.parent().expect("item without parent"),
586            GeneratedFunctionTarget::InEmptyItemList(it) => it.clone(),
587            GeneratedFunctionTarget::InImpl(it) => it.syntax().clone(),
588        }
589    }
590
591    fn insert_impl_at(&self, edit: &mut SourceChangeBuilder, impl_: ast::Impl) {
592        match self {
593            GeneratedFunctionTarget::AfterItem(item) => {
594                let item = edit.make_syntax_mut(item.clone());
595                let position = if item.parent().is_some() {
596                    ted::Position::after(&item)
597                } else {
598                    ted::Position::first_child_of(&item)
599                };
600
601                let indent = IndentLevel::from_node(&item);
602                let leading_ws = make::tokens::whitespace(&format!("\n{indent}"));
603                impl_.indent(indent);
604
605                ted::insert_all(position, vec![leading_ws.into(), impl_.syntax().clone().into()]);
606            }
607            GeneratedFunctionTarget::InEmptyItemList(item_list) => {
608                let item_list = edit.make_syntax_mut(item_list.clone());
609                let insert_after =
610                    item_list.children_with_tokens().find_or_first(|child| child.kind() == T!['{']);
611                let position = match insert_after {
612                    Some(child) => ted::Position::after(child),
613                    None => ted::Position::first_child_of(&item_list),
614                };
615
616                let indent = IndentLevel::from_node(&item_list);
617                let leading_indent = indent + 1;
618                let leading_ws = make::tokens::whitespace(&format!("\n{leading_indent}"));
619                impl_.indent(indent);
620
621                ted::insert_all(position, vec![leading_ws.into(), impl_.syntax().clone().into()]);
622            }
623            GeneratedFunctionTarget::InImpl(_) => {
624                unreachable!("can't insert an impl inside an impl")
625            }
626        }
627    }
628
629    fn insert_fn_at(&self, edit: &mut SourceChangeBuilder, func: ast::Fn) {
630        match self {
631            GeneratedFunctionTarget::AfterItem(item) => {
632                let item = edit.make_syntax_mut(item.clone());
633                let position = if item.parent().is_some() {
634                    ted::Position::after(&item)
635                } else {
636                    ted::Position::first_child_of(&item)
637                };
638
639                let indent = IndentLevel::from_node(&item);
640                let leading_ws = make::tokens::whitespace(&format!("\n\n{indent}"));
641                func.indent(indent);
642
643                ted::insert_all_raw(
644                    position,
645                    vec![leading_ws.into(), func.syntax().clone().into()],
646                );
647            }
648            GeneratedFunctionTarget::InEmptyItemList(item_list) => {
649                let item_list = edit.make_syntax_mut(item_list.clone());
650                let insert_after =
651                    item_list.children_with_tokens().find_or_first(|child| child.kind() == T!['{']);
652                let position = match insert_after {
653                    Some(child) => ted::Position::after(child),
654                    None => ted::Position::first_child_of(&item_list),
655                };
656
657                let indent = IndentLevel::from_node(&item_list);
658                let leading_indent = indent + 1;
659                let leading_ws = make::tokens::whitespace(&format!("\n{leading_indent}"));
660                let trailing_ws = make::tokens::whitespace(&format!("\n{indent}"));
661                func.indent(leading_indent);
662
663                ted::insert_all(
664                    position,
665                    vec![leading_ws.into(), func.syntax().clone().into(), trailing_ws.into()],
666                );
667            }
668            GeneratedFunctionTarget::InImpl(impl_) => {
669                let impl_ = edit.make_mut(impl_.clone());
670
671                let leading_indent = impl_.indent_level() + 1;
672                func.indent(leading_indent);
673
674                impl_.get_or_create_assoc_item_list().add_item(func.into());
675            }
676        }
677    }
678}

718/// Gets parameter bounds and where predicates in scope and filters out irrelevant ones. Returns
719/// `None` when it fails to get scope information.
720///
721/// See comment on `filter_unnecessary_bounds()` for what bounds we consider relevant.
722///
723/// NOTE: Generic parameters returned from this function may cause name clash at `target`. We don't
724/// currently do anything about it because it's actually easy to resolve it after the assist: just
725/// use the Rename functionality.
726fn fn_generic_params(
727    ctx: &AssistContext<'_>,
728    necessary_params: FxHashSet<hir::GenericParam>,
729    target: &GeneratedFunctionTarget,
730) -> Option<(Option<ast::GenericParamList>, Option<ast::WhereClause>)> {
731    if necessary_params.is_empty() {
732        // Not really needed but fast path.
733        return Some((None, None));
734    }
735
736    // 1. Get generic parameters (with bounds) and where predicates in scope.
737    let (generic_params, where_preds) = params_and_where_preds_in_scope(ctx);
738
739    // 2. Extract type parameters included in each bound.
740    let mut generic_params = generic_params
741        .into_iter()
742        .filter_map(|it| compute_contained_params_in_generic_param(ctx, it))
743        .collect();
744    let mut where_preds = where_preds
745        .into_iter()
746        .filter_map(|it| compute_contained_params_in_where_pred(ctx, it))
747        .collect();
748
749    // 3. Filter out unnecessary bounds.
750    filter_unnecessary_bounds(&mut generic_params, &mut where_preds, necessary_params);
751    filter_bounds_in_scope(&mut generic_params, &mut where_preds, ctx, target);
752
753    let generic_params: Vec<ast::GenericParam> =
754        generic_params.into_iter().map(|it| it.node.clone_for_update()).collect();
755    let where_preds: Vec<ast::WherePred> =
756        where_preds.into_iter().map(|it| it.node.clone_for_update()).collect();
757
758    let (generic_params, where_preds): (Vec<ast::GenericParam>, Vec<ast::WherePred>) =
759        if let Some(param) = generic_params.first()
760            && let source_scope = ctx.sema.scope(param.syntax())?
761            && let target_scope = ctx.sema.scope(&target.parent())?
762            && source_scope.module() != target_scope.module()
763        {
764            // 4. Rewrite paths
765            let transform = PathTransform::generic_transformation(&target_scope, &source_scope);
766            let generic_params = generic_params.iter().map(|it| it.syntax());
767            let where_preds = where_preds.iter().map(|it| it.syntax());
768            transform
769                .apply_all(generic_params.chain(where_preds))
770                .into_iter()
771                .filter_map(|it| {
772                    if let Some(it) = ast::GenericParam::cast(it.clone()) {
773                        Some(either::Either::Left(it))
774                    } else {
775                        ast::WherePred::cast(it).map(either::Either::Right)
776                    }
777                })
778                .partition_map(|it| it)
779        } else {
780            (generic_params, where_preds)
781        };
782
783    let generic_param_list = make::generic_param_list(generic_params);
784    let where_clause =
785        if where_preds.is_empty() { None } else { Some(make::where_clause(where_preds)) };
786
787    Some((Some(generic_param_list), where_clause))
788}

790fn params_and_where_preds_in_scope(
791    ctx: &AssistContext<'_>,
792) -> (Vec<ast::GenericParam>, Vec<ast::WherePred>) {
793    let Some(body) = containing_body(ctx) else {
794        return Default::default();
795    };
796
797    let mut generic_params = Vec::new();
798    let mut where_clauses = Vec::new();
799
800    // There are two items where generic parameters currently in scope may be declared: the item
801    // the cursor is at, and its parent (if any).
802    //
803    // We handle parent first so that their generic parameters appear first in the generic
804    // parameter list of the function we're generating.
805    let db = ctx.db();
806    if let Some(parent) = body.as_assoc_item(db).map(|it| it.container(db)) {
807        match parent {
808            hir::AssocItemContainer::Impl(it) => {
809                let (params, clauses) = get_bounds_in_scope(ctx, it);
810                generic_params.extend(params);
811                where_clauses.extend(clauses);
812            }
813            hir::AssocItemContainer::Trait(it) => {
814                let (params, clauses) = get_bounds_in_scope(ctx, it);
815                generic_params.extend(params);
816                where_clauses.extend(clauses);
817            }
818        }
819    }
820
821    // Other defs with body may inherit generic parameters from its parent, but never have their
822    // own generic parameters.
823    if let hir::DefWithBody::Function(it) = body {
824        let (params, clauses) = get_bounds_in_scope(ctx, it);
825        generic_params.extend(params);
826        where_clauses.extend(clauses);
827    }
828
829    (generic_params, where_clauses)
830}

832fn containing_body(ctx: &AssistContext<'_>) -> Option<hir::DefWithBody> {
833    let item: ast::Item = ctx.find_node_at_offset()?;
834    let def = match item {
835        ast::Item::Fn(it) => ctx.sema.to_def(&it)?.into(),
836        ast::Item::Const(it) => ctx.sema.to_def(&it)?.into(),
837        ast::Item::Static(it) => ctx.sema.to_def(&it)?.into(),
838        _ => return None,
839    };
840    Some(def)
841}

843fn get_bounds_in_scope<D>(
844    ctx: &AssistContext<'_>,
845    def: D,
846) -> (impl Iterator<Item = ast::GenericParam>, impl Iterator<Item = ast::WherePred>)
847where
848    D: HasSource,
849    D::Ast: HasGenericParams,
850{
851    // This function should be only called with `Impl`, `Trait`, or `Function`, for which it's
852    // infallible to get source ast.
853    let node = ctx.sema.source(def).expect("definition's source couldn't be found").value;
854    let generic_params = node.generic_param_list().into_iter().flat_map(|it| it.generic_params());
855    let where_clauses = node.where_clause().into_iter().flat_map(|it| it.predicates());
856    (generic_params, where_clauses)
857}

859#[derive(Debug)]
860struct ParamBoundWithParams {
861    node: ast::GenericParam,
862    /// Generic parameter `node` introduces.
863    ///
864    /// ```text
865    /// impl<T> S<T> {
866    ///     fn f<U: Trait<T>>() {}
867    ///          ^ this
868    /// }
869    /// ```
870    ///
871    /// `U` in this example.
872    self_ty_param: hir::GenericParam,
873    /// Generic parameters contained in the trait reference of this bound.
874    ///
875    /// ```text
876    /// impl<T> S<T> {
877    ///     fn f<U: Trait<T>>() {}
878    ///             ^^^^^^^^ params in this part
879    /// }
880    /// ```
881    ///
882    /// `T` in this example.
883    other_params: FxHashSet<hir::GenericParam>,
884}

886#[derive(Debug)]
887struct WherePredWithParams {
888    node: ast::WherePred,
889    /// Generic parameters contained in the "self type" of this where predicate.
890    ///
891    /// ```text
892    /// Struct<T, U>: Trait<T, Assoc = V>,
893    /// ^^^^^^^^^^^^ params in this part
894    /// ```
895    ///
896    /// `T` and `U` in this example.
897    self_ty_params: FxHashSet<hir::GenericParam>,
898    /// Generic parameters contained in the trait reference of this where predicate.
899    ///
900    /// ```text
901    /// Struct<T, U>: Trait<T, Assoc = V>,
902    ///               ^^^^^^^^^^^^^^^^^^^ params in this part
903    /// ```
904    ///
905    /// `T` and `V` in this example.
906    other_params: FxHashSet<hir::GenericParam>,
907}

909fn compute_contained_params_in_generic_param(
910    ctx: &AssistContext<'_>,
911    node: ast::GenericParam,
912) -> Option<ParamBoundWithParams> {
913    match &node {
914        ast::GenericParam::TypeParam(ty) => {
915            let self_ty_param = ctx.sema.to_def(ty)?.into();
916
917            let other_params = ty
918                .type_bound_list()
919                .into_iter()
920                .flat_map(|it| it.bounds())
921                .flat_map(|bound| bound.syntax().descendants())
922                .filter_map(|node| filter_generic_params(ctx, node))
923                .collect();
924
925            Some(ParamBoundWithParams { node, self_ty_param, other_params })
926        }
927        ast::GenericParam::ConstParam(ct) => {
928            let self_ty_param = ctx.sema.to_def(ct)?.into();
929            Some(ParamBoundWithParams { node, self_ty_param, other_params: FxHashSet::default() })
930        }
931        ast::GenericParam::LifetimeParam(_) => {
932            // FIXME: It might be a good idea to handle lifetime parameters too.
933            None
934        }
935    }
936}

938fn compute_contained_params_in_where_pred(
939    ctx: &AssistContext<'_>,
940    node: ast::WherePred,
941) -> Option<WherePredWithParams> {
942    let self_ty = node.ty()?;
943    let bound_list = node.type_bound_list()?;
944
945    let self_ty_params = self_ty
946        .syntax()
947        .descendants()
948        .filter_map(|node| filter_generic_params(ctx, node))
949        .collect();
950
951    let other_params = bound_list
952        .bounds()
953        .flat_map(|bound| bound.syntax().descendants())
954        .filter_map(|node| filter_generic_params(ctx, node))
955        .collect();
956
957    Some(WherePredWithParams { node, self_ty_params, other_params })
958}

969/// Filters out irrelevant bounds from `generic_params` and `where_preds`.
970///
971/// Say we have a trait bound `Struct<T>: Trait<U>`. Given `necessary_params`, when is it relevant
972/// and when not? Some observations:
973/// - When `necessary_params` contains `T`, it's likely that we want this bound, but now we have
974///   an extra param to consider: `U`.
975/// - On the other hand, when `necessary_params` contains `U` (but not `T`), then it's unlikely
976///   that we want this bound because it doesn't really constrain `U`.
977///
978/// (FIXME?: The latter clause might be overstating. We may want to include the bound if the self
979/// type does *not* include generic params at all - like `Option<i32>: From<U>`)
980///
981/// Can we make this a bit more formal? Let's define "dependency" between generic parameters and
982/// trait bounds:
983/// - A generic parameter `T` depends on a trait bound if `T` appears in the self type (i.e. left
984///   part) of the bound.
985/// - A trait bound depends on a generic parameter `T` if `T` appears in the bound.
986///
987/// Using the notion, what we want is all the bounds that params in `necessary_params`
988/// *transitively* depend on!
989///
990/// Now it's not hard to solve: we build a dependency graph and compute all reachable nodes from
991/// nodes that represent params in `necessary_params` by usual and boring DFS.
992///
993/// The time complexity is O(|generic_params| + |where_preds| + |necessary_params|).
994fn filter_unnecessary_bounds(
995    generic_params: &mut Vec<ParamBoundWithParams>,
996    where_preds: &mut Vec<WherePredWithParams>,
997    necessary_params: FxHashSet<hir::GenericParam>,
998) {
999    // All `self_ty_param` should be unique as they were collected from `ast::GenericParamList`s.
1000    let param_map: FxHashMap<hir::GenericParam, usize> =
1001        generic_params.iter().map(|it| it.self_ty_param).zip(0..).collect();
1002    let param_count = param_map.len();
1003    let generic_params_upper_bound = param_count + generic_params.len();
1004    let node_count = generic_params_upper_bound + where_preds.len();
1005
1006    // | node index range                        | what the node represents |
1007    // |-----------------------------------------|--------------------------|
1008    // | 0..param_count                          | generic parameter        |
1009    // | param_count..generic_params_upper_bound | `ast::GenericParam`      |
1010    // | generic_params_upper_bound..node_count  | `ast::WherePred`         |
1011    let mut graph = Graph::new(node_count);
1012    for (pred, pred_idx) in generic_params.iter().zip(param_count..) {
1013        let param_idx = param_map[&pred.self_ty_param];
1014        graph.add_edge(param_idx, pred_idx);
1015        graph.add_edge(pred_idx, param_idx);
1016
1017        for param in &pred.other_params {
1018            let param_idx = param_map[param];
1019            graph.add_edge(pred_idx, param_idx);
1020        }
1021    }
1022    for (pred, pred_idx) in where_preds.iter().zip(generic_params_upper_bound..) {
1023        for param in &pred.self_ty_params {
1024            let param_idx = param_map[param];
1025            graph.add_edge(param_idx, pred_idx);
1026            graph.add_edge(pred_idx, param_idx);
1027        }
1028        for param in &pred.other_params {
1029            let param_idx = param_map[param];
1030            graph.add_edge(pred_idx, param_idx);
1031        }
1032    }
1033
1034    let starting_nodes = necessary_params.iter().flat_map(|param| param_map.get(param).copied());
1035    let reachable = graph.compute_reachable_nodes(starting_nodes);
1036
1037    // Not pretty, but effective. If only there were `Vec::retain_index()`...
1038    let mut idx = param_count;
1039    generic_params.retain(|_| {
1040        idx += 1;
1041        reachable[idx - 1]
1042    });
1043    stdx::always!(idx == generic_params_upper_bound, "inconsistent index");
1044    where_preds.retain(|_| {
1045        idx += 1;
1046        reachable[idx - 1]
1047    });
1048}

1050/// Filters out bounds from impl if we're generating the function into the same impl we're
1051/// generating from.
1052fn filter_bounds_in_scope(
1053    generic_params: &mut Vec<ParamBoundWithParams>,
1054    where_preds: &mut Vec<WherePredWithParams>,
1055    ctx: &AssistContext<'_>,
1056    target: &GeneratedFunctionTarget,
1057) -> Option<()> {
1058    let target_impl = target.parent().ancestors().find_map(ast::Impl::cast)?;
1059    let target_impl = ctx.sema.to_def(&target_impl)?;
1060    // It's sufficient to test only the first element of `generic_params` because of the order of
1061    // insertion (see `params_and_where_preds_in_scope()`).
1062    let def = generic_params.first()?.self_ty_param.parent();
1063    if def != hir::GenericDef::Impl(target_impl) {
1064        return None;
1065    }
1066
1067    // Now we know every element that belongs to an impl would be in scope at `target`, we can
1068    // filter them out just by looking at their parent.
1069    generic_params.retain(|it| !matches!(it.self_ty_param.parent(), hir::GenericDef::Impl(_)));
1070    where_preds.retain(|it| {
1071        it.node.syntax().parent().and_then(|it| it.parent()).and_then(ast::Impl::cast).is_none()
1072    });
1073
1074    Some(())
1075}

1077/// Makes duplicate argument names unique by appending incrementing numbers.
1078///
1079/// ```ignore
1080/// let mut names: Vec<String> =
1081///     vec!["foo".into(), "foo".into(), "bar".into(), "baz".into(), "bar".into()];
1082/// deduplicate_arg_names(&mut names);
1083/// let expected: Vec<String> =
1084///     vec!["foo_1".into(), "foo_2".into(), "bar_1".into(), "baz".into(), "bar_2".into()];
1085/// assert_eq!(names, expected);
1086/// ```
1087fn deduplicate_arg_names(arg_names: &mut [String]) {
1088    let mut arg_name_counts = FxHashMap::default();
1089    for name in arg_names.iter() {
1090        *arg_name_counts.entry(name).or_insert(0) += 1;
1091    }
1092    let duplicate_arg_names: FxHashSet<String> = arg_name_counts
1093        .into_iter()
1094        .filter(|(_, count)| *count >= 2)
1095        .map(|(name, _)| name.clone())
1096        .collect();
1097
1098    let mut counter_per_name = FxHashMap::default();
1099    for arg_name in arg_names.iter_mut() {
1100        if duplicate_arg_names.contains(arg_name) {
1101            let counter = counter_per_name.entry(arg_name.clone()).or_insert(1);
1102            arg_name.push('_');
1103            arg_name.push_str(&counter.to_string());
1104            *counter += 1;
1105        }
1106    }
1107}

1109fn fn_arg_name(sema: &Semantics<'_, RootDatabase>, arg_expr: &ast::Expr) -> String {
1110    let name = (|| match arg_expr {
1111        ast::Expr::CastExpr(cast_expr) => Some(fn_arg_name(sema, &cast_expr.expr()?)),
1112        expr => {
1113            let name_ref = expr
1114                .syntax()
1115                .descendants()
1116                .filter_map(ast::NameRef::cast)
1117                .filter(|name| name.ident_token().is_some())
1118                .last()?;
1119            if let Some(NameRefClass::Definition(Definition::Const(_) | Definition::Static(_), _)) =
1120                NameRefClass::classify(sema, &name_ref)
1121            {
1122                return Some(name_ref.to_string().to_lowercase());
1123            };
1124            Some(to_lower_snake_case(&name_ref.to_string()))
1125        }
1126    })();
1127    match name {
1128        Some(mut name) if name.starts_with(|c: char| c.is_ascii_digit()) => {
1129            name.insert_str(0, "arg");
1130            name
1131        }
1132        Some(name) => name,
1133        None => "arg".to_owned(),
1134    }
1135}

1161        } else {

1163        }

1168}

1170/// Returns the position inside the current mod or file
1171/// directly after the current block
1172/// We want to write the generated function directly after
1173/// fns, impls or macro calls, but inside mods
1174fn next_space_for_fn_after_call_site(expr: ast::CallableExpr) -> Option<GeneratedFunctionTarget> {
1175    let mut ancestors = expr.syntax().ancestors().peekable();
1176    let mut last_ancestor: Option<SyntaxNode> = None;
1177    while let Some(next_ancestor) = ancestors.next() {
1178        match next_ancestor.kind() {
1179            SyntaxKind::SOURCE_FILE => {
1180                break;
1181            }
1182            SyntaxKind::ITEM_LIST => {
1183                if ancestors.peek().map(|a| a.kind()) == Some(SyntaxKind::MODULE) {
1184                    break;
1185                }
1186            }
1187            _ => {}
1188        }
1189        last_ancestor = Some(next_ancestor);
1190    }
1191    last_ancestor.map(GeneratedFunctionTarget::AfterItem)
1192}

1194fn next_space_for_fn_in_module(
1195    db: &dyn hir::db::HirDatabase,
1196    target_module: hir::Module,
1197) -> (FileId, GeneratedFunctionTarget) {
1198    let module_source = target_module.definition_source(db);
1199    let file = module_source.file_id.original_file(db);
1200    let assist_item = match &module_source.value {
1201        hir::ModuleSource::SourceFile(it) => match it.items().last() {
1202            Some(last_item) => GeneratedFunctionTarget::AfterItem(last_item.syntax().clone()),
1203            None => GeneratedFunctionTarget::AfterItem(it.syntax().clone()),
1204        },
1205        hir::ModuleSource::Module(it) => match it.item_list().and_then(|it| it.items().last()) {
1206            Some(last_item) => GeneratedFunctionTarget::AfterItem(last_item.syntax().clone()),
1207            None => {
1208                let item_list =
1209                    it.item_list().expect("module definition source should have an item list");
1210                GeneratedFunctionTarget::InEmptyItemList(item_list.syntax().clone())
1211            }
1212        },
1213        hir::ModuleSource::BlockExpr(it) => {
1214            if let Some(last_item) =
1215                it.statements().take_while(|stmt| matches!(stmt, ast::Stmt::Item(_))).last()
1216            {
1217                GeneratedFunctionTarget::AfterItem(last_item.syntax().clone())
1218            } else {
1219                GeneratedFunctionTarget::InEmptyItemList(it.syntax().clone())
1220            }
1221        }
1222    };
1223
1224    (file.file_id(db), assist_item)
1225}

1230    Crate,
1231    Pub,

1234fn calculate_necessary_visibility(
1235    current_module: Module,
1236    target_module: Module,
1237    ctx: &AssistContext<'_>,
1238) -> Visibility {
1239    let db = ctx.db();
1240    let current_module = current_module.nearest_non_block_module(db);
1241    let target_module = target_module.nearest_non_block_module(db);
1242
1243    if target_module.krate(ctx.db()) != current_module.krate(ctx.db()) {
1244        Visibility::Pub
1245    } else if current_module.path_to_root(db).contains(&target_module) {
1246        Visibility::None
1247    } else {
1248        Visibility::Crate
1249    }
1250}

1252// This is never intended to be used as a generic graph structure. If there's ever another need of
1253// graph algorithm, consider adding a library for that (and replace the following).
1254/// Minimally implemented directed graph structure represented by adjacency list.
1255struct Graph {
1256    edges: Vec<Vec<usize>>,
1257}
1258
1259impl Graph {
1260    fn new(node_count: usize) -> Self {
1261        Self { edges: vec![Vec::new(); node_count] }
1262    }
1263
1264    fn add_edge(&mut self, from: usize, to: usize) {
1265        self.edges[from].push(to);
1266    }
1267
1268    fn edges_for(&self, node_idx: usize) -> &[usize] {
1269        &self.edges[node_idx]
1270    }
1271
1272    fn len(&self) -> usize {
1273        self.edges.len()
1274    }
1275
1276    fn compute_reachable_nodes(
1277        &self,
1278        starting_nodes: impl IntoIterator<Item = usize>,
1279    ) -> Vec<bool> {
1280        let mut visitor = Visitor::new(self);
1281        for idx in starting_nodes {
1282            visitor.mark_reachable(idx);
1283        }
1284        visitor.visited
1285    }
1286}
1287
1288struct Visitor<'g> {
1289    graph: &'g Graph,
1290    visited: Vec<bool>,
1291    // Stack is held in this struct so we can reuse its buffer.
1292    stack: Vec<usize>,
1293}
1294
1295impl<'g> Visitor<'g> {
1296    fn new(graph: &'g Graph) -> Self {
1297        let visited = vec![false; graph.len()];
1298        Self { graph, visited, stack: Vec::new() }
1299    }
1300
1301    fn mark_reachable(&mut self, start_idx: usize) {
1302        // non-recursive DFS
1303        stdx::always!(self.stack.is_empty());
1304
1305        self.stack.push(start_idx);
1306        while let Some(idx) = self.stack.pop() {
1307            if !self.visited[idx] {
1308                self.visited[idx] = true;
1309                for &neighbor in self.graph.edges_for(idx) {
1310                    if !self.visited[neighbor] {
1311                        self.stack.push(neighbor);
1312                    }
1313                }
1314            }
1315        }
1316    }
1317}
