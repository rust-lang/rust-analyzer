30// Assist: generate_function
31//
32// Adds a stub function with a signature matching the function under the cursor.
33//
34// ```
35// struct Baz;
36// fn baz() -> Baz { Baz }
37// fn foo() {
38//     bar$0("", baz());
39// }
40//
41// ```
42// ->
43// ```
44// struct Baz;
45// fn baz() -> Baz { Baz }
46// fn foo() {
47//     bar("", baz());
48// }
49//
50// fn bar(arg: &'static str, baz: Baz) ${0:-> _} {
51//     todo!()
52// }
53//
54// ```
55pub(crate) fn generate_function(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
56    gen_fn(acc, ctx).or_else(|| gen_method(acc, ctx))
57}

59fn gen_fn(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
60    let path_expr: ast::PathExpr = ctx.find_node_at_offset()?;
61    let call = path_expr.syntax().parent().and_then(ast::CallExpr::cast)?;
62    let path = path_expr.path()?;
63    let name_ref = path.segment()?.name_ref()?;
64    if ctx.sema.resolve_path(&path).is_some() {
65        // The function call already resolves, no need to add a function
66        return None;
67    }
68
69    let fn_name = &*name_ref.text();
70    let TargetInfo { target_module, adt_info, target, file } =
71        fn_target_info(ctx, path, &call, fn_name)?;
72
73    if let Some(m) = target_module
74        && !is_editable_crate(m.krate(), ctx.db())
75    {
76        return None;
77    }
78
79    let function_builder =
80        FunctionBuilder::from_call(ctx, &call, fn_name, target_module, target, &adt_info)?;
81    let text_range = call.syntax().text_range();
82    let label = format!("Generate {} function", function_builder.fn_name);
83    add_func_to_accumulator(acc, ctx, text_range, function_builder, file, adt_info, label)
84}

104fn fn_target_info(
105    ctx: &AssistContext<'_>,
106    path: ast::Path,
107    call: &CallExpr,
108    fn_name: &str,
109) -> Option<TargetInfo> {
110    match path.qualifier() {
111        Some(qualifier) => match ctx.sema.resolve_path(&qualifier) {
112            Some(hir::PathResolution::Def(hir::ModuleDef::Module(module))) => {
113                get_fn_target_info(ctx, Some(module), call.clone())
114            }
115            Some(hir::PathResolution::Def(hir::ModuleDef::Adt(adt))) => {
116                if let hir::Adt::Enum(_) = adt {
117                    // Don't suggest generating function if the name starts with an uppercase letter
118                    if fn_name.starts_with(char::is_uppercase) {
119                        return None;
120                    }
121                }
122
123                assoc_fn_target_info(ctx, call, adt, fn_name)
124            }
125            Some(hir::PathResolution::SelfType(impl_)) => {
126                let adt = impl_.self_ty(ctx.db()).as_adt()?;
127                assoc_fn_target_info(ctx, call, adt, fn_name)
128            }
129            _ => None,
130        },
131        _ => get_fn_target_info(ctx, None, call.clone()),
132    }
133}

135fn gen_method(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
136    let call: ast::MethodCallExpr = ctx.find_node_at_offset()?;
137    if ctx.sema.resolve_method_call(&call).is_some() {
138        return None;
139    }
140
141    let fn_name = call.name_ref()?;
142    let receiver_ty = ctx.sema.type_of_expr(&call.receiver()?)?.original().strip_references();
143    let adt = receiver_ty.as_adt()?;
144
145    let target_module = adt.module(ctx.sema.db);
146    if !is_editable_crate(target_module.krate(), ctx.db()) {
147        return None;
148    }
149
150    let (impl_, file) = get_adt_source(ctx, &adt, fn_name.text().as_str())?;
151    let target = get_method_target(ctx, &impl_, &adt)?;
152
153    let function_builder = FunctionBuilder::from_method_call(
154        ctx,
155        &call,
156        &fn_name,
157        receiver_ty,
158        target_module,
159        target,
160    )?;
161    let text_range = call.syntax().text_range();
162    let adt_info = AdtInfo::new(adt, impl_.is_some());
163    let label = format!("Generate {} method", function_builder.fn_name);
164    add_func_to_accumulator(acc, ctx, text_range, function_builder, file, Some(adt_info), label)
165}

167fn add_func_to_accumulator(
168    acc: &mut Assists,
169    ctx: &AssistContext<'_>,
170    text_range: TextRange,
171    function_builder: FunctionBuilder,
172    file: FileId,
173    adt_info: Option<AdtInfo>,
174    label: String,
175) -> Option<()> {
176    acc.add(AssistId::generate("generate_function"), label, text_range, |edit| {
177        edit.edit_file(file);
178
179        let target = function_builder.target.clone();
180        let edition = function_builder.target_edition;
181        let func = function_builder.render(ctx.config.snippet_cap, edit);
182
183        if let Some(adt) = adt_info
184            .and_then(|adt_info| if adt_info.impl_exists { None } else { Some(adt_info.adt) })
185        {
186            let name = make::ty_path(make::ext::ident_path(&format!(
187                "{}",
188                adt.name(ctx.db()).display(ctx.db(), edition)
189            )));
190
191            // FIXME: adt may have generic params.
192            let impl_ = make::impl_(None, None, None, name, None, None).clone_for_update();
193
194            func.indent(IndentLevel(1));
195            impl_.get_or_create_assoc_item_list().add_item(func.into());
196            target.insert_impl_at(edit, impl_);
197        } else {
198            target.insert_fn_at(edit, func);
199        }
200    })
201}

203fn get_adt_source(
204    ctx: &AssistContext<'_>,
205    adt: &hir::Adt,
206    fn_name: &str,
207) -> Option<(Option<ast::Impl>, FileId)> {
208    let range = adt.source(ctx.sema.db)?.syntax().original_file_range_rooted(ctx.sema.db);
209
210    let file = ctx.sema.parse(range.file_id);
211    let adt_source =
212        ctx.sema.find_node_at_offset_with_macros(file.syntax(), range.range.start())?;
213    find_struct_impl(ctx, &adt_source, &[fn_name.to_owned()])
214        .map(|impl_| (impl_, range.file_id.file_id(ctx.db())))
215}

232    /// Prepares a generated function that matches `call`.
233    /// The function is generated in `target_module` or next to `call`
234    fn from_call(
235        ctx: &AssistContext<'_>,
236        call: &ast::CallExpr,
237        fn_name: &str,
238        target_module: Option<Module>,
239        target: GeneratedFunctionTarget,
240        adt_info: &Option<AdtInfo>,
241    ) -> Option<Self> {
242        let target_module =
243            target_module.or_else(|| ctx.sema.scope(target.syntax()).map(|it| it.module()))?;
244        let target_edition = target_module.krate().edition(ctx.db());
245
246        let current_module = ctx.sema.scope(call.syntax())?.module();
247        let visibility = calculate_necessary_visibility(current_module, target_module, ctx);
248        let fn_name = make::name(fn_name);
249        let mut necessary_generic_params = FxHashSet::default();
250        let params = fn_args(
251            ctx,
252            target_module,
253            ast::CallableExpr::Call(call.clone()),
254            &mut necessary_generic_params,
255        )?;
256
257        let await_expr = call.syntax().parent().and_then(ast::AwaitExpr::cast);
258        let is_async = await_expr.is_some();
259
260        let ret_type;
261        let should_focus_return_type;
262        let fn_body;
263
264        // If generated function has the name "new" and is an associated function, we generate fn body
265        // as a constructor and assume a "Self" return type.
266        if let Some(body) =
267            make_fn_body_as_new_function(ctx, &fn_name.text(), adt_info, target_edition)
268        {
269            ret_type = Some(make::ret_type(make::ty_path(make::ext::ident_path("Self"))));
270            should_focus_return_type = false;
271            fn_body = body;
272        } else {
273            let expr_for_ret_ty = await_expr.map_or_else(|| call.clone().into(), |it| it.into());
274            (ret_type, should_focus_return_type) = make_return_type(
275                ctx,
276                &expr_for_ret_ty,
277                target_module,
278                &mut necessary_generic_params,
279            );
280            let placeholder_expr = match ctx.config.expr_fill_default {
281                ExprFillDefaultMode::Todo => make::ext::expr_todo(),
282                ExprFillDefaultMode::Underscore => make::ext::expr_underscore(),
283                ExprFillDefaultMode::Default => make::ext::expr_todo(),
284            };
285            fn_body = make::block_expr(vec![], Some(placeholder_expr));
286        };
287
288        let (generic_param_list, where_clause) =
289            fn_generic_params(ctx, necessary_generic_params, &target)?;
290
291        Some(Self {
292            target,
293            fn_name,
294            generic_param_list,
295            where_clause,
296            params,
297            fn_body,
298            ret_type,
299            should_focus_return_type,
300            visibility,
301            is_async,
302            target_edition,
303        })
304    }

361    fn render(self, cap: Option<SnippetCap>, edit: &mut SourceChangeBuilder) -> ast::Fn {
362        let visibility = match self.visibility {
363            Visibility::None => None,
364            Visibility::Crate => Some(make::visibility_pub_crate()),
365            Visibility::Pub => Some(make::visibility_pub()),
366        };
367        let type_params =
368            self.generic_param_list.filter(|list| list.generic_params().next().is_some());
369        let fn_def = make::fn_(
370            None,
371            visibility,
372            self.fn_name,
373            type_params,
374            self.where_clause,
375            self.params,
376            self.fn_body,
377            self.ret_type,
378            self.is_async,
379            false, // FIXME : const and unsafe are not handled yet.
380            false,
381            false,
382        )
383        .clone_for_update();
384
385        let ret_type = fn_def.ret_type();
386        // PANIC: we guarantee we always create a function body with a tail expr
387        let tail_expr = fn_def
388            .body()
389            .expect("generated function should have a body")
390            .tail_expr()
391            .expect("function body should have a tail expression");
392
393        if let Some(cap) = cap {
394            if self.should_focus_return_type {
395                // Focus the return type if there is one
396                match ret_type {
397                    Some(ret_type) => {
398                        edit.add_placeholder_snippet(cap, ret_type);
399                    }
400                    None => {
401                        edit.add_placeholder_snippet(cap, tail_expr);
402                    }
403                }
404            } else {
405                edit.add_placeholder_snippet(cap, tail_expr);
406            }
407        }
408
409        fn_def
410    }

413/// Makes an optional return type along with whether the return type should be focused by the cursor.
414/// If we cannot infer what the return type should be, we create a placeholder type.
415///
416/// The rule for whether we focus a return type or not (and thus focus the function body),
417/// is rather simple:
418/// * If we could *not* infer what the return type should be, focus it (so the user can fill-in
419///   the correct return type).
420/// * If we could infer the return type, don't focus it (and thus focus the function body) so the
421///   user can change the `todo!` function body.
422fn make_return_type(
423    ctx: &AssistContext<'_>,
424    expr: &ast::Expr,
425    target_module: Module,
426    necessary_generic_params: &mut FxHashSet<hir::GenericParam>,
427) -> (Option<ast::RetType>, bool) {
428    let (ret_ty, should_focus_return_type) = {
429        match ctx.sema.type_of_expr(expr).map(TypeInfo::original) {
430            Some(ty) if ty.is_unknown() => (Some(make::ty_placeholder()), true),
431            None => (Some(make::ty_placeholder()), true),
432            Some(ty) if ty.is_unit() => (None, false),
433            Some(ty) => {
434                necessary_generic_params.extend(ty.generic_params(ctx.db()));
435                let rendered = ty.display_source_code(ctx.db(), target_module.into(), true);
436                match rendered {
437                    Ok(rendered) => (Some(make::ty(&rendered)), false),
438                    Err(_) => (Some(make::ty_placeholder()), true),
439                }
440            }
441        }
442    };
443    let ret_type = ret_ty.map(make::ret_type);
444    (ret_type, should_focus_return_type)
445}

447fn make_fn_body_as_new_function(
448    ctx: &AssistContext<'_>,
449    fn_name: &str,
450    adt_info: &Option<AdtInfo>,
451    edition: Edition,
452) -> Option<ast::BlockExpr> {
453    if fn_name != "new" {
454        return None;
455    };
456    let adt_info = adt_info.as_ref()?;
457
458    let path_self = make::ext::ident_path("Self");
459    let placeholder_expr = match ctx.config.expr_fill_default {
460        ExprFillDefaultMode::Todo => make::ext::expr_todo(),
461        ExprFillDefaultMode::Underscore => make::ext::expr_underscore(),
462        ExprFillDefaultMode::Default => make::ext::expr_todo(),
463    };
464    let tail_expr = if let Some(strukt) = adt_info.adt.as_struct() {
465        match strukt.kind(ctx.db()) {
466            StructKind::Record => {
467                let fields = strukt
468                    .fields(ctx.db())
469                    .iter()
470                    .map(|field| {
471                        make::record_expr_field(
472                            make::name_ref(&format!(
473                                "{}",
474                                field.name(ctx.db()).display(ctx.db(), edition)
475                            )),
476                            Some(placeholder_expr.clone()),
477                        )
478                    })
479                    .collect::<Vec<_>>();
480
481                make::record_expr(path_self, make::record_expr_field_list(fields)).into()
482            }
483            StructKind::Tuple => {
484                let args = strukt
485                    .fields(ctx.db())
486                    .iter()
487                    .map(|_| placeholder_expr.clone())
488                    .collect::<Vec<_>>();
489
490                make::expr_call(make::expr_path(path_self), make::arg_list(args)).into()
491            }
492            StructKind::Unit => make::expr_path(path_self),
493        }
494    } else {
495        placeholder_expr
496    };
497
498    let fn_body = make::block_expr(vec![], Some(tail_expr));
499    Some(fn_body)
500}

540fn assoc_fn_target_info(
541    ctx: &AssistContext<'_>,
542    call: &CallExpr,
543    adt: hir::Adt,
544    fn_name: &str,
545) -> Option<TargetInfo> {
546    let current_module = ctx.sema.scope(call.syntax())?.module();
547    let module = adt.module(ctx.sema.db);
548    let target_module = if current_module == module { None } else { Some(module) };
549    if current_module.krate() != module.krate() {

554    let adt_info = AdtInfo::new(adt, impl_.is_some());
555    Some(TargetInfo::new(target_module, Some(adt_info), target, file))

558#[derive(Clone)]
559enum GeneratedFunctionTarget {
560    AfterItem(SyntaxNode),
561    InEmptyItemList(SyntaxNode),
562    InImpl(ast::Impl),
563}
564
565impl GeneratedFunctionTarget {
566    fn syntax(&self) -> &SyntaxNode {
567        match self {
568            GeneratedFunctionTarget::AfterItem(it) => it,
569            GeneratedFunctionTarget::InEmptyItemList(it) => it,
570            GeneratedFunctionTarget::InImpl(it) => it.syntax(),
571        }
572    }
573
574    fn parent(&self) -> SyntaxNode {
575        match self {
576            GeneratedFunctionTarget::AfterItem(it) => it.parent().expect("item without parent"),
577            GeneratedFunctionTarget::InEmptyItemList(it) => it.clone(),
578            GeneratedFunctionTarget::InImpl(it) => it.syntax().clone(),
579        }
580    }
581
582    fn insert_impl_at(&self, edit: &mut SourceChangeBuilder, impl_: ast::Impl) {
583        match self {
584            GeneratedFunctionTarget::AfterItem(item) => {
585                let item = edit.make_syntax_mut(item.clone());
586                let position = if item.parent().is_some() {
587                    ted::Position::after(&item)
588                } else {
589                    ted::Position::first_child_of(&item)
590                };
591
592                let indent = IndentLevel::from_node(&item);
593                let leading_ws = make::tokens::whitespace(&format!("\n{indent}"));
594                impl_.indent(indent);
595
596                ted::insert_all(position, vec![leading_ws.into(), impl_.syntax().clone().into()]);
597            }
598            GeneratedFunctionTarget::InEmptyItemList(item_list) => {
599                let item_list = edit.make_syntax_mut(item_list.clone());
600                let insert_after =
601                    item_list.children_with_tokens().find_or_first(|child| child.kind() == T!['{']);
602                let position = match insert_after {
603                    Some(child) => ted::Position::after(child),
604                    None => ted::Position::first_child_of(&item_list),
605                };
606
607                let indent = IndentLevel::from_node(&item_list);
608                let leading_indent = indent + 1;
609                let leading_ws = make::tokens::whitespace(&format!("\n{leading_indent}"));
610                impl_.indent(indent);
611
612                ted::insert_all(position, vec![leading_ws.into(), impl_.syntax().clone().into()]);
613            }
614            GeneratedFunctionTarget::InImpl(_) => {
615                unreachable!("can't insert an impl inside an impl")
616            }
617        }
618    }
619
620    fn insert_fn_at(&self, edit: &mut SourceChangeBuilder, func: ast::Fn) {
621        match self {
622            GeneratedFunctionTarget::AfterItem(item) => {
623                let item = edit.make_syntax_mut(item.clone());
624                let position = if item.parent().is_some() {
625                    ted::Position::after(&item)
626                } else {
627                    ted::Position::first_child_of(&item)
628                };
629
630                let indent = IndentLevel::from_node(&item);
631                let leading_ws = make::tokens::whitespace(&format!("\n\n{indent}"));
632                func.indent(indent);
633
634                ted::insert_all_raw(
635                    position,
636                    vec![leading_ws.into(), func.syntax().clone().into()],
637                );
638            }
639            GeneratedFunctionTarget::InEmptyItemList(item_list) => {
640                let item_list = edit.make_syntax_mut(item_list.clone());
641                let insert_after =
642                    item_list.children_with_tokens().find_or_first(|child| child.kind() == T!['{']);
643                let position = match insert_after {
644                    Some(child) => ted::Position::after(child),
645                    None => ted::Position::first_child_of(&item_list),
646                };
647
648                let indent = IndentLevel::from_node(&item_list);
649                let leading_indent = indent + 1;
650                let leading_ws = make::tokens::whitespace(&format!("\n{leading_indent}"));
651                let trailing_ws = make::tokens::whitespace(&format!("\n{indent}"));
652                func.indent(leading_indent);
653
654                ted::insert_all(
655                    position,
656                    vec![leading_ws.into(), func.syntax().clone().into(), trailing_ws.into()],
657                );
658            }
659            GeneratedFunctionTarget::InImpl(impl_) => {
660                let impl_ = edit.make_mut(impl_.clone());
661
662                let leading_indent = impl_.indent_level() + 1;
663                func.indent(leading_indent);
664
665                impl_.get_or_create_assoc_item_list().add_item(func.into());
666            }
667        }
668    }
669}

705        params,

709/// Gets parameter bounds and where predicates in scope and filters out irrelevant ones. Returns
710/// `None` when it fails to get scope information.
711///
712/// See comment on `filter_unnecessary_bounds()` for what bounds we consider relevant.
713///
714/// NOTE: Generic parameters returned from this function may cause name clash at `target`. We don't
715/// currently do anything about it because it's actually easy to resolve it after the assist: just
716/// use the Rename functionality.
717fn fn_generic_params(
718    ctx: &AssistContext<'_>,
719    necessary_params: FxHashSet<hir::GenericParam>,
720    target: &GeneratedFunctionTarget,
721) -> Option<(Option<ast::GenericParamList>, Option<ast::WhereClause>)> {
722    if necessary_params.is_empty() {
723        // Not really needed but fast path.
724        return Some((None, None));
725    }
726
727    // 1. Get generic parameters (with bounds) and where predicates in scope.
728    let (generic_params, where_preds) = params_and_where_preds_in_scope(ctx);
729
730    // 2. Extract type parameters included in each bound.
731    let mut generic_params = generic_params
732        .into_iter()
733        .filter_map(|it| compute_contained_params_in_generic_param(ctx, it))
734        .collect();
735    let mut where_preds = where_preds
736        .into_iter()
737        .filter_map(|it| compute_contained_params_in_where_pred(ctx, it))
738        .collect();
739
740    // 3. Filter out unnecessary bounds.
741    filter_unnecessary_bounds(&mut generic_params, &mut where_preds, necessary_params);
742    filter_bounds_in_scope(&mut generic_params, &mut where_preds, ctx, target);
743
744    let generic_params: Vec<ast::GenericParam> =
745        generic_params.into_iter().map(|it| it.node.clone_for_update()).collect();
746    let where_preds: Vec<ast::WherePred> =
747        where_preds.into_iter().map(|it| it.node.clone_for_update()).collect();
748
749    let (generic_params, where_preds): (Vec<ast::GenericParam>, Vec<ast::WherePred>) =
750        if let Some(param) = generic_params.first()
751            && let source_scope = ctx.sema.scope(param.syntax())?
752            && let target_scope = ctx.sema.scope(&target.parent())?
753            && source_scope.module() != target_scope.module()
754        {
755            // 4. Rewrite paths
756            let transform = PathTransform::generic_transformation(&target_scope, &source_scope);
757            let generic_params = generic_params.iter().map(|it| it.syntax());
758            let where_preds = where_preds.iter().map(|it| it.syntax());
759            transform
760                .apply_all(generic_params.chain(where_preds))
761                .into_iter()
762                .filter_map(|it| {
763                    if let Some(it) = ast::GenericParam::cast(it.clone()) {
764                        Some(either::Either::Left(it))
765                    } else {
766                        ast::WherePred::cast(it).map(either::Either::Right)
767                    }
768                })
769                .partition_map(|it| it)
770        } else {
771            (generic_params, where_preds)
772        };
773
774    let generic_param_list = make::generic_param_list(generic_params);
775    let where_clause =
776        if where_preds.is_empty() { None } else { Some(make::where_clause(where_preds)) };
777
778    Some((Some(generic_param_list), where_clause))
779}

781fn params_and_where_preds_in_scope(
782    ctx: &AssistContext<'_>,
783) -> (Vec<ast::GenericParam>, Vec<ast::WherePred>) {
784    let Some(body) = containing_body(ctx) else {
785        return Default::default();
786    };
787
788    let mut generic_params = Vec::new();
789    let mut where_clauses = Vec::new();
790
791    // There are two items where generic parameters currently in scope may be declared: the item
792    // the cursor is at, and its parent (if any).
793    //
794    // We handle parent first so that their generic parameters appear first in the generic
795    // parameter list of the function we're generating.
796    let db = ctx.db();
797    if let Some(parent) = body.as_assoc_item(db).map(|it| it.container(db)) {
798        match parent {
799            hir::AssocItemContainer::Impl(it) => {
800                let (params, clauses) = get_bounds_in_scope(ctx, it);
801                generic_params.extend(params);
802                where_clauses.extend(clauses);
803            }
804            hir::AssocItemContainer::Trait(it) => {
805                let (params, clauses) = get_bounds_in_scope(ctx, it);
806                generic_params.extend(params);
807                where_clauses.extend(clauses);
808            }
809        }
810    }
811
812    // Other defs with body may inherit generic parameters from its parent, but never have their
813    // own generic parameters.
814    if let hir::DefWithBody::Function(it) = body {
815        let (params, clauses) = get_bounds_in_scope(ctx, it);
816        generic_params.extend(params);
817        where_clauses.extend(clauses);
818    }
819
820    (generic_params, where_clauses)
821}

823fn containing_body(ctx: &AssistContext<'_>) -> Option<hir::DefWithBody> {
824    let item: ast::Item = ctx.find_node_at_offset()?;
825    let def = match item {
826        ast::Item::Fn(it) => ctx.sema.to_def(&it)?.into(),
827        ast::Item::Const(it) => ctx.sema.to_def(&it)?.into(),
828        ast::Item::Static(it) => ctx.sema.to_def(&it)?.into(),
829        _ => return None,
830    };
831    Some(def)
832}

834fn get_bounds_in_scope<D>(
835    ctx: &AssistContext<'_>,
836    def: D,
837) -> (impl Iterator<Item = ast::GenericParam>, impl Iterator<Item = ast::WherePred>)
838where
839    D: HasSource,
840    D::Ast: HasGenericParams,
841{
842    // This function should be only called with `Impl`, `Trait`, or `Function`, for which it's
843    // infallible to get source ast.
844    let node = ctx.sema.source(def).expect("definition's source couldn't be found").value;
845    let generic_params = node.generic_param_list().into_iter().flat_map(|it| it.generic_params());
846    let where_clauses = node.where_clause().into_iter().flat_map(|it| it.predicates());
847    (generic_params, where_clauses)

851struct ParamBoundWithParams {
852    node: ast::GenericParam,
853    /// Generic parameter `node` introduces.
854    ///
855    /// ```text
856    /// impl<T> S<T> {
857    ///     fn f<U: Trait<T>>() {}
858    ///          ^ this
859    /// }
860    /// ```
861    ///
862    /// `U` in this example.
863    self_ty_param: hir::GenericParam,
864    /// Generic parameters contained in the trait reference of this bound.
865    ///
866    /// ```text
867    /// impl<T> S<T> {
868    ///     fn f<U: Trait<T>>() {}
869    ///             ^^^^^^^^ params in this part
870    /// }
871    /// ```
872    ///
873    /// `T` in this example.
874    other_params: FxHashSet<hir::GenericParam>,
875}

877#[derive(Debug)]
878struct WherePredWithParams {
879    node: ast::WherePred,
880    /// Generic parameters contained in the "self type" of this where predicate.
881    ///
882    /// ```text
883    /// Struct<T, U>: Trait<T, Assoc = V>,
884    /// ^^^^^^^^^^^^ params in this part
885    /// ```
886    ///
887    /// `T` and `U` in this example.
888    self_ty_params: FxHashSet<hir::GenericParam>,
889    /// Generic parameters contained in the trait reference of this where predicate.
890    ///
891    /// ```text
892    /// Struct<T, U>: Trait<T, Assoc = V>,
893    ///               ^^^^^^^^^^^^^^^^^^^ params in this part
894    /// ```
895    ///
896    /// `T` and `V` in this example.
897    other_params: FxHashSet<hir::GenericParam>,
898}

900fn compute_contained_params_in_generic_param(
901    ctx: &AssistContext<'_>,
902    node: ast::GenericParam,
903) -> Option<ParamBoundWithParams> {
904    match &node {
905        ast::GenericParam::TypeParam(ty) => {
906            let self_ty_param = ctx.sema.to_def(ty)?.into();
907
908            let other_params = ty
909                .type_bound_list()
910                .into_iter()
911                .flat_map(|it| it.bounds())
912                .flat_map(|bound| bound.syntax().descendants())
913                .filter_map(|node| filter_generic_params(ctx, node))
914                .collect();
915
916            Some(ParamBoundWithParams { node, self_ty_param, other_params })
917        }
918        ast::GenericParam::ConstParam(ct) => {
919            let self_ty_param = ctx.sema.to_def(ct)?.into();
920            Some(ParamBoundWithParams { node, self_ty_param, other_params: FxHashSet::default() })
921        }
922        ast::GenericParam::LifetimeParam(_) => {
923            // FIXME: It might be a good idea to handle lifetime parameters too.
924            None
925        }
926    }
927}

929fn compute_contained_params_in_where_pred(
930    ctx: &AssistContext<'_>,
931    node: ast::WherePred,
932) -> Option<WherePredWithParams> {
933    let self_ty = node.ty()?;
934    let bound_list = node.type_bound_list()?;
935
936    let self_ty_params = self_ty
937        .syntax()
938        .descendants()
939        .filter_map(|node| filter_generic_params(ctx, node))
940        .collect();
941
942    let other_params = bound_list
943        .bounds()
944        .flat_map(|bound| bound.syntax().descendants())
945        .filter_map(|node| filter_generic_params(ctx, node))
946        .collect();
947
948    Some(WherePredWithParams { node, self_ty_params, other_params })
949}

960/// Filters out irrelevant bounds from `generic_params` and `where_preds`.
961///
962/// Say we have a trait bound `Struct<T>: Trait<U>`. Given `necessary_params`, when is it relevant
963/// and when not? Some observations:
964/// - When `necessary_params` contains `T`, it's likely that we want this bound, but now we have
965///   an extra param to consider: `U`.
966/// - On the other hand, when `necessary_params` contains `U` (but not `T`), then it's unlikely
967///   that we want this bound because it doesn't really constrain `U`.
968///
969/// (FIXME?: The latter clause might be overstating. We may want to include the bound if the self
970/// type does *not* include generic params at all - like `Option<i32>: From<U>`)
971///
972/// Can we make this a bit more formal? Let's define "dependency" between generic parameters and
973/// trait bounds:
974/// - A generic parameter `T` depends on a trait bound if `T` appears in the self type (i.e. left
975///   part) of the bound.
976/// - A trait bound depends on a generic parameter `T` if `T` appears in the bound.
977///
978/// Using the notion, what we want is all the bounds that params in `necessary_params`
979/// *transitively* depend on!
980///
981/// Now it's not hard to solve: we build a dependency graph and compute all reachable nodes from
982/// nodes that represent params in `necessary_params` by usual and boring DFS.
983///
984/// The time complexity is O(|generic_params| + |where_preds| + |necessary_params|).
985fn filter_unnecessary_bounds(
986    generic_params: &mut Vec<ParamBoundWithParams>,
987    where_preds: &mut Vec<WherePredWithParams>,
988    necessary_params: FxHashSet<hir::GenericParam>,
989) {
990    // All `self_ty_param` should be unique as they were collected from `ast::GenericParamList`s.
991    let param_map: FxHashMap<hir::GenericParam, usize> =
992        generic_params.iter().map(|it| it.self_ty_param).zip(0..).collect();
993    let param_count = param_map.len();
994    let generic_params_upper_bound = param_count + generic_params.len();
995    let node_count = generic_params_upper_bound + where_preds.len();
996
997    // | node index range                        | what the node represents |
998    // |-----------------------------------------|--------------------------|
999    // | 0..param_count                          | generic parameter        |
1000    // | param_count..generic_params_upper_bound | `ast::GenericParam`      |
1001    // | generic_params_upper_bound..node_count  | `ast::WherePred`         |
1002    let mut graph = Graph::new(node_count);
1003    for (pred, pred_idx) in generic_params.iter().zip(param_count..) {
1004        let param_idx = param_map[&pred.self_ty_param];
1005        graph.add_edge(param_idx, pred_idx);
1006        graph.add_edge(pred_idx, param_idx);
1007
1008        for param in &pred.other_params {
1009            let param_idx = param_map[param];
1010            graph.add_edge(pred_idx, param_idx);
1011        }
1012    }
1013    for (pred, pred_idx) in where_preds.iter().zip(generic_params_upper_bound..) {
1014        for param in &pred.self_ty_params {
1015            let param_idx = param_map[param];
1016            graph.add_edge(param_idx, pred_idx);
1017            graph.add_edge(pred_idx, param_idx);
1018        }
1019        for param in &pred.other_params {
1020            let param_idx = param_map[param];
1021            graph.add_edge(pred_idx, param_idx);
1022        }
1023    }
1024
1025    let starting_nodes = necessary_params.iter().flat_map(|param| param_map.get(param).copied());
1026    let reachable = graph.compute_reachable_nodes(starting_nodes);
1027
1028    // Not pretty, but effective. If only there were `Vec::retain_index()`...
1029    let mut idx = param_count;
1030    generic_params.retain(|_| {
1031        idx += 1;
1032        reachable[idx - 1]
1033    });
1034    stdx::always!(idx == generic_params_upper_bound, "inconsistent index");
1035    where_preds.retain(|_| {
1036        idx += 1;
1037        reachable[idx - 1]
1038    });
1039}

1041/// Filters out bounds from impl if we're generating the function into the same impl we're
1042/// generating from.
1043fn filter_bounds_in_scope(
1044    generic_params: &mut Vec<ParamBoundWithParams>,
1045    where_preds: &mut Vec<WherePredWithParams>,
1046    ctx: &AssistContext<'_>,
1047    target: &GeneratedFunctionTarget,
1048) -> Option<()> {
1049    let target_impl = target.parent().ancestors().find_map(ast::Impl::cast)?;
1050    let target_impl = ctx.sema.to_def(&target_impl)?;
1051    // It's sufficient to test only the first element of `generic_params` because of the order of
1052    // insertion (see `params_and_where_preds_in_scope()`).
1053    let def = generic_params.first()?.self_ty_param.parent();
1054    if def != hir::GenericDef::Impl(target_impl) {
1055        return None;
1056    }
1057
1058    // Now we know every element that belongs to an impl would be in scope at `target`, we can
1059    // filter them out just by looking at their parent.
1060    generic_params.retain(|it| !matches!(it.self_ty_param.parent(), hir::GenericDef::Impl(_)));
1061    where_preds.retain(|it| {
1062        it.node.syntax().parent().and_then(|it| it.parent()).and_then(ast::Impl::cast).is_none()
1063    });
1064
1065    Some(())
1066}

1068/// Makes duplicate argument names unique by appending incrementing numbers.
1069///
1070/// ```ignore
1071/// let mut names: Vec<String> =
1072///     vec!["foo".into(), "foo".into(), "bar".into(), "baz".into(), "bar".into()];
1073/// deduplicate_arg_names(&mut names);
1074/// let expected: Vec<String> =
1075///     vec!["foo_1".into(), "foo_2".into(), "bar_1".into(), "baz".into(), "bar_2".into()];
1076/// assert_eq!(names, expected);
1077/// ```
1078fn deduplicate_arg_names(arg_names: &mut [String]) {
1079    let mut arg_name_counts = FxHashMap::default();
1080    for name in arg_names.iter() {
1081        *arg_name_counts.entry(name).or_insert(0) += 1;
1082    }
1083    let duplicate_arg_names: FxHashSet<String> = arg_name_counts
1084        .into_iter()
1085        .filter(|(_, count)| *count >= 2)
1086        .map(|(name, _)| name.clone())
1087        .collect();
1088
1089    let mut counter_per_name = FxHashMap::default();
1090    for arg_name in arg_names.iter_mut() {
1091        if duplicate_arg_names.contains(arg_name) {
1092            let counter = counter_per_name.entry(arg_name.clone()).or_insert(1);
1093            arg_name.push('_');
1094            arg_name.push_str(&counter.to_string());
1095            *counter += 1;
1096        }
1097    }
1098}

1102        ast::Expr::CastExpr(cast_expr) => Some(fn_arg_name(sema, &cast_expr.expr()?)),

1115            Some(to_lower_snake_case(&name_ref.to_string()))
1116        }

1118    match name {
1119        Some(mut name) if name.starts_with(|c: char| c.is_ascii_digit()) => {
1120            name.insert_str(0, "arg");
1121            name
1122        }
1123        Some(name) => name,
1124        None => "arg".to_owned(),
1125    }
1126}

1165/// Returns the position inside the current mod or file
1166/// directly after the current block
1167/// We want to write the generated function directly after
1168/// fns, impls or macro calls, but inside mods
1169fn next_space_for_fn_after_call_site(expr: ast::CallableExpr) -> Option<GeneratedFunctionTarget> {
1170    let mut ancestors = expr.syntax().ancestors().peekable();
1171    let mut last_ancestor: Option<SyntaxNode> = None;
1172    while let Some(next_ancestor) = ancestors.next() {
1173        match next_ancestor.kind() {
1174            SyntaxKind::SOURCE_FILE => {
1175                break;
1176            }
1177            SyntaxKind::ITEM_LIST => {
1178                if ancestors.peek().map(|a| a.kind()) == Some(SyntaxKind::MODULE) {
1179                    break;
1180                }
1181            }
1182            _ => {}
1183        }
1184        last_ancestor = Some(next_ancestor);
1185    }
1186    last_ancestor.map(GeneratedFunctionTarget::AfterItem)
1187}

1189fn next_space_for_fn_in_module(
1190    db: &dyn hir::db::HirDatabase,
1191    target_module: hir::Module,
1192) -> (FileId, GeneratedFunctionTarget) {
1193    let module_source = target_module.definition_source(db);
1194    let file = module_source.file_id.original_file(db);
1195    let assist_item = match &module_source.value {
1196        hir::ModuleSource::SourceFile(it) => match it.items().last() {
1197            Some(last_item) => GeneratedFunctionTarget::AfterItem(last_item.syntax().clone()),
1198            None => GeneratedFunctionTarget::AfterItem(it.syntax().clone()),
1199        },
1200        hir::ModuleSource::Module(it) => match it.item_list().and_then(|it| it.items().last()) {
1201            Some(last_item) => GeneratedFunctionTarget::AfterItem(last_item.syntax().clone()),
1202            None => {
1203                let item_list =
1204                    it.item_list().expect("module definition source should have an item list");
1205                GeneratedFunctionTarget::InEmptyItemList(item_list.syntax().clone())
1206            }
1207        },
1208        hir::ModuleSource::BlockExpr(it) => {
1209            if let Some(last_item) =
1210                it.statements().take_while(|stmt| matches!(stmt, ast::Stmt::Item(_))).last()
1211            {
1212                GeneratedFunctionTarget::AfterItem(last_item.syntax().clone())
1213            } else {
1214                GeneratedFunctionTarget::InEmptyItemList(it.syntax().clone())

1229fn calculate_necessary_visibility(
1230    current_module: Module,
1231    target_module: Module,
1232    ctx: &AssistContext<'_>,
1233) -> Visibility {
1234    let db = ctx.db();
1235    let current_module = current_module.nearest_non_block_module(db);
1236    let target_module = target_module.nearest_non_block_module(db);
1237
1238    if target_module.krate() != current_module.krate() {
1239        Visibility::Pub
1240    } else if current_module.path_to_root(db).contains(&target_module) {
1241        Visibility::None
1242    } else {
1243        Visibility::Crate
1244    }
1245}

1247// This is never intended to be used as a generic graph structure. If there's ever another need of
1248// graph algorithm, consider adding a library for that (and replace the following).
1249/// Minimally implemented directed graph structure represented by adjacency list.
1250struct Graph {
1251    edges: Vec<Vec<usize>>,
1252}
1253
1254impl Graph {
1255    fn new(node_count: usize) -> Self {
1256        Self { edges: vec![Vec::new(); node_count] }
1257    }
1258
1259    fn add_edge(&mut self, from: usize, to: usize) {
1260        self.edges[from].push(to);
1261    }
1262
1263    fn edges_for(&self, node_idx: usize) -> &[usize] {
1264        &self.edges[node_idx]
1265    }
1266
1267    fn len(&self) -> usize {
1268        self.edges.len()
1269    }
1270
1271    fn compute_reachable_nodes(
1272        &self,
1273        starting_nodes: impl IntoIterator<Item = usize>,
1274    ) -> Vec<bool> {
1275        let mut visitor = Visitor::new(self);
1276        for idx in starting_nodes {
1277            visitor.mark_reachable(idx);
1278        }
1279        visitor.visited
1280    }
1281}
1282
1283struct Visitor<'g> {
1284    graph: &'g Graph,
1285    visited: Vec<bool>,
1286    // Stack is held in this struct so we can reuse its buffer.
1287    stack: Vec<usize>,
1288}
1289
1290impl<'g> Visitor<'g> {
1291    fn new(graph: &'g Graph) -> Self {
1292        let visited = vec![false; graph.len()];
1293        Self { graph, visited, stack: Vec::new() }
1294    }
1295
1296    fn mark_reachable(&mut self, start_idx: usize) {
1297        // non-recursive DFS
1298        stdx::always!(self.stack.is_empty());
1299
1300        self.stack.push(start_idx);
1301        while let Some(idx) = self.stack.pop() {
1302            if !self.visited[idx] {
1303                self.visited[idx] = true;
1304                for &neighbor in self.graph.edges_for(idx) {
1305                    if !self.visited[neighbor] {
1306                        self.stack.push(neighbor);
1307                    }
1308                }
1309            }
1310        }
1311    }
1312}
