20// Assist: destructure_tuple_binding
21//
22// Destructures a tuple binding in place.
23//
24// ```
25// fn main() {
26//     let $0t = (1,2);
27//     let v = t.0;
28// }
29// ```
30// ->
31// ```
32// fn main() {
33//     let ($0_0, _1) = (1,2);
34//     let v = _0;
35// }
36// ```
37pub(crate) fn destructure_tuple_binding(acc: &mut Assists, ctx: &AssistContext<'_>) -> Option<()> {
38    destructure_tuple_binding_impl(acc, ctx, false)
39}
40
41// And when `with_sub_pattern` enabled (currently disabled):
42// Assist: destructure_tuple_binding_in_sub_pattern
43//
44// Destructures tuple items in sub-pattern (after `@`).
45//
46// ```
47// fn main() {
48//     let $0t = (1,2);
49//     let v = t.0;
50// }
51// ```
52// ->
53// ```
54// fn main() {
55//     let t @ ($0_0, _1) = (1,2);
56//     let v = _0;
57// }
58// ```
59pub(crate) fn destructure_tuple_binding_impl(
60    acc: &mut Assists,
61    ctx: &AssistContext<'_>,
62    with_sub_pattern: bool,
63) -> Option<()> {
64    let ident_pat = ctx.find_node_at_offset::<ast::IdentPat>()?;
65    let data = collect_data(ident_pat, ctx)?;
66
67    if with_sub_pattern {
68        acc.add(
69            AssistId::refactor_rewrite("destructure_tuple_binding_in_sub_pattern"),
70            "Destructure tuple in sub-pattern",
71            data.ident_pat.syntax().text_range(),
72            |edit| destructure_tuple_edit_impl(ctx, edit, &data, true),
73        );
74    }
75
76    acc.add(
77        AssistId::refactor_rewrite("destructure_tuple_binding"),
78        if with_sub_pattern { "Destructure tuple in place" } else { "Destructure tuple" },
79        data.ident_pat.syntax().text_range(),
80        |edit| destructure_tuple_edit_impl(ctx, edit, &data, false),
81    );
82
83    Some(())
84}

86fn destructure_tuple_edit_impl(
87    ctx: &AssistContext<'_>,
88    edit: &mut SourceChangeBuilder,
89    data: &TupleData,
90    in_sub_pattern: bool,
91) {
92    let mut syntax_editor = edit.make_editor(data.ident_pat.syntax());
93    let syntax_factory = SyntaxFactory::with_mappings();
94
95    let assignment_edit =
96        edit_tuple_assignment(ctx, edit, &mut syntax_editor, &syntax_factory, data, in_sub_pattern);
97    let current_file_usages_edit = edit_tuple_usages(data, ctx, &syntax_factory, in_sub_pattern);
98
99    assignment_edit.apply(&mut syntax_editor, &syntax_factory);
100    if let Some(usages_edit) = current_file_usages_edit {
101        usages_edit.into_iter().for_each(|usage_edit| usage_edit.apply(edit, &mut syntax_editor))
102    }
103
104    syntax_editor.add_mappings(syntax_factory.finish_with_mappings());
105    edit.add_file_edits(ctx.vfs_file_id(), syntax_editor);
106}

108fn collect_data(ident_pat: IdentPat, ctx: &AssistContext<'_>) -> Option<TupleData> {
109    if ident_pat.at_token().is_some() {
110        // Cannot destructure pattern with sub-pattern:
111        // Only IdentPat can have sub-pattern,
112        // but not TuplePat (`(a,b)`).
113        cov_mark::hit!(destructure_tuple_subpattern);
114        return None;
115    }
116
117    let ty = ctx.sema.type_of_binding_in_pat(&ident_pat)?;
118    let ref_type = if ty.is_mutable_reference() {
119        Some(RefType::Mutable)
120    } else if ty.is_reference() {
121        Some(RefType::ReadOnly)
122    } else {
123        None
124    };
125    // might be reference
126    let ty = ty.strip_references();
127    // must be tuple
128    let field_types = ty.tuple_fields(ctx.db());
129    if field_types.is_empty() {
130        cov_mark::hit!(destructure_tuple_no_tuple);
131        return None;
132    }
133
134    let usages = ctx.sema.to_def(&ident_pat).and_then(|def| {
135        Definition::Local(def)
136            .usages(&ctx.sema)
137            .in_scope(&SearchScope::single_file(ctx.file_id()))
138            .all()
139            .iter()
140            .next()
141            .map(|(_, refs)| refs.to_vec())
142    });
143
144    let mut name_generator =
145        suggest_name::NameGenerator::new_from_scope_locals(ctx.sema.scope(ident_pat.syntax()));
146
147    let field_names = field_types
148        .into_iter()
149        .enumerate()
150        .map(|(id, ty)| {
151            match name_generator.for_type(&ty, ctx.db(), ctx.edition()) {
152                Some(name) => name,
153                None => name_generator.suggest_name(&format!("_{id}")),
154            }
155            .to_string()
156        })
157        .collect::<Vec<_>>();
158
159    Some(TupleData { ident_pat, ref_type, field_names, usages })
160}

162enum RefType {
163    ReadOnly,
164    Mutable,
165}
166struct TupleData {
167    ident_pat: IdentPat,
168    ref_type: Option<RefType>,
169    field_names: Vec<String>,
170    usages: Option<Vec<FileReference>>,

172fn edit_tuple_assignment(
173    ctx: &AssistContext<'_>,
174    edit: &mut SourceChangeBuilder,
175    editor: &mut SyntaxEditor,
176    make: &SyntaxFactory,
177    data: &TupleData,
178    in_sub_pattern: bool,
179) -> AssignmentEdit {
180    let tuple_pat = {
181        let original = &data.ident_pat;
182        let is_ref = original.ref_token().is_some();
183        let is_mut = original.mut_token().is_some();
184        let fields = data
185            .field_names
186            .iter()
187            .map(|name| ast::Pat::from(make.ident_pat(is_ref, is_mut, make.name(name))));
188        make.tuple_pat(fields)
189    };
190    let is_shorthand_field = data
191        .ident_pat
192        .name()
193        .as_ref()
194        .and_then(ast::RecordPatField::for_field_name)
195        .is_some_and(|field| field.colon_token().is_none());
196
197    if let Some(cap) = ctx.config.snippet_cap {
198        // place cursor on first tuple name
199        if let Some(ast::Pat::IdentPat(first_pat)) = tuple_pat.fields().next() {
200            let annotation = edit.make_tabstop_before(cap);
201            editor.add_annotation(
202                first_pat.name().expect("first ident pattern should have a name").syntax(),
203                annotation,
204            );
205        }
206    }
207
208    AssignmentEdit {
209        ident_pat: data.ident_pat.clone(),
210        tuple_pat,
211        in_sub_pattern,
212        is_shorthand_field,
213    }
214}
215struct AssignmentEdit {
216    ident_pat: ast::IdentPat,
217    tuple_pat: ast::TuplePat,
218    in_sub_pattern: bool,
219    is_shorthand_field: bool,
220}
221
222impl AssignmentEdit {
223    fn apply(self, syntax_editor: &mut SyntaxEditor, syntax_mapping: &SyntaxFactory) {
224        // with sub_pattern: keep original tuple and add subpattern: `tup @ (_0, _1)`
225        if self.in_sub_pattern {
226            self.ident_pat.set_pat_with_editor(
227                Some(self.tuple_pat.into()),
228                syntax_editor,
229                syntax_mapping,
230            )
231        } else if self.is_shorthand_field {
232            syntax_editor.insert(Position::after(self.ident_pat.syntax()), self.tuple_pat.syntax());
233            syntax_editor
234                .insert(Position::after(self.ident_pat.syntax()), syntax_mapping.whitespace(" "));
235            syntax_editor
236                .insert(Position::after(self.ident_pat.syntax()), syntax_mapping.token(T![:]));
237        } else {
238            syntax_editor.replace(self.ident_pat.syntax(), self.tuple_pat.syntax())
239        }
240    }
241}

243fn edit_tuple_usages(
244    data: &TupleData,
245    ctx: &AssistContext<'_>,
246    make: &SyntaxFactory,
247    in_sub_pattern: bool,
248) -> Option<Vec<EditTupleUsage>> {
249    // We need to collect edits first before actually applying them
250    // as mapping nodes to their mutable node versions requires an
251    // unmodified syntax tree.
252    //
253    // We also defer editing usages in the current file first since
254    // tree mutation in the same file breaks when `builder.edit_file`
255    // is called
256
257    let edits = data
258        .usages
259        .as_ref()?
260        .as_slice()
261        .iter()
262        .filter_map(|r| edit_tuple_usage(ctx, make, r, data, in_sub_pattern))
263        .collect_vec();
264
265    Some(edits)
266}
267fn edit_tuple_usage(
268    ctx: &AssistContext<'_>,
269    make: &SyntaxFactory,
270    usage: &FileReference,
271    data: &TupleData,
272    in_sub_pattern: bool,
273) -> Option<EditTupleUsage> {
274    match detect_tuple_index(usage, data) {
275        Some(index) => Some(edit_tuple_field_usage(ctx, make, data, index)),
276        None if in_sub_pattern => {
277            cov_mark::hit!(destructure_tuple_call_with_subpattern);
278            None
279        }
280        None => Some(EditTupleUsage::NoIndex(usage.range)),
281    }
282}

284fn edit_tuple_field_usage(
285    ctx: &AssistContext<'_>,
286    make: &SyntaxFactory,
287    data: &TupleData,
288    index: TupleIndex,
289) -> EditTupleUsage {
290    let field_name = &data.field_names[index.index];
291    let field_name = make.expr_path(make.ident_path(field_name));
292
293    if data.ref_type.is_some() {
294        let (replace_expr, ref_data) = determine_ref_and_parens(ctx, &index.field_expr);
295        EditTupleUsage::ReplaceExpr(replace_expr, ref_data.wrap_expr_with_factory(field_name, make))
296    } else {
297        EditTupleUsage::ReplaceExpr(index.field_expr.into(), field_name)
298    }
299}
300enum EditTupleUsage {
301    /// no index access -> make invalid -> requires handling by user
302    /// -> put usage in block comment
303    ///
304    /// Note: For macro invocations this might result in still valid code:
305    ///   When a macro accepts the tuple as argument, as well as no arguments at all,
306    ///   uncommenting the tuple still leaves the macro call working (see `tests::in_macro_call::empty_macro`).
307    ///   But this is an unlikely case. Usually the resulting macro call will become erroneous.
308    NoIndex(TextRange),
309    ReplaceExpr(ast::Expr, ast::Expr),
310}
311
312impl EditTupleUsage {
313    fn apply(self, edit: &mut SourceChangeBuilder, syntax_editor: &mut SyntaxEditor) {
314        match self {
315            EditTupleUsage::NoIndex(range) => {
316                edit.insert(range.start(), "/*");
317                edit.insert(range.end(), "*/");
318            }
319            EditTupleUsage::ReplaceExpr(target_expr, replace_with) => {
320                syntax_editor.replace(target_expr.syntax(), replace_with.syntax())
321            }
322        }
323    }
324}

326struct TupleIndex {
327    index: usize,
328    field_expr: FieldExpr,
329}
330fn detect_tuple_index(usage: &FileReference, data: &TupleData) -> Option<TupleIndex> {
331    // usage is IDENT
332    // IDENT
333    //  NAME_REF
334    //   PATH_SEGMENT
335    //    PATH
336    //     PATH_EXPR
337    //      PAREN_EXRP*
338    //       FIELD_EXPR
339
340    let node = usage
341        .name
342        .syntax()
343        .ancestors()
344        .skip_while(|s| !ast::PathExpr::can_cast(s.kind()))
345        .skip(1) // PATH_EXPR
346        .find(|s| !ast::ParenExpr::can_cast(s.kind()))?; // skip parentheses
347
348    if let Some(field_expr) = ast::FieldExpr::cast(node) {
349        let idx = field_expr.name_ref()?.as_tuple_field()?;
350        if idx < data.field_names.len() {
351            // special case: in macro call -> range of `field_expr` in applied macro, NOT range in actual file!
352            if field_expr.syntax().ancestors().any(|a| ast::MacroStmts::can_cast(a.kind())) {
353                cov_mark::hit!(destructure_tuple_macro_call);
354
355                // issue: cannot differentiate between tuple index passed into macro or tuple index as result of macro:
356                // ```rust
357                // macro_rules! m {
358                //     ($t1:expr, $t2:expr) => { $t1; $t2.0 }
359                // }
360                // let t = (1,2);
361                // m!(t.0, t)
362                // ```
363                // -> 2 tuple index usages detected!
364                //
365                // -> only handle `t`
366                return None;
367            }
368
369            Some(TupleIndex { index: idx, field_expr })
370        } else {
371            // tuple index out of range
372            None
373        }
374    } else {
375        None
376    }
377}
