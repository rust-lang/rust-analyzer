6// Assist: replace_is_some_with_if_let_some
7//
8// Replace `if x.is_some()` with `if let Some(_tmp) = x` or `if x.is_ok()` with `if let Ok(_tmp) = x`.
9//
10// ```
11// fn main() {
12//     let x = Some(1);
13//     if x.is_som$0e() {}
14// }
15// ```
16// ->
17// ```
18// fn main() {
19//     let x = Some(1);
20//     if let Some(${0:x1}) = x {}
21// }
22// ```
23pub(crate) fn replace_is_method_with_if_let_method(
24    acc: &mut Assists,
25    ctx: &AssistContext<'_>,
26) -> Option<()> {
27    let if_expr = ctx.find_node_at_offset::<ast::IfExpr>()?;
28
29    let cond = if_expr.condition()?;
30    let call_expr = match cond {
31        ast::Expr::MethodCallExpr(call) => call,
32        _ => return None,
33    };
34    if ctx.offset() > if_expr.then_branch()?.stmt_list()?.l_curly_token()?.text_range().end() {
35        return None;
36    }
37
38    let name_ref = call_expr.name_ref()?;
39    match name_ref.text().as_str() {
40        "is_some" | "is_ok" => {
41            let receiver = call_expr.receiver()?;
42
43            let mut name_generator = suggest_name::NameGenerator::new_from_scope_locals(
44                ctx.sema.scope(if_expr.syntax()),
45            );
46            let var_name = if let ast::Expr::PathExpr(path_expr) = receiver.clone() {
47                name_generator.suggest_name(&path_expr.path()?.to_string())
48            } else {
49                name_generator.for_variable(&receiver, &ctx.sema)
50            };
51
52            let (assist_id, message, text) = if name_ref.text() == "is_some" {
53                ("replace_is_some_with_if_let_some", "Replace `is_some` with `if let Some`", "Some")
54            } else {
55                ("replace_is_ok_with_if_let_ok", "Replace `is_ok` with `if let Ok`", "Ok")
56            };
57
58            acc.add(
59                AssistId::refactor_rewrite(assist_id),
60                message,
61                call_expr.syntax().text_range(),
62                |edit| {
63                    let make = SyntaxFactory::with_mappings();
64                    let mut editor = edit.make_editor(call_expr.syntax());
65
66                    let var_pat = make.ident_pat(false, false, make.name(&var_name));
67                    let pat = make.tuple_struct_pat(make.ident_path(text), [var_pat.into()]);
68                    let let_expr = make.expr_let(pat.into(), receiver);
69
70                    if let Some(cap) = ctx.config.snippet_cap
71                        && let Some(ast::Pat::TupleStructPat(pat)) = let_expr.pat()
72                        && let Some(first_var) = pat.fields().next()
73                    {
74                        let placeholder = edit.make_placeholder_snippet(cap);
75                        editor.add_annotation(first_var.syntax(), placeholder);
76                    }
77
78                    editor.replace(call_expr.syntax(), let_expr.syntax());
79                    editor.add_mappings(make.finish_with_mappings());
80                    edit.add_file_edits(ctx.vfs_file_id(), editor);
81                },
82            )
83        }
84        _ => None,
85    }
86}
