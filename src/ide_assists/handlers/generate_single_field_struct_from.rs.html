19// Assist: generate_single_field_struct_from
20//
21// Implement From for a single field structure, ignore trivial types.
22//
23// ```
24// # //- minicore: from, phantom_data
25// use core::marker::PhantomData;
26// struct $0Foo<T> {
27//     id: i32,
28//     _phantom_data: PhantomData<T>,
29// }
30// ```
31// ->
32// ```
33// use core::marker::PhantomData;
34// struct Foo<T> {
35//     id: i32,
36//     _phantom_data: PhantomData<T>,
37// }
38//
39// impl<T> From<i32> for Foo<T> {
40//     fn from(id: i32) -> Self {
41//         Self { id, _phantom_data: PhantomData }
42//     }
43// }
44// ```
45pub(crate) fn generate_single_field_struct_from(
46    acc: &mut Assists,
47    ctx: &AssistContext<'_>,
48) -> Option<()> {
49    let strukt_name = ctx.find_node_at_offset::<ast::Name>()?;
50    let adt = ast::Adt::cast(strukt_name.syntax().parent()?)?;
51    let ast::Adt::Struct(strukt) = adt else {
52        tracing::debug!(?adt);
53        return None;
54    };
55
56    let sema = &ctx.sema;
57    let (names, types) = get_fields(&strukt)?;
58
59    let module = sema.scope(strukt.syntax())?.module();
60    let constructors = make_constructors(ctx, module, &types);
61
62    if constructors.iter().filter(|expr| expr.is_none()).count() != 1 {
63        tracing::debug!(?constructors);
64        return None;
65    }
66    let main_field_i = constructors.iter().position(Option::is_none)?;
67    if from_impl_exists(&strukt, main_field_i, &ctx.sema).is_some() {
68        tracing::debug!(?strukt, ?main_field_i);
69        return None;
70    }
71
72    let main_field_name =
73        names.as_ref().map_or(TokenText::borrowed("value"), |names| names[main_field_i].text());
74    let main_field_ty = types[main_field_i].clone();
75
76    acc.add(
77        AssistId::generate("generate_single_field_struct_from"),
78        "Generate single field `From`",
79        strukt.syntax().text_range(),
80        |builder| {
81            let indent = strukt.indent_level();
82            let ty_where_clause = strukt.where_clause();
83            let type_gen_params = strukt.generic_param_list();
84            let type_gen_args = type_gen_params.as_ref().map(|params| params.to_generic_args());
85            let trait_gen_args = Some(make::generic_arg_list([ast::GenericArg::TypeArg(
86                make::type_arg(main_field_ty.clone()),
87            )]));
88
89            let ty = make::ty(&strukt_name.text());
90
91            let constructor =
92                make_adt_constructor(names.as_deref(), constructors, &main_field_name);
93            let body = make::block_expr([], Some(constructor));
94
95            let fn_ = make::fn_(
96                None,
97                None,
98                make::name("from"),
99                None,
100                None,
101                make::param_list(
102                    None,
103                    [make::param(
104                        make::path_pat(make::path_from_text(&main_field_name)),
105                        main_field_ty,
106                    )],
107                ),
108                body,
109                Some(make::ret_type(make::ty("Self"))),
110                false,
111                false,
112                false,
113                false,
114            )
115            .indent(1.into());
116
117            let cfg_attrs = strukt
118                .attrs()
119                .filter(|attr| attr.as_simple_call().is_some_and(|(name, _arg)| name == "cfg"));
120
121            let impl_ = make::impl_trait(
122                cfg_attrs,
123                false,
124                None,
125                trait_gen_args,
126                type_gen_params,
127                type_gen_args,
128                false,
129                make::ty("From"),
130                ty.clone(),
131                None,
132                ty_where_clause.map(|wc| wc.reset_indent()),
133                None,
134            )
135            .clone_for_update();
136
137            impl_.get_or_create_assoc_item_list().add_item(fn_.into());
138            let impl_ = impl_.indent(indent);
139
140            let mut edit = builder.make_editor(strukt.syntax());
141
142            edit.insert_all(
143                Position::after(strukt.syntax()),
144                vec![
145                    make::tokens::whitespace(&format!("\n\n{indent}")).syntax_element(),
146                    impl_.syntax().syntax_element(),
147                ],
148            );
149
150            builder.add_file_edits(ctx.vfs_file_id(), edit);
151        },
152    )
153}

155fn make_adt_constructor(
156    names: Option<&[ast::Name]>,
157    constructors: Vec<Option<ast::Expr>>,
158    main_field_name: &TokenText<'_>,
159) -> ast::Expr {
160    if let Some(names) = names {
161        let fields = make::record_expr_field_list(names.iter().zip(constructors).map(
162            |(name, initializer)| {
163                make::record_expr_field(make::name_ref(&name.text()), initializer)
164            },
165        ));
166        make::record_expr(make::path_from_text("Self"), fields).into()
167    } else {
168        let arg_list = make::arg_list(constructors.into_iter().map(|expr| {
169            expr.unwrap_or_else(|| make::expr_path(make::path_from_text(main_field_name)))
170        }));
171        make::expr_call(make::expr_path(make::path_from_text("Self")), arg_list).into()
172    }
173}

175fn make_constructors(
176    ctx: &AssistContext<'_>,
177    module: hir::Module,
178    types: &[ast::Type],
179) -> Vec<Option<ast::Expr>> {
180    let (db, sema) = (ctx.db(), &ctx.sema);
181    let cfg = ctx.config.find_path_config(ctx.sema.is_nightly(module.krate(ctx.sema.db)));
182    types
183        .iter()
184        .map(|ty| {
185            let ty = sema.resolve_type(ty)?;
186            if ty.is_unit() {
187                return Some(make::expr_tuple([]).into());
188            }
189            let item_in_ns = ModuleDef::Adt(ty.as_adt()?).into();
190            let edition = module.krate(db).edition(db);
191
192            let ty_path = module.find_path(db, item_for_path_search(db, item_in_ns)?, cfg)?;
193
194            use_trivial_constructor(db, mod_path_to_ast(&ty_path, edition), &ty, edition)
195        })
196        .collect()
197}

199fn get_fields(strukt: &ast::Struct) -> Option<(Option<Vec<ast::Name>>, Vec<ast::Type>)> {
200    Some(match strukt.kind() {
201        ast::StructKind::Unit => return None,
202        ast::StructKind::Record(fields) => {
203            let names = fields.fields().map(|field| field.name()).collect::<Option<_>>()?;
204            let types = fields.fields().map(|field| field.ty()).collect::<Option<_>>()?;
205            (Some(names), types)
206        }
207        ast::StructKind::Tuple(fields) => {
208            (None, fields.fields().map(|field| field.ty()).collect::<Option<_>>()?)
209        }
210    })
211}

213#[tracing::instrument(ret)]
214fn from_impl_exists(
215    strukt: &ast::Struct,
216    main_field_i: usize,
217    sema: &Semantics<'_, RootDatabase>,
218) -> Option<()> {
219    let db = sema.db;
220    let strukt = sema.to_def(strukt)?;
221    let krate = strukt.krate(db);
222    let from_trait = FamousDefs(sema, krate).core_convert_From()?;
223    let interner = DbInterner::new_with(db, krate.base());
224    use hir::next_solver::infer::DbInternerInferExt;
225    let infcx = interner.infer_ctxt().build(TypingMode::non_body_analysis());
226
227    let strukt = strukt.instantiate_infer(&infcx);
228    let field_ty = strukt.fields(db).get(main_field_i)?.ty(db);
229    let struct_ty = strukt.ty(db);
230    tracing::debug!(?strukt, ?field_ty, ?struct_ty);
231    struct_ty.impls_trait(infcx, from_trait, &[field_ty]).then_some(())
232}
