23/// [`CargoWorkspace`] represents the logical structure of, well, a Cargo
24/// workspace. It pretty closely mirrors `cargo metadata` output.
25///
26/// Note that internally, rust-analyzer uses a different structure:
27/// `CrateGraph`. `CrateGraph` is lower-level: it knows only about the crates,
28/// while this knows about `Packages` & `Targets`: purely cargo-related
29/// concepts.
30///
31/// We use absolute paths here, `cargo metadata` guarantees to always produce
32/// abs paths.
33#[derive(Debug, Clone, Eq, PartialEq)]
34pub struct CargoWorkspace {
35    packages: Arena<PackageData>,
36    targets: Arena<TargetData>,
37    workspace_root: AbsPathBuf,
38    target_directory: AbsPathBuf,
39    manifest_path: ManifestPath,
40    is_virtual_workspace: bool,
41    /// Whether this workspace represents the sysroot workspace.
42    is_sysroot: bool,
43    /// Environment variables set in the `.cargo/config` file and the extraEnv
44    /// configuration option.
45    env: Env,
46    requires_rustc_private: bool,
47}
48
49impl ops::Index<Package> for CargoWorkspace {
50    type Output = PackageData;
51    fn index(&self, index: Package) -> &PackageData {
52        &self.packages[index]
53    }
54}

56impl ops::Index<Target> for CargoWorkspace {
57    type Output = TargetData;
58    fn index(&self, index: Target) -> &TargetData {
59        &self.targets[index]
60    }

210#[derive(Debug, Clone, Copy, PartialEq, Eq)]
211pub enum DepKind {
212    /// Available to the library, binary, and dev targets in the package (but not the build script).
213    Normal,
214    /// Available only to test and bench targets (and the library target, when built with `cfg(test)`).
215    Dev,
216    /// Available only to the build script target.
217    Build,
218}
219
220impl DepKind {
221    fn iter(list: &[cargo_metadata::DepKindInfo]) -> impl Iterator<Item = Self> {
222        let mut dep_kinds = [None; 3];
223        if list.is_empty() {
224            dep_kinds[0] = Some(Self::Normal);
225        }
226        for info in list {
227            match info.kind {
228                cargo_metadata::DependencyKind::Normal => dep_kinds[0] = Some(Self::Normal),
229                cargo_metadata::DependencyKind::Development => dep_kinds[1] = Some(Self::Dev),
230                cargo_metadata::DependencyKind::Build => dep_kinds[2] = Some(Self::Build),
231                cargo_metadata::DependencyKind::Unknown => continue,
232            }
233        }
234        dep_kinds.into_iter().flatten()
235    }
236}

238/// Information associated with a package's target
239#[derive(Debug, Clone, Eq, PartialEq)]
240pub struct TargetData {
241    /// Package that provided this target
242    pub package: Package,
243    /// Name as given in the `Cargo.toml` or generated from the file name
244    pub name: String,
245    /// Path to the main source file of the target
246    pub root: AbsPathBuf,
247    /// Kind of target
248    pub kind: TargetKind,
249    /// Required features of the target without which it won't build
250    pub required_features: Vec<String>,
251}
252
253#[derive(Debug, Clone, Copy, PartialEq, Eq)]
254pub enum TargetKind {
255    Bin,
256    /// Any kind of Cargo lib crate-type (dylib, rlib, proc-macro, ...).
257    Lib {
258        /// Is this target a proc-macro
259        is_proc_macro: bool,
260    },
261    Example,
262    Test,
263    Bench,
264    /// Cargo calls this kind `custom-build`
265    BuildScript,
266    Other,
267}
268
269impl TargetKind {
270    pub fn new(kinds: &[cargo_metadata::TargetKind]) -> TargetKind {
271        for kind in kinds {
272            return match kind {
273                cargo_metadata::TargetKind::Bin => TargetKind::Bin,
274                cargo_metadata::TargetKind::Test => TargetKind::Test,
275                cargo_metadata::TargetKind::Bench => TargetKind::Bench,
276                cargo_metadata::TargetKind::Example => TargetKind::Example,
277                cargo_metadata::TargetKind::CustomBuild => TargetKind::BuildScript,
278                cargo_metadata::TargetKind::ProcMacro => TargetKind::Lib { is_proc_macro: true },
279                cargo_metadata::TargetKind::Lib
280                | cargo_metadata::TargetKind::DyLib
281                | cargo_metadata::TargetKind::CDyLib
282                | cargo_metadata::TargetKind::StaticLib
283                | cargo_metadata::TargetKind::RLib => TargetKind::Lib { is_proc_macro: false },
284                _ => continue,
285            };
286        }
287        TargetKind::Other
288    }
289
290    pub fn is_executable(self) -> bool {
291        matches!(self, TargetKind::Bin | TargetKind::Example)
292    }
293
294    pub fn is_proc_macro(self) -> bool {
295        matches!(self, TargetKind::Lib { is_proc_macro: true })
296    }
297
298    /// If this is a valid cargo target, returns the name cargo uses in command line arguments
299    /// and output, otherwise None.
300    /// <https://docs.rs/cargo_metadata/latest/cargo_metadata/enum.TargetKind.html>
301    pub fn as_cargo_target(self) -> Option<&'static str> {
302        match self {
303            TargetKind::Bin => Some("bin"),
304            TargetKind::Lib { is_proc_macro: true } => Some("proc-macro"),
305            TargetKind::Lib { is_proc_macro: false } => Some("lib"),
306            TargetKind::Example => Some("example"),
307            TargetKind::Test => Some("test"),
308            TargetKind::Bench => Some("bench"),
309            TargetKind::BuildScript => Some("custom-build"),
310            TargetKind::Other => None,
311        }
312    }
313}

339impl CargoWorkspace {
340    pub fn new(
341        mut meta: cargo_metadata::Metadata,
342        ws_manifest_path: ManifestPath,
343        cargo_env: Env,
344        is_sysroot: bool,
345    ) -> CargoWorkspace {
346        let mut pkg_by_id = FxHashMap::default();
347        let mut packages = Arena::default();
348        let mut targets = Arena::default();
349
350        let ws_members = &meta.workspace_members;
351
352        let workspace_root = AbsPathBuf::assert(meta.workspace_root);
353        let target_directory = AbsPathBuf::assert(meta.target_directory);
354        let mut is_virtual_workspace = true;
355        let mut requires_rustc_private = false;
356
357        let mut members = FxHashSet::default();
358
359        meta.packages.sort_by(|a, b| a.id.cmp(&b.id));
360        for meta_pkg in meta.packages {
361            let cargo_metadata::Package {
362                name,
363                version,
364                id,
365                source,
366                targets: meta_targets,
367                features,
368                manifest_path,
369                repository,
370                edition,
371                metadata,
372                authors,
373                description,
374                homepage,
375                license,
376                license_file,
377                readme,
378                rust_version,
379                ..
380            } = meta_pkg;
381            let id = Arc::new(id);
382            let meta = from_value::<PackageMetadata>(metadata).unwrap_or_default();
383            let edition = match edition {
384                cargo_metadata::Edition::E2015 => Edition::Edition2015,
385                cargo_metadata::Edition::E2018 => Edition::Edition2018,
386                cargo_metadata::Edition::E2021 => Edition::Edition2021,
387                cargo_metadata::Edition::E2024 => Edition::Edition2024,
388                _ => {
389                    tracing::error!("Unsupported edition `{:?}`", edition);
390                    Edition::CURRENT
391                }
392            };
393            // We treat packages without source as "local" packages. That includes all members of
394            // the current workspace, as well as any path dependency outside the workspace.
395            let is_local = source.is_none();
396            let is_member = ws_members.contains(&id);
397
398            let manifest = ManifestPath::try_from(AbsPathBuf::assert(manifest_path)).unwrap();
399            is_virtual_workspace &= manifest != ws_manifest_path;
400            let pkg = packages.alloc(PackageData {
401                id: id.clone(),
402                name: name.to_string(),
403                version,
404                manifest: manifest.clone(),
405                targets: Vec::new(),
406                is_local,
407                is_member,
408                edition,
409                repository,
410                authors,
411                description,
412                homepage,
413                license,
414                license_file,
415                readme,
416                rust_version,
417                dependencies: Vec::new(),
418                features: features.into_iter().collect(),
419                active_features: Vec::new(),
420                metadata: meta.rust_analyzer.unwrap_or_default(),
421                all_member_deps: None,
422            });
423            if is_member {
424                members.insert(pkg);
425            }
426            let pkg_data = &mut packages[pkg];
427            requires_rustc_private |= pkg_data.metadata.rustc_private;
428            pkg_by_id.insert(id, pkg);
429            for meta_tgt in meta_targets {
430                let cargo_metadata::Target { name, kind, required_features, src_path, .. } =
431                    meta_tgt;
432                let kind = TargetKind::new(&kind);
433                let tgt = targets.alloc(TargetData {
434                    package: pkg,
435                    name,
436                    root: if kind == TargetKind::Bin
437                        && manifest.extension().is_some_and(|ext| ext == "rs")
438                    {
439                        // cargo strips the script part of a cargo script away and places the
440                        // modified manifest file into a special target dir which is then used as
441                        // the source path. We don't want that, we want the original here so map it
442                        // back
443                        manifest.clone().into()
444                    } else {
445                        AbsPathBuf::assert(src_path)
446                    },
447                    kind,
448                    required_features,
449                });
450                pkg_data.targets.push(tgt);
451            }
452        }
453        for mut node in meta.resolve.map_or_else(Vec::new, |it| it.nodes) {
454            let &source = pkg_by_id.get(&node.id).unwrap();
455            node.deps.sort_by(|a, b| a.pkg.cmp(&b.pkg));
456            let dependencies = node
457                .deps
458                .iter()
459                .flat_map(|dep| DepKind::iter(&dep.dep_kinds).map(move |kind| (dep, kind)));
460            for (dep_node, kind) in dependencies {
461                let &pkg = pkg_by_id.get(&dep_node.pkg).unwrap();
462                let dep = PackageDependency { name: dep_node.name.to_string(), pkg, kind };
463                packages[source].dependencies.push(dep);
464            }
465            packages[source]
466                .active_features
467                .extend(node.features.into_iter().map(|it| it.to_string()));
468        }
469
470        fn saturate_all_member_deps(
471            packages: &mut Arena<PackageData>,
472            to_visit: Package,
473            visited: &mut FxHashSet<Package>,
474            members: &FxHashSet<Package>,
475        ) {
476            let pkg_data = &mut packages[to_visit];
477
478            if !visited.insert(to_visit) {
479                return;
480            }
481
482            let deps: Vec<_> = pkg_data
483                .dependencies
484                .iter()
485                .filter_map(|dep| {
486                    let pkg = dep.pkg;
487                    if members.contains(&pkg) { Some(pkg) } else { None }
488                })
489                .collect();
490
491            let mut all_member_deps = FxHashSet::from_iter(deps.iter().copied());
492            for dep in deps {
493                saturate_all_member_deps(packages, dep, visited, members);
494                if let Some(transitives) = &packages[dep].all_member_deps {
495                    all_member_deps.extend(transitives);
496                }
497            }
498
499            packages[to_visit].all_member_deps = Some(all_member_deps);
500        }
501
502        let mut visited = FxHashSet::default();
503        for member in members.iter() {
504            saturate_all_member_deps(&mut packages, *member, &mut visited, &members);
505        }
506
507        CargoWorkspace {
508            packages,
509            targets,
510            workspace_root,
511            target_directory,
512            manifest_path: ws_manifest_path,
513            is_virtual_workspace,
514            requires_rustc_private,
515            is_sysroot,
516            env: cargo_env,
517        }
518    }

551    pub fn parent_manifests(&self, manifest_path: &ManifestPath) -> Option<Vec<ManifestPath>> {
552        let mut found = false;
553        let parent_manifests = self
554            .packages()
555            .filter_map(|pkg| {
556                if !found && &self[pkg].manifest == manifest_path {
557                    found = true
558                }
559                self[pkg].dependencies.iter().find_map(|dep| {
560                    (&self[dep.pkg].manifest == manifest_path).then(|| self[pkg].manifest.clone())
561                })
562            })
563            .collect::<Vec<ManifestPath>>();
564
565        // some packages has this pkg as dep. return their manifests
566        if !parent_manifests.is_empty() {
567            return Some(parent_manifests);
568        }
569
570        // this pkg is inside this cargo workspace, fallback to workspace root
571        if found {
572            return Some(vec![
573                ManifestPath::try_from(self.workspace_root().join("Cargo.toml")).ok()?,
574            ]);
575        }
576
577        // not in this workspace
578        None
579    }

581    /// Returns the union of the features of all member crates in this workspace.
582    pub fn workspace_features(&self) -> FxHashSet<String> {
583        self.packages()
584            .filter_map(|package| {
585                let package = &self[package];
586                if package.is_member {
587                    Some(package.features.keys().cloned().chain(
588                        package.features.keys().map(|key| format!("{}/{key}", package.name)),
589                    ))
590                } else {
591                    None
592                }
593            })
594            .flatten()
595            .collect()
596    }
597
598    fn is_unique(&self, name: &str) -> bool {
599        self.packages.iter().filter(|(_, v)| v.name == name).count() == 1
600    }

616    }

631impl FetchMetadata {
632    /// Builds a command to fetch metadata for the given `cargo_toml` manifest.
633    ///
634    /// Performs a lightweight pre-fetch using the `--no-deps` option,
635    /// available via `FetchMetadata::no_deps_metadata`, to gather basic
636    /// information such as the `target-dir`.
637    ///
638    /// The provided sysroot is used to set the `RUSTUP_TOOLCHAIN`
639    /// environment variable when invoking Cargo, ensuring that the
640    /// rustup proxy selects the correct toolchain.
641    pub(crate) fn new(
642        cargo_toml: &ManifestPath,
643        current_dir: &AbsPath,
644        config: &CargoMetadataConfig,
645        sysroot: &Sysroot,
646        no_deps: bool,
647    ) -> Self {
648        let cargo = sysroot.tool(Tool::Cargo, current_dir, &config.extra_env);
649        let mut command = MetadataCommand::new();
650        command.env(NO_RUSTUP_AUTO_INSTALL_ENV.0, NO_RUSTUP_AUTO_INSTALL_ENV.1);
651        command.cargo_path(cargo.get_program());
652        cargo.get_envs().for_each(|(var, val)| _ = command.env(var, val.unwrap_or_default()));
653        command.manifest_path(cargo_toml.to_path_buf());
654        match &config.features {
655            CargoFeatures::All => {
656                command.features(CargoOpt::AllFeatures);
657            }
658            CargoFeatures::Selected { features, no_default_features } => {
659                if *no_default_features {
660                    command.features(CargoOpt::NoDefaultFeatures);
661                }
662                if !features.is_empty() {
663                    command.features(CargoOpt::SomeFeatures(features.clone()));
664                }
665            }
666        }
667        command.current_dir(current_dir);
668
669        let mut other_options = vec![];
670        // cargo metadata only supports a subset of flags of what cargo usually accepts, and usually
671        // the only relevant flags for metadata here are unstable ones, so we pass those along
672        // but nothing else
673        let mut extra_args = config.extra_args.iter();
674        while let Some(arg) = extra_args.next() {
675            if arg == "-Z"
676                && let Some(arg) = extra_args.next()
677            {
678                other_options.push("-Z".to_owned());
679                other_options.push(arg.to_owned());
680            }
681        }
682
683        let mut lockfile_copy = None;
684        if cargo_toml.is_rust_manifest() {
685            other_options.push("-Zscript".to_owned());
686        } else if let Some(v) = config.toolchain_version.as_ref() {
687            lockfile_copy = make_lockfile_copy(
688                v,
689                &<_ as AsRef<Utf8Path>>::as_ref(cargo_toml).with_extension("lock"),
690            );
691        }
692
693        if !config.targets.is_empty() {
694            other_options.extend(
695                config.targets.iter().flat_map(|it| ["--filter-platform".to_owned(), it.clone()]),
696            );
697        }
698
699        command.other_options(other_options.clone());
700
701        // Pre-fetch basic metadata using `--no-deps`, which:
702        // - avoids fetching registries like crates.io,
703        // - skips dependency resolution and does not modify lockfiles,
704        // - and thus doesn't require progress reporting or copying lockfiles.
705        //
706        // Useful as a fast fallback to extract info like `target-dir`.
707        let cargo_command;
708        let no_deps_result = if no_deps {
709            command.no_deps();
710            cargo_command = command.cargo_command();
711            command.exec()
712        } else {
713            let mut no_deps_command = command.clone();
714            no_deps_command.no_deps();
715            cargo_command = no_deps_command.cargo_command();
716            no_deps_command.exec()
717        }
718        .with_context(|| format!("Failed to run `{cargo_command:?}`"));
719
720        Self {
721            manifest_path: cargo_toml.clone(),
722            command,
723            lockfile_copy,
724            kind: config.kind,
725            no_deps,
726            no_deps_result,
727            other_options,
728        }
729    }
730
731    /// Executes the metadata-fetching command.
732    ///
733    /// A successful result may still contain a metadata error if the full fetch failed,
734    /// but the fallback `--no-deps` pre-fetch succeeded during command construction.
735    pub(crate) fn exec(
736        self,
737        locked: bool,
738        progress: &dyn Fn(String),
739    ) -> anyhow::Result<(cargo_metadata::Metadata, Option<anyhow::Error>)> {
740        let Self {
741            mut command,
742            manifest_path: _,
743            lockfile_copy,
744            kind: _,
745            no_deps,
746            no_deps_result,
747            mut other_options,
748        } = self;
749
750        if no_deps {
751            return no_deps_result.map(|m| (m, None));
752        }
753
754        let mut using_lockfile_copy = false;
755        if let Some(lockfile_copy) = &lockfile_copy {
756            match lockfile_copy.usage {
757                LockfileUsage::WithFlag => {
758                    other_options.push("--lockfile-path".to_owned());
759                    other_options.push(lockfile_copy.path.to_string());
760                }
761                LockfileUsage::WithEnvVar => {
762                    other_options.push("-Zlockfile-path".to_owned());
763                    command.env("CARGO_RESOLVER_LOCKFILE_PATH", lockfile_copy.path.as_os_str());
764                }
765            }
766            using_lockfile_copy = true;
767        }
768        if using_lockfile_copy || other_options.iter().any(|it| it.starts_with("-Z")) {
769            command.env("__CARGO_TEST_CHANNEL_OVERRIDE_DO_NOT_USE_THIS", "nightly");
770            other_options.push("-Zunstable-options".to_owned());
771        }
772        // No need to lock it if we copied the lockfile, we won't modify the original after all/
773        // This way cargo cannot error out on us if the lockfile requires updating.
774        if !using_lockfile_copy && locked {
775            other_options.push("--locked".to_owned());
776        }
777        command.other_options(other_options);
778
779        progress("cargo metadata: started".to_owned());
780
781        let res = (|| -> anyhow::Result<(_, _)> {
782            let mut errored = false;
783            tracing::debug!("Running `{:?}`", command.cargo_command());
784            let output =
785                spawn_with_streaming_output(command.cargo_command(), &mut |_| (), &mut |line| {
786                    errored = errored || line.starts_with("error") || line.starts_with("warning");
787                    if errored {
788                        progress("cargo metadata: ?".to_owned());
789                        return;
790                    }
791                    progress(format!("cargo metadata: {line}"));
792                })?;
793            if !output.status.success() {
794                progress(format!("cargo metadata: failed {}", output.status));
795                let error = cargo_metadata::Error::CargoMetadata {
796                    stderr: String::from_utf8(output.stderr)?,
797                }
798                .into();
799                if !no_deps {
800                    // If we failed to fetch metadata with deps, return pre-fetched result without them.
801                    // This makes r-a still work partially when offline.
802                    if let Ok(metadata) = no_deps_result {
803                        tracing::warn!(
804                            ?error,
805                            "`cargo metadata` failed and returning succeeded result with `--no-deps`"
806                        );
807                        return Ok((metadata, Some(error)));
808                    }
809                }
810                return Err(error);
811            }
812            let stdout = from_utf8(&output.stdout)?
813                .lines()
814                .find(|line| line.starts_with('{'))
815                .ok_or(cargo_metadata::Error::NoJson)?;
816            Ok((cargo_metadata::MetadataCommand::parse(stdout)?, None))
817        })()
818        .with_context(|| format!("Failed to run `{:?}`", command.cargo_command()));
819        progress("cargo metadata: finished".to_owned());
820        res
821    }
822}
