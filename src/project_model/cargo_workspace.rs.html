32/// [`CargoWorkspace`] represents the logical structure of, well, a Cargo
33/// workspace. It pretty closely mirrors `cargo metadata` output.
34///
35/// Note that internally, rust-analyzer uses a different structure:
36/// `CrateGraph`. `CrateGraph` is lower-level: it knows only about the crates,
37/// while this knows about `Packages` & `Targets`: purely cargo-related
38/// concepts.
39///
40/// We use absolute paths here, `cargo metadata` guarantees to always produce
41/// abs paths.
42#[derive(Debug, Clone, Eq, PartialEq)]
43pub struct CargoWorkspace {
44    packages: Arena<PackageData>,
45    targets: Arena<TargetData>,
46    workspace_root: AbsPathBuf,
47    target_directory: AbsPathBuf,
48    manifest_path: ManifestPath,
49    is_virtual_workspace: bool,
50    /// Whether this workspace represents the sysroot workspace.
51    is_sysroot: bool,
52    /// Environment variables set in the `.cargo/config` file and the extraEnv
53    /// configuration option.
54    env: Env,
55    requires_rustc_private: bool,
56}
57
58impl ops::Index<Package> for CargoWorkspace {

60    fn index(&self, index: Package) -> &PackageData {
61        &self.packages[index]
62    }
63}

65impl ops::Index<Target> for CargoWorkspace {
66    type Output = TargetData;
67    fn index(&self, index: Target) -> &TargetData {
68        &self.targets[index]
69    }

196#[derive(Debug, Clone, Copy, PartialEq, Eq)]
197pub enum DepKind {
198    /// Available to the library, binary, and dev targets in the package (but not the build script).
199    Normal,
200    /// Available only to test and bench targets (and the library target, when built with `cfg(test)`).
201    Dev,
202    /// Available only to the build script target.
203    Build,
204}
205
206impl DepKind {
207    fn iter(list: &[cargo_metadata::DepKindInfo]) -> impl Iterator<Item = Self> {
208        let mut dep_kinds = [None; 3];
209        if list.is_empty() {
210            dep_kinds[0] = Some(Self::Normal);
211        }
212        for info in list {
213            match info.kind {
214                cargo_metadata::DependencyKind::Normal => dep_kinds[0] = Some(Self::Normal),
215                cargo_metadata::DependencyKind::Development => dep_kinds[1] = Some(Self::Dev),
216                cargo_metadata::DependencyKind::Build => dep_kinds[2] = Some(Self::Build),
217                cargo_metadata::DependencyKind::Unknown => continue,
218            }
219        }
220        dep_kinds.into_iter().flatten()
221    }
222}

226pub struct TargetData {
227    /// Package that provided this target
228    pub package: Package,
229    /// Name as given in the `Cargo.toml` or generated from the file name
230    pub name: String,
231    /// Path to the main source file of the target
232    pub root: AbsPathBuf,
233    /// Kind of target
234    pub kind: TargetKind,
235    /// Required features of the target without which it won't build
236    pub required_features: Vec<String>,
237}

239#[derive(Debug, Clone, Copy, PartialEq, Eq)]
240pub enum TargetKind {
241    Bin,
242    /// Any kind of Cargo lib crate-type (dylib, rlib, proc-macro, ...).
243    Lib {
244        /// Is this target a proc-macro
245        is_proc_macro: bool,
246    },
247    Example,
248    Test,
249    Bench,
250    /// Cargo calls this kind `custom-build`
251    BuildScript,
252    Other,
253}
254
255impl TargetKind {
256    pub fn new(kinds: &[cargo_metadata::TargetKind]) -> TargetKind {
257        for kind in kinds {
258            return match kind {
259                cargo_metadata::TargetKind::Bin => TargetKind::Bin,
260                cargo_metadata::TargetKind::Test => TargetKind::Test,
261                cargo_metadata::TargetKind::Bench => TargetKind::Bench,
262                cargo_metadata::TargetKind::Example => TargetKind::Example,
263                cargo_metadata::TargetKind::CustomBuild => TargetKind::BuildScript,
264                cargo_metadata::TargetKind::ProcMacro => TargetKind::Lib { is_proc_macro: true },
265                cargo_metadata::TargetKind::Lib
266                | cargo_metadata::TargetKind::DyLib
267                | cargo_metadata::TargetKind::CDyLib
268                | cargo_metadata::TargetKind::StaticLib
269                | cargo_metadata::TargetKind::RLib => TargetKind::Lib { is_proc_macro: false },
270                _ => continue,
271            };
272        }
273        TargetKind::Other
274    }
275
276    pub fn is_executable(self) -> bool {
277        matches!(self, TargetKind::Bin | TargetKind::Example)
278    }
279
280    pub fn is_proc_macro(self) -> bool {
281        matches!(self, TargetKind::Lib { is_proc_macro: true })
282    }
283
284    /// If this is a valid cargo target, returns the name cargo uses in command line arguments
285    /// and output, otherwise None.
286    /// https://docs.rs/cargo_metadata/latest/cargo_metadata/enum.TargetKind.html
287    pub fn as_cargo_target(self) -> Option<&'static str> {
288        match self {
289            TargetKind::Bin => Some("bin"),
290            TargetKind::Lib { is_proc_macro: true } => Some("proc-macro"),
291            TargetKind::Lib { is_proc_macro: false } => Some("lib"),
292            TargetKind::Example => Some("example"),
293            TargetKind::Test => Some("test"),
294            TargetKind::Bench => Some("bench"),
295            TargetKind::BuildScript => Some("custom-build"),
296            TargetKind::Other => None,
297        }
298    }
299}

325impl CargoWorkspace {
326    pub fn new(
327        mut meta: cargo_metadata::Metadata,
328        ws_manifest_path: ManifestPath,
329        cargo_env: Env,
330        is_sysroot: bool,
331    ) -> CargoWorkspace {
332        let mut pkg_by_id = FxHashMap::default();
333        let mut packages = Arena::default();
334        let mut targets = Arena::default();
335
336        let ws_members = &meta.workspace_members;
337
338        let workspace_root = AbsPathBuf::assert(meta.workspace_root);
339        let target_directory = AbsPathBuf::assert(meta.target_directory);
340        let mut is_virtual_workspace = true;
341        let mut requires_rustc_private = false;
342
343        let mut members = FxHashSet::default();
344
345        meta.packages.sort_by(|a, b| a.id.cmp(&b.id));
346        for meta_pkg in meta.packages {
347            let cargo_metadata::Package {
348                name,
349                version,
350                id,
351                source,
352                targets: meta_targets,
353                features,
354                manifest_path,
355                repository,
356                edition,
357                metadata,
358                authors,
359                description,
360                homepage,
361                license,
362                license_file,
363                readme,
364                rust_version,
365                ..
366            } = meta_pkg;
367            let id = Arc::new(id);
368            let meta = from_value::<PackageMetadata>(metadata).unwrap_or_default();
369            let edition = match edition {
370                cargo_metadata::Edition::E2015 => Edition::Edition2015,
371                cargo_metadata::Edition::E2018 => Edition::Edition2018,
372                cargo_metadata::Edition::E2021 => Edition::Edition2021,
373                cargo_metadata::Edition::E2024 => Edition::Edition2024,
374                _ => {
375                    tracing::error!("Unsupported edition `{:?}`", edition);
376                    Edition::CURRENT
377                }
378            };
379            // We treat packages without source as "local" packages. That includes all members of
380            // the current workspace, as well as any path dependency outside the workspace.
381            let is_local = source.is_none();
382            let is_member = ws_members.contains(&id);
383
384            let manifest = ManifestPath::try_from(AbsPathBuf::assert(manifest_path)).unwrap();
385            is_virtual_workspace &= manifest != ws_manifest_path;
386            let pkg = packages.alloc(PackageData {
387                id: id.clone(),
388                name: name.to_string(),
389                version,
390                manifest: manifest.clone(),
391                targets: Vec::new(),
392                is_local,
393                is_member,
394                edition,
395                repository,
396                authors,
397                description,
398                homepage,
399                license,
400                license_file,
401                readme,
402                rust_version,
403                dependencies: Vec::new(),
404                features: features.into_iter().collect(),
405                active_features: Vec::new(),
406                metadata: meta.rust_analyzer.unwrap_or_default(),
407                all_member_deps: None,
408            });
409            if is_member {
410                members.insert(pkg);
411            }
412            let pkg_data = &mut packages[pkg];
413            requires_rustc_private |= pkg_data.metadata.rustc_private;
414            pkg_by_id.insert(id, pkg);
415            for meta_tgt in meta_targets {
416                let cargo_metadata::Target { name, kind, required_features, src_path, .. } =
417                    meta_tgt;
418                let kind = TargetKind::new(&kind);
419                let tgt = targets.alloc(TargetData {
420                    package: pkg,
421                    name,
422                    root: if kind == TargetKind::Bin
423                        && manifest.extension().is_some_and(|ext| ext == "rs")
424                    {
425                        // cargo strips the script part of a cargo script away and places the
426                        // modified manifest file into a special target dir which is then used as
427                        // the source path. We don't want that, we want the original here so map it
428                        // back
429                        manifest.clone().into()
430                    } else {
431                        AbsPathBuf::assert(src_path)
432                    },
433                    kind,
434                    required_features,
435                });
436                pkg_data.targets.push(tgt);
437            }
438        }
439        for mut node in meta.resolve.map_or_else(Vec::new, |it| it.nodes) {
440            let &source = pkg_by_id.get(&node.id).unwrap();
441            node.deps.sort_by(|a, b| a.pkg.cmp(&b.pkg));
442            let dependencies = node
443                .deps
444                .iter()
445                .flat_map(|dep| DepKind::iter(&dep.dep_kinds).map(move |kind| (dep, kind)));
446            for (dep_node, kind) in dependencies {
447                let &pkg = pkg_by_id.get(&dep_node.pkg).unwrap();
448                let dep = PackageDependency { name: dep_node.name.to_string(), pkg, kind };
449                packages[source].dependencies.push(dep);
450            }
451            packages[source]
452                .active_features
453                .extend(node.features.into_iter().map(|it| it.to_string()));
454        }
455
456        fn saturate_all_member_deps(
457            packages: &mut Arena<PackageData>,
458            to_visit: Package,
459            visited: &mut FxHashSet<Package>,
460            members: &FxHashSet<Package>,
461        ) {
462            let pkg_data = &mut packages[to_visit];
463
464            if !visited.insert(to_visit) {
465                return;
466            }
467
468            let deps: Vec<_> = pkg_data
469                .dependencies
470                .iter()
471                .filter_map(|dep| {
472                    let pkg = dep.pkg;
473                    if members.contains(&pkg) { Some(pkg) } else { None }
474                })
475                .collect();
476
477            let mut all_member_deps = FxHashSet::from_iter(deps.iter().copied());
478            for dep in deps {
479                saturate_all_member_deps(packages, dep, visited, members);
480                if let Some(transitives) = &packages[dep].all_member_deps {
481                    all_member_deps.extend(transitives);
482                }
483            }
484
485            packages[to_visit].all_member_deps = Some(all_member_deps);
486        }
487
488        let mut visited = FxHashSet::default();
489        for member in members.iter() {
490            saturate_all_member_deps(&mut packages, *member, &mut visited, &members);
491        }
492
493        CargoWorkspace {
494            packages,
495            targets,
496            workspace_root,
497            target_directory,
498            manifest_path: ws_manifest_path,
499            is_virtual_workspace,
500            requires_rustc_private,
501            is_sysroot,
502            env: cargo_env,
503        }
504    }

537    pub fn parent_manifests(&self, manifest_path: &ManifestPath) -> Option<Vec<ManifestPath>> {
538        let mut found = false;
539        let parent_manifests = self
540            .packages()
541            .filter_map(|pkg| {
542                if !found && &self[pkg].manifest == manifest_path {
543                    found = true
544                }
545                self[pkg].dependencies.iter().find_map(|dep| {
546                    (&self[dep.pkg].manifest == manifest_path).then(|| self[pkg].manifest.clone())
547                })
548            })
549            .collect::<Vec<ManifestPath>>();
550
551        // some packages has this pkg as dep. return their manifests
552        if !parent_manifests.is_empty() {
553            return Some(parent_manifests);
554        }
555
556        // this pkg is inside this cargo workspace, fallback to workspace root
557        if found {
558            return Some(vec![
559                ManifestPath::try_from(self.workspace_root().join("Cargo.toml")).ok()?,
560            ]);
561        }
562
563        // not in this workspace
564        None
565    }

567    /// Returns the union of the features of all member crates in this workspace.
568    pub fn workspace_features(&self) -> FxHashSet<String> {
569        self.packages()
570            .filter_map(|package| {
571                let package = &self[package];
572                if package.is_member {
573                    Some(package.features.keys().cloned().chain(
574                        package.features.keys().map(|key| format!("{}/{key}", package.name)),
575                    ))
576                } else {
577                    None
578                }
579            })
580            .flatten()
581            .collect()
582    }
583
584    fn is_unique(&self, name: &str) -> bool {
585        self.packages.iter().filter(|(_, v)| v.name == name).count() == 1
586    }

602    }

605pub(crate) struct FetchMetadata {
606    command: cargo_metadata::MetadataCommand,
607    #[expect(dead_code)]
608    manifest_path: ManifestPath,
609    lockfile_path: Option<Utf8PathBuf>,
610    #[expect(dead_code)]
611    kind: &'static str,
612    no_deps: bool,
613    no_deps_result: anyhow::Result<cargo_metadata::Metadata>,
614    other_options: Vec<String>,
615}
616
617impl FetchMetadata {
618    /// Builds a command to fetch metadata for the given `cargo_toml` manifest.
619    ///
620    /// Performs a lightweight pre-fetch using the `--no-deps` option,
621    /// available via [`FetchMetadata::no_deps_metadata`], to gather basic
622    /// information such as the `target-dir`.
623    ///
624    /// The provided sysroot is used to set the `RUSTUP_TOOLCHAIN`
625    /// environment variable when invoking Cargo, ensuring that the
626    /// rustup proxy selects the correct toolchain.
627    pub(crate) fn new(
628        cargo_toml: &ManifestPath,
629        current_dir: &AbsPath,
630        config: &CargoMetadataConfig,
631        sysroot: &Sysroot,
632        no_deps: bool,
633    ) -> Self {
634        let cargo = sysroot.tool(Tool::Cargo, current_dir, &config.extra_env);
635        let mut command = MetadataCommand::new();
636        command.env(NO_RUSTUP_AUTO_INSTALL_ENV.0, NO_RUSTUP_AUTO_INSTALL_ENV.1);
637        command.cargo_path(cargo.get_program());
638        cargo.get_envs().for_each(|(var, val)| _ = command.env(var, val.unwrap_or_default()));
639        command.manifest_path(cargo_toml.to_path_buf());
640        match &config.features {
641            CargoFeatures::All => {
642                command.features(CargoOpt::AllFeatures);
643            }
644            CargoFeatures::Selected { features, no_default_features } => {
645                if *no_default_features {
646                    command.features(CargoOpt::NoDefaultFeatures);
647                }
648                if !features.is_empty() {
649                    command.features(CargoOpt::SomeFeatures(features.clone()));
650                }
651            }
652        }
653        command.current_dir(current_dir);
654
655        let mut other_options = vec![];
656        // cargo metadata only supports a subset of flags of what cargo usually accepts, and usually
657        // the only relevant flags for metadata here are unstable ones, so we pass those along
658        // but nothing else
659        let mut extra_args = config.extra_args.iter();
660        while let Some(arg) = extra_args.next() {
661            if arg == "-Z"
662                && let Some(arg) = extra_args.next()
663            {
664                other_options.push("-Z".to_owned());
665                other_options.push(arg.to_owned());
666            }
667        }
668
669        let mut lockfile_path = None;
670        if cargo_toml.is_rust_manifest() {
671            other_options.push("-Zscript".to_owned());
672        } else if config
673            .toolchain_version
674            .as_ref()
675            .is_some_and(|v| *v >= MINIMUM_TOOLCHAIN_VERSION_SUPPORTING_LOCKFILE_PATH)
676        {
677            lockfile_path = Some(<_ as AsRef<Utf8Path>>::as_ref(cargo_toml).with_extension("lock"));
678        }
679
680        if !config.targets.is_empty() {
681            other_options.extend(
682                config.targets.iter().flat_map(|it| ["--filter-platform".to_owned(), it.clone()]),
683            );
684        }
685
686        command.other_options(other_options.clone());
687
688        // Pre-fetch basic metadata using `--no-deps`, which:
689        // - avoids fetching registries like crates.io,
690        // - skips dependency resolution and does not modify lockfiles,
691        // - and thus doesn't require progress reporting or copying lockfiles.
692        //
693        // Useful as a fast fallback to extract info like `target-dir`.
694        let cargo_command;
695        let no_deps_result = if no_deps {
696            command.no_deps();
697            cargo_command = command.cargo_command();
698            command.exec()
699        } else {
700            let mut no_deps_command = command.clone();
701            no_deps_command.no_deps();
702            cargo_command = no_deps_command.cargo_command();
703            no_deps_command.exec()
704        }
705        .with_context(|| format!("Failed to run `{cargo_command:?}`"));
706
707        Self {
708            manifest_path: cargo_toml.clone(),
709            command,
710            lockfile_path,
711            kind: config.kind,
712            no_deps,
713            no_deps_result,
714            other_options,
715        }
716    }
717
718    pub(crate) fn no_deps_metadata(&self) -> Option<&cargo_metadata::Metadata> {
719        self.no_deps_result.as_ref().ok()
720    }
721
722    /// Executes the metadata-fetching command.
723    ///
724    /// A successful result may still contain a metadata error if the full fetch failed,
725    /// but the fallback `--no-deps` pre-fetch succeeded during command construction.
726    pub(crate) fn exec(
727        self,
728        target_dir: &Utf8Path,
729        locked: bool,
730        progress: &dyn Fn(String),
731    ) -> anyhow::Result<(cargo_metadata::Metadata, Option<anyhow::Error>)> {
732        _ = target_dir;
733        let Self {
734            mut command,
735            manifest_path: _,
736            lockfile_path,
737            kind: _,
738            no_deps,
739            no_deps_result,
740            mut other_options,
741        } = self;
742
743        if no_deps {
744            return no_deps_result.map(|m| (m, None));
745        }
746
747        let mut using_lockfile_copy = false;
748        let mut _temp_dir_guard;
749        if let Some(lockfile) = lockfile_path
750            && let Some((temp_dir, target_lockfile)) = make_lockfile_copy(&lockfile)
751        {
752            _temp_dir_guard = temp_dir;
753            other_options.push("--lockfile-path".to_owned());
754            other_options.push(target_lockfile.to_string());
755            using_lockfile_copy = true;
756        }
757        if using_lockfile_copy || other_options.iter().any(|it| it.starts_with("-Z")) {
758            command.env("__CARGO_TEST_CHANNEL_OVERRIDE_DO_NOT_USE_THIS", "nightly");
759            other_options.push("-Zunstable-options".to_owned());
760        }
761        // No need to lock it if we copied the lockfile, we won't modify the original after all/
762        // This way cargo cannot error out on us if the lockfile requires updating.
763        if !using_lockfile_copy && locked {
764            other_options.push("--locked".to_owned());
765        }
766        command.other_options(other_options);
767
768        progress("cargo metadata: started".to_owned());
769
770        let res = (|| -> anyhow::Result<(_, _)> {
771            let mut errored = false;
772            tracing::debug!("Running `{:?}`", command.cargo_command());
773            let output =
774                spawn_with_streaming_output(command.cargo_command(), &mut |_| (), &mut |line| {
775                    errored = errored || line.starts_with("error") || line.starts_with("warning");
776                    if errored {
777                        progress("cargo metadata: ?".to_owned());
778                        return;
779                    }
780                    progress(format!("cargo metadata: {line}"));
781                })?;
782            if !output.status.success() {
783                progress(format!("cargo metadata: failed {}", output.status));
784                let error = cargo_metadata::Error::CargoMetadata {
785                    stderr: String::from_utf8(output.stderr)?,
786                }
787                .into();
788                if !no_deps {
789                    // If we failed to fetch metadata with deps, return pre-fetched result without them.
790                    // This makes r-a still work partially when offline.
791                    if let Ok(metadata) = no_deps_result {
792                        tracing::warn!(
793                            ?error,
794                            "`cargo metadata` failed and returning succeeded result with `--no-deps`"
795                        );
796                        return Ok((metadata, Some(error)));
797                    }
798                }
799                return Err(error);
800            }
801            let stdout = from_utf8(&output.stdout)?
802                .lines()
803                .find(|line| line.starts_with('{'))
804                .ok_or(cargo_metadata::Error::NoJson)?;
805            Ok((cargo_metadata::MetadataCommand::parse(stdout)?, None))
806        })()
807        .with_context(|| format!("Failed to run `{:?}`", command.cargo_command()));
808        progress("cargo metadata: finished".to_owned());
809        res
810    }
