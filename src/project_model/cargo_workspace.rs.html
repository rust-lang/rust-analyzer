20/// [`CargoWorkspace`] represents the logical structure of, well, a Cargo
21/// workspace. It pretty closely mirrors `cargo metadata` output.
22///
23/// Note that internally, rust-analyzer uses a different structure:
24/// `CrateGraph`. `CrateGraph` is lower-level: it knows only about the crates,
25/// while this knows about `Packages` & `Targets`: purely cargo-related
26/// concepts.
27///
28/// We use absolute paths here, `cargo metadata` guarantees to always produce
29/// abs paths.
30#[derive(Debug, Clone, Eq, PartialEq)]
31pub struct CargoWorkspace {
32    packages: Arena<PackageData>,
33    targets: Arena<TargetData>,
34    workspace_root: AbsPathBuf,
35    target_directory: AbsPathBuf,
36    manifest_path: ManifestPath,
37    is_virtual_workspace: bool,
38    /// Whether this workspace represents the sysroot workspace.
39    is_sysroot: bool,
40    /// Environment variables set in the `.cargo/config` file.
41    config_env: Env,
42}
43
44impl ops::Index<Package> for CargoWorkspace {
45    type Output = PackageData;
46    fn index(&self, index: Package) -> &PackageData {
47        &self.packages[index]
48    }
49}

51impl ops::Index<Target> for CargoWorkspace {
52    type Output = TargetData;
53    fn index(&self, index: Target) -> &TargetData {
54        &self.targets[index]
55    }
56}

178#[derive(Debug, Clone, Copy, PartialEq, Eq)]
179pub enum DepKind {
180    /// Available to the library, binary, and dev targets in the package (but not the build script).
181    Normal,
182    /// Available only to test and bench targets (and the library target, when built with `cfg(test)`).
183    Dev,
184    /// Available only to the build script target.
185    Build,
186}
187
188impl DepKind {
189    fn iter(list: &[cargo_metadata::DepKindInfo]) -> impl Iterator<Item = Self> {
190        let mut dep_kinds = [None; 3];
191        if list.is_empty() {
192            dep_kinds[0] = Some(Self::Normal);
193        }
194        for info in list {
195            match info.kind {
196                cargo_metadata::DependencyKind::Normal => dep_kinds[0] = Some(Self::Normal),
197                cargo_metadata::DependencyKind::Development => dep_kinds[1] = Some(Self::Dev),
198                cargo_metadata::DependencyKind::Build => dep_kinds[2] = Some(Self::Build),
199                cargo_metadata::DependencyKind::Unknown => continue,
200            }
201        }
202        dep_kinds.into_iter().flatten()
203    }
204}

206/// Information associated with a package's target
207#[derive(Debug, Clone, Eq, PartialEq)]
208pub struct TargetData {
209    /// Package that provided this target
210    pub package: Package,
211    /// Name as given in the `Cargo.toml` or generated from the file name
212    pub name: String,
213    /// Path to the main source file of the target
214    pub root: AbsPathBuf,
215    /// Kind of target
216    pub kind: TargetKind,
217    /// Required features of the target without which it won't build
218    pub required_features: Vec<String>,
219}
220
221#[derive(Debug, Clone, Copy, PartialEq, Eq)]
222pub enum TargetKind {
223    Bin,
224    /// Any kind of Cargo lib crate-type (dylib, rlib, proc-macro, ...).
225    Lib {
226        /// Is this target a proc-macro
227        is_proc_macro: bool,
228    },
229    Example,
230    Test,
231    Bench,
232    /// Cargo calls this kind `custom-build`
233    BuildScript,
234    Other,
235}
236
237impl TargetKind {
238    fn new(kinds: &[cargo_metadata::TargetKind]) -> TargetKind {
239        for kind in kinds {
240            return match kind {
241                cargo_metadata::TargetKind::Bin => TargetKind::Bin,
242                cargo_metadata::TargetKind::Test => TargetKind::Test,
243                cargo_metadata::TargetKind::Bench => TargetKind::Bench,
244                cargo_metadata::TargetKind::Example => TargetKind::Example,
245                cargo_metadata::TargetKind::CustomBuild => TargetKind::BuildScript,
246                cargo_metadata::TargetKind::ProcMacro => TargetKind::Lib { is_proc_macro: true },
247                cargo_metadata::TargetKind::Lib
248                | cargo_metadata::TargetKind::DyLib
249                | cargo_metadata::TargetKind::CDyLib
250                | cargo_metadata::TargetKind::StaticLib
251                | cargo_metadata::TargetKind::RLib => TargetKind::Lib { is_proc_macro: false },
252                _ => continue,
253            };
254        }
255        TargetKind::Other
256    }
257
258    pub fn is_executable(self) -> bool {
259        matches!(self, TargetKind::Bin | TargetKind::Example)
260    }
261
262    pub fn is_proc_macro(self) -> bool {
263        matches!(self, TargetKind::Lib { is_proc_macro: true })
264    }
265
266    /// If this is a valid cargo target, returns the name cargo uses in command line arguments
267    /// and output, otherwise None.
268    /// https://docs.rs/cargo_metadata/latest/cargo_metadata/enum.TargetKind.html
269    pub fn as_cargo_target(self) -> Option<&'static str> {
270        match self {
271            TargetKind::Bin => Some("bin"),
272            TargetKind::Lib { is_proc_macro: true } => Some("proc-macro"),
273            TargetKind::Lib { is_proc_macro: false } => Some("lib"),
274            TargetKind::Example => Some("example"),
275            TargetKind::Test => Some("test"),
276            TargetKind::Bench => Some("bench"),
277            TargetKind::BuildScript => Some("custom-build"),
278            TargetKind::Other => None,
279        }
280    }
281}

302impl CargoWorkspace {
303    /// Fetches the metadata for the given `cargo_toml` manifest.
304    /// A successful result may contain another metadata error if the initial fetching failed but
305    /// the `--no-deps` retry succeeded.
306    ///
307    /// The sysroot is used to set the `RUSTUP_TOOLCHAIN` env var when invoking cargo
308    /// to ensure that the rustup proxy uses the correct toolchain.
309    pub fn fetch_metadata(
310        cargo_toml: &ManifestPath,
311        current_dir: &AbsPath,
312        config: &CargoMetadataConfig,
313        sysroot: &Sysroot,
314        no_deps: bool,
315        locked: bool,
316        progress: &dyn Fn(String),
317    ) -> anyhow::Result<(cargo_metadata::Metadata, Option<anyhow::Error>)> {
318        let res = Self::fetch_metadata_(
319            cargo_toml,
320            current_dir,
321            config,
322            sysroot,
323            no_deps,
324            locked,
325            progress,
326        );
327        if let Ok((_, Some(ref e))) = res {
328            tracing::warn!(
329                %cargo_toml,
330                ?e,
331                "`cargo metadata` failed, but retry with `--no-deps` succeeded"
332            );
333        }
334        res
335    }
336
337    fn fetch_metadata_(
338        cargo_toml: &ManifestPath,
339        current_dir: &AbsPath,
340        config: &CargoMetadataConfig,
341        sysroot: &Sysroot,
342        no_deps: bool,
343        locked: bool,
344        progress: &dyn Fn(String),
345    ) -> anyhow::Result<(cargo_metadata::Metadata, Option<anyhow::Error>)> {
346        let cargo = sysroot.tool(Tool::Cargo, current_dir);
347        let mut meta = MetadataCommand::new();
348        meta.cargo_path(cargo.get_program());
349        cargo.get_envs().for_each(|(var, val)| _ = meta.env(var, val.unwrap_or_default()));
350        config.extra_env.iter().for_each(|(var, val)| _ = meta.env(var, val));
351        meta.manifest_path(cargo_toml.to_path_buf());
352        match &config.features {
353            CargoFeatures::All => {
354                meta.features(CargoOpt::AllFeatures);
355            }
356            CargoFeatures::Selected { features, no_default_features } => {
357                if *no_default_features {
358                    meta.features(CargoOpt::NoDefaultFeatures);
359                }
360                if !features.is_empty() {
361                    meta.features(CargoOpt::SomeFeatures(features.clone()));
362                }
363            }
364        }
365        meta.current_dir(current_dir);
366
367        let mut other_options = vec![];
368        // cargo metadata only supports a subset of flags of what cargo usually accepts, and usually
369        // the only relevant flags for metadata here are unstable ones, so we pass those along
370        // but nothing else
371        let mut extra_args = config.extra_args.iter();
372        while let Some(arg) = extra_args.next() {
373            if arg == "-Z" {
374                if let Some(arg) = extra_args.next() {
375                    other_options.push("-Z".to_owned());
376                    other_options.push(arg.to_owned());
377                }
378            }
379        }
380
381        if !config.targets.is_empty() {
382            other_options.extend(
383                config.targets.iter().flat_map(|it| ["--filter-platform".to_owned(), it.clone()]),
384            );
385        }
386        // The manifest is a rust file, so this means its a script manifest
387        if cargo_toml.is_rust_manifest() {
388            // Deliberately don't set up RUSTC_BOOTSTRAP or a nightly override here, the user should
389            // opt into it themselves.
390            other_options.push("-Zscript".to_owned());
391        }
392        if locked {
393            other_options.push("--locked".to_owned());
394        }
395        if no_deps {
396            other_options.push("--no-deps".to_owned());
397        }
398        meta.other_options(other_options);
399
400        // FIXME: Fetching metadata is a slow process, as it might require
401        // calling crates.io. We should be reporting progress here, but it's
402        // unclear whether cargo itself supports it.
403        progress("metadata".to_owned());
404
405        (|| -> anyhow::Result<(_, _)> {
406            let output = meta.cargo_command().output()?;
407            if !output.status.success() {
408                let error = cargo_metadata::Error::CargoMetadata {
409                    stderr: String::from_utf8(output.stderr)?,
410                }
411                .into();
412                if !no_deps {
413                    // If we failed to fetch metadata with deps, try again without them.
414                    // This makes r-a still work partially when offline.
415                    if let Ok((metadata, _)) = Self::fetch_metadata_(
416                        cargo_toml,
417                        current_dir,
418                        config,
419                        sysroot,
420                        locked,
421                        true,
422                        progress,
423                    ) {
424                        return Ok((metadata, Some(error)));
425                    }
426                }
427                return Err(error);
428            }
429            let stdout = from_utf8(&output.stdout)?
430                .lines()
431                .find(|line| line.starts_with('{'))
432                .ok_or(cargo_metadata::Error::NoJson)?;
433            Ok((cargo_metadata::MetadataCommand::parse(stdout)?, None))
434        })()
435        .map(|(metadata, error)| {
436            (
437                metadata,
438                error.map(|e| e.context(format!("Failed to run `{:?}`", meta.cargo_command()))),
439            )
440        })
441        .with_context(|| format!("Failed to run `{:?}`", meta.cargo_command()))
442    }

444    pub fn new(
445        mut meta: cargo_metadata::Metadata,
446        ws_manifest_path: ManifestPath,
447        cargo_config_env: Env,
448        is_sysroot: bool,
449    ) -> CargoWorkspace {
450        let mut pkg_by_id = FxHashMap::default();
451        let mut packages = Arena::default();
452        let mut targets = Arena::default();
453
454        let ws_members = &meta.workspace_members;
455
456        let workspace_root = AbsPathBuf::assert(meta.workspace_root);
457        let target_directory = AbsPathBuf::assert(meta.target_directory);
458        let mut is_virtual_workspace = true;
459
460        meta.packages.sort_by(|a, b| a.id.cmp(&b.id));
461        for meta_pkg in meta.packages {
462            let cargo_metadata::Package {
463                name,
464                version,
465                id,
466                source,
467                targets: meta_targets,
468                features,
469                manifest_path,
470                repository,
471                edition,
472                metadata,
473                authors,
474                description,
475                homepage,
476                license,
477                license_file,
478                readme,
479                rust_version,
480                ..
481            } = meta_pkg;
482            let meta = from_value::<PackageMetadata>(metadata).unwrap_or_default();
483            let edition = match edition {
484                cargo_metadata::Edition::E2015 => Edition::Edition2015,
485                cargo_metadata::Edition::E2018 => Edition::Edition2018,
486                cargo_metadata::Edition::E2021 => Edition::Edition2021,
487                cargo_metadata::Edition::E2024 => Edition::Edition2024,
488                _ => {
489                    tracing::error!("Unsupported edition `{:?}`", edition);
490                    Edition::CURRENT
491                }
492            };
493            // We treat packages without source as "local" packages. That includes all members of
494            // the current workspace, as well as any path dependency outside the workspace.
495            let is_local = source.is_none();
496            let is_member = ws_members.contains(&id);
497
498            let manifest = ManifestPath::try_from(AbsPathBuf::assert(manifest_path)).unwrap();
499            is_virtual_workspace &= manifest != ws_manifest_path;
500            let pkg = packages.alloc(PackageData {
501                id: id.repr.clone(),
502                name,
503                version,
504                manifest: manifest.clone(),
505                targets: Vec::new(),
506                is_local,
507                is_member,
508                edition,
509                repository,
510                authors,
511                description,
512                homepage,
513                license,
514                license_file,
515                readme,
516                rust_version,
517                dependencies: Vec::new(),
518                features: features.into_iter().collect(),
519                active_features: Vec::new(),
520                metadata: meta.rust_analyzer.unwrap_or_default(),
521            });
522            let pkg_data = &mut packages[pkg];
523            pkg_by_id.insert(id, pkg);
524            for meta_tgt in meta_targets {
525                let cargo_metadata::Target { name, kind, required_features, src_path, .. } =
526                    meta_tgt;
527                let kind = TargetKind::new(&kind);
528                let tgt = targets.alloc(TargetData {
529                    package: pkg,
530                    name,
531                    root: if kind == TargetKind::Bin
532                        && manifest.extension().is_some_and(|ext| ext == "rs")
533                    {
534                        // cargo strips the script part of a cargo script away and places the
535                        // modified manifest file into a special target dir which is then used as
536                        // the source path. We don't want that, we want the original here so map it
537                        // back
538                        manifest.clone().into()
539                    } else {
540                        AbsPathBuf::assert(src_path)
541                    },
542                    kind,
543                    required_features,
544                });
545                pkg_data.targets.push(tgt);
546            }
547        }
548        for mut node in meta.resolve.map_or_else(Vec::new, |it| it.nodes) {
549            let &source = pkg_by_id.get(&node.id).unwrap();
550            node.deps.sort_by(|a, b| a.pkg.cmp(&b.pkg));
551            let dependencies = node
552                .deps
553                .iter()
554                .flat_map(|dep| DepKind::iter(&dep.dep_kinds).map(move |kind| (dep, kind)));
555            for (dep_node, kind) in dependencies {
556                let &pkg = pkg_by_id.get(&dep_node.pkg).unwrap();
557                let dep = PackageDependency { name: dep_node.name.clone(), pkg, kind };
558                packages[source].dependencies.push(dep);
559            }
560            packages[source].active_features.extend(node.features);
561        }
562
563        CargoWorkspace {
564            packages,
565            targets,
566            workspace_root,
567            target_directory,
568            manifest_path: ws_manifest_path,
569            is_virtual_workspace,
570            is_sysroot,
571            config_env: cargo_config_env,
572        }
573    }

606    pub fn parent_manifests(&self, manifest_path: &ManifestPath) -> Option<Vec<ManifestPath>> {
607        let mut found = false;
608        let parent_manifests = self
609            .packages()
610            .filter_map(|pkg| {
611                if !found && &self[pkg].manifest == manifest_path {
612                    found = true
613                }
614                self[pkg].dependencies.iter().find_map(|dep| {
615                    (&self[dep.pkg].manifest == manifest_path).then(|| self[pkg].manifest.clone())
616                })
617            })
618            .collect::<Vec<ManifestPath>>();
619
620        // some packages has this pkg as dep. return their manifests
621        if !parent_manifests.is_empty() {
622            return Some(parent_manifests);
623        }
624
625        // this pkg is inside this cargo workspace, fallback to workspace root
626        if found {
627            return Some(vec![
628                ManifestPath::try_from(self.workspace_root().join("Cargo.toml")).ok()?,
629            ]);
630        }
631
632        // not in this workspace
633        None
634    }

636    /// Returns the union of the features of all member crates in this workspace.
637    pub fn workspace_features(&self) -> FxHashSet<String> {
638        self.packages()
639            .filter_map(|package| {
640                let package = &self[package];
641                if package.is_member {
642                    Some(package.features.keys().cloned().chain(
643                        package.features.keys().map(|key| format!("{}/{key}", package.name)),
644                    ))
645                } else {
646                    None
647                }
648            })
649            .flatten()
650            .collect()
651    }
652
653    fn is_unique(&self, name: &str) -> bool {
654        self.packages.iter().filter(|(_, v)| v.name == name).count() == 1
655    }
