1//! Loads "sysroot" crate.
2//!
3//! One confusing point here is that normally sysroot is a bunch of `.rlib`s,
4//! but we can't process `.rlib` and need source code instead. The source code
5//! is typically installed with `rustup component add rust-src` command.

31#[derive(Debug, Clone, Eq, PartialEq)]
32pub enum RustLibSrcWorkspace {
33    Workspace { ws: CargoWorkspace, metadata_err: Option<String> },
34    Json(ProjectJson),
35    Stitched(stitched::Stitched),
36    Empty,

39impl fmt::Display for RustLibSrcWorkspace {
40    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
41        match self {
42            RustLibSrcWorkspace::Workspace { ws, .. } => {
43                write!(f, "workspace {}", ws.workspace_root())
44            }
45            RustLibSrcWorkspace::Json(json) => write!(f, "json {}", json.manifest_or_root()),
46            RustLibSrcWorkspace::Stitched(stitched) => {
47                write!(f, "stitched with {} crates", stitched.crates.len())
48            }
49            RustLibSrcWorkspace::Empty => write!(f, "empty"),
50        }
51    }
52}

144    /// Returns a command to run a tool preferring the cargo proxies if the sysroot exists.
145    pub fn tool(
146        &self,
147        tool: Tool,
148        current_dir: impl AsRef<Path>,
149        envs: &FxHashMap<String, Option<String>>,
150    ) -> Command {
151        match self.root() {
152            Some(root) => {
153                // special case rustc, we can look that up directly in the sysroot's bin folder
154                // as it should never invoke another cargo binary
155                if let Tool::Rustc = tool
156                    && let Some(path) =
157                        probe_for_binary(root.join("bin").join(Tool::Rustc.name()).into())
158                {
159                    return toolchain::command(path, current_dir, envs);
160                }
161
162                let mut cmd = toolchain::command(tool.prefer_proxy(), current_dir, envs);
163                if !envs.contains_key("RUSTUP_TOOLCHAIN")
164                    && std::env::var_os("RUSTUP_TOOLCHAIN").is_none()
165                {
166                    cmd.env("RUSTUP_TOOLCHAIN", AsRef::<std::path::Path>::as_ref(root));
167                }
168
169                cmd
170            }
171            _ => toolchain::command(tool.path(), current_dir, envs),
172        }
173    }

175    pub fn discover_proc_macro_srv(&self) -> Option<anyhow::Result<AbsPathBuf>> {
176        let root = self.root()?;
177        Some(
178            ["libexec", "lib"]
179                .into_iter()
180                .map(|segment| root.join(segment).join("rust-analyzer-proc-macro-srv"))
181                .find_map(|server_path| probe_for_binary(server_path.into()))
182                .map(AbsPathBuf::assert)
183                .ok_or_else(|| {
184                    anyhow::format_err!("cannot find proc-macro server in sysroot `{}`", root)
185                }),
186        )

189    fn assemble(
190        sysroot_dir: Option<Result<AbsPathBuf, anyhow::Error>>,
191        rust_lib_src_dir: Option<Result<AbsPathBuf, anyhow::Error>>,
192    ) -> Sysroot {
193        let mut errors = String::new();
194        let root = match sysroot_dir {
195            Some(Ok(sysroot_dir)) => Some(sysroot_dir),
196            Some(Err(e)) => {
197                format_to!(errors, "{e}\n");
198                None
199            }
200            None => None,
201        };
202        let rust_lib_src_root = match rust_lib_src_dir {
203            Some(Ok(rust_lib_src_dir)) => Some(rust_lib_src_dir),
204            Some(Err(e)) => {
205                format_to!(errors, "{e}\n");
206                None
207            }
208            None => None,
209        };
210        Sysroot {
211            root,
212            rust_lib_src_root,
213            workspace: RustLibSrcWorkspace::Empty,
214            error: errors.is_empty().not().then_some(errors),
215        }
216    }

218    pub fn load_workspace(
219        &self,
220        sysroot_source_config: &RustSourceWorkspaceConfig,
221        no_deps: bool,
222        target_dir: &Utf8Path,
223        progress: &dyn Fn(String),
224    ) -> Option<RustLibSrcWorkspace> {
225        assert!(matches!(self.workspace, RustLibSrcWorkspace::Empty), "workspace already loaded");
226        let Self { root: _, rust_lib_src_root: Some(src_root), workspace: _, error: _ } = self
227        else {
228            return None;
229        };
230        if let RustSourceWorkspaceConfig::CargoMetadata(cargo_config) = sysroot_source_config {
231            let library_manifest = ManifestPath::try_from(src_root.join("Cargo.toml")).unwrap();
232            if fs::metadata(&library_manifest).is_ok() {
233                match self.load_library_via_cargo(
234                    &library_manifest,
235                    src_root,
236                    target_dir,
237                    cargo_config,
238                    no_deps,
239                    progress,
240                ) {
241                    Ok(loaded) => return Some(loaded),
242                    Err(e) => {
243                        tracing::error!("`cargo metadata` failed on `{library_manifest}` : {e}")
244                    }
245                }
246            }
247            tracing::debug!("Stitching sysroot library: {src_root}");
248
249            let mut stitched = stitched::Stitched { crates: Default::default() };
250
251            for path in stitched::SYSROOT_CRATES.trim().lines() {
252                let name = path.split('/').next_back().unwrap();
253                let root = [format!("{path}/src/lib.rs"), format!("lib{path}/lib.rs")]
254                    .into_iter()
255                    .map(|it| src_root.join(it))
256                    .filter_map(|it| ManifestPath::try_from(it).ok())
257                    .find(|it| fs::metadata(it).is_ok());
258
259                if let Some(root) = root {
260                    stitched.crates.alloc(stitched::RustLibSrcCrateData {
261                        name: name.into(),
262                        root,
263                        deps: Vec::new(),
264                    });
265                }
266            }
267
268            if let Some(std) = stitched.by_name("std") {
269                for dep in stitched::STD_DEPS.trim().lines() {
270                    if let Some(dep) = stitched.by_name(dep) {
271                        stitched.crates[std].deps.push(dep)
272                    }
273                }
274            }
275
276            if let Some(alloc) = stitched.by_name("alloc") {
277                for dep in stitched::ALLOC_DEPS.trim().lines() {
278                    if let Some(dep) = stitched.by_name(dep) {
279                        stitched.crates[alloc].deps.push(dep)
280                    }
281                }
282            }
283
284            if let Some(proc_macro) = stitched.by_name("proc_macro") {
285                for dep in stitched::PROC_MACRO_DEPS.trim().lines() {
286                    if let Some(dep) = stitched.by_name(dep) {
287                        stitched.crates[proc_macro].deps.push(dep)
288                    }
289                }
290            }
291            return Some(RustLibSrcWorkspace::Stitched(stitched));
292        } else if let RustSourceWorkspaceConfig::Json(project_json) = sysroot_source_config {
293            return Some(RustLibSrcWorkspace::Json(project_json.clone()));
294        }
295
296        None
297    }

299    pub fn set_workspace(&mut self, workspace: RustLibSrcWorkspace) {
300        self.workspace = workspace;
301        if self.error.is_none()
302            && let Some(src_root) = &self.rust_lib_src_root
303        {
304            let has_core = match &self.workspace {
305                RustLibSrcWorkspace::Workspace { ws: workspace, .. } => {
306                    workspace.packages().any(|p| workspace[p].name == "core")
307                }
308                RustLibSrcWorkspace::Json(project_json) => project_json
309                    .crates()
310                    .filter_map(|(_, krate)| krate.display_name.clone())
311                    .any(|name| name.canonical_name().as_str() == "core"),
312                RustLibSrcWorkspace::Stitched(stitched) => stitched.by_name("core").is_some(),
313                RustLibSrcWorkspace::Empty => true,
314            };
315            if !has_core {
316                let var_note = if env::var_os("RUST_SRC_PATH").is_some() {
317                    " (env var `RUST_SRC_PATH` is set and may be incorrect, try unsetting it)"
318                } else {
319                    ", try running `rustup component add rust-src` to possibly fix this"
320                };
321                self.error =
322                    Some(format!("sysroot at `{src_root}` is missing a `core` library{var_note}",));
323            }
324        }
325    }

327    fn load_library_via_cargo(
328        &self,
329        library_manifest: &ManifestPath,
330        current_dir: &AbsPath,
331        target_dir: &Utf8Path,
332        cargo_config: &CargoMetadataConfig,
333        no_deps: bool,
334        progress: &dyn Fn(String),
335    ) -> Result<RustLibSrcWorkspace> {
336        tracing::debug!("Loading library metadata: {library_manifest}");
337        let mut cargo_config = cargo_config.clone();
338        // the sysroot uses `public-dependency`, so we make cargo think it's a nightly
339        cargo_config.extra_env.insert(
340            "__CARGO_TEST_CHANNEL_OVERRIDE_DO_NOT_USE_THIS".to_owned(),
341            Some("nightly".to_owned()),
342        );
343
344        // Make sure we never attempt to write to the sysroot
345        let locked = true;
346        let (mut res, err) =
347            FetchMetadata::new(library_manifest, current_dir, &cargo_config, self, no_deps)
348                .exec(target_dir, locked, progress)?;
349
350        // Patch out `rustc-std-workspace-*` crates to point to the real crates.
351        // This is done prior to `CrateGraph` construction to prevent de-duplication logic from failing.
352        let patches = {
353            let mut fake_core = None;
354            let mut fake_alloc = None;
355            let mut fake_std = None;
356            let mut real_core = None;
357            let mut real_alloc = None;
358            let mut real_std = None;
359            res.packages.iter().enumerate().for_each(|(idx, package)| {
360                match package.name.strip_prefix("rustc-std-workspace-") {
361                    Some("core") => fake_core = Some((idx, package.id.clone())),
362                    Some("alloc") => fake_alloc = Some((idx, package.id.clone())),
363                    Some("std") => fake_std = Some((idx, package.id.clone())),
364                    Some(_) => {
365                        tracing::warn!("unknown rustc-std-workspace-* crate: {}", package.name)
366                    }
367                    None => match &**package.name {
368                        "core" => real_core = Some(package.id.clone()),
369                        "alloc" => real_alloc = Some(package.id.clone()),
370                        "std" => real_std = Some(package.id.clone()),
371                        _ => (),
372                    },
373                }
374            });
375
376            [fake_core.zip(real_core), fake_alloc.zip(real_alloc), fake_std.zip(real_std)]
377                .into_iter()
378                .flatten()
379        };
380
381        if let Some(resolve) = res.resolve.as_mut() {
382            resolve.nodes.retain_mut(|node| {
383                // Replace `rustc-std-workspace` crate with the actual one in the dependency list
384                node.deps.iter_mut().for_each(|dep| {
385                    let real_pkg = patches.clone().find(|((_, fake_id), _)| *fake_id == dep.pkg);
386                    if let Some((_, real)) = real_pkg {
387                        dep.pkg = real;
388                    }
389                });
390                // Remove this node if it's a fake one
391                !patches.clone().any(|((_, fake), _)| fake == node.id)
392            });
393        }
394        // Remove the fake ones from the package list
395        patches.map(|((idx, _), _)| idx).sorted().rev().for_each(|idx| {
396            res.packages.remove(idx);
397        });
398
399        let cargo_workspace =
400            CargoWorkspace::new(res, library_manifest.clone(), Default::default(), true);
401        Ok(RustLibSrcWorkspace::Workspace {
402            ws: cargo_workspace,
403            metadata_err: err.map(|e| format!("{e:#}")),
404        })
405    }
406}

408fn discover_sysroot_dir(
409    current_dir: &AbsPath,
410    extra_env: &FxHashMap<String, Option<String>>,
411) -> Result<AbsPathBuf> {
412    let mut rustc = toolchain::command(Tool::Rustc.path(), current_dir, extra_env);
413    rustc.current_dir(current_dir).args(["--print", "sysroot"]);
414    tracing::debug!("Discovering sysroot by {:?}", rustc);
415    let stdout = utf8_stdout(&mut rustc)?;
416    Ok(AbsPathBuf::assert(Utf8PathBuf::from(stdout)))
417}

419fn discover_rust_lib_src_dir(sysroot_path: &AbsPathBuf) -> Option<AbsPathBuf> {
420    if let Ok(path) = env::var("RUST_SRC_PATH") {
421        if let Ok(path) = AbsPathBuf::try_from(path.as_str()) {
422            let core = path.join("core");
423            if fs::metadata(&core).is_ok() {
424                tracing::debug!("Discovered sysroot by RUST_SRC_PATH: {path}");
425                return Some(path);
426            }
427            tracing::debug!("RUST_SRC_PATH is set, but is invalid (no core: {core:?}), ignoring");
428        } else {
429            tracing::debug!("RUST_SRC_PATH is set, but is invalid, ignoring");
430        }
431    }
432
433    get_rust_lib_src(sysroot_path)
434}
435
436fn discover_rust_lib_src_dir_or_add_component(
437    sysroot_path: &AbsPathBuf,
438    current_dir: &AbsPath,
439    extra_env: &FxHashMap<String, Option<String>>,
440) -> Result<AbsPathBuf> {
441    discover_rust_lib_src_dir(sysroot_path)
442        .or_else(|| {
443            let mut rustup = toolchain::command(Tool::Rustup.prefer_proxy(), current_dir, extra_env);
444            rustup.args(["component", "add", "rust-src"]);
445            tracing::info!("adding rust-src component by {:?}", rustup);
446            utf8_stdout(&mut rustup).ok()?;
447            get_rust_lib_src(sysroot_path)
448        })
449        .ok_or_else(|| {
450            tracing::error!(%sysroot_path, "can't load standard library, try installing `rust-src`");
451            format_err!(
452                "\
453can't load standard library from sysroot
454{sysroot_path}
455(discovered via `rustc --print sysroot`)
456try installing `rust-src` the same way you installed `rustc`"
457            )
458        })
459}

461fn get_rustc_src(sysroot_path: &AbsPath) -> Option<ManifestPath> {
462    let rustc_src = sysroot_path.join("lib/rustlib/rustc-src/rust/compiler/rustc/Cargo.toml");
463    let rustc_src = ManifestPath::try_from(rustc_src).ok()?;
464    tracing::debug!("checking for rustc source code: {rustc_src}");
465    if fs::metadata(&rustc_src).is_ok() { Some(rustc_src) } else { None }
466}
467
468fn get_rust_lib_src(sysroot_path: &AbsPath) -> Option<AbsPathBuf> {
469    let rust_lib_src = sysroot_path.join("lib/rustlib/src/rust/library");
470    tracing::debug!("checking sysroot library: {rust_lib_src}");
471    if fs::metadata(&rust_lib_src).is_ok() { Some(rust_lib_src) } else { None }
472}

488    impl ops::Index<RustLibSrcCrate> for Stitched {
489        type Output = RustLibSrcCrateData;
490        fn index(&self, index: RustLibSrcCrate) -> &RustLibSrcCrateData {
491            &self.crates[index]
492        }

494
495    impl Stitched {
496        pub(crate) fn public_deps(
497            &self,
498        ) -> impl Iterator<Item = (CrateName, RustLibSrcCrate, bool)> + '_ {
499            // core is added as a dependency before std in order to
500            // mimic rustcs dependency order
501            [("core", true), ("alloc", false), ("std", true), ("test", false)]
502                .into_iter()
503                .filter_map(move |(name, prelude)| {
504                    Some((CrateName::new(name).unwrap(), self.by_name(name)?, prelude))
505                })
506        }
507
508        pub(crate) fn proc_macro(&self) -> Option<RustLibSrcCrate> {
509            self.by_name("proc_macro")
510        }
511
512        pub(crate) fn crates(&self) -> impl ExactSizeIterator<Item = RustLibSrcCrate> + '_ {
513            self.crates.iter().map(|(id, _data)| id)
514        }
515
516        pub(super) fn by_name(&self, name: &str) -> Option<RustLibSrcCrate> {
517            let (id, _data) = self.crates.iter().find(|(_id, data)| data.name == name)?;
518            Some(id)
519        }
520    }

522    pub(crate) type RustLibSrcCrate = Idx<RustLibSrcCrateData>;
523
524    #[derive(Debug, Clone, Eq, PartialEq)]
525    pub(crate) struct RustLibSrcCrateData {
526        pub(crate) name: String,
527        pub(crate) root: ManifestPath,
528        pub(crate) deps: Vec<RustLibSrcCrate>,
529    }
