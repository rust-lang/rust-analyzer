8/// Recreates the compile-time environment variables that Cargo sets.
9///
10/// Should be synced with
11/// <https://doc.rust-lang.org/cargo/reference/environment-variables.html#environment-variables-cargo-sets-for-crates>
12///
13/// FIXME: ask Cargo to provide this data instead of re-deriving.
14pub(crate) fn inject_cargo_package_env(env: &mut Env, package: &PackageData) {
15    // FIXME: Missing variables:
16    // CARGO_BIN_NAME, CARGO_BIN_EXE_<name>
17
18    let manifest_dir = package.manifest.parent();
19    env.set("CARGO_MANIFEST_DIR", manifest_dir.as_str());
20    env.set("CARGO_MANIFEST_PATH", package.manifest.as_str());
21
22    env.set("CARGO_PKG_VERSION", package.version.to_string());
23    env.set("CARGO_PKG_VERSION_MAJOR", package.version.major.to_string());
24    env.set("CARGO_PKG_VERSION_MINOR", package.version.minor.to_string());
25    env.set("CARGO_PKG_VERSION_PATCH", package.version.patch.to_string());
26    env.set("CARGO_PKG_VERSION_PRE", package.version.pre.to_string());
27
28    env.set("CARGO_PKG_AUTHORS", package.authors.join(":"));
29
30    env.set("CARGO_PKG_NAME", package.name.clone());
31    env.set("CARGO_PKG_DESCRIPTION", package.description.as_deref().unwrap_or_default());
32    env.set("CARGO_PKG_HOMEPAGE", package.homepage.as_deref().unwrap_or_default());
33    env.set("CARGO_PKG_REPOSITORY", package.repository.as_deref().unwrap_or_default());
34    env.set("CARGO_PKG_LICENSE", package.license.as_deref().unwrap_or_default());
35    env.set(
36        "CARGO_PKG_LICENSE_FILE",
37        package.license_file.as_ref().map(ToString::to_string).unwrap_or_default(),
38    );
39    env.set(
40        "CARGO_PKG_README",
41        package.readme.as_ref().map(ToString::to_string).unwrap_or_default(),
42    );
43
44    env.set(
45        "CARGO_PKG_RUST_VERSION",
46        package.rust_version.as_ref().map(ToString::to_string).unwrap_or_default(),
47    );
48}

63pub(crate) fn cargo_config_env(
64    config: &Option<CargoConfigFile>,
65    extra_env: &FxHashMap<String, Option<String>>,
66) -> Env {
67    use toml::de::*;
68
69    let mut env = Env::default();
70    env.extend(extra_env.iter().filter_map(|(k, v)| v.as_ref().map(|v| (k.clone(), v.clone()))));
71
72    let Some(config_reader) = config.as_ref().and_then(|c| c.read()) else {
73        return env;
74    };
75    let Some(env_toml) = config_reader.get(["env"]).and_then(|it| it.as_table()) else {
76        return env;
77    };
78
79    for (key, entry) in env_toml {
80        let key = key.as_ref().as_ref();
81        let value = match entry.as_ref() {
82            DeValue::String(s) => String::from(s.clone()),
83            DeValue::Table(entry) => {
84                // Each entry MUST have a `value` key.
85                let Some(map) = entry.get("value").and_then(|v| v.as_ref().as_str()) else {
86                    continue;
87                };
88                // If the entry already exists in the environment AND the `force` key is not set to
89                // true, then don't overwrite the value.
90                if extra_env.get(key).is_some_and(Option::is_some)
91                    && !entry.get("force").and_then(|v| v.as_ref().as_bool()).unwrap_or(false)
92                {
93                    continue;
94                }
95
96                if let Some(base) = entry.get("relative").and_then(|v| {
97                    if v.as_ref().as_bool().is_some_and(std::convert::identity) {
98                        config_reader.get_origin_root(v)
99                    } else {
100                        None
101                    }
102                }) {
103                    base.join(map).to_string()
104                } else {
105                    map.to_owned()
106                }
107            }
108            _ => continue,
109        };
110
111        env.insert(key, value);
112    }
113
114    env
115}

117#[test]
118fn parse_output_cargo_config_env_works() {
119    use itertools::Itertools;
120
121    let cwd = paths::AbsPathBuf::try_from(
122        paths::Utf8PathBuf::try_from(std::env::current_dir().unwrap()).unwrap(),
123    )
124    .unwrap();
125    let config_path = cwd.join(".cargo").join("config.toml");
126    let raw = r#"
127env.CARGO_WORKSPACE_DIR.relative = true
128env.CARGO_WORKSPACE_DIR.value = ""
129env.INVALID.relative = "invalidbool"
130env.INVALID.value = "../relative"
131env.RELATIVE.relative = true
132env.RELATIVE.value = "../relative"
133env.TEST.value = "test"
134env.FORCED.value = "test"
135env.FORCED.force = true
136env.UNFORCED.value = "test"
137env.UNFORCED.forced = false
138env.OVERWRITTEN.value = "test"
139env.NOT_AN_OBJECT = "value"
140"#;
141    let raw = raw.lines().map(|l| format!("{l} # {config_path}")).join("\n");
142    let config = CargoConfigFile::from_string_for_test(raw);
143    let extra_env = [
144        ("FORCED", Some("ignored")),
145        ("UNFORCED", Some("newvalue")),
146        ("OVERWRITTEN", Some("newvalue")),
147        ("TEST", None),
148    ]
149    .iter()
150    .map(|(k, v)| (k.to_string(), v.map(ToString::to_string)))
151    .collect();
152    let env = cargo_config_env(&Some(config), &extra_env);
153    assert_eq!(env.get("CARGO_WORKSPACE_DIR").as_deref(), Some(cwd.join("").as_str()));
154    assert_eq!(env.get("RELATIVE").as_deref(), Some(cwd.join("../relative").as_str()));
155    assert_eq!(env.get("INVALID").as_deref(), Some("../relative"));
156    assert_eq!(env.get("TEST").as_deref(), Some("test"));
157    assert_eq!(env.get("FORCED").as_deref(), Some("test"));
158    assert_eq!(env.get("UNFORCED").as_deref(), Some("newvalue"));
159    assert_eq!(env.get("OVERWRITTEN").as_deref(), Some("newvalue"));
160    assert_eq!(env.get("NOT_AN_OBJECT").as_deref(), Some("value"));
161}
