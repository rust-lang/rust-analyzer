2use std::path::Path;
3
4use anyhow::Context;
5use rustc_hash::FxHashMap;
6use toolchain::Tool;
7
8use crate::{ManifestPath, Sysroot, toolchain_info::QueryConfig, utf8_stdout};

10/// For cargo, runs `cargo -Zunstable-options config get build.target` to get the configured project target(s).
11/// For rustc, runs `rustc --print -vV` to get the host target.
12pub fn get(
13    config: QueryConfig<'_>,
14    target: Option<&str>,
15    extra_env: &FxHashMap<String, String>,
16) -> anyhow::Result<Vec<String>> {
17    let _p = tracing::info_span!("target_tuple::get").entered();
18    if let Some(target) = target {
19        return Ok(vec![target.to_owned()]);
20    }
21
22    let (sysroot, current_dir) = match config {
23        QueryConfig::Cargo(sysroot, cargo_toml) => {
24            match cargo_config_build_target(cargo_toml, extra_env, sysroot) {
25                Some(it) => return Ok(it),
26                None => (sysroot, cargo_toml.parent().as_ref()),
27            }
28        }
29        QueryConfig::Rustc(sysroot, current_dir) => (sysroot, current_dir),
30    };
31    rustc_discover_host_tuple(extra_env, sysroot, current_dir).map(|it| vec![it])
32}
33
34fn rustc_discover_host_tuple(
35    extra_env: &FxHashMap<String, String>,
36    sysroot: &Sysroot,
37    current_dir: &Path,
38) -> anyhow::Result<String> {
39    let mut cmd = sysroot.tool(Tool::Rustc, current_dir);
40    cmd.envs(extra_env);
41    cmd.arg("-vV");
42    let stdout = utf8_stdout(&mut cmd)
43        .with_context(|| format!("unable to discover host platform via `{cmd:?}`"))?;
44    let field = "host: ";
45    let target = stdout.lines().find_map(|l| l.strip_prefix(field));
46    if let Some(target) = target {
47        Ok(target.to_owned())
48    } else {
49        // If we fail to resolve the host platform, it's not the end of the world.
50        Err(anyhow::format_err!("rustc -vV did not report host platform, got:\n{}", stdout))
51    }
52}

54fn cargo_config_build_target(
55    cargo_toml: &ManifestPath,
56    extra_env: &FxHashMap<String, String>,
57    sysroot: &Sysroot,
58) -> Option<Vec<String>> {
59    let mut cmd = sysroot.tool(Tool::Cargo, cargo_toml.parent());
60    cmd.envs(extra_env);
61    cmd.current_dir(cargo_toml.parent()).env("RUSTC_BOOTSTRAP", "1");
62    cmd.args(["-Z", "unstable-options", "config", "get", "build.target"]);
63    // if successful we receive `build.target = "target-tuple"`
64    // or `build.target = ["<target 1>", ..]`
65    // this might be `error: config value `build.target` is not set` in which case we
66    // don't wanna log the error
67    utf8_stdout(&mut cmd).and_then(parse_output_cargo_config_build_target).ok()
68}
69
70// Parses `"build.target = [target-tuple, target-tuple, ...]"` or `"build.target = "target-tuple"`
71fn parse_output_cargo_config_build_target(stdout: String) -> anyhow::Result<Vec<String>> {
72    let trimmed = stdout.trim_start_matches("build.target = ").trim_matches('"');
73
74    if !trimmed.starts_with('[') {
75        return Ok([trimmed.to_owned()].to_vec());
76    }
77
78    serde_json::from_str(trimmed).context("Failed to parse `build.target` as an array of target")
79}

85    use crate::{ManifestPath, Sysroot};

89    #[test]
90    fn cargo() {
91        let manifest_path = concat!(env!("CARGO_MANIFEST_DIR"), "/Cargo.toml");
92        let sysroot = Sysroot::empty();
93        let manifest_path =
94            ManifestPath::try_from(AbsPathBuf::assert(Utf8PathBuf::from(manifest_path))).unwrap();
95        let cfg = QueryConfig::Cargo(&sysroot, &manifest_path);
96        assert!(get(cfg, None, &FxHashMap::default()).is_ok());
97    }

101        let sysroot = Sysroot::empty();
