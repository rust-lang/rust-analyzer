36fn string_to_diagnostic_code_leaky(code: &str) -> DiagnosticCode {
37    thread_local! {
38        static LEAK_STORE: RefCell<FxHashMap<String, DiagnosticCode>> = RefCell::new(FxHashMap::default());
39    }
40    LEAK_STORE.with_borrow_mut(|s| match s.get(code) {
41        Some(c) => *c,
42        None => {
43            let v = DiagnosticCode::RustcHardError(format!("E{code}").leak());
44            s.insert(code.to_owned(), v);
45            v
46        }
47    })
48}

50fn detect_errors_from_rustc_stderr_file(p: PathBuf) -> FxHashMap<DiagnosticCode, usize> {
51    let text = read_to_string(p).unwrap();
52    let mut result = FxHashMap::default();
53    {
54        let mut text = &*text;
55        while let Some(p) = text.find("error[E") {
56            text = &text[p + 7..];
57            let code = string_to_diagnostic_code_leaky(&text[..4]);
58            *result.entry(code).or_insert(0) += 1;
59        }
60    }
61    result
62}

64impl Tester {
65    fn new() -> Result<Self> {
66        let mut path = AbsPathBuf::assert_utf8(std::env::temp_dir());
67        path.push("ra-rustc-test");
68        let tmp_file = path.join("ra-rustc-test.rs");
69        std::fs::write(&tmp_file, "")?;
70        let cargo_config = CargoConfig {
71            sysroot: Some(RustLibSource::Discover),
72            all_targets: true,
73            set_test: true,
74            ..Default::default()
75        };
76
77        let mut sysroot = Sysroot::discover(tmp_file.parent().unwrap(), &cargo_config.extra_env);
78        let loaded_sysroot = sysroot.load_workspace(
79            &RustSourceWorkspaceConfig::default_cargo(),
80            false,
81            &Utf8PathBuf::default(),
82            &|_| (),
83        );
84        if let Some(loaded_sysroot) = loaded_sysroot {
85            sysroot.set_workspace(loaded_sysroot);
86        }
87
88        let target_data = target_data::get(
89            QueryConfig::Rustc(&sysroot, tmp_file.parent().unwrap().as_ref()),
90            None,
91            &cargo_config.extra_env,
92        );
93
94        let workspace = ProjectWorkspace {
95            kind: ProjectWorkspaceKind::DetachedFile {
96                file: ManifestPath::try_from(tmp_file).unwrap(),
97                cargo: None,
98            },
99            sysroot,
100            rustc_cfg: vec![],
101            toolchain: None,
102            target: target_data.map_err(|it| it.to_string().into()),
103            cfg_overrides: Default::default(),
104            extra_includes: vec![],
105            set_test: true,
106        };
107        let load_cargo_config = LoadCargoConfig {
108            load_out_dirs_from_check: false,
109            with_proc_macro_server: ProcMacroServerChoice::Sysroot,
110            prefill_caches: false,
111        };
112        let (db, _vfs, _proc_macro) =
113            load_workspace(workspace, &cargo_config.extra_env, &load_cargo_config)?;
114        let host = AnalysisHost::with_database(db);
115        let db = host.raw_database();
116        let krates = Crate::all(db);
117        let root_crate = krates.iter().cloned().find(|krate| krate.origin(db).is_local()).unwrap();
118        let root_file = root_crate.root_file(db);
119        Ok(Self {
120            host,
121            root_file,
122            pass_count: 0,
123            ignore_count: 0,
124            fail_count: 0,
125            stopwatch: StopWatch::start(),
126        })
127    }

129    fn test(&mut self, p: PathBuf) {
130        println!("{}", p.display());
131        if p.parent().unwrap().file_name().unwrap() == "auxiliary" {
132            // These are not tests
133            return;
134        }
135        if IGNORED_TESTS.iter().any(|ig| p.file_name().is_some_and(|x| x == *ig)) {
136            println!("{p:?} IGNORE");
137            self.ignore_count += 1;
138            return;
139        }
140        let stderr_path = p.with_extension("stderr");
141        let expected = if stderr_path.exists() {
142            detect_errors_from_rustc_stderr_file(stderr_path)
143        } else {
144            FxHashMap::default()
145        };
146        let text = read_to_string(&p).unwrap();
147        let mut change = ChangeWithProcMacros::default();
148        // Ignore unstable tests, since they move too fast and we do not intend to support all of them.
149        let mut ignore_test = text.contains("#![feature");
150        // Ignore test with extern crates, as this infra don't support them yet.
151        ignore_test |= text.contains("// aux-build:") || text.contains("// aux-crate:");
152        // Ignore test with extern modules similarly.
153        ignore_test |= text.contains("mod ");
154        // These should work, but they don't, and I don't know why, so ignore them.
155        ignore_test |= text.contains("extern crate proc_macro");
156        let should_have_no_error = text.contains("// check-pass")
157            || text.contains("// build-pass")
158            || text.contains("// run-pass");
159        change.change_file(self.root_file, Some(text));
160        self.host.apply_change(change);
161        let diagnostic_config = DiagnosticsConfig::test_sample();
162
163        let res = std::thread::scope(|s| {
164            let worker = Builder::new()
165                .stack_size(40 * 1024 * 1024)
166                .spawn_scoped(s, {
167                    let diagnostic_config = &diagnostic_config;
168                    let main = std::thread::current();
169                    let analysis = self.host.analysis();
170                    let root_file = self.root_file;
171                    move || {
172                        let res = std::panic::catch_unwind(AssertUnwindSafe(move || {
173                            analysis.full_diagnostics(
174                                diagnostic_config,
175                                ide::AssistResolveStrategy::None,
176                                root_file,
177                            )
178                        }));
179                        main.unpark();
180                        res
181                    }
182                })
183                .unwrap();
184
185            let timeout = Duration::from_secs(5);
186            let now = Instant::now();
187            while now.elapsed() <= timeout && !worker.is_finished() {
188                std::thread::park_timeout(timeout - now.elapsed());
189            }
190
191            if !worker.is_finished() {
192                // attempt to cancel the worker, won't work for chalk hangs unfortunately
193                self.host.request_cancellation();
194            }
195            worker.join().and_then(identity)
196        });
197        let mut actual = FxHashMap::default();
198        let panicked = match res {
199            Err(e) => Some(Either::Left(e)),
200            Ok(Ok(diags)) => {
201                for diag in diags {
202                    if !matches!(diag.code, DiagnosticCode::RustcHardError(_)) {
203                        continue;
204                    }
205                    if !should_have_no_error && !SUPPORTED_DIAGNOSTICS.contains(&diag.code) {
206                        continue;
207                    }
208                    *actual.entry(diag.code).or_insert(0) += 1;
209                }
210                None
211            }
212            Ok(Err(e)) => Some(Either::Right(e)),
213        };
214        // Ignore tests with diagnostics that we don't emit.
215        ignore_test |= expected.keys().any(|k| !SUPPORTED_DIAGNOSTICS.contains(k));
216        if ignore_test {
217            println!("{p:?} IGNORE");
218            self.ignore_count += 1;
219        } else if let Some(panic) = panicked {
220            match panic {
221                Either::Left(panic) => {
222                    if let Some(msg) = panic
223                        .downcast_ref::<String>()
224                        .map(String::as_str)
225                        .or_else(|| panic.downcast_ref::<&str>().copied())
226                    {
227                        println!("{msg:?} ")
228                    }
229                    println!("{p:?} PANIC");
230                }
231                Either::Right(_) => println!("{p:?} CANCELLED"),
232            }
233            self.fail_count += 1;
234        } else if actual == expected {
235            println!("{p:?} PASS");
236            self.pass_count += 1;
237        } else {
238            println!("{p:?} FAIL");
239            println!("actual   (r-a)   = {actual:?}");
240            println!("expected (rustc) = {expected:?}");
241            self.fail_count += 1;
242        }
243    }
244
245    fn report(&mut self) {
246        println!(
247            "Pass count = {}, Fail count = {}, Ignore count = {}",
248            self.pass_count, self.fail_count, self.ignore_count
249        );
250        println!("Testing time and memory = {}", self.stopwatch.elapsed());
251        report_metric("rustc failed tests", self.fail_count, "#");
252        report_metric("rustc testing time", self.stopwatch.elapsed().time.as_millis() as u64, "ms");
253    }
254}

298impl flags::RustcTests {
299    pub fn run(self) -> Result<()> {
300        let mut tester = Tester::new()?;
301        let walk_dir = WalkDir::new(self.rustc_repo.join("tests/ui"));
302        eprintln!("Running tests for tests/ui");
303        for i in walk_dir {
304            let i = i?;
305            let p = i.into_path();
306            if let Some(f) = &self.filter
307                && !p.as_os_str().to_string_lossy().contains(f)
308            {
309                continue;
310            }
311            if p.extension().is_none_or(|x| x != "rs") {
312                continue;
313            }
314            if let Err(e) = std::panic::catch_unwind({
315                let tester = AssertUnwindSafe(&mut tester);
316                let p = p.clone();
317                move || {
318                    let _guard = base_db::DbPanicContext::enter(p.display().to_string());
319                    { tester }.0.test(p);
320                }
321            }) {
322                std::panic::resume_unwind(e);
323            }
324        }
325        tester.report();
326        Ok(())
327    }
