312            shuffle(&mut rng, &mut bodies);
313        }
314
315        hir::attach_db(db, || {
316            if !self.skip_lowering {
317                self.run_body_lowering(db, &vfs, &bodies, verbosity);

320            if !self.skip_inference {

327
328            if !self.skip_data_layout {
329                self.run_data_layout(db, &adts, verbosity);

331
332            if !self.skip_const_eval {
333                self.run_const_eval(db, &bodies, verbosity);

335        });
336
337        file_ids.sort();
338        file_ids.dedup();
339
340        if self.run_all_ide_things {
341            self.run_ide_things(host.analysis(), &file_ids, db, &vfs, verbosity);
342        }
343
344        if self.run_term_search {
345            self.run_term_search(&workspace, db, &vfs, &file_ids, verbosity);
346        }
347
348        hir::clear_tls_solver_cache();
349
350        let db = host.raw_database_mut();
351        db.trigger_lru_eviction();
352
353        let total_span = analysis_sw.elapsed();
354        eprintln!("{:<20} {total_span}", "Total:");
355        report_metric("total time", total_span.time.as_millis() as u64, "ms");
356        if let Some(instructions) = total_span.instructions {
357            report_metric("total instructions", instructions, "#instr");
358        }
359        report_metric("total memory", total_span.memory.allocated.megabytes() as u64, "MB");
360
361        if verbosity.is_verbose() {
362            print_memory_usage(host, vfs);
363        }
364
365        Ok(())
366    }

368    fn run_data_layout(&self, db: &RootDatabase, adts: &[hir::Adt], verbosity: Verbosity) {
369        let mut sw = self.stop_watch();
370        let mut all = 0;
371        let mut fail = 0;
372        for &a in adts {
373            let interner = DbInterner::new_with(db, Some(a.krate(db).base()), None);
374            let generic_params = db.generic_params(a.into());
375            if generic_params.iter_type_or_consts().next().is_some()
376                || generic_params.iter_lt().next().is_some()
377            {
378                // Data types with generics don't have layout.
379                continue;
380            }
381            all += 1;
382            let Err(e) = db.layout_of_adt(
383                hir_def::AdtId::from(a),
384                GenericArgs::new_from_iter(interner, []),
385                db.trait_environment(a.into()),
386            ) else {
387                continue;
388            };
389            if verbosity.is_spammy() {
390                let full_name = full_name_of_item(db, a.module(db), a.name(db));
391                println!("Data layout for {full_name} failed due {e:?}");
392            }
393            fail += 1;
394        }
395        let data_layout_time = sw.elapsed();
396        eprintln!("{:<20} {}", "Data layouts:", data_layout_time);
397        eprintln!("Failed data layouts: {fail} ({}%)", percentage(fail, all));
398        report_metric("failed data layouts", fail, "#");
399        report_metric("data layout time", data_layout_time.time.as_millis() as u64, "ms");
400    }

402    fn run_const_eval(&self, db: &RootDatabase, bodies: &[DefWithBody], verbosity: Verbosity) {
403        let len = bodies
404            .iter()
405            .filter(|body| matches!(body, DefWithBody::Const(_) | DefWithBody::Static(_)))
406            .count();
407        let mut bar = match verbosity {
408            Verbosity::Quiet | Verbosity::Spammy => ProgressReport::hidden(),
409            _ if self.parallel || self.output.is_some() => ProgressReport::hidden(),
410            _ => ProgressReport::new(len),
411        };
412
413        let mut sw = self.stop_watch();
414        let mut all = 0;
415        let mut fail = 0;
416        for &b in bodies {
417            bar.set_message(move || format!("const eval: {}", full_name(db, b, b.module(db))));
418            let res = match b {
419                DefWithBody::Const(c) => c.eval(db),
420                DefWithBody::Static(s) => s.eval(db),
421                _ => continue,
422            };
423            bar.inc(1);
424            all += 1;
425            let Err(error) = res else {
426                continue;
427            };
428            if verbosity.is_spammy() {
429                let full_name =
430                    full_name_of_item(db, b.module(db), b.name(db).unwrap_or(Name::missing()));
431                bar.println(format!("Const eval for {full_name} failed due {error:?}"));
432            }
433            fail += 1;
434        }
435        bar.finish_and_clear();
436        let const_eval_time = sw.elapsed();
437        eprintln!("{:<20} {}", "Const evaluation:", const_eval_time);
438        eprintln!("Failed const evals: {fail} ({}%)", percentage(fail, all));
439        report_metric("failed const evals", fail, "#");
440        report_metric("const eval time", const_eval_time.time.as_millis() as u64, "ms");
441    }

443    /// Invariant: `file_ids` must be sorted and deduped before passing into here
444    fn run_term_search(
445        &self,
446        ws: &ProjectWorkspace,
447        db: &RootDatabase,
448        vfs: &Vfs,
449        file_ids: &[EditionedFileId],
450        verbosity: Verbosity,
451    ) {
452        let cargo_config = CargoConfig {
453            sysroot: match self.no_sysroot {
454                true => None,
455                false => Some(RustLibSource::Discover),
456            },
457            all_targets: true,
458            ..Default::default()
459        };
460
461        let mut bar = match verbosity {
462            Verbosity::Quiet | Verbosity::Spammy => ProgressReport::hidden(),
463            _ if self.parallel || self.output.is_some() => ProgressReport::hidden(),
464            _ => ProgressReport::new(file_ids.len()),
465        };
466
467        #[derive(Debug, Default)]
468        struct Acc {
469            tail_expr_syntax_hits: u64,
470            tail_expr_no_term: u64,
471            total_tail_exprs: u64,
472            error_codes: FxHashMap<String, u32>,
473            syntax_errors: u32,
474        }
475
476        let mut acc: Acc = Default::default();
477        bar.tick();
478        let mut sw = self.stop_watch();
479
480        for &file_id in file_ids {
481            let file_id = file_id.editioned_file_id(db);
482            let sema = hir::Semantics::new(db);
483            let display_target = match sema.first_crate(file_id.file_id()) {
484                Some(krate) => krate.to_display_target(sema.db),
485                None => continue,
486            };
487
488            let parse = sema.parse_guess_edition(file_id.into());
489            let file_txt = db.file_text(file_id.into());
490            let path = vfs.file_path(file_id.into()).as_path().unwrap();
491
492            for node in parse.syntax().descendants() {
493                let expr = match syntax::ast::Expr::cast(node.clone()) {
494                    Some(it) => it,
495                    None => continue,
496                };
497                let block = match syntax::ast::BlockExpr::cast(expr.syntax().clone()) {
498                    Some(it) => it,
499                    None => continue,
500                };
501                let target_ty = match sema.type_of_expr(&expr) {
502                    Some(it) => it.adjusted(),
503                    None => continue, // Failed to infer type
504                };
505
506                let expected_tail = match block.tail_expr() {
507                    Some(it) => it,
508                    None => continue,
509                };
510
511                if expected_tail.is_block_like() {
512                    continue;
513                }
514
515                let range = sema.original_range(expected_tail.syntax()).range;
516                let original_text: String = db
517                    .file_text(file_id.into())
518                    .text(db)
519                    .chars()
520                    .skip(usize::from(range.start()))
521                    .take(usize::from(range.end()) - usize::from(range.start()))
522                    .collect();
523
524                let scope = match sema.scope(expected_tail.syntax()) {
525                    Some(it) => it,
526                    None => continue,
527                };
528
529                let ctx = hir::term_search::TermSearchCtx {
530                    sema: &sema,
531                    scope: &scope,
532                    goal: target_ty,
533                    config: hir::term_search::TermSearchConfig {
534                        enable_borrowcheck: true,
535                        ..Default::default()
536                    },
537                };
538                let found_terms = hir::term_search::term_search(&ctx);
539
540                if found_terms.is_empty() {
541                    acc.tail_expr_no_term += 1;
542                    acc.total_tail_exprs += 1;
543                    // println!("\n{original_text}\n");
544                    continue;
545                };
546
547                fn trim(s: &str) -> String {
548                    s.chars().filter(|c| !c.is_whitespace()).collect()
549                }
550
551                let todo = syntax::ast::make::ext::expr_todo().to_string();
552                let mut formatter = |_: &hir::Type<'_>| todo.clone();
553                let mut syntax_hit_found = false;
554                for term in found_terms {
555                    let generated = term
556                        .gen_source_code(
557                            &scope,
558                            &mut formatter,
559                            FindPathConfig {
560                                prefer_no_std: false,
561                                prefer_prelude: true,
562                                prefer_absolute: false,
563                                allow_unstable: true,
564                            },
565                            display_target,
566                        )
567                        .unwrap();
568                    syntax_hit_found |= trim(&original_text) == trim(&generated);
569
570                    // Validate if type-checks
571                    let mut txt = file_txt.text(db).to_string();
572
573                    let edit = ide::TextEdit::replace(range, generated.clone());
574                    edit.apply(&mut txt);
575
576                    if self.validate_term_search {
577                        std::fs::write(path, txt).unwrap();
578
579                        let res = ws.run_build_scripts(&cargo_config, &|_| ()).unwrap();
580                        if let Some(err) = res.error()
581                            && err.contains("error: could not compile")
582                        {
583                            if let Some(mut err_idx) = err.find("error[E") {
584                                err_idx += 7;
585                                let err_code = &err[err_idx..err_idx + 4];
586                                match err_code {
587                                    "0282" | "0283" => continue, // Byproduct of testing method
588                                    "0277" | "0308" if generated.contains(&todo) => continue, // See https://github.com/rust-lang/rust/issues/69882
589                                    // FIXME: In some rare cases `AssocItem::container_or_implemented_trait` returns `None` for trait methods.
590                                    // Generated code is valid in case traits are imported
591                                    "0599"
592                                        if err.contains(
593                                            "the following trait is implemented but not in scope",
594                                        ) =>
595                                    {
596                                        continue;
597                                    }
598                                    _ => (),
599                                }
600                                bar.println(err);
601                                bar.println(generated);
602                                acc.error_codes
603                                    .entry(err_code.to_owned())
604                                    .and_modify(|n| *n += 1)
605                                    .or_insert(1);
606                            } else {
607                                acc.syntax_errors += 1;
608                                bar.println(format!("Syntax error: \n{err}"));
609                            }
610                        }
611                    }
612                }
613
614                if syntax_hit_found {
615                    acc.tail_expr_syntax_hits += 1;
616                }
617                acc.total_tail_exprs += 1;
618
619                let msg = move || {
620                    format!(
621                        "processing: {:<50}",
622                        trim(&original_text).chars().take(50).collect::<String>()
623                    )
624                };
625                if verbosity.is_spammy() {
626                    bar.println(msg());
627                }
628                bar.set_message(msg);
629            }
630            // Revert file back to original state
631            if self.validate_term_search {
632                std::fs::write(path, file_txt.text(db).to_string()).unwrap();
633            }
634
635            bar.inc(1);
636        }
637        let term_search_time = sw.elapsed();
638
639        bar.println(format!(
640            "Tail Expr syntactic hits: {}/{} ({}%)",
641            acc.tail_expr_syntax_hits,
642            acc.total_tail_exprs,
643            percentage(acc.tail_expr_syntax_hits, acc.total_tail_exprs)
644        ));
645        bar.println(format!(
646            "Tail Exprs found: {}/{} ({}%)",
647            acc.total_tail_exprs - acc.tail_expr_no_term,
648            acc.total_tail_exprs,
649            percentage(acc.total_tail_exprs - acc.tail_expr_no_term, acc.total_tail_exprs)
650        ));
651        if self.validate_term_search {
652            bar.println(format!(
653                "Tail Exprs total errors: {}, syntax errors: {}, error codes:",
654                acc.error_codes.values().sum::<u32>() + acc.syntax_errors,
655                acc.syntax_errors,
656            ));
657            for (err, count) in acc.error_codes {
658                bar.println(format!(
659                    "    E{err}: {count:>5}  (https://doc.rust-lang.org/error_codes/E{err}.html)"
660                ));
661            }
662        }
663        bar.println(format!(
664            "Term search avg time: {}ms",
665            term_search_time.time.as_millis() as u64 / acc.total_tail_exprs
666        ));
667        bar.println(format!("{:<20} {}", "Term search:", term_search_time));
668        report_metric("term search time", term_search_time.time.as_millis() as u64, "ms");
669
670        bar.finish_and_clear();
671    }

673    fn run_mir_lowering(&self, db: &RootDatabase, bodies: &[DefWithBody], verbosity: Verbosity) {
674        let mut bar = match verbosity {
675            Verbosity::Quiet | Verbosity::Spammy => ProgressReport::hidden(),
676            _ if self.parallel || self.output.is_some() => ProgressReport::hidden(),
677            _ => ProgressReport::new(bodies.len()),
678        };
679        let mut sw = self.stop_watch();
680        let mut all = 0;
681        let mut fail = 0;
682        for &body_id in bodies {
683            bar.set_message(move || {
684                format!("mir lowering: {}", full_name(db, body_id, body_id.module(db)))
685            });
686            bar.inc(1);
687            if matches!(body_id, DefWithBody::Variant(_)) {
688                continue;
689            }
690            let module = body_id.module(db);
691            if !self.should_process(db, body_id, module) {
692                continue;
693            }
694
695            all += 1;
696            let Err(e) = db.mir_body(body_id.into()) else {
697                continue;
698            };
699            if verbosity.is_spammy() {
700                let full_name = module
701                    .path_to_root(db)
702                    .into_iter()
703                    .rev()
704                    .filter_map(|it| it.name(db))
705                    .chain(Some(body_id.name(db).unwrap_or_else(Name::missing)))
706                    .map(|it| it.display(db, Edition::LATEST).to_string())
707                    .join("::");
708                bar.println(format!("Mir body for {full_name} failed due {e:?}"));
709            }
710            fail += 1;
711            bar.tick();
712        }
713        let mir_lowering_time = sw.elapsed();
714        bar.finish_and_clear();
715        eprintln!("{:<20} {}", "MIR lowering:", mir_lowering_time);
716        eprintln!("Mir failed bodies: {fail} ({}%)", percentage(fail, all));
717        report_metric("mir failed bodies", fail, "#");
718        report_metric("mir lowering time", mir_lowering_time.time.as_millis() as u64, "ms");
719    }

721    fn run_inference(
722        &self,
723        db: &RootDatabase,
724        vfs: &Vfs,
725        bodies: &[DefWithBody],
726        verbosity: Verbosity,
727    ) {
728        let mut bar = match verbosity {
729            Verbosity::Quiet | Verbosity::Spammy => ProgressReport::hidden(),
730            _ if self.parallel || self.output.is_some() => ProgressReport::hidden(),
731            _ => ProgressReport::new(bodies.len()),
732        };
733
734        if self.parallel {
735            let mut inference_sw = self.stop_watch();
736            bodies
737                .par_iter()
738                .map_with(db.clone(), |snap, &body| {
739                    snap.body(body.into());
740                    snap.infer(body.into());
741                })
742                .count();
743            eprintln!("{:<20} {}", "Parallel Inference:", inference_sw.elapsed());
744        }
745
746        let mut inference_sw = self.stop_watch();
747        bar.tick();
748        let mut num_exprs = 0;
749        let mut num_exprs_unknown = 0;
750        let mut num_exprs_partially_unknown = 0;
751        let mut num_expr_type_mismatches = 0;
752        let mut num_pats = 0;
753        let mut num_pats_unknown = 0;
754        let mut num_pats_partially_unknown = 0;
755        let mut num_pat_type_mismatches = 0;
756        let mut panics = 0;
757        for &body_id in bodies {
758            let name = body_id.name(db).unwrap_or_else(Name::missing);
759            let module = body_id.module(db);
760            let display_target = module.krate().to_display_target(db);
761            if let Some(only_name) = self.only.as_deref()
762                && name.display(db, Edition::LATEST).to_string() != only_name
763                && full_name(db, body_id, module) != only_name
764            {
765                continue;
766            }
767            let msg = move || {
768                if verbosity.is_verbose() {
769                    let source = match body_id {
770                        DefWithBody::Function(it) => it.source(db).map(|it| it.syntax().cloned()),
771                        DefWithBody::Static(it) => it.source(db).map(|it| it.syntax().cloned()),
772                        DefWithBody::Const(it) => it.source(db).map(|it| it.syntax().cloned()),
773                        DefWithBody::Variant(it) => it.source(db).map(|it| it.syntax().cloned()),
774                    };
775                    if let Some(src) = source {
776                        let original_file = src.file_id.original_file(db);
777                        let path = vfs.file_path(original_file.file_id(db));
778                        let syntax_range = src.text_range();
779                        format!(
780                            "processing: {} ({} {:?})",
781                            full_name(db, body_id, module),
782                            path,
783                            syntax_range
784                        )
785                    } else {
786                        format!("processing: {}", full_name(db, body_id, module))
787                    }
788                } else {
789                    format!("processing: {}", full_name(db, body_id, module))
790                }
791            };
792            if verbosity.is_spammy() {
793                bar.println(msg());
794            }
795            bar.set_message(msg);
796            let body = db.body(body_id.into());
797            let inference_result = catch_unwind(AssertUnwindSafe(|| db.infer(body_id.into())));
798            let inference_result = match inference_result {
799                Ok(inference_result) => inference_result,
800                Err(p) => {
801                    if let Some(s) = p.downcast_ref::<&str>() {
802                        eprintln!("infer panicked for {}: {}", full_name(db, body_id, module), s);
803                    } else if let Some(s) = p.downcast_ref::<String>() {
804                        eprintln!("infer panicked for {}: {}", full_name(db, body_id, module), s);
805                    } else {
806                        eprintln!("infer panicked for {}", full_name(db, body_id, module));
807                    }
808                    panics += 1;
809                    bar.inc(1);
810                    continue;
811                }
812            };
813            // This query is LRU'd, so actually calling it will skew the timing results.
814            let sm = || db.body_with_source_map(body_id.into()).1;
815
816            // region:expressions
817            let (previous_exprs, previous_unknown, previous_partially_unknown) =
818                (num_exprs, num_exprs_unknown, num_exprs_partially_unknown);
819            for (expr_id, _) in body.exprs() {
820                let ty = &inference_result[expr_id];
821                num_exprs += 1;
822                let unknown_or_partial = if ty.is_ty_error() {
823                    num_exprs_unknown += 1;
824                    if verbosity.is_spammy() {
825                        if let Some((path, start, end)) = expr_syntax_range(db, vfs, &sm(), expr_id)
826                        {
827                            bar.println(format!(
828                                "{} {}:{}-{}:{}: Unknown type",
829                                path,
830                                start.line + 1,
831                                start.col,
832                                end.line + 1,
833                                end.col,
834                            ));
835                        } else {
836                            bar.println(format!(
837                                "{}: Unknown type",
838                                name.display(db, Edition::LATEST)
839                            ));
840                        }
841                    }
842                    true
843                } else {
844                    let is_partially_unknown = ty.references_non_lt_error();
845                    if is_partially_unknown {
846                        num_exprs_partially_unknown += 1;
847                    }
848                    is_partially_unknown
849                };
850                if self.only.is_some() && verbosity.is_spammy() {
851                    // in super-verbose mode for just one function, we print every single expression
852                    if let Some((_, start, end)) = expr_syntax_range(db, vfs, &sm(), expr_id) {
853                        bar.println(format!(
854                            "{}:{}-{}:{}: {}",
855                            start.line + 1,
856                            start.col,
857                            end.line + 1,
858                            end.col,
859                            ty.display(db, display_target)
860                        ));
861                    } else {
862                        bar.println(format!(
863                            "unknown location: {}",
864                            ty.display(db, display_target)
865                        ));
866                    }
867                }
868                if unknown_or_partial && self.output == Some(OutputFormat::Csv) {
869                    println!(
870                        r#"{},type,"{}""#,
871                        location_csv_expr(db, vfs, &sm(), expr_id),
872                        ty.display(db, display_target)
873                    );
874                }
875                if let Some(mismatch) = inference_result.type_mismatch_for_expr(expr_id) {
876                    num_expr_type_mismatches += 1;
877                    if verbosity.is_verbose() {
878                        if let Some((path, start, end)) = expr_syntax_range(db, vfs, &sm(), expr_id)
879                        {
880                            bar.println(format!(
881                                "{} {}:{}-{}:{}: Expected {}, got {}",
882                                path,
883                                start.line + 1,
884                                start.col,
885                                end.line + 1,
886                                end.col,
887                                mismatch.expected.display(db, display_target),
888                                mismatch.actual.display(db, display_target)
889                            ));
890                        } else {
891                            bar.println(format!(
892                                "{}: Expected {}, got {}",
893                                name.display(db, Edition::LATEST),
894                                mismatch.expected.display(db, display_target),
895                                mismatch.actual.display(db, display_target)
896                            ));
897                        }
898                    }
899                    if self.output == Some(OutputFormat::Csv) {
900                        println!(
901                            r#"{},mismatch,"{}","{}""#,
902                            location_csv_expr(db, vfs, &sm(), expr_id),
903                            mismatch.expected.display(db, display_target),
904                            mismatch.actual.display(db, display_target)
905                        );
906                    }
907                }
908            }
909            if verbosity.is_spammy() {
910                bar.println(format!(
911                    "In {}: {} exprs, {} unknown, {} partial",
912                    full_name(db, body_id, module),
913                    num_exprs - previous_exprs,
914                    num_exprs_unknown - previous_unknown,
915                    num_exprs_partially_unknown - previous_partially_unknown
916                ));
917            }
918            // endregion:expressions
919
920            // region:patterns
921            let (previous_pats, previous_unknown, previous_partially_unknown) =
922                (num_pats, num_pats_unknown, num_pats_partially_unknown);
923            for (pat_id, _) in body.pats() {
924                let ty = &inference_result[pat_id];
925                num_pats += 1;
926                let unknown_or_partial = if ty.is_ty_error() {
927                    num_pats_unknown += 1;
928                    if verbosity.is_spammy() {
929                        if let Some((path, start, end)) = pat_syntax_range(db, vfs, &sm(), pat_id) {
930                            bar.println(format!(
931                                "{} {}:{}-{}:{}: Unknown type",
932                                path,
933                                start.line + 1,
934                                start.col,
935                                end.line + 1,
936                                end.col,
937                            ));
938                        } else {
939                            bar.println(format!(
940                                "{}: Unknown type",
941                                name.display(db, Edition::LATEST)
942                            ));
943                        }
944                    }
945                    true
946                } else {
947                    let is_partially_unknown = ty.references_non_lt_error();
948                    if is_partially_unknown {
949                        num_pats_partially_unknown += 1;
950                    }
951                    is_partially_unknown
952                };
953                if self.only.is_some() && verbosity.is_spammy() {
954                    // in super-verbose mode for just one function, we print every single pattern
955                    if let Some((_, start, end)) = pat_syntax_range(db, vfs, &sm(), pat_id) {
956                        bar.println(format!(
957                            "{}:{}-{}:{}: {}",
958                            start.line + 1,
959                            start.col,
960                            end.line + 1,
961                            end.col,
962                            ty.display(db, display_target)
963                        ));
964                    } else {
965                        bar.println(format!(
966                            "unknown location: {}",
967                            ty.display(db, display_target)
968                        ));
969                    }
970                }
971                if unknown_or_partial && self.output == Some(OutputFormat::Csv) {
972                    println!(
973                        r#"{},type,"{}""#,
974                        location_csv_pat(db, vfs, &sm(), pat_id),
975                        ty.display(db, display_target)
976                    );
977                }
978                if let Some(mismatch) = inference_result.type_mismatch_for_pat(pat_id) {
979                    num_pat_type_mismatches += 1;
980                    if verbosity.is_verbose() {
981                        if let Some((path, start, end)) = pat_syntax_range(db, vfs, &sm(), pat_id) {
982                            bar.println(format!(
983                                "{} {}:{}-{}:{}: Expected {}, got {}",
984                                path,
985                                start.line + 1,
986                                start.col,
987                                end.line + 1,
988                                end.col,
989                                mismatch.expected.display(db, display_target),
990                                mismatch.actual.display(db, display_target)
991                            ));
992                        } else {
993                            bar.println(format!(
994                                "{}: Expected {}, got {}",
995                                name.display(db, Edition::LATEST),
996                                mismatch.expected.display(db, display_target),
997                                mismatch.actual.display(db, display_target)
998                            ));
999                        }
1000                    }
1001                    if self.output == Some(OutputFormat::Csv) {
1002                        println!(
1003                            r#"{},mismatch,"{}","{}""#,
1004                            location_csv_pat(db, vfs, &sm(), pat_id),
1005                            mismatch.expected.display(db, display_target),
1006                            mismatch.actual.display(db, display_target)
1007                        );
1008                    }
1009                }
1010            }
1011            if verbosity.is_spammy() {
1012                bar.println(format!(
1013                    "In {}: {} pats, {} unknown, {} partial",
1014                    full_name(db, body_id, module),
1015                    num_pats - previous_pats,
1016                    num_pats_unknown - previous_unknown,
1017                    num_pats_partially_unknown - previous_partially_unknown
1018                ));
1019            }
1020            // endregion:patterns
1021            bar.inc(1);
1022        }
1023
1024        bar.finish_and_clear();
1025        let inference_time = inference_sw.elapsed();
1026        eprintln!(
1027            "  exprs: {}, ??ty: {} ({}%), ?ty: {} ({}%), !ty: {}",
1028            num_exprs,
1029            num_exprs_unknown,
1030            percentage(num_exprs_unknown, num_exprs),
1031            num_exprs_partially_unknown,
1032            percentage(num_exprs_partially_unknown, num_exprs),
1033            num_expr_type_mismatches
1034        );
1035        eprintln!(
1036            "  pats: {}, ??ty: {} ({}%), ?ty: {} ({}%), !ty: {}",
1037            num_pats,
1038            num_pats_unknown,
1039            percentage(num_pats_unknown, num_pats),
1040            num_pats_partially_unknown,
1041            percentage(num_pats_partially_unknown, num_pats),
1042            num_pat_type_mismatches
1043        );
1044        eprintln!("  panics: {panics}");
1045        eprintln!("{:<20} {}", "Inference:", inference_time);
1046        report_metric("unknown type", num_exprs_unknown, "#");
1047        report_metric("type mismatches", num_expr_type_mismatches, "#");
1048        report_metric("pattern unknown type", num_pats_unknown, "#");
1049        report_metric("pattern type mismatches", num_pat_type_mismatches, "#");
1050        report_metric("inference time", inference_time.time.as_millis() as u64, "ms");
1051    }

1221            let msg = format!("annotations: {}", vfs.file_path(file_id.file_id(db)));
1222            bar.set_message(move || msg.clone());
1223            analysis
1224                .annotations(&annotation_config, analysis.editioned_file_id_to_vfs(file_id))
1225                .unwrap()
1226                .into_iter()
1227                .for_each(|annotation| {
1228                    _ = analysis.resolve_annotation(&annotation_config, annotation);
1229                });
1230            bar.inc(1);
1231        }
1232        bar.finish_and_clear();
1233
1234        let ide_time = sw.elapsed();
1235        eprintln!("{:<20} {} ({} files)", "IDE:", ide_time, file_ids.len());
1236    }

1238    fn should_process(&self, db: &RootDatabase, body_id: DefWithBody, module: hir::Module) -> bool {
1239        if let Some(only_name) = self.only.as_deref() {
1240            let name = body_id.name(db).unwrap_or_else(Name::missing);
1241
1242            if name.display(db, Edition::LATEST).to_string() != only_name
1243                && full_name(db, body_id, module) != only_name
1244            {
1245                return false;
1246            }
1247        }
1248        true
1249    }

1256fn full_name(db: &RootDatabase, body_id: DefWithBody, module: hir::Module) -> String {
1257    module
1258        .krate()
1259        .display_name(db)
1260        .map(|it| it.canonical_name().as_str().to_owned())
1261        .into_iter()
1262        .chain(
1263            module
1264                .path_to_root(db)
1265                .into_iter()
1266                .filter_map(|it| it.name(db))
1267                .rev()
1268                .chain(Some(body_id.name(db).unwrap_or_else(Name::missing)))
1269                .map(|it| it.display(db, Edition::LATEST).to_string()),
1270        )
1271        .join("::")
1272}

1286        (line_index.line_col(text_range.start()), line_index.line_col(text_range.end()));

1349fn shuffle<T>(rng: &mut Rand32, slice: &mut [T]) {
1350    for i in 0..slice.len() {
1351        randomize_first(rng, &mut slice[i..]);
1352    }
1353
1354    fn randomize_first<T>(rng: &mut Rand32, slice: &mut [T]) {
1355        assert!(!slice.is_empty());
1356        let idx = rng.rand_range(0..slice.len() as u32) as usize;
1357        slice.swap(0, idx);
1358    }
1359}

1365#[derive(Default, Debug, Eq, PartialEq)]
1366struct UsizeWithUnderscore(usize);
1367
1368impl fmt::Display for UsizeWithUnderscore {
1369    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
1370        let num_str = self.0.to_string();
1371
1372        if num_str.len() <= 3 {
1373            return write!(f, "{num_str}");
1374        }
1375
1376        let mut result = String::new();
1377
1378        for (count, ch) in num_str.chars().rev().enumerate() {
1379            if count > 0 && count % 3 == 0 {
1380                result.push('_');
1381            }
1382            result.push(ch);
1383        }
1384
1385        let result = result.chars().rev().collect::<String>();
1386        write!(f, "{result}")
1387    }
1388}
1389
1390impl std::ops::AddAssign for UsizeWithUnderscore {
1391    fn add_assign(&mut self, other: UsizeWithUnderscore) {
1392        self.0 += other.0;
1393    }
1394}
