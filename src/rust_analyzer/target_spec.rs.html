17/// A target represents a thing we can build or test.
18///
19/// We use it to calculate the CLI arguments required to build, run or
20/// test the target.
21#[derive(Clone, Debug)]
22pub(crate) enum TargetSpec {
23    Cargo(CargoTargetSpec),
24    ProjectJson(ProjectJsonTargetSpec),
25}
26
27impl TargetSpec {
28    pub(crate) fn for_file(
29        global_state_snapshot: &GlobalStateSnapshot,
30        file_id: FileId,
31    ) -> Cancellable<Option<Self>> {
32        let crate_id = match &*global_state_snapshot.analysis.crates_for(file_id)? {
33            &[crate_id, ..] => crate_id,
34            _ => return Ok(None),
35        };
36
37        Ok(global_state_snapshot.target_spec_for_crate(crate_id))
38    }
39
40    pub(crate) fn target_kind(&self) -> TargetKind {
41        match self {
42            TargetSpec::Cargo(cargo) => cargo.target_kind,
43            TargetSpec::ProjectJson(project_json) => project_json.target_kind,
44        }
45    }
46}

48/// Abstract representation of Cargo target.

74impl ProjectJsonTargetSpec {
75    pub(crate) fn runnable_args(&self, kind: &RunnableKind) -> Option<Runnable> {
76        match kind {
77            RunnableKind::Bin => {
78                for runnable in &self.shell_runnables {
79                    if matches!(runnable.kind, project_model::project_json::RunnableKind::Run) {
80                        let mut runnable = runnable.clone();
81
82                        let replaced_args: Vec<_> = runnable
83                            .args
84                            .iter()
85                            .map(|arg| arg.replace("{label}", &self.label))
86                            .collect();
87                        runnable.args = replaced_args;
88
89                        return Some(runnable);
90                    }
91                }
92
93                None
94            }
95            RunnableKind::Test { test_id, .. } => {
96                for runnable in &self.shell_runnables {
97                    if matches!(runnable.kind, project_model::project_json::RunnableKind::TestOne) {
98                        let mut runnable = runnable.clone();
99
100                        let replaced_args: Vec<_> = runnable
101                            .args
102                            .iter()
103                            .map(|arg| arg.replace("{test_id}", &test_id.to_string()))
104                            .map(|arg| arg.replace("{label}", &self.label))
105                            .collect();
106                        runnable.args = replaced_args;
107
108                        return Some(runnable);
109                    }
110                }
111
112                None
113            }
114            RunnableKind::TestMod { .. } => None,
115            RunnableKind::Bench { .. } => None,
116            RunnableKind::DocTest { .. } => None,
117        }
118    }
119}

121impl CargoTargetSpec {
122    pub(crate) fn runnable_args(
123        snap: &GlobalStateSnapshot,
124        spec: Option<CargoTargetSpec>,
125        kind: &RunnableKind,
126        cfg: &Option<CfgExpr>,
127    ) -> (Vec<String>, Vec<String>) {
128        let config = snap.config.runnables(None);
129        let extra_test_binary_args = config.extra_test_binary_args;
130
131        let mut cargo_args = Vec::new();
132        let mut executable_args = Vec::new();
133
134        match kind {
135            RunnableKind::Test { test_id, attr } => {
136                cargo_args.push(config.test_command);
137                executable_args.push(test_id.to_string());
138                if let TestId::Path(_) = test_id {
139                    executable_args.push("--exact".to_owned());
140                }
141                executable_args.extend(extra_test_binary_args);
142                if attr.ignore {
143                    executable_args.push("--ignored".to_owned());
144                }
145            }
146            RunnableKind::TestMod { path } => {
147                cargo_args.push(config.test_command);
148                executable_args.push(path.clone());
149                executable_args.extend(extra_test_binary_args);
150            }
151            RunnableKind::Bench { test_id } => {
152                cargo_args.push(config.bench_command);
153                executable_args.push(test_id.to_string());
154                if let TestId::Path(_) = test_id {
155                    executable_args.push("--exact".to_owned());
156                }
157                executable_args.extend(extra_test_binary_args);
158            }
159            RunnableKind::DocTest { test_id } => {
160                cargo_args.push("test".to_owned());
161                cargo_args.push("--doc".to_owned());
162                executable_args.push(test_id.to_string());
163                executable_args.extend(extra_test_binary_args);
164            }
165            RunnableKind::Bin => {
166                let subcommand = match spec {
167                    Some(CargoTargetSpec { target_kind: TargetKind::Test, .. }) => {
168                        config.test_command
169                    }
170                    _ => "run".to_owned(),
171                };
172                cargo_args.push(subcommand);
173            }
174        }
175
176        let (allowed_features, target_required_features) = if let Some(mut spec) = spec {
177            let allowed_features = mem::take(&mut spec.features);
178            let required_features = mem::take(&mut spec.required_features);
179            spec.push_to(&mut cargo_args, kind);
180            (allowed_features, required_features)
181        } else {
182            (Default::default(), Default::default())
183        };
184
185        let cargo_config = snap.config.cargo(None);
186
187        match &cargo_config.features {
188            CargoFeatures::All => {
189                cargo_args.push("--all-features".to_owned());
190                for feature in target_required_features {
191                    cargo_args.push("--features".to_owned());
192                    cargo_args.push(feature);
193                }
194            }
195            CargoFeatures::Selected { features, no_default_features } => {
196                let mut feats = Vec::new();
197                if let Some(cfg) = cfg.as_ref() {
198                    required_features(cfg, &mut feats);
199                }
200
201                feats.extend(
202                    features.iter().filter(|&feat| allowed_features.contains(feat)).cloned(),
203                );
204                feats.extend(target_required_features);
205
206                feats.dedup();
207                for feature in feats {
208                    cargo_args.push("--features".to_owned());
209                    cargo_args.push(feature);
210                }
211
212                if *no_default_features {
213                    cargo_args.push("--no-default-features".to_owned());
214                }
215            }
216        }
217        cargo_args.extend(config.cargo_extra_args.iter().cloned());
218        (cargo_args, executable_args)
219    }

221    pub(crate) fn override_command(
222        snap: &GlobalStateSnapshot,
223        spec: Option<CargoTargetSpec>,
224        kind: &RunnableKind,
225    ) -> Option<Vec<String>> {
226        let config = snap.config.runnables(None);
227        let (args, test_name) = match kind {
228            RunnableKind::Test { test_id, .. } => {
229                (config.test_override_command, Some(test_id.to_string()))
230            }
231            RunnableKind::TestMod { path } => (config.test_override_command, Some(path.clone())),
232            RunnableKind::Bench { test_id } => {
233                (config.bench_override_command, Some(test_id.to_string()))
234            }
235            RunnableKind::DocTest { test_id } => {
236                (config.doc_test_override_command, Some(test_id.to_string()))
237            }
238            RunnableKind::Bin => match spec {
239                Some(CargoTargetSpec { target_kind: TargetKind::Test, .. }) => {
240                    (config.test_override_command, None)
241                }
242                _ => (None, None),
243            },
244        };
245        let test_name = test_name.unwrap_or_default();
246
247        let target_arg = |kind| match kind {
248            TargetKind::Bin => "--bin",
249            TargetKind::Test => "--test",
250            TargetKind::Bench => "--bench",
251            TargetKind::Example => "--example",
252            TargetKind::Lib { .. } => "--lib",
253            TargetKind::BuildScript | TargetKind::Other => "",
254        };
255
256        let replace_placeholders = |arg: String| match &spec {
257            Some(spec) => arg
258                .replace("${package}", &spec.package)
259                .replace("${target_arg}", target_arg(spec.target_kind))
260                .replace("${target}", &spec.target)
261                .replace("${test_name}", &test_name),
262            _ => arg,
263        };
264
265        args.map(|args| args.into_iter().map(replace_placeholders).collect())
266    }
267
268    pub(crate) fn push_to(self, buf: &mut Vec<String>, kind: &RunnableKind) {
269        buf.push("--package".to_owned());
270        buf.push(self.package);
271
272        // Can't mix --doc with other target flags
273        if let RunnableKind::DocTest { .. } = kind {
274            return;
275        }
276        match self.target_kind {
277            TargetKind::Bin => {
278                buf.push("--bin".to_owned());
279                buf.push(self.target);
280            }
281            TargetKind::Test => {
282                buf.push("--test".to_owned());
283                buf.push(self.target);
284            }
285            TargetKind::Bench => {
286                buf.push("--bench".to_owned());
287                buf.push(self.target);
288            }
289            TargetKind::Example => {
290                buf.push("--example".to_owned());
291                buf.push(self.target);
292            }
293            TargetKind::Lib { is_proc_macro: _ } => {
294                buf.push("--lib".to_owned());
295            }
296            TargetKind::Other | TargetKind::BuildScript => (),
297        }
298    }

301/// Fill minimal features needed
302fn required_features(cfg_expr: &CfgExpr, features: &mut Vec<String>) {
303    match cfg_expr {
304        CfgExpr::Atom(CfgAtom::KeyValue { key, value }) if *key == sym::feature => {
305            features.push(value.to_string())
306        }
307        CfgExpr::All(preds) => {
308            preds.iter().for_each(|cfg| required_features(cfg, features));
309        }
310        CfgExpr::Any(preds) => {
311            for cfg in preds.iter() {
312                let len_features = features.len();
313                required_features(cfg, features);
314                if len_features != features.len() {
315                    break;
316                }
317            }
318        }
319        _ => {}
320    }

325    use super::*;

327    use ide::Edition;
328    use syntax::{
329        SmolStr,
330        ast::{self, AstNode},

332    use syntax_bridge::{
333        DocCommentDesugarMode,
334        dummy_test_span_utils::{DUMMY, DummyTestSpanMap},
335        syntax_node_to_token_tree,
336    };
337
338    fn check(cfg: &str, expected_features: &[&str]) {
339        let cfg_expr = {
340            let source_file = ast::SourceFile::parse(cfg, Edition::CURRENT).ok().unwrap();
341            let tt = source_file.syntax().descendants().find_map(ast::TokenTree::cast).unwrap();
342            let tt = syntax_node_to_token_tree(
343                tt.syntax(),
344                &DummyTestSpanMap,
345                DUMMY,
346                DocCommentDesugarMode::Mbe,
347            );
348            CfgExpr::parse(&tt)
349        };
350
351        let mut features = vec![];
352        required_features(&cfg_expr, &mut features);
353
354        let expected_features =
355            expected_features.iter().map(|&it| SmolStr::new(it)).collect::<Vec<_>>();
356
357        assert_eq!(features, expected_features);
358    }
