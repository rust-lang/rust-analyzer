60impl CargoOptions {
61    pub(crate) fn apply_on_command(&self, cmd: &mut Command) {
62        for target in &self.target_tuples {
63            cmd.args(["--target", target.as_str()]);
64        }
65        if self.all_targets {
66            if self.set_test {
67                cmd.arg("--all-targets");
68            } else {
69                // No --benches unfortunately, as this implies --tests (see https://github.com/rust-lang/cargo/issues/6454),
70                // and users setting `cfg.seTest = false` probably prefer disabling benches than enabling tests.
71                cmd.args(["--lib", "--bins", "--examples"]);
72            }
73        }
74        if self.all_features {
75            cmd.arg("--all-features");
76        } else {
77            if self.no_default_features {
78                cmd.arg("--no-default-features");
79            }
80            if !self.features.is_empty() {
81                cmd.arg("--features");
82                cmd.arg(self.features.join(" "));
83            }
84        }
85        if let Some(target_dir) = &self.target_dir {
86            cmd.arg("--target-dir").arg(target_dir);
87        }
88    }
89}

117impl fmt::Display for FlycheckConfig {
118    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
119        match self {
120            FlycheckConfig::CargoCommand { command, .. } => write!(f, "cargo {command}"),
121            FlycheckConfig::CustomCommand { command, args, .. } => {
122                // Don't show `my_custom_check --foo $saved_file` literally to the user, as it
123                // looks like we've forgotten to substitute $saved_file.
124                //
125                // Instead, show `my_custom_check --foo ...`. The
126                // actual path is often too long to be worth showing
127                // in the IDE (e.g. in the VS Code status bar).
128                let display_args = args
129                    .iter()
130                    .map(|arg| if arg == SAVED_FILE_PLACEHOLDER { "..." } else { arg })
131                    .collect::<Vec<_>>();
132
133                write!(f, "{command} {}", display_args.join(" "))
134            }
135        }
136    }

139/// Flycheck wraps the shared state and communication machinery used for
140/// running `cargo check` (or other compatible command) and providing
141/// diagnostics based on the output.
142/// The spawned thread is shut down when this struct is dropped.
143#[derive(Debug)]
144pub(crate) struct FlycheckHandle {
145    // XXX: drop order is significant
146    sender: Sender<StateChange>,
147    _thread: stdx::thread::JoinHandle,
148    id: usize,
149    generation: AtomicUsize,
150}
151
152impl FlycheckHandle {
153    pub(crate) fn spawn(
154        id: usize,
155        generation: DiagnosticsGeneration,
156        sender: Sender<FlycheckMessage>,
157        config: FlycheckConfig,
158        sysroot_root: Option<AbsPathBuf>,
159        workspace_root: AbsPathBuf,
160        manifest_path: Option<AbsPathBuf>,
161    ) -> FlycheckHandle {
162        let actor = FlycheckActor::new(
163            id,
164            generation,
165            sender,
166            config,
167            sysroot_root,
168            workspace_root,
169            manifest_path,
170        );
171        let (sender, receiver) = unbounded::<StateChange>();
172        let thread =
173            stdx::thread::Builder::new(stdx::thread::ThreadIntent::Worker, format!("Flycheck{id}"))
174                .spawn(move || actor.run(receiver))
175                .expect("failed to spawn thread");
176        FlycheckHandle { id, generation: generation.into(), sender, _thread: thread }
177    }
178
179    /// Schedule a re-start of the cargo check worker to do a workspace wide check.
180    pub(crate) fn restart_workspace(&self, saved_file: Option<AbsPathBuf>) {
181        let generation = self.generation.fetch_add(1, Ordering::Relaxed) + 1;
182        self.sender
183            .send(StateChange::Restart {
184                generation,
185                scope: FlycheckScope::Workspace,
186                saved_file,
187                target: None,
188            })
189            .unwrap();
190    }
191
192    /// Schedule a re-start of the cargo check worker to do a package wide check.
193    pub(crate) fn restart_for_package(
194        &self,
195        package: Arc<PackageId>,
196        target: Option<Target>,
197        workspace_deps: Option<FxHashSet<Arc<PackageId>>>,
198    ) {
199        let generation = self.generation.fetch_add(1, Ordering::Relaxed) + 1;
200        self.sender
201            .send(StateChange::Restart {
202                generation,
203                scope: FlycheckScope::Package { package, workspace_deps },
204                saved_file: None,
205                target,
206            })
207            .unwrap();
208    }
209
210    /// Stop this cargo check worker.
211    pub(crate) fn cancel(&self) {
212        self.sender.send(StateChange::Cancel).unwrap();
213    }
214
215    pub(crate) fn id(&self) -> usize {
216        self.id
217    }
218
219    pub(crate) fn generation(&self) -> DiagnosticsGeneration {
220        self.generation.load(Ordering::Relaxed)
221    }
222}

236pub(crate) enum FlycheckMessage {

238    AddDiagnostic {
239        id: usize,

241        workspace_root: Arc<AbsPathBuf>,

257impl fmt::Debug for FlycheckMessage {
258    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
259        match self {
260            FlycheckMessage::AddDiagnostic {
261                id,
262                generation,
263                workspace_root,
264                diagnostic,
265                package_id,
266            } => f
267                .debug_struct("AddDiagnostic")
268                .field("id", id)
269                .field("generation", generation)
270                .field("workspace_root", workspace_root)
271                .field("package_id", package_id)
272                .field("diagnostic_code", &diagnostic.code.as_ref().map(|it| &it.code))
273                .finish(),
274            FlycheckMessage::ClearDiagnostics { id, kind } => {
275                f.debug_struct("ClearDiagnostics").field("id", id).field("kind", kind).finish()
276            }
277            FlycheckMessage::Progress { id, progress } => {
278                f.debug_struct("Progress").field("id", id).field("progress", progress).finish()
279            }
280        }
281    }

308/// A [`FlycheckActor`] is a single check instance of a workspace.
309struct FlycheckActor {
310    /// The workspace id of this flycheck instance.
311    id: usize,
312
313    generation: DiagnosticsGeneration,
314    sender: Sender<FlycheckMessage>,
315    config: FlycheckConfig,
316    manifest_path: Option<AbsPathBuf>,
317    /// Either the workspace root of the workspace we are flychecking,
318    /// or the project root of the project.
319    root: Arc<AbsPathBuf>,
320    sysroot_root: Option<AbsPathBuf>,
321    scope: FlycheckScope,
322    /// CargoHandle exists to wrap around the communication needed to be able to
323    /// run `cargo check` without blocking. Currently the Rust standard library
324    /// doesn't provide a way to read sub-process output without blocking, so we
325    /// have to wrap sub-processes output handling in a thread and pass messages
326    /// back over a channel.
327    command_handle: Option<CommandHandle<CargoCheckMessage>>,
328    /// The receiver side of the channel mentioned above.
329    command_receiver: Option<Receiver<CargoCheckMessage>>,
330    diagnostics_cleared_for: FxHashSet<Arc<PackageId>>,
331    diagnostics_received: DiagnosticsReceived,
332}
333
334#[derive(PartialEq)]
335enum DiagnosticsReceived {
336    Yes,
337    No,
338    YesAndClearedForAll,
339}

342enum Event {
343    RequestStateChange(StateChange),
344    CheckEvent(Option<CargoCheckMessage>),
345}

349impl FlycheckActor {
350    fn new(
351        id: usize,
352        generation: DiagnosticsGeneration,
353        sender: Sender<FlycheckMessage>,
354        config: FlycheckConfig,
355        sysroot_root: Option<AbsPathBuf>,
356        workspace_root: AbsPathBuf,
357        manifest_path: Option<AbsPathBuf>,
358    ) -> FlycheckActor {
359        tracing::info!(%id, ?workspace_root, "Spawning flycheck");
360        FlycheckActor {
361            id,
362            generation,
363            sender,
364            config,
365            sysroot_root,
366            root: Arc::new(workspace_root),
367            scope: FlycheckScope::Workspace,
368            manifest_path,
369            command_handle: None,
370            command_receiver: None,
371            diagnostics_cleared_for: Default::default(),
372            diagnostics_received: DiagnosticsReceived::No,
373        }
374    }
375
376    fn report_progress(&self, progress: Progress) {
377        self.send(FlycheckMessage::Progress { id: self.id, progress });
378    }
379
380    fn next_event(&self, inbox: &Receiver<StateChange>) -> Option<Event> {
381        let Some(command_receiver) = &self.command_receiver else {
382            return inbox.recv().ok().map(Event::RequestStateChange);
383        };
384
385        // Biased to give restarts a preference so check outputs don't block a restart or stop
386        select_biased! {
387            recv(inbox) -> msg => msg.ok().map(Event::RequestStateChange),
388            recv(command_receiver) -> msg => Some(Event::CheckEvent(msg.ok())),
389        }
390    }
391
392    fn run(mut self, inbox: Receiver<StateChange>) {
393        'event: while let Some(event) = self.next_event(&inbox) {
394            match event {
395                Event::RequestStateChange(StateChange::Cancel) => {
396                    tracing::debug!(flycheck_id = self.id, "flycheck cancelled");
397                    self.cancel_check_process();
398                }
399                Event::RequestStateChange(StateChange::Restart {
400                    generation,
401                    scope,
402                    saved_file,
403                    target,
404                }) => {
405                    // Cancel the previously spawned process
406                    self.cancel_check_process();
407                    while let Ok(restart) = inbox.recv_timeout(Duration::from_millis(50)) {
408                        // restart chained with a stop, so just cancel
409                        if let StateChange::Cancel = restart {
410                            continue 'event;
411                        }
412                    }
413
414                    let command = self.check_command(&scope, saved_file.as_deref(), target);
415                    self.scope = scope;
416                    self.generation = generation;
417
418                    let Some(command) = command else {
419                        continue;
420                    };
421
422                    let formatted_command = format!("{command:?}");
423
424                    tracing::debug!(?command, "will restart flycheck");
425                    let (sender, receiver) = unbounded();
426                    match CommandHandle::spawn(
427                        command,
428                        CargoCheckParser,
429                        sender,
430                        match &self.config {
431                            FlycheckConfig::CargoCommand { options, .. } => Some(
432                                options
433                                    .target_dir
434                                    .as_deref()
435                                    .unwrap_or(
436                                        Utf8Path::new("target").join("rust-analyzer").as_path(),
437                                    )
438                                    .join(format!("flycheck{}", self.id)),
439                            ),
440                            _ => None,
441                        },
442                    ) {
443                        Ok(command_handle) => {
444                            tracing::debug!(command = formatted_command, "did restart flycheck");
445                            self.command_handle = Some(command_handle);
446                            self.command_receiver = Some(receiver);
447                            self.report_progress(Progress::DidStart);
448                        }
449                        Err(error) => {
450                            self.report_progress(Progress::DidFailToRestart(format!(
451                                "Failed to run the following command: {formatted_command} error={error}"
452                            )));
453                        }
454                    }
455                }
456                Event::CheckEvent(None) => {
457                    tracing::debug!(flycheck_id = self.id, "flycheck finished");
458
459                    // Watcher finished
460                    let command_handle = self.command_handle.take().unwrap();
461                    self.command_receiver.take();
462                    let formatted_handle = format!("{command_handle:?}");
463
464                    let res = command_handle.join();
465                    if let Err(error) = &res {
466                        tracing::error!(
467                            "Flycheck failed to run the following command: {}, error={}",
468                            formatted_handle,
469                            error
470                        );
471                    }
472                    if self.diagnostics_received == DiagnosticsReceived::No {
473                        tracing::trace!(flycheck_id = self.id, "clearing diagnostics");
474                        // We finished without receiving any diagnostics.
475                        // Clear everything for good measure
476                        match &self.scope {
477                            FlycheckScope::Workspace => {
478                                self.send(FlycheckMessage::ClearDiagnostics {
479                                    id: self.id,
480                                    kind: ClearDiagnosticsKind::All(ClearScope::Workspace),
481                                });
482                            }
483                            FlycheckScope::Package { package, workspace_deps } => {
484                                for pkg in
485                                    std::iter::once(package).chain(workspace_deps.iter().flatten())
486                                {
487                                    self.send(FlycheckMessage::ClearDiagnostics {
488                                        id: self.id,
489                                        kind: ClearDiagnosticsKind::All(ClearScope::Package(
490                                            pkg.clone(),
491                                        )),
492                                    });
493                                }
494                            }
495                        }
496                    } else if res.is_ok() {
497                        // We clear diagnostics for packages on
498                        // `[CargoCheckMessage::CompilerArtifact]` but there seem to be setups where
499                        // cargo may not report an artifact to our runner at all. To handle such
500                        // cases, clear stale diagnostics when flycheck completes successfully.
501                        match &self.scope {
502                            FlycheckScope::Workspace => {
503                                self.send(FlycheckMessage::ClearDiagnostics {
504                                    id: self.id,
505                                    kind: ClearDiagnosticsKind::OlderThan(
506                                        self.generation,
507                                        ClearScope::Workspace,
508                                    ),
509                                });
510                            }
511                            FlycheckScope::Package { package, workspace_deps } => {
512                                for pkg in
513                                    std::iter::once(package).chain(workspace_deps.iter().flatten())
514                                {
515                                    self.send(FlycheckMessage::ClearDiagnostics {
516                                        id: self.id,
517                                        kind: ClearDiagnosticsKind::OlderThan(
518                                            self.generation,
519                                            ClearScope::Package(pkg.clone()),
520                                        ),
521                                    });
522                                }
523                            }
524                        }
525                    }
526                    self.clear_diagnostics_state();
527
528                    self.report_progress(Progress::DidFinish(res));
529                }
530                Event::CheckEvent(Some(message)) => match message {
531                    CargoCheckMessage::CompilerArtifact(msg) => {
532                        tracing::trace!(
533                            flycheck_id = self.id,
534                            artifact = msg.target.name,
535                            package_id = msg.package_id.repr,
536                            "artifact received"
537                        );
538                        self.report_progress(Progress::DidCheckCrate(format!(
539                            "{} ({})",
540                            msg.target.name,
541                            msg.target.kind.iter().format_with(", ", |kind, f| f(&kind)),
542                        )));
543                        let package_id = Arc::new(msg.package_id);
544                        if self.diagnostics_cleared_for.insert(package_id.clone()) {
545                            tracing::trace!(
546                                flycheck_id = self.id,
547                                package_id = package_id.repr,
548                                "clearing diagnostics"
549                            );
550                            self.send(FlycheckMessage::ClearDiagnostics {
551                                id: self.id,
552                                kind: ClearDiagnosticsKind::All(ClearScope::Package(package_id)),
553                            });
554                        }
555                    }
556                    CargoCheckMessage::Diagnostic { diagnostic, package_id } => {
557                        tracing::trace!(
558                            flycheck_id = self.id,
559                            message = diagnostic.message,
560                            package_id = package_id.as_ref().map(|it| &it.repr),
561                            "diagnostic received"
562                        );
563                        if self.diagnostics_received == DiagnosticsReceived::No {
564                            self.diagnostics_received = DiagnosticsReceived::Yes;
565                        }
566                        if let Some(package_id) = &package_id {
567                            if self.diagnostics_cleared_for.insert(package_id.clone()) {
568                                tracing::trace!(
569                                    flycheck_id = self.id,
570                                    package_id = package_id.repr,
571                                    "clearing diagnostics"
572                                );
573                                self.send(FlycheckMessage::ClearDiagnostics {
574                                    id: self.id,
575                                    kind: ClearDiagnosticsKind::All(ClearScope::Package(
576                                        package_id.clone(),
577                                    )),
578                                });
579                            }
580                        } else if self.diagnostics_received
581                            != DiagnosticsReceived::YesAndClearedForAll
582                        {
583                            self.diagnostics_received = DiagnosticsReceived::YesAndClearedForAll;
584                            self.send(FlycheckMessage::ClearDiagnostics {
585                                id: self.id,
586                                kind: ClearDiagnosticsKind::All(ClearScope::Workspace),
587                            });
588                        }
589                        self.send(FlycheckMessage::AddDiagnostic {
590                            id: self.id,
591                            generation: self.generation,
592                            package_id,
593                            workspace_root: self.root.clone(),
594                            diagnostic,
595                        });
596                    }
597                },
598            }
599        }
600        // If we rerun the thread, we need to discard the previous check results first
601        self.cancel_check_process();
602    }
603
604    fn cancel_check_process(&mut self) {
605        if let Some(command_handle) = self.command_handle.take() {
606            tracing::debug!(
607                command = ?command_handle,
608                "did cancel flycheck"
609            );
610            command_handle.cancel();
611            self.command_receiver.take();
612            self.report_progress(Progress::DidCancel);
613        }
614        self.clear_diagnostics_state();
615    }
616
617    fn clear_diagnostics_state(&mut self) {
618        self.diagnostics_cleared_for.clear();
619        self.diagnostics_received = DiagnosticsReceived::No;
620    }

622    /// Construct a `Command` object for checking the user's code. If the user
623    /// has specified a custom command with placeholders that we cannot fill,
624    /// return None.
625    fn check_command(
626        &self,
627        scope: &FlycheckScope,
628        saved_file: Option<&AbsPath>,
629        target: Option<Target>,
630    ) -> Option<Command> {
631        match &self.config {
632            FlycheckConfig::CargoCommand { command, options, ansi_color_output } => {
633                let mut cmd =
634                    toolchain::command(Tool::Cargo.path(), &*self.root, &options.extra_env);
635                if let Some(sysroot_root) = &self.sysroot_root
636                    && !options.extra_env.contains_key("RUSTUP_TOOLCHAIN")
637                    && std::env::var_os("RUSTUP_TOOLCHAIN").is_none()
638                {
639                    cmd.env("RUSTUP_TOOLCHAIN", AsRef::<std::path::Path>::as_ref(sysroot_root));
640                }
641                cmd.env("CARGO_LOG", "cargo::core::compiler::fingerprint=info");
642                cmd.arg(command);
643
644                match scope {
645                    FlycheckScope::Workspace => cmd.arg("--workspace"),
646                    FlycheckScope::Package { package, .. } => cmd.arg("-p").arg(&package.repr),
647                };
648
649                if let Some(tgt) = target {
650                    match tgt {
651                        Target::Bin(tgt) => cmd.arg("--bin").arg(tgt),
652                        Target::Example(tgt) => cmd.arg("--example").arg(tgt),
653                        Target::Test(tgt) => cmd.arg("--test").arg(tgt),
654                        Target::Benchmark(tgt) => cmd.arg("--bench").arg(tgt),
655                    };
656                }
657
658                cmd.arg(if *ansi_color_output {
659                    "--message-format=json-diagnostic-rendered-ansi"
660                } else {
661                    "--message-format=json"
662                });
663
664                if let Some(manifest_path) = &self.manifest_path {
665                    cmd.arg("--manifest-path");
666                    cmd.arg(manifest_path);
667                    if manifest_path.extension() == Some("rs") {
668                        cmd.env("__CARGO_TEST_CHANNEL_OVERRIDE_DO_NOT_USE_THIS", "nightly");
669                        cmd.arg("-Zscript");
670                    }
671                }
672
673                cmd.arg("--keep-going");
674
675                options.apply_on_command(&mut cmd);
676                cmd.args(&options.extra_args);
677                Some(cmd)
678            }
679            FlycheckConfig::CustomCommand { command, args, extra_env, invocation_strategy } => {
680                let root = match invocation_strategy {
681                    InvocationStrategy::Once => &*self.root,
682                    InvocationStrategy::PerWorkspace => {
683                        // FIXME: &affected_workspace
684                        &*self.root
685                    }
686                };
687                let mut cmd = toolchain::command(command, root, extra_env);
688
689                // If the custom command has a $saved_file placeholder, and
690                // we're saving a file, replace the placeholder in the arguments.
691                if let Some(saved_file) = saved_file {
692                    for arg in args {
693                        if arg == SAVED_FILE_PLACEHOLDER {
694                            cmd.arg(saved_file);
695                        } else {
696                            cmd.arg(arg);
697                        }
698                    }
699                } else {
700                    for arg in args {
701                        if arg == SAVED_FILE_PLACEHOLDER {
702                            // The custom command has a $saved_file placeholder,
703                            // but we had an IDE event that wasn't a file save. Do nothing.
704                            return None;
705                        }
706
707                        cmd.arg(arg);
708                    }
709                }
710
711                Some(cmd)
712            }
713        }
714    }
715
716    #[track_caller]
717    fn send(&self, check_task: FlycheckMessage) {
718        self.sender.send(check_task).unwrap();
719    }
720}

722#[allow(clippy::large_enum_variant)]
723enum CargoCheckMessage {
724    CompilerArtifact(cargo_metadata::Artifact),
725    Diagnostic { diagnostic: Diagnostic, package_id: Option<Arc<PackageId>> },
726}
727
728struct CargoCheckParser;
729
730impl CargoParser<CargoCheckMessage> for CargoCheckParser {
731    fn from_line(&self, line: &str, error: &mut String) -> Option<CargoCheckMessage> {
732        let mut deserializer = serde_json::Deserializer::from_str(line);
733        deserializer.disable_recursion_limit();
734        if let Ok(message) = JsonMessage::deserialize(&mut deserializer) {
735            return match message {
736                // Skip certain kinds of messages to only spend time on what's useful
737                JsonMessage::Cargo(message) => match message {
738                    cargo_metadata::Message::CompilerArtifact(artifact) if !artifact.fresh => {
739                        Some(CargoCheckMessage::CompilerArtifact(artifact))
740                    }
741                    cargo_metadata::Message::CompilerMessage(msg) => {
742                        Some(CargoCheckMessage::Diagnostic {
743                            diagnostic: msg.message,
744                            package_id: Some(Arc::new(msg.package_id)),
745                        })
746                    }
747                    _ => None,
748                },
749                JsonMessage::Rustc(message) => {
750                    Some(CargoCheckMessage::Diagnostic { diagnostic: message, package_id: None })
751                }
752            };
753        }
754
755        error.push_str(line);
756        error.push('\n');
757        None
758    }
759
760    fn from_eof(&self) -> Option<CargoCheckMessage> {
761        None
762    }
763}

766#[serde(untagged)]
767enum JsonMessage {
768    Cargo(cargo_metadata::Message),
769    Rustc(Diagnostic),
