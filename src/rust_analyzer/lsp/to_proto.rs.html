42pub(crate) fn position(line_index: &LineIndex, offset: TextSize) -> lsp_types::Position {
43    let line_col = line_index.index.line_col(offset);
44    match line_index.encoding {
45        PositionEncoding::Utf8 => lsp_types::Position::new(line_col.line, line_col.col),
46        PositionEncoding::Wide(enc) => {
47            let line_col = line_index.index.to_wide(enc, line_col).unwrap();
48            lsp_types::Position::new(line_col.line, line_col.col)
49        }
50    }
51}

53pub(crate) fn range(line_index: &LineIndex, range: TextRange) -> lsp_types::Range {
54    let start = position(line_index, range.start());
55    let end = position(line_index, range.end());
56    lsp_types::Range::new(start, end)

246pub(crate) fn completion_items(
247    config: &Config,
248    fields_to_resolve: &CompletionFieldsToResolve,
249    line_index: &LineIndex,
250    version: Option<i32>,
251    tdpp: lsp_types::TextDocumentPositionParams,
252    completion_trigger_character: Option<char>,
253    mut items: Vec<CompletionItem>,
254) -> Vec<lsp_types::CompletionItem> {
255    if config.completion_hide_deprecated() {
256        items.retain(|item| !item.deprecated);
257    }
258
259    let max_relevance = items.iter().map(|it| it.relevance.score()).max().unwrap_or_default();
260    let mut res = Vec::with_capacity(items.len());
261    let client_commands = config.client_commands();
262    for item in items {
263        completion_item(
264            &mut res,
265            config,
266            &client_commands,
267            fields_to_resolve,
268            line_index,
269            version,
270            &tdpp,
271            max_relevance,
272            completion_trigger_character,
273            item,
274        );
275    }
276
277    if let Some(limit) = config.completion(None, MiniCore::default()).limit {
278        res.sort_by(|item1, item2| item1.sort_text.cmp(&item2.sort_text));
279        res.truncate(limit);
280    }
281
282    res
283}

285fn completion_item(
286    acc: &mut Vec<lsp_types::CompletionItem>,
287    config: &Config,
288    client_commands: &ClientCommandsConfig,
289    fields_to_resolve: &CompletionFieldsToResolve,
290    line_index: &LineIndex,
291    version: Option<i32>,
292    tdpp: &lsp_types::TextDocumentPositionParams,
293    max_relevance: u32,
294    completion_trigger_character: Option<char>,
295    item: CompletionItem,
296) {
297    let insert_replace_support = config.insert_replace_support().then_some(tdpp.position);
298    let ref_match = item.ref_match();
299
300    let mut additional_text_edits = Vec::new();
301    let mut something_to_resolve = false;
302
303    let filter_text = if fields_to_resolve.resolve_filter_text {
304        something_to_resolve |= !item.lookup().is_empty();
305        None
306    } else {
307        Some(item.lookup().to_owned())
308    };
309
310    let text_edit = if fields_to_resolve.resolve_text_edit {
311        something_to_resolve |= true;
312        None
313    } else {
314        // LSP does not allow arbitrary edits in completion, so we have to do a
315        // non-trivial mapping here.
316        let mut text_edit = None;
317        let source_range = item.source_range;
318        for indel in &item.text_edit {
319            if indel.delete.contains_range(source_range) {
320                // Extract this indel as the main edit
321                text_edit = Some(if indel.delete == source_range {
322                    self::completion_text_edit(line_index, insert_replace_support, indel.clone())
323                } else {
324                    assert!(source_range.end() == indel.delete.end());
325                    let range1 = TextRange::new(indel.delete.start(), source_range.start());
326                    let range2 = source_range;
327                    let indel1 = Indel::delete(range1);
328                    let indel2 = Indel::replace(range2, indel.insert.clone());
329                    additional_text_edits.push(self::text_edit(line_index, indel1));
330                    self::completion_text_edit(line_index, insert_replace_support, indel2)
331                })
332            } else {
333                assert!(source_range.intersect(indel.delete).is_none());
334                let text_edit = self::text_edit(line_index, indel.clone());
335                additional_text_edits.push(text_edit);
336            }
337        }
338        Some(text_edit.unwrap())
339    };
340
341    let insert_text_format = item.is_snippet.then_some(lsp_types::InsertTextFormat::SNIPPET);
342    let tags = if fields_to_resolve.resolve_tags {
343        something_to_resolve |= item.deprecated;
344        None
345    } else {
346        item.deprecated.then(|| vec![lsp_types::CompletionItemTag::DEPRECATED])
347    };
348    let command = if item.trigger_call_info && client_commands.trigger_parameter_hints {
349        if fields_to_resolve.resolve_command {
350            something_to_resolve |= true;
351            None
352        } else {
353            Some(command::trigger_parameter_hints())
354        }
355    } else {
356        None
357    };
358
359    let detail = if fields_to_resolve.resolve_detail {
360        something_to_resolve |= item.detail.is_some();
361        None
362    } else {
363        item.detail.clone()
364    };
365
366    let documentation = if fields_to_resolve.resolve_documentation {
367        something_to_resolve |= item.documentation.is_some();
368        None
369    } else {
370        item.documentation.clone().map(documentation)
371    };
372
373    let mut lsp_item = lsp_types::CompletionItem {
374        label: item.label.primary.to_string(),
375        detail,
376        filter_text,
377        kind: Some(completion_item_kind(item.kind)),
378        text_edit,
379        additional_text_edits: additional_text_edits
380            .is_empty()
381            .not()
382            .then_some(additional_text_edits),
383        documentation,
384        deprecated: item.deprecated.then_some(item.deprecated),
385        tags,
386        command,
387        insert_text_format,
388        ..Default::default()
389    };
390
391    if config.completion_label_details_support() {
392        let has_label_details =
393            item.label.detail_left.is_some() || item.label.detail_right.is_some();
394        if fields_to_resolve.resolve_label_details {
395            something_to_resolve |= has_label_details;
396        } else if has_label_details {
397            lsp_item.label_details = Some(lsp_types::CompletionItemLabelDetails {
398                detail: item.label.detail_left.clone(),
399                description: item.label.detail_right.clone(),
400            });
401        }
402    } else if let Some(label_detail) = &item.label.detail_left {
403        lsp_item.label.push_str(label_detail.as_str());
404    }
405
406    set_score(&mut lsp_item, max_relevance, item.relevance);
407
408    let imports = if config.completion(None, MiniCore::default()).enable_imports_on_the_fly
409        && !item.import_to_add.is_empty()
410    {
411        item.import_to_add
412            .clone()
413            .into_iter()
414            .map(|import_path| lsp_ext::CompletionImport { full_import_path: import_path })
415            .collect()
416    } else {
417        Vec::new()
418    };
419    let (ref_resolve_data, resolve_data) = if something_to_resolve || !imports.is_empty() {
420        let ref_resolve_data = if ref_match.is_some() {
421            let ref_resolve_data = lsp_ext::CompletionResolveData {
422                position: tdpp.clone(),
423                imports: Vec::new(),
424                version,
425                trigger_character: completion_trigger_character,
426                for_ref: true,
427                hash: BASE64_STANDARD.encode(completion_item_hash(&item, true)),
428            };
429            Some(to_value(ref_resolve_data).unwrap())
430        } else {
431            None
432        };
433        let resolve_data = lsp_ext::CompletionResolveData {
434            position: tdpp.clone(),
435            imports,
436            version,
437            trigger_character: completion_trigger_character,
438            for_ref: false,
439            hash: BASE64_STANDARD.encode(completion_item_hash(&item, false)),
440        };
441        (ref_resolve_data, Some(to_value(resolve_data).unwrap()))
442    } else {
443        (None, None)
444    };
445
446    if let Some((label, indel, relevance)) = ref_match {
447        let mut lsp_item_with_ref =
448            lsp_types::CompletionItem { label, data: ref_resolve_data, ..lsp_item.clone() };
449        lsp_item_with_ref
450            .additional_text_edits
451            .get_or_insert_with(Default::default)
452            .push(self::text_edit(line_index, indel));
453        set_score(&mut lsp_item_with_ref, max_relevance, relevance);
454        acc.push(lsp_item_with_ref);
455    };
456
457    lsp_item.data = resolve_data;
458    acc.push(lsp_item);
459
460    fn set_score(
461        res: &mut lsp_types::CompletionItem,
462        max_relevance: u32,
463        relevance: CompletionRelevance,
464    ) {
465        if relevance.is_relevant() && relevance.score() == max_relevance {
466            res.preselect = Some(true);
467        }
468        // The relevance needs to be inverted to come up with a sort score
469        // because the client will sort ascending.
470        let sort_score = relevance.score() ^ 0xFF_FF_FF_FF;
471        // Zero pad the string to ensure values can be properly sorted
472        // by the client. Hex format is used because it is easier to
473        // visually compare very large values, which the sort text
474        // tends to be since it is the opposite of the score.
475        res.sort_text = Some(format!("{sort_score:08x}"));
476    }
477}

479pub(crate) fn signature_help(
480    call_info: SignatureHelp,
481    config: CallInfoConfig,
482    label_offsets: bool,
483) -> lsp_types::SignatureHelp {
484    let (label, parameters) = match (config.params_only, label_offsets) {
485        (concise, false) => {
486            let params = call_info
487                .parameter_labels()
488                .map(|label| lsp_types::ParameterInformation {
489                    label: lsp_types::ParameterLabel::Simple(label.to_owned()),
490                    documentation: None,
491                })
492                .collect::<Vec<_>>();
493            let label =
494                if concise { call_info.parameter_labels().join(", ") } else { call_info.signature };
495            (label, params)
496        }
497        (false, true) => {
498            let params = call_info
499                .parameter_ranges()
500                .iter()
501                .map(|it| {
502                    let start = call_info.signature[..it.start().into()]
503                        .chars()
504                        .map(|c| c.len_utf16())
505                        .sum::<usize>() as u32;
506                    let end = start
507                        + call_info.signature[it.start().into()..it.end().into()]
508                            .chars()
509                            .map(|c| c.len_utf16())
510                            .sum::<usize>() as u32;
511                    [start, end]
512                })
513                .map(|label_offsets| lsp_types::ParameterInformation {
514                    label: lsp_types::ParameterLabel::LabelOffsets(label_offsets),
515                    documentation: None,
516                })
517                .collect::<Vec<_>>();
518            (call_info.signature, params)
519        }
520        (true, true) => {
521            let mut params = Vec::new();
522            let mut label = String::new();
523            let mut first = true;
524            for param in call_info.parameter_labels() {
525                if !first {
526                    label.push_str(", ");
527                }
528                first = false;
529                let start = label.len() as u32;
530                label.push_str(param);
531                let end = label.len() as u32;
532                params.push(lsp_types::ParameterInformation {
533                    label: lsp_types::ParameterLabel::LabelOffsets([start, end]),
534                    documentation: None,
535                });
536            }
537
538            (label, params)
539        }
540    };
541
542    let documentation = call_info.doc.filter(|_| config.docs).map(|doc| {
543        lsp_types::Documentation::MarkupContent(lsp_types::MarkupContent {
544            kind: lsp_types::MarkupKind::Markdown,
545            value: format_docs(&doc),
546        })
547    });
548
549    let active_parameter = call_info.active_parameter.map(|it| it as u32);
550
551    let signature = lsp_types::SignatureInformation {
552        label,
553        documentation,
554        parameters: Some(parameters),
555        active_parameter,
556    };
557    lsp_types::SignatureHelp {
558        signatures: vec![signature],
559        active_signature: Some(0),
560        active_parameter,
561    }
562}

564pub(crate) fn inlay_hint(
565    snap: &GlobalStateSnapshot,
566    fields_to_resolve: &InlayFieldsToResolve,
567    line_index: &LineIndex,
568    file_id: FileId,
569    mut inlay_hint: InlayHint,
570) -> Cancellable<lsp_types::InlayHint> {
571    let hint_needs_resolve = |hint: &InlayHint| -> Option<TextRange> {
572        hint.resolve_parent.filter(|_| {
573            hint.text_edit.as_ref().is_some_and(LazyProperty::is_lazy)
574                || hint.label.parts.iter().any(|part| {
575                    part.linked_location.as_ref().is_some_and(LazyProperty::is_lazy)
576                        || part.tooltip.as_ref().is_some_and(LazyProperty::is_lazy)
577                })
578        })
579    };
580
581    let resolve_range_and_hash = hint_needs_resolve(&inlay_hint).map(|range| {
582        (
583            range,
584            std::hash::BuildHasher::hash_one(
585                &std::hash::BuildHasherDefault::<FxHasher>::default(),
586                &inlay_hint,
587            ),
588        )
589    });
590
591    let mut something_to_resolve = false;
592    let text_edits = inlay_hint
593        .text_edit
594        .take()
595        .and_then(|it| match it {
596            LazyProperty::Computed(it) => Some(it),
597            LazyProperty::Lazy => {
598                something_to_resolve |=
599                    snap.config.visual_studio_code_version().is_none_or(|version| {
600                        VersionReq::parse(">=1.86.0").unwrap().matches(version)
601                    }) && resolve_range_and_hash.is_some()
602                        && fields_to_resolve.resolve_text_edits;
603                None
604            }
605        })
606        .map(|it| text_edit_vec(line_index, it));
607    let (label, tooltip) = inlay_hint_label(
608        snap,
609        fields_to_resolve,
610        &mut something_to_resolve,
611        resolve_range_and_hash.is_some(),
612        inlay_hint.label,
613    )?;
614
615    let data = match resolve_range_and_hash {
616        Some((resolve_range, hash)) if something_to_resolve => Some(
617            to_value(lsp_ext::InlayHintResolveData {
618                file_id: file_id.index(),
619                hash: hash.to_string(),
620                version: snap.file_version(file_id),
621                resolve_range: range(line_index, resolve_range),
622            })
623            .unwrap(),
624        ),
625        _ => None,
626    };
627
628    Ok(lsp_types::InlayHint {
629        position: match inlay_hint.position {
630            ide::InlayHintPosition::Before => position(line_index, inlay_hint.range.start()),
631            ide::InlayHintPosition::After => position(line_index, inlay_hint.range.end()),
632        },
633        padding_left: Some(inlay_hint.pad_left),
634        padding_right: Some(inlay_hint.pad_right),
635        kind: match inlay_hint.kind {
636            InlayKind::Parameter | InlayKind::GenericParameter => {
637                Some(lsp_types::InlayHintKind::PARAMETER)
638            }
639            InlayKind::Type | InlayKind::Chaining => Some(lsp_types::InlayHintKind::TYPE),
640            _ => None,
641        },
642        text_edits,
643        data,
644        tooltip,
645        label,
646    })
647}

649fn inlay_hint_label(
650    snap: &GlobalStateSnapshot,
651    fields_to_resolve: &InlayFieldsToResolve,
652    something_to_resolve: &mut bool,
653    needs_resolve: bool,
654    mut label: InlayHintLabel,
655) -> Cancellable<(lsp_types::InlayHintLabel, Option<lsp_types::InlayHintTooltip>)> {
656    let (label, tooltip) = match &*label.parts {
657        [InlayHintLabelPart { linked_location: None, .. }] => {
658            let InlayHintLabelPart { text, tooltip, .. } = label.parts.pop().unwrap();
659            let tooltip = tooltip.and_then(|it| match it {
660                LazyProperty::Computed(it) => Some(it),
661                LazyProperty::Lazy => {
662                    *something_to_resolve |=
663                        needs_resolve && fields_to_resolve.resolve_hint_tooltip;
664                    None
665                }
666            });
667            let hint_tooltip = match tooltip {
668                Some(ide::InlayTooltip::String(s)) => Some(lsp_types::InlayHintTooltip::String(s)),
669                Some(ide::InlayTooltip::Markdown(s)) => {
670                    Some(lsp_types::InlayHintTooltip::MarkupContent(lsp_types::MarkupContent {
671                        kind: lsp_types::MarkupKind::Markdown,
672                        value: s,
673                    }))
674                }
675                None => None,
676            };
677            (lsp_types::InlayHintLabel::String(text), hint_tooltip)
678        }
679        _ => {
680            let parts = label
681                .parts
682                .into_iter()
683                .map(|part| {
684                    let tooltip = part.tooltip.and_then(|it| match it {
685                        LazyProperty::Computed(it) => Some(it),
686                        LazyProperty::Lazy => {
687                            *something_to_resolve |= fields_to_resolve.resolve_label_tooltip;
688                            None
689                        }
690                    });
691                    let tooltip = match tooltip {
692                        Some(ide::InlayTooltip::String(s)) => {
693                            Some(lsp_types::InlayHintLabelPartTooltip::String(s))
694                        }
695                        Some(ide::InlayTooltip::Markdown(s)) => {
696                            Some(lsp_types::InlayHintLabelPartTooltip::MarkupContent(
697                                lsp_types::MarkupContent {
698                                    kind: lsp_types::MarkupKind::Markdown,
699                                    value: s,
700                                },
701                            ))
702                        }
703                        None => None,
704                    };
705                    let location = part
706                        .linked_location
707                        .and_then(|it| match it {
708                            LazyProperty::Computed(it) => Some(it),
709                            LazyProperty::Lazy => {
710                                *something_to_resolve |= fields_to_resolve.resolve_label_location;
711                                None
712                            }
713                        })
714                        .map(|range| location(snap, range))
715                        .transpose()?;
716                    Ok(lsp_types::InlayHintLabelPart {
717                        value: part.text,
718                        tooltip,
719                        location,
720                        command: None,
721                    })
722                })
723                .collect::<Cancellable<_>>()?;
724            (lsp_types::InlayHintLabel::LabelParts(parts), None)
725        }
726    };
727    Ok((label, tooltip))
728}

730static TOKEN_RESULT_COUNTER: AtomicU32 = AtomicU32::new(1);
731
732pub(crate) fn semantic_tokens(
733    text: &str,
734    line_index: &LineIndex,
735    highlights: Vec<HlRange>,
736    semantics_tokens_augments_syntax_tokens: bool,
737    non_standard_tokens: bool,
738) -> lsp_types::SemanticTokens {
739    let id = TOKEN_RESULT_COUNTER.fetch_add(1, Ordering::SeqCst).to_string();
740    let mut builder = semantic_tokens::SemanticTokensBuilder::new(id);
741
742    for highlight_range in highlights {
743        if highlight_range.highlight.is_empty() {
744            continue;
745        }
746
747        if semantics_tokens_augments_syntax_tokens {
748            match highlight_range.highlight.tag {
749                HlTag::BoolLiteral
750                | HlTag::ByteLiteral
751                | HlTag::CharLiteral
752                | HlTag::Comment
753                | HlTag::Keyword
754                | HlTag::NumericLiteral
755                | HlTag::Operator(_)
756                | HlTag::Punctuation(_)
757                | HlTag::StringLiteral
758                | HlTag::None
759                    if highlight_range.highlight.mods.is_empty() =>
760                {
761                    continue;
762                }
763                _ => (),
764            }
765        }
766
767        let (mut ty, mut mods) = semantic_token_type_and_modifiers(highlight_range.highlight);
768
769        if !non_standard_tokens {
770            ty = match standard_fallback_type(ty) {
771                Some(ty) => ty,
772                None => continue,
773            };
774            mods.standard_fallback();
775        }
776        let token_index = semantic_tokens::type_index(ty);
777        let modifier_bitset = mods.0;
778
779        for mut text_range in line_index.index.lines(highlight_range.range) {
780            if text[text_range].ends_with('\n') {
781                text_range =
782                    TextRange::new(text_range.start(), text_range.end() - TextSize::of('\n'));
783            }
784            let range = range(line_index, text_range);
785            builder.push(range, token_index, modifier_bitset);
786        }
787    }
788
789    builder.build()
790}

905pub(crate) fn folding_range(
906    text: &str,
907    line_index: &LineIndex,
908    line_folding_only: bool,
909    fold: Fold,
910) -> lsp_types::FoldingRange {
911    let kind = match fold.kind {
912        FoldKind::Comment => Some(lsp_types::FoldingRangeKind::Comment),
913        FoldKind::Imports => Some(lsp_types::FoldingRangeKind::Imports),
914        FoldKind::Region => Some(lsp_types::FoldingRangeKind::Region),
915        FoldKind::Modules
916        | FoldKind::Block
917        | FoldKind::ArgList
918        | FoldKind::Consts
919        | FoldKind::Statics
920        | FoldKind::TypeAliases
921        | FoldKind::WhereClause
922        | FoldKind::ReturnType
923        | FoldKind::Array
924        | FoldKind::ExternCrates
925        | FoldKind::MatchArm
926        | FoldKind::Function => None,
927    };
928
929    let range = range(line_index, fold.range);
930
931    if line_folding_only {
932        // Clients with line_folding_only == true (such as VSCode) will fold the whole end line
933        // even if it contains text not in the folding range. To prevent that we exclude
934        // range.end.line from the folding region if there is more text after range.end
935        // on the same line.
936        let has_more_text_on_end_line = text[TextRange::new(fold.range.end(), TextSize::of(text))]
937            .chars()
938            .take_while(|it| *it != '\n')
939            .any(|it| !it.is_whitespace());
940
941        let end_line = if has_more_text_on_end_line {
942            range.end.line.saturating_sub(1)
943        } else {
944            range.end.line
945        };
946
947        lsp_types::FoldingRange {
948            start_line: range.start.line,
949            start_character: None,
950            end_line,
951            end_character: None,
952            kind,
953            collapsed_text: None,
954        }
955    } else {
956        lsp_types::FoldingRange {
957            start_line: range.start.line,
958            start_character: Some(range.start.character),
959            end_line: range.end.line,
960            end_character: Some(range.end.character),
961            kind,
962            collapsed_text: None,
963        }
964    }
965}

971/// Returns a `Url` object from a given path, will lowercase drive letters if present.
972/// This will only happen when processing windows paths.
973///
974/// When processing non-windows path, this is essentially the same as `Url::from_file_path`.
975pub(crate) fn url_from_abs_path(path: &AbsPath) -> lsp_types::Url {
976    let url = lsp_types::Url::from_file_path(path).unwrap();
977    match path.components().next() {
978        Some(Utf8Component::Prefix(prefix))
979            if matches!(prefix.kind(), Utf8Prefix::Disk(_) | Utf8Prefix::VerbatimDisk(_)) =>
980        {
981            // Need to lowercase driver letter
982        }
983        _ => return url,
984    }
985
986    let driver_letter_range = {
987        let (scheme, drive_letter, _rest) = match url.as_str().splitn(3, ':').collect_tuple() {
988            Some(it) => it,
989            None => return url,
990        };
991        let start = scheme.len() + ':'.len_utf8();
992        start..(start + drive_letter.len())
993    };
994
995    // Note: lowercasing the `path` itself doesn't help, the `Url::parse`
996    // machinery *also* canonicalizes the drive letter. So, just massage the
997    // string in place.
998    let mut url: String = url.into();
999    url[driver_letter_range].make_ascii_lowercase();
1000    lsp_types::Url::parse(&url).unwrap()
1001}

1098fn merge_text_and_snippet_edits(
1099    line_index: &LineIndex,
1100    edit: TextEdit,
1101    snippet_edit: SnippetEdit,
1102    client_supports_annotations: bool,
1103) -> Vec<SnippetTextEdit> {
1104    let mut edits: Vec<SnippetTextEdit> = vec![];
1105    let mut snippets = snippet_edit.into_edit_ranges().into_iter().peekable();
1106    let annotation = edit.change_annotation();
1107    let text_edits = edit.into_iter();
1108    // offset to go from the final source location to the original source location
1109    let mut source_text_offset = 0i32;
1110
1111    let offset_range = |range: TextRange, offset: i32| -> TextRange {
1112        // map the snippet range from the target location into the original source location
1113        let start = u32::from(range.start()).checked_add_signed(offset).unwrap_or(0);
1114        let end = u32::from(range.end()).checked_add_signed(offset).unwrap_or(0);
1115
1116        TextRange::new(start.into(), end.into())
1117    };
1118
1119    for current_indel in text_edits {
1120        let new_range = {
1121            let insert_len =
1122                TextSize::try_from(current_indel.insert.len()).unwrap_or(TextSize::from(u32::MAX));
1123            TextRange::at(current_indel.delete.start(), insert_len)
1124        };
1125
1126        // figure out how much this Indel will shift future ranges from the initial source
1127        let offset_adjustment =
1128            u32::from(current_indel.delete.len()) as i32 - u32::from(new_range.len()) as i32;
1129
1130        // insert any snippets before the text edit
1131        for (snippet_index, snippet_range) in snippets.peeking_take_while(|(_, range)| {
1132            offset_range(*range, source_text_offset).end() < new_range.start()
1133        }) {
1134            // adjust the snippet range into the corresponding initial source location
1135            let snippet_range = offset_range(snippet_range, source_text_offset);
1136
1137            let snippet_range = if !stdx::always!(
1138                snippet_range.is_empty(),
1139                "placeholder range {:?} is before current text edit range {:?}",
1140                snippet_range,
1141                new_range
1142            ) {
1143                // only possible for tabstops, so make sure it's an empty/insert range
1144                TextRange::empty(snippet_range.start())
1145            } else {
1146                snippet_range
1147            };
1148
1149            edits.push(snippet_text_edit(
1150                line_index,
1151                true,
1152                Indel { insert: format!("${snippet_index}"), delete: snippet_range },
1153                annotation,
1154                client_supports_annotations,
1155            ))
1156        }
1157
1158        if snippets.peek().is_some_and(|(_, range)| {
1159            new_range.intersect(offset_range(*range, source_text_offset)).is_some()
1160        }) {
1161            // at least one snippet edit intersects this text edit,
1162            // so gather all of the edits that intersect this text edit
1163            let mut all_snippets = snippets
1164                .peeking_take_while(|(_, range)| {
1165                    new_range.intersect(offset_range(*range, source_text_offset)).is_some()
1166                })
1167                .map(|(tabstop, range)| (tabstop, offset_range(range, source_text_offset)))
1168                .collect_vec();
1169
1170            // ensure all of the ranges are wholly contained inside of the new range
1171            all_snippets.retain(|(_, range)| {
1172                    stdx::always!(
1173                        new_range.contains_range(*range),
1174                        "found placeholder range {:?} which wasn't fully inside of text edit's new range {:?}", range, new_range
1175                    )
1176                });
1177
1178            let mut new_text = current_indel.insert;
1179
1180            // find which snippet bits need to be escaped
1181            let escape_places =
1182                new_text.rmatch_indices(['\\', '$', '}']).map(|(insert, _)| insert).collect_vec();
1183            let mut escape_places = escape_places.into_iter().peekable();
1184            let mut escape_prior_bits = |new_text: &mut String, up_to: usize| {
1185                for before in escape_places.peeking_take_while(|insert| *insert >= up_to) {
1186                    new_text.insert(before, '\\');
1187                }
1188            };
1189
1190            // insert snippets, and escaping any needed bits along the way
1191            for (index, range) in all_snippets.iter().rev() {
1192                let text_range = range - new_range.start();
1193                let (start, end) = (text_range.start().into(), text_range.end().into());
1194
1195                if range.is_empty() {
1196                    escape_prior_bits(&mut new_text, start);
1197                    new_text.insert_str(start, &format!("${index}"));
1198                } else {
1199                    escape_prior_bits(&mut new_text, end);
1200                    new_text.insert(end, '}');
1201                    escape_prior_bits(&mut new_text, start);
1202                    new_text.insert_str(start, &format!("${{{index}:"));
1203                }
1204            }
1205
1206            // escape any remaining bits
1207            escape_prior_bits(&mut new_text, 0);
1208
1209            edits.push(snippet_text_edit(
1210                line_index,
1211                true,
1212                Indel { insert: new_text, delete: current_indel.delete },
1213                annotation,
1214                client_supports_annotations,
1215            ))
1216        } else {
1217            // snippet edit was beyond the current one
1218            // since it wasn't consumed, it's available for the next pass
1219            edits.push(snippet_text_edit(
1220                line_index,
1221                false,
1222                current_indel,
1223                annotation,
1224                client_supports_annotations,
1225            ));
1226        }
1227
1228        // update the final source -> initial source mapping offset
1229        source_text_offset += offset_adjustment;
1230    }
1231
1232    // insert any remaining tabstops
1233    edits.extend(snippets.map(|(snippet_index, snippet_range)| {
1234        // adjust the snippet range into the corresponding initial source location
1235        let snippet_range = offset_range(snippet_range, source_text_offset);
1236
1237        let snippet_range = if !stdx::always!(
1238            snippet_range.is_empty(),
1239            "found placeholder snippet {:?} without a text edit",
1240            snippet_range
1241        ) {
1242            TextRange::empty(snippet_range.start())
1243        } else {
1244            snippet_range
1245        };
1246
1247        snippet_text_edit(
1248            line_index,
1249            true,
1250            Indel { insert: format!("${snippet_index}"), delete: snippet_range },
1251            annotation,
1252            client_supports_annotations,
1253        )
1254    }));
1255
1256    edits
1257}

1259pub(crate) fn snippet_text_document_edit(
1260    snap: &GlobalStateSnapshot,
1261    is_snippet: bool,
1262    file_id: FileId,
1263    edit: TextEdit,
1264    snippet_edit: Option<SnippetEdit>,

1286    if snap.analysis.is_library_file(file_id)? && snap.config.change_annotation_support() {

1289        }

1294pub(crate) fn snippet_text_document_ops(
1295    snap: &GlobalStateSnapshot,
1296    file_system_edit: FileSystemEdit,
1297) -> Cancellable<Vec<lsp_ext::SnippetDocumentChangeOperation>> {
1298    let mut ops = Vec::new();
1299    match file_system_edit {
1300        FileSystemEdit::CreateFile { dst, initial_contents } => {
1301            let uri = snap.anchored_path(&dst);
1302            let create_file = lsp_types::ResourceOp::Create(lsp_types::CreateFile {
1303                uri: uri.clone(),
1304                options: None,
1305                annotation_id: None,
1306            });
1307            ops.push(lsp_ext::SnippetDocumentChangeOperation::Op(create_file));
1308            if !initial_contents.is_empty() {
1309                let text_document =
1310                    lsp_types::OptionalVersionedTextDocumentIdentifier { uri, version: None };
1311                let text_edit = lsp_ext::SnippetTextEdit {
1312                    range: lsp_types::Range::default(),
1313                    new_text: initial_contents,
1314                    insert_text_format: Some(lsp_types::InsertTextFormat::PLAIN_TEXT),
1315                    annotation_id: None,
1316                };
1317                let edit_file =
1318                    lsp_ext::SnippetTextDocumentEdit { text_document, edits: vec![text_edit] };
1319                ops.push(lsp_ext::SnippetDocumentChangeOperation::Edit(edit_file));
1320            }
1321        }
1322        FileSystemEdit::MoveFile { src, dst } => {
1323            let old_uri = snap.file_id_to_url(src);
1324            let new_uri = snap.anchored_path(&dst);
1325            let mut rename_file =
1326                lsp_types::RenameFile { old_uri, new_uri, options: None, annotation_id: None };
1327            if snap.analysis.is_library_file(src).ok() == Some(true)
1328                && snap.config.change_annotation_support()
1329            {
1330                rename_file.annotation_id = Some(outside_workspace_annotation_id())
1331            }
1332            ops.push(lsp_ext::SnippetDocumentChangeOperation::Op(lsp_types::ResourceOp::Rename(
1333                rename_file,
1334            )))
1335        }
1336        FileSystemEdit::MoveDir { src, src_id, dst } => {
1337            let old_uri = snap.anchored_path(&src);
1338            let new_uri = snap.anchored_path(&dst);
1339            let mut rename_file =
1340                lsp_types::RenameFile { old_uri, new_uri, options: None, annotation_id: None };
1341            if snap.analysis.is_library_file(src_id).ok() == Some(true)
1342                && snap.config.change_annotation_support()
1343            {
1344                rename_file.annotation_id = Some(outside_workspace_annotation_id())
1345            }
1346            ops.push(lsp_ext::SnippetDocumentChangeOperation::Op(lsp_types::ResourceOp::Rename(
1347                rename_file,
1348            )))
1349        }
1350    }
1351    Ok(ops)
1352}

1354pub(crate) fn snippet_workspace_edit(
1355    snap: &GlobalStateSnapshot,
1356    mut source_change: SourceChange,
1357) -> Cancellable<lsp_ext::SnippetWorkspaceEdit> {
1358    let mut document_changes: Vec<lsp_ext::SnippetDocumentChangeOperation> = Vec::new();
1359
1360    for op in &mut source_change.file_system_edits {
1361        if let FileSystemEdit::CreateFile { dst, initial_contents } = op {
1362            // replace with a placeholder to avoid cloneing the edit
1363            let op = FileSystemEdit::CreateFile {
1364                dst: dst.clone(),
1365                initial_contents: mem::take(initial_contents),
1366            };
1367            let ops = snippet_text_document_ops(snap, op)?;
1368            document_changes.extend_from_slice(&ops);
1369        }
1370    }
1371    for (file_id, (edit, snippet_edit)) in source_change.source_file_edits {
1372        let edit = snippet_text_document_edit(
1373            snap,
1374            source_change.is_snippet,
1375            file_id,
1376            edit,
1377            snippet_edit,
1378        )?;
1379        document_changes.push(lsp_ext::SnippetDocumentChangeOperation::Edit(edit));
1380    }
1381    for op in source_change.file_system_edits {
1382        if !matches!(op, FileSystemEdit::CreateFile { .. }) {
1383            let ops = snippet_text_document_ops(snap, op)?;
1384            document_changes.extend_from_slice(&ops);
1385        }
1386    }
1387    let mut workspace_edit = lsp_ext::SnippetWorkspaceEdit {
1388        changes: None,
1389        document_changes: Some(document_changes),
1390        change_annotations: None,
1391    };
1392    if snap.config.change_annotation_support() {
1393        workspace_edit.change_annotations = Some(
1394            once((
1395                outside_workspace_annotation_id(),
1396                lsp_types::ChangeAnnotation {
1397                    label: String::from("Edit outside of the workspace"),
1398                    needs_confirmation: Some(true),
1399                    description: Some(String::from(
1400                        "This edit lies outside of the workspace and may affect dependencies",
1401                    )),
1402                },
1403            ))
1404            .chain(source_change.annotations.into_iter().map(|(id, annotation)| {
1405                (
1406                    id.to_string(),
1407                    lsp_types::ChangeAnnotation {
1408                        label: annotation.label,
1409                        description: annotation.description,
1410                        needs_confirmation: Some(annotation.needs_confirmation),
1411                    },
1412                )
1413            }))
1414            .collect(),
1415        )
1416    }
1417    Ok(workspace_edit)
1418}

1620                return Ok(None);
1621            };
1622            let (cargo_args, executable_args) =
1623                CargoTargetSpec::runnable_args(snap, None, &runnable.kind, &runnable.cfg);
1624

1627

1629                label,
1630                location: Some(location),
1631                kind: lsp_ext::RunnableKind::Cargo,
1632                args: lsp_ext::RunnableArgs::Cargo(lsp_ext::CargoRunnableArgs {
1633                    workspace_root: None,
1634                    override_cargo: config.override_cargo,
1635                    cargo_args,
1636                    cwd: path.as_path().unwrap().to_path_buf().into(),
1637                    executable_args,
1638                    environment: Default::default(),
1639                }),
1640            }))
1641        }
1642    }

1645pub(crate) fn code_lens(
1646    acc: &mut Vec<lsp_types::CodeLens>,
1647    snap: &GlobalStateSnapshot,
1648    annotation: Annotation,
1649) -> Cancellable<()> {
1650    let client_commands_config = snap.config.client_commands();
1651    match annotation.kind {
1652        AnnotationKind::Runnable(run) => {
1653            let line_index = snap.file_line_index(run.nav.file_id)?;
1654            let annotation_range = range(&line_index, annotation.range);
1655
1656            let update_test = run.update_test;
1657            let title = run.title();
1658            let can_debug = match run.kind {
1659                ide::RunnableKind::DocTest { .. } => false,
1660                ide::RunnableKind::TestMod { .. }
1661                | ide::RunnableKind::Test { .. }
1662                | ide::RunnableKind::Bench { .. }
1663                | ide::RunnableKind::Bin => true,
1664            };
1665            let r = runnable(snap, run)?;
1666
1667            if let Some(r) = r {
1668                let has_root = match &r.args {
1669                    lsp_ext::RunnableArgs::Cargo(c) => c.workspace_root.is_some(),
1670                    lsp_ext::RunnableArgs::Shell(_) => true,
1671                };
1672
1673                let lens_config = snap.config.lens();
1674
1675                if has_root {
1676                    if lens_config.run && client_commands_config.run_single {
1677                        let command = command::run_single(&r, &title);
1678                        acc.push(lsp_types::CodeLens {
1679                            range: annotation_range,
1680                            command: Some(command),
1681                            data: None,
1682                        })
1683                    }
1684                    if lens_config.debug && can_debug && client_commands_config.debug_single {
1685                        let command = command::debug_single(&r);
1686                        acc.push(lsp_types::CodeLens {
1687                            range: annotation_range,
1688                            command: Some(command),
1689                            data: None,
1690                        })
1691                    }
1692                    if lens_config.update_test && client_commands_config.run_single {
1693                        let label = update_test.label();
1694                        if let Some(r) = make_update_runnable(&r, update_test) {
1695                            let command = command::run_single(&r, label.unwrap().as_str());
1696                            acc.push(lsp_types::CodeLens {
1697                                range: annotation_range,
1698                                command: Some(command),
1699                                data: None,
1700                            })
1701                        }
1702                    }
1703                }
1704
1705                if lens_config.interpret {
1706                    let command = command::interpret_single(&r);
1707                    acc.push(lsp_types::CodeLens {
1708                        range: annotation_range,
1709                        command: Some(command),
1710                        data: None,
1711                    })
1712                }
1713            }
1714        }
1715        AnnotationKind::HasImpls { pos, data } => {
1716            if !client_commands_config.show_reference {
1717                return Ok(());
1718            }
1719            let line_index = snap.file_line_index(pos.file_id)?;
1720            let annotation_range = range(&line_index, annotation.range);
1721            let url = url(snap, pos.file_id);
1722            let pos = position(&line_index, pos.offset);
1723
1724            let id = lsp_types::TextDocumentIdentifier { uri: url.clone() };
1725
1726            let doc_pos = lsp_types::TextDocumentPositionParams::new(id, pos);
1727
1728            let goto_params = lsp_types::request::GotoImplementationParams {
1729                text_document_position_params: doc_pos,
1730                work_done_progress_params: Default::default(),
1731                partial_result_params: Default::default(),
1732            };
1733
1734            let command = data.map(|ranges| {
1735                let locations: Vec<lsp_types::Location> = ranges
1736                    .into_iter()
1737                    .filter_map(|target| {
1738                        location(
1739                            snap,
1740                            FileRange { file_id: target.file_id, range: target.full_range },
1741                        )
1742                        .ok()
1743                    })
1744                    .collect();
1745
1746                command::show_references(
1747                    implementation_title(locations.len()),
1748                    &url,
1749                    pos,
1750                    locations,
1751                )
1752            });
1753
1754            acc.push(lsp_types::CodeLens {
1755                range: annotation_range,
1756                command,
1757                data: (|| {
1758                    let version = snap.url_file_version(&url)?;
1759                    Some(
1760                        to_value(lsp_ext::CodeLensResolveData {
1761                            version,
1762                            kind: lsp_ext::CodeLensResolveDataKind::Impls(goto_params),
1763                        })
1764                        .unwrap(),
1765                    )
1766                })(),
1767            })
1768        }
1769        AnnotationKind::HasReferences { pos, data } => {
1770            if !client_commands_config.show_reference {
1771                return Ok(());
1772            }
1773            let line_index = snap.file_line_index(pos.file_id)?;
1774            let annotation_range = range(&line_index, annotation.range);
1775            let url = url(snap, pos.file_id);
1776            let pos = position(&line_index, pos.offset);
1777
1778            let id = lsp_types::TextDocumentIdentifier { uri: url.clone() };
1779
1780            let doc_pos = lsp_types::TextDocumentPositionParams::new(id, pos);
1781
1782            let command = data.map(|ranges| {
1783                let locations: Vec<lsp_types::Location> =
1784                    ranges.into_iter().filter_map(|range| location(snap, range).ok()).collect();
1785
1786                command::show_references(reference_title(locations.len()), &url, pos, locations)
1787            });
1788
1789            acc.push(lsp_types::CodeLens {
1790                range: annotation_range,
1791                command,
1792                data: (|| {
1793                    let version = snap.url_file_version(&url)?;
1794                    Some(
1795                        to_value(lsp_ext::CodeLensResolveData {
1796                            version,
1797                            kind: lsp_ext::CodeLensResolveDataKind::References(doc_pos),
1798                        })
1799                        .unwrap(),
1800                    )
1801                })(),
1802            })
1803        }
1804    }
1805    Ok(())
1806}
