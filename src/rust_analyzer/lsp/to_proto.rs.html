42pub(crate) fn position(line_index: &LineIndex, offset: TextSize) -> lsp_types::Position {
43    let line_col = line_index.index.line_col(offset);
44    match line_index.encoding {
45        PositionEncoding::Utf8 => lsp_types::Position::new(line_col.line, line_col.col),
46        PositionEncoding::Wide(enc) => {
47            let line_col = line_index.index.to_wide(enc, line_col).unwrap();
48            lsp_types::Position::new(line_col.line, line_col.col)
49        }
50    }
51}

53pub(crate) fn range(line_index: &LineIndex, range: TextRange) -> lsp_types::Range {
54    let start = position(line_index, range.start());
55    let end = position(line_index, range.end());
56    lsp_types::Range::new(start, end)

248pub(crate) fn completion_items(
249    config: &Config,
250    fields_to_resolve: &CompletionFieldsToResolve,
251    line_index: &LineIndex,
252    version: Option<i32>,
253    tdpp: lsp_types::TextDocumentPositionParams,
254    completion_trigger_character: Option<char>,
255    mut items: Vec<CompletionItem>,
256) -> Vec<lsp_types::CompletionItem> {
257    if config.completion_hide_deprecated() {
258        items.retain(|item| !item.deprecated);
259    }
260
261    let max_relevance = items.iter().map(|it| it.relevance.score()).max().unwrap_or_default();
262    let mut res = Vec::with_capacity(items.len());
263    let client_commands = config.client_commands();
264    for item in items {
265        completion_item(
266            &mut res,
267            config,
268            &client_commands,
269            fields_to_resolve,
270            line_index,
271            version,
272            &tdpp,
273            max_relevance,
274            completion_trigger_character,
275            item,
276        );
277    }
278
279    if let Some(limit) = config.completion(None, MiniCore::default()).limit {
280        res.sort_by(|item1, item2| item1.sort_text.cmp(&item2.sort_text));
281        res.truncate(limit);
282    }
283
284    res
285}

287fn completion_item(
288    acc: &mut Vec<lsp_types::CompletionItem>,
289    config: &Config,
290    client_commands: &ClientCommandsConfig,
291    fields_to_resolve: &CompletionFieldsToResolve,
292    line_index: &LineIndex,
293    version: Option<i32>,
294    tdpp: &lsp_types::TextDocumentPositionParams,
295    max_relevance: u32,
296    completion_trigger_character: Option<char>,
297    item: CompletionItem,
298) {
299    let insert_replace_support = config.insert_replace_support().then_some(tdpp.position);
300    let ref_match = item.ref_match();
301
302    let mut additional_text_edits = Vec::new();
303    let mut something_to_resolve = false;
304
305    let filter_text = if fields_to_resolve.resolve_filter_text {
306        something_to_resolve |= !item.lookup().is_empty();
307        None
308    } else {
309        Some(item.lookup().to_owned())
310    };
311
312    let text_edit = if fields_to_resolve.resolve_text_edit {
313        something_to_resolve |= true;
314        None
315    } else {
316        // LSP does not allow arbitrary edits in completion, so we have to do a
317        // non-trivial mapping here.
318        let mut text_edit = None;
319        let source_range = item.source_range;
320        for indel in &item.text_edit {
321            if indel.delete.contains_range(source_range) {
322                // Extract this indel as the main edit
323                text_edit = Some(if indel.delete == source_range {
324                    self::completion_text_edit(line_index, insert_replace_support, indel.clone())
325                } else {
326                    assert!(source_range.end() == indel.delete.end());
327                    let range1 = TextRange::new(indel.delete.start(), source_range.start());
328                    let range2 = source_range;
329                    let indel1 = Indel::delete(range1);
330                    let indel2 = Indel::replace(range2, indel.insert.clone());
331                    additional_text_edits.push(self::text_edit(line_index, indel1));
332                    self::completion_text_edit(line_index, insert_replace_support, indel2)
333                })
334            } else {
335                assert!(source_range.intersect(indel.delete).is_none());
336                let text_edit = self::text_edit(line_index, indel.clone());
337                additional_text_edits.push(text_edit);
338            }
339        }
340        Some(text_edit.unwrap())
341    };
342
343    let insert_text_format = item.is_snippet.then_some(lsp_types::InsertTextFormat::SNIPPET);
344    let tags = if fields_to_resolve.resolve_tags {
345        something_to_resolve |= item.deprecated;
346        None
347    } else {
348        item.deprecated.then(|| vec![lsp_types::CompletionItemTag::DEPRECATED])
349    };
350    let command = if item.trigger_call_info && client_commands.trigger_parameter_hints {
351        if fields_to_resolve.resolve_command {
352            something_to_resolve |= true;
353            None
354        } else {
355            Some(command::trigger_parameter_hints())
356        }
357    } else {
358        None
359    };
360
361    let detail = if fields_to_resolve.resolve_detail {
362        something_to_resolve |= item.detail.is_some();
363        None
364    } else {
365        item.detail.clone()
366    };
367
368    let documentation = if fields_to_resolve.resolve_documentation {
369        something_to_resolve |= item.documentation.is_some();
370        None
371    } else {
372        item.documentation.clone().map(documentation)
373    };
374
375    let mut lsp_item = lsp_types::CompletionItem {
376        label: item.label.primary.to_string(),
377        detail,
378        filter_text,
379        kind: Some(completion_item_kind(item.kind)),
380        text_edit,
381        additional_text_edits: additional_text_edits
382            .is_empty()
383            .not()
384            .then_some(additional_text_edits),
385        documentation,
386        deprecated: item.deprecated.then_some(item.deprecated),
387        tags,
388        command,
389        insert_text_format,
390        ..Default::default()
391    };
392
393    if config.completion_label_details_support() {
394        let has_label_details =
395            item.label.detail_left.is_some() || item.label.detail_right.is_some();
396        if fields_to_resolve.resolve_label_details {
397            something_to_resolve |= has_label_details;
398        } else if has_label_details {
399            lsp_item.label_details = Some(lsp_types::CompletionItemLabelDetails {
400                detail: item.label.detail_left.clone(),
401                description: item.label.detail_right.clone(),
402            });
403        }
404    } else if let Some(label_detail) = &item.label.detail_left {
405        lsp_item.label.push_str(label_detail.as_str());
406    }
407
408    set_score(&mut lsp_item, max_relevance, item.relevance);
409
410    let imports = if config.completion(None, MiniCore::default()).enable_imports_on_the_fly
411        && !item.import_to_add.is_empty()
412    {
413        item.import_to_add
414            .clone()
415            .into_iter()
416            .map(|import_path| lsp_ext::CompletionImport { full_import_path: import_path })
417            .collect()
418    } else {
419        Vec::new()
420    };
421    let (ref_resolve_data, resolve_data) = if something_to_resolve || !imports.is_empty() {
422        let ref_resolve_data = if ref_match.is_some() {
423            let ref_resolve_data = lsp_ext::CompletionResolveData {
424                position: tdpp.clone(),
425                imports: Vec::new(),
426                version,
427                trigger_character: completion_trigger_character,
428                for_ref: true,
429                hash: BASE64_STANDARD.encode(completion_item_hash(&item, true)),
430            };
431            Some(to_value(ref_resolve_data).unwrap())
432        } else {
433            None
434        };
435        let resolve_data = lsp_ext::CompletionResolveData {
436            position: tdpp.clone(),
437            imports,
438            version,
439            trigger_character: completion_trigger_character,
440            for_ref: false,
441            hash: BASE64_STANDARD.encode(completion_item_hash(&item, false)),
442        };
443        (ref_resolve_data, Some(to_value(resolve_data).unwrap()))
444    } else {
445        (None, None)
446    };
447
448    if let Some((label, indel, relevance)) = ref_match {
449        let mut lsp_item_with_ref =
450            lsp_types::CompletionItem { label, data: ref_resolve_data, ..lsp_item.clone() };
451        lsp_item_with_ref
452            .additional_text_edits
453            .get_or_insert_with(Default::default)
454            .push(self::text_edit(line_index, indel));
455        set_score(&mut lsp_item_with_ref, max_relevance, relevance);
456        acc.push(lsp_item_with_ref);
457    };
458
459    lsp_item.data = resolve_data;
460    acc.push(lsp_item);
461
462    fn set_score(
463        res: &mut lsp_types::CompletionItem,
464        max_relevance: u32,
465        relevance: CompletionRelevance,
466    ) {
467        if relevance.is_relevant() && relevance.score() == max_relevance {
468            res.preselect = Some(true);
469        }
470        // The relevance needs to be inverted to come up with a sort score
471        // because the client will sort ascending.
472        let sort_score = relevance.score() ^ 0xFF_FF_FF_FF;
473        // Zero pad the string to ensure values can be properly sorted
474        // by the client. Hex format is used because it is easier to
475        // visually compare very large values, which the sort text
476        // tends to be since it is the opposite of the score.
477        res.sort_text = Some(format!("{sort_score:08x}"));
478    }
479}

481pub(crate) fn signature_help(
482    call_info: SignatureHelp,
483    config: CallInfoConfig,
484    label_offsets: bool,
485) -> lsp_types::SignatureHelp {
486    let (label, parameters) = match (config.params_only, label_offsets) {
487        (concise, false) => {
488            let params = call_info
489                .parameter_labels()
490                .map(|label| lsp_types::ParameterInformation {
491                    label: lsp_types::ParameterLabel::Simple(label.to_owned()),
492                    documentation: None,
493                })
494                .collect::<Vec<_>>();
495            let label =
496                if concise { call_info.parameter_labels().join(", ") } else { call_info.signature };
497            (label, params)
498        }
499        (false, true) => {
500            let params = call_info
501                .parameter_ranges()
502                .iter()
503                .map(|it| {
504                    let start = call_info.signature[..it.start().into()]
505                        .chars()
506                        .map(|c| c.len_utf16())
507                        .sum::<usize>() as u32;
508                    let end = start
509                        + call_info.signature[it.start().into()..it.end().into()]
510                            .chars()
511                            .map(|c| c.len_utf16())
512                            .sum::<usize>() as u32;
513                    [start, end]
514                })
515                .map(|label_offsets| lsp_types::ParameterInformation {
516                    label: lsp_types::ParameterLabel::LabelOffsets(label_offsets),
517                    documentation: None,
518                })
519                .collect::<Vec<_>>();
520            (call_info.signature, params)
521        }
522        (true, true) => {
523            let mut params = Vec::new();
524            let mut label = String::new();
525            let mut first = true;
526            for param in call_info.parameter_labels() {
527                if !first {
528                    label.push_str(", ");
529                }
530                first = false;
531                let start = label.len() as u32;
532                label.push_str(param);
533                let end = label.len() as u32;
534                params.push(lsp_types::ParameterInformation {
535                    label: lsp_types::ParameterLabel::LabelOffsets([start, end]),
536                    documentation: None,
537                });
538            }
539
540            (label, params)
541        }
542    };
543
544    let documentation = call_info.doc.filter(|_| config.docs).map(|doc| {
545        lsp_types::Documentation::MarkupContent(lsp_types::MarkupContent {
546            kind: lsp_types::MarkupKind::Markdown,
547            value: format_docs(&doc),
548        })
549    });
550
551    let active_parameter = call_info.active_parameter.map(|it| it as u32);
552
553    let signature = lsp_types::SignatureInformation {
554        label,
555        documentation,
556        parameters: Some(parameters),
557        active_parameter,
558    };
559    lsp_types::SignatureHelp {
560        signatures: vec![signature],
561        active_signature: Some(0),
562        active_parameter,
563    }
564}

566pub(crate) fn inlay_hint(
567    snap: &GlobalStateSnapshot,
568    fields_to_resolve: &InlayFieldsToResolve,
569    line_index: &LineIndex,
570    file_id: FileId,
571    mut inlay_hint: InlayHint,
572) -> Cancellable<lsp_types::InlayHint> {
573    let hint_needs_resolve = |hint: &InlayHint| -> Option<TextRange> {
574        hint.resolve_parent.filter(|_| {
575            hint.text_edit.as_ref().is_some_and(LazyProperty::is_lazy)
576                || hint.label.parts.iter().any(|part| {
577                    part.linked_location.as_ref().is_some_and(LazyProperty::is_lazy)
578                        || part.tooltip.as_ref().is_some_and(LazyProperty::is_lazy)
579                })
580        })
581    };
582
583    let resolve_range_and_hash = hint_needs_resolve(&inlay_hint).map(|range| {
584        (
585            range,
586            std::hash::BuildHasher::hash_one(
587                &std::hash::BuildHasherDefault::<FxHasher>::default(),
588                &inlay_hint,
589            ),
590        )
591    });
592
593    let mut something_to_resolve = false;
594    let text_edits = inlay_hint
595        .text_edit
596        .take()
597        .and_then(|it| match it {
598            LazyProperty::Computed(it) => Some(it),
599            LazyProperty::Lazy => {
600                something_to_resolve |=
601                    snap.config.visual_studio_code_version().is_none_or(|version| {
602                        VersionReq::parse(">=1.86.0").unwrap().matches(version)
603                    }) && resolve_range_and_hash.is_some()
604                        && fields_to_resolve.resolve_text_edits;
605                None
606            }
607        })
608        .map(|it| text_edit_vec(line_index, it));
609    let (label, tooltip) = inlay_hint_label(
610        snap,
611        fields_to_resolve,
612        &mut something_to_resolve,
613        resolve_range_and_hash.is_some(),
614        inlay_hint.label,
615    )?;
616
617    let data = match resolve_range_and_hash {
618        Some((resolve_range, hash)) if something_to_resolve => Some(
619            to_value(lsp_ext::InlayHintResolveData {
620                file_id: file_id.index(),
621                hash: hash.to_string(),
622                version: snap.file_version(file_id),
623                resolve_range: range(line_index, resolve_range),
624            })
625            .unwrap(),
626        ),
627        _ => None,
628    };
629
630    Ok(lsp_types::InlayHint {
631        position: match inlay_hint.position {
632            ide::InlayHintPosition::Before => position(line_index, inlay_hint.range.start()),
633            ide::InlayHintPosition::After => position(line_index, inlay_hint.range.end()),
634        },
635        padding_left: Some(inlay_hint.pad_left),
636        padding_right: Some(inlay_hint.pad_right),
637        kind: match inlay_hint.kind {
638            InlayKind::Parameter | InlayKind::GenericParameter => {
639                Some(lsp_types::InlayHintKind::PARAMETER)
640            }
641            InlayKind::Type | InlayKind::Chaining => Some(lsp_types::InlayHintKind::TYPE),
642            _ => None,
643        },
644        text_edits,
645        data,
646        tooltip,
647        label,
648    })
649}

651fn inlay_hint_label(
652    snap: &GlobalStateSnapshot,
653    fields_to_resolve: &InlayFieldsToResolve,
654    something_to_resolve: &mut bool,
655    needs_resolve: bool,
656    mut label: InlayHintLabel,
657) -> Cancellable<(lsp_types::InlayHintLabel, Option<lsp_types::InlayHintTooltip>)> {
658    let (label, tooltip) = match &*label.parts {
659        [InlayHintLabelPart { linked_location: None, .. }] => {
660            let InlayHintLabelPart { text, tooltip, .. } = label.parts.pop().unwrap();
661            let tooltip = tooltip.and_then(|it| match it {
662                LazyProperty::Computed(it) => Some(it),
663                LazyProperty::Lazy => {
664                    *something_to_resolve |=
665                        needs_resolve && fields_to_resolve.resolve_hint_tooltip;
666                    None
667                }
668            });
669            let hint_tooltip = match tooltip {
670                Some(ide::InlayTooltip::String(s)) => Some(lsp_types::InlayHintTooltip::String(s)),
671                Some(ide::InlayTooltip::Markdown(s)) => {
672                    Some(lsp_types::InlayHintTooltip::MarkupContent(lsp_types::MarkupContent {
673                        kind: lsp_types::MarkupKind::Markdown,
674                        value: s,
675                    }))
676                }
677                None => None,
678            };
679            (lsp_types::InlayHintLabel::String(text), hint_tooltip)
680        }
681        _ => {
682            let parts = label
683                .parts
684                .into_iter()
685                .map(|part| {
686                    let tooltip = part.tooltip.and_then(|it| match it {
687                        LazyProperty::Computed(it) => Some(it),
688                        LazyProperty::Lazy => {
689                            *something_to_resolve |= fields_to_resolve.resolve_label_tooltip;
690                            None
691                        }
692                    });
693                    let tooltip = match tooltip {
694                        Some(ide::InlayTooltip::String(s)) => {
695                            Some(lsp_types::InlayHintLabelPartTooltip::String(s))
696                        }
697                        Some(ide::InlayTooltip::Markdown(s)) => {
698                            Some(lsp_types::InlayHintLabelPartTooltip::MarkupContent(
699                                lsp_types::MarkupContent {
700                                    kind: lsp_types::MarkupKind::Markdown,
701                                    value: s,
702                                },
703                            ))
704                        }
705                        None => None,
706                    };
707                    let location = part
708                        .linked_location
709                        .and_then(|it| match it {
710                            LazyProperty::Computed(it) => Some(it),
711                            LazyProperty::Lazy => {
712                                *something_to_resolve |= fields_to_resolve.resolve_label_location;
713                                None
714                            }
715                        })
716                        .map(|range| location(snap, range))
717                        .transpose()?;
718                    Ok(lsp_types::InlayHintLabelPart {
719                        value: part.text,
720                        tooltip,
721                        location,
722                        command: None,
723                    })
724                })
725                .collect::<Cancellable<_>>()?;
726            (lsp_types::InlayHintLabel::LabelParts(parts), None)
727        }
728    };
729    Ok((label, tooltip))
730}

732static TOKEN_RESULT_COUNTER: AtomicU32 = AtomicU32::new(1);
733
734pub(crate) fn semantic_tokens(
735    text: &str,
736    line_index: &LineIndex,
737    highlights: Vec<HlRange>,
738    semantics_tokens_augments_syntax_tokens: bool,
739    non_standard_tokens: bool,
740) -> lsp_types::SemanticTokens {
741    let id = TOKEN_RESULT_COUNTER.fetch_add(1, Ordering::SeqCst).to_string();
742    let mut builder = semantic_tokens::SemanticTokensBuilder::new(id);
743
744    for highlight_range in highlights {
745        if highlight_range.highlight.is_empty() {
746            continue;
747        }
748
749        if semantics_tokens_augments_syntax_tokens {
750            match highlight_range.highlight.tag {
751                HlTag::BoolLiteral
752                | HlTag::ByteLiteral
753                | HlTag::CharLiteral
754                | HlTag::Comment
755                | HlTag::Keyword
756                | HlTag::NumericLiteral
757                | HlTag::Operator(_)
758                | HlTag::Punctuation(_)
759                | HlTag::StringLiteral
760                | HlTag::None
761                    if highlight_range.highlight.mods.is_empty() =>
762                {
763                    continue;
764                }
765                _ => (),
766            }
767        }
768
769        let (mut ty, mut mods) = semantic_token_type_and_modifiers(highlight_range.highlight);
770
771        if !non_standard_tokens {
772            ty = match standard_fallback_type(ty) {
773                Some(ty) => ty,
774                None => continue,
775            };
776            mods.standard_fallback();
777        }
778        let token_index = semantic_tokens::type_index(ty);
779        let modifier_bitset = mods.0;
780
781        for mut text_range in line_index.index.lines(highlight_range.range) {
782            if text[text_range].ends_with('\n') {
783                text_range =
784                    TextRange::new(text_range.start(), text_range.end() - TextSize::of('\n'));
785            }
786            let range = range(line_index, text_range);
787            builder.push(range, token_index, modifier_bitset);
788        }
789    }
790
791    builder.build()
792}

912pub(crate) fn folding_range(
913    text: &str,
914    line_index: &LineIndex,
915    line_folding_only: bool,
916    fold: Fold,
917) -> lsp_types::FoldingRange {
918    let kind = match fold.kind {
919        FoldKind::Comment => Some(lsp_types::FoldingRangeKind::Comment),
920        FoldKind::Imports => Some(lsp_types::FoldingRangeKind::Imports),
921        FoldKind::Region => Some(lsp_types::FoldingRangeKind::Region),
922        FoldKind::Modules
923        | FoldKind::Block
924        | FoldKind::ArgList
925        | FoldKind::Consts
926        | FoldKind::Statics
927        | FoldKind::TypeAliases
928        | FoldKind::WhereClause
929        | FoldKind::ReturnType
930        | FoldKind::Array
931        | FoldKind::ExternCrates
932        | FoldKind::MatchArm
933        | FoldKind::Function => None,
934    };
935
936    let range = range(line_index, fold.range);
937
938    if line_folding_only {
939        // Clients with line_folding_only == true (such as VSCode) will fold the whole end line
940        // even if it contains text not in the folding range. To prevent that we exclude
941        // range.end.line from the folding region if there is more text after range.end
942        // on the same line.
943        let has_more_text_on_end_line = text[TextRange::new(fold.range.end(), TextSize::of(text))]
944            .chars()
945            .take_while(|it| *it != '\n')
946            .any(|it| !it.is_whitespace());
947
948        let end_line = if has_more_text_on_end_line {
949            range.end.line.saturating_sub(1)
950        } else {
951            range.end.line
952        };
953
954        lsp_types::FoldingRange {
955            start_line: range.start.line,
956            start_character: None,
957            end_line,
958            end_character: None,
959            kind,
960            collapsed_text: None,
961        }
962    } else {
963        lsp_types::FoldingRange {
964            start_line: range.start.line,
965            start_character: Some(range.start.character),
966            end_line: range.end.line,
967            end_character: Some(range.end.character),
968            kind,
969            collapsed_text: None,
970        }
971    }
972}

978/// Returns a `Url` object from a given path, will lowercase drive letters if present.
979/// This will only happen when processing windows paths.
980///
981/// When processing non-windows path, this is essentially the same as `Url::from_file_path`.
982pub(crate) fn url_from_abs_path(path: &AbsPath) -> lsp_types::Url {
983    let url = lsp_types::Url::from_file_path(path).unwrap();
984    match path.components().next() {
985        Some(Utf8Component::Prefix(prefix))
986            if matches!(prefix.kind(), Utf8Prefix::Disk(_) | Utf8Prefix::VerbatimDisk(_)) =>
987        {
988            // Need to lowercase driver letter
989        }
990        _ => return url,
991    }
992
993    let driver_letter_range = {
994        let (scheme, drive_letter, _rest) = match url.as_str().splitn(3, ':').collect_tuple() {
995            Some(it) => it,
996            None => return url,
997        };
998        let start = scheme.len() + ':'.len_utf8();
999        start..(start + drive_letter.len())
1000    };
1001
1002    // Note: lowercasing the `path` itself doesn't help, the `Url::parse`
1003    // machinery *also* canonicalizes the drive letter. So, just massage the
1004    // string in place.
1005    let mut url: String = url.into();
1006    url[driver_letter_range].make_ascii_lowercase();
1007    lsp_types::Url::parse(&url).unwrap()
1008}

1105fn merge_text_and_snippet_edits(
1106    line_index: &LineIndex,
1107    edit: TextEdit,
1108    snippet_edit: SnippetEdit,
1109    client_supports_annotations: bool,
1110) -> Vec<SnippetTextEdit> {
1111    let mut edits: Vec<SnippetTextEdit> = vec![];
1112    let mut snippets = snippet_edit.into_edit_ranges().into_iter().peekable();
1113    let annotation = edit.change_annotation();
1114    let text_edits = edit.into_iter();
1115    // offset to go from the final source location to the original source location
1116    let mut source_text_offset = 0i32;
1117
1118    let offset_range = |range: TextRange, offset: i32| -> TextRange {
1119        // map the snippet range from the target location into the original source location
1120        let start = u32::from(range.start()).checked_add_signed(offset).unwrap_or(0);
1121        let end = u32::from(range.end()).checked_add_signed(offset).unwrap_or(0);
1122
1123        TextRange::new(start.into(), end.into())
1124    };
1125
1126    for current_indel in text_edits {
1127        let new_range = {
1128            let insert_len =
1129                TextSize::try_from(current_indel.insert.len()).unwrap_or(TextSize::from(u32::MAX));
1130            TextRange::at(current_indel.delete.start(), insert_len)
1131        };
1132
1133        // figure out how much this Indel will shift future ranges from the initial source
1134        let offset_adjustment =
1135            u32::from(current_indel.delete.len()) as i32 - u32::from(new_range.len()) as i32;
1136
1137        // insert any snippets before the text edit
1138        for (snippet_index, snippet_range) in snippets.peeking_take_while(|(_, range)| {
1139            offset_range(*range, source_text_offset).end() < new_range.start()
1140        }) {
1141            // adjust the snippet range into the corresponding initial source location
1142            let snippet_range = offset_range(snippet_range, source_text_offset);
1143
1144            let snippet_range = if !stdx::always!(
1145                snippet_range.is_empty(),
1146                "placeholder range {:?} is before current text edit range {:?}",
1147                snippet_range,
1148                new_range
1149            ) {
1150                // only possible for tabstops, so make sure it's an empty/insert range
1151                TextRange::empty(snippet_range.start())
1152            } else {
1153                snippet_range
1154            };
1155
1156            edits.push(snippet_text_edit(
1157                line_index,
1158                true,
1159                Indel { insert: format!("${snippet_index}"), delete: snippet_range },
1160                annotation,
1161                client_supports_annotations,
1162            ))
1163        }
1164
1165        if snippets.peek().is_some_and(|(_, range)| {
1166            new_range.intersect(offset_range(*range, source_text_offset)).is_some()
1167        }) {
1168            // at least one snippet edit intersects this text edit,
1169            // so gather all of the edits that intersect this text edit
1170            let mut all_snippets = snippets
1171                .peeking_take_while(|(_, range)| {
1172                    new_range.intersect(offset_range(*range, source_text_offset)).is_some()
1173                })
1174                .map(|(tabstop, range)| (tabstop, offset_range(range, source_text_offset)))
1175                .collect_vec();
1176
1177            // ensure all of the ranges are wholly contained inside of the new range
1178            all_snippets.retain(|(_, range)| {
1179                    stdx::always!(
1180                        new_range.contains_range(*range),
1181                        "found placeholder range {:?} which wasn't fully inside of text edit's new range {:?}", range, new_range
1182                    )
1183                });
1184
1185            let mut new_text = current_indel.insert;
1186
1187            // find which snippet bits need to be escaped
1188            let escape_places =
1189                new_text.rmatch_indices(['\\', '$', '}']).map(|(insert, _)| insert).collect_vec();
1190            let mut escape_places = escape_places.into_iter().peekable();
1191            let mut escape_prior_bits = |new_text: &mut String, up_to: usize| {
1192                for before in escape_places.peeking_take_while(|insert| *insert >= up_to) {
1193                    new_text.insert(before, '\\');
1194                }
1195            };
1196
1197            // insert snippets, and escaping any needed bits along the way
1198            for (index, range) in all_snippets.iter().rev() {
1199                let text_range = range - new_range.start();
1200                let (start, end) = (text_range.start().into(), text_range.end().into());
1201
1202                if range.is_empty() {
1203                    escape_prior_bits(&mut new_text, start);
1204                    new_text.insert_str(start, &format!("${index}"));
1205                } else {
1206                    escape_prior_bits(&mut new_text, end);
1207                    new_text.insert(end, '}');
1208                    escape_prior_bits(&mut new_text, start);
1209                    new_text.insert_str(start, &format!("${{{index}:"));
1210                }
1211            }
1212
1213            // escape any remaining bits
1214            escape_prior_bits(&mut new_text, 0);
1215
1216            edits.push(snippet_text_edit(
1217                line_index,
1218                true,
1219                Indel { insert: new_text, delete: current_indel.delete },
1220                annotation,
1221                client_supports_annotations,
1222            ))
1223        } else {
1224            // snippet edit was beyond the current one
1225            // since it wasn't consumed, it's available for the next pass
1226            edits.push(snippet_text_edit(
1227                line_index,
1228                false,
1229                current_indel,
1230                annotation,
1231                client_supports_annotations,
1232            ));
1233        }
1234
1235        // update the final source -> initial source mapping offset
1236        source_text_offset += offset_adjustment;
1237    }
1238
1239    // insert any remaining tabstops
1240    edits.extend(snippets.map(|(snippet_index, snippet_range)| {
1241        // adjust the snippet range into the corresponding initial source location
1242        let snippet_range = offset_range(snippet_range, source_text_offset);
1243
1244        let snippet_range = if !stdx::always!(
1245            snippet_range.is_empty(),
1246            "found placeholder snippet {:?} without a text edit",
1247            snippet_range
1248        ) {
1249            TextRange::empty(snippet_range.start())
1250        } else {
1251            snippet_range
1252        };
1253
1254        snippet_text_edit(
1255            line_index,
1256            true,
1257            Indel { insert: format!("${snippet_index}"), delete: snippet_range },
1258            annotation,
1259            client_supports_annotations,
1260        )
1261    }));
1262
1263    edits
1264}

1301pub(crate) fn snippet_text_document_ops(
1302    snap: &GlobalStateSnapshot,
1303    file_system_edit: FileSystemEdit,
1304) -> Cancellable<Vec<lsp_ext::SnippetDocumentChangeOperation>> {
1305    let mut ops = Vec::new();
1306    match file_system_edit {
1307        FileSystemEdit::CreateFile { dst, initial_contents } => {
1308            let uri = snap.anchored_path(&dst);
1309            let create_file = lsp_types::ResourceOp::Create(lsp_types::CreateFile {
1310                uri: uri.clone(),
1311                options: None,
1312                annotation_id: None,
1313            });
1314            ops.push(lsp_ext::SnippetDocumentChangeOperation::Op(create_file));
1315            if !initial_contents.is_empty() {
1316                let text_document =
1317                    lsp_types::OptionalVersionedTextDocumentIdentifier { uri, version: None };
1318                let text_edit = lsp_ext::SnippetTextEdit {
1319                    range: lsp_types::Range::default(),
1320                    new_text: initial_contents,
1321                    insert_text_format: Some(lsp_types::InsertTextFormat::PLAIN_TEXT),
1322                    annotation_id: None,
1323                };
1324                let edit_file =
1325                    lsp_ext::SnippetTextDocumentEdit { text_document, edits: vec![text_edit] };
1326                ops.push(lsp_ext::SnippetDocumentChangeOperation::Edit(edit_file));
1327            }
1328        }
1329        FileSystemEdit::MoveFile { src, dst } => {
1330            let old_uri = snap.file_id_to_url(src);
1331            let new_uri = snap.anchored_path(&dst);
1332            let mut rename_file =
1333                lsp_types::RenameFile { old_uri, new_uri, options: None, annotation_id: None };
1334            if snap.analysis.is_library_file(src).ok() == Some(true)
1335                && snap.config.change_annotation_support()
1336            {
1337                rename_file.annotation_id = Some(outside_workspace_annotation_id())
1338            }
1339            ops.push(lsp_ext::SnippetDocumentChangeOperation::Op(lsp_types::ResourceOp::Rename(
1340                rename_file,
1341            )))
1342        }
1343        FileSystemEdit::MoveDir { src, src_id, dst } => {
1344            let old_uri = snap.anchored_path(&src);
1345            let new_uri = snap.anchored_path(&dst);
1346            let mut rename_file =
1347                lsp_types::RenameFile { old_uri, new_uri, options: None, annotation_id: None };
1348            if snap.analysis.is_library_file(src_id).ok() == Some(true)
1349                && snap.config.change_annotation_support()
1350            {
1351                rename_file.annotation_id = Some(outside_workspace_annotation_id())
1352            }
1353            ops.push(lsp_ext::SnippetDocumentChangeOperation::Op(lsp_types::ResourceOp::Rename(
1354                rename_file,
1355            )))
1356        }
1357    }
1358    Ok(ops)
1359}

1361pub(crate) fn snippet_workspace_edit(
1362    snap: &GlobalStateSnapshot,
1363    mut source_change: SourceChange,
1364) -> Cancellable<lsp_ext::SnippetWorkspaceEdit> {
1365    let mut document_changes: Vec<lsp_ext::SnippetDocumentChangeOperation> = Vec::new();
1366
1367    for op in &mut source_change.file_system_edits {
1368        if let FileSystemEdit::CreateFile { dst, initial_contents } = op {
1369            // replace with a placeholder to avoid cloneing the edit
1370            let op = FileSystemEdit::CreateFile {
1371                dst: dst.clone(),
1372                initial_contents: mem::take(initial_contents),
1373            };
1374            let ops = snippet_text_document_ops(snap, op)?;
1375            document_changes.extend_from_slice(&ops);
1376        }
1377    }
1378    for (file_id, (edit, snippet_edit)) in source_change.source_file_edits {
1379        let edit = snippet_text_document_edit(
1380            snap,
1381            source_change.is_snippet,
1382            file_id,
1383            edit,
1384            snippet_edit,
1385        )?;
1386        document_changes.push(lsp_ext::SnippetDocumentChangeOperation::Edit(edit));
1387    }
1388    for op in source_change.file_system_edits {
1389        if !matches!(op, FileSystemEdit::CreateFile { .. }) {
1390            let ops = snippet_text_document_ops(snap, op)?;
1391            document_changes.extend_from_slice(&ops);
1392        }
1393    }
1394    let mut workspace_edit = lsp_ext::SnippetWorkspaceEdit {
1395        changes: None,
1396        document_changes: Some(document_changes),
1397        change_annotations: None,
1398    };
1399    if snap.config.change_annotation_support() {
1400        workspace_edit.change_annotations = Some(
1401            once((
1402                outside_workspace_annotation_id(),
1403                lsp_types::ChangeAnnotation {
1404                    label: String::from("Edit outside of the workspace"),
1405                    needs_confirmation: Some(true),
1406                    description: Some(String::from(
1407                        "This edit lies outside of the workspace and may affect dependencies",
1408                    )),
1409                },
1410            ))
1411            .chain(source_change.annotations.into_iter().map(|(id, annotation)| {
1412                (
1413                    id.to_string(),
1414                    lsp_types::ChangeAnnotation {
1415                        label: annotation.label,
1416                        description: annotation.description,
1417                        needs_confirmation: Some(annotation.needs_confirmation),
1418                    },
1419                )
1420            }))
1421            .collect(),
1422        )
1423    }
1424    Ok(workspace_edit)
1425}

1560    let target_spec = TargetSpec::for_file(snap, runnable.nav.file_id)?;
1561    let source_root = snap.analysis.source_root_id(runnable.nav.file_id).ok();
1562    let config = snap.config.runnables(source_root);
1563
1564    match target_spec {
1565        Some(TargetSpec::Cargo(spec)) => {
1566            let workspace_root = spec.workspace_root.clone();
1567
1568            let target = spec.target.clone();
1569
1570            let override_command =
1571                CargoTargetSpec::override_command(snap, Some(spec.clone()), &runnable.kind);
1572

1574                snap,
1575                Some(spec.clone()),
1576                &runnable.kind,
1577                &runnable.cfg,
1578            );

1580            let cwd = match runnable.kind {
1581                ide::RunnableKind::Bin => workspace_root.clone(),
1582                _ => spec.cargo_toml.parent().to_owned(),
1583            };
1584
1585            let label = runnable.label(Some(&target));

1588            let environment = spec
1589                .sysroot_root

1591                .into_iter()
1592                .collect();
1593
1594            Ok(match override_command {
1595                Some(override_command) => match override_command.split_first() {
1596                    Some((program, args)) => Some(lsp_ext::Runnable {

1598                        location: Some(location),

1606                    }),

1609                None => Some(lsp_ext::Runnable {
1610                    label,
1611                    location: Some(location),
1612                    kind: lsp_ext::RunnableKind::Cargo,
1613                    args: lsp_ext::RunnableArgs::Cargo(lsp_ext::CargoRunnableArgs {

1631                        program: json_shell_runnable_args.program,
1632                        args: json_shell_runnable_args.args,
1633                        cwd: json_shell_runnable_args.cwd,

1637                        label,
1638                        location: Some(location),

1642                }
1643                None => Ok(None),
1644            }
1645        }
1646        None => {
1647            let Some(path) = snap.file_id_to_file_path(runnable.nav.file_id).parent() else {
1648                return Ok(None);
1649            };
1650            let (cargo_args, executable_args) =
1651                CargoTargetSpec::runnable_args(snap, None, &runnable.kind, &runnable.cfg);
1652
1653            let label = runnable.label(None);
1654            let location = location_link(snap, None, runnable.nav)?;

1656            Ok(Some(lsp_ext::Runnable {
1657                label,
1658                location: Some(location),
1659                kind: lsp_ext::RunnableKind::Cargo,

1661                    workspace_root: None,
1662                    override_cargo: config.override_cargo,
1663                    cargo_args,
1664                    cwd: path.as_path().unwrap().to_path_buf().into(),
1665                    executable_args,
1666                    environment: Default::default(),
1667                }),
1668            }))
1669        }
1670    }

1673pub(crate) fn code_lens(
1674    acc: &mut Vec<lsp_types::CodeLens>,
1675    snap: &GlobalStateSnapshot,
1676    annotation: Annotation,
1677) -> Cancellable<()> {
1678    let client_commands_config = snap.config.client_commands();
1679    match annotation.kind {
1680        AnnotationKind::Runnable(run) => {
1681            let line_index = snap.file_line_index(run.nav.file_id)?;
1682            let annotation_range = range(&line_index, annotation.range);
1683
1684            let update_test = run.update_test;
1685            let title = run.title();
1686            let can_debug = match run.kind {
1687                ide::RunnableKind::DocTest { .. } => false,
1688                ide::RunnableKind::TestMod { .. }
1689                | ide::RunnableKind::Test { .. }
1690                | ide::RunnableKind::Bench { .. }
1691                | ide::RunnableKind::Bin => true,
1692            };
1693            let r = runnable(snap, run)?;
1694
1695            if let Some(r) = r {
1696                let has_root = match &r.args {
1697                    lsp_ext::RunnableArgs::Cargo(c) => c.workspace_root.is_some(),
1698                    lsp_ext::RunnableArgs::Shell(_) => true,
1699                };
1700
1701                let lens_config = snap.config.lens();
1702
1703                if has_root {
1704                    if lens_config.run && client_commands_config.run_single {
1705                        let command = command::run_single(&r, &title);
1706                        acc.push(lsp_types::CodeLens {
1707                            range: annotation_range,
1708                            command: Some(command),
1709                            data: None,
1710                        })
1711                    }
1712                    if lens_config.debug && can_debug && client_commands_config.debug_single {
1713                        let command = command::debug_single(&r);
1714                        acc.push(lsp_types::CodeLens {
1715                            range: annotation_range,
1716                            command: Some(command),
1717                            data: None,
1718                        })
1719                    }
1720                    if lens_config.update_test && client_commands_config.run_single {
1721                        let label = update_test.label();
1722                        if let Some(r) = make_update_runnable(&r, update_test) {
1723                            let command = command::run_single(&r, label.unwrap().as_str());
1724                            acc.push(lsp_types::CodeLens {
1725                                range: annotation_range,
1726                                command: Some(command),
1727                                data: None,
1728                            })
1729                        }
1730                    }
1731                }
1732
1733                if lens_config.interpret {
1734                    let command = command::interpret_single(&r);
1735                    acc.push(lsp_types::CodeLens {
1736                        range: annotation_range,
1737                        command: Some(command),
1738                        data: None,
1739                    })
1740                }
1741            }
1742        }
1743        AnnotationKind::HasImpls { pos, data } => {
1744            if !client_commands_config.show_reference {
1745                return Ok(());
1746            }
1747            let line_index = snap.file_line_index(pos.file_id)?;
1748            let annotation_range = range(&line_index, annotation.range);
1749            let url = url(snap, pos.file_id);
1750            let pos = position(&line_index, pos.offset);
1751
1752            let id = lsp_types::TextDocumentIdentifier { uri: url.clone() };
1753
1754            let doc_pos = lsp_types::TextDocumentPositionParams::new(id, pos);
1755
1756            let goto_params = lsp_types::request::GotoImplementationParams {
1757                text_document_position_params: doc_pos,
1758                work_done_progress_params: Default::default(),
1759                partial_result_params: Default::default(),
1760            };
1761
1762            let command = data.map(|ranges| {
1763                let locations: Vec<lsp_types::Location> = ranges
1764                    .into_iter()
1765                    .filter_map(|target| {
1766                        location(
1767                            snap,
1768                            FileRange { file_id: target.file_id, range: target.full_range },
1769                        )
1770                        .ok()
1771                    })
1772                    .collect();
1773
1774                command::show_references(
1775                    implementation_title(locations.len()),
1776                    &url,
1777                    pos,
1778                    locations,
1779                )
1780            });
1781
1782            acc.push(lsp_types::CodeLens {
1783                range: annotation_range,
1784                command,
1785                data: (|| {
1786                    let version = snap.url_file_version(&url)?;
1787                    Some(
1788                        to_value(lsp_ext::CodeLensResolveData {
1789                            version,
1790                            kind: lsp_ext::CodeLensResolveDataKind::Impls(goto_params),
1791                        })
1792                        .unwrap(),
1793                    )
1794                })(),
1795            })
1796        }
1797        AnnotationKind::HasReferences { pos, data } => {
1798            if !client_commands_config.show_reference {
1799                return Ok(());
1800            }
1801            let line_index = snap.file_line_index(pos.file_id)?;
1802            let annotation_range = range(&line_index, annotation.range);
1803            let url = url(snap, pos.file_id);
1804            let pos = position(&line_index, pos.offset);
1805
1806            let id = lsp_types::TextDocumentIdentifier { uri: url.clone() };
1807
1808            let doc_pos = lsp_types::TextDocumentPositionParams::new(id, pos);
