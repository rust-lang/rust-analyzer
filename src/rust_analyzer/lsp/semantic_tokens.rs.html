7macro_rules! declare_enum {
8    (
9        $(#[$attrs:meta])*
10        $visibility:vis enum $name:ident {
11            $($variant:ident),* $(,)?
12        }
13    ) => {
14        $(#[$attrs])*
15        $visibility enum $name {
16            $($variant,)*
17        }
18
19        impl $name {
20            pub(crate) fn iter() -> Iter<'static, Self> {
21                static ITEMS: &[$name] = &[
22                    $(
23                        $name::$variant,
24                    )*
25                ];
26                ITEMS.iter()
27            }
28        }
29    };
30}

250const STANDARD_MOD: [SupportedModifiers; 6] = [
251    SupportedModifiers::Async,
252    SupportedModifiers::Documentation,
253    SupportedModifiers::Declaration,
254    SupportedModifiers::Static,
255    SupportedModifiers::DefaultLibrary,
256    SupportedModifiers::Deprecated,
257];
258const LAST_STANDARD_MOD: usize = STANDARD_MOD.len() - 1;
259
260#[derive(Default)]
261pub(crate) struct ModifierSet(pub(crate) u32);
262
263impl ModifierSet {
264    pub(crate) fn standard_fallback(&mut self) {
265        // Remove all non standard modifiers
266        self.0 &= !(!0u32 << LAST_STANDARD_MOD)
267    }
268}
269
270impl ops::BitOrAssign<SupportedModifiers> for ModifierSet {
271    fn bitor_assign(&mut self, rhs: SupportedModifiers) {
272        let idx = SupportedModifiers::iter().position(|it| *it == rhs).unwrap();
273        self.0 |= 1 << idx;
274    }
275}

277/// Tokens are encoded relative to each other.
278///
279/// This is a direct port of <https://github.com/microsoft/vscode-languageserver-node/blob/f425af9de46a0187adb78ec8a46b9b2ce80c5412/server/src/sematicTokens.proposed.ts#L45>
280pub(crate) struct SemanticTokensBuilder {
281    id: String,
282    prev_line: u32,
283    prev_char: u32,
284    data: Vec<lsp_types::SemanticToken>,
285}
286
287impl SemanticTokensBuilder {
288    pub(crate) fn new(id: String) -> Self {
289        SemanticTokensBuilder { id, prev_line: 0, prev_char: 0, data: Vec::new() }
290    }
291
292    /// Push a new token onto the builder
293    pub(crate) fn push(&mut self, range: Range, token_index: u32, modifier_bitset: u32) {
294        let mut push_line = range.start.line;
295        let mut push_char = range.start.character;
296
297        if !self.data.is_empty() {
298            push_line -= self.prev_line;
299            if push_line == 0 {
300                push_char -= self.prev_char;
301            }
302        }
303
304        // A token cannot be multiline
305        let token_len = range.end.character - range.start.character;
306
307        let token = lsp_types::SemanticToken {
308            delta_line: push_line,
309            delta_start: push_char,
310            length: token_len,
311            token_type: token_index,
312            token_modifiers_bitset: modifier_bitset,
313        };
314
315        self.data.push(token);
316
317        self.prev_line = range.start.line;
318        self.prev_char = range.start.character;
319    }
320
321    pub(crate) fn build(self) -> SemanticTokens {
322        SemanticTokens { result_id: Some(self.id), data: self.data }
323    }
324}

326pub(crate) fn diff_tokens(
327    old: &[lsp_types::SemanticToken],
328    new: &[lsp_types::SemanticToken],
329) -> Vec<SemanticTokensEdit> {
330    let offset = new.iter().zip(old.iter()).take_while(|&(n, p)| n == p).count();
331
332    let (_, old) = old.split_at(offset);
333    let (_, new) = new.split_at(offset);
334
335    let offset_from_end =
336        new.iter().rev().zip(old.iter().rev()).take_while(|&(n, p)| n == p).count();
337
338    let (old, _) = old.split_at(old.len() - offset_from_end);
339    let (new, _) = new.split_at(new.len() - offset_from_end);
340
341    if old.is_empty() && new.is_empty() {
342        vec![]
343    } else {
344        // The lsp data field is actually a byte-diff but we
345        // travel in tokens so `start` and `delete_count` are in multiples of the
346        // serialized size of `SemanticToken`.
347        vec![SemanticTokensEdit {
348            start: 5 * offset as u32,
349            delete_count: 5 * old.len() as u32,
350            data: Some(new.into()),
351        }]
352    }
353}
