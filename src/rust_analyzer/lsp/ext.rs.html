583#[derive(Serialize, Deserialize, Clone, Copy, PartialEq, Eq, Debug)]
584#[serde(rename_all = "camelCase")]
585pub enum Health {
586    Ok,
587    Warning,
588    Error,
589}
590
591impl ops::BitOrAssign for Health {
592    fn bitor_assign(&mut self, rhs: Self) {
593        *self = match (*self, rhs) {
594            (Health::Error, _) | (_, Health::Error) => Health::Error,
595            (Health::Warning, _) | (_, Health::Warning) => Health::Warning,
596            _ => Health::Ok,
597        }
598    }
599}

961#[cfg(test)]
962mod tests {
963    use super::*;
964    use serde_json::json;
965
966    #[test]
967    fn cargo_runnable_round_trips() {
968        let runnable = Runnable {
969            label: "cargo test -p my-crate".to_owned(),
970            location: None,
971            args: RunnableArgs::Cargo(CargoRunnableArgs {
972                environment: [("RUSTC_TOOLCHAIN".to_owned(), "/toolchain".to_owned())]
973                    .into_iter()
974                    .collect(),
975                cwd: "/project".into(),
976                override_cargo: None,
977                workspace_root: Some("/project".into()),
978                cargo_args: vec![
979                    "test".into(),
980                    "--package".into(),
981                    "my-crate".into(),
982                    "--lib".into(),
983                ],
984                executable_args: vec!["my_test".into(), "--exact".into()],
985            }),
986        };
987        let expected = json!({
988            "label": "cargo test -p my-crate",
989            "kind": "cargo",
990            "args": {
991                "environment": {"RUSTC_TOOLCHAIN": "/toolchain"},
992                "cwd": "/project",
993                "overrideCargo": null,
994                "workspaceRoot": "/project",
995                "cargoArgs": ["test", "--package", "my-crate", "--lib"],
996                "executableArgs": ["my_test", "--exact"],
997            }
998        });
999
1000        let serialized = serde_json::to_value(&runnable).expect("serialized runnable");
1001        assert_eq!(serialized, expected);
1002
1003        let deserialized: Runnable =
1004            serde_json::from_value(expected).expect("cargo runnable should deserialize");
1005        let RunnableArgs::Cargo(cargo) = &deserialized.args else {
1006            panic!("expected Cargo variant, got {:?}", deserialized.args);
1007        };
1008        assert_eq!(cargo.cargo_args, vec!["test", "--package", "my-crate", "--lib"]);
1009        assert_eq!(cargo.executable_args, vec!["my_test", "--exact"]);
1010    }
1011
1012    #[test]
1013    fn shell_runnable_round_trips() {
1014        let runnable = Runnable {
1015            label: "nextest test_one".to_owned(),
1016            location: None,
1017            args: RunnableArgs::Shell(ShellRunnableArgs {
1018                environment: [("RUSTC_TOOLCHAIN".to_owned(), "/toolchain".to_owned())]
1019                    .into_iter()
1020                    .collect(),
1021                cwd: "/project".into(),
1022                program: "cargo".into(),
1023                args: vec!["nextest".into(), "run".into(), "--package".into(), "my-crate".into()],
1024            }),
1025        };
1026        let expected = json!({
1027            "label": "nextest test_one",
1028            "kind": "shell",
1029            "args": {
1030                "environment": {"RUSTC_TOOLCHAIN": "/toolchain"},
1031                "cwd": "/project",
1032                "program": "cargo",
1033                "args": ["nextest", "run", "--package", "my-crate"],
1034            }
1035        });
1036
1037        let serialized = serde_json::to_value(&runnable).expect("serialized runnable");
1038        assert_eq!(serialized, expected);
1039
1040        // Every shell runnable is a structurally valid cargo runnable if the `kind` tag isn't
1041        // used. This test ensures that the `kind` tag is used.
1042        let deserialized: Runnable =
1043            serde_json::from_value(expected).expect("shell runnable should deserialize");
1044        let RunnableArgs::Shell(shell) = &deserialized.args else {
1045            panic!("expected Shell variant, got {:?}", deserialized.args);
1046        };
1047        assert_eq!(shell.program, "cargo");
1048        assert_eq!(shell.args, vec!["nextest", "run", "--package", "my-crate"]);
1049    }
