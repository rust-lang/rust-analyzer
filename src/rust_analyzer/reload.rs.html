1//! Project loading & configuration updates.
2//!
3//! This is quite tricky. The main problem is time and changes -- there's no
4//! fixed "project" rust-analyzer is working with, "current project" is itself
5//! mutable state. For example, when the user edits `Cargo.toml` by adding a new
6//! dependency, project model changes. What's more, switching project model is
7//! not instantaneous -- it takes time to run `cargo metadata` and (for proc
8//! macros) `cargo check`.
9//!
10//! The main guiding principle here is, as elsewhere in rust-analyzer,
11//! robustness. We try not to assume that the project model exists or is
12//! correct. Instead, we try to provide a best-effort service. Even if the
13//! project is currently loading and we don't have a full project model, we
14//! still want to respond to various  requests.

66    /// Is the server quiescent?
67    ///
68    /// This indicates that we've fully loaded the projects and
69    /// are ready to do semantic work.
70    pub(crate) fn is_quiescent(&self) -> bool {
71        self.vfs_done
72            && self.fetch_ws_receiver.is_none()
73            && !self.fetch_workspaces_queue.op_in_progress()
74            && !self.fetch_build_data_queue.op_in_progress()
75            && !self.fetch_proc_macros_queue.op_in_progress()
76            && !self.discover_workspace_queue.op_in_progress()
77            && self.vfs_progress_config_version >= self.vfs_config_version
78    }
79
80    /// Is the server ready to respond to analysis dependent LSP requests?
81    ///
82    /// Unlike `is_quiescent`, this returns false when we're indexing
83    /// the project, because we're holding the salsa lock and cannot
84    /// respond to LSP requests that depend on salsa data.
85    fn is_fully_ready(&self) -> bool {
86        self.is_quiescent() && !self.prime_caches_queue.op_in_progress()
87    }

89    pub(crate) fn update_configuration(&mut self, config: Config) {
90        let _p = tracing::info_span!("GlobalState::update_configuration").entered();
91        let old_config = mem::replace(&mut self.config, Arc::new(config));
92        if self.config.lru_parse_query_capacity() != old_config.lru_parse_query_capacity() {
93            self.analysis_host.update_lru_capacity(self.config.lru_parse_query_capacity());
94        }
95        if self.config.lru_query_capacities_config() != old_config.lru_query_capacities_config() {
96            self.analysis_host.update_lru_capacities(
97                &self.config.lru_query_capacities_config().cloned().unwrap_or_default(),
98            );
99        }
100
101        if self.config.linked_or_discovered_projects() != old_config.linked_or_discovered_projects()
102        {
103            let req = FetchWorkspaceRequest { path: None, force_crate_graph_reload: false };
104            self.fetch_workspaces_queue.request_op("discovered projects changed".to_owned(), req)
105        } else if self.config.flycheck(None) != old_config.flycheck(None) {
106            self.reload_flycheck();
107        }
108
109        if self.analysis_host.raw_database().expand_proc_attr_macros()
110            != self.config.expand_proc_attr_macros()
111        {
112            self.analysis_host.raw_database_mut().set_expand_proc_attr_macros_with_durability(
113                self.config.expand_proc_attr_macros(),
114                Durability::HIGH,
115            );
116        }
117
118        if self.config.cargo(None) != old_config.cargo(None) {
119            let req = FetchWorkspaceRequest { path: None, force_crate_graph_reload: false };
120            self.fetch_workspaces_queue.request_op("cargo config changed".to_owned(), req)
121        }
122
123        if self.config.cfg_set_test(None) != old_config.cfg_set_test(None) {
124            let req = FetchWorkspaceRequest { path: None, force_crate_graph_reload: false };
125            self.fetch_workspaces_queue.request_op("cfg_set_test config changed".to_owned(), req)
126        }
127    }

129    pub(crate) fn current_status(&self) -> lsp_ext::ServerStatusParams {
130        let mut status = lsp_ext::ServerStatusParams {
131            health: lsp_ext::Health::Ok,
132            quiescent: self.is_fully_ready(),
133            message: None,
134        };
135        let mut message = String::new();
136
137        if !self.config.cargo_autoreload_config(None)
138            && self.is_quiescent()
139            && self.fetch_workspaces_queue.op_requested()
140            && self.config.discover_workspace_config().is_none()
141        {
142            status.health |= lsp_ext::Health::Warning;
143            message.push_str("Auto-reloading is disabled and the workspace has changed, a manual workspace reload is required.\n\n");
144        }
145
146        if self.build_deps_changed {
147            status.health |= lsp_ext::Health::Warning;
148            message.push_str(
149                "Proc-macros and/or build scripts have changed and need to be rebuilt.\n\n",
150            );
151        }
152        if self.fetch_build_data_error().is_err() {
153            status.health |= lsp_ext::Health::Warning;
154            message.push_str("Failed to run build scripts of some packages.\n\n");
155            message.push_str(
156                "Please refer to the language server logs for more details on the errors.",
157            );
158        }
159        if let Some(err) = &self.config_errors {
160            status.health |= lsp_ext::Health::Warning;
161            format_to!(message, "{err}\n");
162        }
163        if let Some(err) = &self.last_flycheck_error {
164            status.health |= lsp_ext::Health::Warning;
165            message.push_str(err);
166            message.push('\n');
167        }
168
169        if self.config.linked_or_discovered_projects().is_empty()
170            && self.config.detached_files().is_empty()
171        {
172            status.health |= lsp_ext::Health::Warning;
173            message.push_str("Failed to discover workspace.\n");
174            message.push_str("Consider adding the `Cargo.toml` of the workspace to the [`linkedProjects`](https://rust-analyzer.github.io/book/configuration.html#linkedProjects) setting.\n\n");
175        }
176        if self.fetch_workspace_error().is_err() {
177            status.health |= lsp_ext::Health::Error;
178            message.push_str("Failed to load workspaces.");
179
180            if self.config.has_linked_projects() {
181                message.push_str(
182                    "`rust-analyzer.linkedProjects` have been specified, which may be incorrect. Specified project paths:\n",
183                );
184                message
185                    .push_str(&format!("    {}", self.config.linked_manifests().format("\n    ")));
186                if self.config.has_linked_project_jsons() {
187                    message.push_str("\nAdditionally, one or more project jsons are specified")
188                }
189            }
190            message.push_str("\n\n");
191        }
192
193        if !self.workspaces.is_empty() {
194            self.check_workspaces_msrv().for_each(|e| {
195                status.health |= lsp_ext::Health::Warning;
196                format_to!(message, "{e}");
197            });
198
199            let proc_macro_clients = self.proc_macro_clients.iter().chain(iter::repeat(&None));
200
201            for (ws, proc_macro_client) in self.workspaces.iter().zip(proc_macro_clients) {
202                if let ProjectWorkspaceKind::Cargo { error: Some(error), .. }
203                | ProjectWorkspaceKind::DetachedFile {
204                    cargo: Some((_, _, Some(error))), ..
205                } = &ws.kind
206                {
207                    status.health |= lsp_ext::Health::Warning;
208                    format_to!(
209                        message,
210                        "Failed to read Cargo metadata with dependencies for `{}`: {:#}\n\n",
211                        ws.manifest_or_root(),
212                        error
213                    );
214                }
215                if let Some(err) = ws.sysroot.error() {
216                    status.health |= lsp_ext::Health::Warning;
217                    format_to!(
218                        message,
219                        "Workspace `{}` has sysroot errors: ",
220                        ws.manifest_or_root()
221                    );
222                    message.push_str(err);
223                    message.push_str("\n\n");
224                }
225                if let Some(err) = ws.sysroot.metadata_error() {
226                    status.health |= lsp_ext::Health::Warning;
227                    format_to!(
228                        message,
229                        "Failed to read Cargo metadata with dependencies for sysroot of `{}`: ",
230                        ws.manifest_or_root()
231                    );
232                    message.push_str(err);
233                    message.push_str("\n\n");
234                }
235                if let ProjectWorkspaceKind::Cargo { rustc: Err(Some(err)), .. } = &ws.kind {
236                    status.health |= lsp_ext::Health::Warning;
237                    format_to!(
238                        message,
239                        "Failed loading rustc_private crates for workspace `{}`: ",
240                        ws.manifest_or_root()
241                    );
242                    message.push_str(err);
243                    message.push_str("\n\n");
244                };
245                match proc_macro_client {
246                    Some(Err(err)) => {
247                        status.health |= lsp_ext::Health::Warning;
248                        format_to!(
249                            message,
250                            "Failed spawning proc-macro server for workspace `{}`: {err}",
251                            ws.manifest_or_root()
252                        );
253                        message.push_str("\n\n");
254                    }
255                    Some(Ok(client)) => {
256                        if let Some(err) = client.exited() {
257                            status.health |= lsp_ext::Health::Warning;
258                            format_to!(
259                                message,
260                                "proc-macro server for workspace `{}` exited: {err}",
261                                ws.manifest_or_root()
262                            );
263                            message.push_str("\n\n");
264                        }
265                    }
266                    // sysroot was explicitly not set so we didn't discover a server
267                    None => {}
268                }
269            }
270        }
271
272        if !message.is_empty() {
273            status.message = Some(message.trim_end().to_owned());
274        }
275
276        status
277    }

279    pub(crate) fn fetch_workspaces(
280        &mut self,
281        cause: Cause,
282        path: Option<AbsPathBuf>,
283        force_crate_graph_reload: bool,
284    ) {
285        info!(%cause, "will fetch workspaces");
286
287        self.task_pool.handle.spawn_with_sender(ThreadIntent::Worker, {
288            let linked_projects = self.config.linked_or_discovered_projects();
289            let detached_files: Vec<_> = self
290                .config
291                .detached_files()
292                .iter()
293                .cloned()
294                .map(ManifestPath::try_from)
295                .filter_map(Result::ok)
296                .collect();
297            let cargo_config = self.config.cargo(None);
298            let discover_command = self.config.discover_workspace_config().cloned();
299            let is_quiescent = !(self.discover_workspace_queue.op_in_progress()
300                || self.vfs_progress_config_version < self.vfs_config_version
301                || !self.vfs_done);
302
303            move |sender| {
304                let progress = {
305                    let sender = sender.clone();
306                    move |msg| {
307                        sender
308                            .send(Task::FetchWorkspace(ProjectWorkspaceProgress::Report(msg)))
309                            .unwrap()
310                    }
311                };
312
313                sender.send(Task::FetchWorkspace(ProjectWorkspaceProgress::Begin)).unwrap();
314
315                if let (Some(_command), Some(path)) = (&discover_command, &path) {
316                    let build = linked_projects.iter().find_map(|project| match project {
317                        LinkedProject::InlineProjectJson(it) => it.crate_by_buildfile(path),
318                        _ => None,
319                    });
320
321                    if let Some(build) = build
322                        && is_quiescent
323                    {
324                        let path = AbsPathBuf::try_from(build.build_file)
325                            .expect("Unable to convert to an AbsPath");
326                        let arg = DiscoverProjectParam::Buildfile(path);
327                        sender.send(Task::DiscoverLinkedProjects(arg)).unwrap();
328                    }
329                }
330
331                let mut workspaces: Vec<_> = linked_projects
332                    .iter()
333                    .map(|project| match project {
334                        LinkedProject::ProjectManifest(manifest) => {
335                            debug!(path = %manifest, "loading project from manifest");
336
337                            project_model::ProjectWorkspace::load(
338                                manifest.clone(),
339                                &cargo_config,
340                                &progress,
341                            )
342                        }
343                        LinkedProject::InlineProjectJson(it) => {
344                            let workspace = project_model::ProjectWorkspace::load_inline(
345                                it.clone(),
346                                &cargo_config,
347                                &progress,
348                            );
349                            Ok(workspace)
350                        }
351                    })
352                    .collect();
353
354                let mut i = 0;
355                while i < workspaces.len() {
356                    if let Ok(w) = &workspaces[i] {
357                        let dupes: Vec<_> = workspaces[i + 1..]
358                            .iter()
359                            .positions(|it| it.as_ref().is_ok_and(|ws| ws.eq_ignore_build_data(w)))
360                            .collect();
361                        dupes.into_iter().rev().for_each(|d| {
362                            _ = workspaces.remove(d + i + 1);
363                        });
364                    }
365                    i += 1;
366                }
367
368                if !detached_files.is_empty() {
369                    workspaces.extend(project_model::ProjectWorkspace::load_detached_files(
370                        detached_files,
371                        &cargo_config,
372                    ));
373                }
374
375                info!(?workspaces, "did fetch workspaces");
376                sender
377                    .send(Task::FetchWorkspace(ProjectWorkspaceProgress::End(
378                        workspaces,
379                        force_crate_graph_reload,
380                    )))
381                    .unwrap();
382            }
383        });
384    }

386    pub(crate) fn fetch_build_data(&mut self, cause: Cause) {
387        info!(%cause, "will fetch build data");
388        let workspaces = Arc::clone(&self.workspaces);
389        let config = self.config.cargo(None);
390        let root_path = self.config.root_path().clone();
391
392        self.task_pool.handle.spawn_with_sender(ThreadIntent::Worker, move |sender| {
393            sender.send(Task::FetchBuildData(BuildDataProgress::Begin)).unwrap();
394
395            let progress = {
396                let sender = sender.clone();
397                move |msg| {
398                    sender.send(Task::FetchBuildData(BuildDataProgress::Report(msg))).unwrap()
399                }
400            };
401            let res = ProjectWorkspace::run_all_build_scripts(
402                &workspaces,
403                &config,
404                &progress,
405                &root_path,
406            );
407
408            sender.send(Task::FetchBuildData(BuildDataProgress::End((workspaces, res)))).unwrap();
409        });
410    }

412    pub(crate) fn fetch_proc_macros(
413        &mut self,
414        cause: Cause,
415        mut change: ChangeWithProcMacros,
416        paths: Vec<ProcMacroPaths>,
417    ) {
418        info!(%cause, "will load proc macros");
419        let ignored_proc_macros = self.config.ignored_proc_macros(None).clone();
420        let proc_macro_clients = self.proc_macro_clients.clone();
421
422        self.task_pool.handle.spawn_with_sender(ThreadIntent::Worker, move |sender| {
423            sender.send(Task::LoadProcMacros(ProcMacroProgress::Begin)).unwrap();
424
425            let ignored_proc_macros = &ignored_proc_macros;
426            let progress = {
427                let sender = sender.clone();
428                &move |msg| {
429                    sender.send(Task::LoadProcMacros(ProcMacroProgress::Report(msg))).unwrap()
430                }
431            };
432
433            let mut builder = ProcMacrosBuilder::default();
434            let proc_macro_clients = proc_macro_clients.iter().chain(iter::repeat(&None));
435            for (client, paths) in proc_macro_clients.zip(paths) {
436                for (crate_id, res) in paths.iter() {
437                    let expansion_res = match client {
438                        Some(Ok(client)) => match res {
439                            Ok((crate_name, path)) => {
440                                progress(format!("loading proc-macros: {path}"));
441                                let ignored_proc_macros = ignored_proc_macros
442                                    .iter()
443                                    .find_map(|(name, macros)| {
444                                        eq_ignore_underscore(name, crate_name).then_some(&**macros)
445                                    })
446                                    .unwrap_or_default();
447
448                                load_proc_macro(client, path, ignored_proc_macros)
449                            }
450                            Err(e) => Err(e.clone()),
451                        },
452                        Some(Err(e)) => Err(ProcMacroLoadingError::ProcMacroSrvError(
453                            e.to_string().into_boxed_str(),
454                        )),
455                        None => Err(ProcMacroLoadingError::ProcMacroSrvError(
456                            "proc-macro-srv is not running".into(),
457                        )),
458                    };
459                    builder.insert(*crate_id, expansion_res)
460                }
461            }
462
463            change.set_proc_macros(builder);
464            sender.send(Task::LoadProcMacros(ProcMacroProgress::End(change))).unwrap();
465        });
466    }

468    pub(crate) fn switch_workspaces(&mut self, cause: Cause) {
469        let _p = tracing::info_span!("GlobalState::switch_workspaces").entered();
470        tracing::info!(%cause, "will switch workspaces");
471
472        let Some(FetchWorkspaceResponse { workspaces, force_crate_graph_reload }) =
473            self.fetch_workspaces_queue.last_op_result()
474        else {
475            return;
476        };
477        let switching_from_empty_workspace = self.workspaces.is_empty();
478
479        info!(%cause, ?force_crate_graph_reload, %switching_from_empty_workspace);
480        if self.fetch_workspace_error().is_err() && !switching_from_empty_workspace {
481            if *force_crate_graph_reload {
482                self.recreate_crate_graph(cause, false);
483            }
484            // It only makes sense to switch to a partially broken workspace
485            // if we don't have any workspace at all yet.
486            return;
487        }
488
489        let workspaces =
490            workspaces.iter().filter_map(|res| res.as_ref().ok().cloned()).collect::<Vec<_>>();
491
492        let same_workspaces = workspaces.len() == self.workspaces.len()
493            && workspaces
494                .iter()
495                .zip(self.workspaces.iter())
496                .all(|(l, r)| l.eq_ignore_build_data(r));
497
498        if same_workspaces {
499            if switching_from_empty_workspace {
500                // Switching from empty to empty is a no-op
501                return;
502            }
503            if let Some(FetchBuildDataResponse { workspaces, build_scripts }) =
504                self.fetch_build_data_queue.last_op_result()
505            {
506                if Arc::ptr_eq(workspaces, &self.workspaces) {
507                    info!("set build scripts to workspaces");
508
509                    let workspaces = workspaces
510                        .iter()
511                        .cloned()
512                        .zip(build_scripts)
513                        .map(|(mut ws, bs)| {
514                            ws.set_build_scripts(bs.as_ref().ok().cloned().unwrap_or_default());
515                            ws
516                        })
517                        .collect::<Vec<_>>();
518                    // Workspaces are the same, but we've updated build data.
519                    info!("same workspace, but new build data");
520                    self.workspaces = Arc::new(workspaces);
521                } else {
522                    info!("build scripts do not match the version of the active workspace");
523                    if *force_crate_graph_reload {
524                        self.recreate_crate_graph(cause, switching_from_empty_workspace);
525                    }
526
527                    // Current build scripts do not match the version of the active
528                    // workspace, so there's nothing for us to update.
529                    return;
530                }
531            } else {
532                if *force_crate_graph_reload {
533                    self.recreate_crate_graph(cause, switching_from_empty_workspace);
534                }
535
536                // No build scripts but unchanged workspaces, nothing to do here
537                return;
538            }
539        } else {
540            info!("abandon build scripts for workspaces");
541
542            // Here, we completely changed the workspace (Cargo.toml edit), so
543            // we don't care about build-script results, they are stale.
544            // FIXME: can we abort the build scripts here if they are already running?
545            self.workspaces = Arc::new(workspaces);
546            self.check_workspaces_msrv().for_each(|message| {
547                self.send_notification::<lsp_types::notification::ShowMessage>(
548                    lsp_types::ShowMessageParams { typ: lsp_types::MessageType::WARNING, message },
549                );
550            });
551
552            if self.config.run_build_scripts(None) {
553                self.build_deps_changed = false;
554                self.fetch_build_data_queue.request_op("workspace updated".to_owned(), ());
555
556                if !switching_from_empty_workspace {
557                    // `switch_workspaces()` will be called again when build scripts already run, which should
558                    // take a short time. If we update the workspace now we will invalidate proc macros and cfgs,
559                    // and then when build scripts complete we will invalidate them again.
560                    return;
561                }
562            }
563        }
564
565        if let FilesWatcher::Client = self.config.files().watcher {
566            let filter = self
567                .workspaces
568                .iter()
569                .flat_map(|ws| ws.to_roots())
570                .filter(|it| it.is_local)
571                .map(|it| it.include);
572
573            let mut watchers: Vec<FileSystemWatcher> =
574                if self.config.did_change_watched_files_relative_pattern_support() {
575                    // When relative patterns are supported by the client, prefer using them
576                    filter
577                        .flat_map(|include| {
578                            include.into_iter().flat_map(|base| {
579                                [
580                                    (base.clone(), "**/*.rs"),
581                                    (base.clone(), "**/Cargo.{lock,toml}"),
582                                    (base, "**/rust-analyzer.toml"),
583                                ]
584                            })
585                        })
586                        .map(|(base, pat)| lsp_types::FileSystemWatcher {
587                            glob_pattern: lsp_types::GlobPattern::Relative(
588                                lsp_types::RelativePattern {
589                                    base_uri: lsp_types::OneOf::Right(
590                                        lsp_types::Url::from_file_path(base).unwrap(),
591                                    ),
592                                    pattern: pat.to_owned(),
593                                },
594                            ),
595                            kind: None,
596                        })
597                        .collect()
598                } else {
599                    // When they're not, integrate the base to make them into absolute patterns
600                    filter
601                        .flat_map(|include| {
602                            include.into_iter().flat_map(|base| {
603                                [
604                                    format!("{base}/**/*.rs"),
605                                    format!("{base}/**/Cargo.{{toml,lock}}"),
606                                    format!("{base}/**/rust-analyzer.toml"),
607                                ]
608                            })
609                        })
610                        .map(|glob_pattern| lsp_types::FileSystemWatcher {
611                            glob_pattern: lsp_types::GlobPattern::String(glob_pattern),
612                            kind: None,
613                        })
614                        .collect()
615                };
616
617            // Also explicitly watch any build files configured in JSON project files.
618            for ws in self.workspaces.iter() {
619                if let ProjectWorkspaceKind::Json(project_json) = &ws.kind {
620                    for (_, krate) in project_json.crates() {
621                        let Some(build) = &krate.build else {
622                            continue;
623                        };
624                        watchers.push(lsp_types::FileSystemWatcher {
625                            glob_pattern: lsp_types::GlobPattern::String(
626                                build.build_file.to_string(),
627                            ),
628                            kind: None,
629                        });
630                    }
631                }
632            }
633
634            watchers.extend(
635                iter::once(Config::user_config_dir_path().as_deref())
636                    .chain(self.workspaces.iter().map(|ws| ws.manifest().map(ManifestPath::as_ref)))
637                    .flatten()
638                    .map(|glob_pattern| lsp_types::FileSystemWatcher {
639                        glob_pattern: lsp_types::GlobPattern::String(glob_pattern.to_string()),
640                        kind: None,
641                    }),
642            );
643
644            let registration_options =
645                lsp_types::DidChangeWatchedFilesRegistrationOptions { watchers };
646            let registration = lsp_types::Registration {
647                id: "workspace/didChangeWatchedFiles".to_owned(),
648                method: "workspace/didChangeWatchedFiles".to_owned(),
649                register_options: Some(serde_json::to_value(registration_options).unwrap()),
650            };
651            self.send_request::<lsp_types::request::RegisterCapability>(
652                lsp_types::RegistrationParams { registrations: vec![registration] },
653                |_, _| (),
654            );
655        }
656
657        let files_config = self.config.files();
658        let project_folders = ProjectFolders::new(
659            &self.workspaces,
660            &files_config.exclude,
661            Config::user_config_dir_path().as_deref(),
662        );
663
664        if (self.proc_macro_clients.len() < self.workspaces.len() || !same_workspaces)
665            && self.config.expand_proc_macros()
666        {
667            info!("Spawning proc-macro servers");
668
669            self.proc_macro_clients = Arc::from_iter(self.workspaces.iter().map(|ws| {
670                let path = match self.config.proc_macro_srv() {
671                    Some(path) => path,
672                    None => match ws.find_sysroot_proc_macro_srv()? {
673                        Ok(path) => path,
674                        Err(e) => return Some(Err(e)),
675                    },
676                };
677
678                let env: FxHashMap<_, _> = match &ws.kind {
679                    ProjectWorkspaceKind::Cargo { cargo, .. }
680                    | ProjectWorkspaceKind::DetachedFile { cargo: Some((cargo, ..)), .. } => cargo
681                        .env()
682                        .into_iter()
683                        .map(|(k, v)| (k.clone(), Some(v.clone())))
684                        .chain(
685                            self.config.extra_env(None).iter().map(|(k, v)| (k.clone(), v.clone())),
686                        )
687                        .chain(
688                            ws.sysroot
689                                .root()
690                                .filter(|_| {
691                                    !self.config.extra_env(None).contains_key("RUSTUP_TOOLCHAIN")
692                                        && std::env::var_os("RUSTUP_TOOLCHAIN").is_none()
693                                })
694                                .map(|it| ("RUSTUP_TOOLCHAIN".to_owned(), Some(it.to_string()))),
695                        )
696                        .collect(),
697
698                    _ => Default::default(),
699                };
700                info!("Using proc-macro server at {path}");
701
702                Some(ProcMacroClient::spawn(&path, &env).map_err(|err| {
703                    tracing::error!(
704                        "Failed to run proc-macro server from path {path}, error: {err:?}",
705                    );
706                    anyhow::format_err!(
707                        "Failed to run proc-macro server from path {path}, error: {err:?}",
708                    )
709                }))
710            }))
711        }
712
713        let watch = match files_config.watcher {
714            FilesWatcher::Client => vec![],
715            FilesWatcher::Server => project_folders.watch,
716        };
717        self.vfs_config_version += 1;
718        self.loader.handle.set_config(vfs::loader::Config {
719            load: project_folders.load,
720            watch,
721            version: self.vfs_config_version,
722        });
723        self.source_root_config = project_folders.source_root_config;
724        self.local_roots_parent_map = Arc::new(self.source_root_config.source_root_parent_map());
725
726        info!(?cause, "recreating the crate graph");
727        self.recreate_crate_graph(cause, switching_from_empty_workspace);
728
729        info!("did switch workspaces");
730    }
731
732    fn recreate_crate_graph(&mut self, cause: String, initial_build: bool) {
733        info!(?cause, "Building Crate Graph");
734        self.report_progress(
735            "Building CrateGraph",
736            crate::lsp::utils::Progress::Begin,
737            None,
738            None,
739            None,
740        );
741
742        // crate graph construction relies on these paths, record them so when one of them gets
743        // deleted or created we trigger a reconstruction of the crate graph
744        self.crate_graph_file_dependencies.clear();
745        self.detached_files = self
746            .workspaces
747            .iter()
748            .filter_map(|ws| match &ws.kind {
749                ProjectWorkspaceKind::DetachedFile { file, .. } => Some(file.clone()),
750                _ => None,
751            })
752            .collect();
753
754        self.incomplete_crate_graph = false;
755        let (crate_graph, proc_macro_paths) = {
756            // Create crate graph from all the workspaces
757            let vfs = &self.vfs.read().0;
758            let load = |path: &AbsPath| {
759                let vfs_path = vfs::VfsPath::from(path.to_path_buf());
760                self.crate_graph_file_dependencies.insert(vfs_path.clone());
761                let file_id = vfs.file_id(&vfs_path);
762                self.incomplete_crate_graph |= file_id.is_none();
763                file_id.and_then(|(file_id, excluded)| {
764                    (excluded == vfs::FileExcluded::No).then_some(file_id)
765                })
766            };
767
768            ws_to_crate_graph(&self.workspaces, self.config.extra_env(None), load)
769        };
770        let mut change = ChangeWithProcMacros::default();
771        if initial_build || !self.config.expand_proc_macros() {
772            if self.config.expand_proc_macros() {
773                change.set_proc_macros(
774                    crate_graph
775                        .iter()
776                        .map(|id| (id, Err(ProcMacroLoadingError::NotYetBuilt)))
777                        .collect(),
778                );
779            } else {
780                change.set_proc_macros(
781                    crate_graph
782                        .iter()
783                        .map(|id| (id, Err(ProcMacroLoadingError::Disabled)))
784                        .collect(),
785                );
786            }
787
788            change.set_crate_graph(crate_graph);
789            self.analysis_host.apply_change(change);
790
791            self.finish_loading_crate_graph();
792        } else {
793            change.set_crate_graph(crate_graph);
794            self.fetch_proc_macros_queue.request_op(cause, (change, proc_macro_paths));
795        }
796
797        self.report_progress(
798            "Building CrateGraph",
799            crate::lsp::utils::Progress::End,
800            None,
801            None,
802            None,
803        );
804    }
805
806    pub(crate) fn finish_loading_crate_graph(&mut self) {
807        self.process_changes();
808        self.reload_flycheck();
809    }

811    pub(super) fn fetch_workspace_error(&self) -> Result<(), String> {
812        let mut buf = String::new();
813
814        let Some(FetchWorkspaceResponse { workspaces, .. }) =
815            self.fetch_workspaces_queue.last_op_result()
816        else {
817            return Ok(());
818        };
819
820        if workspaces.is_empty() && self.config.discover_workspace_config().is_none() {
821            stdx::format_to!(buf, "rust-analyzer failed to fetch workspace");
822        } else {
823            for ws in workspaces {
824                if let Err(err) = ws {
825                    stdx::format_to!(buf, "rust-analyzer failed to load workspace: {:#}\n", err);
826                }
827            }
828        }
829
830        if buf.is_empty() {
831            return Ok(());
832        }
833
834        Err(buf)
835    }

837    pub(super) fn fetch_build_data_error(&self) -> Result<(), String> {
838        let mut buf = String::new();
839
840        let Some(FetchBuildDataResponse { build_scripts, .. }) =
841            &self.fetch_build_data_queue.last_op_result()
842        else {
843            return Ok(());
844        };
845
846        for script in build_scripts {
847            match script {
848                Ok(data) => {
849                    if let Some(stderr) = data.error() {
850                        stdx::format_to!(buf, "{:#}\n", stderr)
851                    }
852                }
853                // io errors
854                Err(err) => stdx::format_to!(buf, "{:#}\n", err),
855            }
856        }
857
858        if buf.is_empty() { Ok(()) } else { Err(buf) }
859    }

861    fn reload_flycheck(&mut self) {
862        let _p = tracing::info_span!("GlobalState::reload_flycheck").entered();
863        let config = self.config.flycheck(None);
864        let sender = &self.flycheck_sender;
865        let invocation_strategy = config.invocation_strategy();
866        let next_gen =
867            self.flycheck.iter().map(FlycheckHandle::generation).max().unwrap_or_default() + 1;
868
869        self.flycheck = match invocation_strategy {
870            crate::flycheck::InvocationStrategy::Once => {
871                vec![FlycheckHandle::spawn(
872                    0,
873                    next_gen,
874                    sender.clone(),
875                    config,
876                    None,
877                    self.config.root_path().clone(),
878                    None,
879                )]
880            }
881            crate::flycheck::InvocationStrategy::PerWorkspace => {
882                self.workspaces
883                    .iter()
884                    .enumerate()
885                    .filter_map(|(id, ws)| {
886                        Some((
887                            id,
888                            match &ws.kind {
889                                ProjectWorkspaceKind::Cargo { cargo, .. }
890                                | ProjectWorkspaceKind::DetachedFile {
891                                    cargo: Some((cargo, _, _)),
892                                    ..
893                                } => (cargo.workspace_root(), Some(cargo.manifest_path())),
894                                ProjectWorkspaceKind::Json(project) => {
895                                    // Enable flychecks for json projects if a custom flycheck command was supplied
896                                    // in the workspace configuration.
897                                    match config {
898                                        FlycheckConfig::CustomCommand { .. } => {
899                                            (project.path(), None)
900                                        }
901                                        _ => return None,
902                                    }
903                                }
904                                ProjectWorkspaceKind::DetachedFile { .. } => return None,
905                            },
906                            ws.sysroot.root().map(ToOwned::to_owned),
907                        ))
908                    })
909                    .map(|(id, (root, manifest_path), sysroot_root)| {
910                        FlycheckHandle::spawn(
911                            id,
912                            next_gen,
913                            sender.clone(),
914                            config.clone(),
915                            sysroot_root,
916                            root.to_path_buf(),
917                            manifest_path.map(|it| it.to_path_buf()),
918                        )
919                    })
920                    .collect()
921            }
922        }
923        .into();
924    }

928pub fn ws_to_crate_graph(
929    workspaces: &[ProjectWorkspace],
930    extra_env: &FxHashMap<String, Option<String>>,
931    mut load: impl FnMut(&AbsPath) -> Option<vfs::FileId>,
932) -> (CrateGraphBuilder, Vec<ProcMacroPaths>) {
933    let mut crate_graph = CrateGraphBuilder::default();
934    let mut proc_macro_paths = Vec::default();
935    for ws in workspaces {
936        let (other, mut crate_proc_macros) = ws.to_crate_graph(&mut load, extra_env);
937
938        crate_graph.extend(other, &mut crate_proc_macros);
939        proc_macro_paths.push(crate_proc_macros);
940    }
941
942    crate_graph.shrink_to_fit();
943    proc_macro_paths.shrink_to_fit();
944    (crate_graph, proc_macro_paths)
945}

947pub(crate) fn should_refresh_for_change(
948    path: &AbsPath,
949    change_kind: ChangeKind,
950    additional_paths: &[&str],
951) -> bool {
952    const IMPLICIT_TARGET_FILES: &[&str] = &["build.rs", "src/main.rs", "src/lib.rs"];
953    const IMPLICIT_TARGET_DIRS: &[&str] = &["src/bin", "examples", "tests", "benches"];
954
955    let file_name = match path.file_name() {
956        Some(it) => it,
957        None => return false,
958    };
959
960    if let "Cargo.toml" | "Cargo.lock" = file_name {
961        return true;
962    }
963
964    if additional_paths.contains(&file_name) {
965        return true;
966    }
967
968    if change_kind == ChangeKind::Modify {
969        return false;
970    }
971
972    // .cargo/config{.toml}
973    if path.extension().unwrap_or_default() != "rs" {
974        let is_cargo_config = matches!(file_name, "config.toml" | "config")
975            && path.parent().map(|parent| parent.as_str().ends_with(".cargo")).unwrap_or(false);
976        return is_cargo_config;
977    }
978
979    if IMPLICIT_TARGET_FILES.iter().any(|it| path.as_str().ends_with(it)) {
980        return true;
981    }
982    let parent = match path.parent() {
983        Some(it) => it,
984        None => return false,
985    };
986    if IMPLICIT_TARGET_DIRS.iter().any(|it| parent.as_str().ends_with(it)) {
987        return true;
988    }
989    if file_name == "main.rs" {
990        let grand_parent = match parent.parent() {
991            Some(it) => it,
992            None => return false,
993        };
994        if IMPLICIT_TARGET_DIRS.iter().any(|it| grand_parent.as_str().ends_with(it)) {
995            return true;
996        }
997    }
998    false
999}

1001/// Similar to [`str::eq_ignore_ascii_case`] but instead of ignoring
1002/// case, we say that `-` and `_` are equal.
1003fn eq_ignore_underscore(s1: &str, s2: &str) -> bool {
1004    if s1.len() != s2.len() {
1005        return false;
1006    }
1007
1008    s1.as_bytes().iter().zip(s2.as_bytes()).all(|(c1, c2)| {
1009        let c1_underscore = c1 == &b'_' || c1 == &b'-';
1010        let c2_underscore = c2 == &b'_' || c2 == &b'-';
1011
1012        c1 == c2 || (c1_underscore && c2_underscore)
1013    })
1014}
