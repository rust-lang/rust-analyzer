77pub(crate) fn handle_analyzer_status(
78    snap: GlobalStateSnapshot,
79    params: lsp_ext::AnalyzerStatusParams,
80) -> anyhow::Result<String> {
81    let _p = tracing::info_span!("handle_analyzer_status").entered();
82
83    let mut buf = String::new();
84
85    let mut file_id = None;
86    if let Some(tdi) = params.text_document {
87        match from_proto::file_id(&snap, &tdi.uri) {
88            Ok(Some(it)) => file_id = Some(it),
89            Ok(None) => {}
90            Err(_) => format_to!(buf, "file {} not found in vfs", tdi.uri),
91        }
92    }
93
94    if snap.workspaces.is_empty() {
95        buf.push_str("No workspaces\n")
96    } else {
97        buf.push_str("Workspaces:\n");
98        format_to!(
99            buf,
100            "Loaded {:?} packages across {} workspace{}.\n",
101            snap.workspaces.iter().map(|w| w.n_packages()).sum::<usize>(),
102            snap.workspaces.len(),
103            if snap.workspaces.len() == 1 { "" } else { "s" }
104        );
105
106        format_to!(
107            buf,
108            "Workspace root folders: {:?}",
109            snap.workspaces.iter().map(|ws| ws.manifest_or_root()).collect::<Vec<&AbsPath>>()
110        );
111    }
112    buf.push_str("\nAnalysis:\n");
113    buf.push_str(
114        &snap
115            .analysis
116            .status(file_id)
117            .unwrap_or_else(|_| "Analysis retrieval was cancelled".to_owned()),
118    );
119
120    buf.push_str("\nVersion: \n");
121    format_to!(buf, "{}", crate::version());
122
123    buf.push_str("\nConfiguration: \n");
124    format_to!(buf, "{:#?}", snap.config);
125
126    Ok(buf)
127}

129pub(crate) fn handle_memory_usage(_state: &mut GlobalState, _: ()) -> anyhow::Result<String> {
130    let _p = tracing::info_span!("handle_memory_usage").entered();
131
132    #[cfg(not(feature = "dhat"))]
133    {
134        Err(anyhow::anyhow!(
135            "Memory profiling is not enabled for this build of rust-analyzer.\n\n\

138        ))
139    }

141    {

146            drop(old_profiler);
147            *profiler = Some(dhat::Profiler::builder().file_name(&dhat_output_file).build());
148            Ok(format!(
149                "Memory profile was saved successfully to {dhat_output_file}.\n\n\
150                See https://docs.rs/dhat/latest/dhat/#viewing for how to inspect the profile."
151            ))
152        } else {
153            Err(anyhow::anyhow!(
154                "Please set `rust-analyzer.profiling.memoryProfile` to the path where you want to save the profile."
155            ))
156        }
157    }

218// cargo test requires:
219// - the package is a member of the workspace
220// - the target in the package is not a build script (custom-build)
221// - the package name - the root of the test identifier supplied to this handler can be
222//   a package or a target inside a package.
223// - the target name - if the test identifier is a target, it's needed in addition to the
224//   package name to run the right test
225// - real names - the test identifier uses the namespace form where hyphens are replaced with
226//   underscores. cargo test requires the real name.
227// - the target kind e.g. bin or lib
228fn all_test_targets(cargo: &CargoWorkspace) -> impl Iterator<Item = TestTarget> {
229    cargo.packages().filter(|p| cargo[*p].is_member).flat_map(|p| {
230        let package = &cargo[p];
231        package.targets.iter().filter_map(|t| {
232            let target = &cargo[*t];
233            if target.kind == TargetKind::BuildScript {
234                None
235            } else {
236                Some(TestTarget {
237                    package: package.name.clone(),
238                    target: target.name.clone(),
239                    kind: target.kind,
240                })
241            }
242        })
243    })
244}

246fn find_test_target(namespace_root: &str, cargo: &CargoWorkspace) -> Option<TestTarget> {
247    all_test_targets(cargo).find(|t| namespace_root == t.target.replace('-', "_"))
248}

250pub(crate) fn handle_run_test(
251    state: &mut GlobalState,
252    params: lsp_ext::RunTestParams,
253) -> anyhow::Result<()> {
254    if let Some(_session) = state.test_run_session.take() {
255        state.send_notification::<lsp_ext::EndRunTest>(());
256    }
257
258    let mut handles = vec![];
259    for ws in &*state.workspaces {
260        if let ProjectWorkspaceKind::Cargo { cargo, .. } = &ws.kind {
261            // need to deduplicate `include` to avoid redundant test runs
262            let tests = match params.include {
263                Some(ref include) => include
264                    .iter()
265                    .unique()
266                    .filter_map(|test| {
267                        let (root, remainder) = match test.split_once("::") {
268                            Some((root, remainder)) => (root.to_owned(), Some(remainder)),
269                            None => (test.clone(), None),
270                        };
271                        if let Some(target) = find_test_target(&root, cargo) {
272                            Some((target, remainder))
273                        } else {
274                            tracing::error!("Test target not found for: {test}");
275                            None
276                        }
277                    })
278                    .collect_vec(),
279                None => all_test_targets(cargo).map(|target| (target, None)).collect(),
280            };
281
282            for (target, path) in tests {
283                let handle = CargoTestHandle::new(
284                    path,
285                    state.config.cargo_test_options(None),
286                    cargo.workspace_root(),
287                    target,
288                    state.test_run_sender.clone(),
289                )?;
290                handles.push(handle);
291            }
292        }
293    }
294    // Each process send finished signal twice, once for stdout and once for stderr
295    state.test_run_remaining_jobs = 2 * handles.len();
296    state.test_run_session = Some(handles);
297    Ok(())
298}

300pub(crate) fn handle_discover_test(
301    snap: GlobalStateSnapshot,
302    params: lsp_ext::DiscoverTestParams,
303) -> anyhow::Result<lsp_ext::DiscoverTestResults> {
304    let _p = tracing::info_span!("handle_discover_test").entered();
305    let (tests, scope) = match params.test_id {
306        Some(id) => {
307            let crate_id = id.split_once("::").map(|it| it.0).unwrap_or(&id);
308            (
309                snap.analysis.discover_tests_in_crate_by_test_id(crate_id)?,
310                Some(vec![crate_id.to_owned()]),
311            )
312        }
313        None => (snap.analysis.discover_test_roots()?, None),
314    };
315
316    Ok(lsp_ext::DiscoverTestResults {
317        tests: tests
318            .into_iter()
319            .filter_map(|t| {
320                let line_index = t.file.and_then(|f| snap.file_line_index(f).ok());
321                to_proto::test_item(&snap, t, line_index.as_ref())
322            })
323            .collect(),
324        scope,
325        scope_file: None,
326    })
327}

351pub(crate) fn handle_selection_range(
352    snap: GlobalStateSnapshot,
353    params: lsp_types::SelectionRangeParams,
354) -> anyhow::Result<Option<Vec<lsp_types::SelectionRange>>> {
355    let _p = tracing::info_span!("handle_selection_range").entered();
356    let file_id = try_default!(from_proto::file_id(&snap, &params.text_document.uri)?);
357    let line_index = snap.file_line_index(file_id)?;
358    let res: anyhow::Result<Vec<lsp_types::SelectionRange>> = params
359        .positions
360        .into_iter()
361        .map(|position| {
362            let offset = from_proto::offset(&line_index, position)?;
363            let mut ranges = Vec::new();
364            {
365                let mut range = TextRange::new(offset, offset);
366                loop {
367                    ranges.push(range);
368                    let frange = FileRange { file_id, range };
369                    let next = snap.analysis.extend_selection(frange)?;
370                    if next == range {
371                        break;
372                    } else {
373                        range = next
374                    }
375                }
376            }
377            let mut range = lsp_types::SelectionRange {
378                range: to_proto::range(&line_index, *ranges.last().unwrap()),
379                parent: None,
380            };
381            for &r in ranges.iter().rev().skip(1) {
382                range = lsp_types::SelectionRange {
383                    range: to_proto::range(&line_index, r),
384                    parent: Some(Box::new(range)),
385                }
386            }
387            Ok(range)
388        })
389        .collect();
390
391    Ok(Some(res?))
392}

435            }
436        }
437    }
438
439    Ok(to_proto::text_edit_vec(&line_index, res))

462pub(crate) fn handle_on_type_formatting(
463    snap: GlobalStateSnapshot,
464    params: lsp_types::DocumentOnTypeFormattingParams,
465) -> anyhow::Result<Option<Vec<lsp_ext::SnippetTextEdit>>> {
466    let _p = tracing::info_span!("handle_on_type_formatting").entered();
467    let char_typed = params.ch.chars().next().unwrap_or('\0');
468    if !snap.config.typing_trigger_chars().contains(char_typed) {
469        return Ok(None);
470    }
471
472    let mut position =
473        try_default!(from_proto::file_position(&snap, params.text_document_position)?);
474    let line_index = snap.file_line_index(position.file_id)?;
475
476    // in `ide`, the `on_type` invariant is that
477    // `text.char_at(position) == typed_char`.
478    position.offset -= TextSize::of('.');
479
480    let text = snap.analysis.file_text(position.file_id)?;
481    if stdx::never!(!text[usize::from(position.offset)..].starts_with(char_typed)) {
482        return Ok(None);
483    }
484
485    let edit = snap.analysis.on_char_typed(position, char_typed)?;
486    let edit = match edit {
487        Some(it) => it,
488        None => return Ok(None),
489    };
490
491    // This should be a single-file edit
492    let (_, (text_edit, snippet_edit)) = edit.source_file_edits.into_iter().next().unwrap();
493    stdx::always!(snippet_edit.is_none(), "on type formatting shouldn't use structured snippets");
494
495    let change = to_proto::snippet_text_edit_vec(
496        &line_index,
497        edit.is_snippet,
498        text_edit,
499        snap.config.change_annotation_support(),
500    );
501    Ok(Some(change))
502}

516pub(crate) fn handle_document_diagnostics(
517    snap: GlobalStateSnapshot,
518    params: lsp_types::DocumentDiagnosticParams,
519) -> anyhow::Result<lsp_types::DocumentDiagnosticReportResult> {
520    let file_id = match from_proto::file_id(&snap, &params.text_document.uri)? {
521        Some(it) => it,
522        None => return Ok(empty_diagnostic_report()),
523    };
524    let source_root = snap.analysis.source_root_id(file_id)?;
525    if !snap.analysis.is_local_source_root(source_root)? {
526        return Ok(empty_diagnostic_report());
527    }
528    let source_root = snap.analysis.source_root_id(file_id)?;
529    let config = snap.config.diagnostics(Some(source_root));
530    if !config.enabled {
531        return Ok(empty_diagnostic_report());
532    }
533    let line_index = snap.file_line_index(file_id)?;
534    let supports_related = snap.config.text_document_diagnostic_related_document_support();
535
536    let mut related_documents = FxHashMap::default();
537    let diagnostics = snap
538        .analysis
539        .full_diagnostics(&config, AssistResolveStrategy::None, file_id)?
540        .into_iter()
541        .filter_map(|d| {
542            let file = d.range.file_id;
543            if file == file_id {
544                let diagnostic = convert_diagnostic(&line_index, d);
545                return Some(diagnostic);
546            }
547            if supports_related {
548                let (diagnostics, line_index) = related_documents
549                    .entry(file)
550                    .or_insert_with(|| (Vec::new(), snap.file_line_index(file).ok()));
551                let diagnostic = convert_diagnostic(line_index.as_mut()?, d);
552                diagnostics.push(diagnostic);
553            }
554            None
555        });
556    Ok(lsp_types::DocumentDiagnosticReportResult::Report(
557        lsp_types::DocumentDiagnosticReport::Full(lsp_types::RelatedFullDocumentDiagnosticReport {
558            full_document_diagnostic_report: lsp_types::FullDocumentDiagnosticReport {
559                result_id: Some("rust-analyzer".to_owned()),
560                items: diagnostics.collect(),
561            },
562            related_documents: related_documents.is_empty().not().then(|| {
563                related_documents
564                    .into_iter()
565                    .map(|(id, (items, _))| {
566                        (
567                            to_proto::url(&snap, id),
568                            lsp_types::DocumentDiagnosticReportKind::Full(
569                                lsp_types::FullDocumentDiagnosticReport {
570                                    result_id: Some("rust-analyzer".to_owned()),
571                                    items,
572                                },
573                            ),
574                        )
575                    })
576                    .collect()
577            }),
578        }),
579    ))
580}

582pub(crate) fn handle_document_symbol(
583    snap: GlobalStateSnapshot,
584    params: lsp_types::DocumentSymbolParams,
585) -> anyhow::Result<Option<lsp_types::DocumentSymbolResponse>> {
586    let _p = tracing::info_span!("handle_document_symbol").entered();
587    let file_id = try_default!(from_proto::file_id(&snap, &params.text_document.uri)?);
588    let line_index = snap.file_line_index(file_id)?;
589
590    let mut symbols: Vec<(lsp_types::DocumentSymbol, Option<usize>)> = Vec::new();
591
592    let config = snap.config.document_symbol(None);
593
594    let structure_nodes = snap.analysis.file_structure(
595        &FileStructureConfig { exclude_locals: config.search_exclude_locals },
596        file_id,
597    )?;
598
599    for node in structure_nodes {
600        let mut tags = Vec::new();
601        if node.deprecated {
602            tags.push(SymbolTag::DEPRECATED)
603        };
604
605        #[allow(deprecated)]
606        let symbol = lsp_types::DocumentSymbol {
607            name: node.label,
608            detail: node.detail,
609            kind: to_proto::structure_node_kind(node.kind),
610            tags: Some(tags),
611            deprecated: Some(node.deprecated),
612            range: to_proto::range(&line_index, node.node_range),
613            selection_range: to_proto::range(&line_index, node.navigation_range),
614            children: None,
615        };
616        symbols.push((symbol, node.parent));
617    }
618
619    // Builds hierarchy from a flat list, in reverse order (so that the indices make sense)
620    let document_symbols = {
621        let mut acc = Vec::new();
622        while let Some((mut symbol, parent_idx)) = symbols.pop() {
623            if let Some(children) = &mut symbol.children {
624                children.reverse();
625            }
626            let parent = match parent_idx {
627                None => &mut acc,
628                Some(i) => symbols[i].0.children.get_or_insert_with(Vec::new),
629            };
630            parent.push(symbol);
631        }
632        acc.reverse();
633        acc
634    };
635
636    let res = if snap.config.hierarchical_symbols() {
637        document_symbols.into()
638    } else {
639        let url = to_proto::url(&snap, file_id);
640        let mut symbol_information = Vec::new();
641        for symbol in document_symbols {
642            flatten_document_symbol(&symbol, None, &url, &mut symbol_information);
643        }
644        symbol_information.into()
645    };
646    return Ok(Some(res));
647
648    fn flatten_document_symbol(
649        symbol: &lsp_types::DocumentSymbol,
650        container_name: Option<String>,
651        url: &Url,
652        res: &mut Vec<SymbolInformation>,
653    ) {
654        #[allow(deprecated)]
655        res.push(SymbolInformation {
656            name: symbol.name.clone(),
657            kind: symbol.kind,
658            tags: symbol.tags.clone(),
659            deprecated: symbol.deprecated,
660            location: Location::new(url.clone(), symbol.range),
661            container_name,
662        });
663
664        for child in symbol.children.iter().flatten() {
665            flatten_document_symbol(child, Some(symbol.name.clone()), url, res);
666        }
667    }
668}

670pub(crate) fn handle_workspace_symbol(
671    snap: GlobalStateSnapshot,
672    params: WorkspaceSymbolParams,
673) -> anyhow::Result<Option<lsp_types::WorkspaceSymbolResponse>> {
674    let _p = tracing::info_span!("handle_workspace_symbol").entered();
675
676    let config = snap.config.workspace_symbol(None);
677    let (all_symbols, libs) = decide_search_kind_and_scope(&params, &config);
678
679    let query = {
680        let query: String = params.query.chars().filter(|&c| c != '#' && c != '*').collect();
681        let mut q = Query::new(query);
682        if !all_symbols {
683            q.only_types();
684        }
685        if libs {
686            q.libs();
687        }
688        if config.search_exclude_imports {
689            q.exclude_imports();
690        }
691        q
692    };
693    let mut res = exec_query(&snap, query, config.search_limit)?;
694    if res.is_empty() && !all_symbols {
695        res = exec_query(&snap, Query::new(params.query), config.search_limit)?;
696    }
697
698    return Ok(Some(lsp_types::WorkspaceSymbolResponse::Nested(res)));
699
700    fn decide_search_kind_and_scope(
701        params: &WorkspaceSymbolParams,
702        config: &WorkspaceSymbolConfig,
703    ) -> (bool, bool) {
704        // Support old-style parsing of markers in the query.
705        let mut all_symbols = params.query.contains('#');
706        let mut libs = params.query.contains('*');
707
708        // If no explicit marker was set, check request params. If that's also empty
709        // use global config.
710        if !all_symbols {
711            let search_kind = match params.search_kind {
712                Some(ref search_kind) => search_kind,
713                None => &config.search_kind,
714            };
715            all_symbols = match search_kind {
716                lsp_ext::WorkspaceSymbolSearchKind::OnlyTypes => false,
717                lsp_ext::WorkspaceSymbolSearchKind::AllSymbols => true,
718            }
719        }
720
721        if !libs {
722            let search_scope = match params.search_scope {
723                Some(ref search_scope) => search_scope,
724                None => &config.search_scope,
725            };
726            libs = match search_scope {
727                lsp_ext::WorkspaceSymbolSearchScope::Workspace => false,
728                lsp_ext::WorkspaceSymbolSearchScope::WorkspaceAndDependencies => true,
729            }
730        }
731
732        (all_symbols, libs)
733    }
734
735    fn exec_query(
736        snap: &GlobalStateSnapshot,
737        query: Query,
738        limit: usize,
739    ) -> anyhow::Result<Vec<lsp_types::WorkspaceSymbol>> {
740        let mut res = Vec::new();
741        for nav in snap.analysis.symbol_search(query, limit)? {
742            let container_name = nav.container_name.as_ref().map(|v| v.to_string());
743
744            let info = lsp_types::WorkspaceSymbol {
745                name: match &nav.alias {
746                    Some(alias) => format!("{} (alias for {})", alias, nav.name),
747                    None => format!("{}", nav.name),
748                },
749                kind: nav
750                    .kind
751                    .map(to_proto::symbol_kind)
752                    .unwrap_or(lsp_types::SymbolKind::VARIABLE),
753                // FIXME: Set deprecation
754                tags: None,
755                container_name,
756                location: lsp_types::OneOf::Left(to_proto::location_from_nav(snap, nav)?),
757                data: None,
758            };
759            res.push(info);
760        }
761        Ok(res)
762    }
763}

765pub(crate) fn handle_will_rename_files(
766    snap: GlobalStateSnapshot,
767    params: lsp_types::RenameFilesParams,
768) -> anyhow::Result<Option<lsp_types::WorkspaceEdit>> {
769    let _p = tracing::info_span!("handle_will_rename_files").entered();
770
771    let source_changes: Vec<SourceChange> = params
772        .files
773        .into_iter()
774        .filter_map(|file_rename| {
775            let from = Url::parse(&file_rename.old_uri).ok()?;
776            let to = Url::parse(&file_rename.new_uri).ok()?;
777
778            let from_path = from.to_file_path().ok()?;
779            let to_path = to.to_file_path().ok()?;
780
781            // Limit to single-level moves for now.
782            match (from_path.parent(), to_path.parent()) {
783                (Some(p1), Some(p2)) if p1 == p2 => {
784                    if from_path.is_dir() {
785                        // add '/' to end of url -- from `file://path/to/folder` to `file://path/to/folder/`
786                        let mut old_folder_name = from_path.file_stem()?.to_str()?.to_owned();
787                        old_folder_name.push('/');
788                        let from_with_trailing_slash = from.join(&old_folder_name).ok()?;
789
790                        let imitate_from_url = from_with_trailing_slash.join("mod.rs").ok()?;
791                        let new_file_name = to_path.file_name()?.to_str()?;
792                        Some((
793                            snap.url_to_file_id(&imitate_from_url).ok()?,
794                            new_file_name.to_owned(),
795                        ))
796                    } else {
797                        let old_name = from_path.file_stem()?.to_str()?;
798                        let new_name = to_path.file_stem()?.to_str()?;
799                        match (old_name, new_name) {
800                            ("mod", _) => None,
801                            (_, "mod") => None,
802                            _ => Some((snap.url_to_file_id(&from).ok()?, new_name.to_owned())),
803                        }
804                    }
805                }
806                _ => None,
807            }
808        })
809        .filter_map(|(file_id, new_name)| {
810            snap.analysis.will_rename_file(file_id?, &new_name).ok()?
811        })
812        .collect();
813
814    // Drop file system edits since we're just renaming things on the same level
815    let mut source_changes = source_changes.into_iter();
816    let mut source_change = source_changes.next().unwrap_or_default();
817    source_change.file_system_edits.clear();
818    // no collect here because we want to merge text edits on same file ids
819    source_change.extend(source_changes.flat_map(|it| it.source_file_edits));
820    if source_change.source_file_edits.is_empty() {
821        Ok(None)
822    } else {
823        Ok(Some(to_proto::workspace_edit(&snap, source_change)?))
824    }
825}

896pub(crate) fn handle_parent_module(
897    snap: GlobalStateSnapshot,
898    params: lsp_types::TextDocumentPositionParams,
899) -> anyhow::Result<Option<lsp_types::GotoDefinitionResponse>> {
900    let _p = tracing::info_span!("handle_parent_module").entered();
901    if let Ok(file_path) = &params.text_document.uri.to_file_path() {
902        if file_path.file_name().unwrap_or_default() == "Cargo.toml" {
903            // search workspaces for parent packages or fallback to workspace root
904            let abs_path_buf = match Utf8PathBuf::from_path_buf(file_path.to_path_buf())
905                .ok()
906                .map(AbsPathBuf::try_from)
907            {
908                Some(Ok(abs_path_buf)) => abs_path_buf,
909                _ => return Ok(None),
910            };
911
912            let manifest_path = match ManifestPath::try_from(abs_path_buf).ok() {
913                Some(manifest_path) => manifest_path,
914                None => return Ok(None),
915            };
916
917            let links: Vec<LocationLink> = snap
918                .workspaces
919                .iter()
920                .filter_map(|ws| match &ws.kind {
921                    ProjectWorkspaceKind::Cargo { cargo, .. }
922                    | ProjectWorkspaceKind::DetachedFile { cargo: Some((cargo, _, _)), .. } => {
923                        cargo.parent_manifests(&manifest_path)
924                    }
925                    _ => None,
926                })
927                .flatten()
928                .map(|parent_manifest_path| LocationLink {
929                    origin_selection_range: None,
930                    target_uri: to_proto::url_from_abs_path(&parent_manifest_path),
931                    target_range: Range::default(),
932                    target_selection_range: Range::default(),
933                })
934                .collect::<_>();
935            return Ok(Some(links.into()));
936        }
937
938        // check if invoked at the crate root
939        let file_id = try_default!(from_proto::file_id(&snap, &params.text_document.uri)?);
940        let crate_id = match snap.analysis.crates_for(file_id)?.first() {
941            Some(&crate_id) => crate_id,
942            None => return Ok(None),
943        };
944        let cargo_spec = match TargetSpec::for_file(&snap, file_id)? {
945            Some(TargetSpec::Cargo(it)) => it,
946            Some(TargetSpec::ProjectJson(_)) | None => return Ok(None),
947        };
948
949        if snap.analysis.crate_root(crate_id)? == file_id {
950            let cargo_toml_url = to_proto::url_from_abs_path(&cargo_spec.cargo_toml);
951            let res = vec![LocationLink {
952                origin_selection_range: None,
953                target_uri: cargo_toml_url,
954                target_range: Range::default(),
955                target_selection_range: Range::default(),
956            }]
957            .into();
958            return Ok(Some(res));
959        }
960    }
961
962    // locate parent module by semantics
963    let position = try_default!(from_proto::file_position(&snap, params)?);
964    let navs = snap.analysis.parent_module(position)?;
965    let res = to_proto::goto_definition_response(&snap, None, navs)?;
966    Ok(Some(res))
967}

981pub(crate) fn handle_runnables(
982    snap: GlobalStateSnapshot,
983    params: lsp_ext::RunnablesParams,
984) -> anyhow::Result<Vec<lsp_ext::Runnable>> {
985    let _p = tracing::info_span!("handle_runnables").entered();
986    let file_id = try_default!(from_proto::file_id(&snap, &params.text_document.uri)?);
987    let source_root = snap.analysis.source_root_id(file_id).ok();
988    let line_index = snap.file_line_index(file_id)?;
989    let offset = params.position.and_then(|it| from_proto::offset(&line_index, it).ok());
990    let target_spec = TargetSpec::for_file(&snap, file_id)?;
991
992    let mut res = Vec::new();
993    for runnable in snap.analysis.runnables(file_id)? {
994        if should_skip_for_offset(&runnable, offset)
995            || should_skip_target(&runnable, target_spec.as_ref())
996        {
997            continue;
998        }
999
1000        let update_test = runnable.update_test;
1001        if let Some(mut runnable) = to_proto::runnable(&snap, runnable)? {
1002            if let Some(runnable) = to_proto::make_update_runnable(&runnable, update_test) {
1003                res.push(runnable);
1004            }
1005
1006            if let lsp_ext::RunnableArgs::Cargo(r) = &mut runnable.args
1007                && let Some(TargetSpec::Cargo(CargoTargetSpec {
1008                    sysroot_root: Some(sysroot_root),
1009                    ..
1010                })) = &target_spec
1011            {
1012                r.environment.insert("RUSTC_TOOLCHAIN".to_owned(), sysroot_root.to_string());
1013            };
1014
1015            res.push(runnable);
1016        }
1017    }
1018
1019    // Add `cargo check` and `cargo test` for all targets of the whole package
1020    let config = snap.config.runnables(source_root);
1021    match target_spec {
1022        Some(TargetSpec::Cargo(spec)) => {
1023            let is_crate_no_std = snap.analysis.is_crate_no_std(spec.crate_id)?;
1024            for cmd in ["check", "run", "test"] {
1025                if cmd == "run" && spec.target_kind != TargetKind::Bin {
1026                    continue;
1027                }
1028                let cwd = if cmd != "test" || spec.target_kind == TargetKind::Bin {
1029                    spec.workspace_root.clone()
1030                } else {
1031                    spec.cargo_toml.parent().to_path_buf()
1032                };
1033                let mut cargo_args =
1034                    vec![cmd.to_owned(), "--package".to_owned(), spec.package.clone()];
1035                let all_targets = cmd != "run" && !is_crate_no_std;
1036                if all_targets {
1037                    cargo_args.push("--all-targets".to_owned());
1038                }
1039                cargo_args.extend(config.cargo_extra_args.iter().cloned());
1040                res.push(lsp_ext::Runnable {
1041                    label: format!(
1042                        "cargo {cmd} -p {}{all_targets}",
1043                        spec.package,
1044                        all_targets = if all_targets { " --all-targets" } else { "" }
1045                    ),
1046                    location: None,
1047                    kind: lsp_ext::RunnableKind::Cargo,
1048                    args: lsp_ext::RunnableArgs::Cargo(lsp_ext::CargoRunnableArgs {
1049                        workspace_root: Some(spec.workspace_root.clone().into()),
1050                        cwd: cwd.into(),
1051                        override_cargo: config.override_cargo.clone(),
1052                        cargo_args,
1053                        executable_args: Vec::new(),
1054                        environment: spec
1055                            .sysroot_root
1056                            .as_ref()
1057                            .map(|root| ("RUSTC_TOOLCHAIN".to_owned(), root.to_string()))
1058                            .into_iter()
1059                            .collect(),
1060                    }),
1061                })
1062            }
1063        }
1064        Some(TargetSpec::ProjectJson(_)) => {}
1065        None => {
1066            if !snap.config.linked_or_discovered_projects().is_empty()
1067                && let Some(path) = snap.file_id_to_file_path(file_id).parent()
1068            {
1069                let mut cargo_args = vec!["check".to_owned(), "--workspace".to_owned()];
1070                cargo_args.extend(config.cargo_extra_args.iter().cloned());
1071                res.push(lsp_ext::Runnable {
1072                    label: "cargo check --workspace".to_owned(),
1073                    location: None,
1074                    kind: lsp_ext::RunnableKind::Cargo,
1075                    args: lsp_ext::RunnableArgs::Cargo(lsp_ext::CargoRunnableArgs {
1076                        workspace_root: None,
1077                        cwd: path.as_path().unwrap().to_path_buf().into(),
1078                        override_cargo: config.override_cargo,
1079                        cargo_args,
1080                        executable_args: Vec::new(),
1081                        environment: Default::default(),
1082                    }),
1083                });
1084            };
1085        }
1086    }
1087    Ok(res)
1088}

1158pub(crate) fn handle_completion_resolve(
1159    snap: GlobalStateSnapshot,
1160    mut original_completion: CompletionItem,
1161) -> anyhow::Result<CompletionItem> {
1162    let _p = tracing::info_span!("handle_completion_resolve").entered();
1163
1164    if !all_edits_are_disjoint(&original_completion, &[]) {
1165        return Err(invalid_params_error(
1166            "Received a completion with overlapping edits, this is not LSP-compliant".to_owned(),
1167        )
1168        .into());
1169    }
1170
1171    let Some(data) = original_completion.data.take() else {
1172        return Ok(original_completion);
1173    };
1174
1175    let resolve_data: lsp_ext::CompletionResolveData = serde_json::from_value(data)?;
1176
1177    let file_id = from_proto::file_id(&snap, &resolve_data.position.text_document.uri)?
1178        .expect("we never provide completions for excluded files");
1179    let line_index = snap.file_line_index(file_id)?;
1180    // FIXME: We should fix up the position when retrying the cancelled request instead
1181    let Ok(offset) = from_proto::offset(&line_index, resolve_data.position.position) else {
1182        return Ok(original_completion);
1183    };
1184    let source_root = snap.analysis.source_root_id(file_id)?;
1185
1186    let mut forced_resolve_completions_config =
1187        snap.config.completion(Some(source_root), snap.minicore());
1188    forced_resolve_completions_config.fields_to_resolve = CompletionFieldsToResolve::empty();
1189
1190    let position = FilePosition { file_id, offset };
1191    let Some(completions) = snap.analysis.completions(
1192        &forced_resolve_completions_config,
1193        position,
1194        resolve_data.trigger_character,
1195    )?
1196    else {
1197        return Ok(original_completion);
1198    };
1199    let Ok(resolve_data_hash) = BASE64_STANDARD.decode(resolve_data.hash) else {
1200        return Ok(original_completion);
1201    };
1202
1203    let Some(corresponding_completion) = completions.into_iter().find(|completion_item| {
1204        // Avoid computing hashes for items that obviously do not match
1205        // r-a might append a detail-based suffix to the label, so we cannot check for equality
1206        original_completion.label.starts_with(completion_item.label.primary.as_str())
1207            && resolve_data_hash == completion_item_hash(completion_item, resolve_data.for_ref)
1208    }) else {
1209        return Ok(original_completion);
1210    };
1211
1212    let mut resolved_completions = to_proto::completion_items(
1213        &snap.config,
1214        &forced_resolve_completions_config.fields_to_resolve,
1215        &line_index,
1216        snap.file_version(position.file_id),
1217        resolve_data.position,
1218        resolve_data.trigger_character,
1219        vec![corresponding_completion],
1220    );
1221    let Some(mut resolved_completion) = resolved_completions.pop() else {
1222        return Ok(original_completion);
1223    };
1224
1225    if !resolve_data.imports.is_empty() {
1226        let additional_edits = snap
1227            .analysis
1228            .resolve_completion_edits(
1229                &forced_resolve_completions_config,
1230                position,
1231                resolve_data.imports.into_iter().map(|import| import.full_import_path),
1232            )?
1233            .into_iter()
1234            .flat_map(|edit| edit.into_iter().map(|indel| to_proto::text_edit(&line_index, indel)))
1235            .collect::<Vec<_>>();
1236
1237        if !all_edits_are_disjoint(&resolved_completion, &additional_edits) {
1238            return Err(LspError::new(
1239                ErrorCode::InternalError as i32,
1240                "Import edit overlaps with the original completion edits, this is not LSP-compliant"
1241                    .into(),
1242            )
1243            .into());
1244        }
1245
1246        if let Some(original_additional_edits) = resolved_completion.additional_text_edits.as_mut()
1247        {
1248            original_additional_edits.extend(additional_edits)
1249        } else {
1250            resolved_completion.additional_text_edits = Some(additional_edits);
1251        }
1252    }
1253
1254    Ok(resolved_completion)
1255}

1343    snap: GlobalStateSnapshot,
1344    params: RenameParams,
1345) -> anyhow::Result<Option<WorkspaceEdit>> {
1346    let _p = tracing::info_span!("handle_rename").entered();
1347    let position = try_default!(from_proto::file_position(&snap, params.text_document_position)?);
1348
1349    let source_root = snap.analysis.source_root_id(position.file_id).ok();

1352        .analysis
1353        .rename(position, &params.new_name, &config)?
1354        .map_err(to_proto::rename_error)?;

1357    // When a module gets renamed by renaming the mod declaration this causes the file to move

1363        change.source_file_edits.clear();

1365
1366    let workspace_edit = to_proto::workspace_edit(&snap, change)?;
1367
1368    if let Some(lsp_types::DocumentChanges::Operations(ops)) =
1369        workspace_edit.document_changes.as_ref()
1370    {

1372            if let lsp_types::DocumentChangeOperation::Op(doc_change_op) = op {
1373                resource_ops_supported(&snap.config, resolve_resource_op(doc_change_op))?

1448pub(crate) fn handle_code_action(
1449    snap: GlobalStateSnapshot,
1450    params: lsp_types::CodeActionParams,
1451) -> anyhow::Result<Option<Vec<lsp_ext::CodeAction>>> {
1452    let _p = tracing::info_span!("handle_code_action").entered();
1453
1454    if !snap.config.code_action_literals() {
1455        // We intentionally don't support command-based actions, as those either
1456        // require either custom client-code or server-initiated edits. Server
1457        // initiated edits break causality, so we avoid those.
1458        return Ok(None);
1459    }
1460
1461    let file_id = try_default!(from_proto::file_id(&snap, &params.text_document.uri)?);
1462    let line_index = snap.file_line_index(file_id)?;
1463    let frange = try_default!(from_proto::file_range(&snap, &params.text_document, params.range)?);
1464    let source_root = snap.analysis.source_root_id(file_id)?;
1465
1466    let mut assists_config = snap.config.assist(Some(source_root));
1467    assists_config.allowed = params
1468        .context
1469        .only
1470        .clone()
1471        .map(|it| it.into_iter().filter_map(from_proto::assist_kind).collect());
1472
1473    let mut res: Vec<lsp_ext::CodeAction> = Vec::new();
1474
1475    let code_action_resolve_cap = snap.config.code_action_resolve();
1476    let resolve = if code_action_resolve_cap {
1477        AssistResolveStrategy::None
1478    } else {
1479        AssistResolveStrategy::All
1480    };
1481    let assists = snap.analysis.assists_with_fixes(
1482        &assists_config,
1483        &snap.config.diagnostic_fixes(Some(source_root)),
1484        resolve,
1485        frange,
1486    )?;
1487    let client_commands = snap.config.client_commands();
1488    for (index, assist) in assists.into_iter().enumerate() {
1489        let resolve_data = if code_action_resolve_cap {
1490            Some((index, params.clone(), snap.file_version(file_id)))
1491        } else {
1492            None
1493        };
1494        let code_action = to_proto::code_action(&snap, &client_commands, assist, resolve_data)?;
1495
1496        // Check if the client supports the necessary `ResourceOperation`s.
1497        let changes = code_action.edit.as_ref().and_then(|it| it.document_changes.as_ref());
1498        if let Some(changes) = changes {
1499            for change in changes {
1500                if let lsp_ext::SnippetDocumentChangeOperation::Op(res_op) = change {
1501                    resource_ops_supported(&snap.config, resolve_resource_op(res_op))?
1502                }
1503            }
1504        }
1505
1506        res.push(code_action)
1507    }
1508
1509    // Fixes from `cargo check`.
1510    for fix in snap
1511        .check_fixes
1512        .iter()
1513        .flat_map(|it| it.values())
1514        .filter_map(|it| it.get(&frange.file_id))
1515        .flatten()
1516    {
1517        // FIXME: this mapping is awkward and shouldn't exist. Refactor
1518        // `snap.check_fixes` to not convert to LSP prematurely.
1519        let intersect_fix_range = fix
1520            .ranges
1521            .iter()
1522            .copied()
1523            .filter_map(|range| from_proto::text_range(&line_index, range).ok())
1524            .any(|fix_range| fix_range.intersect(frange.range).is_some());
1525        if intersect_fix_range {
1526            res.push(fix.action.clone());
1527        }
1528    }
1529
1530    Ok(Some(res))
1531}

1533pub(crate) fn handle_code_action_resolve(
1534    snap: GlobalStateSnapshot,
1535    mut code_action: lsp_ext::CodeAction,
1536) -> anyhow::Result<lsp_ext::CodeAction> {
1537    let _p = tracing::info_span!("handle_code_action_resolve").entered();
1538    let Some(params) = code_action.data.take() else {
1539        return Ok(code_action);
1540    };
1541
1542    let file_id = from_proto::file_id(&snap, &params.code_action_params.text_document.uri)?
1543        .expect("we never provide code actions for excluded files");
1544    if snap.file_version(file_id) != params.version {
1545        return Err(invalid_params_error("stale code action".to_owned()).into());
1546    }
1547    let line_index = snap.file_line_index(file_id)?;
1548    let range = from_proto::text_range(&line_index, params.code_action_params.range)?;
1549    let frange = FileRange { file_id, range };
1550    let source_root = snap.analysis.source_root_id(file_id)?;
1551
1552    let mut assists_config = snap.config.assist(Some(source_root));
1553    assists_config.allowed = params
1554        .code_action_params
1555        .context
1556        .only
1557        .map(|it| it.into_iter().filter_map(from_proto::assist_kind).collect());
1558
1559    let (assist_index, assist_resolve) = match parse_action_id(&params.id) {
1560        Ok(parsed_data) => parsed_data,
1561        Err(e) => {
1562            return Err(invalid_params_error(format!(
1563                "Failed to parse action id string '{}': {e}",
1564                params.id
1565            ))
1566            .into());
1567        }
1568    };
1569
1570    let expected_assist_id = assist_resolve.assist_id.clone();
1571    let expected_kind = assist_resolve.assist_kind;
1572
1573    let assists = snap.analysis.assists_with_fixes(
1574        &assists_config,
1575        &snap.config.diagnostic_fixes(Some(source_root)),
1576        AssistResolveStrategy::Single(assist_resolve),
1577        frange,
1578    )?;
1579
1580    let assist = match assists.get(assist_index) {
1581        Some(assist) => assist,
1582        None => return Err(invalid_params_error(format!(
1583            "Failed to find the assist for index {} provided by the resolve request. Resolve request assist id: {}",
1584            assist_index, params.id,
1585        ))
1586        .into())
1587    };
1588    if assist.id.0 != expected_assist_id || assist.id.1 != expected_kind {
1589        return Err(invalid_params_error(format!(
1590            "Mismatching assist at index {} for the resolve parameters given. Resolve request assist id: {}, actual id: {:?}.",
1591            assist_index, params.id, assist.id
1592        ))
1593        .into());
1594    }
1595    let ca = to_proto::code_action(&snap, &snap.config.client_commands(), assist.clone(), None)?;
1596    code_action.edit = ca.edit;
1597    code_action.command = ca.command;
1598
1599    if let Some(edit) = code_action.edit.as_ref()
1600        && let Some(changes) = edit.document_changes.as_ref()
1601    {
1602        for change in changes {
1603            if let lsp_ext::SnippetDocumentChangeOperation::Op(res_op) = change {
1604                resource_ops_supported(&snap.config, resolve_resource_op(res_op))?
1605            }
1606        }
1607    }
1608
1609    Ok(code_action)
1610}

1612fn parse_action_id(action_id: &str) -> anyhow::Result<(usize, SingleResolve), String> {
1613    let id_parts = action_id.split(':').collect::<Vec<_>>();
1614    match id_parts.as_slice() {
1615        [assist_id_string, assist_kind_string, index_string, subtype_str] => {
1616            let assist_kind: AssistKind = assist_kind_string.parse()?;
1617            let index: usize = match index_string.parse() {
1618                Ok(index) => index,
1619                Err(e) => return Err(format!("Incorrect index string: {e}")),
1620            };
1621            let assist_subtype = subtype_str.parse::<usize>().ok();
1622            Ok((
1623                index,
1624                SingleResolve {
1625                    assist_id: assist_id_string.to_string(),
1626                    assist_kind,
1627                    assist_subtype,
1628                },
1629            ))
1630        }
1631        _ => Err("Action id contains incorrect number of segments".to_owned()),
1632    }
1633}

1786pub(crate) fn handle_inlay_hints_resolve(
1787    snap: GlobalStateSnapshot,
1788    mut original_hint: InlayHint,
1789) -> anyhow::Result<InlayHint> {
1790    let _p = tracing::info_span!("handle_inlay_hints_resolve").entered();
1791
1792    let Some(data) = original_hint.data.take() else {
1793        return Ok(original_hint);
1794    };
1795    let resolve_data: lsp_ext::InlayHintResolveData = serde_json::from_value(data)?;
1796    let file_id = FileId::from_raw(resolve_data.file_id);
1797    if resolve_data.version != snap.file_version(file_id) {
1798        tracing::warn!("Inlay hint resolve data is outdated");
1799        return Ok(original_hint);
1800    }
1801    let Some(hash) = resolve_data.hash.parse().ok() else {
1802        return Ok(original_hint);
1803    };
1804    anyhow::ensure!(snap.file_exists(file_id), "Invalid LSP resolve data");
1805
1806    let line_index = snap.file_line_index(file_id)?;
1807    let range = from_proto::text_range(&line_index, resolve_data.resolve_range)?;
1808
1809    let mut forced_resolve_inlay_hints_config = snap.config.inlay_hints(snap.minicore());
1810    forced_resolve_inlay_hints_config.fields_to_resolve = InlayFieldsToResolve::empty();
1811    let resolve_hints = snap.analysis.inlay_hints_resolve(
1812        &forced_resolve_inlay_hints_config,
1813        file_id,
1814        range,
1815        hash,
1816        |hint| {
1817            std::hash::BuildHasher::hash_one(
1818                &std::hash::BuildHasherDefault::<ide_db::FxHasher>::default(),
1819                hint,
1820            )
1821        },
1822    )?;
1823
1824    Ok(resolve_hints
1825        .and_then(|it| {
1826            to_proto::inlay_hint(
1827                &snap,
1828                &forced_resolve_inlay_hints_config.fields_to_resolve,
1829                &line_index,
1830                file_id,
1831                it,
1832            )
1833            .ok()
1834        })
1835        .filter(|hint| hint.position == original_hint.position)
1836        .filter(|hint| hint.kind == original_hint.kind)
1837        .unwrap_or(original_hint))
1838}

2334                ),

2336            }
2337        }
2338        _ => false,

2342fn run_rustfmt(
2343    snap: &GlobalStateSnapshot,
2344    text_document: TextDocumentIdentifier,
2345    range: Option<lsp_types::Range>,
2346) -> anyhow::Result<Option<Vec<lsp_types::TextEdit>>> {
2347    let file_id = try_default!(from_proto::file_id(snap, &text_document.uri)?);
2348    let file = snap.analysis.file_text(file_id)?;
2349
2350    // Determine the edition of the crate the file belongs to (if there's multiple, we pick the
2351    // highest edition).
2352    let Ok(editions) = snap
2353        .analysis
2354        .relevant_crates_for(file_id)?
2355        .into_iter()
2356        .map(|crate_id| snap.analysis.crate_edition(crate_id))
2357        .collect::<Result<Vec<_>, _>>()
2358    else {
2359        return Ok(None);
2360    };
2361    let edition = editions.iter().copied().max();
2362
2363    let line_index = snap.file_line_index(file_id)?;
2364    let source_root_id = snap.analysis.source_root_id(file_id).ok();
2365
2366    // try to chdir to the file so we can respect `rustfmt.toml`
2367    // FIXME: use `rustfmt --config-path` once
2368    // https://github.com/rust-lang/rustfmt/issues/4660 gets fixed
2369    let current_dir = match text_document.uri.to_file_path() {
2370        Ok(mut path) => {
2371            // pop off file name
2372            if path.pop() && path.is_dir() { path } else { std::env::current_dir()? }
2373        }
2374        Err(_) => {
2375            tracing::error!(
2376                text_document = ?text_document.uri,
2377                "Unable to get path, rustfmt.toml might be ignored"
2378            );
2379            std::env::current_dir()?
2380        }
2381    };
2382
2383    let mut command = match snap.config.rustfmt(source_root_id) {
2384        RustfmtConfig::Rustfmt { extra_args, enable_range_formatting } => {
2385            // FIXME: Set RUSTUP_TOOLCHAIN
2386            let mut cmd = toolchain::command(
2387                toolchain::Tool::Rustfmt.path(),
2388                current_dir,
2389                snap.config.extra_env(source_root_id),
2390            );
2391            cmd.args(extra_args);
2392
2393            if let Some(edition) = edition {
2394                cmd.arg("--edition");
2395                cmd.arg(edition.to_string());
2396            }
2397
2398            if let Some(range) = range {
2399                if !enable_range_formatting {
2400                    return Err(LspError::new(
2401                        ErrorCode::InvalidRequest as i32,
2402                        String::from(
2403                            "rustfmt range formatting is unstable. \
2404                            Opt-in by using a nightly build of rustfmt and setting \
2405                            `rustfmt.rangeFormatting.enable` to true in your LSP configuration",
2406                        ),
2407                    )
2408                    .into());
2409                }
2410
2411                let frange = try_default!(from_proto::file_range(snap, &text_document, range)?);
2412                let start_line = line_index.index.line_col(frange.range.start()).line;
2413                let end_line = line_index.index.line_col(frange.range.end()).line;
2414
2415                cmd.arg("--unstable-features");
2416                cmd.arg("--file-lines");
2417                cmd.arg(
2418                    json!([{
2419                        "file": "stdin",
2420                        // LineCol is 0-based, but rustfmt is 1-based.
2421                        "range": [start_line + 1, end_line + 1]
2422                    }])
2423                    .to_string(),
2424                );
2425            }
2426
2427            cmd
2428        }
2429        RustfmtConfig::CustomCommand { command, args } => {
2430            let cmd = Utf8PathBuf::from(&command);
2431            let target_spec = TargetSpec::for_file(snap, file_id)?;
2432            let extra_env = snap.config.extra_env(source_root_id);
2433            let mut cmd = match target_spec {
2434                Some(TargetSpec::Cargo(_)) => {
2435                    // approach: if the command name contains a path separator, join it with the project root.
2436                    // however, if the path is absolute, joining will result in the absolute path being preserved.
2437                    // as a fallback, rely on $PATH-based discovery.
2438                    let cmd_path = if command.contains(std::path::MAIN_SEPARATOR)
2439                        || (cfg!(windows) && command.contains('/'))
2440                    {
2441                        snap.config.root_path().join(cmd).into()
2442                    } else {
2443                        cmd
2444                    };
2445                    toolchain::command(cmd_path, current_dir, extra_env)
2446                }
2447                _ => toolchain::command(cmd, current_dir, extra_env),
2448            };
2449
2450            cmd.args(args);
2451            cmd
2452        }
2453    };
2454
2455    let output = {
2456        let _p = tracing::info_span!("rustfmt", ?command).entered();
2457
2458        let mut rustfmt = command
2459            .stdin(Stdio::piped())
2460            .stdout(Stdio::piped())
2461            .stderr(Stdio::piped())
2462            .spawn()
2463            .context(format!("Failed to spawn {command:?}"))?;
2464
2465        rustfmt.stdin.as_mut().unwrap().write_all(file.as_bytes())?;
2466
2467        rustfmt.wait_with_output()?
2468    };
2469
2470    let captured_stdout = String::from_utf8(output.stdout)?;
2471    let captured_stderr = String::from_utf8(output.stderr).unwrap_or_default();
2472
2473    if !output.status.success() {
2474        let rustfmt_not_installed =
2475            captured_stderr.contains("not installed") || captured_stderr.contains("not available");
2476
2477        return match output.status.code() {
2478            Some(1) if !rustfmt_not_installed => {
2479                // While `rustfmt` doesn't have a specific exit code for parse errors this is the
2480                // likely cause exiting with 1. Most Language Servers swallow parse errors on
2481                // formatting because otherwise an error is surfaced to the user on top of the
2482                // syntax error diagnostics they're already receiving. This is especially jarring
2483                // if they have format on save enabled.
2484                tracing::warn!(
2485                    ?command,
2486                    %captured_stderr,
2487                    "rustfmt exited with status 1"
2488                );
2489                Ok(None)
2490            }
2491            // rustfmt panicked at lexing/parsing the file
2492            Some(101)
2493                if !rustfmt_not_installed
2494                    && (captured_stderr.starts_with("error[")
2495                        || captured_stderr.starts_with("error:")) =>
2496            {
2497                Ok(None)
2498            }
2499            _ => {
2500                // Something else happened - e.g. `rustfmt` is missing or caught a signal
2501                tracing::error!(
2502                    ?command,
2503                    %output.status,
2504                    %captured_stdout,
2505                    %captured_stderr,
2506                    "rustfmt failed"
2507                );
2508                Ok(None)
2509            }
2510        };
2511    }
2512
2513    let (new_text, new_line_endings) = LineEndings::normalize(captured_stdout);
2514
2515    if line_index.endings != new_line_endings {
2516        // If line endings are different, send the entire file.
2517        // Diffing would not work here, as the line endings might be the only
2518        // difference.
2519        Ok(Some(to_proto::text_edit_vec(
2520            &line_index,
2521            TextEdit::replace(TextRange::up_to(TextSize::of(&*file)), new_text),
2522        )))
2523    } else if *file == new_text {
2524        // The document is already formatted correctly -- no edits needed.
2525        Ok(None)
2526    } else {
2527        Ok(Some(to_proto::text_edit_vec(&line_index, diff(&file, &new_text))))
2528    }
2529}

2577/// Searches for the directory of a Rust crate given this crate's root file path.
2578///
2579/// # Arguments
2580///
2581/// * `root_file_path`: The path to the root file of the crate.
2582///
2583/// # Returns
2584///
2585/// An `Option` value representing the path to the directory of the crate with the given
2586/// name, if such a crate is found. If no crate with the given name is found, this function
2587/// returns `None`.
2588fn crate_path(root_file_path: &VfsPath) -> Option<VfsPath> {
2589    let mut current_dir = root_file_path.parent();
2590    while let Some(path) = current_dir {
2591        let cargo_toml_path = path.join("../Cargo.toml")?;
2592        if fs::metadata(cargo_toml_path.as_path()?).is_ok() {
2593            let crate_path = cargo_toml_path.parent()?;
2594            return Some(crate_path);
2595        }
2596        current_dir = path.parent();
2597    }
2598    None
2599}

2635pub(crate) fn diff(left: &str, right: &str) -> TextEdit {
2636    use dissimilar::Chunk;
2637
2638    let chunks = dissimilar::diff(left, right);
2639
2640    let mut builder = TextEdit::builder();
2641    let mut pos = TextSize::default();
2642
2643    let mut chunks = chunks.into_iter().peekable();
2644    while let Some(chunk) = chunks.next() {
2645        if let (Chunk::Delete(deleted), Some(&Chunk::Insert(inserted))) = (chunk, chunks.peek()) {
2646            chunks.next().unwrap();
2647            let deleted_len = TextSize::of(deleted);
2648            builder.replace(TextRange::at(pos, deleted_len), inserted.into());
2649            pos += deleted_len;
2650            continue;
2651        }
2652
2653        match chunk {
2654            Chunk::Equal(text) => {
2655                pos += TextSize::of(text);
2656            }
2657            Chunk::Delete(deleted) => {
2658                let deleted_len = TextSize::of(deleted);
2659                builder.delete(TextRange::at(pos, deleted_len));
2660                pos += deleted_len;
2661            }
2662            Chunk::Insert(inserted) => {
2663                builder.insert(pos, inserted.into());
2664            }
2665        }
2666    }
2667    builder.finish()
2668}
