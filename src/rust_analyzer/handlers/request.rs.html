74pub(crate) fn handle_analyzer_status(
75    snap: GlobalStateSnapshot,
76    params: lsp_ext::AnalyzerStatusParams,
77) -> anyhow::Result<String> {
78    let _p = tracing::info_span!("handle_analyzer_status").entered();
79
80    let mut buf = String::new();
81
82    let mut file_id = None;
83    if let Some(tdi) = params.text_document {
84        match from_proto::file_id(&snap, &tdi.uri) {
85            Ok(Some(it)) => file_id = Some(it),
86            Ok(None) => {}
87            Err(_) => format_to!(buf, "file {} not found in vfs", tdi.uri),
88        }
89    }
90
91    if snap.workspaces.is_empty() {
92        buf.push_str("No workspaces\n")
93    } else {
94        buf.push_str("Workspaces:\n");
95        format_to!(
96            buf,
97            "Loaded {:?} packages across {} workspace{}.\n",
98            snap.workspaces.iter().map(|w| w.n_packages()).sum::<usize>(),
99            snap.workspaces.len(),
100            if snap.workspaces.len() == 1 { "" } else { "s" }
101        );
102
103        format_to!(
104            buf,
105            "Workspace root folders: {:?}",
106            snap.workspaces.iter().map(|ws| ws.manifest_or_root()).collect::<Vec<&AbsPath>>()
107        );
108    }
109    buf.push_str("\nAnalysis:\n");
110    buf.push_str(
111        &snap
112            .analysis
113            .status(file_id)
114            .unwrap_or_else(|_| "Analysis retrieval was cancelled".to_owned()),
115    );
116
117    buf.push_str("\nVersion: \n");
118    format_to!(buf, "{}", crate::version());
119
120    buf.push_str("\nConfiguration: \n");
121    format_to!(buf, "{:#?}", snap.config);
122
123    Ok(buf)
124}

197// cargo test requires:
198// - the package is a member of the workspace
199// - the target in the package is not a build script (custom-build)
200// - the package name - the root of the test identifier supplied to this handler can be
201//   a package or a target inside a package.
202// - the target name - if the test identifier is a target, it's needed in addition to the
203//   package name to run the right test
204// - real names - the test identifier uses the namespace form where hyphens are replaced with
205//   underscores. cargo test requires the real name.
206// - the target kind e.g. bin or lib
207fn all_test_targets(cargo: &CargoWorkspace) -> impl Iterator<Item = TestTarget> {
208    cargo.packages().filter(|p| cargo[*p].is_member).flat_map(|p| {
209        let package = &cargo[p];
210        package.targets.iter().filter_map(|t| {
211            let target = &cargo[*t];
212            if target.kind == TargetKind::BuildScript {
213                None
214            } else {
215                Some(TestTarget {
216                    package: package.name.clone(),
217                    target: target.name.clone(),
218                    kind: target.kind,
219                })
220            }
221        })
222    })
223}

225fn find_test_target(namespace_root: &str, cargo: &CargoWorkspace) -> Option<TestTarget> {
226    all_test_targets(cargo).find(|t| namespace_root == t.target.replace('-', "_"))
227}

229pub(crate) fn handle_run_test(
230    state: &mut GlobalState,
231    params: lsp_ext::RunTestParams,
232) -> anyhow::Result<()> {
233    if let Some(_session) = state.test_run_session.take() {
234        state.send_notification::<lsp_ext::EndRunTest>(());
235    }
236
237    let mut handles = vec![];
238    for ws in &*state.workspaces {
239        if let ProjectWorkspaceKind::Cargo { cargo, .. } = &ws.kind {
240            // need to deduplicate `include` to avoid redundant test runs
241            let tests = match params.include {
242                Some(ref include) => include
243                    .iter()
244                    .unique()
245                    .filter_map(|test| {
246                        let (root, remainder) = match test.split_once("::") {
247                            Some((root, remainder)) => (root.to_owned(), Some(remainder)),
248                            None => (test.clone(), None),
249                        };
250                        if let Some(target) = find_test_target(&root, cargo) {
251                            Some((target, remainder))
252                        } else {
253                            tracing::error!("Test target not found for: {test}");
254                            None
255                        }
256                    })
257                    .collect_vec(),
258                None => all_test_targets(cargo).map(|target| (target, None)).collect(),
259            };
260
261            for (target, path) in tests {
262                let handle = CargoTestHandle::new(
263                    path,
264                    state.config.cargo_test_options(None),
265                    cargo.workspace_root(),
266                    target,
267                    state.test_run_sender.clone(),
268                )?;
269                handles.push(handle);
270            }
271        }
272    }
273    // Each process send finished signal twice, once for stdout and once for stderr
274    state.test_run_remaining_jobs = 2 * handles.len();
275    state.test_run_session = Some(handles);
276    Ok(())
277}

330pub(crate) fn handle_selection_range(
331    snap: GlobalStateSnapshot,
332    params: lsp_types::SelectionRangeParams,
333) -> anyhow::Result<Option<Vec<lsp_types::SelectionRange>>> {
334    let _p = tracing::info_span!("handle_selection_range").entered();
335    let file_id = try_default!(from_proto::file_id(&snap, &params.text_document.uri)?);
336    let line_index = snap.file_line_index(file_id)?;
337    let res: anyhow::Result<Vec<lsp_types::SelectionRange>> = params
338        .positions
339        .into_iter()
340        .map(|position| {
341            let offset = from_proto::offset(&line_index, position)?;
342            let mut ranges = Vec::new();
343            {
344                let mut range = TextRange::new(offset, offset);
345                loop {
346                    ranges.push(range);
347                    let frange = FileRange { file_id, range };
348                    let next = snap.analysis.extend_selection(frange)?;
349                    if next == range {
350                        break;
351                    } else {
352                        range = next
353                    }
354                }
355            }
356            let mut range = lsp_types::SelectionRange {
357                range: to_proto::range(&line_index, *ranges.last().unwrap()),
358                parent: None,
359            };
360            for &r in ranges.iter().rev().skip(1) {
361                range = lsp_types::SelectionRange {
362                    range: to_proto::range(&line_index, r),
363                    parent: Some(Box::new(range)),
364                }
365            }
366            Ok(range)
367        })
368        .collect();
369
370    Ok(Some(res?))
371}

441pub(crate) fn handle_on_type_formatting(
442    snap: GlobalStateSnapshot,
443    params: lsp_types::DocumentOnTypeFormattingParams,
444) -> anyhow::Result<Option<Vec<lsp_ext::SnippetTextEdit>>> {
445    let _p = tracing::info_span!("handle_on_type_formatting").entered();
446    let char_typed = params.ch.chars().next().unwrap_or('\0');
447    if !snap.config.typing_trigger_chars().contains(char_typed) {
448        return Ok(None);
449    }
450
451    let mut position =
452        try_default!(from_proto::file_position(&snap, params.text_document_position)?);
453    let line_index = snap.file_line_index(position.file_id)?;
454
455    // in `ide`, the `on_type` invariant is that
456    // `text.char_at(position) == typed_char`.
457    position.offset -= TextSize::of('.');
458
459    let text = snap.analysis.file_text(position.file_id)?;
460    if stdx::never!(!text[usize::from(position.offset)..].starts_with(char_typed)) {
461        return Ok(None);
462    }
463
464    let edit = snap.analysis.on_char_typed(position, char_typed)?;
465    let edit = match edit {
466        Some(it) => it,
467        None => return Ok(None),
468    };
469
470    // This should be a single-file edit
471    let (_, (text_edit, snippet_edit)) = edit.source_file_edits.into_iter().next().unwrap();
472    stdx::always!(snippet_edit.is_none(), "on type formatting shouldn't use structured snippets");
473
474    let change = to_proto::snippet_text_edit_vec(
475        &line_index,
476        edit.is_snippet,
477        text_edit,
478        snap.config.change_annotation_support(),
479    );
480    Ok(Some(change))
481}

495pub(crate) fn handle_document_diagnostics(
496    snap: GlobalStateSnapshot,
497    params: lsp_types::DocumentDiagnosticParams,
498) -> anyhow::Result<lsp_types::DocumentDiagnosticReportResult> {
499    let file_id = match from_proto::file_id(&snap, &params.text_document.uri)? {
500        Some(it) => it,
501        None => return Ok(empty_diagnostic_report()),
502    };
503    let source_root = snap.analysis.source_root_id(file_id)?;
504    if !snap.analysis.is_local_source_root(source_root)? {
505        return Ok(empty_diagnostic_report());
506    }
507    let source_root = snap.analysis.source_root_id(file_id)?;
508    let config = snap.config.diagnostics(Some(source_root));
509    if !config.enabled {
510        return Ok(empty_diagnostic_report());
511    }
512    let line_index = snap.file_line_index(file_id)?;
513    let supports_related = snap.config.text_document_diagnostic_related_document_support();
514
515    let mut related_documents = FxHashMap::default();
516    let diagnostics = snap
517        .analysis
518        .full_diagnostics(&config, AssistResolveStrategy::None, file_id)?
519        .into_iter()
520        .filter_map(|d| {
521            let file = d.range.file_id;
522            if file == file_id {
523                let diagnostic = convert_diagnostic(&line_index, d);
524                return Some(diagnostic);
525            }
526            if supports_related {
527                let (diagnostics, line_index) = related_documents
528                    .entry(file)
529                    .or_insert_with(|| (Vec::new(), snap.file_line_index(file).ok()));
530                let diagnostic = convert_diagnostic(line_index.as_mut()?, d);
531                diagnostics.push(diagnostic);
532            }
533            None
534        });
535    Ok(lsp_types::DocumentDiagnosticReportResult::Report(
536        lsp_types::DocumentDiagnosticReport::Full(lsp_types::RelatedFullDocumentDiagnosticReport {
537            full_document_diagnostic_report: lsp_types::FullDocumentDiagnosticReport {
538                result_id: Some("rust-analyzer".to_owned()),
539                items: diagnostics.collect(),
540            },
541            related_documents: related_documents.is_empty().not().then(|| {
542                related_documents
543                    .into_iter()
544                    .map(|(id, (items, _))| {
545                        (
546                            to_proto::url(&snap, id),
547                            lsp_types::DocumentDiagnosticReportKind::Full(
548                                lsp_types::FullDocumentDiagnosticReport {
549                                    result_id: Some("rust-analyzer".to_owned()),
550                                    items,
551                                },
552                            ),
553                        )
554                    })
555                    .collect()
556            }),
557        }),
558    ))
559}

561pub(crate) fn handle_document_symbol(
562    snap: GlobalStateSnapshot,
563    params: lsp_types::DocumentSymbolParams,
564) -> anyhow::Result<Option<lsp_types::DocumentSymbolResponse>> {
565    let _p = tracing::info_span!("handle_document_symbol").entered();
566    let file_id = try_default!(from_proto::file_id(&snap, &params.text_document.uri)?);
567    let line_index = snap.file_line_index(file_id)?;
568
569    let mut parents: Vec<(lsp_types::DocumentSymbol, Option<usize>)> = Vec::new();
570
571    for symbol in snap.analysis.file_structure(file_id)? {
572        let mut tags = Vec::new();
573        if symbol.deprecated {
574            tags.push(SymbolTag::DEPRECATED)
575        };
576
577        #[allow(deprecated)]
578        let doc_symbol = lsp_types::DocumentSymbol {
579            name: symbol.label,
580            detail: symbol.detail,
581            kind: to_proto::structure_node_kind(symbol.kind),
582            tags: Some(tags),
583            deprecated: Some(symbol.deprecated),
584            range: to_proto::range(&line_index, symbol.node_range),
585            selection_range: to_proto::range(&line_index, symbol.navigation_range),
586            children: None,
587        };
588        parents.push((doc_symbol, symbol.parent));
589    }
590
591    // Builds hierarchy from a flat list, in reverse order (so that indices
592    // makes sense)
593    let document_symbols = {
594        let mut acc = Vec::new();
595        while let Some((mut node, parent_idx)) = parents.pop() {
596            if let Some(children) = &mut node.children {
597                children.reverse();
598            }
599            let parent = match parent_idx {
600                None => &mut acc,
601                Some(i) => parents[i].0.children.get_or_insert_with(Vec::new),
602            };
603            parent.push(node);
604        }
605        acc.reverse();
606        acc
607    };
608
609    let res = if snap.config.hierarchical_symbols() {
610        document_symbols.into()
611    } else {
612        let url = to_proto::url(&snap, file_id);
613        let mut symbol_information = Vec::<SymbolInformation>::new();
614        for symbol in document_symbols {
615            flatten_document_symbol(&symbol, None, &url, &mut symbol_information);
616        }
617        symbol_information.into()
618    };
619    return Ok(Some(res));
620
621    fn flatten_document_symbol(
622        symbol: &lsp_types::DocumentSymbol,
623        container_name: Option<String>,
624        url: &Url,
625        res: &mut Vec<SymbolInformation>,
626    ) {
627        #[allow(deprecated)]
628        res.push(SymbolInformation {
629            name: symbol.name.clone(),
630            kind: symbol.kind,
631            tags: symbol.tags.clone(),
632            deprecated: symbol.deprecated,
633            location: Location::new(url.clone(), symbol.range),
634            container_name,
635        });
636
637        for child in symbol.children.iter().flatten() {
638            flatten_document_symbol(child, Some(symbol.name.clone()), url, res);
639        }
640    }

643pub(crate) fn handle_workspace_symbol(
644    snap: GlobalStateSnapshot,
645    params: WorkspaceSymbolParams,
646) -> anyhow::Result<Option<lsp_types::WorkspaceSymbolResponse>> {
647    let _p = tracing::info_span!("handle_workspace_symbol").entered();
648
649    let config = snap.config.workspace_symbol(None);
650    let (all_symbols, libs) = decide_search_scope_and_kind(&params, &config);
651
652    let query = {
653        let query: String = params.query.chars().filter(|&c| c != '#' && c != '*').collect();
654        let mut q = Query::new(query);
655        if !all_symbols {
656            q.only_types();
657        }
658        if libs {
659            q.libs();
660        }
661        q
662    };
663    let mut res = exec_query(&snap, query, config.search_limit)?;
664    if res.is_empty() && !all_symbols {
665        res = exec_query(&snap, Query::new(params.query), config.search_limit)?;
666    }
667
668    return Ok(Some(lsp_types::WorkspaceSymbolResponse::Nested(res)));
669
670    fn decide_search_scope_and_kind(
671        params: &WorkspaceSymbolParams,
672        config: &WorkspaceSymbolConfig,
673    ) -> (bool, bool) {
674        // Support old-style parsing of markers in the query.
675        let mut all_symbols = params.query.contains('#');
676        let mut libs = params.query.contains('*');
677
678        // If no explicit marker was set, check request params. If that's also empty
679        // use global config.
680        if !all_symbols {
681            let search_kind = match params.search_kind {
682                Some(ref search_kind) => search_kind,
683                None => &config.search_kind,
684            };
685            all_symbols = match search_kind {
686                lsp_ext::WorkspaceSymbolSearchKind::OnlyTypes => false,
687                lsp_ext::WorkspaceSymbolSearchKind::AllSymbols => true,
688            }
689        }
690
691        if !libs {
692            let search_scope = match params.search_scope {
693                Some(ref search_scope) => search_scope,
694                None => &config.search_scope,
695            };
696            libs = match search_scope {
697                lsp_ext::WorkspaceSymbolSearchScope::Workspace => false,
698                lsp_ext::WorkspaceSymbolSearchScope::WorkspaceAndDependencies => true,
699            }
700        }
701
702        (all_symbols, libs)
703    }
704
705    fn exec_query(
706        snap: &GlobalStateSnapshot,
707        query: Query,
708        limit: usize,
709    ) -> anyhow::Result<Vec<lsp_types::WorkspaceSymbol>> {
710        let mut res = Vec::new();
711        for nav in snap.analysis.symbol_search(query, limit)? {
712            let container_name = nav.container_name.as_ref().map(|v| v.to_string());
713
714            let info = lsp_types::WorkspaceSymbol {
715                name: match &nav.alias {
716                    Some(alias) => format!("{} (alias for {})", alias, nav.name),
717                    None => format!("{}", nav.name),
718                },
719                kind: nav
720                    .kind
721                    .map(to_proto::symbol_kind)
722                    .unwrap_or(lsp_types::SymbolKind::VARIABLE),
723                // FIXME: Set deprecation
724                tags: None,
725                container_name,
726                location: lsp_types::OneOf::Left(to_proto::location_from_nav(snap, nav)?),
727                data: None,
728            };
729            res.push(info);
730        }
731        Ok(res)
732    }
733}

735pub(crate) fn handle_will_rename_files(
736    snap: GlobalStateSnapshot,
737    params: lsp_types::RenameFilesParams,
738) -> anyhow::Result<Option<lsp_types::WorkspaceEdit>> {
739    let _p = tracing::info_span!("handle_will_rename_files").entered();
740
741    let source_changes: Vec<SourceChange> = params
742        .files
743        .into_iter()
744        .filter_map(|file_rename| {
745            let from = Url::parse(&file_rename.old_uri).ok()?;
746            let to = Url::parse(&file_rename.new_uri).ok()?;
747
748            let from_path = from.to_file_path().ok()?;
749            let to_path = to.to_file_path().ok()?;
750
751            // Limit to single-level moves for now.
752            match (from_path.parent(), to_path.parent()) {
753                (Some(p1), Some(p2)) if p1 == p2 => {
754                    if from_path.is_dir() {
755                        // add '/' to end of url -- from `file://path/to/folder` to `file://path/to/folder/`
756                        let mut old_folder_name = from_path.file_stem()?.to_str()?.to_owned();
757                        old_folder_name.push('/');
758                        let from_with_trailing_slash = from.join(&old_folder_name).ok()?;
759
760                        let imitate_from_url = from_with_trailing_slash.join("mod.rs").ok()?;
761                        let new_file_name = to_path.file_name()?.to_str()?;
762                        Some((
763                            snap.url_to_file_id(&imitate_from_url).ok()?,
764                            new_file_name.to_owned(),
765                        ))
766                    } else {
767                        let old_name = from_path.file_stem()?.to_str()?;
768                        let new_name = to_path.file_stem()?.to_str()?;
769                        match (old_name, new_name) {
770                            ("mod", _) => None,
771                            (_, "mod") => None,
772                            _ => Some((snap.url_to_file_id(&from).ok()?, new_name.to_owned())),
773                        }
774                    }
775                }
776                _ => None,
777            }
778        })
779        .filter_map(|(file_id, new_name)| {
780            snap.analysis.will_rename_file(file_id?, &new_name).ok()?
781        })
782        .collect();
783
784    // Drop file system edits since we're just renaming things on the same level
785    let mut source_changes = source_changes.into_iter();
786    let mut source_change = source_changes.next().unwrap_or_default();
787    source_change.file_system_edits.clear();
788    // no collect here because we want to merge text edits on same file ids
789    source_change.extend(source_changes.flat_map(|it| it.source_file_edits));
790    if source_change.source_file_edits.is_empty() {
791        Ok(None)
792    } else {
793        Ok(Some(to_proto::workspace_edit(&snap, source_change)?))
794    }
795}

863pub(crate) fn handle_parent_module(
864    snap: GlobalStateSnapshot,
865    params: lsp_types::TextDocumentPositionParams,
866) -> anyhow::Result<Option<lsp_types::GotoDefinitionResponse>> {
867    let _p = tracing::info_span!("handle_parent_module").entered();
868    if let Ok(file_path) = &params.text_document.uri.to_file_path() {
869        if file_path.file_name().unwrap_or_default() == "Cargo.toml" {
870            // search workspaces for parent packages or fallback to workspace root
871            let abs_path_buf = match Utf8PathBuf::from_path_buf(file_path.to_path_buf())
872                .ok()
873                .map(AbsPathBuf::try_from)
874            {
875                Some(Ok(abs_path_buf)) => abs_path_buf,
876                _ => return Ok(None),
877            };
878
879            let manifest_path = match ManifestPath::try_from(abs_path_buf).ok() {
880                Some(manifest_path) => manifest_path,
881                None => return Ok(None),
882            };
883
884            let links: Vec<LocationLink> = snap
885                .workspaces
886                .iter()
887                .filter_map(|ws| match &ws.kind {
888                    ProjectWorkspaceKind::Cargo { cargo, .. }
889                    | ProjectWorkspaceKind::DetachedFile { cargo: Some((cargo, _, _)), .. } => {
890                        cargo.parent_manifests(&manifest_path)
891                    }
892                    _ => None,
893                })
894                .flatten()
895                .map(|parent_manifest_path| LocationLink {
896                    origin_selection_range: None,
897                    target_uri: to_proto::url_from_abs_path(&parent_manifest_path),
898                    target_range: Range::default(),
899                    target_selection_range: Range::default(),
900                })
901                .collect::<_>();
902            return Ok(Some(links.into()));
903        }
904
905        // check if invoked at the crate root
906        let file_id = try_default!(from_proto::file_id(&snap, &params.text_document.uri)?);
907        let crate_id = match snap.analysis.crates_for(file_id)?.first() {
908            Some(&crate_id) => crate_id,
909            None => return Ok(None),
910        };
911        let cargo_spec = match TargetSpec::for_file(&snap, file_id)? {
912            Some(TargetSpec::Cargo(it)) => it,
913            Some(TargetSpec::ProjectJson(_)) | None => return Ok(None),
914        };
915
916        if snap.analysis.crate_root(crate_id)? == file_id {
917            let cargo_toml_url = to_proto::url_from_abs_path(&cargo_spec.cargo_toml);
918            let res = vec![LocationLink {
919                origin_selection_range: None,
920                target_uri: cargo_toml_url,
921                target_range: Range::default(),
922                target_selection_range: Range::default(),
923            }]
924            .into();
925            return Ok(Some(res));
926        }
927    }
928
929    // locate parent module by semantics
930    let position = try_default!(from_proto::file_position(&snap, params)?);
931    let navs = snap.analysis.parent_module(position)?;
932    let res = to_proto::goto_definition_response(&snap, None, navs)?;
933    Ok(Some(res))
934}

948pub(crate) fn handle_runnables(
949    snap: GlobalStateSnapshot,
950    params: lsp_ext::RunnablesParams,
951) -> anyhow::Result<Vec<lsp_ext::Runnable>> {
952    let _p = tracing::info_span!("handle_runnables").entered();
953    let file_id = try_default!(from_proto::file_id(&snap, &params.text_document.uri)?);
954    let source_root = snap.analysis.source_root_id(file_id).ok();
955    let line_index = snap.file_line_index(file_id)?;
956    let offset = params.position.and_then(|it| from_proto::offset(&line_index, it).ok());
957    let target_spec = TargetSpec::for_file(&snap, file_id)?;
958
959    let mut res = Vec::new();
960    for runnable in snap.analysis.runnables(file_id)? {
961        if should_skip_for_offset(&runnable, offset)
962            || should_skip_target(&runnable, target_spec.as_ref())
963        {
964            continue;
965        }
966
967        let update_test = runnable.update_test;
968        if let Some(mut runnable) = to_proto::runnable(&snap, runnable)? {
969            if let Some(runnable) = to_proto::make_update_runnable(&runnable, update_test) {
970                res.push(runnable);
971            }
972
973            if let lsp_ext::RunnableArgs::Cargo(r) = &mut runnable.args {
974                if let Some(TargetSpec::Cargo(CargoTargetSpec {
975                    sysroot_root: Some(sysroot_root),
976                    ..
977                })) = &target_spec
978                {
979                    r.environment.insert("RUSTC_TOOLCHAIN".to_owned(), sysroot_root.to_string());
980                }
981            };
982
983            res.push(runnable);
984        }
985    }
986
987    // Add `cargo check` and `cargo test` for all targets of the whole package
988    let config = snap.config.runnables(source_root);
989    match target_spec {
990        Some(TargetSpec::Cargo(spec)) => {
991            let is_crate_no_std = snap.analysis.is_crate_no_std(spec.crate_id)?;
992            for cmd in ["check", "run", "test"] {
993                if cmd == "run" && spec.target_kind != TargetKind::Bin {
994                    continue;
995                }
996                let cwd = if cmd != "test" || spec.target_kind == TargetKind::Bin {
997                    spec.workspace_root.clone()
998                } else {
999                    spec.cargo_toml.parent().to_path_buf()
1000                };
1001                let mut cargo_args =
1002                    vec![cmd.to_owned(), "--package".to_owned(), spec.package.clone()];
1003                let all_targets = cmd != "run" && !is_crate_no_std;
1004                if all_targets {
1005                    cargo_args.push("--all-targets".to_owned());
1006                }
1007                cargo_args.extend(config.cargo_extra_args.iter().cloned());
1008                res.push(lsp_ext::Runnable {
1009                    label: format!(
1010                        "cargo {cmd} -p {}{all_targets}",
1011                        spec.package,
1012                        all_targets = if all_targets { " --all-targets" } else { "" }
1013                    ),
1014                    location: None,
1015                    kind: lsp_ext::RunnableKind::Cargo,
1016                    args: lsp_ext::RunnableArgs::Cargo(lsp_ext::CargoRunnableArgs {
1017                        workspace_root: Some(spec.workspace_root.clone().into()),
1018                        cwd: cwd.into(),
1019                        override_cargo: config.override_cargo.clone(),
1020                        cargo_args,
1021                        executable_args: Vec::new(),
1022                        environment: spec
1023                            .sysroot_root
1024                            .as_ref()
1025                            .map(|root| ("RUSTC_TOOLCHAIN".to_owned(), root.to_string()))
1026                            .into_iter()
1027                            .collect(),
1028                    }),
1029                })
1030            }
1031        }
1032        Some(TargetSpec::ProjectJson(_)) => {}
1033        None => {
1034            if !snap.config.linked_or_discovered_projects().is_empty() {
1035                if let Some(path) = snap.file_id_to_file_path(file_id).parent() {
1036                    let mut cargo_args = vec!["check".to_owned(), "--workspace".to_owned()];
1037                    cargo_args.extend(config.cargo_extra_args.iter().cloned());
1038                    res.push(lsp_ext::Runnable {
1039                        label: "cargo check --workspace".to_owned(),
1040                        location: None,
1041                        kind: lsp_ext::RunnableKind::Cargo,
1042                        args: lsp_ext::RunnableArgs::Cargo(lsp_ext::CargoRunnableArgs {
1043                            workspace_root: None,
1044                            cwd: path.as_path().unwrap().to_path_buf().into(),
1045                            override_cargo: config.override_cargo,
1046                            cargo_args,
1047                            executable_args: Vec::new(),
1048                            environment: Default::default(),
1049                        }),
1050                    });
1051                };
1052            }
1053        }
1054    }
1055    Ok(res)
1056}

1126pub(crate) fn handle_completion_resolve(
1127    snap: GlobalStateSnapshot,
1128    mut original_completion: CompletionItem,
1129) -> anyhow::Result<CompletionItem> {
1130    let _p = tracing::info_span!("handle_completion_resolve").entered();
1131
1132    if !all_edits_are_disjoint(&original_completion, &[]) {
1133        return Err(invalid_params_error(
1134            "Received a completion with overlapping edits, this is not LSP-compliant".to_owned(),
1135        )
1136        .into());
1137    }
1138
1139    let Some(data) = original_completion.data.take() else {
1140        return Ok(original_completion);
1141    };
1142
1143    let resolve_data: lsp_ext::CompletionResolveData = serde_json::from_value(data)?;
1144
1145    let file_id = from_proto::file_id(&snap, &resolve_data.position.text_document.uri)?
1146        .expect("we never provide completions for excluded files");
1147    let line_index = snap.file_line_index(file_id)?;
1148    // FIXME: We should fix up the position when retrying the cancelled request instead
1149    let Ok(offset) = from_proto::offset(&line_index, resolve_data.position.position) else {
1150        return Ok(original_completion);
1151    };
1152    let source_root = snap.analysis.source_root_id(file_id)?;
1153
1154    let mut forced_resolve_completions_config = snap.config.completion(Some(source_root));
1155    forced_resolve_completions_config.fields_to_resolve = CompletionFieldsToResolve::empty();
1156
1157    let position = FilePosition { file_id, offset };
1158    let Some(completions) = snap.analysis.completions(
1159        &forced_resolve_completions_config,
1160        position,
1161        resolve_data.trigger_character,
1162    )?
1163    else {
1164        return Ok(original_completion);
1165    };
1166    let Ok(resolve_data_hash) = BASE64_STANDARD.decode(resolve_data.hash) else {
1167        return Ok(original_completion);
1168    };
1169
1170    let Some(corresponding_completion) = completions.into_iter().find(|completion_item| {
1171        // Avoid computing hashes for items that obviously do not match
1172        // r-a might append a detail-based suffix to the label, so we cannot check for equality
1173        original_completion.label.starts_with(completion_item.label.primary.as_str())
1174            && resolve_data_hash == completion_item_hash(completion_item, resolve_data.for_ref)
1175    }) else {
1176        return Ok(original_completion);
1177    };
1178
1179    let mut resolved_completions = to_proto::completion_items(
1180        &snap.config,
1181        &forced_resolve_completions_config.fields_to_resolve,
1182        &line_index,
1183        snap.file_version(position.file_id),
1184        resolve_data.position,
1185        resolve_data.trigger_character,
1186        vec![corresponding_completion],
1187    );
1188    let Some(mut resolved_completion) = resolved_completions.pop() else {
1189        return Ok(original_completion);
1190    };
1191
1192    if !resolve_data.imports.is_empty() {
1193        let additional_edits = snap
1194            .analysis
1195            .resolve_completion_edits(
1196                &forced_resolve_completions_config,
1197                position,
1198                resolve_data.imports.into_iter().map(|import| import.full_import_path),
1199            )?
1200            .into_iter()
1201            .flat_map(|edit| edit.into_iter().map(|indel| to_proto::text_edit(&line_index, indel)))
1202            .collect::<Vec<_>>();
1203
1204        if !all_edits_are_disjoint(&resolved_completion, &additional_edits) {
1205            return Err(LspError::new(
1206                ErrorCode::InternalError as i32,
1207                "Import edit overlaps with the original completion edits, this is not LSP-compliant"
1208                    .into(),
1209            )
1210            .into());
1211        }
1212
1213        if let Some(original_additional_edits) = resolved_completion.additional_text_edits.as_mut()
1214        {
1215            original_additional_edits.extend(additional_edits)
1216        } else {
1217            resolved_completion.additional_text_edits = Some(additional_edits);
1218        }
1219    }
1220
1221    Ok(resolved_completion)
1222}

1309pub(crate) fn handle_rename(
1310    snap: GlobalStateSnapshot,
1311    params: RenameParams,
1312) -> anyhow::Result<Option<WorkspaceEdit>> {
1313    let _p = tracing::info_span!("handle_rename").entered();
1314    let position = try_default!(from_proto::file_position(&snap, params.text_document_position)?);
1315
1316    let mut change =
1317        snap.analysis.rename(position, &params.new_name)?.map_err(to_proto::rename_error)?;
1318
1319    // this is kind of a hack to prevent double edits from happening when moving files
1320    // When a module gets renamed by renaming the mod declaration this causes the file to move
1321    // which in turn will trigger a WillRenameFiles request to the server for which we reply with a
1322    // a second identical set of renames, the client will then apply both edits causing incorrect edits
1323    // with this we only emit source_file_edits in the WillRenameFiles response which will do the rename instead
1324    // See https://github.com/microsoft/vscode-languageserver-node/issues/752 for more info
1325    if !change.file_system_edits.is_empty() && snap.config.will_rename() {
1326        change.source_file_edits.clear();
1327    }
1328
1329    let workspace_edit = to_proto::workspace_edit(&snap, change)?;
1330
1331    if let Some(lsp_types::DocumentChanges::Operations(ops)) =
1332        workspace_edit.document_changes.as_ref()
1333    {
1334        for op in ops {
1335            if let lsp_types::DocumentChangeOperation::Op(doc_change_op) = op {
1336                resource_ops_supported(&snap.config, resolve_resource_op(doc_change_op))?
1337            }
1338        }
1339    }
1340
1341    Ok(Some(workspace_edit))

1407pub(crate) fn handle_code_action(
1408    snap: GlobalStateSnapshot,
1409    params: lsp_types::CodeActionParams,
1410) -> anyhow::Result<Option<Vec<lsp_ext::CodeAction>>> {
1411    let _p = tracing::info_span!("handle_code_action").entered();
1412
1413    if !snap.config.code_action_literals() {
1414        // We intentionally don't support command-based actions, as those either
1415        // require either custom client-code or server-initiated edits. Server
1416        // initiated edits break causality, so we avoid those.
1417        return Ok(None);
1418    }
1419
1420    let file_id = try_default!(from_proto::file_id(&snap, &params.text_document.uri)?);
1421    let line_index = snap.file_line_index(file_id)?;
1422    let frange = try_default!(from_proto::file_range(&snap, &params.text_document, params.range)?);
1423    let source_root = snap.analysis.source_root_id(file_id)?;
1424
1425    let mut assists_config = snap.config.assist(Some(source_root));
1426    assists_config.allowed = params
1427        .context
1428        .only
1429        .clone()
1430        .map(|it| it.into_iter().filter_map(from_proto::assist_kind).collect());
1431
1432    let mut res: Vec<lsp_ext::CodeAction> = Vec::new();
1433
1434    let code_action_resolve_cap = snap.config.code_action_resolve();
1435    let resolve = if code_action_resolve_cap {
1436        AssistResolveStrategy::None
1437    } else {
1438        AssistResolveStrategy::All
1439    };
1440    let assists = snap.analysis.assists_with_fixes(
1441        &assists_config,
1442        &snap.config.diagnostics(Some(source_root)),
1443        resolve,
1444        frange,
1445    )?;
1446    for (index, assist) in assists.into_iter().enumerate() {
1447        let resolve_data = if code_action_resolve_cap {
1448            Some((index, params.clone(), snap.file_version(file_id)))
1449        } else {
1450            None
1451        };
1452        let code_action = to_proto::code_action(&snap, assist, resolve_data)?;
1453
1454        // Check if the client supports the necessary `ResourceOperation`s.
1455        let changes = code_action.edit.as_ref().and_then(|it| it.document_changes.as_ref());
1456        if let Some(changes) = changes {
1457            for change in changes {
1458                if let lsp_ext::SnippetDocumentChangeOperation::Op(res_op) = change {
1459                    resource_ops_supported(&snap.config, resolve_resource_op(res_op))?
1460                }
1461            }
1462        }
1463
1464        res.push(code_action)
1465    }
1466
1467    // Fixes from `cargo check`.
1468    for fix in snap
1469        .check_fixes
1470        .iter()
1471        .flat_map(|it| it.values())
1472        .filter_map(|it| it.get(&frange.file_id))
1473        .flatten()
1474    {
1475        // FIXME: this mapping is awkward and shouldn't exist. Refactor
1476        // `snap.check_fixes` to not convert to LSP prematurely.
1477        let intersect_fix_range = fix
1478            .ranges
1479            .iter()
1480            .copied()
1481            .filter_map(|range| from_proto::text_range(&line_index, range).ok())
1482            .any(|fix_range| fix_range.intersect(frange.range).is_some());
1483        if intersect_fix_range {
1484            res.push(fix.action.clone());
1485        }
1486    }
1487
1488    Ok(Some(res))
1489}

1491pub(crate) fn handle_code_action_resolve(
1492    snap: GlobalStateSnapshot,
1493    mut code_action: lsp_ext::CodeAction,
1494) -> anyhow::Result<lsp_ext::CodeAction> {
1495    let _p = tracing::info_span!("handle_code_action_resolve").entered();
1496    let Some(params) = code_action.data.take() else {
1497        return Ok(code_action);
1498    };
1499
1500    let file_id = from_proto::file_id(&snap, &params.code_action_params.text_document.uri)?
1501        .expect("we never provide code actions for excluded files");
1502    if snap.file_version(file_id) != params.version {
1503        return Err(invalid_params_error("stale code action".to_owned()).into());
1504    }
1505    let line_index = snap.file_line_index(file_id)?;
1506    let range = from_proto::text_range(&line_index, params.code_action_params.range)?;
1507    let frange = FileRange { file_id, range };
1508    let source_root = snap.analysis.source_root_id(file_id)?;
1509
1510    let mut assists_config = snap.config.assist(Some(source_root));
1511    assists_config.allowed = params
1512        .code_action_params
1513        .context
1514        .only
1515        .map(|it| it.into_iter().filter_map(from_proto::assist_kind).collect());
1516
1517    let (assist_index, assist_resolve) = match parse_action_id(&params.id) {
1518        Ok(parsed_data) => parsed_data,
1519        Err(e) => {
1520            return Err(invalid_params_error(format!(
1521                "Failed to parse action id string '{}': {e}",
1522                params.id
1523            ))
1524            .into());
1525        }
1526    };
1527
1528    let expected_assist_id = assist_resolve.assist_id.clone();
1529    let expected_kind = assist_resolve.assist_kind;
1530
1531    let assists = snap.analysis.assists_with_fixes(
1532        &assists_config,
1533        &snap.config.diagnostics(Some(source_root)),
1534        AssistResolveStrategy::Single(assist_resolve),
1535        frange,
1536    )?;
1537
1538    let assist = match assists.get(assist_index) {
1539        Some(assist) => assist,
1540        None => return Err(invalid_params_error(format!(
1541            "Failed to find the assist for index {} provided by the resolve request. Resolve request assist id: {}",
1542            assist_index, params.id,
1543        ))
1544        .into())
1545    };
1546    if assist.id.0 != expected_assist_id || assist.id.1 != expected_kind {
1547        return Err(invalid_params_error(format!(
1548            "Mismatching assist at index {} for the resolve parameters given. Resolve request assist id: {}, actual id: {:?}.",
1549            assist_index, params.id, assist.id
1550        ))
1551        .into());
1552    }
1553    let ca = to_proto::code_action(&snap, assist.clone(), None)?;
1554    code_action.edit = ca.edit;
1555    code_action.command = ca.command;
1556
1557    if let Some(edit) = code_action.edit.as_ref() {
1558        if let Some(changes) = edit.document_changes.as_ref() {
1559            for change in changes {
1560                if let lsp_ext::SnippetDocumentChangeOperation::Op(res_op) = change {
1561                    resource_ops_supported(&snap.config, resolve_resource_op(res_op))?
1562                }
1563            }
1564        }
1565    }
1566
1567    Ok(code_action)
1568}

1570fn parse_action_id(action_id: &str) -> anyhow::Result<(usize, SingleResolve), String> {
1571    let id_parts = action_id.split(':').collect::<Vec<_>>();
1572    match id_parts.as_slice() {
1573        [assist_id_string, assist_kind_string, index_string, subtype_str] => {
1574            let assist_kind: AssistKind = assist_kind_string.parse()?;
1575            let index: usize = match index_string.parse() {
1576                Ok(index) => index,
1577                Err(e) => return Err(format!("Incorrect index string: {e}")),
1578            };
1579            let assist_subtype = subtype_str.parse::<usize>().ok();
1580            Ok((
1581                index,
1582                SingleResolve {
1583                    assist_id: assist_id_string.to_string(),
1584                    assist_kind,
1585                    assist_subtype,
1586                },
1587            ))
1588        }
1589        _ => Err("Action id contains incorrect number of segments".to_owned()),
1590    }
1591}

2327            if let Some(range) = range {
2328                if !enable_range_formatting {
2329                    return Err(LspError::new(
2330                        ErrorCode::InvalidRequest as i32,
2331                        String::from(
2332                            "rustfmt range formatting is unstable. \
2333                            Opt-in by using a nightly build of rustfmt and setting \
2334                            `rustfmt.rangeFormatting.enable` to true in your LSP configuration",
2335                        ),
2336                    )
2337                    .into());
2338                }
2339
2340                let frange = try_default!(from_proto::file_range(snap, &text_document, range)?);
2341                let start_line = line_index.index.line_col(frange.range.start()).line;
2342                let end_line = line_index.index.line_col(frange.range.end()).line;
2343
2344                cmd.arg("--unstable-features");
2345                cmd.arg("--file-lines");
2346                cmd.arg(
2347                    json!([{
2348                        "file": "stdin",
2349                        // LineCol is 0-based, but rustfmt is 1-based.
2350                        "range": [start_line + 1, end_line + 1]
2351                    }])
2352                    .to_string(),
2353                );
2354            }
2355
2356            cmd
2357        }
2358        RustfmtConfig::CustomCommand { command, args } => {
2359            let cmd = Utf8PathBuf::from(&command);
2360            let target_spec = TargetSpec::for_file(snap, file_id)?;
2361            let mut cmd = match target_spec {
2362                Some(TargetSpec::Cargo(_)) => {
2363                    // approach: if the command name contains a path separator, join it with the project root.
2364                    // however, if the path is absolute, joining will result in the absolute path being preserved.
2365                    // as a fallback, rely on $PATH-based discovery.
2366                    let cmd_path = if command.contains(std::path::MAIN_SEPARATOR)
2367                        || (cfg!(windows) && command.contains('/'))
2368                    {
2369                        snap.config.root_path().join(cmd).into()
2370                    } else {
2371                        cmd
2372                    };
2373                    toolchain::command(cmd_path, current_dir)
2374                }
2375                _ => toolchain::command(cmd, current_dir),
2376            };
2377
2378            cmd.envs(snap.config.extra_env(source_root_id));
2379            cmd.args(args);
2380            cmd
2381        }
2382    };
2383
2384    let output = {
2385        let _p = tracing::info_span!("rustfmt", ?command).entered();
2386
2387        let mut rustfmt = command
2388            .stdin(Stdio::piped())
2389            .stdout(Stdio::piped())
2390            .stderr(Stdio::piped())
2391            .spawn()
2392            .context(format!("Failed to spawn {command:?}"))?;
2393
2394        rustfmt.stdin.as_mut().unwrap().write_all(file.as_bytes())?;
2395
2396        rustfmt.wait_with_output()?
2397    };
2398
2399    let captured_stdout = String::from_utf8(output.stdout)?;
2400    let captured_stderr = String::from_utf8(output.stderr).unwrap_or_default();
2401
2402    if !output.status.success() {
2403        let rustfmt_not_installed =
2404            captured_stderr.contains("not installed") || captured_stderr.contains("not available");
2405
2406        return match output.status.code() {
2407            Some(1) if !rustfmt_not_installed => {
2408                // While `rustfmt` doesn't have a specific exit code for parse errors this is the
2409                // likely cause exiting with 1. Most Language Servers swallow parse errors on
2410                // formatting because otherwise an error is surfaced to the user on top of the
2411                // syntax error diagnostics they're already receiving. This is especially jarring
2412                // if they have format on save enabled.
2413                tracing::warn!(
2414                    ?command,
2415                    %captured_stderr,
2416                    "rustfmt exited with status 1"
2417                );
2418                Ok(None)
2419            }
2420            // rustfmt panicked at lexing/parsing the file
2421            Some(101)
2422                if !rustfmt_not_installed
2423                    && (captured_stderr.starts_with("error[")
2424                        || captured_stderr.starts_with("error:")) =>
2425            {
2426                Ok(None)
2427            }
2428            _ => {
2429                // Something else happened - e.g. `rustfmt` is missing or caught a signal
2430                Err(LspError::new(
2431                    -32900,
2432                    format!(
2433                        r#"rustfmt exited with:
2434                           Status: {}
2435                           stdout: {captured_stdout}
2436                           stderr: {captured_stderr}"#,
2437                        output.status,
2438                    ),
2439                )
2440                .into())
2441            }
2442        };
2443    }
2444
2445    let (new_text, new_line_endings) = LineEndings::normalize(captured_stdout);
2446
2447    if line_index.endings != new_line_endings {
2448        // If line endings are different, send the entire file.
2449        // Diffing would not work here, as the line endings might be the only
2450        // difference.
2451        Ok(Some(to_proto::text_edit_vec(
2452            &line_index,
2453            TextEdit::replace(TextRange::up_to(TextSize::of(&*file)), new_text),
2454        )))
2455    } else if *file == new_text {
2456        // The document is already formatted correctly -- no edits needed.
2457        Ok(None)
2458    } else {
2459        Ok(Some(to_proto::text_edit_vec(&line_index, diff(&file, &new_text))))
2460    }
2461}

2509/// Searches for the directory of a Rust crate given this crate's root file path.
2510///
2511/// # Arguments
2512///
2513/// * `root_file_path`: The path to the root file of the crate.
2514///
2515/// # Returns
2516///
2517/// An `Option` value representing the path to the directory of the crate with the given
2518/// name, if such a crate is found. If no crate with the given name is found, this function
2519/// returns `None`.
2520fn crate_path(root_file_path: &VfsPath) -> Option<VfsPath> {
2521    let mut current_dir = root_file_path.parent();
2522    while let Some(path) = current_dir {
2523        let cargo_toml_path = path.join("../Cargo.toml")?;
2524        if fs::metadata(cargo_toml_path.as_path()?).is_ok() {
2525            let crate_path = cargo_toml_path.parent()?;
2526            return Some(crate_path);
2527        }
2528        current_dir = path.parent();
2529    }
2530    None
2531}

2567pub(crate) fn diff(left: &str, right: &str) -> TextEdit {
2568    use dissimilar::Chunk;
2569
2570    let chunks = dissimilar::diff(left, right);
2571
2572    let mut builder = TextEdit::builder();
2573    let mut pos = TextSize::default();
2574
2575    let mut chunks = chunks.into_iter().peekable();
2576    while let Some(chunk) = chunks.next() {
2577        if let (Chunk::Delete(deleted), Some(&Chunk::Insert(inserted))) = (chunk, chunks.peek()) {
2578            chunks.next().unwrap();
2579            let deleted_len = TextSize::of(deleted);
2580            builder.replace(TextRange::at(pos, deleted_len), inserted.into());
2581            pos += deleted_len;
2582            continue;
2583        }
2584
2585        match chunk {
2586            Chunk::Equal(text) => {
2587                pos += TextSize::of(text);
2588            }
2589            Chunk::Delete(deleted) => {
2590                let deleted_len = TextSize::of(deleted);
2591                builder.delete(TextRange::at(pos, deleted_len));
2592                pos += deleted_len;
2593            }
2594            Chunk::Insert(inserted) => {
2595                builder.insert(pos, inserted.into());
2596            }
2597        }
2598    }
2599    builder.finish()
2600}
