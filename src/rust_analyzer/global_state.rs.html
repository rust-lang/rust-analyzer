144    /// `workspaces` field stores the data we actually use, while the `OpQueue`
145    /// stores the result of the last fetch.
146    ///
147    /// If the fetch (partially) fails, we do not update the current value.
148    ///
149    /// The handling of build data is subtle. We fetch workspace in two phases:
150    ///
151    /// *First*, we run `cargo metadata`, which gives us fast results for
152    /// initial analysis.
153    ///
154    /// *Second*, we run `cargo check` which runs build scripts and compiles
155    /// proc macros.
156    ///
157    /// We need both for the precise analysis, but we want rust-analyzer to be
158    /// at least partially available just after the first phase. That's because
159    /// first phase is much faster, and is much less likely to fail.
160    ///
161    /// This creates a complication -- by the time the second phase completes,
162    /// the results of the first phase could be invalid. That is, while we run
163    /// `cargo check`, the user edits `Cargo.toml`, we notice this, and the new
164    /// `cargo metadata` completes before `cargo check`.
165    ///
166    /// An additional complication is that we want to avoid needless work. When
167    /// the user just adds comments or whitespace to Cargo.toml, we do not want
168    /// to invalidate any salsa caches.
169    pub(crate) workspaces: Arc<Vec<ProjectWorkspace>>,
170    pub(crate) crate_graph_file_dependencies: FxHashSet<vfs::VfsPath>,
171    pub(crate) detached_files: FxHashSet<ManifestPath>,
172
173    // op queues
174    pub(crate) fetch_workspaces_queue: OpQueue<FetchWorkspaceRequest, FetchWorkspaceResponse>,
175    pub(crate) fetch_build_data_queue: OpQueue<(), FetchBuildDataResponse>,
176    pub(crate) fetch_proc_macros_queue: OpQueue<(ChangeWithProcMacros, Vec<ProcMacroPaths>), bool>,
177    pub(crate) prime_caches_queue: OpQueue,
178    pub(crate) discover_workspace_queue: OpQueue,

193    pub(crate) incomplete_crate_graph: bool,

224    pub(crate) fn new(sender: Sender<lsp_server::Message>, config: Config) -> GlobalState {
225        let loader = {
226            let (sender, receiver) = unbounded::<vfs::loader::Message>();
227            let handle: vfs_notify::NotifyHandle = vfs::loader::Handle::spawn(sender);
228            let handle = Box::new(handle) as Box<dyn vfs::loader::Handle>;
229            Handle { handle, receiver }
230        };
231
232        let task_pool = {
233            let (sender, receiver) = unbounded();
234            let handle = TaskPool::new_with_threads(sender, config.main_loop_num_threads());
235            Handle { handle, receiver }
236        };
237        let fmt_pool = {
238            let (sender, receiver) = unbounded();
239            let handle = TaskPool::new_with_threads(sender, 1);
240            Handle { handle, receiver }
241        };
242        let cancellation_pool = thread::Pool::new(1);
243
244        let task_queue = {
245            let (sender, receiver) = unbounded();
246            TaskQueue { sender, receiver }
247        };
248
249        let mut analysis_host = AnalysisHost::new(config.lru_parse_query_capacity());
250        if let Some(capacities) = config.lru_query_capacities_config() {
251            analysis_host.update_lru_capacities(capacities);
252        }
253        let (flycheck_sender, flycheck_receiver) = unbounded();
254        let (test_run_sender, test_run_receiver) = unbounded();
255
256        let (discover_sender, discover_receiver) = unbounded();
257
258        let mut this = GlobalState {
259            sender,
260            req_queue: ReqQueue::default(),
261            task_pool,
262            fmt_pool,
263            cancellation_pool,
264            loader,
265            config: Arc::new(config.clone()),
266            analysis_host,
267            diagnostics: Default::default(),
268            mem_docs: MemDocs::default(),
269            semantic_tokens_cache: Arc::new(Default::default()),
270            shutdown_requested: false,
271            last_reported_status: lsp_ext::ServerStatusParams {
272                health: lsp_ext::Health::Ok,
273                quiescent: true,
274                message: None,
275            },
276            source_root_config: SourceRootConfig::default(),
277            local_roots_parent_map: Arc::new(FxHashMap::default()),
278            config_errors: Default::default(),
279
280            proc_macro_clients: Arc::from_iter([]),
281
282            build_deps_changed: false,
283
284            flycheck: Arc::from_iter([]),
285            flycheck_sender,
286            flycheck_receiver,
287            last_flycheck_error: None,
288
289            test_run_session: None,
290            test_run_sender,
291            test_run_receiver,
292            test_run_remaining_jobs: 0,
293
294            discover_handle: None,
295            discover_sender,
296            discover_receiver,
297
298            fetch_ws_receiver: None,
299
300            vfs: Arc::new(RwLock::new((vfs::Vfs::default(), Default::default()))),
301            vfs_config_version: 0,
302            vfs_progress_config_version: 0,
303            vfs_span: None,
304            vfs_done: true,
305            wants_to_switch: None,
306
307            workspaces: Arc::from(Vec::new()),
308            crate_graph_file_dependencies: FxHashSet::default(),
309            detached_files: FxHashSet::default(),

312            fetch_proc_macros_queue: OpQueue::default(),
313
314            prime_caches_queue: OpQueue::default(),
315            discover_workspace_queue: OpQueue::default(),

318            incomplete_crate_graph: false,
319
320            minicore: MiniCoreRustAnalyzerInternalOnly::default(),
321        };
322        // Apply any required database inputs from the config.
323        this.update_configuration(config);
324        this
325    }

327    pub(crate) fn process_changes(&mut self) -> bool {
328        let _p = span!(Level::INFO, "GlobalState::process_changes").entered();
329        // We cannot directly resolve a change in a ratoml file to a format
330        // that can be used by the config module because config talks
331        // in `SourceRootId`s instead of `FileId`s and `FileId` -> `SourceRootId`
332        // mapping is not ready until `AnalysisHost::apply_changes` has been called.
333        let mut modified_ratoml_files: FxHashMap<FileId, (ChangeKind, vfs::VfsPath)> =
334            FxHashMap::default();
335
336        let mut change = ChangeWithProcMacros::default();
337        let mut guard = self.vfs.write();
338        let changed_files = guard.0.take_changes();
339        if changed_files.is_empty() {
340            return false;
341        }
342
343        let (change, modified_rust_files, workspace_structure_change) =
344            self.cancellation_pool.scoped(|s| {
345                // start cancellation in parallel, this will kick off lru eviction
346                // allowing us to do meaningful work while waiting
347                let analysis_host = AssertUnwindSafe(&mut self.analysis_host);
348                s.spawn(thread::ThreadIntent::LatencySensitive, || {
349                    { analysis_host }.0.request_cancellation()
350                });
351
352                // downgrade to read lock to allow more readers while we are normalizing text
353                let guard = RwLockWriteGuard::downgrade_to_upgradable(guard);
354                let vfs: &Vfs = &guard.0;
355
356                let mut workspace_structure_change = None;
357                // A file was added or deleted
358                let mut has_structure_changes = false;
359                let mut bytes = vec![];
360                let mut modified_rust_files = vec![];
361                for file in changed_files.into_values() {
362                    let vfs_path = vfs.file_path(file.file_id);
363                    if let Some(("rust-analyzer", Some("toml"))) = vfs_path.name_and_extension() {
364                        // Remember ids to use them after `apply_changes`
365                        modified_ratoml_files.insert(file.file_id, (file.kind(), vfs_path.clone()));
366                    }
367
368                    if let Some(path) = vfs_path.as_path() {
369                        has_structure_changes |= file.is_created_or_deleted();
370
371                        if file.is_modified() && path.extension() == Some("rs") {
372                            modified_rust_files.push(file.file_id);
373                        }
374
375                        let additional_files = self
376                            .config
377                            .discover_workspace_config()
378                            .map(|cfg| {
379                                cfg.files_to_watch.iter().map(String::as_str).collect::<Vec<&str>>()
380                            })
381                            .unwrap_or_default();
382
383                        let path = path.to_path_buf();
384                        if file.is_created_or_deleted() {
385                            workspace_structure_change.get_or_insert((path, false)).1 |=
386                                self.crate_graph_file_dependencies.contains(vfs_path);
387                        } else if reload::should_refresh_for_change(
388                            &path,
389                            file.kind(),
390                            &additional_files,
391                        ) {
392                            trace!(?path, kind = ?file.kind(), "refreshing for a change");
393                            workspace_structure_change.get_or_insert((path.clone(), false));
394                        }
395                    }
396
397                    // Clear native diagnostics when their file gets deleted
398                    if !file.exists() {
399                        self.diagnostics.clear_native_for(file.file_id);
400                    }
401
402                    let text = if let vfs::Change::Create(v, _) | vfs::Change::Modify(v, _) =
403                        file.change
404                    {
405                        String::from_utf8(v).ok().map(|text| {
406                            // FIXME: Consider doing normalization in the `vfs` instead? That allows
407                            // getting rid of some locking
408                            let (text, line_endings) = LineEndings::normalize(text);
409                            (text, line_endings)
410                        })
411                    } else {
412                        None
413                    };
414                    // delay `line_endings_map` changes until we are done normalizing the text
415                    // this allows delaying the re-acquisition of the write lock
416                    bytes.push((file.file_id, text));
417                }
418                let (vfs, line_endings_map) = &mut *RwLockUpgradableReadGuard::upgrade(guard);
419                bytes.into_iter().for_each(|(file_id, text)| {
420                    let text = match text {
421                        None => None,
422                        Some((text, line_endings)) => {
423                            line_endings_map.insert(file_id, line_endings);
424                            Some(text)
425                        }
426                    };
427                    change.change_file(file_id, text);
428                });
429                if has_structure_changes {
430                    let roots = self.source_root_config.partition(vfs);
431                    change.set_roots(roots);
432                }
433                (change, modified_rust_files, workspace_structure_change)
434            });
435
436        self.analysis_host.apply_change(change);
437        if !modified_ratoml_files.is_empty()
438            || !self.config.same_source_root_parent_map(&self.local_roots_parent_map)
439        {
440            let config_change = {
441                let _p = span!(Level::INFO, "GlobalState::process_changes/config_change").entered();
442                let user_config_path = (|| {
443                    let mut p = Config::user_config_dir_path()?;
444                    p.push("rust-analyzer.toml");
445                    Some(p)
446                })();
447
448                let user_config_abs_path = user_config_path.as_deref();
449
450                let mut change = ConfigChange::default();
451                let db = self.analysis_host.raw_database();
452
453                // FIXME @alibektas : This is silly. There is no reason to use VfsPaths when there is SourceRoots. But how
454                // do I resolve a "workspace_root" to its corresponding id without having to rely on a cargo.toml's ( or project json etc.) file id?
455                let workspace_ratoml_paths = self
456                    .workspaces
457                    .iter()
458                    .map(|ws| {
459                        VfsPath::from({
460                            let mut p = ws.workspace_root().to_owned();
461                            p.push("rust-analyzer.toml");
462                            p
463                        })
464                    })
465                    .collect_vec();
466
467                for (file_id, (change_kind, vfs_path)) in modified_ratoml_files {
468                    tracing::info!(%vfs_path, ?change_kind, "Processing rust-analyzer.toml changes");
469                    if vfs_path.as_path() == user_config_abs_path {
470                        tracing::info!(%vfs_path, ?change_kind, "Use config rust-analyzer.toml changes");
471                        change.change_user_config(Some(db.file_text(file_id).text(db).clone()));
472                    }
473
474                    // If change has been made to a ratoml file that
475                    // belongs to a non-local source root, we will ignore it.
476                    let source_root_id = db.file_source_root(file_id).source_root_id(db);
477                    let source_root = db.source_root(source_root_id).source_root(db);
478
479                    if !source_root.is_library {
480                        let entry = if workspace_ratoml_paths.contains(&vfs_path) {
481                            tracing::info!(%vfs_path, ?source_root_id, "workspace rust-analyzer.toml changes");
482                            change.change_workspace_ratoml(
483                                source_root_id,
484                                vfs_path.clone(),
485                                Some(db.file_text(file_id).text(db).clone()),
486                            )
487                        } else {
488                            tracing::info!(%vfs_path, ?source_root_id, "crate rust-analyzer.toml changes");
489                            change.change_ratoml(
490                                source_root_id,
491                                vfs_path.clone(),
492                                Some(db.file_text(file_id).text(db).clone()),
493                            )
494                        };
495
496                        if let Some((kind, old_path, old_text)) = entry {
497                            // SourceRoot has more than 1 RATOML files. In this case lexicographically smaller wins.
498                            if old_path < vfs_path {
499                                tracing::error!(
500                                    "Two `rust-analyzer.toml` files were found inside the same crate. {vfs_path} has no effect."
501                                );
502                                // Put the old one back in.
503                                match kind {
504                                    RatomlFileKind::Crate => {
505                                        change.change_ratoml(source_root_id, old_path, old_text);
506                                    }
507                                    RatomlFileKind::Workspace => {
508                                        change.change_workspace_ratoml(
509                                            source_root_id,
510                                            old_path,
511                                            old_text,
512                                        );
513                                    }
514                                }
515                            }
516                        }
517                    } else {
518                        tracing::info!(%vfs_path, "Ignoring library rust-analyzer.toml");
519                    }
520                }
521                change.change_source_root_parent_map(self.local_roots_parent_map.clone());
522                change
523            };
524
525            let (config, e, should_update) = self.config.apply_change(config_change);
526            self.config_errors = e.is_empty().not().then_some(e);
527
528            if should_update {
529                self.update_configuration(config);
530            } else {
531                // No global or client level config was changed. So we can naively replace config.
532                self.config = Arc::new(config);
533            }
534        }
535
536        // FIXME: `workspace_structure_change` is computed from `should_refresh_for_change` which is
537        // path syntax based. That is not sufficient for all cases so we should lift that check out
538        // into a `QueuedTask`, see `handle_did_save_text_document`.
539        // Or maybe instead of replacing that check, kick off a semantic one if the syntactic one
540        // didn't find anything (to make up for the lack of precision).
541        {
542            if !matches!(&workspace_structure_change, Some((.., true))) {
543                _ = self
544                    .deferred_task_queue
545                    .sender
546                    .send(crate::main_loop::QueuedTask::CheckProcMacroSources(modified_rust_files));
547            }
548            // FIXME: ideally we should only trigger a workspace fetch for non-library changes
549            // but something's going wrong with the source root business when we add a new local
550            // crate see https://github.com/rust-lang/rust-analyzer/issues/13029
551            if let Some((path, force_crate_graph_reload)) = workspace_structure_change {
552                let _p = span!(Level::INFO, "GlobalState::process_changes/ws_structure_change")
553                    .entered();
554                self.enqueue_workspace_fetch(path, force_crate_graph_reload);
555            }
556        }
557
558        true
559    }

561    pub(crate) fn snapshot(&self) -> GlobalStateSnapshot {
562        GlobalStateSnapshot {
563            config: Arc::clone(&self.config),
564            workspaces: Arc::clone(&self.workspaces),
565            analysis: self.analysis_host.analysis(),
566            vfs: Arc::clone(&self.vfs),
567            minicore: self.minicore.clone(),
568            check_fixes: Arc::clone(&self.diagnostics.check_fixes),
569            mem_docs: self.mem_docs.clone(),
570            semantic_tokens_cache: Arc::clone(&self.semantic_tokens_cache),
571            proc_macros_loaded: !self.config.expand_proc_macros()
572                || self.fetch_proc_macros_queue.last_op_result().copied().unwrap_or(false),
573            flycheck: self.flycheck.clone(),
574        }

613    pub(crate) fn respond(&mut self, response: lsp_server::Response) {
614        if let Some((method, start)) = self.req_queue.incoming.complete(&response.id) {
615            if let Some(err) = &response.error
616                && err.message.starts_with("server panicked")
617            {
618                self.poke_rust_analyzer_developer(format!("{}, check the log", err.message));
619            }
620
621            let duration = start.elapsed();
622            tracing::debug!(name: "message response", method, %response.id, duration = format_args!("{:0.2?}", duration));
623            self.send(response.into());
624        }
625    }

627    pub(crate) fn cancel(&mut self, request_id: lsp_server::RequestId) {
628        if let Some(response) = self.req_queue.incoming.cancel(request_id) {
629            self.send(response.into());
630        }
631    }

642    pub(crate) fn publish_diagnostics(
643        &mut self,
644        uri: Url,
645        version: Option<i32>,
646        mut diagnostics: Vec<lsp_types::Diagnostic>,
647    ) {
648        // We put this on a separate thread to avoid blocking the main thread with serialization work
649        self.task_pool.handle.spawn_with_sender(stdx::thread::ThreadIntent::Worker, {
650            let sender = self.sender.clone();
651            move |_| {
652                // VSCode assumes diagnostic messages to be non-empty strings, so we need to patch
653                // empty diagnostics. Neither the docs of VSCode nor the LSP spec say whether
654                // diagnostic messages are actually allowed to be empty or not and patching this
655                // in the VSCode client does not work as the assertion happens in the protocol
656                // conversion. So this hack is here to stay, and will be considered a hack
657                // until the LSP decides to state that empty messages are allowed.
658
659                // See https://github.com/rust-lang/rust-analyzer/issues/11404
660                // See https://github.com/rust-lang/rust-analyzer/issues/13130
661                let patch_empty = |message: &mut String| {
662                    if message.is_empty() {
663                        " ".clone_into(message);
664                    }
665                };
666
667                for d in &mut diagnostics {
668                    patch_empty(&mut d.message);
669                    if let Some(dri) = &mut d.related_information {
670                        for dri in dri {
671                            patch_empty(&mut dri.message);
672                        }
673                    }
674                }
675
676                let not = lsp_server::Notification::new(
677                    <lsp_types::notification::PublishDiagnostics as lsp_types::notification::Notification>::METHOD.to_owned(),
678                    lsp_types::PublishDiagnosticsParams { uri, diagnostics, version },
679                );
680                _ = sender.send(not.into());
681            }
682        });
683    }

685    pub(crate) fn check_workspaces_msrv(&self) -> impl Iterator<Item = String> + '_ {
686        self.workspaces.iter().filter_map(|ws| {
687            if let Some(toolchain) = &ws.toolchain
688                && *toolchain < crate::MINIMUM_SUPPORTED_TOOLCHAIN_VERSION
689            {
690                return Some(format!(
691                    "Workspace `{}` is using an outdated toolchain version `{}` but \
692                        rust-analyzer only supports `{}` and higher.\n\
693                        Consider using the rust-analyzer rustup component for your toolchain or
694                        upgrade your toolchain to a supported version.\n\n",
695                    ws.manifest_or_root(),
696                    toolchain,
697                    crate::MINIMUM_SUPPORTED_TOOLCHAIN_VERSION,
698                ));
699            }
700            None
701        })
702    }

704    fn enqueue_workspace_fetch(&mut self, path: AbsPathBuf, force_crate_graph_reload: bool) {
705        let already_requested = self.fetch_workspaces_queue.op_requested()
706            && !self.fetch_workspaces_queue.op_in_progress();
707        if self.fetch_ws_receiver.is_none() && already_requested {
708            // Don't queue up a new fetch request if we already have done so
709            // Otherwise we will re-fetch in quick succession which is unnecessary
710            // Note though, that if one is already in progress, we *want* to re-queue
711            // as the in-progress fetch might not have the latest changes in it anymore
712            // FIXME: We should cancel the in-progress fetch here
713            return;
714        }
715
716        self.fetch_ws_receiver = Some((
717            crossbeam_channel::after(Duration::from_millis(100)),
718            FetchWorkspaceRequest { path: Some(path), force_crate_graph_reload },
719        ));
720    }

722    pub(crate) fn debounce_workspace_fetch(&mut self) {
723        if let Some((fetch_receiver, _)) = &mut self.fetch_ws_receiver {
724            *fetch_receiver = crossbeam_channel::after(Duration::from_millis(100));
725        }
726    }

782    pub(crate) fn target_spec_for_crate(&self, crate_id: Crate) -> Option<TargetSpec> {
783        let file_id = self.analysis.crate_root(crate_id).ok()?;
784        let path = self.vfs_read().file_path(file_id).clone();
785        let path = path.as_path()?;
786
787        for workspace in self.workspaces.iter() {
788            match &workspace.kind {
789                ProjectWorkspaceKind::Cargo { cargo, .. }
790                | ProjectWorkspaceKind::DetachedFile { cargo: Some((cargo, _, _)), .. } => {
791                    let Some(target_idx) = cargo.target_by_root(path) else {
792                        continue;
793                    };
794
795                    let target_data = &cargo[target_idx];
796                    let package_data = &cargo[target_data.package];
797
798                    return Some(TargetSpec::Cargo(CargoTargetSpec {
799                        workspace_root: cargo.workspace_root().to_path_buf(),
800                        cargo_toml: package_data.manifest.clone(),
801                        crate_id,
802                        package: cargo.package_flag(package_data),
803                        package_id: package_data.id.clone(),
804                        target: target_data.name.clone(),
805                        target_kind: target_data.kind,
806                        required_features: target_data.required_features.clone(),
807                        features: package_data.features.keys().cloned().collect(),
808                        sysroot_root: workspace.sysroot.root().map(ToOwned::to_owned),
809                    }));
810                }
811                ProjectWorkspaceKind::Json(project) => {
812                    let Some(krate) = project.crate_by_root(path) else {
813                        continue;
814                    };
815                    let Some(build) = krate.build else {
816                        continue;
817                    };
818
819                    return Some(TargetSpec::ProjectJson(ProjectJsonTargetSpec {
820                        label: build.label,
821                        target_kind: build.target_kind,
822                        shell_runnables: project.runnables().to_owned(),
823                    }));
824                }
825                ProjectWorkspaceKind::DetachedFile { .. } => {}
826            };
827        }
828
829        None
830    }

834        package: &Arc<PackageId>,
835    ) -> Option<FxHashSet<Arc<PackageId>>> {
