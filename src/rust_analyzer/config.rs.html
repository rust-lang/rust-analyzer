49// Conventions for configuration keys to preserve maximal extendability without breakage:
50//  - Toggles (be it binary true/false or with more options in-between) should almost always suffix as `_enable`
51//    This has the benefit of namespaces being extensible, and if the suffix doesn't fit later it can be changed without breakage.
52//  - In general be wary of using the namespace of something verbatim, it prevents us from adding subkeys in the future
53//  - Don't use abbreviations unless really necessary
54//  - foo_command = overrides the subcommand, foo_overrideCommand allows full overwriting, extra args only applies for foo_command

472        /// Enables automatic discovery of projects using [`DiscoverWorkspaceConfig::command`].
473        ///
474        /// [`DiscoverWorkspaceConfig`] also requires setting `progress_label` and `files_to_watch`.
475        /// `progress_label` is used for the title in progress indicators, whereas `files_to_watch`
476        /// is used to determine which build system-specific files should be watched in order to
477        /// reload rust-analyzer.
478        ///
479        /// Below is an example of a valid configuration:
480        /// ```json
481        /// "rust-analyzer.workspace.discoverConfig": {
482        ///     "command": [
483        ///         "rust-project",
484        ///         "develop-json"
485        ///     ],
486        ///     "progressLabel": "rust-analyzer",
487        ///     "filesToWatch": [
488        ///         "BUCK"
489        ///     ]
490        /// }
491        /// ```
492        ///
493        /// ## On `DiscoverWorkspaceConfig::command`
494        ///
495        /// **Warning**: This format is provisional and subject to change.
496        ///
497        /// [`DiscoverWorkspaceConfig::command`] *must* return a JSON object corresponding to
498        /// `DiscoverProjectData::Finished`:
499        ///
500        /// ```norun
501        /// #[derive(Debug, Clone, Deserialize, Serialize)]
502        /// #[serde(tag = "kind")]
503        /// #[serde(rename_all = "snake_case")]
504        /// enum DiscoverProjectData {
505        ///     Finished { buildfile: Utf8PathBuf, project: ProjectJsonData },
506        ///     Error { error: String, source: Option<String> },
507        ///     Progress { message: String },
508        /// }
509        /// ```
510        ///
511        /// As JSON, `DiscoverProjectData::Finished` is:
512        ///
513        /// ```json
514        /// {
515        ///     // the internally-tagged representation of the enum.
516        ///     "kind": "finished",
517        ///     // the file used by a non-Cargo build system to define
518        ///     // a package or target.
519        ///     "buildfile": "rust-analyzer/BUILD",
520        ///     // the contents of a rust-project.json, elided for brevity
521        ///     "project": {
522        ///         "sysroot": "foo",
523        ///         "crates": []
524        ///     }
525        /// }
526        /// ```
527        ///
528        /// It is encouraged, but not required, to use the other variants on `DiscoverProjectData`
529        /// to provide a more polished end-user experience.
530        ///
531        /// `DiscoverWorkspaceConfig::command` may *optionally* include an `{arg}`, which will be
532        /// substituted with the JSON-serialized form of the following enum:
533        ///
534        /// ```norun
535        /// #[derive(PartialEq, Clone, Debug, Serialize)]
536        /// #[serde(rename_all = "camelCase")]
537        /// pub enum DiscoverArgument {
538        ///    Path(AbsPathBuf),
539        ///    Buildfile(AbsPathBuf),
540        /// }
541        /// ```
542        ///
543        /// The JSON representation of `DiscoverArgument::Path` is:
544        ///
545        /// ```json
546        /// {
547        ///     "path": "src/main.rs"
548        /// }
549        /// ```
550        ///
551        /// Similarly, the JSON representation of `DiscoverArgument::Buildfile` is:
552        ///
553        /// ```json
554        /// {
555        ///     "buildfile": "BUILD"
556        /// }
557        /// ```
558        ///
559        /// `DiscoverArgument::Path` is used to find and generate a `rust-project.json`, and
560        /// therefore, a workspace, whereas `DiscoverArgument::buildfile` is used to to update an
561        /// existing workspace. As a reference for implementors, buck2's `rust-project` will likely
562        /// be useful: https://github.com/facebook/buck2/tree/main/integrations/rust-project.
563        workspace_discoverConfig: Option<DiscoverWorkspaceConfig> = None,
564    }

1074    /// Changes made to client and global configurations will partially not be reflected even after `.apply_change()` was called.
1075    /// The return tuple's bool component signals whether the `GlobalState` should call its `update_configuration()` method.
1076    fn apply_change_with_sink(&self, change: ConfigChange) -> (Config, bool) {
1077        let mut config = self.clone();
1078        config.validation_errors = ConfigErrors::default();
1079
1080        let mut should_update = false;
1081
1082        if let Some(change) = change.user_config_change {
1083            tracing::info!("updating config from user config toml: {:#}", change);
1084            if let Ok(table) = toml::from_str(&change) {
1085                let mut toml_errors = vec![];
1086                validate_toml_table(
1087                    GlobalWorkspaceLocalConfigInput::FIELDS,
1088                    &table,
1089                    &mut String::new(),
1090                    &mut toml_errors,
1091                );
1092                config.user_config = Some((
1093                    GlobalWorkspaceLocalConfigInput::from_toml(table, &mut toml_errors),
1094                    ConfigErrors(
1095                        toml_errors
1096                            .into_iter()
1097                            .map(|(a, b)| ConfigErrorInner::Toml { config_key: a, error: b })
1098                            .map(Arc::new)
1099                            .collect(),
1100                    ),
1101                ));
1102                should_update = true;
1103            }
1104        }
1105
1106        if let Some(mut json) = change.client_config_change {
1107            tracing::info!("updating config from JSON: {:#}", json);
1108
1109            if !(json.is_null() || json.as_object().is_some_and(|it| it.is_empty())) {
1110                let detached_files = get_field_json::<Vec<Utf8PathBuf>>(
1111                    &mut json,
1112                    &mut Vec::new(),
1113                    "detachedFiles",
1114                    None,
1115                )

1292    pub fn apply_change(&self, change: ConfigChange) -> (Config, ConfigErrors, bool) {

3418            return None;
3419        }
3420        s.parse().ok()
3421    }
3422
3423    if pointer.is_empty() {
3424        return None;
3425    }
3426    if !pointer.starts_with('/') {
3427        return None;
3428    }
3429    let mut parts = pointer.split('/').skip(1);
3430    let first = parts.next()?;
3431    let init = toml.get(first)?;
3432    parts.map(|x| x.replace("~1", "/").replace("~0", "~")).try_fold(init, |target, token| {
3433        match target {
3434            toml::Value::Table(table) => table.get(&token),
3435            toml::Value::Array(list) => parse_index(&token).and_then(move |x| list.get(x)),

3438    })

3941fn validate_toml_table(
3942    known_ptrs: &[&[&'static str]],
3943    toml: &toml::Table,
3944    ptr: &mut String,
3945    error_sink: &mut Vec<(String, toml::de::Error)>,
3946) {
3947    let verify = |ptr: &String| known_ptrs.iter().any(|ptrs| ptrs.contains(&ptr.as_str()));
3948
3949    let l = ptr.len();
3950    for (k, v) in toml {
3951        if !ptr.is_empty() {
3952            ptr.push('_');
3953        }
3954        ptr.push_str(k);
3955
3956        match v {
3957            // This is a table config, any entry in it is therefore valid
3958            toml::Value::Table(_) if verify(ptr) => (),
3959            toml::Value::Table(table) => validate_toml_table(known_ptrs, table, ptr, error_sink),
3960            _ if !verify(ptr) => error_sink
3961                .push((ptr.replace('_', "/"), toml::de::Error::custom("unexpected field"))),
3962            _ => (),
3963        }
3964
3965        ptr.truncate(l);
3966    }
3967}
