21#[proc_macro_attribute]
22pub fn query_group(args: TokenStream, input: TokenStream) -> TokenStream {
23    match query_group_impl(args, input.clone()) {
24        Ok(tokens) => tokens,
25        Err(e) => token_stream_with_error(input, e),
26    }
27}

41struct SalsaAttr {
42    name: String,
43    tts: TokenStream,
44    span: Span,
45}
46
47impl std::fmt::Debug for SalsaAttr {
48    fn fmt(&self, fmt: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
49        write!(fmt, "{:?}", self.name)
50    }
51}
52
53impl TryFrom<syn::Attribute> for SalsaAttr {
54    type Error = syn::Attribute;
55
56    fn try_from(attr: syn::Attribute) -> Result<SalsaAttr, syn::Attribute> {
57        if is_not_salsa_attr_path(attr.path()) {
58            return Err(attr);
59        }
60
61        let span = attr.span();
62
63        let name = attr.path().segments[1].ident.to_string();
64        let tts = match attr.meta {
65            syn::Meta::Path(path) => path.into_token_stream(),
66            syn::Meta::List(ref list) => {
67                let tts = list
68                    .into_token_stream()
69                    .into_iter()
70                    .skip(attr.path().to_token_stream().into_iter().count());
71                proc_macro2::TokenStream::from_iter(tts)
72            }
73            syn::Meta::NameValue(nv) => nv.into_token_stream(),
74        }
75        .into();
76
77        Ok(SalsaAttr { name, tts, span })
78    }
79}
80
81fn is_not_salsa_attr_path(path: &syn::Path) -> bool {
82    path.segments.first().map(|s| s.ident != "salsa").unwrap_or(true) || path.segments.len() != 2
83}
84
85fn filter_attrs(attrs: Vec<Attribute>) -> (Vec<Attribute>, Vec<SalsaAttr>) {
86    let mut other = vec![];
87    let mut salsa = vec![];
88    // Leave non-salsa attributes untouched. These are
89    // attributes that don't start with `salsa::` or don't have
90    // exactly two segments in their path.
91    for attr in attrs {
92        match SalsaAttr::try_from(attr) {
93            Ok(it) => salsa.push(it),
94            Err(it) => other.push(it),
95        }
96    }
97    (other, salsa)
98}

100#[derive(Debug, Clone, PartialEq, Eq)]
101enum QueryKind {
102    Input,
103    Tracked,
104    TrackedWithSalsaStruct,
105    Transparent,
106}

108#[derive(Default, Debug, Clone)]
109struct Cycle {
110    cycle_fn: Option<(syn::Ident, Path)>,
111    cycle_initial: Option<(syn::Ident, Path)>,
112    cycle_result: Option<(syn::Ident, Path)>,
113}
114
115impl Parse for Cycle {
116    fn parse(input: ParseStream<'_>) -> syn::Result<Self> {
117        let options = Punctuated::<Option, Token![,]>::parse_terminated(input)?;
118        let mut cycle_fn = None;
119        let mut cycle_initial = None;
120        let mut cycle_result = None;
121        for option in options {
122            let name = option.name.to_string();
123            match &*name {
124                "cycle_fn" => {
125                    if cycle_fn.is_some() {
126                        return Err(syn::Error::new_spanned(&option.name, "duplicate option"));
127                    }
128                    cycle_fn = Some((option.name, option.value));
129                }
130                "cycle_initial" => {
131                    if cycle_initial.is_some() {
132                        return Err(syn::Error::new_spanned(&option.name, "duplicate option"));
133                    }
134                    cycle_initial = Some((option.name, option.value));
135                }
136                "cycle_result" => {
137                    if cycle_result.is_some() {
138                        return Err(syn::Error::new_spanned(&option.name, "duplicate option"));
139                    }
140                    cycle_result = Some((option.name, option.value));
141                }
142                _ => {
143                    return Err(syn::Error::new_spanned(
144                        &option.name,
145                        "unknown cycle option. Accepted values: `cycle_result`, `cycle_fn`, `cycle_initial`",
146                    ));
147                }
148            }
149        }
150        return Ok(Self { cycle_fn, cycle_initial, cycle_result });
151
152        struct Option {
153            name: syn::Ident,
154            value: Path,
155        }
156
157        impl Parse for Option {
158            fn parse(input: ParseStream<'_>) -> syn::Result<Self> {
159                let name = input.parse()?;
160                input.parse::<Token![=]>()?;
161                let value = input.parse()?;
162                Ok(Self { name, value })
163            }
164        }
165    }
166}

168pub(crate) fn query_group_impl(
169    _args: proc_macro::TokenStream,
170    input: proc_macro::TokenStream,
171) -> Result<proc_macro::TokenStream, syn::Error> {
172    let mut item_trait = syn::parse::<ItemTrait>(input)?;
173
174    let supertraits = &item_trait.supertraits;
175
176    let db_attr: Attribute = parse_quote! {
177        #[salsa_macros::db]
178    };
179    item_trait.attrs.push(db_attr);
180
181    let trait_name_ident = &item_trait.ident.clone();
182    let input_struct_name = format_ident!("{}Data", trait_name_ident);
183    let create_data_ident = format_ident!("create_data_{}", trait_name_ident);
184
185    let mut input_struct_fields: Vec<InputStructField> = vec![];
186    let mut trait_methods = vec![];
187    let mut setter_trait_methods = vec![];
188    let mut lookup_signatures = vec![];
189
190    for item in &mut item_trait.items {
191        if let syn::TraitItem::Fn(method) = item {
192            let method_name = &method.sig.ident;
193            let signature = &method.sig;
194
195            let (_attrs, salsa_attrs) = filter_attrs(method.attrs.clone());
196
197            let mut query_kind = QueryKind::TrackedWithSalsaStruct;
198            let mut invoke = None;
199            let mut cycle = None;
200            let mut lru = None;
201
202            let params: Vec<FnArg> = signature.inputs.clone().into_iter().collect();
203            let pat_and_tys = params
204                .into_iter()
205                .filter(|fn_arg| matches!(fn_arg, FnArg::Typed(_)))
206                .map(|fn_arg| match fn_arg {
207                    FnArg::Typed(pat_type) => pat_type,
208                    FnArg::Receiver(_) => unreachable!("this should have been filtered out"),
209                })
210                .collect::<Vec<syn::PatType>>();
211
212            for SalsaAttr { name, tts, span } in salsa_attrs {
213                match name.as_str() {
214                    "cycle" => {
215                        let c = syn::parse::<Parenthesized<Cycle>>(tts)?;
216                        cycle = Some(c.0);
217                    }
218                    "input" => {
219                        if !pat_and_tys.is_empty() {
220                            return Err(syn::Error::new(
221                                span,
222                                "input methods cannot have a parameter",
223                            ));
224                        }
225                        query_kind = QueryKind::Input;
226                    }
227                    "invoke_interned" => {
228                        let path = syn::parse::<Parenthesized<Path>>(tts)?;
229                        invoke = Some(path.0.clone());
230                        query_kind = QueryKind::Tracked;
231                    }
232                    "invoke" => {
233                        let path = syn::parse::<Parenthesized<Path>>(tts)?;
234                        invoke = Some(path.0.clone());
235                        if query_kind != QueryKind::Transparent {
236                            query_kind = QueryKind::TrackedWithSalsaStruct;
237                        }
238                    }
239                    "tracked" if method.default.is_some() => {
240                        query_kind = QueryKind::TrackedWithSalsaStruct;
241                    }
242                    "lru" => {
243                        let lru_count = syn::parse::<Parenthesized<syn::LitInt>>(tts)?;
244                        let lru_count = lru_count.0.base10_parse::<u32>()?;
245
246                        lru = Some(lru_count);
247                    }
248                    "transparent" => {
249                        query_kind = QueryKind::Transparent;
250                    }
251                    _ => return Err(syn::Error::new(span, format!("unknown attribute `{name}`"))),
252                }
253            }
254
255            let syn::ReturnType::Type(_, return_ty) = signature.output.clone() else {
256                return Err(syn::Error::new(signature.span(), "Queries must have a return type"));
257            };
258
259            if let syn::Type::Path(ref ty_path) = *return_ty
260                && matches!(query_kind, QueryKind::Input)
261            {
262                let field = InputStructField {
263                    name: method_name.to_token_stream(),
264                    ty: ty_path.path.to_token_stream(),
265                };
266
267                input_struct_fields.push(field);
268            }
269
270            if let Some(block) = &mut method.default {
271                SelfToDbRewriter.visit_block_mut(block);
272            }
273
274            match (query_kind, invoke) {
275                // input
276                (QueryKind::Input, None) => {
277                    let query = InputQuery {
278                        signature: method.sig.clone(),
279                        create_data_ident: create_data_ident.clone(),
280                    };
281                    let value = Queries::InputQuery(query);
282                    trait_methods.push(value);
283
284                    let setter = InputSetter {
285                        signature: method.sig.clone(),
286                        return_type: *return_ty.clone(),
287                        create_data_ident: create_data_ident.clone(),
288                    };
289                    setter_trait_methods.push(SetterKind::Plain(setter));
290
291                    let setter = InputSetterWithDurability {
292                        signature: method.sig.clone(),
293                        return_type: *return_ty.clone(),
294                        create_data_ident: create_data_ident.clone(),
295                    };
296                    setter_trait_methods.push(SetterKind::WithDurability(setter));
297                }
298                // tracked function. it might have an invoke, or might not.
299                (QueryKind::Tracked, invoke) => {
300                    let method = TrackedQuery {
301                        trait_name: trait_name_ident.clone(),
302                        generated_struct: Some(GeneratedInputStruct {
303                            input_struct_name: input_struct_name.clone(),
304                            create_data_ident: create_data_ident.clone(),
305                        }),
306                        signature: signature.clone(),
307                        pat_and_tys: pat_and_tys.clone(),
308                        invoke,
309                        cycle,
310                        lru,
311                        default: method.default.take(),
312                    };
313
314                    trait_methods.push(Queries::TrackedQuery(method));
315                }
316                (QueryKind::TrackedWithSalsaStruct, invoke) => {
317                    let method = TrackedQuery {
318                        trait_name: trait_name_ident.clone(),
319                        generated_struct: None,
320                        signature: signature.clone(),
321                        pat_and_tys: pat_and_tys.clone(),
322                        invoke,
323                        cycle,
324                        lru,
325                        default: method.default.take(),
326                    };
327
328                    trait_methods.push(Queries::TrackedQuery(method))
329                }
330                (QueryKind::Transparent, invoke) => {
331                    let method = Transparent {
332                        signature: method.sig.clone(),
333                        pat_and_tys: pat_and_tys.clone(),
334                        invoke,
335                        default: method.default.take(),
336                    };
337                    trait_methods.push(Queries::Transparent(method));
338                }
339                (QueryKind::Input, Some(path)) => {
340                    return Err(syn::Error::new(
341                        path.span(),
342                        "Inputs cannot be used with an `#[invoke]`".to_string(),
343                    ));
344                }
345            }
346        }
347    }
348
349    let fields = input_struct_fields
350        .into_iter()
351        .map(|input| {
352            let name = input.name;
353            let ret = input.ty;
354            quote! { #name: Option<#ret> }
355        })
356        .collect::<Vec<proc_macro2::TokenStream>>();
357
358    let input_struct = quote! {
359        #[salsa_macros::input]
360        pub(crate) struct #input_struct_name {
361            #(#fields),*
362        }
363    };
364
365    let field_params = std::iter::repeat_n(quote! { None }, fields.len())
366        .collect::<Vec<proc_macro2::TokenStream>>();
367
368    let create_data_method = quote! {
369        #[allow(non_snake_case)]
370        #[salsa_macros::tracked]
371        fn #create_data_ident(db: &dyn #trait_name_ident) -> #input_struct_name {
372            #input_struct_name::new(db, #(#field_params),*)
373        }
374    };
375
376    let mut setter_signatures = vec![];
377    let mut setter_methods = vec![];
378    for trait_item in setter_trait_methods
379        .iter()
380        .map(|method| method.to_token_stream())
381        .map(|tokens| syn::parse2::<syn::TraitItemFn>(tokens).unwrap())
382    {
383        let mut methods_sans_body = trait_item.clone();
384        methods_sans_body.default = None;
385        methods_sans_body.semi_token = Some(syn::Token![;](trait_item.span()));
386
387        setter_signatures.push(TraitItem::Fn(methods_sans_body));
388        setter_methods.push(TraitItem::Fn(trait_item));
389    }
390
391    item_trait.items.append(&mut setter_signatures);
392    item_trait.items.append(&mut lookup_signatures);
393
394    let trait_impl = quote! {
395        #[salsa_macros::db]
396        impl<DB> #trait_name_ident for DB
397        where
398            DB: #supertraits,
399        {
400            #(#trait_methods)*
401
402            #(#setter_methods)*
403        }
404    };
405    RemoveAttrsFromTraitMethods.visit_item_trait_mut(&mut item_trait);
406
407    let out = quote! {
408        #item_trait
409
410        #trait_impl
411
412        #input_struct
413
414        #create_data_method
415    }
416    .into();
417
418    Ok(out)
419}

421/// Parenthesis helper
422pub(crate) struct Parenthesized<T>(pub(crate) T);
423
424impl<T> syn::parse::Parse for Parenthesized<T>
425where
426    T: syn::parse::Parse,
427{
428    fn parse(input: ParseStream<'_>) -> syn::Result<Self> {
429        let content;
430        syn::parenthesized!(content in input);
431        content.parse::<T>().map(Parenthesized)
432    }
433}
434
435struct RemoveAttrsFromTraitMethods;
436
437impl VisitMut for RemoveAttrsFromTraitMethods {
438    fn visit_item_trait_mut(&mut self, i: &mut syn::ItemTrait) {
439        for item in &mut i.items {
440            if let TraitItem::Fn(trait_item_fn) = item {
441                trait_item_fn.attrs = vec![];
442            }
443        }
444    }
445}

447pub(crate) fn token_stream_with_error(mut tokens: TokenStream, error: syn::Error) -> TokenStream {
448    tokens.extend(TokenStream::from(error.into_compile_error()));
449    tokens
450}
