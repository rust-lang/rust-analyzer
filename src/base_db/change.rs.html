1//! Defines a unit of change that can applied to the database to get the next
2//! state. Changes are transactional.
3
4use std::fmt;
5
6use rustc_hash::FxHashSet;
7use salsa::{Durability, Setter as _};
8use triomphe::Arc;
9use vfs::FileId;
10
11use crate::{
12    CrateGraphBuilder, CratesIdMap, LibraryRoots, LocalRoots, RootQueryDb, SourceRoot, SourceRootId,
13};
14
15/// Encapsulate a bunch of raw `.set` calls on the database.
16#[derive(Default)]
17pub struct FileChange {
18    pub roots: Option<Vec<SourceRoot>>,
19    pub files_changed: Vec<(FileId, Option<String>)>,
20    pub crate_graph: Option<CrateGraphBuilder>,
21}
22
23impl fmt::Debug for FileChange {
24    fn fmt(&self, fmt: &mut fmt::Formatter<'_>) -> fmt::Result {
25        let mut d = fmt.debug_struct("Change");
26        if let Some(roots) = &self.roots {
27            d.field("roots", roots);
28        }
29        if !self.files_changed.is_empty() {
30            d.field("files_changed", &self.files_changed.len());
31        }
32        if self.crate_graph.is_some() {
33            d.field("crate_graph", &self.crate_graph);
34        }
35        d.finish()
36    }
37}
38
39impl FileChange {
40    pub fn set_roots(&mut self, roots: Vec<SourceRoot>) {
41        self.roots = Some(roots);
42    }
43
44    pub fn change_file(&mut self, file_id: FileId, new_text: Option<String>) {
45        self.files_changed.push((file_id, new_text))
46    }
47
48    pub fn set_crate_graph(&mut self, graph: CrateGraphBuilder) {
49        self.crate_graph = Some(graph);
50    }
51
52    pub fn apply(self, db: &mut dyn RootQueryDb) -> Option<CratesIdMap> {
53        let _p = tracing::info_span!("FileChange::apply").entered();
54        if let Some(roots) = self.roots {
55            let mut local_roots = FxHashSet::default();
56            let mut library_roots = FxHashSet::default();
57            for (idx, root) in roots.into_iter().enumerate() {
58                let root_id = SourceRootId(idx as u32);
59                if root.is_library {
60                    library_roots.insert(root_id);
61                } else {
62                    local_roots.insert(root_id);
63                }
64                let durability = source_root_durability(&root);
65                for file_id in root.iter() {
66                    db.set_file_source_root_with_durability(file_id, root_id, durability);
67                }
68
69                db.set_source_root_with_durability(root_id, Arc::new(root), durability);
70            }
71            LocalRoots::get(db).set_roots(db).to(local_roots);
72            LibraryRoots::get(db).set_roots(db).to(library_roots);
73        }
74
75        for (file_id, text) in self.files_changed {
76            let source_root_id = db.file_source_root(file_id);
77            let source_root = db.source_root(source_root_id.source_root_id(db));
78
79            let durability = file_text_durability(&source_root.source_root(db));
80            // XXX: can't actually remove the file, just reset the text
81            let text = text.unwrap_or_default();
82            db.set_file_text_with_durability(file_id, &text, durability)
83        }
84
85        if let Some(crate_graph) = self.crate_graph {
86            return Some(crate_graph.set_in_db(db));
87        }
88        None
89    }
90}
91
92fn source_root_durability(source_root: &SourceRoot) -> Durability {
93    if source_root.is_library { Durability::MEDIUM } else { Durability::LOW }
94}
95
96fn file_text_durability(source_root: &SourceRoot) -> Durability {
97    if source_root.is_library { Durability::HIGH } else { Durability::LOW }
98}
