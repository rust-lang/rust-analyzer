29#[derive(Debug, Clone, PartialEq, Eq, Hash)]
30pub enum ProcMacroLoadingError {
31    Disabled,
32    FailedToBuild,
33    ExpectedProcMacroArtifact,
34    MissingDylibPath,
35    NotYetBuilt,
36    NoProcMacros,
37    ProcMacroSrvError(Box<str>),
38}
39impl ProcMacroLoadingError {
40    pub fn is_hard_error(&self) -> bool {
41        match self {
42            ProcMacroLoadingError::Disabled | ProcMacroLoadingError::NotYetBuilt => false,
43            ProcMacroLoadingError::ExpectedProcMacroArtifact
44            | ProcMacroLoadingError::FailedToBuild
45            | ProcMacroLoadingError::MissingDylibPath
46            | ProcMacroLoadingError::NoProcMacros
47            | ProcMacroLoadingError::ProcMacroSrvError(_) => true,
48        }
49    }
50}
51
52impl Error for ProcMacroLoadingError {}
53impl fmt::Display for ProcMacroLoadingError {
54    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
55        match self {
56            ProcMacroLoadingError::ExpectedProcMacroArtifact => {
57                write!(f, "proc-macro crate did not build proc-macro artifact")
58            }
59            ProcMacroLoadingError::Disabled => write!(f, "proc-macro expansion is disabled"),
60            ProcMacroLoadingError::FailedToBuild => write!(f, "proc-macro failed to build"),
61            ProcMacroLoadingError::MissingDylibPath => {
62                write!(
63                    f,
64                    "proc-macro crate built but the dylib path is missing, this indicates a problem with your build system."
65                )
66            }
67            ProcMacroLoadingError::NotYetBuilt => write!(f, "proc-macro not yet built"),
68            ProcMacroLoadingError::NoProcMacros => {
69                write!(f, "proc macro library has no proc macros")
70            }
71            ProcMacroLoadingError::ProcMacroSrvError(msg) => {
72                write!(f, "proc macro server error: {msg}")
73            }
74        }
75    }

81/// Files are grouped into source roots. A source root is a directory on the
82/// file systems which is watched for changes. Typically it corresponds to a
83/// Rust crate. Source roots *might* be nested: in this case, a file belongs to
84/// the nearest enclosing source root. Paths to files are always relative to a
85/// source root, and the analyzer does not know the root path of the source root at
86/// all. So, a file from one source root can't refer to a file in another source
87/// root by path.
88#[derive(Clone, Debug, PartialEq, Eq)]
89pub struct SourceRoot {
90    /// Sysroot or crates.io library.
91    ///
92    /// Libraries are considered mostly immutable, this assumption is used to
93    /// optimize salsa's query structure
94    pub is_library: bool,
95    file_set: FileSet,
96}
97
98impl SourceRoot {
99    pub fn new_local(file_set: FileSet) -> SourceRoot {
100        SourceRoot { is_library: false, file_set }
101    }
102
103    pub fn new_library(file_set: FileSet) -> SourceRoot {
104        SourceRoot { is_library: true, file_set }
105    }
106
107    pub fn path_for_file(&self, file: &FileId) -> Option<&VfsPath> {
108        self.file_set.path_for_file(file)
109    }
110
111    pub fn file_for_path(&self, path: &VfsPath) -> Option<&FileId> {
112        self.file_set.file_for_path(path)
113    }
114
115    pub fn resolve_path(&self, path: AnchoredPath<'_>) -> Option<FileId> {
116        self.file_set.resolve_path(path)
117    }
118
119    pub fn iter(&self) -> impl Iterator<Item = FileId> + '_ {
120        self.file_set.iter()
121    }
122}

326/// The crate data from which we derive the `Crate`.
327///
328/// We want this to contain as little data as possible, because if it contains dependencies and
329/// something changes, this crate and all of its dependencies ids are invalidated, which causes
330/// pretty much everything to be recomputed. If the crate id is not invalidated, only this crate's
331/// information needs to be recomputed.
332///
333/// *Most* different crates have different root files (actually, pretty much all of them).
334/// Still, it is possible to have crates distinguished by other factors (e.g. dependencies).
335/// So we store only the root file - unless we find that this crate has the same root file as
336/// another crate, in which case we store all data for one of them (if one is a dependency of
337/// the other, we store for it, because it has more dependencies to be invalidated).
338#[derive(Debug, Clone, PartialEq, Eq, Hash)]
339pub struct UniqueCrateData {
340    root_file_id: FileId,
341    disambiguator: Option<Box<(BuiltCrateData, HashableCfgOptions)>>,
342}

344#[derive(Debug, Clone, PartialEq, Eq, Hash)]
345pub struct CrateData<Id> {
346    pub root_file_id: FileId,
347    pub edition: Edition,
348    /// The dependencies of this crate.
349    ///
350    /// Note that this may contain more dependencies than the crate actually uses.
351    /// A common example is the test crate which is included but only actually is active when
352    /// declared in source via `extern crate test`.
353    pub dependencies: Vec<Dependency<Id>>,
354    pub origin: CrateOrigin,
355    /// Extra crate-level attributes, including the surrounding `#![]`.
356    pub crate_attrs: Box<[Box<str>]>,
357    pub is_proc_macro: bool,
358    /// The working directory to run proc-macros in invoked in the context of this crate.
359    /// This is the workspace root of the cargo workspace for workspace members, the crate manifest
360    /// dir otherwise.
361    // FIXME: This ought to be a `VfsPath` or something opaque.
362    pub proc_macro_cwd: Arc<AbsPathBuf>,
363}

466impl Crate {
467    /// Returns an iterator over all transitive dependencies of the given crate,
468    /// including the crate itself.
469    ///
470    /// **Warning**: do not use this query in `hir-*` crates! It kills incrementality across crate metadata modifications.
471    pub fn transitive_deps(self, db: &dyn salsa::Database) -> Vec<Crate> {
472        // There is a bit of duplication here and in `CrateGraphBuilder` in the same method, but it's not terrible
473        // and removing that is a bit difficult.
474        let mut worklist = vec![self];
475        let mut deps_seen = FxHashSet::default();
476        let mut deps = Vec::new();
477
478        while let Some(krate) = worklist.pop() {
479            if !deps_seen.insert(krate) {
480                continue;
481            }
482            deps.push(krate);
483
484            worklist.extend(krate.data(db).dependencies.iter().map(|dep| dep.crate_id));
485        }
486        deps
487    }
488
489    /// Returns all transitive reverse dependencies of the given crate,
490    /// including the crate itself.
491    ///
492    /// **Warning**: do not use this query in `hir-*` crates! It kills incrementality across crate metadata modifications.
493    pub fn transitive_rev_deps(self, db: &dyn RootQueryDb) -> Box<[Crate]> {
494        let mut worklist = vec![self];
495        let mut rev_deps = FxHashSet::default();
496        rev_deps.insert(self);
497
498        let mut inverted_graph = FxHashMap::<_, Vec<_>>::default();
499        db.all_crates().iter().for_each(|&krate| {
500            krate
501                .data(db)
502                .dependencies
503                .iter()
504                .for_each(|dep| inverted_graph.entry(dep.crate_id).or_default().push(krate))
505        });
506
507        while let Some(krate) = worklist.pop() {
508            if let Some(crate_rev_deps) = inverted_graph.get(&krate) {
509                crate_rev_deps
510                    .iter()
511                    .copied()
512                    .filter(|&rev_dep| rev_deps.insert(rev_dep))
513                    .for_each(|rev_dep| worklist.push(rev_dep));
514            }
515        }
516
517        rev_deps.into_iter().collect::<Box<_>>()
518    }
519}

525impl CrateGraphBuilder {
526    pub fn add_crate_root(
527        &mut self,
528        root_file_id: FileId,
529        edition: Edition,
530        display_name: Option<CrateDisplayName>,
531        version: Option<String>,
532        mut cfg_options: CfgOptions,
533        mut potential_cfg_options: Option<CfgOptions>,
534        mut env: Env,
535        origin: CrateOrigin,
536        crate_attrs: Vec<String>,
537        is_proc_macro: bool,
538        proc_macro_cwd: Arc<AbsPathBuf>,
539        ws_data: Arc<CrateWorkspaceData>,
540    ) -> CrateBuilderId {
541        env.entries.shrink_to_fit();
542        cfg_options.shrink_to_fit();
543        if let Some(potential_cfg_options) = &mut potential_cfg_options {
544            potential_cfg_options.shrink_to_fit();
545        }
546        let crate_attrs: Vec<_> = crate_attrs
547            .into_iter()
548            .map(|raw_attr| format!("#![{raw_attr}]").into_boxed_str())
549            .collect();
550        self.arena.alloc(CrateBuilder {
551            basic: CrateData {
552                root_file_id,
553                edition,
554                dependencies: Vec::new(),
555                origin,
556                crate_attrs: crate_attrs.into_boxed_slice(),
557                is_proc_macro,
558                proc_macro_cwd,
559            },
560            extra: ExtraCrateData { version, display_name, potential_cfg_options },
561            cfg_options,
562            env,
563            ws_data,
564        })
565    }
566
567    pub fn add_dep(
568        &mut self,
569        from: CrateBuilderId,
570        dep: DependencyBuilder,
571    ) -> Result<(), CyclicDependenciesError> {
572        let _p = tracing::info_span!("add_dep").entered();
573
574        // Check if adding a dep from `from` to `to` creates a cycle. To figure
575        // that out, look for a  path in the *opposite* direction, from `to` to
576        // `from`.
577        if let Some(path) = self.find_path(&mut FxHashSet::default(), dep.crate_id, from) {
578            let path =
579                path.into_iter().map(|it| (it, self[it].extra.display_name.clone())).collect();
580            let err = CyclicDependenciesError { path };
581            assert!(err.from().0 == from && err.to().0 == dep.crate_id);
582            return Err(err);
583        }
584
585        self.arena[from].basic.dependencies.push(dep);
586        Ok(())
587    }
588
589    pub fn set_in_db(self, db: &mut dyn RootQueryDb) -> CratesIdMap {
590        // For some reason in some repositories we have duplicate crates, so we use a set and not `Vec`.
591        // We use an `IndexSet` because the list needs to be topologically sorted.
592        let mut all_crates = FxIndexSet::with_capacity_and_hasher(self.arena.len(), FxBuildHasher);
593        let mut visited = FxHashMap::default();
594        let mut visited_root_files = FxHashSet::default();
595
596        let old_all_crates = db.all_crates();
597
598        let crates_map = db.crates_map();
599        // salsa doesn't compare new input to old input to see if they are the same, so here we are doing all the work ourselves.
600        for krate in self.iter() {
601            go(
602                &self,
603                db,
604                &crates_map,
605                &mut visited,
606                &mut visited_root_files,
607                &mut all_crates,
608                krate,
609            );
610        }
611
612        if old_all_crates.len() != all_crates.len()
613            || old_all_crates.iter().any(|&krate| !all_crates.contains(&krate))
614        {
615            db.set_all_crates_with_durability(
616                Arc::new(Vec::from_iter(all_crates).into_boxed_slice()),
617                Durability::MEDIUM,
618            );
619        }
620
621        return visited;
622
623        fn go(
624            graph: &CrateGraphBuilder,
625            db: &mut dyn RootQueryDb,
626            crates_map: &CratesMap,
627            visited: &mut FxHashMap<CrateBuilderId, Crate>,
628            visited_root_files: &mut FxHashSet<FileId>,
629            all_crates: &mut FxIndexSet<Crate>,
630            source: CrateBuilderId,
631        ) -> Crate {
632            if let Some(&crate_id) = visited.get(&source) {
633                return crate_id;
634            }
635            let krate = &graph[source];
636            let dependencies = krate
637                .basic
638                .dependencies
639                .iter()
640                .map(|dep| BuiltDependency {
641                    crate_id: go(
642                        graph,
643                        db,
644                        crates_map,
645                        visited,
646                        visited_root_files,
647                        all_crates,
648                        dep.crate_id,
649                    ),
650                    name: dep.name.clone(),
651                    prelude: dep.prelude,
652                    sysroot: dep.sysroot,
653                })
654                .collect::<Vec<_>>();
655            let crate_data = BuiltCrateData {
656                dependencies,
657                edition: krate.basic.edition,
658                is_proc_macro: krate.basic.is_proc_macro,
659                origin: krate.basic.origin.clone(),
660                crate_attrs: krate.basic.crate_attrs.clone(),
661                root_file_id: krate.basic.root_file_id,
662                proc_macro_cwd: krate.basic.proc_macro_cwd.clone(),
663            };
664            let disambiguator = if visited_root_files.insert(krate.basic.root_file_id) {
665                None
666            } else {
667                Some(Box::new((crate_data.clone(), krate.cfg_options.to_hashable())))
668            };
669
670            let unique_crate_data =
671                UniqueCrateData { root_file_id: krate.basic.root_file_id, disambiguator };
672            let crate_input = match crates_map.0.entry(unique_crate_data) {
673                Entry::Occupied(entry) => {
674                    let old_crate = *entry.get();
675                    if crate_data != *old_crate.data(db) {
676                        old_crate.set_data(db).with_durability(Durability::MEDIUM).to(crate_data);
677                    }
678                    if krate.extra != *old_crate.extra_data(db) {
679                        old_crate
680                            .set_extra_data(db)
681                            .with_durability(Durability::MEDIUM)
682                            .to(krate.extra.clone());
683                    }
684                    if krate.cfg_options != *old_crate.cfg_options(db) {
685                        old_crate
686                            .set_cfg_options(db)
687                            .with_durability(Durability::MEDIUM)
688                            .to(krate.cfg_options.clone());
689                    }
690                    if krate.env != *old_crate.env(db) {
691                        old_crate
692                            .set_env(db)
693                            .with_durability(Durability::MEDIUM)
694                            .to(krate.env.clone());
695                    }
696                    if krate.ws_data != *old_crate.workspace_data(db) {
697                        old_crate
698                            .set_workspace_data(db)
699                            .with_durability(Durability::MEDIUM)
700                            .to(krate.ws_data.clone());
701                    }
702                    old_crate
703                }
704                Entry::Vacant(entry) => {
705                    let input = Crate::builder(
706                        crate_data,
707                        krate.extra.clone(),
708                        krate.ws_data.clone(),
709                        krate.cfg_options.clone(),
710                        krate.env.clone(),
711                    )
712                    .durability(Durability::MEDIUM)
713                    .new(db);
714                    entry.insert(input);
715                    input
716                }
717            };
718            all_crates.insert(crate_input);
719            visited.insert(source, crate_input);
720            crate_input
721        }
722    }

728    /// Returns an iterator over all transitive dependencies of the given crate,
729    /// including the crate itself.
730    pub fn transitive_deps(&self, of: CrateBuilderId) -> impl Iterator<Item = CrateBuilderId> {
731        let mut worklist = vec![of];
732        let mut deps = FxHashSet::default();
733
734        while let Some(krate) = worklist.pop() {
735            if !deps.insert(krate) {
736                continue;
737            }
738
739            worklist.extend(self[krate].basic.dependencies.iter().map(|dep| dep.crate_id));
740        }
741
742        deps.into_iter()
743    }

745    /// Returns all crates in the graph, sorted in topological order (ie. dependencies of a crate
746    /// come before the crate itself).
747    fn crates_in_topological_order(&self) -> Vec<CrateBuilderId> {
748        let mut res = Vec::new();
749        let mut visited = FxHashSet::default();
750
751        for krate in self.iter() {
752            go(self, &mut visited, &mut res, krate);
753        }
754
755        return res;
756
757        fn go(
758            graph: &CrateGraphBuilder,
759            visited: &mut FxHashSet<CrateBuilderId>,
760            res: &mut Vec<CrateBuilderId>,
761            source: CrateBuilderId,
762        ) {
763            if !visited.insert(source) {
764                return;
765            }
766            for dep in graph[source].basic.dependencies.iter() {
767                go(graph, visited, res, dep.crate_id)
768            }
769            res.push(source)
770        }
771    }

773    /// Extends this crate graph by adding a complete second crate
774    /// graph and adjust the ids in the [`ProcMacroPaths`] accordingly.
775    ///
776    /// This will deduplicate the crates of the graph where possible.
777    /// Furthermore dependencies are sorted by crate id to make deduplication easier.
778    ///
779    /// Returns a map mapping `other`'s IDs to the new IDs in `self`.
780    pub fn extend(
781        &mut self,
782        mut other: CrateGraphBuilder,
783        proc_macros: &mut ProcMacroPaths,
784    ) -> FxHashMap<CrateBuilderId, CrateBuilderId> {
785        // Sorting here is a bit pointless because the input is likely already sorted.
786        // However, the overhead is small and it makes the `extend` method harder to misuse.
787        self.arena
788            .iter_mut()
789            .for_each(|(_, data)| data.basic.dependencies.sort_by_key(|dep| dep.crate_id));
790
791        let m = self.arena.len();
792        let topo = other.crates_in_topological_order();
793        let mut id_map: FxHashMap<CrateBuilderId, CrateBuilderId> = FxHashMap::default();
794        for topo in topo {
795            let crate_data = &mut other.arena[topo];
796
797            crate_data
798                .basic
799                .dependencies
800                .iter_mut()
801                .for_each(|dep| dep.crate_id = id_map[&dep.crate_id]);
802            crate_data.basic.dependencies.sort_by_key(|dep| dep.crate_id);
803
804            let find = self.arena.iter().take(m).find_map(|(k, v)| (v == crate_data).then_some(k));
805            let new_id = find.unwrap_or_else(|| self.arena.alloc(crate_data.clone()));
806            id_map.insert(topo, new_id);
807        }
808
809        *proc_macros =
810            mem::take(proc_macros).into_iter().map(|(id, macros)| (id_map[&id], macros)).collect();
811        id_map
812    }
813
814    fn find_path(
815        &self,
816        visited: &mut FxHashSet<CrateBuilderId>,
817        from: CrateBuilderId,
818        to: CrateBuilderId,
819    ) -> Option<Vec<CrateBuilderId>> {
820        if !visited.insert(from) {
821            return None;
822        }
823
824        if from == to {
825            return Some(vec![to]);
826        }
827
828        for dep in &self[from].basic.dependencies {
829            let crate_id = dep.crate_id;
830            if let Some(mut path) = self.find_path(visited, crate_id, to) {
831                path.push(from);
832                return Some(path);
833            }
834        }
835
836        None
837    }

839    /// Removes all crates from this crate graph except for the ones in `to_keep` and fixes up the dependencies.
840    /// Returns a mapping from old crate ids to new crate ids.
841    pub fn remove_crates_except(
842        &mut self,
843        to_keep: &[CrateBuilderId],
844    ) -> Vec<Option<CrateBuilderId>> {
845        let mut id_map = vec![None; self.arena.len()];
846        self.arena = std::mem::take(&mut self.arena)
847            .into_iter()
848            .filter_map(|(id, data)| if to_keep.contains(&id) { Some((id, data)) } else { None })
849            .enumerate()
850            .map(|(new_id, (id, data))| {
851                id_map[id.into_raw().into_u32() as usize] =
852                    Some(CrateBuilderId::from_raw(RawIdx::from_u32(new_id as u32)));
853                data
854            })
855            .collect();
856        for (_, data) in self.arena.iter_mut() {
857            data.basic.dependencies.iter_mut().for_each(|dep| {
858                dep.crate_id =
859                    id_map[dep.crate_id.into_raw().into_u32() as usize].expect("crate was filtered")
860            });
861        }
862        id_map
863    }

932#[derive(Debug)]
933pub struct CyclicDependenciesError {
934    path: Vec<(CrateBuilderId, Option<CrateDisplayName>)>,
935}
936
937impl CyclicDependenciesError {
938    fn from(&self) -> &(CrateBuilderId, Option<CrateDisplayName>) {
939        self.path.first().unwrap()
940    }
941    fn to(&self) -> &(CrateBuilderId, Option<CrateDisplayName>) {
942        self.path.last().unwrap()
943    }
944}
945
946impl fmt::Display for CyclicDependenciesError {
947    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
948        let render = |(id, name): &(CrateBuilderId, Option<CrateDisplayName>)| match name {
949            Some(it) => format!("{it}({id:?})"),
950            None => format!("{id:?}"),
951        };
952        let path = self.path.iter().rev().map(render).collect::<Vec<String>>().join(" -> ");
953        write!(
954            f,
955            "cyclic deps: {} -> {}, alternative path: {}",
956            render(self.from()),
957            render(self.to()),
958            path
959        )
960    }
