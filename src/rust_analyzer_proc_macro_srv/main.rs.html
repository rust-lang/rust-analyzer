1//! A standalone binary for `proc-macro-srv`.
2//! Driver for proc macro server
3#![cfg_attr(feature = "in-rust-tree", feature(rustc_private))]
4#![cfg_attr(not(feature = "sysroot-abi"), allow(unused_crate_dependencies))]

18fn main() -> std::io::Result<()> {
19    let v = std::env::var("RUST_ANALYZER_INTERNALS_DO_NOT_USE");
20    if v.is_err() {
21        eprintln!(
22            "This is an IDE implementation detail, you can use this tool by exporting RUST_ANALYZER_INTERNALS_DO_NOT_USE."
23        );
24        eprintln!(
25            "Note that this tool's API is highly unstable and may break without prior notice"
26        );
27        std::process::exit(122);
28    }
29    let matches = Command::new("proc-macro-srv")
30        .args(&[
31            clap::Arg::new("format")
32                .long("format")
33                .action(clap::ArgAction::Set)
34                .default_value("json-legacy")
35                .value_parser(clap::builder::EnumValueParser::<ProtocolFormatArg>::new()),
36            clap::Arg::new("version")
37                .long("version")
38                .action(clap::ArgAction::SetTrue)
39                .help("Prints the version of the proc-macro-srv"),
40        ])
41        .get_matches();
42    if matches.get_flag("version") {
43        println!("rust-analyzer-proc-macro-srv {}", version::version());
44        return Ok(());
45    }
46    let &format = matches
47        .get_one::<ProtocolFormatArg>("format")
48        .expect("format value should always be present");
49
50    let mut stdin = std::io::BufReader::new(std::io::stdin());
51    let mut stdout = std::io::stdout();
52
53    run(&mut stdin, &mut stdout, format.into())
54}

56/// Wrapper for CLI argument parsing that implements `ValueEnum`.
57#[derive(Copy, Clone)]
58struct ProtocolFormatArg(ProtocolFormat);
59
60impl From<ProtocolFormatArg> for ProtocolFormat {
61    fn from(arg: ProtocolFormatArg) -> Self {
62        arg.0
63    }
64}
65
66impl ValueEnum for ProtocolFormatArg {
67    fn value_variants<'a>() -> &'a [Self] {
68        &[
69            ProtocolFormatArg(ProtocolFormat::JsonLegacy),
70            ProtocolFormatArg(ProtocolFormat::BidirectionalPostcardPrototype),
71        ]
72    }
73
74    fn to_possible_value(&self) -> Option<clap::builder::PossibleValue> {
75        match self.0 {
76            ProtocolFormat::JsonLegacy => Some(clap::builder::PossibleValue::new("json-legacy")),
77            ProtocolFormat::BidirectionalPostcardPrototype => {
78                Some(clap::builder::PossibleValue::new("bidirectional-postcard-prototype"))
79            }
80        }
81    }
82
83    fn from_str(input: &str, _ignore_case: bool) -> Result<Self, String> {
84        match input {
85            "json-legacy" => Ok(ProtocolFormatArg(ProtocolFormat::JsonLegacy)),
86            "bidirectional-postcard-prototype" => {
87                Ok(ProtocolFormatArg(ProtocolFormat::BidirectionalPostcardPrototype))
88            }
89            _ => Err(format!("unknown protocol format: {input}")),
90        }
91    }
92}

103            .to_owned(),
104    ))
