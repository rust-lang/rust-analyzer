36pub(super) fn type_info_of(
37    sema: &Semantics<'_, RootDatabase>,
38    _config: &HoverConfig<'_>,
39    expr_or_pat: &Either<ast::Expr, ast::Pat>,
40    edition: Edition,
41    display_target: DisplayTarget,
42) -> Option<HoverResult> {
43    let ty_info = match expr_or_pat {
44        Either::Left(expr) => sema.type_of_expr(expr)?,
45        Either::Right(pat) => sema.type_of_pat(pat)?,
46    };
47    type_info(sema, _config, ty_info, edition, display_target)
48}

61pub(super) fn try_expr(
62    sema: &Semantics<'_, RootDatabase>,
63    _config: &HoverConfig<'_>,
64    try_expr: &ast::TryExpr,
65    edition: Edition,
66    display_target: DisplayTarget,
67) -> Option<HoverResult> {
68    let inner_ty = sema.type_of_expr(&try_expr.expr()?)?.original;
69    let mut ancestors = try_expr.syntax().ancestors();
70    let mut body_ty = loop {
71        let next = ancestors.next()?;
72        break match_ast! {
73            match next {
74                ast::Fn(fn_) => sema.to_def(&fn_)?.ret_type(sema.db),
75                ast::Item(__) => return None,
76                ast::ClosureExpr(closure) => sema.type_of_expr(&closure.body()?)?.original,
77                ast::BlockExpr(block_expr) => if matches!(block_expr.modifier(), Some(ast::BlockModifier::Async(_) | ast::BlockModifier::Try(_)| ast::BlockModifier::Const(_))) {
78                    sema.type_of_expr(&block_expr.into())?.original
79                } else {
80                    continue;
81                },
82                _ => continue,
83            }
84        };
85    };
86
87    if inner_ty == body_ty {
88        return None;
89    }
90
91    let mut inner_ty = inner_ty;
92    let mut s = "Try Target".to_owned();
93
94    let adts = inner_ty.as_adt().zip(body_ty.as_adt());
95    if let Some((hir::Adt::Enum(inner), hir::Adt::Enum(body))) = adts {
96        let famous_defs = FamousDefs(sema, sema.scope(try_expr.syntax())?.krate());
97        // special case for two options, there is no value in showing them
98        if let Some(option_enum) = famous_defs.core_option_Option()
99            && inner == option_enum
100            && body == option_enum
101        {
102            cov_mark::hit!(hover_try_expr_opt_opt);
103            return None;
104        }
105
106        // special case two results to show the error variants only
107        if let Some(result_enum) = famous_defs.core_result_Result()
108            && inner == result_enum
109            && body == result_enum
110        {
111            let error_type_args =
112                inner_ty.type_arguments().nth(1).zip(body_ty.type_arguments().nth(1));
113            if let Some((inner, body)) = error_type_args {
114                inner_ty = inner;
115                body_ty = body;
116                "Try Error".clone_into(&mut s);
117            }
118        }
119    }
120
121    let mut res = HoverResult::default();
122
123    let mut targets: Vec<hir::ModuleDef> = Vec::new();
124    let mut push_new_def = |item: hir::ModuleDef| {
125        if !targets.contains(&item) {
126            targets.push(item);
127        }
128    };
129    walk_and_push_ty(sema.db, &inner_ty, &mut push_new_def);
130    walk_and_push_ty(sema.db, &body_ty, &mut push_new_def);
131    if let Some(actions) = HoverAction::goto_type_from_targets(sema, targets, edition) {
132        res.actions.push(actions);
133    }
134
135    let inner_ty = inner_ty.display(sema.db, display_target).to_string();
136    let body_ty = body_ty.display(sema.db, display_target).to_string();
137    let ty_len_max = inner_ty.len().max(body_ty.len());
138
139    let l = "Propagated as: ".len() - " Type: ".len();
140    let static_text_len_diff = l as isize - s.len() as isize;
141    let tpad = static_text_len_diff.max(0) as usize;
142    let ppad = static_text_len_diff.min(0).unsigned_abs();
143
144    res.markup = format!(
145        "```text\n{} Type: {:>pad0$}\nPropagated as: {:>pad1$}\n```\n",
146        s,
147        inner_ty,
148        body_ty,
149        pad0 = ty_len_max + tpad,
150        pad1 = ty_len_max + ppad,
151    )
152    .into();
153    Some(res)
154}

156pub(super) fn deref_expr(
157    sema: &Semantics<'_, RootDatabase>,
158    _config: &HoverConfig<'_>,
159    deref_expr: &ast::PrefixExpr,
160    edition: Edition,
161    display_target: DisplayTarget,
162) -> Option<HoverResult> {
163    let inner_ty = sema.type_of_expr(&deref_expr.expr()?)?.original;
164    let TypeInfo { original, adjusted } =
165        sema.type_of_expr(&ast::Expr::from(deref_expr.clone()))?;
166
167    let mut res = HoverResult::default();
168    let mut targets: Vec<hir::ModuleDef> = Vec::new();
169    let mut push_new_def = |item: hir::ModuleDef| {
170        if !targets.contains(&item) {
171            targets.push(item);
172        }
173    };
174    walk_and_push_ty(sema.db, &inner_ty, &mut push_new_def);
175    walk_and_push_ty(sema.db, &original, &mut push_new_def);
176
177    res.markup = if let Some(adjusted_ty) = adjusted {
178        walk_and_push_ty(sema.db, &adjusted_ty, &mut push_new_def);
179        let original = original.display(sema.db, display_target).to_string();
180        let adjusted = adjusted_ty.display(sema.db, display_target).to_string();
181        let inner = inner_ty.display(sema.db, display_target).to_string();
182        let type_len = "To type: ".len();
183        let coerced_len = "Coerced to: ".len();
184        let deref_len = "Dereferenced from: ".len();
185        let max_len = (original.len() + type_len)
186            .max(adjusted.len() + coerced_len)
187            .max(inner.len() + deref_len);
188        format!(
189            "```text\nDereferenced from: {:>ipad$}\nTo type: {:>apad$}\nCoerced to: {:>opad$}\n```\n",
190            inner,
191            original,
192            adjusted,
193            ipad = max_len - deref_len,
194            apad = max_len - type_len,
195            opad = max_len - coerced_len,
196        )
197        .into()
198    } else {
199        let original = original.display(sema.db, display_target).to_string();
200        let inner = inner_ty.display(sema.db, display_target).to_string();
201        let type_len = "To type: ".len();
202        let deref_len = "Dereferenced from: ".len();
203        let max_len = (original.len() + type_len).max(inner.len() + deref_len);
204        format!(
205            "```text\nDereferenced from: {:>ipad$}\nTo type: {:>apad$}\n```\n",
206            inner,
207            original,
208            ipad = max_len - deref_len,
209            apad = max_len - type_len,
210        )
211        .into()
212    };
213    if let Some(actions) = HoverAction::goto_type_from_targets(sema, targets, edition) {
214        res.actions.push(actions);
215    }
216
217    Some(res)
218}

264pub(super) fn keyword(
265    sema: &Semantics<'_, RootDatabase>,
266    config: &HoverConfig<'_>,
267    token: &SyntaxToken,
268    edition: Edition,
269    display_target: DisplayTarget,
270) -> Option<HoverResult> {
271    if !token.kind().is_keyword(edition) || !config.documentation || !config.keywords {
272        return None;
273    }
274    let parent = token.parent()?;
275    let famous_defs = FamousDefs(sema, sema.scope(&parent)?.krate());
276
277    let KeywordHint { description, keyword_mod, actions } =
278        keyword_hints(sema, token, parent, edition, display_target);
279
280    let doc_owner = find_std_module(&famous_defs, &keyword_mod, edition)?;
281    let (docs, range_map) = doc_owner.docs_with_rangemap(sema.db)?;
282    let (markup, range_map) =
283        markup(Some(docs.into()), Some(range_map), description, None, None, String::new());
284    let markup = process_markup(sema.db, Definition::Module(doc_owner), &markup, range_map, config);
285    Some(HoverResult { markup, actions })
286}

288/// Returns missing types in a record pattern.
289/// Only makes sense when there's a rest pattern in the record pattern.
290/// i.e. `let S {a, ..} = S {a: 1, b: 2}`
291pub(super) fn struct_rest_pat(
292    sema: &Semantics<'_, RootDatabase>,
293    _config: &HoverConfig<'_>,
294    pattern: &ast::RecordPat,
295    edition: Edition,
296    display_target: DisplayTarget,
297) -> HoverResult {
298    let missing_fields = sema.record_pattern_missing_fields(pattern);
299
300    // if there are no missing fields, the end result is a hover that shows ".."
301    // should be left in to indicate that there are no more fields in the pattern
302    // example, S {a: 1, b: 2, ..} when struct S {a: u32, b: u32}
303
304    let mut res = HoverResult::default();
305    let mut targets: Vec<hir::ModuleDef> = Vec::new();
306    let mut push_new_def = |item: hir::ModuleDef| {
307        if !targets.contains(&item) {
308            targets.push(item);
309        }
310    };
311    for (_, t) in &missing_fields {
312        walk_and_push_ty(sema.db, t, &mut push_new_def);
313    }
314
315    res.markup = {
316        let mut s = String::from(".., ");
317        for (f, _) in &missing_fields {
318            s += f.display(sema.db, display_target).to_string().as_ref();
319            s += ", ";
320        }
321        // get rid of trailing comma
322        s.truncate(s.len() - 2);
323
324        Markup::fenced_block(&s)
325    };
326    if let Some(actions) = HoverAction::goto_type_from_targets(sema, targets, edition) {
327        res.actions.push(actions);
328    }
329    res
330}

332pub(super) fn try_for_lint(attr: &ast::Attr, token: &SyntaxToken) -> Option<HoverResult> {
333    let (path, tt) = attr.as_simple_call()?;
334    if !tt.syntax().text_range().contains(token.text_range().start()) {
335        return None;
336    }
337    let (is_clippy, lints) = match &*path {
338        "feature" => (false, FEATURES),
339        "allow" | "deny" | "expect" | "forbid" | "warn" => {
340            let is_clippy = algo::non_trivia_sibling(token.clone().into(), Direction::Prev)
341                .filter(|t| t.kind() == T![:])
342                .and_then(|t| algo::non_trivia_sibling(t, Direction::Prev))
343                .filter(|t| t.kind() == T![:])
344                .and_then(|t| algo::non_trivia_sibling(t, Direction::Prev))
345                .is_some_and(|t| {
346                    t.kind() == T![ident] && t.into_token().is_some_and(|t| t.text() == "clippy")
347                });
348            if is_clippy { (true, CLIPPY_LINTS) } else { (false, DEFAULT_LINTS) }
349        }
350        _ => return None,
351    };
352
353    let tmp;
354    let needle = if is_clippy {
355        tmp = format!("clippy::{}", token.text());
356        &tmp
357    } else {
358        token.text()
359    };
360
361    let lint =
362        lints.binary_search_by_key(&needle, |lint| lint.label).ok().map(|idx| &lints[idx])?;
363    Some(HoverResult {
364        markup: Markup::from(format!("```\n{}\n```\n---\n\n{}", lint.label, lint.description)),
365        ..Default::default()
366    })
367}

369pub(super) fn process_markup(
370    db: &RootDatabase,
371    def: Definition,
372    markup: &Markup,
373    markup_range_map: Option<DocsRangeMap>,
374    config: &HoverConfig<'_>,
375) -> Markup {
376    let markup = markup.as_str();
377    let markup = if config.links_in_hover {
378        rewrite_links(db, markup, def, markup_range_map)
379    } else {
380        remove_links(markup)
381    };
382    Markup::from(markup)
383}

385fn definition_owner_name(db: &RootDatabase, def: Definition, edition: Edition) -> Option<String> {
386    match def {
387        Definition::Field(f) => {
388            let parent = f.parent_def(db);
389            let parent_name = parent.name(db);
390            let parent_name = parent_name.display(db, edition).to_string();
391            return match parent {
392                VariantDef::Variant(variant) => {
393                    let enum_name = variant.parent_enum(db).name(db);
394                    Some(format!("{}::{parent_name}", enum_name.display(db, edition)))
395                }
396                _ => Some(parent_name),
397            };
398        }
399        Definition::Variant(e) => Some(e.parent_enum(db).name(db)),
400        Definition::GenericParam(generic_param) => match generic_param.parent() {
401            hir::GenericDef::Adt(it) => Some(it.name(db)),
402            hir::GenericDef::Trait(it) => Some(it.name(db)),
403            hir::GenericDef::TypeAlias(it) => Some(it.name(db)),
404
405            hir::GenericDef::Impl(i) => i.self_ty(db).as_adt().map(|adt| adt.name(db)),
406            hir::GenericDef::Function(it) => {
407                let container = it.as_assoc_item(db).and_then(|assoc| match assoc.container(db) {
408                    hir::AssocItemContainer::Trait(t) => Some(t.name(db)),
409                    hir::AssocItemContainer::Impl(i) => {
410                        i.self_ty(db).as_adt().map(|adt| adt.name(db))
411                    }
412                });
413                match container {
414                    Some(name) => {
415                        return Some(format!(
416                            "{}::{}",
417                            name.display(db, edition),
418                            it.name(db).display(db, edition)
419                        ));
420                    }
421                    None => Some(it.name(db)),
422                }
423            }
424            hir::GenericDef::Const(it) => {
425                let container = it.as_assoc_item(db).and_then(|assoc| match assoc.container(db) {
426                    hir::AssocItemContainer::Trait(t) => Some(t.name(db)),
427                    hir::AssocItemContainer::Impl(i) => {
428                        i.self_ty(db).as_adt().map(|adt| adt.name(db))
429                    }
430                });
431                match container {
432                    Some(name) => {
433                        return Some(format!(
434                            "{}::{}",
435                            name.display(db, edition),
436                            it.name(db)?.display(db, edition)
437                        ));
438                    }
439                    None => it.name(db),
440                }
441            }
442            hir::GenericDef::Static(it) => Some(it.name(db)),
443        },
444        Definition::DeriveHelper(derive_helper) => Some(derive_helper.derive().name(db)),
445        d => {
446            if let Some(assoc_item) = d.as_assoc_item(db) {
447                match assoc_item.container(db) {
448                    hir::AssocItemContainer::Trait(t) => Some(t.name(db)),
449                    hir::AssocItemContainer::Impl(i) => {
450                        i.self_ty(db).as_adt().map(|adt| adt.name(db))
451                    }
452                }
453            } else {
454                return d.as_extern_assoc_item(db).map(|_| "<extern>".to_owned());
455            }
456        }
457    }
458    .map(|name| name.display(db, edition).to_string())
459}

461pub(super) fn path(
462    db: &RootDatabase,
463    module: hir::Module,
464    item_name: Option<String>,
465    edition: Edition,
466) -> String {
467    let crate_name = module.krate().display_name(db).as_ref().map(|it| it.to_string());
468    let module_path = module
469        .path_to_root(db)
470        .into_iter()
471        .rev()
472        .flat_map(|it| it.name(db).map(|name| name.display(db, edition).to_string()));
473    crate_name.into_iter().chain(module_path).chain(item_name).join("::")
474}

476pub(super) fn definition(
477    db: &RootDatabase,
478    def: Definition,
479    famous_defs: Option<&FamousDefs<'_, '_>>,
480    notable_traits: &[(Trait, Vec<(Option<Type<'_>>, Name)>)],
481    macro_arm: Option<u32>,
482    render_extras: bool,
483    subst_types: Option<&Vec<(Symbol, Type<'_>)>>,
484    config: &HoverConfig<'_>,
485    edition: Edition,
486    display_target: DisplayTarget,
487) -> (Markup, Option<DocsRangeMap>) {
488    let mod_path = definition_path(db, &def, edition);
489    let label = match def {
490        Definition::Trait(trait_) => trait_
491            .display_limited(db, config.max_trait_assoc_items_count, display_target)
492            .to_string(),
493        Definition::Adt(adt @ (Adt::Struct(_) | Adt::Union(_))) => {
494            adt.display_limited(db, config.max_fields_count, display_target).to_string()
495        }
496        Definition::Variant(variant) => {
497            variant.display_limited(db, config.max_fields_count, display_target).to_string()
498        }
499        Definition::Adt(adt @ Adt::Enum(_)) => {
500            adt.display_limited(db, config.max_enum_variants_count, display_target).to_string()
501        }
502        Definition::SelfType(impl_def) => {
503            let self_ty = &impl_def.self_ty(db);
504            match self_ty.as_adt() {
505                Some(adt) => {
506                    adt.display_limited(db, config.max_fields_count, display_target).to_string()
507                }
508                None => self_ty.display(db, display_target).to_string(),
509            }
510        }
511        Definition::Macro(it) => {
512            let mut label = it.display(db, display_target).to_string();
513            if let Some(macro_arm) = macro_arm {
514                format_to!(label, " // matched arm #{}", macro_arm);
515            }
516            label
517        }
518        Definition::Function(fn_) => {
519            fn_.display_with_container_bounds(db, true, display_target).to_string()
520        }
521        _ => def.label(db, display_target),
522    };
523    let (docs, range_map) =
524        if let Some((docs, doc_range)) = def.docs_with_rangemap(db, famous_defs, display_target) {
525            (Some(docs), doc_range)
526        } else {
527            (None, None)
528        };
529    let value = || match def {
530        Definition::Variant(it) => {
531            if !it.parent_enum(db).is_data_carrying(db) {
532                match it.eval(db) {
533                    Ok(it) => {
534                        Some(if it >= 10 { format!("{it} ({it:#X})") } else { format!("{it}") })
535                    }
536                    Err(err) => {
537                        let res = it.value(db).map(|it| format!("{it:?}"));
538                        if env::var_os("RA_DEV").is_some() {
539                            let res = res.as_deref().unwrap_or("");
540                            Some(format!(
541                                "{res} ({})",
542                                render_const_eval_error(db, err, display_target)
543                            ))
544                        } else {
545                            res
546                        }
547                    }
548                }
549            } else {
550                None
551            }
552        }
553        Definition::Const(it) => {
554            let body = it.eval(db);
555            Some(match body {
556                Ok(it) => match it.render_debug(db) {
557                    Ok(it) => it,
558                    Err(err) => {
559                        let it = it.render(db, display_target);
560                        if env::var_os("RA_DEV").is_some() {
561                            format!(
562                                "{it}\n{}",
563                                render_const_eval_error(db, err.into(), display_target)
564                            )
565                        } else {
566                            it
567                        }
568                    }
569                },
570                Err(err) => {
571                    let source = it.source(db)?;
572                    let mut body = source.value.body()?.syntax().clone();
573                    if let Some(macro_file) = source.file_id.macro_file() {
574                        let span_map = db.expansion_span_map(macro_file);
575                        body = prettify_macro_expansion(db, body, &span_map, it.krate(db).into());
576                    }
577                    if env::var_os("RA_DEV").is_some() {
578                        format!("{body}\n{}", render_const_eval_error(db, err, display_target))
579                    } else {
580                        body.to_string()
581                    }
582                }
583            })
584        }
585        Definition::Static(it) => {
586            let body = it.eval(db);
587            Some(match body {
588                Ok(it) => match it.render_debug(db) {
589                    Ok(it) => it,
590                    Err(err) => {
591                        let it = it.render(db, display_target);
592                        if env::var_os("RA_DEV").is_some() {
593                            format!(
594                                "{it}\n{}",
595                                render_const_eval_error(db, err.into(), display_target)
596                            )
597                        } else {
598                            it
599                        }
600                    }
601                },
602                Err(err) => {
603                    let source = it.source(db)?;
604                    let mut body = source.value.body()?.syntax().clone();
605                    if let Some(macro_file) = source.file_id.macro_file() {
606                        let span_map = db.expansion_span_map(macro_file);
607                        body = prettify_macro_expansion(db, body, &span_map, it.krate(db).into());
608                    }
609                    if env::var_os("RA_DEV").is_some() {
610                        format!("{body}\n{}", render_const_eval_error(db, err, display_target))
611                    } else {
612                        body.to_string()
613                    }
614                }
615            })
616        }
617        _ => None,
618    };
619
620    let layout_info = || match def {
621        Definition::Field(it) => render_memory_layout(
622            config.memory_layout,
623            || it.layout(db),
624            |_| {
625                let var_def = it.parent_def(db);
626                match var_def {
627                    hir::VariantDef::Struct(s) => {
628                        Adt::from(s).layout(db).ok().and_then(|layout| layout.field_offset(it))
629                    }
630                    _ => None,
631                }
632            },
633            |_| None,
634            |_| None,
635        ),
636        Definition::Adt(it @ Adt::Struct(strukt)) => render_memory_layout(
637            config.memory_layout,
638            || it.layout(db),
639            |_| None,
640            |layout| {
641                let mut field_size =
642                    |i: usize| Some(strukt.fields(db).get(i)?.layout(db).ok()?.size());
643                if strukt.repr(db).is_some_and(|it| it.inhibit_struct_field_reordering()) {
644                    Some(("tail padding", layout.tail_padding(&mut field_size)?))
645                } else {
646                    Some(("largest padding", layout.largest_padding(&mut field_size)?))
647                }
648            },
649            |_| None,
650        ),
651        Definition::Adt(it) => render_memory_layout(
652            config.memory_layout,
653            || it.layout(db),
654            |_| None,
655            |_| None,
656            |_| None,
657        ),
658        Definition::Variant(it) => render_memory_layout(
659            config.memory_layout,
660            || it.layout(db),
661            |_| None,
662            |_| None,
663            |layout| layout.enum_tag_size(),
664        ),
665        Definition::TypeAlias(it) => render_memory_layout(
666            config.memory_layout,
667            || it.ty(db).layout(db),
668            |_| None,
669            |_| None,
670            |_| None,
671        ),
672        Definition::Local(it) => render_memory_layout(
673            config.memory_layout,
674            || it.ty(db).layout(db),
675            |_| None,
676            |_| None,
677            |_| None,
678        ),
679        Definition::SelfType(it) => render_memory_layout(
680            config.memory_layout,
681            || it.self_ty(db).layout(db),
682            |_| None,
683            |_| None,
684            |_| None,
685        ),
686        _ => None,
687    };
688
689    let drop_info = || {
690        if !config.show_drop_glue {
691            return None;
692        }
693        let drop_info = match def {
694            Definition::Field(field) => {
695                DropInfo { drop_glue: field.ty(db).to_type(db).drop_glue(db), has_dtor: None }
696            }
697            Definition::Adt(Adt::Struct(strukt)) => {
698                let struct_drop_glue = strukt.ty_params(db).drop_glue(db);
699                let mut fields_drop_glue = strukt
700                    .fields(db)
701                    .iter()
702                    .map(|field| field.ty(db).to_type(db).drop_glue(db))
703                    .max()
704                    .unwrap_or(DropGlue::None);
705                let has_dtor = match (fields_drop_glue, struct_drop_glue) {
706                    (DropGlue::None, _) => struct_drop_glue != DropGlue::None,
707                    (_, DropGlue::None) => {
708                        // This is `ManuallyDrop`.
709                        fields_drop_glue = DropGlue::None;
710                        false
711                    }
712                    (_, _) => struct_drop_glue > fields_drop_glue,
713                };
714                DropInfo { drop_glue: fields_drop_glue, has_dtor: Some(has_dtor) }
715            }
716            // Unions cannot have fields with drop glue.
717            Definition::Adt(Adt::Union(union)) => DropInfo {
718                drop_glue: DropGlue::None,
719                has_dtor: Some(union.ty_params(db).drop_glue(db) != DropGlue::None),
720            },
721            Definition::Adt(Adt::Enum(enum_)) => {
722                let enum_drop_glue = enum_.ty_params(db).drop_glue(db);
723                let fields_drop_glue = enum_
724                    .variants(db)
725                    .iter()
726                    .map(|variant| {
727                        variant
728                            .fields(db)
729                            .iter()
730                            .map(|field| field.ty(db).to_type(db).drop_glue(db))
731                            .max()
732                            .unwrap_or(DropGlue::None)
733                    })
734                    .max()
735                    .unwrap_or(DropGlue::None);
736                DropInfo {
737                    drop_glue: fields_drop_glue,
738                    has_dtor: Some(enum_drop_glue > fields_drop_glue),
739                }
740            }
741            Definition::Variant(variant) => {
742                let fields_drop_glue = variant
743                    .fields(db)
744                    .iter()
745                    .map(|field| field.ty(db).to_type(db).drop_glue(db))
746                    .max()
747                    .unwrap_or(DropGlue::None);
748                DropInfo { drop_glue: fields_drop_glue, has_dtor: None }
749            }
750            Definition::TypeAlias(type_alias) => {
751                DropInfo { drop_glue: type_alias.ty_params(db).drop_glue(db), has_dtor: None }
752            }
753            Definition::Local(local) => {
754                DropInfo { drop_glue: local.ty(db).drop_glue(db), has_dtor: None }
755            }
756            _ => return None,
757        };
758        let rendered_drop_glue = if drop_info.has_dtor == Some(true) {
759            "impl Drop"
760        } else {
761            match drop_info.drop_glue {
762                DropGlue::HasDropGlue => "needs Drop",
763                DropGlue::None => "no Drop",
764                DropGlue::DependOnParams => "type param may need Drop",
765            }
766        };
767
768        Some(rendered_drop_glue.to_owned())
769    };
770
771    let dyn_compatibility_info = || match def {
772        Definition::Trait(it) => {
773            let mut dyn_compatibility_info = String::new();
774            render_dyn_compatibility(db, &mut dyn_compatibility_info, it.dyn_compatibility(db));
775            Some(dyn_compatibility_info)
776        }
777        _ => None,
778    };
779
780    let variance_info = || match def {
781        Definition::GenericParam(it) => it.variance(db).as_ref().map(ToString::to_string),
782        _ => None,
783    };
784
785    let mut extra = String::new();
786    if render_extras {
787        if let Some(notable_traits) =
788            render_notable_trait(db, notable_traits, edition, display_target)
789        {
790            extra.push_str("\n___\n");
791            extra.push_str(&notable_traits);
792        }
793        if let Some(variance_info) = variance_info() {
794            extra.push_str("\n___\n");
795            extra.push_str(&variance_info);
796        }
797        if let Some(layout_info) = layout_info() {
798            extra.push_str("\n___\n");
799            extra.push_str(&layout_info);
800            if let Some(drop_info) = drop_info() {
801                extra.push_str(", ");
802                extra.push_str(&drop_info)
803            }
804        } else if let Some(drop_info) = drop_info() {
805            extra.push_str("\n___\n");
806            extra.push_str(&drop_info);
807        }
808        if let Some(dyn_compatibility_info) = dyn_compatibility_info() {
809            extra.push_str("\n___\n");
810            extra.push_str(&dyn_compatibility_info);
811        }
812    }
813    let mut desc = String::new();
814    desc.push_str(&label);
815    if let Some(value) = value() {
816        desc.push_str(" = ");
817        desc.push_str(&value);
818    }
819
820    let subst_types = match config.max_subst_ty_len {
821        SubstTyLen::Hide => String::new(),
822        SubstTyLen::LimitTo(_) | SubstTyLen::Unlimited => {
823            let limit = if let SubstTyLen::LimitTo(limit) = config.max_subst_ty_len {
824                Some(limit)
825            } else {
826                None
827            };
828            subst_types
829                .map(|subst_type| {
830                    subst_type
831                        .iter()
832                        .filter(|(_, ty)| !ty.is_unknown())
833                        .format_with(", ", |(name, ty), fmt| {
834                            fmt(&format_args!(
835                                "`{name}` = `{}`",
836                                ty.display_truncated(db, limit, display_target)
837                            ))
838                        })
839                        .to_string()
840                })
841                .unwrap_or_default()
842        }
843    };
844
845    markup(
846        docs.map(Into::into),
847        range_map,
848        desc,
849        extra.is_empty().not().then_some(extra),
850        mod_path,
851        subst_types,
852    )
853}

861pub(super) fn literal(
862    sema: &Semantics<'_, RootDatabase>,
863    token: SyntaxToken,
864    display_target: DisplayTarget,
865) -> Option<Markup> {
866    let lit = token.parent().and_then(ast::Literal::cast)?;
867    let ty = if let Some(p) = lit.syntax().parent().and_then(ast::Pat::cast) {
868        sema.type_of_pat(&p)?
869    } else {
870        sema.type_of_expr(&ast::Expr::Literal(lit))?
871    }
872    .original;
873
874    let value = match_ast! {
875        match token {
876            ast::String(string)     => string.value().as_ref().map_err(|e| format!("{e:?}")).map(ToString::to_string),
877            ast::ByteString(string) => string.value().as_ref().map_err(|e| format!("{e:?}")).map(|it| format!("{it:?}")),
878            ast::CString(string)    => string.value().as_ref().map_err(|e| format!("{e:?}")).map(|it| std::str::from_utf8(it).map_or_else(|e| format!("{e:?}"), ToOwned::to_owned)),
879            ast::Char(char)         => char  .value().as_ref().map_err(|e| format!("{e:?}")).map(ToString::to_string),
880            ast::Byte(byte)         => byte  .value().as_ref().map_err(|e| format!("{e:?}")).map(|it| format!("0x{it:X}")),
881            ast::FloatNumber(num) => {
882                let text = num.value_string();
883                if ty.as_builtin().map(|it| it.is_f16()).unwrap_or(false) {
884                    match text.parse::<f16>() {
885                        Ok(num) => Ok(format!("{num} (bits: 0x{:X})", num.to_bits())),
886                        Err(e) => Err(e.0.to_owned()),
887                    }
888                } else if ty.as_builtin().map(|it| it.is_f32()).unwrap_or(false) {
889                    match text.parse::<f32>() {
890                        Ok(num) => Ok(format!("{num} (bits: 0x{:X})", num.to_bits())),
891                        Err(e) => Err(e.to_string()),
892                    }
893                } else if ty.as_builtin().map(|it| it.is_f128()).unwrap_or(false) {
894                    match text.parse::<f128>() {
895                        Ok(num) => Ok(format!("{num} (bits: 0x{:X})", num.to_bits())),
896                        Err(e) => Err(e.0.to_owned()),
897                    }
898                } else {
899                    match text.parse::<f64>() {
900                        Ok(num) => Ok(format!("{num} (bits: 0x{:X})", num.to_bits())),
901                        Err(e) => Err(e.to_string()),
902                    }
903                }
904            },
905            ast::IntNumber(num) => match num.value() {
906                Ok(num) => Ok(format!("{num} (0x{num:X}|0b{num:b})")),
907                Err(e) => Err(e.to_string()),
908            },
909            _ => return None
910        }
911    };
912    let ty = ty.display(sema.db, display_target);
913
914    let mut s = format!("```rust\n{ty}\n```\n---\n\n");
915    match value {
916        Ok(value) => {
917            let backtick_len = value.chars().filter(|c| *c == '`').count();
918            let spaces_len = value.chars().filter(|c| *c == ' ').count();
919            let backticks = "`".repeat(backtick_len + 1);
920            let space_char = if spaces_len == value.len() { "" } else { " " };
921
922            if let Some(newline) = value.find('\n') {
923                format_to!(
924                    s,
925                    "value of literal (truncated up to newline): {backticks}{space_char}{}{space_char}{backticks}",
926                    &value[..newline]
927                )
928            } else {
929                format_to!(
930                    s,
931                    "value of literal: {backticks}{space_char}{value}{space_char}{backticks}"
932                )
933            }
934        }
935        Err(error) => format_to!(s, "invalid literal: {error}"),
936    }
937    Some(s.into())
938}

940fn render_notable_trait(
941    db: &RootDatabase,
942    notable_traits: &[(Trait, Vec<(Option<Type<'_>>, Name)>)],
943    edition: Edition,
944    display_target: DisplayTarget,
945) -> Option<String> {
946    let mut desc = String::new();
947    let mut needs_impl_header = true;
948    for (trait_, assoc_types) in notable_traits {
949        desc.push_str(if mem::take(&mut needs_impl_header) {
950            "Implements notable traits: `"
951        } else {
952            "`, `"
953        });
954        format_to!(desc, "{}", trait_.name(db).display(db, edition));
955        if !assoc_types.is_empty() {
956            desc.push('<');
957            format_to!(
958                desc,
959                "{}",
960                assoc_types.iter().format_with(", ", |(ty, name), f| {
961                    f(&name.display(db, edition))?;
962                    f(&" = ")?;
963                    match ty {
964                        Some(ty) => f(&ty.display(db, display_target)),
965                        None => f(&"?"),
966                    }
967                })
968            );
969            desc.push('>');
970        }
971    }
972    if desc.is_empty() {
973        None
974    } else {
975        desc.push('`');
976        Some(desc)
977    }
978}

980fn type_info(
981    sema: &Semantics<'_, RootDatabase>,
982    config: &HoverConfig<'_>,
983    ty: TypeInfo<'_>,
984    edition: Edition,
985    display_target: DisplayTarget,
986) -> Option<HoverResult> {
987    if let Some(res) = closure_ty(sema, config, &ty, edition, display_target) {
988        return Some(res);
989    };
990    let db = sema.db;
991    let TypeInfo { original, adjusted } = ty;
992    let mut res = HoverResult::default();
993    let mut targets: Vec<hir::ModuleDef> = Vec::new();
994    let mut push_new_def = |item: hir::ModuleDef| {
995        if !targets.contains(&item) {
996            targets.push(item);
997        }
998    };
999    walk_and_push_ty(db, &original, &mut push_new_def);
1000
1001    res.markup = if let Some(adjusted_ty) = adjusted {
1002        walk_and_push_ty(db, &adjusted_ty, &mut push_new_def);
1003
1004        let notable = if let Some(notable) =
1005            render_notable_trait(db, &notable_traits(db, &original), edition, display_target)
1006        {
1007            format!("{notable}\n")
1008        } else {
1009            String::new()
1010        };
1011
1012        let original = original.display(db, display_target).to_string();
1013        let adjusted = adjusted_ty.display(db, display_target).to_string();
1014        let static_text_diff_len = "Coerced to: ".len() - "Type: ".len();
1015        format!(
1016            "```text\nType: {:>apad$}\nCoerced to: {:>opad$}\n{notable}```\n",
1017            original,
1018            adjusted,
1019            apad = static_text_diff_len + adjusted.len().max(original.len()),
1020            opad = original.len(),
1021        )
1022        .into()
1023    } else {
1024        let mut desc = format!("```rust\n{}\n```", original.display(db, display_target));
1025        if let Some(extra) =
1026            render_notable_trait(db, &notable_traits(db, &original), edition, display_target)
1027        {
1028            desc.push_str("\n---\n");
1029            desc.push_str(&extra);
1030        };
1031        desc.into()
1032    };
1033    if let Some(actions) = HoverAction::goto_type_from_targets(sema, targets, edition) {
1034        res.actions.push(actions);
1035    }
1036    Some(res)
1037}

1039fn closure_ty(
1040    sema: &Semantics<'_, RootDatabase>,
1041    config: &HoverConfig<'_>,
1042    TypeInfo { original, adjusted }: &TypeInfo<'_>,
1043    edition: Edition,
1044    display_target: DisplayTarget,
1045) -> Option<HoverResult> {
1046    let c = original.as_closure()?;
1047    let mut captures_rendered = c.captured_items(sema.db)
1048        .into_iter()
1049        .map(|it| {
1050            let borrow_kind = match it.kind() {
1051                CaptureKind::SharedRef => "immutable borrow",
1052                CaptureKind::UniqueSharedRef => "unique immutable borrow ([read more](https://doc.rust-lang.org/stable/reference/types/closure.html#unique-immutable-borrows-in-captures))",
1053                CaptureKind::MutableRef => "mutable borrow",
1054                CaptureKind::Move => "move",
1055            };
1056            format!("* `{}` by {}", it.display_place(sema.db), borrow_kind)
1057        })
1058        .join("\n");
1059    if captures_rendered.trim().is_empty() {
1060        "This closure captures nothing".clone_into(&mut captures_rendered);
1061    }
1062    let mut targets: Vec<hir::ModuleDef> = Vec::new();
1063    let mut push_new_def = |item: hir::ModuleDef| {
1064        if !targets.contains(&item) {
1065            targets.push(item);
1066        }
1067    };
1068    walk_and_push_ty(sema.db, original, &mut push_new_def);
1069    c.capture_types(sema.db).into_iter().for_each(|ty| {
1070        walk_and_push_ty(sema.db, &ty, &mut push_new_def);
1071    });
1072
1073    let adjusted = if let Some(adjusted_ty) = adjusted {
1074        walk_and_push_ty(sema.db, adjusted_ty, &mut push_new_def);
1075        format!(
1076            "\nCoerced to: {}",
1077            adjusted_ty
1078                .display(sema.db, display_target)
1079                .with_closure_style(hir::ClosureStyle::ImplFn)
1080        )
1081    } else {
1082        String::new()
1083    };
1084    let mut markup = format!("```rust\n{}\n```", c.display_with_impl(sema.db, display_target));
1085
1086    if let Some(trait_) = c.fn_trait(sema.db).get_id(sema.db, original.krate(sema.db).into()) {
1087        push_new_def(hir::Trait::from(trait_).into())
1088    }
1089    if let Some(layout) = render_memory_layout(
1090        config.memory_layout,
1091        || original.layout(sema.db),
1092        |_| None,
1093        |_| None,
1094        |_| None,
1095    ) {
1096        format_to!(markup, "\n---\n{layout}");
1097    }
1098    format_to!(markup, "{adjusted}\n\n## Captures\n{}", captures_rendered,);
1099
1100    let mut res = HoverResult::default();
1101    if let Some(actions) = HoverAction::goto_type_from_targets(sema, targets, edition) {
1102        res.actions.push(actions);
1103    }
1104    res.markup = markup.into();
1105    Some(res)
1106}

1108fn definition_path(db: &RootDatabase, &def: &Definition, edition: Edition) -> Option<String> {
1109    if matches!(
1110        def,
1111        Definition::TupleField(_)
1112            | Definition::Label(_)
1113            | Definition::Local(_)
1114            | Definition::BuiltinAttr(_)
1115            | Definition::BuiltinLifetime(_)
1116            | Definition::BuiltinType(_)
1117            | Definition::InlineAsmRegOrRegClass(_)
1118            | Definition::InlineAsmOperand(_)
1119    ) {
1120        return None;
1121    }
1122    let rendered_parent = definition_owner_name(db, def, edition);
1123    def.module(db).map(|module| path(db, module, rendered_parent, edition))
1124}

1126fn markup(
1127    docs: Option<String>,
1128    range_map: Option<DocsRangeMap>,
1129    rust: String,
1130    extra: Option<String>,
1131    mod_path: Option<String>,
1132    subst_types: String,
1133) -> (Markup, Option<DocsRangeMap>) {
1134    let mut buf = String::new();
1135
1136    if let Some(mod_path) = mod_path
1137        && !mod_path.is_empty()
1138    {
1139        format_to!(buf, "```rust\n{}\n```\n\n", mod_path);
1140    }
1141    format_to!(buf, "```rust\n{}\n```", rust);
1142
1143    if let Some(extra) = extra {
1144        buf.push_str(&extra);
1145    }
1146
1147    if !subst_types.is_empty() {
1148        format_to!(buf, "\n___\n{subst_types}");
1149    }
1150
1151    if let Some(doc) = docs {
1152        format_to!(buf, "\n___\n\n");
1153        let offset = TextSize::new(buf.len() as u32);
1154        let buf_range_map = range_map.map(|range_map| range_map.shift_docstring_line_range(offset));
1155        format_to!(buf, "{}", doc);
1156
1157        (buf.into(), buf_range_map)
1158    } else {
1159        (buf.into(), None)
1160    }
1161}

1163fn render_memory_layout(
1164    config: Option<MemoryLayoutHoverConfig>,
1165    layout: impl FnOnce() -> Result<Layout, LayoutError>,
1166    offset: impl FnOnce(&Layout) -> Option<u64>,
1167    padding: impl FnOnce(&Layout) -> Option<(&str, u64)>,
1168    tag: impl FnOnce(&Layout) -> Option<usize>,
1169) -> Option<String> {
1170    let config = config?;
1171    let layout = layout().ok()?;
1172
1173    let mut label = String::new();
1174
1175    if let Some(render) = config.size {
1176        let size = match tag(&layout) {
1177            Some(tag) => layout.size() as usize - tag,
1178            None => layout.size() as usize,
1179        };
1180        format_to!(label, "size = ");
1181        match render {
1182            MemoryLayoutHoverRenderKind::Decimal => format_to!(label, "{size}"),
1183            MemoryLayoutHoverRenderKind::Hexadecimal => format_to!(label, "{size:#X}"),
1184            MemoryLayoutHoverRenderKind::Both if size >= 10 => {
1185                format_to!(label, "{size} ({size:#X})")
1186            }
1187            MemoryLayoutHoverRenderKind::Both => format_to!(label, "{size}"),
1188        }
1189        format_to!(label, ", ");
1190    }
1191
1192    if let Some(render) = config.alignment {
1193        let align = layout.align();
1194        format_to!(label, "align = ");
1195        match render {
1196            MemoryLayoutHoverRenderKind::Decimal => format_to!(label, "{align}",),
1197            MemoryLayoutHoverRenderKind::Hexadecimal => format_to!(label, "{align:#X}",),
1198            MemoryLayoutHoverRenderKind::Both if align >= 10 => {
1199                format_to!(label, "{align} ({align:#X})")
1200            }
1201            MemoryLayoutHoverRenderKind::Both => {
1202                format_to!(label, "{align}")
1203            }
1204        }
1205        format_to!(label, ", ");
1206    }
1207
1208    if let Some(render) = config.offset
1209        && let Some(offset) = offset(&layout)
1210    {
1211        format_to!(label, "offset = ");
1212        match render {
1213            MemoryLayoutHoverRenderKind::Decimal => format_to!(label, "{offset}"),
1214            MemoryLayoutHoverRenderKind::Hexadecimal => format_to!(label, "{offset:#X}"),
1215            MemoryLayoutHoverRenderKind::Both if offset >= 10 => {
1216                format_to!(label, "{offset} ({offset:#X})")
1217            }
1218            MemoryLayoutHoverRenderKind::Both => {
1219                format_to!(label, "{offset}")
1220            }
1221        }
1222        format_to!(label, ", ");
1223    }
1224
1225    if let Some(render) = config.padding
1226        && let Some((padding_name, padding)) = padding(&layout)
1227    {
1228        format_to!(label, "{padding_name} = ");
1229        match render {
1230            MemoryLayoutHoverRenderKind::Decimal => format_to!(label, "{padding}"),
1231            MemoryLayoutHoverRenderKind::Hexadecimal => format_to!(label, "{padding:#X}"),
1232            MemoryLayoutHoverRenderKind::Both if padding >= 10 => {
1233                format_to!(label, "{padding} ({padding:#X})")
1234            }
1235            MemoryLayoutHoverRenderKind::Both => {
1236                format_to!(label, "{padding}")
1237            }
1238        }
1239        format_to!(label, ", ");
1240    }
1241
1242    if config.niches
1243        && let Some(niches) = layout.niches()
1244    {
1245        if niches > 1024 {
1246            if niches.is_power_of_two() {
1247                format_to!(label, "niches = 2{}, ", pwr2_to_exponent(niches));
1248            } else if is_pwr2plus1(niches) {
1249                format_to!(label, "niches = 2{} + 1, ", pwr2_to_exponent(niches - 1));
1250            } else if is_pwr2minus1(niches) {
1251                format_to!(label, "niches = 2{} - 1, ", pwr2_to_exponent(niches + 1));
1252            } else {
1253                format_to!(label, "niches = a lot, ");
1254            }
1255        } else {
1256            format_to!(label, "niches = {niches}, ");
1257        }
1258    }
1259    label.pop(); // ' '
1260    label.pop(); // ','
1261    Some(label)
1262}

1264struct KeywordHint {
1265    description: String,
1266    keyword_mod: String,
1267    actions: Vec<HoverAction>,
1268}
1269
1270impl KeywordHint {
1271    fn new(description: String, keyword_mod: String) -> Self {
1272        Self { description, keyword_mod, actions: Vec::default() }
1273    }
1274}
1275
1276fn keyword_hints(
1277    sema: &Semantics<'_, RootDatabase>,
1278    token: &SyntaxToken,
1279    parent: syntax::SyntaxNode,
1280    edition: Edition,
1281    display_target: DisplayTarget,
1282) -> KeywordHint {
1283    match token.kind() {
1284        T![await] | T![loop] | T![match] | T![unsafe] | T![as] | T![try] | T![if] | T![else] => {
1285            let keyword_mod = format!("{}_keyword", token.text());
1286
1287            match ast::Expr::cast(parent).and_then(|site| sema.type_of_expr(&site)) {
1288                // ignore the unit type ()
1289                Some(ty) if !ty.adjusted.as_ref().unwrap_or(&ty.original).is_unit() => {
1290                    let mut targets: Vec<hir::ModuleDef> = Vec::new();
1291                    let mut push_new_def = |item: hir::ModuleDef| {
1292                        if !targets.contains(&item) {
1293                            targets.push(item);
1294                        }
1295                    };
1296                    walk_and_push_ty(sema.db, &ty.original, &mut push_new_def);
1297
1298                    let ty = ty.adjusted();
1299                    let description =
1300                        format!("{}: {}", token.text(), ty.display(sema.db, display_target));
1301
1302                    KeywordHint {
1303                        description,
1304                        keyword_mod,
1305                        actions: HoverAction::goto_type_from_targets(sema, targets, edition)
1306                            .into_iter()
1307                            .collect(),
1308                    }
1309                }
1310                _ => KeywordHint {
1311                    description: token.text().to_owned(),
1312                    keyword_mod,
1313                    actions: Vec::new(),
1314                },
1315            }
1316        }
1317        T![fn] => {
1318            let module = match ast::FnPtrType::cast(parent) {
1319                // treat fn keyword inside function pointer type as primitive
1320                Some(_) => format!("prim_{}", token.text()),
1321                None => format!("{}_keyword", token.text()),
1322            };
1323            KeywordHint::new(token.text().to_owned(), module)
1324        }
1325        T![Self] => KeywordHint::new(token.text().to_owned(), "self_upper_keyword".into()),
1326        _ => KeywordHint::new(token.text().to_owned(), format!("{}_keyword", token.text())),
1327    }
1328}

1330fn render_dyn_compatibility(
1331    db: &RootDatabase,
1332    buf: &mut String,
1333    safety: Option<DynCompatibilityViolation>,
1334) {
1335    let Some(osv) = safety else {
1336        buf.push_str("Is dyn-compatible");
1337        return;
1338    };
1339    buf.push_str("Is not dyn-compatible due to ");
1340    match osv {
1341        DynCompatibilityViolation::SizedSelf => {
1342            buf.push_str("having a `Self: Sized` bound");
1343        }
1344        DynCompatibilityViolation::SelfReferential => {
1345            buf.push_str("having a bound that references `Self`");
1346        }
1347        DynCompatibilityViolation::Method(func, mvc) => {
1348            let name = hir::Function::from(func).name(db);
1349            format_to!(buf, "having a method `{}` that is not dispatchable due to ", name.as_str());
1350            let desc = match mvc {
1351                MethodViolationCode::StaticMethod => "missing a receiver",
1352                MethodViolationCode::ReferencesSelfInput => "having a parameter referencing `Self`",
1353                MethodViolationCode::ReferencesSelfOutput => "the return type referencing `Self`",
1354                MethodViolationCode::ReferencesImplTraitInTrait => {
1355                    "the return type containing `impl Trait`"
1356                }
1357                MethodViolationCode::AsyncFn => "being async",
1358                MethodViolationCode::WhereClauseReferencesSelf => {
1359                    "a where clause referencing `Self`"
1360                }
1361                MethodViolationCode::Generic => "having a const or type generic parameter",
1362                MethodViolationCode::UndispatchableReceiver => {
1363                    "having a non-dispatchable receiver type"
1364                }
1365            };
1366            buf.push_str(desc);
1367        }
1368        DynCompatibilityViolation::AssocConst(const_) => {
1369            let name = hir::Const::from(const_).name(db);
1370            if let Some(name) = name {
1371                format_to!(buf, "having an associated constant `{}`", name.as_str());
1372            } else {
1373                buf.push_str("having an associated constant");
1374            }
1375        }
1376        DynCompatibilityViolation::GAT(alias) => {
1377            let name = hir::TypeAlias::from(alias).name(db);
1378            format_to!(buf, "having a generic associated type `{}`", name.as_str());
1379        }
1380        DynCompatibilityViolation::HasNonCompatibleSuperTrait(super_trait) => {
1381            let name = hir::Trait::from(super_trait).name(db);
1382            format_to!(buf, "having a dyn-incompatible supertrait `{}`", name.as_str());
1383        }
1384    }
1385}

1387fn is_pwr2minus1(val: u128) -> bool {
1388    val == u128::MAX || (val + 1).is_power_of_two()
1389}
1390
1391fn is_pwr2plus1(val: u128) -> bool {
1392    val != 0 && (val - 1).is_power_of_two()
1393}

1395/// Formats a power of two as an exponent of two, i.e. 16 => ⁴. Note that `num` MUST be a power
1396/// of 2, or this function will panic.
1397fn pwr2_to_exponent(num: u128) -> String {
1398    const DIGITS: [char; 10] = ['⁰', '¹', '²', '³', '⁴', '⁵', '⁶', '⁷', '⁸', '⁹'];
1399    assert_eq!(num.count_ones(), 1);
1400    num.trailing_zeros()
1401        .to_string()
1402        .chars()
1403        .map(|c| c.to_digit(10).unwrap() as usize)
1404        .map(|idx| DIGITS[idx])
1405        .collect::<String>()
1406}
