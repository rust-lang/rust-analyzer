22pub(super) fn ra_fixture(
23    hl: &mut Highlights,
24    sema: &Semantics<'_, RootDatabase>,
25    config: &HighlightConfig<'_>,
26    literal: &ast::String,
27    expanded: &ast::String,
28) -> Option<()> {
29    let (analysis, fixture_analysis) = Analysis::from_ra_fixture_with_on_cursor(
30        sema,
31        literal.clone(),
32        expanded,
33        config.minicore,
34        &mut |range| {
35            hl.add(HlRange {
36                range,
37                highlight: HlTag::Keyword | HlMod::Injected,
38                binding_hash: None,
39            });
40        },
41    )?;
42
43    if let Some(range) = literal.open_quote_text_range() {
44        hl.add(HlRange { range, highlight: HlTag::StringLiteral.into(), binding_hash: None })
45    }
46
47    for tmp_file_id in fixture_analysis.files() {
48        for mut hl_range in analysis
49            .highlight(
50                HighlightConfig {
51                    syntactic_name_ref_highlighting: false,
52                    comments: true,
53                    punctuation: true,
54                    operator: true,
55                    strings: true,
56                    specialize_punctuation: config.specialize_punctuation,
57                    specialize_operator: config.operator,
58                    inject_doc_comment: config.inject_doc_comment,
59                    macro_bang: config.macro_bang,
60                    // What if there is a fixture inside a fixture? It's fixtures all the way down.
61                    // (In fact, we have a fixture inside a fixture in our test suite!)
62                    minicore: config.minicore,
63                },
64                tmp_file_id,
65            )
66            .unwrap()
67        {
68            for range in fixture_analysis.map_range_up(tmp_file_id, hl_range.range) {
69                hl_range.range = range;
70                hl_range.highlight |= HlMod::Injected;
71                hl.add(hl_range);
72            }
73        }
74    }
75
76    if let Some(range) = literal.close_quote_text_range() {
77        hl.add(HlRange { range, highlight: HlTag::StringLiteral.into(), binding_hash: None })
78    }
79
80    Some(())
81}

83const RUSTDOC_FENCE_LENGTH: usize = 3;
84const RUSTDOC_FENCES: [&str; 2] = ["```", "~~~"];

86/// Injection of syntax highlighting of doctests and intra doc links.
87pub(super) fn doc_comment(
88    hl: &mut Highlights,
89    sema: &Semantics<'_, RootDatabase>,
90    config: &HighlightConfig<'_>,
91    src_file_id: EditionedFileId,
92    node: &SyntaxNode,
93) {
94    let (attributes, def) = match doc_attributes(sema, node) {
95        Some(it) => it,
96        None => return,
97    };
98    let src_file_id: HirFileId = src_file_id.into();
99
100    // Extract intra-doc links and emit highlights for them.
101    if let Some((docs, doc_mapping)) = docs_with_rangemap(sema.db, &attributes) {
102        extract_definitions_from_docs(&docs)
103            .into_iter()
104            .filter_map(|(range, link, ns)| {
105                doc_mapping
106                    .map(range)
107                    .filter(|(mapping, _)| mapping.file_id == src_file_id)
108                    .and_then(|(InFile { value: mapped_range, .. }, attr_id)| {
109                        Some(mapped_range).zip(resolve_doc_path_for_def(
110                            sema.db,
111                            def,
112                            &link,
113                            ns,
114                            attr_id.is_inner_attr(),
115                        ))
116                    })
117            })
118            .for_each(|(range, def)| {
119                hl.add(HlRange {
120                    range,
121                    highlight: module_def_to_hl_tag(def)
122                        | HlMod::Documentation
123                        | HlMod::Injected
124                        | HlMod::IntraDocLink,
125                    binding_hash: None,
126                })
127            })
128    }
129
130    // Extract doc-test sources from the docs and calculate highlighting for them.
131
132    let mut inj = RangeMapper::default();
133    inj.add_unmapped("fn doctest() {\n");
134
135    let attrs_source_map = attributes.source_map(sema.db);
136
137    let mut is_codeblock = false;
138    let mut is_doctest = false;
139
140    let mut new_comments = Vec::new();
141    let mut string;
142
143    for attr in attributes.by_key(sym::doc).attrs() {
144        let InFile { file_id, value: src } = attrs_source_map.source_of(attr);
145        if file_id != src_file_id {
146            continue;
147        }
148        let (line, range) = match &src {
149            Either::Left(it) => {
150                string = match find_doc_string_in_attr(attr, it) {
151                    Some(it) => it,
152                    None => continue,
153                };
154                let text = string.text();
155                let text_range = string.syntax().text_range();
156                match string.quote_offsets() {
157                    Some(QuoteOffsets { contents, .. }) => {
158                        (&text[contents - text_range.start()], contents)
159                    }
160                    None => (text, text_range),
161                }
162            }
163            Either::Right(comment) => {
164                let value = comment.prefix().len();
165                let range = comment.syntax().text_range();
166                (
167                    &comment.text()[value..],
168                    TextRange::new(range.start() + TextSize::try_from(value).unwrap(), range.end()),
169                )
170            }
171        };
172
173        let mut range_start = range.start();
174        for line in line.split('\n') {
175            let line_len = TextSize::from(line.len() as u32);
176            let prev_range_start = {
177                let next_range_start = range_start + line_len + TextSize::from(1);
178                mem::replace(&mut range_start, next_range_start)
179            };
180            let mut pos = TextSize::from(0);
181
182            match RUSTDOC_FENCES.into_iter().find_map(|fence| line.find(fence)) {
183                Some(idx) => {
184                    is_codeblock = !is_codeblock;
185                    // Check whether code is rust by inspecting fence guards
186                    let guards = &line[idx + RUSTDOC_FENCE_LENGTH..];
187                    let is_rust = is_rust_fence(guards);
188                    is_doctest = is_codeblock && is_rust;
189                    continue;
190                }
191                None if !is_doctest => continue,
192                None => (),
193            }
194
195            // whitespace after comment is ignored
196            if let Some(ws) = line[pos.into()..].chars().next().filter(|c| c.is_whitespace()) {
197                pos += TextSize::of(ws);
198            }
199            // lines marked with `#` should be ignored in output, we skip the `#` char
200            if line[pos.into()..].starts_with('#') {
201                pos += TextSize::of('#');
202            }
203
204            new_comments.push(TextRange::at(prev_range_start, pos));
205            inj.add(&line[pos.into()..], TextRange::new(pos, line_len) + prev_range_start);
206            inj.add_unmapped("\n");
207        }
208    }
209
210    if new_comments.is_empty() {
211        return; // no need to run an analysis on an empty file
212    }
213
214    inj.add_unmapped("\n}");
215
216    let (analysis, tmp_file_id) = Analysis::from_single_file(inj.take_text());
217
218    if let Ok(ranges) = analysis.with_db(|db| {
219        super::highlight(
220            db,
221            &HighlightConfig {
222                syntactic_name_ref_highlighting: true,
223                comments: true,
224                punctuation: true,
225                operator: true,
226                strings: true,
227                specialize_punctuation: config.specialize_punctuation,
228                specialize_operator: config.operator,
229                inject_doc_comment: config.inject_doc_comment,
230                macro_bang: config.macro_bang,
231                minicore: config.minicore,
232            },
233            tmp_file_id,
234            None,
235        )
236    }) {
237        for HlRange { range, highlight, binding_hash } in ranges {
238            for range in inj.map_range_up(range) {
239                hl.add(HlRange { range, highlight: highlight | HlMod::Injected, binding_hash });
240            }
241        }
242    }
243
244    for range in new_comments {
245        hl.add(HlRange {
246            range,
247            highlight: HlTag::Comment | HlMod::Documentation,
248            binding_hash: None,
249        });
250    }
251}

253fn find_doc_string_in_attr(attr: &hir::Attr, it: &ast::Attr) -> Option<ast::String> {
254    match it.expr() {
255        // #[doc = lit]
256        Some(ast::Expr::Literal(lit)) => match lit.kind() {
257            ast::LiteralKind::String(it) => Some(it),
258            _ => None,
259        },
260        // #[cfg_attr(..., doc = "", ...)]
261        None => {
262            // We gotta hunt the string token manually here
263            let text = attr.string_value()?.as_str();
264            // FIXME: We just pick the first string literal that has the same text as the doc attribute
265            // This means technically we might highlight the wrong one
266            it.syntax()
267                .descendants_with_tokens()
268                .filter_map(NodeOrToken::into_token)
269                .filter_map(ast::String::cast)
270                .find(|string| string.text().get(1..string.text().len() - 1) == Some(text))
271        }
272        _ => None,
273    }
274}

276fn module_def_to_hl_tag(def: Definition) -> HlTag {
277    let symbol = match def {
278        Definition::Module(_) | Definition::Crate(_) | Definition::ExternCrateDecl(_) => {
279            SymbolKind::Module
280        }
281        Definition::Function(_) => SymbolKind::Function,
282        Definition::Adt(hir::Adt::Struct(_)) => SymbolKind::Struct,
283        Definition::Adt(hir::Adt::Enum(_)) => SymbolKind::Enum,
284        Definition::Adt(hir::Adt::Union(_)) => SymbolKind::Union,
285        Definition::Variant(_) => SymbolKind::Variant,
286        Definition::Const(_) => SymbolKind::Const,
287        Definition::Static(_) => SymbolKind::Static,
288        Definition::Trait(_) => SymbolKind::Trait,
289        Definition::TypeAlias(_) => SymbolKind::TypeAlias,
290        Definition::BuiltinLifetime(_) => SymbolKind::LifetimeParam,
291        Definition::BuiltinType(_) => return HlTag::BuiltinType,
292        Definition::Macro(_) => SymbolKind::Macro,
293        Definition::Field(_) | Definition::TupleField(_) => SymbolKind::Field,
294        Definition::SelfType(_) => SymbolKind::Impl,
295        Definition::Local(_) => SymbolKind::Local,
296        Definition::GenericParam(gp) => match gp {
297            hir::GenericParam::TypeParam(_) => SymbolKind::TypeParam,
298            hir::GenericParam::ConstParam(_) => SymbolKind::ConstParam,
299            hir::GenericParam::LifetimeParam(_) => SymbolKind::LifetimeParam,
300        },
301        Definition::Label(_) => SymbolKind::Label,
302        Definition::BuiltinAttr(_) => SymbolKind::BuiltinAttr,
303        Definition::ToolModule(_) => SymbolKind::ToolModule,
304        Definition::DeriveHelper(_) => SymbolKind::DeriveHelper,
305        Definition::InlineAsmRegOrRegClass(_) => SymbolKind::InlineAsmRegOrRegClass,
306        Definition::InlineAsmOperand(_) => SymbolKind::Local,
307    };
308    HlTag::Symbol(symbol)
