1//! This module takes a (parsed) definition of `macro_rules` invocation, a
2//! `tt::TokenTree` representing an argument of macro invocation, and produces a
3//! `tt::TokenTree` for the result of the expansion.
4
5mod matcher;
6mod transcriber;
7
8use intern::Symbol;
9use rustc_hash::FxHashMap;
10use span::{Edition, Span};
11
12use crate::{ExpandError, ExpandErrorKind, ExpandResult, MatchedArmIndex, parser::MetaVarKind};
13
14pub(crate) fn expand_rules(
15    rules: &[crate::Rule],
16    input: &tt::TopSubtree<Span>,
17    marker: impl Fn(&mut Span) + Copy,
18    call_site: Span,
19    def_site_edition: Edition,
20) -> ExpandResult<(tt::TopSubtree<Span>, MatchedArmIndex)> {
21    let mut match_: Option<(matcher::Match<'_>, &crate::Rule, usize)> = None;
22    for (idx, rule) in rules.iter().enumerate() {
23        let new_match = matcher::match_(&rule.lhs, input, def_site_edition);
24
25        if new_match.err.is_none() {
26            // If we find a rule that applies without errors, we're done.
27            // Unconditionally returning the transcription here makes the
28            // `test_repeat_bad_var` test fail.
29            let ExpandResult { value, err: transcribe_err } =
30                transcriber::transcribe(&rule.rhs, &new_match.bindings, marker, call_site);
31            if transcribe_err.is_none() {
32                return ExpandResult::ok((value, Some(idx as u32)));
33            }
34        }
35        // Use the rule if we matched more tokens, or bound variables count
36        if let Some((prev_match, _, _)) = &match_ {
37            if (new_match.unmatched_tts, -(new_match.bound_count as i32))
38                < (prev_match.unmatched_tts, -(prev_match.bound_count as i32))
39            {
40                match_ = Some((new_match, rule, idx));
41            }
42        } else {
43            match_ = Some((new_match, rule, idx));
44        }
45    }
46    if let Some((match_, rule, idx)) = match_ {
47        // if we got here, there was no match without errors
48        let ExpandResult { value, err: transcribe_err } =
49            transcriber::transcribe(&rule.rhs, &match_.bindings, marker, call_site);
50        ExpandResult { value: (value, idx.try_into().ok()), err: match_.err.or(transcribe_err) }
51    } else {
52        ExpandResult::new(
53            (tt::TopSubtree::empty(tt::DelimSpan::from_single(call_site)), None),
54            ExpandError::new(call_site, ExpandErrorKind::NoMatchingRule),
55        )
56    }
57}

59/// The actual algorithm for expansion is not too hard, but is pretty tricky.
60/// `Bindings` structure is the key to understanding what we are doing here.
61///
62/// On the high level, it stores mapping from meta variables to the bits of
63/// syntax it should be substituted with. For example, if `$e:expr` is matched
64/// with `1 + 1` by macro_rules, the `Binding` will store `$e -> 1 + 1`.
65///
66/// The tricky bit is dealing with repetitions (`$()*`). Consider this example:
67///
68/// ```not_rust
69/// macro_rules! foo {
70///     ($($ i:ident $($ e:expr),*);*) => {
71///         $(fn $ i() { $($ e);*; })*
72///     }
73/// }
74/// foo! { foo 1,2,3; bar 4,5,6 }
75/// ```
76///
77/// Here, the `$i` meta variable is matched first with `foo` and then with
78/// `bar`, and `$e` is matched in turn with `1`, `2`, `3`, `4`, `5`, `6`.
79///
80/// To represent such "multi-mappings", we use a recursive structures: we map
81/// variables not to values, but to *lists* of values or other lists (that is,
82/// to the trees).
83///
84/// For the above example, the bindings would store
85///
86/// ```not_rust
87/// i -> [foo, bar]
88/// e -> [[1, 2, 3], [4, 5, 6]]
89/// ```
90///
91/// We construct `Bindings` in the `match_lhs`. The interesting case is
92/// `TokenTree::Repeat`, where we use `push_nested` to create the desired
93/// nesting structure.
94///
95/// The other side of the puzzle is `expand_subtree`, where we use the bindings
96/// to substitute meta variables in the output template. When expanding, we
97/// maintain a `nesting` stack of indices which tells us which occurrence from
98/// the `Bindings` we should take. We push to the stack when we enter a
99/// repetition.
100///
101/// In other words, `Bindings` is a *multi* mapping from `Symbol` to
102/// `tt::TokenTree`, where the index to select a particular `TokenTree` among
103/// many is not a plain `usize`, but a `&[usize]`.
104#[derive(Debug, Default, Clone)]
105struct Bindings<'a> {
106    inner: FxHashMap<Symbol, Binding<'a>>,
107}
108
109#[derive(Debug, Clone)]
110enum Binding<'a> {
111    Fragment(Fragment<'a>),
112    Nested(Vec<Binding<'a>>),
113    Empty,
114    Missing(MetaVarKind),
115}
116
117#[derive(Debug, Default, Clone)]
118enum Fragment<'a> {
119    #[default]
120    Empty,
121    /// token fragments are just copy-pasted into the output
122    Tokens(tt::TokenTreesView<'a, Span>),
123    /// Expr ast fragments are surrounded with `()` on transcription to preserve precedence.
124    /// Note that this impl is different from the one currently in `rustc` --
125    /// `rustc` doesn't translate fragments into token trees at all.
126    ///
127    /// At one point in time, we tried to use "fake" delimiters here à la
128    /// proc-macro delimiter=none. As we later discovered, "none" delimiters are
129    /// tricky to handle in the parser, and rustc doesn't handle those either.
130    ///
131    /// The span of the outer delimiters is marked on transcription.
132    Expr(tt::TokenTreesView<'a, Span>),
133    /// There are roughly two types of paths: paths in expression context, where a
134    /// separator `::` between an identifier and its following generic argument list
135    /// is mandatory, and paths in type context, where `::` can be omitted.
136    ///
137    /// Unlike rustc, we need to transform the parsed fragments back into tokens
138    /// during transcription. When the matched path fragment is a type-context path
139    /// and is trasncribed as an expression-context path, verbatim transcription
140    /// would cause a syntax error. We need to fix it up just before transcribing;
141    /// see `transcriber::fix_up_and_push_path_tt()`.
142    Path(tt::TokenTreesView<'a, Span>),
143    TokensOwned(tt::TopSubtree<Span>),
144}
145
146impl Fragment<'_> {
147    fn is_empty(&self) -> bool {
148        match self {
149            Fragment::Empty => true,
150            Fragment::Tokens(it) => it.len() == 0,
151            Fragment::Expr(it) => it.len() == 0,
152            Fragment::Path(it) => it.len() == 0,
153            Fragment::TokensOwned(it) => it.0.is_empty(),
154        }
155    }
156}
