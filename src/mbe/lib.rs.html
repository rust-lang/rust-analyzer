36#[derive(Debug, PartialEq, Eq, Clone)]
37pub enum ParseError {
38    UnexpectedToken(Box<str>),
39    Expected(Box<str>),
40    InvalidRepeat,
41    RepetitionEmptyTokenTree,
42}
43
44impl ParseError {
45    fn expected(e: &str) -> ParseError {
46        ParseError::Expected(e.into())
47    }
48
49    fn unexpected(e: &str) -> ParseError {
50        ParseError::UnexpectedToken(e.into())
51    }
52}
53
54impl fmt::Display for ParseError {
55    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
56        match self {
57            ParseError::UnexpectedToken(it) => f.write_str(it),
58            ParseError::Expected(it) => f.write_str(it),
59            ParseError::InvalidRepeat => f.write_str("invalid repeat"),
60            ParseError::RepetitionEmptyTokenTree => f.write_str("empty token tree in repetition"),
61        }
62    }
63}

65#[derive(Debug, PartialEq, Eq, Clone, Hash)]
66pub struct ExpandError {
67    pub inner: Arc<(Span, ExpandErrorKind)>,
68}
69#[derive(Debug, PartialEq, Eq, Clone, Hash)]
70pub enum ExpandErrorKind {
71    BindingError(Box<Box<str>>),
72    UnresolvedBinding(Box<Box<str>>),
73    LeftoverTokens,
74    LimitExceeded,
75    NoMatchingRule,
76    UnexpectedToken,
77}
78
79impl ExpandError {
80    fn new(span: Span, kind: ExpandErrorKind) -> ExpandError {
81        ExpandError { inner: Arc::new((span, kind)) }
82    }
83    fn binding_error(span: Span, e: impl Into<Box<str>>) -> ExpandError {
84        ExpandError { inner: Arc::new((span, ExpandErrorKind::BindingError(Box::new(e.into())))) }
85    }
86}
87impl fmt::Display for ExpandError {
88    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
89        self.inner.1.fmt(f)
90    }
91}
92
93impl fmt::Display for ExpandErrorKind {
94    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
95        match self {
96            ExpandErrorKind::NoMatchingRule => f.write_str("no rule matches input tokens"),
97            ExpandErrorKind::UnexpectedToken => f.write_str("unexpected token in input"),
98            ExpandErrorKind::BindingError(e) => f.write_str(e),
99            ExpandErrorKind::UnresolvedBinding(binding) => {
100                f.write_str("could not find binding ")?;
101                f.write_str(binding)
102            }
103            ExpandErrorKind::LimitExceeded => f.write_str("Expand exceed limit"),
104            ExpandErrorKind::LeftoverTokens => f.write_str("leftover tokens"),
105        }
106    }
107}

128/// This struct contains AST for a single `macro_rules` definition. What might
129/// be very confusing is that AST has almost exactly the same shape as
130/// `tt::TokenTree`, but there's a crucial difference: in macro rules, `$ident`
131/// and `$()*` have special meaning (see `Var` and `Repeat` data structures)
132#[derive(Clone, Debug, PartialEq, Eq)]
133pub struct DeclarativeMacro {
134    rules: Box<[Rule]>,
135    err: Option<Box<ParseError>>,
136}
137
138#[derive(Clone, Debug, PartialEq, Eq)]
139struct Rule {
140    lhs: MetaTemplate,
141    rhs: MetaTemplate,
142}
143
144impl DeclarativeMacro {
145    pub fn from_err(err: ParseError) -> DeclarativeMacro {
146        DeclarativeMacro { rules: Box::default(), err: Some(Box::new(err)) }
147    }
148
149    /// The old, `macro_rules! m {}` flavor.
150    pub fn parse_macro_rules(
151        tt: &tt::TopSubtree<Span>,
152        ctx_edition: impl Copy + Fn(SyntaxContext) -> Edition,
153    ) -> DeclarativeMacro {
154        // Note: this parsing can be implemented using mbe machinery itself, by
155        // matching against `$($lhs:tt => $rhs:tt);*` pattern, but implementing
156        // manually seems easier.
157        let mut src = tt.iter();
158        let mut rules = Vec::new();
159        let mut err = None;
160
161        while !src.is_empty() {
162            let rule = match Rule::parse(ctx_edition, &mut src) {
163                Ok(it) => it,
164                Err(e) => {
165                    err = Some(Box::new(e));
166                    break;
167                }
168            };
169            rules.push(rule);
170            if let Err(()) = src.expect_char(';') {
171                if !src.is_empty() {
172                    err = Some(Box::new(ParseError::expected("expected `;`")));
173                }
174                break;
175            }
176        }
177
178        for Rule { lhs, .. } in &rules {
179            if let Err(e) = validate(lhs) {
180                err = Some(Box::new(e));
181                break;
182            }
183        }
184
185        DeclarativeMacro { rules: rules.into_boxed_slice(), err }
186    }
187
188    /// The new, unstable `macro m {}` flavor.
189    pub fn parse_macro2(
190        args: Option<&tt::TopSubtree<Span>>,
191        body: &tt::TopSubtree<Span>,
192        ctx_edition: impl Copy + Fn(SyntaxContext) -> Edition,
193    ) -> DeclarativeMacro {
194        let mut rules = Vec::new();
195        let mut err = None;
196
197        if let Some(args) = args {
198            cov_mark::hit!(parse_macro_def_simple);
199
200            let rule = (|| {
201                let lhs = MetaTemplate::parse_pattern(ctx_edition, args.iter())?;
202                let rhs = MetaTemplate::parse_template(ctx_edition, body.iter())?;
203
204                Ok(crate::Rule { lhs, rhs })
205            })();
206
207            match rule {
208                Ok(rule) => rules.push(rule),
209                Err(e) => err = Some(Box::new(e)),
210            }
211        } else {
212            cov_mark::hit!(parse_macro_def_rules);
213            let mut src = body.iter();
214            while !src.is_empty() {
215                let rule = match Rule::parse(ctx_edition, &mut src) {
216                    Ok(it) => it,
217                    Err(e) => {
218                        err = Some(Box::new(e));
219                        break;
220                    }
221                };
222                rules.push(rule);
223                if let Err(()) = src.expect_any_char(&[';', ',']) {
224                    if !src.is_empty() {
225                        err = Some(Box::new(ParseError::expected(
226                            "expected `;` or `,` to delimit rules",
227                        )));
228                    }
229                    break;
230                }
231            }
232        }
233
234        for Rule { lhs, .. } in &rules {
235            if let Err(e) = validate(lhs) {
236                err = Some(Box::new(e));
237                break;
238            }
239        }
240
241        DeclarativeMacro { rules: rules.into_boxed_slice(), err }
242    }
243
244    pub fn err(&self) -> Option<&ParseError> {
245        self.err.as_deref()
246    }
247
248    pub fn num_rules(&self) -> usize {
249        self.rules.len()
250    }
251
252    pub fn expand(
253        &self,
254        tt: &tt::TopSubtree<Span>,
255        marker: impl Fn(&mut Span) + Copy,
256        call_site: Span,
257        def_site_edition: Edition,
258    ) -> ExpandResult<(tt::TopSubtree<Span>, MatchedArmIndex)> {
259        expander::expand_rules(&self.rules, tt, marker, call_site, def_site_edition)
260    }
261}

263impl Rule {
264    fn parse(
265        edition: impl Copy + Fn(SyntaxContext) -> Edition,
266        src: &mut TtIter<'_, Span>,
267    ) -> Result<Self, ParseError> {
268        let (_, lhs) =
269            src.expect_subtree().map_err(|()| ParseError::expected("expected subtree"))?;
270        src.expect_char('=').map_err(|()| ParseError::expected("expected `=`"))?;
271        src.expect_char('>').map_err(|()| ParseError::expected("expected `>`"))?;
272        let (_, rhs) =
273            src.expect_subtree().map_err(|()| ParseError::expected("expected subtree"))?;
274
275        let lhs = MetaTemplate::parse_pattern(edition, lhs)?;
276        let rhs = MetaTemplate::parse_template(edition, rhs)?;
277
278        Ok(crate::Rule { lhs, rhs })
279    }
280}

282fn validate(pattern: &MetaTemplate) -> Result<(), ParseError> {
283    for op in pattern.iter() {
284        match op {
285            Op::Subtree { tokens, .. } => validate(tokens)?,
286            Op::Repeat { tokens: subtree, separator, .. } => {
287                // Checks that no repetition which could match an empty token
288                // https://github.com/rust-lang/rust/blob/a58b1ed44f5e06976de2bdc4d7dc81c36a96934f/src/librustc_expand/mbe/macro_rules.rs#L558
289                let lsh_is_empty_seq = separator.is_none() && subtree.iter().all(|child_op| {
290                    match *child_op {
291                        // vis is optional
292                        Op::Var { kind: Some(kind), .. } => kind == MetaVarKind::Vis,
293                        Op::Repeat {
294                            kind: parser::RepeatKind::ZeroOrMore | parser::RepeatKind::ZeroOrOne,
295                            ..
296                        } => true,
297                        _ => false,
298                    }
299                });
300                if lsh_is_empty_seq {
301                    return Err(ParseError::RepetitionEmptyTokenTree);
302                }
303                validate(subtree)?
304            }
305            _ => (),
306        }
307    }
308    Ok(())
309}

313#[derive(Debug, Clone, Eq, PartialEq)]
314pub struct ValueResult<T, E> {
315    pub value: T,
316    pub err: Option<E>,
317}
318
319impl<T: Default, E> Default for ValueResult<T, E> {
320    fn default() -> Self {
321        Self { value: Default::default(), err: Default::default() }
322    }
323}
324
325impl<T, E> ValueResult<T, E> {
326    pub fn new(value: T, err: E) -> Self {
327        Self { value, err: Some(err) }
328    }
329
330    pub fn ok(value: T) -> Self {
331        Self { value, err: None }
332    }
333
334    pub fn only_err(err: E) -> Self
335    where
336        T: Default,
337    {
338        Self { value: Default::default(), err: Some(err) }
339    }
340
341    pub fn zip_val<U>(self, other: U) -> ValueResult<(T, U), E> {
342        ValueResult { value: (self.value, other), err: self.err }
343    }
344
345    pub fn map<U>(self, f: impl FnOnce(T) -> U) -> ValueResult<U, E> {
346        ValueResult { value: f(self.value), err: self.err }
347    }
348
349    pub fn map_err<E2>(self, f: impl FnOnce(E) -> E2) -> ValueResult<T, E2> {
350        ValueResult { value: self.value, err: self.err.map(f) }
351    }
352
353    pub fn result(self) -> Result<T, E> {
354        self.err.map_or(Ok(self.value), Err)
355    }
356}
357
358impl<T: Default, E> From<Result<T, E>> for ValueResult<T, E> {
359    fn from(result: Result<T, E>) -> Self {
360        result.map_or_else(Self::only_err, Self::ok)
361    }
362}

364pub fn expect_fragment<'t>(
365    tt_iter: &mut TtIter<'t, Span>,
366    entry_point: ::parser::PrefixEntryPoint,
367    edition: ::parser::Edition,
368    delim_span: DelimSpan<Span>,
369) -> ExpandResult<tt::TokenTreesView<'t, Span>> {
370    use ::parser;
371    let buffer = tt_iter.remaining();
372    // FIXME: Pass the correct edition per token. Due to the split between mbe and hir-expand it's complicated.
373    let parser_input = to_parser_input(buffer, &mut |_ctx| edition);
374    let tree_traversal = entry_point.parse(&parser_input, edition);
375    let mut cursor = buffer.cursor();
376    let mut error = false;
377    for step in tree_traversal.iter() {
378        match step {
379            parser::Step::Token { kind, mut n_input_tokens } => {
380                if kind == ::parser::SyntaxKind::LIFETIME_IDENT {
381                    n_input_tokens = 2;
382                }
383                for _ in 0..n_input_tokens {
384                    cursor.bump_or_end();
385                }
386            }
387            parser::Step::FloatSplit { .. } => {
388                // FIXME: We need to split the tree properly here, but mutating the token trees
389                // in the buffer is somewhat tricky to pull off.
390                cursor.bump_or_end();
391            }
392            parser::Step::Enter { .. } | parser::Step::Exit => (),
393            parser::Step::Error { .. } => error = true,
394        }
395    }
396
397    let err = if error || !cursor.is_root() {
398        Some(ExpandError::binding_error(
399            buffer.cursor().token_tree().map_or(delim_span.close, |tt| tt.first_span()),
400            format!("expected {entry_point:?}"),
401        ))
402    } else {
403        None
404    };
405
406    while !cursor.is_root() {
407        cursor.bump_or_end();
408    }
409
410    let res = cursor.crossed();
411    tt_iter.flat_advance(res.len());
412
413    ExpandResult { value: res, err }
